20/*!
21 * \file src/ir/function.cc
22 * \brief The function data structure.
23 */
24#include <tvm/ir/function.h>
25#include <tvm/runtime/registry.h>
26// NOTE: reverse dependency on relay, tir/
27// These dependencies do not happen at the interface-level,
28// and are only used in minimum cases where they are clearly marked.
29//
30// Rationale: We calls into the type specific WithAttr function
31#include <tvm/relay/function.h>
32#include <tvm/tir/function.h>
33
34namespace tvm {
35
36TVM_REGISTER_GLOBAL("ir.BaseFunc_Attrs").set_body_typed([](BaseFunc func) { return func->attrs; });
37
38TVM_REGISTER_GLOBAL("ir.BaseFuncCopy").set_body_typed([](BaseFunc func) { return func; });
39
40TVM_REGISTER_GLOBAL("ir.BaseFuncWithAttr")
41 .set_body_typed([](BaseFunc func, String key, ObjectRef value) -> BaseFunc {
42 if (func->IsInstance<tir::PrimFuncNode>()) {
43 return WithAttr(Downcast<tir::PrimFunc>(std::move(func)), key, value);
44 } else if (func->IsInstance<relay::FunctionNode>()) {
45 return WithAttr(Downcast<relay::Function>(std::move(func)), key, value);
46 } else {
47 LOG(FATAL) << "Do not support function type " << func->GetTypeKey();
48 return func;
49 }
50 });
51
52} // namespace tvm
