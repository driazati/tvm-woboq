88/* Diagnostic Context */
89TVM_REGISTER_NODE_TYPE(DiagnosticContextNode);
90
91void DiagnosticContext::Render() {
92 (*this)->renderer.Render(*this);
93
94 int errs = 0;
95 if ((*this)->diagnostics.size()) {
96 for (auto diagnostic : (*this)->diagnostics) {
97 if (diagnostic->level == DiagnosticLevel::kError) {
98 errs += 1;
99 }
100 }
101 }
102
103 if (errs) {
104 (*this)->renderer = DiagnosticRenderer();
105 LOG(FATAL) << "DiagnosticError: one or more error diagnostics were "
106 << "emitted, please check diagnostic render for output.";
107 }
108}

111 .set_body_typed([](DiagnosticRenderer renderer, DiagnosticContext ctx) {
112 renderer.Render(ctx);
113 });

147/* Default Terminal Renderer. */
148static const char* DEFAULT_RENDERER = "diagnostics.DefaultRenderer";
149static const char* OVERRIDE_RENDERER = "diagnostics.OverrideRenderer";
150
151DiagnosticRenderer GetRenderer() {
152 auto override_pf = tvm::runtime::Registry::Get(OVERRIDE_RENDERER);
153 tvm::runtime::TypedPackedFunc<ObjectRef()> pf;
154 if (override_pf) {
155 pf = tvm::runtime::TypedPackedFunc<ObjectRef()>(*override_pf);
156 } else {
157 auto default_pf = tvm::runtime::Registry::Get(DEFAULT_RENDERER);
158 ICHECK(default_pf != nullptr)
159 << "Can not find registered function for " << DEFAULT_RENDERER << "." << std::endl
160 << "Either this is an internal error or the default function was overloaded incorrectly.";
161 pf = tvm::runtime::TypedPackedFunc<ObjectRef()>(*default_pf);
162 }
163 return Downcast<DiagnosticRenderer>(pf());
164}
165
166DiagnosticContext DiagnosticContext::Default(const IRModule& module) {
167 auto renderer = GetRenderer();
168 return DiagnosticContext(module, renderer);
169}

175std::ostream& EmitDiagnosticHeader(std::ostream& out, const Span& span, DiagnosticLevel level,
176 std::string msg) {
177 rang::fg diagnostic_color = rang::fg::reset;
178 std::string diagnostic_type;
179
180 switch (level) {
181 case DiagnosticLevel::kWarning: {
182 diagnostic_color = rang::fg::yellow;
183 diagnostic_type = "warning";
184 break;
185 }
186 case DiagnosticLevel::kError: {
187 diagnostic_color = rang::fg::red;
188 diagnostic_type = "error";
189 break;
190 }
191 case DiagnosticLevel::kBug: {
192 diagnostic_color = rang::fg::blue;
193 diagnostic_type = "bug";
194 break;
195 }
196 case DiagnosticLevel::kNote: {
197 diagnostic_color = rang::fg::reset;
198 diagnostic_type = "note";
199 break;
200 }
201 case DiagnosticLevel::kHelp: {
202 diagnostic_color = rang::fg::reset;
203 diagnostic_type = "help";
204 break;
205 }
206 }
207
208 out << rang::style::bold << diagnostic_color << diagnostic_type << ": " << rang::fg::reset << msg
209 << std::endl
210 << rang::fg::blue << " --> " << rang::fg::reset << rang::style::reset
211 << span->source_name->name << ":" << span->line << ":" << span->column << std::endl;
212
213 return out;
214}

216/*! \brief Generate an error message at a specific line and column with the
217 * annotated message.
218 *
219 * The error is written directly to the `out` std::ostream.
220 *
221 * \param out The output ostream.
222 * \param line The line at which to report a diagnostic.
223 * \param line The column at which to report a diagnostic.
224 * \param msg The message to attach.
225 */
226void ReportAt(const DiagnosticContext& context, std::ostream& out, const Span& span,
227 const Diagnostic& diagnostic) {
228 if (!span.defined()) {
229 out << diagnostic->message << std::endl;
230 return;
231 }
232
233 ICHECK(context->module->source_map.defined());
234 auto it = context->module->source_map->source_map.find(span->source_name);
235
236 // If the source name is not in the current source map, sources were not annotated.
237 if (it == context->module->source_map->source_map.end()) {
238 LOG(FATAL) << "The source maps are not populated for this module. "
239 << "Please use `tvm.relay.transform.AnnotateSpans` to attach source maps for error "
240 "reporting.\n"
241 << "Error: " << diagnostic->message;
242 }
243
244 auto source = (*it).second;
245 VLOG(1) << "Source: " << std::endl << source->source;
246
247 VLOG(1) << "ReportAt "
248 << "span = " << span << " msg = " << diagnostic->message;
249
250 auto line_text = source.GetLine(span->line);
251
252 std::stringstream line_header_s;
253 line_header_s << " " << span->line << " ";
254 auto line_header = line_header_s.str();
255
256 std::stringstream no_line_header_s;
257 for (size_t i = 0; i < line_header.size(); i++) {
258 no_line_header_s << " ";
259 }
260 auto no_line_header = no_line_header_s.str();
261
262 EmitDiagnosticHeader(out, span, diagnostic->level, diagnostic->message)
263 << no_line_header << "| " << std::endl
264 << line_header << "| " << line_text << std::endl
265 << no_line_header << "| ";
266
267 std::stringstream marker;
268 for (size_t i = 1; i <= line_text.size(); i++) {
269 if (static_cast<int>(i) >= span->column && static_cast<int>(i) < span->end_column) {
270 marker << "^";
271 } else {
272 marker << " ";
273 }
274 }
275 out << marker.str();
276 out << std::endl;
277}

279// TODO(@jroesch): eventually modularize the rendering interface to provide control of how to
280// format errors.
281DiagnosticRenderer TerminalRenderer(std::ostream& out) {
282 return DiagnosticRenderer([&](const DiagnosticContext& ctx) {
283 for (auto diagnostic : ctx->diagnostics) {
284 ReportAt(ctx, out, diagnostic->span, diagnostic);
285 }
286 });
287}
