45IRModule::IRModule(tvm::Map<GlobalVar, BaseFunc> functions,
46 tvm::Map<GlobalTypeVar, TypeData> type_definitions,
47 std::unordered_set<String> import_set, parser::SourceMap source_map,
48 DictAttrs attrs) {
49 auto n = make_object<IRModuleNode>();
50 n->functions = std::move(functions);
51 n->type_definitions = std::move(type_definitions);
52 n->global_type_var_map_ = {};
53 n->global_var_map_ = {};
54 n->constructor_tag_map_ = {};
55 n->import_set_ = std::move(import_set);
56 n->source_map = source_map;
57 n->attrs = std::move(attrs);
58
59 for (const auto& kv : n->functions) {
60 // set global var map
61 ICHECK(n->global_var_map_.count(kv.first->name_hint) == 0)
62 << "Duplicate global function name " << kv.first->name_hint;
63 n->global_var_map_.Set(kv.first->name_hint, kv.first);
64 }
65
66 for (const auto& kv : n->type_definitions) {
67 // set global typevar map
68 ICHECK(n->global_type_var_map_.count(kv.first->name_hint) == 0)
69 << "Duplicate global type definition name " << kv.first->name_hint;
70 n->global_type_var_map_.Set(kv.first->name_hint, kv.first);
71 n->RegisterConstructors(kv.first, kv.second);
72 }
73 data_ = std::move(n);
74}

76bool IRModuleNode::SEqualReduce(const IRModuleNode* other, SEqualReducer equal) const {
77 if (functions.size() != other->functions.size()) return false;
78 if (!equal(this->attrs, other->attrs)) return false;
79 for (const auto& kv : this->functions) {
80 if (!other->ContainGlobalVar(kv.first->name_hint)) return false;
81 if (!equal(kv.second, other->Lookup(kv.first->name_hint))) return false;
82 }
83 if (type_definitions.size() != other->type_definitions.size()) return false;
84 for (const auto& kv : this->type_definitions) {
85 if (!other->ContainGlobalTypeVar(kv.first->name_hint)) return false;
86 if (!equal(kv.second, other->LookupTypeDef(kv.first->name_hint))) return false;
87 }
88 return true;
89}

91void IRModuleNode::SHashReduce(SHashReducer hash_reduce) const {
92 using KV = std::pair<std::string, ObjectRef>;
93 // hash the functions.
94 std::vector<KV> temp;
95
96 auto reduce_temp = [&]() {
97 // sort by the hash key of the keys.
98 std::sort(temp.begin(), temp.end(),
99 [](const KV& lhs, const KV& rhs) { return lhs.first < rhs.first; });
100
101 hash_reduce(static_cast<uint64_t>(temp.size()));
102 // hash the content
103 for (size_t i = 0; i < temp.size(); ++i) {
104 hash_reduce(temp[i].first);
105 hash_reduce(temp[i].second);
106 }
107 };
108
109 for (const auto& kv : this->functions) {
110 temp.emplace_back(kv.first->name_hint, kv.second);
111 }
112 reduce_temp();
113
114 temp.clear();
115 for (const auto& kv : this->type_definitions) {
116 temp.emplace_back(kv.first->name_hint, kv.second);
117 }
118 reduce_temp();
119 hash_reduce(this->attrs);
120}

122bool IRModuleNode::ContainGlobalVar(const String& name) const {
123 return global_var_map_.find(name) != global_var_map_.end();
124}

126bool IRModuleNode::ContainGlobalTypeVar(const String& name) const {
127 return global_type_var_map_.find(name) != global_type_var_map_.end();
128}
129
130GlobalVar IRModuleNode::GetGlobalVar(const String& name) const {
131 auto it = global_var_map_.find(name);
132 if (it == global_var_map_.end()) {
133 std::ostringstream msg;
134 msg << "ValueError: Cannot find global var \"" << name << "\" in the Module\n"
135 << "candidates are: [";
136 int counter = 0;
137 for (auto kv : global_var_map_) {
138 if (counter++ != 0) {
139 msg << ", ";
140 }
141 msg << "\"" << kv.first << "\"";
142 }
143 msg << "]";
144 LOG(FATAL) << msg.str();
145 }
146 return (*it).second;
147}

149tvm::Array<GlobalVar> IRModuleNode::GetGlobalVars() const {
150 std::vector<GlobalVar> global_vars;
151 for (const auto& pair : global_var_map_) {
152 global_vars.push_back(pair.second);
153 }
154 return tvm::Array<GlobalVar>(global_vars);

157GlobalTypeVar IRModuleNode::GetGlobalTypeVar(const String& name) const {
158 ICHECK(global_type_var_map_.defined());
159 auto it = global_type_var_map_.find(name);
160 ICHECK(it != global_type_var_map_.end())
161 << "Cannot find global type var " << name << " in the Module";
162 return (*it).second;
163}

165Constructor IRModuleNode::GetConstructor(const String& adt, const String& cons) const {
166 TypeData typeDef = this->LookupTypeDef(adt);
167 for (Constructor c : typeDef->constructors) {
168 if (cons.compare(c->name_hint) == 0) {
169 return c;
170 }
171 }
172
173 LOG(FATAL) << adt << " does not contain constructor " << cons;
174 return {};
175}

177tvm::Array<GlobalTypeVar> IRModuleNode::GetGlobalTypeVars() const {
178 std::vector<GlobalTypeVar> global_type_vars;
179 for (const auto& pair : global_type_var_map_) {
180 global_type_vars.push_back(pair.second);
181 }
182 return tvm::Array<GlobalTypeVar>(global_type_vars);
183}

185void WarnIfMalformed(const IRModule& mod, relay::Function func) {
186 func = Downcast<relay::Function>(relay::DeDup(func));
187 // Type check the item before we add it to the module.
188 auto fv = relay::FreeVars(func);
189 auto ftv = relay::FreeTypeVars(func, mod);
190 // TODO(@jroesch): refactor to use diagnostic context
191 ICHECK_EQ(fv.size(), 0) << "Function:" << std::endl
192 << PrettyPrint(func) << std::endl
193 << "contains free variables: " << fv;
194 ICHECK_EQ(ftv.size(), 0) << "Function:" << std::endl
195 << PrettyPrint(func) << std::endl
196 << "contains free type variables: " << fv;
197}

199void IRModuleNode::Add(const GlobalVar& var, const BaseFunc& f, bool update) {
200 BaseFunc checked_func = f;
201 if (auto* ptr = f.as<relay::FunctionNode>()) {
202 WarnIfMalformed(GetRef<IRModule>(this), GetRef<relay::Function>(ptr));
203 }
204
205 AddUnchecked(var, checked_func);
206}
207
208void IRModuleNode::AddUnchecked(const GlobalVar& var, const BaseFunc& func) {
209 this->functions.Set(var, func);
210
211 auto it = global_var_map_.find(var->name_hint);
212 if (it != global_var_map_.end()) {
213 ICHECK_EQ((*it).second, var);
214 } else {
215 ICHECK(global_var_map_.count(var->name_hint) == 0)
216 << "Duplicate global function name " << PrettyPrint(var);
217 }
218
219 global_var_map_.Set(var->name_hint, var);
220}

222void IRModuleNode::RegisterConstructors(const GlobalTypeVar& var, const TypeData& type) {
223 // We hash the global type var name to use as a globally unique prefix for tags.
224 // The hash will be used as the most significant byte of the tag, with the index of
225 // the constructor in the less significant bytes
226 size_t hash = std::hash<std::string>()(var->name_hint);
227 int32_t prefix = static_cast<int32_t>(hash & 0xff) << 24;
228 for (size_t i = 0; i < type->constructors.size(); ++i) {
229 type->constructors[i]->tag = prefix | static_cast<int32_t>(i);
230 constructor_tag_map_[type->constructors[i]->tag] = type->constructors[i];
231 }
232}

234void IRModuleNode::AddTypeDef(const GlobalTypeVar& var, const TypeData& type, bool update) {
235 // TODO(@jroesch): we have temporarily removed kind checking here, and will consolidate
236 // to the type checker in follow up PR.
237 AddTypeDefUnchecked(var, type, update);
238}

240void IRModuleNode::AddTypeDefUnchecked(const GlobalTypeVar& var, const TypeData& type,
241 bool update) {
242 this->type_definitions.Set(var, type);
243 if (!update) {
244 // set global type var map
245 ICHECK(global_type_var_map_.count(var->name_hint) == 0)
246 << "Duplicate global type definition name " << PrettyPrint(var);
247 }
248 global_type_var_map_.Set(var->name_hint, var);
249 RegisterConstructors(var, type);
250}

260void IRModuleNode::Remove(const GlobalVar& var) {
261 auto functions_node = this->functions.CopyOnWrite();
262 functions_node->erase(var);
263 auto gvar_node = global_var_map_.CopyOnWrite();
264 gvar_node->erase(var->name_hint);
265}

267BaseFunc IRModuleNode::Lookup(const GlobalVar& var) const {
268 auto it = functions.find(var);
269 ICHECK(it != functions.end()) << "There is no definition of " << PrettyPrint(var);
270 return (*it).second;
271}

273BaseFunc IRModuleNode::Lookup(const String& name) const {
274 GlobalVar id = this->GetGlobalVar(name);

278TypeData IRModuleNode::LookupTypeDef(const GlobalTypeVar& var) const {

295String IRModuleNode::GetUniqueName(const String& name) {
296 String result = name;
297 int suffix = 0;
298 while (true) {
299 auto it = global_var_map_.find(result);
300 if (it == global_var_map_.end()) {
301 return result;
302 }
303 std::ostringstream os;
304 os << name << "_" << ++suffix;
305 result = os.str();
306 }

309/*!
310 * \brief Renames global type/term variables to prefer the GlobalTypeVar/GlobalVar in the lhs
311 * ('one') side above the rhs ('two').
312 */
313struct Renamer : relay::ExprMutator, TypeMutator {
314 Map<String, GlobalVar> defs;
315 Map<String, GlobalTypeVar> types;
316 std::unordered_map<int32_t, Constructor> ctors;
317
318 Renamer(Map<String, GlobalVar> defs_one, Map<String, GlobalVar> defs_two,
319 Map<String, GlobalTypeVar> types_one, Map<String, GlobalTypeVar> types_two,
320 std::unordered_map<int32_t, Constructor> ctors_one,
321 std::unordered_map<int32_t, Constructor> ctor_two) {
322 for (auto pair : defs_one) {
323 defs.Set(pair.first, pair.second);
324 }
325
326 for (auto pair : defs_two) {
327 auto it = defs.find(pair.first);
328 if (it == defs.end()) {
329 defs.Set(pair.first, pair.second);
330 }
331 }
332
333 for (auto pair : types_one) {
334 types.Set(pair.first, pair.second);
335 }
336
337 for (auto pair : types_two) {
338 auto it = types.find(pair.first);
339 if (it == types.end()) {
340 types.Set(pair.first, pair.second);
341 }
342 }
343 }
344
345 relay::Expr VisitExpr_(const GlobalVarNode* node) override { return defs.at(node->name_hint); }
346
347 Type VisitType_(const GlobalTypeVarNode* node) override { return types.at(node->name_hint); }
348};

350void IRModuleNode::Update(const IRModule& mod) {
351 Renamer renamer(this->global_var_map_, mod->global_var_map_, this->global_type_var_map_,
352 mod->global_type_var_map_, this->constructor_tag_map_, mod->constructor_tag_map_);
353
354 this->global_var_map_ = renamer.defs;
355 this->global_type_var_map_ = renamer.types;
356 this->constructor_tag_map_ = renamer.ctors;
357
358 for (auto pair : mod->type_definitions) {
359 auto tvar = renamer.types.at(pair.first->name_hint);
360 auto ty = renamer.ExprMutator::VisitType(pair.second);
361 this->AddTypeDefUnchecked(tvar, Downcast<TypeData>(ty), true);
362 }
363
364 for (auto pair : mod->functions) {
365 if (auto rfn = pair.second.as<relay::FunctionNode>()) {
366 auto gvar = renamer.defs.at(pair.first->name_hint);
367 auto fn = renamer.VisitExpr(GetRef<relay::Function>(rfn));
368 this->AddUnchecked(gvar, Downcast<BaseFunc>(fn));
369 } else {
370 // TODO(@jroesch): rename into IRModule.
371 this->AddUnchecked(pair.first, pair.second);
372 }
373 }
374}

381std::pair<IRModule, GlobalVar> IRModule::FromExprInContext(
382 const RelayExpr& expr, const tvm::Map<GlobalVar, BaseFunc>& global_funcs,
383 const tvm::Map<GlobalTypeVar, TypeData>& type_definitions,
384 std::unordered_set<String> import_set) {
385 auto mod = IRModule(global_funcs, type_definitions, std::move(import_set));
386 String gv_name;
387
388 // All global definitions must be functions.
389 BaseFunc func;
390 if (auto* func_node = expr.as<BaseFuncNode>()) {
391 func = GetRef<BaseFunc>(func_node);
392 if (auto opt = func->GetAttr<String>(tvm::attr::kGlobalSymbol)) {
393 // Function literal has been annotated with it's required global symbol.
394 gv_name = opt.value();
395 }
396 } else {
397 func = relay::Function(relay::FreeVars(expr), expr, Type(), relay::FreeTypeVars(expr, mod), {});
398 }
399
400 if (gv_name.empty()) {
401 // Bind function to 'main' (though rename if would clash with existing 'main').
402 gv_name = mod->GetUniqueName("main");
403 }
404
405 GlobalVar main_gv(gv_name);
406 mod->Add(main_gv, func);
407 return {mod, main_gv};
408}

410IRModule IRModule::FromExpr(const RelayExpr& expr, const Map<GlobalVar, BaseFunc>& global_funcs,
411 const Map<GlobalTypeVar, TypeData>& type_definitions) {
412 return FromExprInContext(expr, global_funcs, type_definitions).first;
413}

415void IRModuleNode::Import(const String& path) {
416 if (this->import_set_.count(path) == 0) {
417 this->import_set_.insert(path);
418 std::fstream src_file(path, std::fstream::in);
419 std::string file_contents{std::istreambuf_iterator<char>(src_file),
420 std::istreambuf_iterator<char>()};
421 auto mod_to_import = parser::ParseModule(path, file_contents, GetRef<IRModule>(this));
422 Update(mod_to_import);
423 }
424}

455TVM_REGISTER_GLOBAL("ir.Module_Add").set_body([](TVMArgs args, TVMRetValue* ret) {
456 IRModule mod = args[0];
457 GlobalVar var = args[1];
458 ObjectRef val = args[2];
459 bool update = args[3];
460 ICHECK(val->IsInstance<RelayExprNode>());
461
462 if (val->IsInstance<BaseFuncNode>()) {
463 mod->Add(var, Downcast<BaseFunc>(val), update);
464 } else if (val->IsInstance<GlobalVarNode>()) {
465 GlobalVar gv = Downcast<GlobalVar>(val);
466 auto mod_copy = IRModule(make_object<IRModuleNode>(*mod.operator->()));
467 mod_copy = relay::transform::EtaExpand(
468 /* expand_constructor */ false,
469 /* expand_global_var */ true)(mod_copy);
470 auto func = mod_copy->Lookup(gv->name_hint);
471 mod->Add(var, Downcast<relay::Function>(func), update);
472 } else {
473 auto func = relay::Function({}, Downcast<RelayExpr>(val), Type(nullptr), {});
474 mod->Add(var, func, update);
475 }
476 *ret = mod;
477});
