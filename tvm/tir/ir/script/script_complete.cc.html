36/*! \brief Generate surrounding loops automatically */
37class ScriptCompleter : public StmtMutator {
38 public:
39 explicit ScriptCompleter(Map<Var, Buffer>* buffer_var_map) : buffer_var_map_(buffer_var_map) {}
40 /*! \brief Whether the stmt contains at least one block. */
41 bool contains_block = false;
42
43 private:
44 Map<Var, Buffer>* buffer_var_map_;
45 Stmt VisitStmt_(const BlockRealizeNode* op) override {
46 contains_block = true;
47 for (const PrimExpr& value : op->iter_values) {
48 CHECK(value.dtype().is_int())
49 << "BlockRealize iter_value expected a IntImm, but got " << value.dtype();
50 }
51 return StmtMutator::VisitStmt_(op);
52 }
53
54 Stmt VisitStmt_(const BlockNode* op) override {
55 // Buffers allocated in the block can be accessed by its body.
56 for (const auto& alloc_buffer : op->alloc_buffers) {
57 buffer_var_map_->Set(alloc_buffer->data, alloc_buffer);
58 }
59 for (const auto& match_buffer : op->match_buffers) {
60 const Buffer& target_buffer = match_buffer->buffer;
61 buffer_var_map_->Set(target_buffer->data, target_buffer);
62 }
63 Block block = Downcast<Block>(StmtMutator::VisitStmt_(op));
64 // Remove buffers allocated inside block to detect its access region
65 for (const auto& alloc_buffer : op->alloc_buffers) {
66 buffer_var_map_->erase(alloc_buffer->data);
67 }
68 for (const auto& match_buffer : op->match_buffers) {
69 const Buffer& target_buffer = match_buffer->buffer;
70 buffer_var_map_->erase(target_buffer->data);
71 }
72 // Get access detection mask
73 // 0 for provided region, 1 and 3 for need detect read, 2 and 3 for need detect write
74 int mask = 0;
75 auto it = op->annotations.find(attr::script_parsing_detect_access);
76 if (it != op->annotations.end()) {
77 mask = Downcast<IntImm>((*it).second)->value;
78 }
79 // ignore root block or blocks which already has reads/writes regions
80 if (mask != 0) {
81 auto access_region = GetBlockAccessRegion(block, *buffer_var_map_);
82 const Array<BufferRegion>& reads = access_region[0];
83 const Array<BufferRegion>& writes = access_region[1];
84 const Array<BufferRegion>& opaque = access_region[2];
85 CHECK(opaque.empty())
86 << "ValueError: Can not auto detect buffer access region from tir.Load, tir.Store or "
87 "direct access by buffer data. Please annotation the access region manually";
88 auto n = CopyOnWrite(block.operator->());
89 if (mask & 1) n->reads = reads;
90 if (mask & 2) n->writes = writes;
91 n->annotations = op->annotations;
92 n->annotations.erase(attr::script_parsing_detect_access);
93 return Block(n);
94 } else {
95 return std::move(block);
96 }
97 }
98};
99
100PrimFunc ScriptComplete(PrimFunc func, const Array<Buffer>& root_allocates) {
101 Map<Var, Buffer> buffer_var_map;
102 for (const auto& pair : func->buffer_map) {
103 const Buffer& buffer = pair.second;
104 buffer_var_map.Set(buffer->data, buffer);
105 }
106 for (const auto& alloc : root_allocates) {
107 buffer_var_map.Set(alloc->data, alloc);
108 }
109 bool contain_root = root_allocates.empty() && func->body->IsInstance<BlockRealizeNode>() &&
110 Downcast<BlockRealize>(func->body)->block->iter_vars.empty();
111 ScriptCompleter script_completer(&buffer_var_map);
112 // generate surrounding loops automatically
113 Stmt res = script_completer(func->body);
114 // generate root block automatically
115 if ((script_completer.contains_block || root_allocates.size()) && !contain_root) {
116 res = Block({}, {}, {}, "root", res, NullOpt, root_allocates);
117 res = BlockRealize({}, Bool(true), Downcast<Block>(res));
118 }
119 if (func->body.same_as(res)) {
120 return func;
121 } else {
122 auto fptr = func.CopyOnWrite();
123 fptr->body = res;
124 return func;
125 }
126}
127
128TVM_REGISTER_GLOBAL("script.Complete").set_body_typed(ScriptComplete);
