37Stmt MergeNest(const std::vector<Stmt>& nest, Stmt body) {
38 // use reverse iteration
39 for (auto ri = nest.rbegin(); ri != nest.rend(); ++ri) {
40 Stmt s = *ri;
41 if (const auto* for_ = s.as<ForNode>()) {
42 auto n = make_object<ForNode>(*for_);
43 ICHECK(is_no_op(n->body));
44 n->body = body;
45 body = Stmt(n);
46 } else if (const auto* let = s.as<LetStmtNode>()) {
47 auto n = make_object<LetStmtNode>(*let);
48 ICHECK(is_no_op(n->body));
49 n->body = body;
50 body = Stmt(n);
51 } else if (const auto* attr = s.as<AttrStmtNode>()) {
52 auto n = make_object<AttrStmtNode>(*attr);
53 ICHECK(is_no_op(n->body));
54 n->body = body;
55 body = Stmt(n);
56 } else if (const auto* ite = s.as<IfThenElseNode>()) {
57 auto n = make_object<IfThenElseNode>(*ite);
58 ICHECK(is_no_op(n->then_case));
59 ICHECK(!n->else_case.defined());
60 n->then_case = body;
61 body = Stmt(n);
62 } else if (const auto* seq = s.as<SeqStmtNode>()) {
63 auto n = make_object<SeqStmtNode>(*seq);
64 ICHECK(n->size() != 0 && is_no_op(n->seq[n->size() - 1]));
65 n->seq.Set(n->size() - 1, body);
66 body = Stmt(n);
67 } else if (const auto* assert_ = s.as<AssertStmtNode>()) {
68 auto n = make_object<AssertStmtNode>(*assert_);
69 ICHECK(is_no_op(n->body));
70 n->body = body;
71 body = Stmt(n);
72 } else if (const auto* alloc = s.as<AllocateNode>()) {
73 auto n = make_object<AllocateNode>(*alloc);
74 ICHECK(is_no_op(n->body));
75 n->body = body;
76 body = Stmt(n);
77 } else {
78 LOG(FATAL) << "not supported nest type";
79 }
80 }
81 return body;
82}

84Stmt MergeNest(const std::vector<std::vector<Stmt>>& nest, Stmt body) {
85 for (auto ri = nest.rbegin(); ri != nest.rend(); ++ri) {
86 body = MergeNest(*ri, body);
87 }
88 return body;
89}

91class IRConvertSSA final : public StmtExprMutator {
92 public:
93 PrimExpr VisitExpr_(const VarNode* op) final {
94 if (scope_.count(op) && !scope_[op].empty()) {
95 return scope_[op].back();
96 } else {
97 return GetRef<PrimExpr>(op);
98 }
99 }
100 PrimExpr VisitExpr_(const LetNode* op) final {
101 const Var& v = op->var;
102 if (defined_.count(v.get())) {
103 PrimExpr value = this->VisitExpr(op->value);
104 ScopedRedefine redefine(this, v);
105 PrimExpr body = this->VisitExpr(op->body);
106 return Let(redefine.new_var, value, body);
107 } else {
108 defined_.insert(v.get());
109 return StmtExprMutator::VisitExpr_(op);
110 }
111 }
112
113 PrimExpr VisitExpr_(const LoadNode* op) final {
114 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
115 return PrimExpr();
116 }
117
118 Stmt VisitStmt_(const StoreNode* op) final {
119 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
120 return Stmt();
121 }
122
123 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
124 auto node = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
125 auto output = VisitBufferAccess(std::move(node));
126 return std::move(output);
127 }
128
129 Stmt VisitStmt_(const BufferStoreNode* op) final {
130 auto node = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
131 auto output = VisitBufferAccess(std::move(node));
132 return std::move(output);
133 }
134
135 template <typename Node>
136 Node VisitBufferAccess(Node node) {
137 Buffer new_buf = GetRemappedBuffer(node->buffer);
138 if (!new_buf.same_as(node->buffer)) {
139 auto writer = node.CopyOnWrite();
140 writer->buffer = new_buf;
141 }
142
143 return node;
144 }
145
146 Buffer GetRemappedBuffer(Buffer buf) {
147 // Determine the buffer var that should be in the updated buffer,
148 // given the current scope. If no redefines are present, then the
149 // buffer var is unchanged.
150 Var new_buffer_var = buf->data;
151 auto var_it = scope_.find(buf->data.get());
152 if (var_it != scope_.end() && !var_it->second.empty()) {
153 new_buffer_var = var_it->second.back();
154 }
155
156 // If no mapping is required, return the original buffer.
157 if (new_buffer_var.same_as(buf->data)) {
158 return buf;
159 }
160
161 // If the current scope already has a mapping of this buffer, use
162 // the mapped buffer.
163 auto key = buf.get();
164 std::vector<Buffer>& buffers = buf_remap_[key];
165 if (buffers.size() && buffers.back()->data.same_as(new_buffer_var)) {
166 return buffers.back();
167 }
168
169 // Otherwise, make and return a new buffer object that uses the
170 // new buffer, pushing it onto the scoped stack of existing
171 // buffers. This will be popped when the new_buffer_var
172 // redefinition is popped.
173 Buffer new_buf(new_buffer_var, buf->dtype, buf->shape, buf->strides, buf->elem_offset,
174 buf->name, buf->data_alignment, buf->offset_factor, buf->buffer_type,
175 buf->axis_separators, buf->span);
176 buffers.push_back(new_buf);
177 return new_buf;
178 }

180 Stmt VisitStmt_(const LetStmtNode* op) final {
181 const Var& v = op->var;
182 if (defined_.count(v.get())) {
183 PrimExpr value = this->VisitExpr(op->value);
184 ScopedRedefine redefine(this, v);
185 Stmt body = this->VisitStmt(op->body);
186 return LetStmt(redefine.new_var, value, body);
187 } else {
188 defined_.insert(v.get());
189 return StmtExprMutator::VisitStmt_(op);
190 }
191 }
192 Stmt VisitStmt_(const ForNode* op) final {
193 const Var& v = op->loop_var;
194 if (defined_.count(v.get())) {
195 ScopedRedefine redefine(this, v);
196 Stmt stmt = StmtExprMutator::VisitStmt_(op);
197 op = stmt.as<ForNode>();
198 return For(redefine.new_var, op->min, op->extent, op->kind, op->body, op->thread_binding,
199 op->annotations);
200 } else {
201 defined_.insert(v.get());
202 return StmtExprMutator::VisitStmt_(op);
203 }
204 }
205 Stmt VisitStmt_(const AllocateNode* op) final {
206 const Var& v = op->buffer_var;
207 if (defined_.count(v.get())) {
208 ScopedRedefine redefine(this, v);
209 Stmt stmt = StmtExprMutator::VisitStmt_(op);
210 op = stmt.as<AllocateNode>();
211 return Allocate(redefine.new_var, op->dtype, op->extents, op->condition, op->body);
212 } else {
213 defined_.insert(v.get());
214 return StmtExprMutator::VisitStmt_(op);
215 }
216 }
217 Stmt VisitStmt_(const AttrStmtNode* op) final {
218 if (const VarNode* v = op->node.as<VarNode>()) {
219 Stmt stmt = StmtExprMutator::VisitStmt_(op);
220 op = stmt.as<AttrStmtNode>();
221 if (scope_.count(v) && scope_[v].size() != 0) {
222 return AttrStmt(scope_[v].back(), op->attr_key, op->value, op->body);
223 } else {
224 return stmt;
225 }
226 } else {
227 return StmtExprMutator::VisitStmt_(op);
228 }
229 }
230
231 private:
232 struct ScopedRedefine {
233 ScopedRedefine(IRConvertSSA* parent, Var old_var) : parent(parent), old_var(old_var) {
234 if (old_var->type_annotation.defined()) {
235 new_var = Var(old_var->name_hint, old_var->type_annotation);
236 } else {
237 new_var = Var(old_var->name_hint, old_var->dtype);
238 }
239 parent->scope_[old_var.get()].push_back(new_var);
240 }
241
242 ~ScopedRedefine() {
243 parent->scope_[old_var.get()].pop_back();
244 for (auto& kv : parent->buf_remap_) {
245 std::vector<Buffer>& buffers = kv.second;
246 if (buffers.size() && (buffers.back()->data.get() == new_var.get())) {
247 buffers.pop_back();
248 }
249 }
250 }
251
252 IRConvertSSA* parent;
253 Var old_var;
254 Var new_var;
255 };
256
257 std::unordered_map<const VarNode*, std::vector<Var>> scope_;
258 std::unordered_set<const VarNode*> defined_;
259 std::unordered_map<const BufferNode*, std::vector<Buffer>> buf_remap_;
260};

270Array<PrimExpr> ConvertIndices(const MatchBufferRegion& match_buffer,
271 const Array<PrimExpr>& indices) {
272 const Buffer& target = match_buffer->buffer;
273 const BufferRegion& source = match_buffer->source;
274 ICHECK_EQ(indices.size(), target->shape.size());
275
276 arith::Analyzer analyzer;
277 Array<PrimExpr> result;
278 result.reserve(source->region.size());
279 size_t offset = source->region.size() - indices.size();
280 for (size_t i = 0; i < offset; ++i) {
281 const Range& range = source->region[i];
282 ICHECK(analyzer.CanProve(range->extent == 1));
283 result.push_back(range->min);
284 }
285 for (size_t i = 0; i < indices.size(); ++i) {
286 const Range& range = source->region[i + offset];
287 const PrimExpr& index = indices[i];
288 result.push_back(range->min + index);
289 }
290 return result;
291}
292
293Region ConvertRegion(const MatchBufferRegion& match_buffer, const Region& region) {
294 const Buffer& target = match_buffer->buffer;
295 const BufferRegion& source = match_buffer->source;
296 ICHECK_EQ(region.size(), target->shape.size());
297
298 arith::Analyzer analyzer;
299 Region result;
300 result.reserve(source->region.size());
301 size_t offset = source->region.size() - region.size();
302 for (size_t i = 0; i < offset; ++i) {
303 const Range& source_range = source->region[i];
304 ICHECK(analyzer.CanProve(source_range->extent == 1));
305 result.push_back(Range::FromMinExtent(source_range->min, 1));
306 }
307 for (size_t i = 0; i < region.size(); ++i) {
308 const Range& source_range = source->region[i + offset];
309 const Range& target_range = region[i];
310 result.push_back(
311 Range::FromMinExtent(source_range->min + target_range->min, target_range->extent));
312 }
313 return result;
314}

321Map<Var, Range> ConditionalBoundsContext::GetVarBoundsFromCondition() {
322 // extract equations and related vars from condition expression.
323 // currently only extract simple integral equations which could be solvable.
324 arith::Analyzer analyzer;
325 PrimExpr condition = is_true_branch_ ? condition_ : analyzer.Simplify(!condition_);
326 Array<PrimExpr> equations;
327 Array<Var> vars;
328 std::function<void(const PrimExpr&)> fvisit = [&equations, &vars, &fvisit](const PrimExpr& e) {
329 if (e->IsInstance<GENode>() || e->IsInstance<GTNode>() || e->IsInstance<LENode>() ||
330 e->IsInstance<LTNode>() || e->IsInstance<EQNode>() || e->IsInstance<NENode>()) {
331 bool is_simple = true;
332 std::vector<Var> cand_vars;
333 PostOrderVisit(e, [&cand_vars, &is_simple, &e](const ObjectRef& obj) {
334 if (obj.same_as(e)) {
335 return;
336 } else if (const VarNode* var = obj.as<VarNode>()) {
337 if (var->dtype.is_int() || var->dtype.is_uint()) {
338 cand_vars.push_back(GetRef<Var>(var));
339 }
340 } else {
341 is_simple &= obj->IsInstance<AddNode>() || obj->IsInstance<SubNode>() ||
342 obj->IsInstance<MulNode>() || obj->IsInstance<FloorDivNode>() ||
343 obj->IsInstance<FloorModNode>() || obj->IsInstance<IntImmNode>();
344 }
345 });
346 if (is_simple && !cand_vars.empty()) {
347 for (const Var& new_var : cand_vars) {
348 if (!std::any_of(vars.begin(), vars.end(),
349 [&new_var](const Var& v) { return v.same_as(new_var); })) {
350 vars.push_back(new_var);
351 }
352 }
353 equations.push_back(Downcast<PrimExpr>(e));
354 }
355 } else if (e->IsInstance<AndNode>()) {
356 And op = Downcast<And>(e);
357 fvisit(op->a);
358 fvisit(op->b);
359 } else if (e->IsInstance<CallNode>()) {
360 Call op = Downcast<Call>(e);
361 if (op->op.same_as(builtin::likely())) {
362 fvisit(op->args[0]);
363 }
364 }
365 };
366 fvisit(condition);
367 if (equations.empty() || vars.empty()) {
368 return Map<Var, Range>();
369 }
370 // build dom ranges for related vars
371 Map<Var, Range> ranges;
372 for (const Var& v : vars) {
373 arith::IntSet dom;
374 auto relax_it = relax_map_->find(v.get());
375 if (relax_it != relax_map_->end()) {
376 dom = relax_it->second;
377 } else {
378 auto hint_it = hint_map_->find(v.get());
379 if (hint_it != hint_map_->end()) {
380 dom = hint_it->second;
381 }
382 }
383 if (dom.defined()) {
384 ranges.Set(v, Range::FromMinExtent(dom.min(), analyzer.Simplify(dom.max() - dom.min() + 1)));
385 }
386 }
387 // solve constraints
388 arith::IntConstraints constraint(vars, ranges, equations);
389 auto result = arith::SolveInequalitiesToRange(constraint);
390 return result->ranges;
391}

393ConditionalBoundsContext::ConditionalBoundsContext(
394 const PrimExpr& condition, std::unordered_map<const VarNode*, arith::IntSet>* relax_map,
395 std::unordered_map<const VarNode*, arith::IntSet>* hint_map, bool is_true_branch)
396 : condition_(condition),
397 relax_map_(relax_map),
398 hint_map_(hint_map),
399 is_true_branch_(is_true_branch) {}
400
401void ConditionalBoundsContext::EnterWithScope() {
402 for (const auto& p : GetVarBoundsFromCondition()) {
403 const auto* var = p.first.get();
404 arith::IntSet new_dom = arith::IntSet::FromRange(p.second);
405 auto relax_it = relax_map_->find(var);
406 if (relax_it != relax_map_->end()) {
407 // this is a bound for relaxed var
408 origin_map_.emplace(var, relax_it->second);
409 relax_it->second = arith::Intersect({relax_it->second, new_dom});
410 } else {
411 // this is a bound for free var
412 auto hint_it = hint_map_->find(var);
413 if (hint_it != hint_map_->end()) {
414 origin_map_.emplace(var, hint_it->second);
415 hint_it->second = arith::Intersect({hint_it->second, new_dom});
416 } else {
417 origin_map_.emplace(var, arith::IntSet::Nothing());
418 hint_map_->insert(hint_it, {var, new_dom});
419 }
420 }
421 }
422}
423
424void ConditionalBoundsContext::ExitWithScope() {
425 for (const auto& p : origin_map_) {
426 const auto* var = p.first;
427 auto relax_it = relax_map_->find(var);
428 if (relax_it != relax_map_->end()) {
429 // recover bound for relaxed var
430 relax_it->second = p.second;
431 } else {
432 // recover bound for free var
433 auto hint_it = hint_map_->find(var);
434 ICHECK(hint_it != hint_map_->end());
435 if (p.second.IsNothing()) {
436 hint_map_->erase(hint_it);
437 } else {
438 hint_it->second = p.second;
439 }
440 }
441 }
442}

444std::pair<PrimExpr, PrimExpr> GetAsyncWaitAttributes(const AttrStmtNode* op) {
445 ICHECK(op && op->attr_key == tir::attr::async_wait_queue_scope);
446 auto inner = op->body.as<AttrStmtNode>();
447 ICHECK(inner && inner->attr_key == tir::attr::async_wait_inflight_count);
448 return std::make_pair(op->value, inner->value);
449}
