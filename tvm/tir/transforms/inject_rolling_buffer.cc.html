20/*!
21 * \file inject_rolling_buffer.cc
22 * \brief Inject rolling buffer statements.
23
24 Rolling buffers are buffers where one of the dimensions has been made into
25 a circular buffer. Two optimizations are implemented in order to accomplish
26 this: sliding window and storage folding. In particular, the sliding window
27 optimization is applied to the entire buffer (to avoid recomputing elements)
28 and storage folding is then applied to just the rolling dimension.
29
30 Rolling buffers must be inside a loop with only part of the buffer used per
31 iteration. The outermost axis will be rolled over.
32
33 For more information, see the RFC:
34 https://discuss.tvm.apache.org/t/rfc-introducing-a-rolling-buffer-scheduling-primitive/9836
35 */
36#include <tvm/arith/analyzer.h>
37#include <tvm/runtime/registry.h>
38#include <tvm/tir/stmt_functor.h>
39#include <tvm/tir/transform.h>
40
41#include "ir_utils.h"

43namespace tvm {
44namespace tir {

46using arith::IntSet;
47
48struct RollingBufferInfo {
49 int rolling_axis;
50 int rolling_extent;
51 std::vector<int> axis_overlaps;
52 std::vector<Optional<Var>> axis_iter_vars;
53};
54
55class RollingBufferInjector : public StmtExprMutator {
56 std::vector<For> for_loops{};
57 std::set<Buffer> rolling_buffers{};
58 std::map<Buffer, BufferRealize> buffer_to_buffer_realize{};
59 std::map<Buffer, std::vector<AttrStmt>> buffer_to_attrs{};
60 std::map<Buffer, RollingBufferInfo> rolling_buffer_to_info{};
61 // The actual key type is Var, ObjectRef has been used because
62 // of the ambiguous overload for ‘operator<’
63 std::map<ObjectRef, std::vector<BufferRealize>> hoist_buffer_to_for{};
64
65 public:
66 RollingBufferInjector() {}
67
68 Stmt Inject(Stmt stmt) { return ConvertSSA(operator()(std::move(stmt))); }
69
70 Stmt VisitStmt_(const ForNode* op) final {
71 // Manage the stack of iter_vars
72 for_loops.push_back(GetRef<For>(op));
73
74 auto stmt{StmtExprMutator::VisitStmt_(op)};
75 op = stmt.as<ForNode>();
76
77 // Manage the stack of iter_vars
78 for_loops.pop_back();
79
80 auto it{hoist_buffer_to_for.find(op->loop_var)};
81 if (it != hoist_buffer_to_for.end()) {
82 // If the loop corresponds to an iter_var that needs a BufferRealize
83 // hoisting to its scope, perform the hoisting
84 Stmt body{GetRef<For>(op)};
85 for (auto realise : it->second) {
86 auto attrs{buffer_to_attrs[realise->buffer]};
87 Stmt new_realize{BufferRealize(realise->buffer, realise->bounds, realise->condition, body,
88 realise->span)};
89 // The attributes attached to the BufferRealize need hoisting too
90 for (auto attr : attrs) {
91 if (attr->attr_key == attr::rolling_buffer_scope) {
92 continue;
93 }
94 new_realize = AttrStmt(attr->node, attr->attr_key, attr->value, new_realize, attr->span);
95 }
96 body = new_realize;
97 }
98 return body;
99 } else {
100 return stmt;
101 }
102 }
103
104 Stmt VisitStmt_(const AttrStmtNode* op) final {
105 if (auto b = op->node.as<BufferNode>()) {
106 auto buffer = GetRef<Buffer>(b);
107 // Keep a dictionary associating attribute statements with the buffers
108 // they reference. We'll need this if the buffer gets hoisted and we
109 // need to hoist all of its attributes at the same time.
110 buffer_to_attrs[buffer].push_back(GetRef<AttrStmt>(op));
111
112 if (op->attr_key == attr::rolling_buffer_scope && Downcast<IntImm>(op->value)->value) {
113 // If the attribute is indicating that a buffer should be a rolling
114 // buffer, then update the rolling_buffers set to include the buffer
115 rolling_buffers.insert(buffer);
116
117 auto it{buffer_to_buffer_realize.find(buffer)};
118 ICHECK(it != buffer_to_buffer_realize.end())
119 << "Rolling buffer injection failed: no BufferRealize found";
120 BufferRealize buffer_realize = it->second;
121
122 // If a BufferRealize has been identified as needing to be made into
123 // a rolling buffer, begin the analysis.
124 std::vector<Optional<Var>> bound_iter_vars{};
125 std::vector<int> bound_overlaps{};
126 // We use the bound information of the BufferRealize to calculate
127 // how we can legally roll
128 auto stride{0};
129 Optional<Var> iter_var{};
130 for (auto bound : buffer_realize->bounds) {
131 if (auto floor_div = bound->min.as<FloorDivNode>()) {
132 // Handle the case of fractional strides
133 // They take this form: floordiv(hh.outer, 2)
134 // Strip the floordiv and keep track of the divisor
135 auto divisor{Downcast<IntImm>(floor_div->b)->value};
136 bound = Range::FromMinExtent(floor_div->a, bound->extent, bound->span);
137 stride = std::ceil(stride / divisor);
138 }
139 if (bound->min.as<IntImmNode>()) {
140 // If the bound is an int, we can't roll over it
141 iter_var = nullptr;
142 } else if (auto var = bound->min.as<VarNode>()) {
143 // If the bound is just a Var, that implies the stride is 1
144 iter_var = GetRef<Var>(var);
145 stride = 1;
146 } else {
147 // Otherwise, it's the iter var multiplied by the stride
148 // If not we're in unknown behaviour, so assert
149 auto mul = bound->min.as<MulNode>();
150 ICHECK(mul) << "Rolling buffer injection failed: the buffer striding is unsupported";
151 auto a = mul->a.as<VarNode>();
152 ICHECK(a) << "Rolling buffer injection failed: the buffer striding is unsupported";
153 auto b = mul->b.as<IntImmNode>();
154 ICHECK(b) << "Rolling buffer injection failed: the buffer striding is unsupported";
155 iter_var = GetRef<Var>(a);
156 stride = b->value;
157 }
158 bound_iter_vars.push_back(iter_var);
159 if (iter_var) {
160 bound_overlaps.push_back(Downcast<IntImm>(bound->extent)->value - stride);
161 } else {
162 bound_overlaps.push_back(0);
163 }
164 }
165 // Pick the outermost iter_var that's mentioned in the bounds
166 // to be the rolling axis
167 Optional<Var> roll_iter_var{};
168 int roll_axis{1};
169 for (auto loop : for_loops) {
170 auto loop_var{loop->loop_var};
171 iter_var = loop_var;
172
173 auto it{std::find_if(
174 bound_iter_vars.begin(), bound_iter_vars.end(),
175 [&](Optional<Var> var) { return var && (var.get() == loop_var.get()); })};
176
177 if (it != bound_iter_vars.end()) {
178 auto i{std::distance(bound_iter_vars.begin(), it)};
179 roll_iter_var = loop_var;
180 roll_axis = i;
181 break;
182 }
183 }
184 // We must have found an axis to roll over
185 ICHECK(roll_iter_var) << "Rolling buffer injection failed: no rolling axis found";
186 ICHECK(roll_axis != -1) << "Rolling buffer injection failed: no rolling axis found";
187
188 RollingBufferInfo rolling_buffer_info = {
189 roll_axis,
190 static_cast<int>(Downcast<IntImm>(buffer_realize->bounds[roll_axis]->extent)->value),
191 bound_overlaps,
192 bound_iter_vars,
193 };
194 rolling_buffer_to_info[buffer] = rolling_buffer_info;
195 Array<Range> new_bounds{};
196 auto shape{buffer->shape};
197 for (size_t i{0}; i < shape.size(); ++i) {
198 auto extent{shape[i]};
199 if (static_cast<int>(i) == rolling_buffer_info.rolling_axis) {
200 new_bounds.push_back(Range(0, rolling_buffer_info.rolling_extent));
201 } else {
202 new_bounds.push_back(Range(0, extent));
203 }
204 }
205 BufferRealize new_realize{BufferRealize(buffer, new_bounds, buffer_realize->condition,
206 buffer_realize->body, buffer_realize->span)};
207 hoist_buffer_to_for[iter_var.value()].push_back(new_realize);
208 }
209 }
210
211 auto stmt{StmtExprMutator::VisitStmt_(op)};
212 op = stmt.as<AttrStmtNode>();
213
214 if (rolling_buffers.count(GetRef<Buffer>(op->node.as<BufferNode>()))) {
215 // Remove the attribute statements attached to rolling buffers
216 // because they will have been hoisted to the relevant rolling
217 // scope
218 return op->body;
219 } else {
220 return stmt;
221 }
222 }
223
224 Stmt VisitStmt_(const BufferRealizeNode* op) final {
225 buffer_to_buffer_realize.insert({op->buffer, GetRef<BufferRealize>(op)});
226
227 auto stmt{StmtExprMutator::VisitStmt_(op)};
228 op = stmt.as<BufferRealizeNode>();
229
230 if (rolling_buffers.count(op->buffer)) {
231 // Remove the original BufferRealize for rolling buffers
232 // because they will have been hoisted to the relevant rolling
233 // scope
234 return op->body;
235 } else {
236 return stmt;
237 }
238 }
239
240 Stmt VisitStmt_(const BufferStoreNode* op) final {
241 auto stmt{StmtExprMutator::VisitStmt_(op)};
242 op = stmt.as<BufferStoreNode>();
243
244 auto it{rolling_buffer_to_info.find(op->buffer)};
245 if (it != rolling_buffer_to_info.end()) {
246 auto rolling_buffer_info{it->second};
247 std::vector<PrimExpr> indices{};
248 // First modify the access indices to use modulo arithmetic
249 // for the rolling axis
250 for (size_t i{0}; i < op->indices.size(); ++i) {
251 auto index{op->indices[i]};
252 if (static_cast<int>(i) == rolling_buffer_info.rolling_axis) {
253 indices.push_back(FloorMod(index, rolling_buffer_info.rolling_extent));
254 } else {
255 indices.push_back(index);
256 }
257 }
258 Stmt buffer_store = BufferStore(op->buffer, op->value, indices, op->span);
259 // Then wrap the BufferStores in some Ifs to avoid recomputing elements
260 for (size_t i{0}; i < rolling_buffer_info.axis_iter_vars.size(); ++i) {
261 auto iter_var{rolling_buffer_info.axis_iter_vars[i]};
262 if (iter_var && rolling_buffer_info.axis_overlaps[i] > 0) {
263 Var var{iter_var.value()};
264 const Map<Var, IntSet> dmap{std::make_pair(var, IntSet::Interval(0, 0))};
265 auto term_2{arith::Analyzer{}.int_set(op->indices[i], dmap).min()};
266 auto condition = Or(LT(var, 1), GE(term_2, rolling_buffer_info.axis_overlaps[i]));
267 buffer_store = IfThenElse(likely(condition), buffer_store);
268 }
269 }
270 return buffer_store;
271 } else {
272 return stmt;
273 }
274 }
275
276 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
277 auto expr{StmtExprMutator::VisitExpr_(op)};
278 op = expr.as<BufferLoadNode>();
279
280 auto it{rolling_buffer_to_info.find(op->buffer)};
281 if (it != rolling_buffer_to_info.end()) {
282 auto rolling_buffer_info{it->second};
283 std::vector<PrimExpr> indices{};
284 // Modify the access indices to use modulo arithmetic
285 // for the rolling axis
286 for (size_t i{0}; i < op->indices.size(); ++i) {
287 auto index{op->indices[i]};
288 if (static_cast<int>(i) == rolling_buffer_info.rolling_axis) {
289 indices.push_back(FloorMod(index, rolling_buffer_info.rolling_extent));
290 } else {
291 indices.push_back(index);
292 }
293 }
294 return BufferLoad(op->buffer, indices, op->span);
295 } else {
296 return expr;
297 }
298 }
299}; // namespace tir

301namespace transform {
302
303Pass InjectRollingBuffer() {
304 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
305 auto* n = f.CopyOnWrite();
306 n->body = RollingBufferInjector().Inject(std::move(n->body));
307 return f;
308 };
309 return CreatePrimFuncPass(pass_func, 0, "tir.InjectRollingBuffer", {});
310}
