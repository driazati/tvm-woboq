20/*!
21 * \file convert_block_to_opaque.cc
22 * \brief Convert the blocks to opaque blocks which do not have block vars.
23 */
24
25#include <tvm/tir/stmt_functor.h>
26#include <tvm/tir/transform.h>
27
28#include "ir_utils.h"
29
30namespace tvm {
31namespace tir {
32
33/*!
34 * \brief Substitute expr via BlockRealize value bindings and convert each block into opaque
35 * blocks.
36 */
37class OpaqueBlockConverter : public StmtExprMutator {
38 public:
39 static Stmt Substitute(const PrimFunc& f) {
40 OpaqueBlockConverter substituter;
41 return substituter.VisitStmt(f->body);
42 }
43
44 private:
45 OpaqueBlockConverter() = default;
46
47 PrimExpr VisitExpr_(const VarNode* var) final {
48 auto it = var_substitutes_.find(var);
49 if (it != var_substitutes_.end()) {
50 return it->second;
51 }
52 return GetRef<Var>(var);
53 }
54
55 Stmt VisitStmt_(const BlockNode* block) final {
56 ICHECK(!block->init.defined())
57 << "Block Init part is not allowed in pass ConvertBlocksToOpaque";
58 Block new_block = Downcast<Block>(StmtExprMutator::VisitStmt_(block));
59 if (!new_block->iter_vars.empty()) {
60 new_block.CopyOnWrite()->iter_vars.clear();
61 }
62 return std::move(new_block);
63 }
64
65 Stmt VisitStmt_(const BlockRealizeNode* realize) final {
66 const auto* block_op = realize->block.get();
67 ICHECK(!block_op->init.defined());
68 // Step 1. Update "block vars => binding values" for substitution.
69 ICHECK_EQ(block_op->iter_vars.size(), realize->iter_values.size());
70 for (int i = 0, n = block_op->iter_vars.size(); i < n; ++i) {
71 IterVar block_var = block_op->iter_vars[i];
72 PrimExpr v = this->VisitExpr(realize->iter_values[i]);
73 var_substitutes_.emplace(block_var->var.get(), v);
74 }
75 // Step 2. Visit recursively.
76 BlockRealize new_realize = Downcast<BlockRealize>(StmtExprMutator::VisitStmt_(realize));
77 if (!new_realize->iter_values.empty()) {
78 new_realize.CopyOnWrite()->iter_values.clear();
79 }
80 return std::move(new_realize);
81 }
82
83 /*! \brief The map from block vars to their binding values. */
84 std::unordered_map<const VarNode*, PrimExpr> var_substitutes_;
85};
86
87PrimFunc ConvertBlocksToOpaque(PrimFunc f) {
88 // Only apply this pass to TIR that is not from TE schedules
89 if (!IsFromLegacyTESchedule(f)) {
90 PrimFuncNode* fptr = f.CopyOnWrite();
91 fptr->body = OpaqueBlockConverter::Substitute(f);
92 return f;
93 } else {
94 return f;
95 }
96}
97
98namespace transform {
99
100Pass ConvertBlocksToOpaque() {
101 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
102 return ConvertBlocksToOpaque(std::move(f));
103 };
104 return CreatePrimFuncPass(pass_func, 0, "tir.ConvertBlocksToOpaque", {});
105}

107TVM_REGISTER_GLOBAL("tir.transform.ConvertBlocksToOpaque").set_body_typed(ConvertBlocksToOpaque);
