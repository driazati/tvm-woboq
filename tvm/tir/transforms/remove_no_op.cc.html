39// Mark the statement of each stage.
40class NoOpRemover : public StmtMutator {
41 public:
42 Stmt VisitStmt_(const LetStmtNode* op) final {
43 Stmt stmt = StmtMutator::VisitStmt_(op);
44 op = stmt.as<LetStmtNode>();
45 return is_no_op(op->body) ? MakeEvaluate(op->value) : stmt;
46 }
47 Stmt VisitStmt_(const AttrStmtNode* op) final {
48 if (op->attr_key == "pragma_debug_skip_region") {
49 return MakeEvaluate(0);
50 } else if (op->attr_key == attr::async_wait_queue_scope) {
51 auto wait_attrs = GetAsyncWaitAttributes(op);
52 auto wait_cnt = wait_attrs.second;
53 arith::Analyzer ana;
54 if (ana.CanProve(wait_cnt < 0)) {
55 // A negative wait count can arise if it depends on a loop variable.
56 // For example, a wait count 1 - i can be negative after loop unrolling.
57 // We assume that such wait is a nop.
58 auto inner = op->body.as<AttrStmtNode>();
59 ICHECK(inner);
60 return StmtMutator::VisitStmt(inner->body);
61 }
62 }
63
64 Stmt stmt = StmtMutator::VisitStmt_(op);
65 op = stmt.as<AttrStmtNode>();
66 return is_no_op(op->body) ? MakeEvaluate(op->value) : stmt;
67 }
68 Stmt VisitStmt_(const IfThenElseNode* op) final {
69 Stmt stmt = StmtMutator::VisitStmt_(op);
70 op = stmt.as<IfThenElseNode>();
71 if (op->else_case.defined()) {
72 if (is_no_op(op->else_case)) {
73 if (is_no_op(op->then_case)) {
74 return MakeEvaluate(op->condition);
75 } else {
76 return IfThenElse(op->condition, op->then_case);
77 }
78 } else {
79 return stmt;
80 }
81 } else {
82 if (is_no_op(op->then_case)) {
83 return MakeEvaluate(op->condition);
84 } else {
85 return stmt;
86 }
87 }
88 }
89 Stmt VisitStmt_(const ForNode* op) final {
90 Stmt stmt = StmtMutator::VisitStmt_(op);
91 op = stmt.as<ForNode>();
92 if (is_zero(op->extent)) {
93 return Evaluate(0);
94 }
95 return is_no_op(op->body) ? MakeEvaluate({op->min, op->extent}) : stmt;
96 }
97 Stmt VisitStmt_(const AllocateNode* op) final {
98 Stmt stmt = StmtMutator::VisitStmt_(op);
99 op = stmt.as<AllocateNode>();
100 return is_no_op(op->body) ? MakeEvaluate(op->extents) : stmt;
101 }
102
103 Stmt VisitStmt_(const ProducerRealizeNode* op) final {
104 Stmt stmt = StmtMutator::VisitStmt_(op);
105 op = stmt.as<ProducerRealizeNode>();
106 return is_no_op(op->body) ? op->body : stmt;
107 }
108 Stmt VisitStmt_(const EvaluateNode* op) final {
109 if (SideEffect(op->value) > CallEffectKind::kReadState) return GetRef<Stmt>(op);
110 return Evaluate(0);
111 }
112
113 Stmt VisitStmt_(const SeqStmtNode* op) final {
114 Stmt ret = StmtMutator::VisitSeqStmt_(op, true);
115 op = ret.as<SeqStmtNode>();
116 ICHECK(op != nullptr);
117 bool need_compact = false;
118 for (size_t i = 0; i < op->size(); ++i) {
119 if (is_no_op(op->seq[i])) need_compact = true;
120 }
121 if (need_compact) {
122 auto n = CopyOnWrite(op);
123 size_t top = 0;
124 for (size_t i = 0; i < n->seq.size(); ++i) {
125 if (!is_no_op(n->seq[i])) {
126 n->seq.Set(top++, n->seq[i]);
127 }
128 }
129 if (top == 1) {
130 return n->seq[0];
131 } else {
132 n->seq.resize(top);
133 return Stmt(n);
134 }
135 } else {
136 if (op->size() == 1) {
137 return op->seq[0];
138 } else {
139 return ret;
140 }
141 }
142 }
143
144 private:
145 Stmt MakeEvaluate(PrimExpr value) {
146 if (SideEffect(value) > CallEffectKind::kReadState) {
147 return Evaluate(value);
148 } else {
149 return Evaluate(0);
150 }
151 }
152 Stmt MakeEvaluate(const Array<PrimExpr>& values) {
153 Stmt stmt;
154 for (PrimExpr e : values) {
155 if (SideEffect(e) > CallEffectKind::kReadState) {
156 if (stmt.defined()) {
157 stmt = SeqStmt({stmt, Evaluate(e)});
158 } else {
159 stmt = Evaluate(e);
160 }
161 }
162 }
163 return stmt.defined() ? stmt : Evaluate(0);
164 }
165};
166
167Stmt RemoveNoOp(Stmt stmt) { return NoOpRemover()(std::move(stmt)); }
168
169namespace transform {
170
171Pass RemoveNoOp() {
172 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
173 auto* n = f.CopyOnWrite();
174 n->body = NoOpRemover()(std::move(n->body));
175 return f;
176 };
177 return CreatePrimFuncPass(pass_func, 0, "tir.RemoveNoOp", {});
178}

180TVM_REGISTER_GLOBAL("tir.transform.RemoveNoOp").set_body_typed(RemoveNoOp);
