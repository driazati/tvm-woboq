51// Detect double buffer variables.
52class DoubleBufferDetector : public StmtExprVisitor {
53 public:
54 void VisitStmt_(const AttrStmtNode* op) final {
55 if (op->attr_key == attr::double_buffer_scope) {
56 touched_.insert(op->node.as<VarNode>());
57 StmtExprVisitor::VisitStmt_(op);
58 } else {
59 StmtExprVisitor::VisitStmt_(op);
60 }
61 }
62
63 void VisitExpr_(const VarNode* op) final {
64 if (touched_.count(op)) {
65 touched_.erase(op);
66 }
67 }
68 // The set of touched variable.
69 std::unordered_set<const VarNode*> touched_;
70};
71
72class StripDoubleBufferWrite : public StmtMutator {
73 public:
74 Stmt VisitStmt_(const AttrStmtNode* op) final {
75 if (op->attr_key == attr::double_buffer_write) {
76 return VisitStmt(op->body);
77 } else {
78 return StmtMutator::VisitStmt_(op);
79 }
80 }
81};
82
83class DoubleBufferInjector : public StmtExprMutator {
84 public:
85 explicit DoubleBufferInjector(int split_loop) : split_loop_(split_loop) {}
86
87 Stmt Inject(Stmt stmt) {
88 DoubleBufferDetector detector;
89 detector(stmt);
90 if (detector.touched_.empty()) return stmt;
91 for (const VarNode* v : detector.touched_) {
92 dbuffer_info_[v] = StorageEntry();
93 }
94 return ConvertSSA(operator()(std::move(stmt)));
95 }
96
97 Stmt VisitStmt_(const AttrStmtNode* op) final {
98 if (op->attr_key == attr::double_buffer_scope) {
99 return MakeProducer(op);
100 } else {
101 return StmtExprMutator::VisitStmt_(op);
102 }
103 }
104
105 Stmt VisitStmt_(const AllocateNode* op) final {
106 const VarNode* buf = op->buffer_var.as<VarNode>();
107 auto it = dbuffer_info_.find(buf);
108 if (it != dbuffer_info_.end()) {
109 it->second.scope = GetPtrStorageScope(op->buffer_var);
110
111 ICHECK_EQ(op->extents.size(), 1) << "InjectDoubleBuffer expects flat 1-d buffers. "
112 << "Has StorageFlatten (TE-based schedules) or "
113 << "FlattenBuffer (TIR-based schedules) been run?";
114 it->second.stride = op->extents[0];
115 Stmt stmt = StmtExprMutator::VisitStmt_(op);
116 op = stmt.as<AllocateNode>();
117
118 Array<PrimExpr> new_extents = {op->extents[0] * make_const(op->extents[0].dtype(), 2)};
119 ICHECK(it->second.loop != nullptr);
120 auto& alloc_nest = loop_allocs_[it->second.loop];
121 alloc_nest.emplace_back(
122 Allocate(op->buffer_var, op->dtype, new_extents, op->condition, Evaluate(0)));
123 return op->body;
124 } else {
125 return StmtExprMutator::VisitStmt_(op);
126 }
127 }
128
129 Stmt VisitStmt_(const ForNode* op) final {
130 loop_nest_.push_back(op);
131 Stmt stmt = StmtExprMutator::VisitStmt_(op);
132 auto it = loop_pre_.find(op);
133 if (it != loop_pre_.end()) {
134 const ForNode* old_loop = stmt.as<ForNode>();
135 if (split_loop_ != 0) {
136 // Explicitly unroll the loop
137 ICHECK(split_loop_ % 2 == 0 || split_loop_ == 1)
138 << "It is better to split with multiple of 2";
139 ICHECK(is_zero(old_loop->min));
140 PrimExpr zero = old_loop->min;
141 PrimExpr new_ext = old_loop->extent - make_const(old_loop->loop_var.dtype(), 1);
142 PrimExpr factor = make_const(new_ext.dtype(), split_loop_);
143 PrimExpr outer_ext = new_ext / factor;
144 PrimExpr tail_base = outer_ext * factor;
145 Var outer_var(old_loop->loop_var->name_hint + ".outer", old_loop->loop_var.dtype());
146 std::unordered_map<const VarNode*, PrimExpr> vmap;
147 std::vector<Stmt> loop_seq;
148 for (int32_t i = 0; i < split_loop_; ++i) {
149 vmap[old_loop->loop_var.get()] = outer_var * factor + make_const(factor.dtype(), i);
150 loop_seq.emplace_back(Substitute(old_loop->body, vmap));
151 }
152 Stmt loop = For(outer_var, zero, outer_ext, old_loop->kind, SeqStmt::Flatten(loop_seq));
153 // tail
154 std::vector<Stmt> tail_seq;
155 Stmt tail_body = StripDoubleBufferWrite()(old_loop->body);
156 for (int32_t i = 0; i < split_loop_; ++i) {
157 PrimExpr idx = tail_base + make_const(tail_base.dtype(), i);
158 vmap[old_loop->loop_var.get()] = idx;
159 tail_seq.emplace_back(IfThenElse(idx < old_loop->extent, Substitute(tail_body, vmap)));
160 }
161 stmt = SeqStmt::Flatten(loop, tail_seq);
162 }
163 stmt = SeqStmt::Flatten(it->second, stmt);
164 }
165 it = loop_allocs_.find(op);
166 if (it != loop_allocs_.end()) {
167 stmt = MergeNest(it->second, stmt);
168 }
169 loop_nest_.pop_back();
170 return stmt;
171 }

173 PrimExpr VisitExpr_(const LoadNode* op) final {
174 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
175 return PrimExpr();
176 }
177
178 Stmt VisitStmt_(const StoreNode* op) final {
179 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
180 return Stmt();
181 }
182
183 Stmt VisitStmt_(const BufferStoreNode* op) final {
184 auto node = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
185
186 auto it = dbuffer_info_.find(node->buffer->data.get());
187 if (it != dbuffer_info_.end()) {
188 const StorageEntry& e = it->second;
189 ICHECK(in_double_buffer_scope_);
190 ICHECK(e.switch_write_var.defined());
191
192 ICHECK_EQ(node->indices.size(), 1) << "InjectDoubleBuffer expects flat 1-d buffers. "
193 << "Has StorageFlatten (TE-based schedules) or "
194 << "FlattenBuffer (TIR-based schedules) been run?";
195
196 auto writer = node.CopyOnWrite();
197 writer->buffer = GetRemappedBuffer(node->buffer, e.stride);
198 writer->indices = {e.switch_write_var * e.stride + node->indices[0]};
199 }
200
201 return std::move(node);
202 }
203
204 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
205 auto node = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
206
207 auto it = dbuffer_info_.find(node->buffer->data.get());
208 if (it != dbuffer_info_.end()) {
209 const StorageEntry& e = it->second;
210 ICHECK(e.switch_read_var.defined());
211
212 ICHECK_EQ(node->indices.size(), 1) << "InjectDoubleBuffer expects flat 1-d buffers. "
213 << "Has StorageFlatten (TE-based schedules) or "
214 << "FlattenBuffer (TIR-based schedules) been run?";
215
216 auto writer = node.CopyOnWrite();
217 writer->buffer = GetRemappedBuffer(node->buffer, e.stride);
218 writer->indices = {e.switch_read_var * e.stride + node->indices[0]};
219 }
220
221 return std::move(node);
222 }
223
224 Buffer GetRemappedBuffer(Buffer buf, PrimExpr stride) {
225 auto key = buf.get();
226 auto it = buf_remap_.find(key);
227 if (it != buf_remap_.end()) {
228 return it->second;
229 }
230
231 ICHECK(stride.defined());
232 // TODO(Lunderberg): Move this pass to before
233 // StorageFlatten/FlattenBuffer. That will simplify the
234 // implementation, to be the insertion of a new dimension for the
235 // buffer, rather than adjusting the other indices.
236 ICHECK_EQ(buf->shape.size(), 1) << "InjectDoubleBuffer expects flat 1-d buffers. "
237 << "Has StorageFlatten (TE-based schedules) or "
238 << "FlattenBuffer (TIR-based schedules) been run?";
239 auto writer = buf.CopyOnWrite();
240 writer->shape = {buf->shape[0] * stride};
241
242 buf_remap_[key] = buf;
243 return buf;
244 }
245
246 PrimExpr VisitExpr_(const VarNode* op) final {
247 ICHECK(!dbuffer_info_.count(op));
248 return GetRef<PrimExpr>(op);
249 }
250
251 private:
252 Stmt MakeProducer(const AttrStmtNode* op) {
253 const Var buffer = Downcast<Var>(op->node);
254 ICHECK_NE(loop_nest_.size(), 0U) << "Double buffer scope must be inside a loop";
255 auto it = dbuffer_info_.find(buffer.get());
256 if (it == dbuffer_info_.end()) {
257 LOG(WARNING) << "Skip double buffer scope " << op->node;
258 return this->VisitStmt(op->body);
259 }
260 StorageEntry& e = it->second;
261 e.loop = loop_nest_.back();
262 PrimExpr zero = make_const(e.loop->loop_var.dtype(), 0);
263 PrimExpr one = make_const(e.loop->loop_var.dtype(), 1);
264 PrimExpr two = make_const(e.loop->loop_var.dtype(), 2);
265 PrimExpr loop_shift = e.loop->loop_var + one;
266 e.switch_write_var = Var(e.loop->loop_var->name_hint + ".db", e.loop->loop_var.dtype());
267 e.switch_read_var = indexmod(e.loop->loop_var, two);
268 in_double_buffer_scope_ = true;
269 Stmt body = this->VisitStmt(op->body);
270 in_double_buffer_scope_ = false;
271 std::unordered_map<const VarNode*, PrimExpr> vmap;
272 vmap[e.switch_write_var.get()] = zero;
273 vmap[e.loop->loop_var.get()] = zero;
274 loop_pre_[e.loop].emplace_back(Substitute(body, vmap));
275 vmap[e.loop->loop_var.get()] = loop_shift;
276 vmap[e.switch_write_var.get()] = indexmod(loop_shift, two);
277 body = Substitute(body, vmap);
278 body = AttrStmt(buffer, attr::double_buffer_write, 1, body);
279 body = IfThenElse(loop_shift < e.loop->extent, body);
280 return body;
281 }
282 // Storage entry for those who need double buffering.
283 struct StorageEntry {
284 // The size of the buffer
285 PrimExpr stride;
286 // The loop we need
287 const ForNode* loop{nullptr};
288 // The switch variable.
289 Var switch_write_var;
290 // The switch variable for reading.
291 PrimExpr switch_read_var;
292 // The storage scope.
293 std::string scope;
294 };
295 // Whether split loop
296 int32_t split_loop_;
297 // Whether we are inside double buffer scope.
298 bool in_double_buffer_scope_{false};
299 // The current loop next
300 std::vector<const ForNode*> loop_nest_;
301 // The allocs to be appended before the loop
302 std::unordered_map<const ForNode*, std::vector<Stmt> > loop_allocs_;
303 // The stmt to be appended before the loop
304 std::unordered_map<const ForNode*, std::vector<Stmt> > loop_pre_;
305 // The allocation size of the buffer
306 std::unordered_map<const VarNode*, StorageEntry> dbuffer_info_;
307 // The updated Buffer objects
308 std::unordered_map<const BufferNode*, Buffer> buf_remap_;
309};

313Pass InjectDoubleBuffer() {
314 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
315 auto* n = f.CopyOnWrite();
316 auto cfg = ctx->GetConfig<InjectDoubleBufferConfig>("tir.InjectDoubleBuffer");
317 if (!cfg.defined()) {
318 cfg = AttrsWithDefaultValues<InjectDoubleBufferConfig>();
319 }
320 n->body = DoubleBufferInjector(cfg.value()->split_loop).Inject(std::move(n->body));
321 return f;
322 };
323 return CreatePrimFuncPass(pass_func, 0, "tir.InjectDoubleBuffer", {});
324}
