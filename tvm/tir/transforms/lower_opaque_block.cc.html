32/*!
33 * \brief Remove Block to ensure that the TIR can not be scheduled again.
34 */
35class OpaqueBlockLower : public StmtExprMutator {
36 private:
37 Stmt VisitStmt_(const BlockRealizeNode* op) final {
38 // We have convert blocks into opaque blocks in previous passes.
39 ICHECK(op->iter_values.empty()) << "Non-opaque blocks are not allowed in FlattenBuffer. Please "
40 "call pass ConvertBlocksToOpaque before.";
41 // Step 1. Visit the body
42 Block new_block = Downcast<Block>(this->VisitStmt(op->block));
43 PrimExpr predicate = this->VisitExpr(op->predicate);
44 // Step 2. Transform the `predicate` to if-then-else
45 Stmt body = new_block->body;
46 if (!is_one(predicate)) {
47 body = IfThenElse(predicate, std::move(body));
48 }
49 // Step 3. Handle allocations in reverse order
50 for (size_t i = new_block->alloc_buffers.size(); i > 0; --i) {
51 const Buffer& buffer = new_block->alloc_buffers[i - 1];
52 Array<PrimExpr> new_shape = buffer->shape;
53 if (buffer->strides.size()) {
54 ICHECK_EQ(buffer->shape.size(), buffer->strides.size());
55 for (size_t i = buffer->strides.size() - 1; i > 0; --i) {
56 ICHECK(is_zero(floormod(buffer->strides[i - 1], buffer->strides[i])));
57 new_shape.Set(i, buffer->strides[i - 1] / buffer->strides[i]);
58 }
59 }
60 body = Allocate(buffer->data, buffer->dtype, new_shape, const_true(), std::move(body));
61 }
62 return body;
63 }
64
65 Stmt VisitStmt_(const ForNode* op) final {
66 // Step 1. Update unit loop info.
67 PrimExpr min = this->VisitExpr(op->min);
68 PrimExpr extent = this->VisitExpr(op->extent);
69 if (is_one(extent) && op->annotations.empty()) {
70 // handling unit loop
71 unit_loop_vars_[op->loop_var] = min;
72 }
73 // Step 2. Visit recursively
74 Stmt body = this->VisitStmt(op->body);
75 // Step 3. Create new For loop accordingly
76 if (op->kind == ForKind::kThreadBinding) {
77 // Case 1. Thread binding
78 ICHECK(op->thread_binding.defined());
79 String thread_tag = op->thread_binding.value()->thread_tag;
80 body = MakeLaunchThread(min, extent, op->loop_var, thread_tag, body);
81 } else if (is_one(extent) && op->annotations.empty()) {
82 // Case 2. Unit loop
83 return body;
84 } else {
85 // Case 3. An ordinary loop
86 body = For(op->loop_var, std::move(min), std::move(extent), op->kind, std::move(body));
87 }
88 // Step 4. Handle annotations
89 std::set<std::string> ordered_ann_keys;
90 for (const auto& annotation : op->annotations) {
91 ordered_ann_keys.insert(annotation.first);
92 }
93 for (auto it = ordered_ann_keys.rbegin(); it != ordered_ann_keys.rend(); ++it) {
94 const std::string& ann_key = *it;
95 const ObjectRef& ann_value = op->annotations.at(ann_key);
96 if (attr::IsPragmaKey(ann_key)) {
97 body =
98 AttrStmt(op->loop_var, ann_key, ConvertAttrValue(ann_key, ann_value), std::move(body));
99 }
100 }
101 return body;
102 }
103
104 PrimExpr VisitExpr_(const VarNode* op) final {
105 Var var = GetRef<Var>(op);
106 auto it = unit_loop_vars_.find(var);
107 if (it == unit_loop_vars_.end()) {
108 return std::move(var);
109 } else {
110 PrimExpr expr = it->second;
111 if (expr.dtype() != var.dtype()) {
112 expr = tvm::cast(var.dtype(), std::move(expr));
113 }
114 return expr;
115 }
116 }
117
118 static Stmt MakeLaunchThread(PrimExpr min, PrimExpr extent, Var var, String thread_tag,
119 Stmt body) {
120 IterVar iter_var(/*dom=*/Range::FromMinExtent(min, extent),
121 /*var=*/std::move(var),
122 /*iter_type=*/IterVarType::kThreadIndex,
123 /*thread_tag=*/thread_tag);
124 String attr_key = (thread_tag == "vthread" || thread_tag == "vthread.x" ||
125 thread_tag == "vthread.y" || thread_tag == "vthread.z")
126 ? attr::virtual_thread
127 : attr::thread_extent;
128 return AttrStmt(/*node=*/std::move(iter_var),
129 /*attr_key=*/std::move(attr_key),
130 /*value=*/std::move(extent),
131 /*body=*/std::move(body));
132 }
133
134 /*! \brief Convert attr value from annotation map into PrimExpr. */
135 PrimExpr ConvertAttrValue(const String& key, const ObjectRef& obj) {
136 if (!obj.defined()) {
137 return PrimExpr();
138 } else if (const PrimExprNode* expr = obj.as<PrimExprNode>()) {
139 return GetRef<PrimExpr>(expr);
140 } else if (const StringObj* str = obj.as<StringObj>()) {
141 return std::move(StringImm(str->data));
142 } else {
143 LOG(FATAL) << "Illegal attribute of key " << key << ", value type " << obj->GetTypeKey()
144 << " not supported";
145 return PrimExpr();
146 }
147 }
148
149 /*! \brief Record the loop_var and loop start value of unit loops, whose extent is one. */
150 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectPtrEqual> unit_loop_vars_;
151};
152
153PrimFunc LowerOpaqueBlock(PrimFunc f) {
154 // Only apply this pass to TIR that is not from TE schedules
155 if (!IsFromLegacyTESchedule(f)) {
156 auto fptr = f.CopyOnWrite();
157 fptr->body = OpaqueBlockLower()(std::move(fptr->body));
158 return f;
159 } else {
160 return f;
161 }
162}

164namespace transform {
165
166Pass LowerOpaqueBlock() {
167 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
168 return LowerOpaqueBlock(std::move(f));
169 };
170 return CreatePrimFuncPass(pass_func, 0, "tir.LowerOpaqueBlock", {});
171}
