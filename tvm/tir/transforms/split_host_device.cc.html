42// use/def analysis, also delete unreferenced lets
43class VarUseDefAnalysis : public StmtExprMutator {
44 public:
45 Stmt VisitStmt_(const AttrStmtNode* op) final {
46 if (op->attr_key == attr::thread_extent) {
47 IterVar iv = Downcast<IterVar>(op->node);
48 ICHECK_NE(iv->thread_tag.length(), 0U);
49 // thread_extent can appear multiple times
50 // use the first appearance as def.
51 if (!use_count_.count(iv->var.get())) {
52 this->HandleDef(iv->var.get());
53 thread_axis_.push_back(iv);
54 thread_extent_.push_back(op->value);
55 }
56
57 PrimExpr value = op->value;
58 if (visit_thread_extent_) {
59 value = this->VisitExpr(value);
60 }
61 Stmt body = this->VisitStmt(op->body);
62 if (value.same_as(op->value) && body.same_as(op->body)) {
63 return GetRef<Stmt>(op);
64 }
65 return AttrStmt(op->node, op->attr_key, value, body);
66 } else {
67 return StmtExprMutator::VisitStmt_(op);
68 }
69 }
70
71 Stmt VisitStmt_(const LetStmtNode* op) final {
72 this->HandleDef(op->var.get());
73 Stmt body = this->VisitStmt(op->body);
74 // eliminate unreferenced let
75 if (use_count_.at(op->var.get()) == 0 && SideEffect(op->value) <= CallEffectKind::kReadState &&
76 simplify_let_) {
77 return body;
78 } else {
79 PrimExpr value = this->VisitExpr(op->value);
80 if (body.same_as(op->body) && value.same_as(op->value)) {
81 return GetRef<Stmt>(op);
82 } else {
83 return LetStmt(op->var, value, body);
84 }
85 }
86 }
87
88 Stmt VisitStmt_(const ForNode* op) final {
89 this->HandleDef(op->loop_var.get());
90 return StmtExprMutator::VisitStmt_(op);
91 }
92
93 Stmt VisitStmt_(const AllocateNode* op) final {
94 this->HandleDef(op->buffer_var.get());
95 auto storage_scope = runtime::StorageScope::Create(GetPtrStorageScope(op->buffer_var));
96 if (storage_scope.rank == runtime::StorageRank::kShared && storage_scope.tag == ".dyn") {
97 ICHECK_EQ(use_dyn_shmem_, false) << "Only one dynamic shared memory allocation is allowed.";
98 ICHECK_GT(op->extents.size(), 0);
99 dyn_shmem_size_ = op->extents[0];
100 for (size_t i = 1; i < op->extents.size(); ++i) {
101 dyn_shmem_size_ *= op->extents[i];
102 }
103 dyn_shmem_size_ = dyn_shmem_size_ * (op->dtype.bytes());
104 use_dyn_shmem_ = true;
105 }
106 return StmtExprMutator::VisitStmt_(op);
107 }
108
109 Stmt VisitStmt_(const AllocateConstNode* op) final {
110 this->HandleDef(op->buffer_var.get());
111 return StmtExprMutator::VisitStmt_(op);
112 }
113
114 Stmt VisitStmt_(const StoreNode* op) final {
115 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
116 return Stmt();
117 }
118
119 Stmt VisitStmt_(const BufferStoreNode* op) final {
120 VisitBuffer(op->buffer);
121 return StmtExprMutator::VisitStmt_(op);
122 }
123
124 PrimExpr VisitExpr_(const LetNode* op) final {
125 // Weaker SSA condition
126 // A single var can be binded in multiple lets
127 // but they have to bind to the same value.
128 // This is used to allow cases when we reuse a single let
129 // expression to construct a nested expr.
130 // (let x = 1 in x + 1) * (let x = 1 in x + 1)
131 auto it = let_binding_.find(op->var);
132 PrimExpr value = this->VisitExpr(op->value);
133 if (it != let_binding_.end()) {
134 ICHECK(deep_equal_(it->second->value, value))
135 << "Let cannot bind the same var to two different values";
136 return GetRef<PrimExpr>(it->second);
137 } else {
138 this->HandleDef(op->var.get());
139 let_binding_[op->var] = op;
140 }
141 PrimExpr body = this->VisitExpr(op->body);
142 // eliminate unreferenced let
143 if (use_count_.at(op->var.get()) == 0 && SideEffect(op->value) <= CallEffectKind::kReadState &&
144 simplify_let_) {
145 return body;
146 } else {
147 if (body.same_as(op->body) && value.same_as(op->value)) {
148 return GetRef<PrimExpr>(op);
149 } else {
150 return Let(op->var, value, body);
151 }
152 }
153 }
154
155 PrimExpr VisitExpr_(const VarNode* op) final {
156 this->HandleUse(GetRef<PrimExpr>(op));
157 return StmtExprMutator::VisitExpr_(op);
158 }
159
160 PrimExpr VisitExpr_(const ReduceNode* op) final {
161 for (const auto& iv : op->axis) {
162 this->HandleDef(iv->var.get());
163 }
164 return StmtExprMutator::VisitExpr_(op);
165 }
166
167 PrimExpr VisitExpr_(const LoadNode* op) final {
168 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
169 return PrimExpr();
170 }
171
172 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
173 VisitBuffer(op->buffer);
174 return StmtExprMutator::VisitExpr_(op);
175 }
176
177 void VisitBuffer(Buffer buffer) {
178 this->HandleUse(buffer->data);
179 auto visit_arr = [&](Array<PrimExpr> arr) {
180 for (const auto& element : arr) {
181 this->VisitExpr(element);
182 }
183 };
184
185 visit_arr(buffer->shape);
186 visit_arr(buffer->strides);
187 }
188
189 void HandleDef(const VarNode* v) {
190 ICHECK(!def_count_.count(v)) << "variable " << v->name_hint
191 << " has already been defined, the Stmt is not SSA";
192 ICHECK(!use_count_.count(v)) << "variable " << v->name_hint
193 << " has been used before definition!";
194 use_count_[v] = 0;
195 def_count_[v] = 1;
196 }
197
198 void HandleUse(const PrimExpr& v) {
199 ICHECK(v.as<VarNode>());
200 Var var = Downcast<Var>(v);
201 auto it = use_count_.find(var.get());
202 if (it != use_count_.end()) {
203 if (it->second >= 0) {
204 ++it->second;
205 }
206 } else {
207 undefined_.push_back(var);
208 use_count_[var.get()] = -1;
209 }
210 }
211
212 // The fields are publically readible to
213 // be accessible to the users.
214 bool visit_thread_extent_{true};
215 bool simplify_let_{true};
216 Array<Var> undefined_;
217 Array<IterVar> thread_axis_;
218 Array<PrimExpr> thread_extent_;
219 PrimExpr dyn_shmem_size_{0};
220 bool use_dyn_shmem_{false};
221 std::unordered_map<const VarNode*, int> use_count_;
222 std::unordered_map<const VarNode*, int> def_count_;
223
224 private:
225 ExprDeepEqual deep_equal_;
226 std::unordered_map<Var, const LetNode*, ObjectPtrHash, ObjectPtrEqual> let_binding_;
227};
228
229Array<Var> UndefinedVars(const Stmt& stmt, const Array<Var>& args) {
230 VarUseDefAnalysis m;
231 m.simplify_let_ = false;
232 for (Var arg : args) {
233 m.use_count_[arg.get()] = 0;
234 }
235 m(stmt);
236 return m.undefined_;
237}
238
239Array<Var> UndefinedVars(const PrimExpr& expr) {
240 VarUseDefAnalysis m;
241 m.simplify_let_ = false;
242 m(expr);
243 return m.undefined_;
244}

246class HostDeviceSplitter : public StmtMutator {
247 public:
248 explicit HostDeviceSplitter(IRModule* device_mod, Target device_target, std::string name_prefix)
249 : device_mod_(device_mod), device_target_(device_target), name_prefix_(name_prefix) {}
250
251 Stmt VisitStmt_(const AllocateNode* op) final {
252 handle_data_type_[op->buffer_var.get()] = make_const(op->dtype, 0);
253 return StmtMutator::VisitStmt_(op);
254 }
255
256 Stmt VisitStmt_(const AttrStmtNode* op) final {
257 if (op->attr_key == attr::thread_extent || op->attr_key == attr::pipeline_exec_scope ||
258 op->attr_key == attr::device_scope) {
259 return SplitDeviceFunc(GetRef<Stmt>(op));
260 }
261 return StmtMutator::VisitStmt_(op);
262 }
263
264 private:
265 Stmt SplitDeviceFunc(Stmt body) {
266 std::ostringstream os;
267 os << name_prefix_ << "_kernel" << device_func_counter_++;
268 std::string kernel_symbol = os.str();
269 // isolate the device function.
270 VarUseDefAnalysis m;
271 m.visit_thread_extent_ = false;
272 body = m(std::move(body));
273
274 Array<Var> params;
275 Array<PrimExpr> arguments;
276 Map<tir::Var, PrimExpr> remap_vars;
277
278 // Strictly order the arguments: Var pointers, positional arguments.
279 for (Var var : m.undefined_) {
280 if (var.dtype().is_handle()) {
281 // Create a new version of v.
282 auto it = handle_data_type_.find(var.get());
283 if (it != handle_data_type_.end()) {
284 String storage_scope;
285 if (auto* ptr_type = var->type_annotation.as<PointerTypeNode>()) {
286 storage_scope = ptr_type->storage_scope;
287 }
288 tir::Var new_var(var->name_hint,
289 PointerType(PrimType((*it).second->dtype), storage_scope));
290 params.push_back(new_var);
291 remap_vars.Set(var, new_var);
292 } else {
293 params.push_back(var);
294 }
295 arguments.push_back(var);
296 }
297 }
298 // positional arguments
299 for (Var var : m.undefined_) {
300 if (!var.dtype().is_handle()) {
301 params.push_back(var);
302 arguments.push_back(var);
303 }
304 }
305 PrimFunc device_func(params, Substitute(body, remap_vars));
306 device_func = WithAttr(std::move(device_func), tir::attr::kDeviceThreadAxis, m.thread_axis_);
307 device_func = WithAttr(std::move(device_func), tvm::attr::kCallingConv,
308 Integer(CallingConv::kDeviceKernelLaunch));
309 device_func =
310 WithAttr(std::move(device_func), tvm::attr::kGlobalSymbol, runtime::String(kernel_symbol));
311 device_func = WithAttr(std::move(device_func), tir::attr::kNoAlias, Integer(1));
312 device_func = WithAttr(std::move(device_func), tvm::attr::kTarget, device_target_);
313 device_func = WithAttr(std::move(device_func), tir::attr::kIsGlobalFunc, Integer(1));
314 if (m.use_dyn_shmem_) {
315 device_func =
316 WithAttr(std::move(device_func), tir::attr::kDeviceUseDynSharedMemory, Integer(1));
317 }
318 (*device_mod_)->Add(GlobalVar(kernel_symbol), device_func);
319
320 // generate calls to the device function
321 Array<PrimExpr> call_args;
322 call_args.push_back(StringImm(kernel_symbol));
323 for (PrimExpr arg : arguments) {
324 call_args.push_back(arg);
325 }
326 for (PrimExpr ext : m.thread_extent_) {
327 call_args.push_back(ext);
328 }
329 if (m.use_dyn_shmem_) {
330 call_args.push_back(m.dyn_shmem_size_);
331 }
332 return Evaluate(Call(DataType::Int(32), builtin::tvm_call_packed(), call_args));
333 }
334
335 // target ir module
336 IRModule* device_mod_;
337 // Device target
338 Target device_target_;
339 // function name hint
340 std::string name_prefix_;
341 // Number of device functions.
342 int device_func_counter_{0};
343 std::unordered_map<const VarNode*, PrimExpr> handle_data_type_;
344};
345
346PrimFunc SplitHostDevice(PrimFunc&& func, IRModule* device_mod) {
347 auto target = func->GetAttr<Target>(tvm::attr::kTarget);
348 ICHECK(target.defined()) << "SplitHostDevice: Require the target attribute";
349 auto global_symbol = func->GetAttr<String>(tvm::attr::kGlobalSymbol);
350 ICHECK(global_symbol.defined())
351 << "SplitHostDevice: Expect PrimFunc to have the global_symbol attribute";
352
353 HostDeviceSplitter splitter(device_mod, target.value(),
354 static_cast<std::string>(global_symbol.value()));
355
356 auto* n = func.CopyOnWrite();
357 n->body = splitter(std::move(n->body));
358 // set the host target to None.
359 func = WithAttr(std::move(func), tvm::attr::kTarget, Target(nullptr));
360 return std::move(func);
361}

363namespace transform {
364
365Pass SplitHostDevice() {
366 auto pass_func = [](IRModule mod, PassContext ctx) {
367 IRModuleNode* mod_ptr = mod.CopyOnWrite();
368 auto* func_dict = mod_ptr->functions.CopyOnWrite();
369 IRModule device_mod = IRModule(Map<GlobalVar, BaseFunc>({}));
370
371 for (auto& kv : *func_dict) {
372 if (kv.second->IsInstance<PrimFuncNode>()) {
373 PrimFunc func = Downcast<PrimFunc>(std::move(kv.second));
374 ICHECK(device_mod.defined()) << "The device module must be defined.";
375 kv.second = SplitHostDevice(std::move(func), &device_mod);
376 }
377 }
378 mod->Update(device_mod);
379 return mod;
380 };
381
382 return tvm::transform::CreateModulePass(pass_func, 0, "tir.SplitHostDevice", {});
383}
