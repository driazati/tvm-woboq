34namespace tvm {
35namespace arith {
36
37using namespace tir;
38
39class StmtSimplifier : public IRMutatorWithAnalyzer {
40 public:
41 explicit StmtSimplifier(Analyzer* analyzer) : IRMutatorWithAnalyzer(analyzer) {}
42
43 using Parent = IRMutatorWithAnalyzer;
44 using Parent::VisitStmt;
45 using Parent::VisitStmt_;
46
47 PrimExpr VisitExpr(const PrimExpr& expr) final { return analyzer_->Simplify(expr); }
48
49 Stmt Simplify(Stmt stmt) { return operator()(std::move(stmt)); }
50
51 Stmt VisitStmt_(const ForNode* op) final {
52 analyzer_->Bind(op->loop_var, Range::FromMinExtent(op->min, op->extent));
53 With<ConstraintContext> ctx1(analyzer_, op->loop_var >= op->min);
54 With<ConstraintContext> ctx2(analyzer_, op->loop_var < op->min + op->extent);
55 return Parent::VisitStmt_(op);
56 }
57
58 bool CanInlineLetStmt(const LetStmtNode* op) {
59 if (is_const_number(op->value)) return true;
60 if (op->value.as<VarNode>()) return true;
61 // Won't face the deep expression explosion problem as in Let expression.
62 // attempt to inline as much as possible if the value integer type(can be index).
63 if (!op->value.dtype().is_int()) return false;
64 return SideEffect(op->value) <= CallEffectKind::kPure;
65 }
66
67 Stmt VisitStmt_(const LetStmtNode* op) {
68 PrimExpr value = this->VisitExpr(op->value);
69 if (CanInlineLetStmt(op)) {
70 // it is fine to discard the let binding
71 // because the call to simplify will always inline the var.
72 analyzer_->Bind(op->var, value);
73 return this->VisitStmt(op->body);
74 } else if (SideEffect(op->value) <= CallEffectKind::kPure) {
75 // Even if we aren't replacing all occurrences, they may be
76 // necessary for proving conditional statements.
77 non_inlined_bindings_.Set(op->var, value);
78 }
79 Stmt body = this->VisitStmt(op->body);
80 if (value.same_as(op->value) && body.same_as(op->body)) {
81 return GetRef<Stmt>(op);
82 } else {
83 auto n = this->CopyOnWrite(op);
84 n->value = std::move(value);
85 n->body = std::move(body);
86 return Stmt(n);
87 }
88 }
89
90 Stmt VisitStmt_(const IfThenElseNode* op) {
91 PrimExpr cond = analyzer_->Simplify(Substitute(op->condition, non_inlined_bindings_));
92 if (const int64_t* as_int = as_const_int(cond)) {
93 if (*as_int) {
94 return this->VisitStmt(op->then_case);
95 } else if (op->else_case.defined()) {
96 return this->VisitStmt(op->else_case);
97 } else {
98 return Evaluate(0);
99 }
100 }
101 return Parent::VisitStmt_(op);
102 }
103
104 PrimExpr VisitExpr_(const CallNode* op) {
105 if (op->op.same_as(builtin::if_then_else())) {
106 PrimExpr cond = this->VisitExpr(op->args[0]);
107 cond = analyzer_->Simplify(Substitute(std::move(cond), non_inlined_bindings_));
108 if (const int64_t* as_int = as_const_int(cond)) {
109 if (*as_int) {
110 return this->VisitExpr(op->args[1]);
111 } else {
112 return this->VisitExpr(op->args[2]);
113 }
114 }
115 }
116 return Parent::VisitExpr_(op);
117 }
118
119 Stmt VisitStmt_(const StoreNode* op) final {
120 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
121 return Stmt();
122 }
123
124 // eliminate useless stores
125 Stmt VisitStmt_(const BufferStoreNode* op) final {
126 BufferStore store = Downcast<BufferStore>(Parent::VisitStmt_(op));
127 if (const BufferLoadNode* load = store->value.as<BufferLoadNode>()) {
128 if (load->buffer->data.same_as(store->buffer->data) &&
129 ArrayDeepEqual(load->indices, store->indices) &&
130 tir::ExprDeepEqual()(load->buffer->elem_offset, store->buffer->elem_offset) &&
131 ArrayDeepEqual(load->buffer->shape, store->buffer->shape) &&
132 ArrayDeepEqual(load->buffer->strides, store->buffer->strides)) {
133 return Evaluate(0);
134 }
135 }
136 return std::move(store);
137 }
138
139 private:
140 bool ArrayDeepEqual(const Array<PrimExpr>& lhs, const Array<PrimExpr>& rhs) {
141 if (lhs.size() != rhs.size()) {
142 return false;
143 }
144 for (size_t i = 0; i < lhs.size(); i++) {
145 if (!tir::ExprDeepEqual()(lhs[i], rhs[i])) {
146 return false;
147 }
148 }
149 return true;
150 }
151
152 Map<Var, PrimExpr> non_inlined_bindings_;
153};
154
155} // namespace arith
156
157namespace tir {
158namespace transform {
159
160Pass Simplify() {
161 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
162 auto* n = f.CopyOnWrite();
163 arith::Analyzer analyzer;
164 n->body = arith::StmtSimplifier(&analyzer).Simplify(std::move(n->body));
165 return f;
166 };
167 return CreatePrimFuncPass(pass_func, 0, "tir.Simplify", {});
168}
169
170TVM_REGISTER_GLOBAL("tir.transform.Simplify").set_body_typed(Simplify);
