41struct UnrollLoopConfigNode : public tvm::AttrsNode<UnrollLoopConfigNode> {
42 int auto_max_step;
43 int auto_max_depth;
44 int auto_max_extent;
45 int explicit_unroll;
46
47 TVM_DECLARE_ATTRS(UnrollLoopConfigNode, "tir.transform.UnrollLoopConfig") {
48 TVM_ATTR_FIELD(auto_max_step)
49 .describe("Threshold of number of steps in the loop to be automatically unrolled")
50 .set_default(0);
51 TVM_ATTR_FIELD(auto_max_depth)
52 .describe("The maximum nested level of loops that can be automatically unrolled.")
53 .set_default(8);
54 TVM_ATTR_FIELD(auto_max_extent)
55 .describe("The maximum extent of loop that will be unrolled.")
56 .set_default(0);
57 TVM_ATTR_FIELD(explicit_unroll)
58 .describe("Whether to explicitly unroll the loop instead of setting a pragma")
59 .set_default(true);
60 }
61};
62
63class UnrollLoopConfig : public Attrs {
64 public:
65 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(UnrollLoopConfig, Attrs, UnrollLoopConfigNode);
66};

68TVM_REGISTER_NODE_TYPE(UnrollLoopConfigNode);
69TVM_REGISTER_PASS_CONFIG_OPTION("tir.UnrollLoop", UnrollLoopConfig);

71class LoopUnroller : public StmtExprMutator {
72 public:
73 explicit LoopUnroller(int auto_max_step, int auto_max_depth, int auto_max_extent,
74 bool explicit_unroll)
75 : auto_max_step_(auto_max_step),
76 auto_max_depth_(auto_max_depth),
77 auto_max_extent_(auto_max_extent),
78 explicit_unroll_(explicit_unroll) {}
79
80 Stmt VisitStmt_(const AttrStmtNode* op) final {
81 if (op->attr_key == "pragma_auto_unroll_max_step") {
82 int value = static_cast<int>(Downcast<Integer>(op->value)->value);
83 std::swap(value, auto_max_step_);
84 Stmt ret = this->VisitStmt(op->body);
85 std::swap(value, auto_max_step_);
86 return ret;
87 } else if (op->attr_key == "pragma_unroll_explicit") {
88 bool explicit_unroll = Downcast<Integer>(op->value)->value;
89 std::swap(explicit_unroll, explicit_unroll_);
90 Stmt ret = this->VisitStmt(op->body);
91 std::swap(explicit_unroll, explicit_unroll_);
92 return ret;
93 } else {
94 return StmtExprMutator::VisitStmt_(op);
95 }
96 }
97
98 Stmt VisitStmt_(const ForNode* op) {
99 Stmt stmt = StmtExprMutator::VisitStmt_(op);
100 op = stmt.as<ForNode>();
101 int value = GetExtent(op);
102 // condition for auto unroll
103 bool auto_unroll = (op->kind == ForKind::kSerial && value >= 0 && normal_loop_depth_ == 0 &&
104 unroll_depth_ <= auto_max_depth_);
105
106 auto_unroll =
107 auto_unroll && (value * step_count_ <= auto_max_step_ || value <= auto_max_extent_);
108
109 if (op->kind == ForKind::kUnrolled) {
110 ICHECK_GE(value, 0) << "Cannot unroll non-constant loop";
111 auto_unroll = true;
112 }
113
114 if (auto_unroll) {
115 step_count_ *= value;
116 unroll_depth_ += 1;
117 } else {
118 normal_loop_depth_ += 1;
119 }
120
121 if ((auto_unroll && explicit_unroll_) ||
122 // unroll loops with extent = 1, no matter how many steps in body
123 (0 <= value && value <= auto_max_extent_ && auto_max_extent_ == 1)) {
124 return Unroll(op);
125 } else {
126 if (auto_unroll) {
127 if (op->kind != ForKind::kUnrolled) {
128 return For(op->loop_var, op->min, op->extent, ForKind::kUnrolled, op->body,
129 op->thread_binding, op->annotations);
130 }
131 }
132 return stmt;
133 }
134 }
135
136 Stmt VisitStmt_(const StoreNode* op) final {
137 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
138 return Stmt();
139 }
140
141 Stmt VisitStmt_(const BufferStoreNode* op) final {
142 ++step_count_;
143 return StmtExprMutator::VisitStmt_(op);
144 }
145
146 Stmt VisitStmt_(const EvaluateNode* op) final {
147 ++step_count_;
148 return StmtExprMutator::VisitStmt_(op);
149 }
150
151 Stmt VisitStmt_(const SeqStmtNode* op) final {
152 auto fmutate = [this](const Stmt& s) {
153 int step_count = step_count_;
154 int unroll_depth = unroll_depth_;
155 int normal_loop_depth = normal_loop_depth_;
156 step_count_ = 0;
157 unroll_depth_ = 0;
158 normal_loop_depth_ = 0;
159 Stmt ret = this->VisitStmt(s);
160 step_count_ += step_count;
161 normal_loop_depth_ = std::max(normal_loop_depth, normal_loop_depth_);
162 unroll_depth_ = std::max(unroll_depth_, unroll_depth);
163 return ret;
164 };
165 return StmtMutator::VisitSeqStmt_(op, false, fmutate);
166 }
167
168 Stmt Unroll(const ForNode* op) {
169 int value = GetExtent(op);
170 // For loop must have a constant integer extent
171 ICHECK_NE(value, -1) << "loop doesn't have a constant integer extent";
172 if (value == 0) return Evaluate(0);
173 Stmt body = op->body;
174 Map<Var, PrimExpr> vmap;
175 Array<Stmt> unrolled;
176 for (int i = 0; i < value; ++i) {
177 vmap.Set(op->loop_var, op->min + make_const(op->loop_var.dtype(), i));
178 Stmt step = Substitute(body, vmap);
179 unrolled.push_back(step);
180 }
181 return SeqStmt::Flatten(unrolled);
182 }
183
184 private:
185 // returns the extent of the loop if it's a constant integer, otherwise return -1
186 int GetExtent(const ForNode* op) {
187 // constant folding.
188 PrimExpr extent = analyzer_.Simplify(op->extent);
189 const IntImmNode* v1 = extent.as<IntImmNode>();
190 int value = -1;
191 // integers that do not fit in int32_t are treated as symbolic,
192 // as it's impossible to unroll such large loops
193 if (v1 != nullptr && v1->value <= std::numeric_limits<int>::max()) {
194 value = static_cast<int>(v1->value);
195 }
196 return value;
197 }
198
199 // maximum number of step to perform auto unroll.
200 int auto_max_step_;
201 int auto_max_depth_;
202 // max extent of loop to auto unroll
203 // this not not count the total steps, only count the number of loops
204 int auto_max_extent_;
205 bool explicit_unroll_;
206 // Number of normal loops in scope
207 int normal_loop_depth_{0};
208 // number of unrolled cases in current scope.
209 int unroll_depth_{0};
210 // Number of total steps unrolled
211 int step_count_{0};
212 // analyzer
213 arith::Analyzer analyzer_;
214};
215
216Stmt UnrollLoop(Stmt stmt, UnrollLoopConfig cfg) {
217 Stmt ret = LoopUnroller(cfg->auto_max_step, cfg->auto_max_depth, cfg->auto_max_extent,
218 cfg->explicit_unroll)(stmt);
219 if (!ret.same_as(stmt)) {
220 return ConvertSSA(ret);
221 } else {
222 return ret;
223 }
224}

226namespace transform {
227
228Pass UnrollLoop() {
229 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
230 auto* n = f.CopyOnWrite();
231 auto cfg = ctx->GetConfig<UnrollLoopConfig>("tir.UnrollLoop");
232 if (!cfg.defined()) {
233 cfg = AttrsWithDefaultValues<UnrollLoopConfig>();
234 }
235 n->body = UnrollLoop(std::move(f->body), cfg.value());
236 return f;
237 };
238 return CreatePrimFuncPass(pass_func, 0, "tir.UnrollLoop", {});
239}
