97/*!
98 * \brief A helper mutator which recursively replaces the old buffer with the new buffer and
99 * collects the block sref reuse information for the following replacement.
100 *
101 * If the buffer to be replaced in used as the source in `match_buffers`, depending the specific
102 * use cases, the target buffers in `match_buffers` may also need to be mutated. In this
103 * case, this class should be subclassed to explicitly handle `match_buffers`.
104 */
105class ReplaceBufferMutator : public StmtExprMutator {
106 public:
107 /*!
108 * \brief The constructor
109 * \param old_buffer The old buffer
110 * \param new_buffer The new buffer
111 * \param block_sref_reuse Optional map to record mapping between old and new blocks that reuse
112 * sref.
113 */
114 ReplaceBufferMutator(const Buffer& old_buffer, Buffer new_buffer,
115 Map<Block, Block>* block_sref_reuse);
116
117 protected:
118 PrimExpr VisitExpr_(const VarNode* var) final;
119
120 template <typename Node>
121 Node VisitBufferAccess(Node node) {
122 auto it = buffer_var_map_.find(node->buffer->data.get());
123 if (it != buffer_var_map_.end()) {
124 node.CopyOnWrite()->buffer = it->second;
125 }
126 return node;
127 }
128
129 Stmt VisitStmt_(const BufferStoreNode* op) final;
130
131 PrimExpr VisitExpr_(const BufferLoadNode* op) final;
132
133 virtual MatchBufferRegion VisitMatchBufferRegion(const MatchBufferRegion& match_buffer);
134
135 Stmt VisitStmt_(const BlockNode* block) final;
136
137 /*!
138 * \brief A mapping which maps old buffer vars to new buffers, including the buffers defined in
139 * MatchBufferRegion.
140 */
141 std::unordered_map<const VarNode*, Buffer> buffer_var_map_;
142 /*! \brief The block sref reuse map for the following replacement */
143 Map<Block, Block>* block_sref_reuse_;
144};

146/******** Block Removal ********/
147
148/*!
149 * \brief Construct a new AST, with a specific sref tree leaf removed.
150 * The leaf's ancestors who have only a single child will be removed too.
151 * \param leaf_block_sref The block/loop sref to the sref tree leaf to be removed
152 * \param src_stmt The root of the subtree where the replacement begins
153 * \param tgt_stmt The root of the subtree after the replacement
154 * \return A boolean indicating if the leaf can be removed successfully
155 * \note Read before use:
156 * 1) Removal is not conducted beyond scope-level.
157 * 2) This method only works properly when the scope root is a stage pipeline.
158 *
159 * An example of the removal plan, say we are removing the leaf block "B" from the AST.
160 *
161 * \code
162 * with block([], "scope_root"):
163 * ...
164 * with block([128, 128], "B") as [vi, vj]:
165 * B[vi, vj] = A[vi, vj] + 1.0
166 * with block([128, 128], "C") as [vi, vj]:
167 * C[vi, vj] = B[vi, vj] * 2.0
168 * \endcode
169 *
170 * Ths method does not mutate the AST, instead it returns the a `(src_stmt, tgt_stmt)` pair as a
171 * plan to substitute certain pieces of the IR.
172 *
173 * In our example, it returns block "scope_root" as `src_stmt`, and the result `tgt_stmt` is:
174 *
175 * \code
176 * with block([], "scope_root"):
177 * ...
178 * with block([128, 128], "C") as [vi, vj]:
179 * C[vi, vj] = B[vi, vj] * 2.0
180 * \endcode
181 */
182void LeafBlockRemovalPlan(const ScheduleState& self, const StmtSRef& leaf_block_sref,
183 Stmt* src_stmt, Stmt* tgt_stmt);

199/*!
200 * \brief Simplifier for indices of buffer access and block buffer access regions.
201 */
202class BlockBufferAccessSimplifier : public arith::IRMutatorWithAnalyzer {
203 public:
204 /*!
205 * \brief Simplify indices of buffer access and block buffer access regions in the statement
206 * \param stmt The statement to be simplified
207 * \param analyzer The arithmetic analyzer
208 * \return The simplified statement
209 */
210 static Stmt Simplify(const Stmt& stmt, arith::Analyzer* analyzer) {
211 BlockBufferAccessSimplifier simplifier(analyzer);
212 return simplifier(stmt);
213 }
214
215 private:
216 explicit BlockBufferAccessSimplifier(arith::Analyzer* analyzer)
217 : IRMutatorWithAnalyzer(analyzer) {}
218
219 using IRMutatorWithAnalyzer::VisitExpr_;
220 using IRMutatorWithAnalyzer::VisitStmt_;
221
222 void SimplifyAccessRegion(Array<BufferRegion>* old_access_regions);
223 Stmt VisitStmt_(const BlockNode* op) final;
224 Stmt VisitStmt_(const BufferStoreNode* op) final;
225 PrimExpr VisitExpr_(const BufferLoadNode* op) final;
226
227 template <typename Node>
228 Node VisitBufferAccess(Node node) {
229 node.CopyOnWrite()->indices.MutateByApply(
230 [this](const PrimExpr& expr) { return analyzer_->Simplify(expr); });
231 return node;
232 }
233};
