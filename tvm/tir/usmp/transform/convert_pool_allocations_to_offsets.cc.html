41/*!
42 * \brief The StmtExpr mutator class to replace allocate nodes
43 * with offsets within memory pools
44 *
45 * This mutator class will add Pool variables recursively to every PrimFunc
46 * starting from the main PrimFunc. For all allocate nodes, that have been
47 * memory planned, will be mutated into an offset using a Let binding.
48 */
49class PoolAllocationToOffsetConverter : public StmtExprMutator {
50 public:
51 PoolAllocationToOffsetConverter(const IRModule& module,
52 const Map<tir::Stmt, PoolAllocation>& pool_allocations,
53 bool emit_tvmscript_printable = false)
54 : pool_allocations_(pool_allocations), emit_tvmscript_printable_(emit_tvmscript_printable) {
55 module_ = module->ShallowCopy();
56 for (const auto& kv : pool_allocations) {
57 size_t extent_size = -1;
58 if (kv.first->IsInstance<AllocateNode>()) {
59 Allocate allocate_node = Downcast<Allocate>(kv.first);
60 extent_size = CalculateExtentsSize(allocate_node.operator->()).IntValue();
61 } else if (kv.first->IsInstance<AllocateConstNode>()) {
62 AllocateConst allocate_const_node = Downcast<AllocateConst>(kv.first);
63 extent_size = CalculateExtentsSize(allocate_const_node.operator->()).IntValue();
64 } else {
65 ICHECK(false) << "Not supported node type " << kv.first->GetTypeKey();
66 }
67 PoolAllocation pool_allocation = kv.second;
68 PoolInfo pool_info = pool_allocation->pool_info;
69 int byte_pool_offset = pool_allocation->byte_offset->value;
70 int required_pool_size_for_allocation = byte_pool_offset + extent_size;
71 if (all_pools_sizes_.find(pool_info) == all_pools_sizes_.end()) {
72 all_pools_sizes_[pool_info] = required_pool_size_for_allocation;
73 } else {
74 int prev_required_pool_size = all_pools_sizes_[pool_info];
75 if (prev_required_pool_size < required_pool_size_for_allocation) {
76 all_pools_sizes_[pool_info] = required_pool_size_for_allocation;
77 }
78 }
79 }
80
81 for (const auto& kv : all_pools_sizes_) {
82 PoolInfo pi = kv.first;
83 int allocated_size = kv.second;
84 allocated_pool_ordering_.push_back(AllocatedPoolInfo(pi, allocated_size));
85 }
86 std::sort(allocated_pool_ordering_.begin(), allocated_pool_ordering_.end(),
87 [](const AllocatedPoolInfo& lhs, const AllocatedPoolInfo& rhs) {
88 if (lhs->pool_info->pool_name < rhs->pool_info->pool_name) {
89 return true;
90 }
91 return false;
92 });
93 }
94 IRModule operator()();
95
96 private:
97 PrimExpr VisitExpr_(const CallNode* op) override;
98 Stmt VisitStmt_(const AllocateNode* op) override;
99 PrimExpr VisitExpr_(const BufferLoadNode* op) override;
100 Stmt VisitStmt_(const BufferStoreNode* op) override;
101
102 Stmt VisitStmt_(const AllocateConstNode* op) override;
103 LetStmt ToLetStmt(const PoolAllocation& pool_allocation, const Var& buffer_var, const Stmt& body);
104 /*! \brief This is a structure where the modified function
105 * signature is kept while body of the function is mutated
106 */
107 struct ScopeInfo {
108 Array<tir::Var> params;
109 Map<PoolInfo, tir::Var> pools_to_params;
110 Array<AllocatedPoolInfo> allocated_pool_params;
111 Map<tir::Var, Buffer> buffer_map;
112 };
113
114 /*! \brief The function scope information that are needed
115 * in the mutation of the function need to be stacked and
116 * popped when each function is entered/exited in the
117 * mutation process.
118 */
119 std::stack<ScopeInfo> scope_stack;
120 /*! \brief Each PrimFunc signature needs to be updated
121 * with pool variables. This is a helper function to
122 * capture the updated information to ScopeInfo object.
123 */
124 ScopeInfo UpdateFunctionScopeInfo(const PrimFunc& original_func);
125 /*! \brief This is a helper to create the PrimFunc with
126 * pool variables that calls the UpdateFunctionScopeInfo
127 * inside of it.
128 */
129 PrimFunc CreatePrimFuncWithPoolParams(const PrimFunc& original_primfunc);
130 /*! \brief This is a helper to append the pool args to
131 * the callsite of the function.
132 */
133 Array<PrimExpr> AppendPoolParamsToArgs(Array<PrimExpr> args, bool has_device_context);
134 /*! \brief Some arguments that used to be Allocate nodes
135 * should be replaced by Let nodes in the pass that loads
136 * the space from a pool variable.
137 */
138 Array<PrimExpr> ReplaceAllocateArgsWithLetArgs(const Array<PrimExpr>& args);
139 /*! \brief Obtain a resource handle if its there
140 */
141 Optional<Var> GetResourceHandle(const PrimFunc& func);
142 /*! \brief Get the Buffer object representing the mapped access into
143 * the pool.
144 */
145 Buffer GetRemappedBuffer(Buffer buf);
146
147 /*! \brief The tir::Var map to PoolInfo objects */
148 Map<tir::Var, PoolInfo> primfunc_args_to_pool_info_map_;
149 /*! \brief The buffer var map to their allocate nodes */
150 Map<tir::Var, tir::Stmt> allocate_var_to_stmt_map_;
151 /*! \brief The IRModule being constructed/mutated */
152 IRModule module_;
153 /*! \brief The input allocate node to PoolAllocation map */
154 Map<tir::Stmt, PoolAllocation> pool_allocations_;
155 /*! \brief The set of ordered pools to ensure an unique order of args for functions */
156 std::vector<AllocatedPoolInfo> allocated_pool_ordering_;
157 /*! \brief The storage of calculated pool size at init */
158 std::unordered_map<PoolInfo, int, ObjectPtrHash, ObjectPtrEqual> all_pools_sizes_;
159 /*! \brief After mutation, each allocate buffer is replaced with tir::Var that is let bounded
160 * to position from a pool as designated by a PoolAllocation
161 */
162 Map<tir::Var, tir::Var> allocate_var_to_let_var_;
163 /*! \brief A map from the original buffer object
164 *
165 * Each key-value pair in this map satisfies
166 * ``allocate_buf_to_let_var[key->data] = value->data``. However,
167 * since more than one `tir::Buffer` may use the same Var, they must
168 * be tracked separately.
169 */
170 Map<tir::Buffer, tir::Buffer> original_buf_to_let_buf_;
171
172 Map<String, Bool> signature_has_device_context_;
173 /*! \brief A counter to give references to pools a reproducible unique set of names */
174 int pool_var_count_ = 0;
175 /*! \brief This toggles to remove non tvmscript printable items for IRModule for unit tests */
176 bool emit_tvmscript_printable_ = false;
177 /*! \brief A counter to give references to pools a reproducible unique set of names */
178 std::unordered_set<PrimFunc, ObjectPtrHash, ObjectPtrEqual> visited_primfuncs;
179
180 Map<PoolInfo, Array<ConstantInfo>> pool_initializations_;
181 void AppdendConstInitializationData(ScopeInfo si);
182};

184Optional<Var> PoolAllocationToOffsetConverter::GetResourceHandle(const PrimFunc& func) {
185 if (!func->params.empty() &&
186 func->buffer_map.find(func->params.back()) == func->buffer_map.end()) {
187 return func->params.back();
188 }
189 return Optional<Var>();
190}
191
192PoolAllocationToOffsetConverter::ScopeInfo PoolAllocationToOffsetConverter::UpdateFunctionScopeInfo(
193 const PrimFunc& original_func) {
194 ScopeInfo si;
195
196 Optional<Var> resource_handle = GetResourceHandle(original_func);
197 si.params = original_func->params;
198 if (resource_handle) {
199 si.params.pop_back();
200 ICHECK(si.params.size() == original_func->params.size() - 1);
201 }
202 si.buffer_map = original_func->buffer_map;
203 Map<tir::Var, PoolInfo> ret;
204 for (const AllocatedPoolInfo& allocated_pool_info : allocated_pool_ordering_) {
205 PoolInfo pool_info = allocated_pool_info->pool_info;
206 String pool_ref_name = pool_info->pool_name + "_" + std::to_string(pool_var_count_++);
207 String var_name = pool_ref_name + "_var";
208 DataType elem_dtype = DataType::UInt(8);
209 Var buffer_var(var_name, PointerType(PrimType(elem_dtype), "global"));
210 Var pool_var = Var(var_name, PointerType(PrimType(elem_dtype), "global"));
211 si.params.push_back(pool_var);
212 si.pools_to_params.Set(pool_info, pool_var);
213 si.allocated_pool_params.push_back(AllocatedPoolInfo(
214 allocated_pool_info->pool_info, allocated_pool_info->allocated_size, si.params.size() - 1));
215
216 int pool_size = all_pools_sizes_[pool_info];
217 String buffer_var_name = pool_ref_name + "_buffer_var";
218 si.buffer_map.Set(pool_var,
219 Buffer(buffer_var /* data */, elem_dtype /* dtype */, {pool_size} /* shape */,
220 {1} /* strides */, 0 /* elem_offset */, buffer_var_name /* name */,
221 16 /* data_alignment */, 1 /* offset_factor */,
222 BufferType::kDefault /* buffer-type */));
223 }
224 if (resource_handle) {
225 si.params.push_back(resource_handle.value());
226 }
227 return si;
228}

230PrimFunc PoolAllocationToOffsetConverter::CreatePrimFuncWithPoolParams(
231 const PrimFunc& original_primfunc) {
232 // Only create the new function if it was not modified with pool params
233 if (visited_primfuncs.find(original_primfunc) == visited_primfuncs.end()) {
234 ScopeInfo si = UpdateFunctionScopeInfo(original_primfunc);
235 this->scope_stack.push(si);
236 Stmt new_body = this->VisitStmt(original_primfunc->body);
237 this->scope_stack.pop();
238 DictAttrs original_attrs = original_primfunc->attrs;
239 // We dont need attrs of PrimFunc that might include non printable attrs such as target
240 // for unit tests where emit_tvmscript_printable_ is to be used.
241 if (emit_tvmscript_printable_) {
242 original_attrs = DictAttrs();
243 }
244 PrimFunc ret = PrimFunc(si.params, new_body, original_primfunc->ret_type, si.buffer_map,
245 si.buffer_map, original_attrs);
246 if (!emit_tvmscript_printable_) {
247 ret = WithAttr(ret, tvm::attr::kPoolArgs, si.allocated_pool_params);
248 }
249 visited_primfuncs.insert(ret);
250 return ret;
251 }
252 return original_primfunc;
253}

255Array<PrimExpr> PoolAllocationToOffsetConverter::AppendPoolParamsToArgs(Array<PrimExpr> args,
256 bool has_device_context) {
257 Array<PrimExpr> new_args;
258 PrimExpr resource_handle_arg;
259 // name, params...params[, context]
260 if (has_device_context) {
261 resource_handle_arg = args.back();
262 args.pop_back();
263 }
264 for (const auto& arg : args) {
265 new_args.push_back(VisitExpr(arg));
266 }
267 ScopeInfo top_scope = this->scope_stack.top();
268 for (const auto& pools_vars : top_scope.pools_to_params) {
269 tir::Var pool_var = pools_vars.second;
270 Buffer buffer_var = top_scope.buffer_map[pool_var];
271 new_args.push_back(buffer_var->data);
272 }
273 if (resource_handle_arg.defined()) {
274 new_args.push_back(resource_handle_arg);
275 }
276 return new_args;
277}

279Array<PrimExpr> PoolAllocationToOffsetConverter::ReplaceAllocateArgsWithLetArgs(
280 const Array<PrimExpr>& args) {
281 Array<PrimExpr> ret;
282 for (const PrimExpr& arg : args) {
283 if (arg->IsInstance<VarNode>() &&
284 allocate_var_to_let_var_.find(Downcast<Var>(arg)) != allocate_var_to_let_var_.end()) {
285 ret.push_back(allocate_var_to_let_var_[Downcast<Var>(arg)]);
286 } else {
287 ret.push_back(VisitExpr(arg));
288 }
289 }
290 return ret;
291}

293PrimExpr PoolAllocationToOffsetConverter::VisitExpr_(const CallNode* op) {
294 if (op->op.same_as(builtin::call_extern()) || op->op.same_as(builtin::tvm_call_cpacked())) {
295 String func_name = Downcast<StringImm>(op->args[0])->value;
296 Array<PrimExpr> new_args;
297 if (module_->ContainGlobalVar(func_name) &&
298 module_->Lookup(func_name)->IsInstance<PrimFuncNode>()) {
299 GlobalVar gv = module_->GetGlobalVar(func_name);
300 PrimFunc func = Downcast<PrimFunc>(module_->Lookup(gv));
301
302 if (!signature_has_device_context_.count(func_name)) {
303 if (op->args.size() == func->params.size() + 2) {
304 signature_has_device_context_.Set(func_name, Bool(true));
305 } else {
306 signature_has_device_context_.Set(func_name, Bool(false));
307 }
308 }
309
310 PrimFunc prim_func = CreatePrimFuncWithPoolParams(func);
311 module_->Update(gv, prim_func);
312 new_args = AppendPoolParamsToArgs(op->args, signature_has_device_context_[func_name]);
313 new_args = ReplaceAllocateArgsWithLetArgs(new_args);
314 } else {
315 new_args = ReplaceAllocateArgsWithLetArgs(op->args);
316 }
317 return Call(op->dtype, op->op, new_args);
318 }
319 if (op->op->IsInstance<PrimFuncNode>()) {
320 String func_name = Downcast<StringImm>(op->args[0])->value;
321 PrimFunc func = Downcast<PrimFunc>(op->op);
322 PrimFunc prim_func = CreatePrimFuncWithPoolParams(func);
323 Array<PrimExpr> new_args =
324 AppendPoolParamsToArgs(op->args, signature_has_device_context_[func_name]);
325 new_args = ReplaceAllocateArgsWithLetArgs(new_args);
326 return Call(op->dtype, prim_func, new_args);
327 }
328 return StmtExprMutator::VisitExpr_(op);
329}

331LetStmt PoolAllocationToOffsetConverter::ToLetStmt(const PoolAllocation& pool_allocation,
332 const Var& buffer_var, const Stmt& body) {
333 ScopeInfo scope_info = scope_stack.top();
334 Var param = scope_info.pools_to_params[pool_allocation->pool_info];
335 BufferLoad load_node = BufferLoad(scope_info.buffer_map[param], {pool_allocation->byte_offset});
336 Call address_of_load = Call(DataType::Handle(), builtin::address_of(), {load_node});
337
338 Type let_var_type = buffer_var->type_annotation;
339 if (emit_tvmscript_printable_) {
340 // Strip the storage_scope from the variable type, as TVMScript
341 // doesn't parsethe scoped pointers (e.g. ``T.Ptr[global T.int32]``)
342 // correctly.
343 let_var_type = PointerType(Downcast<PointerType>(let_var_type)->element_type);
344 }
345 Var let_var(buffer_var->name_hint + "_let", let_var_type);
346 allocate_var_to_let_var_.Set(buffer_var, let_var);
347 Stmt new_body = VisitStmt(body);
348 allocate_var_to_let_var_.erase(buffer_var);
349 return LetStmt(let_var, address_of_load, new_body);
350}
351
352Stmt PoolAllocationToOffsetConverter::VisitStmt_(const AllocateNode* op) {
353 if (pool_allocations_.count(GetRef<Allocate>(op))) {
354 return ToLetStmt(pool_allocations_[GetRef<Stmt>(op)], op->buffer_var, op->body);
355 }
356 return StmtExprMutator::VisitStmt_(op);
357}
358
359Stmt PoolAllocationToOffsetConverter::VisitStmt_(const AllocateConstNode* op) {
360 if (pool_allocations_.count(GetRef<AllocateConst>(op))) {
361 const auto& result = ToLetStmt(pool_allocations_[GetRef<Stmt>(op)], op->buffer_var, op->body);
362
363 PoolInfo pool_info = pool_allocations_[GetRef<Stmt>(op)]->pool_info;
364 if (pool_initializations_.find(pool_info) == pool_initializations_.end()) {
365 pool_initializations_.Set(pool_info, {});
366 }
367
368 auto consts = pool_initializations_[pool_info];
369 consts.push_back({result->var->name_hint, pool_allocations_[GetRef<Stmt>(op)]->byte_offset,
370 op->data.value()});
371
372 pool_initializations_.Set(pool_info, consts);
373 return result;
374 }
375 return StmtExprMutator::VisitStmt_(op);
376}
377
378Stmt PoolAllocationToOffsetConverter::VisitStmt_(const BufferStoreNode* op) {
379 BufferStore store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
380
381 Buffer remapped = GetRemappedBuffer(store->buffer);
382 if (!op->buffer.same_as(remapped)) {
383 store.CopyOnWrite()->buffer = remapped;
384 }
385 return std::move(store);
386}
387
388PrimExpr PoolAllocationToOffsetConverter::VisitExpr_(const BufferLoadNode* op) {
389 BufferLoad load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
390
391 Buffer remapped = GetRemappedBuffer(load->buffer);
392 if (!op->buffer.same_as(remapped)) {
393 load.CopyOnWrite()->buffer = remapped;
394 }
395 return std::move(load);
396}
397
398Buffer PoolAllocationToOffsetConverter::GetRemappedBuffer(Buffer original) {
399 {
400 auto it = original_buf_to_let_buf_.find(original);
401 if (it != original_buf_to_let_buf_.end()) {
402 return (*it).second;
403 }
404 }
405
406 Buffer remapped = original;
407
408 auto it = allocate_var_to_let_var_.find(original->data);
409 if (it != allocate_var_to_let_var_.end()) {
410 remapped = Buffer((*it).second, original->dtype, original->shape, original->strides,
411 original->elem_offset, original->name, original->data_alignment,
412 original->offset_factor, original->buffer_type, original->axis_separators,
413 original->span);
414 }
415
416 original_buf_to_let_buf_.Set(original, remapped);
417 return remapped;
418}

420void PoolAllocationToOffsetConverter::AppdendConstInitializationData(
421 PoolAllocationToOffsetConverter::ScopeInfo si) {
422 for (AllocatedPoolInfo api : si.allocated_pool_params) {
423 const auto& it = pool_initializations_.find(api->pool_info);
424 if (it != pool_initializations_.end()) {
425 auto* pi = const_cast<ConstantPoolInfoNode*>(api->pool_info.as<ConstantPoolInfoNode>());
426 pi->constant_info_array = (*it).second;
427 }
428 }
429}

431IRModule PoolAllocationToOffsetConverter::operator()() {
432 GlobalVar gv = module_->GetGlobalVar(::tvm::runtime::symbol::tvm_module_main);
433 PrimFunc main_func = Downcast<PrimFunc>(module_->Lookup(gv));
434 ScopeInfo si = UpdateFunctionScopeInfo(main_func);
435 this->scope_stack.push(si);
436 Stmt main_func_body = this->VisitStmt(main_func->body);
437 this->scope_stack.pop();
438 AppdendConstInitializationData(si);
439 // We dont need attrs of PrimFunc that might include non printable attrs such as target
440 // for unit tests where emit_tvmscript_printable_ is to be used.
441 if (!emit_tvmscript_printable_) {
442 main_func = PrimFunc(si.params, main_func_body, main_func->ret_type, si.buffer_map, {},
443 main_func->attrs);
444 main_func = WithAttr(main_func, tvm::attr::kPoolArgs, si.allocated_pool_params);
445 } else {
446 main_func =
447 PrimFunc(si.params, main_func_body, main_func->ret_type, si.buffer_map, {}, DictAttrs());
448 }
449 module_->Update(gv, main_func);
450 if (!emit_tvmscript_printable_) {
451 return WithAttr(this->module_, tvm::attr::kPoolArgs, si.allocated_pool_params);
452 }
453 return this->module_;
454}

458tvm::transform::Pass ConvertPoolAllocationsToOffsets(
459 const Map<tir::Stmt, PoolAllocation>& pool_allocations, Bool emit_tvmscript_printable) {
460 auto pass_func = [=](IRModule m, tvm::transform::PassContext ctx) {
461 return Downcast<IRModule>(PoolAllocationToOffsetConverter(
462 m, pool_allocations, emit_tvmscript_printable->value != 0)());
463 };
464 return tvm::transform::CreateModulePass(pass_func, 0, "tir.usmp.ConvertPoolAllocationsToOffsets",
465 {});
466}
