66bool ShouldAnnotateEntryFunc(const IRModule mod) {
67 Optional<tvm::relay::Executor> executor = mod->GetAttr<tvm::relay::Executor>("executor");
68 const bool aot_executor = executor.defined() && executor.value()->name == "aot";
69 const bool single_entry_func = (mod->functions.size() == 1);
70 return single_entry_func && !aot_executor;
71}

73/*! \return The default host target for a given device target */
74Target DefaultTargetHost(Target target) {
75 if (target.defined() && target->kind->device_type == kDLCPU) {
76 return target;
77 } else {
78 if (LLVMEnabled()) {
79 return Target("llvm");
80 } else {
81 return Target("stackvm");
82 }

86void GetBinds(const Array<ObjectRef>& args, bool compact,
87 const std::unordered_map<te::Tensor, tir::Buffer>& binds,
88 Map<te::Tensor, tir::Buffer>* out_binds, Array<ObjectRef>* out_arg_list) {
89 *out_binds = binds;
90
91 for (const ObjectRef& x : args) {
92 if (const te::TensorNode* tensor_node = x.as<te::TensorNode>()) {
93 te::Tensor x_ref = GetRef<te::Tensor>(tensor_node);
94 if (out_binds->find(x_ref) == out_binds->end()) {
95 tir::Buffer buf = tir::BufferWithOffsetAlignment(x_ref->shape, x_ref->dtype,
96 x_ref->op->name, -1, 0, compact);
97 out_binds->Set(x_ref, buf);
98 out_arg_list->push_back(buf);
99 } else {
100 out_arg_list->push_back((*out_binds)[x_ref]);
101 }
102 } else if (x.as<te::BufferNode>() || x.as<tir::VarNode>()) {
103 out_arg_list->push_back(x);
104 } else {
105 LOG(FATAL)
106 << "Expected type of the elements of args to be te::Tensor, te::Buffer or tir::Var, "
107 << "but got a " << x->GetTypeKey();
108 }
109 }
110}

113 const std::unordered_map<te::Tensor, tir::Buffer>& binds,
114 Map<te::Tensor, tir::Buffer>* out_binds, Array<ObjectRef>* out_arg_list) {
115 Array<ObjectRef> ref_args;

142Array<tvm::transform::Pass> CreatePassList(bool disable_loop_partition) {
143 transform::PassContext pass_ctx = transform::PassContext::Current();
144
145 bool disable_vectorize = pass_ctx->GetConfig<Bool>("tir.disable_vectorize", Bool(false)).value();
146 bool disable_storage_rewrite =
147 pass_ctx->GetConfig<Bool>("tir.disable_storage_rewrite", Bool(false)).value();
148 bool instrument_bound_checkers =
149 pass_ctx->GetConfig<Bool>("tir.instrument_bound_checkers", Bool(false)).value();
150 bool disable_cse_tir = pass_ctx->GetConfig<Bool>("tir.disable_cse_tir", Bool(false)).value();
151 bool enable_equiv_terms_in_cse_tir =
152 pass_ctx->GetConfig<Bool>("tir.enable_equiv_terms_in_cse_tir", Bool(false)).value();
153
154 // Get any user-added passes
155 Array<Array<ObjectRef>> add_lower_pass =
156 pass_ctx->GetConfig<Array<Array<ObjectRef>>>("tir.add_lower_pass", Array<Array<ObjectRef>>())
157 .value();
158
159 Array<transform::Pass> user_lower_phase0 = Array<transform::Pass>();
160 Array<transform::Pass> user_lower_phase1 = Array<transform::Pass>();
161 Array<transform::Pass> user_lower_phase2 = Array<transform::Pass>();
162 Array<transform::Pass> user_lower_phase3 = Array<transform::Pass>();
163
164 // phase passes is of the form
165 // [[phase_number, pass], [phase_number, pass]... ]
166 for (Array<ObjectRef> phase_pass : add_lower_pass) {
167 const IntImmNode* phase_num = phase_pass[0].as<IntImmNode>();
168 ICHECK(phase_num)
169 << "Expected the first entry in the inner Array of tir.add_lower_pass to be an integer";
170 int phase_num_val = phase_num->value;
171
172 CHECK_GE(phase_num_val, 0);
173
174 const tvm::transform::PassNode* pass_node = phase_pass[1].as<tvm::transform::PassNode>();
175 tvm::transform::Pass pass = GetRef<tvm::transform::Pass>(pass_node);
176 // Copy the pass into the correct phase
177 if (phase_num_val == 0) {
178 user_lower_phase0.push_back(pass);
179 } else if (phase_num_val == 1) {
180 user_lower_phase1.push_back(pass);
181 } else if (phase_num_val == 2) {
182 user_lower_phase2.push_back(pass);
183 } else if (phase_num_val >= 3) {
184 user_lower_phase3.push_back(pass);
185 }
186 }
187
188 // Construct the pass list, inserting the user provided passes at the end of the phase
189
190 // PHASE 0
191 Array<tvm::transform::Pass> pass_list = user_lower_phase0;
192
193 // PHASE 1
194 pass_list.push_back(tir::transform::InjectPrefetch());
195 pass_list.push_back(tir::transform::TextureFlatten());
196 pass_list.push_back(tir::transform::StorageFlatten(64, instrument_bound_checkers));
197 pass_list.push_back(tir::transform::LowerCrossThreadReduction());
198 pass_list.push_back(tir::transform::LowerInitBlock());
199 pass_list.push_back(tir::transform::PlanAndUpdateBufferAllocationLocation());
200 pass_list.push_back(tir::transform::ConvertBlocksToOpaque());
201 pass_list.push_back(tir::transform::UnifyThreadBinding());
202 pass_list.push_back(tir::transform::CompactBufferAllocation());
203 pass_list.push_back(tir::transform::LowerMatchBuffer());
204 pass_list.push_back(tir::transform::InjectSoftwarePipeline());
205 pass_list.push_back(tir::transform::LowerOpaqueBlock());
206 pass_list.push_back(tir::transform::FlattenBuffer());
207 pass_list.push_back(tir::transform::LowerVtcmAlloc());
208 pass_list.push_back(tir::transform::BF16Legalize());
209 pass_list.push_back(tir::transform::NarrowDataType(32));
210 pass_list.push_back(tir::transform::Simplify());
211
212 // Add user-defined phase-1 passes
213 pass_list.insert(pass_list.end(), user_lower_phase1.begin(), user_lower_phase1.end());
214
215 // PHASE 2
216 if (!disable_loop_partition) {
217 pass_list.push_back(tir::transform::LoopPartition());
218 }
219
220 pass_list.push_back(tir::transform::VectorizeLoop(!disable_vectorize));
221 pass_list.push_back(tir::transform::InjectVirtualThread());
222 pass_list.push_back(tir::transform::InjectDoubleBuffer());
223 if (!disable_storage_rewrite) {
224 pass_list.push_back(tir::transform::StorageRewrite());
225 }
226 pass_list.push_back(tir::transform::UnrollLoop());
227
228 // Add user-defined phase-2 passes
229 pass_list.insert(pass_list.end(), user_lower_phase2.begin(), user_lower_phase2.end());
230
231 // PHASE 3
232 pass_list.push_back(tir::transform::RenormalizeSplitPattern());
233 pass_list.push_back(tir::transform::Simplify());
234 pass_list.push_back(tir::transform::RemoveNoOp());
235 pass_list.push_back(tir::transform::RewriteUnsafeSelect());
236 pass_list.push_back(tir::transform::HoistIfThenElse());
237
238 // Add user-defined phase-3 passes
239 pass_list.insert(pass_list.end(), user_lower_phase3.begin(), user_lower_phase3.end());
240
241 if (instrument_bound_checkers) {
242 pass_list.push_back(tir::transform::InstrumentBoundCheckers());
243 }
244
245 pass_list.push_back(
246 tir::transform::CommonSubexprElimTIR(!disable_cse_tir, enable_equiv_terms_in_cse_tir));
247
248 return pass_list;
249}

262// Convert te schedule to IRModule
263IRModule ScheduleToModule(te::Schedule sch, const Array<ObjectRef>& args, const std::string& name,
264 const std::unordered_map<te::Tensor, tir::Buffer>& binds) {
265 sch = sch.normalize();
266
267 transform::PassContext pass_ctx = transform::PassContext::Current();
268 bool debug_keep_trivial_loop =
269 pass_ctx->GetConfig<Bool>("tir.debug_keep_trivial_loop", Bool(false)).value();
270
271 // Before TIR transformation.
272 tir::Stmt stmt = te::ScheduleOps(sch, te::InferBound(sch), debug_keep_trivial_loop);
273 bool compact = te::VerifyCompactBuffer(stmt);
274
275 Map<te::Tensor, tir::Buffer> out_binds;
276 Array<ObjectRef> out_arg_list;
277 GetBinds(args, compact, binds, &out_binds, &out_arg_list);
278
279 // Build the function, converting from te::Tensor to tir::Buffer
280 tir::PrimFunc f = te::SchedulePostProcToPrimFunc(out_arg_list, std::move(stmt), out_binds);
281 f = WithAttr(std::move(f), "global_symbol", runtime::String(name));
282
283 // Mark this schedule as being converted from an TE schedule. Makes sure that
284 // the correct TE passes are run.
285 f = WithAttr(std::move(f), "from_legacy_te_schedule", Bool(true));
286
287 bool noalias = pass_ctx->GetConfig<Bool>("tir.noalias", Bool(true)).value();
288
289 if (noalias) {
290 f = WithAttr(std::move(f), "tir.noalias", Bool(true));
291 }
292 return IRModule(Map<GlobalVar, BaseFunc>({{GlobalVar(name), f}}));
293}

295TVM_REGISTER_GLOBAL("driver.schedule_to_module")
296 .set_body_typed([](te::Schedule sch, const Array<ObjectRef>& args, const String& name,
297 const Map<te::Tensor, tir::Buffer>& binds) {
298 std::unordered_map<te::Tensor, tir::Buffer> c_binds;

300 if (binds.defined()) {
301 for (auto kv : binds) {
302 c_binds.insert({kv.first, kv.second});
303 }
304 }
305 IRModule mod = ScheduleToModule(std::move(sch), args, name, c_binds);
306 return mod;
307 });

369/**
370 * This function takes the input module that contains both the device and host opts.
371 * Then, it applies transformation on the original module before splitting into separate modules for
372 * device and host. Then it also applies transformations on the new splitted modules.
373 */
374std::pair<IRModule, IRModule> SplitMixedModule(IRModule mod_mixed, const Target& target_arg,
375 const Target& target_host_arg) {
376 Target target = target_arg, target_host = target_host_arg;
377 CheckAndUpdateHostConsistency(&target, &target_host);
378
379 ICHECK(mod_mixed.defined()) << "This module must be defined";
380
381 mod_mixed = ApplyPasses(mod_mixed, MixedModulePassManager(mod_mixed, target));
382
383 IRModule host_mod = ApplyPasses(mod_mixed, HostModulePassManager(mod_mixed, target_host));
384
385 IRModule device_mod = ApplyPasses(mod_mixed, DeviceModulePassManager(mod_mixed, target));
386
387 auto keys = target->GetKeys();
388
389 CheckAndUpdateHostConsistency(&target, &target_host);
390
391 bool target_is_gpu = std::find(keys.begin(), keys.end(), "gpu") != keys.end();
392 if (target_is_gpu && device_mod->functions.size() == 0) {
393 DLOG(WARNING) << "Specified target " << target->str()
394 << " but cannot find device code. Did you forget to bind?";
395 }
396
397 return {host_mod, device_mod};
398}

400runtime::Module TIRToRuntime(const Map<Target, IRModule>& inputs_arg,
401 const Target& target_host_arg) {
402 auto pass_ctx = transform::PassContext::Current();
403
404 std::vector<runtime::Module> device_modules;
405 Map<Target, IRModule> inputs = inputs_arg;
406 Target target_host = target_host_arg;
407
408 // Fetch previous defined target host in targets
409 CheckAndUpdateHostConsistency(&inputs, &target_host);
410
411 if (!target_host.defined()) {
412 for (const auto& it : inputs) {
413 if (it.first->kind->device_type == kDLCPU || it.first->kind->device_type == kDLMicroDev) {
414 target_host = it.first;
415 break;
416 }
417 }
418 }
419
420 if (!target_host.defined()) {
421 target_host = DefaultTargetHost(target_host);
422 }
423
424 // Update target host for all targets
425 CheckAndUpdateHostConsistency(&inputs, &target_host);
426
427 // Take the attrs from the first module so the eventual modules have them.
428 // Ideally this would just be one unified module all the way through;
429 IRModule first_module = (*inputs.begin()).second;
430 IRModule mhost_all = IRModule(Map<GlobalVar, BaseFunc>(), {}, {}, {}, first_module->attrs);
431
432 ICHECK(mhost_all.defined()) << "The host module must be defined";
433
434 for (const auto& it : inputs) {
435 if (it.second.defined()) {
436 const Target& target = it.first;
437 const IRModule& ir_module = it.second;
438 auto pair = SplitMixedModule(ir_module, target, target_host);
439 auto& host_mod = pair.first;
440 auto& device_mod = pair.second;
441
442 ICHECK(host_mod.defined()) << "The split host module must be defined";
443
444 ICHECK(mhost_all.defined()) << "The host module must be defined";
445
446 // We don't want library modules going back into host codegen
447 // unless they're supposed to. Here if we overrode the target host
448 // to allow lowering previously we check that it's meant to be placed
449 // back into the host Module.
450 bool overrides_host_target = target->kind->device_type == target_host->kind->device_type;
451 bool non_host_target_kind = target->kind != target_host->kind;
452 if (overrides_host_target && non_host_target_kind) {
453 device_modules.push_back(codegen::Build(host_mod, it.first));
454 } else {
455 mhost_all->Update(host_mod);
456 }
457
458 if (device_mod->functions.size() != 0) {
459 device_modules.push_back(codegen::Build(device_mod, it.first));
460 }
461 }
462 }
463
464 runtime::Module mhost = codegen::Build(mhost_all, target_host);
465 for (const auto& it : device_modules) {
466 if (it.operator->()) {
467 mhost.Import(it);
468 }
469 }
470
471 return mhost;
472}

486// Build for heterogeneous execution when target is a string.
487runtime::Module build(const Map<String, IRModule>& inputs_arg, const Target& target_host_arg) {
488 Map<Target, IRModule> updated_inputs;
489 Target target_host = target_host_arg;
490 for (const auto& it : inputs_arg) {
491 Target target = Target(it.first);
492 CheckAndUpdateHostConsistency(&target, &target_host);
493 Optional<String> device = target->GetAttr<String>("device");
494 if (device.defined() && device.value() == "vta") {
495 target = Target("ext_dev");
496 }
497 updated_inputs.Set(target, it.second);
498 }
499 return TIRToRuntime(updated_inputs, target_host);
500}

512transform::Sequential MixedModulePassManager(IRModule mixed_mod, Target target) {
513 transform::PassContext pass_ctx = transform::PassContext::Current();
514
515 Array<Pass> mixed_pass_list;
516
517 mixed_pass_list.push_back(tir::transform::BindTarget(target));
518
519 mixed_pass_list.push_back(tir::transform::VerifyMemory());
520
521 if (ShouldAnnotateEntryFunc(mixed_mod)) {
522 mixed_pass_list.push_back(tir::transform::AnnotateEntryFunc());
523 }
524
525 bool detect_global_barrier =
526 pass_ctx->GetConfig<Bool>("tir.detect_global_barrier", Bool(false)).value();
527 if (detect_global_barrier) {
528 mixed_pass_list.push_back(tir::transform::ThreadSync("global"));
529 }
530
531 mixed_pass_list.push_back(tir::transform::ThreadSync("shared"));
532 mixed_pass_list.push_back(tir::transform::ThreadSync("shared.dyn"));
533 mixed_pass_list.push_back(tir::transform::MergeDynamicSharedMemoryAllocations());
534 mixed_pass_list.push_back(tir::transform::ThreadSync("warp"));
535 mixed_pass_list.push_back(tir::transform::InferFragment());
536 mixed_pass_list.push_back(tir::transform::LowerThreadAllreduce());
537
538 bool use_ptx_async_copy =
539 pass_ctx->GetConfig<Bool>("tir.use_ptx_async_copy", Bool(false)).value();
540
541 if (use_ptx_async_copy) {
542 mixed_pass_list.push_back(tir::transform::InjectPTXAsyncCopy());
543 }
544
545 bool unpacked_api = mixed_mod->GetAttr<relay::Executor>(tvm::attr::kExecutor)
546 .value_or(relay::Executor::Create("graph", {}))
547 ->GetAttr<Bool>("unpacked-api")
548 .value_or(Bool(false));
549 if (unpacked_api) {
550 mixed_pass_list.push_back(tir::transform::MakeUnpackedAPI());
551 } else {
552 mixed_pass_list.push_back(tir::transform::MakePackedAPI(-1));
553 }
554 mixed_pass_list.push_back(tir::transform::SplitHostDevice());
555
556 return transform::Sequential(mixed_pass_list);
557}

564transform::Sequential HostModulePassManager(IRModule mixed_mod, Target target_host) {
565 Array<tvm::transform::Pass> host_pass_list;
566
567 runtime::TypedPackedFunc<bool(tir::PrimFunc)> fcond = [](const tir::PrimFunc& f) {
568 return f->GetAttr<Integer>(tvm::attr::kCallingConv, Integer(CallingConv::kDefault)) !=
569 CallingConv::kDeviceKernelLaunch;
570 };
571 host_pass_list.push_back(tir::transform::Filter(fcond));
572
573 ICHECK(mixed_mod.defined()) << "This module must be defined";
574
575 host_pass_list.push_back(tir::transform::BindTarget(target_host));
576
