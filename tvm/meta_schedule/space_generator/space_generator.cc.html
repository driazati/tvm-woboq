21namespace tvm {
22namespace meta_schedule {
23
24void PySpaceGeneratorNode::InitializeWithTuneContext(const TuneContext& context) {
25 ICHECK(f_initialize_with_tune_context != nullptr)
26 << "PySpaceGenerator's InitializeWithTuneContext method not implemented!";
27 f_initialize_with_tune_context(context);
28}
29
30Array<tir::Schedule> PySpaceGeneratorNode::GenerateDesignSpace(const IRModule& mod) {
31 ICHECK(f_generate_design_space != nullptr)
32 << "PySpaceGenerator's GenerateDesignSpace method not implemented!";
33 return f_generate_design_space(mod);
34}
35
36SpaceGenerator SpaceGenerator::PySpaceGenerator(
37 PySpaceGeneratorNode::FInitializeWithTuneContext f_initialize_with_tune_context,
38 PySpaceGeneratorNode::FGenerateDesignSpace f_generate_design_space) {
39 ObjectPtr<PySpaceGeneratorNode> n = make_object<PySpaceGeneratorNode>();
40 n->f_initialize_with_tune_context = std::move(f_initialize_with_tune_context);
41 n->f_generate_design_space = std::move(f_generate_design_space);
42 return SpaceGenerator(n);
43}

45TVM_REGISTER_OBJECT_TYPE(SpaceGeneratorNode);

48TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorInitializeWithTuneContext")
49 .set_body_method<SpaceGenerator>(&SpaceGeneratorNode::InitializeWithTuneContext);
50TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorGenerateDesignSpace")
51 .set_body_method<SpaceGenerator>(&SpaceGeneratorNode::GenerateDesignSpace);
52TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorPySpaceGenerator")
53 .set_body_typed(SpaceGenerator::PySpaceGenerator);
