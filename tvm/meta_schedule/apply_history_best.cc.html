28template <class FunctionType, class RetType, class Callback>
29Optional<RetType> GetOnlyOneFunctionCommon(const IRModule& mod, Callback on_found) {
30 if (mod->functions.size() != 1) {
31 return NullOpt;
32 }
33 for (const auto& kv : mod->functions) {
34 const BaseFunc& func = kv.second;
35 if (!func->IsInstance<typename FunctionType::ContainerType>()) {
36 return NullOpt;
37 } else {
38 return on_found(kv);
39 }
40 }
41 return NullOpt;
42}
43
44template <class FunctionType>
45Optional<GlobalVar> GetOnlyOneFunctionKey(const IRModule& mod) {
46 return GetOnlyOneFunctionCommon<FunctionType, GlobalVar>(mod, [](auto kv) { return kv.first; });
47}
48
49template <class FunctionType>
50Optional<FunctionType> GetOnlyOneFunction(const IRModule& mod) {
51 return GetOnlyOneFunctionCommon<FunctionType, FunctionType>(
52 mod, [](auto kv) { return Downcast<FunctionType>(kv.second); });
53}
54
55template <class FunctionType>
56bool HasOnlyOneFunction(const IRModule& mod) {
57 return GetOnlyOneFunction<FunctionType>(mod).defined();
58}

60/**************** Context Manager ****************/
61
62class ApplyHistoryBestInternal {
63 public:
64 static void EnterScope(ApplyHistoryBest ctx) { ctx.EnterWithScope(); }
65 static void ExitScope(ApplyHistoryBest ctx) { ctx.ExitWithScope(); }
66};
67
68struct ApplyHistoryBestThreadLocalEntry {
69 Optional<ApplyHistoryBest> ctx;
70};
71
72using ApplyHistoryBestThreadLocalStore = dmlc::ThreadLocalStore<ApplyHistoryBestThreadLocalEntry>;
73
74Optional<ApplyHistoryBest> ApplyHistoryBest::Current() {
75 return ApplyHistoryBestThreadLocalStore::Get()->ctx;
76}
77
78void ApplyHistoryBest::EnterWithScope() {
79 Optional<ApplyHistoryBest>& ctx = ApplyHistoryBestThreadLocalStore::Get()->ctx;
80 CHECK(!ctx.defined()) << "ValueError: Nested ApplyHistoryBest context managers are not allowed";
81 ctx = *this;
82}
83
84void ApplyHistoryBest::ExitWithScope() {
85 Optional<ApplyHistoryBest>& ctx = ApplyHistoryBestThreadLocalStore::Get()->ctx;
86 ICHECK(ctx.defined());
87 ctx = NullOpt;
88}

92ApplyHistoryBest::ApplyHistoryBest(Database database,
93 ApplyHistoryBestNode::FTEFilterFunc te_filter_func,
94 PackedFunc logging_func) {
95 ObjectPtr<ApplyHistoryBestNode> n = make_object<ApplyHistoryBestNode>();
96 n->database = database;
97 n->te_filter_func = te_filter_func;
98 n->logging_func = logging_func;
99 if (te_filter_func == nullptr) {
100 n->te_filter_func = DefaultTaskFilter;
101 }
102 data_ = n;
103}

105Optional<IRModule> ApplyHistoryBestNode::Query(runtime::String task_name, IRModule mod,
106 Target target, Optional<Array<IRModule>> dispatched,
107 FTakeTuningRecord f_take_tuning_record,
108 FDirectDispatch f_direct_dispatch) {
109 ICHECK(dispatched.defined());
110 ICHECK_EQ(dispatched.value().size(), 1);
111 ICHECK(HasOnlyOneFunction<relay::Function>(mod)) << mod;
112 IRModule prim_mod = dispatched.value()[0];
113 ICHECK(HasOnlyOneFunction<tir::PrimFunc>(prim_mod)) << prim_mod;
114
115 // Keep the original func name to be returned later.
116 GlobalVar gv = GetOnlyOneFunctionKey<tir::PrimFunc>(prim_mod).value();
117
118 // Unify func name to make sure it can be found in database
119 const auto* parse_mod_func = runtime::Registry::Get("tvm.meta_schedule.tune.parse_mod");
120 ICHECK(parse_mod_func) << "Parse mod function not defined!";
121 prim_mod = (*parse_mod_func)(prim_mod);
122
123 if (f_direct_dispatch != nullptr) {
124 Optional<IRModule> mod = f_direct_dispatch(prim_mod);
125 if (mod.defined()) {
126 TVM_PY_LOG(INFO, logging_func) << "Direct dispatch applied for workload: " << task_name;
127 return mod.value();
128 }
129 }
130 if (database->HasWorkload(prim_mod)) {
131 Array<TuningRecord> records = database->GetTopK(database->CommitWorkload(prim_mod), 1);
132 if (records.size() == 1) {
133 if (f_take_tuning_record != nullptr) {
134 f_take_tuning_record(records[0]);
135 }
136 tir::Schedule sch =
137 tir::Schedule::Traced(records[0]->workload->mod, /*seed=*/-1, /*debug_mask=*/0,
138 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kNone);
139 records[0]->trace->ApplyToSchedule(sch, false);
140 tir::PrimFunc func = GetOnlyOneFunction<tir::PrimFunc>(sch->mod()).value();
141 // Make sure we return the updated PrimFunc paired with the original func name.
142 return IRModule({{gv, func}});
143 }
144 }
145 TVM_PY_LOG(WARNING, logging_func) << "Cannot find workload: " << task_name;
146 return NullOpt;
147}
