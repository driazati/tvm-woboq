30ExtractedTask::ExtractedTask(String task_name, IRModule mod, Target target,
31 Array<IRModule> dispatched, int weight) {
32 ObjectPtr<ExtractedTaskNode> n = make_object<ExtractedTaskNode>();
33 n->task_name = task_name;
34 n->mod = mod;
35 n->target = target;
36 n->dispatched = dispatched;
37 n->weight = weight;
38 data_ = n;
39}

41Optional<tir::PrimFunc> DefaultTaskFilterImpl(const Array<te::Tensor>& args, bool allow_extern_op) {
42 using namespace ::tvm::te;
43 std::vector<Tensor> stack;
44 std::unordered_set<const TensorNode*> visited;
45 for (const Tensor& v : args) {
46 for (const PrimExpr& e : v->shape) {
47 // Dynamic shape is not supported for now
48 if (!e->IsInstance<IntImmNode>()) {
49 return NullOpt;
50 }
51 }
52 if (!visited.count(v.get())) {
53 visited.insert(v.get());
54 stack.push_back(v);
55 }
56 }
57 while (!stack.empty()) {
58 Tensor tensor = stack.back();
59 stack.pop_back();
60 if (tensor->op->IsInstance<PlaceholderOpNode>()) {
61 // do nothing
62 } else if (tensor->op->IsInstance<ComputeOpNode>() ||
63 (allow_extern_op && tensor->op->IsInstance<ExternOpNode>())) {
64 Array<Tensor> inputs = tensor->op->InputTensors();
65 for (const Tensor& v : inputs) {
66 if (!visited.count(v.get())) {
67 visited.insert(v.get());
68 stack.push_back(v);
69 }
70 }
71 } else {
72 return NullOpt;
73 }
74 }
75 PrimFunc func = te::CreatePrimFunc(args);
76 bool dynamic_loop_extent = false;
77 PostOrderVisit(func->body, [&dynamic_loop_extent](const ObjectRef& obj) -> void {
78 if (const auto* loop = obj.as<tir::ForNode>()) {
79 if (!loop->extent->IsInstance<IntImmNode>()) {
80 dynamic_loop_extent = true;
81 }
82 }
83 });
84 if (dynamic_loop_extent) {
85 return NullOpt;
86 }
87 return func;
88}
89
90Optional<tir::PrimFunc> DefaultTaskFilter(const Array<te::Tensor>& args) {
91 return DefaultTaskFilterImpl(args, false);
92}
93
94Optional<tir::PrimFunc> DefaultTaskFilterAllowExtern(const Array<te::Tensor>& args) {
95 return DefaultTaskFilterImpl(args, true);
96}
