39HardwareParams::HardwareParams(int num_cores, int vector_unit_bytes, int cache_line_bytes,
40 int max_shared_memory_per_block, int max_local_memory_per_block,
41 int max_threads_per_block, int max_vthread_extent, int warp_size) {
42 auto node = make_object<HardwareParamsNode>();
43 node->num_cores = num_cores;
44 node->vector_unit_bytes = vector_unit_bytes;
45 node->cache_line_bytes = cache_line_bytes;
46 node->max_shared_memory_per_block = max_shared_memory_per_block;
47 node->max_local_memory_per_block = max_local_memory_per_block;
48 node->max_threads_per_block = max_threads_per_block;
49 node->max_vthread_extent = max_vthread_extent;
50 node->warp_size = warp_size;
51 data_ = std::move(node);
52}

54HardwareParams HardwareParamsNode::GetDefaultHardwareParams(const Target& target,
55 const Target& target_host) {
56 // There is no use of target_host so no updates here in the function.
57 const auto device_type = target->kind->device_type;
58 if (device_type == kDLCPU) {
59 return HardwareParams(tvm::runtime::threading::MaxConcurrency(), 64, 64, 0, 0, 0, 0, 0);
60 } else if (device_type == kDLCUDA || device_type == kDLROCM) {
61 auto dev = Device{static_cast<DLDeviceType>(device_type), 0};
62 auto device_name = device_type == kDLCUDA ? "device_api.cuda" : "device_api.rocm";
63 auto func = tvm::runtime::Registry::Get(device_name);
64 ICHECK(func != nullptr) << "Cannot find CUDA device_api in registry";
65 auto device_api = static_cast<tvm::runtime::DeviceAPI*>(((*func)()).operator void*());
66
67 tvm::runtime::TVMRetValue ret;
68 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kMaxSharedMemoryPerBlock, &ret);
69 int max_shared_memory_per_block = ret;
70
71 // There is no explicit local memory limition in CUDA runtime,
72 // so we can use INT32_MAX to disalbe the check on local_memory.
73 int max_local_memory_per_block = INT32_MAX;
74
75 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kMaxThreadsPerBlock, &ret);
76 int max_threads_per_block = ret;
77
78 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kWarpSize, &ret);
79 int warp_size = ret;
80
81 int max_vthread_extent = warp_size / 4;
82 return HardwareParams(-1, 16, 64, max_shared_memory_per_block, max_local_memory_per_block,
83 max_threads_per_block, max_vthread_extent, warp_size);
84 } else if (device_type == kDLMetal) {
85 // Reference: https://developer.apple.com/metal/Metal-Feature-Set-Tables.pdf
86 // This setting looks working for Metal GPUs later than A10
87 int max_shared_memory_per_block = 32 * 1024;
88 int max_local_memory_per_block = INT32_MAX; // skip the check on local memory
89 int max_threads_per_block = 1024;
90 int warp_size = 8;
91 int max_vthread_extent = warp_size / 4;
92 return HardwareParams(-1, 16, 64, max_shared_memory_per_block, max_local_memory_per_block,
93 max_threads_per_block, max_vthread_extent, warp_size);
94 } else if (target->kind->device_type == kDLOpenCL) {
95 if (target->GetAttr<String>("device", "") == "mali") {
96 // We cannot use device API to get hardware attributes like CUDA,
97 // because like Mali target is normally on the remote machine.
98 int max_shared_memory_per_block = 32768;
99 int max_local_memory_per_block = INT32_MAX; // skip the check on local memory
100 int max_threads_per_block = 256;
101 int warp_size = 1;
102 int max_vthread_extent = 1;
103 return HardwareParams(-1, 16, 64, max_shared_memory_per_block, max_local_memory_per_block,
104 max_threads_per_block, max_vthread_extent, warp_size);
105 } else {
106 // add other opencl target
107 auto dev = Device{static_cast<DLDeviceType>(device_type), 0};
108 auto device_name = "device_api.opencl";
109 auto func = tvm::runtime::Registry::Get(device_name);
110 ICHECK(func != nullptr) << "Cannot find OpenCL device_api in registry";
111 auto device_api = static_cast<tvm::runtime::DeviceAPI*>(((*func)()).operator void*());
112
113 tvm::runtime::TVMRetValue ret;
114 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kMaxSharedMemoryPerBlock, &ret);
115 int max_shared_memory_per_block = ret;
116
117 int max_local_memory_per_block = INT32_MAX;
118
119 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kMaxThreadsPerBlock, &ret);
120 int max_threads_per_block = ret;
121
122 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kWarpSize, &ret);
123 int warp_size = ret;
124
125 if (warp_size == 1) {
126 LOG(WARNING)
127 << "Warp size 1 is not recommended for OpenCL devices. Tuning might crash or stuck";
128 }
129
130 int max_vthread_extent = std::max(1, warp_size / 4);
131 return HardwareParams(-1, 16, 64, max_shared_memory_per_block, max_local_memory_per_block,
132 max_threads_per_block, max_vthread_extent, warp_size);
133 }
134 } else if (device_type == kDLVulkan) {
135 auto dev = Device{static_cast<DLDeviceType>(device_type), 0};
136 auto device_name = "device_api.vulkan";
137 auto func = tvm::runtime::Registry::Get(device_name);
138 ICHECK(func != nullptr) << "Cannot find Vulkan device_api in registry";
139 auto device_api = static_cast<tvm::runtime::DeviceAPI*>(((*func)()).operator void*());
140
141 tvm::runtime::TVMRetValue ret;
142 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kMaxSharedMemoryPerBlock, &ret);
143 int max_shared_memory_per_block = ret;
144
145 int max_local_memory_per_block = INT32_MAX;
146
147 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kMaxThreadsPerBlock, &ret);
148 int max_threads_per_block = ret;
149
150 device_api->GetAttr(dev, tvm::runtime::DeviceAttrKind::kWarpSize, &ret);
151 int warp_size = ret;
152
153 int max_vthread_extent = std::max(1, warp_size / 4);
154
155 return HardwareParams(-1, 16, 64, max_shared_memory_per_block, max_local_memory_per_block,
156 max_threads_per_block, max_vthread_extent, warp_size);
157 } else {
158 LOG(FATAL) << "No default hardware parameters for target: " << target;
159 }
160 return HardwareParams();
161}

163SearchTask::SearchTask(ComputeDAG compute_dag, String workload_key, Target target,
164 Target target_host, Optional<HardwareParams> hardware_params,
165 LayoutRewriteOption layout_rewrite_option, Array<String> task_input_names,
166 String desc) {
167 CheckAndUpdateHostConsistency(&target, &target_host);
168 auto node = make_object<SearchTaskNode>();
169 node->compute_dag = std::move(compute_dag);
170 node->workload_key = std::move(workload_key);
171 node->desc = std::move(desc);
172 node->target = std::move(target);
173 node->target_host = std::move(target_host);
174 if (hardware_params) {
175 node->hardware_params = hardware_params.value();
176 } else {
177 node->hardware_params =
178 HardwareParamsNode::GetDefaultHardwareParams(node->target, node->target_host);
179 }
180 node->layout_rewrite_option = layout_rewrite_option;
181 node->task_input_names = std::move(task_input_names);
182 data_ = std::move(node);
183}

197 });

199TVM_REGISTER_GLOBAL("auto_scheduler.SearchTask")
200 .set_body_typed([](ComputeDAG compute_dag, String workload_key, Target target,
201 Target target_host, Optional<HardwareParams> hardware_params,
202 int layout_rewrite_option, Array<String> task_input_names, String desc) {
203 CheckAndUpdateHostConsistency(&target, &target_host);
204 return SearchTask(compute_dag, workload_key, target, target_host, hardware_params,
205 LayoutRewriteOption(layout_rewrite_option), task_input_names, desc);
206 });
