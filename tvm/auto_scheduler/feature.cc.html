166// Return whether a var is in an expr
167bool VarInExpr(const Var& var, const PrimExpr& expr) {
168 bool find = false;
169
170 PostOrderVisit(expr, [&find, &var](const ObjectRef& node) {
171 if (find) {
172 return;
173 }
174
175 if (const VarNode* op = node.as<VarNode>()) {
176 if (op == var.get()) {
177 find = true;
178 }
179 }
180 });
181
182 return find;
183}

185// Get position encoding for annotation
186AnnotationPosType GetAnnotationPosEncoding(const Var& var, const Array<PrimExpr>& spatial_args,
187 const Array<IterVar>& axis,
188 const Array<IterVar>& reduce_axis) {
189 // Try to match spatial args first
190 size_t find_i = 0;
191 size_t find_ct = 0;
192 for (size_t i = 0; i < spatial_args.size(); ++i) {
193 if (VarInExpr(var, spatial_args[i])) {
194 find_i = i;
195 find_ct += 1;
196 }
197 }
198
199 if (find_ct == 0) {
200 // If it is not found in spacial args, then it is a reduce iterator.
201 // Use name to match
202 const std::string& var_name = var->name_hint;
203 for (size_t i = 0; i < reduce_axis.size(); ++i) {
204 if (var_name.find(reduce_axis[i]->var->name_hint) != std::string::npos) {
205 find_i = i;
206 find_ct++;
207 }
208 }
209 if (find_ct >= 1) {
210 if (find_i == 0) {
211 return AnnotationPosType::kPosInnerReduce;
212 } else if (find_i == reduce_axis.size() - 1) {
213 return AnnotationPosType::kPosOuterReduce;
214 } else {
215 return AnnotationPosType::kPosMiddleReduce;
216 }
217 } else {
218 // If the axis is not found in both spatial args and reduce axis,
219 // then this stage must compute_at somewhere under this axis and this axis is simplified out
220 // We assume it is an outer spatial
221 return AnnotationPosType::kPosOuterSpatial;
222 }
223 } else if (find_ct == 1) {
224 if (find_i == spatial_args.size() - 1) {
225 return AnnotationPosType::kPosInnerSpatial;
226 } else if (find_i == 0) {
227 return AnnotationPosType::kPosOuterSpatial;
228 } else {
229 return AnnotationPosType::kPosMiddleSpatial;
230 }
231 } else {
232 return AnnotationPosType::kPosMixed;
233 }
234}

236// Return the maximum extent of a for loop
237int64_t GetLoopExtent(const ForNode* node, const Analyzer& ana) {
238 int64_t bound = ana.const_int_bound(node->extent)->max_value;
239 if (bound == ConstIntBound::kPosInf) {
240 return 1; // Analyzer could not determine a valid bound, use 1 instead.
241 } else {
242 return bound;
243 }

338// Extract all buffer accesses in an expr
339class BufferAccessExtractor : public StmtExprVisitor {
340 public:
341 void ExtractReads(const PrimExpr& expr) { this->VisitExpr(expr); }
342
343 void InsertAccess(const Var& buf, BufferAccessType acc_type, const Array<PrimExpr>& indices) {
344 BufferAccess& acc = buf_accesses[buf];
345 acc.acc_type = acc_type;
346 acc.indices.push_back(std::vector<PrimExpr>(indices.begin(), indices.end()));
347 }
348
349 void VisitExpr_(const BufferLoadNode* op) final {
350 AddAccess(op->buffer->data, op->indices);
351 StmtExprVisitor::VisitExpr_(op);
352 }
353
354 void AddAccess(const Var& buffer, const Array<PrimExpr>& indices) {
355 BufferAccess& acc = buf_accesses[buffer];
356 switch (acc.acc_type) {
357 case BufferAccessType::kRead:
358 break;
359 case BufferAccessType::kWrite:
360 acc.acc_type = BufferAccessType::kReadWrite;
361 break;
362 case BufferAccessType::kReadWrite:
363 break;
364 case BufferAccessType::kUnknownRW:
365 default:
366 acc.acc_type = BufferAccessType::kRead;
367 break;
368 }
369
370 if (acc.acc_type != BufferAccessType::kReadWrite) {
371 // If a buffer is both read and written, in the tvm DSL, it must be a update,
372 // so the indices should be the same. Then we can skip appending indices for it.
373 // Otherwise we do the following.
374 buf_accesses[buffer].indices.push_back(std::vector<PrimExpr>(indices.begin(), indices.end()));
375 }
376 }
377
378 BufferMap<BufferAccess> buf_accesses;
379};

381// Compute the coefficient for an loop iterator in an expression
382// Note: we use an approximation strategy to find coefficient.
383// Hopefully, it is faster than DetectLinearEquation and can handle more cases (non-linear)
384class CoefficientExtractor : public StmtExprVisitor {
385 public:
386 void VisitExpr_(const MulNode* node) final {
387 StmtExprVisitor::VisitExpr_(node);
388 if (visited_var) {
389 if (!visited_add) {
390 if (auto a = node->a.as<IntImmNode>()) {
391 visited_mul = true;
392 stride = a->value;
393 } else if (auto b = node->b.as<IntImmNode>()) {
394 visited_mul = true;
395 stride = b->value;
396 }
397 }
398 }
399 }
400
401 void VisitExpr_(const AddNode* node) final {
402 StmtExprVisitor::VisitExpr_(node);
403 if (visited_var) {
404 if (!visited_mul) {
405 visited_add = true;
406 stride = 1;
407 }
408 }
409 }
410
411 void VisitExpr_(const VarNode* node) final {
412 if (node == var_) {
413 visited_var = true;
414 // This is a magic default stride in case our approximation strategy fails
415 stride = 2;
416 }
417 }
418
419 int ExtractCoefficient(const PrimExpr& expr, const VarNode* var) {
420 visited_var = visited_mul = visited_add = false;
421 var_ = var;
422
423 this->VisitExpr(expr);
424
425 if (visited_var && !visited_mul && !visited_add) {
426 return 1;
427 } else {
428 return stride;
429 }
430 }
431
432 bool visited_var{false};
433 bool visited_mul{false};
434 bool visited_add{false};
435 int stride{0};
436
437 private:
438 const VarNode* var_{nullptr};
439};

441// Compute stride for the accesses to a buffer
442int64_t ComputeStride(const std::vector<std::vector<PrimExpr>>& indices,
443 const std::vector<int>& shape, const VarNode* stride_var) {
444 int64_t min_stride = std::numeric_limits<int64_t>::max();
445 bool find = false;
446 CoefficientExtractor extractor;
447
448 for (const auto& index : indices) {
449 int64_t shape_stride = 1;
450 for (int i = static_cast<int>(index.size()) - 1; i >= 0; i--) {
451 int coefficient = extractor.ExtractCoefficient(index[i], stride_var);
452 if (extractor.visited_var) {
453 find = true;
454 min_stride = std::min(min_stride, std::abs(coefficient) * shape_stride);
455 break;
456 }
457 shape_stride *= shape[i];
458 }
459 }
460
461 return find ? min_stride : 0;
462}

464// Compute touched bytes and cache lines for accesses to a buffer
465void ComputeRegion(const std::vector<std::vector<PrimExpr>>& indices, arith::Analyzer* ana,
466 std::vector<int>* region) {
467 region->clear();
468
469 if (indices.empty()) {
470 return;
471 }
472
473 region->reserve(indices[0].size());
474
475 if (indices.size() == 1) {
476 for (const auto& index : indices[0]) {
477 ConstIntBound bound = ana->const_int_bound(index);
478 region->push_back(bound->max_value - bound->min_value + 1);
479 }
480 } else {
481 // future(lmzheng): implement a more accurate IntSet?
482 for (size_t i = 0; i < indices[0].size(); ++i) {
483 int64_t minimum = ConstIntBound::kPosInf, maximum = ConstIntBound::kNegInf;
484 for (size_t j = 0; j < indices.size(); ++j) {
485 ConstIntBound bound = ana->const_int_bound(indices[j][i]);
486
487 minimum = std::min(minimum, bound->min_value);
488 maximum = std::max(maximum, bound->max_value);
489 }
490 region->push_back(maximum - minimum + 1);
491 }
492 }
493}

495// Compute reuse distance and reuse ratio for accesses to a buffer
496// return values: reuse_type, reuse_dis_iter, reuse_dis_bytes, reuse_ct
497std::tuple<ReuseType, float, float, float> ComputeReuse(
498 const Var& buf, const std::vector<std::vector<PrimExpr>>& indices,
499 const std::vector<const ForNode*>& for_loop_stack,
500 const std::unordered_map<const ForNode*,
501 BufferMap<std::vector<std::tuple<BufferAccessType, int64_t, int>>>>&
502 for_touch_regions,
503 const Analyzer& ana) {
504 float reuse_dis_iter = 1.0f;
505 float reuse_dis_bytes = -1.0f;
506
507 for (int i = static_cast<int>(for_loop_stack.size()) - 1; i >= 0; --i) {
508 const ForNode* cur_for = for_loop_stack[i];
509 bool find = false;
510
511 for (size_t j = 0; j < indices.size(); j++) {
512 for (size_t k = 0; k < indices[j].size(); k++) {
513 if (VarInExpr(cur_for->loop_var, indices[j][k])) {
514 find = true;
515 break;
516 }
517 }
518 if (find) {
519 break;
520 }
521 }
522
523 int64_t extent = GetLoopExtent(for_loop_stack[i], ana);
524 if (find) {
525 // accumulate/update reuse distance
526 reuse_dis_iter *= extent;
527 reuse_dis_bytes = 0.0f;
528 for (const auto& iter : for_touch_regions.at(cur_for)) {
529 for (const auto& access : iter.second) {
530 reuse_dis_bytes += std::get<1>(access) * std::get<2>(access);
531 }
532 }
533 } else {
534 // Have LoopMultipleRead reuse
535 if (reuse_dis_bytes < 0) {
536 // For the reuse in the innermost axis, the above code won't be executed.
537 // So we compute bytes here
538 reuse_dis_bytes = 0.0f;
539 for (const auto& iter : for_touch_regions.at(cur_for)) {
540 for (const auto& access : iter.second) {
541 reuse_dis_bytes += 1 * std::get<2>(access);
542 }
543 }
544 }
545 return std::make_tuple(ReuseType::kLoopMultipleRead, reuse_dis_iter, reuse_dis_bytes, extent);
546 }
547
548 const BufferMap<std::vector<std::tuple<BufferAccessType, int64_t, int>>>& buffer_map =
549 for_touch_regions.at(cur_for);
550
551 int serial_reuse = static_cast<int>(buffer_map.at(buf).size()) - 1;
552 if (serial_reuse > 0) {
553 int64_t extent = GetLoopExtent(cur_for, ana);
554
555 // Have SerialMultipleReadWrite reuse
556 reuse_dis_iter = std::numeric_limits<float>::max();
557 for (const auto& acc_info : buffer_map.at(buf)) {
558 reuse_dis_iter = std::min(reuse_dis_iter, static_cast<float>(std::get<1>(acc_info)));
559 }
560
561 reuse_dis_bytes = 0.0f;
562 for (const auto& iter : for_touch_regions.at(cur_for)) {
563 for (const auto& access : iter.second) {
564 reuse_dis_bytes += std::get<1>(access) * std::get<2>(access);
565 }
566 }
567
568 return std::make_tuple(ReuseType::kSerialMultipleReadWrite, reuse_dis_iter / extent,
569 reuse_dis_bytes / extent, serial_reuse);
570 }
571 }
572
573 return std::make_tuple(ReuseType::kNoReuse, 0, 0, 0);
574}

576// Extract features for every BufferStore statement
577//
578// This visitor assumes that loop bounds do no depend on data or on parent loop
579// bounds. For example, `for i in .. { for j in range(i, ..) }` would result in
580// inaccurate features. This visitor also does not take conditionals into
581// consideration when creating features. Each branch of the conditional is
582// taken at the same time.
583class PerStoreFeatureExtractor : public StmtExprVisitor {
584 public:
585 explicit PerStoreFeatureExtractor(int cache_line_size, const Map<Var, Buffer>& existing_buffers)
586 : cache_line_size_(cache_line_size) {
587 for (const auto& buffer : existing_buffers) {
588 buffer_shapes[buffer.first] = buffer.second->shape;
589 buffer_dtypes[buffer.first] = buffer.second->dtype;
590 // Also need to add a reference from the buffers internal variable. This
591 // is usually how buffers are referenced within the body of a PrimFunc
592 buffer_shapes[buffer.second->data] = buffer.second->shape;
593 buffer_dtypes[buffer.second->data] = buffer.second->dtype;
594 }
595 }

597 void VisitStmt_(const AttrStmtNode* node) final {
598 if (node->attr_key == tir::attr::thread_extent || node->attr_key == tir::attr::virtual_thread) {
599 const Var& var = node->node.as<IterVarNode>()->var;
600 int extent = GetIntImm(node->value);
601
602 int* plen = nullptr;
603
604 const std::string& name = var.get()->name_hint;
605 if (node->attr_key == tir::attr::thread_extent) {
606 if (name == "blockIdx.x") {
607 plen = &blockIdx_x_len_;
608 } else if (name == "blockIdx.y") {
609 plen = &block_idx_y_len_;
610 } else if (name == "blockIdx.z") {
611 plen = &block_idx_z_len_;
612 } else if (name == "threadIdx.x") {
613 plen = &threadIdx_x_len_;
614 } else if (name == "threadIdx.y") {
615 plen = &thread_idx_y_len_;
616 } else if (name == "threadIdx.z") {
617 plen = &thread_idx_z_len_;
618 } else {
619 LOG(FATAL) << "invalid thread itervar " + name;
620 }
621 } else {
622 plen = &vthread_len_;
623 }
624
625 int extent_before = *plen;
626 if (node->attr_key == tir::attr::thread_extent) {
627 *plen = extent;
628 } else {
629 *plen *= extent;
630 }
631
632 is_gpu_ = true;
633
634 // make a fake for node for blockIdx.x or threadIdx.x
635 Stmt fake_for_node = For(var, 0, extent, ForKind::kParallel, node->body);
636
637 outer_loop_prod_ *= extent;
638 for_loop_stack_.push_back(fake_for_node.as<ForNode>());
639 variable_definition_stack_.push_back({});
640 StmtExprVisitor::VisitStmt_(node);
641 variable_definition_stack_.pop_back();
642 for_loop_stack_.pop_back();
643 outer_loop_prod_ /= extent;
644
645 *plen = extent_before;

822 // Extract buffer access related features (group 2)
823 void ExtractBufferAccessFeature(const Var& buffer, const Array<PrimExpr>& indices,
824 const PrimExpr& value, const MathOpCounter& math_op_counter,
825 double* cur_compute_ops, std::vector<float>* compute_ops_list,
826 std::vector<float>* mem_bytes_list) {
827 FeatureSet& fea = buffer_features[buffer];
828
829 // Extract all buffer accesses
830 std::vector<BufferAccessFeature> acc_feas;
831 BufferAccessExtractor buf_extractor;
832 buf_extractor.InsertAccess(buffer, BufferAccessType::kWrite, indices);
833 buf_extractor.ExtractReads(value);
834
835 mem_bytes_list->reserve(for_loop_stack_.size());
836 compute_ops_list->reserve(for_loop_stack_.size());
837
838 *cur_compute_ops = math_op_counter.float_mad + math_op_counter.float_addsub +
839 math_op_counter.float_mul + math_op_counter.float_divmod +
840 math_op_counter.float_cmp + math_op_counter.float_math_func +
841 math_op_counter.float_other_func;
842
843 ICHECK_EQ(for_loop_stack_.size(), variable_definition_stack_.size())
844 << "variable_definition_stack_ should mirror for_loop_stack_ in size";
845 std::vector<int> tmp_region;
846 for (int i = static_cast<int>(for_loop_stack_.size()) - 1; i >= 0; i--) {
847 const ForNode* p_for = for_loop_stack_[i];
848
849 // Construct a local analyzer context which contains definitions (for and
850 // let) from innermost loops up to and including `i`. For loop variable
851 // definitions in loops more outer than `i` are set to 1 so that we can
852 // get per-loop-iteration features. Note that we add these definitions
853 // from outermost to innermost because inner definitions may depend on
854 // outer ones.
855 Analyzer local_analyzer;
856 for (int j = 0; j < i; j++) {
857 local_analyzer.Bind(for_loop_stack_.at(j)->loop_var,
858 Range::FromMinExtent(for_loop_stack_.at(j)->min, 1));
859 }
860 for (int j = i; j < static_cast<int>(for_loop_stack_.size()); j++) {
861 local_analyzer.Bind(
862 for_loop_stack_.at(j)->loop_var,
863 Range::FromMinExtent(for_loop_stack_.at(j)->min, for_loop_stack_.at(j)->extent));
864 for (auto definition : variable_definition_stack_.at(j)) {
865 local_analyzer.Bind(std::get<0>(definition), std::get<1>(definition));
866 }
867 }
868
869 // Note, here we do overwrite.
870 // So if there are multiple BufferStoreNode, the last one will overwrite the first few.
871 // e.g. The update part in gemm will overwrite the init part.
872 BufferMap<std::vector<std::tuple<BufferAccessType, int64_t, int>>>& buffer_regions_map =
873 for_touch_regions_[p_for];
874
875 int64_t mem_bytes = 0;
876 for (const auto& x : buf_extractor.buf_accesses) {
877 const Var& t = x.first;
878 const BufferAccess& acc = x.second;
879
880 ComputeRegion(acc.indices, &local_analyzer, &tmp_region);
881 int64_t touched_size = ElementProduct(tmp_region);
882 buffer_regions_map[t].push_back(
883 std::make_tuple(acc.acc_type, touched_size, buffer_dtypes.at(t).bytes()));
884 mem_bytes += touched_size * buffer_dtypes.at(t).bytes();
885 }
886
887 mem_bytes_list->push_back(mem_bytes);
888 *cur_compute_ops *= GetLoopExtent(for_loop_stack_[i], local_analyzer);
889 compute_ops_list->push_back(*cur_compute_ops);
890 }
891
892 // Buffer access related features (per buffer)
893 for (const auto& x : buf_extractor.buf_accesses) {
894 const Var& t = x.first;
895 const BufferAccess& acc = x.second;
896
897 std::vector<int> int_shape;
898 for (const auto& dim : buffer_shapes.at(t)) {
899 int_shape.push_back(GetIntImm(dim));
900 }
901
902 size_t ele_bytes = buffer_dtypes.at(t).bytes();
903
904 // calculate bytes
905 float bytes = outer_loop_prod_ * ele_bytes;
906 float unique_bytes;
907
908 // calculate cache lines
909 int64_t stride;
910 float lines;
911 float unique_lines;
912
913 if (for_loop_stack_.empty()) {
914 unique_bytes = ele_bytes;
915 stride = 0;
916 lines = 1.0f;
917 unique_lines = 1.0f;
918 } else {
919 unique_bytes =
920 std::get<1>(for_touch_regions_[for_loop_stack_.front()][t].front()) * ele_bytes;
921
922 stride = 0;
923 int64_t reduce_ratio = 1;
924
925 int i;
926 for (i = static_cast<int>(for_loop_stack_.size()) - 1; i >= 0; i--) {
927 stride = ComputeStride(acc.indices, int_shape, for_loop_stack_[i]->loop_var.get());
928 if (stride != 0) {
929 break;
930 }
931 reduce_ratio *= GetLoopExtent(for_loop_stack_.back(), ana_);
932 }
933
934 lines = outer_loop_prod_ / reduce_ratio *
935 std::min(1.0f, 1.0f * stride * ele_bytes / cache_line_size_);
936 lines = std::max(lines, 1.0f);
937
938 // convert `stride` back to the stride of the innermost iterator
939 stride = (i == static_cast<int>(for_loop_stack_.size()) - 1 ? stride : 0);
940
941 float n_continuous = ele_bytes;
942 for (int i = std::min(static_cast<int>(tmp_region.size()) - 1,
943 static_cast<int>(int_shape.size()) - 1);
944 i >= 0; i--) {
945 if (tmp_region[i] == int_shape[i]) {
946 n_continuous *= tmp_region[i];
947 break;
948 }
949 }
950 unique_lines = unique_bytes / std::min(n_continuous, static_cast<float>(cache_line_size_));
951 unique_lines = std::max(unique_lines, 1.0f);
952 }
953
954 ReuseType reuse_type;
955 float reuse_dis_iter, reuse_dis_bytes, reuse_ct;
956 std::tie(reuse_type, reuse_dis_iter, reuse_dis_bytes, reuse_ct) =
957 ComputeReuse(t, acc.indices, for_loop_stack_, for_touch_regions_, ana_);
958
959 acc_feas.emplace_back();
960 BufferAccessFeature& acc_fea = acc_feas.back();
961
962 // TODO(tkonolige): save buffer names and use those instead?
963 acc_fea.buffer_name = t->name_hint;
964 acc_fea.acc_type = acc.acc_type;
965 acc_fea.stride = stride;
966 acc_fea.bytes = bytes;
967 acc_fea.unique_bytes = unique_bytes;
968 acc_fea.lines = lines;
969 acc_fea.unique_lines = unique_lines;
970 acc_fea.reuse_type = reuse_type;
971 acc_fea.reuse_dis_iter = reuse_dis_iter;
972 acc_fea.reuse_dis_bytes = reuse_dis_bytes;
973 acc_fea.reuse_ct = reuse_ct;
974 if (acc_fea.reuse_ct > 0.5) {
975 acc_fea.bytes_d_reuse_ct = bytes / reuse_ct;
976 acc_fea.unique_bytes_d_reuse_ct = unique_bytes / reuse_ct;
977 acc_fea.lines_d_reuse_ct = lines / reuse_ct;
978 acc_fea.unique_lines_d_reuse_ct = unique_lines / reuse_ct;
979 } else {
980 // no reuse, multiply by a magic number '2'
981 acc_fea.bytes_d_reuse_ct = bytes * 2;
982 acc_fea.unique_bytes_d_reuse_ct = unique_bytes * 2;
983 acc_fea.lines_d_reuse_ct = lines * 2;
984 acc_fea.unique_lines_d_reuse_ct = unique_lines * 2;
985 }
986 }
987
988 fea.access_feas = acc_feas;
989 }

991 // Extract arithmetic intensity related feature (group 3)
992 void ExtractArithmeticIntensityFeature(const Var& buffer, double cur_compute_ops,
993 const std::vector<float>& compute_ops_list,
994 const std::vector<float>& mem_bytes_list) {
995 FeatureSet& fea = buffer_features[buffer];
996
997 // Compute arithmetic intensity curve (y axis : arithmetic intensity, x axis : flops).
998 // We use piecewise linear interpolation to fit this curve.
999 int pt = 0;
1000 if (cur_compute_ops <= 0 || compute_ops_list.empty()) {
1001 std::fill(fea.arith_intensity_curve,
1002 fea.arith_intensity_curve + ARITH_INTENSITY_CURVE_SAMPLE_N, 0.0);
1003 } else {
1004 for (size_t i = 0; i < ARITH_INTENSITY_CURVE_SAMPLE_N; ++i) {
1005 float cur_compute_ops = compute_ops_list.back() * (i + 1) / ARITH_INTENSITY_CURVE_SAMPLE_N;
1006 while (compute_ops_list[pt] < cur_compute_ops - 1e-4) {
1007 pt++;
1008 }
1009 ICHECK_LT(pt, compute_ops_list.size());
1010
1011 float value;
1012 if (pt == 0) {
1013 value = compute_ops_list[pt] / mem_bytes_list[pt];
1014 } else {
1015 float base = compute_ops_list[pt - 1] / mem_bytes_list[pt - 1];
1016 float slope = (compute_ops_list[pt] / mem_bytes_list[pt] -
1017 compute_ops_list[pt - 1] / mem_bytes_list[pt - 1]) /
1018 (compute_ops_list[pt] - compute_ops_list[pt - 1]);
1019 value = base + slope * (cur_compute_ops - compute_ops_list[pt - 1]);
1020 }
1021 fea.arith_intensity_curve[i] = value;
1022 }
1023 }

1356void GetPerStoreFeaturesWorkerFunc(const SearchTask& task, const State& state, int max_n_bufs,
1357 std::vector<float>* feature, std::atomic<int>* error_ct) {
1358 te::Schedule sch;
1359 Array<te::Tensor> tensors;
1360
1361 std::tie(sch, tensors) = task->compute_dag.ApplySteps(state->transform_steps);
1362
1363 // When inlining, replace const matrices with const values.
1364 // Produces wrong IR, but good enough for feature extraction, and
1365 // can improve the speed of feature extraction/search. Must be
1366 // called before ScheduleToModule to have an effect.
1367 sch = sch.normalize_for_feature_extraction();
1368
1369 try {
1370 const std::string& name = "main";
1371 auto pass_ctx = tvm::transform::PassContext::Current();
1372
1373 auto mod = ScheduleToModule(sch, Array<ObjectRef>{tensors.begin(), tensors.end()}, name,
1374 std::unordered_map<te::Tensor, te::Buffer>());
1375
1376 bool disable_vectorize =
1377 pass_ctx->GetConfig<Bool>("tir.disable_vectorize", Bool(false)).value();
1378 bool instrument_bound_checkers =
1379 pass_ctx->GetConfig<Bool>("tir.instrument_bound_checkers", Bool(false)).value();
1380
1381 if (IsGPUTask(task)) {
1382 auto pass_list = Array<tvm::transform::Pass>();
1383 // Phase 0
1384 pass_list.push_back(tir::transform::InjectPrefetch());
1385 pass_list.push_back(tir::transform::StorageFlatten(64, instrument_bound_checkers));
1386 // Phase 1
1387 pass_list.push_back(tir::transform::NarrowDataType(32));
1388 pass_list.push_back(tir::transform::Simplify());
1389 pass_list.push_back(tir::transform::VectorizeLoop(!disable_vectorize));
1390 pass_list.push_back(tir::transform::InjectVirtualThread());
1391 pass_list.push_back(tir::transform::StorageRewrite());
1392 pass_list.push_back(tir::transform::Simplify());
1393 tvm::Map<String, tvm::PrimExpr> gpu_params{
1394 {"max_shared_memory_per_block", task->hardware_params->max_shared_memory_per_block},
1395 {"max_local_memory_per_block", task->hardware_params->max_local_memory_per_block},
1396 {"max_threads_per_block", task->hardware_params->max_threads_per_block},
1397 {"max_vector_bytes", task->hardware_params->vector_unit_bytes},
1398 {"max_vthread", task->hardware_params->max_vthread_extent},
1399 };
1400 pass_list.push_back(tir::transform::VerifyGPUCode(gpu_params));
1401 const auto& optimize = tir::transform::Sequential(pass_list);
1402 optimize(mod);
1403 }
1404 const auto& optimize =
1405 tir::transform::Sequential(Array<tvm::transform::Pass>{tir::transform::Simplify()});
1406 mod = optimize(std::move(mod));
1407 PrimFunc prim_func = Downcast<PrimFunc>(mod->Lookup(name));
1408 GetPerStoreFeature(prim_func, task->hardware_params->cache_line_bytes, max_n_bufs, feature);
1409 } catch (Error& e) {
1410 (*error_ct)++;
1411 }

1444void GetPerStoreFeaturesFromFile(const std::string& filename, int max_lines, int max_n_bufs,
1445 std::vector<std::vector<float>>* features,
1446 std::vector<float>* normalized_throughputs,
1447 std::vector<int>* task_ids) {
1448 Array<State> states;
1449 std::vector<SearchTask> tasks;
1450
1451 normalized_throughputs->clear();
1452 task_ids->clear();
1453
1454 // (workload_key, target) -> (search_task, task_id)
1455 std::unordered_map<std::pair<std::string, std::string>, std::pair<SearchTask, size_t>> task_cache;
1456 // task_id -> min_cost
1457 std::vector<float> min_costs;
1458
1459 const auto* workload_key_to_tensors =
1460 tvm::runtime::Registry::Get("auto_scheduler.workload_key_to_tensors");
1461 ICHECK(workload_key_to_tensors != nullptr);
1462
1463 // read from file
1464 RecordReader reader(filename);
1465 auto cur_inp = make_object<MeasureInputNode>();
1466 auto cur_res = make_object<MeasureResultNode>();
1467 while (reader->ReadNext(cur_inp.get(), cur_res.get())) {
1468 float cost = static_cast<float>(FloatArrayMean(cur_res->costs));
1469 const std::string& workload_key = cur_inp->task->workload_key;
1470
1471 SearchTask task;
1472 size_t task_id;
1473 std::pair<std::string, std::string> key(workload_key, cur_inp->task->target->str());
1474 auto find_res = task_cache.find(key);
1475 if (find_res == task_cache.end()) {
1476 // rebuild task
1477 Array<te::Tensor> tensors = (*workload_key_to_tensors)(workload_key);
1478 Target target = cur_inp->task->target;
1479 Target target_host = cur_inp->task->target_host;
1480 CheckAndUpdateHostConsistency(&target, &target_host);
1481 task = SearchTask(ComputeDAG(tensors), workload_key, target, target_host,
1482 cur_inp->task->hardware_params, cur_inp->task->layout_rewrite_option,
1483 cur_inp->task->task_input_names);
1484 task_id = task_cache.size();
1485
1486 // compute min cost for each task
1487 task_cache.insert(std::make_pair(key, std::make_pair(task, task_id)));
1488 min_costs.push_back(cost);
1489 } else {
1490 std::tie(task, task_id) = find_res->second;
1491 min_costs[task_id] = std::min(min_costs[task_id], cost);
1492 }
1493
1494 tasks.push_back(std::move(task));
1495 task_ids->push_back(task_id);
1496 states.push_back(cur_inp->state);
1497 normalized_throughputs->push_back(cost);
1498
1499 if (max_lines > 0 && static_cast<int>(states.size()) >= max_lines) {
1500 break;
1501 }
1502 }
1503
1504 for (size_t i = 0; i < normalized_throughputs->size(); ++i) {
1505 (*normalized_throughputs)[i] = min_costs[(*task_ids)[i]] / (*normalized_throughputs)[i];
1506 }
1507
1508 GetPerStoreFeaturesFromStates(states, tasks, 0, max_n_bufs, features);
1509}

1587/*
1588 * \brief Serialize a two-dimensional variable-size feature vector with normalized throughputs
1589 * and task ids to a one-dimensional flatten byte array.
1590 *
1591 * For faster data copy between c++ and python, the c++ part returns features in a single
1592 * flatten array using a packed format. The python part then unpacks the flatten array.
1593 *
1594 * The packed format for n records is:
1595 * {
1596 * int n;
1597 * int sizes[n+2]; // The sizes for the following arrays
1598 *
1599 * float features_0[size[0]]; // The features for record 0
1600 * float features_1[size[1]]; // The features for record 1
1601 * ...
1602 * float features_i[size[i]]; // The features for record i
1603 * ... // until i == n - 1
1604 *
1605 * float throughputs[sizes[n]]; // The normalized throughputs for n records
1606 * int task_ids[size[n+1]]; // The task ids for n records
1607 *
1608 * }
1609 * To implement this format, we also store int as float, so we can store all numbers
1610 * into a single float array.
1611 */
1612TVMByteArray SerializeFeatures(std::vector<std::vector<float>>&& features,
1613 std::vector<float>&& normalized_throughputs,
1614 std::vector<int>&& task_ids, std::vector<char>* out_data) {
1615 size_t total_bytes = 0;
1616 std::vector<int> size_vector;
1617
1618 int n = features.size();
1619
1620 // serialize sizes
1621 size_t size_vector_size = 1 + n + 2;
1622 total_bytes += size_vector_size * sizeof(int);
1623
1624 size_vector.reserve(size_vector_size);
1625 size_vector.push_back(features.size());
1626 for (const auto& x : features) {
1627 size_vector.push_back(static_cast<int>(x.size()));
1628 total_bytes += sizeof(float) * x.size();
1629 }
1630 size_vector.push_back(static_cast<int>(normalized_throughputs.size()));
1631 total_bytes += sizeof(float) * normalized_throughputs.size();
1632 size_vector.push_back(static_cast<int>(task_ids.size()));
1633 total_bytes += sizeof(int) * task_ids.size();
1634
1635 ICHECK_EQ(size_vector.size(), size_vector_size);
1636
1637 // allocate memory
1638 out_data->reserve(total_bytes);
1639 char* ptr = out_data->data();
1640
1641 // serialize size_vector
1642 memmove(ptr, reinterpret_cast<char*>(size_vector.data()), size_vector.size() * sizeof(int));
1643 ptr += size_vector.size() * sizeof(int);
1644
1645 // serialize features
1646 for (auto& x : features) {
1647 memmove(ptr, x.data(), sizeof(float) * x.size());
1648 ptr += sizeof(float) * x.size();
1649 x.clear();
1650 }
1651
1652 // serialize normalized_throughputs
1653 memmove(ptr, reinterpret_cast<char*>(normalized_throughputs.data()),
1654 normalized_throughputs.size() * sizeof(int));
1655 ptr += normalized_throughputs.size() * sizeof(int);
1656
1657 // serialize task_ids
1658 memmove(ptr, reinterpret_cast<char*>(task_ids.data()), task_ids.size() * sizeof(int));
1659 ptr += task_ids.size() * sizeof(int);
1660
1661 ICHECK_EQ(ptr - out_data->data(), total_bytes);
1662
1663 return TVMByteArray{out_data->data(), total_bytes};
1664}
