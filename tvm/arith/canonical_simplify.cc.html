62inline PrimExpr ModImpl(PrimExpr a, PrimExpr b, DivMode mode) {
63 if (mode == kTruncDiv) {
64 return truncmod(a, b);
65 } else {
66 ICHECK_EQ(mode, kFloorDiv);
67 return floormod(a, b);
68 }
69}
70
71inline PrimExpr DivImpl(PrimExpr a, PrimExpr b, DivMode mode) {
72 if (mode == kTruncDiv) {
73 return truncdiv(a, b);
74 } else {
75 ICHECK_EQ(mode, kFloorDiv);
76 return floordiv(a, b);
77 }
78}

80/*!
81 * \brief check if value fits in dtype
82 * \param value The value to be analyzed
83 * \param dtype The target dtype
84 * \param analyzer The analyzer
85 * \return whether value fits in dtype
86 */
87bool CastIsSafe(DataType dtype, PrimExpr value, Analyzer* analyzer) {
88 if (!IsIndexType(dtype)) {
89 return false;
90 }
91 ConstIntBound bound = analyzer->const_int_bound(value);
92 int64_t ubound = Downcast<IntImm>(max_value(dtype))->value;
93 int64_t lbound = Downcast<IntImm>(min_value(dtype))->value;
94 if (value.dtype().bits() <= dtype.bits() || // upcast is safe
95 (bound->max_value <= ubound && bound->min_value >= lbound)) {
96 return true;
97 }
98 return false;
99}

101/*!
102 * \brief Internal "Split normal form" of expression.
103 *
104 * This is a special expression that represents
105 * a scaled value derived from a split of an index.
106 *
107 * result = ((index % upper_factor) / lower_factor) * scale
108 */
109class SplitExprNode : public CanonicalExprNode {
110 public:
111 /*! \brief The base index expression. */
112 PrimExpr index;
113 /*! \brief The division factor ratio. */
114 int64_t lower_factor{1};
115 /*!
116 * \brief The upper factor.
117 * invariance: (upper_factor == kPosInf || upper_factor % lower_factor == 0)
118 */
119 int64_t upper_factor{kPosInf};
120 /*! \brief scale to the expression. */
121 int64_t scale{1};
122 /*! \brief Division mode. */
123 DivMode div_mode{kTruncDiv};
124
125 /*! \brief verify that this is a valid entry. */
126 void Verify() const { ICHECK(upper_factor == kPosInf || upper_factor % lower_factor == 0); }
127
128 PrimExpr NormalizeWithScale(int64_t sscale) const {
129 PrimExpr res = this->index;
130 DataType dtype = this->dtype;
131 if (this->scale == 0) {
132 return make_const(dtype, 0);
133 }
134 if (this->upper_factor != SplitExprNode::kPosInf) {
135 res = ModImpl(res, make_const(dtype, this->upper_factor), div_mode);
136 }
137 if (this->lower_factor != 1) {
138 res = DivImpl(res, make_const(dtype, this->lower_factor), div_mode);
139 }
140 sscale *= this->scale;
141 if (sscale != 1) {
142 ICHECK(!dtype.is_uint() || sscale > 0);
143 res = res * make_const(dtype, sscale);
144 }
145 return res;
146 }
147
148 PrimExpr Normalize() const final { return NormalizeWithScale(1); }
149
150 void MulToSelf(int64_t scale) { this->scale *= scale; }
151
152 /*!
153 * \brief check if cast can be pushed to sub-expressions
154 * \param dtype The target datatype
155 * \param analyzer The analyzer
156 * \return whether the cast can be safely pushed to children
157 */
158 bool CanPushCastToChildren(DataType dtype, Analyzer* analyzer) const {
159 // cast(dtype, index % upper_factor / lower_factor * scale) ==
160 // cast(dtype, index) % upper_factor / lower_factor * scale
161 // iff it is an upcast (dtype.bits >= self.dtype.bits) or all of
162 // its intermediate results fit in the range of dtype
163 if (dtype.bits() >= this->dtype.bits()) {
164 return true; // upcast is safe
165 }
166 PrimExpr res = this->index;
167 if (this->scale == 0) {
168 return true;
169 }
170 if (!CastIsSafe(dtype, res, analyzer)) {
171 return false;
172 }
173 if (this->upper_factor != SplitExprNode::kPosInf) {
174 res = ModImpl(res, make_const(this->dtype, this->upper_factor), div_mode);
175 if (!CastIsSafe(dtype, res, analyzer)) {
176 return false;
177 }
178 }
179 if (this->lower_factor != 1) {
180 res = DivImpl(res, make_const(this->dtype, this->lower_factor), div_mode);
181 if (!CastIsSafe(dtype, res, analyzer)) {
182 return false;
183 }
184 }
185 if (this->scale != 1) {
186 ICHECK(!this->dtype.is_uint() || this->scale > 0);
187 res = res * make_const(this->dtype, this->scale);
188 if (!CastIsSafe(dtype, res, analyzer)) {
189 return false;
190 }
191 }
192 return true;
193 }
194
195 /*!
196 * \brief self = cast(dtype, self)
197 * \param dtype The target datatype
198 */
199 void PushCastToChildren(DataType dtype) {
200 this->index = cast(dtype, this->index);
201 this->dtype = dtype;
202 }
203
204 inline bool IndexEqual(const SplitExpr& other) const;
205 inline bool DivModeCompatibleTo(DivMode mode) const;
206
207 /*! \brief positive infty */
208 static const constexpr int64_t kPosInf = ConstIntBoundNode::kPosInf;
209 static constexpr const char* _type_key = "arith.SplitExpr";
210 TVM_DECLARE_FINAL_OBJECT_INFO(SplitExprNode, CanonicalExprNode);
211};

230/*!
231 * \brief Normal form that represents sum of expressions.
232 *
233 * result = sum(args) + base.
234 */
235class SumExprNode : public CanonicalExprNode {
236 public:
237 /*!
238 * \brief arguments to be summed up.
239 *
240 * args are divided into segments with the same index.
241 * within each segment, the SplitExpr is ordered in descending order of lower_factor.
242 */
243 std::vector<SplitExpr> args;
244 /*! \brief Base value in the summation. */
245 int64_t base{0};
246 /*! \brief The expression equals zero. */
247 bool IsZero() const { return base == 0 && args.size() == 0; }
248 /*!
249 * \brief Return the normal Expr that is equivalent to self.
250 * \return The normal expression.
251 */
252 PrimExpr Normalize() const final {
253 // quick path 1.
254 if (this->args.size() == 0) {
255 return make_const(this->dtype, this->base);
256 }
257 return Normalize_(this->dtype, SimplifySplitExprs(args), base);
258 }
259 /*!
260 * \brief Whether self is divisible by scale.
261 * \param scale The scale to be applied.
262 */
263 bool DivisibleBy(int64_t scale) {
264 if (base % scale != 0) return false;
265 for (size_t i = 0; i < this->args.size(); ++i) {
266 if (args[i]->scale % scale != 0) return false;
267 }
268 return true;
269 }
270 /*!
271 * \brief mul scale to self.
272 * \param scale The scale to be applied.
273 */
274 void MulToSelf(int64_t scale) {
275 this->base *= scale;
276 for (size_t i = 0; i < this->args.size(); ++i) {
277 args[i].CopyOnWrite()->scale *= scale;
278 }
279 }
280 /*!
281 * \brief divide by scale.
282 * \param scale The scale to be applied.
283 */
284 void DivideBy(int64_t scale) {
285 ICHECK_EQ(this->base % scale, 0);
286 this->base /= scale;
287 for (size_t i = 0; i < this->args.size(); ++i) {
288 ICHECK_EQ(args[i]->scale % scale, 0);
289 args[i].CopyOnWrite()->scale /= scale;
290 }
291 }
292 /*!
293 * \brief add constant value to self.
294 * \param value to be added.
295 */
296 void AddToSelf(int64_t value) { this->base += value; }
297 /*!
298 * \brief self += other * scale;
299 * \param other The expression to be added.
300 * \param scale The additional scale on value.
301 */
302 void AddToSelf(SplitExpr other, int64_t scale) {
303 if (other->scale == 0) return;
304 // We need to maintain the segment invariance:
305 // Same index are stored close to each other.
306 // sorted from big lower_factor to small one.
307 size_t start = 0;
308 for (; start < args.size(); ++start) {
309 if (args[start]->IndexEqual(other)) break;
310 }
311 for (size_t j = start; j < args.size(); ++j) {
312 if (!args[j]->IndexEqual(other) || other->lower_factor > args[j]->lower_factor) {
313 other.CopyOnWrite()->scale *= scale;
314 this->args.insert(this->args.begin() + j, other);
315 return;
316 }
317 if (other->lower_factor == args[j]->lower_factor &&
318 other->upper_factor == args[j]->upper_factor &&
319 other->DivModeCompatibleTo(args[j]->div_mode)) {
320 args[j].CopyOnWrite()->scale += other->scale * scale;
321 return;
322 }
323 }
324 // Insert other in the end.
325 other.CopyOnWrite()->scale *= scale;
326 this->args.emplace_back(std::move(other));
327 }

329 void AddToSelf(const SumExpr& other, int64_t scale);

331 /*!
332 * \brief check if cast can be pushed to sub-expressions
333 * \param dtype The target datatype
334 * \param analyzer The analyzer
335 * \return whether the cast can be safely pushed to children
336 */
337 bool CanPushCastToChildren(DataType dtype, Analyzer* analyzer) const {
338 // cast(dtype, arg_1 + arg_2 + ... arg_n) ==
339 // cast(dtype, arg_1) + ... + cast(dtype, arg_n)
340 // iff it is an upcast (dtype.bits >= self.dtype.bits) or all of
341 // its intermediate results fit in the range of dtype
342 if (dtype.bits() >= this->dtype.bits()) {
343 return true; // upcast is safe
344 }
345 PrimExpr res = make_const(dtype, 0);
346 for (size_t i = 0; i < args.size(); ++i) {
347 if (args[i]->scale > 0) {
348 res = res + args[i]->Normalize();
349 if (!CastIsSafe(dtype, res, analyzer)) {
350 return false;
351 }
352 }
353 }
354 if (base > 0) {
355 res = res + make_const(dtype, base);
356 if (!CastIsSafe(dtype, res, analyzer)) {
357 return false;
358 }
359 }
360 // negative scales follows using sub.
361 for (size_t i = 0; i < args.size(); ++i) {
362 if (args[i]->scale < 0) {
363 res = res - args[i]->NormalizeWithScale(-1);
364 if (!CastIsSafe(dtype, res, analyzer)) {
365 return false;
366 }
367 }
368 }
369 if (base < 0) {
370 res = res - make_const(dtype, -base);
371 if (!CastIsSafe(dtype, res, analyzer)) {
372 return false;
373 }
374 }
375 for (const auto& arg : args) {
376 if (!arg->CanPushCastToChildren(dtype, analyzer)) {
377 return false;
378 }
379 }
380 return true;
381 }
382
383 /*!
384 * \brief self = cast(dtype, self)
385 * \param dtype The target datatype
386 */
387 void PushCastToChildren(DataType dtype) {
388 for (auto& arg : args) {
389 arg.CopyOnWrite()->PushCastToChildren(dtype);
390 }
391 this->dtype = dtype;
392 }

394 static constexpr const char* _type_key = "arith.SumExpr";
395 TVM_DECLARE_FINAL_OBJECT_INFO(SumExprNode, CanonicalExprNode);

397 private:
398 /*!
399 * \brief Simplify the args by merging SplitExprs
400 * \param args The original list of arguments.
401 * \return simplified version.
402 */
403 static std::vector<SplitExpr> SimplifySplitExprs(std::vector<SplitExpr> args) {
404 // NOTE: This algorithm relies on the factor that args are divided into segments
405 // and each segment is sorted in descending order of lower_factor.
406 for (size_t i = 0; i < args.size(); ++i) {
407 if (args[i]->scale == 0) continue;
408 for (size_t j = i + 1; j < args.size(); ++j) {
409 SplitExpr& lhs = args[i];
410 SplitExpr& rhs = args[j];
411 if (!lhs->IndexEqual(rhs)) break;
412 if (lhs->upper_factor < rhs->lower_factor) break;
413 if (lhs->upper_factor == rhs->upper_factor && lhs->lower_factor == rhs->lower_factor &&
414 lhs->DivModeCompatibleTo(rhs->div_mode)) {
415 // folding same co-efficient.
416 rhs.CopyOnWrite()->scale += lhs->scale;
417 lhs.CopyOnWrite()->scale = 0;
418 } else if (lhs->lower_factor == rhs->upper_factor && rhs->scale != 0 &&
419 lhs->scale % rhs->scale == 0 &&
420 lhs->lower_factor == (lhs->scale / rhs->scale) * rhs->lower_factor &&
421 lhs->DivModeCompatibleTo(rhs->div_mode)) {
422 // Rules used in the proof:
423 //
424 // Rule 1: (x % (c * s)) / c = (x / c) % s
425 // Proof:
426 // x can always be decomposed into p * c * s + q * c + r
427 // where 0 <= q * c + r < c * s and 0 <= r < c.
428 // Then, lhs = ((p * c * s + q * c + r) % (c * s)) / c = (q * c + r) / c = q
429 // rhs = ((p * c * s + q * c + r) / c) % s = (p * s + q) % s = q
430 // Thus, lhs = rhs
431 //
432 // The above proof is for the floordiv.
433 // The same rule also holds for truncdiv(division rule in C).
434 // Because both sides only involve mul, div and mod,
435 // we can take abs of x, c and s, apply the floordiv proof,
436 // and finally add the sign back.
437 //
438 // Rule 2: (x / s) * s + x % s = x (true for both trunc and floor div)
439 //
440 // General merge condition and proof:
441 // - x = lhs->index % lhs->upper_factor
442 // - s = lhs->scale / rhs->scale
443 // - c = rhs->lower_factor
444 //
445 // (x / (c * s)) * s + (x % (c * s)) / c
446 // => ((x / c) / s) * s + ((x / c) % s)
447 // => (x / c)
448 //
449 // Examples:
450 //
451 // (z / 6) * 6 + ((z % 6) / 3) * 3
452 // => ((z / 6) * 2 + (z % 6) / 3) * 3
453 // => (z / 3) * 3
454 // note: x = z, c = 3, s = 2
455 //
456 // ((z % 12) / 6) * 6 + ((z % 6) / 3) * 3
457 // => (((z % 12) / 6) * 2 + ((z % 12) % 6) / 3) * 3
458 // => ((z % 12) / 3) * 3
459 // note: x = z % 12, c = 3, s = 2
460 // note also the invariance lhs->upper_factor % lhs->lower_factor == 0
461 //
462 SplitExprNode* merged = rhs.CopyOnWrite();
463 merged->upper_factor = lhs->upper_factor;
464 // reset args[i] to be zero.
465 lhs.CopyOnWrite()->scale = 0;
466 break;
467 }
468 }
469 }
470 // sort by the entry
471 // Here we simply sort by descending order of scales.
472 // For now, we do not compare by index because that comparison
473 // can be runtime dependent and create inderminism.
474 // we do not sort by index for now because it can be costly
475 // to deep compare Exprs, and address of Vars can be runtime dependent.
476 //
477 auto fcompare = [](const SplitExpr& lhs, const SplitExpr& rhs) {
478 // order by scale first
479 if (lhs->scale > rhs->scale) return true;
480 if (lhs->scale < rhs->scale) return false;
481 // then order by factor
482 if (lhs->lower_factor > rhs->lower_factor) return true;
483 if (lhs->lower_factor < rhs->lower_factor) return false;
484 // then order by upper factor
485 if (lhs->upper_factor > rhs->upper_factor) return true;
486 if (lhs->upper_factor < rhs->upper_factor) return false;
487 // then order by div mode
488 if (lhs->div_mode > rhs->div_mode) return true;
489 if (lhs->div_mode < rhs->div_mode) return false;
490 // tie.
491 // TODO(tvm-team) We might consider index as the last comparison point,
492 // after we make deep comparator more derministic.
493 // Specifically, we can consider comparing names of vars and break ties with address.
494 return false;
495 };
496 std::stable_sort(args.begin(), args.end(), fcompare);
497 return args;
498 }
499 static PrimExpr Normalize_(DataType dtype, const std::vector<SplitExpr>& args, int64_t base) {
500 // Positive scales first
501 PrimExpr res = make_const(dtype, 0);
502 for (size_t i = 0; i < args.size(); ++i) {
503 if (args[i]->scale > 0) {
504 res = res + args[i]->Normalize();
505 }
506 }
507 if (base > 0) {
508 res = res + make_const(dtype, base);
509 }
510 // negative scales follows using sub.
511 for (size_t i = 0; i < args.size(); ++i) {
512 if (args[i]->scale < 0) {
513 res = res - args[i]->NormalizeWithScale(-1);
514 }
515 }
516 if (base < 0) {
517 res = res - make_const(dtype, -base);
518 }
519 return res;
520 }
521};

529void SumExprNode::AddToSelf(const SumExpr& other, int64_t scale) {
530 // NOTE: it is rare to have a balanced long expression,
531 // linear scan is fine for our case.
532 for (size_t i = 0; i < other->args.size(); ++i) {
533 this->AddToSelf(other->args[i], scale);
534 }
535 this->AddToSelf(other->base * scale);
536}

606 private:
607 /*!
608 * \brief compute lhs / cval
609 * \param lhs The left operand.
610 * \param cval The constant value.
611 * \param div_mode The division mode.
612 * \return The result expression;
613 */
614 SplitExpr SplitDivConst(SplitExpr lhs, int64_t cval, DivMode div_mode);
615 /*!
616 * \brief compute lhs % cval
617 * \param lhs The left operand.
618 * \param cval The constant value.
619 * \param div_mode The division mode.
620 * \return The result expression;
621 */
622 SplitExpr SplitModConst(SplitExpr lhs, int64_t cval, DivMode div_mode);
623 /*!
624 * \brief Separate psum into divisible and non-divisible parts.
625 * \param psum The sum expression.
626 * \param coeff The co-efficient.
627 * \param out_divisible The result divisible component.
628 * \param out_non_divisible The non-divisible component.
629 */
630 void SeparateDivisibleParts(const SumExprNode* psum, int64_t coeff, SumExpr* out_divisible,
631 SumExpr* out_non_divisible);
632 /*!
633 * \brief Normalize expr to normal expr.
634 * \param expr The input expression.
635 * \return Normalized expr.
636 */
637 PrimExpr Normalize(PrimExpr expr) {
638 if (const auto* op = expr.as<CanonicalExprNode>()) {
639 return op->Normalize();
640 } else {
641 return expr;
642 }
643 }
644 /*!
645 * \brief Create a SplitExpr from expr.
646 * \param expr The input expr.
647 * \return The transformed SplitExpr.
648 */
649 SplitExpr ToSplitExpr(PrimExpr expr) {
650 if (const auto* op = expr.as<SplitExprNode>()) {
651 return GetRef<SplitExpr>(op);
652 }
653 if (const auto* op = expr.as<SumExprNode>()) {
654 if (op->base == 0 && op->args.size() == 1) return op->args[0];
655 }
656 if (const auto* op = expr.as<CanonicalExprNode>()) {
657 expr = op->Normalize();
658 }
659 ObjectPtr<SplitExprNode> n = make_object<SplitExprNode>();
660 n->dtype = expr.dtype();
661 n->index = std::move(expr);
662 n->div_mode = kTruncDiv;
663 return SplitExpr(n);
664 }
665 /*!
666 * \brief Convert expr to an equivalent SplitExpr
667 * that has the specified div_mode.
668 *
669 * This function will return the same expr if its
670 * div_mode already satisfies the need.
671 *
672 * \param expr The input expr.
673 * \param div_mode The new div_mode.
674 * \return The transformed SplitExpr.
675 */
676 SplitExpr ConvertDivMode(SplitExpr expr, DivMode div_mode) {
677 if (expr->div_mode == div_mode) return expr;
678 if (expr->DivModeCompatibleTo(div_mode)) {
679 expr.CopyOnWrite()->div_mode = div_mode;
680 return expr;
681 }
682 expr = ToSplitExpr(Normalize(expr));
683 ICHECK(expr->DivModeCompatibleTo(div_mode));
684 expr.CopyOnWrite()->div_mode = div_mode;
685 return expr;
686 }
687 /*!
688 * \brief Create a SumExpr from expr.
689 * \param expr The input expr.
690 * \return The transformed SumExpr.
691 */
692 SumExpr ToSumExpr(PrimExpr expr) {
693 if (const auto* op = expr.as<SumExprNode>()) {
694 return GetRef<SumExpr>(op);
695 }
696 ObjectPtr<SumExprNode> n = make_object<SumExprNode>();
697 n->dtype = expr.dtype();
698 if (const auto* op = expr.as<IntImmNode>()) {
699 n->base = op->value;
700 return SumExpr(n);
701 } else {
702 n->args.emplace_back(ToSplitExpr(expr));
703 return SumExpr(n);
704 }
705 }
706 // Simplify the combiner used in reduce.
707 PrimExpr SimplifyReduceCombiner(const ReduceNode* op);
708};

760PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const MulNode* op) {
761 if (!IsIndexType(op->dtype)) {
762 return Rewriter::VisitExpr_(op);
763 }
764 // normalize
765 PrimExpr a = this->CanonicalMutate(op->a);
766 PrimExpr b = this->CanonicalMutate(op->b);
767
768 // const folding
769 PrimExpr const_res = TryConstFold<Mul>(a, b);
770 if (const_res.defined()) return const_res;

772 // x * c
773 if (a.as<IntImmNode>()) {
774 std::swap(a, b);
775 }
776 if (const auto* bconst = b.as<IntImmNode>()) {
777 if (a.as<SumExprNode>()) {

781 } else {

783 ret.CopyOnWrite()->MulToSelf(bconst->value);

785 }
786 }
787
788 // normal path.
789 a = Normalize(a);
790 b = Normalize(b);
791 if (op->a.same_as(a) && op->b.same_as(b)) {
792 return GetRef<PrimExpr>(op);
793 } else {
794 return Mul(a, b);

798void CanonicalSimplifier::Impl::SeparateDivisibleParts(const SumExprNode* psum, int64_t coeff,
799 SumExpr* out_divisible,
800 SumExpr* out_non_divisible) {
801 auto divisible = make_object<SumExprNode>();
802 auto non_divisible = make_object<SumExprNode>();
803 divisible->dtype = psum->dtype;
804 non_divisible->dtype = psum->dtype;
805
806 if (psum->base % coeff == 0) {
807 divisible->base = psum->base;
808 } else {
809 non_divisible->base = psum->base;
810 }
811 for (const auto& e : psum->args) {
812 if (e->scale % coeff == 0) {
813 divisible->args.push_back(e);
814 } else {
815 non_divisible->args.push_back(e);
816 }
817 }
818 *out_divisible = SumExpr(divisible);
819 *out_non_divisible = SumExpr(non_divisible);
820}

822SplitExpr CanonicalSimplifier::Impl::SplitDivConst(SplitExpr lhs, int64_t cval, DivMode div_mode) {
823 ICHECK_GT(cval, 0);
824 lhs = ConvertDivMode(lhs, div_mode);
825
826 // the following rule works for both floordiv and truncdiv
827 if (lhs->scale % cval == 0) {
828 lhs.CopyOnWrite()->scale /= cval;
829 return lhs;
830 }
831
832 if (cval % lhs->scale == 0) {
833 int64_t scaled_cval = cval / lhs->scale;
834 if (lhs->upper_factor == SplitExprNode::kPosInf ||
835 lhs->upper_factor % (lhs->lower_factor * scaled_cval) == 0) {
836 // directly fold division.
837 lhs.CopyOnWrite()->scale = 1;
838 lhs.CopyOnWrite()->lower_factor *= scaled_cval;
839 lhs->Verify();
840 return lhs;
841 } else if (lhs->upper_factor <= (lhs->lower_factor * scaled_cval)) {
842 // (x % c1) / c2 => 0 when c2 >= c1
843 return ToSplitExpr(make_zero(lhs.dtype()));
844 } else {
845 // move the upper_factor modular into index.
846 lhs.CopyOnWrite()->index =
847 ModImpl(lhs->index, make_const(lhs.dtype(), lhs->upper_factor), div_mode);
848 lhs.CopyOnWrite()->upper_factor = SplitExprNode::kPosInf;
849 lhs.CopyOnWrite()->scale = 1;
850 lhs.CopyOnWrite()->lower_factor *= scaled_cval;
851 lhs->Verify();
852 return lhs;
853 }
854 }
855 // directly return the split with cval == 1
856 lhs = ToSplitExpr(Normalize(lhs));
857 ICHECK(lhs->DivModeCompatibleTo(div_mode));
858 ICHECK_EQ(lhs->scale, 1);
859 lhs.CopyOnWrite()->lower_factor *= cval;
860 lhs.CopyOnWrite()->div_mode = div_mode;
861 return lhs;
862}

864PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const DivNode* op) {
865 if (!IsIndexType(op->dtype)) {
866 return Rewriter::VisitExpr_(op);
867 }
868
869 PrimExpr a = this->CanonicalMutate(op->a);
870 PrimExpr b = this->CanonicalMutate(op->b);
871
872 // const folding
873 PrimExpr const_res = TryConstFold<Div>(a, b);
874 if (const_res.defined()) return const_res;
875 PVar<IntImm> c1;
876 // x / c1
877 if (c1.Match(b) && c1.Eval()->value > 0) {
878 int64_t cval = c1.Eval()->value;
879 if (cval == 1) return a;
880
881 if (const auto* psum = a.as<SumExprNode>()) {
882 SumExpr lhs, extra;
883 SeparateDivisibleParts(psum, cval, &lhs, &extra);
884 // can be divided by cval
885 if (extra->IsZero()) {
886 lhs.CopyOnWrite()->DivideBy(cval);
887 return std::move(lhs);
888 }
889 // both lhs and extra are non-negative
890 if (analyzer_->CanProveGreaterEqual(lhs->Normalize(), 0) &&
891 analyzer_->CanProveGreaterEqual(extra->Normalize(), 0)) {
892 lhs.CopyOnWrite()->DivideBy(cval);
893 PrimExpr temp = Normalize(extra);
894 if (const auto* pconst = temp.as<IntImmNode>()) {
895 lhs.CopyOnWrite()->AddToSelf(pconst->value / cval);
896 } else {
897 // if 0 <= extra < cval, it means the extra can be eliminated.
898 if (TryCompare(temp, cval) != kLT) {
899 lhs.CopyOnWrite()->AddToSelf(SplitDivConst(ToSplitExpr(temp), cval, kTruncDiv), 1);
900 }
901 }
902 return std::move(lhs);
903 }
904 } else {
905 // if a >= 0 && a < cval, then result == 0
906 auto cbound = analyzer_->const_int_bound(Normalize(a));
907 if (cbound->min_value >= 0 && cbound->max_value < cval) {
908 return make_zero(a.dtype());
909 }
910 }
911 return SplitDivConst(ToSplitExpr(std::move(a)), cval, kTruncDiv);
912 }
913 // normal path
914 a = Normalize(a);
915 b = Normalize(b);
916 if (op->a.same_as(a) && op->b.same_as(b)) {
917 return GetRef<PrimExpr>(op);
918 } else {
919 return Div(a, b);
920 }
921}

923PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const FloorDivNode* op) {
924 if (!IsIndexType(op->dtype)) {
925 return Rewriter::VisitExpr_(op);
926 }
927 PrimExpr a = this->CanonicalMutate(op->a);
928 PrimExpr b = this->CanonicalMutate(op->b);
929
930 // const folding
931 PrimExpr const_res = TryConstFold<FloorDiv>(a, b);
932 if (const_res.defined()) return const_res;
933 PVar<IntImm> c1;
934 // x / c1
935 if (c1.Match(b) && c1.Eval()->value > 0) {
936 int64_t cval = c1.Eval()->value;
937 if (cval == 1) return a;
938
939 if (const auto* psum = a.as<SumExprNode>()) {
940 SumExpr lhs, extra;
941 SeparateDivisibleParts(psum, cval, &lhs, &extra);
942 if (extra->IsZero()) {
943 lhs.CopyOnWrite()->DivideBy(cval);
944 return std::move(lhs);
945 }
946 // continue simplification.
947 lhs.CopyOnWrite()->DivideBy(cval);
948 PrimExpr temp = Normalize(extra);
949 if (const auto* pconst = temp.as<IntImmNode>()) {
950 lhs.CopyOnWrite()->AddToSelf(floordiv(pconst->value, cval));
951 } else {
952 // if 0 <= extra < cval, it means the extra can be eliminated.
953 if (!(TryCompare(temp, cval) == kLT && analyzer_->CanProveGreaterEqual(temp, 0))) {
954 lhs.CopyOnWrite()->AddToSelf(SplitDivConst(ToSplitExpr(temp), cval, kFloorDiv), 1);
955 }
956 }
957 return std::move(lhs);
958 } else {
959 // if a >= 0 && a < cval, then result == 0
960 auto cbound = analyzer_->const_int_bound(Normalize(a));
961 if (cbound->min_value >= 0 && cbound->max_value < cval) {
962 return make_zero(a.dtype());
963 }
964 }
965 return SplitDivConst(ToSplitExpr(std::move(a)), cval, kFloorDiv);
966 }
967 // normal path
968 a = Normalize(a);
969 b = Normalize(b);
970 if (op->a.same_as(a) && op->b.same_as(b)) {
971 return GetRef<PrimExpr>(op);
972 } else {
973 return FloorDiv(a, b);
974 }

977SplitExpr CanonicalSimplifier::Impl::SplitModConst(SplitExpr lhs, int64_t cval, DivMode div_mode) {
978 ICHECK_GT(cval, 0);
979 lhs = ConvertDivMode(lhs, div_mode);
980
981 if (lhs->scale % cval == 0) {
982 lhs.CopyOnWrite()->scale = 0;
983 return lhs;
984 }
985 if (cval % lhs->scale == 0) {
986 // The rationale:
987 // (index % upper) / lower * scale % cval, given cval = scaled_cval * scale
988 // by the rule (x * c1) % (c2 * c1) => (x % c2) * c1,
989 // = (index % upper) / lower % scaled_cval * scale
990 // by the rule (x / c1) % c2 => (x % (c1 * c2)) / c1,
991 // = (index % upper) % (new_upper_factor) / lower * scale
992 int64_t scaled_cval = cval / lhs->scale;
993 int64_t new_upper_factor = lhs->lower_factor * scaled_cval;
994 // try to see if we can reduce the existing upper modular.
995 if (lhs->upper_factor == SplitExprNode::kPosInf || lhs->upper_factor % new_upper_factor == 0) {
996 // we gained a new upper factor that is smaller
997 // than the original one
998 // Perhaps there are more chances in simplifying the index
999 // Do a recursive call to simplify the mod with the new factor.
1000 if (new_upper_factor < lhs->upper_factor && lhs->upper_factor != SplitExprNode::kPosInf) {
1001 auto updated = ToSplitExpr(this->VisitExpr(
1002 ModImpl(lhs->index, make_const(lhs.dtype(), new_upper_factor), div_mode)));
1003 // re-apply the lower_factor
1004 if (lhs->lower_factor != 1) {
1005 auto ret = SplitDivConst(updated, lhs->lower_factor, div_mode);
1006 ret.CopyOnWrite()->MulToSelf(lhs->scale);
1007 return ret;
1008 } else {
1009 updated.CopyOnWrite()->MulToSelf(lhs->scale);
1010 return updated;
1011 }
1012 } else {
1013 lhs.CopyOnWrite()->upper_factor = new_upper_factor;
1014 return lhs;
1015 }
1016 } else if (new_upper_factor % lhs->upper_factor == 0) {
1017 // (x % 2) % 4 => x % 2
1018 return lhs;
1019 }
1020 }
1021 // Normalize the value.
1022 lhs = ToSplitExpr(Normalize(lhs));
1023 ICHECK(lhs->DivModeCompatibleTo(div_mode));
1024 ICHECK_EQ(lhs->scale, 1);
1025 ICHECK_EQ(lhs->lower_factor, 1);
1026 lhs.CopyOnWrite()->div_mode = div_mode;
1027 lhs.CopyOnWrite()->upper_factor = cval;
1028 return lhs;
1029}

1031PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const ModNode* op) {
1032 if (!IsIndexType(op->dtype)) {
1033 return Rewriter::VisitExpr_(op);
1034 }
1035 // normalize
1036 PrimExpr a = this->CanonicalMutate(op->a);
1037 PrimExpr b = this->CanonicalMutate(op->b);
1038
1039 // const folding
1040 PrimExpr const_res = TryConstFold<Mod>(a, b);
1041 if (const_res.defined()) return const_res;
1042
1043 PVar<IntImm> c1;
1044 // x % c1
1045 if (c1.Match(b) && c1.Eval()->value > 0) {
1046 int64_t cval = c1.Eval()->value;
1047 if (const auto* psum = a.as<SumExprNode>()) {
1048 SumExpr lhs, extra;
1049 SeparateDivisibleParts(psum, cval, &lhs, &extra);
1050 if (extra->IsZero()) {
1051 return make_zero(a.dtype());
1052 }
1053 // both lhs and extra are non-negative
1054 if (analyzer_->CanProveGreaterEqual(lhs->Normalize(), 0) &&
1055 analyzer_->CanProveGreaterEqual(extra->Normalize(), 0)) {
1056 PrimExpr temp = Normalize(extra);
1057 if (temp.as<IntImmNode>()) {
1058 return truncmod(temp, c1.Eval());
1059 } else {
1060 // If temp < cval && temp >=0 then can remove the mod.
1061 if (TryCompare(temp, cval) == kLT) {
1062 return temp;
1063 } else {
1064 // contonue to use logic below.
1065 a = extra;
1066 psum = a.as<SumExprNode>();
1067 ICHECK(psum != nullptr);
1068 }
1069 }
1070 }
1071 // Simplify the offset constant if necessary.
1072 // (x - 5) % 3 => (x - 2) % 3 if x - 5 >= 0
1073 auto cbound = analyzer_->const_int_bound(Normalize(a));
1074 int64_t new_base = psum->base % cval;
1075 if (cbound->min_value >= 0 && cbound->min_value - psum->base + new_base >= 0) {
1076 SumExpr sum_expr = Downcast<SumExpr>(a);
1077 sum_expr.CopyOnWrite()->base = new_base;
1078 return SplitModConst(ToSplitExpr(std::move(sum_expr)), cval, kTruncDiv);
1079 }
1080 } else {
1081 // if a >= 0 && a < cval, then result == 0
1082 auto cbound = analyzer_->const_int_bound(Normalize(a));
1083 if (cbound->min_value >= 0 && cbound->max_value < cval) {
1084 return a;
1085 }
1086 }
1087 return SplitModConst(ToSplitExpr(std::move(a)), cval, kTruncDiv);
1088 }
1089 // normal path
1090 a = Normalize(a);
1091 b = Normalize(b);
1092 if (op->a.same_as(a) && op->b.same_as(b)) {
1093 return GetRef<PrimExpr>(op);
1094 } else {
1095 return Mod(a, b);
1096 }
1097}

1157// Simplify reduce expression.
1158PrimExpr CanonicalSimplifier::Impl::SimplifyReduceCombiner(const ReduceNode* op) {
1159 // First simplify the results
1160 Array<PrimExpr> simplified_result;
1161 for (const auto& res : op->combiner->result) {
1162 PrimExpr new_res = this->VisitExpr(res);
1163 simplified_result.push_back(new_res);
1164 }
1165
1166 // Which components to keep
1167 std::vector<int> used(op->combiner->result.size(), false);
1168
1169 // This function recursively marks the used components starting from
1170 // the index idx
1171 std::function<void(int)> mark_used;
1172 mark_used = [&used, &simplified_result, op, &mark_used](size_t idx) {
1173 // if the idx-th component was marked as used before, do nothing
1174 if (used[idx]) return;
1175 used[idx] = true;
1176
1177 // check if the idx-th result expr uses some lhs or rhs variables
1178 // and recursively mark the corresponding components
1179 for (size_t i = 0; i < simplified_result.size(); ++i)
1180 if (!used[i]) {
1181 if (UsesVar(simplified_result[idx],
1182 [v = op->combiner->lhs[i].get()](const VarNode* var) { return var == v; }) ||
1183 UsesVar(simplified_result[idx],
1184 [v = op->combiner->rhs[i].get()](const VarNode* var) { return var == v; }))
1185 mark_used(i);
1186 }
1187 };
1188
1189 // mark all used components starting from the value_index
1190 mark_used(op->value_index);
1191
1192 // components which have side effects should also be preserved
1193 for (size_t i = 0; i < used.size(); ++i) {
1194 if (SideEffect(op->source[i]) > CallEffectKind::kReadState ||
1195 SideEffect(op->combiner->identity_element[i]) > CallEffectKind::kReadState ||
1196 SideEffect(op->combiner->result[i]) > CallEffectKind::kReadState ||
1197 (!op->init.empty() && SideEffect(op->init[i]) > CallEffectKind::kReadState)) {
1198 mark_used(i);
1199 }
1200 }
1201
1202 int new_value_index = op->value_index;
1203 Array<PrimExpr> new_result;
1204 Array<PrimExpr> new_identity;
1205 Array<Var> new_lhs;
1206 Array<Var> new_rhs;
1207 Array<PrimExpr> new_source;
1208 Array<PrimExpr> new_init;
1209
1210 // new stuff is old stuff which is used
1211 for (size_t i = 0; i < used.size(); ++i) {
1212 if (used[i]) {
1213 // We simplify the result and identity, but not the source
1214 new_result.push_back(simplified_result[i]);
1215 new_identity.push_back(this->VisitExpr(op->combiner->identity_element[i]));
1216 new_lhs.push_back(op->combiner->lhs[i]);
1217 new_rhs.push_back(op->combiner->rhs[i]);
1218 new_source.push_back(op->source[i]);
1219 if (!op->init.empty()) new_init.push_back(op->init[i]);
1220 } else if (static_cast<int>(i) < op->value_index) {
1221 // value_index should also be adjusted
1222 new_value_index--;
1223 }
1224 }
1225
1226 CommReducer new_combiner = CommReducer(new_lhs, new_rhs, new_result, new_identity);
1227 return Reduce(new_combiner, new_source, op->axis, op->condition, new_value_index, new_init);
1228}
1229
1230PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const ReduceNode* op) {
1231 // Recursively call simplification when necessary.
1232 PrimExpr ret = RewriteSimplifier::Impl::VisitExpr_(op);
1233 op = ret.as<ReduceNode>();
1234 // already been simplified by const reduction axis removal
1235 if (op == nullptr) return ret;
1236 if (op->axis.empty()) {
1237 if (!op->init.empty()) {
1238 return this->VisitExpr(Select(op->condition,
1239 (*op->combiner.get())(op->init, op->source)[op->value_index],
1240 op->init[op->value_index]));
1241 }
1242 // Note that here we assume that the identity element is indeed identity. Without this
1243 // assumption we would have to perform a single iteration of the loop, i.e. use
1244 // `(*op->combiner.get())(op->combineop->identity_element, op->source)[op->value_index]`
1245 // instead of `op->source[op->value_index]`. The former may be more difficult to simplify.
1246 return this->VisitExpr(Select(op->condition, op->source[op->value_index],
1247 op->combiner->identity_element[op->value_index]));
1248 }
1249 // combiner simplification.
1250 ret = SimplifyReduceCombiner(op);
1251 return ret;
1252}

1254PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const CastNode* op) {
1255 if (!IsIndexType(op->dtype)) {
1256 return Rewriter::VisitExpr_(op);
1257 }
1258 // normalize
1259 PrimExpr value = this->CanonicalMutate(op->value);
1260 // PushCastToChildren
1261 if (value.as<SumExprNode>()) {
1262 SumExpr se = Downcast<SumExpr>(value);
1263 if (se->CanPushCastToChildren(op->dtype, analyzer_)) {
1264 se.CopyOnWrite()->PushCastToChildren(op->dtype);
1265 return std::move(se);
1266 }
1267 }
1268 if (value.as<SplitExprNode>()) {
1269 SplitExpr se = Downcast<SplitExpr>(value);
1270 if (se->CanPushCastToChildren(op->dtype, analyzer_)) {
1271 se.CopyOnWrite()->PushCastToChildren(op->dtype);
1272 return std::move(se);
1273 }
1274 }
1275 return Rewriter::VisitExpr_(op);
1276}
