38/*!
39 * \brief Try to run binary compute with constant folding.
40 *
41 * \param a The left operand.
42 * \param b The right operand.
43 * \tparam Op The operator type.
44 *
45 * \note a and b Must already matched data types with each other.
46 * \return nullptr if constant fold fails, otherwise return folded result.
47 */
48template <typename Op>
49inline PrimExpr TryConstFold(PrimExpr a, PrimExpr b);

51/*!
52 * \brief Try to run unary compute with constant folding.
53 *
54 * \param a The left operand.
55 * \tparam Op The operator type.
56 *
57 * \note a and b Must already matched data types with each other.
58 * \return nullptr if constant fold fails, otherwise return folded result.
59 */
60template <typename Op>
61inline PrimExpr TryConstFold(PrimExpr a);

63/*!
64 * \brief Check whether type is used to represent index.
65 *
66 * Index types are frequently used in shape computation
67 * and need to be aggressively constant-folded.
68 *
69 * \param type The type to represent index.
70 * \return the checked result.
71 */
72inline bool IsIndexType(const DataType& type) {
73 return type.is_int() && type.lanes() == 1 && (type.bits() == 32 || type.bits() == 64);
74}
75
76#define TVM_ARITH_CONST_PROPAGATION(BODY) \
77 using tir::FloatImmNode; \
78 const IntImmNode* pa = a.as<IntImmNode>(); \
79 const IntImmNode* pb = b.as<IntImmNode>(); \
80 const FloatImmNode* fa = a.as<FloatImmNode>(); \
81 const FloatImmNode* fb = b.as<FloatImmNode>(); \
82 BODY;
83
84#define TVM_INDEX_CONST_PROPAGATION(BODY) \
85 const IntImmNode* pa = a.as<IntImmNode>(); \
86 const IntImmNode* pb = b.as<IntImmNode>(); \
87 const DataType& ta = a.dtype(); \
88 const DataType& tb = b.dtype(); \
89 if (arith::IsIndexType(ta) && arith::IsIndexType(tb)) { \
90 BODY; \
91 }

93// specialization of constant folders.
94template <>
95inline PrimExpr TryConstFold<tir::Add>(PrimExpr a, PrimExpr b) {
96 TVM_ARITH_CONST_PROPAGATION({
97 const DataType& rtype = a.dtype();
98 if (pa && pb) return IntImm(rtype, pa->value + pb->value);
99 if (pa && pa->value == 0) return b;
100 if (pb && pb->value == 0) return a;
101 if (fa && fb) return FloatImm(rtype, fa->value + fb->value);
102 if (fa && fa->value == 0) return b;
103 if (fb && fb->value == 0) return a;
104 });
105 return PrimExpr();
106}

108template <>
109inline PrimExpr TryConstFold<tir::Sub>(PrimExpr a, PrimExpr b) {
110 TVM_ARITH_CONST_PROPAGATION({
111 ICHECK(!((pa && pa->dtype.is_uint() && pa->value == 0U) &&
112 (pb && pb->dtype.is_uint() && pb->value > 0U)))
113 << "Checked failed. Minuend 's value is 0U and it's dtype is uint "
114 << "while Subtrahend's dtype is uint; which will cause a negative uint";
115 const DataType& rtype = a.dtype();
116 if (pa && pb) return IntImm(rtype, pa->value - pb->value);
117 if (pb && pb->value == 0) return a;
118 if (fa && fb) return FloatImm(rtype, fa->value - fb->value);
119 if (fb && fb->value == 0) return a;
120 });
121 return PrimExpr();
122}

124template <>
125inline PrimExpr TryConstFold<tir::Mul>(PrimExpr a, PrimExpr b) {
126 TVM_ARITH_CONST_PROPAGATION({
127 const DataType& rtype = a.dtype();
128 if (pa && pb) return IntImm(rtype, pa->value * pb->value);
129 if (pa) {
130 if (pa->value == 1) return b;
131 if (pa->value == 0) return a;
132 }
133 if (pb) {
134 if (pb->value == 1) return a;
135 if (pb->value == 0) return b;
136 }
137 if (fa && fb) return FloatImm(rtype, fa->value * fb->value);
138 if (fa) {
139 if (fa->value == 1) return b;
140 if (fa->value == 0) return a;
141 }
142 if (fb) {
143 if (fb->value == 1) return a;
144 if (fb->value == 0) return b;
145 }
146 });
147 return PrimExpr();
148}

150template <>
151inline PrimExpr TryConstFold<tir::Div>(PrimExpr a, PrimExpr b) {
152 TVM_ARITH_CONST_PROPAGATION({
153 const DataType& rtype = a.dtype();
154 if (pa && pb) {
155 // due to division and mod can have different modes
156 // NOTE: this will assumes truc div.
157 ICHECK_NE(pb->value, 0) << "Divide by zero";
158 return IntImm(rtype, pa->value / pb->value);
159 }
160 if (pa) {
161 if (pa->value == 0) return a;
162 }
163 if (pb) {
164 if (pb->value == 1) return a;
165 ICHECK_NE(pb->value, 0) << "Divide by zero";
166 }
167 if (fa && fb && fb->value != 0) {
168 return FloatImm(rtype, fa->value / fb->value);
169 }
170 if (fa && fa->value == 0) return a;
171 if (fb) {
172 if (fb->value == 1) return a;
173 ICHECK_NE(fb->value, 0) << "Divide by zero";
174 }
175 });
176 return PrimExpr();
177}

179template <>
180inline PrimExpr TryConstFold<tir::Mod>(PrimExpr a, PrimExpr b) {
181 TVM_INDEX_CONST_PROPAGATION({
182 const DataType& rtype = a.dtype();
183 if (pa && pb) {
184 ICHECK_NE(pb->value, 0) << "Divide by zero";
185 return IntImm(rtype, pa->value % pb->value);
186 }
187 if (pa) {
188 if (pa->value == 0) return a;
189 }
190 if (pb) {
191 if (pb->value == 1) return tir::make_zero(rtype);
192 ICHECK_NE(pb->value, 0) << "Divide by zero";
193 }
194 });
195 return PrimExpr();
196}

198template <>
199inline PrimExpr TryConstFold<tir::FloorDiv>(PrimExpr a, PrimExpr b) {
200 TVM_ARITH_CONST_PROPAGATION({
201 const DataType& rtype = a.dtype();
202 if (pa && pb) {
203 ICHECK_NE(pb->value, 0) << "Divide by zero";
204 return IntImm(rtype, arith::floordiv(pa->value, pb->value));
205 }
206 if (pa) {
207 if (pa->value == 0) return a;
208 }
209 if (pb) {
210 if (pb->value == 1) return a;
211 ICHECK_NE(pb->value, 0) << "Divide by zero";
212 }
213 if (fa && fb && fb->value != 0) {
214 return FloatImm(rtype, std::floor(fa->value / fb->value));
215 }
216 if (fa && fa->value == 0) return a;
217 if (fb) {
218 if (fb->value == 1) return a;
219 ICHECK_NE(fb->value, 0) << "Divide by zero";
220 }
221 });
222 return PrimExpr();
223}

244template <>
245inline PrimExpr TryConstFold<tir::Min>(PrimExpr a, PrimExpr b) {
246 TVM_ARITH_CONST_PROPAGATION({
247 const DataType& rtype = a.dtype();
248 if (pa && pb) return IntImm(rtype, std::min(pa->value, pb->value));
249 if (fa && fb) return FloatImm(rtype, std::min(fa->value, fb->value));
250 });
251 if (a.same_as(b)) return a;
252 return PrimExpr();

320template <>
321inline PrimExpr TryConstFold<tir::And>(PrimExpr a, PrimExpr b) {
322 const IntImmNode* pa = a.as<IntImmNode>();
323 const IntImmNode* pb = b.as<IntImmNode>();
324 if (pa && pa->value) return b;
325 if (pa && !pa->value) return a;
326 if (pb && pb->value) return a;
327 if (pb && !pb->value) return b;
328 return PrimExpr();
329}
330
331template <>
332inline PrimExpr TryConstFold<tir::Or>(PrimExpr a, PrimExpr b) {
333 const IntImmNode* pa = a.as<IntImmNode>();
334 const IntImmNode* pb = b.as<IntImmNode>();
335 if (pa && pa->value) return a;
336 if (pa && !pa->value) return b;
337 if (pb && pb->value) return b;
338 if (pb && !pb->value) return a;
339 return PrimExpr();
340}

342template <>
343inline PrimExpr TryConstFold<tir::Not>(PrimExpr a) {
344 const IntImmNode* pa = a.as<IntImmNode>();
345 if (pa) {
346 return IntImm(DataType::UInt(1), !(pa->value));
347 }
348 return PrimExpr();

351/*! \brief Helper namespace for symbolic value limits */
352struct SymbolicLimits {
353 /*! \brief positive infinity */
354 static PrimExpr pos_inf_;
355 /*! \brief negative infinity */
356 static PrimExpr neg_inf_;
357};
358
359/*!
360 * \brief Opaque expression representing positive infinity.
361 *
362 * It can can only be used as parameter of by min/max
363 * for integer analysis and cannot be used in normal expressions.
364 *
365 * \return positive infinity.
366 */
367inline PrimExpr pos_inf() { return SymbolicLimits::pos_inf_; }
368
369/*!
370 * \brief Check if value is positive infinity.
371 * \param value The value to be checked.
372 *
373 * \return The check result.
374 */
375inline bool is_pos_inf(const PrimExpr& value) { return value.same_as(SymbolicLimits::pos_inf_); }
376
377/*!
378 * \brief Opaque expression representing negative infinity.
379 *
380 * It can can only be used as parameter of by min/max
381 * for integer analysis and cannot be used in normal expressions.
382 *
383 * \return negative infinity.
384 */
385inline PrimExpr neg_inf() { return SymbolicLimits::neg_inf_; }
386
387/*!
388 * \brief Check if value is negative infinity.
389 * \param value The value to be checked.
390 *
391 * \return The check result.
392 */
393inline bool is_neg_inf(const PrimExpr& value) { return value.same_as(SymbolicLimits::neg_inf_); }
