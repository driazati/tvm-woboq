239/*!
240 * \brief Meta node collector
241 * If we decide to put some node into meta, then all the sub-nodes inside
242 * it need to be put in meta as well, since when parsing we need to know
243 * whether two refs are the same
244 */
245class MetaCollector : public StmtExprVisitor {
246 public:
247 explicit MetaCollector(TextMetaDataContext* meta) : meta_(meta) {}
248
249 void Collect(const ObjectRef& n) {
250 // these nodes can be print directly(StringLiteral or use identifier to identify)
251 if (!n.defined() || n.as<StringImmNode>() || n.as<StringObj>() || n.as<SizeVarNode>() ||
252 n.as<VarNode>() || n.as<BufferNode>() || n.as<IterVarNode>()) {
253 return;
254 }
255 if (n->IsInstance<StmtNode>()) {
256 VisitStmt(Downcast<Stmt>(n));
257 } else if (n->IsInstance<PrimExprNode>()) {
258 VisitExpr(Downcast<PrimExpr>(n));
259 }
260 }
261
262 void VisitStmt(const Stmt& n) override {
263 meta_->GetMetaNode(n);
264 StmtVisitor::VisitStmt(n);
265 }
266
267 void VisitExpr(const PrimExpr& n) override {
268 meta_->GetMetaNode(n);
269 ExprVisitor::VisitExpr(n);
270 }
271
272 private:
273 TextMetaDataContext* meta_;
274};

418class TextPrinter {
419 public:
420 explicit TextPrinter(bool show_meta_data,
421 const runtime::TypedPackedFunc<std::string(ObjectRef)>& annotate,
422 bool show_warning = true)
423 : show_meta_data_(show_meta_data),
424 show_warning_(show_warning),
425 annotate_(annotate),
426 relay_text_printer_(show_meta_data, &meta_, annotate),
427 tir_text_printer_(show_meta_data, &meta_) {}
428
429 /*! \brief whether show meta data */
430 bool show_meta_data_;
431
432 /*! \brief whether show the meta data warning message */
433 bool show_warning_;
434
435 /*! \brief meta data context */
436 TextMetaDataContext meta_;
437 /*! \brief additional comment function */
438 runtime::TypedPackedFunc<std::string(ObjectRef)> annotate_;
439 /*! \brief Relay Text Printer */
440 relay::RelayTextPrinter relay_text_printer_;
441 /*! \brief TIR Text Printer */
442 tir::TIRTextPrinter tir_text_printer_;
443
444 bool GetVarName(::tvm::tir::Var v, std::string* s) { return tir_text_printer_.GetVarName(v, s); }
445
446 Doc PrintFinal(const ObjectRef& node) {
447 Doc doc;
448 if (node.defined() && node->IsInstance<IRModuleNode>()) {
449 doc << PrintMod(Downcast<IRModule>(node));
450 } else if (node.defined() &&
451 (node->IsInstance<tir::PrimFuncNode>() || node->IsInstance<PrimExprNode>() ||
452 node->IsInstance<tir::StmtNode>())) {
453 doc << tir_text_printer_.Print(node);
454 } else {
455 doc << relay_text_printer_.PrintFinal(node);
456 }
457 if (!meta_.empty()) {
458 doc << Doc::NewLine();
459 if (show_meta_data_) {
460 doc << "#[metadata]" << Doc::NewLine() << meta_.GetMetaSection();
461 } else if (show_warning_) {
462 doc << "/* For debugging purposes the metadata section has been omitted." << Doc::NewLine()
463 << " * If you would like to see the full metadata section you can set the "
464 << Doc::NewLine() << " * option to `True` when invoking `astext`. " << Doc::NewLine()
465 << " */";
466 }
467 }
468 return doc;
469 }
