20#include <tvm/relay/transform.h>
21#include <tvm/target/target.h>
22
23#include "./codegen_c.h"
24
25namespace tvm {
26namespace relay {
27namespace contrib {
28
29/*!
30 * \brief This demonstration external codegen target emits C/C++ for compilation by the native c
31 * compiler on CPU.
32 * - Patterns: None, functions must be explicitly marked as "Primitive" and "Compiler=ccompiler".
33 * - Custom compiler: relay/backend/contrib/codegen_c/codegen.cc
34 */
35TVM_REGISTER_TARGET_KIND("ccompiler", kDLCPU)
36 .set_attr<Bool>(tvm::attr::kIsExternalCodegen, Bool(true))
37 .set_attr<FTVMRelayToTIR>(tvm::attr::kRelayToTIR, CCompilerPass())
38 // Value is prepended to every output CModule.
39 .add_attr_option<String>("header", String(""));
40
41} // namespace contrib
42} // namespace relay
43} // namespace tvm
