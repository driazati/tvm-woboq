34Array<meta_schedule::ExtractedTask> ExtractTask(
35 IRModule mod, Target target, Map<String, runtime::NDArray> params,
36 runtime::TypedPackedFunc<Optional<tir::PrimFunc>(const Array<te::Tensor>&)> filter_func) {
37 using meta_schedule::ExtractedTask;
38 if (filter_func == nullptr) {
39 filter_func = tvm::meta_schedule::DefaultTaskFilter;
40 }
41 backend::BindParamsInModule(mod, params);
42 // is_vm=true for backward compatibility
43 Array<Pass> pass_seqs = relay::backend::GetPassPrefix(/*is_homogenous=*/true, /*is_vm=*/true);
44 pass_seqs.push_back(transform::FuseOps());
45 mod = transform::Sequential(pass_seqs)(std::move(mod));
46
47 std::vector<ExtractedTask> tasks;
48 std::unordered_map<tec::CCacheKey, ExtractedTask> cache;
49 PostOrderVisit(mod->Lookup("main"), [&target, &tasks, &cache, &filter_func](const Expr& exp) {
50 if (exp->IsInstance<FunctionNode>()) {
51 Function relay_func = Downcast<Function>(exp);
52 if (!relay_func->HasNonzeroAttr(attr::kPrimitive)) {
53 return;
54 }
55 tec::CCacheKey cache_key(relay_func, target);
56 auto it = cache.find(cache_key);
57 if (it != cache.end()) {
58 it->second->weight += 1;
59 return;
60 }
61 Array<te::Tensor> inputs_outputs{nullptr};
62 std::string fused_name;
63 std::tie(inputs_outputs, fused_name) =
64 tec::LowerTECompute(relay_func, target, /*return_inputs=*/true);
65 if (Optional<tir::PrimFunc> prim_func = filter_func(inputs_outputs)) {
66 GlobalVar prim_fn_var(fused_name);
67 IRModule relay_mod({{prim_fn_var, relay_func}});
68 IRModule tir_mod({{prim_fn_var, prim_func.value()}});
69 ExtractedTask extracted_task(fused_name, relay_mod, target, {tir_mod}, 1);
70 tasks.push_back(extracted_task);
71 cache.emplace(cache_key, extracted_task);
72 }
73 }
74 });
75 // Tasks are extracted via post order visit, return the reversed list.
76 std::reverse(tasks.begin(), tasks.end());
77 std::unordered_map<std::string, int> name_map;
78 for (ExtractedTask task : tasks) {
79 task->task_name = tec::GetUniqueName(task->task_name, &name_map);
80 }
81 return tasks;
82}
