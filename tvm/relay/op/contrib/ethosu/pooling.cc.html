35bool EthosuPoolingRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
36 const TypeReporter& reporter) {
37 int ifm_index = 0;
38 int result_index = 2;
39 ICHECK_EQ(types.size(), result_index + 1);
40
41 const auto* ifm = types[ifm_index].as<TensorTypeNode>();
42 if (ifm == nullptr) return false;
43
44 const auto* param = attrs.as<EthosuPoolingAttrs>();
45 ICHECK(param != nullptr) << "EthosuPoolingAttrs cannot be nullptr.";
46
47 const String operator_name = "ethosu_pooling";
48
49 if (param->pooling_type != "AVG" && param->pooling_type != "MAX") {
50 reporter->GetDiagCtx().EmitFatal(Diagnostic::Error(reporter->GetSpan())
51 << "Invalid operator: expected " << operator_name
52 << " type 'AVG' or 'MAX' but was " << param->pooling_type);
53 return false;
54 }
55
56 CheckDataType(reporter, ifm->dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name, "ifm",
57 param->pooling_type);
58
59 CheckUpscaleMethod(reporter, param->upscale, {"NONE", "ZEROS", "NEAREST"}, operator_name);
60
61 Array<IndexExpr> ifm_shape = ifm->shape;
62 if (param->upscale != "NONE") {
63 ifm_shape = EthosuInferUpscaledInput(ifm_shape, param->ifm_layout);
64 }
65
66 // Assign ofm shape
67 auto ofm_shape = EthosuInferKernelOutput(
68 ifm_shape, param->ifm_layout, param->ofm_layout, param->pool_shape, param->ofm_channels,
69 Array<IndexExpr>({1, 1}), param->strides, param->padding);
70 reporter->Assign(types[result_index], TensorType(ofm_shape, ifm->dtype));
71 return true;
72}
73
74Expr MakeEthosuPooling(Expr ifm, Expr lut, String pooling_type, double ifm_scale,
75 int ifm_zero_point, double ofm_scale, int ofm_zero_point,
76 Array<IndexExpr> pool_shape, IndexExpr ofm_channels,
77 Array<IndexExpr> strides, Array<IndexExpr> padding, String activation,
78 int clip_min, int clip_max, String rounding_mode, String upscale,
79 String ifm_layout, String ofm_layout) {
80 auto attrs = make_object<EthosuPoolingAttrs>();
81 attrs->pooling_type = std::move(pooling_type);
82 attrs->ifm_scale = ifm_scale;
83 attrs->ifm_zero_point = ifm_zero_point;
84 attrs->ofm_scale = ofm_scale;
85 attrs->ofm_zero_point = ofm_zero_point;
86 attrs->pool_shape = std::move(pool_shape);
87 attrs->ofm_channels = std::move(ofm_channels);
88 attrs->strides = std::move(strides);
89 attrs->padding = std::move(padding);
90 attrs->activation = std::move(activation);
91 attrs->clip_min = clip_min;
92 attrs->clip_max = clip_max;
93 attrs->rounding_mode = std::move(rounding_mode);
94 attrs->upscale = std::move(upscale);
95 attrs->ifm_layout = std::move(ifm_layout);
96 attrs->ofm_layout = std::move(ofm_layout);
97 static const Op& op = Op::Get("contrib.ethosu.pooling");
98 return Call(op, {ifm, lut}, Attrs(attrs), {});
99}
100
101TVM_REGISTER_GLOBAL("relay.op._make.ethosu_pooling").set_body_typed(MakeEthosuPooling);
102
103RELAY_REGISTER_OP("contrib.ethosu.pooling")
104 .describe(R"code(Arm(R) Ethos(TM)-U NPU 2D quantized pooling operator.
105
106This Relay operator corresponds to the hardware-implemented quantized
107pooling operation found on Ethos(TM)-U NPU. It accepts either NHWC
108or NHCWB16 format for the input data (input feature map, or IFM).
109
110Reference: https://developer.arm.com/documentation/102420/0200/
111
112- **ifm**: NHWC - (1, ifm_height, ifm_width, ifm_channels)
113 NHCWB16 - (1, ifm_height, ifm_channels // 16, ifm_width, 16)
114- **ofm**: (1, ofm_height, ofm_width, ofm_channels)
115
116)code" TVM_ADD_FILELINE)
117 .set_attrs_type<EthosuPoolingAttrs>()
118 .set_num_inputs(2)
119 .add_argument("ifm", "Tensor", "The Input Feature Map tensor (IFM).")
120 .add_argument("lut", "Tensor", "The look-up table of values to use if activation = 'LUT'")
121 .set_support_level(11)
122 .add_type_rel("EthosuPooling", EthosuPoolingRel);
