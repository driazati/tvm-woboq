43 uint32_t tindex = self->type_index();

51// Hash handler that handles free vars
52// by assigning an unique counter in the order of their ocurrence.
53//
54// This algorithm depends on the determinism of the traversal of SHash function.
55// In particular, when we traverse unordered_map, we should first sort
56// the entries by keys(or hash of keys) before traversing.
57
58class VarCountingSHashHandler : public SHashReducer::Handler {
59 public:
60 /*! \brief Pending reduce tasks. */
61 struct Task {
62 /*!
63 * \brief The object operand to be hashed.
64 * If the object is nullptr, then the reduced hash is already set
65 * the correct value.
66 */
67 ObjectRef object;
68 /*! \brief The partially reduce hash value.*/
69 size_t reduced_hash;
70 /*! \brief The expected location in the result stack. */
71 size_t result_stack_index = std::numeric_limits<size_t>::max();
72 /*! \brief Whether the children has been expanded via SEqualReduce */
73 bool children_expanded{false};
74 /*! \brief Whether the node is graph node. */
75 bool graph_node_hash{false};
76 /*! \brief whether to map the free variables. */
77 bool map_free_vars;
78
79 Task() = default;
80 explicit Task(ObjectRef object, size_t reduced_hash, bool map_free_vars)
81 : object(object), reduced_hash(reduced_hash), map_free_vars(map_free_vars) {}
82 };
83
84 VarCountingSHashHandler() {}
85
86 void MarkGraphNode() final {
87 // need to push to pending tasks in this case
88 ICHECK(!allow_push_to_stack_ && !task_stack_.empty());
89 task_stack_.back().graph_node_hash = true;
90 }
91
92 bool LookupHashedValue(const ObjectRef& key, size_t* hash_value) final {
93 auto it = hash_memo_.find(key);
94 if (it != hash_memo_.end()) {
95 hash_value[0] = it->second;
96 return true;
97 }
98 return false;
99 }
100
101 void SHashReduceHashedValue(size_t hashed_value) final {
102 pending_tasks_.emplace_back(Task(ObjectRef(nullptr), hashed_value, false));
103 }
104
105 void SHashReduceFreeVar(const runtime::Object* var, bool map_free_vars) final {
106 ICHECK(!hash_memo_.count(GetRef<ObjectRef>(var)));
107 if (map_free_vars) {
108 // use counter value.
109 size_t value = std::hash<size_t>()(free_var_counter_++);
110 pending_tasks_.emplace_back(Task(ObjectRef(nullptr), value, false));
111 } else {
112 // use pointer hash
113 size_t value = std::hash<const runtime::Object*>()(var);
114 pending_tasks_.emplace_back(Task(ObjectRef(nullptr), value, false));
115 }
116 }
117
118 void SHashReduce(const ObjectRef& object, bool map_free_vars) final {
119 // Directly push the result
120 // Note: it is still important to push the result to pendng tasks
121 // so that the reduction order of hash values stays the same.
122 if (!object.defined()) {
123 pending_tasks_.emplace_back(Task(ObjectRef(nullptr), 0, false));
124 return;
125 }
126 auto it = hash_memo_.find(object);
127 if (it != hash_memo_.end()) {
128 pending_tasks_.emplace_back(Task(ObjectRef(nullptr), it->second, false));
129 } else {
130 // Push a pending task with initial value.
131 pending_tasks_.emplace_back(Task(object, object->GetTypeKeyHash(), map_free_vars));
132 }
133 }
134
135 size_t Hash(const ObjectRef& object, bool map_free_vars) {
136 ICHECK_EQ(task_stack_.size(), 0U);
137 ICHECK_EQ(pending_tasks_.size(), 0U);
138 ICHECK_EQ(result_stack_.size(), 0U);
139
140 this->SHashReduce(object, map_free_vars);
141 ICHECK_EQ(pending_tasks_.size(), 1U);
142 ICHECK(allow_push_to_stack_);
143 task_stack_.emplace_back(std::move(pending_tasks_.back()));
144 pending_tasks_.clear();
145
146 this->RunTasks();
147
148 ICHECK_EQ(result_stack_.size(), 1U);
149 size_t ret = result_stack_.back();
150 result_stack_.pop_back();
151 return ret;
152 }
153
154 protected:
155 /*!
156 * \brief Pop the top entry of the task stack and push the hash into the result stack.
157 */
158 void PopTaskStack() {
159 const auto& entry = task_stack_.back();
160 result_stack_.push_back(entry.reduced_hash);
161 task_stack_.pop_back();
162 }
163 /*!
164 * \brief Compute the reduced hash value for the task.
165 * \param task The indicated task.
166 */
167 size_t ReduceHash(const Task& task) {
168 size_t stack_begin = task.result_stack_index;
169 ICHECK_LE(stack_begin, result_stack_.size());
170
171 // combine in the reverse order of the stack.
172 size_t reduced_hash = task.reduced_hash;
173 for (size_t i = result_stack_.size(); i != stack_begin; --i) {
174 reduced_hash = support::HashCombine(reduced_hash, result_stack_[i - 1]);
175 }
176 result_stack_.resize(stack_begin);
177 return reduced_hash;
178 }
179 // run the tasks.
180 void RunTasks() {
181 while (task_stack_.size() != 0) {
182 // Caution: entry becomes invalid when the stack changes
183 auto& entry = task_stack_.back();
184 if (entry.children_expanded) {
185 // reduce hash
186 entry.reduced_hash = ReduceHash(entry);
187 // When all the children has expanded and visited.
188 // entry.reduced_hash contains the reduced hash result.
189 auto it = hash_memo_.find(entry.object);
190 if (it != hash_memo_.end()) {
191 // use the pre-computed hash for the object.
192 entry.reduced_hash = it->second;
193 } else {
194 // Append the graph node counter to the hash
195 // so that we can distinguish DAG from trees.
196 if (entry.graph_node_hash) {
197 entry.reduced_hash = support::HashCombine(entry.reduced_hash,
198 std::hash<size_t>()(graph_node_counter_++));
199 }
200 hash_memo_[entry.object] = entry.reduced_hash;
201 }
202 // send value to parent.
203 this->PopTaskStack();
204 } else if (!entry.object.defined()) {
205 // Directly send value to parent
206 this->PopTaskStack();
207 } else {
208 // check if there are already hash for object.
209 auto it = hash_memo_.find(entry.object);
210 if (it != hash_memo_.end()) {
211 entry.reduced_hash = it->second;
212 this->PopTaskStack();
213 } else {
214 // NOTE: important to modify entry before visit.
215 // as entry becomes invalid after we change the stack.
216 entry.children_expanded = true;
217 entry.result_stack_index = result_stack_.size();
218
219 ICHECK_EQ(pending_tasks_.size(), 0U);
220 allow_push_to_stack_ = false;
221 // dispatch hash, reduce to the current slot.
222 this->DispatchSHash(entry.object, entry.map_free_vars);
223 allow_push_to_stack_ = true;
224 // Move pending tasks to the stack until the marked point.
225 while (pending_tasks_.size() != 0) {
226 task_stack_.emplace_back(std::move(pending_tasks_.back()));
227 pending_tasks_.pop_back();
228 }
229 }
230 }
231 }
232 }
233
234 // The default equal as registered in the structural equal vtable.
235 void DispatchSHash(const ObjectRef& object, bool map_free_vars) {
236 ICHECK(object.defined());
237 vtable_->SHashReduce(object.get(), SHashReducer(this, map_free_vars));
238 }
239
240 private:
241 // free var counter.
242 size_t free_var_counter_{0};
243 // graph node counter.
244 size_t graph_node_counter_{0};
245 // record current stack top
246 bool allow_push_to_stack_{true};
247 // list of pending tasks to be pushed to the stack.
248 std::vector<Task> pending_tasks_;
249 // Internal task stack to executed the task
250 std::vector<Task> task_stack_;
251 // Internal stack to store the result poped from the task stack.
252 std::vector<size_t> result_stack_;
253 // reflection vtable
254 ReflectionVTable* vtable_ = ReflectionVTable::Global();
255 // map from lhs to rhs
256 std::unordered_map<ObjectRef, size_t, ObjectPtrHash, ObjectPtrEqual> hash_memo_;
257};

332void NDArrayContainerTrait::SHashReduce(const runtime::NDArray::Container* key,
333 SHashReducer hash_reduce) {
334 ICHECK_EQ(key->dl_tensor.device.device_type, kDLCPU) << "can only compare CPU tensor";
335 ICHECK(runtime::IsContiguous(key->dl_tensor)) << "Can only hash contiguous tensor";
336 hash_reduce(runtime::DataType(key->dl_tensor.dtype));
337 hash_reduce(key->dl_tensor.ndim);
338 for (int i = 0; i < key->dl_tensor.ndim; ++i) {
339 hash_reduce(key->dl_tensor.shape[i]);
340 }
341 hash_reduce->SHashReduceHashedValue(runtime::String::HashBytes(
342 static_cast<const char*>(key->dl_tensor.data), runtime::GetDataSize(key->dl_tensor)));
343}

345bool NDArrayContainerTrait::SEqualReduce(const runtime::NDArray::Container* lhs,
346 const runtime::NDArray::Container* rhs,
347 SEqualReducer equal) {
348 if (lhs == rhs) return true;
349
350 auto ldt = lhs->dl_tensor.dtype;
351 auto rdt = rhs->dl_tensor.dtype;
352 ICHECK_EQ(lhs->dl_tensor.device.device_type, kDLCPU) << "can only compare CPU tensor";
353 ICHECK_EQ(rhs->dl_tensor.device.device_type, kDLCPU) << "can only compare CPU tensor";
354 ICHECK(runtime::IsContiguous(lhs->dl_tensor)) << "Can only compare contiguous tensor";
355 ICHECK(runtime::IsContiguous(rhs->dl_tensor)) << "Can only compare contiguous tensor";
356
357 if (lhs->dl_tensor.ndim != rhs->dl_tensor.ndim) return false;
358 for (int i = 0; i < lhs->dl_tensor.ndim; ++i) {
359 if (!equal(lhs->dl_tensor.shape[i], rhs->dl_tensor.shape[i])) return false;
360 }
361 if (ldt.code == rdt.code && ldt.lanes == rdt.lanes && ldt.bits == rdt.bits) {
362 size_t data_size = runtime::GetDataSize(lhs->dl_tensor);
363 return std::memcmp(lhs->dl_tensor.data, rhs->dl_tensor.data, data_size) == 0;
364 } else {
365 return false;
366 }
367}

388struct ArrayNodeTrait {
389 static constexpr const std::nullptr_t VisitAttrs = nullptr;
390
391 static void SHashReduce(const ArrayNode* key, SHashReducer hash_reduce) {
392 hash_reduce(static_cast<uint64_t>(key->size()));
393 for (size_t i = 0; i < key->size(); ++i) {
394 hash_reduce(key->at(i));
395 }
396 }
397
398 static bool SEqualReduce(const ArrayNode* lhs, const ArrayNode* rhs, SEqualReducer equal) {
399 if (equal.IsPathTracingEnabled()) {
400 return SEqualReduceTraced(lhs, rhs, equal);
401 }
402
403 if (lhs->size() != rhs->size()) return false;
404 for (size_t i = 0; i < lhs->size(); ++i) {
405 if (!equal(lhs->at(i), rhs->at(i))) return false;
406 }
407 return true;
408 }
409
410 private:
411 static bool SEqualReduceTraced(const ArrayNode* lhs, const ArrayNode* rhs,
412 const SEqualReducer& equal) {
413 size_t min_size = std::min(lhs->size(), rhs->size());
414 const ObjectPathPair& array_paths = equal.GetCurrentObjectPaths();
415
416 for (size_t index = 0; index < min_size; ++index) {
417 ObjectPathPair element_paths = {array_paths->lhs_path->ArrayIndex(index),
418 array_paths->rhs_path->ArrayIndex(index)};
419 if (!equal(lhs->at(index), rhs->at(index), element_paths)) {
420 return false;
421 }
422 }
423
424 if (lhs->size() == rhs->size()) {
425 return true;
426 }
427
428 // If the array length is mismatched, don't report it immediately.
429 // Instead, defer the failure until we visit all children.
430 //
431 // This is for human readability. For example, say we have two sequences
432 //
433 // (1) a b c d e f g h i j k l m
434 // (2) a b c d e g h i j k l m
435 //
436 // If we directly report a mismatch at the end of the array right now,
437 // the user will see that array (1) has an element `m` at index 12 but array (2)
438 // has no index 12 because it's too short:
439 //
440 // (1) a b c d e f g h i j k l m
441 // ^error here
442 // (2) a b c d e g h i j k l m
443 // ^ error here
444 //
445 // This is not very helpful. Instead, if we defer reporting this mismatch until all elements
446 // are fully visited, we can be much more helpful with pointing out the location:
447 //
448 // (1) a b c d e f g h i j k l m
449 // ^
450 // error here
451 //
452 // (2) a b c d e g h i j k l m
453 // ^
454 // error here
455 if (lhs->size() > min_size) {
456 equal->DeferFail({array_paths->lhs_path->ArrayIndex(min_size),
457 array_paths->rhs_path->MissingArrayElement(min_size)});
458 } else {
459 equal->DeferFail({array_paths->lhs_path->MissingArrayElement(min_size),
460 array_paths->rhs_path->ArrayIndex(min_size)});
461 }
462
463 // Can return `true` pretending that everything is good since we have deferred the failure.
464 return true;
465 }
466};

469 return ::tvm::runtime::make_object<ArrayNode>();

472struct MapNodeTrait {
473 static constexpr const std::nullptr_t VisitAttrs = nullptr;
474
475 static void SHashReduceForOMap(const MapNode* key, SHashReducer hash_reduce) {
476 // SHash's var handling depends on the determinism of traversal.
477 // NOTE: only book-keep the mapped hash keys.
478 // This resolves common use cases where we want to store
479 // Map<Var, Value> where Var is defined in the function
480 // parameters.
481 using KV = std::pair<size_t, ObjectRef>;
482 std::vector<KV> temp;
483 for (const auto& kv : *key) {
484 size_t hashed_value;
485 if (hash_reduce->LookupHashedValue(kv.first, &hashed_value)) {
486 temp.emplace_back(hashed_value, kv.second);
487 }
488 }
489 // sort by the hash key of the keys.
490 std::sort(temp.begin(), temp.end(),
491 [](const KV& lhs, const KV& rhs) { return lhs.first < rhs.first; });
492 // add size to the hash
493 hash_reduce(static_cast<uint64_t>(key->size()));
494 // hash the content
495 for (size_t i = 0; i < temp.size();) {
496 size_t k = i + 1;
497 for (; k < temp.size() && temp[k].first == temp[i].first; ++k) {
498 }
499 // ties are rare, but we need to skip them to make the hash determinsitic
500 if (k == i + 1) {
501 hash_reduce->SHashReduceHashedValue(temp[i].first);
502 hash_reduce(temp[i].second);
503 }
504 i = k;
505 }
506 }
507
508 static void SHashReduceForSMap(const MapNode* key, SHashReducer hash_reduce) {
509 // NOTE: only book-keep the mapped hash keys.
510 // This resolves common use cases where we want to store
511 // Map<Var, Value> where Var is defined in the function
512 // parameters.
513 using KV = std::pair<String, ObjectRef>;
514 std::vector<KV> temp;
515 for (const auto& kv : *key) {
516 temp.push_back(std::make_pair(Downcast<String>(kv.first), kv.second));
517 }
518 // sort by the hash key of the keys.
519 std::sort(temp.begin(), temp.end(),
520 [](const KV& lhs, const KV& rhs) { return lhs.first < rhs.first; });
521 // NOTE: we won't have ties
522 // add size to the hash after sorting.
523 hash_reduce(static_cast<uint64_t>(key->size()));
524 // hash the content
525 for (size_t i = 0; i < temp.size(); ++i) {
526 hash_reduce(temp[i].first);
527 hash_reduce(temp[i].second);
528 }
529 }

531 static void SHashReduce(const MapNode* key, SHashReducer hash_reduce) {
532 bool is_str_map = std::all_of(key->begin(), key->end(), [](const auto& v) {
533 return v.first->template IsInstance<StringObj>();
534 });
535 if (is_str_map) {
536 SHashReduceForSMap(key, hash_reduce);
537 } else {
538 SHashReduceForOMap(key, hash_reduce);
539 }

542 static bool SEqualReduceForOMap(const MapNode* lhs, const MapNode* rhs, SEqualReducer equal) {
543 for (const auto& kv : *lhs) {
544 // Only allow equal checking if the keys are already mapped
545 // This resolves common use cases where we want to store
546 // Map<Var, Value> where Var is defined in the function
547 // parameters.
548 ObjectRef rhs_key = equal->MapLhsToRhs(kv.first);
549 if (!rhs_key.defined()) return false;
550 auto it = rhs->find(rhs_key);
551 if (it == rhs->end()) return false;
552 if (!equal(kv.second, it->second)) return false;
553 }
554 return true;
555 }

557 static bool SEqualReduceForSMap(const MapNode* lhs, const MapNode* rhs, SEqualReducer equal) {
558 for (const auto& kv : *lhs) {
559 auto it = rhs->find(kv.first);
560 if (it == rhs->end()) return false;
561 if (!equal(kv.second, it->second)) return false;
562 }
563 return true;
564 }

566 static bool IsStringMap(const MapNode* map) {
567 return std::all_of(map->begin(), map->end(),
568 [](const auto& v) { return v.first->template IsInstance<StringObj>(); });

571 static bool SEqualReduceTracedForOMap(const MapNode* lhs, const MapNode* rhs,
572 const SEqualReducer& equal) {
573 const ObjectPathPair& map_paths = equal.GetCurrentObjectPaths();
574
575 std::vector<const Object*> seen_rhs_keys;
576
577 // First, check that every key from `lhs` is also in `rhs`,
578 // and their values are mapped to each other.
579 for (const auto& kv : *lhs) {
580 ObjectPath lhs_path = map_paths->lhs_path->MapValue(kv.first);
581
582 ObjectRef rhs_key = equal->MapLhsToRhs(kv.first);
583 if (!rhs_key.defined()) {
584 equal.RecordMismatchPaths({lhs_path, map_paths->rhs_path->MissingMapEntry()});
585 return false;
586 }
587
588 auto it = rhs->find(rhs_key);
589 if (it == rhs->end()) {
590 equal.RecordMismatchPaths({lhs_path, map_paths->rhs_path->MissingMapEntry()});
591 return false;
592 }
593
594 if (!equal(kv.second, it->second, {lhs_path, map_paths->rhs_path->MapValue(it->first)})) {
595 return false;
596 }
597
598 seen_rhs_keys.push_back(it->first.get());
599 }
600
601 std::sort(seen_rhs_keys.begin(), seen_rhs_keys.end());
602
603 // Second, check that we have visited every `rhs` key when iterating over `lhs`.
604 for (const auto& kv : *rhs) {
605 if (!std::binary_search(seen_rhs_keys.begin(), seen_rhs_keys.end(), kv.first.get())) {
606 equal.RecordMismatchPaths(
607 {map_paths->lhs_path->MissingMapEntry(), map_paths->rhs_path->MapValue(kv.first)});
608 return false;
609 }
610 }
611
612 ICHECK(lhs->size() == rhs->size());
613 return true;
614 }

616 static bool SEqualReduceTracedForSMap(const MapNode* lhs, const MapNode* rhs,
617 const SEqualReducer& equal) {
618 const ObjectPathPair& map_paths = equal.GetCurrentObjectPaths();
619
620 // First, check that every key from `lhs` is also in `rhs`, and their values are equal.
621 for (const auto& kv : *lhs) {
622 ObjectPath lhs_path = map_paths->lhs_path->MapValue(kv.first);
623 auto it = rhs->find(kv.first);
624 if (it == rhs->end()) {
625 equal.RecordMismatchPaths({lhs_path, map_paths->rhs_path->MissingMapEntry()});
626 return false;
627 }
628
629 if (!equal(kv.second, it->second, {lhs_path, map_paths->rhs_path->MapValue(it->first)})) {
630 return false;
631 }
632 }
633
634 // Second, make sure every key from `rhs` is also in `lhs`.
635 for (const auto& kv : *rhs) {
636 ObjectPath rhs_path = map_paths->rhs_path->MapValue(kv.first);
637 if (!lhs->count(kv.first)) {
638 equal.RecordMismatchPaths({map_paths->lhs_path->MissingMapEntry(), rhs_path});
639 return false;
640 }
641 }
642
643 ICHECK(lhs->size() == rhs->size());
644 return true;
645 }
646
647 static bool SEqualReduceTraced(const MapNode* lhs, const MapNode* rhs,
648 const SEqualReducer& equal) {
649 if (IsStringMap(lhs)) {
650 return SEqualReduceTracedForSMap(lhs, rhs, equal);
651 } else {
652 return SEqualReduceTracedForOMap(lhs, rhs, equal);
653 }
654 }
655
656 static bool SEqualReduce(const MapNode* lhs, const MapNode* rhs, SEqualReducer equal) {
657 if (equal.IsPathTracingEnabled()) {
658 return SEqualReduceTraced(lhs, rhs, equal);
659 }
660
661 if (rhs->size() != lhs->size()) return false;
662 if (rhs->size() == 0) return true;
663 bool ls = IsStringMap(lhs);
664 bool rs = IsStringMap(rhs);
665 if (ls != rs) {
666 return false;
667 }
668 return (ls && rs) ? SEqualReduceForSMap(lhs, rhs, equal) : SEqualReduceForOMap(lhs, rhs, equal);
669 }
670};
