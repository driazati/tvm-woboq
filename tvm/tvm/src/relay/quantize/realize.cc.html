44Expr QRealizeIntExprNode::Realize() const {
45 Expr data = this->data;
46 // dequantize
47 data = Cast(data, DataType::Float(32));
48 data = Multiply(data, this->dom_scale);
49 return data;
50}

52QRealizeIntExpr::QRealizeIntExpr(Expr data, Expr dom_scale, DataType dtype) {
53 ObjectPtr<QRealizeIntExprNode> n = make_object<QRealizeIntExprNode>();
54 n->data = std::move(data);
55 n->dom_scale = std::move(dom_scale);
56 n->dtype = std::move(dtype);

64/* calculate `data * s1 / s2`, use shift if possible */
65inline Expr MulAndDiv(Expr data, float s1, float s2, DataType dtype,
66 const Array<IndexExpr>& data_shape) {
67 const QConfig& cfg = QConfig::Current();
68 // here we assume the dtype of data is dtype activation
69 if (s1 == s2) return data;
70
71 float factor = s1 / s2;
72 float shift_factor = std::log2(factor);
73 ICHECK_GT(shift_factor, 0);
74 if (static_cast<int>(shift_factor) == shift_factor) {
75 return LeftShift(data, MakeConstantScalar(dtype, static_cast<int>(shift_factor)));
76 } else if (static_cast<int>(factor) == factor) {
77 return Multiply(data, MakeConstantScalar(dtype, factor));
78 } else {
79 if (cfg->rounding == "UPWARD") {
80 auto [fixed_point_multiplier, shift] = qnn::GetFixedPointMultiplierShift(factor);
81 data = relay::FixedPointMultiply(data, fixed_point_multiplier, shift);
82 } else {
83 data = qnn::FixedPointMultiplyToNearest(data, factor, data_shape);
84 }
85
86 return Cast(data, dtype);
87 }
88}

90Expr QuantizeRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
91 const QConfig& cfg = QConfig::Current();
92 // do not handle data type cast
93 const auto param = ref_call->attrs.as<SimulatedQuantizeAttrs>();
94 ICHECK_EQ(param->rounding, "round");
95
96 Expr dom_scale = new_args[1];
97 Expr clip_min = new_args[2];
98 Expr clip_max = new_args[3];
99
100 float dom_scale_imm = GetScalarFromConstant<float>(dom_scale);
101 float clip_min_imm = GetScalarFromConstant<float>(clip_min);
102 float clip_max_imm = GetScalarFromConstant<float>(clip_max);
103
104 // x * idom_scale = y * odom_scale
105 // => y = x * idom_scale / odom_scale
106 if (const auto* n = new_args[0].as<QRealizeIntExprNode>()) {
107 // int32->int8
108 Expr data = n->data;
109 float idom_scale_imm = GetScalarFromConstant<float>(n->dom_scale);
110 float odom_scale_imm = GetScalarFromConstant<float>(dom_scale);
111 if (idom_scale_imm == odom_scale_imm) {
112 // same domain scale, only clip
113 data = Clip(data, clip_min_imm, clip_max_imm);
114 return QRealizeIntExpr(data, dom_scale, n->dtype);
115 }
116
117 float shift_nbit = std::log2(odom_scale_imm / idom_scale_imm);
118 ICHECK_NE(shift_nbit, 0);
119 if (static_cast<int>(shift_nbit) == shift_nbit) {
120 if (shift_nbit > 0) {
121 // use right shift
122 if (cfg->round_for_shift) {
123 float round_bias = std::pow(2.0, shift_nbit - 1);
124 data = Add(data, MakeConstantScalar(cfg->dtype_activation, static_cast<int>(round_bias)));
125 }
126 data = RightShift(data,
127 MakeConstantScalar(cfg->dtype_activation, static_cast<int>(shift_nbit)));
128 } else {
129 data = LeftShift(data,
130 MakeConstantScalar(cfg->dtype_activation, static_cast<int>(-shift_nbit)));
131 }
132 data = Clip(data, clip_min_imm, clip_max_imm);
133 return QRealizeIntExpr(data, dom_scale, n->dtype);
134 } else {
135 data = Cast(data, DataType::Int(64));
136 if (cfg->rounding == "UPWARD") {
137 auto [fixed_point_multiplier, shift] =
138 qnn::GetFixedPointMultiplierShift(idom_scale_imm / odom_scale_imm);
139 data = relay::FixedPointMultiply(data, fixed_point_multiplier, shift);
140 } else {
141 data = qnn::FixedPointMultiplyToNearest(data, idom_scale_imm / odom_scale_imm,
142 ref_call->type_as<TensorTypeNode>()->shape);
143 }
144 data = Cast(Clip(data, clip_min_imm, clip_max_imm), n->dtype);
145 return QRealizeIntExpr(data, dom_scale, n->dtype);
146 }
147 }
148
149 // quantize from real
150 ICHECK(!new_args[0]->IsInstance<TempExprNode>());
151 Expr data = new_args[0];
152 Expr scaled_data = Multiply(data, MakeConstantScalar(DataType::Float(32), 1 / dom_scale_imm));
153 Expr round_data = Clip(Round(scaled_data), clip_min_imm, clip_max_imm);
154 return QRealizeIntExpr(round_data, dom_scale, DataType::Float(32));
155}

157Expr FoldConstantOpt(const Expr& expr) {
158 auto mod = IRModule::FromExpr(expr);
159 mod = transform::FoldConstant()(mod);
160 auto entry_func = Downcast<Function>(mod->Lookup("main"));
161 return expr.as<FunctionNode>() == nullptr ? entry_func->body : entry_func;
162}

167Expr Conv2dRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
168 const QConfig& cfg = QConfig::Current();
169 ICHECK_EQ(new_args.size(), 2);
170 if (new_args[0].as<QRealizeIntExprNode>() && new_args[1].as<QRealizeIntExprNode>()) {
171 const auto* lhs = new_args[0].as<QRealizeIntExprNode>();
172 const auto* rhs = new_args[1].as<QRealizeIntExprNode>();
173 Expr ldata = lhs->data;
174 if (lhs->dtype != cfg->dtype_input) {
175 ldata = Cast(ldata, cfg->dtype_input);
176 }
177 Expr rdata = Cast(rhs->data, cfg->dtype_weight);
178
179 const auto ref_attrs = ref_call->attrs.as<Conv2DAttrs>();
180 auto attrs = make_object<Conv2DAttrs>();
181 *attrs = *ref_attrs;
182 DataType out_dtype = cfg->dtype_activation;
183 attrs->out_dtype = out_dtype;
184
185 Expr ret = Call(ref_call->op, {ldata, rdata}, Attrs(attrs), ref_call->type_args);
186 Expr mul = Multiply(lhs->dom_scale, rhs->dom_scale);
187 Expr dom_scale = FoldConstantOpt(mul);
188 return QRealizeIntExpr(ret, dom_scale, out_dtype);
189 }
190 ICHECK(!new_args[0]->IsInstance<TempExprNode>() || !new_args[1]->IsInstance<TempExprNode>());
191 return Expr(nullptr);
192}

256Expr MulRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
257 const QConfig& cfg = QConfig::Current();
258 ICHECK_EQ(new_args.size(), 2);
259 if (new_args[0].as<QRealizeIntExprNode>() && new_args[1].as<QRealizeIntExprNode>()) {
260 // execute the operation with activation data type.
261 const auto* lhs = new_args[0].as<QRealizeIntExprNode>();
262 const auto* rhs = new_args[1].as<QRealizeIntExprNode>();
263 Expr ldata = lhs->data;
264 Expr rdata = rhs->data;
265
266 DataType dtype = cfg->dtype_activation;
267 if (lhs->dtype != dtype) {
268 ldata = Cast(ldata, dtype);
269 }
270 if (rhs->dtype != dtype) {
271 rdata = Cast(rdata, dtype);
272 }
273
274 Expr ret = ForwardOp(ref_call, {ldata, rdata});
275 Expr mul = Multiply(lhs->dom_scale, rhs->dom_scale);
276 Expr dom_scale = FoldConstantOpt(mul);
277 return QRealizeIntExpr(ret, dom_scale, dtype);
278 }
279 ICHECK(!new_args[0]->IsInstance<TempExprNode>() || !new_args[1]->IsInstance<TempExprNode>());
280 return Expr(nullptr);
281}

285float ChooseDomScale(const std::vector<const QRealizeIntExprNode*>& nptrs) {
286 if (nptrs.size() == 2) {
287 // x = a * s1, y = b * s2
288 // x + y = (a * s1 / s2 + b) * s2, if s1 > s2
289 // = (a + b * s2 / s1) * s1, if s2 > s1
290 float s1 = GetScalarFromConstant<float>(nptrs[0]->dom_scale);
291 float s2 = GetScalarFromConstant<float>(nptrs[1]->dom_scale);
292 return s1 > s2 ? s2 : s1;
293 } else {
294 const QConfig& cfg = QConfig::Current();
295 float scale = cfg->global_scale;
296 return scale / std::pow(2.0, cfg->nbit_activation - 1);
297 }
298}

300/* \brief Unify the dom scale of arguments */
301Array<Expr> UnifyDTypeScale(const Array<Expr>& ref_args, const Array<Expr>& args,
302 DataType* dtype_ptr, Expr* scale_ptr,
303 DataType dtype = DataType::Void()) {
304 static const Op& simulated_quantize = Op::Get("relay.op.annotation.simulated_quantize");
305 const QConfig& cfg = QConfig::Current();
306
307 std::vector<const QRealizeIntExprNode*> nptrs;
308 Array<Expr> ret;
309 for (auto arg : args) {
310 const auto* nptr = arg.as<QRealizeIntExprNode>();
311 ICHECK(nptr);
312 nptrs.push_back(nptr);
313 ret.push_back(nptr->data);
314 }
315
316 // unify the data type
317 ICHECK_EQ(ref_args.size(), args.size());
318
319 if (dtype.is_void()) {
320 if (ret.size() == 2 && nptrs[1]->dtype == cfg->dtype_input) {
321 dtype = cfg->dtype_input;
322 } else {
323 dtype = cfg->dtype_activation;
324 }
325 }
326
327 for (size_t i = 0; i < ret.size(); ++i) {
328 auto ref_arg = ref_args[i].as<CallNode>();
329 if (nptrs[i]->dtype != dtype) {
330 ret.Set(i, Cast(ret[i], dtype));
331 } else if (ref_arg && ref_arg->op.same_as(simulated_quantize) &&
332 ref_arg->attrs.as<SimulatedQuantizeAttrs>()->kind == kQInput) {
333 auto new_arg = Cast(ret[i], cfg->dtype_input);
334 new_arg = StopFusion(new_arg);
335 ret.Set(i, Cast(new_arg, dtype));
336 }
337 }
338
339 // unify the dom_scale
340 float s = ChooseDomScale(nptrs);
341 Expr dom_scale = MakeConstantScalar(DataType::Float(32), s);
342 for (size_t i = 0; i < ret.size(); ++i) {
343 float cur_s = GetScalarFromConstant<float>(nptrs[i]->dom_scale);
344 ret.Set(i, MulAndDiv(ret[i], cur_s, s, dtype, ref_args[i]->type_as<TensorTypeNode>()->shape));
345 }
346
347 *dtype_ptr = dtype;
348 *scale_ptr = dom_scale;
349 return ret;
350}

352Expr AddRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
353 ICHECK_EQ(new_args.size(), 2);
354 if (new_args[0].as<QRealizeIntExprNode>() && new_args[1].as<QRealizeIntExprNode>()) {
355 DataType dtype;
356 Expr dom_scale;
357 // execute the operation with activation data type.
358 const QConfig& cfg = QConfig::Current();
359 Array<Expr> ret_args =
360 UnifyDTypeScale(ref_call->args, new_args, &dtype, &dom_scale, cfg->dtype_activation);
361 for (size_t i = 0; i < ret_args.size(); ++i) {
362 // do not fuse float32 arg
363 if (new_args[i].as<QRealizeIntExprNode>()->dtype == DataType::Float(32)) {
364 ret_args.Set(i, StopFusion(ret_args[i]));
365 }
366 }
367 Expr ret = ForwardOp(ref_call, ret_args);
368 return QRealizeIntExpr(ret, dom_scale, dtype);
369 }
370
371 ICHECK(!new_args[0]->IsInstance<TempExprNode>() && !new_args[1]->IsInstance<TempExprNode>());
372 return Expr(nullptr);
373}

377Expr ClipRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
378 ICHECK_EQ(new_args.size(), 1);
379 if (const auto* n = new_args[0].as<QRealizeIntExprNode>()) {
380 const auto ref_attrs = ref_call->attrs.as<ClipAttrs>();
381 auto attrs = make_object<ClipAttrs>();
382 double dom_scale = GetScalarFromConstant<float>(n->dom_scale);
383 attrs->a_min = ref_attrs->a_min / dom_scale;
384 attrs->a_max = ref_attrs->a_max / dom_scale;
385
386 Expr ret = Call(ref_call->op, {n->data}, Attrs(attrs), ref_call->type_args);
387 return QRealizeIntExpr(ret, n->dom_scale, n->dtype);
388 }
389 ICHECK(!new_args[0]->IsInstance<TempExprNode>());
390 return Expr(nullptr);
391}

395Expr ConcatenateRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
396 ICHECK_EQ(new_args.size(), 1);
397 ICHECK_EQ(ref_call->args.size(), 1);
398
399 const auto* tuple = new_args[0].as<TupleNode>();
400 const auto* ref_tuple = ref_call->args[0].as<TupleNode>();
401 ICHECK(tuple);
402 ICHECK(ref_tuple);
403 const Array<Expr>& arr = tuple->fields;
404 const Array<Expr>& ref_arr = ref_tuple->fields;
405
406 if (arr[0].as<QRealizeIntExprNode>()) {
407 DataType dtype;
408 Expr dom_scale;
409 Array<Expr> ret_args = UnifyDTypeScale(ref_arr, arr, &dtype, &dom_scale);
410 Expr ret = ForwardOp(ref_call, {Tuple(ret_args)});
411 return QRealizeIntExpr(ret, dom_scale, dtype);
412 } else {
413 for (auto arg : new_args) {
414 ICHECK(!arg->IsInstance<TempExprNode>());
415 }
416 return Expr(nullptr);
417 }
418}

501Expr BatchMatmulRealize(const Call& ref_call, const Array<Expr>& new_args, const ObjectRef& ctx) {
502 const QConfig& cfg = QConfig::Current();
503 ICHECK_EQ(new_args.size(), 2);
504 if (!new_args[0]->IsInstance<TempExprNode>() || !new_args[1]->IsInstance<TempExprNode>()) {
505 return Expr(nullptr);
506 }
507 const auto* lhs = new_args[0].as<QRealizeIntExprNode>();
508 const auto* rhs = new_args[1].as<QRealizeIntExprNode>();
509
510 Expr ldata = lhs->data;
511 Expr rdata = rhs->data;
512 DataType dtype_input = cfg->dtype_input;
513 DataType dtype_weight = cfg->dtype_weight;
514
515 if (lhs->dtype != dtype_input) {
516 ldata = Cast(ldata, dtype_input);
517 }
518 if (rhs->dtype != dtype_weight) {
519 rdata = Cast(rdata, dtype_weight);
520 }
521
522 const auto ref_attrs = ref_call->attrs.as<BatchMatmulAttrs>();
523 auto attrs = make_object<BatchMatmulAttrs>();
524 *attrs = *ref_attrs;
525 DataType out_dtype = cfg->dtype_activation;
526 attrs->out_dtype = out_dtype;
527
528 Expr ret = Call(ref_call->op, {ldata, rdata}, Attrs(attrs), ref_call->type_args);
529 Expr mul = Multiply(lhs->dom_scale, rhs->dom_scale);
530 Expr dom_scale = FoldConstantOpt(mul);
531 return QRealizeIntExpr(ret, dom_scale, out_dtype);
532}
