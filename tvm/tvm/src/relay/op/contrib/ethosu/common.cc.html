37Array<IndexExpr> EthosuInferElementwiseOutputShape(Array<IndexExpr> ifm_shape, String ifm_layout,
38 String ofm_layout, IndexExpr ofm_channels) {
39 // In the case of NHCWB16, convert the ifm shape to NHW (C not required for this function)
40 if (ifm_layout == "NHCWB16") {
41 ifm_shape = {ifm_shape[0], ifm_shape[1], ifm_shape[3]};
42 }
43 Array<IndexExpr> oshape({ifm_shape[0], ifm_shape[1], ifm_shape[2], ofm_channels});
44
45 // If the ofm is NHCWB16, convert the layout
46 if (ofm_layout == "NHCWB16") {
47 int channel_bricks = 1 + (oshape[3].as<IntImmNode>()->value - 1) / 16;
48 oshape = {oshape[0], oshape[1], channel_bricks, oshape[2], 16};
49 }
50
51 return oshape;
52}

54Array<IndexExpr> EthosuInferKernelOutput(Array<IndexExpr> ifm_shape, String ifm_layout,
55 String ofm_layout, Array<IndexExpr> kernel_shape,
56 IndexExpr ofm_channels, Array<IndexExpr> dilation,
57 Array<IndexExpr> strides, Array<IndexExpr> padding) {
58 // In the case of NHCWB16, convert the ifm shape to NHW (C not required for this function)
59 if (ifm_layout == "NHCWB16") {
60 ifm_shape = {ifm_shape[0], ifm_shape[1], ifm_shape[3]};
61 }
62 Array<IndexExpr> output_shape({ifm_shape[0], 0, 0, ofm_channels});
63
64 IndexExpr dilated_ksize_y = 1 + (kernel_shape[0] - 1) * dilation[0];
65 IndexExpr dilated_ksize_x = 1 + (kernel_shape[1] - 1) * dilation[1];
66 IndexExpr pad_h, pad_w;
67 GetPaddingHeightWidth(padding, &pad_h, &pad_w);
68 output_shape.Set(1, indexdiv(ifm_shape[1] + pad_h - dilated_ksize_y, strides[0]) + 1);
69 output_shape.Set(2, indexdiv(ifm_shape[2] + pad_w - dilated_ksize_x, strides[1]) + 1);
70
71 // If the ofm is NHCWB16, convert the layout
72 if (ofm_layout == "NHCWB16") {
73 int channel_bricks = 1 + (output_shape[3].as<IntImmNode>()->value - 1) / 16;
74 output_shape = {output_shape[0], output_shape[1], channel_bricks, output_shape[2], 16};
75 }
76
77 return output_shape;
78}

80Array<IndexExpr> EthosuInferUpscaledInput(Array<IndexExpr> ifm_shape, String ifm_layout) {
81 if (ifm_layout == "NHCWB16") {
82 ifm_shape = {ifm_shape[0], ifm_shape[1], ifm_shape[3], ifm_shape[2] * 16};
83 }
84
85 const int scale_factor = 2;
86 Array<IndexExpr> new_ifm_shape = {ifm_shape[0], ifm_shape[1] * scale_factor,
87 ifm_shape[2] * scale_factor, ifm_shape[3]};
88
89 if (ifm_layout == "NHCWB16") {
90 int channel_bricks = 1 + (new_ifm_shape[3].as<IntImmNode>()->value - 1) / 16;
91 new_ifm_shape = {new_ifm_shape[0], new_ifm_shape[1], channel_bricks, new_ifm_shape[2], 16};
92 }
93
94 return new_ifm_shape;
95}

102void CheckDataType(const TypeReporter& reporter, const DataType& data_type,
103 const std::initializer_list<DataType>& allowed_data_types,
104 const String& operator_name, const String& tensor_name,
105 const String& operator_type) {
106 for (const auto& i : allowed_data_types) {
107 if (data_type == i) {
108 return;
109 }
110 }
111
112 std::ostringstream message;
113 message << "Invalid operator: expected " << operator_name << " ";
114 if (operator_type != "") {
115 message << operator_type << " ";
116 }
117 message << "to have type in {";
118 for (auto it = allowed_data_types.begin(); it != allowed_data_types.end(); ++it) {
119 message << *it;
120 if (std::next(it) != allowed_data_types.end()) {
121 message << ", ";
122 }
123 }
124 message << "}";
125 message << " for " << tensor_name << " but was " << data_type << ".";
126
127 reporter->GetDiagCtx().EmitFatal(Diagnostic::Error(reporter->GetSpan()) << message.str());
128}

157void CheckDataTypeMatch(const TypeReporter& reporter, const DataType& data_type,
158 const DataType& data_type2, const String& operator_name,
159 const String& tensor_name, const String& tensor_name2,
160 const String& operator_type) {
161 if (data_type == data_type2) {
162 return;
163 }
164
165 std::ostringstream message;
166 message << "Invalid operator: expected " << operator_name << " ";
167 if (operator_type != "") {
168 message << operator_type << " ";
169 }
170 message << "data types for " << tensor_name << " and " << tensor_name2 << " to match, but was "
171 << data_type << " and " << data_type2;
172
173 reporter->GetDiagCtx().EmitFatal(Diagnostic::Error(reporter->GetSpan()) << message.str());
174}
