24
25#ifndef TVM_RELAY_OP_CONTRIB_ETHOSU_COMMON_H_
26#define TVM_RELAY_OP_CONTRIB_ETHOSU_COMMON_H_
27
28#include <tvm/relay/expr.h>
29
30namespace tvm {
31namespace relay {
32namespace op {
33namespace contrib {
34namespace ethosu {
35
36/*! \brief Infer the output tensor shape for binary elementwise operators.
37 * \param ifm_shape The shape of Input Feature Map.
38 * \param ifm_layout The layout of the IFM (NHWC or NHCWB16).
39 * \param ofm_layout The layout of the OFM (NHWC or NHCWB16).
40 * \param ofm_channels The number of Output Feature Map channels.
41 * \return The shape of the output tensor.
42 */
43Array<IndexExpr> EthosuInferElementwiseOutputShape(Array<IndexExpr> ifm_shape, String ifm_layout,
44 String ofm_layout, IndexExpr ofm_channels);
45
46/*! \brief Infer the output tensor shape for convolution and pooling operators.
47 * \param ifm_shape The shape of Input Feature Map.
48 * \param ifm_layout The layout of the IFM (NHWC or NHCWB16).
49 * \param ofm_layout The layout of the OFM (NHWC or NHCWB16).
50 * \param kernel_shape Kernel shape in format (height, width).
51 * \param ofm_channels The number of Output Feature Map channels.
52 * \param dilation The 2-dimensional dilation as (dilation_height, dilation_width).
53 * \param strides The 2 dimensional strides as (stride_height, stride_width).
54 * \param padding The 4 dimensional padding as (pad_top, pad_left, pad_bottom, pad_right).
55 * \return The shape of the output tensor.
56 */
57Array<IndexExpr> EthosuInferKernelOutput(Array<IndexExpr> ifm_shape, String ifm_layout,
58 String ofm_layout, Array<IndexExpr> kernel_shape,
59 IndexExpr ofm_channels, Array<IndexExpr> dilation,
60 Array<IndexExpr> strides, Array<IndexExpr> padding);
61
62/*! \brief Infer the Output Feature Map shape for operations that use upscaling.
63 * \param ifm_shape The shape of the Input Feature Map.
64 * \param ifm_layout The layout of the Input Feature Map.
65 */
66Array<IndexExpr> EthosuInferUpscaledInput(Array<IndexExpr> ifm_shape, String ifm_layout);
67
68/*! \brief Get data type from string representation.
69 * \param dtype Data type in lower case format followed by number of bits e.g. "int8".
70 */
71DataType DataTypeFromString(const String& dtype);
72
73/*! \brief Check the data type for a given input matches one given in allowed_data_types. Raise a
74 * type inference error if not.
75 * \param reporter The infer type reporter.
76 * \param data_type The data type to check.
77 * \param allowed_data_types An initializer list of allowed data types.
78 * \param operator_name The name of the operator to report.
79 * \param tensor_name The name of the tensor to report e.g. "ifm", "ofm".
80 * \param operator_type The type of the operator to report e.g. "ADD" for binary_elementwise.
81 */
82void CheckDataType(const TypeReporter& reporter, const DataType& data_type,
83 const std::initializer_list<DataType>& allowed_data_types,
84 const String& operator_name, const String& tensor_name,
85 const String& operator_type = "");
86
87/*! \brief Check the upscale method matches one given in allowed_upscale_methods. Raise a type
88 * inference error if not.
89 * \param reporter The infer type reporter.
90 * \param upscale_method The upscale method string to check.
91 * \param allowed_upscale_methods An initializer list of allowed upscale methods.
92 * \param operator_name The name of the operator to report.
93 * \param operator_type The type of the operator to report e.g. "ADD" for binary_elementwise.
94 */
95void CheckUpscaleMethod(const TypeReporter& reporter, const String& upscale_method,
96 const std::initializer_list<String>& allowed_upscale_methods,
97 const String& operator_name, const String& operator_type = "");
98
99/*! \brief Check the data type matches that of the second data type provided. Raise a type inference
100 * error if not.
101 * \param reporter The infer type reporter.
102 * \param data_type The data type to check.
103 * \param data_type2 The second data type to check.
104 * \param operator_name The name of the operator to report.
105 * \param tensor_name The name of the tensor to report e.g. "ifm", "ofm".
106 * \param tensor_name2 The name of the second tensor to report e.g. "ifm2".
107 * \param operator_type The type of the operator to report e.g. "ADD" for binary_elementwise.
108 */
109void CheckDataTypeMatch(const TypeReporter& reporter, const DataType& data_type,
110 const DataType& data_type2, const String& operator_name,
111 const String& tensor_name, const String& tensor_name2,
112 const String& operator_type = "");
113
114} // namespace ethosu
115} // namespace contrib
116} // namespace op
117} // namespace relay
118} // namespace tvm
