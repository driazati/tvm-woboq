35bool EthosuUnaryElementwiseRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
36 const TypeReporter& reporter) {
37 const int ifm_index = 0;
38 const int result_index = 2;
39 ICHECK_EQ(types.size(), result_index + 1);
40
41 const auto* ifm = types[ifm_index].as<TensorTypeNode>();
42 if (ifm == nullptr) return false;
43
44 const auto* param = attrs.as<EthosuUnaryElementwiseAttrs>();
45 CHECK(param != nullptr) << "EthosuUnaryElementwiseAttrs cannot be nullptr.";
46
47 const String operator_name = "ethosu_unary_elementwise";
48 const String operator_type = param->operator_type;
49 if (operator_type != "ABS" && operator_type != "CLZ") {
50 reporter->GetDiagCtx().EmitFatal(Diagnostic::Error(reporter->GetSpan())
51 << "Invalid operator: expected << " << operator_name
52 << " 'ABS' or 'CLZ' for operator_type but was"
53 << operator_type);
54 return false;
55 }
56
57 const DataType ifm_dtype = ifm->dtype;
58 if (operator_type == "CLZ") {
59 CheckDataType(reporter, ifm_dtype, {DataType::Int(32)}, operator_name, "ifm", operator_type);
60 } else {
61 CheckDataType(reporter, ifm_dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name, "ifm",
62 operator_type);
63 }
64
65 // Assign ofm type
66 auto ofm_shape = EthosuInferElementwiseOutputShape(ifm->shape, param->ifm_layout,
67 param->ofm_layout, param->ofm_channels);
68 reporter->Assign(types[result_index], TensorType(ofm_shape, ifm_dtype));
69 return true;
70}
71
72Expr MakeEthosuUnaryElementwise(Expr ifm, Expr lut, String operator_type, double ifm_scale,
73 int ifm_zero_point, double ofm_scale, int ofm_zero_point,
74 IndexExpr ofm_channels, String activation, int clip_min,
75 int clip_max, String rounding_mode, String ifm_layout,
76 String ofm_layout) {
77 auto attrs = make_object<EthosuUnaryElementwiseAttrs>();
78
79 attrs->operator_type = std::move(operator_type);
80 attrs->ifm_scale = ifm_scale;
81 attrs->ifm_zero_point = ifm_zero_point;
82 attrs->ofm_scale = ofm_scale;
83 attrs->ofm_zero_point = ofm_zero_point;
84 attrs->ofm_channels = std::move(ofm_channels);
85 attrs->activation = std::move(activation);
86 attrs->clip_min = clip_min;
87 attrs->clip_max = clip_max;
88 attrs->rounding_mode = std::move(rounding_mode);
89 attrs->ifm_layout = std::move(ifm_layout);
90 attrs->ofm_layout = std::move(ofm_layout);
91
92 static const Op& op = Op::Get("contrib.ethosu.unary_elementwise");
93 return Call(op, {ifm, lut}, Attrs(attrs), {});
94}
95
96TVM_REGISTER_GLOBAL("relay.op._make.ethosu_unary_elementwise")
97 .set_body_typed(MakeEthosuUnaryElementwise);
98
99RELAY_REGISTER_OP("contrib.ethosu.unary_elementwise")
100 .describe(R"code(Quantized unary elementwise operator for Arm(R) Ethos(TM)-U NPUs.
101
102This Relay operator corresponds to the hardware-implemented quantized
103unary elementwise operation found on NPUs. It accepts either NHWC
104or NHCWB16 format for the inputs data (input feature maps, or IFMs).
105
106Reference: https://developer.arm.com/documentation/102420/0200/
107
108- **ifm**: NHWC - (1, ifm_height, ifm_width, ifm_channels)
109 NHCWB16 - (1, ifm_height, ifm_channels // 16, ifm_width, 16)
110- **ofm**: (1, ofm_height, ofm_width, ofm_channels)
111
112)code" TVM_ADD_FILELINE)
113 .set_attrs_type<EthosuUnaryElementwiseAttrs>()
114 .set_num_inputs(2)
115 .add_argument("ifm", "Tensor", "The Input Feature Map tensor (IFM).")
116 .add_argument("lut", "Tensor", "The look-up table values to use if activation = 'LUT'")
117 .set_support_level(11)
118 .add_type_rel("EthosuUnaryElementwise", EthosuUnaryElementwiseRel);
