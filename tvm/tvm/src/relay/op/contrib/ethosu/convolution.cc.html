42bool EthosuConv2DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
43 const TypeReporter& reporter) {
44 CHECK_EQ(types.size(), 5);
45 const auto* ifm = types[0].as<TensorTypeNode>();
46 const auto* weight = types[1].as<TensorTypeNode>();
47 const auto* scale_bias = types[2].as<TensorTypeNode>();
48 if (ifm == nullptr || weight == nullptr) return false;
49 const auto* param = attrs.as<EthosuConv2DAttrs>();
50 CHECK(param != nullptr) << "EthosuConv2DAttrs cannot be nullptr.";
51 const String operator_name = "ethosu_conv2d";
52
53 CheckDataType(reporter, ifm->dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name, "ifm");
54 CheckDataType(reporter, weight->dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name,
55 "weight");
56 CheckDataType(reporter, scale_bias->dtype, {DataType::UInt(8)}, operator_name, "scale bias");
57
58 CheckUpscaleMethod(reporter, param->upscale, {"NONE", "ZEROS", "NEAREST"}, operator_name);
59
60 // The scale_bias should be provided as a tensor of size {ofm_channels, 10}
61 reporter->Assign(types[2], TensorType({weight->shape[0], 10}, DataType::UInt(8)));
62
63 // Assign weight type {ofm_channels, kernel_height, kernel_width, ifm_channels}
64 reporter->Assign(types[1], TensorType({param->ofm_channels, param->kernel_shape[0],
65 param->kernel_shape[1], weight->shape[3]},
66 weight->dtype));
67
68 Array<IndexExpr> ifm_shape = ifm->shape;
69 if (param->upscale != "NONE") {
70 ifm_shape = EthosuInferUpscaledInput(ifm_shape, param->ifm_layout);
71 }
72
73 // Assign ofm type
74 auto ofm_shape =
75 EthosuInferKernelOutput(ifm_shape, param->ifm_layout, param->ofm_layout, param->kernel_shape,
76 param->ofm_channels, param->dilation, param->strides, param->padding);
77
78 reporter->Assign(types[4], TensorType(ofm_shape, ifm->dtype));
79 return true;
80}
81
82Expr MakeEthosuConv2D(Expr ifm, Expr weight, Expr scale_bias, Expr lut, double ifm_scale,
83 int ifm_zero_point, int weight_zero_point, double ofm_scale,
84 int ofm_zero_point, Array<IndexExpr> kernel_shape, IndexExpr ofm_channels,
85 Array<IndexExpr> strides, Array<IndexExpr> padding, Array<IndexExpr> dilation,
86 String activation, int clip_min, int clip_max, String rounding_mode,
87 String upscale, String ifm_layout, String ofm_layout) {
88 auto attrs = make_object<EthosuConv2DAttrs>();
89 attrs->ifm_scale = ifm_scale;
90 attrs->ifm_zero_point = ifm_zero_point;
91 attrs->weight_zero_point = weight_zero_point;
92 attrs->ofm_scale = ofm_scale;
93 attrs->ofm_zero_point = ofm_zero_point;
94 attrs->kernel_shape = std::move(kernel_shape);
95 attrs->ofm_channels = std::move(ofm_channels);
96 attrs->strides = std::move(strides);
97 attrs->padding = std::move(padding);
98 attrs->dilation = std::move(dilation);
99 attrs->activation = std::move(activation);
100 attrs->clip_min = clip_min;
101 attrs->clip_max = clip_max;
102 attrs->rounding_mode = std::move(rounding_mode);
103 attrs->upscale = std::move(upscale);
104 attrs->ifm_layout = std::move(ifm_layout);
105 attrs->ofm_layout = std::move(ofm_layout);
106 static const Op& op = Op::Get("contrib.ethosu.conv2d");
107 return Call(op, {ifm, weight, scale_bias, lut}, Attrs(attrs), {});
108}
109
110TVM_REGISTER_GLOBAL("relay.op._make.ethosu_conv2d").set_body_typed(MakeEthosuConv2D);
111
112RELAY_REGISTER_OP("contrib.ethosu.conv2d")
113 .describe(R"code(Arm(R) Ethos(TM)-U NPU 2D quantized convolution operator.
114
115This Relay operator corresponds to the hardware-implemented quantized
116convolution operation found on Ethos(TM)-U NPU. It accepts either NHWC
117or NHCWB16 format for the input data (Input Feature Map, or IFM) and
118OHWI format for the kernel weights.
119
120Reference: https://developer.arm.com/documentation/102420/0200/
121
122Note that the per-channel weight scale and bias tensor must be packed together into
123a combined tensor of uint80s. This is represented in TVM by a (channels, 10) tensor
124of type uint8. For more detail, refer to the Technical Reference Manual linked above.
125
126- **ifm**: NHWC - (1, ifm_height, ifm_width, ifm_channels)
127 NHCWB16 - (1, ifm_height, ifm_channels // 16, ifm_width, 16)
128- **weight**: (ofm_channels, kernel_shape[0], kernel_shape[1], ifm_channels)
129- **scale_bias**: (ofm_channels, 10)
130- **ofm**: (1, ofm_height, ofm_width, ofm_channels)
131
132)code" TVM_ADD_FILELINE)
133 .set_attrs_type<EthosuConv2DAttrs>()
134 .set_num_inputs(4)
135 .add_argument("ifm", "Tensor", "The Input Feature Map tensor (IFM).")
136 .add_argument("weight", "Tensor", "The weight tensor.")
137 .add_argument("scale_bias", "Tensor", "The packed per-channel weight scale and bias tensor.")
138 .add_argument("lut", "Tensor", "The look-up table of values to use if activation = 'LUT'.")
139 .set_support_level(11)
140 .add_type_rel("EthosuConv2D", EthosuConv2DRel);
