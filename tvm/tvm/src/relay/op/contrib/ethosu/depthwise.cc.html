41bool EthosuDepthwiseConv2DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
42 const TypeReporter& reporter) {
43 ICHECK_EQ(types.size(), 5);
44 const auto* ifm = types[0].as<TensorTypeNode>();
45 const auto* weight = types[1].as<TensorTypeNode>();
46 const auto* scale_bias = types[2].as<TensorTypeNode>();
47 if (ifm == nullptr || weight == nullptr) return false;
48
49 const auto* param = attrs.as<EthosuDepthwiseConv2DAttrs>();
50 ICHECK(param != nullptr) << "EthosuDepthwiseConv2DAttrs cannot be nullptr.";
51
52 const String operator_name = "ethosu_depthwise_conv2d";
53
54 CheckDataType(reporter, ifm->dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name, "ifm");
55 CheckDataType(reporter, weight->dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name,
56 "weight");
57 CheckDataType(reporter, scale_bias->dtype, {DataType::UInt(8)}, operator_name, "scale bias");
58
59 DataType ofm_dtype = DataTypeFromString(param->ofm_dtype);
60 auto ofm_dtypes = {DataType::UInt(8), DataType::Int(8), DataType::Int(16), DataType::Int(32)};
61 CheckDataType(reporter, ofm_dtype, ofm_dtypes, operator_name, "ofm");
62
63 // Collect the ifm, weight and ofm tensors for using in the inference function
64 Array<Type> tensor_types = {types[0], types[1], types[4]};
65
66 // Assign weight type {ofm_channels, kernel_height, kernel_width, 1}
67 reporter->Assign(types[1], TensorType({param->ofm_channels, param->kernel_shape[0],
68 param->kernel_shape[1], weight->shape[3]},
69 weight->dtype));
70
71 // Assign ofm type
72 auto ofm_shape =
73 EthosuInferKernelOutput(ifm->shape, param->ifm_layout, param->ofm_layout, param->kernel_shape,
74 param->ofm_channels, param->dilation, param->strides, param->padding);
75
76 reporter->Assign(types[4], TensorType(ofm_shape, ofm_dtype));
77
78 return true;
79}

81Expr MakeEthosuDepthwiseConv2D(Expr ifm, Expr weight, Expr scale_bias, Expr lut, double ifm_scale,
82 int ifm_zero_point, int weight_zero_point, double ofm_scale,
83 int ofm_zero_point, Array<IndexExpr> kernel_shape,
84 IndexExpr ofm_channels, Array<IndexExpr> strides,
85 Array<IndexExpr> padding, Array<IndexExpr> dilation,
86 String activation, int clip_min, int clip_max, String rounding_mode,
87 String upscale, String ifm_layout, String ofm_layout,
88 String ofm_dtype) {
89 auto attrs = make_object<EthosuDepthwiseConv2DAttrs>();
90 attrs->ifm_scale = ifm_scale;
91 attrs->ifm_zero_point = ifm_zero_point;
92 attrs->weight_zero_point = weight_zero_point;
93 attrs->ofm_scale = ofm_scale;
94 attrs->ofm_zero_point = ofm_zero_point;
95 attrs->kernel_shape = std::move(kernel_shape);
96 attrs->ofm_channels = std::move(ofm_channels);
97 attrs->strides = std::move(strides);
98 attrs->padding = std::move(padding);
99 attrs->dilation = std::move(dilation);
100 attrs->activation = std::move(activation);
101 attrs->clip_min = clip_min;
102 attrs->clip_max = clip_max;
103 attrs->rounding_mode = std::move(rounding_mode);
104 attrs->upscale = std::move(upscale);
105 attrs->ifm_layout = std::move(ifm_layout);
106 attrs->ofm_layout = std::move(ofm_layout);
107 attrs->ofm_dtype = std::move(ofm_dtype);
108 static const Op& op = Op::Get("contrib.ethosu.depthwise_conv2d");
109 return Call(op, {ifm, weight, scale_bias, lut}, Attrs(attrs), {});
110}
111
112TVM_REGISTER_GLOBAL("relay.op._make.ethosu_depthwise_conv2d")
113 .set_body_typed(MakeEthosuDepthwiseConv2D);
114
115RELAY_REGISTER_OP("contrib.ethosu.depthwise_conv2d")
116 .describe(R"code(Arm(R) Ethos(TM)-U NPU 2D quantized depthwise operator.
117
118This Relay operator corresponds to the hardware-implemented quantized
119depthwise operation found on Ethos(TM)-U NPU. It accepts either NHWC or NHCWB16 format
120for the input data (input feature map, or IFM) and OHWI format for the kernel weights.
121
122- **ifm**: NHWC - (1, ifm_height, ifm_width, ifm_channels)
123 NHCWB16 - (1, ifm_height, ifm_channels // 16, ifm_width, 16)
124- **weight**: (ofm_channels, kernel_shape[0], kernel_shape[1], 1 (depth multiplier))
125- **scale_bias**: (ofm_channels, 10)
126- **ofm**: (1, ofm_height, ofm_width, ofm_channels)
127
128)code" TVM_ADD_FILELINE)
129 .set_attrs_type<EthosuDepthwiseConv2DAttrs>()
130 .set_num_inputs(4)
131 .add_argument("ifm", "Tensor", "The Input Feature Map tensor (IFM).")
132 .add_argument("weight", "Tensor", "The weight tensor.")
133 .add_argument("scale_bias", "Tensor", "The packed per-channel weight scale and bias tensor.")
134 .add_argument("lut", "Tensor", "The look-up table of values to use if activation = 'LUT'")
135 .set_support_level(11)
136 .add_type_rel("EthosuDepthwiseConv2D", EthosuDepthwiseConv2DRel);
