35bool EthosuIdentityRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
36 const TypeReporter& reporter) {
37 const int ifm_index = 0;
38 const int result_index = 2;
39 ICHECK_EQ(types.size(), result_index + 1);
40
41 const auto* ifm = types[ifm_index].as<TensorTypeNode>();
42 if (ifm == nullptr) return false;
43
44 const auto* param = attrs.as<EthosuIdentityAttrs>();
45 ICHECK(param != nullptr) << "EthosuIdentityAttrs cannot be nullptr.";
46
47 const String operator_name = "ethosu_identity";
48
49 CheckDataType(reporter, ifm->dtype, {DataType::UInt(8), DataType::Int(8)}, operator_name, "ifm");
50
51 if (ifm->shape.size() > 4) {
52 reporter->GetDiagCtx().EmitFatal(
53 Diagnostic::Error(reporter->GetSpan())
54 << "Invalid operator: Input Feature Map should be at most 4 dimensional, but was "
55 << ifm->shape);
56 return false;
57 }
58
59 // Assign ofm type
60 auto ofm_shape = ifm->shape;
61 reporter->Assign(types[result_index], TensorType(ofm_shape, ifm->dtype));
62 return true;
63}
64
65Expr MakeEthosuIdentity(Expr ifm, Expr lut, double ifm_scale, int ifm_zero_point, double ofm_scale,
66 int ofm_zero_point, String activation) {
67 auto attrs = make_object<EthosuIdentityAttrs>();
68 attrs->ifm_scale = ifm_scale;
69 attrs->ifm_zero_point = ifm_zero_point;
70 attrs->ofm_scale = ofm_scale;
71 attrs->ofm_zero_point = ofm_zero_point;
72 attrs->activation = std::move(activation);
73 static const Op& op = Op::Get("contrib.ethosu.identity");
74 return Call(op, {ifm, lut}, Attrs(attrs), {});
75}
76
77TVM_REGISTER_GLOBAL("relay.op._make.ethosu_identity").set_body_typed(MakeEthosuIdentity);
78
79RELAY_REGISTER_OP("contrib.ethosu.identity")
80 .describe(R"code(Arm(R) Ethos(TM)-U NPU identity operator.
81
82This Relay operator performs the identity pooling operation on the NPU with a capability
83to requantize the data. It accepts input tensors of 4 dimensions or less.
84
85)code" TVM_ADD_FILELINE)
86 .set_attrs_type<EthosuIdentityAttrs>()
87 .set_num_inputs(2)
88 .add_argument("ifm", "Tensor", "The Input Feature Map tensor (IFM).")
89 .add_argument("lut", "Tensor", "The look-up table values to use if activation = 'LUT'.")
90 .set_support_level(11)
91 .add_type_rel("EthosuIdentity", EthosuIdentityRel);
