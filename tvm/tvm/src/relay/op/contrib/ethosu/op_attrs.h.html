36/*! \brief Attributes used by the Ethos(TM)-U NPU binary elementwise operators */
37struct EthosuBinaryElementwiseAttrs : public tvm::AttrsNode<EthosuBinaryElementwiseAttrs> {
38 String operator_type;
39 double ifm_scale;
40 int ifm_zero_point;
41 double ifm2_scale;
42 int ifm2_zero_point;
43 double ofm_scale;
44 int ofm_zero_point;
45 IndexExpr ifm_channels;
46 IndexExpr ifm2_channels;
47 bool reversed_operands;
48 String activation;
49 int clip_min;
50 int clip_max;
51 String rounding_mode;
52 String ifm_layout;
53 String ifm2_layout;
54 String ofm_layout;
55 String ofm_dtype;
56
57 TVM_DECLARE_ATTRS(EthosuBinaryElementwiseAttrs, "relay.attrs.EthosuBinaryElementwiseAttrs") {
58 TVM_ATTR_FIELD(operator_type)
59 .describe(
60 "The type of the binary elementwise operator."
61 "'ADD'"
62 "'SUB'"
63 "'MUL'"
64 "'MIN'"
65 "'MAX'"
66 "'SHR'"
67 "'SHL'");
68 TVM_ATTR_FIELD(ifm_scale).describe("The quantization scale for the Input Feature Map tensor.");
69 TVM_ATTR_FIELD(ifm_zero_point)
70 .describe("The quantization zero point for the Input Feature Map tensor.");
71 TVM_ATTR_FIELD(ifm2_scale)
72 .describe("The quantization scale for the Input Feature Map tensor 2.");
73 TVM_ATTR_FIELD(ifm2_zero_point)
74 .describe("The quantization zero point for the Input Feature Map tensor 2.");
75 TVM_ATTR_FIELD(ofm_scale).describe("The quantization scale for the Output Feature Map tensor.");
76 TVM_ATTR_FIELD(ofm_zero_point)
77 .describe("The quantization zero point for the Output Feature Map tensor.");
78 TVM_ATTR_FIELD(ifm_channels).describe("The number of the Input Feature Map channels.");
79 TVM_ATTR_FIELD(ifm2_channels).describe("The number of the Input Feature Map 2 channels.");
80 TVM_ATTR_FIELD(reversed_operands)
81 .describe("True if IFM2 is the first operand and IFM is the second operand.")
82 .set_default(false);
83 TVM_ATTR_FIELD(activation)
84 .describe(
85 "The activation function to use. "
86 "'NONE' - no activation function. "
87 "'CLIP' - clip the output between clip_min and clip_max. "
88 "'TANH' - tanh activation function. "
89 "'SIGMOID' - sigmoid activation function. "
90 "'LUT' - use a look-up table to perform the activation function."
91 "Available activations for activation type:"
92 "{int8, uint8}: 'NONE', 'CLIP', 'TANH', 'SIGMOID', 'LUT'"
93 "{int32}: 'NONE'")
94 .set_default("NONE");
95 TVM_ATTR_FIELD(clip_min)
96 .describe("The minimum clipping value if activation = 'CLIP'.")
97 .set_default(0);
98 TVM_ATTR_FIELD(clip_max)
99 .describe("The maximum clipping value if activation = 'CLIP'.")
100 .set_default(0);
101 TVM_ATTR_FIELD(rounding_mode)
102 .describe(
103 "The rounding mode to apply to the Output Feature Map tensor. "
104 "'TFL' - Tensorflow Lite rounding scheme. "
105 "'TRUNCATE' - Truncate towards zero."
106 "'NATURAL' - Round to nearest value, with x.5 rounded up towards +infinity.")
107 .set_default("TFL");
108 TVM_ATTR_FIELD(ifm_layout)
109 .describe("The layout of the Input Feature Map tensor. Can be 'NHWC' or 'NHCWB16'.")
110 .set_default("NHWC");
111 TVM_ATTR_FIELD(ifm2_layout)
112 .describe("The layout of the Input Feature Map tensor 2. Can be 'NHWC' or 'NHCWB16'.")
113 .set_default("NHWC");
114 TVM_ATTR_FIELD(ofm_layout)
115 .describe("The layout of the Output Feature Map tensor. Can be 'NHWC' or 'NHCWB16'.")
116 .set_default("NHWC");
117 TVM_ATTR_FIELD(ofm_dtype).describe(
118 "The Output Feature Map tensor type."
119 "MUL, ADD, SUB {IFM}->{OFM}:"
120 " {uint8, int8 int32} -> {uint8, int8, int32}, any pairing"
121 "MAX, MIN:"
122 " IFM and OFM must be of the same type, one of:"
123 " {int8, uint8}"
124 "SHR {IFM}->{OFM}:"
125 " {int32}->{int8, uint8, int32}, any pairing"
126 "SHL:"
127 " {int32}->{int32} only");
128 }
129};
130
131TVM_REGISTER_NODE_TYPE(EthosuBinaryElementwiseAttrs);

133/*! \brief Attributes used by the Ethos(TM)-U NPU convolution operator */
134struct EthosuConv2DAttrs : public tvm::AttrsNode<EthosuConv2DAttrs> {

139 int ofm_zero_point;

141 IndexExpr ofm_channels;
142 Array<IndexExpr> strides;
143 Array<IndexExpr> padding;
144 Array<IndexExpr> dilation;
145 String activation;
146 int clip_min;
147 int clip_max;
148 String rounding_mode;
149 String upscale;
150 String ifm_layout;
151 String ofm_layout;
152
153 TVM_DECLARE_ATTRS(EthosuConv2DAttrs, "relay.attrs.EthosuConv2DAttrs") {
154 TVM_ATTR_FIELD(ifm_scale).describe("The quantization scale for the Input Feature Map tensor.");
155 TVM_ATTR_FIELD(ifm_zero_point)
156 .describe("The quantization zero point for the Input Feature Map tensor.");
157 TVM_ATTR_FIELD(weight_zero_point)
158 .describe("The quantization zero point for the weight tensor.");
159 TVM_ATTR_FIELD(ofm_scale).describe("The quantization scale for the Output Feature Map tensor.");
160 TVM_ATTR_FIELD(ofm_zero_point)
161 .describe("The quantization zero point for the Output Feature Map tensor.");
162 TVM_ATTR_FIELD(kernel_shape)
163 .describe("The 2 dimensional kernel shape as (kernel_height, kernel_width).")
164 .set_default(NullValue<Array<IndexExpr>>());
165 TVM_ATTR_FIELD(ofm_channels)
166 .describe("The number of the Output Feature Map channels.")
167 .set_default(NullValue<IndexExpr>());
168 TVM_ATTR_FIELD(strides)
169 .set_default(Array<IndexExpr>({1, 1}))
170 .describe("The 2 dimensional strides as (stride_height, stride_width).");
171 TVM_ATTR_FIELD(padding)
172 .set_default(Array<IndexExpr>({0, 0, 0, 0}))
173 .describe("The 4 dimensional padding as (pad_top, pad_left, pad_bottom, pad_right).");
174 TVM_ATTR_FIELD(dilation)
175 .set_default(Array<IndexExpr>({1, 1}))
176 .describe("The 2 dimensional dilation as (dilation_height, dilation_width).");
177 TVM_ATTR_FIELD(activation)
178 .describe(
179 "The activation function to use. "
180 "'NONE' - no activation function. "
181 "'CLIP' - clip the output between clip_min and clip_max. "
182 "'TANH' - tanh activation function. "
183 "'SIGMOID' - sigmoid activation function. "
184 "'LUT' - use a look-up table to perform the activation function.")
185 .set_default("NONE");
186 TVM_ATTR_FIELD(clip_min)
187 .describe("The minimum clipping value if activation = 'CLIP'.")
188 .set_default(0);
189 TVM_ATTR_FIELD(clip_max)
190 .describe("The maximum clipping value if activation = 'CLIP'.")
191 .set_default(0);
192 TVM_ATTR_FIELD(rounding_mode)
193 .describe(
194 "The rounding mode to apply to the Output Feature Map tensor. "
195 "'TFL' - Tensorflow Lite rounding scheme. "
196 "'TRUNCATE' - Truncate towards zero."
197 "'NATURAL' - Round to nearest value, with x.5 rounded up towards +infinity.")
198 .set_default("TFL");
199 TVM_ATTR_FIELD(upscale)
200 .describe(
201 "The 2x2 upscaling mode to apply to the Input Feature Map tensor. "
202 "'NONE' - no upscaling. "
203 "'NEAREST' - upscale using nearest neighbour. "
204 "'ZEROS' - upscale using zeros.")
205 .set_default("NONE");
206 TVM_ATTR_FIELD(ifm_layout)
207 .set_default("NHWC")
208 .describe("The layout of the Input Feature Map tensor. Can be 'NHWC' or 'NHCWB16'.");
209 TVM_ATTR_FIELD(ofm_layout)
210 .set_default("NHWC")
211 .describe("The layout of the Output Feature Map tensor. Can be 'NHWC' or 'NHCWB16'.");
212 }
213};
