34template <typename T>
35InferCorrectLayoutOutput ResizeInferCorrectLayout(const Attrs& attrs,
36 const Array<Layout>& new_in_layouts,
37 const Array<Layout>& old_in_layouts,
38 const Array<tvm::relay::Type>& old_in_types) {
39 const auto* attrs_ptr = attrs.as<T>();
40 CHECK(attrs_ptr);
41 ObjectPtr<T> params = make_object<T>(*attrs_ptr);
42
43 if (new_in_layouts.defined()) {
44 ICHECK_EQ(new_in_layouts.size(), 1);
45
46 Layout raw_layout(params->layout);
47 Layout new_layout = new_in_layouts[0];
48 Layout old_layout = old_in_layouts[0];
49 if (!new_layout.Equals(old_layout) && raw_layout.Equals(old_layout) &&
50 new_layout->axes.size() == old_layout->axes.size()) {
51 // Follow input layout
52 params->layout = new_layout.name();
53 }
54 }
55
56 return InferCorrectLayoutOutput({params->layout}, {params->layout}, Attrs(params));
57}

61bool Resize1DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
62 const TypeReporter& reporter) {
63 ICHECK_EQ(types.size(), 2);
64 const auto* data = types[0].as<TensorTypeNode>();
65 if (data == nullptr) return false;
66
67 static const Layout kNCW("NCW");
68
69 const Resize1DAttrs* param = attrs.as<Resize1DAttrs>();
70 ICHECK(param != nullptr);
71 ICHECK(param->size.size() == 1);
72 ICHECK(param->roi.size() == 2);
73 const Layout in_layout(param->layout);
74 auto layout_converter = tir::BijectiveLayout(in_layout, kNCW);
75 ICHECK(layout_converter.defined())
76 << "Resize only support input layouts that are convertible from NCW."
77 << " But got " << in_layout;
78
79 auto oshape = layout_converter.ForwardShape(data->shape);
80 oshape.Set(2, param->size[0]);
81
82 DataType out_dtype = param->out_dtype;
83 if (out_dtype.bits() == 0) {
84 out_dtype = data->dtype;
85 }
86
87 // assign output type
88 reporter->Assign(types[1], TensorType(layout_converter.BackwardShape(oshape), out_dtype));
89 return true;
90}

139bool Resize2DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
140 const TypeReporter& reporter) {

146

150 ICHECK(param->roi.size() == 4);

158 oshape.Set(2, param->size[0]);

161 DataType out_dtype = param->out_dtype;
162 if (out_dtype.bits() == 0) {
163 out_dtype = data->dtype;
164 }
165
166 // assign output type
167 reporter->Assign(types[1], TensorType(layout_converter.BackwardShape(oshape), out_dtype));

218bool Resize3DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
219 const TypeReporter& reporter) {
220 ICHECK_EQ(types.size(), 2);
221 const auto* data = types[0].as<TensorTypeNode>();
222 if (data == nullptr) return false;
223
224 static const Layout kNCDHW("NCDHW");
225
226 const Resize3DAttrs* param = attrs.as<Resize3DAttrs>();
227 ICHECK(param != nullptr);
228 ICHECK(param->size.size() == 3);
229 ICHECK(param->roi.size() == 6);
230 const Layout in_layout(param->layout);
231 auto layout_converter = tir::BijectiveLayout(in_layout, kNCDHW);
232 ICHECK(layout_converter.defined())
233 << "Resize3d only support input layouts that are convertible from NCDHW."
234 << " But got " << in_layout;
235
236 auto oshape = layout_converter.ForwardShape(data->shape);
237 oshape.Set(2, param->size[0]);
238 oshape.Set(3, param->size[1]);
239 oshape.Set(4, param->size[2]);
240
241 DataType out_dtype = param->out_dtype;
242 if (out_dtype.bits() == 0) {
243 out_dtype = data->dtype;
244 }
245
246 // assign output type
247 reporter->Assign(types[1], TensorType(layout_converter.BackwardShape(oshape), out_dtype));
248 return true;
249}

297
298bool CropAndResizeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
299 const TypeReporter& reporter) {
300 ICHECK_EQ(types.size(), 4);
301 const auto* data = types[0].as<TensorTypeNode>();
302 const auto* boxes = types[1].as<TensorTypeNode>();
303 const auto* box_indices = types[2].as<TensorTypeNode>();
304 if (data == nullptr || boxes == nullptr || box_indices == nullptr) return false;
305
306 const CropAndResizeAttrs* param = attrs.as<CropAndResizeAttrs>();
307 ICHECK(param != nullptr);
308 auto crop_size = param->crop_size;
309
310 DataType out_dtype = param->out_dtype;
311 if (out_dtype.bits() == 0) {
312 out_dtype = data->dtype;
313 }
314
315 // 4-D tensor of shape [num_boxes, crop_height, crop_width, depth]
316 static const Layout kNCHW("NCHW");
317 const Layout in_layout(param->layout);
318 auto layout_converter = tir::BijectiveLayout(in_layout, kNCHW);
319 auto oshape = layout_converter.ForwardShape(data->shape);
320 oshape.Set(0, boxes->shape[0]);
321 oshape.Set(2, crop_size[0]);
322 oshape.Set(3, crop_size[1]);
323 auto bshape = layout_converter.BackwardShape(oshape);
324 // assign output type
325 reporter->Assign(types[3], TensorType(bshape, out_dtype));
326 return true;
327}
