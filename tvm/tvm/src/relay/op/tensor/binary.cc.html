34#define RELAY_BINARY_COMPUTE(FTOPI) \
35 [](const Attrs& attrs, const Array<te::Tensor>& inputs, \
36 const Type& out_type) -> Array<te::Tensor> { \
37 ICHECK_EQ(inputs.size(), 2U); \
38 return {FTOPI(inputs[0], inputs[1])}; \
39 }
