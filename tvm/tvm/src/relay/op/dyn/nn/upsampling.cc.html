41bool UpSamplingRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
42 const TypeReporter& reporter) {
43 // types = [data_type, scale_h_type, scale_w_type, ret_type]
44 ICHECK_EQ(types.size(), 4);
45 const auto* data = types[0].as<TensorTypeNode>();
46 const auto* scale_h = types[1].as<TensorTypeNode>();
47 const auto* scale_w = types[2].as<TensorTypeNode>();
48 if (data == nullptr) return false;
49 if (scale_h == nullptr) return false;
50 if (scale_w == nullptr) return false;
51
52 ICHECK_EQ(scale_h->shape.size(), 0);
53 ICHECK_EQ(scale_w->shape.size(), 0);
54 static const Layout kNCHW("NCHW");
55
56 const UpSamplingAttrs* param = attrs.as<UpSamplingAttrs>();
57 ICHECK(param);
58 const Layout in_layout(param->layout);
59
60 auto layout_converter = tir::BijectiveLayout(in_layout, kNCHW);
61 ICHECK(layout_converter.defined())
62 << "UpSampling only supports input layouts that are convertible from NCHW."
63 << " But got " << in_layout;
64
65 auto nchw_oshape = layout_converter.ForwardShape(data->shape);
66
67 nchw_oshape.Set(2, Any());
68 nchw_oshape.Set(3, Any());
69 auto oshape = layout_converter.BackwardShape(nchw_oshape);
70
71 reporter->Assign(types[3], TensorType(oshape, data->dtype));
72 return true;
73}

75// Positional relay function to create upsampling operator
76// used by frontend FFI.
77Expr MakeUpSampling(Expr data, Expr scale_h, Expr scale_w, String layout, String method,
78 bool align_corners) {
79 auto attrs = make_object<UpSamplingAttrs>();
80 attrs->layout = std::move(layout);
81 attrs->method = std::move(method);
82 attrs->align_corners = align_corners;
83
84 static const Op& op = Op::Get("dyn.nn.upsampling");
85 return Call(op, {data, scale_h, scale_w}, Attrs(attrs), {});
86}
87
88TVM_REGISTER_GLOBAL("relay.op.dyn.nn._make.upsampling").set_body_typed(MakeUpSampling);
89
90RELAY_REGISTER_OP("dyn.nn.upsampling")
91 .describe(
92 R"code(Perform upsampling on input array with nearest neighbour or bilinear interpolation.
93
94- **data**: data is 4D array of shape
95 (batch_size, channels, in_height, in_width) for NCHW
96 (batch_size, in_height, in_width, channels) for NHWC
97
98- **scale_h**: scale_h is a double of the amount to scale height by
99
100- **scale_w**: scale_w is a double of the amount to scale width by
101
102- **out**: Output is 4D array of shape
103 for layout NCHW
104 (batch_size, channels, in_height*scale_h, in_width*scale_w)
105
106 for layout NHWC
107 (batch_size, in_height*scale_h, in_width*scale_w, channels)
108
109)code" TVM_ADD_FILELINE)
110 .set_attrs_type<UpSamplingAttrs>()
111 .set_num_inputs(3)
112 .add_argument("data", "Tensor", "The input tensor.")
113 .add_argument("scale_h", "double", "The scale for the height.")
114 .add_argument("scale_w", "double", "The scale for the width.")
115 .set_support_level(2)
116 .add_type_rel("DynamicUpSampling", UpSamplingRel)
117 .set_attr<FInferCorrectLayout>("FInferCorrectLayout",
118 UpsamplingInferCorrectLayout<UpSamplingAttrs>)
119 .set_attr<TOpPattern>("TOpPattern", kInjective);

121// UpSampling3D
122bool UpSampling3DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
123 const TypeReporter& reporter) {
124 // types = [data_type, scale_d_type, scale_h_type, scale_w_type, ret_type]
125 ICHECK_EQ(types.size(), 5);
126 const auto* data = types[0].as<TensorTypeNode>();
127 if (data == nullptr) return false;
128
129 static const Layout kNCDHW("NCDHW");
130
131 const UpSampling3DAttrs* param = attrs.as<UpSampling3DAttrs>();
132 ICHECK(param != nullptr);
133 const Layout in_layout(param->layout);
134
135 auto layout_converter = tir::BijectiveLayout(in_layout, kNCDHW);
136 ICHECK(layout_converter.defined())
137 << "UpSampling3D only support input layouts that are convertible from NCDHW."
138 << " But got " << in_layout;
139
140 auto ncdhw_oshape = layout_converter.ForwardShape(data->shape);
141
142 ncdhw_oshape.Set(2, Any());
143 ncdhw_oshape.Set(3, Any());
144 ncdhw_oshape.Set(4, Any());
145
146 auto oshape = layout_converter.BackwardShape(ncdhw_oshape);
147
148 reporter->Assign(types[4], TensorType(oshape, data->dtype));
149 return true;
150}

152Expr MakeUpSampling3D(Expr data, Expr scale_d, Expr scale_h, Expr scale_w, String layout,
153 String method, String coordinate_transformation_mode) {
154 auto attrs = make_object<UpSampling3DAttrs>();
155 attrs->layout = std::move(layout);
156 attrs->method = std::move(method);
157 attrs->coordinate_transformation_mode = coordinate_transformation_mode;
158
159 static const Op& op = Op::Get("dyn.nn.upsampling3d");
160 return Call(op, {data, scale_d, scale_h, scale_w}, Attrs(attrs), {});
161}
162
163TVM_REGISTER_GLOBAL("relay.op.dyn.nn._make.upsampling3d").set_body_typed(MakeUpSampling3D);
164
165RELAY_REGISTER_OP("dyn.nn.upsampling3d")
166 .describe(R"code(Perform upsampling on input array with nearest neighbour or
167bilinear interpolation.
168
169- **data**: data is 5D array of shape
170 (batch_size, channels, in_depth, in_height, in_width) for NCDHW
171 (batch_size, in_depth, in_height, in_width, channels) for NDHWC
172
173- **scale_d**: scale_d is a double of the amount to scale depth by
174
175- **scale_h**: scale_h is a double of the amount to scale height by
176
177- **scale_w**: scale_w is a double of the amount to scale width by
178
179- **out**: Output is 5D array of shape
180 for layout NCDHW
181 (batch_size, channels, in_depth*scale_d, in_height*scale_h, in_width*scale_w)
182
183 for layout NDHWC
184 (batch_size, in_depth*scale_d, in_height*scale_h, in_width*scale_w, channels)
185
186)code" TVM_ADD_FILELINE)
187 .set_attrs_type<UpSampling3DAttrs>()
188 .set_num_inputs(4)
189 .add_argument("data", "Tensor", "The input tensor.")
190 .add_argument("scale_d", "double", "The scale for the depth.")
191 .add_argument("scale_h", "double", "The scale for the height.")
192 .add_argument("scale_w", "double", "The scale for the width.")
193 .set_support_level(2)
194 .add_type_rel("DynamicUpSampling3D", UpSampling3DRel)
195 .set_attr<FInferCorrectLayout>("FInferCorrectLayout",
196 UpsamplingInferCorrectLayout<UpSampling3DAttrs>)
197 .set_attr<TOpPattern>("TOpPattern", kInjective);
