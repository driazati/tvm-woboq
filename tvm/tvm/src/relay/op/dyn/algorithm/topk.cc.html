32bool TopKRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
33 const TypeReporter& reporter) {
34 // `types` contains: [data, k, result]
35 const TopKAttrs* param = attrs.as<TopKAttrs>();
36 ICHECK_EQ(types.size(), 3);
37 const auto* data = types[0].as<TensorTypeNode>();
38 const auto* k = types[1].as<TensorTypeNode>();
39 if (data == nullptr) {
40 ICHECK(types[0].as<IncompleteTypeNode>())
41 << "tile: expect input type to be TensorType but get " << types[0];
42 return false;
43 }
44 if (k == nullptr) {
45 ICHECK(types[1].as<IncompleteTypeNode>())
46 << "tile: expect input type to be TensorType but get " << types[1];
47 return false;
48 }
49 ICHECK(k->shape.size() <= 1) << "Parameter k must be a Scalar or a Tensor of shape (1, )";
50 if (k->shape.size() == 1) {
51 const IntImmNode* k_shape = k->shape[0].as<IntImmNode>();
52 ICHECK(k_shape) << "Parameter k must have static shape";
53 ICHECK_EQ(k_shape->value, 1) << "Parameter k must be a Scalar or a Tensor of shape (1, )";
54 }
55 int ndim = data->shape.size();
56 int axis = param->axis;
57 if (axis < 0) {
58 axis += ndim;
59 }
60 ICHECK(axis >= 0 && axis < ndim);
61 Array<IndexExpr> out_shape;
62 for (int i = 0; i < ndim; ++i) {
63 if (i != axis) {
64 out_shape.push_back(data->shape[i]);
65 } else {
66 out_shape.push_back(Any());
67 }
68 }
69 auto values_ty = TensorType(out_shape, data->dtype);
70 auto indices_ty = TensorType(out_shape, param->dtype);
71 if (param->ret_type == "both") {
72 reporter->Assign(types[2], TupleType({values_ty, indices_ty}));
73 } else if (param->ret_type == "values") {
74 reporter->Assign(types[2], values_ty);
75 } else if (param->ret_type == "indices") {
76 reporter->Assign(types[2], indices_ty);
77 } else {
78 LOG(FATAL) << "Unsupported ret type: " << param->ret_type;
79 }
80 return true;
81}
82
83Expr MakeTopK(Expr data, Expr k, int axis, String ret_type, bool is_ascend, DataType dtype) {
84 auto attrs = make_object<TopKAttrs>();
85 attrs->axis = axis;
86 attrs->ret_type = ret_type;
87 attrs->is_ascend = is_ascend;
88 attrs->dtype = dtype;
89 static const Op& op = Op::Get("dyn.topk");
90 return Call(op, {data, k}, Attrs(attrs), {});
91}
92
93TVM_REGISTER_GLOBAL("relay.op.dyn._make.topk").set_body_typed(MakeTopK);
94
95RELAY_REGISTER_OP("dyn.topk")
96 .describe(R"doc(Get the top k elements in an input tensor along the given axis.
97)doc" TVM_ADD_FILELINE)
98 .set_num_inputs(2)
99 .set_attrs_type<TopKAttrs>()
100 .add_argument("data", "Tensor", "Input data.")
101 .add_argument("k", "Tensor", "Number of top elements.")
102 .set_support_level(6)
103 .add_type_rel("DynTopK", TopKRel);
