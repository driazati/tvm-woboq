47bool ReshapeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
48 const TypeReporter& reporter) {
49 // types: [data, newshape, result]
50 ICHECK_EQ(types.size(), 3);
51
52 const auto* data = types[0].as<TensorTypeNode>();
53 if (data == nullptr) {
54 ICHECK(types[0].as<IncompleteTypeNode>())
55 << "reshape: expect input type to be TensorType but get " << types[0];
56 return false;
57 }
58
59 Array<IndexExpr> oshape;
60 const auto* newshape = types[1].as<TensorTypeNode>();
61 if (newshape == nullptr) {
62 ICHECK(types[1].as<IncompleteTypeNode>())
63 << "reshape: expect input type to be TensorType but get " << types[1];
64 return false;
65 }
66
67 const IntImmNode* rank = newshape->shape[0].as<IntImmNode>();
68 ICHECK(rank != nullptr) << "Dynamic Reshape doesn't support Dynamic Rank";
69 for (int i = 0; i < rank->value; i++) {
70 oshape.push_back(Any());
71 }
72
73 reporter->Assign(types[2], TensorType(oshape, data->dtype));
74 return true;
75}

77Array<te::Tensor> ReshapeCompute(const Attrs& attrs, const Array<te::Tensor>& inputs,
78 const Type& out_type) {
79 const auto* out_ttype = out_type.as<TensorTypeNode>();
80 ICHECK(out_ttype != nullptr);
81 Array<IndexExpr> newshape;
82 for (auto val : out_ttype->shape) {
83 if (val->IsInstance<tir::AnyNode>()) {
84 newshape.push_back(val.as<tir::AnyNode>()->ToVar());
85 } else {
86 newshape.push_back(val);
87 }
88 }
89 return {topi::reshape(inputs[0], newshape)};
90}

101RELAY_REGISTER_OP("dyn.reshape")
102 .describe(R"code(Reshapes the input array based on the values in the newshape array.
103
104 To give user more convenience in without doing manual shape inference,
105 some dimensions of the shape can take special values from the set {0, -1, -3}.
106 The significance of each is explained below:
107
108 ``0`` copy this dimension from the input to the output shape.
109
110 .. code-block:: python
111
112 data.shape = (2,3,4), newshape = (4,0,2), result.shape = (4,3,2)
113 data.shape = (2,3,4), newshape = (2,0,0), result.shape = (2,3,4)
114
115 ``-1`` infers the dimension of the output shape by using the remainder of
116 the input dimensions keeping the size of the new array same as that of the input array.
117 At most one dimension of shape can be -1.
118
119 .. code-block:: python
120
121 data.shape = (2,3,4), newshape = (6,1,-1), result.shape = (6,1,4)
122 data.shape = (2,3,4), newshape = (3,-1,8), result.shape = (3,1,8)
123 data.shape = (2,3,4), newshape = (-1,), result.shape = (24,)
124
125 ``-3`` use the product of two consecutive dimensions of the input shape
126 as the output dimension.
127
128 .. code-block:: python
129
130 data.shape = (2,3,4), newshape = (-3,4), result.shape = (6,4)
131 data.shape = (2,3,4,5), newshape = (-3,-3), result.shape = (6,20)
132 data.shape = (2,3,4), newshape = (0,-3), result.shape = (2,12)
133
134 Special values -2 and -4 from the standard reshape op would introduce dynamic rank
135 in this op. Thus, they are not permitted.
136
137 )code" TVM_ADD_FILELINE)
138 .set_num_inputs(2)
139 .set_attrs_type<ReshapeAttrs>()
140 .add_argument("data", "Tensor", "The input tensor.")
141 .add_argument("newshape", "Tensor", "The shape of output tensor.")
142 .set_support_level(3)
143 .add_type_rel("DynamicReshape", ReshapeRel)
144 .set_attr<FTVMCompute>("FTVMCompute", ReshapeCompute)
145 .set_attr<TOpPattern>("TOpPattern", kInjective)
146 .set_attr<TReshapeOp>("TReshapeOp", true);

151bool TileRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
152 const TypeReporter& reporter) {
153 // `types` contains: [data, reps, result]
154 ICHECK_EQ(types.size(), 3);
155 const auto* data = types[0].as<TensorTypeNode>();
156 const auto* reps = types[1].as<TensorTypeNode>();
157 if (data == nullptr) {
158 ICHECK(types[0].as<IncompleteTypeNode>())
159 << "tile: expect input type to be TensorType but get " << types[0];
160 return false;
161 }
162 if (reps == nullptr) {
163 ICHECK(types[1].as<IncompleteTypeNode>())
164 << "tile: expect input type to be TensorType but get " << types[1];
165 return false;
166 }
167 const IntImmNode* reps_shape = reps->shape[0].as<IntImmNode>();
168 ICHECK(reps_shape) << "Parameter reps must have static shape";
169 const size_t ndim = data->shape.size();
170 const size_t rndim = reps_shape->value;
171 size_t tndim = (ndim > rndim) ? ndim : rndim;
172 std::vector<IndexExpr> oshape;
173 oshape.reserve(tndim);
174 for (size_t i = 0; i < tndim; ++i) {
175 oshape.emplace_back(Any());
176 }
177 reporter->Assign(types[2], TensorType(oshape, data->dtype));
178 return true;
179}

214bool BroadCastToRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
215 const TypeReporter& reporter) {
216 // types = [data_type, broadcast_shape_type, ret_type]
217 ICHECK_EQ(types.size(), 3);
218
219 const auto* input_type = types[0].as<TensorTypeNode>();
220 const auto* target_type = types[1].as<TensorTypeNode>();
221 if (target_type == nullptr) {
222 return false;
223 }
224 if (input_type == nullptr) {
225 return false;
226 }
227 auto out_dtype = input_type->dtype;
228 // rank must be static
229 const IntImmNode* rank = target_type->shape[0].as<IntImmNode>();
230 ICHECK(rank)
231 << "Target shape must have static rank"; // rank must be static even in dyn pass
232 // could add support for dyn rank in futures
233
234 std::vector<IndexExpr> oshape;
235 for (int i = 0; i < rank->value; ++i) {
236 oshape.push_back(Any());
237 }
238
239 reporter->Assign(types[2], TensorType(oshape, out_dtype));
240 return true;
241}

328bool OneHotRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
329 const TypeReporter& reporter) {
330 // `types` contains: [indices, on_value, off_value, result]
331 ICHECK_EQ(types.size(), 5);
332 const auto* indices = types[0].as<TensorTypeNode>();
333 ICHECK(indices);
334
335 const auto param = attrs.as<OneHotAttrs>();
336
337 Array<IndexExpr> oshape;
338 int ndim = indices->shape.size() + 1;
339 int indices_index = 0;
340 int true_axis = (param->axis == -1) ? indices->shape.size() : param->axis;
341 for (int i = 0; i < ndim; i++) {
342 if (i == true_axis) {
343 oshape.push_back(Any());
344 } else {
345 oshape.push_back(indices->shape[indices_index++]);
346 }
347 }
348
349 reporter->Assign(types[4], TensorType(oshape, param->dtype));
350 return true;
351}

398bool FullRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
399 const TypeReporter& reporter) {
400 ICHECK_EQ(types.size(), 3);
401 const InitOpAttrs* param = attrs.as<InitOpAttrs>();
402 const auto* fill_value = types[0].as<TensorTypeNode>();
403 const auto* fill_shape = types[1].as<TensorTypeNode>();
404 if (fill_value == nullptr) {
405 return false;
406 }
407 if (fill_shape == nullptr) {
408 return false;
409 }
410
411 DataType out_dtype = param->dtype;
412 if (out_dtype.bits() == 0) {
413 out_dtype = fill_value->dtype;
414 }
415
416 ICHECK_EQ(fill_value->shape.size(), 0)
417 << "Fill value should be a scalar but has dimension " << fill_value->shape.size() << ".";
418
419 const IntImmNode* rank = fill_shape->shape[0].as<IntImmNode>();
420 ICHECK(rank) << "Parameter shape must have static rank";
421
422 std::vector<IndexExpr> oshape;
423 for (int i = 0; i < rank->value; ++i) {
424 oshape.push_back(Any());
425 }
426 reporter->Assign(types[2], TensorType(oshape, out_dtype));
427 return true;
428}

456bool StridedSliceRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
457 const TypeReporter& reporter) {
458 // [data, begin, end, strides, out]
459 ICHECK_EQ(types.size(), 5);
460 const StridedSliceAttrs* param = attrs.as<StridedSliceAttrs>();
461 if (param == nullptr) {
462 return false;
463 }
464 const auto* data = types[0].as<TensorTypeNode>();
465 if (data == nullptr) {
466 return false;
467 }
468 auto dshape = data->shape;
469 int64_t num_axis = dshape.size();
470
471 const auto* begin = types[1].as<TensorTypeNode>();
472 if (begin == nullptr) {
473 return false;
474 }
475 ICHECK(begin);
476
477 // calculate output shape
478 std::vector<IndexExpr> oshape(num_axis);
479 int64_t num_dynamic_axes = begin->shape[0].as<IntImmNode>()->value;
480 for (int64_t i = 0; i < num_dynamic_axes; ++i) {
481 oshape[i] = Any();
482 }
483
484 for (int64_t i = num_dynamic_axes; i < num_axis; ++i) {
485 oshape[i] = dshape[i];
486 }
487
488 reporter->Assign(types[4], TensorType(oshape, data->dtype));
489 return true;
490}

492Array<te::Tensor> StridedSliceCompute(const Attrs& attrs, const Array<te::Tensor>& inputs,
493 const Type& out_type) {
494 te::Tensor data = inputs[0];
495 te::Tensor begin = inputs[1];
496 te::Tensor end = inputs[2];
497 te::Tensor strides = inputs[3];
498 // Dynamic computation
499 int64_t data_rank = data->shape.size();
500 int64_t num_dynamic_axes = begin->shape[0].as<IntImmNode>()->value;
501 ICHECK(end->shape[0].as<IntImmNode>()->value == num_dynamic_axes &&
502 strides->shape[0].as<IntImmNode>()->value == num_dynamic_axes)
503 << "begin, end, strides should have the same length if they are dynamic variables";
504 ICHECK(num_dynamic_axes <= data_rank)
505 << "the number of dynamic axes to slice should be less than or equal to the data rank";
506 return Array<te::Tensor>{topi::dynamic_strided_slice(data, begin, end, strides)};
507}

555bool SparseToDenseRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
556 const TypeReporter& reporter) {
557 ICHECK_EQ(num_inputs, 4);
558 auto sparse_indices = types[0].as<TensorTypeNode>();
559 auto sparse_values = types[1].as<TensorTypeNode>();
560 auto default_value = types[2].as<TensorTypeNode>();
561 auto output_shape = types[3].as<TensorTypeNode>();
562
563 if (sparse_indices == nullptr || sparse_values == nullptr || default_value == nullptr ||
564 output_shape == nullptr) {
565 return false;
566 }
567
568 CHECK(sparse_indices->dtype.is_int()) << "sparse_indices must be tensor of integers";
569
570 CHECK_LE(sparse_indices->shape.size(), 3)
571 << "sparse_indices must be a tensor of either 0D, 1D or 2D";
572
573 CHECK_LE(sparse_values->shape.size(), 2) << "sparse_values must be a tensor of either 0D, 1D";
574
575 CHECK_EQ(default_value->shape.size(), 0) << "default_value should be a scalar";
576
577 Array<IndexExpr> oshape;
578 for (int i = 0; i < output_shape->shape[0].as<IntImmNode>()->value; i++) {
579 oshape.push_back(Any());
580 }
581 reporter->Assign(types[4], TensorType(oshape, sparse_values->dtype));
582 return true;
583}

629bool ExpandDimsRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
630 const TypeReporter& reporter) {
631 ICHECK_EQ(num_inputs, 2);
632 const auto* data_type = types[0].as<TensorTypeNode>();
633 if (data_type == nullptr) {
634 ICHECK(types[0].as<IncompleteTypeNode>())
635 << "expand_dims: expect input type to be TensorType but get " << types[0];
636 return false;
637 }
638
639 const auto* param = attrs.as<DynExpandDimsAttrs>();
640
641 // We don't know the output shape until we see the value of the axis input
642 int ndim = data_type->shape.size();
643 Array<IndexExpr> oshape(ndim + param->num_newaxis, Any());
644
645 const auto* axis_type = types[1].as<TensorTypeNode>();
646 ICHECK(axis_type->shape.size() == 0) << "Axis should be a scalar got shape " << axis_type->shape;
647
648 // Set output shape
649 reporter->Assign(types[2], TensorType(oshape, data_type->dtype));
650 return true;
651}
652
653Array<te::Tensor> ExpandDimsCompute(const Attrs& attrs, const Array<te::Tensor>& inputs,
654 const Type& out_type) {
655 // inputs = [Input tensor, axis to expand]
656 ICHECK_EQ(inputs.size(), 2);
657
658 const auto* param = attrs.as<DynExpandDimsAttrs>();
659
660 Array<IndexExpr> ishape = inputs[0]->shape;
661 const TensorTypeNode* out_ttype = out_type.as<TensorTypeNode>();
662 int ndim_out = out_ttype->shape.size();
663 int ndim_in = ishape.size();
664 ICHECK_EQ(ndim_in + param->num_newaxis, ndim_out);
665
666 Array<IndexExpr> newshape;
667 for (auto val : out_ttype->shape) {
668 // These vars will be populated by the VM executor with the results
669 // of the shape_func for the op.
670 newshape.push_back(val.as<tir::AnyNode>()->ToVar());
671 }
672
673 return {topi::reshape(inputs[0], newshape)};
674}

700bool DynSqueezeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
701 const TypeReporter& reporter) {
702 // [data, axes, output]
703 ICHECK_EQ(types.size(), 3);
704 const auto* data = types[0].as<TensorTypeNode>();
705 if (data == nullptr) {
706 return false;
707 }
708 const auto* axes = types[1].as<TensorTypeNode>();
709 if (axes == nullptr) {
710 return false;
711 }
712
713 ICHECK_EQ(axes->shape.size(), 1) << "Got" << axes->shape.size() << "expected 1";
714 ICHECK(axes->shape[0].as<IntImmNode>()) << "axes expected to be static rank";
715 size_t output_rank = data->shape.size() - axes->shape[0].as<IntImmNode>()->value;
716 std::vector<IndexExpr> result_shape(output_rank, Any());
717 reporter->Assign(types[2], TensorType(result_shape, data->dtype));
718 return true;
719}
