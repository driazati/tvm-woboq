34TVM_REGISTER_NODE_TYPE(Resize2DAttrs);
35
36bool Resize2DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
37 const TypeReporter& reporter) {
38 // {data, size, roi, out}
39 ICHECK_EQ(types.size(), 4);
40 const auto* data = types[0].as<TensorTypeNode>();
41 if (data == nullptr) return false;
42
43 static const Layout kNCHW("NCHW");
44
45 const Resize2DAttrs* param = attrs.as<Resize2DAttrs>();
46 ICHECK(param != nullptr);
47 const Layout in_layout(param->layout);
48 auto layout_converter = tir::BijectiveLayout(in_layout, kNCHW);
49 ICHECK(layout_converter.defined())
50 << "Resize only support input layouts that are convertible from NCHW."
51 << " But got " << in_layout;
52
53 auto oshape = layout_converter.ForwardShape(data->shape);
54 oshape.Set(2, Any());
55 oshape.Set(3, Any());
56
57 DataType out_dtype = param->out_dtype;
58 if (out_dtype.bits() == 0) {
59 out_dtype = data->dtype;
60 }
61
62 // assign output type
63 reporter->Assign(types[3], TensorType(layout_converter.BackwardShape(oshape), out_dtype));
64 return true;
65}
66
67// Positional relay function to create image operator
68// used by frontend FFI.
69Expr MakeResize2D(Expr data, Expr size, Expr roi, String layout, String method,
70 String coordinate_transformation_mode, String rounding_method, double cubic_alpha,
71 double cubic_exclude, double extrapolation_value, DataType out_dtype) {
72 auto attrs = make_object<Resize2DAttrs>();
73 attrs->layout = std::move(layout);
74 attrs->method = std::move(method);
75 attrs->coordinate_transformation_mode = coordinate_transformation_mode;
76 attrs->rounding_method = rounding_method;
77 attrs->cubic_alpha = cubic_alpha;
78 attrs->cubic_exclude = cubic_exclude;
79 attrs->extrapolation_value = extrapolation_value;
80 attrs->out_dtype = out_dtype;
81 static const Op& op = Op::Get("dyn.image.resize2d");
82 return Call(op, {data, size, roi}, Attrs(attrs), {});
83}
84
85TVM_REGISTER_GLOBAL("relay.op.dyn.image._make.resize2d").set_body_typed(MakeResize2D);
86
87RELAY_REGISTER_OP("dyn.image.resize2d")
88 .describe(R"code(Perform resize to input array with nearest neighbour or bilinear interpolation.
89
90- **data**: data is 4D array of shape
91 (batch_size, channels, in_height, in_width) for NCHW
92 (batch_size, in_height, in_width, channels) for NHWC
93
94- **size**: data is 2D array of shape (2,) with values
95 (new_height, new_width)
96
97- **out**: Output is 4D array of shape
98 for layout NCHW
99 (batch_size, channels, size[0], size[1])
100
101 for layout NHWC
102 (batch_size, size[0], size[1], channels)
103)code" TVM_ADD_FILELINE)
104 .set_attrs_type<Resize2DAttrs>()
105 .set_num_inputs(3)
106 .add_argument("data", "Tensor", "The input tensor.")
107 .add_argument("size", "Tensor", "The output size tensor.")
108 .add_argument("roi", "Tensor", "The region of interest for tf_crop_and_resize.")
109 .set_support_level(5)
110 .add_type_rel("DynResize2D", Resize2DRel)
111 .set_attr<TOpPattern>("TOpPattern", kInjective);
