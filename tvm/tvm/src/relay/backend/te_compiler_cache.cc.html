104Array<IndexExpr> GetShape(const Array<IndexExpr>& shape) {
105 // for now, we always use int32 shape when possible
106 // even if the result of shape inference becomes int64.
107 Array<IndexExpr> res;
108 for (IndexExpr val : shape) {
109 const int64_t* pval = tir::as_const_int(val);
110 if (pval != nullptr) {
111#ifndef TVM_INDEX_DEFAULT_I64
112 ICHECK_LE(pval[0], std::numeric_limits<int32_t>::max())
113 << "dimension must be less then int32_t's max value";
114 ICHECK_GE(pval[0], std::numeric_limits<int32_t>::min())
115 << "dimension must be less then int32_t's max value";
116 res.push_back(IntImm(DataType::Int(32), *pval));
117#else
118 res.push_back(val);
119#endif // TVM_INDEX_DEFAULT_I64
120 } else if (val->IsInstance<tir::AnyNode>()) {
121 // currently all 'any' we meet in shape function are non-negative.
122 res.push_back(val.as<tir::AnyNode>()->ToSizeVar());
123 } else {
124 res.push_back(val);
125 }
126 }
127 return res;
128}

130// Helper class that is used during lowering to TE.
131// It matches sequence of Ops and lower them into single TOPI operation. All supported patterns are
132// enumerated in "supported_patterns_".
133class QnnPatternMatcher {
134 public:
135 QnnPatternMatcher()
136 : qnn_conv2d_op_(Op::Get("qnn.conv2d")),
137 qnn_dense_op_(Op::Get("qnn.dense")),
138 qnn_requantize_op_(Op::Get("qnn.requantize")),
139 bias_add_op_(Op::Get("add")) {}
140
141 // Memoize visited operations
142 void Register(const CallNode* call_node) {
143 ICHECK(call_node->op.as<OpNode>());
144 Op op = Downcast<Op>(call_node->op);
145 if (op == qnn_conv2d_op_) {
146 registered_ops_.push_front(P_QConv2d);
147 ICHECK(anchor_op_ == nullptr);
148 anchor_op_ = call_node;
149 } else if (op == qnn_requantize_op_) {
150 registered_ops_.push_front(P_QRequantize);
151 } else if (op == bias_add_op_) {
152 registered_ops_.push_front(P_BiasAdd);
153 } else if (op == qnn_dense_op_) {
154 registered_ops_.push_front(P_QDense);
155 ICHECK(anchor_op_ == nullptr);
156 anchor_op_ = call_node;
157 } else {
158 registered_ops_.push_front(P_Opaque);
159 }
160 }
161
162 // Check whether given Op is a part of matched pattern.
163 bool find(const Op& op) {
164 if (registered_ops_.empty()) return false;
165
166 if (op == qnn_conv2d_op_ || op == qnn_requantize_op_ || op == bias_add_op_ ||
167 op == qnn_dense_op_) {
168 for (const auto& pat : supported_patterns_) {
169 auto it =
170 std::search(registered_ops_.begin(), registered_ops_.end(), pat.begin(), pat.end());
171 if (it != registered_ops_.end()) return true;
172 }
173 }
174 return false;
175 }
176
177 // returns whether given Op is last in the pattern sequence.
178 bool IsLeafOp(const Op& op) { return op == qnn_requantize_op_; }
179
180 const CallNode* GetAnchorOp() { return anchor_op_; }
181
182 void Clear() { registered_ops_.clear(); }
183
184 private:
185 const Op& qnn_conv2d_op_;
186 const Op& qnn_dense_op_;
187 const Op& qnn_requantize_op_;
188 const Op& bias_add_op_;
189
190 // Main (complicated) operation in the primitive (for example qnn.conv2d, qnn.dense etc.).
191 const CallNode* anchor_op_ = nullptr;
192
193 enum POper { P_QConv2d, P_QDense, P_BiasAdd, P_QRequantize, P_Opaque };
194
195 std::deque<POper> registered_ops_;
196
197 const std::vector<std::deque<POper>> supported_patterns_ = {
198 {P_QDense, P_BiasAdd, P_QRequantize}, // Pattern qnn.dense -> bias_add -> qnn.requantize
199 {P_QDense, P_QRequantize}, // Patter qnn.dense -> qnn.requantize
200 {P_QConv2d, P_BiasAdd, P_QRequantize}, // Pattern qnn.conv2d -> bias_add -> qnn.requantize
201 {P_QConv2d, P_QRequantize} // Patter qnn.conv2d -> qnn.requantize
202 };
203};

205// Lowers Relay primitive Function to TE Compute
206class LowerToTECompute : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
207 public:
208 LowerToTECompute(Target target, NameSupply constants_name_supply)
209 : target_(target),
210 device_copy_op_(Op::Get("device_copy")),
211 constants_name_supply_(constants_name_supply) {}

213 Array<te::Tensor> Lower(const Function& relay_func) {
214 for (Var param : relay_func->params) {
215 Array<tvm::te::Tensor> inputs;
216 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
217 auto name_hint = param->vid->name_hint;
218 tvm::te::Tensor tensor = tvm::te::placeholder(
219 GetShape(ttype->shape), ttype->dtype, (name_hint == "") ? "placeholder" : name_hint);
220 inputs.push_back(tensor);
221 fn_inputs_.push_back(tensor);
222 }
223 memo_[param] = inputs;
224 }
225 readable_name_stream_ << "fused";
226
227 Array<te::Tensor> outputs = this->VisitExpr(relay_func->body);
228
229 candidate_name_ = readable_name_stream_.str();
230
231 constexpr static size_t kMaxFuncNameLength = 80;
232 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
233 // whenever the value of kMaxFuncNameLength changes
234 if (candidate_name_.size() > kMaxFuncNameLength) {
235 std::stringstream truncated_name;
236 truncated_name << candidate_name_.substr(0, kMaxFuncNameLength);
237 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name_) << "_";
238 candidate_name_ = truncated_name.str();
239 }
240
241 return outputs;
242 }

244 Array<te::Tensor> VisitExpr_(const VarNode* op) final {
245 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(GetRef<Var>(op));
246 }
247
248 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
249 using tir::make_const;
250 void* data = op->data->data;
251 DataType dtype = DataType(op->data->dtype);
252 if (op->is_scalar()) {
253 auto value = te::compute(
254 {},
255 [&](const Array<tvm::tir::Var>&) {
256 if (dtype == DataType::Int(16)) {
257 return make_const(dtype, static_cast<const int16_t*>(data)[0]);
258 } else if (dtype == DataType::Int(8)) {
259 return make_const(dtype, static_cast<const int8_t*>(data)[0]);
260 } else if (dtype == DataType::UInt(8) || dtype == DataType::Bool()) {
261 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
262 } else if (dtype == DataType::Int(32)) {
263 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
264 } else if (dtype == DataType::Int(64)) {
265 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
266 } else if (dtype == DataType::Float(16)) {
267 return make_const(dtype, __gnu_h2f_ieee(static_cast<const uint16_t*>(data)[0]));
268 } else if (dtype == DataType::Float(32)) {
269 return make_const(dtype, static_cast<const float*>(data)[0]);
270 } else if (dtype == DataType::Float(64)) {
271 return make_const(dtype, static_cast<const double*>(data)[0]);
272 } else {
273 LOG(FATAL) << dtype << " not handled";
274 }
275 },
276 "compile_engine_const", topi::kBroadcast);
277 scalars_.push_back(value->op);
278 return {value};
279 } else {
280 const auto* ttype = op->checked_type().as<TensorTypeNode>();
281 std::stringstream ss;
282 std::string s = readable_name_stream_.str();
283 std::replace(s.begin(), s.end(), '.', '_');
284 ss << constants_name_supply_->FreshName(s + "_constant");
285 tvm::te::Tensor tensor = tvm::te::placeholder(GetShape(ttype->shape), ttype->dtype, ss.str());
286 constant_tensors_[op] = tensor;
287 return {tensor};
288 }
289 }

291 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
292 static auto flower_call = tvm::runtime::Registry::Get("relay.backend.lower_call");
293 ICHECK(flower_call) << "relay.backend.lower_call is not registered.";
294
295 pattern_matcher_.Register(call_node);
296
297 Array<te::Tensor> inputs;
298 // int count_tuple = 0;
299 for (Expr arg : call_node->args) {
300 if (arg->checked_type().as<TupleTypeNode>()) {
301 // ++count_tuple;
302 }
303 for (te::Tensor tensor : VisitExpr(arg)) {
304 inputs.push_back(tensor);
305 }
306 }
307
308 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
309 Op op = Downcast<Op>(call_node->op);
310
311 // TODO(mbs): device_copy cleanup
312 ICHECK_NE(op, device_copy_op_) << "device_copy cannot be lowered";
313
314 Array<te::Tensor> outputs;
315
316 if (pattern_matcher_.find(op)) {
317 if (pattern_matcher_.IsLeafOp(op)) {
318 // Lower anchor op when pattern leaf op was reached
319 auto anchor_op = pattern_matcher_.GetAnchorOp();
320 LoweredOutput lowered_out =
321 (*flower_call)(GetRef<Call>(anchor_op), inputs, target_, call_node->checked_type());
322 outputs = lowered_out->outputs;
323 Op a_op = Downcast<Op>(anchor_op->op);
324 op_implementations_[a_op.operator->()] = lowered_out->implementation;
325
326 pattern_matcher_.Clear();
327 } else {
328 // Forward inputs as "outputs" for successor.
329 readable_name_stream_ << '_' << op->name;
330 return inputs;
331 }
332 } else {
333 LoweredOutput lowered_out = (*flower_call)(GetRef<Call>(call_node), inputs, target_);
334 outputs = lowered_out->outputs;
335 op_implementations_[op.operator->()] = lowered_out->implementation;
336 }
337
338 if (outputs.size() != 1) {
339 const auto* tuple_type = call_node->checked_type().as<TupleTypeNode>();
340 ICHECK(tuple_type) << "Expected output to be a tuple type "
341 << PrettyPrint(call_node->checked_type());
342
343 ICHECK_EQ(tuple_type->fields.size(), outputs.size());
344 }
345
346 readable_name_stream_ << '_' << op->name;
347 return outputs;
348 }

350 Array<te::Tensor> VisitExpr_(const FunctionNode* op) final {
351 LOG(FATAL) << "Primitive Functions can not contain nested functions.";
352 }
353
354 Array<te::Tensor> VisitExpr_(const LetNode* op) final {
355 Array<te::Tensor> val = VisitExpr(op->value);
356 ICHECK(!memo_.count(op->var));
357 memo_[op->var] = val;
358 return VisitExpr(op->body);
359 }
360
361 Array<te::Tensor> VisitExpr_(const TupleNode* op) final {
362 Array<te::Tensor> fields;
363 for (Expr field : op->fields) {
364 // TODO(mbs): Generalize to be equivalent to FlattenTupleType.
365 ICHECK(field->checked_type().as<TensorTypeNode>()) << "Only allow Tuple of Tensor";
366 Array<te::Tensor> res = VisitExpr(field);
367 ICHECK_EQ(res.size(), 1);
368 fields.push_back(res[0]);
369 }
370 return fields;
371 }
372
373 Array<te::Tensor> VisitExpr_(const TupleGetItemNode* op) final {
374 const auto* tuple_type = op->tuple->type_as<TupleTypeNode>();
375 Array<te::Tensor> tuple = VisitExpr(op->tuple);
376 ICHECK_EQ(tuple_type->fields.size(), tuple.size());
377 ICHECK_GE(op->index, 0);
378 ICHECK_LT(static_cast<size_t>(op->index), tuple.size());
379 return {tuple[op->index]};
380 }
381
382 public:
383 // Additional outputs
384 Array<tvm::te::Tensor> fn_inputs_;
385 Array<te::Operation> scalars_;
386 std::unordered_map<const ConstantNode*, te::Tensor> constant_tensors_;
387 std::unordered_map<const OpNode*, OpImplementation> op_implementations_;
388 std::string candidate_name_;
389
390 private:
391 QnnPatternMatcher pattern_matcher_;
392
393 tvm::Target target_;
394 std::ostringstream readable_name_stream_;
395 // Cache device copy op for equivalence checking to reduce registry lookup
396 // overhead for each invocation of call node when retrieving schedules.
397 const Op& device_copy_op_;
398 // A NameSupply object passed from a caller, used to assign unique names to constants
399 // across different invocations of LowerToTECompute.
400 NameSupply constants_name_supply_;
401};

405class LayoutFreeConstantCollector : public StmtVisitor {
406 public:
407 Array<runtime::NDArray> constants;
408
409 private:
410 void VisitStmt_(const BlockNode* op) final {
411 StmtVisitor::VisitStmt_(op);
412 if (Optional<ObjectRef> ann = op->annotations.Get("layout_free_placeholders")) {
413 for (Buffer buffer : Downcast<Array<Buffer>>(ann)) {
414 layout_free_buffer_vars_.insert(buffer->data.get());
415 }
416 }
417 }
418
419 void VisitStmt_(const AllocateConstNode* op) final {
420 StmtVisitor::VisitStmt_(op);
421 if (auto it = layout_free_buffer_vars_.find(op->buffer_var.get());
422 it != layout_free_buffer_vars_.end()) {
423 constants.push_back(op->data.value());
424 }
425 }
426
427 std::unordered_set<const tir::VarNode*> layout_free_buffer_vars_;
428};

433// Replace constants in AllocateConst nodes according to the given mapping
434class AllocateConstReplaceConstant : public StmtExprMutator {
435 public:
436 explicit AllocateConstReplaceConstant(const NDArrayMap& constant_map)
437 : constant_map_(constant_map) {}
438
439 static PrimFunc Rewrite(PrimFunc f, const NDArrayMap& constant_map) {
440 AllocateConstReplaceConstant rewriter(constant_map);
441 PrimFuncNode* n = f.CopyOnWrite();
442 n->body = rewriter(std::move(n->body));
443 return f;
444 }
445
446 private:
447 Stmt VisitStmt_(const AllocateConstNode* op) final {
448 if (auto it = constant_map_.find(op->data.value()); it != constant_map_.end()) {
449 auto rewriten_constant = it->second;
450 Array<PrimExpr> rewritten_extents;
451 for (auto s : rewriten_constant.Shape()) {
452 rewritten_extents.push_back(PrimExpr(static_cast<int>(s)));
453 }
454 return AllocateConst(op->buffer_var, op->dtype, rewritten_extents, rewriten_constant,
455 op->body, op->annotations, op->span);
456 }
457 return StmtExprMutator::VisitStmt_(op);
458 }
459
460 NDArrayMap constant_map_;
461};

463// Construct a schedule for a given Relay primitive function and target.
464class ScheduleBuilder : public ExprVisitor {
465 public:
466 explicit ScheduleBuilder(Target target)
467 : target_(target),
468 mod_eq_structural_(meta_schedule::ModuleEquality::Create("ignore-ndarray")) {
469 // Whether to use auto_scheduler schedule.
470 use_auto_scheduler_ = backend::IsAutoSchedulerEnabled();
471 if (backend::IsMetaScheduleEnabled()) {
472 database_ = meta_schedule::Database::Current();
473 CHECK(database_.defined()) << "ValueError: `use_meta_schedule` is enabled in Relay "
474 "build, but no `meta_schedule.Database` context is provided. ";
475 } else {
476 database_ = NullOpt;
477 }
478 }
479
480 CachedFunc Create(const Function& relay_func, GlobalVarSupply global_var_supply,
481 NameSupply constant_name_supply) {
482 LowerToTECompute lower_te_compute(target_, constant_name_supply);
483 Array<te::Tensor> tensor_outs = lower_te_compute.Lower(relay_func);
484 Array<te::Tensor> fn_inputs = lower_te_compute.fn_inputs_;
485 VisitExpr(relay_func->body);
486
487 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
488 // no other GlobalVar ctors should appear inside the lowering machinery.
489 auto prim_fn_var = global_var_supply->FreshGlobal(lower_te_compute.candidate_name_);
490 prim_fn_var->checked_type_ = relay_func->checked_type();
491
492 // Fusion over tupled results may leave identity relationships
493 // between inputs and outputs, copy identity output tensors,
494 // since tir lowering do not support aliasing output to input buffer.
495 for (size_t i = 0; i < tensor_outs.size(); ++i) {
496 if (tensor_outs[i]->op.as<te::PlaceholderOpNode>()) {
497 tensor_outs.Set(i, topi::identity(tensor_outs[i]));
498 }
499 }
500
501 te::Schedule schedule{nullptr};
502 tir::PrimFunc prim_func{nullptr};
503 // No need to register schedule for device copy op.
504 if (anchor_attrs_.as<DeviceCopyAttrs>() == nullptr) {
505 if (use_auto_scheduler_) {
506 const auto* fauto_schedule =
507 runtime::Registry::Get("auto_scheduler.relay_integration.auto_schedule_topi_compute");
508 ICHECK(fauto_schedule != nullptr)
509 << "auto_scheduler.relay_integration.auto_schedule_topi_compute is not registered";
510 ObjectRef obj = (*fauto_schedule)(prim_fn_var->name_hint, tensor_outs);
511 if (obj.defined()) {
512 schedule = Downcast<te::Schedule>(obj);
513 }
514 }
515 if (database_) {
516 using tvm::meta_schedule::TuningRecord;
517 using tvm::tir::IndexMap;
518 using tvm::tir::Instruction;
519 using tvm::tir::InstructionKind;
520 using tvm::tir::PrimFunc;
521 using tvm::tir::Schedule;
522 backend::FTECompilerTIRConverter tir_converter = backend::GetTIRConverter();
523 Array<te::Tensor> te_args = Concat(fn_inputs, tensor_outs);
524 Array<runtime::NDArray> constants;
525 for (auto [const_node, te_tensor] : lower_te_compute.constant_tensors_) {
526 te_args.push_back(te_tensor);
527 constants.push_back(const_node->data);
528 }
529 if (Optional<PrimFunc> f = tir_converter(te_args, constants)) {
530 IRModule query_mod = backend::PrimFuncToIRModule(f.value());
531 if (Optional<TuningRecord> opt_record = database_.value()->QueryTuningRecord(
532 /*mod=*/query_mod,
533 /*target=*/target_,
534 /*workload_name=*/prim_fn_var->name_hint)) {
535 LayoutFreeConstantCollector const_collector;
536 const_collector(f.value()->body);
537
538 static InstructionKind kind_transform_layout = InstructionKind::Get("TransformLayout");
539 TuningRecord record = opt_record.value();
540 for (const Instruction& inst : record->trace->insts) {
541 if (inst->kind.same_as(kind_transform_layout)) {
542 ICHECK_EQ(inst->attrs.size(), 4);
543 auto index_map = Downcast<IndexMap>(inst->attrs[2]);
544
545 if (!const_collector.constants.empty()) {
546 // In this case, RewriteLayout is acting on an AllocateConst node.
547 // After tuning, we reach this code path twice: First by
548 // the Relay MetaScheduleLayoutRewrite pass, and next by the final
549 // compilation (Relay to TE schedule lowering).
550 //
551 // Due to Relay MetaScheduleLayoutRewrite and FoldConstant passes,
552 // the Relay subgraph for which we query the database during the
553 // final compilation has its weight tensor transformed according to
554 // the index map, determined during tuning. For example,
555 //
556 // fn (%p0: Tensor[(1, 56, 56, 64), float32]) {
557 // %0 = nn.conv2d(%p0, meta[relay.Constant][0],
558 // /*ty=Tensor[(4, 2, 2, 3, 3, 32, 8), float32]*/, ...);
559 // add(%0, meta[relay.Constant][1])
560 // }
561 //
562 // Note that the database does not have an entry corresponding to such subgraphs,
563 // since an input subgraph to the tuning system always has its weight tensor in
564 // the original layout, e.g.
565 //
566 // fn (%p0: Tensor[(1, 56, 56, 64), float32]) {
567 // %0 = nn.conv2d(%p0, meta[relay.Constant][0],
568 // /*ty=Tensor[(3, 3, 64, 64), float32]*/, ...);
569 // add(%0, meta[relay.Constant][1])
570 // }
571 //
572 // Thus, in both of the two cases where we reach this code path, we need careful
573 // logic to make sure that (1) the database lookup during the final compilation
574 // succeeds and (2) the application of a schedule trace is well defined.
575
576 ICHECK(const_collector.constants.size() == 1)
577 << "Only one layout-free constant is supported by RewriteLayout for now";
578 auto constant = const_collector.constants[0];
579
580 if (constant.Shape().size() == index_map->initial_indices.size()) {
581 // This is the first case, reached during the MetaScheduleLayoutRewrite pass.
582 //
583 // A layout-free constant having the same rank as an input to the index map
584 // is assumed to be transformed by this index map.
585 // TODO(masahi): If there are multiple layout-free constants in one
586 // TIR mod (e.g. conv2d -> conv2d fusion), this assumption does not hold.
587 // We need to determine which constant the given index map acts on.
588 //
589 // We know that, during the final compilation, we will query the database
590 // for a subgraph that the tuner has never seen. We workaround this problem
591 // by adding a dummy entry to the database. The dummy entry is carefully
592 // constructed so that the lookup during the final compilation would succeed.
593 runtime::NDArray rewritten_constant = index_map->MapNDArray(constant);
594 auto f_dummy = AllocateConstReplaceConstant::Rewrite(
595 f.value(), {{constant, rewritten_constant}});
596 auto workload_dummy =
597 database_.value()->CommitWorkload(backend::PrimFuncToIRModule(f_dummy));
598 TuningRecord rec_dummy(record->trace, workload_dummy, record->run_secs,
599 record->target, record->args_info);
600 database_.value()->CommitTuningRecord(rec_dummy);
601 } else {
602 // The constant is already transformed, so this is the second case, reached
603 // during the final compilation.
604 //
605 // The schedule trace is supposed to be applied to the weight in its original
606 // layout. But as explained above, the Relay subgraph we get in this case
607 // has its weight tensor transformed according to the corresponding index map.
608 // So effectively, we undo the layout transformation on the weight to restore
609 // the original PrimFunc that the schedule trace is supposed to act on.
610 ICHECK(index_map->inverse_index_map);
611 auto inverse_map = Downcast<IndexMap>(index_map->inverse_index_map.value());
612 ICHECK(constant.Shape().size() == inverse_map->initial_indices.size());
613 runtime::NDArray orig_constant = inverse_map->MapNDArray(constant);
614 auto f_ = AllocateConstReplaceConstant::Rewrite(f.value(),
615 {{constant, orig_constant}});
616 query_mod = backend::PrimFuncToIRModule(f_);
617 }
618 }
619 MetaScheduleLayoutRewriter::LayoutQueuePush(index_map);
620 }
621 }
622
623 Schedule sch = Schedule::Traced(query_mod, /*seed=*/-1, /*debug_mask=*/0,
624 tir::ScheduleErrorRenderLevel::kDetail);
625
626 if (!mod_eq_structural_->Equal(query_mod, opt_record.value()->workload->mod)) {
627 // When the database lookup succeeds while structural equality check fails,
628 // it implies that the anchor block based equality has been used during tuning.
629 // The trace in the record cannot directly be applied to this query module.
630 meta_schedule::ScheduleUsingAnchorTrace(sch, record->trace, target_);
631 } else {
632 record->trace->ApplyToSchedule(sch, /*remove_postproc=*/false);
633 }
634
635 IRModule mod = sch->mod();
636 ICHECK_EQ(mod->functions.size(), 1);
637 mod = tir::transform::RemoveWeightLayoutRewriteBlock(/*skip_ndarray_rewrite*/ false)(
638 std::move(mod));
639 prim_func = Downcast<PrimFunc>(mod->Lookup("main"));
640 // Need to copy attrs from relay function over to prim func. Most notably the structural
641 // hash.
642 prim_func = WithAttrs(prim_func, relay_func->attrs->dict);
643 } else {
644 int dispatch = backend::UseMetaScheduleDispatch();
645 // (dispatch & 2): controls whether to print TVMScript for missing TIR
646 // (dispatch & 4): controls whether to raise fatal errors for missing TIR
647 if (dispatch & 2) {
648 LOG(WARNING) << "Cannot find workload: " << prim_fn_var->name_hint << "\n"
649 << tir::AsTVMScript(f.value());
650 } else {
651 LOG(WARNING) << "Cannot find workload: " << prim_fn_var->name_hint;
652 }
653 if (dispatch & 4) {
654 LOG(FATAL);
655 }
656 }
657 }
658 }
659 // Use TOPI schedule if user specified, or the function has no auto_scheduler schedule.
660 if (!schedule.defined() && !prim_func.defined()) {
661 if (anchor_op_.defined()) {
662 auto anchor_impl = lower_te_compute.op_implementations_.find(anchor_op_.operator->());
663 ICHECK(anchor_impl != lower_te_compute.op_implementations_.end());
664 schedule = anchor_impl->second.Schedule(anchor_attrs_, tensor_outs, target_);
665 } else {
666 auto default_sched = GenericFunc::Get("schedule_injective");
667 ICHECK(default_sched.defined()) << "schedule_injective not registered for " << target_;
668 With<Target> tctx(target_);
669 schedule = default_sched(tensor_outs);
670 }
671 }
672 if (schedule.defined()) {
673 for (const auto& scalar : lower_te_compute.scalars_) {
674 if (schedule->Contain(scalar)) {
675 schedule[scalar].compute_inline();
676 }
677 }
678 }
679 }
680
681 IRModule funcs = IRModule(Map<GlobalVar, BaseFunc>({}));
682 return CachedFunc(target_, prim_fn_var, fn_inputs, tensor_outs, schedule, prim_func, {}, funcs,
683 lower_te_compute.constant_tensors_);
684 }
685
686 void VisitExpr_(const CallNode* call_node) final {
687 static auto fpattern = Op::GetAttrMap<TOpPattern>("TOpPattern");
688
689 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
690 Op op = Downcast<Op>(call_node->op);
691
692 for (Expr arg : call_node->args) {
693 VisitExpr(arg);
694 }
695
696 int op_pattern = fpattern[op];
697 if (!use_auto_scheduler_ && !database_.defined() && op_pattern >= kCommReduce) {
698 ICHECK(!anchor_op_.defined() || anchor_op_pattern_ < kCommReduce)
699 << "Cannot apply TOPI schedule to a primitive function with two complicated ops"
700 << " anchor=" << anchor_op_ << " current=" << op;
701 }
702 if (op_pattern >= anchor_op_pattern_) {
703 anchor_op_ = op;
704 anchor_attrs_ = call_node->attrs;
705 anchor_op_pattern_ = op_pattern;
706 }
707 }

709 private:
710 tvm::Target target_;
711 Op anchor_op_;
712 Attrs anchor_attrs_;
713 int anchor_op_pattern_{0};
714 bool use_auto_scheduler_;
715 Optional<meta_schedule::Database> database_;
716 std::unique_ptr<meta_schedule::ModuleEquality> mod_eq_structural_;
717};

732class MakeShapeFunc : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
733 public:
734 MakeShapeFunc() {}
735
736 CachedFunc Create(const Function& prim_func, const Target& target,
737 GlobalVarSupply global_var_supply) {
738 VLOG_CONTEXT << "MakeShapeFunc";
739 TShapeDataDependent shape_func_param_states;
740
741 for (auto param : prim_func->params) {
742 param_states_[param] = kNoNeed;
743 Array<tvm::te::Tensor> data_inputs;
744 Array<tvm::te::Tensor> shape_inputs;
745
746 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
747 // Add data placeholder (in case we discover we need it below)
748 Shape shape = GetShape(ttype->shape);
749 tvm::te::Tensor data_tensor =
750 tvm::te::placeholder(shape, ttype->dtype, "data_" + param->vid->name_hint);
751 data_inputs.push_back(data_tensor);
752 // Add shape placeholder (in case we discover we need it below)
753 int64_t ndim = shape.size();
754 Shape sshape;
755 if (ndim > 0) {
756 sshape.push_back(tvm::Integer(ndim));
757 }
758 tvm::te::Tensor shape_tensor =
759 tvm::te::placeholder(sshape, DataType::Int(64), "shape_" + param->vid->name_hint);
760 shape_inputs.push_back(shape_tensor);
761 }
762 param_data_[param] = data_inputs;
763 param_shapes_[param] = shape_inputs;
764 }
765
766 // Setup the name;
767 readable_name_stream_ << "shape_func";
768
769 // Create the tensor expressions representing the output shapes.
770 Array<te::Tensor> outputs = VisitExpr(prim_func->body);
771
772 // Generate a name.
773 auto candidate_name = readable_name_stream_.str();
774
775 constexpr static size_t kMaxFuncNameLength = 80;
776 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
777 // whenever the value of kMaxFuncNameLength changes
778 if (candidate_name.size() > kMaxFuncNameLength) {
779 std::stringstream truncated_name;
780 truncated_name << candidate_name.substr(0, kMaxFuncNameLength);
781 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name) << "_";
782 candidate_name = truncated_name.str();
783 }
784
785 // Set all the inputs correctly, and accumulate their types from the p.o.v. of the
786 // shape function rather than the primitive it is derived for.
787 Array<te::Tensor> inputs;
788 Array<Type> shape_function_arg_types;
789 for (auto param : prim_func->params) {
790 int state = param_states_[param];
791 shape_func_param_states.push_back(IntImm(DataType::Int(32), state));
792 if (state & kNeedInputData) {
793 // Pass the primitive arguments directly (though in flattened form and on the host)
794 for (auto t : param_data_[param]) {
795 inputs.push_back(t);
796 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
797 }
798 }
799 if (state & kNeedInputShape) {
800 // Pass the shapes of the primitive arguments (also on the host)
801 for (auto t : param_shapes_[param]) {
802 inputs.push_back(t);
803 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
804 }
805 }
806 }
807
808 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
809 // no other GlobalVar ctors should appear inside the lowering machinery.
810 auto prim_fn_gvar = global_var_supply->FreshGlobal(candidate_name);
811
812 // Gather the result types, again from the p.o.v. of the shape function rather than
813 // the primitive it is derived for.
814 Array<Type> shape_function_res_types;
815 for (const auto& t : outputs) {
816 shape_function_res_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
817 }
818
819 // Assign the shape function its true type.
820 FuncType type(shape_function_arg_types, TupleType(shape_function_res_types),
821 /*type_params=*/{}, /*type_constraints=*/{});
822 VLOG(1) << "shape function '" << prim_fn_gvar->name_hint << "' has type:" << std::endl
823 << PrettyPrint(type) << std::endl
824 << "corresponding to primitive of type:" << std::endl
825 << PrettyPrint(prim_func->checked_type());
826 prim_fn_gvar->checked_type_ = std::move(type);
827
828 // generate schedule for shape func
829 Array<te::Operation> out_ops;
830 for (auto t : outputs) {
831 out_ops.push_back(t->op);
832 }
833 te::Schedule schedule = te::create_schedule(out_ops);
834 tvm::te::AutoInlineInjective(schedule);
835 for (const auto& scalar : scalars_) {
836 auto scalar_op = scalar->op;
837 if (schedule->Contain(scalar_op)) {
838 schedule[scalar_op].compute_inline();
839 }
840 }
841
842 Array<te::Tensor> all_args = Array<te::Tensor>(inputs);
843 for (te::Tensor arg : outputs) {
844 all_args.push_back(arg);
845 }
846
847 using tvm::transform::PassContext;
848 With<PassContext> fresh_pass_ctx_scope(PassContext::Create());
849
850 std::unordered_map<te::Tensor, tir::Buffer> binds;
851 IRModule lowered_module =
852 tvm::LowerSchedule(schedule, all_args, prim_fn_gvar->name_hint, binds, global_var_supply);
853 return CachedFunc(target, prim_fn_gvar, inputs, outputs, schedule, tir::PrimFunc{nullptr},
854 shape_func_param_states, lowered_module);
855 }

857 Array<te::Tensor> VisitExpr(const Expr& expr) final {
858 if (expr.as<VarNode>()) {
859 // Do not memoize vars because shape functions could use either the data
860 // or the shape of a var each time.
861 return ExprFunctor::VisitExpr(expr);
862 }
863 // For other case, do memoized visit
864 return backend::MemoizedExprTranslator<Array<te::Tensor>>::VisitExpr(expr);
865 }
866
867 Array<te::Tensor> VisitExpr_(const VarNode* var_node) final {
868 auto var = GetRef<Var>(var_node);
869 auto it = param_arg_map_.find(var);
870 if (it != param_arg_map_.end()) {
871 // This var is a parameter of a nested function. Visit the corresponding argument in the
872 // function call site.
873 return VisitExpr(it->second);
874 }
875 if (param_states_.find(var) == param_states_.end()) {
876 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(var);
877 } else {
878 ICHECK(data_dependents_per_input_.size());
879 auto data_dependent = data_dependents_per_input_.back();
880 if (data_dependent) {
881 param_states_[var] |= kNeedInputData;
882 return param_data_[var];
883 } else {
884 param_states_[var] |= kNeedInputShape;
885 return param_shapes_[var];
886 }
887 }
888 }
889
890 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
891 using tir::make_const;
892 ICHECK(data_dependents_per_input_.size());
893 bool data_dependent = data_dependents_per_input_.back();
894 if (!op->is_scalar()) {
895 // This is a constant weight, extract the shape of the weight tensor.
896 // This can not be data dependent.
897 CHECK(!data_dependent);
898 auto ttype = op->checked_type().as<TensorTypeNode>();
899 int ndim = static_cast<int>(ttype->shape.size());
900 Array<PrimExpr> out_shape{ndim};
901 te::Tensor value = tvm::te::compute(
902 out_shape,
903 [&](const Array<tvm::tir::Var>& indices) {
904 auto idx = indices[0];
905 PrimExpr ret = make_const(DataType::Int(64), 0);
906 for (int i = 0; i < ndim; i++) {
907 ret = tvm::if_then_else(idx == i, ttype->shape[i], ret);
908 }
909 return ret;
910 },
911 "shape_const", topi::kBroadcast);
912 scalars_.push_back(value);
913 return {value};
914 }
915 if (data_dependent) {
916 void* data = op->data->data;
917 DataType dtype = DataType(op->data->dtype);
918 auto value = tvm::te::compute(
919 {},
920 [&](const Array<tvm::tir::Var>&) {
921 if (dtype == DataType::Int(32)) {
922 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
923 } else if (dtype == DataType::Int(64)) {
924 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
925 } else if (dtype == DataType::Float(32)) {
926 return make_const(dtype, static_cast<const float*>(data)[0]);
927 } else if (dtype == DataType::Float(64)) {
928 return make_const(dtype, static_cast<const double*>(data)[0]);
929 } else if (dtype == DataType::Bool()) {
930 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
931 } else {
932 LOG(FATAL) << "not handled";
933 }
934 },
935 "data_const", topi::kBroadcast);
936 scalars_.push_back(value);
937 return {value};
938 } else {
939 auto value = tvm::te::compute(
940 {}, [&](const Array<tvm::tir::Var>&) { return tir::make_const(DataType::Int(64), 0); },
941 "shape_const", topi::kBroadcast);
942 scalars_.push_back(value);
943 return {value};
944 }
945 }

947 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
948 VLOG(1) << "considering call:" << std::endl << PrettyPrint(GetRef<Call>(call_node));
949 if (auto* func = call_node->op.as<FunctionNode>()) {
950 VLOG(1) << "user function";
951 for (size_t i = 0; i < func->params.size(); ++i) {
952 param_arg_map_[func->params[i]] = call_node->args[i];
953 }
954 return VisitExpr(func->body);
955 }
956
957 static auto fshape_func = Op::GetAttrMap<FShapeFunc>("FShapeFunc");
958 static auto tshape_data_dependent = Op::GetAttrMap<TShapeDataDependent>("TShapeDataDependent");
959 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
960 Op op = Downcast<Op>(call_node->op);
961 ICHECK(data_dependents_per_input_.empty() || !data_dependents_per_input_.back())
962 << "Error in op fusion: output of the shape func is fed to a "
963 << "data-dependent shape func";
964 ICHECK_GT(fshape_func.count(op), 0) << "Internal error, cannot find ShapeFunc for " << op->name;
965 ICHECK_GT(tshape_data_dependent.count(op), 0)
966 << "Internal error, cannot find TShapeDataDependent for " << op->name;
967
968 Array<Integer> dep_spec = tshape_data_dependent[op];
969 if (dep_spec.size() == 1) {
970 // This is for cases when data dependence is specified per op
971 // Replicate 0 or 1 flag to all arguments
972 for (size_t i = 1; i < call_node->args.size(); ++i) {
973 dep_spec.push_back(dep_spec[0]);
974 }
975 }
976
977 // Visit all inputs
978 Array<te::Tensor> inputs;
979 int count_tuple = 0;
980 for (size_t i = 0; i < call_node->args.size(); ++i) {
981 Expr arg = call_node->args[i];
982 if (arg->checked_type().as<TupleTypeNode>()) {
983 ++count_tuple;
984 }
985 data_dependents_per_input_.push_back(dep_spec[i]->value != 0);
986 for (te::Tensor tensor : VisitExpr(arg)) {
987 inputs.push_back(tensor);
988 }
989 data_dependents_per_input_.pop_back();
990 }
991 if (count_tuple) {
992 ICHECK_EQ(call_node->args.size(), 1U) << "Only allow function with a single tuple input";
993 }
994 // Get output ndims
995 auto ret_type = call_node->checked_type();
996 Array<IndexExpr> out_ndims;
997 for (const auto& ttype : FlattenTupleType(ret_type)) {
998 out_ndims.push_back(IntImm(DataType::Int(32), ttype->shape.size()));
999 }
1000
1001 // Call shape function
1002 Array<te::Tensor> outputs = fshape_func[op](call_node->attrs, inputs, out_ndims);
1003 VLOG(1) << "shape function for '" << op->name << "' with inputs:" << std::endl
1004 << inputs << std::endl
1005 << "yielded outputs:" << std::endl
1006 << outputs;
1007 readable_name_stream_ << "_" << op->name;
1008 return outputs;
1009 }

1065std::tuple<Array<te::Tensor>, Array<runtime::NDArray>, std::string> LowerTECompute(
1066 const Function& source_func, Target target, NameSupply constant_name_supply,
1067 bool return_inputs) {
1068 LowerToTECompute lower_te_compute(target, constant_name_supply);
1069 Array<te::Tensor> outputs = lower_te_compute.Lower(source_func);
1070 // Following ScheduleBuilder, remove placeholder ops from outputs.
1071 tvm::Array<te::Tensor> tensor_outs;
1072 for (const auto& tensor : outputs) {
1073 if (!tensor->op.as<te::PlaceholderOpNode>()) {
1074 tensor_outs.push_back(tensor);
1075 }
1076 }
1077
1078 tvm::Array<runtime::NDArray> constants;
1079 for (auto [const_node, te_tensor] : lower_te_compute.constant_tensors_) {
1080 tensor_outs.push_back(te_tensor);
1081 constants.push_back(const_node->data);
1082 }
1083
1084 if (return_inputs) {
1085 return std::make_tuple(Concat(lower_te_compute.fn_inputs_, tensor_outs), constants,
1086 lower_te_compute.candidate_name_);
1087 }
1088 return std::make_tuple(tensor_outs, constants, lower_te_compute.candidate_name_);
1089}
