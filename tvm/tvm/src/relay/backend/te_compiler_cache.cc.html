98Array<IndexExpr> GetShape(const Array<IndexExpr>& shape) {
99 // for now, we always use int32 shape when possible
100 // even if the result of shape inference becomes int64.
101 Array<IndexExpr> res;
102 for (IndexExpr val : shape) {
103 const int64_t* pval = tir::as_const_int(val);
104 if (pval != nullptr) {
105#ifndef TVM_INDEX_DEFAULT_I64
106 ICHECK_LE(pval[0], std::numeric_limits<int32_t>::max())
107 << "dimension must be less then int32_t's max value";
108 ICHECK_GE(pval[0], std::numeric_limits<int32_t>::min())
109 << "dimension must be less then int32_t's max value";
110 res.push_back(IntImm(DataType::Int(32), *pval));
111#else
112 res.push_back(val);
113#endif // TVM_INDEX_DEFAULT_I64
114 } else if (val->IsInstance<tir::AnyNode>()) {
115 // currently all 'any' we meet in shape function are non-negative.
116 res.push_back(val.as<tir::AnyNode>()->ToSizeVar());
117 } else {
118 res.push_back(val);
119 }
120 }
121 return res;
122}

124// Lowers Relay primitive Function to TE Compute
125class LowerToTECompute : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
126 public:
127 explicit LowerToTECompute(Target target)
128 : target_(target), device_copy_op_(Op::Get("device_copy")) {}
129
130 Array<te::Tensor> Lower(const Function& relay_func) {
131 for (Var param : relay_func->params) {
132 Array<tvm::te::Tensor> inputs;
133 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
134 tvm::te::Tensor tensor = tvm::te::placeholder(GetShape(ttype->shape), ttype->dtype);
135 inputs.push_back(tensor);
136 fn_inputs_.push_back(tensor);
137 }
138 memo_[param] = inputs;
139 }
140 readable_name_stream_ << "fused";
141
142 Array<te::Tensor> outputs = this->VisitExpr(relay_func->body);
143
144 candidate_name_ = readable_name_stream_.str();
145
146 constexpr static size_t kMaxFuncNameLength = 80;
147 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
148 // whenever the value of kMaxFuncNameLength changes
149 if (candidate_name_.size() > kMaxFuncNameLength) {
150 std::stringstream truncated_name;
151 truncated_name << candidate_name_.substr(0, kMaxFuncNameLength);
152 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name_) << "_";
153 candidate_name_ = truncated_name.str();
154 }
155
156 return outputs;
157 }

159 Array<te::Tensor> VisitExpr_(const VarNode* op) final {
160 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(GetRef<Var>(op));
161 return {};
162 }

164 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
165 using tir::make_const;
166 void* data = op->data->data;
167 DataType dtype = DataType(op->data->dtype);
168 if (op->is_scalar()) {
169 auto value = te::compute(
170 {},
171 [&](const Array<tvm::tir::Var>&) {
172 if (dtype == DataType::Int(16)) {
173 return make_const(dtype, static_cast<const int16_t*>(data)[0]);
174 } else if (dtype == DataType::Int(8)) {
175 return make_const(dtype, static_cast<const int8_t*>(data)[0]);
176 } else if (dtype == DataType::UInt(8) || dtype == DataType::Bool()) {
177 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
178 } else if (dtype == DataType::Int(32)) {
179 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
180 } else if (dtype == DataType::Int(64)) {
181 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
182 } else if (dtype == DataType::Float(16)) {
183 return make_const(dtype, __gnu_h2f_ieee(static_cast<const uint16_t*>(data)[0]));
184 } else if (dtype == DataType::Float(32)) {
185 return make_const(dtype, static_cast<const float*>(data)[0]);
186 } else if (dtype == DataType::Float(64)) {
187 return make_const(dtype, static_cast<const double*>(data)[0]);
188 } else {
189 LOG(FATAL) << dtype << " not handled";
190 return tvm::PrimExpr();
191 }
192 },
193 "compile_engine_const", topi::kBroadcast);
194 scalars_.push_back(value->op);
195 return {value};
196 } else {
197 const auto* ttype = op->checked_type().as<TensorTypeNode>();
198 std::stringstream ss;
199 std::string s = readable_name_stream_.str();
200 std::replace(s.begin(), s.end(), '.', '_');
201 ss << s << "_constant_" << const_index++;
202 tvm::te::Tensor tensor = tvm::te::placeholder(GetShape(ttype->shape), ttype->dtype, ss.str());
203 constant_tensors_[op] = tensor;
204 return {tensor};
205 }
206 }

208 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
209 static auto flower_call = tvm::runtime::Registry::Get("relay.backend.lower_call");
210 ICHECK(flower_call) << "relay.backend.lower_call is not registered.";
211
212 Array<te::Tensor> inputs;
213 int count_tuple = 0;
214 for (Expr arg : call_node->args) {
215 if (arg->checked_type().as<TupleTypeNode>()) {
216 ++count_tuple;
217 }
218 for (te::Tensor tensor : VisitExpr(arg)) {
219 inputs.push_back(tensor);
220 }
221 }
222
223 if (count_tuple) {
224 ICHECK_EQ(call_node->args.size(), 1U)
225 << "Only functions with a single tuple input are allowed, but " << count_tuple
226 << " were provided.";
227 }
228
229 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
230 Op op = Downcast<Op>(call_node->op);
231
232 // TODO(mbs): device_copy cleanup
233 ICHECK_NE(op, device_copy_op_) << "device_copy cannot be lowered";
234
235 LoweredOutput lowered_out = (*flower_call)(GetRef<Call>(call_node), inputs, target_);
236 Array<te::Tensor> outputs = lowered_out->outputs;
237 op_implementations_[op.operator->()] = lowered_out->implementation;
238
239 if (outputs.size() != 1) {
240 const auto* tuple_type = call_node->checked_type().as<TupleTypeNode>();
241 ICHECK(tuple_type) << "Expected output to be a tuple type "
242 << PrettyPrint(call_node->checked_type());
243
244 ICHECK_EQ(tuple_type->fields.size(), outputs.size());
245 }
246
247 readable_name_stream_ << '_' << op->name;
248 return outputs;
249 }
250
251 Array<te::Tensor> VisitExpr_(const FunctionNode* op) final {
252 LOG(FATAL) << "Primitive Functions can not contain nested functions.";
253 return Array<te::Tensor>();
254 }
255
256 Array<te::Tensor> VisitExpr_(const LetNode* op) final {
257 Array<te::Tensor> val = VisitExpr(op->value);
258 ICHECK(!memo_.count(op->var));
259 memo_[op->var] = val;
260 return VisitExpr(op->body);
261 }
262
263 Array<te::Tensor> VisitExpr_(const TupleNode* op) final {
264 Array<te::Tensor> fields;
265 for (Expr field : op->fields) {
266 // TODO(mbs): Generalize to be equivalent to FlattenTupleType.
267 ICHECK(field->checked_type().as<TensorTypeNode>()) << "Only allow Tuple of Tensor";
268 Array<te::Tensor> res = VisitExpr(field);
269 ICHECK_EQ(res.size(), 1);
270 fields.push_back(res[0]);
271 }
272 return fields;
273 }
274
275 Array<te::Tensor> VisitExpr_(const TupleGetItemNode* op) final {
276 const auto* tuple_type = op->tuple->type_as<TupleTypeNode>();
277 Array<te::Tensor> tuple = VisitExpr(op->tuple);
278 ICHECK_EQ(tuple_type->fields.size(), tuple.size());
279 ICHECK_GE(op->index, 0);
280 ICHECK_LT(static_cast<size_t>(op->index), tuple.size());
281 return {tuple[op->index]};
282 }
283
284 public:
285 // Additional outputs
286 Array<tvm::te::Tensor> fn_inputs_;
287 Array<te::Operation> scalars_;
288 std::unordered_map<const ConstantNode*, te::Tensor> constant_tensors_;
289 std::unordered_map<const OpNode*, OpImplementation> op_implementations_;
290 std::string candidate_name_;
291
292 private:
293 tvm::Target target_;
294 std::ostringstream readable_name_stream_;
295 // Index of the global constants
296 static int const_index;
297 // Cache device copy op for equivalence checking to reduce registry lookup
298 // overhead for each invocation of call node when retrieving schedules.
299 const Op& device_copy_op_;

304// Construct a schedule for a given Relay primitive function and target.
305class ScheduleBuilder : public ExprVisitor {
306 public:
307 explicit ScheduleBuilder(Target target) : target_(target) {
308 // Whether to use auto_scheduler schedule.
309 use_auto_scheduler_ = backend::IsAutoSchedulerEnabled();
310 if (backend::IsMetaScheduleEnabled()) {
311 database_ = meta_schedule::Database::Current();
312 CHECK(database_.defined()) << "ValueError: `use_meta_schedule` is enabled in Relay "
313 "build, but no `meta_schedule.Database` context is provided. ";
314 } else {
315 database_ = NullOpt;
316 }
317 }
318
319 CachedFunc Create(const Function& relay_func, GlobalVarSupply global_var_supply) {
320 LowerToTECompute lower_te_compute(target_);
321 Array<te::Tensor> tensor_outs = lower_te_compute.Lower(relay_func);
322 Array<te::Tensor> fn_inputs = lower_te_compute.fn_inputs_;
323 VisitExpr(relay_func->body);
324
325 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
326 // no other GlobalVar ctors should appear inside the lowering machinery.
327 auto prim_fn_var = global_var_supply->FreshGlobal(lower_te_compute.candidate_name_);
328 prim_fn_var->checked_type_ = relay_func->checked_type();
329
330 // Fusion over tupled results may leave identity relationships
331 // between inputs and outputs, copy identity output tensors,
332 // since tir lowering do not support aliasing output to input buffer.
333 for (size_t i = 0; i < tensor_outs.size(); ++i) {
334 if (tensor_outs[i]->op.as<te::PlaceholderOpNode>()) {
335 tensor_outs.Set(i, topi::identity(tensor_outs[i]));
336 }
337 }
338
339 te::Schedule schedule{nullptr};
340 tir::PrimFunc prim_func{nullptr};
341 // No need to register schedule for device copy op.
342 if (anchor_attrs_.as<DeviceCopyAttrs>() == nullptr) {
343 if (use_auto_scheduler_) {
344 const auto* fauto_schedule =
345 runtime::Registry::Get("auto_scheduler.relay_integration.auto_schedule_topi_compute");
346 ICHECK(fauto_schedule != nullptr)
347 << "auto_scheduler.relay_integration.auto_schedule_topi_compute is not registered";
348 ObjectRef obj = (*fauto_schedule)(prim_fn_var->name_hint, tensor_outs);
349 if (obj.defined()) {
350 schedule = Downcast<te::Schedule>(obj);
351 }
352 }
353 if (database_) {
354 using tvm::meta_schedule::TuningRecord;
355 using tvm::tir::IndexMap;
356 using tvm::tir::Instruction;
357 using tvm::tir::InstructionKind;
358 using tvm::tir::PrimFunc;
359 using tvm::tir::Schedule;
360 backend::FTECompilerTIRConverter tir_converter = backend::GetTIRConverter();
361 Array<te::Tensor> te_args = Concat(fn_inputs, tensor_outs);
362 Array<runtime::NDArray> constants;
363 for (auto [const_node, te_tensor] : lower_te_compute.constant_tensors_) {
364 te_args.push_back(te_tensor);
365 constants.push_back(const_node->data);
366 }
367 if (Optional<PrimFunc> f = tir_converter(te_args, constants)) {
368 if (Optional<TuningRecord> opt_record = database_.value()->QueryTuningRecord(
369 /*mod=*/backend::PrimFuncToIRModule(f.value()),
370 /*target=*/target_,
371 /*workload_name=*/prim_fn_var->name_hint)) {
372 static InstructionKind kind_transform_layout = InstructionKind::Get("TransformLayout");
373 TuningRecord record = opt_record.value();
374 for (const Instruction& inst : record->trace->insts) {
375 if (inst->kind.same_as(kind_transform_layout)) {
376 ICHECK_EQ(inst->attrs.size(), 3);
377 MetaScheduleLayoutRewriter::LayoutQueuePush(Downcast<IndexMap>(inst->attrs[2]));
378 }
379 }
380 Schedule sch = Schedule::Traced(record->workload->mod, /*seed=*/-1, /*debug_mask=*/0,
381 tir::ScheduleErrorRenderLevel::kDetail);
382 record->trace->ApplyToSchedule(sch, /*remove_postproc=*/false);
383 IRModule mod = sch->mod();
384 ICHECK_EQ(mod->functions.size(), 1);
385 mod = tir::transform::RemoveWeightLayoutRewriteBlock()(std::move(mod));
386 prim_func = Downcast<PrimFunc>(mod->Lookup("main"));
387 } else {
388 LOG(WARNING) << "Cannot find workload: " << prim_fn_var->name_hint;
389 }
390 }
391 }
392 // Use TOPI schedule if user specified, or the function has no auto_scheduler schedule.
393 if (!schedule.defined() && !prim_func.defined()) {
394 if (anchor_op_.defined()) {
395 auto anchor_impl = lower_te_compute.op_implementations_.find(anchor_op_.operator->());
396 ICHECK(anchor_impl != lower_te_compute.op_implementations_.end());
397 schedule = anchor_impl->second.Schedule(anchor_attrs_, tensor_outs, target_);
398 } else {
399 auto default_sched = GenericFunc::Get("schedule_injective");
400 ICHECK(default_sched.defined()) << "schedule_injective not registered for " << target_;
401 With<Target> tctx(target_);
402 schedule = default_sched(tensor_outs);
403 }
404 }
405 if (schedule.defined()) {
406 for (const auto& scalar : lower_te_compute.scalars_) {
407 if (schedule->Contain(scalar)) {
408 schedule[scalar].compute_inline();
409 }
410 }
411 }
412 }
413
414 IRModule funcs = IRModule(Map<GlobalVar, BaseFunc>({}));
415 return CachedFunc(target_, prim_fn_var, fn_inputs, tensor_outs, schedule, prim_func, {}, funcs,
416 lower_te_compute.constant_tensors_);
417 }
418
419 void VisitExpr_(const CallNode* call_node) final {
420 static auto fpattern = Op::GetAttrMap<TOpPattern>("TOpPattern");
421
422 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
423 Op op = Downcast<Op>(call_node->op);
424
425 for (Expr arg : call_node->args) {
426 VisitExpr(arg);
427 }
428
429 int op_pattern = fpattern[op];
430 if (!use_auto_scheduler_ && !database_.defined() && op_pattern >= kCommReduce) {
431 ICHECK(!anchor_op_.defined() || anchor_op_pattern_ < kCommReduce)
432 << "Cannot apply TOPI schedule to a primitive function with two complicated ops"
433 << " anchor=" << anchor_op_ << " current=" << op;
434 }
435 if (op_pattern >= anchor_op_pattern_) {
436 anchor_op_ = op;
437 anchor_attrs_ = call_node->attrs;
438 anchor_op_pattern_ = op_pattern;
439 }
440 }
441
442 private:
443 tvm::Target target_;
444 Op anchor_op_;
445 Attrs anchor_attrs_;
446 int anchor_op_pattern_{0};
447 bool use_auto_scheduler_;
448 Optional<meta_schedule::Database> database_;
449};

463// Creates shape function from functor.
464class MakeShapeFunc : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
465 public:
466 MakeShapeFunc() {}
467
468 CachedFunc Create(const Function& prim_func, const Target& target,
469 GlobalVarSupply global_var_supply) {
470 VLOG_CONTEXT << "MakeShapeFunc";
471 TShapeDataDependent shape_func_param_states;
472
473 for (auto param : prim_func->params) {
474 param_states_[param] = kNoNeed;
475 Array<tvm::te::Tensor> data_inputs;
476 Array<tvm::te::Tensor> shape_inputs;
477
478 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
479 // Add data placeholder (in case we discover we need it below)
480 Shape shape = GetShape(ttype->shape);
481 tvm::te::Tensor data_tensor = tvm::te::placeholder(shape, ttype->dtype);
482 data_inputs.push_back(data_tensor);
483 // Add shape placeholder (in case we discover we need it below)
484 int64_t ndim = shape.size();
485 Shape sshape;
486 if (ndim > 0) {
487 sshape.push_back(tvm::Integer(ndim));
488 }
489 tvm::te::Tensor shape_tensor = tvm::te::placeholder(sshape, DataType::Int(64));
490 shape_inputs.push_back(shape_tensor);
491 }
492 param_data_[param] = data_inputs;
493 param_shapes_[param] = shape_inputs;
494 }
495
496 // Setup the name;
497 readable_name_stream_ << "shape_func";
498
499 // Create the tensor expressions representing the output shapes.
500 Array<te::Tensor> outputs = VisitExpr(prim_func->body);
501
502 // Generate a name.
503 auto candidate_name = readable_name_stream_.str();
504
505 constexpr static size_t kMaxFuncNameLength = 80;
506 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
507 // whenever the value of kMaxFuncNameLength changes
508 if (candidate_name.size() > kMaxFuncNameLength) {
509 std::stringstream truncated_name;
510 truncated_name << candidate_name.substr(0, kMaxFuncNameLength);
511 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name) << "_";
512 candidate_name = truncated_name.str();
513 }
514
515 // Set all the inputs correctly, and accumulate their types from the p.o.v. of the
516 // shape function rather than the primitive it is derived for.
517 Array<te::Tensor> inputs;
518 Array<Type> shape_function_arg_types;
519 for (auto param : prim_func->params) {
520 int state = param_states_[param];
521 shape_func_param_states.push_back(IntImm(DataType::Int(32), state));
522 if (state & kNeedInputData) {
523 // Pass the primitive arguments directly (though in flattened form and on the host)
524 for (auto t : param_data_[param]) {
525 inputs.push_back(t);
526 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
527 }
528 }
529 if (state & kNeedInputShape) {
530 // Pass the shapes of the primitive arguments (also on the host)
531 for (auto t : param_shapes_[param]) {
532 inputs.push_back(t);
533 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
534 }
535 }
536 }
537
538 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
539 // no other GlobalVar ctors should appear inside the lowering machinery.
540 auto prim_fn_gvar = global_var_supply->FreshGlobal(candidate_name);
541
542 // Gather the result types, again from the p.o.v. of the shape function rather than
543 // the primitive it is derived for.
544 Array<Type> shape_function_res_types;
545 for (const auto& t : outputs) {
546 shape_function_res_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
547 }
548
549 // Assign the shape function its true type.
550 FuncType type(shape_function_arg_types, TupleType(shape_function_res_types),
551 /*type_params=*/{}, /*type_constraints=*/{});
552 VLOG(1) << "shape function '" << prim_fn_gvar->name_hint << "' has type:" << std::endl
553 << PrettyPrint(type) << std::endl
554 << "corresponding to primitive of type:" << std::endl
555 << PrettyPrint(prim_func->checked_type());
556 prim_fn_gvar->checked_type_ = std::move(type);
557
558 // generate schedule for shape func
559 Array<te::Operation> out_ops;
560 for (auto t : outputs) {
561 out_ops.push_back(t->op);
562 }
563 te::Schedule schedule = te::create_schedule(out_ops);
564 tvm::te::AutoInlineInjective(schedule);
565 for (const auto& scalar : scalars_) {
566 auto scalar_op = scalar->op;
567 if (schedule->Contain(scalar_op)) {
568 schedule[scalar_op].compute_inline();
569 }
570 }
571
572 Array<te::Tensor> all_args = Array<te::Tensor>(inputs);
573 for (te::Tensor arg : outputs) {
574 all_args.push_back(arg);
575 }
576
577 using tvm::transform::PassContext;
578 With<PassContext> fresh_pass_ctx_scope(PassContext::Create());
579
580 std::unordered_map<te::Tensor, tir::Buffer> binds;
581 IRModule lowered_module =
582 tvm::LowerSchedule(schedule, all_args, prim_fn_gvar->name_hint, binds, global_var_supply);
583 return CachedFunc(target, prim_fn_gvar, inputs, outputs, schedule, tir::PrimFunc{nullptr},
584 shape_func_param_states, lowered_module);
585 }

587 Array<te::Tensor> VisitExpr(const Expr& expr) final {
588 if (expr.as<VarNode>()) {
589 // Do not memoize vars because shape functions could use either the data
590 // or the shape of a var each time.
591 return ExprFunctor::VisitExpr(expr);
592 }
593 // For other case, do memoized visit
594 return backend::MemoizedExprTranslator<Array<te::Tensor>>::VisitExpr(expr);
595 }
596
597 Array<te::Tensor> VisitExpr_(const VarNode* var_node) final {
598 auto var = GetRef<Var>(var_node);
599 auto it = param_arg_map_.find(var);
600 if (it != param_arg_map_.end()) {
601 // This var is a parameter of a nested function. Visit the corresponding argument in the
602 // function call site.
603 return VisitExpr(it->second);
604 }
605 if (param_states_.find(var) == param_states_.end()) {
606 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(var);
607 return {};
608 } else {
609 ICHECK(data_dependents_per_input_.size());
610 auto data_dependent = data_dependents_per_input_.back();
611 if (data_dependent) {
612 param_states_[var] |= kNeedInputData;
613 return param_data_[var];
614 } else {
615 param_states_[var] |= kNeedInputShape;
616 return param_shapes_[var];
617 }
618 }
619 }
620
621 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
622 using tir::make_const;
623 ICHECK(data_dependents_per_input_.size());
624 bool data_dependent = data_dependents_per_input_.back();
625 if (!op->is_scalar()) {
626 // This is a constant weight, extract the shape of the weight tensor.
627 // This can not be data dependent.
628 CHECK(!data_dependent);
629 auto ttype = op->checked_type().as<TensorTypeNode>();
630 int ndim = static_cast<int>(ttype->shape.size());
631 Array<PrimExpr> out_shape{ndim};
632 te::Tensor value = tvm::te::compute(
633 out_shape,
634 [&](const Array<tvm::tir::Var>& indices) {
635 auto idx = indices[0];
636 PrimExpr ret = make_const(DataType::Int(64), 0);
637 for (int i = 0; i < ndim; i++) {
638 ret = tvm::if_then_else(idx == i, ttype->shape[i], ret);
639 }
640 return ret;
641 },
642 "shape_const", topi::kBroadcast);
643 scalars_.push_back(value);
644 return {value};
645 }
646 if (data_dependent) {
647 void* data = op->data->data;
648 DataType dtype = DataType(op->data->dtype);
649 auto value = tvm::te::compute(
650 {},
651 [&](const Array<tvm::tir::Var>&) {
652 if (dtype == DataType::Int(32)) {
653 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
654 } else if (dtype == DataType::Int(64)) {
655 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
656 } else if (dtype == DataType::Float(32)) {
657 return make_const(dtype, static_cast<const float*>(data)[0]);
658 } else if (dtype == DataType::Float(64)) {
659 return make_const(dtype, static_cast<const double*>(data)[0]);
660 } else if (dtype == DataType::Bool()) {
661 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
662 } else {
663 LOG(FATAL) << "not handled";
664 return tvm::PrimExpr();
665 }
666 },
667 "data_const", topi::kBroadcast);
668 scalars_.push_back(value);
669 return {value};
670 } else {
671 auto value = tvm::te::compute(
672 {}, [&](const Array<tvm::tir::Var>&) { return tir::make_const(DataType::Int(64), 0); },
673 "shape_const", topi::kBroadcast);
674 scalars_.push_back(value);
675 return {value};
676 }
677 }
678
679 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
680 VLOG(1) << "considering call:" << std::endl << PrettyPrint(GetRef<Call>(call_node));
681 if (auto* func = call_node->op.as<FunctionNode>()) {
682 VLOG(1) << "user function";
683 for (size_t i = 0; i < func->params.size(); ++i) {
684 param_arg_map_[func->params[i]] = call_node->args[i];
685 }
686 return VisitExpr(func->body);
687 }
688
689 static auto fshape_func = Op::GetAttrMap<FShapeFunc>("FShapeFunc");
690 static auto tshape_data_dependent = Op::GetAttrMap<TShapeDataDependent>("TShapeDataDependent");
691 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
692 Op op = Downcast<Op>(call_node->op);
693 ICHECK(data_dependents_per_input_.empty() || !data_dependents_per_input_.back())
694 << "Error in op fusion: output of the shape func is fed to a "
695 << "data-dependent shape func";
696 ICHECK_GT(fshape_func.count(op), 0) << "Internal error, cannot find ShapeFunc for " << op->name;
697 ICHECK_GT(tshape_data_dependent.count(op), 0)
698 << "Internal error, cannot find TShapeDataDependent for " << op->name;
699
700 Array<Integer> dep_spec = tshape_data_dependent[op];
701 if (dep_spec.size() == 1) {
702 // This is for cases when data dependence is specified per op
703 // Replicate 0 or 1 flag to all arguments
704 for (size_t i = 1; i < call_node->args.size(); ++i) {
705 dep_spec.push_back(dep_spec[0]);
706 }
707 }
708
709 // Visit all inputs
710 Array<te::Tensor> inputs;
711 int count_tuple = 0;
712 for (size_t i = 0; i < call_node->args.size(); ++i) {
713 Expr arg = call_node->args[i];
714 if (arg->checked_type().as<TupleTypeNode>()) {
715 ++count_tuple;
716 }
717 data_dependents_per_input_.push_back(dep_spec[i]->value != 0);
718 for (te::Tensor tensor : VisitExpr(arg)) {
719 inputs.push_back(tensor);
720 }
721 data_dependents_per_input_.pop_back();
722 }
723 if (count_tuple) {
724 ICHECK_EQ(call_node->args.size(), 1U) << "Only allow function with a single tuple input";
725 }
726 // Get output ndims
727 auto ret_type = call_node->checked_type();
728 Array<IndexExpr> out_ndims;
729 for (const auto& ttype : FlattenTupleType(ret_type)) {
730 out_ndims.push_back(IntImm(DataType::Int(32), ttype->shape.size()));
731 }
732
733 // Call shape function
734 Array<te::Tensor> outputs = fshape_func[op](call_node->attrs, inputs, out_ndims);
735 VLOG(1) << "shape function for '" << op->name << "' with inputs:" << std::endl
736 << inputs << std::endl
737 << "yielded outputs:" << std::endl
738 << outputs;
739 readable_name_stream_ << "_" << op->name;
740 return outputs;
741 }
742
743 Array<te::Tensor> VisitExpr_(const FunctionNode* op) final {
744 LOG(FATAL) << "Nested functions are not allowed to be visited.";
745 return Array<te::Tensor>();
746 }
747
748 Array<te::Tensor> VisitExpr_(const LetNode* op) final {
749 Array<te::Tensor> val = VisitExpr(op->value);
750 ICHECK(!memo_.count(op->var));
751 memo_[op->var] = val;
752 return VisitExpr(op->body);
753 }
754
755 Array<te::Tensor> VisitExpr_(const TupleNode* op) final {
756 Array<te::Tensor> fields;
757 for (Expr field : op->fields) {
758 ICHECK(field->checked_type().as<TensorTypeNode>())
759 << "Expected a Tuple of Tensor, but got " << PrettyPrint(field->checked_type());
760 Array<te::Tensor> res = VisitExpr(field);
761 ICHECK_EQ(res.size(), 1);
762 fields.push_back(res[0]);
763 }
764 return fields;
765 }
766
767 Array<te::Tensor> VisitExpr_(const TupleGetItemNode* op) final {
768 Array<te::Tensor> input_shapes = VisitExpr(op->tuple);
769 Array<te::Tensor> out;
770 out.push_back(input_shapes[op->index]);
771 return out;
772 }

775 /*! \brief String stream for function name */
776 std::ostringstream readable_name_stream_;
777 /*! \brief Map from parameter to its shape function usage state */
778 std::unordered_map<Expr, int, ObjectPtrHash, ObjectPtrEqual> param_states_;
779 /*! \brief Map from parameter to list of data placeholder */
780 std::unordered_map<Expr, Array<te::Tensor>, ObjectPtrHash, ObjectPtrEqual> param_data_;
781 /*! \brief Map from parameter to list of shape placeholder */
782 std::unordered_map<Expr, Array<te::Tensor>, ObjectPtrHash, ObjectPtrEqual> param_shapes_;

788 * call site.

798std::tuple<Array<te::Tensor>, Array<runtime::NDArray>, std::string> LowerTECompute(
799 const Function& source_func, Target target, bool return_inputs) {
800 LowerToTECompute lower_te_compute(target);
801 Array<te::Tensor> outputs = lower_te_compute.Lower(source_func);
802 // Following ScheduleBuilder, remove placeholder ops from outputs.
803 tvm::Array<te::Tensor> tensor_outs;
804 for (const auto& tensor : outputs) {
805 if (!tensor->op.as<te::PlaceholderOpNode>()) {
806 tensor_outs.push_back(tensor);
807 }
808 }
809
810 tvm::Array<runtime::NDArray> constants;
811 for (auto [const_node, te_tensor] : lower_te_compute.constant_tensors_) {
812 tensor_outs.push_back(te_tensor);
813 constants.push_back(const_node->data);
814 }
815
816 if (return_inputs) {
817 return std::make_tuple(Concat(lower_te_compute.fn_inputs_, tensor_outs), constants,
818 lower_te_compute.candidate_name_);
819 }
820 return std::make_tuple(tensor_outs, constants, lower_te_compute.candidate_name_);
821}
