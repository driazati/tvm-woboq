100Array<IndexExpr> GetShape(const Array<IndexExpr>& shape) {
101 // for now, we always use int32 shape when possible
102 // even if the result of shape inference becomes int64.
103 Array<IndexExpr> res;
104 for (IndexExpr val : shape) {
105 const int64_t* pval = tir::as_const_int(val);
106 if (pval != nullptr) {
107#ifndef TVM_INDEX_DEFAULT_I64
108 ICHECK_LE(pval[0], std::numeric_limits<int32_t>::max())
109 << "dimension must be less then int32_t's max value";
110 ICHECK_GE(pval[0], std::numeric_limits<int32_t>::min())
111 << "dimension must be less then int32_t's max value";
112 res.push_back(IntImm(DataType::Int(32), *pval));
113#else
114 res.push_back(val);
115#endif // TVM_INDEX_DEFAULT_I64
116 } else if (val->IsInstance<tir::AnyNode>()) {
117 // currently all 'any' we meet in shape function are non-negative.
118 res.push_back(val.as<tir::AnyNode>()->ToSizeVar());
119 } else {
120 res.push_back(val);
121 }
122 }
123 return res;
124}

126// Helper class that is used during lowering to TE.
127// It matches sequence of Ops and lower them into single TOPI operation. All supported patterns are
128// enumerated in "supported_patterns_".
129class QnnPatternMatcher {
130 public:
131 QnnPatternMatcher()
132 : qnn_conv2d_op_(Op::Get("qnn.conv2d")),
133 qnn_dense_op_(Op::Get("qnn.dense")),
134 qnn_requantize_op_(Op::Get("qnn.requantize")),
135 bias_add_op_(Op::Get("add")) {}
136
137 // Memoize visited operations
138 void Register(const CallNode* call_node) {
139 ICHECK(call_node->op.as<OpNode>());
140 Op op = Downcast<Op>(call_node->op);
141 if (op == qnn_conv2d_op_) {
142 registered_ops_.push_front(P_QConv2d);
143 ICHECK(anchor_op_ == nullptr);
144 anchor_op_ = call_node;
145 } else if (op == qnn_requantize_op_) {
146 registered_ops_.push_front(P_QRequantize);
147 } else if (op == bias_add_op_) {
148 registered_ops_.push_front(P_BiasAdd);
149 } else if (op == qnn_dense_op_) {
150 registered_ops_.push_front(P_QDense);
151 ICHECK(anchor_op_ == nullptr);
152 anchor_op_ = call_node;
153 } else {
154 registered_ops_.push_front(P_Opaque);
155 }
156 }
157
158 // Check whether given Op is a part of matched pattern.
159 bool find(const Op& op) {
160 if (registered_ops_.empty()) return false;
161
162 if (op == qnn_conv2d_op_ || op == qnn_requantize_op_ || op == bias_add_op_ ||
163 op == qnn_dense_op_) {
164 for (const auto& pat : supported_patterns_) {
165 auto it =
166 std::search(registered_ops_.begin(), registered_ops_.end(), pat.begin(), pat.end());
167 if (it != registered_ops_.end()) return true;
168 }
169 }
170 return false;
171 }
172
173 // returns whether given Op is last in the pattern sequence.
174 bool IsLeafOp(const Op& op) { return op == qnn_requantize_op_; }
175
176 const CallNode* GetAnchorOp() { return anchor_op_; }
177
178 void Clear() { registered_ops_.clear(); }
179
180 private:
181 const Op& qnn_conv2d_op_;
182 const Op& qnn_dense_op_;
183 const Op& qnn_requantize_op_;
184 const Op& bias_add_op_;
185
186 // Main (complicated) operation in the primitive (for example qnn.conv2d, qnn.dense etc.).
187 const CallNode* anchor_op_ = nullptr;
188
189 enum POper { P_QConv2d, P_QDense, P_BiasAdd, P_QRequantize, P_Opaque };
190
191 std::deque<POper> registered_ops_;
192
193 const std::vector<std::deque<POper>> supported_patterns_ = {
194 {P_QDense, P_BiasAdd, P_QRequantize}, // Pattern qnn.dense -> bias_add -> qnn.requantize
195 {P_QDense, P_QRequantize}, // Patter qnn.dense -> qnn.requantize
196 {P_QConv2d, P_BiasAdd, P_QRequantize}, // Pattern qnn.conv2d -> bias_add -> qnn.requantize
197 {P_QConv2d, P_QRequantize} // Patter qnn.conv2d -> qnn.requantize
198 };
199};

201// Lowers Relay primitive Function to TE Compute
202class LowerToTECompute : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
203 public:
204 explicit LowerToTECompute(Target target)
205 : target_(target), device_copy_op_(Op::Get("device_copy")) {}

207 Array<te::Tensor> Lower(const Function& relay_func) {
208 for (Var param : relay_func->params) {
209 Array<tvm::te::Tensor> inputs;
210 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
211 auto name_hint = param->vid->name_hint;
212 tvm::te::Tensor tensor = tvm::te::placeholder(
213 GetShape(ttype->shape), ttype->dtype, (name_hint == "") ? "placeholder" : name_hint);
214 inputs.push_back(tensor);
215 fn_inputs_.push_back(tensor);
216 }
217 memo_[param] = inputs;
218 }
219 readable_name_stream_ << "fused";
220
221 Array<te::Tensor> outputs = this->VisitExpr(relay_func->body);
222
223 candidate_name_ = readable_name_stream_.str();
224
225 constexpr static size_t kMaxFuncNameLength = 80;
226 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
227 // whenever the value of kMaxFuncNameLength changes
228 if (candidate_name_.size() > kMaxFuncNameLength) {
229 std::stringstream truncated_name;
230 truncated_name << candidate_name_.substr(0, kMaxFuncNameLength);
231 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name_) << "_";
232 candidate_name_ = truncated_name.str();
233 }
234
235 return outputs;
236 }

238 Array<te::Tensor> VisitExpr_(const VarNode* op) final {
239 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(GetRef<Var>(op));
240 return {};
241 }
242
243 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
244 using tir::make_const;
245 void* data = op->data->data;
246 DataType dtype = DataType(op->data->dtype);
247 if (op->is_scalar()) {
248 auto value = te::compute(
249 {},
250 [&](const Array<tvm::tir::Var>&) {
251 if (dtype == DataType::Int(16)) {
252 return make_const(dtype, static_cast<const int16_t*>(data)[0]);
253 } else if (dtype == DataType::Int(8)) {
254 return make_const(dtype, static_cast<const int8_t*>(data)[0]);
255 } else if (dtype == DataType::UInt(8) || dtype == DataType::Bool()) {
256 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
257 } else if (dtype == DataType::Int(32)) {
258 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
259 } else if (dtype == DataType::Int(64)) {
260 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
261 } else if (dtype == DataType::Float(16)) {
262 return make_const(dtype, __gnu_h2f_ieee(static_cast<const uint16_t*>(data)[0]));
263 } else if (dtype == DataType::Float(32)) {
264 return make_const(dtype, static_cast<const float*>(data)[0]);
265 } else if (dtype == DataType::Float(64)) {
266 return make_const(dtype, static_cast<const double*>(data)[0]);
267 } else {
268 LOG(FATAL) << dtype << " not handled";
269 return tvm::PrimExpr();
270 }
271 },
272 "compile_engine_const", topi::kBroadcast);
273 scalars_.push_back(value->op);
274 return {value};
275 } else {
276 const auto* ttype = op->checked_type().as<TensorTypeNode>();
277 std::stringstream ss;
278 std::string s = readable_name_stream_.str();
279 std::replace(s.begin(), s.end(), '.', '_');
280 ss << s << "_constant_" << const_index++;
281 tvm::te::Tensor tensor = tvm::te::placeholder(GetShape(ttype->shape), ttype->dtype, ss.str());
282 constant_tensors_[op] = tensor;
283 return {tensor};
284 }
285 }
286
287 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
288 static auto flower_call = tvm::runtime::Registry::Get("relay.backend.lower_call");
289 ICHECK(flower_call) << "relay.backend.lower_call is not registered.";
290
291 pattern_matcher_.Register(call_node);
292
293 Array<te::Tensor> inputs;
294 int count_tuple = 0;
295 for (Expr arg : call_node->args) {
296 if (arg->checked_type().as<TupleTypeNode>()) {
297 ++count_tuple;
298 }
299 for (te::Tensor tensor : VisitExpr(arg)) {
300 inputs.push_back(tensor);
301 }
302 }
303
304 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
305 Op op = Downcast<Op>(call_node->op);
306
307 // TODO(mbs): device_copy cleanup
308 ICHECK_NE(op, device_copy_op_) << "device_copy cannot be lowered";
309
310 Array<te::Tensor> outputs;
311
312 if (pattern_matcher_.find(op)) {
313 if (pattern_matcher_.IsLeafOp(op)) {
314 // Lower anchor op when pattern leaf op was reached
315 auto anchor_op = pattern_matcher_.GetAnchorOp();
316 LoweredOutput lowered_out =
317 (*flower_call)(GetRef<Call>(anchor_op), inputs, target_, call_node->checked_type());
318 outputs = lowered_out->outputs;
319 Op a_op = Downcast<Op>(anchor_op->op);
320 op_implementations_[a_op.operator->()] = lowered_out->implementation;
321
322 pattern_matcher_.Clear();
323 } else {
324 // Forward inputs as "outputs" for successor.
325 readable_name_stream_ << '_' << op->name;
326 return inputs;
327 }
328 } else {
329 LoweredOutput lowered_out = (*flower_call)(GetRef<Call>(call_node), inputs, target_);
330 outputs = lowered_out->outputs;
331 op_implementations_[op.operator->()] = lowered_out->implementation;
332 }
333
334 if (outputs.size() != 1) {
335 const auto* tuple_type = call_node->checked_type().as<TupleTypeNode>();
336 ICHECK(tuple_type) << "Expected output to be a tuple type "
337 << PrettyPrint(call_node->checked_type());
338
339 ICHECK_EQ(tuple_type->fields.size(), outputs.size());
340 }
341
342 readable_name_stream_ << '_' << op->name;
343 return outputs;
344 }
345
346 Array<te::Tensor> VisitExpr_(const FunctionNode* op) final {
347 LOG(FATAL) << "Primitive Functions can not contain nested functions.";
348 return Array<te::Tensor>();
349 }
350
351 Array<te::Tensor> VisitExpr_(const LetNode* op) final {
352 Array<te::Tensor> val = VisitExpr(op->value);
353 ICHECK(!memo_.count(op->var));
354 memo_[op->var] = val;
355 return VisitExpr(op->body);
356 }
357
358 Array<te::Tensor> VisitExpr_(const TupleNode* op) final {
359 Array<te::Tensor> fields;
360 for (Expr field : op->fields) {
361 // TODO(mbs): Generalize to be equivalent to FlattenTupleType.
362 ICHECK(field->checked_type().as<TensorTypeNode>()) << "Only allow Tuple of Tensor";
363 Array<te::Tensor> res = VisitExpr(field);
364 ICHECK_EQ(res.size(), 1);
365 fields.push_back(res[0]);
366 }
367 return fields;
368 }
369
370 Array<te::Tensor> VisitExpr_(const TupleGetItemNode* op) final {
371 const auto* tuple_type = op->tuple->type_as<TupleTypeNode>();
372 Array<te::Tensor> tuple = VisitExpr(op->tuple);
373 ICHECK_EQ(tuple_type->fields.size(), tuple.size());
374 ICHECK_GE(op->index, 0);
375 ICHECK_LT(static_cast<size_t>(op->index), tuple.size());
376 return {tuple[op->index]};
377 }
378
379 public:
380 // Additional outputs
381 Array<tvm::te::Tensor> fn_inputs_;
382 Array<te::Operation> scalars_;
383 std::unordered_map<const ConstantNode*, te::Tensor> constant_tensors_;
384 std::unordered_map<const OpNode*, OpImplementation> op_implementations_;
385 std::string candidate_name_;
386
387 private:
388 QnnPatternMatcher pattern_matcher_;
389
390 tvm::Target target_;
391 std::ostringstream readable_name_stream_;
392 // Index of the global constants
393 static int const_index;
394 // Cache device copy op for equivalence checking to reduce registry lookup
395 // overhead for each invocation of call node when retrieving schedules.
396 const Op& device_copy_op_;
397};

403class LayoutFreeConstantCollector : public StmtVisitor {
404 public:
405 Array<runtime::NDArray> constants;
406
407 private:
408 void VisitStmt_(const BlockNode* op) final {
409 StmtVisitor::VisitStmt_(op);
410 if (Optional<ObjectRef> ann = op->annotations.Get("layout_free_placeholders")) {
411 for (Buffer buffer : Downcast<Array<Buffer>>(ann)) {
412 layout_free_buffer_vars_.insert(buffer->data.get());
413 }
414 }
415 }
416
417 void VisitStmt_(const AllocateConstNode* op) final {
418 StmtVisitor::VisitStmt_(op);
419 if (auto it = layout_free_buffer_vars_.find(op->buffer_var.get());
420 it != layout_free_buffer_vars_.end()) {
421 constants.push_back(op->data.value());
422 }
423 }
424
425 std::unordered_set<const tir::VarNode*> layout_free_buffer_vars_;
426};

428using NDArrayMap =
429 std::unordered_map<runtime::NDArray, runtime::NDArray, ObjectPtrHash, ObjectPtrEqual>;

431// Replace constants in AllocateConst nodes according to the given mapping
432class AllocateConstReplaceConstant : public StmtExprMutator {
433 public:
434 explicit AllocateConstReplaceConstant(const NDArrayMap& constant_map)
435 : constant_map_(constant_map) {}
436
437 static PrimFunc Rewrite(PrimFunc f, const NDArrayMap& constant_map) {
438 AllocateConstReplaceConstant rewriter(constant_map);
439 PrimFuncNode* n = f.CopyOnWrite();
440 n->body = rewriter(std::move(n->body));
441 return f;
442 }
443
444 private:
445 Stmt VisitStmt_(const AllocateConstNode* op) final {
446 if (auto it = constant_map_.find(op->data.value()); it != constant_map_.end()) {
447 auto rewriten_constant = it->second;
448 Array<PrimExpr> rewritten_extents;
449 for (auto s : rewriten_constant.Shape()) {
450 rewritten_extents.push_back(PrimExpr(static_cast<int>(s)));
451 }
452 return AllocateConst(op->buffer_var, op->dtype, rewritten_extents, rewriten_constant,
453 op->body, op->annotations, op->span);
454 }
455 return StmtExprMutator::VisitStmt_(op);
456 }
457
458 NDArrayMap constant_map_;
459};

461// Construct a schedule for a given Relay primitive function and target.
462class ScheduleBuilder : public ExprVisitor {
463 public:
464 explicit ScheduleBuilder(Target target) : target_(target) {
465 // Whether to use auto_scheduler schedule.
466 use_auto_scheduler_ = backend::IsAutoSchedulerEnabled();
467 if (backend::IsMetaScheduleEnabled()) {
468 database_ = meta_schedule::Database::Current();
469 CHECK(database_.defined()) << "ValueError: `use_meta_schedule` is enabled in Relay "
470 "build, but no `meta_schedule.Database` context is provided. ";
471 } else {
472 database_ = NullOpt;
473 }
474 }
475
476 CachedFunc Create(const Function& relay_func, GlobalVarSupply global_var_supply) {
477 LowerToTECompute lower_te_compute(target_);
478 Array<te::Tensor> tensor_outs = lower_te_compute.Lower(relay_func);
479 Array<te::Tensor> fn_inputs = lower_te_compute.fn_inputs_;
480 VisitExpr(relay_func->body);
481
482 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
483 // no other GlobalVar ctors should appear inside the lowering machinery.
484 auto prim_fn_var = global_var_supply->FreshGlobal(lower_te_compute.candidate_name_);
485 prim_fn_var->checked_type_ = relay_func->checked_type();
486
487 // Fusion over tupled results may leave identity relationships
488 // between inputs and outputs, copy identity output tensors,
489 // since tir lowering do not support aliasing output to input buffer.
490 for (size_t i = 0; i < tensor_outs.size(); ++i) {
491 if (tensor_outs[i]->op.as<te::PlaceholderOpNode>()) {
492 tensor_outs.Set(i, topi::identity(tensor_outs[i]));
493 }
494 }
495
496 te::Schedule schedule{nullptr};
497 tir::PrimFunc prim_func{nullptr};
498 // No need to register schedule for device copy op.
499 if (anchor_attrs_.as<DeviceCopyAttrs>() == nullptr) {
500 if (use_auto_scheduler_) {
501 const auto* fauto_schedule =
502 runtime::Registry::Get("auto_scheduler.relay_integration.auto_schedule_topi_compute");
503 ICHECK(fauto_schedule != nullptr)
504 << "auto_scheduler.relay_integration.auto_schedule_topi_compute is not registered";
505 ObjectRef obj = (*fauto_schedule)(prim_fn_var->name_hint, tensor_outs);
506 if (obj.defined()) {
507 schedule = Downcast<te::Schedule>(obj);
508 }
509 }
510 if (database_) {
511 using tvm::meta_schedule::TuningRecord;
512 using tvm::tir::IndexMap;
513 using tvm::tir::Instruction;
514 using tvm::tir::InstructionKind;
515 using tvm::tir::PrimFunc;
516 using tvm::tir::Schedule;
517 backend::FTECompilerTIRConverter tir_converter = backend::GetTIRConverter();
518 Array<te::Tensor> te_args = Concat(fn_inputs, tensor_outs);
519 Array<runtime::NDArray> constants;
520 for (auto [const_node, te_tensor] : lower_te_compute.constant_tensors_) {
521 te_args.push_back(te_tensor);
522 constants.push_back(const_node->data);
523 }
524 if (Optional<PrimFunc> f = tir_converter(te_args, constants)) {
525 IRModule query_mod = backend::PrimFuncToIRModule(f.value());
526 if (Optional<TuningRecord> opt_record = database_.value()->QueryTuningRecord(
527 /*mod=*/query_mod,
528 /*target=*/target_,
529 /*workload_name=*/prim_fn_var->name_hint)) {
530 LayoutFreeConstantCollector const_collector;
531 const_collector(f.value()->body);
532
533 static InstructionKind kind_transform_layout = InstructionKind::Get("TransformLayout");
534 TuningRecord record = opt_record.value();
535 for (const Instruction& inst : record->trace->insts) {
536 if (inst->kind.same_as(kind_transform_layout)) {
537 ICHECK_EQ(inst->attrs.size(), 4);
538 auto index_map = Downcast<IndexMap>(inst->attrs[2]);
539
540 if (!const_collector.constants.empty()) {
541 // In this case, RewriteLayout is acting on an AllocateConst node.
542 // After tuning, we reach this code path twice: First by
543 // the Relay MetaScheduleLayoutRewrite pass, and next by the final
544 // compilation (Relay to TE schedule lowering).
545 //
546 // Due to Relay MetaScheduleLayoutRewrite and FoldConstant passes,
547 // the Relay subgraph for which we query the database during the
548 // final compilation has its weight tensor transformed according to
549 // the index map, determined during tuning. For example,
550 //
551 // fn (%p0: Tensor[(1, 56, 56, 64), float32]) {
552 // %0 = nn.conv2d(%p0, meta[relay.Constant][0],
553 // /*ty=Tensor[(4, 2, 2, 3, 3, 32, 8), float32]*/, ...);
554 // add(%0, meta[relay.Constant][1])
555 // }
556 //
557 // Note that the database does not have an entry corresponding to such subgraphs,
558 // since an input subgraph to the tuning system always has its weight tensor in
559 // the original layout, e.g.
560 //
561 // fn (%p0: Tensor[(1, 56, 56, 64), float32]) {
562 // %0 = nn.conv2d(%p0, meta[relay.Constant][0],
563 // /*ty=Tensor[(3, 3, 64, 64), float32]*/, ...);
564 // add(%0, meta[relay.Constant][1])
565 // }
566 //
567 // Thus, in both of the two cases where we reach this code path, we need careful
568 // logic to make sure that (1) the database lookup during the final compilation
569 // succeeds and (2) the application of a schedule trace is well defined.
570
571 ICHECK(const_collector.constants.size() == 1)
572 << "Only one layout-free constant is supported by RewriteLayout for now";
573 auto constant = const_collector.constants[0];
574
575 if (constant.Shape().size() == index_map->initial_indices.size()) {
576 // This is the first case, reached during the MetaScheduleLayoutRewrite pass.
577 //
578 // A layout-free constant having the same rank as an input to the index map
579 // is assumed to be transformed by this index map.
580 // TODO(masahi): If there are multiple layout-free constants in one
581 // TIR mod (e.g. conv2d -> conv2d fusion), this assumption does not hold.
582 // We need to determine which constant the given index map acts on.
583 //
584 // We know that, during the final compilation, we will query the database
585 // for a subgraph that the tuner has never seen. We workaround this problem
586 // by adding a dummy entry to the database. The dummy entry is carefully
587 // constructed so that the lookup during the final compilation would succeed.
588 runtime::NDArray rewritten_constant = index_map->MapNDArray(constant);
589 auto f_dummy = AllocateConstReplaceConstant::Rewrite(
590 f.value(), {{constant, rewritten_constant}});
591 auto workload_dummy =
592 database_.value()->CommitWorkload(backend::PrimFuncToIRModule(f_dummy));
593 TuningRecord rec_dummy(record->trace, workload_dummy, record->run_secs,
594 record->target, record->args_info);
595 database_.value()->CommitTuningRecord(rec_dummy);
596 } else {
597 // The constant is already transformed, so this is the second case, reached
598 // during the final compilation.
599 //
600 // The schedule trace is supposed to be applied to the weight in its original
601 // layout. But as explained above, the Relay subgraph we get in this case
602 // has its weight tensor transformed according to the corresponding index map.
603 // So effectively, we undo the layout transformation on the weight to restore
604 // the original PrimFunc that the schedule trace is supposed to act on.
605 ICHECK(index_map->inverse_index_map);
606 auto inverse_map = Downcast<IndexMap>(index_map->inverse_index_map.value());
607 ICHECK(constant.Shape().size() == inverse_map->initial_indices.size());
608 runtime::NDArray orig_constant = inverse_map->MapNDArray(constant);
609 auto f_ = AllocateConstReplaceConstant::Rewrite(f.value(),
610 {{constant, orig_constant}});
611 query_mod = backend::PrimFuncToIRModule(f_);
612 }
613 }
614 MetaScheduleLayoutRewriter::LayoutQueuePush(index_map);
615 }
616 }
617 Schedule sch = Schedule::Traced(query_mod, /*seed=*/-1, /*debug_mask=*/0,
618 tir::ScheduleErrorRenderLevel::kDetail);
619 record->trace->ApplyToSchedule(sch, /*remove_postproc=*/false);
620 IRModule mod = sch->mod();
621 ICHECK_EQ(mod->functions.size(), 1);
622 mod = tir::transform::RemoveWeightLayoutRewriteBlock(/*skip_ndarray_rewrite*/ false)(
623 std::move(mod));
624 prim_func = Downcast<PrimFunc>(mod->Lookup("main"));
625 // Need to copy attrs from relay function over to prim func. Most notably the structural
626 // hash.
627 prim_func = WithAttrs(prim_func, relay_func->attrs->dict);
628 } else {
629 int dispatch = backend::UseMetaScheduleDispatch();
630 // (dispatch & 2): controls whether to print TVMScript for missing TIR
631 // (dispatch & 4): controls whether to raise fatal errors for missing TIR
632 if (dispatch & 2) {
633 LOG(WARNING) << "Cannot find workload: " << prim_fn_var->name_hint << "\n"
634 << tir::AsTVMScript(f.value());
635 } else {
636 LOG(WARNING) << "Cannot find workload: " << prim_fn_var->name_hint;
637 }
638 if (dispatch & 4) {
639 LOG(FATAL);
640 }
641 }
642 }
643 }
644 // Use TOPI schedule if user specified, or the function has no auto_scheduler schedule.
645 if (!schedule.defined() && !prim_func.defined()) {
646 if (anchor_op_.defined()) {
647 auto anchor_impl = lower_te_compute.op_implementations_.find(anchor_op_.operator->());
648 ICHECK(anchor_impl != lower_te_compute.op_implementations_.end());
649 schedule = anchor_impl->second.Schedule(anchor_attrs_, tensor_outs, target_);
650 } else {
651 auto default_sched = GenericFunc::Get("schedule_injective");
652 ICHECK(default_sched.defined()) << "schedule_injective not registered for " << target_;
653 With<Target> tctx(target_);
654 schedule = default_sched(tensor_outs);
655 }
656 }
657 if (schedule.defined()) {
658 for (const auto& scalar : lower_te_compute.scalars_) {
659 if (schedule->Contain(scalar)) {
660 schedule[scalar].compute_inline();
661 }
662 }
663 }
664 }
665
666 IRModule funcs = IRModule(Map<GlobalVar, BaseFunc>({}));
667 return CachedFunc(target_, prim_fn_var, fn_inputs, tensor_outs, schedule, prim_func, {}, funcs,
668 lower_te_compute.constant_tensors_);
669 }
670
671 void VisitExpr_(const CallNode* call_node) final {
672 static auto fpattern = Op::GetAttrMap<TOpPattern>("TOpPattern");
673
674 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
675 Op op = Downcast<Op>(call_node->op);
676
677 for (Expr arg : call_node->args) {
678 VisitExpr(arg);
679 }
680
681 int op_pattern = fpattern[op];
682 if (!use_auto_scheduler_ && !database_.defined() && op_pattern >= kCommReduce) {
683 ICHECK(!anchor_op_.defined() || anchor_op_pattern_ < kCommReduce)
684 << "Cannot apply TOPI schedule to a primitive function with two complicated ops"
685 << " anchor=" << anchor_op_ << " current=" << op;
686 }
687 if (op_pattern >= anchor_op_pattern_) {
688 anchor_op_ = op;
689 anchor_attrs_ = call_node->attrs;
690 anchor_op_pattern_ = op_pattern;
691 }
692 }
693
694 private:
695 tvm::Target target_;
696 Op anchor_op_;
697 Attrs anchor_attrs_;
698 int anchor_op_pattern_{0};
699 bool use_auto_scheduler_;
700 Optional<meta_schedule::Database> database_;
701};

715// Creates shape function from functor.
716class MakeShapeFunc : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
717 public:
718 MakeShapeFunc() {}
719
720 CachedFunc Create(const Function& prim_func, const Target& target,
721 GlobalVarSupply global_var_supply) {
722 VLOG_CONTEXT << "MakeShapeFunc";
723 TShapeDataDependent shape_func_param_states;
724
725 for (auto param : prim_func->params) {
726 param_states_[param] = kNoNeed;
727 Array<tvm::te::Tensor> data_inputs;
728 Array<tvm::te::Tensor> shape_inputs;
729
730 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
731 // Add data placeholder (in case we discover we need it below)
732 Shape shape = GetShape(ttype->shape);
733 tvm::te::Tensor data_tensor =
734 tvm::te::placeholder(shape, ttype->dtype, "data_" + param->vid->name_hint);
735 data_inputs.push_back(data_tensor);
736 // Add shape placeholder (in case we discover we need it below)
737 int64_t ndim = shape.size();
738 Shape sshape;
739 if (ndim > 0) {
740 sshape.push_back(tvm::Integer(ndim));
741 }
742 tvm::te::Tensor shape_tensor =
743 tvm::te::placeholder(sshape, DataType::Int(64), "shape_" + param->vid->name_hint);
744 shape_inputs.push_back(shape_tensor);
745 }
746 param_data_[param] = data_inputs;
747 param_shapes_[param] = shape_inputs;
748 }
749
750 // Setup the name;
751 readable_name_stream_ << "shape_func";
752
753 // Create the tensor expressions representing the output shapes.
754 Array<te::Tensor> outputs = VisitExpr(prim_func->body);
755
756 // Generate a name.
757 auto candidate_name = readable_name_stream_.str();
758
759 constexpr static size_t kMaxFuncNameLength = 80;
760 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
761 // whenever the value of kMaxFuncNameLength changes
762 if (candidate_name.size() > kMaxFuncNameLength) {
763 std::stringstream truncated_name;
764 truncated_name << candidate_name.substr(0, kMaxFuncNameLength);
765 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name) << "_";
766 candidate_name = truncated_name.str();
767 }
768
769 // Set all the inputs correctly, and accumulate their types from the p.o.v. of the
770 // shape function rather than the primitive it is derived for.
771 Array<te::Tensor> inputs;
772 Array<Type> shape_function_arg_types;
773 for (auto param : prim_func->params) {
774 int state = param_states_[param];
775 shape_func_param_states.push_back(IntImm(DataType::Int(32), state));
776 if (state & kNeedInputData) {
777 // Pass the primitive arguments directly (though in flattened form and on the host)
778 for (auto t : param_data_[param]) {
779 inputs.push_back(t);
780 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
781 }
782 }
783 if (state & kNeedInputShape) {
784 // Pass the shapes of the primitive arguments (also on the host)
785 for (auto t : param_shapes_[param]) {
786 inputs.push_back(t);
787 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
788 }
789 }
790 }
791
792 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
793 // no other GlobalVar ctors should appear inside the lowering machinery.
794 auto prim_fn_gvar = global_var_supply->FreshGlobal(candidate_name);
795
796 // Gather the result types, again from the p.o.v. of the shape function rather than
797 // the primitive it is derived for.
798 Array<Type> shape_function_res_types;
799 for (const auto& t : outputs) {
800 shape_function_res_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
801 }
802
803 // Assign the shape function its true type.
804 FuncType type(shape_function_arg_types, TupleType(shape_function_res_types),
805 /*type_params=*/{}, /*type_constraints=*/{});
806 VLOG(1) << "shape function '" << prim_fn_gvar->name_hint << "' has type:" << std::endl
807 << PrettyPrint(type) << std::endl
808 << "corresponding to primitive of type:" << std::endl
809 << PrettyPrint(prim_func->checked_type());
810 prim_fn_gvar->checked_type_ = std::move(type);
811
812 // generate schedule for shape func
813 Array<te::Operation> out_ops;
814 for (auto t : outputs) {
815 out_ops.push_back(t->op);
816 }
817 te::Schedule schedule = te::create_schedule(out_ops);
818 tvm::te::AutoInlineInjective(schedule);
819 for (const auto& scalar : scalars_) {
820 auto scalar_op = scalar->op;
821 if (schedule->Contain(scalar_op)) {
822 schedule[scalar_op].compute_inline();
823 }
824 }
825
826 Array<te::Tensor> all_args = Array<te::Tensor>(inputs);
827 for (te::Tensor arg : outputs) {
828 all_args.push_back(arg);
829 }
830
831 using tvm::transform::PassContext;
832 With<PassContext> fresh_pass_ctx_scope(PassContext::Create());
833
834 std::unordered_map<te::Tensor, tir::Buffer> binds;
835 IRModule lowered_module =
836 tvm::LowerSchedule(schedule, all_args, prim_fn_gvar->name_hint, binds, global_var_supply);
837 return CachedFunc(target, prim_fn_gvar, inputs, outputs, schedule, tir::PrimFunc{nullptr},
838 shape_func_param_states, lowered_module);
839 }

841 Array<te::Tensor> VisitExpr(const Expr& expr) final {
842 if (expr.as<VarNode>()) {
843 // Do not memoize vars because shape functions could use either the data
844 // or the shape of a var each time.
845 return ExprFunctor::VisitExpr(expr);
846 }
847 // For other case, do memoized visit
848 return backend::MemoizedExprTranslator<Array<te::Tensor>>::VisitExpr(expr);
849 }
850
851 Array<te::Tensor> VisitExpr_(const VarNode* var_node) final {
852 auto var = GetRef<Var>(var_node);
853 auto it = param_arg_map_.find(var);
854 if (it != param_arg_map_.end()) {
855 // This var is a parameter of a nested function. Visit the corresponding argument in the
856 // function call site.
857 return VisitExpr(it->second);
858 }
859 if (param_states_.find(var) == param_states_.end()) {
860 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(var);
861 return {};
862 } else {
863 ICHECK(data_dependents_per_input_.size());
864 auto data_dependent = data_dependents_per_input_.back();
865 if (data_dependent) {
866 param_states_[var] |= kNeedInputData;
867 return param_data_[var];
868 } else {
869 param_states_[var] |= kNeedInputShape;
870 return param_shapes_[var];
871 }
872 }
873 }
874
875 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
876 using tir::make_const;
877 ICHECK(data_dependents_per_input_.size());
878 bool data_dependent = data_dependents_per_input_.back();
879 if (!op->is_scalar()) {
880 // This is a constant weight, extract the shape of the weight tensor.
881 // This can not be data dependent.
882 CHECK(!data_dependent);
883 auto ttype = op->checked_type().as<TensorTypeNode>();
884 int ndim = static_cast<int>(ttype->shape.size());
885 Array<PrimExpr> out_shape{ndim};
886 te::Tensor value = tvm::te::compute(
887 out_shape,
888 [&](const Array<tvm::tir::Var>& indices) {
889 auto idx = indices[0];
890 PrimExpr ret = make_const(DataType::Int(64), 0);
891 for (int i = 0; i < ndim; i++) {
892 ret = tvm::if_then_else(idx == i, ttype->shape[i], ret);
893 }
894 return ret;
895 },
896 "shape_const", topi::kBroadcast);
897 scalars_.push_back(value);
898 return {value};
899 }
900 if (data_dependent) {
901 void* data = op->data->data;
902 DataType dtype = DataType(op->data->dtype);
903 auto value = tvm::te::compute(
904 {},
905 [&](const Array<tvm::tir::Var>&) {
906 if (dtype == DataType::Int(32)) {
907 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
908 } else if (dtype == DataType::Int(64)) {
909 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
910 } else if (dtype == DataType::Float(32)) {
911 return make_const(dtype, static_cast<const float*>(data)[0]);
912 } else if (dtype == DataType::Float(64)) {
913 return make_const(dtype, static_cast<const double*>(data)[0]);
914 } else if (dtype == DataType::Bool()) {
915 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
916 } else {
917 LOG(FATAL) << "not handled";
918 return tvm::PrimExpr();
919 }
920 },
921 "data_const", topi::kBroadcast);
922 scalars_.push_back(value);
923 return {value};
924 } else {
925 auto value = tvm::te::compute(
926 {}, [&](const Array<tvm::tir::Var>&) { return tir::make_const(DataType::Int(64), 0); },
927 "shape_const", topi::kBroadcast);
928 scalars_.push_back(value);
929 return {value};
930 }
931 }

933 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
934 VLOG(1) << "considering call:" << std::endl << PrettyPrint(GetRef<Call>(call_node));
935 if (auto* func = call_node->op.as<FunctionNode>()) {
936 VLOG(1) << "user function";
937 for (size_t i = 0; i < func->params.size(); ++i) {
938 param_arg_map_[func->params[i]] = call_node->args[i];
939 }
940 return VisitExpr(func->body);
941 }
942
943 static auto fshape_func = Op::GetAttrMap<FShapeFunc>("FShapeFunc");
944 static auto tshape_data_dependent = Op::GetAttrMap<TShapeDataDependent>("TShapeDataDependent");
945 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
946 Op op = Downcast<Op>(call_node->op);
947 ICHECK(data_dependents_per_input_.empty() || !data_dependents_per_input_.back())
948 << "Error in op fusion: output of the shape func is fed to a "
949 << "data-dependent shape func";
950 ICHECK_GT(fshape_func.count(op), 0) << "Internal error, cannot find ShapeFunc for " << op->name;
951 ICHECK_GT(tshape_data_dependent.count(op), 0)
952 << "Internal error, cannot find TShapeDataDependent for " << op->name;
953
954 Array<Integer> dep_spec = tshape_data_dependent[op];
955 if (dep_spec.size() == 1) {
956 // This is for cases when data dependence is specified per op
957 // Replicate 0 or 1 flag to all arguments
958 for (size_t i = 1; i < call_node->args.size(); ++i) {
959 dep_spec.push_back(dep_spec[0]);
960 }
961 }
962
963 // Visit all inputs
964 Array<te::Tensor> inputs;
965 int count_tuple = 0;
966 for (size_t i = 0; i < call_node->args.size(); ++i) {
967 Expr arg = call_node->args[i];
968 if (arg->checked_type().as<TupleTypeNode>()) {
969 ++count_tuple;
970 }
971 data_dependents_per_input_.push_back(dep_spec[i]->value != 0);
972 for (te::Tensor tensor : VisitExpr(arg)) {
973 inputs.push_back(tensor);
974 }
975 data_dependents_per_input_.pop_back();
976 }
977 if (count_tuple) {
978 ICHECK_EQ(call_node->args.size(), 1U) << "Only allow function with a single tuple input";
979 }
980 // Get output ndims
981 auto ret_type = call_node->checked_type();
982 Array<IndexExpr> out_ndims;
983 for (const auto& ttype : FlattenTupleType(ret_type)) {
984 out_ndims.push_back(IntImm(DataType::Int(32), ttype->shape.size()));
985 }
986
987 // Call shape function
988 Array<te::Tensor> outputs = fshape_func[op](call_node->attrs, inputs, out_ndims);
989 VLOG(1) << "shape function for '" << op->name << "' with inputs:" << std::endl
990 << inputs << std::endl
991 << "yielded outputs:" << std::endl
992 << outputs;
993 readable_name_stream_ << "_" << op->name;
994 return outputs;
995 }

998 LOG(FATAL) << "Nested functions are not allowed to be visited.";
999 return Array<te::Tensor>();

1052std::tuple<Array<te::Tensor>, Array<runtime::NDArray>, std::string> LowerTECompute(
1053 const Function& source_func, Target target, bool return_inputs) {
1054 LowerToTECompute lower_te_compute(target);
1055 Array<te::Tensor> outputs = lower_te_compute.Lower(source_func);
1056 // Following ScheduleBuilder, remove placeholder ops from outputs.
1057 tvm::Array<te::Tensor> tensor_outs;
1058 for (const auto& tensor : outputs) {
1059 if (!tensor->op.as<te::PlaceholderOpNode>()) {
1060 tensor_outs.push_back(tensor);
1061 }
1062 }
1063
1064 tvm::Array<runtime::NDArray> constants;
1065 for (auto [const_node, te_tensor] : lower_te_compute.constant_tensors_) {
1066 tensor_outs.push_back(te_tensor);
1067 constants.push_back(const_node->data);
1068 }
1069
1070 if (return_inputs) {
1071 return std::make_tuple(Concat(lower_te_compute.fn_inputs_, tensor_outs), constants,
1072 lower_te_compute.candidate_name_);
1073 }
1074 return std::make_tuple(tensor_outs, constants, lower_te_compute.candidate_name_);
1075}
