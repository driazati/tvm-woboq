65void ExtractTransformLayout(const meta_schedule::TuningRecord& record) {
66 static tir::InstructionKind kind_transform_layout = tir::InstructionKind::Get("TransformLayout");
67 for (const tir::Instruction& inst : record->trace->insts) {
68 if (inst->kind.same_as(kind_transform_layout)) {
69 ICHECK_EQ(inst->attrs.size(), 3);
70 relay::MetaScheduleLayoutRewriter::LayoutQueuePush(Downcast<tir::IndexMap>(inst->attrs[2]));
71 }
72 }
73}

108Array<IndexExpr> GetShape(const Array<IndexExpr>& shape) {
109 // for now, we always use int32 shape when possible
110 // even if the result of shape inference becomes int64.
111 Array<IndexExpr> res;
112 for (IndexExpr val : shape) {
113 const int64_t* pval = tir::as_const_int(val);
114 if (pval != nullptr) {
115#ifndef TVM_INDEX_DEFAULT_I64
116 ICHECK_LE(pval[0], std::numeric_limits<int32_t>::max())
117 << "dimension must be less then int32_t's max value";
118 ICHECK_GE(pval[0], std::numeric_limits<int32_t>::min())
119 << "dimension must be less then int32_t's max value";
120 res.push_back(IntImm(DataType::Int(32), *pval));
121#else
122 res.push_back(val);
123#endif // TVM_INDEX_DEFAULT_I64
124 } else if (val->IsInstance<tir::AnyNode>()) {
125 // currently all 'any' we meet in shape function are non-negative.
126 res.push_back(val.as<tir::AnyNode>()->ToSizeVar());
127 } else {
128 res.push_back(val);
129 }
130 }
131 return res;
132}

134// Lowers Relay primitive Function to TE Compute
135class LowerToTECompute : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
136 public:
137 explicit LowerToTECompute(Target target)
138 : target_(target), device_copy_op_(Op::Get("device_copy")) {}
139
140 Array<te::Tensor> Lower(const Function& relay_func) {
141 for (Var param : relay_func->params) {
142 Array<tvm::te::Tensor> inputs;
143 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
144 tvm::te::Tensor tensor = tvm::te::placeholder(GetShape(ttype->shape), ttype->dtype);
145 inputs.push_back(tensor);
146 fn_inputs_.push_back(tensor);
147 }
148 memo_[param] = inputs;
149 }
150 readable_name_stream_ << "fused";
151
152 Array<te::Tensor> outputs = this->VisitExpr(relay_func->body);
153
154 candidate_name_ = readable_name_stream_.str();
155
156 constexpr static size_t kMaxFuncNameLength = 80;
157 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
158 // whenever the value of kMaxFuncNameLength changes
159 if (candidate_name_.size() > kMaxFuncNameLength) {
160 std::stringstream truncated_name;
161 truncated_name << candidate_name_.substr(0, kMaxFuncNameLength);
162 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name_) << "_";
163 candidate_name_ = truncated_name.str();
164 }
165
166 return outputs;
167 }

169 Array<te::Tensor> VisitExpr_(const VarNode* op) final {
170 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(GetRef<Var>(op));
171 return {};
172 }

174 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
175 using tir::make_const;
176 void* data = op->data->data;
177 DataType dtype = DataType(op->data->dtype);
178 if (op->is_scalar()) {
179 auto value = te::compute(
180 {},
181 [&](const Array<tvm::tir::Var>&) {
182 if (dtype == DataType::Int(16)) {
183 return make_const(dtype, static_cast<const int16_t*>(data)[0]);
184 } else if (dtype == DataType::Int(8)) {
185 return make_const(dtype, static_cast<const int8_t*>(data)[0]);
186 } else if (dtype == DataType::UInt(8) || dtype == DataType::Bool()) {
187 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
188 } else if (dtype == DataType::Int(32)) {
189 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
190 } else if (dtype == DataType::Int(64)) {
191 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
192 } else if (dtype == DataType::Float(16)) {
193 return make_const(dtype, __gnu_h2f_ieee(static_cast<const uint16_t*>(data)[0]));
194 } else if (dtype == DataType::Float(32)) {
195 return make_const(dtype, static_cast<const float*>(data)[0]);
196 } else if (dtype == DataType::Float(64)) {
197 return make_const(dtype, static_cast<const double*>(data)[0]);
198 } else {
199 LOG(FATAL) << dtype << " not handled";
200 return tvm::PrimExpr();
201 }
202 },
203 "compile_engine_const", topi::kBroadcast);
204 scalars_.push_back(value->op);
205 return {value};
206 } else {
207 const auto* ttype = op->checked_type().as<TensorTypeNode>();
208 std::stringstream ss;
209 std::string s = readable_name_stream_.str();
210 std::replace(s.begin(), s.end(), '.', '_');
211 ss << s << "_constant_" << const_index++;
212 tvm::te::Tensor tensor = tvm::te::placeholder(GetShape(ttype->shape), ttype->dtype, ss.str());
213 constant_tensors_[op] = tensor;
214 return {tensor};
215 }
216 }

218 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
219 static auto flower_call = tvm::runtime::Registry::Get("relay.backend.lower_call");
220 ICHECK(flower_call) << "relay.backend.lower_call is not registered.";
221
222 Array<te::Tensor> inputs;
223 int count_tuple = 0;
224 for (Expr arg : call_node->args) {
225 if (arg->checked_type().as<TupleTypeNode>()) {
226 ++count_tuple;
227 }
228 for (te::Tensor tensor : VisitExpr(arg)) {
229 inputs.push_back(tensor);
230 }
231 }
232
233 if (count_tuple) {
234 ICHECK_EQ(call_node->args.size(), 1U)
235 << "Only functions with a single tuple input are allowed, but " << count_tuple
236 << " were provided.";
237 }
238
239 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
240 Op op = Downcast<Op>(call_node->op);
241
242 // TODO(mbs): device_copy cleanup
243 ICHECK_NE(op, device_copy_op_) << "device_copy cannot be lowered";
244
245 LoweredOutput lowered_out = (*flower_call)(GetRef<Call>(call_node), inputs, target_);
246 Array<te::Tensor> outputs = lowered_out->outputs;
247 op_implementations_[op.operator->()] = lowered_out->implementation;
248
249 if (outputs.size() != 1) {
250 const auto* tuple_type = call_node->checked_type().as<TupleTypeNode>();
251 ICHECK(tuple_type) << "Expected output to be a tuple type "
252 << PrettyPrint(call_node->checked_type());
253
254 ICHECK_EQ(tuple_type->fields.size(), outputs.size());
255 }
256
257 readable_name_stream_ << '_' << op->name;
258 return outputs;
259 }

261 Array<te::Tensor> VisitExpr_(const FunctionNode* op) final {
262 LOG(FATAL) << "Primitive Functions can not contain nested functions.";
263 return Array<te::Tensor>();
264 }

266 Array<te::Tensor> VisitExpr_(const LetNode* op) final {
267 Array<te::Tensor> val = VisitExpr(op->value);
268 ICHECK(!memo_.count(op->var));
269 memo_[op->var] = val;
270 return VisitExpr(op->body);
271 }
272
273 Array<te::Tensor> VisitExpr_(const TupleNode* op) final {
274 Array<te::Tensor> fields;
275 for (Expr field : op->fields) {
276 // TODO(mbs): Generalize to be equivalent to FlattenTupleType.
277 ICHECK(field->checked_type().as<TensorTypeNode>()) << "Only allow Tuple of Tensor";
278 Array<te::Tensor> res = VisitExpr(field);
279 ICHECK_EQ(res.size(), 1);
280 fields.push_back(res[0]);
281 }
282 return fields;
283 }
284
285 Array<te::Tensor> VisitExpr_(const TupleGetItemNode* op) final {
286 const auto* tuple_type = op->tuple->type_as<TupleTypeNode>();
287 Array<te::Tensor> tuple = VisitExpr(op->tuple);
288 ICHECK_EQ(tuple_type->fields.size(), tuple.size());
289 ICHECK_GE(op->index, 0);
290 ICHECK_LT(static_cast<size_t>(op->index), tuple.size());
291 return {tuple[op->index]};
292 }
293
294 public:
295 // Additional outputs
296 Array<tvm::te::Tensor> fn_inputs_;
297 Array<te::Operation> scalars_;
298 std::unordered_map<const ConstantNode*, te::Tensor> constant_tensors_;
299 std::unordered_map<const OpNode*, OpImplementation> op_implementations_;
300 std::string candidate_name_;
301
302 private:
303 tvm::Target target_;
304 std::ostringstream readable_name_stream_;
305 // Index of the global constants
306 static int const_index;
307 // Cache device copy op for equivalence checking to reduce registry lookup
308 // overhead for each invocation of call node when retrieving schedules.
309 const Op& device_copy_op_;
310};

314// Construct a schedule for a given Relay primitive function and target.
315class ScheduleBuilder : public ExprVisitor {
316 public:
317 explicit ScheduleBuilder(Target target) : target_(target) {
318 // Whether to use auto_scheduler schedule.
319 use_auto_scheduler_ = backend::IsAutoSchedulerEnabled();
320 if (backend::IsMetaScheduleEnabled()) {
321 meta_schedule_ctx_ = meta_schedule::ApplyHistoryBest::Current();
322 CHECK(meta_schedule_ctx_.defined()) << "ValueError: `use_meta_schedule` is enabled in Relay "
323 "build, but no ApplyHistoryBest context is provided. ";
324 } else {
325 meta_schedule_ctx_ = NullOpt;
326 }
327 }
328
329 CachedFunc Create(const Function& relay_func, GlobalVarSupply global_var_supply) {
330 LowerToTECompute lower_te_compute(target_);
331 Array<te::Tensor> tensor_outs = lower_te_compute.Lower(relay_func);
332 Array<te::Tensor> fn_inputs = lower_te_compute.fn_inputs_;
333 VisitExpr(relay_func->body);
334
335 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
336 // no other GlobalVar ctors should appear inside the lowering machinery.
337 auto prim_fn_var = global_var_supply->FreshGlobal(lower_te_compute.candidate_name_);
338 prim_fn_var->checked_type_ = relay_func->checked_type();
339
340 // Fusion over tupled results may leave identity relationships
341 // between inputs and outputs, copy identity output tensors,
342 // since tir lowering do not support aliasing output to input buffer.
343 for (size_t i = 0; i < tensor_outs.size(); ++i) {
344 if (tensor_outs[i]->op.as<te::PlaceholderOpNode>()) {
345 tensor_outs.Set(i, topi::identity(tensor_outs[i]));
346 }
347 }
348
349 te::Schedule schedule{nullptr};
350 tir::PrimFunc prim_func{nullptr};
351 // No need to register schedule for device copy op.
352 if (anchor_attrs_.as<DeviceCopyAttrs>() == nullptr) {
353 if (use_auto_scheduler_) {
354 const auto* fauto_schedule =
355 runtime::Registry::Get("auto_scheduler.relay_integration.auto_schedule_topi_compute");
356 ICHECK(fauto_schedule != nullptr)
357 << "auto_scheduler.relay_integration.auto_schedule_topi_compute is not registered";
358 ObjectRef obj = (*fauto_schedule)(prim_fn_var->name_hint, tensor_outs);
359 if (obj.defined()) {
360 schedule = Downcast<te::Schedule>(obj);
361 }
362 }
363 if (meta_schedule_ctx_) {
364 Array<te::Tensor> te_args = Concat(fn_inputs, tensor_outs);
365 if (Optional<tir::PrimFunc> tir_func =
366 meta_schedule_ctx_.value()->te_filter_func(te_args)) {
367 IRModule relay_mod({{prim_fn_var, relay_func}});
368 IRModule tir_mod({{prim_fn_var, tir_func.value()}});
369 if (Optional<IRModule> opt_scheduled_mod = meta_schedule_ctx_.value()->Query(
370 /*task_name=*/prim_fn_var->name_hint, //
371 /*mod=*/relay_mod, //
372 /*target=*/target_, //
373 /*dispatched=*/Array<IRModule>{tir_mod}, //
374 /*f_take_tuning_record=*/ExtractTransformLayout)) {
375 IRModule scheduled_mod =
376 tir::transform::RemoveWeightLayoutRewriteBlock()(opt_scheduled_mod.value());
377 ICHECK_EQ(scheduled_mod->functions.count(prim_fn_var), 1);
378 prim_func = Downcast<tir::PrimFunc>(scheduled_mod->functions[prim_fn_var]);
379 }
380 }
381 }
382 // Use TOPI schedule if user specificed, or the function has no auto_scheduler schedule.
383 if (!schedule.defined() && !prim_func.defined()) {
384 if (anchor_op_.defined()) {
385 auto anchor_impl = lower_te_compute.op_implementations_.find(anchor_op_.operator->());
386 ICHECK(anchor_impl != lower_te_compute.op_implementations_.end());
387 schedule = anchor_impl->second.Schedule(anchor_attrs_, tensor_outs, target_);
388 } else {
389 auto default_sched = GenericFunc::Get("schedule_injective");
390 ICHECK(default_sched.defined()) << "schedule_injective not registered for " << target_;
391 With<Target> tctx(target_);
392 schedule = default_sched(tensor_outs);
393 }
394 }
395 if (schedule.defined()) {
396 for (const auto& scalar : lower_te_compute.scalars_) {
397 if (schedule->Contain(scalar)) {
398 schedule[scalar].compute_inline();
399 }
400 }
401 }
402 }
403
404 IRModule funcs = IRModule(Map<GlobalVar, BaseFunc>({}));
405 return CachedFunc(target_, prim_fn_var, fn_inputs, tensor_outs, schedule, prim_func, {}, funcs,
406 lower_te_compute.constant_tensors_);
407 }
408
409 void VisitExpr_(const CallNode* call_node) final {
410 static auto fpattern = Op::GetAttrMap<TOpPattern>("TOpPattern");
411
412 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
413 Op op = Downcast<Op>(call_node->op);
414
415 for (Expr arg : call_node->args) {
416 VisitExpr(arg);
417 }
418
419 int op_pattern = fpattern[op];
420 if (!use_auto_scheduler_ && !meta_schedule_ctx_.defined() && op_pattern >= kCommReduce) {
421 ICHECK(!anchor_op_.defined() || anchor_op_pattern_ < kCommReduce)
422 << "Cannot apply TOPI schedule to a primitive function with two complicated ops"
423 << " anchor=" << anchor_op_ << " current=" << op;
424 }
425 if (op_pattern >= anchor_op_pattern_) {
426 anchor_op_ = op;
427 anchor_attrs_ = call_node->attrs;
428 anchor_op_pattern_ = op_pattern;
429 }
430 }
431
432 private:
433 tvm::Target target_;
434 Op anchor_op_;
435 Attrs anchor_attrs_;
436 int anchor_op_pattern_{0};
437 bool use_auto_scheduler_;
438 Optional<meta_schedule::ApplyHistoryBest> meta_schedule_ctx_;
439};

453// Creates shape function from functor.
454class MakeShapeFunc : public backend::MemoizedExprTranslator<Array<te::Tensor>> {
455 public:
456 MakeShapeFunc() {}
457
458 CachedFunc Create(const Function& prim_func, const Target& target,
459 GlobalVarSupply global_var_supply) {
460 VLOG_CONTEXT << "MakeShapeFunc";
461 TShapeDataDependent shape_func_param_states;
462
463 for (auto param : prim_func->params) {
464 param_states_[param] = kNoNeed;
465 Array<tvm::te::Tensor> data_inputs;
466 Array<tvm::te::Tensor> shape_inputs;
467
468 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
469 // Add data placeholder (in case we discover we need it below)
470 Shape shape = GetShape(ttype->shape);
471 tvm::te::Tensor data_tensor = tvm::te::placeholder(shape, ttype->dtype);
472 data_inputs.push_back(data_tensor);
473 // Add shape placeholder (in case we discover we need it below)
474 int64_t ndim = shape.size();
475 Shape sshape;
476 if (ndim > 0) {
477 sshape.push_back(tvm::Integer(ndim));
478 }
479 tvm::te::Tensor shape_tensor = tvm::te::placeholder(sshape, DataType::Int(64));
480 shape_inputs.push_back(shape_tensor);
481 }
482 param_data_[param] = data_inputs;
483 param_shapes_[param] = shape_inputs;
484 }
485
486 // Setup the name;
487 readable_name_stream_ << "shape_func";
488
489 // Create the tensor expressions representing the output shapes.
490 Array<te::Tensor> outputs = VisitExpr(prim_func->body);
491
492 // Generate a name.
493 auto candidate_name = readable_name_stream_.str();
494
495 constexpr static size_t kMaxFuncNameLength = 80;
496 // WARNING: Please make sure to also update TVM_CRT_MAX_STRLEN_FUNCTION_NAME
497 // whenever the value of kMaxFuncNameLength changes
498 if (candidate_name.size() > kMaxFuncNameLength) {
499 std::stringstream truncated_name;
500 truncated_name << candidate_name.substr(0, kMaxFuncNameLength);
501 truncated_name << "_" << std::hex << std::hash<std::string>{}(candidate_name) << "_";
502 candidate_name = truncated_name.str();
503 }
504
505 // Set all the inputs correctly, and accumulate their types from the p.o.v. of the
506 // shape function rather than the primitive it is derived for.
507 Array<te::Tensor> inputs;
508 Array<Type> shape_function_arg_types;
509 for (auto param : prim_func->params) {
510 int state = param_states_[param];
511 shape_func_param_states.push_back(IntImm(DataType::Int(32), state));
512 if (state & kNeedInputData) {
513 // Pass the primitive arguments directly (though in flattened form and on the host)
514 for (auto t : param_data_[param]) {
515 inputs.push_back(t);
516 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
517 }
518 }
519 if (state & kNeedInputShape) {
520 // Pass the shapes of the primitive arguments (also on the host)
521 for (auto t : param_shapes_[param]) {
522 inputs.push_back(t);
523 shape_function_arg_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
524 }
525 }
526 }
527
528 // TODO(mbs): This should be the definitive global by which the PrimFunc is known and
529 // no other GlobalVar ctors should appear inside the lowering machinery.
530 auto prim_fn_gvar = global_var_supply->FreshGlobal(candidate_name);
531
532 // Gather the result types, again from the p.o.v. of the shape function rather than
533 // the primitive it is derived for.
534 Array<Type> shape_function_res_types;
535 for (const auto& t : outputs) {
536 shape_function_res_types.push_back(TensorType(t->GetShape(), t->GetDataType()));
537 }
538
539 // Assign the shape function its true type.
540 FuncType type(shape_function_arg_types, TupleType(shape_function_res_types),
541 /*type_params=*/{}, /*type_constraints=*/{});
542 VLOG(1) << "shape function '" << prim_fn_gvar->name_hint << "' has type:" << std::endl
543 << PrettyPrint(type) << std::endl
544 << "corresponding to primitive of type:" << std::endl
545 << PrettyPrint(prim_func->checked_type());
546 prim_fn_gvar->checked_type_ = std::move(type);
547
548 // generate schedule for shape func
549 Array<te::Operation> out_ops;
550 for (auto t : outputs) {
551 out_ops.push_back(t->op);
552 }
553 te::Schedule schedule = te::create_schedule(out_ops);
554 tvm::te::AutoInlineInjective(schedule);
555 for (const auto& scalar : scalars_) {
556 auto scalar_op = scalar->op;
557 if (schedule->Contain(scalar_op)) {
558 schedule[scalar_op].compute_inline();
559 }
560 }
561
562 Array<te::Tensor> all_args = Array<te::Tensor>(inputs);
563 for (te::Tensor arg : outputs) {
564 all_args.push_back(arg);
565 }
566
567 using tvm::transform::PassContext;
568 With<PassContext> fresh_pass_ctx_scope(PassContext::Create());
569
570 std::unordered_map<te::Tensor, tir::Buffer> binds;
571 IRModule lowered_module =
572 tvm::LowerSchedule(schedule, all_args, prim_fn_gvar->name_hint, binds, global_var_supply);
573 return CachedFunc(target, prim_fn_gvar, inputs, outputs, schedule, tir::PrimFunc{nullptr},
574 shape_func_param_states, lowered_module);
575 }

577 Array<te::Tensor> VisitExpr(const Expr& expr) final {
578 if (expr.as<VarNode>()) {
579 // Do not memoize vars because shape functions could use either the data
580 // or the shape of a var each time.
581 return ExprFunctor::VisitExpr(expr);
582 }
583 // For other case, do memoized visit
584 return backend::MemoizedExprTranslator<Array<te::Tensor>>::VisitExpr(expr);
585 }
586
587 Array<te::Tensor> VisitExpr_(const VarNode* var_node) final {
588 auto var = GetRef<Var>(var_node);
589 auto it = param_arg_map_.find(var);
590 if (it != param_arg_map_.end()) {
591 // This var is a parameter of a nested function. Visit the corresponding argument in the
592 // function call site.
593 return VisitExpr(it->second);
594 }
595 if (param_states_.find(var) == param_states_.end()) {
596 LOG(FATAL) << "Unexpected free variable " << PrettyPrint(var);
597 return {};
598 } else {
599 ICHECK(data_dependents_per_input_.size());
600 auto data_dependent = data_dependents_per_input_.back();
601 if (data_dependent) {
602 param_states_[var] |= kNeedInputData;
603 return param_data_[var];
604 } else {
605 param_states_[var] |= kNeedInputShape;
606 return param_shapes_[var];
607 }
608 }
609 }
610
611 Array<te::Tensor> VisitExpr_(const ConstantNode* op) final {
612 using tir::make_const;
613 ICHECK(data_dependents_per_input_.size());
614 bool data_dependent = data_dependents_per_input_.back();
615 if (!op->is_scalar()) {
616 // This is a constant weight, extract the shape of the weight tensor.
617 // This can not be data dependent.
618 CHECK(!data_dependent);
619 auto ttype = op->checked_type().as<TensorTypeNode>();
620 int ndim = static_cast<int>(ttype->shape.size());
621 Array<PrimExpr> out_shape{ndim};
622 te::Tensor value = tvm::te::compute(
623 out_shape,
624 [&](const Array<tvm::tir::Var>& indices) {
625 auto idx = indices[0];
626 PrimExpr ret = make_const(DataType::Int(64), 0);
627 for (int i = 0; i < ndim; i++) {
628 ret = tvm::if_then_else(idx == i, ttype->shape[i], ret);
629 }
630 return ret;
631 },
632 "shape_const", topi::kBroadcast);
633 scalars_.push_back(value);
634 return {value};
635 }
636 if (data_dependent) {
637 void* data = op->data->data;
638 DataType dtype = DataType(op->data->dtype);
639 auto value = tvm::te::compute(
640 {},
641 [&](const Array<tvm::tir::Var>&) {
642 if (dtype == DataType::Int(32)) {
643 return make_const(dtype, static_cast<const int32_t*>(data)[0]);
644 } else if (dtype == DataType::Int(64)) {
645 return make_const(dtype, static_cast<const int64_t*>(data)[0]);
646 } else if (dtype == DataType::Float(32)) {
647 return make_const(dtype, static_cast<const float*>(data)[0]);
648 } else if (dtype == DataType::Float(64)) {
649 return make_const(dtype, static_cast<const double*>(data)[0]);
650 } else if (dtype == DataType::Bool()) {
651 return make_const(dtype, static_cast<const uint8_t*>(data)[0]);
652 } else {
653 LOG(FATAL) << "not handled";
654 return tvm::PrimExpr();
655 }
656 },
657 "data_const", topi::kBroadcast);
658 scalars_.push_back(value);
659 return {value};
660 } else {
661 auto value = tvm::te::compute(
662 {}, [&](const Array<tvm::tir::Var>&) { return tir::make_const(DataType::Int(64), 0); },
663 "shape_const", topi::kBroadcast);
664 scalars_.push_back(value);
665 return {value};
666 }
667 }
668
669 Array<te::Tensor> VisitExpr_(const CallNode* call_node) final {
670 VLOG(1) << "considering call:" << std::endl << PrettyPrint(GetRef<Call>(call_node));
671 if (auto* func = call_node->op.as<FunctionNode>()) {
672 VLOG(1) << "user function";
673 for (size_t i = 0; i < func->params.size(); ++i) {
674 param_arg_map_[func->params[i]] = call_node->args[i];
675 }
676 return VisitExpr(func->body);
677 }
678
679 static auto fshape_func = Op::GetAttrMap<FShapeFunc>("FShapeFunc");
680 static auto tshape_data_dependent = Op::GetAttrMap<TShapeDataDependent>("TShapeDataDependent");
681 ICHECK(call_node->op.as<OpNode>()) << "Primitive function only allows call into primitive ops";
682 Op op = Downcast<Op>(call_node->op);
683 ICHECK(data_dependents_per_input_.empty() || !data_dependents_per_input_.back())
684 << "Error in op fusion: output of the shape func is fed to a "
685 << "data-dependent shape func";
686 ICHECK_GT(fshape_func.count(op), 0) << "Internal error, cannot find ShapeFunc for " << op->name;
687 ICHECK_GT(tshape_data_dependent.count(op), 0)
688 << "Internal error, cannot find TShapeDataDependent for " << op->name;
689
690 Array<Integer> dep_spec = tshape_data_dependent[op];
691 if (dep_spec.size() == 1) {
692 // This is for cases when data dependence is specified per op
693 // Replicate 0 or 1 flag to all arguments
694 for (size_t i = 1; i < call_node->args.size(); ++i) {
695 dep_spec.push_back(dep_spec[0]);
696 }
697 }
698
699 // Visit all inputs
700 Array<te::Tensor> inputs;
701 int count_tuple = 0;
702 for (size_t i = 0; i < call_node->args.size(); ++i) {
703 Expr arg = call_node->args[i];
704 if (arg->checked_type().as<TupleTypeNode>()) {
705 ++count_tuple;
706 }
707 data_dependents_per_input_.push_back(dep_spec[i]->value != 0);
708 for (te::Tensor tensor : VisitExpr(arg)) {
709 inputs.push_back(tensor);
710 }
711 data_dependents_per_input_.pop_back();
712 }
713 if (count_tuple) {
714 ICHECK_EQ(call_node->args.size(), 1U) << "Only allow function with a single tuple input";
715 }
716 // Get output ndims
717 auto ret_type = call_node->checked_type();
718 Array<IndexExpr> out_ndims;
719 for (const auto& ttype : FlattenTupleType(ret_type)) {
720 out_ndims.push_back(IntImm(DataType::Int(32), ttype->shape.size()));
721 }
722
723 // Call shape function
724 Array<te::Tensor> outputs = fshape_func[op](call_node->attrs, inputs, out_ndims);
725 VLOG(1) << "shape function for '" << op->name << "' with inputs:" << std::endl
726 << inputs << std::endl
727 << "yielded outputs:" << std::endl
728 << outputs;
729 readable_name_stream_ << "_" << op->name;
730 return outputs;
731 }
732
733 Array<te::Tensor> VisitExpr_(const FunctionNode* op) final {
734 LOG(FATAL) << "Nested functions are not allowed to be visited.";
735 return Array<te::Tensor>();
736 }
737
738 Array<te::Tensor> VisitExpr_(const LetNode* op) final {
739 Array<te::Tensor> val = VisitExpr(op->value);
740 ICHECK(!memo_.count(op->var));
741 memo_[op->var] = val;
742 return VisitExpr(op->body);
743 }
744
745 Array<te::Tensor> VisitExpr_(const TupleNode* op) final {
746 Array<te::Tensor> fields;
747 for (Expr field : op->fields) {
748 ICHECK(field->checked_type().as<TensorTypeNode>())
749 << "Expected a Tuple of Tensor, but got " << PrettyPrint(field->checked_type());
750 Array<te::Tensor> res = VisitExpr(field);
751 ICHECK_EQ(res.size(), 1);
752 fields.push_back(res[0]);
753 }
754 return fields;
755 }
756
757 Array<te::Tensor> VisitExpr_(const TupleGetItemNode* op) final {
758 Array<te::Tensor> input_shapes = VisitExpr(op->tuple);
759 Array<te::Tensor> out;
760 out.push_back(input_shapes[op->index]);
761 return out;
762 }

766 std::ostringstream readable_name_stream_;
767 /*! \brief Map from parameter to its shape function usage state */
768 std::unordered_map<Expr, int, ObjectPtrHash, ObjectPtrEqual> param_states_;
769 /*! \brief Map from parameter to list of data placeholder */
770 std::unordered_map<Expr, Array<te::Tensor>, ObjectPtrHash, ObjectPtrEqual> param_data_;
771 /*! \brief Map from parameter to list of shape placeholder */

788std::pair<Array<te::Tensor>, std::string> LowerTECompute(const Function& source_func, Target target,
789 bool return_inputs) {
790 LowerToTECompute lower_te_compute(target);
791 Array<te::Tensor> outputs = lower_te_compute.Lower(source_func);
792 // Following ScheduleBuilder, remove placeholder ops from outputs.
793 tvm::Array<te::Tensor> tensor_outs;
794 for (const auto& tensor : outputs) {
795 if (!tensor->op.as<te::PlaceholderOpNode>()) {
796 tensor_outs.push_back(tensor);
797 }
798 }
799 if (return_inputs) {
800 return std::make_pair(Concat(lower_te_compute.fn_inputs_, tensor_outs),
801 lower_te_compute.candidate_name_);
802 }
803 return std::make_pair(tensor_outs, lower_te_compute.candidate_name_);
804}
