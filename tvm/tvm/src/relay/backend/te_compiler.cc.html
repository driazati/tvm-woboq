20/*!
21 * \file relay/backend/te_compiler.cc
22 * \brief Manages the transition from Relay "Primitive" \p Functions to TIR \p PrimFuncs. Also
23 * handles invocation of external codegen.
24 *
25 * \p LowerTEPass handles the following (as a monolithic blob of code):
26 *
27 * - Most importantly, any function with the "Primitive" attribute is first converted to TE by
28 * \p LowerToTECompute (see te_compiler_cache.cc) using each operator's 'compute' function.
29 * The TE is then 'scheduled' to TIR using the 'anchor' operator's 'schedule' function. Both
30 * of those functions come from the \p OpStrategy returned by the Python
31 * 'relay.backend.lower_call' function (see te_compiler.py).
32 * The TIR is packed as a \p PrimFunc and introduced as a new global function. Calls to the
33 * original "Primitive" function are then rewritten to the form:
34 * \code
35 * call_lowered(@new_global, (... original args...), attributes)
36 * \endcode
37 *
38 * - The above "Primitive" function can appear:
39 * - As a global function
40 * - As a let-bound function
41 * - As an inline function, ie the 'op' of calls.
42 * In all three cases it is possible for the same "Primitive" function to be called multiple
43 * times, and that sharing must be respected.
44 *
45 * - "Primitive" functions must have a "global_symbol" attribute matching their desired or
46 * existing global name. Care is taken to ensure GlobalVars with the same name are shared.
47 *
48 * - It is possible for multiple structurally equal "Primitive" functions to appear in the same
49 * \p IRModule. Only one implementation should be generated, and all calls should share that
50 * implementation.
51 *
52 * - When later converting to DPS (see memory_alloc.cc) we must handle functions who's result
53 * tensor shapes depend at runtime on the input tensor shapes and/or data.
54 * - That dependency is first described in TE form (see \p MakeShapeFunc in
55 * te_compiler_cache.cc), then scheduled to yield a 'dynamic shape function' \p PrimFunc.
56 * This relies on each operator's "FShapeFunc" and "TShapeDataDependent" attributes.
57 * Since shapes are rank-1 tensors everything can be reflected back down into the regular
58 * TE/TIR forms.
59 * - Then the call_lowered attributes must record everything about the dynamic shape function
60 * later needed by memory_alloc.cc. We call this 'cross linking' the call with the shape
61 * function.
62 *
63 * - Two external codegen mechanisms are supported, both triggered by "Primitive" functions which
64 * also have a "Compiler" attribute bound to $compiler:
65 * - Function-at-a-time (old style): The primitive function is passed to the function
66 * registered as 'relay.ext.$compiler'. The function returns a runtime::Module which
67 * should return true for \p ImplementsFunction for the function's global name. That
68 * module is added to the IRModule's "external_mods" attributes.
69 * - IRModule-at-a-item (new style): The \p RelayToTIRTargetHook sub-pass looks for
70 * $compiler names which correspond to TargetKind names with a \p RelayToTIR attribute.
71 * The \p Pass bound to that attribute is run, and each such 'custom' pass can do what
72 * it likes, including replacing Functions with PrimFuncs, or adding new runtime::Modules
73 * to the IRModule's "external_mods" attribute.
74 *
75 * - Calls to functions added by external codegen are also rewritten to call_lowered form, and
76 * may also require cross-linking to dynamic shape functions. However, since the functions
77 * are/will be implemented by a runtime::Module all the Relay type information is no longer
78 * available. So the Relay definitions for these "Primitive" "Compiler" functions are retained
79 * in the \p IRModule, but marked with the "Extern" attribute to signal the function is now
80 * just for carrying metadata.
81 *
82 * - Some operators are handled specially:
83 * - 'reshape', since it's a no-op on the underlying tensor buffer, and this is handled by
84 * condition tests in many passes.
85 * - 'debug', since it's intercepted differently depending on runtimes.
86 *
87 * TODO(mbs): This desperately deserves a refactor to separate all these concerns. See Relax.
88 */

138 explicit TECompilerImpl(Optional<IRModule> opt_mod, Optional<String> opt_mod_name) {
139 String mod_name = opt_mod_name.value_or("");
140 NameSupply name_supply = NameSupply(mod_name /* prefix */);
141 global_var_supply_ = GlobalVarSupply(name_supply);
142 // Make sure we don't collide with any existing globals in the module.
143 if (opt_mod) {
144 for (const auto& kv : opt_mod.value()->functions) {
145 global_var_supply_->name_supply_->ReserveName(kv.first->name_hint, false);
146 }
147 }
148 }

152 return LowerInternal(key, global_var_supply_)->cached_func;
153 }

159 return LowerInternal(key, global_var_supply_)->cached_func;
160 }

162 // For now, build one module per function.
163 PackedFunc JIT(const CCacheKey& key) final {
164 CCacheValue value = LowerInternal(key, GlobalVarSupply(NameSupply("")));
165 if (value->packed_func != nullptr) {
166 return value->packed_func;
167 }
168 auto m = build(value->cached_func->funcs, key->target, Target(nullptr));
169 value->packed_func = m.GetFunction(value->cached_func->prim_fn_var->name_hint);
170 return value->packed_func;
171 }

177 IRModule GetLoweredFunctions() {
178 VLOG(1) << "GetLoweredFunctions";
179 IRModule mod;
180 // Extract lowered functions from the cache
181 for (const auto& it : cache_) {
182 auto source_func = it.first;
183 auto lowered_func = it.second;
184
185 IRModule lowered_mod = lowered_func->cached_func->funcs;
186
187 // Annotate functions with their target and put them in the return module
188 for (const auto& kv : lowered_mod->functions) {
189 const GlobalVar& var = kv.first;
190 const BaseFunc& func = kv.second;
191
192 // Only add functions that are not external functions
193 if (!func->GetAttr<String>(attr::kCompiler).defined()) {
194 ICHECK(func->IsInstance<tir::PrimFuncNode>())
195 << "Expected all functions that are not external to be PrimFuncs, but found:"
196 << std::endl
197 << PrettyPrint(func);
198 const tir::PrimFunc& prim_func = Downcast<tir::PrimFunc>(func);
199 mod->Update(var, WithAttr(prim_func, tvm::attr::kTarget, source_func->target));
200 }
201 }
202 }
203
204 // Extract lowered dynamic shape functions from the shape cache
205 for (const auto& it : shape_func_cache_) {
206 auto source_func = it.first;
207 auto lowered_func = it.second;
208 auto target = source_func->target;
209 IRModule lowered_mod = lowered_func->cached_func->funcs;
210
211 // Annotate functions with their target and put them in the return module
212 for (auto kv : lowered_mod->functions) {
213 const GlobalVar& var = kv.first;
214 const BaseFunc& func = kv.second;
215 const tir::PrimFunc& prim_func = Downcast<tir::PrimFunc>(func);
216 mod->Update(var, WithAttr(prim_func, tvm::attr::kTarget, source_func->target));
217 }
218 }
219
220 return mod;
221 }

223 void AddExterns(IRModule module) {
224 // Everything tagged with "Compiler" has been compiled, so remove those definitions.
225 std::vector<GlobalVar> to_be_deleted;
226 for (const auto& kv : module->functions) {
227 if (kv.second->GetAttr<String>(attr::kCompiler).defined()) {
228 to_be_deleted.push_back(kv.first);
229 }
230 }
231 for (const auto& global_var : to_be_deleted) {
232 VLOG(1) << "Removing definition for external codegened '" << global_var->name_hint << "'";
233 module->Remove(global_var);
234 }
235 // HOWEVER we still need a Relay definition to go with those now external functions, so
236 // retrieve them from the cache and mark them with "ExternalSymbol".
237 for (const auto& kv1 : cache_) {
238 auto src_func = kv1.first->source_func;
239 ICHECK(src_func.defined());
240 if (src_func->GetAttr<String>(attr::kCompiler).defined()) {
241 for (const auto& kv2 : kv1.second->cached_func->funcs->functions) {
242 if (const auto* function_node = kv2.second.as<FunctionNode>()) {
243 // Abandon the existing function annotations.
244
245 // Unfortunately, Optional<DictAttrs>() is indistinguishable from
246 // NullValue<DictAttrs>(), and DictAttrs() is nullptr, so to erase the attributes, we
247 // need pass in DictAttrs<Map<String, ObjectRef>()), which is a DictAttrs containing no
248 // attributes.
249 Function function =
250 WithFields(GetRef<Function>(function_node), function_node->params,
251 function_node->body, function_node->ret_type, function_node->type_params,
252 /* erase attributes */ DictAttrs(Map<String, ObjectRef>()));
253 // Mark function as 'extern'.
254 function = WithAttr(std::move(function), attr::kExtern, Integer(1));
255 module->Add(kv2.first, function);
256 }
257 }
258 }
259 }
260 }

262 Array<tvm::runtime::Module> LowerExternalFunctions() {
263 Array<tvm::runtime::Module> ret;
264 std::vector<CCacheKey> cached_ext_funcs;
265
266 for (const auto& it : cache_) {
267 auto src_func = it.first->source_func;
268 ICHECK(src_func.defined());
269 Optional<String> opt_compiler = src_func->GetAttr<String>(attr::kCompiler);
270 if (opt_compiler.defined()) {
271 Optional<String> opt_symbol_name = src_func->GetAttr<String>(tvm::attr::kGlobalSymbol);
272 ICHECK(opt_symbol_name.defined()) << "No external symbol is set for:" << std::endl
273 << PrettyPrint(src_func);
274 VLOG(1) << "using external codegen '" << opt_compiler.value() << "' for name '"
275 << opt_symbol_name.value() << "' and function:" << std::endl
276 << PrettyPrint(src_func);
277 cached_ext_funcs.push_back(it.first);
278
279 std::string ext_name = "relay.ext." + opt_compiler.value();
280 auto pf = tvm::runtime::Registry::Get(ext_name);
281 ICHECK(pf) << "Failed to find the external codegen tool for " << ext_name;
282 // No need to keep compiler attribute at this point, functions have been
283 // extracted for specific codegen.
284 src_func = WithAttr(std::move(src_func), attr::kCompiler, NullValue<ObjectRef>());
285 VLOG_CONTEXT << opt_compiler.value();
286 With<Target> with_target(it.first->target);
287 runtime::Module ext_mod = (*pf)(src_func);
288 if (ext_mod.defined()) {
289 // TODO(mbs): Can this be an ICHECKs?
290 if (!ext_mod->ImplementsFunction(opt_symbol_name.value())) {
291 VLOG(1) << "Note that the external codegen for '" << opt_compiler.value()
292 << "' returned a runtime module which does not appear to implement '"
293 << opt_symbol_name.value() << "'";
294 }
295 ret.push_back(ext_mod);
296 } else {
297 // It is valid for the external codegen function to return null:
298 // - Unit tests can use it.
299 // - The true compilation may have already been handled by a RelayToTIR custom pass
300 // on the Target's kind. The original Relay functions will be left in place so
301 // that we can capture that their function names are now externally defined.
302 VLOG(1) << "Note that no external runtime module was generated by external codegen '"
303 << opt_compiler.value() << "'";
304 }
305 }
306 }
307
308 // No need to cache external functions as we collected them all to create
309 // external runtime modules.
310 for (const auto& it : cached_ext_funcs) {
311 cache_.erase(it);
312 }
313 return ret;
314 }

341 // implement lowered func
342 CCacheValue LowerInternal(const CCacheKey& key, GlobalVarSupply global_var_supply) {
343 VLOG(1) << "lowering:" << std::endl
344 << PrettyPrint(key->source_func) << std::endl
345 << "for target:" << std::endl
346 << key->target->ToDebugString();
347 std::lock_guard<std::mutex> lock(mutex_);
348 CCacheValue value;
349 auto it = cache_.find(key);
350 if (it != cache_.end()) {
351 VLOG(1) << "already lowered to name:" << std::endl
352 << PrettyPrint(it->second->cached_func->prim_fn_var);
353 it->second->use_count += 1;
354 if (it->second->cached_func.defined()) return it->second;
355 value = it->second;
356 } else {
357 value = CCacheValue(make_object<CCacheValueNode>());
358 value->use_count = 1;
359 cache_[key] = value;
360 }
361 cur_ccache_key_ = key;
362
363 Optional<String> opt_compiler = key->source_func->GetAttr<String>(attr::kCompiler);
364 if (opt_compiler.defined()) {
365 // Don't compile now since we don't have anywhere to put the resulting runtime module.
366 // Instead place the original definition in the cache and wait for LowerExternalFunctions.
367 IRModule ir_module({}, {});
368 Optional<String> opt_global_symbol =
369 key->source_func->GetAttr<String>(tvm::attr::kGlobalSymbol);
370 ICHECK(opt_global_symbol.defined()) << "External function has not been attached a name yet.";
371 // Note that the source_func may already be bound to a global function in the module
372 // we are compiling, in which case we should not attempt to make its name unique w.r.t.
373 // the module's globals. Furthermore, the external codegen tool must bind the compiled
374 // function to the "global_symbol" attribute on the source_func. So do not use GetUniqueName
375 // here.
376 auto global_var = global_var_supply->UniqueGlobalFor(opt_global_symbol.value(), false);
377 global_var->checked_type_ = key->source_func->checked_type();
378 ir_module->Add(global_var, key->source_func);
379 value->cached_func = CachedFunc(key->target, global_var, {}, {}, te::Schedule{nullptr},
380 tir::PrimFunc{nullptr}, {}, ir_module);
381 // Collect these here as it's removed in LowerExternalFunctions()
382 device_contexts_.Set(value->cached_func->prim_fn_var, opt_compiler.value());
383 VLOG(1) << "preparing to use external codegen '" << opt_compiler.value()
384 << "' with name:" << std::endl
385 << PrettyPrint(value->cached_func->prim_fn_var) << std::endl
386 << "and definitions:" << std::endl
387 << PrettyPrint(value->cached_func->funcs);
388 return value;
389 }
390
391 // Enforce use the target.
392 With<Target> target_scope(key->target);
393
394 ICHECK(!value->cached_func.defined());
395 value->cached_func = PrimFuncFor(key->source_func, key->target, global_var_supply);
396
397 if (value->cached_func->prim_func.defined()) {
398 VLOG(1) << "Lowering PrimFunc";
399 IRModule lowered = tvm::LowerPrimFunc(value->cached_func->prim_func.value(),
400 value->cached_func->prim_fn_var->name_hint, false);
401 ICHECK_EQ(lowered->functions.size(), 1);
402 for (const auto& kv : lowered->functions) {
403 value->cached_func->funcs->Add(value->cached_func->prim_fn_var, kv.second);
404 }
405 } else {
406 // NOTE: array will copy on write.
407 Array<te::Tensor> all_args = Array<te::Tensor>(value->cached_func->inputs);
408 for (te::Tensor arg : value->cached_func->outputs) {
409 all_args.push_back(arg);
410 }
411 Array<runtime::NDArray> all_consts;
412 for (auto kv : value->cached_func->constant_tensors) {
413 all_args.push_back(kv.second);
414 all_consts.push_back(kv.first->data);
415 }
416 // lower the function
417 std::unordered_map<te::Tensor, tir::Buffer> binds;
418
419 // If we have memory scopes, need to create tir::Buffer knowing this info
420 size_t i = 0; // for corresponding from tensor array
421 for (Var param : key->source_func->params) {
422 if (!param->virtual_device()->memory_scope.empty()) {
423 for (const auto& ttype : FlattenTupleType(param->checked_type())) {
424 te::Tensor x_ref = value->cached_func->inputs[i];
425 // verification if we have synced params and tensors
426 ICHECK(ttype->dtype == x_ref->dtype && ttype->shape.size() == x_ref->shape.size())
427 << "function parameter does not correspond to prepared tensor";
428 binds[x_ref] =
429 tir::BufferWithOffsetAlignment(x_ref->shape, x_ref->dtype, x_ref->op->name, -1, 0,
430 false, param->virtual_device()->memory_scope);
431 }
432 }
433 i++;
434 }
435 if (key->virtual_device != VirtualDevice::FullyUnconstrained() &&
436 !key->virtual_device->memory_scope.empty() &&
437 key->virtual_device->memory_scope != "global") {
438 ICHECK(value->cached_func->outputs.size() == 1)
439 << "Expect only one output for defined memory scope";
440 te::Tensor x_ref = value->cached_func->outputs[0];
441 binds[x_ref] =
442 tir::BufferWithOffsetAlignment(x_ref->shape, x_ref->dtype, x_ref->op->name, -1, 0,
443 false, key->virtual_device->memory_scope);
444 }
445 auto func_name = value->cached_func->prim_fn_var->name_hint;
446 VLOG(1) << "scheduling";
447 IRModule scheduled_module = tvm::LowerSchedule(value->cached_func->schedule, all_args,
448 func_name, binds, global_var_supply);
449 scheduled_module->Update(tir::transform::BindParams(all_consts)(scheduled_module));
450 for (const auto& kv : scheduled_module->functions) {
451 GlobalVar global_var = kv.first;
452 auto func = kv.second;
453 // Propagate the structural hash of the relay function to the tir
454 // function so associations can be made between the two.
455 Optional<String> hash = key->source_func->attrs.GetAttr<String>("hash");
456 if (hash) {
457 func = WithAttrs(Downcast<tir::PrimFunc>(func), {{String("hash"), hash.value()}});
458 }
459 value->cached_func->funcs->Add(global_var, func);
460 }
461 ICHECK(value->cached_func->funcs->Lookup(value->cached_func->prim_fn_var)
462 .as<tir::PrimFuncNode>());
463 }
464 VLOG(1) << "lowered to name:" << std::endl
465 << PrettyPrint(value->cached_func->prim_fn_var) << std::endl
466 << "with definitions:" << std::endl
467 << PrettyPrint(value->cached_func->funcs);
468
469 return value;
470 }
471
472 // implement lowered shape func
473 CCacheValue LowerShapeFuncInternal(const CCacheKey& key) {
474 VLOG(1) << "lowering dynamic shape function for:" << std::endl
475 << PrettyPrint(key->source_func) << std::endl
476 << "for target:" << std::endl
477 << key->target->ToDebugString();
478 std::lock_guard<std::mutex> lock(mutex_);
479 CCacheValue value;
480 auto it = shape_func_cache_.find(key);
481 if (it != shape_func_cache_.end()) {
482 it->second->use_count += 1;
483 if (it->second->cached_func.defined()) return it->second;
484 value = it->second;
485 } else {
486 value = CCacheValue(make_object<CCacheValueNode>());
487 value->use_count = 0;
488 shape_func_cache_[key] = value;
489 }
490 // Enforce use the target.
491 With<Target> target_scope(key->target);
492
493 ICHECK(!value->cached_func.defined());
494
495 using tvm::transform::PassContext;
496 With<PassContext> fresh_pass_ctx_scope(PassContext::Create());
497 value->cached_func = ShapeFuncFor(key->source_func, key->target, global_var_supply_);
498
499 ICHECK(
500 value->cached_func->funcs->Lookup(value->cached_func->prim_fn_var).as<tir::PrimFuncNode>());
501
502 VLOG(1) << "lowered to name:" << std::endl
503 << PrettyPrint(value->cached_func->prim_fn_var) << std::endl
504 << "with definitions:" << std::endl
505 << PrettyPrint(value->cached_func->funcs);
506 return value;
507 }

509 Map<String, Integer> GetOpWeights() const {
510 Map<String, Integer> weights;
511 for (const auto& kv : cache_) {
512 auto value = kv.second;
513 auto name = value->cached_func->prim_fn_var->name_hint;
514 weights.Set(name, value->use_count);
515 }
516 return weights;
517 }

587/*!
588 * \brief Rewrites call expressions to Relay Functions marked as "primitive"
589 * to calls to the corresponding TIR PrimFunc for the appropriate target.
590 *
591 * \code
592 * %0 = fn(...) { prim_op(...) } OR let %p = fn(...) { prim_op(...) }
593 * ... %0(...) ... ... %p(...) ...
594 * ==>
595 * def @q(..., target=<target>) { <tir body> }
596 * ... @q(...) ...
597 * \endcode
598 *
599 * Requires FuseOps, ToANormalForm, EtaExpand and InferType to have run.
600 *
601 * FuseOps is needed to identify and lift all prim op calls:
602 * \code
603 * ... prim_op(...) ...
604 * ==>
605 * %0 = fn(...) { prim_op(...) }
606 * ... %0(...) ...
607 * \endcode
608 *
609 * ToANormalForm is needed so we only need to consider vars and function literals as the call
610 * target.
611 *
612 * EtaExpand is needed to ensures all calls to primitives are direct:
613 * \code
614 * let %p1 = fn(...) { prim_op1(...) }
615 * let %p2 = fn(...) { prim_op2(...) }
616 * let %p = if (...) { %p1 } else { %p2 }
617 * ... %p(...) ...
618 * ==>
619 * let %p1 = fn(...) { prim_op1(...) }
620 * let %p2 = fn(...) { prim_op2(...) }
621 * let %p = fn(...) { if (...) { %p1(...) } else { %p2(...) } }
622 * ... %p(...) ...
623 * \endcode
624 */
625class LowerTensorExprMutator : public DeviceAwareExprMutator {
626 public:
627 LowerTensorExprMutator(IRModule module, ProcessFn process_fn, CompilationConfig config,
628 TECompiler compiler)
629 : DeviceAwareExprMutator(module),
630 module_(std::move(module)),
631 process_fn_(std::move(process_fn)),
632 config_(std::move(config)),
633 compiler_(std::move(compiler)),
634 debug_op_(Op::Get("debug")) {}

636 /*!
637 * \brief Returns the primitive function associated with \p expr, or nullptr if none.
638 */
639 BaseFunc ResolveToPrimitive(const Expr& expr) {
640 // NOTE: We can't assume expr->checked_type_ is defined, so can't early exit for first-order
641 // expressions.
642 if (const auto* global_var_node = expr.as<GlobalVarNode>()) {
643 if (!module_->ContainGlobalVar(global_var_node->name_hint)) {
644 // TODO(mbs): extern function cleanup
645 // Assume the function is extern and thus no longer in the IRModule.
646 return {};
647 } else {
648 BaseFunc base_func = module_->Lookup(GetRef<GlobalVar>(global_var_node));
649 return ResolveToPrimitive(base_func);
650 }
651 } else if (const auto* prim_func_node = expr.as<tir::PrimFuncNode>()) {
652 return GetRef<tir::PrimFunc>(prim_func_node);
653 } else if (const auto* var_node = expr.as<VarNode>()) {
654 auto itr = primitive_functions_.find(var_node);
655 if (itr == primitive_functions_.end()) {
656 // Not bound to a primitive function.
657 return {};
658 } else {
659 return itr->second;
660 }
661 } else if (const auto* function_node = expr.as<FunctionNode>()) {
662 if (function_node->HasNonzeroAttr(attr::kExtern)) {
663 // We have a regular call to an 'extern' function. The call itself needs to be rewritten
664 // to call_lowered form, and any required dynamic shape functions generated and
665 // cross-linked.
666 return GetRef<Function>(function_node);
667 } else if (function_node->HasNonzeroAttr(attr::kPrimitive)) {
668 if (const auto* call_node = function_node->body.as<CallNode>()) {
669 if (call_node->op == debug_op_) {
670 // Debug 'primitives' are not lowered.
671 return {};
672 }
673 }
674 // We have a regular call to a 'primitive' function (possibly with a 'Compiler' attribute).
675 // We need to lower and rewrite the call.
676 return GetRef<Function>(function_node);
677 } else {
678 // Not marked as primitive during partitioning or TVM fusion.
679 return {};
680 }
681 } else {
682 return {};
683 }
684 }
685
686 /*!
687 * \brief Returns a 'call_lowered' call to \p prim_fn_var with \p args and \p span with all the
688 * required attributes filled in. Generally \p prim_fn_var will correspond to the lowered or
689 * externally codegen-ed form of \p original_function, where \p lowered_functions binds all
690 * the required lowered functions.
691 *
692 * The call's attributes will capture:
693 * - Any attributes on the original_function.
694 * - All the lowered functions.
695 * TODO(mbs): Pretty sure that's no longer needed.
696 * - Details needed to cross-link the call to it's dynamic shape function, if any.
697 */
698 Expr MakeLoweredCall(const BaseFunc& original_function, const GlobalVar& prim_fn_var,
699 Array<Expr> args, Span span, const Target& target,
700 const Map<GlobalVar, BaseFunc>& lowered_functions) {
701 auto opt_compiler = original_function->GetAttr<String>(attr::kCompiler);
702
703 // Add some metadata on top of the *original function* and invoke the callback so it can
704 // be captured.
705 // TODO(@areusch, @jroesch): this metadata is for AOT, this should be our interface for AOT
706 Map<GlobalVar, tir::PrimFunc> prim_fns;
707 Array<GlobalVar> all_prim_fn_vars;
708 for (const auto& kv : lowered_functions) {
709 if (opt_compiler) {
710 // We expect the original function to have just the "Extern" attribute signaling the
711 // function (will be) compiled externally.
712 ICHECK(kv.second.as<FunctionNode>())
713 << PrettyPrint(kv.first) << " must be bound to an (external) Function";
714 } else {
715 // We expect one or more PrimFuncs, one of which corresponds to 'the' lowered primitive,
716 // and the rest are in support of that via tir::Calls.
717 ICHECK(kv.second.as<tir::PrimFuncNode>())
718 << PrettyPrint(kv.first) << " must be bound to a PrimFunc";
719 prim_fns.Set(kv.first, Downcast<tir::PrimFunc>(kv.second));
720 all_prim_fn_vars.push_back(kv.first);
721 }
722 }
723
724 // Alas, WithAttr cannot work with base classes.
725 if (const auto* prim_func_node = original_function.as<te::PrimFuncNode>()) {
726 auto func_with_metadata = GetRef<te::PrimFunc>(prim_func_node);
727 func_with_metadata = WithAttr(func_with_metadata, "prim_fn_var", prim_fn_var);
728 func_with_metadata = WithAttr(func_with_metadata, "prim_funcs", prim_fns);
729 func_with_metadata = WithAttr(func_with_metadata, tvm::attr::kTarget, target);
730 this->process_fn_(func_with_metadata);
731 } else {
732 const auto* function_node = original_function.as<FunctionNode>();
733 ICHECK(function_node);
734 auto func_with_metadata = GetRef<Function>(function_node);
735 func_with_metadata = WithAttr(func_with_metadata, "prim_fn_var", prim_fn_var);
736 func_with_metadata = WithAttr(func_with_metadata, "prim_funcs", prim_fns);
737 func_with_metadata = WithAttr(func_with_metadata, tvm::attr::kTarget, target);
738 this->process_fn_(func_with_metadata);
739 }
740
741 // Now prepare the attributes of the call_lowered.
742 CallLoweredAttrs call_lowered_attrs;
743
744 // TODO(mbs): "reshape" cleanup.
745 if (!opt_compiler && original_function->HasNonzeroAttr(attr::kReshapeOnly)) {
746 call_lowered_attrs.metadata.Set(attr::kReshapeOnly, tvm::Integer(1));
747 }
748
749 call_lowered_attrs.metadata.Set("relay_attrs", original_function->attrs);
750 call_lowered_attrs.metadata.Set("all_prim_fn_vars", all_prim_fn_vars);
751
752 if (const auto* function_node = original_function.as<FunctionNode>()) {
753 if (IsDynamic(function_node->ret_type)) {
754 // Create a dynamic shape function to calculate the expected shape of the results of
755 // the lowered function.
756 // Shape function keys use the original function as their 'function', but the generic 'cpu'
757 // target as the target since all shape functions run on the host cpu irrespective of where
758 // the primitive runs.
759 CCacheKey shape_key(GetRef<Function>(function_node), config_->host_virtual_device->target);
760 CachedFunc lowered_shape_func = compiler_->LowerShapeFunc(shape_key);
761
762 // Capture the shape function's global var and parameters 'states' in call
763 // annotations so calling convention can be recovered.
764 // TODO(mbs): Shape cleanup.
765 call_lowered_attrs.metadata.Set("prim_shape_fn_var", lowered_shape_func->prim_fn_var);
766 call_lowered_attrs.metadata.Set("prim_shape_fn_states",
767 lowered_shape_func->shape_func_param_states);
768 call_lowered_attrs.metadata.Set(
769 "prim_shape_fn_num_inputs",
770 Integer(static_cast<int>(lowered_shape_func->inputs.size())));
771 call_lowered_attrs.metadata.Set(
772 "prim_shape_fn_num_outputs",
773 Integer(static_cast<int>(lowered_shape_func->outputs.size())));
774 Array<GlobalVar> all_prim_shape_fn_vars;
775 for (const auto& kv : lowered_shape_func->funcs->functions) {
776 CHECK(kv.second.as<tir::PrimFuncNode>()) << "must be a prim fn";
777 all_prim_shape_fn_vars.push_back(kv.first);
778 }
779 call_lowered_attrs.metadata.Set("all_prim_shape_fn_vars", all_prim_shape_fn_vars);
780 }
781 }
782
783 return CallLowered(prim_fn_var, std::move(args), std::move(call_lowered_attrs),
784 std::move(span));
785 }

810 Expr DeviceAwareVisitExpr_(const FunctionNode* function_node) override {

812 function_node->HasNonzeroAttr(attr::kExtern)) {

819
820 Expr DeviceAwareVisitExpr_(const CallNode* call_node) override {
821 // We can see six forms of calls:
822 // 1. A 'normal' Relay call to a Function with the "Primitive" attribute and not "Compiler"
823 // attribute. We will need to lower that to a global PrimFunc and rewrite the call to:
824 // call_lowered(@new_global, (arg1, ..., argn), <attributes>)
825 // If needed, the call needs to be cross-linked with any dynamic shape functions.
826 // (However, some primitives are special and handled separately.)
827 // 2. A 'normal' Relay call to a Function with the "Primitive" and "Compiler" attributes. We
828 // will need to invoke the "relay.ext.<compiler>" function to yield a runtime module, and
829 // rewrite the call to the same form as above. Dynamic shape function cross-linking may
830 // also be needed.
831 // 3. A 'normal' Relay call to a Function with the "Extern" attribute. This function has
832 // already been compiled by an external codegen and a definition for it exists in some
833 // runtime module. Again, we rewrite to call_lowered form, and cross-link with a dynamic
834 // shape function if needed.
835 // 4. A 'normal' Relay call to a PrimFunc which has already been supplied via a global
836 // definition. We rewrite those to use the call_lowered form, but otherwise nothing else
837 // needs to be done.
838 // 5. A 'call_lowered' call from an earlier invocation of this pass or otherwise deliberately
839 // inserted. It has all the required attributes, and any associated dynamic shape function
840 // has been generated and cross-linked. These calls are not changed.
841 // 6. A 'normal' Relay call to a Relay Function without any special attribute. These
842 // calls are not changed.
843 //
844 // Note that ResolveToPrimitive will yield non-null only for cases 1-4.
845
846 // Prepare the arguments and op.
847 Array<Expr> new_args;
848 for (const auto& arg : call_node->args) {
849 new_args.push_back(VisitExpr(arg));
850 }
851 Expr new_op = VisitExpr(call_node->op);
852
853 // Look for (possibly indirect) calls to primitives.
854 BaseFunc primitive_func = ResolveToPrimitive(call_node->op);
855 if (!primitive_func.defined()) {
856 // Cases 5 and 6: Leave as ordinary call.
857 if (const auto* function_node = call_node->op.as<FunctionNode>()) {
858 process_fn_(GetRef<Function>(function_node));
859 }
860 return WithFields(GetRef<Call>(call_node), std::move(new_op), std::move(new_args));
861 }
862
863 // Special case for case 1: device_copies are left as calls to primitive operators
864 // so that each backend can handle them directly.
865 // TODO(mbs): device_copy cleanup. Would be better for FuseOps to just leave device_copy alone.
866 if (const auto* function_node = primitive_func.as<FunctionNode>()) {
867 DeviceCopyProps device_copy_props = GetDeviceCopyProps(function_node->body);
868 if (device_copy_props.body.defined()) {
869 ICHECK_EQ(new_args.size(), 1);
870 return DeviceCopy(new_args[0], device_copy_props.src_virtual_device,
871 device_copy_props.dst_virtual_device);
872 }
873 }
874
875 ICHECK(call_node->type_args.empty()) << "lowered functions cannot be polymorphic";
876
877 // Case 4: If the function has already been lowered we just need to update the call.
878 if (const auto* prim_func_node = primitive_func.as<tir::PrimFuncNode>()) {
879 // Function should already be Target annotated by this point
880 // but the TE Compiler metadata is still needed for the callback
881 // TODO(Mousius) - Robustify this to not assume we're in the GlobalVar for Target Hooks
882 Optional<Target> opt_target = primitive_func->GetAttr<Target>(tvm::attr::kTarget);
883 ICHECK(opt_target.defined());
884 auto prim_fn_var = Downcast<GlobalVar>(call_node->op);
885 tir::PrimFunc prim_func = GetRef<tir::PrimFunc>(prim_func_node);
886 Map<GlobalVar, BaseFunc> prim_fns = {{prim_fn_var, prim_func}};
887 return MakeLoweredCall(primitive_func, prim_fn_var, std::move(new_args), call_node->span,
888 opt_target.value(), prim_fns);
889 }
890
891 // Determine the target for lowering or external codegen.
892 Target target;
893 Optional<String> opt_compiler = primitive_func->GetAttr<String>(attr::kCompiler);
894 if (opt_compiler.defined()) {
895 // This function needs to be compiled with external codegen.
896 Optional<Target> opt_target = config_->FindPrimitiveTargetForKind(opt_compiler.value());
897 if (opt_target.defined()) {
898 // The target is what's supplied by the compilation config for kind matching the
899 // "Compiler" name.
900 target = opt_target.value();
901 } else {
902 // Legacy fallback.
903 target = Target("ext_dev");
904 }
905 } else {
906 // The target corresponding to the call_node expression's annotation.
907 VirtualDevice virtual_device = GetVirtualDevice(GetRef<Call>(call_node));
908 ICHECK(!virtual_device->IsFullyUnconstrained()) << PrettyPrint(GetRef<Call>(call_node));
909 target = virtual_device->target;
910 ICHECK(target.defined());
911 }
912
913 if (primitive_func->HasNonzeroAttr(attr::kExtern)) {
914 // Case 3: Function has already been compiled.
915 GlobalVar prim_fn_var = Downcast<GlobalVar>(call_node->op);
916 return MakeLoweredCall(primitive_func, prim_fn_var, std::move(new_args), call_node->span,
917 target, /*lowered_functions=*/{});
918 } else {
919 // Cases 1 and 2: lower the primitive function for the desired target, possibly using external
920 // codegen.
921 CCacheKey key(Downcast<Function>(primitive_func), target,
922 GetVirtualDevice(GetRef<Call>(call_node)));
923 CachedFunc cfunc = compiler_->Lower(key);
924 ICHECK(cfunc.defined());
925 return MakeLoweredCall(primitive_func, cfunc->prim_fn_var, std::move(new_args),
926 call_node->span, target, cfunc->funcs->functions);
927 }
928 }

930 IRModule module_;
931 ProcessFn process_fn_;
932 /*! \brief All available targets. */
933 CompilationConfig config_;
934 // Map from in-scope let-bound variables to Functions known to be primitive, or PrimFuncs which
935 // have already been lowered. We'll rewrite these to the fresh global vars bound to the lowered
936 // primitive function as we go. Those vars will be bound in the target device-type specific
937 // module we'll ultimately emit for each required device-type. Note that a primitive may be
938 // lowered for multiple device types, each which will be assigned a fresh var.
939 std::unordered_map<const VarNode*, BaseFunc> primitive_functions_;
940 TECompiler compiler_;
941 // Cache ops that need to be frequently used later to reduce lookup overhead.
942 const Op& debug_op_;
943};

954backend::FunctionInfo UpdateMainWorkspaceSize(const IRModule& mod, const CompilationConfig& config,
955 Map<Expr, backend::StorageInfo> storage_info_map) {
956 Function func = Downcast<Function>(mod->Lookup("main"));
957
958 VLOG_CONTEXT << "UpdateMainWorkspaceSize";
959 VLOG(1) << "calculating FunctionInfo for main:" << std::endl << PrettyPrint(func);
960
961 // This is a Map<device,Map<storage_id, size>>
962 // TODO(mbs): Collapsing VirtualDevices to just device type.
963 std::unordered_map<DLDeviceType, std::unordered_map<int, int>, backend::EnumClassHash>
964 sid_workspace;
965 // This is a Map<device, size_of_inputs_and_outputs>
966 std::unordered_map<DLDeviceType, int, backend::EnumClassHash> device_io;
967 // This is a Map<device, size_of_constants>
968 std::unordered_map<DLDeviceType, int, backend::EnumClassHash> device_consts;
969
970 // Initialize the mapping from all storage identifiers to workspace sizes,
971 // the amount of device io, and the device constants.
972 for (const auto& kv : storage_info_map) {
973 const backend::StorageInfo& storage_info = kv.second;
974 const std::vector<int64_t>& storage_ids = storage_info->storage_ids;
975 const std::vector<VirtualDevice>& virtual_devices = storage_info->virtual_devices;
976 CHECK_EQ(storage_ids.size(), virtual_devices.size());
977 for (uint32_t i = 0; i < virtual_devices.size(); i++) {
978 DLDeviceType device_type = virtual_devices[i]->device_type();
979 sid_workspace[device_type][storage_ids[i]] = 0;
980 device_io[device_type] = 0;
981 device_consts[device_type] = 0;
982 }
983 }
984
985 // Iterate the storage map to compute all the tensor sizes in the program.
986 // There are 3 cases in this code:
987 //
988 // First we need to compute the sizes of all
989 // inline constants.
990 //
991 // Second we compute the size of any bound variable as these are input and output
992 // sizes of the program.
993 //
994 // Finally for all other expressions we check which storage identifier they have
995 // been assigned and we compute the maximal size of the storage, as tensors can
996 // share storage with other tensors which are the same size or larger.
997 //
998 // In this final case there is only one allocation for all tensors which share storage
999 // which will be the maximal size of all tensors which were assigned to it.
1000 for (const auto& kv : storage_info_map) {
1001 const Expr& expr = kv.first;
1002 const backend::StorageInfo& storage_info = kv.second;
1003 int64_t size_bytes = backend::CalculateRelayExprSizeBytes(expr->checked_type());
1004 VLOG(1) << "expression:" << std::endl
1005 << PrettyPrint(expr) << std::endl
1006 << "of type:" << std::endl
1007 << PrettyPrint(expr->checked_type()) << std::endl
1008 << "has size " << size_bytes << " and storage info:" << std::endl
1009 << storage_info;
1010 const std::vector<int64_t>& storage_ids = storage_info->storage_ids;
1011 const std::vector<VirtualDevice>& virtual_devices = storage_info->virtual_devices;
1012
1013 if (expr->IsInstance<ConstantNode>()) {
1014 for (const auto& virtual_device : virtual_devices) {
1015 DLDeviceType device_type = virtual_device->device_type();
1016 ICHECK_EQ(device_consts.count(device_type), 1);
1017 device_consts[device_type] += size_bytes;
1018 }
1019 } else if (expr->IsInstance<VarNode>() || expr.same_as(func->body)) {
1020 CHECK(size_bytes == 0 || virtual_devices.size() >= 1) << "must be at least one device";
1021 for (const auto& virtual_device : virtual_devices) {
1022 DLDeviceType device_type = virtual_device->device_type();
1023 device_io[device_type] += size_bytes;
1024 }
1025 } else {
1026 // TODO(@electriclilies): This code is never being called which means sid_workspace is not
1027 // updated.. This means that storage info is probably not being created correctly. Or is not
1028 // equivalent to what was here previously
1029 for (uint32_t i = 0; i < storage_ids.size(); i++) {
1030 // Here we record the largest size of the tensor
1031 // that share the same storage id, because storage_id will
1032 // be shared between multiple tensors that are not live simultaneously.
1033 DLDeviceType device_type = virtual_devices[i]->device_type();
1034 if (size_bytes > sid_workspace[device_type][storage_ids[i]]) {
1035 sid_workspace[device_type][storage_ids[i]] = size_bytes;
1036 }
1037 }
1038 }
1039 }
1040
1041 // This is a Map<device, workspace_size>
1042 std::unordered_map<DLDeviceType, int, backend::EnumClassHash> device_workspace;
1043 // Once we know the sizes of sids, we need to accumulate per device
1044 for (const auto& dev_sid_size : sid_workspace) {
1045 auto dev = dev_sid_size.first;
1046 device_workspace[dev] = 0;
1047 for (const auto& sid_size : dev_sid_size.second) {
1048 device_workspace[dev] += sid_size.second;
1049 }
1050 }
1051
1052 Map<Target, Integer> workspace_sizes;
1053 Map<Target, Integer> io_sizes;
1054 Map<Target, Integer> constant_sizes;
1055 Map<Target, tir::PrimFunc> tir_primfuncs;
1056 Map<Target, Function> relay_primfuncs;
1057
1058 // Initialize all target workspaces to zero
1059 for (const auto& target : config->primitive_targets) {
1060 workspace_sizes.Set(target, 0);
1061 }
1062
1063 for (const auto& dev_and_size : device_workspace) {
1064 Target target = config->FindPrimitiveTargetForDeviceOrFail(dev_and_size.first);
1065 workspace_sizes.Set(target, dev_and_size.second);
1066 relay_primfuncs.Set(target, func);
1067 }
1068 for (const auto& dev_and_size : device_io) {
1069 Target target = config->FindPrimitiveTargetForDeviceOrFail(dev_and_size.first);
1070 io_sizes.Set(target, dev_and_size.second);
1071 }
1072
1073 for (const auto& dev_and_size : device_consts) {
1074 Target target = config->FindPrimitiveTargetForDeviceOrFail(dev_and_size.first);
1075 ICHECK_EQ(constant_sizes.count(target), 0);
1076 constant_sizes.Set(target, dev_and_size.second);
1077 }
1078
1079 backend::FunctionInfo func_info(std::move(workspace_sizes), std::move(io_sizes),
1080 std::move(constant_sizes), std::move(tir_primfuncs),
1081 std::move(relay_primfuncs));
1082 VLOG(1) << "func_info: " << func_info;
1083 return std::move(func_info);
1084}

1086/*!
1087 * \brief A function to create the function metadata for an input function (ie calculate buffer
1088 * input/output sizes)
1089 * \param func The function to calculate function metadata for
1090 * \param function_metadata The map that stores all the function metadatas
1091 */
1092void UpdateFunctionMetadata(BaseFunc func,
1093 Map<String, backend::FunctionInfo>& function_metadata, // NOLINT(*)
1094 Integer workspace_byte_alignment) {
1095 VLOG_CONTEXT << "UpdateFunctionMetadata";
1096 VLOG(1) << "updating function metadata for:" << std::endl << PrettyPrint(func);
1097 // Originally UpdateFunctionMetadata took in CCachedFunc and looped through all the funcs stored
1098 // there Now the goal is to take only one func because process_fn should be controlling the
1099 // iteration However, to do the workspace calculations we need the primfuncs. So process_fn
1100 // needs to either access the cached funcs or be directly passed primfuncs This is bad and
1101 // ideally we don't want process_fn to look at primfuncs There's also the question now of what
1102 // the function metadatas are and how they are used if we can do something else to replicate the
1103 // behavior of the function metadatas that might be good (ie annotating functions or something).
1104 Map<Target, Integer> workspace_sizes;
1105 Map<Target, Integer> io_sizes;
1106 Map<Target, Integer> constant_sizes;
1107 Map<Target, tir::PrimFunc> tir_primfuncs;
1108 Map<Target, Function> relay_primfuncs;
1109
1110 Optional<Map<GlobalVar, tir::PrimFunc>> prim_fns =
1111 func->GetAttr<Map<GlobalVar, tir::PrimFunc>>("prim_funcs");
1112 CHECK(prim_fns) << "primitive functions not set on Relay function by TECompiler.";
1113
1114 Optional<GlobalVar> prim_fn_var = func->GetAttr<GlobalVar>("prim_fn_var");
1115 CHECK(prim_fn_var) << "prim_fn_var must be set on Relay functions by TECompiler.";
1116
1117 Optional<Target> relay_target = func->GetAttr<Target>(tvm::attr::kTarget);
1118 CHECK(relay_target) << "target must be set on Relay functions by the TECompiler.";
1119
1120 for (const auto& kv : prim_fns.value()) {
1121 auto prim_fn = Downcast<tir::PrimFunc>(kv.second);
1122 CHECK(prim_fn.defined()) << "the primitive function must be defined";
1123
1124 Integer workspace_size = CalculateWorkspaceBytes(prim_fn, workspace_byte_alignment);
1125
1126 // Workspace sizes
1127 Target prim_fn_target;
1128 if (prim_fn->attrs->dict.count(tvm::attr::kTarget)) {
1129 prim_fn_target = Downcast<Target>(prim_fn->attrs->dict[tvm::attr::kTarget]);
1130 } else {
1131 prim_fn_target = relay_target.value();
1132 }
1133
1134 workspace_sizes.Set(prim_fn_target, workspace_size);
1135
1136 // Calculating size for I/O
1137 // TODO(mbs): See also the other three utils for calculating tensor bytesize.
1138 for (auto const& param : prim_fn->params) {
1139 bool not_a_buffer = prim_fn->buffer_map.count(param) == 0;
1140 if (not_a_buffer) {
1141 io_sizes.Set(prim_fn_target, 0);
1142 continue;
1143 }
1144
1145 auto p_shape = prim_fn->buffer_map[param]->shape;
1146 int num_of_elements = 1;
1147 for (const auto& dim_index_expr : p_shape) {
1148 if (dim_index_expr->IsInstance<IntImmNode>()) {
1149 num_of_elements *= dim_index_expr.as<IntImmNode>()->value;
1150 } else {
1151 // If shape is dynamic, we cannot calculate workspace in compile time.
1152 num_of_elements = 0;
1153 }
1154 }
1155 int element_size = prim_fn->buffer_map[param]->dtype.bytes();
1156 io_sizes.Set(prim_fn_target, element_size * num_of_elements);
1157 }
1158
1159 constant_sizes.Set(prim_fn_target, 0);
1160 tir_primfuncs.Set(prim_fn_target, prim_fn);
1161 if (func->IsInstance<FunctionNode>()) {
1162 relay_primfuncs.Set(prim_fn_target, Downcast<Function>(func));
1163 }
1164 }
1165
1166 backend::FunctionInfo fi = backend::FunctionInfo(
1167 std::move(workspace_sizes), std::move(io_sizes), std::move(constant_sizes),
1168 std::move(tir_primfuncs), std::move(relay_primfuncs));
1169
1170 VLOG(1) << "FunctionInfo: " << PrettyPrint(prim_fn_var.value()) << " = " << PrettyPrint(fi);
1171
1172 // The primitive function name here corresponds to the string we will use to generate
1173 // this Relay function at the low level.
1174 function_metadata.Set(prim_fn_var.value()->name_hint, fi);
1175}

1177/*! \brief Main lowering driving. */
1178IRModule LowerTE(const IRModule& module, const String& module_name, ProcessFn process_fn,
1179 CompilationConfig config) {
1180 TECompiler compiler(module, module_name);
1181
1182 // TODO(mbs): This is all unnecessarily convoluted. Better would be to accumulate the rewritten
1183 // module as we go (including rewritten Functions, lowered primitives, and runtime modules
1184 // generated by external toolchains), and use a pair of maps over vars and global vars
1185 // to global vars to remember which functions have already been lowered.
1186
1187 // Lower all the callees in module:
1188 // - Functions tagged with "Compiler" are unchanged (checked by CreateFunctionPass)
1189 // - Functions tagged with "Primitive" are unchanged (checked by LowerTensorExprMutator)
1190 // - Called functions tagged with "Compiler" are copied into the compiler cache with a fresh
1191 // GlobalVar, and calls updated (sticking with regular Relay Call).
1192 // - Calls to functions tagged with "Primitive" are compiled to PrimFuncs, and calls updated
1193 // (using call_lowered convention).
1194 IRModule updated_module =
1195 LowerTensorExpr(compiler, std::move(process_fn), std::move(config))(module);
1196
1197 // The Functions tagged with "Compiler" are now residing in the cache ready to be
1198 // compiled by LowerExternalFunctions. However we still need a record of them in the
1199 // IRModule so that the various executors can see which function names need to be
1200 // retrieved. They may, however, have been renamed.
1201 compiler->AddExterns(updated_module);
1202
1203 // Add the lowered functions.
1204 IRModule lowered_module = compiler->GetLoweredFunctions();
1205 VLOG(1) << "capturing " << lowered_module->functions.size() << " new lowered functions";
1206 for (const auto& kv : lowered_module->functions) {
1207 if (updated_module->ContainGlobalVar(kv.first->name_hint)) {
1208 LOG(FATAL) << "duplicate bindings for '" << kv.first->name_hint
1209 << "'. Existing is:" << std::endl
1210 << PrettyPrint(updated_module->Lookup(kv.first->name_hint)) << std::endl
1211 << "while new is:" << std::endl
1212 << PrettyPrint(kv.second);
1213 }
1214 updated_module->Add(kv.first, kv.second);
1215 }
1216
1217 // Invoke external codegen for all Functions in the cache tagged with "Compiler", and
1218 // annotate the module with the resulting runtime modules.
1219 // TODO(mbs): runtime modules should be first class rather than attributes.
1220 Array<runtime::Module> external_mods =
1221 module->GetAttr<Array<runtime::Module>>(tvm::attr::kExternalMods).value_or({});
1222 Array<runtime::Module> new_external_mods = compiler->LowerExternalFunctions();
1223 VLOG(1) << "capturing " << external_mods.size() << " existing and " << new_external_mods.size()
1224 << " new external modules";
1225 for (const auto& mod : new_external_mods) {
1226 external_mods.push_back(mod); // copy-on-write.
1227 }
1228
1229 // Annotate the module with C Device API context mapping (this is until we have Targets
1230 // annotated for the C Device API)
1231 // TODO(Mousius) - Remove "device_contexts" as soon as we have the graph annotated properly with
1232 // Targets
1233 Map<GlobalVar, String> device_contexts =
1234 module->GetAttr<Map<GlobalVar, String>>("device_contexts", Map<GlobalVar, String>()).value();
1235 Map<GlobalVar, String> new_device_contexts = compiler->GetDeviceContexts();
1236 VLOG(1) << "capturing " << device_contexts.size() << " existing and "
1237 << new_device_contexts.size() << " new device contexts for external functions";
1238 for (const auto& kv : new_device_contexts) {
1239 ICHECK_EQ(device_contexts.count(kv.first), 0);
1240 device_contexts.Set(kv.first, kv.second); // copy-on-write.
1241 }
1242
1243 updated_module = WithAttrs(updated_module, {{tvm::attr::kExternalMods, std::move(external_mods)},
1244 {"device_contexts", std::move(device_contexts)}});
1245
1246 if (backend::IsAutoSchedulerEnabled()) {
1247 // Capture all the 'operator weights', ie usage counts for each PrimFunc.
1248 Map<String, Integer> op_weights =
1249 module->GetAttr<Map<String, Integer>>("op_weights", Map<String, Integer>()).value();
1250 Map<String, Integer> new_op_weights = compiler->GetOpWeights();
1251 VLOG(1) << "capturing " << op_weights.size() << " existing and " << new_op_weights.size()
1252 << " new operator weights for PrimFuncs";
1253 for (const auto& kv : new_op_weights) {
1254 ICHECK_EQ(op_weights.count(kv.first), 0);
1255 op_weights.Set(kv.first, kv.second); // copy-on-write.
1256 }
1257 updated_module = WithAttr(updated_module, "op_weights", std::move(op_weights));
1258 }
1259
1260 return updated_module;
1261}

1263Map<Target, IRModule> GetPerTargetModules(IRModule mod) {
1264 std::unordered_map<Target, IRModule, backend::TargetStrHash, backend::TargetStrEqual>
1265 per_target_modules;
1266 for (const auto& kv : mod->functions) {
1267 const GlobalVar& var = kv.first;
1268 const BaseFunc& func = kv.second;
1269 if (func->IsInstance<tir::PrimFuncNode>()) {
1270 // Extract target
1271 Optional<Target> target = func->GetAttr<Target>(tvm::attr::kTarget);
1272 ICHECK(target) << "Target should be set at this point";
1273
1274 // Put the function in per_target_modules
1275 if (!per_target_modules.count(target.value())) {
1276 // Initialize the IRModule for this target with the attributes from the input IRModule
1277 IRModule target_module = IRModule({}, {}, {}, {}, mod->attrs);
1278 // Add the function to the IRModule
1279 target_module->Add(var, func);
1280 per_target_modules[target.value()] = target_module;
1281 } else {
1282 // The IRModule for this target is initialized, so just add the function.
1283 IRModule target_module = per_target_modules.at(target.value());
1284 target_module->Add(var, func);
1285 }
1286 } else if (!func->IsInstance<relay::FunctionNode>()) {
1287 LOG(FATAL)
1288 << "The function types in the IRModule should be RelayFunction or PrimFunc, but got "
1289 << func->GetTypeKey();
1290 }
1291 }
1292 return per_target_modules;
1293}

1295Pass LowerTE(String module_name, CompilationConfig complilation_config, ProcessFn process_fn) {
1296 runtime::TypedPackedFunc<IRModule(IRModule, PassContext)> pass_func = [=](IRModule module,
1297 PassContext ctx) {
1298 return LowerTE(module, module_name, process_fn, complilation_config);
1299 };
1300
1301 return tvm::transform::Sequential(
1302 {tvm::relay::transform::RelayToTIRTargetHook(complilation_config),
1303 tvm::transform::CreateModulePass(pass_func, 0, "LowerTE", {"InferType"}), InferType(),
1304 tvm::tir::transform::ExtractPrimFuncConstants()});
1305}
