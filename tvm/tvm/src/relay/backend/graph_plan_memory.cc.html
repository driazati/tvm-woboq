53/*! A representation of a block of memory required at runtime on some device. */
54struct StorageToken {
55 /*! \brief Reference counter */
56 int ref_counter{0};
57 /*! \brief number of bytes */
58 size_t max_bytes{0};
59 /*! \brief The corresponding tensor type. */
60 TensorType ttype{nullptr};
61 /*! \brief VirtualDevice on which the memory will reside. */
62 VirtualDevice virtual_device = VirtualDevice::FullyUnconstrained();
63 /*! \brief The storage id */
64 int64_t storage_id{-1};
65
66 bool is_valid() const { return !virtual_device->IsFullyUnconstrained(); }
67
68 bool is_compatible(const StorageToken& that) const {
69 return virtual_device == that.virtual_device;
70 }
71
72 std::string ToString() const {
73 std::ostringstream os;
74 os << "{storage_id: " << storage_id << ", max_bytes: " << max_bytes
75 << ", ttype: " << PrettyPrint(ttype) << ", virtual_device: " << virtual_device << "}";
76 return os.str();
77 }
78};

80class StorageAllocaBaseVisitor : public transform::DeviceAwareExprVisitor {
81 public:
82 StorageAllocaBaseVisitor() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
83
84 // run the visitor on a global function.
85 void Run(const Function& func) { VisitExpr(func); }
86
87 using transform::DeviceAwareExprVisitor::VisitExpr_;
88
89 void VisitExpr_(const ConstantNode* op) final { this->CreateToken(op, false); }
90
91 void VisitExpr_(const VarNode* op) final {
92 // Do nothing.
93 }
94
95 void DeviceAwareVisitExpr_(const FunctionNode* func_node) final {
96 if (function_nesting() > 1) {
97 // do not recurse into sub functions.
98 return;
99 }
100 if (func_node->HasNonzeroAttr(attr::kPrimitive)) {
101 // No storage needed for primitive functions.
102 return;
103 }
104 for (const auto& param : func_node->params) {
105 CreateToken(param.get(), /*can_realloc=*/false);
106 }
107 // Process the function body, and make sure all result tokens are considered 'alive'.
108 for (StorageToken* tok : GetToken(func_node->body)) {
109 tok->ref_counter += 1;
110 }
111 }
112
113 void VisitExpr_(const GlobalVarNode* op) final {
114 // Do nothing.
115 }
116
117 void VisitExpr_(const OpNode* op) final {
118 // Do nothing.
119 }
120
121 void VisitExpr_(const TupleNode* op) final {
122 std::vector<StorageToken*> fields;
123 for (Expr field : op->fields) {
124 auto tokens = GetToken(field);
125 fields.insert(fields.end(), tokens.begin(), tokens.end());
126 }
127 token_map_[op] = fields;
128 }
129
130 void VisitExpr_(const TupleGetItemNode* op) final {
131 const auto& tok = GetToken(op->tuple);
132 ICHECK_LT(static_cast<size_t>(op->index), tok.size());
133 token_map_[op] = {tok[op->index]};
134 }
135
136 void VisitExpr_(const IfNode* op) final { LOG(FATAL) << "if is not supported."; }
137
138 void PreVisitLetBinding_(const Var& var, const Expr& value) final {
139 token_map_[var.get()] = GetToken(value);
140 }
141
142 void PostVisitLet_(const LetNode* let_node) final {
143 token_map_[let_node] = GetToken(let_node->body);
144 }
145
146 protected:
147 /*! \brief internal token map */
148 std::unordered_map<const ExprNode*, std::vector<StorageToken*>> token_map_;
149 /*! \brief empty token map */
150 const std::vector<StorageToken*> no_tokens_;
151
152 /*!
153 * \brief Get the necessary token.
154 * \param expr The expression.
155 * \return The corresponding token.
156 */
157 const std::vector<StorageToken*>& GetToken(const Expr& expr) {
158 this->VisitExpr(expr);
159 // See through on_device calls.
160 Expr real_expr = IgnoreOnDevice(expr);
161
162 // Functions don't require data storage, represented by the empty token
163 if (real_expr->checked_type().as<FuncTypeNode>()) {
164 return no_tokens_;
165 }
166 this->VisitExpr(real_expr);
167 auto it = token_map_.find(real_expr.get());
168 ICHECK(it != token_map_.end()) << "Expression not found in storage map:" << std::endl
169 << PrettyPrint(real_expr);
170 return it->second;
171 }
172
173 /*!
174 * \brief Allocates (or reuses if \p can_realloc is true) a storage token for holding
175 * the result of evaluating \p op.
176 */
177 void CreateToken(const ExprNode* expr_node, bool can_realloc) {
178 return CreateTokenOnDevice(expr_node, GetVirtualDevice(GetRef<Expr>(expr_node)), can_realloc);
179 }
180
181 /*!
182 * \brief Allocates (or reuses if \p can_realloc is true) a storage token for holding
183 * the result of evaluating \p op on \p device_type.
184 */
185 virtual void CreateTokenOnDevice(const ExprNode* op, const VirtualDevice& virtual_device,
186 bool can_realloc) = 0;
187};

189/*! \brief Associate storage with every expression without any concern for sharing. */
190class StorageAllocaInit : protected StorageAllocaBaseVisitor {
191 public:
192 explicit StorageAllocaInit(support::Arena* arena) : arena_(arena) {}
193
194 /*! \return The internal token map */
195 std::unordered_map<const ExprNode*, std::vector<StorageToken*>> GetInitTokenMap(
196 const Function& func) {
197 this->Run(func);
198 return std::move(token_map_);
199 }
200
201 protected:
202 using StorageAllocaBaseVisitor::VisitExpr_;
203
204 void CreateTokenOnDevice(const ExprNode* op, const VirtualDevice& virtual_device,
205 bool can_realloc) override {
206 ICHECK(!token_map_.count(op));
207 std::vector<StorageToken*> tokens;
208 for (const auto& ttype : FlattenTupleType(op->checked_type())) {
209 auto* token = arena_->make<StorageToken>();
210 token->ttype = ttype;
211 token->virtual_device = virtual_device;
212 tokens.push_back(token);
213 }
214 token_map_[op] = tokens;
215 }
216
217 using StorageAllocaBaseVisitor::DeviceAwareVisitExpr_;
218
219 void DeviceAwareVisitExpr_(const CallNode* call_node) final {
220 // create token for the call node.
221 CreateToken(call_node, true);
222
223 // for each input, visit argument token.
224 for (Expr arg : call_node->args) {
225 for (StorageToken* tok : GetToken(arg)) {
226 tok->ref_counter += 1;
227 }
228 }
229 }
230
231 private:
232 // allocator
233 support::Arena* arena_;
234 Map<Expr, Array<String>> node_storage_map_;
235};

237/*! \brief Associate storage with every expression, reusing storage where possible. */
238class StorageAllocator : public StorageAllocaBaseVisitor {
239 public:
240 StorageAllocator() = default;
241
242 /*!
243 * \return total number of bytes allocated
244 */
245 size_t TotalAllocBytes() const {
246 size_t total = 0;
247 for (const auto* p : data_) {
248 total += p->max_bytes;
249 }
250 return total;
251 }
252
253 // Run storage allocation for a function.
254 StaticMemoryPlan Plan(const Function& func) {
255 VLOG_CONTEXT << "StorageAllocator";
256 VLOG(1) << "planning:" << std::endl << PrettyPrint(func);
257 prototype_ = StorageAllocaInit(&arena_).GetInitTokenMap(func);
258 this->Run(func);
259
260 // The value of smap contains two integer arrays where the first array
261 // contains the planned storage ids and the second holds the device types.
262 Map<Expr, backend::StorageInfo> smap;
263 int num_annotated_nodes = 0;
264 int num_nodes = 0;
265
266 for (const auto& kv : token_map_) {
267 std::vector<int64_t> storage_ids;
268 storage_ids.reserve(kv.second.size());
269 std::vector<VirtualDevice> virtual_devices;
270 virtual_devices.reserve(kv.second.size());
271 std::vector<int64_t> sid_sizes_byte;
272 sid_sizes_byte.reserve(kv.second.size());
273
274 for (StorageToken* tok : kv.second) {
275 VLOG(1) << "token: " << tok->ToString();
276 if (tok->is_valid()) {
277 num_annotated_nodes++;
278 }
279 num_nodes++;
280 storage_ids.push_back(tok->storage_id);
281 virtual_devices.push_back(tok->virtual_device);
282 sid_sizes_byte.push_back(allocator_.GetMemorySize(tok));
283 }
284 auto storage_info = backend::StorageInfo(std::move(storage_ids), std::move(virtual_devices),
285 std::move(sid_sizes_byte));
286 smap.Set(GetRef<Expr>(kv.first), storage_info);
287 }
288 // Either all or none of the nodes should be annotated.
289 VLOG(1) << "num annotated nodes / num_nodes: " << num_annotated_nodes << " / " << num_nodes
290 << std::endl;
291 if (num_annotated_nodes != 0 && num_annotated_nodes != num_nodes) {
292 LOG(FATAL) << num_annotated_nodes << " out of " << num_nodes
293 << "expressions are assigned with virtual device types. Either all "
294 "or none of the expressions are expected to be annotated.";
295 }
296 return backend::StaticMemoryPlan(smap);
297 }
298
299 protected:
300 // override create token by getting token as prototype requirements.
301 void CreateTokenOnDevice(const ExprNode* op, const VirtualDevice& virtual_device,
302 bool can_realloc) final {
303 ICHECK(!token_map_.count(op));
304 auto it = prototype_.find(op);
305 ICHECK(it != prototype_.end());
306 std::vector<StorageToken*> tokens;
307
308 for (StorageToken* tok : it->second) {
309 ICHECK(tok->virtual_device == virtual_device);
310 if (can_realloc) {
311 tokens.push_back(allocator_.Request(tok));
312 } else {
313 // Allocate a new token,
314 StorageToken* allocated_tok = allocator_.Alloc(tok);
315 allocated_tok->virtual_device = tok->virtual_device;
316 // ensure it never get de-allocated.
317 allocated_tok->ref_counter += 1;
318 tokens.push_back(allocated_tok);
319 }
320 }
321 token_map_[op] = tokens;
322 }
323
324 // Mark op to reuse the input_token
325 // tie the two memories together
326 void ReuseInputToken(const ExprNode* op, StorageToken* input_token) {
327 ICHECK(!token_map_.count(op));
328 auto it = prototype_.find(op);
329 ICHECK(it != prototype_.end());
330 ICHECK_EQ(it->second.size(), 1U);
331 StorageToken* prototype = it->second[0];
332 // add the reference counter of the output
333 // so the input token can only be deleted after references
334 // to both are expired
335 input_token->ref_counter += prototype->ref_counter;
336 // reuse the input token
337 token_map_[op] = {input_token};
338 }
339
340 using StorageAllocaBaseVisitor::DeviceAwareVisitExpr_;
341
342 // The call map
343 void DeviceAwareVisitExpr_(const CallNode* call_node) final {
344 std::vector<StorageToken*> args;
345 // for each input, visit argument token.
346
347 for (const Expr& arg : call_node->args) {
348 // Note: GetToken skips GlobalVars and handles tuples properly, so we don't need to treat
349 // call_lowered specially.
350 for (StorageToken* tok : GetToken(arg)) {
351 args.push_back(tok);
352 }
353 }
354
355 // Under the flat-memory setting.
356 // we can force aliasing the input and output of reshape
357 // to make it an nop. Note that this is not true
358 // for non-flat memory case. Given the current graph plan memory
359 // only works for flat memory case, we will go with this choice
360 //
361 // TODO(tvm-team) Update checks of flat memory enablement when we support
362 // opaque-nd memory planning to skip this path.
363 // TODO(mbs): "reshape" cleanup.
364 CallLoweredProps call_lowered_props = GetCallLoweredProps(call_node);
365 if (call_lowered_props.lowered_func.defined() && IsReshapeOnly(call_lowered_props)) {
366 ICHECK_EQ(call_lowered_props.arguments.size(), 1U);
367 ReuseInputToken(call_node, args[0]);
368 } else {
369 // create token for the call node.
370 CreateToken(call_node, true);
371 }
372
373 // check if there is orphaned output that can be released immediately.
374 for (StorageToken* tok : token_map_.at(call_node)) {
375 allocator_.CheckForRelease(tok);
376 }
377 for (StorageToken* tok : args) {
378 tok->ref_counter -= 1;
379 allocator_.CheckForRelease(tok);
380 }
381 }

383 /**
384 * @brief Memory manager for flattened 1d memory (buffers)
385 */
386 class TokenAllocator1D {
387 public:
388 /*!
389 * \brief ceil(size/word_size) to get number of words.
390 * \param size The original size.
391 * \param word_size The element size.
392 */
393 static size_t DivRoundUp(size_t size, size_t word_size) {
394 return (size + word_size - 1) / word_size;
395 }
396
397 /*!
398 * \brief Get the memory requirement.
399 * \param prototype The prototype token.
400 * \return The required memory size.
401 *
402 * TODO(mbs): Gf GetMemorySizeBytes in aot_executor_codegen.cc,
403 * CalculateRelayExprSizeBytes in utils.cc
404 */
405 size_t GetMemorySize(StorageToken* prototype) {
406 TensorType ttype = prototype->ttype;
407 ICHECK(ttype.defined());
408 size_t size = 1;
409 for (IndexExpr dim : ttype->shape) {
410 const int64_t* pval = tir::as_const_int(dim);
411 ICHECK(pval != nullptr) << "Cannot allocate memory symbolic tensor shape " << ttype->shape;
412 ICHECK_GE(*pval, 0) << "Cannot allocate memory for tensor with negative shape" << *pval;
413 size *= static_cast<size_t>(pval[0]);
414 }
415 size *= DivRoundUp(ttype->dtype.bits() * ttype->dtype.lanes(), 8);
416 return size;
417 }
418 /*!
419 * \brief Request a storage token for a given prototype.
420 * \param prototype. The prototype storage token.
421 * \return The result token.
422 */
423 StorageToken* Request(StorageToken* prototype) {
424 // calculate the size;
425 size_t size = GetMemorySize(prototype);
426 // search memory block in [size / match_range_, size * match_range_)
427 if (match_range_ == 0) {
428 return nullptr;
429 }
430 auto begin = free_.lower_bound(size / match_range_);
431 auto mid = free_.lower_bound(size);
432 auto end = free_.upper_bound(size * match_range_);
433 // search for memory blocks larger than requested
434 for (auto it = mid; it != end; ++it) {
435 StorageToken* tok = it->second;
436 if (!tok->is_compatible(*prototype)) continue;
437 ICHECK_EQ(tok->ref_counter, 0);
438 // Use exect matching strategy
439 tok->max_bytes = std::max(size, tok->max_bytes);
440 tok->ref_counter = prototype->ref_counter;
441 // find a exact match, erase from map and return
442 free_.erase(it);
443 return tok;
444 }
445 // then search for memory blocks smaller than requested space
446 for (auto it = mid; it != begin;) {
447 --it;
448 StorageToken* tok = it->second;
449 if (!tok->is_compatible(*prototype)) continue;
450 ICHECK_EQ(tok->ref_counter, 0);
451 // Use exect matching strategy
452 tok->max_bytes = std::max(size, tok->max_bytes);
453 tok->ref_counter = prototype->ref_counter;
454 // erase from map and return
455 free_.erase(it);
456 return tok;
457 }
458 return nullptr;
459 }
460 /*!
461 * \brief Alloacte a storage token by consuming prototype
462 * \param prototype The prototype token.
463 * \param size The size of memory being requested.
464 */
465 StorageToken* Alloc(StorageToken* prototype, int64_t storage_id) {
466 size_t size = GetMemorySize(prototype);
467 prototype->max_bytes = size;
468 prototype->storage_id = storage_id;
469 data_.push_back(prototype);
470 return prototype;
471 }
472 /*!
473 * \brief Check if we can release token.
474 * \param tok The token to be released.
475 */
476 void CheckForRelease(StorageToken* tok) {
477 ICHECK_GE(tok->storage_id, 0);
478 ICHECK_GE(tok->ref_counter, 0);
479 if (tok->ref_counter == 0) {
480 free_.insert({tok->max_bytes, tok});
481 }
482 }
483
484 private:
485 // scale used for rough match
486 const size_t match_range_{16};
487 // free list of storage entry
488 std::multimap<size_t, StorageToken*> free_;
489 // all the storage resources available
490 std::vector<StorageToken*> data_;
491 };

493 /**
494 * @brief Memory manager for 2d memory (textures)
495 */
496 class TokenAllocator2D {
497 public:
498 /*!
499 * \brief Request a storage token for a given prototype.
500 * \param prototype. The prototype storage token.
501 * \return The result token.
502 */
503 StorageToken* Request(StorageToken* prototype) {
504 auto shape = GetSize2D(prototype);
505 int64_t requested_size = shape.height * shape.width;
506 int64_t min_added_size = std::numeric_limits<int64_t>::max();
507 int64_t min_wasted_size = std::numeric_limits<int64_t>::max();
508 int64_t best_storage_id = -1;
509 MemBlock best_mem, new_mem;
510 for (int64_t free_id : free_list_) {
511 MemBlock& cached = blocks_[free_id];
512 // Can only reuse texture 2d blocks of the same type
513 if (cached.token_->ttype->dtype != prototype->ttype->dtype) {
514 continue;
515 }
516 int64_t cached_size = cached.x_ * cached.y_;
517 new_mem.x_ = std::max(cached.x_, shape.width);
518 new_mem.y_ = std::max(cached.y_, shape.height);
519 int64_t expanded_size = new_mem.x_ * new_mem.y_;
520 int64_t added_size = expanded_size - cached_size;
521 int64_t wasted_size = expanded_size - requested_size;
522 // Prioritize minimization of added size first, then minimize
523 // wasted size among blocks which would not require expansion
524 if ((min_added_size > 0 && added_size < min_added_size) ||
525 (min_added_size == 0 && wasted_size < min_wasted_size)) {
526 min_added_size = added_size;
527 min_wasted_size = wasted_size;
528 best_storage_id = free_id;
529 best_mem = new_mem;
530 }
531 }
532
533 if (min_added_size <= requested_size) {
534 best_mem.token_ = blocks_[best_storage_id].token_;
535 // Reset the reference counter of the now live token
536 best_mem.token_->ref_counter = prototype->ref_counter;
537 blocks_[best_storage_id] = best_mem;
538 free_list_.erase(best_storage_id);
539 return best_mem.token_;
540 }
541 return nullptr;
542 }
543 /*!
544 * \brief Alloacte a storage token by consuming prototype
545 * \param prototype The prototype token.
546 * \param size The size of memory being requested.
547 */
548 StorageToken* Alloc(StorageToken* prototype, int64_t storage_id) {
549 auto shape = GetSize2D(prototype);
550 MemBlock block;
551 block.x_ = shape.width;
552 block.y_ = shape.height;
553 prototype->storage_id = storage_id;
554 block.token_ = prototype;
555 blocks_[prototype->storage_id] = block;
556 return prototype;
557 }
558 /*!
559 * \brief Check if we can release token.
560 * \param tok The token to be released.
561 */
562 void CheckForRelease(StorageToken* tok) {
563 ICHECK_GE(tok->storage_id, 0);
564 ICHECK_GE(tok->ref_counter, 0);
565 if (tok->ref_counter == 0) {
566 free_list_.insert(tok->storage_id);
567 }
568 }
569 /*!
570 * \brief Get the texture 2d size requirement
571 * \param prototype The prototype token.
572 * \return The required texture 2d memory size in (width, height, channel).
573 */
574 Texture2DShape GetSize2D(StorageToken* prototype) {
575 TensorType ttype = prototype->ttype;
576 ICHECK(ttype.defined());
577 size_t axis = runtime::DefaultTextureLayoutSeparator(ttype->shape.size(),
578 prototype->virtual_device->memory_scope);
579 struct Shape {
580 const Array<PrimExpr>& shape;
581 int64_t operator[](size_t i) const { return *tir::as_const_int(shape[i]); }
582 };
583 return runtime::ApplyTexture2DFlattening<int64_t>(Shape{ttype->shape}, ttype->shape.size(),
584 axis);
585 }
586
587 private:
588 struct MemBlock {
589 StorageToken* token_;
590 int64_t x_;
591 int64_t y_;
592 };
593
594 std::unordered_map<int64_t, MemBlock> blocks_;
595 std::unordered_set<int64_t> free_list_;
596 };

598 class TokenAllocator {
599 public:
600 StorageToken* Alloc(StorageToken* proto) {
601 return Is2DStorage(proto) ? token_2d_.Alloc(proto, storage_ids_++)
602 : token_1d_.Alloc(proto, storage_ids_++);
603 }
604 StorageToken* Request(StorageToken* proto) {
605 StorageToken* token =
606 Is2DStorage(proto) ? token_2d_.Request(proto) : token_1d_.Request(proto);
607 return token ? token : this->Alloc(proto);
608 }
609 void CheckForRelease(StorageToken* tok) {
610 return Is2DStorage(tok) ? token_2d_.CheckForRelease(tok) : token_1d_.CheckForRelease(tok);
611 }
612
613 size_t GetMemorySize(StorageToken* tok) {
614 // TODO(amalyshe): figure out who requries sizes and for what
615 // size in case of texture is not enough - we can return any value if it
616 // assumed to be used for memory allocatoion or we can return real size
617 // if it is just for information
618 return Is2DStorage(tok) ? 0 : token_1d_.GetMemorySize(tok);
619 }
620 static bool Is2DStorage(StorageToken* tok) {
621 return relay::Is2DStorage(tok->virtual_device->memory_scope);
622 }
623
624 private:
625 int64_t storage_ids_{0};
626 TokenAllocator1D token_1d_;
627 TokenAllocator2D token_2d_;
628 };
629
630 private:
631 // allocator
632 support::Arena arena_;
633 // scale used for rough match
634 size_t match_range_{16};
635 // free list of storage entry
636 std::multimap<size_t, StorageToken*> free_;
637 // all the storage resources available
638 std::vector<StorageToken*> data_;
639 /*! \brief internal prototype token map */
640 std::unordered_map<const ExprNode*, std::vector<StorageToken*>> prototype_;
641 /*! \brief token allocator for optimizing 1d and 2d token alloc requests */
642 TokenAllocator allocator_;
643};
