32std::string ToCFunctionStyle(const std::string& original_name) {
33 ICHECK(!original_name.empty()) << "Function name is empty";
34 ICHECK_EQ(original_name.find("TVM"), 0) << "Function not TVM prefixed";
35
36 int tvm_prefix_length = 3;
37 std::string function_name("TVM");
38
39 bool new_block = true;
40 for (const char& symbol : original_name.substr(tvm_prefix_length)) {
41 if (std::isalpha(symbol)) {
42 if (new_block) {
43 function_name.push_back(std::toupper(symbol));
44 new_block = false;
45 } else {
46 function_name.push_back(std::tolower(symbol));
47 }
48 } else if (symbol == '_') {
49 new_block = true;
50 }
51 }
52 return function_name;
53}
54
55std::string ToCVariableStyle(const std::string& original_name) {
56 ICHECK(!original_name.empty()) << "Variable name is empty";
57 ICHECK_EQ(original_name.find("TVM"), 0) << "Variable not TVM prefixed";
58
59 std::string variable_name;
60 variable_name.resize(original_name.size());
61
62 std::transform(original_name.begin(), original_name.end(), variable_name.begin(), ::tolower);
63 return variable_name;
64}
65
66std::string ToCConstantStyle(const std::string& original_name) {
67 ICHECK_EQ(original_name.find("TVM"), 0) << "Constant not TVM prefixed";
68 std::string constant_name = ToCVariableStyle(original_name);
69
70 std::transform(constant_name.begin(), constant_name.end(), constant_name.begin(), ::toupper);
71 return constant_name;
72}

74std::string CombineNames(const Array<String>& names) {
75 std::stringstream combine_stream;
76 ICHECK(!names.empty()) << "Name segments empty";
77
78 for (const String& name : names) {
79 ICHECK(!name.empty()) << "Name segment is empty";
80 combine_stream << name << "_";
81 }
82
83 std::string combined_name = combine_stream.str();
84 combined_name.pop_back();
85 return combined_name;
86}
