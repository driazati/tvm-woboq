33Array<meta_schedule::ExtractedTask> ExtractTask(IRModule mod, Target target,
34 Map<String, runtime::NDArray> params) {
35 using meta_schedule::ExtractedTask;
36 backend::FTECompilerTIRConverter tir_converter = backend::GetTIRConverter();
37 backend::BindParamsInModule(mod, params);
38 // is_vm=true for backward compatibility
39 Array<Pass> pass_seqs = relay::backend::GetPassPrefix(/*is_homogenous=*/true, /*is_vm=*/true);
40 pass_seqs.push_back(transform::FuseOps());
41
42 mod = transform::Sequential(pass_seqs)(std::move(mod));
43
44 std::vector<ExtractedTask> tasks;
45 std::unordered_map<tec::CCacheKey, ExtractedTask> cache;
46 PostOrderVisit(mod->Lookup("main"), [&target, &tasks, &cache, &tir_converter](const Expr& exp) {
47 if (exp->IsInstance<FunctionNode>()) {
48 Function relay_func = Downcast<Function>(exp);
49 if (!relay_func->HasNonzeroAttr(attr::kPrimitive)) {
50 return;
51 }
52 tec::CCacheKey cache_key(relay_func, target);
53 auto it = cache.find(cache_key);
54 if (it != cache.end()) {
55 it->second->weight += 1;
56 return;
57 }
58 auto [inputs_outputs, constants, fused_name] =
59 tec::LowerTECompute(relay_func, target, /*return_inputs=*/true);
60 if (Optional<tir::PrimFunc> f = tir_converter(inputs_outputs, constants)) {
61 IRModule relay_mod({{GlobalVar(fused_name), relay_func}});
62 ExtractedTask task(fused_name, relay_mod, target, {PrimFuncToIRModule(f.value())}, 1);
63 tasks.push_back(task);
64 cache.emplace(cache_key, task);
65 }
66 }
67 });
68 // Tasks are extracted via post order visit, return the reversed list.
69 std::reverse(tasks.begin(), tasks.end());
70 NameSupply name_supply = NameSupply("");
71 for (ExtractedTask task : tasks) {
72 task->task_name = name_supply->FreshName(task->task_name);
73 }
74 return tasks;
75}

77TVM_REGISTER_GLOBAL("relay.backend.MetaScheduleExtractTask").set_body_typed(ExtractTask);
