37class OpCounter : public ExprVisitor {
38 public:
39 static size_t GetOpCount(relay::Function func) {
40 OpCounter counter;
41 counter(func->body);
42 return counter.count;
43 }
44
45 private:
46 void VisitExpr_(const CallNode* call) final {
47 if (call->op->IsInstance<OpNode>()) {
48 ++count;
49 }
50 ExprVisitor::VisitExpr_(call);
51 }
52
53 size_t count{0};
54};
55
56Array<meta_schedule::ExtractedTask> ExtractTask(IRModule mod, Target target,
57 Map<String, runtime::NDArray> params,
58 String mod_eq_name) {
59 using meta_schedule::ExtractedTask;
60 using meta_schedule::ModuleEqual;
61 using meta_schedule::ModuleHash;
62 backend::FTECompilerTIRConverter tir_converter = backend::GetTIRConverter();
63 backend::BindParamsInModule(mod, params);
64 // is_vm=true for backward compatibility
65 Array<Pass> pass_seqs = relay::backend::GetPassPrefix(/*is_homogenous=*/true, /*is_vm=*/true);
66 pass_seqs.push_back(transform::FuseOps());
67
68 mod = transform::Sequential(pass_seqs)(std::move(mod));
69
70 std::vector<ExtractedTask> tasks;
71
72 auto mod_eq = meta_schedule::ModuleEquality::Create(mod_eq_name);
73
74 std::unordered_map<IRModule, ExtractedTask, ModuleHash, ModuleEqual> cache(
75 /*bucket_count*/ 0, ModuleHash(*mod_eq), ModuleEqual(*mod_eq));
76
77 std::vector<std::tuple<std::string, Function, IRModule>> lower_results;
78
79 NameSupply constant_name_supply("");
80
81 PostOrderVisit(mod->Lookup("main"), [&](const Expr& exp) {
82 if (exp->IsInstance<FunctionNode>()) {
83 Function relay_func = Downcast<Function>(exp);
84 if (!relay_func->HasNonzeroAttr(attr::kPrimitive)) {
85 return;
86 }
87 auto [inputs_outputs, constants, fused_name] =
88 tec::LowerTECompute(relay_func, target, constant_name_supply, /*return_inputs=*/true);
89
90 if (Optional<tir::PrimFunc> f = tir_converter(inputs_outputs, constants)) {
91 IRModule tir_mod = PrimFuncToIRModule(f.value());
92 lower_results.push_back(std::make_tuple(fused_name, relay_func, tir_mod));
93 }
94 }
95 });
96
97 std::vector<int> indices(lower_results.size());
98 std::iota(indices.begin(), indices.end(), 0);
99
100 if (mod_eq_name == "anchor-block") {
101 std::vector<size_t> op_counts(lower_results.size());
102 for (size_t i = 0; i < op_counts.size(); ++i) {
103 op_counts[i] = OpCounter::GetOpCount(std::get<1>(lower_results[i]));
104 }
105
106 // When anchor-block based equality is used, tuning tasks "nn_conv2d_add_nn_relu" and
107 // "nn_conv2d_add_add_nn_relu", for example, can be identified as equal. Thus, one of
108 // them will be filtered by the cache below.
109 //
110 // To make sure that we tune "nn_conv2d_add_nn_relu" and not "nn_conv2d_add_add_nn_relu",
111 // we sort the TE lowering results based on the number of relay ops. This way,
112 // "nn_conv2d_add_nn_relu" will be added to the cache first, and "nn_conv2d_add_add_nn_relu"
113 // will be filtered.
114 std::sort(indices.begin(), indices.end(),
115 [&op_counts](int i1, int i2) { return op_counts[i1] < op_counts[i2]; });
116 }
117
118 for (auto i : indices) {
119 const auto& [fused_name, relay_func, tir_mod] = lower_results[i];
120 auto it = cache.find(tir_mod);
121 if (it != cache.end()) {
122 it->second->weight += 1;
123 continue;
124 }
125 // Note that the cache is key-ed on the tir mod, rather than the relay mod
126 IRModule relay_mod({{GlobalVar(fused_name), relay_func}});
127 ExtractedTask task(fused_name, relay_mod, target, {tir_mod}, 1);
128 tasks.push_back(task);
129 cache.emplace(tir_mod, task);
130 }
131
132 // Tasks are extracted via post order visit, return the reversed list.
133 std::reverse(tasks.begin(), tasks.end());
134 NameSupply name_supply = NameSupply("");
135 for (ExtractedTask task : tasks) {
136 task->task_name = name_supply->FreshName(task->task_name);
137 }
138 return tasks;
139}
