63/**
64 * This is an on demand allocator for AOT. A new temporary
65 * (storage allocator identifier) is allocated for each operation.
66 */
67class AOTOnDemandAllocator : public transform::DeviceAwareExprVisitor {
68 public:
69 AOTOnDemandAllocator() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
70
71 // run the visitor on a global function.
72 void Run(const Function& func) { VisitExpr(func); }
73
74 std::vector<int> GetReturnIds() const { return return_ids_; }
75 std::vector<TensorType> GetReturnTtypes() const { return return_ttypes_; }
76
77 StorageMap GetStorageMap() const { return storage_device_map_; }
78
79 using ExprVisitor::VisitExpr_;
80
81 void VisitExpr_(const ConstantNode* op) final {
82 CreateStorage(op);
83 AssignReturnSid(GetRef<Expr>(op));
84 }
85
86 void DeviceAwareVisitExpr_(const CallNode* call_node) final {
87 // AOTOnDemandAllocator is run both before and after lowering, so we need to handle the case
88 // where the op of the call is a generic function
89
90 Expr func;
91 Array<Expr> args;
92
93 CallLoweredProps call_lowered_props = GetCallLoweredProps(call_node);
94 if (call_lowered_props.lowered_func.defined()) {
95 func = call_lowered_props.lowered_func;
96 args = call_lowered_props.arguments;
97 } else { // Relay functions that have not been lowered and lowered extern functions
98 func = call_node->op;
99 args = call_node->args;
100 if (call_node->op.as<GlobalVarNode>()) { // Lowered extern function
101 ICHECK(!(call_node->attrs.defined())) << "Extern functions should have null attributes.";
102 } else { // Relay function which has not been lowered yet
103 ICHECK(call_node->op.as<FunctionNode>())
104 << "Expected the call to be to a lowered primfunc, a lowered extern function or a "
105 "unlowered Relay function.";
106 }
107 }
108 VisitExpr(func);
109 CreateStorage(call_node);
110 for (const Expr& arg : args) {
111 VisitExpr(arg);
112 }
113 AssignReturnSid(GetRef<Expr>(call_node));
114 }
115
116 void VisitExpr_(const VarNode* op) final { AssignReturnSid(GetRef<Expr>(op)); }
117
118 void DeviceAwareVisitExpr_(const FunctionNode* func_node) final {
119 if (function_nesting() > 1) {
120 // do not recurse into sub functions.
121 return;
122 }
123 if (func_node->HasNonzeroAttr(attr::kPrimitive)) {
124 // No storage needed for primitive functions.
125 return;
126 }
127 for (const auto& param : func_node->params) {
128 CreateStorage(param.get());
129 }
130 VisitExpr(func_node->body);
131 }
132
133 void VisitExpr_(const GlobalVarNode* op) final {
134 // Do nothing.
135 }
136
137 void VisitExpr_(const OpNode* op) final {
138 // Do nothing.
139 }
140
141 void VisitExpr_(const TupleNode* op) final {
142 std::vector<int64_t> storage_ids;
143 std::vector<VirtualDevice> virtual_devices;
144 std::vector<int64_t> storage_sizes_in_bytes;
145 Expr expr = GetRef<Expr>(op);
146 for (Expr field : op->fields) {
147 auto sid = GetStorage(field);
148 storage_ids.insert(storage_ids.end(), sid->storage_ids.begin(), sid->storage_ids.end());
149 virtual_devices.insert(virtual_devices.end(), sid->virtual_devices.begin(),
150 sid->virtual_devices.end());
151 storage_sizes_in_bytes.insert(storage_sizes_in_bytes.end(),
152 sid->storage_sizes_in_bytes.begin(),
153 sid->storage_sizes_in_bytes.end());
154 }
155 storage_device_map_[expr] = StorageInfo(storage_ids, virtual_devices, storage_sizes_in_bytes);
156 AssignReturnSid(expr);
157 }
158
159 void VisitExpr_(const TupleGetItemNode* op) final {
160 Expr expr = GetRef<Expr>(op);
161 auto sids = GetStorage(op->tuple);
162 ICHECK_LT(static_cast<size_t>(op->index), sids->storage_ids.size());
163 storage_device_map_[expr] =
164 StorageInfo({sids->storage_ids[op->index]}, {sids->virtual_devices[op->index]},
165 {sids->storage_sizes_in_bytes[op->index]});
166 AssignReturnSid(expr);
167 }
168
169 void VisitExpr_(const IfNode* op) final { LOG(FATAL) << "if is not supported."; }
170
171 void PreVisitLetBinding_(const Var& var, const Expr& value) final {
172 VisitExpr(value);
173 StorageInfo si = GetStorage(value);
174 storage_device_map_[var] = si;
175 }
176
177 private:
178 void AssignReturnSid(Expr e) {
179 if (storage_device_map_.find(e) != storage_device_map_.end()) {
180 StorageInfo& sinfo = storage_device_map_[e];
181 return_ids_.clear();
182 for (auto sid : sinfo->storage_ids) {
183 return_ids_.push_back(sid);
184 }
185 return_ttypes_.clear();
186 return_ttypes_ = FlattenTupleType(e->checked_type());
187 }
188 }
189 /*!
190 * \brief ceil(size/word_size) to get number of words.
191 * \param size The original size.
192 * \param word_size The element size.
193 */
194 static size_t DivRoundUp(size_t size, size_t word_size) {
195 return (size + word_size - 1) / word_size;
196 }
197 /*!
198 * \brief Get the memory requirement.
199 * \param prototype The prototype token.
200 * \return The required memory size.
201 *
202 * TODO(mbs): Cf CalculateRelayExprSizeBytes in utils.cc, GetMemorySize is graph_plan_memory.cc
203 */
204 size_t GetMemorySizeBytes(const TensorType& ttype) {
205 size_t size = 1;
206 for (IndexExpr dim : ttype->shape) {
207 const int64_t* pval = tir::as_const_int(dim);
208 ICHECK(pval != nullptr) << "Cannot allocate memory symbolic tensor shape " << ttype->shape;
209 ICHECK_GE(*pval, 0) << "Cannot allocate memory for tensor with negative shape" << *pval;
210 size *= static_cast<size_t>(pval[0]);
211 }
212 size *= DivRoundUp(ttype->dtype.bits() * ttype->dtype.lanes(), 8);
213 return size;
214 }
215 /*!
216 * \brief Get the necessary storage for the expression.
217 * \param expr The expression.
218 * \return The corresponding token.
219 */
220 StorageInfo GetStorage(const Expr& expr) {
221 // See through "on_device" calls.
222 Expr true_expr = IgnoreOnDevice(expr);
223 VisitExpr(true_expr);
224 auto it = storage_device_map_.find(true_expr);
225 ICHECK(it != storage_device_map_.end()) << "Could not find " << true_expr->GetTypeKey() << " "
226 << PrettyPrint(true_expr) << " in storage device map";
227 return it->second;
228 }
229
230 /*!
231 * \brief Create storage for the expression.
232 */
233 void CreateStorage(const ExprNode* op) {
234 Expr expr = GetRef<Expr>(op);
235 return CreateStorage(expr, GetVirtualDevice(expr));
236 }
237
238 /*!
239 * \brief Create storage to hold the result of evaluating \p expr in \p virtual_device.
240 */
241 void CreateStorage(const Expr& expr, const VirtualDevice& virtual_device) {
242 ICHECK(!virtual_device->IsFullyUnconstrained())
243 << "invalid virtual device for expr:" << std::endl
244 << PrettyPrint(expr);
245 std::vector<int64_t> storage_ids;
246 std::vector<VirtualDevice> virtual_devices;
247 std::vector<int64_t> storage_sizes_in_bytes;
248 for (const auto& ttype : FlattenTupleType(expr->checked_type())) {
249 storage_ids.push_back(next_available_sid_++);
250 virtual_devices.push_back(virtual_device);
251 storage_sizes_in_bytes.push_back(GetMemorySizeBytes(ttype));
252 }
253 storage_device_map_[expr] = StorageInfo(std::move(storage_ids), std::move(virtual_devices),
254 std::move(storage_sizes_in_bytes));
255 }
256
257 /*! \brief mapping of expression -> storageInfo */
258 StorageMap storage_device_map_;
259 /*! \brief current id of the temporary allocated */
260 int next_available_sid_{0};
261 /*! \brief the set of intermediate tensors that are return variables */
262 std::vector<int> return_ids_;
263 /*! \brief the data types of the return values */
264 std::vector<TensorType> return_ttypes_;
265};

267/*! \brief Code generator for AOT executor */
268class AOTExecutorCodegen : public MixedModeVisitor {
269 protected:
270 /*! \brief Describes the type of kernel call emitted. */
271 enum CallType {
272 /*!
273 * \brief Emit PackedFunc calls bound just-in-time using TVMBackend* functions.
274 *
275 * When this type is selected, assumes all operators must be called via TVMFuncCall. Given the
276 * implementation of TVMFuncCall in the C++ runtime, this in practice implies that those
277 * functions are of type TVMBackendPackedCFunc.
278 *
279 * The following code is emitted at call sites to call a function named `func`:
280 * void* func_ptr = TVMBackendGetFuncFromEnv("func");
281 * TVMFuncCall(func_ptr, values, tcodes, num_args, ret_values, ret_tcodes)
282 *
283 * The arguments given to the tir::Call node are encoded into `values`, `tcodes`, and `num_args`
284 * by LowerTVMBuiltin TIR transform.
285 *
286 * If `resource_handle` is passed to `func`, it is determined by TVMFuncCall (often,
287 * `resource_handle` is registered with the C++ runtime to provide a `this` equivalent when
288 * `func` is implemented in C).
289 *
290 * Compatible with both C++ and C runtimes, implemented with the C runtime only.
291 */
292 kPacked, // Emit tir.call_packed and wrap all arguments in DLTensor.
293
294 /*!
295 * \brief Directly call a TVMBackendPackedCFunc named according to the tir::Call.
296 *
297 * When this type is selected, assumes all operators are implemented in functions of type
298 * `TVMBackendPackedCFunc` and should be called directly. That is, presumes at the time of
299 * downstream compilation that there is a symbol named after the 0th arg to tir::Call of
300 * type `TVMBackendPackedCFunc`. This situation should occur when target_host == target.
301 *
302 * The following code is emitted at call sites to call a function named `func`:
303 * func(values, tcodes, num_args, ret_values, ret_tcodes, resource_handle)
304 *
305 * The arguments given to the tir::Call node are encoded into `values`, `tcodes`, and `num_args`
306 * by LowerTVMBuiltin TIR transform.
307 *
308 * `resource_handle` is encoded as the final argument to the tir::Call node. In practice, it is
309 * always the device context parameter when not null. At present, the implementation does not
310 * support forwarding device context parameters to CPacked.
311 *
312 * Compatible with the C runtime and C++ runtime (so long as target_host == target). Implemented
313 * in the same scenarios.
314 */
315 kCPacked, // Emit tir.call_cpacked and wrap all arguments in DLTensor.
316
317 /*! \brief Directly call a function accepting the `data` arrays as args.
318 *
319 * When this type is selected, assumes all operaotrs are implemented in C functions whose
320 * arguments are 1-to-1 with those in the tir::Call. DLTensor arguments are encoded as just the
321 * `data` parameters (i.e. no DLTensor object is passed along).
322 *
323 * The following code is emitted at call sites to a function named `func`:
324 * func(void* arg0, void* arg1, ..., void* argN) // no resource_handle
325 * -or-
326 * func(void* arg0, void* arg1, ..., void* argN, void* resource_handle) // with resource_handle
327 *
328 * `resource_handle` is encoded as the final argument to the tir::Call node. In practice, it is
329 * always the device context parameter when not null.
330 *
331 * Compatible with the C runtime and C++ runtime (so long as target_host == target). Implemented
332 * with the C runtime only.
333 */
334 kUnpacked, // Emit tir.call_extern passing only the `data` part of DLTensors.
335 };

337 /*!
338 * \brief Return a vector of variables that represents the sids for the given Relay Expr
339 */
340 std::vector<tir::Var> PackSid(Expr expr) {
341 std::vector<tir::Var> buffer_vars;
342
343 ICHECK(storage_device_map_.find(expr) != storage_device_map_.end())
344 << "Storage map did not contain constant expr " << PrettyPrint(expr);
345 StorageInfo& sinfo = storage_device_map_[expr];
346
347 // Note that an expression can have multiple sids associated with it
348 // e.g., returning multiple values from a function
349 for (auto sid : sinfo->storage_ids) {
350 // Determine if an sid is an output buffer
351 auto output_iter = std::find(return_sid_.begin(), return_sid_.end(), sid);
352 if (output_iter != return_sid_.end()) {
353 int output_index = std::distance(return_sid_.begin(), output_iter);
354 buffer_vars.push_back(GetBufferVarForIO(input_vars_.size() + output_index));
355 continue;
356 }
357
358 auto sid_value = sids_table_[sid];
359 buffer_vars.push_back(sid_value);
360 }
361 return buffer_vars;
362 }
363
364 /*!
365 * brief Given an expression return the variable(s) associated with that expression
366 */
367 std::vector<te::Var> FindExpr(Expr arg) {
368 auto input_iter = std::find(input_vars_.begin(), input_vars_.end(), arg);
369 if (input_iter != input_vars_.end()) {
370 // Input variable
371 int main_index = std::distance(input_vars_.begin(), input_iter);
372 return {GetBufferVarForIO(main_index)};
373 } else {
374 // Storage identifier (i.e., intermediate memory)
375 return PackSid(arg);
376 }
377 }

394 void PushArgs(const Expr& expr, const std::vector<tir::Var>& sids, Array<PrimExpr>* args) {
395 const TupleNode* t = expr.as<TupleNode>();
396 if (t != nullptr) {

414 /*!
415 * brief Create a function call
416 * \param call_lowered_props The lowered function and the arguments to call it with
417 * \param result_expr The call we got func and args from (so as to recover the storage
418 * ids to hold the result).
419 */
420 void CreateFuncCall(CallLoweredProps call_lowered_props, const Expr& result_expr) {
421 std::string func_name = call_lowered_props.lowered_func->name_hint;
422 tvm::Array<PrimExpr> args{tvm::tir::StringImm(func_name)};
423 std::vector<tir::Stmt> create_func_call_stmts;
424
425 // Pack the inputs
426 for (const Expr& arg : call_lowered_props.arguments) {
427 if (params_by_expr_.find(arg) != params_by_expr_.end()) {
428 auto param_handle = tvm::tir::Call(DataType::Handle(), tvm::tir::builtin::lookup_param(),
429 {tir::StringImm(params_by_expr_[arg])});
430 // NOTE: this cast looks like a no-op, but is required for compilation downstream.
431 // Because DataType::Handle has default bits=64, but CodeGenC does not observe this field,
432 // adding this cast forces the codegen to insert the cast. In this case, a cast is required
433 // because param_handle is actually code-generated as `const void*`, and the `const` piece
434 // needs to be removed.
435 args.push_back(tvm::tir::Cast(DataType::Handle(32, 1), param_handle));
436 } else {
437 auto sids = FindExpr(arg);
438 PushArgs(arg, sids, &args);
439 }
440 }
441
442 // Pack the return(s) value. A call node can produce multiple outputs
443 auto result_expr_sid = PackSid(result_expr);
444 PushArgs(result_expr, result_expr_sid, &args);
445
446 GlobalVar global_var = call_lowered_props.lowered_func;
447 bool has_c_device_api_context = device_contexts_.count(global_var) != 0;
448 tir::Var device_context;
449 tir::Stmt func_call;
450
451 switch (call_type_) {
452 case CallType::kUnpacked: {
453 // call_extern calling convention with optional context
454 if (has_c_device_api_context) {
455 device_context = device_contexts_.Get(global_var).value();
456 args.push_back(device_context);
457 }
458 func_call = tir::Evaluate(AddCheckReturn(
459 tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::call_extern(), args)));
460 break;
461 }
462 case CallType::kCPacked: {
463 if (has_c_device_api_context) {
464 device_context = device_contexts_.Get(global_var).value();
465 args.push_back(device_context);
466 } else {
467 // NOTE: LowerTVMBuiltin expects some device_context placeholder.
468 args.push_back(tir::make_zero(DataType::Handle()));
469 }
470 func_call = tir::Evaluate(
471 tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::tvm_call_cpacked(), args));
472 create_func_call_stmts.push_back(func_call);
473 break;
474 }
475 case CallType::kPacked: {
476 // call_packed does not accept a device context.
477 CHECK(!has_c_device_api_context) << "CallType::kPacked does not accept a device context";
478 func_call = tir::Evaluate(AddCheckReturn(
479 tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::tvm_call_packed(), args)));
480 create_func_call_stmts.push_back(func_call);
481 break;
482 }
483 default:
484 ICHECK(false) << "Unknown CallType: " << call_type_;
485 }
486
487 ICHECK(func_call.defined()) << "Must define func_call";
488
489 if (has_c_device_api_context) {
490 func_call = tir::SeqStmt(Array<tir::Stmt>({
491 GenerateDeviceHook(device_context, "Open"),
492 func_call,
493 GenerateDeviceHook(device_context, "Close"),
494 }));
495 }
496
497 tir::Stmt body = tir::SeqStmt({func_call});
498 stmts_.push_back(body);
499 }

501 /*!
502 * \brief Copy a variable to the output. This function is mainly used in edge cases
503 * when we want to return an input or a parameter.
504 * TODO(giuseros): we should try to avoid unnecessary copy to the output, e.g., in a
505 * copy-on-write fashion.
506 */
507 void CopyToOutput(PrimExpr out, PrimExpr in, bool pack_input, size_t size) {
508 // Define intermediate DLTensor to load/store the data
509 tir::Buffer tmp_read =
510 tir::decl_buffer({IntImm(DataType::UInt(64), size)}, DataType::UInt(8), "tmp_read");
511 tir::Buffer tmp_write =
512 tir::decl_buffer({IntImm(DataType::UInt(64), size)}, DataType::UInt(8), "tmp_write");
513 te::Var loop_idx("i", DataType::Int(32));
514 auto retval_i = tir::BufferLoad(tmp_read, {loop_idx});
515 // Copy the variable from the input to the output
516 tir::Stmt copy = tir::For(
517 loop_idx, 0, tir::make_const(DataType::Int(32, 1), size, Span()), tir::ForKind::kSerial,
518 tir::BufferStore(tmp_write, tir::Let(tmp_read->data, in, retval_i), {loop_idx}));
519 stmts_.push_back(tir::LetStmt(tmp_write->data, out, copy));
520 }

522 /*
523 * \brief Collects device context variables for passing to operators
524 */
525 void CollectDeviceVariables(const Map<GlobalVar, String>& device_contexts) {
526 Map<TargetKind, tir::Var> target_contexts;
527 TargetKindAttrMap<Bool> target_attr_map = tvm::TargetKind::GetAttrMap<Bool>("use_device_api");
528
529 for (const auto& it : device_contexts) {
530 const GlobalVar& global_var = it.first;
531 const std::string device_context_name = it.second;
532
533 Optional<TargetKind> target_kind = tvm::TargetKind::Get(device_context_name);
534 if (!target_kind || !target_attr_map.count(target_kind.value())) {
535 return;
536 }
537 if (target_attr_map[target_kind.value()]) {
538 std::string context_name = tvm::runtime::SanitizeName(device_context_name);
539 tir::Var device_context_var("device_context_" + context_name, DataType::Handle());
540
541 auto pair = target_contexts.find(target_kind.value());
542 if (pair != target_contexts.end()) {
543 device_context_var = (*pair).second;
544 } else {
545 main_signature_.push_back(device_context_var);
546 devices_.Set(context_name, device_context_var);
547 target_contexts.Set(target_kind.value(), device_context_var);
548 }
549
550 device_contexts_.Set(global_var, device_context_var);
551 }
552 }
553 }

555 /**
556 * \brief Generates a call to a given hook for all Devices found for C Device API
557 * \param Name of hook to generate statements for
558 * \return Statement with function calls for each device
559 */
560 tir::Stmt GenerateAllDeviceHook(const String& hook) {
561 std::vector<tir::Stmt> device_hooks;
562 for (const auto& it : devices_) {
563 const String& device_name = it.first;
564 const tir::Var& context = it.second;
565 Array<String> sections = {"Device", device_name, hook};
566 String device_hook_name = ToCFunctionStyle(PrefixName(sections));
567
568 tir::Evaluate device_hook(
569 AddCheckReturn(tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::call_extern(),
570 {tvm::tir::StringImm(device_hook_name), context})));
571 device_hooks.push_back(device_hook);
572 }
573 return tir::SeqStmt(device_hooks);
574 }

590 return tir::Evaluate(
591 AddCheckReturn(tir::Call(DataType::Int(32), tvm::tir::builtin::call_extern(),
592 {tvm::tir::StringImm(device_hook), context})));
593 }

607 void VisitExpr_(const CallNode* call_node) override {
608 OnDeviceProps on_device_props = GetOnDeviceProps(call_node);
609 if (on_device_props.body.defined()) {
610 VisitExpr(on_device_props.body);
611 return;
612 }
613
614 DeviceCopyProps device_copy_props = GetDeviceCopyProps(call_node);
615 CallLoweredProps call_lowered_props = GetCallLoweredProps(call_node);
616
617 if (device_copy_props.body.defined()) {
618 // TODO(mbs): device_copy cleaunp
619 // Suspect treating as no-op is better since already built into the StorageInfo?
620 LOG(FATAL) << "The AOT executor does not currently support device_copy";
621 return;
622 }
623
624 // At this point we should only see calls of the form call_lowered(@callee, (args...)),
625 // where @callee can be a PrimFunc we've compiled or an external function supplied via
626 // some other mechanism.
627 ICHECK(call_lowered_props.lowered_func.defined())
628 << "AOT does not support calling Relay functions. Attempting to call:" << std::endl
629 << PrettyPrint(GetRef<Call>(call_node));
630 for (const auto& arg : call_lowered_props.arguments) {
631 // Evaluate the args
632 VisitExpr(arg);
633 }
634 CreateFuncCall(call_lowered_props, GetRef<Call>(call_node));
635 }

637 void VisitExpr_(const VarNode* op) override {
638 Expr expr = GetRef<Expr>(op);
639 StorageInfo& sinfo = storage_device_map_[expr];
640
641 // Let bound vars refer to a value, so these should not be considered "output" vars.
642 if (let_bound_vars_.find(GetRef<Var>(op)) != let_bound_vars_.end()) {
643 return;
644 }
645
646 // If the Var node is an output node we need to copy the content of the variable to the output
647 // It's safe to check the SID here because Var StorageToken are never reallocated
648 auto output_iter = std::find(return_sid_.begin(), return_sid_.end(), sinfo->storage_ids[0]);
649 if (output_iter != return_sid_.end()) {
650 int output_index = std::distance(return_sid_.begin(), output_iter);
651 if (params_by_expr_.find(expr) != params_by_expr_.end()) {
652 auto param_handle = tvm::tir::Call(DataType::Handle(), tvm::tir::builtin::lookup_param(),
653 {tir::StringImm(params_by_expr_[expr])});
654 CopyToOutput(GetBufferVarForIO(input_vars_.size() + output_index), param_handle,
655 /*pack_input*/ false, sinfo->storage_sizes_in_bytes[0]);
656 } else {
657 auto var_expr = FindExpr(expr);
658 CopyToOutput(GetBufferVarForIO(input_vars_.size() + output_index), var_expr[0],
659 /*pack_input*/ false, sinfo->storage_sizes_in_bytes[0]);
660 }
661 }
662 }
663
664 void VisitExpr_(const ConstantNode* op) override {
665 Expr expr = GetRef<Expr>(op);
666 ICHECK(storage_device_map_.find(expr) != storage_device_map_.end())
667 << "Storage map did not contain constant expr " << PrettyPrint(expr);
668 StorageInfo& sinfo = storage_device_map_[expr];
669 std::stringstream ss;
670 ss << "constant_" << constant_map_.size();
671
672 tir::Var constant(ss.str(), PointerType(PrimType(DataType(op->data->dtype))));
673 constant_map_[constant] = op;
674 auto sid = sinfo->storage_ids[0];
675 sids_table_[sid] = constant;
676
677 // If the Constant node is an output node we need to copy the content of the parameter to the
678 // output. A node can only produce a single output
679 auto output_iter = std::find(return_sid_.begin(), return_sid_.end(), sid);
680 if (output_iter != return_sid_.end()) {
681 int output_index = std::distance(return_sid_.begin(), output_iter);
682 auto param_handle = tvm::tir::Call(DataType::Handle(), tvm::tir::builtin::lookup_param(),
683 {tir::StringImm(ss.str())});
684 CopyToOutput(GetBufferVarForIO(input_vars_.size() + output_index), constant,
685 /* pack_input */ false, sinfo->storage_sizes_in_bytes[0]);
686 }
687 }

736 // Create the main PrimFunc to execute the graph. Please note that
737 // the packed function calls don't pack their arguments. The AOT
738 // runner function needs to be legalized by the LegalizePackedCalls pass.
739 tir::PrimFunc CreateMainFunc(String mod_name, unsigned int relay_params) {
740 tir::Stmt body = tir::SeqStmt(stmts_);
741 // Allocate the sids
742 std::unordered_map<int, bool> allocated;
743
744 for (auto kv : storage_device_map_) {
745 // Only allocate sids that are needed
746 const bool is_input =
747 (std::find(input_vars_.begin(), input_vars_.end(), kv.first) != input_vars_.end());
748 const bool is_param = (params_by_expr_.find(kv.first) != params_by_expr_.end());
749 if (is_input || is_param) {
750 continue;
751 }
752
753 for (unsigned int i = 0; i < kv.second->storage_ids.size(); i++) {
754 int size = kv.second->storage_sizes_in_bytes[i];
755 int sid = kv.second->storage_ids[i];
756
757 if (std::find(return_sid_.begin(), return_sid_.end(), sid) != return_sid_.end()) {
758 continue;
759 }
760
761 // Make sure it hasn't already been allocated, this can happen
762 // with let-bound var/value pairs.
763 if (allocated.find(sid) != allocated.end()) {
764 continue;
765 }
766
767 allocated[sid] = constant_map_.count(sids_table_[sid]);
768
769 // TODO(giuseros): we should allocate this once outside the PrimFunc
770 // so we don't pay the price of allocation for every inference
771 if (!allocated[sid]) {
772 PointerType ptype = Downcast<PointerType>(sids_table_[sid]->type_annotation);
773 DataType element_type = Downcast<PrimType>(ptype->element_type)->dtype;
774 body = tir::Allocate(sids_table_[sid], element_type, {size}, tir::const_true(), body);
775 }
776 allocated[sid] = true;
777 }
778 }
779
780 for (auto kv : constant_map_) {
781 auto buffer_var = kv.first;
782 auto dtype = DataType(kv.second->data->dtype);
783
784 int ndim = kv.second->data->ndim;
785 Array<PrimExpr> extents;
786
787 for (int i = 0; i < ndim; i++) {
788 int shape = kv.second->data->shape[i];
789 extents.push_back(tir::make_const(DataType::Int(32), shape, Span()));
790 }
791 body = tir::AllocateConst(buffer_var, dtype, extents, kv.second->data, body);
792 }
793
794 // Define the PrimFunc attributes
795 Map<String, ObjectRef> dict_attrs;
796 String run_func_name = runtime::get_name_mangled(mod_name, runtime::symbol::tvm_module_main);
797 dict_attrs.Set("global_symbol", run_func_name);
798 dict_attrs.Set("runner_function", Bool(true));
799 dict_attrs.Set(tvm::attr::kTarget, config_->host_target);
800
801 tir::Stmt device_activations = GenerateAllDeviceHook("Activate");
802 tir::Stmt device_deactivations = GenerateAllDeviceHook("Deactivate");
803 tir::Stmt final_body = tir::SeqStmt({device_activations, body, device_deactivations});
804
805 // Make the PrimFunc
806 return tir::PrimFunc(main_signature_, final_body, VoidType(), main_buffer_map_,
807 DictAttrs(dict_attrs));
808 }

874 /*!
875 * \brief Calculate workspace sizes for PrimFuncs in the IRModule
876 */
877 Map<String, FunctionInfo> CalculateWorkspaceSizes(
878 const IRModule& lowered_mod, const Map<String, FunctionInfo>& function_metadata) {
879 Integer workspace_byte_alignment = GetModuleWorkspaceByteAlignment(lowered_mod);
880 Map<String, FunctionInfo> updated_function_metadata;
881 for (const auto& kv : lowered_mod->functions) {
882 GlobalVar global_var = kv.first;
883 BaseFunc base_func = kv.second;
884 if (base_func->IsInstance<tir::PrimFuncNode>()) {
885 tir::PrimFunc pfunc = Downcast<tir::PrimFunc>(base_func);
886 Target tgt = pfunc->GetAttr<Target>(tvm::attr::kTarget).value();
887 const auto& ws = CalculateWorkspaceBytes(pfunc, workspace_byte_alignment);
888 if (function_metadata.count(global_var->name_hint)) {
889 updated_function_metadata.Set(global_var->name_hint,
890 function_metadata[global_var->name_hint]);
891 updated_function_metadata[global_var->name_hint]->workspace_sizes.Set(tgt, ws);
892 } else {
893 FunctionInfo finfo{{{tgt, ws}}, {}, {}, {{tgt, pfunc}}, {}};
894 updated_function_metadata.Set(global_var->name_hint, finfo);
895 }
896 }
897 }
898 return updated_function_metadata;
899 }

902 * \brief Run USMP to plan memory for lowered IRModule.
903 */
904 IRModule PlanMemoryWithUSMP(const IRModule& mod) {
905 VLOG(1) << "Planning memory with USMP for module:" << std::endl << PrettyPrint(mod);
906 Integer workspace_byte_alignment = GetModuleWorkspaceByteAlignment(mod);
907 IRModule lowered_mod = mod->ShallowCopy();
908 lowered_mod = tir::transform::UnifiedStaticMemoryPlanner()(lowered_mod);
909 function_metadata_ = CalculateWorkspaceSizes(lowered_mod, function_metadata_);
910 Optional<Array<tir::usmp::AllocatedPoolInfo>> allocated_pool_infos =
911 lowered_mod->GetAttr<Array<tir::usmp::AllocatedPoolInfo>>(tvm::attr::kPoolArgs);
912 backend::FunctionInfo main_func_info =
913 lowered_mod->GetAttr<backend::FunctionInfo>("main_func_info").value();
914 main_func_info->workspace_sizes.clear();
915 if (allocated_pool_infos) {
916 for (const tir::usmp::AllocatedPoolInfo& allocated_pool_info : allocated_pool_infos.value()) {
917 for (const auto& tgt : allocated_pool_info->pool_info->targets) {
918 VLOG(1) << "USMP requires target " << tgt->ToDebugString() << " to have pool size "
919 << allocated_pool_info->allocated_size->value;
920 size_t size = allocated_pool_info->allocated_size->value;
921 if (allocated_pool_info->pool_info->IsInstance<ConstantPoolInfoNode>()) {
922 size += main_func_info->constant_sizes.count(tgt)
923 ? main_func_info->constant_sizes[tgt]->value
924 : 0;
925 main_func_info->constant_sizes.Set(tgt, size);
926 } else if (allocated_pool_info->pool_info->IsInstance<WorkspacePoolInfoNode>()) {
927 size += main_func_info->workspace_sizes.count(tgt)
928 ? main_func_info->workspace_sizes[tgt]->value
929 : 0;
930 main_func_info->workspace_sizes.Set(tgt, size);
931 } else {
932 LOG(FATAL) << "Unknown pool type: " << allocated_pool_info->pool_info->GetTypeKey();
933 }
934 }
935 }
936 }
937 function_metadata_.Set(runtime::symbol::tvm_module_main, main_func_info);
938 return lowered_mod;
939 }

1041 LoweredOutput Codegen(IRModule mod, relay::Function func, String mod_name) {
1042 VLOG_CONTEXT << "AOT";
1043
1044 Runtime runtime_config = mod->GetAttr<Runtime>(tvm::attr::kRuntime).value();
1045 Integer workspace_byte_alignment = GetModuleWorkspaceByteAlignment(mod);
1046
1047 Executor executor_config = mod->GetAttr<Executor>(tvm::attr::kExecutor).value();
1048 std::string interface_api =
1049 executor_config->GetAttr<String>("interface-api").value_or("packed");
1050 bool unpacked_api = executor_config->GetAttr<Bool>("unpacked-api").value_or(Bool(false));
1051
1052 // Validate choice of unpacked_api and use_call_cpacked_
1053 if (runtime_config->name == kTvmRuntimeCrt) {
1054 if (unpacked_api == true) {
1055 call_type_ = CallType::kUnpacked;
1056 } else if (unpacked_api == false && interface_api == "packed") {
1057 call_type_ = CallType::kCPacked;
1058 } else {
1059 CHECK(interface_api == "packed" || unpacked_api == true)
1060 << "Either need interface_api == \"packed\" (got: " << interface_api
1061 << ") or unpacked-api == true (got: " << unpacked_api << ") when targeting c runtime";
1062 ICHECK(false) << "Unhandled executor option config: interface-api=" << interface_api
1063 << ", unpacked-api=" << unpacked_api;
1064 }
1065 } else if (runtime_config->name == kTvmRuntimeCpp) {
1066 if (unpacked_api == false && interface_api == "packed") {
1067 call_type_ = CallType::kCPacked;
1068 } else {
1069 CHECK(static_cast<bool>(unpacked_api) == false && interface_api == "packed")
1070 << "Need unpacked-api == false (got: " << unpacked_api
1071 << ") and interface-api == \"packed\" (got: " << interface_api
1072 << ") when targeting c++ runtime";
1073 ICHECK(false) << "Unhandled executor option config: interface-api=" << interface_api
1074 << ", unpacked-api=" << unpacked_api;
1075 }
1076 } else {
1077 ICHECK(false) << "runtime_config (" << runtime_config->name
1078 << ") is not one of the expected values";
1079 }
1080
1081 mod = transform::ToANormalForm()(mod);
1082 mod = transform::InferType()(mod);
1083 mod = transform::AnnotateUsedMemory()(mod);
1084
1085 IRModule lowered_mod =
1086 tec::LowerTE(mod_name, config_, [this, workspace_byte_alignment](BaseFunc func) {
1087 // We need to maintain the constant map for external
1088 // functions so we pass this processing function which
1089 // allows us to process each function as we lower it.
1090 if (func->GetAttr<String>(attr::kCompiler).defined()) {
1091 UpdateConstants(func, &params_);
1092 }
1093
1094 // TODO(@areusch, @jroesch): We should refactor this to
1095 // execute as a further pass, instead writing data to the
1096 // lowering process directly.
1097 tec::UpdateFunctionMetadata(func, this->function_metadata_, workspace_byte_alignment);
1098 })(mod);
1099
1100 transform::PassContext pass_ctx = transform::PassContext::Current();
1101 bool enable_remove_reshapes =
1102 pass_ctx->GetConfig<Bool>("relay.remove_standalone_reshapes.enable", Bool(true)).value();
1103 if (enable_remove_reshapes) {
1104 lowered_mod = transform::RemoveStandaloneReshapes()(lowered_mod);
1105 }
1106 auto lowered_main = lowered_mod->Lookup("main");
1107 auto lowered_main_func = GetRef<Function>(lowered_main.as<FunctionNode>());
1108
1109 // Post-lowering storage map for writing main func
1110 AOTOnDemandAllocator final_aot_allocator;
1111 final_aot_allocator.Run(lowered_main_func);
1112 storage_device_map_ = final_aot_allocator.GetStorageMap();
1113
1114 // TODO(@electriclilies, @jroesch, @Mousius): remove UpdateMainWorkspaceSize
1115 StaticMemoryPlan memory_plan(storage_device_map_);
1116 backend::FunctionInfo func_info =
1117 tec::UpdateMainWorkspaceSize(lowered_mod, config_, memory_plan->expr_to_storage_info);
1118 lowered_mod = WithAttr(lowered_mod, "main_func_info", func_info);
1119
1120 for (auto input : lowered_main_func->params) {
1121 input_vars_.push_back(input);
1122 std::string input_name = SanitizeName(input->name_hint());
1123 // We dont want the compiler changing input names in the
1124 // event of a sanitization collision. Therefore, enforcing
1125 // the var created to use the input_name strictly.
1126 CreateIOVar(input, input_name, /*use_unique_name = */ false);
1127 }
1128
1129 // Define the storage allocator ids
1130 for (auto kv : storage_device_map_) {
1131 for (auto sid : kv.second->storage_ids) {
1132 // The buffer_var is created with storage_scope to be global.workspace to be serviced by
1133 // TVMBackendAllocWorkspace(TVMBAW) calls, explicitly. The reasoning being the executor
1134 // allocates should be serviced by TVMBAWs as the data could be accessed by many devices and
1135 // should not be lowered to the stack. For more details please refer to the discussion here:
1136 // https://github.com/apache/tvm/issues/9022
1137 te::Var buffer_var(MakeString("sid_", sid),
1138 PointerType(PrimType(DataType::Int(8)), "global.workspace"));
1139 sids_table_[sid] = buffer_var;
1140 }
1141 }
1142
1143 // Retrieve the return sids
1144 return_sid_ = final_aot_allocator.GetReturnIds();
1145 // Insert outputs to main func signature
1146 // If output tensor names were provided use them
1147 if (auto opt = func->GetAttr<Array<String>>("output_tensor_names")) {
1148 Array<String> output_tensor_names = opt.value();
1149 Expr output_expr = lowered_main_func->body;
1150 if (output_expr->checked_type()->IsInstance<TupleTypeNode>()) {
1151 TupleType output_tuple_type = Downcast<TupleType>(output_expr->checked_type());
1152 for (unsigned i = 0; i < output_tuple_type->fields.size(); i++) {
1153 // AoT Executor Codegen does not create these names,
1154 // thus should be used as they are provided.
1155 CreateIOVar(output_tuple_type->fields[i], output_tensor_names[i],
1156 /*use_unique_name = */ false);
1157 }
1158 } else {
1159 // AoT Executor Codegen does not create these names,
1160 // thus should be used as they are provided.
1161 CreateIOVar(lowered_main_func->body, output_tensor_names[0], /*use_unique_name = */ false);
1162 }
1163 } else {
1164 // If output tensor names are not provided we will generate output(x)
1165 // where x is a counter to create unique names.
1166 CreateIOVar(lowered_main_func->body, "output");
1167 }
1168
1169 CollectDeviceVariables(lowered_mod->GetAttr<Map<GlobalVar, String>>("device_contexts").value());
1170 VisitExpr(lowered_main_func->body);
1171
1172 // Create the runner function. Please note that the function is not legal yet
1173 // because the packed calls arguments are not wrapped in TVMValues. To make this happen we need
1174 // to run the LegalizePackedCalls pass.
1175 LoweredOutput ret;
1176
1177 // Collect any constants extracted by external codegen.
1178 ret.params = std::unordered_map<std::string, tvm::runtime::NDArray>();
1179 Map<String, runtime::NDArray> const_name_to_constant =
1180 lowered_mod->GetAttr<Map<String, runtime::NDArray>>(tvm::attr::kConstNameToConstant)
1181 .value_or({});
1182 for (const auto& kv : const_name_to_constant) {
1183 ICHECK(ret.params.emplace(kv.first, kv.second).second);
1184 }
1185
1186 // Collect any constants extracted during lowering.
1187 for (const auto& kv : params_) {
1188 ICHECK(ret.params.emplace(kv.first, kv.second).second);
1189 }
1190
1191 // AoT Executor codegen works completely on TIR beyond this point, hence removing relay main
1192 // function and replacing it with its TIR version. We should try to make this a Pass.
1193 lowered_mod->Remove(lowered_mod->GetGlobalVar("main"));
1194 auto tir_main_func = CreateMainFunc(mod_name, lowered_main_func->params.size());
1195 // Extract additional information around main TIR PrimFunc arguments
1196 Array<String> devices = ListDevices();
1197 const auto main_func_params_end_iterator =
1198 tir_main_func->params.begin() + tir_main_func->params.size();
1199 const auto outputs_begin_iterator =
1200 main_func_params_end_iterator - return_sid_.size() - devices.size();
1201 Array<tir::Var> inputs = Array<tir::Var>(tir_main_func->params.begin(), outputs_begin_iterator);
1202 Array<TensorType> input_tensor_types;
1203 for (auto i : inputs) {
1204 input_tensor_types.push_back(io_tensor_types_[i]);
1205 }
1206 Array<tir::Var> outputs =
1207 Array<tir::Var>(outputs_begin_iterator, main_func_params_end_iterator - devices.size());
1208
1209 lowered_mod->Update(GlobalVar(::tvm::runtime::symbol::tvm_module_main), tir_main_func);
1210 // Parallel for loops are not supported in AoT codegen.
1211 lowered_mod = tir::transform::ConvertForLoopsToSerial()(lowered_mod);
1212
1213 bool enable_usmp = pass_ctx->GetConfig<Bool>(kUSMPEnableOption, Bool(false)).value();
1214 if (enable_usmp) {
1215 lowered_mod = PlanMemoryWithUSMP(lowered_mod);
1216 } else {
1217 lowered_mod = PlanMemoryWithStorageRewrite(lowered_mod);
1218 }
1219 ret.function_metadata = std::move(function_metadata_);
1220
1221 // Legalize AOT if needed. This means that all the packed calls
1222 // need to be wrapped in TVMValues (unless unpacked_api is set)
1223 if (call_type_ == CallType::kCPacked || call_type_ == CallType::kPacked) {
1224 auto pack_calls = tir::transform::LegalizePackedCalls();
1225 lowered_mod = pack_calls(lowered_mod);
1226 }
1227
1228 // Collect any runtime modules generated by external codegen.
1229 ret.external_mods =
1230 lowered_mod->GetAttr<Array<tvm::runtime::Module>>(tvm::attr::kExternalMods).value_or({});
1231
1232 // This is the point where we separate the functions in the module by target
1233 VLOG(1) << "lowered module:" << std::endl << PrettyPrint(lowered_mod);
1234 ret.lowered_funcs = tec::GetPerTargetModules(lowered_mod);
1235 VLOG(1) << "per-target modules:";
1236 for (const auto& kv : ret.lowered_funcs) {
1237 VLOG(1) << "target:" << std::endl
1238 << kv.first->ToDebugString() << std::endl
1239 << "maps to:" << std::endl
1240 << PrettyPrint(kv.second);
1241 }
1242
1243 // Extract USMP metadata to pass onto metadata sources
1244 Map<tir::Var, tir::usmp::AllocatedPoolInfo> pool_var_info;
1245 std::vector<tir::Var> pool_vars;
1246 tir_main_func =
1247 Downcast<tir::PrimFunc>(lowered_mod->Lookup(::tvm::runtime::symbol::tvm_module_main));
1248 Optional<Array<tir::usmp::AllocatedPoolInfo>> allocated_pool_infos =
1249 tir_main_func->GetAttr<Array<tir::usmp::AllocatedPoolInfo>>(tvm::attr::kPoolArgs);
1250 if (allocated_pool_infos) {
1251 for (const tir::usmp::AllocatedPoolInfo& allocated_pool_info : allocated_pool_infos.value()) {
1252 int pool_var_index = allocated_pool_info->pool_var_idx.value()->value;
1253 pool_vars.push_back(tir_main_func->params[pool_var_index]);
1254 pool_var_info.Set(tir_main_func->params[pool_var_index], allocated_pool_info);
1255 }
1256 }
1257 Map<String, tir::usmp::PoolAllocation> io_pool_allocations =
1258 lowered_mod
1259 ->GetAttr<Map<String, tir::usmp::PoolAllocation>>(tvm::attr::kIOTensorPoolAllocations)
1260 .value_or({});
1261
1262 std::vector<String> output_var_names;
1263 if (auto opt = func->GetAttr<Array<String>>("output_tensor_names")) {
1264 Array<String> output_tensor_names = opt.value();
1265 for (size_t i = 0; i < output_tensor_names.size(); ++i) {
1266 output_var_names.push_back(output_tensor_names[i]);
1267 }
1268 }
1269
1270 // If output names have not been specified then generate default output names
1271 if (output_var_names.size() == 0) {
1272 if (return_sid_.size() == 1) {
1273 output_var_names.push_back(String("output"));
1274 } else {
1275 for (size_t i = 0; i < return_sid_.size(); ++i) {
1276 output_var_names.push_back(String("output" + std::to_string(i)));
1277 }
1278 }
1279 }
1280
1281 Array<TensorType> output_tensor_types{final_aot_allocator.GetReturnTtypes()};
1282
1283 ret.metadata = ExecutorCodegenMetadata(
1284 inputs, input_tensor_types, output_var_names, output_tensor_types, pool_vars, devices,
1285 runtime::kTvmExecutorAot, mod_name, interface_api, unpacked_api,
1286 GetModuleWorkspaceByteAlignment(mod), GetModuleConstantByteAlignment(mod), pool_var_info,
1287 io_pool_allocations);
1288 return ret;
1289 }

1292 * \brief Get list of devices found
1293 * \return List of devices
1294 */

1296 std::vector<String> device_names(devices_.size());
1297 std::transform(devices_.begin(), devices_.end(), device_names.begin(),
1298 [](const auto& it) -> String { return it.first; });
1299 return device_names;

1307 if (name == "init") {

1309 ICHECK_EQ(args.num_args, 2) << "The expected of arguments are: "
1310 << "runtime::Module mod and Array<Target> targets";

1317 IRModule mod = args[0];
1318 Function func = args[1];
1319 String mod_name = args[2];
1320 this->output_ = this->codegen_->Codegen(mod, func, mod_name);
1321 });
1322 } else if (name == "list_params_name") {
1323 return PackedFunc(
1324 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = list_params_name(); });
1325 } else if (name == "get_param_by_name") {
1326 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
1327 String key = args[0];
1328 *rv = get_param_by_name(key);
1329 });
1330 } else if (name == "get_irmodule") {
1331 return PackedFunc(
1332 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = get_irmodule(); });
1333 } else if (name == "get_external_modules") {
1334 return PackedFunc(
1335 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = get_external_modules(); });
1336 } else if (name == "get_function_metadata") {
1337 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
1338 *rv = this->output_.function_metadata;
1339 });
1340 } else if (name == "get_devices") {
1341 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
1342 *rv = this->codegen_->ListDevices();
1343 });
1344 } else if (name == "get_executor_codegen_metadata") {
1345 return PackedFunc(
1346 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = output_.metadata; });
