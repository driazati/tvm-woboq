39BaseAddress::BaseAddress(String name, Integer primfunc_param_idx, Integer region, Integer size,
40 Bool is_runtime_allocation) {
41 auto base_address_node = make_object<BaseAddressNode>();
42 base_address_node->name = name;
43 base_address_node->primfunc_param_idx = primfunc_param_idx;
44 base_address_node->region = region;
45 base_address_node->size = size;
46 base_address_node->is_runtime_allocation = is_runtime_allocation;
47 data_ = std::move(base_address_node);
48}
49
50TVM_REGISTER_NODE_TYPE(BaseAddressNode);
51TVM_REGISTER_GLOBAL("relay.ext.ethos-u.BaseAddress")
52 .set_body_typed([](String name, Integer primfunc_param_idx, Integer region, Integer size,
53 Bool is_runtime_allocation) {
54 if (is_runtime_allocation.defined()) {
55 return BaseAddress(name, primfunc_param_idx, region, size, is_runtime_allocation);
56 } else {
57 return BaseAddress(name, primfunc_param_idx, region, size);
58 }
59 });
60
61CompilationArtifact::CompilationArtifact(String function_name, String command_stream,
62 String encoded_constants,
63 Array<BaseAddress> base_addresses) {
64 auto compilation_artifact_node = make_object<CompilationArtifactNode>();
65 compilation_artifact_node->function_name = function_name;
66 compilation_artifact_node->command_stream = command_stream;
67 compilation_artifact_node->encoded_constants = encoded_constants;
68 compilation_artifact_node->base_addresses = base_addresses;
69 data_ = std::move(compilation_artifact_node);
70}
71
72TVM_REGISTER_NODE_TYPE(CompilationArtifactNode);
73TVM_REGISTER_GLOBAL("relay.ext.ethos-u.CompilationArtifact")
74 .set_body_typed([](String function_name, String command_stream, String encoded_constants,
75 Array<BaseAddress> base_addresses) {
76 return CompilationArtifact(function_name, command_stream, encoded_constants, base_addresses);
77 });
78
79TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
80 .set_dispatch<CompilationArtifactNode>([](const ObjectRef& ref, ReprPrinter* p) {
81 auto* node = static_cast<const CompilationArtifactNode*>(ref.get());
82 p->stream << "CompilationArtifactNode(\n"
83 << "function_name=" << node->function_name
84 << ",\n command_stream=" << node->command_stream
85 << ",\n encoded_constants=" << node->encoded_constants
86 << ",\n base_addresses=" << node->base_addresses << ")";
87 });
