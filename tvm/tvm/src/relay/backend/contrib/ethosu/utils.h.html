37/*!
38 * \brief Base addresses are input pointers to
39 * the driver that get accessed by the command stream
40 * using offsets to read/write data.
41 */
42struct BaseAddressNode : public Object {
43 /*! \brief The identifier, usually it the param name of the PrimFunc that gets lowered */
44 String name;
45 /*! \brief The index in the params array of the PrimFunc. This is needed to keep aligned
46 * between the PrimFunc arguments ordering and argument ordering of generated code */
47 Integer primfunc_param_idx;
48 /*! \brief The region used by the command stream. This needs to match with base address
49 * index passed into the driver */
50 Integer region;
51 /*! \brief The size of the buffer accessible by this base address */
52 Integer size;
53 /*! \brief This is a runtime allocation that needs to be done in the function */
54 Bool is_runtime_allocation{Bool(false)};
55
56 void VisitAttrs(tvm::AttrVisitor* v) {
57 v->Visit("name", &name);
58 v->Visit("primfunc_param_idx", &primfunc_param_idx);
59 v->Visit("region", &region);
60 v->Visit("size", &size);
61 v->Visit("is_runtime_allocation", &is_runtime_allocation);
62 }
63
64 bool SEqualReduce(const BaseAddressNode* other, SEqualReducer equal) const {
65 return equal(name, other->name) && equal(primfunc_param_idx, other->primfunc_param_idx) &&
66 equal(region, other->region) && equal(size, other->size) &&
67 equal(is_runtime_allocation, other->is_runtime_allocation);
68 }
69
70 void SHashReduce(SHashReducer hash_reduce) const {
71 hash_reduce(name);
72 hash_reduce(primfunc_param_idx);
73 hash_reduce(region);
74 hash_reduce(size);
75 hash_reduce(is_runtime_allocation);
76 }
77
78 static constexpr const char* _type_key = "relay.ext.ethos-u.BaseAddress";
79 TVM_DECLARE_FINAL_OBJECT_INFO(BaseAddressNode, Object);
80};
81
82class BaseAddress : public ObjectRef {
83 public:
84 TVM_DLL BaseAddress(String name, Integer primfunc_param_idx, Integer region, Integer size,
85 Bool is_runtime_allocation = Bool(false));
86 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(BaseAddress, ObjectRef, BaseAddressNode);
87};

89/*!
90 * \brief Captures all the binary artifactes required to create
91 * the C-source runtime module
92 */
93struct CompilationArtifactNode : public Object {
94 /*! \brief The function name for this artifact belongs to */
95 String function_name;
96 /*! \brief The binary command stream (CS) in hex format */
97 String command_stream;
98 /*! \brief The encoded biases and weights in hex format */
99 String encoded_constants;
100 /*! \brief The information regarding the base addresses */
101 Array<BaseAddress> base_addresses;
102
103 void VisitAttrs(tvm::AttrVisitor* v) {
104 v->Visit("function_name", &function_name);
105 v->Visit("command_stream", &command_stream);
106 v->Visit("encoded_constants", &encoded_constants);
107 v->Visit("base_addresses", &base_addresses);
108 }
109
110 bool SEqualReduce(const CompilationArtifactNode* other, SEqualReducer equal) const {
111 return equal(function_name, other->function_name) &&
112 equal(command_stream, other->command_stream) &&
113 equal(encoded_constants, other->encoded_constants) &&
114 equal(base_addresses, other->base_addresses);
115 }
116
117 void SHashReduce(SHashReducer hash_reduce) const {
118 hash_reduce(function_name);
119 hash_reduce(command_stream);
120 hash_reduce(encoded_constants);
121 hash_reduce(base_addresses);
122 }
123
124 static constexpr const char* _type_key = "relay.ext.ethos-u.CompilationArtifact";
125 TVM_DECLARE_FINAL_OBJECT_INFO(CompilationArtifactNode, Object);
126};
127
128class CompilationArtifact : public ObjectRef {
129 public:
130 TVM_DLL CompilationArtifact(String function_name, String command_stream, String encoded_constants,
131 Array<BaseAddress> base_addresses);
132 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(CompilationArtifact, ObjectRef, CompilationArtifactNode);
133};
