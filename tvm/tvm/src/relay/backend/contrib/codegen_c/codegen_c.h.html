71// The base class to generate the declaration functions in C.
72class CodegenCBase {
73 public:
74 virtual ~CodegenCBase() {}
75
76 protected:
77 /*! \brief Print indents using spaces. */
78 void PrintIndents() {
79 for (int i = 0; i < indent_; i++) {
80 code_stream_ << ' ';
81 }
82 }
83
84 /*!
85 * \brief Enter a new scope.
86 */
87 void EnterScope() { indent_ += 2; }
88
89 /*!
90 * \brief Exit a scope.
91 */
92 void ExitScope() {
93 ICHECK_GE(indent_, 2U) << "Wrong ident found.";
94 indent_ -= 2;
95 }
96
97 /*!
98 * \brief Creates a runtime function header
99 */
100 void PrintRuntimeFunctionHeader(std::string func_name) {
101 code_stream_ << "#ifdef __cplusplus\n";
102 code_stream_ << "extern \"C\" {\n";
103 code_stream_ << "#endif\n";
104 code_stream_ << "TVM_DLL int32_t ";
105 code_stream_ << func_name << "(";
106 code_stream_ << "TVMValue* args, ";
107 code_stream_ << "int* type_code, ";
108 code_stream_ << "int num_args, ";
109 code_stream_ << "TVMValue* out_value, ";
110 code_stream_ << "int* out_type_code) {\n";
111 }
112
113 /*!
114 * \brief Adds a line to convert TVMValue args to DLTensors
115 */
116 void PrintArgToData(int idx) {
117 PrintIndents();
118 code_stream_ << "DLTensor* arg" << idx << " = ";
119 code_stream_ << "(DLTensor*)(((TVMValue*)args)[" << idx << "].v_handle);\n";
120 }
121
122 /*!
123 * \brief Adds a line to convert TVMValue rets to DLTensors
124 */
125 void PrintRetToData(int idx) {
126 PrintIndents();
127 code_stream_ << "DLTensor* ret" << idx << " = ";
128 code_stream_ << "(DLTensor*)(((TVMValue*)args)[" << idx << "].v_handle);\n";
129 }
130
131 /*!
132 * \brief Gerenate C code for the external function.
133 *
134 * \param func_name The name of the external function.
135 * \param args arguments to the external function.
136 *
137 * \code
138 *
139 * Array<NDArray> foo_consts;
140 *
141 * // An example code for the generated C function.
142 * int foo_wrapper_(DLTensor* arg0,
143 * DLTensor* arg1,
144 * DLTensor* out) {
145 * foo_((float*)(arg0->data),
146 * (float*)(arg1->data),
147 * (float*)(out->data));
148 * return 0;
149 * }
150 *
151 * TVM_DLL_EXPORT_TYPED_FUNC(foo, foo_wrapper_);
152 *
153 * int foo_init_wrapper_(Array<NDArray> arr) {
154 * foo_consts = arr;
155 * return 0;
156 * }
157 *
158 * TVM_DLL_EXPORT_TYPED_FUNC(__init_foo, foo_init_wrapper_);
159 *
160 * \endcode
161 */
162 void GenerateBackendCFunc(const std::string& func_name, const Array<Var>& args,
163 const std::string& const_arr_name, const std::vector<Output>& outs,
164 bool pass_dl_tensor = false) {
165 // Print signature
166 code_stream_ << "\n";
167
168 code_stream_ << "int " << func_name << "_wrapper_(";
169 for (size_t i = 0; i < args.size(); i++) {
170 code_stream_ << "DLTensor* arg" << i << ",\n";
171 code_stream_ << "\t";
172 }
173 for (size_t i = 0; i < outs.size() - 1; i++) {
174 code_stream_ << "DLTensor* out" << i << ",\n";
175 code_stream_ << "\t";
176 }
177 code_stream_ << "DLTensor* out" << outs.size() - 1 << ") {\n";
178
179 EnterScope();
180
181 // Generate the internal call.
182 PrintIndents();
183 code_stream_ << func_name << "_(";
184 for (size_t i = 0; i < args.size(); i++) {
185 if (pass_dl_tensor) {
186 code_stream_ << "arg" << i << ",\n";
187 } else {
188 const auto& dtype_str = GetDtypeString(args[i]);
189 code_stream_ << "(" << dtype_str << "*)(arg" << i << "->data),\n";
190 }
191 PrintIndents();
192 }
193 for (size_t i = 0; i < outs.size() - 1; i++) {
194 if (pass_dl_tensor) {
195 code_stream_ << "out" << i << ",\n";
196 } else {
197 code_stream_ << "(" << outs[i].dtype << "*)(out" << i << "->data),\n";
198 }
199 PrintIndents();
200 }
201 if (pass_dl_tensor) {
202 code_stream_ << "out" << outs.size() - 1 << ");\n";
203 } else {
204 code_stream_ << "(" << outs.back().dtype << "*)(out" << outs.size() - 1 << "->data));\n";
205 }
206 PrintIndents();
207 code_stream_ << "return 0;\n";
208 ExitScope();
209 code_stream_ << "}\n\n";
210
211 // Create the external function
212 PrintRuntimeFunctionHeader(func_name);
213 EnterScope();
214 for (size_t i = 0; i < args.size(); i++) {
215 PrintArgToData(i);
216 }
217 for (size_t i = 0; i < outs.size(); i++) {
218 PrintRetToData(args.size() + i);
219 }
220 PrintIndents();
221 code_stream_ << func_name << "_wrapper_(";
222 for (size_t i = 0; i < args.size(); i++) {
223 code_stream_ << "arg" << i << ",";
224 }
225 for (size_t i = 0; i < outs.size() - 1; i++) {
226 code_stream_ << "ret" << args.size() + i << ",";
227 }
228 code_stream_ << "ret" << args.size() + outs.size() - 1 << ");\n";
229 PrintIndents();
230 code_stream_ << "return 0;\n";
231 ExitScope();
232 code_stream_ << "}\n";
233 code_stream_ << "#ifdef __cplusplus\n";
234 code_stream_ << "}\n";
235 code_stream_ << "#endif\n";
236
237 if (!const_arr_name.empty()) {
238 // If there are constants, insert the __init_ and the wrapper
239 // This segment would be generated in C++ because of the usage
240 // of tvm::runtime::Array. This is not ideal, but this to demonstrate
241 // constant copying process used packed imports in other external
242 // codegen. Moreover, in microTVM we dont expect this part to be generated.
243 code_stream_ << "#ifdef __cplusplus\n";
244 code_stream_ << "int " << func_name
245 << "_init_wrapper_(tvm::runtime::Array<tvm::runtime::NDArray> arr) {\n";
246 EnterScope();
247 PrintIndents();
248 code_stream_ << func_name << "_consts = arr;\n";
249 code_stream_ << "return 0;\n";
250 ExitScope();
251 code_stream_ << "}\n\n";
252 code_stream_ << "TVM_DLL_EXPORT_TYPED_FUNC(__init_" << func_name << ", " << func_name
253 << "_init_wrapper_);\n\n";
254 code_stream_ << "#endif\n";
255 }
256 }

258 /*!
259 * \brief Emit the code for external runtime.
260 *
261 * \param out The outputs.
262 *
263 * \return The code string.
264 */
265 virtual std::string JIT(const std::vector<Output>& out) = 0;
266
267 /*!
268 * \brief A common interface that is used by various external runtime to
269 * generate the wrapper to invoke external kernels.
270 *
271 * \param ext_func_id The unique id of an external function. It will be used
272 * during runtime to pick the correct external function.
273 * \param args The arguments used by the external function.
274 * \param buf_decl The declaration of temporary buffers that used to store the
275 * intermeidate of each external kernel.
276 * \param body The statements of the external function.
277 * \param out The name and id pairs for output.
278 *
279 * \return The emitted code string.
280 */
281 std::string JitImpl(const std::string& ext_func_id, const Array<Var>& args,
282 const std::vector<std::string>& buf_decl,
283 const std::vector<std::string>& body, const std::string& const_arr_name,
284 const std::vector<Output>& outs) {
285 // Create a declaration for global ndarrays that contain constant data.
286 if (!const_arr_name.empty()) {
287 code_stream_ << "#ifdef __cplusplus\n";
288 code_stream_ << const_arr_name << "\n\n";
289 code_stream_ << "#endif\n";
290 }
291 // Create the signature. For example, it could be:
292 // void dnnl_0_(float* in0, float* in1, float* out0, float* out1) {}
293 code_stream_ << "void " << ext_func_id << "_(";
294
295 for (const auto& arg : args) {
296 const auto& dtype_str = GetDtypeString(arg);
297 code_stream_ << dtype_str << "* " << arg->name_hint() << ", ";
298 }
299 for (size_t i = 0; i < outs.size() - 1; ++i) {
300 code_stream_ << outs[i].dtype << "* out" << i << ", ";
301 }
302 code_stream_ << outs.back().dtype << "* out" << outs.size() - 1 << ") {\n";
303 this->EnterScope();
304
305 // Function body
306 for (auto decl : buf_decl) {
307 this->PrintIndents();
308 code_stream_ << decl << "\n";
309 }
310 code_stream_ << "\n";
311 for (auto stmt : body) {
312 this->PrintIndents();
313 code_stream_ << stmt << "\n";
314 }
315
316 // Copy output
317 for (size_t i = 0; i < outs.size(); ++i) {
318 if (!outs[i].need_copy) {
319 continue;
320 }
321 this->PrintIndents();
322 code_stream_ << "memcpy(out" << i << ", " << outs[i].name << ", 4 * " << outs[i].size
323 << ");\n";
324 }
325
326 // Free buffers
327 for (size_t i = 0; i < buf_decl.size(); i++) {
328 this->PrintIndents();
329 code_stream_ << "free(buf_" << i << ");\n";
330 }
331
332 this->ExitScope();
333 code_stream_ << "}\n";
334
335 // Create the wrapper to call the ext_func
336 this->GenerateBackendCFunc(ext_func_id, args, const_arr_name, outs);
337 return code_stream_.str();
338 }

340 /*!
341 * \brief Returns dtype string
342 *
343 * \param var Var to get the dtype of
344 *
345 * \return The dtype string.
346 */
347 std::string GetDtypeString(const Var& var) {
348 auto ttype = var->checked_type().as<TensorTypeNode>();
349 ICHECK(ttype) << "Expect TensorTypeNode";
350 return GetDtypeString(ttype);
351 }
352
353 /*!
354 * \brief Returns dtype string
355 *
356 * \param ttype TensorTypeNode* to get the dtype of
357 *
358 * \return The dtype string.
359 */
360 std::string GetDtypeString(const TensorTypeNode* ttype) {
361 std::string dtype;
362 if (runtime::TypeMatch(ttype->dtype, kDLFloat, 32)) {
363 dtype = "float";
364 } else if (runtime::TypeMatch(ttype->dtype, kDLFloat, 16)) {
365 dtype = "half";
366 } else if (runtime::TypeMatch(ttype->dtype, kDLBfloat, 16)) {
367 dtype = "bfloat";
368 } else if (runtime::TypeMatch(ttype->dtype, kDLInt, 32)) {
369 dtype = "int";
370 } else if (runtime::TypeMatch(ttype->dtype, kDLInt, 64)) {
371 dtype = "int64_t";
372 } else {
373 LOG(FATAL) << "Unsupported dtype " << ttype->dtype;
374 }
375
376 return dtype;
377 }

379 /*!
380 * \brief Creates a checker to check if the NDArray pool is initialized
381 *
382 * \param symobl The Symbol of the current function
383 *
384 * \return The created checker
385 */
386 std::string CreateInitChecker(const std::string& symbol) const {
387 std::ostringstream oss;
388 oss << "ICHECK(!" << symbol
389 << "_consts.empty()) << \"C source module hasn't been initialized.\";\n";
390 return oss.str();
391 }
392
393 /*!
394 * \brief Generates the global ndarray pool declaration
395 *
396 * \param symobl The Symbol of the current function
397 *
398 * \return The created declaration
399 */
400 std::string CreateNDArrayPool(const std::string& symbol) const {
401 return "tvm::runtime::Array<tvm::runtime::NDArray> " + symbol + "_consts;";
402 }
403
404 /*!
405 * \brief Generates the reference to the data of a constant ndarray
406 *
407 * \param symobl The Symbol of the current function
408 * \param symobl const_id The index of the constant
409 *
410 * \return The created reference
411 */
412 std::string CreateDataReference(const std::string& symbol, size_t const_id) const {
413 return "(float*)(" + symbol + "_consts[" + std::to_string(const_id) + "]->data)";
414 }
415
416 /*!
417 * \brief Returns the variable name for a constant variable
418 *
419 * \param symobl The Symbol of the current function
420 * \param symobl const_id The index of the constant
421 *
422 * \return The created variable name
423 */
424 std::string CreateConstVar(const std::string& symbol, size_t const_id) const {
425 return symbol + "_const_" + std::to_string(const_id);
426 }
427
428 /*! \brief The external function source code stream. */
429 std::ostringstream code_stream_;
430
431 private:
432 /*! \brief Indent of the source code. */
433 int indent_{0};
434};
