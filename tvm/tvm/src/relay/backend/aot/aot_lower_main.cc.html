41/*!
42 * \brief Looks at the expressions in a given function and produces an Expr to
43 * StorageInfo map by assigning one or more StorageInfos to the expressions that
44 * require storage.
45 *
46 * This pass is leveraged by AOTMainLowerer to perform an initial naive allocation
47 * for tensors in the Relay main function. The resulting storage map is then lowered
48 * into TIR allocations by AOTMainLowerer where the allocation can be subsequently
49 * optimized by later passes (e.g. USMP).
50 */
51class ExprAllocator : public transform::DeviceAwareExprVisitor {
52 public:
53 ExprAllocator() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
54
55 // run the visitor on a global function.
56 void Run(const Function& func) { VisitExpr(func); }
57
58 std::vector<int> GetReturnSIDs() const { return return_sids_; }
59
60 StorageMap GetStorageMap() const { return expr_storage_map_; }
61
62 using ExprVisitor::VisitExpr_;
63
64 void DeviceAwareVisitExpr_(const CallNode* call_node) final {
65 Array<Expr> args;
66
67 CallLoweredProps call_lowered_props = GetCallLoweredProps(call_node);
68 if (call_lowered_props.lowered_func.defined()) {
69 args = call_lowered_props.arguments;
70 } else { // Relay functions that have not been lowered and lowered extern functions
71 args = call_node->args;
72 if (call_node->op.as<GlobalVarNode>()) { // Lowered extern function
73 ICHECK(!(call_node->attrs.defined())) << "Extern functions should have null attributes.";
74 } else { // Relay function which has not been lowered yet
75 ICHECK(call_node->op.as<FunctionNode>())
76 << "Expected the call to be to a lowered primfunc, a lowered extern function or a "
77 "unlowered Relay function.";
78 }
79 }
80 CreateStorage(call_node);
81 for (const Expr& arg : args) {
82 VisitExpr(arg);
83 }
84 AssignReturnSID(GetRef<Expr>(call_node));
85 }
86
87 void DeviceAwareVisitExpr_(const FunctionNode* func_node) final {
88 if (function_nesting() > 1) {
89 // Do not recurse into sub functions.
90 return;
91 }
92 for (const auto& param : func_node->params) {
93 CreateStorage(param.get());
94 }
95 VisitExpr(func_node->body);
96 }
97
98 void PreVisitLetBinding_(const Var& var, const Expr& value) final {
99 VisitExpr(value);
100 StorageInfo si = GetStorage(value);
101 expr_storage_map_[var] = si;
102 }
103
104 void VisitExpr_(const ConstantNode* op) final {
105 CreateStorage(op);
106 AssignReturnSID(GetRef<Expr>(op));
107 }
108
109 void VisitExpr_(const VarNode* op) final { AssignReturnSID(GetRef<Expr>(op)); }
110
111 void VisitExpr_(const TupleNode* op) final {
112 std::vector<int64_t> storage_ids;
113 std::vector<VirtualDevice> virtual_devices;
114 std::vector<int64_t> storage_sizes_in_bytes;
115 Expr expr = GetRef<Expr>(op);
116 for (Expr field : op->fields) {
117 auto sid = GetStorage(field);
118 storage_ids.insert(storage_ids.end(), sid->storage_ids.begin(), sid->storage_ids.end());
119 virtual_devices.insert(virtual_devices.end(), sid->virtual_devices.begin(),
120 sid->virtual_devices.end());
121 storage_sizes_in_bytes.insert(storage_sizes_in_bytes.end(),
122 sid->storage_sizes_in_bytes.begin(),
123 sid->storage_sizes_in_bytes.end());
124 }
125 expr_storage_map_[expr] = StorageInfo(storage_ids, virtual_devices, storage_sizes_in_bytes);
126 AssignReturnSID(expr);
127 }
128
129 void VisitExpr_(const TupleGetItemNode* op) final {
130 Expr expr = GetRef<Expr>(op);
131 auto sids = GetStorage(op->tuple);
132 ICHECK_LT(static_cast<size_t>(op->index), sids->storage_ids.size());
133 expr_storage_map_[expr] =
134 StorageInfo({sids->storage_ids[op->index]}, {sids->virtual_devices[op->index]},
135 {sids->storage_sizes_in_bytes[op->index]});
136 AssignReturnSID(expr);
137 }
138
139 void VisitExpr_(const IfNode* op) final { LOG(FATAL) << "'If' is not supported."; }
140
141 private:
142 /*!
143 * \brief Assign the expression's storage IDs as the return storage IDs.
144 * \note This is called when visiting every expression on the understanding
145 * that the returned expression will be visited last.
146 */
147 void AssignReturnSID(const Expr& e) {
148 if (expr_storage_map_.find(e) != expr_storage_map_.end()) {
149 StorageInfo& sinfo = expr_storage_map_[e];
150 return_sids_.clear();
151 for (auto sid : sinfo->storage_ids) {
152 return_sids_.push_back(sid);
153 }
154 }
155 }
156
157 /*!
158 * \brief Get the necessary storage for the expression.
159 * \param expr The expression.
160 * \return The corresponding token.
161 */
162 StorageInfo GetStorage(const Expr& expr) {
163 // See through "on_device" calls.
164 Expr true_expr = IgnoreOnDevice(expr);
165 VisitExpr(true_expr);
166 auto it = expr_storage_map_.find(true_expr);
167 ICHECK(it != expr_storage_map_.end()) << "Could not find " << true_expr->GetTypeKey() << " "
168 << PrettyPrint(true_expr) << " in storage device map";
169 return it->second;
170 }
171
172 /*!
173 * \brief Create storage for the expression.
174 */
175 void CreateStorage(const ExprNode* op) {
176 Expr expr = GetRef<Expr>(op);
177 return CreateStorage(expr, GetVirtualDevice(expr));
178 }
179
180 /*!
181 * \brief Create storage to hold the result of evaluating \p expr in \p virtual_device.
182 */
183 void CreateStorage(const Expr& expr, const VirtualDevice& virtual_device) {
184 ICHECK(!virtual_device->IsFullyUnconstrained())
185 << "invalid virtual device for expr:" << std::endl
186 << PrettyPrint(expr);
187 std::vector<int64_t> storage_ids;
188 std::vector<VirtualDevice> virtual_devices;
189 std::vector<int64_t> storage_sizes_in_bytes;
190 for (const auto& ttype : FlattenTupleType(expr->checked_type())) {
191 storage_ids.push_back(next_available_sid_++);
192 virtual_devices.push_back(virtual_device);
193 storage_sizes_in_bytes.push_back(GetMemorySizeBytes(ttype->shape, ttype->dtype));
194 }
195 expr_storage_map_[expr] = StorageInfo(std::move(storage_ids), std::move(virtual_devices),
196 std::move(storage_sizes_in_bytes));
197 }
198
199 /*! \brief Map between Exprs and StorageInfos */
200 StorageMap expr_storage_map_;
201 /*! \brief The next available storage ID to be used */
202 int next_available_sid_{0};
203 /*! \brief The storage IDs that correspond to return values */
204 std::vector<int> return_sids_;
205};
206
207std::tuple<StorageMap, std::vector<int>> CreateStorage(const Function& func) {
208 ExprAllocator expr_allocator;
209 expr_allocator.Run(func);
210 return std::make_tuple(expr_allocator.GetStorageMap(), expr_allocator.GetReturnSIDs());
211}

213class AOTMainLowerer : public MixedModeVisitor {
214 public:
215 AOTMainLowerer(tvm::CompilationConfig config, CallType call_type)
216 : config_(config), call_type_(call_type) {}

218 IRModule Lower(IRModule mod, String mod_name) {
219 VLOG_CONTEXT << "AOT";
220 IRModule lowered_mod = GetRef<IRModule>(mod.CopyOnWrite());
221
222 auto lowered_main = lowered_mod->Lookup("main");
223 auto lowered_main_func = GetRef<Function>(lowered_main.as<FunctionNode>());
224
225 // Assign StorageInfo to all the Relay exprs and get the return SIDs
226 std::tie(expr_storage_map_, return_sid_) = CreateStorage(lowered_main_func);
227
228 for (auto input : lowered_main_func->params) {
229 input_vars_.push_back(input);
230 std::string input_name = SanitizeName(input->name_hint());
231 // We don't want the compiler changing input names in the
232 // event of a sanitization collision. Therefore, enforcing
233 // the var created to use the input_name strictly.
234 CreateIOVar(input, input_name, /*use_unique_name = */ false);
235 }
236
237 // Define the storage allocator ids
238 for (auto kv : expr_storage_map_) {
239 for (auto sid : kv.second->storage_ids) {
240 // The buffer_var is created with storage_scope to be global.workspace to be serviced by
241 // TVMBackendAllocWorkspace(TVMBAW) calls, explicitly. The reasoning being the executor
242 // allocates should be serviced by TVMBAWs as the data could be accessed by many devices and
243 // should not be lowered to the stack. For more details please refer to the discussion here:
244 // https://github.com/apache/tvm/issues/9022
245 tir::Var buffer_var(MakeString("sid_", sid),
246 PointerType(PrimType(DataType::Int(8)), "global.workspace"));
247 sids_table_[sid] = buffer_var;
248 }
249 }
250
251 // Create output vars for the TIR main func
252 // If output tensor names were provided use them
253 if (auto opt = lowered_main->GetAttr<Array<String>>("output_tensor_names")) {
254 Array<String> output_tensor_names = opt.value();
255 Expr output_expr = lowered_main_func->body;
256 if (output_expr->checked_type()->IsInstance<TupleTypeNode>()) {
257 TupleType output_tuple_type = Downcast<TupleType>(output_expr->checked_type());
258 for (unsigned i = 0; i < output_tuple_type->fields.size(); i++) {
259 // AoT Executor Codegen does not create these names,
260 // thus should be used as they are provided.
261 CreateIOVar(output_tuple_type->fields[i], output_tensor_names[i],
262 /*use_unique_name = */ false);
263 }
264 } else {
265 // AoT Executor Codegen does not create these names,
266 // thus should be used as they are provided.
267 CreateIOVar(lowered_main_func->body, output_tensor_names[0], /*use_unique_name = */ false);
268 }
269 } else {
270 // If output tensor names are not provided we will generate output(x)
271 // where x is a counter to create unique names.
272 if (lowered_main_func->body->checked_type()->IsInstance<TupleTypeNode>()) {
273 CreateIOVar(lowered_main_func->body, "output");
274 } else {
275 CreateIOVar(lowered_main_func->body, "output", /*use_unique_name = */ false);
276 }
277 }
278
279 CollectDeviceVariables(lowered_mod->GetAttr<Map<GlobalVar, String>>("device_contexts")
280 .value_or(Map<GlobalVar, String>()));
281 VisitExpr(lowered_main_func->body);
282
283 // Remove the Relay main and replace it with the lowered TIR version
284 lowered_mod->Remove(lowered_mod->GetGlobalVar("main"));
285 auto tir_main_func = CreateMainFunc(mod_name);
286 lowered_mod->Update(GlobalVar(runtime::symbol::tvm_module_main), tir_main_func);
287 lowered_mod = tir::transform::RemoveNoOp()(lowered_mod);
288 return lowered_mod;
289 }

291 void VisitExpr_(const CallNode* call_node) override {
292 OnDeviceProps on_device_props = GetOnDeviceProps(call_node);
293 if (on_device_props.body.defined()) {
294 VisitExpr(on_device_props.body);
295 return;
296 }
297
298 DeviceCopyProps device_copy_props = GetDeviceCopyProps(call_node);
299 CallLoweredProps call_lowered_props = GetCallLoweredProps(call_node);
300
301 if (device_copy_props.body.defined()) {
302 // TODO(mbs): device_copy cleaunp
303 // Suspect treating as no-op is better since already built into the StorageInfo?
304 LOG(FATAL) << "The AOT executor does not currently support device_copy";
305 return;
306 }
307
308 // At this point we should only see calls of the form call_lowered(@callee, (args...)),
309 // where @callee can be a PrimFunc we've compiled or an external function supplied via
310 // some other mechanism.
311 ICHECK(call_lowered_props.lowered_func.defined())
312 << "AOT does not support calling Relay functions. Attempting to call:" << std::endl
313 << PrettyPrint(GetRef<Call>(call_node));
314 for (const auto& arg : call_lowered_props.arguments) {
315 // Evaluate the args
316 VisitExpr(arg);
317 }
318 CreateFuncCall(call_lowered_props, GetRef<Call>(call_node));
319 }
320
321 void VisitExpr_(const VarNode* op) override {
322 Expr expr = GetRef<Expr>(op);
323 StorageInfo& sinfo = expr_storage_map_[expr];
324
325 // Let bound vars refer to a value, so these should not be considered "output" vars.
326 if (let_bound_vars_.find(GetRef<Var>(op)) != let_bound_vars_.end()) {
327 return;
328 }
329
330 // If the Var node is an output node we need to copy the content of the variable to the output
331 // It's safe to check the SID here because Var StorageToken are never reallocated
332 auto output_iter = std::find(return_sid_.begin(), return_sid_.end(), sinfo->storage_ids[0]);
333 if (output_iter != return_sid_.end()) {
334 int output_index = std::distance(return_sid_.begin(), output_iter);
335 auto var_expr = FindExpr(expr);
336 CopyToOutput(GetBufferVarForIO(input_vars_.size() + output_index), var_expr[0],
337 /*pack_input*/ false, sinfo->storage_sizes_in_bytes[0]);
338 }
339 }
340
341 void VisitExpr_(const ConstantNode* op) override {
342 Expr expr = GetRef<Expr>(op);
343 ICHECK(expr_storage_map_.find(expr) != expr_storage_map_.end())
344 << "Storage map did not contain constant expr " << PrettyPrint(expr);
345 StorageInfo& sinfo = expr_storage_map_[expr];
346 std::stringstream ss;
347 ss << "constant_" << constant_map_.size();
348
349 tir::Var constant(ss.str(), PointerType(PrimType(DataType(op->data->dtype))));
350 constant_map_[constant] = op;
351 auto sid = sinfo->storage_ids[0];
352 sids_table_[sid] = constant;
353
354 // If the Constant node is an output node we need to copy the content of the parameter to the
355 // output. A node can only produce a single output
356 auto output_iter = std::find(return_sid_.begin(), return_sid_.end(), sid);
357 if (output_iter != return_sid_.end()) {
358 int output_index = std::distance(return_sid_.begin(), output_iter);
359 auto param_handle = tvm::tir::Call(DataType::Handle(), tvm::tir::builtin::lookup_param(),
360 {tir::StringImm(ss.str())});
361 CopyToOutput(GetBufferVarForIO(input_vars_.size() + output_index), constant,
362 /* pack_input */ false, sinfo->storage_sizes_in_bytes[0]);
363 }
364 }
365
366 void VisitExpr_(const TupleNode* op) override {
367 for (auto field : op->fields) {
368 VisitExpr(field);
369 }
370 }
371
372 void VisitExpr_(const LetNode* op) override {
373 auto pre_visit = [this](const LetNode* op) {
374 let_bound_vars_.insert(op->var);
375 this->VisitExpr(op->value);
376 };
377 auto post_visit = [this](const LetNode* op) {
378 this->VisitExpr(op->body);
379 this->visit_counter_[op] += 1;
380 };
381 ExpandANormalForm(op, pre_visit, post_visit);
382 }

384 void VisitExpr_(const TupleGetItemNode* op) override { VisitExpr(op->tuple); }
385 void VisitExpr_(const OpNode* op) override {
386 if (GetRef<Op>(op) != CallLoweredOp() && GetRef<Op>(op) != OnDeviceOp()) {
387 LOG(FATAL) << "All OpNodes except for call_lowered should have been expanded";
388 }
389 }
390 void VisitExpr_(const IfNode* op) override {
391 LOG(FATAL) << "All GlobalVarNodes should be removed before AOT executor's Codegen is called";
392 }
393 void VisitExpr_(const FunctionNode* op) override {
394 ICHECK(op->GetAttr<String>(attr::kCompiler).defined())
395 << "FunctionNode only supported by custom codegen";
396 }
397 void VisitExpr_(const RefCreateNode* op) override {
398 LOG(FATAL) << "AOT executor does not support references (found RefCreateNode)";
399 }
400 void VisitExpr_(const RefReadNode* op) override {
401 LOG(FATAL) << "AOT executor does not support references (found RefReadNode)";
402 }
403 void VisitExpr_(const RefWriteNode* op) override {
404 LOG(FATAL) << "AOT executor does not support references (found RefWriteNode)";
405 }
406 void VisitExpr_(const ConstructorNode* op) override {
407 LOG(FATAL) << "AOT executor does not support ADTs (found ConstructorNode)";

414 /*!
415 * \brief Create the main PrimFunc to execute the graph.
416 * \note The packed function calls don't pack their arguments. The AOT
417 * runner function needs to be legalized by the LegalizePackedCalls pass.
418 */
419 tir::PrimFunc CreateMainFunc(String mod_name) {
420 tir::Stmt body = tir::SeqStmt(stmts_);
421 // Allocate the sids
422 std::unordered_map<int, bool> allocated;
423 std::vector<std::pair<int64_t, int64_t>> sids_to_allocate;
424
425 for (auto kv : expr_storage_map_) {
426 // Only allocate sids that are needed
427 const bool is_input =
428 (std::find(input_vars_.begin(), input_vars_.end(), kv.first) != input_vars_.end());
429 if (is_input) {
430 continue;
431 }
432
433 for (unsigned int i = 0; i < kv.second->storage_ids.size(); i++) {
434 sids_to_allocate.push_back(
435 std::make_pair(kv.second->storage_ids[i], kv.second->storage_sizes_in_bytes[i]));
436 }
437 }
438
439 // Sort the SID allocation to make output deterministic
440 std::sort(sids_to_allocate.begin(), sids_to_allocate.end());
441
442 for (auto p : sids_to_allocate) {
443 int sid = p.first;
444 int size = p.second;
445
446 if (std::find(return_sid_.begin(), return_sid_.end(), sid) != return_sid_.end()) {
447 continue;
448 }
449
450 // Make sure it hasn't already been allocated, this can happen
451 // with let-bound var/value pairs.
452 if (allocated.find(sid) != allocated.end()) {
453 continue;
454 }
455
456 allocated[sid] = constant_map_.count(sids_table_[sid]);
457
458 // TODO(giuseros): we should allocate this once outside the PrimFunc
459 // so we don't pay the price of allocation for every inference
460 if (!allocated[sid]) {
461 PointerType ptype = Downcast<PointerType>(sids_table_[sid]->type_annotation);
462 DataType element_type = Downcast<PrimType>(ptype->element_type)->dtype;
463 body = tir::Allocate(sids_table_[sid], element_type, {size}, tir::const_true(), body);
464 }
465 allocated[sid] = true;
466 }
467
468 for (auto kv : constant_map_) {
469 auto buffer_var = kv.first;
470 auto dtype = DataType(kv.second->data->dtype);
471
472 int ndim = kv.second->data->ndim;
473 Array<PrimExpr> extents;
474
475 for (int i = 0; i < ndim; i++) {
476 int shape = kv.second->data->shape[i];
477 extents.push_back(tir::make_const(DataType::Int(32), shape, Span()));
478 }
479 body = tir::AllocateConst(buffer_var, dtype, extents, kv.second->data, body);
480 }
481
482 // Define the PrimFunc attributes
483 Map<String, ObjectRef> dict_attrs;
484 String run_func_name = runtime::get_name_mangled(mod_name, runtime::symbol::tvm_module_main);
485 dict_attrs.Set("global_symbol", run_func_name);
486 dict_attrs.Set("runner_function", Bool(true));
487 dict_attrs.Set(tvm::attr::kTarget, config_->host_target);
488 Array<tir::Var> input_vars =
489 Array<tir::Var>(main_signature_.begin(), main_signature_.begin() + input_vars_.size());
490 dict_attrs.Set("input_vars", input_vars);
491 Array<tir::Var> output_vars =
492 Array<tir::Var>(main_signature_.begin() + input_vars_.size(),
493 main_signature_.begin() + input_vars_.size() + return_sid_.size());
494 dict_attrs.Set("output_vars", output_vars);
495
496 tir::Stmt device_activations = GenerateAllDeviceHook("Activate");
497 tir::Stmt device_deactivations = GenerateAllDeviceHook("Deactivate");
498 tir::Stmt final_body = tir::SeqStmt({device_activations, body, device_deactivations});
499
500 // Make the PrimFunc
501 return tir::PrimFunc(main_signature_, final_body, VoidType(), main_buffer_map_, {},
502 DictAttrs(dict_attrs));
503 }

505 /*!
506 * \brief Collects device context variables for passing to operators
507 */
508 void CollectDeviceVariables(const Map<GlobalVar, String>& device_contexts) {
509 Map<TargetKind, tir::Var> target_contexts;
510 TargetKindAttrMap<Bool> target_attr_map = tvm::TargetKind::GetAttrMap<Bool>("use_device_api");
511
512 for (const auto& it : device_contexts) {
513 const GlobalVar& global_var = it.first;
514 const std::string device_context_name = it.second;
515
516 Optional<TargetKind> target_kind = tvm::TargetKind::Get(device_context_name);
517 if (!target_kind || !target_attr_map.count(target_kind.value())) {
518 return;
519 }
520 if (target_attr_map[target_kind.value()]) {
521 std::string context_name = SanitizeName(device_context_name);
522 tir::Var device_context_var("device_context_" + context_name, DataType::Handle());
523
524 auto pair = target_contexts.find(target_kind.value());
525 if (pair != target_contexts.end()) {
526 device_context_var = (*pair).second;
527 } else {
528 main_signature_.push_back(device_context_var);
529 devices_.Set(context_name, device_context_var);
530 target_contexts.Set(target_kind.value(), device_context_var);
531 }
532
533 device_contexts_.Set(global_var, device_context_var);
534 }
535 }
536 }

538 /*!
539 * \brief Return a vector of variables that represents the sids for the given Relay Expr
540 */
541 std::vector<tir::Var> PackSid(Expr expr) {
542 std::vector<tir::Var> buffer_vars;
543
544 ICHECK(expr_storage_map_.find(expr) != expr_storage_map_.end())
545 << "Storage map did not contain constant expr " << PrettyPrint(expr);
546 StorageInfo& sinfo = expr_storage_map_[expr];
547
548 // Note that an expression can have multiple sids associated with it
549 // e.g., returning multiple values from a function
550 for (auto sid : sinfo->storage_ids) {
551 // Determine if an sid is an output buffer
552 auto output_iter = std::find(return_sid_.begin(), return_sid_.end(), sid);
553 if (output_iter != return_sid_.end()) {
554 int output_index = std::distance(return_sid_.begin(), output_iter);
555 buffer_vars.push_back(GetBufferVarForIO(input_vars_.size() + output_index));
556 continue;
557 }
558
559 auto sid_value = sids_table_[sid];
560 buffer_vars.push_back(sid_value);
561 }
562 return buffer_vars;
563 }

565 /*!
566 * \brief Given an expression return the variable(s) associated with that expression
567 */
568 std::vector<tir::Var> FindExpr(Expr arg) {
569 auto input_iter = std::find(input_vars_.begin(), input_vars_.end(), arg);
570 if (input_iter != input_vars_.end()) {
571 // Input variable
572 int main_index = std::distance(input_vars_.begin(), input_iter);
573 return {GetBufferVarForIO(main_index)};
574 } else {
575 // Storage identifier (i.e., intermediate memory)
576 return PackSid(arg);
577 }
578 }
579
580 void PushArgs(const Expr& expr, const std::vector<tir::Var>& sids, Array<PrimExpr>* args) {
581 const TupleNode* t = expr.as<TupleNode>();
582 if (t != nullptr) {
583 CHECK_EQ(sids.size(), t->fields.size()) << "Relay tuple does not map 1:1 into TIR; AOT can't "
584 "handle this type of Relay Expr in a CallNode.";
585 }
586
587 args->insert(args->end(), sids.begin(), sids.end());
588 }

600 /*!
601 * \brief Create a function call
602 * \param call_lowered_props The lowered function and the arguments to call it with
603 * \param result_expr The call we got func and args from (so as to recover the storage
604 * ids to hold the result).
605 */
606 void CreateFuncCall(CallLoweredProps call_lowered_props, const Expr& result_expr) {
607 std::string func_name = call_lowered_props.lowered_func->name_hint;
608 tvm::Array<PrimExpr> args{tvm::tir::StringImm(func_name)};
609 std::vector<tir::Stmt> create_func_call_stmts;
610
611 // Pack the inputs
612 for (const Expr& arg : call_lowered_props.arguments) {
613 auto sids = FindExpr(arg);
614 PushArgs(arg, sids, &args);
615 }
616
617 // Pack the return(s) value. A call node can produce multiple outputs
618 auto result_expr_sid = PackSid(result_expr);
619 PushArgs(result_expr, result_expr_sid, &args);
620
621 GlobalVar global_var = call_lowered_props.lowered_func;
622 bool has_c_device_api_context = device_contexts_.count(global_var) != 0;
623 tir::Var device_context;
624 tir::Stmt func_call;
625
626 switch (call_type_) {
627 case CallType::kUnpacked: {
628 // call_extern calling convention with optional context
629 if (has_c_device_api_context) {
630 device_context = device_contexts_.Get(global_var).value();
631 args.push_back(device_context);
632 }
633 func_call = tir::Evaluate(AddCheckReturn(
634 tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::call_extern(), args)));
635 break;
636 }
637 case CallType::kCPacked: {
638 if (has_c_device_api_context) {
639 device_context = device_contexts_.Get(global_var).value();
640 args.push_back(device_context);
641 } else {
642 // NOTE: LowerTVMBuiltin expects some device_context placeholder.
643 args.push_back(tir::make_zero(DataType::Handle()));
644 }
645 func_call = tir::Evaluate(
646 tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::tvm_call_cpacked(), args));
647 create_func_call_stmts.push_back(func_call);
648 break;
649 }
650 case CallType::kPacked: {
651 // call_packed does not accept a device context.
652 CHECK(!has_c_device_api_context) << "CallType::kPacked does not accept a device context";
653 func_call = tir::Evaluate(AddCheckReturn(
654 tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::tvm_call_packed(), args)));
655 create_func_call_stmts.push_back(func_call);
656 break;
657 }
658 default:
659 ICHECK(false) << "Unknown CallType: " << call_type_;
660 }
661
662 ICHECK(func_call.defined()) << "Must define func_call";
663
664 if (has_c_device_api_context) {
665 func_call = tir::SeqStmt(Array<tir::Stmt>({
666 GenerateDeviceHook(device_context, "Open"),
667 func_call,
668 GenerateDeviceHook(device_context, "Close"),
669 }));
670 }
671
672 tir::Stmt body = tir::SeqStmt({func_call});
673 stmts_.push_back(body);
674 }

676 /*!
677 * \brief Copy a variable to the output. This function is mainly used in edge cases
678 * when we want to return an input or a parameter.
679 * TODO(giuseros): we should try to avoid unnecessary copy to the output, e.g., in a
680 * copy-on-write fashion.
681 */
682 void CopyToOutput(PrimExpr out, PrimExpr in, bool pack_input, size_t size) {
683 // Define intermediate DLTensor to load/store the data
684 tir::Buffer tmp_read =
685 tir::decl_buffer({IntImm(DataType::UInt(64), size)}, DataType::UInt(8), "tmp_read");
686 tir::Buffer tmp_write =
687 tir::decl_buffer({IntImm(DataType::UInt(64), size)}, DataType::UInt(8), "tmp_write");
688 te::Var loop_idx("i", DataType::Int(32));
689 auto retval_i = tir::BufferLoad(tmp_read, {loop_idx});
690 // Copy the variable from the input to the output
691 tir::Stmt copy = tir::For(
692 loop_idx, 0, tir::make_const(DataType::Int(32, 1), size, Span()), tir::ForKind::kSerial,
693 tir::BufferStore(tmp_write, tir::Let(tmp_read->data, in, retval_i), {loop_idx}));
694 stmts_.push_back(tir::LetStmt(tmp_write->data, out, copy));
695 }

697 /*!
698 * \brief Generates a call to a given hook for all Devices found for C Device API
699 * \param hook Name of hook to generate statements for
700 * \return Statement with function calls for each device
701 */
702 tir::Stmt GenerateAllDeviceHook(const String& hook) {
703 std::vector<tir::Stmt> device_hooks;
704 for (const auto& it : devices_) {
705 const String& device_name = it.first;
706 const tir::Var& context = it.second;
707 Array<String> sections = {"Device", device_name, hook};
708 String device_hook_name = ToCFunctionStyle(PrefixName(sections));
709
710 tir::Evaluate device_hook(
711 AddCheckReturn(tvm::tir::Call(DataType::Int(32), tvm::tir::builtin::call_extern(),
712 {tvm::tir::StringImm(device_hook_name), context})));
713 device_hooks.push_back(device_hook);
714 }
715 return tir::SeqStmt(device_hooks);
716 }
717
718 /*!
719 * \brief Generates a call to a given hook for a single Device function
720 * \param context Device context to call hook on
721 * \param hook Name of hook to generate statements for
722 * \return Statement with function call to Device API
723 */
724 tir::Stmt GenerateDeviceHook(const tir::Var& context, const String& hook) {
725 const auto& it = std::find_if(std::begin(devices_), std::end(devices_), [&](const auto& it) {
726 return it.second->name_hint == context->name_hint;
727 });
728 const String& device_name = (*it).first;
729 Array<String> sections = {"Device", device_name, hook};
730 String device_hook = ToCFunctionStyle(PrefixName(sections));
731
732 return tir::Evaluate(
733 AddCheckReturn(tir::Call(DataType::Int(32), tvm::tir::builtin::call_extern(),
734 {tvm::tir::StringImm(device_hook), context})));
735 }

763 CreateIOVar(expr->checked_type(), original_name, use_unique_name);
764 }
765
766 /*!
767 * \brief Create tir::Var for input/output while updating the buffer_maps.
768 * \param expr The expression to evaluate.
769 * \param original_name The name of the tir::Var.
770 * \param use_unique_name Whether to generate a new unique name where a name conflicts.
771 */
772 void CreateIOVar(const Type& type, const std::string& original_name,
773 bool use_unique_name = true) {
774 if (type->IsInstance<TupleTypeNode>()) {
775 TupleType tuple_type = Downcast<TupleType>(type);
776 for (unsigned i = 0; i < tuple_type->fields.size(); i++) {
777 CreateIOVar(tuple_type->fields[i], original_name);
778 }
779 } else {
780 std::string name = original_name;
781 if (use_unique_name) {
782 name = GetUniqueIOVarName(original_name);
783 }
784 tir::Var var = tir::Var(name, DataType::Handle());
785 main_signature_.push_back(var);
786 auto tensor_type = type.as<TensorTypeNode>();
787 ICHECK(tensor_type) << "Expected TensorType node but was " << type->GetTypeKey();
788 DataType elem_type = tensor_type->dtype;
789 tir::Var buffer_var =
790 tir::Var(name + "_buffer_var", PointerType(PrimType(elem_type), "global"));
791 tir::Buffer buffer = tir::Buffer(buffer_var, elem_type, tensor_type->shape, {}, 0,
792 name + "_buffer", 16, 1, tir::BufferType::kDefault);
793 main_buffer_map_.Set(var, buffer);
794 }
795 }

797 /*!
798 * \brief Create a unique name for I/O Var
799 */
800 std::string GetUniqueIOVarName(std::string name) {
801 if (io_var_names_.find(name) == io_var_names_.end()) {
802 io_var_names_[name] = 1;
803 return name + std::to_string(io_var_names_[name] - 1);
804 } else {
805 io_var_names_[name] = io_var_names_[name] + 1;
806 return name + std::to_string(io_var_names_[name] - 1);
807 }
808 }
