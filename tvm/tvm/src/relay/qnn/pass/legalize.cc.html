25#include <tvm/relay/qnn/transform.h>
26
27namespace tvm {
28namespace relay {
29namespace qnn {
30
31namespace transform {
32
33// QnnLegalize pass is a wrapper for relay::legalize::Legalize pass.
34Pass QnnLegalize() {
35 runtime::TypedPackedFunc<Function(Function, IRModule, relay::transform::PassContext)> pass_func =
36 [=](Function f, IRModule m, relay::transform::PassContext pc) {
37 return Downcast<Function>(relay::legalize::Legalize(f, "FTVMQnnLegalize"));
38 };
39 return relay::transform::CreateFunctionPass(pass_func, 1, "QnnLegalize", {"InferType"});
40}
41
42// QnnCanonicalize pass is a wrapper for relay::legalize::Legalize pass.
43Pass QnnCanonicalize() {
44 runtime::TypedPackedFunc<Function(Function, IRModule, relay::transform::PassContext)> pass_func =
45 [=](Function f, IRModule m, relay::transform::PassContext pc) {
46 return Downcast<Function>(relay::legalize::Legalize(f, "FTVMQnnCanonicalize"));
47 };
48 return relay::transform::CreateFunctionPass(pass_func, 1, "QnnCanonicalize", {"InferType"});
49}
50
51Pass Legalize() {
52 Array<Pass> pass_seqs;
53 pass_seqs.push_back(QnnLegalize());
54 pass_seqs.push_back(QnnCanonicalize());
55 relay::transform::Pass seq = relay::transform::Sequential(pass_seqs, "qnn.Legalize");
56 return seq;
57}
58
59TVM_REGISTER_GLOBAL("relay.qnn._transform.Legalize").set_body_typed(Legalize);
