20/*!
21 * \file relay/qnn/pass/legalize.cc
22 * \brief The Legalize wrapper for QNN.
23 */
24
25#include <tvm/relay/qnn/transform.h>
26
27namespace tvm {
28namespace relay {
29namespace qnn {
30
31namespace transform {
32
33Pass Legalize() {
34 Array<Pass> pass_seqs;
35 pass_seqs.push_back(relay::transform::Legalize("FTVMQnnLegalize"));
36 pass_seqs.push_back(relay::transform::Legalize("FTVMQnnCanonicalize"));
37 relay::transform::Pass seq = relay::transform::Sequential(pass_seqs, "qnn.Legalize");
38 return seq;
39}
40
41TVM_REGISTER_GLOBAL("relay.qnn._transform.Legalize").set_body_typed(Legalize);
42
43} // namespace transform
44
45} // namespace qnn
