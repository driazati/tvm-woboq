51static inline int32_t GetQmin(const DataType& dtype) {
52 ICHECK_LE(dtype.bits(), 32) << "QNN ops support int32 or lower precision";
53 if (dtype.is_int() || dtype.is_uint()) {
54 auto min_value_expr = tvm::min_value(dtype);
55 auto* min_value = tir::as_const_int(min_value_expr);
56 ICHECK(min_value != nullptr);
57 return static_cast<int32_t>(min_value[0]);
58 } else {
59 LOG(FATAL) << "Type not supported " << dtype;
60 return -1; // To hide the warning
61 }
62}
63
64static inline int32_t GetQmax(const DataType& dtype) {
65 ICHECK_LE(dtype.bits(), 32) << "QNN ops support int32 or lower precision";
66 if (dtype.is_int() || dtype.is_uint()) {
67 auto max_value_expr = tvm::max_value(dtype);
68 auto* max_value = tir::as_const_int(max_value_expr);
69 ICHECK(max_value != nullptr);
70 return static_cast<int32_t>(max_value[0]);
71 } else {
72 LOG(FATAL) << "Type not supported " << dtype;
73 return -1; // To hide the warning
74 }
75}

77/*
78 * \brief Convert FP32 representation into fixed point representation.
79 * \param double_multplier The input FP32 number.
80 * \return The pair of multiplier and shift for fixed point representation.
81 * \note Converts a floating point number so that it can be represented by
82 * integers. The representation is
83 * float_number = (significand) * 2^(exponent)
84 *
85 * The significand is a number between 0.5 and 1. This is represented by
86 * an integer number. For example, if it is int32, then the decimal point
87 * exists between bit 31 and 30 from LSB (or between first and second bit
88 * from the left).
89 *
90 * Some examples are
91 * 0.25 = (0.5) * 2^(-1)
92 * 0.125 = (0.5) * 2^(-2)
93 *
94 * Credit to TFLite reference implementation.
95 */
96std::pair<int32_t, int32_t> GetFixedPointMultiplierShift(double double_multiplier);

106static inline Expr Requantize(const Expr& data, const Array<IndexExpr>& input_shape,
107 const Expr& input_scale, const Expr& input_zero_point,
108 const Expr& output_scale, const Expr& output_zero_point,
109 const DataType& out_dtype, const int& axis = -1,
110 const std::string& rounding = "None",
111 const std::string& compute_dtype = "None") {
112 auto attrs = make_object<RequantizeAttrs>();
113 attrs->axis = axis;
114 attrs->out_dtype = std::move(out_dtype);
115 const RequantizeConfig& cfg = RequantizeConfig::Current();
116 attrs->rounding =
117 SelectRequntizeParameter(rounding, cfg->get_rounding(), cfg->is_default, "rounding");
118 attrs->compute_dtype = SelectRequntizeParameter(compute_dtype, cfg->get_compute_dtype(),
119 cfg->is_default, "compute_dtype");
120 return RequantizeLower(data, input_scale, input_zero_point, output_scale, output_zero_point,
121 attrs.operator->(), input_shape, attrs->out_dtype);
122}

164/*
165 * \brief Fixed point multiplication between integer tensor with floating point
166 * scalar. This implementation rounds to the nearest value when it is midway
167 * between two representable values.
168 * \param tensor The quantized input tensor of dtype int64.
169 * \param multiplier The scalar multiplier.
170 * \param input_shape Shape of the input tensor.
171 * \return The sequence of Relay ops for fixed point multiplication with TONEARES rounding.
172
173 * \note Original compuation is scale_fp32 * quantized_tensor. To convert into
174 * integer computation, the multiplication with fp32 scalar can be
175 * replaced by multiplication with an int value and then right shifting
176 * the result. This approximates the floating point computation with a
177 * fixed point computation.
178 *
179 * Computation of fixed point multiplication is consist of following
180 steps:
181 * 1) Multiply the fixed point multiplier with quantized tensor.
182 * 2) Round the result.
183 * 3) Right shift the result
184 */
185Expr FixedPointMultiplyToNearest(Expr tensor, double multiplier,
186 const Array<IndexExpr>& input_shape);

188/*
189 * \brief Fixed point multiplication between integer tensor with floating point
190 scalar where the input tensor is per-axis/per-channel quantized..
191 * \param tensor The quantized input tensor of dtype int64.
192 * \param multiplier The scalar multiplier.
193 * \param input_shape Shape of the input tensor.
194 * \param channel_axis The channel_axis along which the input tensor is quantized. Default value is
195 -1 which corresponds to the last channel_axis.
196 * \param rounding "UPWARD" or "TONEAREST". The rounding direction when the value
197 is midway between" "two representable values.
198 * \return The sequence of Relay ops for fixed point multiplication.
199
200 * \note Original compuation is scale_fp32 * quantized_tensor. To convert into
201 * integer computation, the multiplication with fp32 vector can be
202 * replaced by multiplication with an int vector and then right shifting
203 * the result. This approximates the floating point computation with a
204 * fixed point computation.
205 *
206 * Computation of fixed point multiplication is consist of following
207 steps:
208 * 1) Multiply the fixed point multiplier with quantized tensor.
209 * 2) Round the result.
210 * 3) Right shift the result
211 */
212Expr FixedPointMultiplyPerChannel(Expr tensor, std::vector<double> multiplier,
213 const Array<IndexExpr>& input_shape, int channel_axis,
214 const std::string& rounding);

232/*
233 * \brief Checks whether an expr type is scalar of a given data type.
234 * \param expr_type The type of expr to be checked.
235 * \param dtype The expected dtype.
236 * \return True if the type is a scalar of given dtype
237 */
238static inline bool IsScalarType(const Type& expr_type, const DataType& dtype) {
239 const auto* tensor_type = expr_type.as<TensorTypeNode>();
240 ICHECK(tensor_type) << "Only tensor type can be checked for scalar values. But got"
241 << AsText(expr_type, false);
242 ICHECK_EQ(tensor_type->shape.size(), 0);
243 ICHECK(tensor_type->dtype == dtype) << "Expected " << dtype << " but got " << tensor_type->dtype;
244 return true;
245}

259/*
260 * \brief Checks and assigns types to scale and zero points.
261 * \param expr_type The type of expr to be checked.
262 * \param dtype The expected dtype.
263 * \param shape The shape at C dim of original tensor.
264 * \param reporter The type reported of original InferType call.
265 */
266static inline void AssignType(const Type& expr_type, const DataType& dtype, const IndexExpr& shape,
267 const TypeReporter& reporter) {
268 // Scale/Zero_points can be either const scalar or a vector with C axis num elems.
269 const auto* tensor_type = expr_type.as<TensorTypeNode>();
270 ICHECK(tensor_type) << "Can assign type to Tensor type only. But got "
271 << AsText(expr_type, false);
272 const auto tensor_dtype = tensor_type->dtype;
273 ICHECK(tensor_dtype == dtype) << "Expected type is " << dtype << " but received " << tensor_dtype;
274 if (tensor_type->shape.size() != 0) {
275 reporter->Assign(expr_type, TensorType({shape}, tensor_type->dtype));
276 }
277}

279static inline std::vector<float> GetFloatVectorFromConstant(const Expr& expr) {
280 const auto* n = expr.as<ConstantNode>();
281 std::vector<float> vals;
282 ICHECK(n) << "Expr must be a constant expr - " << AsText(expr, false);
283 int64_t num_elems = 1;
284 auto shape = n->data.Shape();
285 for (size_t i = 0; i < shape.size(); i++) {
286 num_elems *= shape[i];
287 }
288 for (int64_t i = 0; i < num_elems; i++) {
289 vals.push_back(static_cast<float*>(n->data->data)[i]);
290 }
291 return vals;
292}
