33/*
34 * \brief Canonicalizes the QNN subtract op.
35 * \param attrs The empty attribute.
36 * \param new_args The new mutated args to the call node.
37 * \param arg_types The types of input and output.
38 * \return The sequence of Relay ops for add op.
39 */
40Expr QnnSubtractCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
41 const Array<tvm::relay::Type>& arg_types) {
42 // Get the args.
43 QnnBinaryOpArguments args(new_args);
44
45 // Get the input dtype and shape.
46 QnnBinaryOpTensorType input_type(arg_types, 0);
47
48 const auto* broadcast_attrs = attrs.as<BroadcastAttrs>();
49 ICHECK(broadcast_attrs != nullptr);
50
51 auto lhs_axis = broadcast_attrs->lhs_axis;
52 auto rhs_axis = broadcast_attrs->rhs_axis;
53
54 // TODO(shoubhik) - The lowering can be further optimized. Instead of inserting requantize in
55 // the start, we can insert requantize at the end if both input tensors have same qnn params. In
56 // that case, we can first subtract the tensors, add the zero point, and requantize at the end.
57 // This can be done in future.
58
59 // Since the input qnn params can be different than output qnn params, we first requantize the
60 // input tensors to the output qnn params. Then we call relay.subtract on the requantized inputs.
61 // This subtraction results in extra subtraction of the output zero point. We further add
62 // the zero point. The whole process can be represented using following equations
63 //
64 // scale_c * (Q_c - zp_c) = scale_a * (Q_a - zp_a) - scale_b * (Q_b - zp_b)
65 //
66 // After requantizing Q_a and Q_b, equation becomes,
67 // scale_c * (Q_c - zp_c) = scale_c * (Q_a' - zp_c) - scale_c * (Q_b' - zp_c)
68 // scale_c * (Q_c - zp_c) = scale_c * (Q_a' - Q_b')
69 //
70 // Comparing the LHS and RHS, it results in
71 // Q_c = Q_a' - Q_b' + zp_c
72 // The subtract op is done in int32 precision.
73
74 // Requantize LHS if necessary. Computes Q_a'
75 auto requantized_lhs =
76 RequantizeOrUpcast(args.lhs, args.lhs_scale, args.lhs_zero_point, args.output_scale,
77 args.output_zero_point, input_type.shape, lhs_axis);
78 // Requantize RHS if necessary. Computes Q_b'
79 auto requantized_rhs =
80 RequantizeOrUpcast(args.rhs, args.rhs_scale, args.rhs_zero_point, args.output_scale,
81 args.output_zero_point, input_type.shape, rhs_axis);
82
83 // Computes Q_a' - Q_b'
84 auto output = Subtract(requantized_lhs, requantized_rhs);
85
86 // Add zero point. Computes (Q_a' - Q_b') + zp_c
87 auto zero_scalar = MakeConstantScalar(DataType::Int(32), 0);
88 if (!IsEqualScalar(args.output_zero_point, zero_scalar)) {
89 output = Add(output, args.output_zero_point);
90 }
91
92 // Go back to lower precision.
93 return ConvertDtype(output, input_type.dtype);
94}
95
96// QNN Subtraction operator.
97QNN_REGISTER_BINARY_OP("subtract")
98 .describe("Elementwise subtract with broadcasting for quantized tensors.")
99 .set_support_level(11)
100 .set_attr<FTVMLegalize>("FTVMQnnCanonicalize", QnnSubtractCanonicalize);
