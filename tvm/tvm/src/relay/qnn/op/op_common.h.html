57/*
58 * \brief Simple struct to organize the inputs to the Qnn
59 * binary operators. The main reason to have a struct
60 * is to be able to perform the common checks needed at a
61 * central location.
62 */
63struct QnnBinaryOpArguments {
64 Expr lhs;
65 Expr rhs;
66 Expr lhs_scale;
67 Expr lhs_zero_point;
68 Expr rhs_scale;
69 Expr rhs_zero_point;
70 Expr output_scale;
71 Expr output_zero_point;
72
73 explicit QnnBinaryOpArguments(const Array<Expr>& new_args) {
74 ICHECK_EQ(new_args.size(), kNumQnnBinaryOpInputs);
75 int idx = 0;
76 lhs = new_args[idx++];
77 rhs = new_args[idx++];
78 lhs_scale = new_args[idx++];
79 lhs_zero_point = new_args[idx++];
80 rhs_scale = new_args[idx++];
81 rhs_zero_point = new_args[idx++];
82 output_scale = new_args[idx++];
83 output_zero_point = new_args[idx++];
84 ICHECK_EQ(idx, kNumQnnBinaryOpInputs);
85 }
86};

159/*
160 * \brief Converts the expression from expression's dtype
161 * to target dtype. This is mainly used for converting
162 * computations done in Int32 to lower precision Int8 or
163 * UInt8.
164 * \param expr The expression to whose dtype needs conversion.
165 * \param target_dtype The dtype of the target expression
166 * \return New expression with target dtype and possibly lower
167 * precision.
168 */
169inline Expr ConvertDtype(const Expr& expr, const DataType& target_dtype) {
170 auto q_min = GetQmin(target_dtype);
171 auto q_max = GetQmax(target_dtype);
172 auto output = Clip(expr, q_min, q_max);
173 return Cast(output, target_dtype);
174}
175
176/*
177 * \brief Requantizes the given expression if expression's
178 * scale and zero point both do not match target scale and
179 * zero point. This is mainly needed for requantizing the
180 * input tensors with output tensor's scale and zero point
181 * to ease the computation of final quantized tensor.
182 * \param expr The expression on which the check needs to be performed.
183 * \param expr_scale The scale of the expression.
184 * \param expr_zero_point The zero point of the expression.
185 * \param target_scale The scale of the output tensor.
186 * \param target_zero_point The zero point of the output tensor.
187 * \param expr_shape The shape of the input expression.
188 * \return New expression that is requantized to target scale and zero
189 * point if the expression scale and zero points are different otherwise
190 * it simply casts the given expression to Int32 as no requantization is
191 * needed in this case.
192 */
193inline Expr RequantizeOrUpcast(const Expr& expr, const Expr& expr_scale,
194 const Expr& expr_zero_point, const Expr& target_scale,
195 const Expr& target_zero_point, const Array<PrimExpr>& expr_shape,
196 const int& axis = -1,
197 const DataType& target_dtype = DataType::Int(32)) {
198 auto result = expr;
199 if (!IsEqualScalar(expr_scale, target_scale) ||
200 !IsEqualScalar(expr_zero_point, target_zero_point)) {
201 result = Requantize(expr, expr_shape, expr_scale, expr_zero_point, target_scale,
202 target_zero_point, target_dtype, axis);
203 } else {
204 result = Cast(result, target_dtype);
205 }
206 return result;
207}

209/*! \brief Infer layout for QNN binary broadcast operators */
210inline InferCorrectLayoutOutput QnnBinaryBroadcastLayout(
211 const Attrs& attrs, const Array<Layout>& new_in_layouts, const Array<Layout>& old_in_layouts,
212 const Array<tvm::relay::Type>& old_in_types) {
213 // Use Relay Binary Broadcast Infer correct layout.
214 auto layouts = BinaryBroadcastLayout(attrs, new_in_layouts, old_in_layouts, old_in_types);
215
216 // Fill the layouts of remaining input tensors - scales and zero points. The layouts of these
217 // tensors can be treated as C.
218 Layout channel_layout = Layout("C");
219 Array<Layout> input_layouts = {layouts->input_layouts[0],
220 layouts->input_layouts[1],
221 channel_layout,
222 channel_layout,
223 channel_layout,
224 channel_layout,
225 channel_layout,
226 channel_layout};
227 Array<Layout> output_layouts = layouts->output_layouts;
228 return InferCorrectLayoutOutput(input_layouts, output_layouts, attrs);
229}

231static inline bool QnnBroadcastRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
232 const TypeReporter& reporter) {
233 // Expected Types: lhs, rhs, lhs_scale, lhs_zero_point, rhs_scale, rhs_zero_point, output_scale,
234 // output_zero_point, out_type
235 ICHECK_EQ(types.size(), kNumQnnBinaryOpArgTypes);
236
237 // Check the lhs and rhs types
238 for (size_t i = 0; i < 2; ++i) {
239 if (types[i].as<IncompleteTypeNode>()) {
240 return false;
241 }
242 }
243 // Check the scale and zero point types
244 for (size_t i = 2; i < 8; ++i) {
245 if (types[i].as<IncompleteTypeNode>()) {
246 return false;
247 }
248 }
249
250 const auto* lhs_data = types[0].as<TensorTypeNode>();
251 const auto* rhs_data = types[1].as<TensorTypeNode>();
252
253 if (lhs_data == nullptr || rhs_data == nullptr) {
254 return false;
255 }
256
257 const BroadcastAttrs* broadcast_attrs = attrs.as<BroadcastAttrs>();
258 ICHECK(broadcast_attrs);
259
260 auto lhs_rank = static_cast<int>(lhs_data->shape.size());
261 auto rhs_rank = static_cast<int>(rhs_data->shape.size());
262
263 auto get_channel_axis = [](int rank, int axis_from_attr) {
264 if (rank <= 1) return 0;
265 if (axis_from_attr < 0) return rank + axis_from_attr;
266 return axis_from_attr;
267 };
268
269 const int lhs_axis = get_channel_axis(lhs_rank, broadcast_attrs->lhs_axis);
270 const int rhs_axis = get_channel_axis(rhs_rank, broadcast_attrs->rhs_axis);
271
272 // If zero point and scale are scalar then axis doesn't matter.
273 bool lhs_scale_is_scalar = (types[2].as<TensorTypeNode>())->shape.size() == 0;
274 bool lhs_zp_is_scalar = (types[3].as<TensorTypeNode>())->shape.size() == 0;
275 bool rhs_scale_is_scalar = (types[4].as<TensorTypeNode>())->shape.size() == 0;
276 bool rhs_zp_is_scalar = (types[5].as<TensorTypeNode>())->shape.size() == 0;
277
278 if (!(lhs_scale_is_scalar && lhs_zp_is_scalar)) {
279 ICHECK_LT(lhs_axis, lhs_rank > 0 ? lhs_rank : 1)
280 << "lhs_axis " << broadcast_attrs->lhs_axis << " is out of range";
281 ICHECK_GE(lhs_axis, 0) << "lhs_axis " << broadcast_attrs->lhs_axis << " is out of range";
282 }
283
284 if (!(rhs_scale_is_scalar && rhs_zp_is_scalar)) {
285 ICHECK_LT(rhs_axis, rhs_rank > 0 ? rhs_rank : 1)
286 << "rhs_axis " << broadcast_attrs->rhs_axis << " is out of range";
287 ICHECK_GE(rhs_axis, 0) << "rhs_axis " << broadcast_attrs->rhs_axis << " is out of range";
288 }
289
290 PrimExpr lhs_axis_shape;
291 if (lhs_rank > 0) {
292 lhs_axis_shape = lhs_data->shape[lhs_axis];
293 } else {
294 lhs_axis_shape = Integer(1);
295 }
296
297 PrimExpr rhs_axis_shape;
298 if (rhs_rank > 0) {
299 rhs_axis_shape = rhs_data->shape[rhs_axis];
300 } else {
301 rhs_axis_shape = Integer(1);
302 }
303
304 // Check and assign types for scale and zero points.
305 AssignType(types[2], DataType::Float(32), lhs_axis_shape, reporter); // lhs_scale
306 AssignType(types[3], DataType::Int(32), lhs_axis_shape, reporter); // lhs_zero_point
307 AssignType(types[4], DataType::Float(32), rhs_axis_shape, reporter); // rhs_scale
308 AssignType(types[5], DataType::Int(32), rhs_axis_shape, reporter); // rhs_zero_point
309
310 ICHECK(IsScalarType(types[6], DataType::Float(32))); // output_scale
311 ICHECK(IsScalarType(types[7], DataType::Int(32))); // output_zero_point
312
313 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
314 // BroadcastRel infer type function.
315 Array<Type> tensor_types = {types[0], types[1], types[8]};
316 return BroadcastRel(tensor_types, 3, attrs, reporter);
317}

319/*! Quick helper macro
320 * - Expose a positional make function to construct the node.
321 * - Register op to the registry.
322 *
323 * We make the decision to always only expose positional argument.
324 * We will do rewrapping in the frontend to support language
325 * sugars such as keyword arguments and default value.
326 *
327 * \param OpName the name of registry.
328 */
329#define QNN_REGISTER_BINARY_OP(OpName) \
330 TVM_REGISTER_GLOBAL("relay.qnn.op._make." OpName) \
331 .set_body_typed([](Expr lhs, Expr rhs, Expr lhs_scale, Expr lhs_zero_point, Expr rhs_scale, \
332 Expr rhs_zero_point, Expr output_scale, Expr output_zero_point, \
333 int lhs_axis, int rhs_axis) { \
334 static const Op& op = Op::Get("qnn." OpName); \
335 auto attrs = make_object<BroadcastAttrs>(); \
336 attrs->lhs_axis = lhs_axis; \
337 attrs->rhs_axis = rhs_axis; \
338 return Call(op, \
339 {lhs, rhs, lhs_scale, lhs_zero_point, rhs_scale, rhs_zero_point, output_scale, \
340 output_zero_point}, \
341 Attrs(attrs), {}); \
342 }); \
343 RELAY_REGISTER_OP("qnn." OpName) \
344 .set_attrs_type<BroadcastAttrs>() \
345 .set_num_inputs(kNumQnnBinaryOpInputs) \
346 .add_argument("lhs", "Tensor", "The left hand side quantized tensor.") \
347 .add_argument("rhs", "Tensor", "The right hand side quantized tensor.") \
348 .add_argument("lhs_scale", "Tensor", "The scale of the lhs tensor.") \
349 .add_argument("lhs_zero_point", "Tensor", "The zero_point of the lhs tensor.") \
350 .add_argument("rhs_scale", "Tensor", "The scale of the rhs tensor.") \
351 .add_argument("rhs_zero_point", "Tensor", "The zero_point of the rhs tensor.") \
352 .add_argument("output_scale", "Tensor", "The scale of the output tensor.") \
353 .add_argument("output_zero_point", "Tensor", "The zero_point of the output tensor.") \
354 .add_argument("lhs_axis", "Tensor", "The channel quantization of the lhs tensor.") \
355 .add_argument("rhs_axis", "Tensor", "The channel quantization of the rhs tensor.") \
356 .add_type_rel("QnnBroadcast", QnnBroadcastRel) \
357 .set_attr<TNonComputational>("TNonComputational", true) \
358 .set_attr<FInferCorrectLayout>("FInferCorrectLayout", QnnBinaryBroadcastLayout)
359
360static inline bool QnnElementwiseUnaryFuncRel(const Array<Type>& types, int num_inputs,
361 const Attrs& attrs, const TypeReporter& reporter) {
362 // Expected Types: data, scale, zero_point, output_scale, output_zero_point
363 ICHECK_EQ(types.size(), 6);
364 const auto* x = types[0].as<TensorTypeNode>();
365 if (x == nullptr) return false;
366 ICHECK(x->dtype == DataType::Int(8) || x->dtype == DataType::UInt(8))
367 << "Expected quantized type(int8, uint8) for input but was " << x->dtype;
368
369 // Check the types of scale and zero points.
370 for (size_t i = 1; i < 5; ++i) {
371 if (types[i].as<IncompleteTypeNode>()) {
372 return false;
373 }
374 }
375 ICHECK(IsScalarType(types[1], DataType::Float(32))); // scale
376 ICHECK(IsScalarType(types[2], DataType::Int(32))); // zero_point
377 ICHECK(IsScalarType(types[3], DataType::Float(32))); // output_scale
378 ICHECK(IsScalarType(types[4], DataType::Int(32))); // output_zero_point
379
380 // Assign types for scale and zero points.
381 reporter->Assign(types[1], TensorType({}, DataType::Float(32))); // scale
382 reporter->Assign(types[2], TensorType({}, DataType::Int(32))); // zero_point
383 reporter->Assign(types[3], TensorType({}, DataType::Float(32))); // output_scale
384 reporter->Assign(types[4], TensorType({}, DataType::Int(32))); // output_zero_point
385
386 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
387 // IdentityRel infer type function.
388 Array<Type> tensor_types = {types[0], types[5]};
389 return IdentityRel(tensor_types, 2, attrs, reporter);
390}

392static inline Expr LegalizeExpr(const Expr& expr) {
393 // Canonicalizations should not contain qnn ops, so use this
394 // to lower expressions automatically after using things like qnn.dequantize
395 // in the lowering process.
396 auto mod = IRModule::FromExpr(expr);
397 mod = transform::Legalize()(mod);
398 if (expr.as<FunctionNode>()) {
399 return mod->Lookup("main");
400 } else {
401 return mod->Lookup("main").as<FunctionNode>()->body;
402 }
403}

426 .add_argument("zero_point", "Tensor", "The quantization zero_point of the input tensor.") \
427 .add_argument("output_scale", "Tensor", "The quantization scale of the output tensor.") \
428 .add_argument("output_zero_point", "Tensor", \
429 "The quantization zero_point of the output tensor.") \
430 .set_support_level(11) \
431 .add_type_rel("qnn." OpName, QnnElementwiseUnaryFuncRel) \
432 .set_attr<TNonComputational>("TNonComputational", true)

434/*! Quick helper macro
435 * Create a default canonicalization for a QNN operator, which dequantizes the operator
436 * runs the calculation using the provided Call func, and then requantizes.
437 *
438 * FloatingPointFunc is usually a handle from "src/relay/transforms/pattern_utils.h"
439 *
440 * \param FloatingPointFunc the floating point function with function signature `Expr Erf(Expr e)`
441 */
442#define QNN_UNARY_OP_DEFAULT_CANONICALIZATION(FloatingPointFunc) \
443 [](const Attrs& attrs, const Array<Expr>& new_args, const Array<tvm::relay::Type>& arg_types) { \
444 QnnUnaryOpArguments args(new_args); \
445 QnnUnaryOpTensorType input_type(arg_types, 0); \
446 Expr dequantized_arg = MakeDequantize(args.x, args.scale, args.zero_point, -1); \
447 Expr output = FloatingPointFunc(dequantized_arg); \
448 Expr result = \
449 MakeQuantize(output, args.output_scale, args.output_zero_point, -1, input_type.dtype); \
450 return LegalizeExpr(result); \
451 }
