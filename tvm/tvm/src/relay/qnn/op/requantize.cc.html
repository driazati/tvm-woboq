41InferCorrectLayoutOutput RequantizeInferCorrectLayout(const Attrs& attrs,
42 const Array<Layout>& new_in_layouts,
43 const Array<Layout>& old_in_layouts,
44 const Array<tvm::relay::Type>& old_in_types) {
45 const auto* attrs_ptr = attrs.as<RequantizeAttrs>();
46 ICHECK(attrs_ptr);
47 ObjectPtr<RequantizeAttrs> param = make_object<RequantizeAttrs>(*attrs_ptr);
48
49 Array<Array<IndexExpr>> old_in_shapes;
50 for (auto old_in_t : old_in_types) {
51 ICHECK(old_in_t.as<TensorTypeNode>());
52 old_in_shapes.push_back(old_in_t.as<TensorTypeNode>()->shape);
53 }
54
55 Array<Layout> input_layouts, output_layouts;
56 if (new_in_layouts.defined()) {
57 // Adapt to new layout. The axis has to change.
58 // Record original reduce axis. Convert to the modified layout axis.
59 ICHECK_EQ(new_in_layouts.size(), 5);
60 ICHECK_EQ(old_in_layouts.size(), 5);
61
62 // 1) Get the axis.
63 int axis = param->axis;
64 axis = (axis == -1) ? old_in_shapes[0].size() - 1 : axis;
65
66 // 2) Collect the original axis
67 std::string old_dim = old_in_layouts[0][axis].name();
68
69 // 3) Collect the new axes by walking new_layout.
70 tvm::Integer new_axis;
71 std::string new_layout_string = "";
72 int axis_index = 0;
73 for (auto iter_var : new_in_layouts[0]->axes) {
74 const auto& layout_axis = LayoutAxis::Get(iter_var);
75 const std::string& layout_dim = layout_axis.name();
76 if (old_dim == layout_dim) {
77 new_axis = tvm::Integer(axis_index);
78 }
79 // Collect only the primal axis.
80 if (layout_axis.IsPrimal()) {
81 new_layout_string += layout_dim;
82 axis_index++;
83 }
84 }
85
86 // 4) Set the new axis and layout.
87 Layout new_layout = Layout(new_layout_string);
88
89 // Fill the layouts of remaining input tensors - scales and zero points. The layouts of these
90 // tensors can be treated as channel layout.
91 Layout channel_layout = Layout("C");
92 input_layouts = {new_layout, channel_layout, channel_layout, channel_layout, channel_layout};
93 output_layouts = {new_layout};
94 param->axis = new_axis.IntValue();
95 } else if (old_in_layouts.defined()) {
96 // If the new layout is undefined, set the old layout as the inferred layout.
97 ICHECK_EQ(old_in_layouts.size(), 5);
98
99 Layout old_layout = old_in_layouts[0];
100
101 // Fill the layouts of remaining input tensors - scales and zero points. The layouts of these
102 // tensors can be treated as channel layout.
103 Layout channel_layout = Layout("C");
104 input_layouts = {old_layout, channel_layout, channel_layout, channel_layout, channel_layout};
105 output_layouts = {old_layout};
106 } else {
107 // Set the layouts to undef.
108 Layout undef = Layout::Undef();
109 input_layouts = Array<Layout>(5, undef);
110 output_layouts = {undef};
111 }
112
113 return InferCorrectLayoutOutput(input_layouts, output_layouts, Attrs(param));
114}

116bool has_current_target_sse41_support() {
117 auto target = Target::Current(true);
118 Optional<String> mcpu =
119 target.defined() ? target->GetAttr<String>("mcpu") : Optional<String>(nullptr);
120 auto target_has_sse41_fn_ptr = tvm::runtime::Registry::Get("tvm.topi.x86.utils.target_has_sse41");
121 ICHECK(target_has_sse41_fn_ptr) << "Function tvm.topi.x86.utils.target_has_sse41 not found";
122 return mcpu && (*target_has_sse41_fn_ptr)(mcpu.value());
123}

125/*
126 * \brief TONEAREST is the standard rounding where the value is rounded away
127 * from zero at midpoints (for example, -1.5 rounds to -2).
128 * \param input_tensor The input tensor to rounding op.
129 * \return The sequence of existing Relay ops.
130 */
131template <int Bits>
132Expr Tonearest(const Expr& input_tensor) {
133 if (has_current_target_sse41_support()) return Round(input_tensor);
134
135 auto half = MakeConstantScalar(DataType::Float(Bits), 0.5f);
136 auto zero = MakeConstantScalar(DataType::Float(Bits), 0.f);
137 auto pos_one = MakeConstantScalar(DataType::Float(Bits), +1.f);
138 auto neg_one = MakeConstantScalar(DataType::Float(Bits), -1.f);
139 auto multiplier = Where(Less(input_tensor, zero), neg_one, pos_one);
140 auto half_multiplied = Multiply(half, multiplier);
141 auto input_tensor_biased = Add(input_tensor, half_multiplied);
142 auto input_tensor_biased_multiplied = Multiply(input_tensor_biased, multiplier);
143 auto input_tensor_biased_multiplied_int =
144 Cast(input_tensor_biased_multiplied, DataType::Int(Bits));
145 auto input_tensor_biased_multiplied_float =
146 Cast(input_tensor_biased_multiplied_int, DataType::Float(Bits));
147 auto input_tensor_rounded = Multiply(input_tensor_biased_multiplied_float, multiplier);
148 return Where(IsFinite(input_tensor), input_tensor_rounded, input_tensor);
149}
150
151/*
152 * \brief UPWARD is the standard rounding except at midpoints where the value
153 * is rounded to positive infinity (for example, -1.5 rounds to -1).
154 * \param input_tensor The input tensor to rounding op.
155 * \return The sequence of existing Relay ops.
156 */
157template <int Bits>
158Expr Upward(const Expr& input_tensor) {
159 auto half = MakeConstantScalar(DataType::Float(Bits), 0.5f);
160 auto input_tensor_biased = Add(input_tensor, half);
161 if (has_current_target_sse41_support()) return Floor(input_tensor_biased);
162
163 auto zero = MakeConstantScalar(DataType::Float(Bits), 0.f);
164 auto one = MakeConstantScalar(DataType::Float(Bits), +1.f);
165 auto input_tensor_biased_int = Cast(input_tensor_biased, DataType::Int(Bits));
166 auto input_tensor_biased_float = Cast(input_tensor_biased_int, DataType::Float(Bits));
167 auto is_subtraction_not_necessary =
168 LogicalOr(Equal(input_tensor_biased, input_tensor_biased_float),
169 GreaterEqual(input_tensor_biased, zero));
170 auto input_tensor_rounded = Where(is_subtraction_not_necessary, input_tensor_biased_float,
171 Subtract(input_tensor_biased_float, one));
172 return Where(IsFinite(input_tensor), input_tensor_rounded, input_tensor);
173}

177/*
178 * \brief Lower requantize to a sequence of ops.
179 * \param input_tensor The input tensor to requantize op.
180 * \param param The requantize op attrs.
181 * \param input_shape The input tensor shape of the requantize op.
182 * \return The sequence of existing Relay ops.
183 * \note RequantizationInt using only integer computation. Here, the computation is
184 * converted to a fixed point computation by computing output multiplier
185 * and shift. This is useful, if the target device does not support/have
186 * very expensive floating point computations.
187 *
188 * The whole computation this can be broken down into following steps
189 * 1) Calculate the integer multiplier and integer shift.
190 * 2) Subtract the input integer zero point.
191 * 3) Perform fixed point multiplication.
192 * 4) Add the output zero point.
193 * 5) Cast to the out_dtype.
194 */
195Expr RequantizeLowerInt(const Expr& input_tensor, const Expr& input_scale,
196 const Expr& input_zero_point, const Expr& output_scale,
197 const Expr& output_zero_point, const RequantizeAttrs* param,
198 const Array<IndexExpr>& input_shape, const DataType& out_dtype) {
199 auto tensor = Cast(input_tensor, DataType::Int(32));
200 auto zero_scalar = MakeConstantScalar(DataType::Int(32), 0);
201 if (!IsEqualScalar(input_zero_point, zero_scalar)) {
202 // Broadcast input zero point if needed.
203 int rank = static_cast<int>(input_shape.size());
204 int axis = (param->axis < 0) ? ((rank > 0) ? rank + param->axis : 0) : param->axis;
205 Expr input_zero_broadcast = ExpandBiasToMatchAxis(Reshape(input_zero_point,
206 {
207 -1,
208 }),
209 rank, {axis});
210 tensor = Subtract(tensor, Cast(input_zero_broadcast, DataType::Int(32)));
211 }
212
213 // 2) If the input and output scales are same, we can skip the fixed point multiplication. Check
214 // if the input scale is per-tensor or per-channel. If it is per-tensor, there is single scale for
215 // the whole tensor. For per-channel (aka per-axis), there is a vector of scales for the input
216 // tensor. Depending on the quantization type, the fixed point multiplication routing is called.
217 auto scaled_int32_t = tensor;
218 float output_scale_float = GetScalarFromConstant<float>(output_scale);
219 if (IsConstScalar(input_scale)) {
220 // This is per-tensor quantization. Single scale.
221 float input_scale_float = GetScalarFromConstant<float>(input_scale);
222 double double_multiplier =
223 static_cast<double>(input_scale_float) / static_cast<double>(output_scale_float);
224 // Skip if input and output scales are same.
225 if (!IsEqualScalar(input_scale, output_scale)) {
226 int32_t fixed_point_multiplier, shift;
227 std::tie(fixed_point_multiplier, shift) = GetFixedPointMultiplierShift(double_multiplier);
228
229 const bool is_upward_rounding = (param->rounding == "UPWARD");
230
231 // When using upward rounding (i.e., x.5 rounded to x+1), leverage
232 // the FixedPointMultiply operator
233 scaled_int32_t =
234 (is_upward_rounding
235 ? FixedPointMultiply(scaled_int32_t, fixed_point_multiplier, shift)
236 : FixedPointMultiplyToNearest(scaled_int32_t, double_multiplier, input_shape));
237 }
238
239 } else {
240 // This is per-channel (per=axis) quantization.
241 std::vector<double> double_multipliers;
242 auto input_axis_scales = GetFloatVectorFromConstant(input_scale);
243 for (auto input_axis_scale : input_axis_scales) {
244 double multiplier =
245 static_cast<double>(input_axis_scale) / static_cast<double>(output_scale_float);
246 double_multipliers.push_back(multiplier);
247 }
248 int axis = param->axis;
249 axis = (axis == -1) ? input_shape.size() - 1 : axis;
250 scaled_int32_t = FixedPointMultiplyPerChannel(scaled_int32_t, double_multipliers, input_shape,
251 axis, param->rounding);
252 }
253
254 // 3) Add the output zero point.
255 auto shifted_int32_t = scaled_int32_t;
256 if (!IsEqualScalar(output_zero_point, zero_scalar)) {
257 shifted_int32_t = Add(Cast(output_zero_point, DataType::Int(32)), scaled_int32_t);
258 }
259
260 // 4) Clip to the out_dtype min/max. Skip clipping if out_dtype is Int32. The fixed point
261 // multiplication keeps the value in int32 range.
262 if (out_dtype == DataType::Int(32)) {
263 return shifted_int32_t;
264 }
265
266 auto q_min = GetQmin(out_dtype);
267 auto q_max = GetQmax(out_dtype);
268 auto clipped_t = Clip(shifted_int32_t, q_min, q_max);
269 return Cast(clipped_t, out_dtype);
270}

274/*
275 * \brief Lower requantize to a sequence of ops.
276 * \param input_tensor The input tensor to requantize op.
277 * \param param The requantize op attrs.
278 * \param input_shape The input tensor shape of the requantize op.
279 * \return The sequence of existing Relay ops.
280 * \note RequantizationFP using floating computation. All multiplication/sub/sum
281 * occurs in floating point data type and only at the end is converted to
282 * int32 data type and clamped for output data type.
283 *
284 * The whole computation this can be broken down into following steps
285 * 1) Subtract the input zero point.
286 * 2) Perform multiplication.
287 * 3) Add the output zero point.
288 * 4) Cast to the out_dtype.
289 */
290template <int Bits>
291Expr RequantizeLowerFP(const Expr& input_tensor, const Expr& input_scale,
292 const Expr& input_zero_point, const Expr& output_scale,
293 const Expr& output_zero_point, const RequantizeAttrs* param,
294 const Array<IndexExpr>& input_shape, const DataType& out_dtype) {
295 auto tensor = Cast(input_tensor, DataType::Float(Bits));
296 auto zero_scalar = MakeConstantScalar(DataType::Int(32), 0);
297 if (!IsEqualScalar(input_zero_point, zero_scalar)) {
298 // Broadcast input zero point if needed.
299 int rank = static_cast<int>(input_shape.size());
300 int axis = (param->axis < 0) ? ((rank > 0) ? rank + param->axis : 0) : param->axis;
301 Expr input_zero_broadcast = ExpandBiasToMatchAxis(Reshape(input_zero_point,
302 {
303 -1,
304 }),
305 rank, {axis});
306 tensor = Subtract(tensor, Cast(input_zero_broadcast, DataType::Float(Bits)));
307 }
308
309 // 2) If the input and output scales are same, we can skip the multiplication. Check
310 // if the input scale is per-tensor or per-channel. If it is per-tensor, there is single scale for
311 // the whole tensor. For per-channel (aka per-axis), there is a vector of scales for the input
312 // tensor. Depending on the quantization type, the fixed point multiplication routing is called.
313 auto scaled_fp_t = tensor;
314 double output_scale_float = GetScalarFromConstant<float>(output_scale);
315 if (IsConstScalar(input_scale)) {
316 // This is per-tensor quantization. Single scale.
317 double input_scale_float = GetScalarFromConstant<float>(input_scale);
318 double double_multiplier = static_cast<double>(input_scale_float) / output_scale_float;
319 // Skip if input and output scales are same.
320 if (!IsEqualScalar(input_scale, output_scale)) {
321 double multiplier = double_multiplier;
322 auto m_scalar = MakeConstantScalar(DataType::Float(Bits), multiplier);
323 scaled_fp_t = Multiply(m_scalar, scaled_fp_t);
324 }
325
326 } else {
327 // This is per-channel (per=axis) quantization.
328 std::vector<double> double_multipliers;
329 auto input_axis_scales = GetFloatVectorFromConstant(input_scale);
330 double output_scale_float = GetScalarFromConstant<float>(output_scale);
331 for (auto input_axis_scale : input_axis_scales) {
332 double multiplier = static_cast<double>(input_axis_scale) / output_scale_float;
333 double_multipliers.push_back(multiplier);
334 }
335 int axis = param->axis;
336 axis = (axis == -1) ? input_shape.size() - 1 : axis;
337
338 auto fixed_pt_multiplier_expr = MakeConstantTensor(
339 DataType::Float(Bits), {(int64_t)double_multipliers.size()}, double_multipliers);
340 size_t n_dim = input_shape.size();
341 auto exp_fixed_pt_multiplier_expr =
342 ExpandBiasToMatchAxis(fixed_pt_multiplier_expr, n_dim, {axis});
343
344 scaled_fp_t = Multiply(scaled_fp_t, exp_fixed_pt_multiplier_expr);
345 }
346
347 // 3) Add the output zero point.
348 auto shifted_fp_t = scaled_fp_t;
349 if (!IsEqualScalar(output_zero_point, zero_scalar)) {
350 shifted_fp_t = Add(shifted_fp_t, Cast(output_zero_point, DataType::Float(Bits)));
351 }
352
353 if (param->rounding == "UPWARD") {
354 shifted_fp_t = Upward<Bits>(shifted_fp_t);
355 } else /*if (param->rounding == "TONEAREST")*/ {
356 shifted_fp_t = Tonearest<Bits>(shifted_fp_t);
357 }
358
359 shifted_fp_t = Cast(shifted_fp_t, DataType::Int(32));
360 // 4) Clip to the out_dtype min/max. Skip clipping if out_dtype is Int32. The fixed point
361 // multiplication keeps the value in int32 range.
362 if (out_dtype == DataType::Int(32)) {
363 return shifted_fp_t;
364 }
365
366 auto q_min = GetQmin(out_dtype);
367 auto q_max = GetQmax(out_dtype);
368 auto clipped_t = Clip(shifted_fp_t, q_min, q_max);
369 return Cast(clipped_t, out_dtype);
370}

372// Lowering of qnn.requantize op
373/*
374 * \brief Lower requantize to a sequence of ops.
375 * \param input_tensor The input tensor to requantize op.
376 * \param param The requantize op attrs.
377 * \param input_shape The input tensor shape of the requantize op.
378 * \return The sequence of existing Relay ops.
379 */
380Expr RequantizeLower(const Expr& input_tensor, const Expr& input_scale,
381 const Expr& input_zero_point, const Expr& output_scale,
382 const Expr& output_zero_point, const RequantizeAttrs* param,
383 const Array<IndexExpr>& input_shape, const DataType& out_dtype) {
384 // Check rounding validity.
385 ICHECK(param->rounding == "UPWARD" || param->rounding == "TONEAREST")
386 << "QNN requantize supports two rounding modes - UPWARD and "
387 << "TONEAREST";
388 // Check compute_dtype validity.
389 ICHECK(param->compute_dtype == "int64" || param->compute_dtype == "float32" ||
390 param->compute_dtype == "float64")
391 << "QNN requantize supports three compute_dtype variants - \"int64\", \"float32\" and "
392 "\"float64\"";
393 if (param->compute_dtype == "float32") {
394 return RequantizeLowerFP<32>(input_tensor, input_scale, input_zero_point, output_scale,
395 output_zero_point, param, input_shape, out_dtype);
396 } else if (param->compute_dtype == "float64") {
397 return RequantizeLowerFP<64>(input_tensor, input_scale, input_zero_point, output_scale,
398 output_zero_point, param, input_shape, out_dtype);
399 } else /*if (param->compute_dtype == "int64") */ {
400 return RequantizeLowerInt(input_tensor, input_scale, input_zero_point, output_scale,
401 output_zero_point, param, input_shape, out_dtype);
402 }
403}

405/*
406 * \brief Forward rewrite the requantize op.
407 * \param ref_call The original call that will be lowered.
408 * \param new_args The new mutated args to the call node.
409 * \param ctx The node context.
410 * \return The sequence of Relay ops for requantize op.
411 * \note Lowering of the requantize operation. The requantize operator converts
412 * one quantized tensor to another quantized tensor. For the output
413 * tensor, we are provided with output scale and zero point. The
414 * computation looks like this
415 *
416 * Q_output = zp_output + (scale_input)/(scale_ouptut) * (Q_input - zp_input)
417 */
418Expr RequantizeQnnCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
419 const Array<tvm::relay::Type>& types) {
420 ICHECK_EQ(new_args.size(), 5);
421 auto& quantized_data = new_args[0];
422 auto& input_scale = new_args[1];
423 auto& input_zero_point = new_args[2];
424 auto& output_scale = new_args[3];
425 auto& output_zero_point = new_args[4];
426 const auto* param = attrs.as<RequantizeAttrs>();
427 const RequantizeConfig& cfg = RequantizeConfig::Current();
428
429 ICHECK(param != nullptr);
430
431 const_cast<RequantizeAttrs*>(param)->rounding =
432 SelectRequntizeParameter(param->rounding, cfg->get_rounding(), cfg->is_default, "rounding");
433 const_cast<RequantizeAttrs*>(param)->compute_dtype = SelectRequntizeParameter(
434 param->compute_dtype, cfg->get_compute_dtype(), cfg->is_default, "compute_dtype");
435
436 // Find input shape.
437 ICHECK_EQ(types.size(), 6);
438 auto in_type = types[0];
439 auto in_tensor_type = in_type.as<TensorTypeNode>();
440 ICHECK(in_tensor_type != nullptr) << "Type information missing."
441 << " Please run infer_type pass.";
442 Array<IndexExpr> input_shape = in_tensor_type->shape;
443
444 // Find the output dtype.
445 auto out_type = types[5];
446 auto out_tensor_type = out_type.as<TensorTypeNode>();
447 ICHECK(out_tensor_type != nullptr) << "Type information missing."
448 << " Please run infer_type pass.";
449 auto out_dtype = out_tensor_type->dtype;
450 return RequantizeLower(quantized_data, input_scale, input_zero_point, output_scale,
451 output_zero_point, param, input_shape, out_dtype);
452}

454/*
455 * \brief Infer shape function of Requantize op.
456 * \param types The types of input args.
457 * \param num_inputs The number of inputs.
458 * \param attrs The op attributes.
459 * \param reporter The type reporter that sets the dtype and shapes.
460 * \return True if the infer shape succeeded.
461 */
462bool RequantizeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
463 const TypeReporter& reporter) {
464 // Expected Types: data, input_scale, input_zero_point, output_scale, output_zero_point, output
465 ICHECK_EQ(types.size(), 6);
466 const auto* data = types[0].as<TensorTypeNode>();
467
468 if (data == nullptr) {
469 return false;
470 }
471
472 // Check the scale and zero point types
473 for (size_t i = 3; i < 5; ++i) {
474 if (types[i].as<IncompleteTypeNode>()) {
475 return false;
476 }
477 }
478 const auto in_dtype = data->dtype;
479 ICHECK(in_dtype == DataType::Int(8) || in_dtype == DataType::UInt(8) ||
480 in_dtype == DataType::Int(32) || in_dtype == DataType::Int(64))
481 << "Input type should be one of [int8, uint8, int32, int64] but was " << in_dtype;
482
483 const RequantizeAttrs* requantize_attrs = attrs.as<RequantizeAttrs>();
484 int axis = requantize_attrs->axis;
485 auto rank = static_cast<int>(data->shape.size());
486 axis = (axis < 0) ? ((rank > 0) ? data->shape.size() + axis : 0) : axis;
487 ICHECK_LT(axis, rank > 0 ? rank : 1) << "axis " << requantize_attrs->axis << " is out of range";
488 ICHECK_GE(axis, 0) << "axis " << requantize_attrs->axis << " is out of range";
489
490 PrimExpr axis_shape;
491 if (rank > 0) {
492 axis_shape = data->shape[axis];
493 } else {
494 axis_shape = Integer(1);
495 }
496 // Check and assign types for scale and zero points.
497 AssignType(types[1], DataType::Float(32), axis_shape, reporter); // input_scale
498 AssignType(types[2], DataType::Int(32), axis_shape, reporter); // input_zero_pt
499 // For now, requantize output tensor is limited to full tensor uniform quantization.
500 ICHECK(IsScalarType(types[3], DataType::Float(32))); // output_scale
501 ICHECK(IsScalarType(types[4], DataType::Int(32))); // output_zero_point
502
503 const Array<tvm::PrimExpr> oshape = data->shape;
504 // assign output type
505 auto out_dtype = requantize_attrs->out_dtype;
506 ICHECK(out_dtype == DataType::Int(8) || out_dtype == DataType::UInt(8) ||
507 out_dtype == DataType::Int(16) || out_dtype == DataType::Int(32))
508 << "Output type should be one of [int8, uint8, int16, int32] but was " << out_dtype;
509 reporter->Assign(types[5], TensorType(oshape, out_dtype));
510 return true;
511}

513// Positional relay function to create qnn requantize operator
514// used by frontend FFI.
515Expr MakeRequantize(Expr data, Expr input_scale, Expr input_zero_point, Expr output_scale,
516 Expr output_zero_point, int axis, String rounding, String compute_dtype,
517 DataType out_dtype) {
518 auto attrs = make_object<RequantizeAttrs>();
519 attrs->axis = axis;
520 attrs->rounding = std::move(rounding);
521 attrs->out_dtype = std::move(out_dtype);
522 attrs->compute_dtype = std::move(compute_dtype);
523 static const Op& op = Op::Get("qnn.requantize");
524 return Call(op, {data, input_scale, input_zero_point, output_scale, output_zero_point},
525 Attrs(attrs), {});
526}

528RELAY_REGISTER_OP("qnn.requantize")
529 .describe(R"code(Requantize operator.
530The requantize operator converts one quantized tensor to another quantized
531tensor. For the output tensor, we are provided with output scale and zero
532point. The computation looks like this
533
534Q_output = zp_output + (scale_input)/(scale_output) * (Q_input - zp_input)
535
536)code" TVM_ADD_FILELINE)
