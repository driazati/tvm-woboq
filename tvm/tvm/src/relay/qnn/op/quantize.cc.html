39bool QuantizeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
40 const TypeReporter& reporter) {
41 ICHECK_EQ(types.size(), 4);
42 const auto* data = types[0].as<TensorTypeNode>();
43
44 if (data == nullptr) {
45 return false;
46 }
47
48 const auto input_dtype = data->dtype;
49 ICHECK(input_dtype == DataType::Float(32))
50 << "Input type should be one of float32 but was " << input_dtype;
51
52 const auto* quantize_attrs = attrs.as<QuantizeAttrs>();
53 int axis = quantize_attrs->axis;
54 auto rank = static_cast<int>(data->shape.size());
55 axis = (axis < 0) ? ((rank > 0) ? data->shape.size() + axis : 0) : axis;
56
57 // If zero point and scale are scalar then axis doesnt matter.
58 bool scale_is_scalar, zp_is_scalar;
59
60 if (auto ttype = types[1].as<TensorTypeNode>()) {
61 scale_is_scalar = ttype->shape.size() == 0;
62 } else {
63 ICHECK(types[1].as<IncompleteTypeNode>())
64 << "Quantize: expect to be TensorType but get " << types[1];
65 return false;
66 }
67
68 if (auto ttype = types[2].as<TensorTypeNode>()) {
69 zp_is_scalar = ttype->shape.size() == 0;
70 } else {
71 ICHECK(types[2].as<IncompleteTypeNode>())
72 << "Quantize: expect to be TensorType but get " << types[2];
73 return false;
74 }
75
76 if (!(scale_is_scalar && zp_is_scalar)) {
77 ICHECK_LT(axis, rank > 0 ? rank : 1) << "axis " << quantize_attrs->axis << " is out of range";
78 ICHECK_GE(axis, 0) << "axis " << quantize_attrs->axis << " is out of range";
79 }
80
81 PrimExpr axis_shape;
82 if (rank > 0) {
83 axis_shape = data->shape[axis];
84 } else {
85 axis_shape = Integer(1);
86 }
87 // Check and assign types for scale and zero points.
88 AssignType(types[1], DataType::Float(32), axis_shape, reporter); // scale
89 AssignType(types[2], DataType::Int(32), axis_shape, reporter); // zero point
90
91 const Array<tvm::PrimExpr> oshape = data->shape;
92 const DataType out_dtype = quantize_attrs->out_dtype;
93 ICHECK(out_dtype == DataType::Int(8) || out_dtype == DataType::UInt(8) ||
94 out_dtype == DataType::Int(16) || out_dtype == DataType::Int(32))
95 << "Output type should be one of [int8, unit8, int16, int32] but was " << out_dtype;
96 // assign output type
97 reporter->Assign(types[3], TensorType(oshape, out_dtype));
98 return true;
99}

101Expr MakeQuantize(Expr data, Expr output_scale, Expr output_zero_point, int axis,
102 DataType out_dtype) {
103 auto attrs = make_object<QuantizeAttrs>();
104 attrs->axis = axis;
105 attrs->out_dtype = std::move(out_dtype);
106 // result_quantized_value = result_zero_point + result_real_value / result_scale.
107 // A more detailed explanation can be found here -
108 // https://github.com/google/gemmlowp/blob/master/doc/quantization.md
109 static const Op& op = Op::Get("qnn.quantize");
110 return Call(op, {data, output_scale, output_zero_point}, Attrs(attrs), {});
111}
112
113Expr QuantizeLower(const Expr& input_tensor, const Expr& output_scale,
114 const Expr& output_zero_point, const Array<tvm::relay::Type>& types,
115 const QuantizeAttrs* attrs) {
116 ICHECK_EQ(types.size(), 4);
117 auto in_type = types[0];
118 auto in_tensor_type = in_type.as<TensorTypeNode>();
119 ICHECK(in_tensor_type != nullptr) << "Type information missing."
120 << " Please run infer_type pass.";
121 Array<IndexExpr> input_shape = in_tensor_type->shape;
122
123 const auto out_dtype = attrs->out_dtype;
124 auto axis = attrs->axis;
125
126 size_t n_dim = input_shape.size();
127
128 // Wrap axis from negative to positive if needed.
129 if (axis < 0) {
130 axis = static_cast<int>(n_dim) + axis;
131 }
132
133 auto expanded_output_scale = output_scale;
134 if (!IsConstScalar(output_scale) && !IsScalarType(types[1])) {
135 expanded_output_scale = ExpandBiasToMatchAxis(output_scale, n_dim, {axis});
136 }
137
138 auto expanded_output_zero_point = output_zero_point;
139 if (!IsConstScalar(output_zero_point) && !IsScalarType(types[2])) {
140 expanded_output_zero_point = ExpandBiasToMatchAxis(output_zero_point, n_dim, {axis});
141 }
142
143 const int32_t min_val = GetQmin(out_dtype);
144 const int32_t max_val = GetQmax(out_dtype);
145 auto scale_data = Round(Divide(input_tensor, expanded_output_scale));
146 auto add_zero_point = Add(scale_data, Cast(expanded_output_zero_point, DataType::Float(32)));
147 auto clamped_output = Clip(add_zero_point, min_val, max_val);
148 return Cast(clamped_output, out_dtype);
149}
150
151Expr QuantizeQnnCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
152 const Array<tvm::relay::Type>& types) {
153 ICHECK_EQ(new_args.size(), 3);
154 auto& data = new_args[0];
155 auto& output_scale = new_args[1];
156 auto& output_zero_point = new_args[2];
157 const auto* quantize_attrs = attrs.as<QuantizeAttrs>();
158 ICHECK(quantize_attrs != nullptr);
159
160 return QuantizeLower(data, output_scale, output_zero_point, types, quantize_attrs);
161}

163RELAY_REGISTER_OP("qnn.quantize")
164 .describe(R"code(Quantizes the input and produces quantized output.
165The input can be either float or quantized(int8, unit8). If the input is float,
166this op takes scale and zero point and quantize the float value to
167quantized output, in int8 or uint8 format. If the input is quantized value,
168the op requantize the input (of a certain type, with a given scale and zero
169point) to the output of the same or different type with a same or different
170scale and zero point.
171- **data**: Tensor of any shape to quantize. The input data can be of floating point
172 or quantized.
173)code" TVM_ADD_FILELINE)
174 .set_attrs_type<QuantizeAttrs>()
175 .set_num_inputs(3)
176 .add_argument("data", "Tensor", "The tensor to quantize.")
177 .add_argument("output_scale", "Tensor", "The quantization scale of the output tensor.")
178 .add_argument("output_zero_point", "Tensor",
179 "The quantization zero_point of the output tensor.")
180 .set_support_level(11)
181 .add_type_rel("Quantize", QuantizeRel)
182 .set_attr<TNonComputational>("TNonComputational", true)
183 .set_attr<FTVMLegalize>("FTVMQnnCanonicalize", QuantizeQnnCanonicalize);
