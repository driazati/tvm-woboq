20/*!
21 * \file src/relay/qnn/op/simulated_quantize.cc
22 * \brief QNN simulated quantize operator. Mimics the behavior
23 * of QNN quantize in floating point with added flexibility.
24 */
25
26#include <tvm/relay/analysis.h>
27#include <tvm/relay/op_attr_types.h>
28#include <tvm/relay/qnn/attrs.h>
29
30#include "../../transforms/pattern_utils.h"
31#include "../utils.h"
32
33namespace tvm {
34namespace relay {
35namespace qnn {
36
37TVM_REGISTER_NODE_TYPE(SimulatedQuantizeAttrs);
38
39bool SimulatedQuantizeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
40 const TypeReporter& reporter) {
41 // types = [data_type, datatype_type, scale_type, zp_type, ret_type]
42 ICHECK_EQ(types.size(), 5);
43 const auto* data = types[0].as<TensorTypeNode>();
44 const auto* dtype = types[1].as<TensorTypeNode>();
45
46 if ((data == nullptr) || (dtype == nullptr)) {
47 return false;
48 }
49
50 // assign output type
51 reporter->Assign(types[4], TensorType(data->shape, data->dtype));
52 return true;
53}
54
55Expr MakeSimulatedQuantize(Expr data, Expr out_dtype, Expr output_scale, Expr output_zero_point,
56 int axis) {
57 auto attrs = make_object<SimulatedQuantizeAttrs>();
58 attrs->axis = axis;
59 static const Op& op = Op::Get("qnn.simulated_quantize");
60 return Call(op, {data, out_dtype, output_scale, output_zero_point}, Attrs(attrs), {});
61}
62
63RELAY_REGISTER_OP("qnn.simulated_quantize")
64 .describe(R"code(Simulates the functionality of qnn.quantize but allows more flexible
65 dynamic input type conversion and always outputs float values.
66)code" TVM_ADD_FILELINE)
67 .set_attrs_type<SimulatedQuantizeAttrs>()
68 .set_num_inputs(4)
69 .add_argument("data", "Tensor", "The tensor to quantize.")
70 .add_argument("out_dtype", "Tensor",
71 "A code corresponding to the type of quantization to apply.")
72 .add_argument("output_scale", "Tensor", "The quantization scale of the output tensor.")
73 .add_argument("output_zero_point", "Tensor",
74 "The quantization zero_point of the output tensor.")
75 .set_support_level(11)
76 .add_type_rel("QNNSimulatedQuantize", SimulatedQuantizeRel);
77
78TVM_REGISTER_GLOBAL("relay.qnn.op._make.simulated_quantize").set_body_typed(MakeSimulatedQuantize);
