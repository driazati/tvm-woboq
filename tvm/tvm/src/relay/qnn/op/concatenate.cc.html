39bool QnnConcatenateRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
40 const TypeReporter& reporter) {
41 // Expected Types: data, input_scales, input_zero_points, output_scale, output_zero_point,
42 // out_type
43 ICHECK_EQ(types.size(), 6);
44
45 if (types[0].as<IncompleteTypeNode>()) {
46 return false;
47 }
48 // Check the scale and zero point types
49 const auto* input_scales_tuple = types[1].as<TupleTypeNode>();
50 if (input_scales_tuple == nullptr) {
51 if (types[1].as<IncompleteTypeNode>()) {
52 return false;
53 } else {
54 throw CompileError(
55 ErrorBuilder()
56 << "qnn concatenate requires a tuple of scales as the second argument, found "
57 << PrettyPrint(types[1]));
58 }
59 }
60 for (const auto& input_scale : input_scales_tuple->fields) {
61 if (input_scale.as<IncompleteTypeNode>()) {
62 return false;
63 }
64 ICHECK(IsScalarType(input_scale, DataType::Float(32))); // input_scales[idx]
65 }
66
67 const auto* input_zero_points_tuple = types[2].as<TupleTypeNode>();
68 if (input_zero_points_tuple == nullptr) {
69 if (types[2].as<IncompleteTypeNode>()) {
70 return false;
71 } else {
72 throw CompileError(
73 ErrorBuilder()
74 << "qnn concatenate requires a tuple of zero_points as the third argument, found "
75 << PrettyPrint(types[2]));
76 }
77 }
78 for (const auto& input_zero_point : input_zero_points_tuple->fields) {
79 if (input_zero_point.as<IncompleteTypeNode>()) {
80 return false;
81 }
82 ICHECK(IsScalarType(input_zero_point, DataType::Int(32))); // input_zero_points[idx]
83 }
84
85 for (size_t i = 3; i < 5; ++i) {
86 if (types[i].as<IncompleteTypeNode>()) {
87 return false;
88 }
89 }
90 ICHECK(IsScalarType(types[3], DataType::Float(32))); // output_scale
91 ICHECK(IsScalarType(types[4], DataType::Int(32))); // output_zero_point
92
93 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
94 // Concatenate infer type function.
95 Array<Type> tensor_types = {types[0], types[5]};
96 return ConcatenateRel<ConcatenateAttrs>(tensor_types, 2, attrs, reporter);
97}

99InferCorrectLayoutOutput QnnConcatenateLayout(const Attrs& attrs,
100 const Array<Layout>& new_in_layouts,
101 const Array<Layout>& old_in_layouts,
102 const Array<tvm::relay::Type>& old_in_types) {
103 // Collect the layouts and types to reuse Relay Concatenate Infer Correct Layout.
104 ICHECK_EQ(old_in_types.size(), 5);
105 auto input_tuple_type = old_in_types[0].as<TupleTypeNode>();
106 ICHECK(input_tuple_type);
107 auto num_input_tensors = input_tuple_type->fields.size();
108
109 Array<Layout> relay_new_in_layouts(nullptr);
110 if (new_in_layouts.defined()) {
111 relay_new_in_layouts =
112 Array<Layout>(new_in_layouts.begin(), new_in_layouts.begin() + num_input_tensors);
113 }
114 Array<Layout> relay_old_in_layouts(nullptr);
115 if (old_in_layouts.defined()) {
116 relay_old_in_layouts =
117 Array<Layout>(old_in_layouts.begin(), old_in_layouts.begin() + num_input_tensors);
118 }
119
120 // Use Relay Concatenate Infer Correct layout to infer the layouts for data tensors.
121 auto concat_new_layout =
122 ConcatenateLayout(attrs, relay_new_in_layouts, relay_old_in_layouts, {old_in_types[0]});
123
124 // Fill the layouts of remaining input tensors - scales and zero points. The layouts of these
125 // tensors can be treated as channel layout. Total number of these tensors are 2 * num of data
126 // tensors (scale and zero point for each input data tensor) + 2 for the output data tensor.
127 Layout channel_layout = Layout("C");
128 Array<Layout> input_layouts = concat_new_layout->input_layouts;
129
130 for (size_t i = 0; i < 2 * num_input_tensors + 2; i++) {
131 input_layouts.push_back(channel_layout);
132 }
133 Array<Layout> output_layouts = concat_new_layout->output_layouts;
134 return InferCorrectLayoutOutput(input_layouts, output_layouts, concat_new_layout->new_attrs);
135}

137Expr MakeQnnConcatenate(Expr data, Expr input_scales, Expr input_zero_points, Expr output_scale,
138 Expr output_zero_point, int axis) {
139 auto attrs = make_object<ConcatenateAttrs>();
140 attrs->axis = axis;
141 static const Op& op = Op::Get("qnn.concatenate");
142 return Call(op, {data, input_scales, input_zero_points, output_scale, output_zero_point},
143 Attrs(attrs), {});
144}

146/*
147 * \brief Canonicalizes the QNN concatenate op.
148 * \param attrs The QNN concatenate attrs.
149 * \param new_args The new mutated args to the call node.
150 * \param arg_types The types of input and output.
151 * \return The sequence of Relay ops for concatenate op.
152 */
153Expr ConcatenateQnnCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
154 const Array<tvm::relay::Type>& arg_types) {
155 // Get the attrs.
156 ICHECK_EQ(new_args.size(), 5);
157 auto& data = new_args[0];
158 auto& input_scales = new_args[1];
159 auto& input_zero_points = new_args[2];
160 auto& output_scale = new_args[3];
161 auto& output_zero_point = new_args[4];
162 const auto* concatenate_attrs = attrs.as<ConcatenateAttrs>();
163 ICHECK(concatenate_attrs != nullptr);
164
165 // Get the input dtype and shape.
166 ICHECK_GE(arg_types.size(), 1);
167 auto tuple_type = arg_types[0].as<TupleTypeNode>();
168 ICHECK(tuple_type != nullptr);
169
170 // FIXME (anijain2305) - The lowering can be further optimized. Instead of inserting requantize in
171 // the start, we can insert requantize at the end if and only if all the input tensors have same
172 // qnn params. This can be done in future.
173
174 // If the output qnn params do not match the input qnn params, we can call requantize on the input
175 // expr first, followed by a concatenate on the requantized input exprs.
176
177 Array<Expr> tuple_exprs;
178 if (data->IsInstance<TupleNode>()) {
179 tuple_exprs = data.as<TupleNode>()->fields;
180 } else if (data->IsInstance<CallNode>()) { // if the data is a CallNode, use TupleGetItems
181 auto call = Downcast<Call>(data);
182 for (size_t i = 0; i < tuple_type->fields.size(); i++) {
183 tuple_exprs.push_back(TupleGetItem(call, i));
184 }
185 }
186 ICHECK(!tuple_exprs.empty());
187
188 auto tuple_input_scales = input_scales.as<TupleNode>();
189 ICHECK(tuple_input_scales != nullptr);
190
191 auto tuple_input_zero_points = input_zero_points.as<TupleNode>();
192 ICHECK(tuple_input_zero_points != nullptr);
193
194 int idx = 0;
195 Array<Expr> requantized_exprs;
196 for (auto quantized_expr : tuple_exprs) {
197 // Get the input scale for the idx quantized input tensor.
198 auto input_scale = tuple_input_scales->fields[idx];
199
200 // Get the zero point for the idx quantized input tensor.
201 auto input_zero_point = tuple_input_zero_points->fields[idx];
202
203 // Check if output and input qnn params are same. If not, requantize.
204 if (!IsEqualScalar(input_scale, output_scale) ||
205 !IsEqualScalar(input_zero_point, output_zero_point)) {
206 // Get the input shape and dtype.
207 auto tensor_type = tuple_type->fields[idx].as<TensorTypeNode>();
208 auto input_dtype = tensor_type->dtype;
209 auto input_shape = tensor_type->shape;
210
211 // Requantize the input.
212 auto requantized_expr = Requantize(quantized_expr, input_shape, input_scale, input_zero_point,
213 output_scale, output_zero_point, input_dtype);
214 requantized_exprs.push_back(requantized_expr);
215 } else {
216 requantized_exprs.push_back(quantized_expr);
217 }
218 idx++;
219 }
220 return MakeConcatenate(Tuple(requantized_exprs), concatenate_attrs->axis);
221}

223RELAY_REGISTER_OP("qnn.concatenate")
224 .describe(R"code(Concatenate the quantized input tensors along the given axis.
225)code" TVM_ADD_FILELINE)
226 .set_attrs_type<ConcatenateAttrs>()
227 .set_num_inputs(5)
228 .add_argument("data", "Tensor", "The tensor to concatenate.")
229 .add_argument("input_scales", "Tensor", "The quantization scales of the input tensors.")
230 .add_argument("input_zero_points", "Tensor",
231 "The quantization zero_points of the input tensors.")
232 .add_argument("output_scale", "Tensor", "The quantization scale of the output tensor.")
233 .add_argument("output_zero_point", "Tensor",
234 "The quantization zero_point of the output tensor.")
235 .set_support_level(11)
236 .add_type_rel("QnnConcatenate", QnnConcatenateRel)
237 .set_attr<TNonComputational>("TNonComputational", true)
238 .set_attr<FTVMLegalize>("FTVMQnnCanonicalize", ConcatenateQnnCanonicalize)
239 .set_attr<FInferCorrectLayout>("FInferCorrectLayout", QnnConcatenateLayout);
240
241TVM_REGISTER_GLOBAL("relay.qnn.op._make.concatenate").set_body_typed(MakeQnnConcatenate);
