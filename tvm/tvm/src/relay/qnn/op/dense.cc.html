40bool QnnDenseRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
41 const TypeReporter& reporter) {
42 // Expected Types: data, weight, input_zero_point, weight_zero_point, input_scale, weight_scale,
43 // out_type
44 ICHECK_EQ(types.size(), 7);
45 const auto* data = types[0].as<TensorTypeNode>();
46 const auto* weight = types[1].as<TensorTypeNode>();
47 if (data == nullptr || weight == nullptr) return false;
48 const auto* param = attrs.as<DenseAttrs>();
49 ICHECK(param != nullptr) << "DenseAttrs cannot be nullptr.";
50 ICHECK(data->dtype == DataType::Int(8) || data->dtype == DataType::UInt(8) ||
51 data->dtype == DataType::Int(16) || data->dtype == DataType::UInt(16))
52 << "Expected quantized dense type(int8, uint8, int16, uint16) for input but was "
53 << data->dtype;
54 ICHECK(weight->dtype == DataType::Int(8) || weight->dtype == DataType::UInt(8))
55 << "Expected quantized dense type(int8, uint8) for weight but was " << weight->dtype;
56 ICHECK(param->out_dtype == DataType::Int(32) || param->out_dtype == DataType::Int(64))
57 << "Expected quantized dense type(int32, int64) for output but was " << param->out_dtype;
58
59 // Check the types of scale and zero points.
60 for (size_t i = 2; i < 5; ++i) {
61 if (types[i].as<IncompleteTypeNode>()) {
62 return false;
63 }
64 }
65 ICHECK(IsScalarType(types[2], DataType::Int(32))); // input_zero_point
66 ICHECK(IsScalarType(types[4], DataType::Float(32))); // input_scale
67 // weight_zero_point can be a scalar or a vector of the same shape as the weight_scale
68 AssignType(types[5], DataType::Float(32), param->units, reporter); // weight_scale
69
70 ICHECK(param->out_dtype.bits() > 0) << "Output dtype bits should be greater than 0.";
71
72 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
73 // Dense infer type function.
74 Array<Type> tensor_types = {types[0], types[1], types[6]};
75 return MatmulRel<DenseAttrs>(tensor_types, 3, attrs, reporter);
76}

78// Positional relay function to create quantized dense operator used by frontend FFI.
79Expr MakeQuantizedDense(Expr data, Expr weight, Expr input_zero_point, Expr kernel_zero_point,
80 Expr input_scale, Expr kernel_scale, IndexExpr units, DataType out_dtype) {
81 auto attrs = make_object<DenseAttrs>();
82 attrs->units = std::move(units);
83 attrs->out_dtype = out_dtype;
84 static const Op& op = Op::Get("qnn.dense");
85 return Call(op, {data, weight, input_zero_point, kernel_zero_point, input_scale, kernel_scale},
86 Attrs(attrs), {});
87}
88
89Expr DenseFirstTerm(const Expr& quantized_data, const Expr& quantized_kernel,
90 const DenseAttrs* attrs) {
91 return Dense(quantized_data, quantized_kernel, attrs->units, attrs->out_dtype);
92}
93
94Expr DenseSecondTerm(const Expr& quantized_data, const Expr& kernel_zero_point,
95 const int out_dim_size) {
96 Array<Integer> axes = {1};
97 Expr reduced_t2 = Sum(Cast(quantized_data, DataType::Int(32)), axes, true, false);
98 Expr multiplied_t2;
99 if (!IsConstScalar(kernel_zero_point)) {
100 multiplied_t2 = Multiply(kernel_zero_point, MakeRepeat(reduced_t2, out_dim_size, 1));
101 } else {
102 multiplied_t2 = Multiply(kernel_zero_point, reduced_t2);
103 }
104 return multiplied_t2;
105}
106
107Expr DenseThirdTerm(const Expr& quantized_kernel, const Expr& input_zero_point) {
108 Array<Integer> axes = {1};
109 return Multiply(input_zero_point,
110 Sum(Cast(quantized_kernel, DataType::Int(32)), axes, false, false));
111}
112
113Expr DenseFourthTerm(int input_zero_point_int, int kernel_zero_point_int, int reduction_dim_size) {
114 int32_t scalar_term = input_zero_point_int * kernel_zero_point_int * reduction_dim_size;
115 return MakeConstantScalar(DataType::Int(32), scalar_term);
116}
117
118Expr DenseFourthTerm(const Expr& input_zero_point, const Expr& kernel_zero_point,
119 int reduction_dim_size) {
120 auto reduction_dim = MakeConstantScalar(DataType::Int(32), reduction_dim_size);
121 return Multiply(Multiply(input_zero_point, kernel_zero_point), reduction_dim);
122}
123
124Expr DenseCombineTerms(const Expr& term1, const Expr& term2, const Expr& term3, const Expr& term4) {
125 auto data_term = Subtract(term1, term2);
126 // Putting constant terms together, so that constant folding can fold it.
127 auto const_term = Subtract(term4, term3);
128 return Add(data_term, const_term);
129}
130/*
131 * \brief Forward rewrite the qnn dense op.
132 * \param attrs The QNN dense attrs.
133 * \param new_args The new mutated args to the call node.
134 * \param arg_types The types of input and output.
135 * \return The sequence of Relay ops for qnn cov2d op.
136 * \note Lowering of the qnn.dense operator
137 * A quantized tensor is represented in following manner
138 * A = scale_a x (QA - zp_A)
139 * where QA is quantized tensor, scale_a and zp_A are quantization
140 * params.
141 *
142 * Quantized dense multiplies two quantized tensors and returns a
143 * quantized tensor of default dtype of int32, with scale equaling to the
144 * product of scales of input tensors, and a zero point of zero.
145 *
146 * The lowering for asymmetric quantized dense looks as follows. More details at
147 * https://discuss.tvm.ai/t/tf-lite-quantized-conv2d-operator-conversion/2651/8
148 * The computation gets unrolled into following 4 terms
149 * C(m, n) = Sigma(k) (A(m, k) * W(n, k))
150 *
151 * RHS becomes
152 * Sigma(k) ([QA(m, k) - zp_a] * [QW(n, k) - zp_w])
153 *
154 * Unrolling leads to following sequence
155 * Sigma(k) QA(m, k) * QW(n, k) // Term1
156 * - Sigma(k) zp_w * QA(m, k) // Term2
157 * - Sigma(k) zp_a * QW(n, k) // Term3
158 * - Sigma(k) * zp_a * zp_w // Term4
159 *
160 * Term3 and Term4 can be computed at compile time.
161 */
162Expr QnnDenseCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
163 const Array<tvm::relay::Type>& arg_types) {
164 ICHECK_EQ(new_args.size(), 6);
165 Expr quantized_data = new_args[0];
166 Expr quantized_kernel = new_args[1];
167 Expr input_zero_point = new_args[2];
168 Expr kernel_zero_point = new_args[3];
169
170 const auto in_shape = get_shape(arg_types[0]);
171 const auto w_shape = get_shape(arg_types[1]);
172 const int reduction_dim_size = get_const_int(in_shape[1]);
173 const int out_dim_size = get_const_int(w_shape[0]);
174
175 const auto* qnn_dense_attrs = attrs.as<DenseAttrs>();
176
177 auto term1 = DenseFirstTerm(quantized_data, quantized_kernel, qnn_dense_attrs);
178 auto term2 = DenseSecondTerm(quantized_data, kernel_zero_point, out_dim_size);
179 auto term3 = DenseThirdTerm(quantized_kernel, input_zero_point);
180
181 // Extract the integer zero points.
182
183 if (!IsConstScalar(input_zero_point) || !IsConstScalar(kernel_zero_point)) {
184 auto term4 = DenseFourthTerm(input_zero_point, kernel_zero_point, reduction_dim_size);
185 return DenseCombineTerms(term1, term2, term3, term4);
186 }
187
188 auto kernel_zero_point_int = GetScalarFromConstant<int>(kernel_zero_point);
189 auto input_zero_point_int = GetScalarFromConstant<int>(input_zero_point);
190
191 // Get all the terms as described in the comments.
192 auto term4 = DenseFourthTerm(input_zero_point_int, kernel_zero_point_int, reduction_dim_size);
193
194 // Combine those 4 terms depending on the zero points to get the best lowering.
195 if (input_zero_point_int == 0 && kernel_zero_point_int == 0) {
196 // term 2, 3 and 4 become zero.
197 return term1;
198 } else if (input_zero_point_int == 0 && kernel_zero_point_int != 0) {
199 // term 3 and term 4 become zero.
200 return Subtract(term1, term2);
201 } else if (input_zero_point_int != 0 && kernel_zero_point_int == 0) {
202 // term 2 and term 4 become zero.
203 return Subtract(term1, term3);
204 } else {
205 return DenseCombineTerms(term1, term2, term3, term4);
206 }
207}

209RELAY_REGISTER_OP("qnn.dense")
210 .describe(R"code(Applies a linear transformation: :math:`Y = XW^T`.
211- **data**: quantized(int8, unit8) `(x1, x2, ..., xn, input_dim)`
212- **weight**: quantized(int8, unit8) `(units, input_dim)`
213- **out**: quantized(int32) `(x1, x2, ..., xn, units)`.
214)code" TVM_ADD_FILELINE)
215 .set_attrs_type<DenseAttrs>()
216 .set_num_inputs(6)
217 .add_argument("data", "quantized nD Tensor", "Input data.")
218 .add_argument("weight", "quantized 2D Tensor", "Weight matrix.")
219 .add_argument("input_scale", "Tensor", "The quantization scale of the input tensor.")
220 .add_argument("input_zero_point", "Tensor", "The quantization zero_point of the input tensor.")
221 .add_argument("weight_scale", "Tensor", "The quantization scale of the weight tensor.")
222 .add_argument("weight_zero_point", "Tensor",
223 "The quantization zero_point of the weight tensor.")
224 .set_support_level(11)
225 .add_type_rel("QDense", QnnDenseRel)
226 .set_attr<TNonComputational>("TNonComputational", true)
227 .set_attr<FTVMLegalize>("FTVMQnnCanonicalize", QnnDenseCanonicalize);
228
