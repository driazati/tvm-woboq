38namespace tvm {
39namespace relay {
40namespace qnn {
41
42class RequantizeConfig;
43/*!
44 * \brief Container for build configuration options
45 */
46class RequantizeConfigNode : public Object {
47 std::string rounding;
48 std::string compute_dtype;
49
50 public:
51 explicit RequantizeConfigNode(bool is_default = false) : is_default(is_default) {}
52
53 std::string get_rounding() const {
54 if (!rounding.empty()) return rounding;
55 return "UPWARD";
56 }
57
58 std::string get_compute_dtype() const {
59 if (!compute_dtype.empty()) return compute_dtype;
60
61 // For the x86 architecture, the float32 computation is expected to give significant speedup,
62 // with little loss in the accuracy of the requantize operation.
63 auto target = Target::Current(true);
64 auto target_has_sse41 = tvm::runtime::Registry::Get("tvm.topi.x86.utils.target_has_sse41");
65 ICHECK(target_has_sse41) << "Function tvm.topi.x86.utils.target_has_sse41 not found";
66 if (target.defined() && target->kind->name == "llvm" &&
67 (target->GetAttr<String>("mcpu") &&
68 (*target_has_sse41)(target->GetAttr<String>("mcpu").value()))) {
69 return "float32";
70 }
71
72 return "int64";
73 }
74
75 const bool is_default = false;
76
77 void VisitAttrs(AttrVisitor* v) {
78 v->Visit("rounding", &rounding);
79 v->Visit("compute_dtype", &compute_dtype);
80 }
81
82 static constexpr const char* _type_key = "relay.qnn.op.RequantizeConfig";
83 TVM_DECLARE_FINAL_OBJECT_INFO(RequantizeConfigNode, Object);
84};
85
86/*!
87 * \brief Container for build configuration options
88 */
89class RequantizeConfig : public ObjectRef {
90 public:
91 RequantizeConfig() {}
92 explicit RequantizeConfig(ObjectPtr<Object> n) : ObjectRef(n) {}
93
94 const RequantizeConfigNode* operator->() const {
95 return static_cast<const RequantizeConfigNode*>(get());
96 }
97
98 RequantizeConfigNode* operator->() { return static_cast<RequantizeConfigNode*>(get_mutable()); }
99
100 /*!
101 * \brief Push a new BuildConfig context onto the thread local stack.
102 * \param build_config The configuration to set as the current context.
103 */
104 static void EnterRequantizeConfigScope(const RequantizeConfig& requantize_config);
105
106 /*!
107 * \brief Pop a build config off the thread local context stack, restoring the previous
108 * configuration as the current context.
109 */
110 static void ExitRequantizeConfigScope();
111
112 /*!
113 * \brief Get the current BuildConfig context from thread local storage, or a default
114 * configuration if a BuildConfig scope has not been entered.
115 * \return The configuration that is the current context.
116 */
117 static RequantizeConfig& Current();
118
119 using ContainerType = RequantizeConfigNode;
120};
