33namespace tvm {
34namespace relay {
35namespace qnn {
36
37/*! \brief Entry to hold the BuildConfig context stack. */
38struct TVMRequantizeConfigThreadLocalEntry {
39 /*! \brief The default build config if the stack is empty */
40 RequantizeConfig default_config;
41
42 /*! \brief The current build config context */
43 std::stack<RequantizeConfig> context_stack;
44
45 TVMRequantizeConfigThreadLocalEntry() : default_config(make_object<RequantizeConfigNode>(true)) {}
46};
47
48/*! \brief Thread local store to hold the BuildConfig context stack. */
49typedef dmlc::ThreadLocalStore<TVMRequantizeConfigThreadLocalEntry>
50 TVMRequantizeConfigThreadLocalStore;
51
52void RequantizeConfig::EnterRequantizeConfigScope(const RequantizeConfig& build_config) {
53 TVMRequantizeConfigThreadLocalEntry* entry = TVMRequantizeConfigThreadLocalStore::Get();
54 entry->context_stack.push(build_config);
55}
56
57void RequantizeConfig::ExitRequantizeConfigScope() {
58 TVMRequantizeConfigThreadLocalEntry* entry = TVMRequantizeConfigThreadLocalStore::Get();
59 entry->context_stack.pop();
60}
61
62RequantizeConfig& RequantizeConfig::Current() {
63 TVMRequantizeConfigThreadLocalEntry* entry = TVMRequantizeConfigThreadLocalStore::Get();
64 if (entry->context_stack.size() > 0) {
65 return entry->context_stack.top();
66 }
67
68 return entry->default_config;
69}
70
71TVM_REGISTER_NODE_TYPE(RequantizeConfigNode);
72
73TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
74 .set_dispatch<RequantizeConfigNode>([](const ObjectRef& ref, ReprPrinter* p) {
75 auto* op = static_cast<const RequantizeConfigNode*>(ref.get());
76 p->stream << "requantize_config(";
77 p->stream << "rounding==" << op->get_rounding() << ", ";
78 p->stream << "compute_dtype==" << op->get_compute_dtype();
79 p->stream << ")";
80 });
81
82TVM_REGISTER_GLOBAL("relay._requantize._GetCurrentRequantizeConfig")
83 .set_body_typed([]() -> RequantizeConfig { return RequantizeConfig::Current(); });
84
85TVM_REGISTER_GLOBAL("relay._requantize._EnterRequantizeConfigScope")
86 .set_body_typed(RequantizeConfig::EnterRequantizeConfigScope);
87
88TVM_REGISTER_GLOBAL("relay._requantize._ExitRequantizeConfigScope")
89 .set_body_typed(RequantizeConfig::ExitRequantizeConfigScope);
