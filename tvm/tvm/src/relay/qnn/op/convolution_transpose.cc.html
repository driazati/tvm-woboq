42inline Expr MakeQnnConv2DTranspose(Expr data, Expr weight, Expr input_zero_point,
43 Expr kernel_zero_point, Expr input_scale, Expr kernel_scale,
44 Array<IndexExpr> strides, Array<IndexExpr> padding,
45 Array<IndexExpr> dilation, int groups, IndexExpr channels,
46 Array<IndexExpr> kernel_size, std::string data_layout,
47 std::string kernel_layout, std::string out_layout,
48 Array<IndexExpr> output_padding, DataType out_dtype) {
49 auto attrs = make_object<Conv2DTransposeAttrs>();
50 attrs->strides = std::move(strides);
51 attrs->padding = std::move(padding);
52 attrs->dilation = std::move(dilation);
53 attrs->groups = groups;
54 attrs->channels = std::move(channels);
55 attrs->kernel_size = std::move(kernel_size);
56 attrs->data_layout = std::move(data_layout);
57 attrs->kernel_layout = std::move(kernel_layout);
58 attrs->out_layout = std::move(out_layout);
59 attrs->output_padding = std::move(output_padding);
60 attrs->out_dtype = std::move(out_dtype);
61 const Op& op = Op::Get("qnn.conv2d_transpose");
62 return Call(op, {data, weight, input_zero_point, kernel_zero_point, input_scale, kernel_scale},
63 Attrs(attrs), {});
64}

66InferCorrectLayoutOutput QnnConvTransposeInferCorrectLayout(
67 const Attrs& attrs, const Array<Layout>& new_in_layouts, const Array<Layout>& old_in_layouts,
68 const Array<tvm::relay::Type>& old_in_types) {
69 // Use Relay Conv2D transpose Infer correct layout.
70 auto conv_transpose_new_layouts = ConvInferCorrectLayout<Conv2DTransposeAttrs>(
71 attrs, new_in_layouts, old_in_layouts, old_in_types);
72
73 // Fill the layouts of remaining input tensors - scales and zero points. The layouts of these
74 // tensors can be treated as channel layout.
75 Layout channel_layout = Layout("C");
76 Array<Layout> input_layouts = {conv_transpose_new_layouts->input_layouts[0],
77 conv_transpose_new_layouts->input_layouts[1],
78 channel_layout,
79 channel_layout,
80 channel_layout,
81 channel_layout};
82 Array<Layout> output_layouts = conv_transpose_new_layouts->output_layouts;
83 return InferCorrectLayoutOutput(input_layouts, output_layouts, attrs);
84}

86bool QnnConv2DTransposeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
87 const TypeReporter& reporter) {
88 // Expected Types: data, weight, input_zero_point, weight_zero_point, input_scale, weight_scale,
89 // out_type
90 ICHECK_EQ(types.size(), 7);
91 const auto* data = types[0].as<TensorTypeNode>();
92 const auto* weight = types[1].as<TensorTypeNode>();
93 if (data == nullptr || weight == nullptr) return false;
94 const auto* param = attrs.as<Conv2DTransposeAttrs>();
95 ICHECK(param != nullptr) << "Conv2DTransposeAttrs cannot be nullptr.";
96 ICHECK(data->dtype == DataType::Int(8) || data->dtype == DataType::UInt(8))
97 << "Expected qnn conv2d type(int8, uint8) for input but was " << data->dtype;
98 ICHECK(weight->dtype == DataType::Int(8) || weight->dtype == DataType::UInt(8))
99 << "Expected qnn conv2d type(int8, uint8) for weight but was " << weight->dtype;
100 ICHECK(param->out_dtype == DataType::Int(16) || param->out_dtype == DataType::Int(32))
101 << "Expected qnn conv2d type(int32, int16) for output but was " << param->out_dtype;
102 ICHECK(param->out_dtype.bits() > 0) << "Output dtype bits should be greater than 0.";
103
104 // Check the types of scale and zero points.
105 for (size_t i = 2; i < 5; ++i) {
106 if (types[i].as<IncompleteTypeNode>()) {
107 return false;
108 }
109 }
110
111 const auto* weight_zp_type = types[3].as<TensorTypeNode>();
112 ICHECK(weight_zp_type->dtype == DataType::Int(32)); // weight_zero_point
113
114 bool input_zp_is_scalar = (types[2].as<TensorTypeNode>())->shape.size() == 0 ||
115 get_const_int((types[2].as<TensorTypeNode>())->Size()) == 1;
116 bool input_scale_is_scalar = (types[4].as<TensorTypeNode>())->shape.size() == 0 ||
117 get_const_int((types[4].as<TensorTypeNode>())->Size()) == 1;
118
119 ICHECK(input_scale_is_scalar && input_zp_is_scalar)
120 << "Zero point or scale should be scalar or a vector with one element.";
121
122 // Assign types for input scale and zero point.
123 AssignType(types[2], DataType::Int(32), Integer(1), reporter); // input_zero_point
124 AssignType(types[4], DataType::Float(32), Integer(1), reporter); // input_scale
125
126 // Kernel scale can be a vector of length output_channels or a scalar.
127 if (param->groups == 1) {
128 size_t axis = param->kernel_layout.find('O');
129 ICHECK(axis != std::string::npos) << "Kernel layout attribute is not defined";
130 AssignType(types[5], DataType::Float(32), weight->shape[axis], reporter); // weight_scale
131 } else {
132 // Here, total number of output channels depend on depth multiplier.
133 size_t o_axis = param->kernel_layout.find('O');
134 size_t i_axis = param->kernel_layout.find('I');
135 ICHECK(o_axis != std::string::npos || i_axis != std::string::npos)
136 << "Kernel layout attribute is not defined";
137 AssignType(types[5], DataType::Float(32), weight->shape[i_axis] * weight->shape[o_axis],
138 reporter); // kernel scale
139 }
140
141 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
142 // Conv2D infer type function.
143 Array<Type> tensor_types = {types[0], types[1], types[6]};
144 return Conv2DTransposeRel(tensor_types, 3, attrs, reporter);
145}

147RELAY_REGISTER_OP("qnn.conv2d_transpose")
148 .describe(R"code(Quantized transposed 2D convolution layer (sometimes called Deconvolution).
149This operator deconvolves quantized weight with quantized data. The scale of the
150output quantized tensor is the product of the weight_scale and input_scale of
151the input quantized tensors. The zero point of the output quantized tensor is
1520. By default, the dtype of output is int32. Please also refer to Requantize
153operator to understand how to scale back the int32 output to (u)int8.
154- **data**: This depends on the `layout` parameter. Input is 4D array of shape
155 (batch_size, in_channels, height, width) if `layout` is `NCHW`.
156- **weight**: (channels, in_channels, kernel_size[0], kernel_size[1])
157- **out**: This depends on the `layout` parameter. Output is 4D array of shape
158 (batch_size, channels, out_height, out_width) if `layout` is `NCHW`.
159)code" TVM_ADD_FILELINE)
160 .set_attrs_type<Conv2DTransposeAttrs>()
161 .set_num_inputs(6)
162 .add_argument("data", "Tensor", "The quantized input data tensor.")
163 .add_argument("weight", "Tensor", "The quantized weight tensor.")
164 .add_argument("input_scale", "Tensor", "The quantization scale of the input tensor.")
165 .add_argument("input_zero_point", "Tensor", "The quantization zero_point of the input tensor.")
166 .add_argument("weight_scale", "Tensor", "The quantization scale of the weight tensor.")
167 .add_argument("weight_zero_point", "Tensor",
168 "The quantization zero_point of the weight tensor.")
169 .set_support_level(11)
170 .add_type_rel("QnnConv2DTranspose", QnnConv2DTransposeRel)
171 .set_attr<TNonComputational>("TNonComputational", true)
172 .set_attr<FInferCorrectLayout>("FInferCorrectLayout", QnnConvTransposeInferCorrectLayout);
173
174TVM_REGISTER_GLOBAL("relay.qnn.op._make.conv2d_transpose").set_body_typed(MakeQnnConv2DTranspose);
