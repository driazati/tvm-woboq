38// relay.op.qnn.batch_matmul
39
40bool QnnBatchMatmulRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
41 const TypeReporter& reporter) {
42 // Expected Types: x, y, x_zero_point, y_zero_point, x_scale, y_scale,
43 // out_type
44 ICHECK_EQ(types.size(), 7);
45 const auto* x = types[0].as<TensorTypeNode>();
46 const auto* y = types[1].as<TensorTypeNode>();
47 if (x == nullptr || y == nullptr) return false;
48 const auto* param = attrs.as<BatchMatmulAttrs>();
49 ICHECK(param != nullptr) << "BatchMatmulAttrs cannot be nullptr.";
50 ICHECK(x->dtype == DataType::Int(8) || x->dtype == DataType::UInt(8))
51 << "Expected quantized batch_matmul type(int8, uint8) for input but was " << x->dtype;
52 ICHECK(y->dtype == DataType::Int(8) || y->dtype == DataType::UInt(8))
53 << "Expected quantized batch_matmul type(int8, uint8) for weight but was " << y->dtype;
54 ICHECK(param->out_dtype == DataType::Int(32))
55 << "Expected quantized batch_matmul type(int32) for output but was " << param->out_dtype;
56
57 // Check the types of scale and zero points.
58 for (size_t i = 2; i < 5; ++i) {
59 if (types[i].as<IncompleteTypeNode>()) {
60 return false;
61 }
62 }
63 ICHECK(IsScalarType(types[2], DataType::Int(32))); // x_zero_point
64 ICHECK(IsScalarType(types[3], DataType::Int(32))); // y_zero_point
65 ICHECK(IsScalarType(types[4], DataType::Float(32))); // x_scale
66 ICHECK(IsScalarType(types[5], DataType::Float(32))); // y_scale
67
68 ICHECK(param->out_dtype.bits() > 0) << "Output dtype bits should be greater than 0.";
69
70 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
71 // BatchMatmul infer type function.
72 Array<Type> tensor_types = {types[0], types[1], types[6]};
73 return BatchMatmulRel<BatchMatmulAttrs>(tensor_types, 3, attrs, reporter);
74}

76// Positional relay function to create quantized batch_matmul operator used by frontend FFI.
77Expr MakeQuantizedBatchMatmul(Expr x, Expr y, Expr x_zero_point, Expr y_zero_point, Expr x_scale,
78 Expr y_scale, DataType out_dtype) {
79 auto attrs = make_object<BatchMatmulAttrs>();
80 attrs->out_dtype = out_dtype;
81 // For legacy reason, currently `qnn.batch_matmul` only supports
82 // (transpose_a=false, transpose_b=true)
83 // TODO(jcf94): extent to support all tensor format
84 attrs->transpose_a = false;
85 attrs->transpose_b = true;
86 static const Op& op = Op::Get("qnn.batch_matmul");
87 return Call(op, {x, y, x_zero_point, y_zero_point, x_scale, y_scale}, Attrs(attrs), {});
88}
89
90Expr BatchMatmulFirstTerm(const Expr& quantized_x, const Expr& quantized_y,
91 const BatchMatmulAttrs* attrs) {
92 ICHECK(attrs->transpose_a == false && attrs->transpose_b == true)
93 << "Currently qnn.batch_matmul only supports (transpose_a=false, transpose_b=true).";
94 return MakeBatchMatmul(quantized_x, quantized_y, attrs->out_dtype, attrs->transpose_a,
95 attrs->transpose_b);
96}
97
98Expr BatchMatmulSecondTerm(const Expr& x_quantized_data, const Expr& y_zero_point) {
99 Array<Integer> axes = {2};
100 return Multiply(y_zero_point, Sum(Cast(x_quantized_data, DataType::Int(32)), axes, true, false));
101}
102
103Expr BatchMatmulThirdTerm(const Expr& y_quantized_data, const Expr& x_zero_point,
104 int broadcast_dim_size) {
105 Array<Integer> axes = {2};
106 auto reducemult =
107 Multiply(x_zero_point, Sum(Cast(y_quantized_data, DataType::Int(32)), axes, true, false));
108 Array<Integer> newshape;
109
110 // dimension of 0 in reshape copies old dimension size
111 newshape = {0, 1, broadcast_dim_size};
112 return Reshape(reducemult, newshape);
113}
114
115Expr BatchMatmulFourthTerm(int x_zero_point_int, int y_zero_point_int, int reduction_dim_size) {
116 int32_t scalar_term = x_zero_point_int * y_zero_point_int * reduction_dim_size;
117 return MakeConstantScalar(DataType::Int(32), scalar_term);
118}
119
120Expr BatchMatmulCombineTerms(const Expr& term1, const Expr& term2, const Expr& term3,
121 const Expr& term4) {
122 auto data1_term = Subtract(term1, term2);
123 auto data2_term = Subtract(term4, term3);
124 return Add(data1_term, data2_term);
125}

127/*
128 * \brief Forward rewrite the qnn batch_matmul op.
129 * \param attrs The QNN batch_matmul attrs.
130 * \param new_args The new mutated args to the call node.
131 * \param arg_types The types of input and output.
132 * \return The sequence of Relay ops for qnn batch_matmul op.
133 * \note Lowering of the qnn.batch_matmul operator
134 * A quantized tensor is represented in following manner
135 * A = scale_a x (QA - zp_A)
136 * where QA is quantized tensor, scale_a and zp_A are quantization
137 * params.
138 *
139 * Quantized batch_matmul multiplies two quantized tensors and returns a
140 * quantized tensor of default dtype of int32, with scale equaling to the
141 * product of scales of input tensors, and a zero point of zero.
142 *
143 * The lowering for asymmetric quantized batch_matmul looks similar to
144 * quantized conv2d and dense and originally was discussed here:
145 * https://discuss.tvm.apache.org/t/tf-lite-quantized-conv2d-operator-conversion/2651/7
146 *
147 * The computation gets unrolled into following 4 terms
148 * C(m, n) = Sigma(k) (X(m, k) * Y(n, k))
149 *
150 * RHS becomes
151 * Sigma(k) ([QX(m, k) - zp_x] * [QY(n, k) - zp_y])
152 *
153 * Unrolling leads to following sequence
154 * Sigma(k) QX(m, k) * QX(n, k) // Term1
155 * - Sigma(k) zp_y * QX(m, k) // Term2
156 * - Sigma(k) zp_x * QY(n, k) // Term3
157 * - Sigma(k) * zp_x * zp_y // Term4
158 *
159 * Term4 can be computed at compile time, everything else depending on the
160 * input type.
161 */
162Expr QnnBatchMatmulCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
163 const Array<tvm::relay::Type>& arg_types) {
164 ICHECK_EQ(new_args.size(), 6);
165 Expr quantized_x = new_args[0];
166 Expr quantized_y = new_args[1];
167 Expr x_zero_point = new_args[2];
168 Expr y_zero_point = new_args[3];
169
170 const auto in_shape = get_shape(arg_types[0]);
171 const int reduction_dim_size = get_const_int(in_shape[2]);
172
173 const auto y_shape = get_shape(arg_types[1]);
174 const int broadcast_dim_size = get_const_int(y_shape[1]);
175
176 const auto* qnn_batch_matmul_attrs = attrs.as<BatchMatmulAttrs>();
177
178 // Extract the integer zero points.
179 auto y_zero_point_int = GetScalarFromConstant<int>(y_zero_point);
180 auto x_zero_point_int = GetScalarFromConstant<int>(x_zero_point);
181
182 // Get all the terms as described in the comments.
183 auto term1 = BatchMatmulFirstTerm(quantized_x, quantized_y, qnn_batch_matmul_attrs);
184 auto term2 = BatchMatmulSecondTerm(quantized_x, y_zero_point);
185 auto term3 = BatchMatmulThirdTerm(quantized_y, x_zero_point, broadcast_dim_size);
186 auto term4 = BatchMatmulFourthTerm(x_zero_point_int, y_zero_point_int, reduction_dim_size);
187
188 // Combine those 4 terms depending on the zero points to get the best lowering.
189 if (x_zero_point_int == 0 && y_zero_point_int == 0) {
190 // term 2, 3 and 4 become zero.
191 return term1;
192 } else if (x_zero_point_int == 0 && y_zero_point_int != 0) {
193 // term 3 and term 4 become zero.
194 return Subtract(term1, term2);
195 } else if (x_zero_point_int != 0 && y_zero_point_int == 0) {
196 // term 2 and term 4 become zero.
197 return Subtract(term1, term3);
198 } else {
199 return BatchMatmulCombineTerms(term1, term2, term3, term4);
200 }
201}

203RELAY_REGISTER_OP("qnn.batch_matmul")
204 .describe(R"code(Compute batch matrix multiplication of `tensor_a` and `tensor_b`.
205
206Note we expect tensor_b to be transposed to copy the standard nn.batch_matmul conventions.
207
208.. math::
209
210 batch\_matmul(A, B)[i, :, :] = matmul(A[i, :, :], B[i, :, :]^T)
211
212- **data**: quantized(int8, unit8) `(i, m, k)`
213- **weight**: quantized(int8, unit8) `(i, n, k)`
214- **out**: quantized(int32) `(i, m, n)`.
215
216)code" TVM_ADD_FILELINE)
217 .set_attrs_type<BatchMatmulAttrs>()
218 .set_num_inputs(6)
219 .add_argument("x", "quantized 2D Tensor", "First input data.")
220 .add_argument("y", "quantized 2D Tensor", "Second input data.")
221 .add_argument("x_scale", "Tensor", "The quantization scale of the x input tensor.")
222 .add_argument("x_zero_point", "Tensor", "The quantization zero_point of the x input tensor.")
223 .add_argument("y_scale", "Tensor", "The quantization scale of the y input tensor.")
224 .add_argument("y_zero_point", "Tensor", "The quantization zero_point of the y input tensor.")
225 .set_support_level(11)
226 .add_type_rel("QBatchMatmul", QnnBatchMatmulRel)
227 .set_attr<TNonComputational>("TNonComputational", true)
228 .set_attr<FTVMLegalize>("FTVMQnnCanonicalize", QnnBatchMatmulCanonicalize);
