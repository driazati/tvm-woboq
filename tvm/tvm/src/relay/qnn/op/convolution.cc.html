43bool QnnConv2DRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
44 const TypeReporter& reporter) {
45 // Expected Types: data, weight, input_zero_point, weight_zero_point, input_scale, weight_scale,
46 // out_type
47 ICHECK_EQ(types.size(), 7);
48 const auto* data = types[0].as<TensorTypeNode>();
49 const auto* weight = types[1].as<TensorTypeNode>();
50 if (data == nullptr || weight == nullptr) return false;
51 const auto* param = attrs.as<Conv2DAttrs>();
52 ICHECK(param != nullptr) << "Conv2DAttrs cannot be nullptr.";
53 ICHECK(data->dtype == DataType::Int(8) || data->dtype == DataType::UInt(8) ||
54 data->dtype == DataType::Int(16))
55 << "Expected qnn conv2d type(int8, uint8, int16) for input but was " << data->dtype;
56 ICHECK(weight->dtype == DataType::Int(8) || weight->dtype == DataType::UInt(8))
57 << "Expected qnn conv2d type(int8, uint8) for weight but was " << weight->dtype;
58 ICHECK(param->out_dtype == DataType::Int(16) || param->out_dtype == DataType::Int(32) ||
59 param->out_dtype == DataType::Int(64))
60 << "Expected qnn conv2d type(int16, int32, int64) for output but was " << param->out_dtype;
61 ICHECK(param->out_dtype.bits() > 0) << "Output dtype bits should be greater than 0.";
62
63 // Check the types of scale and zero points.
64 for (size_t i = 2; i < 5; ++i) {
65 if (types[i].as<IncompleteTypeNode>()) {
66 return false;
67 }
68 }
69 ICHECK(IsScalarType(types[2], DataType::Int(32))); // input_zero_point
70 ICHECK(IsScalarType(types[4], DataType::Float(32))); // input_scale
71 // Kernel scale can be a vector of length output_channels or a scalar.
72 if (param->groups == 1) {
73 size_t axis = param->kernel_layout.operator std::string().find('O');
74 ICHECK(axis != std::string::npos) << "Kernel layout attribute is not defined";
75 AssignType(types[5], DataType::Float(32), weight->shape[axis], reporter); // weight_scale
76 } else {
77 // Here, total number of output channels depend on depth multiplier.
78 size_t o_axis = param->kernel_layout.operator std::string().find('O');
79 size_t i_axis = param->kernel_layout.operator std::string().find('I');
80 ICHECK(o_axis != std::string::npos || i_axis != std::string::npos)
81 << "Kernel layout attribute is not defined";
82 AssignType(types[5], DataType::Float(32), weight->shape[i_axis] * weight->shape[o_axis],
83 reporter); // weight_scale
84 }
85
86 // Collect the input tensor and output tensor devoid of scale and zero points to reuse Relay
87 // Conv2D infer type function.
88 Array<Type> tensor_types = {types[0], types[1], types[6]};
89 return Conv2DRel(tensor_types, 3, attrs, reporter);
90}

92InferCorrectLayoutOutput QnnConvInferCorrectLayout(const Attrs& attrs,
93 const Array<Layout>& new_in_layouts,
94 const Array<Layout>& old_in_layouts,
95 const Array<tvm::relay::Type>& old_in_types) {
96 // Use Relay Conv2D Infer correct layout.
97 auto conv_new_layouts =
98 ConvInferCorrectLayout<Conv2DAttrs>(attrs, new_in_layouts, old_in_layouts, old_in_types);
99
100 // Fill the layouts of remaining input tensors - scales and zero points. The layouts of these
101 // tensors can be treated as channel layout.
102 Layout channel_layout = Layout("C");
103 Array<Layout> input_layouts = {conv_new_layouts->input_layouts[0],
104 conv_new_layouts->input_layouts[1],
105 channel_layout,
106 channel_layout,
107 channel_layout,
108 channel_layout};
109 Array<Layout> output_layouts = conv_new_layouts->output_layouts;
110 return InferCorrectLayoutOutput(input_layouts, output_layouts, attrs);
111}

113bool is_depthwise(const Conv2DAttrs* param) {
114 return param->channels.defined() && tvm::tir::ExprDeepEqual()(param->channels, param->groups) &&
115 param->groups != 1;
116}

118// Workload - batch_size, in_channels, out_channels, kernel_h, kernel_w, channel_multiplier
119using WorkloadType = std::tuple<int, int, int, int, int, int>;
120
121/*
122 * \brief Get the conv parameters like batch_size, kernel_height etc.
123 * \param ref_call The original callnode.
124 * \param param The qnn conv2d attributes.
125 * \return A tuple of workload.
126 */
127WorkloadType GetWorkload(const Array<tvm::relay::Type>& arg_types, const Conv2DAttrs* param) {
128 // Get conv parameters.
129 const auto in_shape = get_shape(arg_types[0]);
130 int batch_size, in_channels;
131 if (param->data_layout == "NCHW") {
132 batch_size = get_const_int(in_shape[0]);
133 in_channels = get_const_int(in_shape[1]);
134 } else if (param->data_layout == "NHWC") {
135 batch_size = get_const_int(in_shape[0]);
136 in_channels = get_const_int(in_shape[3]);
137 } else {
138 LOG(FATAL) << "qnn.conv2d does not support " << param->data_layout << " layout";
139 }
140
141 const auto kernel_shape = get_shape(arg_types[1]);
142 int out_channels, kernel_h, kernel_w;
143 int channel_multiplier = -1;
144 bool depthwise = is_depthwise(param);
145 int index_k = 0;
146 int index_h = 2;
147 int index_w = 3;
148 int index_c = 1;
149 if (param->kernel_layout == "HWIO") {
150 index_k = 3;
151 index_h = 0;
152 index_w = 1;
153 index_c = 2;
154 } else if (param->kernel_layout == "HWOI") {
155 index_k = 2;
156 index_h = 0;
157 index_w = 1;
158 index_c = 3;
159 } else if (param->kernel_layout == "OHWI") {
160 index_k = 0;
161 index_h = 1;
162 index_w = 2;
163 index_c = 3;
164 } else if (param->kernel_layout != "OIHW") {
165 LOG(FATAL) << "qnn.conv2d does not support " << param->kernel_layout << " layout";
166 }
167
168 kernel_h = get_const_int(kernel_shape[index_h]);
169 kernel_w = get_const_int(kernel_shape[index_w]);
170 out_channels = get_const_int(kernel_shape[index_k]);
171
172 if (depthwise) {
173 channel_multiplier = get_const_int(kernel_shape[index_c]);
174 }
175
176 return std::make_tuple(batch_size, in_channels, out_channels, kernel_h, kernel_w,
177 channel_multiplier);
178}

180/*
181 * \brief Fallback to simpler lowering for dilation (when non-zero kernel point) or grouped conv.
182 * \param data The input expr.
183 * \param weight The weight expr.
184 * \param input_zero_point The input zero point expr.
185 * \param kernel_zero_point The kernel zero point expr.
186 * \param param The qnn conv2d attributes.
187 * \return The fallback lowered sequence of Relay expr.
188 * \note In case of dilation with non-zero kernel zero point, normal lowering would require a
189 * dilated pool. Since, we don't have dilated pool, we fallback to a simpler sequence of Relay
190 * operations. This will potentially lead to performance degradation as the convolution is called on
191 * int32 tensors instead of int8 tensors.
192 */
193Expr Conv2DFallBack(const Expr& data, const Expr& weight, const Expr& input_zero_point,
194 const Expr& kernel_zero_point, const Conv2DAttrs* param) {
195 // Upcast the parameters to be at least int32 to avoid overflow
196 auto upcast_bits = param->out_dtype.bits() < 32 ? 32 : param->out_dtype.bits();
197
198 auto zp_data = Cast(input_zero_point, DataType::Int(upcast_bits));
199 auto zp_kernel = Cast(kernel_zero_point, DataType::Int(upcast_bits));
200
201 auto shifted_data = Cast(data, DataType::Int(upcast_bits));
202 auto zero_scalar = MakeConstantScalar(DataType::Int(upcast_bits), 0);
203 if (!IsEqualScalar(input_zero_point, zero_scalar)) {
204 shifted_data = Subtract(Cast(data, DataType::Int(upcast_bits)), zp_data);
205 }
206
207 auto shifted_kernel = Cast(weight, DataType::Int(upcast_bits));
208 if (!IsEqualScalar(kernel_zero_point, zero_scalar)) {
209 shifted_kernel = Subtract(Cast(weight, DataType::Int(upcast_bits)), zp_kernel);
210 }
211
212 return Conv2D(shifted_data, shifted_kernel, param->strides, param->padding, param->dilation,
213 param->groups, param->channels, param->kernel_size, param->data_layout,
214 param->kernel_layout, param->out_layout, param->out_dtype);
215}

217/*
218 * \brief Pad the input data.
219 * \param data The input expr.
220 * \param input_zero_point The input zero point expr.
221 * \return The padded input expr.
222 * \note For quantized convolution, the input has to be padded with zero point
223 * instead of zero. This might lead to performance degradation as pad
224 * cannot be fused with conv in Relay. In case we see performance
225 * degradation, we can change the conv2D API to accept a pad_const value.
226 */
227Expr Conv2DPadInput(const Expr& data, const Expr& input_zero_point, const Conv2DAttrs* param) {
228 // 1) Pad the input data
229 auto padded_data = data;
230 auto pad_top_value = get_const_int(param->padding[0]);
231 auto pad_left_value = get_const_int(param->padding[1]);
232 auto pad_bottom_value = get_const_int(param->padding[2]);
233 auto pad_right_value = get_const_int(param->padding[3]);
234 bool do_pad =
235 pad_top_value != 0 || pad_left_value != 0 || pad_bottom_value != 0 || pad_right_value != 0;
236 if (do_pad) {
237 Array<IndexExpr> pad_n({0, 0});
238 Array<IndexExpr> pad_c({0, 0});
239 Array<IndexExpr> pad_h({param->padding[0], param->padding[2]});
240 Array<IndexExpr> pad_w({param->padding[1], param->padding[3]});
241
242 Array<Array<IndexExpr>> pad_width;
243 if (param->data_layout == "NCHW") {
244 pad_width = {pad_n, pad_c, pad_h, pad_w};
245 } else if (param->data_layout == "NHWC") {
246 pad_width = {pad_n, pad_h, pad_w, pad_c};
247 } else {
248 LOG(FATAL) << "qnn.conv2d does not support " << param->data_layout << " layout";
249 }
250 padded_data = Pad(data, pad_width, input_zero_point, "constant");
251 }
252 return padded_data;
253}

255/*
256 * \brief Calculates the second term in the qnn.conv2d depthwise lowering sequence.
257 * \param padded_data The padded data expr.
258 * \param kernel_zero_point The kernel zero point expr.
259 * \param param The qnn conv2d attributes.
260 * \param kernel_h The height of kernel.
261 * \param kernel_w The width of kernel.
262 * \param channel_multiplier The channel/depth multiplier.
263 * \return The sequence of Relay operators for term2.
264 * \note The term2 looks like this
265 *
266 * Sigma(r, s) zp_w * Qa(n, oc/cm, oh + r, ow + s)
267 *
268 * Second term is not directly representable by one Relay operator.
269 * However, deeper analysis shows that we can reduce r,s using avg_pool2d,
270 * followed by repeat on the C axis by cm times.
271 */
272Expr DepthwiseConv2DSecondTerm(const Expr& padded_data, const Expr& kernel_zero_point,
273 const Conv2DAttrs* param, int kernel_h, int kernel_w,
274 int channel_multiplier) {
275 auto casted_t2 = Cast(padded_data, DataType::Int(32));
276
277 // We can reduce the H and W axis by using avg_pool2d. However, avg_pool2d averages the sum.
278 // Since, this is integer division (floor), we can first multiply the data by the pool_size and
279 // then perform avg_pool2d. Reversing this causes inaccuracy due to floor division. If the
280 // pool_size and strides are 1x1, we don't need avg_pool2d.
281 auto reduced_t2 = casted_t2;
282 if (kernel_h * kernel_w != 1) {
283 auto scaled_hw_t2 =
284 Multiply(casted_t2, MakeConstantScalar(DataType::Int(32), kernel_h * kernel_w));
285 Array<IndexExpr> padding({0, 0});
286 reduced_t2 = AvgPool2D(scaled_hw_t2, param->kernel_size, param->strides, param->dilation,
287 padding, param->data_layout,
288 "", // out_layout
289 false, // ceil_mode
290 false); // count_include_pad
291 } else {
292 int stride1 = get_const_int(param->strides[0]);
293 int stride2 = get_const_int(param->strides[1]);
294 if (stride1 * stride2 != 1) {
295 Array<IndexExpr> padding({0, 0});
296 reduced_t2 = AvgPool2D(reduced_t2, param->kernel_size, param->strides, param->dilation,
297 padding, param->data_layout,
298 "", // out_layout
299 false, // ceil_mode
300 false); // count_include_pad
301 }
302 }
303
304 auto multiplied_t2 = reduced_t2;
305 auto one_scalar = MakeConstantScalar(DataType::Int(32), 1);
306 if (!IsEqualScalar(kernel_zero_point, one_scalar)) {
307 if (!IsConstScalar(kernel_zero_point)) {
308 multiplied_t2 = Multiply(MakeRepeat(kernel_zero_point, channel_multiplier, 0), reduced_t2);
309 } else {
310 multiplied_t2 = Multiply(kernel_zero_point, reduced_t2);
311 }
312 }
313
314 // Reduce the C dimension. Find the dimension.
315 int axis_t2 = 0;
316 if (param->data_layout == "NCHW") {
317 axis_t2 = 1;
318 } else if (param->data_layout == "NHWC") {
319 axis_t2 = 3;
320 } else {
321 LOG(FATAL) << "qnn.conv2d does not support " << param->data_layout << " layout";
322 }
323 auto repeated_t2 = multiplied_t2;
324 if (channel_multiplier != 1) {
325 repeated_t2 = MakeRepeat(multiplied_t2, channel_multiplier, axis_t2);
326 }
327 return repeated_t2;
328}

330/*
331 * \brief Calculates the third term in the qnn.conv2d depthwise lowering sequence.
332 * \param weight The weight expr.
333 * \param input_zero_point The input zero point expr.
334 * \param param The qnn conv2d attributes.
335 * \param out_channels The number of output channels.
336 * \param channel_multiplier The channel/depth multiplier.
337 * \return The sequence of Relay operatos for term3.
338 * \note The term3 looks like this
339 *
340 * Sigma(r, s) zp_a * Qw(oc/m, oc%m, r, s)
341 *
342 * This can be achieved by calling reduce on r and s axis. The tensor can be then reshaped to
343 * (1, oc, 1, 1) as (oc/m, oc%m) are just contiguous memory locations.
344 */
345Expr DepthwiseConv2DThirdTerm(const Expr& weight, const Expr& input_zero_point,
346 const Conv2DAttrs* param, int out_channels, int channel_multiplier) {
347 // Find which dimensions are R, S.
348 Array<Integer> axes_t3;
349 if (param->kernel_layout == "OIHW") {
350 // For OIHW kernel layout, HW are reduce axis
351 axes_t3 = {2, 3};
352 } else if (param->kernel_layout == "HWIO") {
353 axes_t3 = {0, 1};
354 } else if (param->kernel_layout == "HWOI") {
355 axes_t3 = {0, 1};
356 } else {
357 LOG(FATAL) << "qnn.conv2d does not support " << param->kernel_layout << " layout";
358 }
359 auto reduced_t3 = Sum(Cast(weight, DataType::Int(32)), axes_t3, false, false);
360
361 // Find the newshape depending on NCHW/NHWC layout.
362 Array<Integer> newshape;
363 if (param->data_layout == "NCHW") {
364 newshape = {1, out_channels * channel_multiplier, 1, 1};
365 } else if (param->data_layout == "NHWC") {
366 newshape = {1, 1, 1, out_channels * channel_multiplier};
367 } else {
368 LOG(FATAL) << "qnn.conv2d does not support " << param->data_layout << " layout";
369 }
370 auto reshaped_t3 = Reshape(reduced_t3, newshape);
371
372 auto one_scalar = MakeConstantScalar(DataType::Int(32), 1);
373 if (IsEqualScalar(input_zero_point, one_scalar)) {
374 return reshaped_t3;
375 }
376 return Multiply(input_zero_point, reshaped_t3);
377}

379/*
380 * \brief Calculates the fourth term in the qnn.conv2d depthwise lowering sequence.
381 * \param input_zero_point_int The int value of input zero point.
382 * \param kernel_zero_point_int The int value of kernel zero point.
383 * \param kernel_h The height of kernel.
384 * \param kernel_w The width of kernel.
385 * \return The sequence of Relay operators for term4.
386 * \note The term4 looks like this
387 *
388 * Sigma(r, s) zp_a * zp_w
389 */
390Expr DepthwiseConv2DFourthTerm(int input_zero_point_int, int kernel_zero_point_int, int kernel_h,
391 int kernel_w) {
392 int scalar_term4 = input_zero_point_int * kernel_zero_point_int * kernel_h * kernel_w;
393 return MakeConstantScalar(DataType::Int(32), scalar_term4);
394}

396/*
397 * \brief Calculates the fourth term in the qnn.conv2d depthwise lowering sequence
398 for non-constant zero_points.
399 * \param input_zero_point The Expr for the input zero point.
400 * \param kernel_zero_point The Expr for the kernel zero point.
401 * \param kernel_h The height of kernel.
402 * \param kernel_w The width of kernel.
403 * \return The sequence of Relay operators for term4.
404 * \note The term4 looks like this
405 *
406 * Sigma(r, s) zp_a * zp_w
407 */
408Expr DepthwiseConv2DFourthTerm(const Expr& input_zero_point, const Expr& kernel_zero_point,
409 int kernel_h, int kernel_w) {
410 Expr scalar_term4 = MakeConstantScalar(DataType::Int(32), kernel_h * kernel_w);
411 Expr variable_term4 = Multiply(input_zero_point, kernel_zero_point);
412 return Multiply(scalar_term4, variable_term4);
413}

415/*
416 * \brief Calculates the first term in the qnn.conv2d lowering sequence.
417 * \param data The input expr.
418 * \param weight The weight expr.
419 * \param param The qnn conv2d attributes.
420 * \return The sequence of Relay operators for term1.
421 * \note The term1 is
422 * Sigma(c,r,s) QW(k, c, r, s) * QA(n, c, h + r, w + s)

425Expr Conv2DFirstTerm(const Expr& padded_data, const Expr& weight, const Conv2DAttrs* param) {
426 // Lowering for Term 1
427 Array<IndexExpr> padding({0, 0, 0, 0});
428 return Conv2D(padded_data, weight, param->strides, padding, param->dilation, param->groups,
429 param->channels, param->kernel_size, param->data_layout, param->kernel_layout,
430 param->out_layout, param->out_dtype);
431}

433/*
434 * \brief Calculates the second term in the qnn.conv2d lowering sequence.
435 * \param padded_data The padded data expr.
436 * \param kernel_zero_point The kernel zero point expr.
437 * \param param The qnn conv2d attributes.
438 * \param kernel_h The height of kernel.
439 * \param kernel_w The width of kernel.
440 * \return The sequence of Relay operators for term2.
441 * \note The term2 looks like this
442 *
443 * Sigma(c,r,s) zp_w * QA(n, c, h + r, w + s)
444 *
445 * Second term is not directly representable by one Relay operator.
446 * However, deeper analysis shows that we can reduce r,s using avg_pool2d,
447 * followed by a reduce on the C axis. Using avg_pool2d also gives an
448 * opportunity to reuse alter_op_layout infrastructure.
449 */
450Expr Conv2DSecondTerm(const Expr& padded_data, const Expr& kernel_zero_point,
451 const Conv2DAttrs* param, int kernel_h, int kernel_w, int out_channels) {
452 auto casted_t2 = Cast(padded_data, DataType::Int(32));
453
454 // We can reduce the H and W axis by using avg_pool2d. However, avg_pool2d averages the sum.
455 // Since, this is integer division (floor), we can first multiply the data by the pool_size and
456 // then perform avg_pool2d. Reversing this causes inaccuracy due to floor division.
457 Array<IndexExpr> padding({0, 0});
458
459 // Reduce the C dimension. Find the dimension.
460 Array<Integer> axes_t2;
461 if (param->data_layout == "NCHW") {
462 axes_t2 = {1};
463 } else if (param->data_layout == "NHWC") {
464 axes_t2 = {3};
465 } else {
466 LOG(FATAL) << "qnn.conv2d does not support " << param->data_layout << " layout";
467 }
468 // Keep dims true to retain 4D tensor
469 auto reduced_c_t2 = Sum(casted_t2, axes_t2, true, false);
470
471 // If the pool_size and strides are 1x1, we don't need avg_pool2d.
472 auto reduced_t2 = reduced_c_t2;
473 if (kernel_h * kernel_w != 1) {
474 reduced_c_t2 =
475 Multiply(reduced_c_t2, MakeConstantScalar(DataType::Int(32), kernel_h * kernel_w));
476 reduced_t2 = AvgPool2D(reduced_c_t2, param->kernel_size, param->strides, param->dilation,
477 padding, param->data_layout,
478 "", // out_layout
479 false, // ceil_mode
480 false); // count_include_pad
481 } else {
482 int stride1 = get_const_int(param->strides[0]);
483 int stride2 = get_const_int(param->strides[1]);
484 if (stride1 * stride2 != 1) {
485 reduced_t2 = AvgPool2D(reduced_c_t2, param->kernel_size, param->strides, param->dilation,
486 padding, param->data_layout,
487 "", // out_layout
488 false, // ceil_mode
489 false); // count_include_pad
490 }
491 }
492
493 auto multiplied_t2 = reduced_t2;
494 auto one_scalar = MakeConstantScalar(DataType::Int(32), 1);
495 if (!IsEqualScalar(kernel_zero_point, one_scalar)) {
496 if (!IsConstScalar(kernel_zero_point)) {
497 Layout layout(param->data_layout);
498 int channel_axis = layout.IndexOf(LayoutAxis::Get('C'));
499 reduced_t2 = MakeRepeat(reduced_t2, out_channels, channel_axis);
500 }
501 multiplied_t2 = Multiply(kernel_zero_point, reduced_t2);
502 }
503 return multiplied_t2;
504}

506/*
507 * \brief Calculates the third term in the qnn.conv2d lowering sequence.
508 * \param weight The weight expr.
509 * \param input_zero_point The input zero point expr.
510 * \param param The qnn conv2d attributes.
511 * \param out_channels The number of output channels.
512 * \return The sequence of Relay operators for term3.
513 * \note The term3 looks like this
514 *
515 * Sigma(c,r,s) zp_a * QW(k, c, r, s)
516 *
517 * This can be achieved by calling reduce on c, r and s axis, resulting in
518 * a 1D tensor. The tensor is then reshaped to conform to NHWC/NCHW
519 * format.
520 */
521Expr Conv2DThirdTerm(const Expr& weight, const Expr& input_zero_point, const Conv2DAttrs* param,
522 int out_channels) {
523 // Find which dimensions are C, R, S.
524 Array<Integer> axes_t3;
525 if (param->kernel_layout == "OIHW") {
526 // For OIHW kernel layout, IHW are reduce axis
527 axes_t3 = {1, 2, 3};
528 } else if (param->kernel_layout == "HWIO") {
529 axes_t3 = {0, 1, 2};
530 } else if (param->kernel_layout == "HWOI") {
531 axes_t3 = {0, 1, 3};
532 } else if (param->kernel_layout == "OHWI") {
533 axes_t3 = {1, 2, 3};
534 } else {
535 LOG(FATAL) << "qnn.conv2d does not support " << param->kernel_layout << " layout";
536 }
537 auto reduced_t3 = Sum(Cast(weight, DataType::Int(32)), axes_t3, false, false);
538
539 // Find the newshape depending on NCHW/NHWC layout.
540 Array<Integer> newshape;
541 if (param->data_layout == "NCHW") {
542 newshape = {1, out_channels, 1, 1};
543 } else if (param->data_layout == "NHWC") {
544 newshape = {1, 1, 1, out_channels};
545 } else {
546 LOG(FATAL) << "qnn.conv2d does not support " << param->data_layout << " layout";
547 }
548 auto reshaped_t3 = Reshape(reduced_t3, newshape);
549
550 auto one_scalar = MakeConstantScalar(DataType::Int(32), 1);
551 if (IsEqualScalar(input_zero_point, one_scalar)) {
552 return reshaped_t3;
553 }
554 return Multiply(input_zero_point, reshaped_t3);
555}

557/*
558 * \brief Calculates the fourth term in the qnn.conv2d lowering sequence.
559 * \param input_zero_point_int The int value of input zero point.
560 * \param kernel_zero_point_int The int value of kernel zero point.
561 * \param in_channels The number of input channels.
562 * \param kernel_h The height of kernel.
563 * \param kernel_w The width of kernel.
564 * \param param The qnn conv2d attributes.
565 * \return The sequence of Relay operators for term4.
566 * \note The term4 looks like this
567 *
568 * Sigma(c,r,s) zp_a * zp_w
569 *
570 */
571Expr Conv2DFourthTerm(int input_zero_point_int, int kernel_zero_point_int, int in_channels,
572 int kernel_h, int kernel_w, const Conv2DAttrs* param) {
573 auto upcast_bits = param->out_dtype.bits() < 32 ? 32 : param->out_dtype.bits();
574 int scalar_term4 =
575 input_zero_point_int * kernel_zero_point_int * in_channels * kernel_h * kernel_w;
576 return MakeConstantScalar(DataType::Int(upcast_bits), scalar_term4);
577}

579/*
580 * \brief Calculates the fourth term in the qnn.conv2d lowering sequence
581 for non-constant zero_points.
582 * \param input_zero_point The Expr for the input zero point.
583 * \param kernel_zero_point The Expr for the kernel zero point.
584 * \param in_channels The number of input channels.
585 * \param kernel_h The height of kernel.
586 * \param kernel_w The width of kernel.
587 * \param param The qnn conv2d attributes.
588 * \return The sequence of Relay operators for term4.
589 * \note The term4 looks like this
590 *
591 * Sigma(c,r,s) zp_a * zp_w
592 *
593 */
594Expr Conv2DFourthTerm(const Expr& input_zero_point, const Expr& kernel_zero_point, int in_channels,
595 int kernel_h, int kernel_w, const Conv2DAttrs* param) {
596 auto upcast_bits = param->out_dtype.bits() < 32 ? 32 : param->out_dtype.bits();
597 Expr scalar_term4 =
598 MakeConstantScalar(DataType::Int(upcast_bits), in_channels * kernel_h * kernel_w);
599 Expr variable_term4 = Multiply(input_zero_point, kernel_zero_point);
600 return Multiply(scalar_term4, variable_term4);
601}

603/*
604 * \brief Combines different terms of qnn conv2d lowering.
605 * \param term1 The term1 of qnn conv2d lowering.
606 * \param term2 The term2 of qnn conv2d lowering.
607 * \param term3 The term3 of qnn conv2d lowering.
608 * \param term4 The term4 of qnn conv2d lowering.
609 * \param input_zero_point_int The int value of input zero point.
610 * \param kernel_zero_point_int The int value of kernel zero point.
611 * \param param The qnn conv2d attributes.
612 * \return The combined sequence of relay operations.
613 * \note The combined operation looks like this
614 *
615 * Sigma(c,r,s) QW(k, c, r, s) * QA(n, c, h + r, w + s) // Term1
616 * - Sigma(c,r,s) zp_w * QA(n, c, h + r, w + s) // Term2
617 * - Sigma(c,r,s) zp_a * QW(k, c, r, s) // Term3
618 * + Sigma(c,r,s) zp_a * zp_w // Term4
619 *
620 */
621Expr Conv2DCombineTerms(const Expr& term1, const Expr& term2, const Expr& term3, const Expr& term4,
622 int input_zero_point_int, int kernel_zero_point_int) {
623 if (input_zero_point_int == 0 && kernel_zero_point_int == 0) {
624 // term 2, 3 and 4 become zero.
625 return term1;
626 } else if (input_zero_point_int == 0 && kernel_zero_point_int != 0) {
627 // term 3 and term 4 become zero.
628 return Subtract(term1, term2);
629 } else if (input_zero_point_int != 0 && kernel_zero_point_int == 0) {
630 // term 2 and term 4 become zero.
631 return Subtract(term1, term3);
632 } else {
633 auto data_term = Subtract(term1, term2);
634 // Putting constant terms together, so that constant folding can fold it.
635 auto const_term = Subtract(term4, term3);
636 return Add(data_term, const_term);
637 }
638}

640/*
641 * \brief Forward rewrite the qnn conv2d op.
642 * \param attrs The QNN conv2d attrs.
643 * \param new_args The new mutated args to the call node.
644 * \param arg_types The types of input and output.
645 * \return The sequence of Relay ops for qnn cov2d op.
646 * \node Lowering of the qnn.conv2d operator
647 * A quantized tensor is represented in following manner
648 * A = scale_a x (QA - zp_A)
649 * where QA is quantized tensor, scale_a and zp_A are quantization
650 * params.
651 *
652 * Quantized convolution will convolve two quantized tensors and returns a
653 * quantized tensor of default dtype of int32, with scale equaling to the
654 * product of scales of input tensors, and a zero point of zero.
655 *
656 * For symmetric quantization, the zp_* for all tensors is 0. So, the
657 * lowering of qnn.conv2d is
658 *
659 * QA(n, ic, oh + r, ow + s) (conv) QW(oc, ic, r, s)
660 *
661 * For asymmetric computation, we can perform similar unrolling. We can
662 * find more details at
663 * https://discuss.tvm.ai/t/tf-lite-quantized-conv2d-operator-conversion/2651/8?u=janimesh
664 * The computation gets unrolled into following 4 terms
665 *
666 * Sigma(c,r,s) QW(k, c, r, s) * QA(n, c, h + r, w + s) // Term1
667 * - Sigma(c,r,s) zp_w * QA(n, c, h + r, w + s) // Term2
668 * - Sigma(c,r,s) zp_a * QW(k, c, r, s) // Term3
669 * + Sigma(c,r,s) zp_a * zp_w // Term4
670 *
671 * Term3 and Term4 can be computed at compile time.
672 *
673 * Key points to notice:
674 * 1) Padding is done explicitly because the input has to be padded with
675 * zero point. This might leave some performance opportunity at the
676 * table. Can be avoided by modifying conv2d API to accept the
677 * pad_const_value.
678 * 2) Second term is not directly representable by one Relay operator.
679 * However, deeper analysis shows that we can reduce r,s using
680 * avg_pool2d, followed by a reduce on the C axis. Using avg_pool2d also
681 * gives an opportunity to reuse alter_op_layout infrastructure.
682 * 3) For dilated conv, in current lowering, we need dilated pool. So as
683 * a workaround, we fall back to simpler lowering using int32 conv if
684 * the conv is dilated. We fallback also in case of grouped conv.
685 *
686 * For depthwise, we can similarly unroll the computation. The initial compute is as follows
687 * where cm = channel_multiplier
688 *
689 * Qc(n, oc, oh, ow) = Sigma(r, s) (Qw(oc/m, oc%/m, r, s) - zp_w)
690 * * (Qa(n, oc/cm, oh + r, ow + s) - zp_a)
691 *
692 * This can be written as
693 *
694 * Sigma(r, s) Qw(oc/m, oc%/m, r, s) * Qa(n, oc/cm, oh + r, ow + s)
695 * - Sigma(r, s) zp_w * Qa(n, oc/cm, oh + r, ow + s)
696 * - Sigma(r, s) zp_a * Qw(oc/m, oc%m, r, s)
697 * - Sigma(r, s) zp_a * zp_w
698 *
699 * The whole process can be broken down into following steps
700 * * Assertion checks for existing support, fallback if necessary
701 * * Pad the input.
702 * * Get Term1.
703 * * Get Term2.
704 * * Get Term3.
705 * * Get Term4.
706 * * Combine the terms.
707 */
708Expr QnnConv2DCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
709 const Array<tvm::relay::Type>& arg_types) {
710 ICHECK_EQ(new_args.size(), 6);
711 Expr data = new_args[0];
712 Expr weight = new_args[1];
713 Expr input_zero_point = new_args[2];
714 Expr kernel_zero_point = new_args[3];
715 const auto* param = attrs.as<Conv2DAttrs>();
716 ICHECK(param != nullptr);
717 // Assertion checks for existing support.
718 ICHECK(param->data_layout == "NCHW" || param->data_layout == "NHWC")
719 << "qnn.conv2d supports only NCHW/NHWC input data layout.";
720 ICHECK(param->kernel_layout == "OIHW" || param->kernel_layout == "HWIO" ||
721 param->kernel_layout == "HWOI" || param->kernel_layout == "OHWI")
722 << "qnn.conv2d supports only OIHW/HWIO/HWOI/OHWI kernel data layout.";
723 ICHECK(param->kernel_size.defined()) << "qnn.conv2d requires kernel size to be specified.";
724
725 int batch_size, in_channels, out_channels, kernel_h, kernel_w, channel_multiplier;
726 std::tie(batch_size, in_channels, out_channels, kernel_h, kernel_w, channel_multiplier) =
727 GetWorkload(arg_types, param);
728
729 // zero points are allowed to be non-scalar. Let's check if that's the case.
730 bool dynamic_zp = false;
731 // Use -1 zero point as a default for dynamic.
732 int input_zero_point_int = -1;
733 int kernel_zero_point_int = -1;
734
735 // Input zero point can either be a constant or a scalar expression.
736 if (IsConstScalar(input_zero_point) && (IsConstScalar(kernel_zero_point))) {
737 // Extract the integer zero points.
738 input_zero_point_int = GetScalarFromConstant<int>(input_zero_point);
739 kernel_zero_point_int = GetScalarFromConstant<int>(kernel_zero_point);
740 } else {
741 // Make kernel_zero_point expression a 1-D tensor for consistent shape.
742 kernel_zero_point = Reshape(kernel_zero_point, {
743 -1,
744 });
745 dynamic_zp = true;
746 }
747
748 // Fallback to int32 conv if there is dilation with non-zero kernel point or grouped conv2d
749 // For dilated conv, if the kernel zero point is non-zero, the pooling operator also has to
750 // traverse the elements in dilated manner. Currently, we do not have strided pool. So, in case of
751 // dilated conv with non-zero kernel point, we fall back to simpler but slow lowering.
752
753 ICHECK_EQ(param->dilation.size(), 2) << "qnn.conv2d only supports 2D dilation";
754 auto dilation_h = get_const_int(param->dilation[0]);
755 auto dilation_w = get_const_int(param->dilation[1]);
756 // Check if qnn supports the conv2d parameters. If not, fallback to regular conv2d.
757 bool supported_dilation = (kernel_zero_point_int == 0) || (dilation_h == 1 && dilation_w == 1);
758 bool supported_groups = (param->groups == 1 || is_depthwise(param));
759 bool conv2d_params_supported = supported_dilation && supported_groups;
760
761 // If we need to fall back to default conv2d, kernel zp may need to be broadcast to kernel_layout.
762 // Otherwise, we broadcast it to data_layout for qnn lowering.
763 if (dynamic_zp) {
764 if (!conv2d_params_supported) {
765 Layout kernel_layout(param->kernel_layout);
766 int kernel_axis = kernel_layout.IndexOf(LayoutAxis::Get("O"));
767 kernel_zero_point = ExpandBiasToMatchAxis(kernel_zero_point, 4, {kernel_axis});
768 } else {
769 Layout data_layout(param->data_layout);
770 int channel_axis = data_layout.IndexOf(LayoutAxis::Get("C"));
771 kernel_zero_point = ExpandBiasToMatchAxis(kernel_zero_point, 4, {channel_axis});
772 }
773 }
774
775 if (!conv2d_params_supported) {
776 return Conv2DFallBack(data, weight, input_zero_point, kernel_zero_point, param);
777 } else if (is_depthwise(param)) {
778 ICHECK_NE(channel_multiplier, -1);
779 auto padded_data = Conv2DPadInput(data, input_zero_point, param);
780 auto term1 = Conv2DFirstTerm(padded_data, weight, param);
781 auto term2 = DepthwiseConv2DSecondTerm(padded_data, kernel_zero_point, param, kernel_h,
782 kernel_w, channel_multiplier);
783 auto term3 =
784 DepthwiseConv2DThirdTerm(weight, input_zero_point, param, out_channels, channel_multiplier);
785 Expr term4;
786 if (dynamic_zp) {
787 term4 = DepthwiseConv2DFourthTerm(input_zero_point, kernel_zero_point, kernel_h, kernel_w);
788 } else {
789 term4 = DepthwiseConv2DFourthTerm(input_zero_point_int, kernel_zero_point_int, kernel_h,
790 kernel_w);
791 }
792 return Conv2DCombineTerms(term1, term2, term3, term4, input_zero_point_int,
793 kernel_zero_point_int);
794 }
795
796 auto padded_data = Conv2DPadInput(data, input_zero_point, param);
797 auto term1 = Conv2DFirstTerm(padded_data, weight, param);
798 auto term2 =
799 Conv2DSecondTerm(padded_data, kernel_zero_point, param, kernel_h, kernel_w, out_channels);
800 auto term3 = Conv2DThirdTerm(weight, input_zero_point, param, out_channels);
801 Expr term4;
802 if (dynamic_zp) {
803 term4 = Conv2DFourthTerm(input_zero_point, kernel_zero_point, in_channels, kernel_h, kernel_w,
804 param);
805 } else {
806 term4 = Conv2DFourthTerm(input_zero_point_int, kernel_zero_point_int, in_channels, kernel_h,
807 kernel_w, param);
808 }
809 return Conv2DCombineTerms(term1, term2, term3, term4, input_zero_point_int,
810 kernel_zero_point_int);
811}

814// used by frontend FFI.
815Expr MakeQnnConv2D(Expr data, Expr weight, Expr input_zero_point, Expr kernel_zero_point,
816 Expr input_scale, Expr kernel_scale, Array<IndexExpr> strides,
817 Array<IndexExpr> padding, Array<IndexExpr> dilation, int groups,
818 IndexExpr channels, Array<IndexExpr> kernel_size, String data_layout,

821 attrs->strides = std::move(strides);
822 attrs->padding = std::move(padding);
823 attrs->dilation = std::move(dilation);
824 attrs->groups = groups;
825 attrs->channels = std::move(channels);
826 attrs->kernel_size = std::move(kernel_size);
827 attrs->data_layout = std::move(data_layout);
828 attrs->kernel_layout = std::move(kernel_layout);
829 attrs->out_layout = std::move(out_layout);
830 attrs->out_dtype = std::move(out_dtype);

832 return Call(op, {data, weight, input_zero_point, kernel_zero_point, input_scale, kernel_scale},
833 Attrs(attrs), {});
834}
