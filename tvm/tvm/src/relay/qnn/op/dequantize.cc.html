39bool DequantizeRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
40 const TypeReporter& reporter) {
41 ICHECK_EQ(types.size(), 4);
42 const auto* data = types[0].as<TensorTypeNode>();
43
44 if (data == nullptr) {
45 return false;
46 }
47
48 const auto input_dtype = data->dtype;
49 ICHECK(input_dtype == DataType::Int(8) || input_dtype == DataType::UInt(8) ||
50 input_dtype == DataType::Int(16) || input_dtype == DataType::Int(32))
51 << "Input type should be one of the quantized types [unit8, int8, int16, int32] but was "
52 << input_dtype;
53
54 const auto* dequantize_attrs = attrs.as<DequantizeAttrs>();
55 int axis = dequantize_attrs->axis;
56 auto rank = static_cast<int>(data->shape.size());
57 axis = (axis < 0) ? ((rank > 0) ? data->shape.size() + axis : 0) : axis;
58
59 // If zero point and scale are scalar or have arbitrary rank with one element,
60 // then axis doesn't matter.
61 bool scale_is_scalar = (types[1].as<TensorTypeNode>())->shape.size() == 0 ||
62 get_const_int((types[1].as<TensorTypeNode>())->Size()) == 1;
63 bool zp_is_scalar = (types[2].as<TensorTypeNode>())->shape.size() == 0 ||
64 get_const_int((types[2].as<TensorTypeNode>())->Size()) == 1;
65
66 if (!scale_is_scalar || !zp_is_scalar) {
67 ICHECK_LT(axis, rank > 0 ? rank : 1) << "axis " << dequantize_attrs->axis << " is out of range";
68 ICHECK_GE(axis, 0) << "axis " << dequantize_attrs->axis << " is out of range";
69 }
70
71 PrimExpr axis_shape;
72 if (!scale_is_scalar || !zp_is_scalar) {
73 axis_shape = data->shape[axis];
74 } else {
75 axis_shape = Integer(1);
76 }
77 // Check and assign types for scale and zero points.
78 AssignType(types[1], DataType::Float(32), axis_shape, reporter); // scale
79 AssignType(types[2], DataType::Int(32), axis_shape, reporter); // zero point
80 const Array<tvm::PrimExpr> oshape = data->shape;
81 // assign output type, output will always be float 32.
82 reporter->Assign(types[3], TensorType(oshape, DataType::Float(32)));
83 return true;
84}
85
86Expr MakeDequantize(Expr data, Expr input_scale, Expr input_zero_point, int axis) {
87 // real_value = scale * (quantized_value - zero_point)
88 // A more detailed explanation can be found here -
89 // https://github.com/google/gemmlowp/blob/master/doc/quantization.md
90 auto attrs = make_object<DequantizeAttrs>();
91 attrs->axis = axis;
92 static const Op& op = Op::Get("qnn.dequantize");
93 return Call(op, {data, input_scale, input_zero_point}, Attrs(attrs), {});
94}
95
96Expr DequantizeLower(const Expr& input_tensor, const Expr& input_scale,
97 const Expr& input_zero_point, const Array<tvm::relay::Type>& types,
98 const DequantizeAttrs* attrs) {
99 auto axis = attrs->axis;
100
101 ICHECK_EQ(types.size(), 4);
102 auto in_type = types[0];
103 auto in_tensor_type = in_type.as<TensorTypeNode>();
104 ICHECK(in_tensor_type != nullptr) << "Type information missing"
105 << " Please run infer_type pass.";
106 Array<IndexExpr> input_shape = in_tensor_type->shape;
107
108 size_t n_dim = input_shape.size();
109
110 // Wrap axis from negative to positive if needed.
111 if (axis < 0) {
112 axis = static_cast<int>(n_dim) + axis;
113 }
114
115 // Expand scale and zero point if the input tensor is channel quantized
116 auto expanded_input_scale = input_scale;
117 if (!IsConstScalar(input_scale) && !IsScalarType(types[1])) {
118 expanded_input_scale = ExpandBiasToMatchAxis(input_scale, n_dim, {axis});
119 }
120
121 auto expanded_input_zero_point = input_zero_point;
122 if (!IsConstScalar(input_zero_point) && !IsScalarType(types[2])) {
123 expanded_input_zero_point = ExpandBiasToMatchAxis(input_zero_point, n_dim, {axis});
124 }
125
126 auto shift = Subtract(Cast(input_tensor, DataType::Int(32)), expanded_input_zero_point);
127 auto scaled_output = Multiply(Cast(shift, DataType::Float(32)), expanded_input_scale);
128 return scaled_output;
129}
130
131Expr DequantizeQnnCanonicalize(const Attrs& attrs, const Array<Expr>& new_args,
132 const Array<tvm::relay::Type>& types) {
133 ICHECK_EQ(new_args.size(), 3);
134 auto& data = new_args[0];
135 auto& input_scale = new_args[1];
136 auto& input_zero_point = new_args[2];
137 ICHECK_EQ(types.size(), 4);
138
139 // Get attrs.
140 const auto* dequantize_attrs = attrs.as<DequantizeAttrs>();
141 ICHECK(dequantize_attrs != nullptr);
142
143 return DequantizeLower(data, input_scale, input_zero_point, types, dequantize_attrs);
144}

146RELAY_REGISTER_OP("qnn.dequantize")
147 .describe(R"code(Dequantizes the input and produces float32 output.
148The input is always quantized (int8, uint8) and will be converted to float32 given input scale and zero_point.
149- **data**: Quantized tensor of any shape to dequantize. The input data can be of floating point
150)code" TVM_ADD_FILELINE)
151 .set_attrs_type<DequantizeAttrs>()
152 .set_num_inputs(3)
153 .add_argument("data", "Tensor", "The tensor to dequantize.")
154 .add_argument("input_scale", "Tensor", "The quantization scale of the input tensor.")
155 .add_argument("input_zero_point", "Tensor", "The quantization zero_point of the input tensor.")
156 .set_support_level(11)
157 .add_type_rel("Dequantize", DequantizeRel)
158 .set_attr<TNonComputational>("TNonComputational", true)
159 .set_attr<FTVMLegalize>("FTVMQnnCanonicalize", DequantizeQnnCanonicalize);
160
161TVM_REGISTER_GLOBAL("relay.qnn.op._make.dequantize").set_body_typed(MakeDequantize);
