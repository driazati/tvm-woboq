20/*!
21 * \file src/relay/collage/utils.h
22 * \brief Misc helpers.
23 */
24
25#ifndef TVM_RELAY_COLLAGE_UTILS_H_
26#define TVM_RELAY_COLLAGE_UTILS_H_
27
28#include <tvm/relay/expr.h>
29#include <tvm/relay/function.h>
30#include <tvm/relay/op_attr_types.h>
31#include <tvm/runtime/container/string.h>
32
33#include <string>
34
35namespace tvm {
36namespace relay {
37namespace collage {
38
39/*!
40 * \brief Distinguished partition spec names.
41 */
42constexpr const char* kTVMSpecNamePrefix = "tvm_";
43constexpr const char* kHostSpecName = "host";
44
45/*!
46 * \brief Returns the partition spec name to use for \p target. For external codegen targets the
47 * spec name is just the target kind name. For TVM native targets the spec name is of the form
48 * "tvm_<kind_name>".
49 */
50String GetSpecName(const Target& target);
51
52/*! \brief Returns \p "<left>+<right>". */
53String UnionLabels(String left, String right);
54
55/*! \brief Returns \p "<outer>.<inner>". */
56String NestLabels(String outer, String inner);
57
58/*! \brief Returns abbreviation for \p kind. */
59std::string KindToString(OpPatternKind kind);
60
61/*! \brief Returns maximum of \p left and \p right. */
62OpPatternKind CombineKinds(OpPatternKind left, OpPatternKind right);
63
64/*!
65 * \brief Returns true if \p expr can be safely inlined in body of function extracted
66 * from sub-graph, even if \p expr was not technically matched by the pattern which produced
67 * the sub-graph.
68 */
69bool CanInline(const Expr& expr);
70
71/*!
72 * \brief Returns true if \p op_node can be directly handled by the VM.
73 */
74bool IsSpecialOp(const OpNode* op_node);
75
76/*!
77 * \brief Return true if the Relay expression node given by \p expr cannot be evaluated by
78 * the VM and must end up in a kernel.
79 */
80bool MustBeLowered(const Expr& expr);
81
82} // namespace collage
83} // namespace relay
84} // namespace tvm
85
86#endif // TVM_RELAY_COLLAGE_UTILS_H_
