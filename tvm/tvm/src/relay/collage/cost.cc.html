27namespace tvm {
28namespace relay {
29namespace collage {
30
31std::string Cost::ToString() const {
32 if (is_invalid()) {
33 return "invalid";
34 } else if (is_unknown()) {
35 return "unknown";
36 } else if (value_ == 0.0) {
37 return "0";
38 } else {
39 return std::to_string(value_ * 1e6) + "us";
40 }
41}
