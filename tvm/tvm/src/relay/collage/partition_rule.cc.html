80std::vector<CandidatePartition> DFPatternPartitionRuleNode::AllCandidates(
81 const DataflowGraph& dataflow_graph, const PartitionSpec& spec) const {
82 VLOG(1) << "running DFPatternPartitionRule(" << rule_name_ << ")";
83 std::vector<CandidatePartition> result;
84 DFPatternMatcher matcher(&dataflow_graph.indexed_graph());
85 for (PostDfsIndex index = 0; index < dataflow_graph.size(); ++index) {
86 Expr sub_expr = dataflow_graph.index_to_node(index)->ref();
87 if (!matcher.Match(pattern_, sub_expr)) {
88 continue;
89 }
90 if (!predicate_(sub_expr)) {
91 VLOG(1) << "DFPatternPartitionRule(" << rule_name_ << ") has failing predicate";
92 continue;
93 }
94 IndexSet inside = MatcherToIndexSet(matcher);
95 OpPatternKind kind;
96 String label;
97 std::tie(kind, label) = SubGraphKindAndLabel(dataflow_graph, inside);
98 SubGraph sub_graph(dataflow_graph, std::move(inside), kind, std::move(label));
99 String rule_name = rule_name_.empty() ? sub_graph->label_ : rule_name_;
100 CandidatePartition candidate(std::move(rule_name), std::move(sub_graph), spec);
101 VLOG(2) << "DFPatternPartitionRule(" << rule_name_ << ") yields " << candidate->ToString();
102 result.emplace_back(std::move(candidate));
103 }
104 VLOG(1) << "DFPatternPartitionRule(" << rule_name_ << ") produced " << result.size()
105 << " candidates";
106 return result;
107}

130std::vector<CandidatePartition> CompositePartitionRuleNode::AllCandidates(
131 const DataflowGraph& dataflow_graph, const PartitionSpec& spec) const {
132 std::vector<CandidatePartition> candidates = sub_rule_->AllCandidates(dataflow_graph, spec);
133 VLOG(1) << "running CompositePartitionRule(" << rule_name_ << ") over " << candidates.size()
134 << " sub-candidates";
135 std::vector<CandidatePartition> result;
136 FunctionAttrsMap attrs;
137 attrs.Set(attr::kComposite, rule_name_);
138 for (auto& candidate : candidates) {
139 String rule_name = NestLabels(rule_name_, candidate->rule_name_);
140 SubGraph sub_graph = candidate->sub_graph_.WithAttrs(dataflow_graph, attrs);
141 CandidatePartition new_candidate = WithSubGraph(
142 WithRuleName(std::move(candidate), std::move(rule_name)), std::move(sub_graph));
143 VLOG(2) << "CompositePartitionRule(" << rule_name_ << ") yields " << new_candidate->ToString();
144 result.emplace_back(std::move(new_candidate));
145 }
146 VLOG(1) << "CompositePartitionRule(" << rule_name_ << ") produced " << result.size()
147 << " candidates";
148 return result;
149}

251std::vector<CandidatePartition> OpCallByKindPartitionRuleNode::AllCandidates(
252 const DataflowGraph& dataflow_graph, const PartitionSpec& spec) const {
253 VLOG(1) << "running OpCallByKindPartitionRule(" << rule_name_ << ")";
254 std::vector<CandidatePartition> result;
255 for (PostDfsIndex index = 0; index < dataflow_graph.size(); ++index) {
256 auto node = dataflow_graph.index_to_node(index);
257 Expr sub_expr = node->ref();
258 if (sub_expr->IsInstance<CallNode>()) {
259 OpPatternKind kind;
260 String label;
261 std::tie(kind, label) = SubExprKindAndLabel(sub_expr);
262 if (kind <= kOutEWiseFusable) {
263 IndexSet inside(dataflow_graph.size(), {index});
264 SubGraph sub_graph(dataflow_graph, std::move(inside), kind, std::move(label));
265 String rule_name = NestLabels(rule_name_, sub_graph->label_);
266 CandidatePartition candidate(std::move(rule_name), std::move(sub_graph), spec);
267 VLOG(2) << "OpCallByKindPartitionRule(" << rule_name_ << ") yields "
268 << candidate->ToString();
269 result.emplace_back(std::move(candidate));
270 }
271 }
272 }
273 VLOG(1) << "OpCallByKindPartitionRule(" << rule_name_ << ") produced " << result.size()
274 << " candidates";
275 return result;
276}

294std::vector<CandidatePartition> CombinePartitionRuleNode::AllCandidates(
295 const DataflowGraph& dataflow_graph, const PartitionSpec& spec) const {
296 // We'll accumulate all the candidates here, starting with those from the sub-rule.
297 // Once a candidate is added to this vector it is immutable.
298 std::vector<CandidatePartition> candidates = sub_rule_->AllCandidates(dataflow_graph, spec);
299 VLOG(1) << "running CombinePartitionRule(" << rule_name_ << ") over " << candidates.size()
300 << " sub-candidates";
301 CandidateSet result_set(std::move(candidates));
302
303 size_t num_rounds = 0;
304 AppendAllResultsContext ctxt(&dataflow_graph, max_depth_, &result_set);
305 while (result_set.PrepareForNextRound()) {
306 VLOG_CONTEXT << "round " << ++num_rounds;
307 VLOG(1) << "checking " << result_set.size() << " candidates (" << result_set.first_new_index()
308 << " existing)";
309 for (const auto& combiner_rule : combiner_rules_) {
310 combiner_rule->AppendAllResults(&ctxt);
311 }
312 }
313
314 std::vector<CandidatePartition> result;
315 for (auto& candidate : result_set.MovedCurrentCandidates()) {
316 String rule_name = NestLabels(rule_name_, candidate->rule_name_);
317 CandidatePartition new_candidate = WithRuleName(std::move(candidate), std::move(rule_name));
318 VLOG(2) << "CombinePartitionRule(" << rule_name_ << ") yields " << new_candidate->ToString();
319 result.emplace_back(std::move(new_candidate));
320 }
321 VLOG(1) << "CombinePartitionRule(" << rule_name_ << ") produced " << result.size()
322 << " candidates";
323 return result;
324}

327 PartitionRuleNode::AppendBodyItems(body_items);

354std::vector<CandidatePartition> OnlyValidPartitionRuleNode::AllCandidates(

363 continue;

365 String rule_name = NestLabels(rule_name_, candidate->rule_name_);
366 CandidatePartition new_candidate = WithRuleName(std::move(candidate), std::move(rule_name));
367 VLOG(2) << "OnlyValidPartitionRule(" << rule_name_ << ") yields " << new_candidate->ToString();
368 result.emplace_back(std::move(new_candidate));

370 VLOG(1) << "OnlyValidPartitionRule(" << rule_name_ << ") produced " << result.size()
371 << " candidates";
372 return result;

398std::vector<CandidatePartition> HostPartitionRuleNode::AllCandidates(
399 const DataflowGraph& dataflow_graph, const PartitionSpec& spec) const {
400 VLOG(1) << "running HostPartitionRule(" << rule_name_ << ")";
401 std::vector<CandidatePartition> result;
402 for (PostDfsIndex index = 0; index < dataflow_graph.size(); ++index) {
403 if (MustBeLowered(dataflow_graph.index_to_node(index)->ref())) {
404 continue;
405 }
406 IndexSet inside(dataflow_graph.size(), {index});
407 OpPatternKind kind;
408 String label;
409 std::tie(kind, label) = SubGraphKindAndLabel(dataflow_graph, inside);
410 SubGraph sub_graph(dataflow_graph, std::move(inside), kind, label);
411 String rule_name = NestLabels(rule_name_, sub_graph->label_);
412 // We'll a zero cost for the candidate since we'll never want to actually estimate the cost
413 // of this 'partition'.
414 CandidatePartition candidate(std::move(rule_name), std::move(sub_graph), spec, Cost::Zero());
415 VLOG(2) << "HostPartitionRule(" << rule_name_ << ") yields " << candidate->ToString();
416 result.push_back(candidate);
417 }
418 VLOG(1) << "HostPartitionRule(" << rule_name_ << ") produced " << result.size() << " candidates";
419 return result;
420}
