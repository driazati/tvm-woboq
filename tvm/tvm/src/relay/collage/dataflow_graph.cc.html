21 * \file src/relay/collage/dataflow_graph.cc
22 * \brief A representation of the dataflow for an overall Relay expression.
23 */
24
25#include "./dataflow_graph.h"
26
27namespace tvm {
28namespace relay {
29namespace collage {
30
31DataflowGraph::DataflowGraph(Expr expr) : expr_(std::move(expr)) {
32 indexed_graph_ = CreateIndexedGraph(expr_);
33 downstream_map_.reserve(indexed_graph_->size());
34 for (PostDfsIndex index = 0; index < indexed_graph_->size(); ++index) {
35 const Node* node = indexed_graph_->index_to_node(index);
36 std::unordered_set<const Node*> downstream_nodes;
37 node->AccumulateDownstreamNodes(&downstream_nodes);
38 IndexSet index_set(indexed_graph_->size());
39 for (const Node* downstream_node : downstream_nodes) {
40 index_set.Add(downstream_node->index_);
41 }
42 downstream_map_.emplace_back(std::move(index_set));
43 }
44}
