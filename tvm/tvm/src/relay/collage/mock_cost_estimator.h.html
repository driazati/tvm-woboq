26#define TVM_RELAY_COLLAGE_MOCK_COST_ESTIMATOR_H_
27
28#include <tvm/relay/function.h>
29
30#include "./cost.h"
31#include "./cost_estimator.h"
32
33namespace tvm {
34namespace relay {
35namespace collage {
36
37/*!
38 * \brief A mock cost estimator which can determine the cost of a candidate based on both
39 * the candidate's target and the number of operator calls inside it.
40 *
41 * The help unit tests the estimator also ICHECK fails if:
42 * - the module has inlined "Compiler" functions
43 * - @main has non-tensor arguments (eg a tuple)
44 * - more than the given number of candidate modules are measured
45 *
46 * To support unit testing only.
47 */
48class MockCostEstimatorNode : public CostEstimatorNode {
49 public:
50 Cost Estimate(const IRModule& mod, const Target& target) const override;
51
52 static constexpr const char* _type_key = "relay.collage.MockCostEstimator";
53 TVM_DECLARE_FINAL_OBJECT_INFO(MockCostEstimatorNode, CostEstimatorNode);
54
55 protected:
56 /*!
57 * \brief Map from target kind name to assumed baseline cost (in integer seconds) for all
58 * operator calls.
59 */
60 Map<String, Integer> target_costs_;
61
62 /*!
63 * \brief If non-zero, the maximum number of distinct modules which may be estimated.
64 */
65 Integer max_estimates_;
66
67 /*! \brief Number of calls to Estimate. */
68 mutable size_t num_estimates_ = 0;
69
70 friend class MockCostEstimator;
71};
72
73class MockCostEstimator : public CostEstimator {
74 public:
75 explicit MockCostEstimator(Map<String, Integer> target_costs, Integer max_estimates = 0);
76
77 TVM_DEFINE_OBJECT_REF_METHODS(MockCostEstimator, CostEstimator, MockCostEstimatorNode);
78};
79
80} // namespace collage
81} // namespace relay
82} // namespace tvm
