22 * \brief A source of fresh variable names.

24
25#ifndef TVM_RELAY_COLLAGE_NAME_SUPPLY_H_
26#define TVM_RELAY_COLLAGE_NAME_SUPPLY_H_
27
28#include <string>
29#include <unordered_map>
30#include <utility>
31
32namespace tvm {
33namespace relay {
34namespace collage {
35
36/*! \brief A supply of fresh names. */
37class NameSupply {
38 public:
39 explicit NameSupply(std::string prefix) : prefix_(std::move(prefix)) {}
40
41 NameSupply MakeSubNameSupply();
42
43 void Reserve(const std::string& existing) { next_free_index_.emplace(existing, 1); }
44
45 std::string Fresh(const std::initializer_list<std::string>& hints);
46
47 private:
48 /*! \brief Prefix for all names. May be empty. */
49 std::string prefix_;
50 /*! \brief Next unused index for variables with given basename. */
51 std::unordered_map<std::string, int> next_free_index_;
52};
53
54} // namespace collage
55} // namespace relay
56} // namespace tvm
57
58#endif // TVM_RELAY_COLLAGE_NAME_SUPPLY_H_
