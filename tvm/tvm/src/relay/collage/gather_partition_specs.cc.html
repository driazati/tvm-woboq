35PartitionRule MakeCombinePartitionRule(PartitionRule sub_rule, Array<CombinerRule> combiner_rules,
36 size_t max_depth) {
37 if (combiner_rules.empty()) {
38 return sub_rule;
39 } else {
40 return CombinePartitionRule("", std::move(sub_rule), std::move(combiner_rules), max_depth);
41 }
42}

44/*! \brief Returns the primitive combiner rules which mimic TVM's \p FuseOps. */
45Array<CombinerRule> TVMCombinerRules() {
46 Array<SimpleCombinerRule> simple_rules;
47 // Mimic the FuseOps rules.
48 simple_rules.push_back(ByKindSimpleCombinerRule(kOutEWiseFusable, kBroadcast));
49 simple_rules.push_back(ByKindSimpleCombinerRule(kBroadcast, kCommReduce));
50 simple_rules.push_back(ByKindSimpleCombinerRule(kInjective, kInjective));
51
52 Array<CombinerRule> combiner_rules;
53 // Fire the simple fusion rules
54 combiner_rules.push_back(AllSimpleCombinerRule("combiner", std::move(simple_rules)));
55 // Fuse tuple arguments
56 combiner_rules.push_back(TupleArgCombinerRule("tuple"));
57 // Fuse tuple projection
58 combiner_rules.push_back(TupleProjCombinerRule("proj"));
59
60 return combiner_rules;
61}

63size_t GetMaxDepth(std::string key) {
64 tvm::transform::PassContext ctxt = tvm::transform::PassContext::Current();
65 std::string config_key = "relay.collage." + key;
66 Optional<Integer> opt_max_depth = ctxt->GetConfig(config_key, Optional<Integer>());
67 ICHECK(opt_max_depth.defined()) << "missing binding for '" << config_key << " in pass context";
68 ICHECK(opt_max_depth.value()->value > 0)
69 << "invalid value for '" << config_key << " in pass context";
70 return static_cast<size_t>(opt_max_depth.value()->value);
71}

73/*! \brief Returns partition rule mimicking TVM FuseOps. */
74PartitionRule MakeTVMPartitionRule() {
75 size_t max_depth = GetMaxDepth("tvm_max_depth");
76 // Build singleton candidates for all calls to ops <= kOutEWiseFusable.
77 OpCallByKindPartitionRule op_call_by_kind("");
78 // Combine candidates according to the TVM fusion rules.
79 PartitionRule combine =
80 MakeCombinePartitionRule(std::move(op_call_by_kind), TVMCombinerRules(), max_depth);
81 // Discard invalid candidates.
82 SubGraphConfig sub_graph_config;
83 sub_graph_config.allow_taps = false;
84 sub_graph_config.max_depth = max_depth;
85 sub_graph_config.max_exits = 1;
86 return OnlyValidPartitionRule("", std::move(combine), sub_graph_config);
87 // NOTE: We don't wrap by a "Primitive" since we want to defer making TVM fusion decisions until
88 // after running more Relay passes.
89}

91/*!
92 * \brief Returns the fusion style for \p compiler.
93 *
94 * TODO(mbs): Defer to per-BYOC integration definition.
95 */
96BYOCStyle BYOCFusionStyleForCompiler(const String& compiler) {
97 if (compiler == "cutlass" || compiler == "cublas" || compiler == "cudnn") {
98 return kNoFusionBYOCStyle;
99 } else if (compiler == "tensorrt") {
100 return kTVMFusionBYOCStyle;
101 } else {
102 return kArbitraryFusionBYOCStyle;
103 }
104}
105
106/*!
107 * \brief Returns the primitive combiner rules which allow for any touching candidates
108 * to be fused provided they don't have kind \p kOpaque.
109 */
110Array<CombinerRule> BYOCCombinerRules(const String& compiler) {
111 Array<SimpleCombinerRule> simple_rules;
112 Array<CombinerRule> combiner_rules;
113 switch (BYOCFusionStyleForCompiler(compiler)) {
114 case kNoFusionBYOCStyle:
115 break;
116 case kTVMFusionBYOCStyle:
117 // Conservatively assume the BYOC toolchain follows the same rules as for TVM's FuseOps.
118 simple_rules.push_back(ByKindSimpleCombinerRule(kOutEWiseFusable, kBroadcast));
119 simple_rules.push_back(ByKindSimpleCombinerRule(kBroadcast, kCommReduce));
120 simple_rules.push_back(ByKindSimpleCombinerRule(kInjective, kInjective));
121 combiner_rules.push_back(AllSimpleCombinerRule("combiner", std::move(simple_rules)));
122 break;
123 case kArbitraryFusionBYOCStyle:
124 // Just try all combinations up to the max_depth limit.
125 simple_rules.push_back(ByKindSimpleCombinerRule(kOutEWiseFusable, kOutEWiseFusable));
126 combiner_rules.push_back(AllSimpleCombinerRule("combiner", std::move(simple_rules)));
127 break;
128 }
129 return combiner_rules;
130}

132/*!
133 * \brief Returns partition rule mimicking one entry in the patterns list passed to the
134 * MergeComposite pass.
135 */
136PartitionRule MakeLabelledDFPatternPartitionRule(
137 const std::string& compiler, String rule_name, DFPattern dataflow_pattern,
138 TPatternPredicate predicate = DefaultPatternPredicate) {
139 DFPatternPartitionRule patterns("", std::move(dataflow_pattern), std::move(predicate));
140 return CompositePartitionRule(std::move(rule_name), std::move(patterns));
141}

143/*!
144 * \brief Returns partition rule mimicking
145 * MergeComposite/AnnotateTarget/MergeCompilerRegions/PartitionGraph passes for "compiler"
146 * attribute of \p target.
147 */
148PartitionRule MakePatternBYOCPartitionRule(const std::string& compiler,
149 Array<PartitionRule> sub_rules) {
150 size_t max_depth = GetMaxDepth("byoc_max_depth");
151 // Union all the individual pattern rules.
152 UnionPartitionRule unioned("", std::move(sub_rules));
153 PartitionRule combine =
154 MakeCombinePartitionRule(std::move(unioned), BYOCCombinerRules(compiler), max_depth);
155 // Ignore invalid candidates.
156 SubGraphConfig sub_graph_config;
157 sub_graph_config.allow_taps = false;
158 sub_graph_config.max_depth = max_depth;
159 sub_graph_config.max_exits = 1;
160 OnlyValidPartitionRule valid("", std::move(combine), sub_graph_config);
161 // Wrap the candidates in a "Primitive" function with a "Compiler" attribute.
162 return PrimitivePartitionRule("", std::move(valid));
163}

182Array<PartitionSpec> GatherPartitionSpecs(const CompilationConfig& config) {
183 Array<PartitionSpec> result;
184 for (const auto& primitive_target : config->primitive_targets) {
185 String spec_name = GetSpecName(primitive_target);
186 PartitionRule rule;
187 if (primitive_target.IsExternalCodegen()) {
188 // Transition to the Python side so we can get access to the BYOC pattern registry.
189 // That will bounce right back into the above construction helpers.
190 static const runtime::PackedFunc* make_byoc_partition_rule =
191 runtime::Registry::Get("tvm.relay.collage.make_byoc_partition_rule");
192 ICHECK(make_byoc_partition_rule);
193 rule = (*make_byoc_partition_rule)(spec_name); // spec_name == primitive_target->kind->name
194 VLOG(1) << "Target " << primitive_target->ToDebugString() << " is for BYOC spec_name "
195 << spec_name << " and has default partition rule:\n"
196 << rule->ToString();
197 } else {
198 rule = MakeTVMPartitionRule();
199 VLOG(1) << "Target " << primitive_target->ToDebugString() << " is for TVM spec_name "
200 << spec_name << " and has default partition rule:\n"
201 << rule->ToString();
202 }
203 result.push_back(PartitionSpec(spec_name, primitive_target, rule));
204 }
205
206 // Add one more spec to cover the host target.
207 result.push_back(PartitionSpec(kHostSpecName, config->host_target, MakeHostPartitionRule()));
208
209 return result;
210}
