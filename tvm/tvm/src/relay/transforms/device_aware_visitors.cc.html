31// TODO(mbs): This machinery can be used a) on expressions/modules which have not had
32// device planning run, and b) on expressions for which we've not kept track of their
33// containing module. For now we'll handle b) by being forgiving as possible when recovering
34// the device for an expression, and we'll support a) the same way. But better would be
35// to ICHECK fail when, eg, a variable is not in scope or the lexical device stack is empty.
36
37LexicalOnDeviceMixin::LexicalOnDeviceMixin(const Optional<IRModule>& maybe_mod) {
38 if (maybe_mod) {
39 for (const auto& kv : maybe_mod.value()->functions) {
40 if (const auto* function_node = kv.second.as<FunctionNode>()) {
41 VirtualDevice virtual_device = function_node->virtual_device();
42 if (!virtual_device->IsFullyUnconstrained()) {
43 VLOG(2) << "global '" << kv.first->name_hint << "' has virtual device " << virtual_device;
44 global_var_virtual_devices_.emplace(kv.first, virtual_device);
45 }
46 }
47 }
48 }
49}
50
51VirtualDevice LexicalOnDeviceMixin::GetVirtualDevice(const Expr& expr) const {
52 OnDeviceProps props = GetOnDeviceProps(expr);
53 if (props.body.defined() && props.is_fixed()) {
54 return props.virtual_device;
55 } else if (const auto* var_node = expr.as<VarNode>()) {
56 // Lookup variable binding.
57 auto itr = var_virtual_devices_.find(GetRef<Var>(var_node));
58 if (itr != var_virtual_devices_.end()) {
59 return itr->second;
60 }
61 // else: fallthrough to unconstrained
62 } else if (const auto* global_var_node = expr.as<GlobalVarNode>()) {
63 // Lookup global variable.
64 auto itr = global_var_virtual_devices_.find(GetRef<GlobalVar>(global_var_node));
65 if (itr != global_var_virtual_devices_.end()) {
66 return itr->second;
67 }
68 // else: fallthrough to unconstrained
69 } else if (const auto* function_node = expr.as<FunctionNode>()) {
70 if (function_node->HasNonzeroAttr(attr::kPrimitive)) {
71 if (!expr_virtual_devices_.empty()) {
72 // Use the currently in-scope device type.
73 return expr_virtual_devices_.back();
74 }
75 // else: fallthrough to unconstrained
76 } else {
77 return function_node->virtual_device();
78 }
79 } else {
80 if (!expr_virtual_devices_.empty()) {
81 // Use the currently in-scope device type.
82 return expr_virtual_devices_.back();
83 }
84 // else: fallthrough to unconstrained
85 }
86 return VirtualDevice::FullyUnconstrained();
87}

89void LexicalOnDeviceMixin::EnterFunctionBody() { ++function_nesting_; }
90
91void LexicalOnDeviceMixin::ExitFunctionBody() {
92 ICHECK_GT(function_nesting_, 0);
93 --function_nesting_;
94}

96void LexicalOnDeviceMixin::PushVirtualDevice(const VirtualDevice& virtual_device) {
97 if (virtual_device->IsFullyUnconstrained()) {
98 return;
99 }
100 expr_virtual_devices_.emplace_back(virtual_device);
101}
102
103void LexicalOnDeviceMixin::PopVirtualDevice() {
104 if (expr_virtual_devices_.empty()) {
105 return;
106 }
107 expr_virtual_devices_.pop_back();
108}
109
110void LexicalOnDeviceMixin::PushBoundVar(Var var, const VirtualDevice& virtual_device) {
111 if (virtual_device->IsFullyUnconstrained()) {
112 return;
113 }
114 ICHECK(var_virtual_devices_.find(var) == var_virtual_devices_.end());
115 var_virtual_devices_.emplace(std::move(var), virtual_device);
116}
117
118void LexicalOnDeviceMixin::PopBoundVar(const Var& var) {
119 auto itr = var_virtual_devices_.find(var);
120 if (itr == var_virtual_devices_.end()) {
121 return;
122 }
123 var_virtual_devices_.erase(itr);
124}

129void DeviceAwareExprVisitor::VisitExpr_(const FunctionNode* function_node) {
130 if (function_node->HasNonzeroAttr(attr::kPrimitive)) {
131 // No tracking inside primitive functions.
132 DeviceAwareVisitExpr_(function_node);
133 } else {
134 // Function parameters come into scope.
135 for (auto param : function_node->params) {
136 PushBoundVar(param, param->virtual_device());
137 }
138 // Entering scope of function body.
139 PushVirtualDevice(function_node->virtual_device());
140 EnterFunctionBody();
141
142 DeviceAwareVisitExpr_(function_node);
143
144 // Leaving scope of function body.
145 ExitFunctionBody();
146 PopVirtualDevice();
147 // Function parameters go out of scope.
148 for (size_t i = 0; i < function_node->params.size(); ++i) {
149 PopBoundVar(function_node->params[i]);
150 }
151 }
152}

154void DeviceAwareExprVisitor::VisitExpr_(const LetNode* let_node) {
155 PreVisitLetBlock_(let_node);
156 std::vector<const LetNode*> bindings;
157 Expr expr = GetRef<Expr>(let_node);
158 while (const auto* inner_let_node = expr.as<LetNode>()) {
159 // Let-bound var (in pre visited version) goes into scope.
160 // (We'll just assume this is a letrec).
161 PushBoundVar(inner_let_node->var, GetVirtualDevice(inner_let_node->value));
162 PreVisitLetBinding_(inner_let_node->var, inner_let_node->value);
163 bindings.emplace_back(inner_let_node);
164 expr = inner_let_node->body;
165 }
166
167 VisitExpr(expr);
168
169 for (auto itr = bindings.rbegin(); itr != bindings.rend(); ++itr) {
170 // Let-bound var goes out of scope.
171 PopBoundVar((*itr)->var);
172 PostVisitLet_(*itr);
173 }
174 PostVisitLetBlock_(let_node);
175}

177void DeviceAwareExprVisitor::VisitExpr_(const CallNode* call_node) {
178 OnDeviceProps props = GetOnDeviceProps(call_node);
179 if (props.body.defined() && props.is_fixed()) {
180 // Entering lexical scope of fixed "on_device" call.
181 PushVirtualDevice(props.virtual_device);
182 VisitExpr(props.body);
183 // Leaving lexical scope of "on_device" call.
184 PopVirtualDevice();
185 } else {
186 DeviceAwareVisitExpr_(call_node);
187 }
188}

215Expr DeviceAwareExprMutator::VisitExpr_(const FunctionNode* function_node) {
216 if (function_node->HasNonzeroAttr(attr::kPrimitive)) {
217 // No tracking inside primitive functions.
218 return DeviceAwareVisitExpr_(function_node);
219 } else {
220 // Function parameters come into scope.
221 for (auto param : function_node->params) {
222 PushBoundVar(param, param->virtual_device());
223 }
224 // Entering scope of function body.
225 PushVirtualDevice(function_node->virtual_device());
226 EnterFunctionBody();
227
228 Expr result = DeviceAwareVisitExpr_(function_node);
229
230 // Leaving scope of function body.
231 ExitFunctionBody();
232 PopVirtualDevice();
233 // Function parameters go out of scope.
234 for (size_t i = 0; i < function_node->params.size(); ++i) {
235 PopBoundVar(function_node->params[i]);
236 }
237
238 return result;
239 }
240}

242Expr DeviceAwareExprMutator::VisitExpr_(const LetNode* let_node) {
243 PreVisitLetBlock_(let_node);
244 std::vector<std::tuple<Var, Expr, Span, const LetNode*>> bindings;
245 Expr expr = GetRef<Expr>(let_node);
246 while (const auto* inner_let_node = expr.as<LetNode>()) {
247 // Let-bound var (in pre visited version) goes into scope.
248 // (We'll just assume this is a letrec.)
249 PushBoundVar(inner_let_node->var, GetVirtualDevice(inner_let_node->value));
250 std::pair<Var, Expr> pair = PreVisitLetBinding_(inner_let_node->var, inner_let_node->value);
251 bindings.emplace_back(pair.first, pair.second, inner_let_node->span, inner_let_node);
252 expr = inner_let_node->body;
253 }
254
255 expr = VisitExpr(expr);
256
257 for (auto itr = bindings.rbegin(); itr != bindings.rend(); ++itr) {
258 // Let-bound var goes out of scope.
259 const LetNode* pre_let_node = std::get<3>(*itr);
260 PopBoundVar(pre_let_node->var);
261 Let post_let = Let(/*var=*/std::get<0>(*itr), /*value=*/std::get<1>(*itr),
262 /*body=*/expr, /*span=*/std::get<2>(*itr));
263 expr = PostVisitLet_(pre_let_node, post_let.get());
264 }
265 return PostVisitLetBlock_(let_node, expr.as<LetNode>());
266}

268Expr DeviceAwareExprMutator::VisitExpr_(const CallNode* call_node) {
269 OnDeviceProps props = GetOnDeviceProps(call_node);
270 if (props.body.defined() && props.is_fixed()) {
271 // Entering lexical scope of fixed "on_device" call.
272 PushVirtualDevice(props.virtual_device);
273 Expr expr = VisitExpr(props.body);
274 // Leaving lexical scope of "on_device" call.
275 PopVirtualDevice();
276 return MaybeOnDeviceWithProps(expr, props);
277 } else {
278 return DeviceAwareVisitExpr_(call_node);
279 }
280}

293std::pair<Var, Expr> DeviceAwareExprMutator::PreVisitLetBinding_(const Var& var,

298Expr DeviceAwareExprMutator::PostVisitLet_(const LetNode* pre_let_node,
299 const LetNode* post_let_node) {
300 if (pre_let_node->var == post_let_node->var && pre_let_node->value == post_let_node->value &&
301 pre_let_node->body == post_let_node->body) {
302 return GetRef<Expr>(pre_let_node);
303 } else {
304 return GetRef<Expr>(post_let_node);
305 }
306}

318std::unordered_map<const ExprNode*, VirtualDevice> RecoverVirtualDeviceMap(const IRModule& mod,
319 const Expr& expr) {
320 class Visitor : public DeviceAwareExprVisitor {
321 public:
322 explicit Visitor(const Optional<IRModule>& maybe_mod) : DeviceAwareExprVisitor(maybe_mod) {}
323
324 void VisitExpr(const Expr& expr) final {
325 if (expr->IsInstance<OpNode>() || expr->IsInstance<ConstructorNode>()) {
326 // Don't record for ops or constructors since they are 'device polymorphic'.
327 } else {
328 map_[expr.get()] = GetVirtualDevice(expr);
329 }
330 DeviceAwareExprVisitor::VisitExpr(expr);
331 }
332
333 std::unordered_map<const ExprNode*, VirtualDevice> map_;
334 };
335
336 Visitor visitor(mod);
337 visitor.VisitExpr(expr);
338 return std::move(visitor.map_);
339}

341// Export the helper function for testing.
342TVM_REGISTER_GLOBAL("relay.transform.RecoverVirtualDeviceMap")
343 .set_body_typed([](const IRModule& mod, const Expr& expr) {
344 std::unordered_map<const ExprNode*, VirtualDevice> raw_map =
345 RecoverVirtualDeviceMap(mod, expr);
346 Map<Expr, VirtualDevice> map;
347 for (const auto& kv : raw_map) {
348 map.Set(GetRef<Expr>(kv.first), kv.second);
349 }
350 return map;
351 });
