20/*!
21 * \file annotate_texture_storage.cc
22 * \brief Collection of target specific relay passes which
23 * storage scope related information.
24 *
25 * - CollectStorageInfo returns a mapping from relay expr
26 * to a map of storage scopes for each call argument.
27 * These scopes are used during memory planning as well
28 * as downstream when doing codegen and in the graph runtime when doing runtime dataspace
29 * allocations.
30 *
31 * - AnnotateMemoryScope calls *target.CollectStorageInfo for all target been represented
32 * in the graph and rewrites graph modifying or inserting of VirtualDevice with required
33 * memory_scope collected from the CollectStorageInfo
34 */

53/**
54 * @brief Analyzes the graph and returns mapping of expressions vs desired memory scope
55 */
56class StorageInfo : private transform::DeviceAwareExprVisitor {
57 public:
58 StorageInfo() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
59
60 static Map<Expr, Map<Expr, Array<String>>> GetStorageMap(const Expr& expr) {
61 StorageInfo storage_info;
62 storage_info.VisitExpr(expr);
63 storage_info.LegalizeProducerStorage();
64 Map<Expr, Map<Expr, Array<String>>> storage_map = storage_info.accept_textures_;
65 for (auto& kv : storage_info.storage_scope_) {
66 std::vector<String> storage_scopes;
67 std::copy(kv.second.begin(), kv.second.end(), std::back_inserter(storage_scopes));
68 Map<Expr, Array<String>> ent;
69 ent.Set(Expr(), Array<String>{storage_scopes});
70 storage_map.Set(GetRef<Expr>(kv.first), ent);
71 }
72
73 // Filling the input arguments by "global" scope to handle PlanDevice algo which propagates
74 // virtual devices from outputs to inputs. At the same time outputs must be unconstrained
75 // to avoid useless device_copy
76 for (const auto& cs : storage_info.consumer_storage_scopes_) {
77 // we have record in consumers that mean that potentially consumer
78 // dealt with textures anyhow, it's safe to mark this expr as global scope
79 // even without verification of the consumer's outputs scope
80 if (storage_info.CanConsumeTextures(cs.second) &&
81 storage_map.find(GetRef<Expr>(cs.first)) == storage_map.end()) {
82 Map<Expr, Array<String>> ent;
83 ent.Set(Expr(), Array<String>{"global"});
84 storage_map.Set(GetRef<Expr>(cs.first), ent);
85 }
86 }
87
88 // initial algo assumes mapping of outputs of the expr that is not enough, need to update
89 // VirtualDevice for function variables to get proper codegen. Adding vars to storage_map
90 for (const auto& a : storage_info.args_to_vars_) {
91 if (storage_map.count(a.first)) {
92 for (const auto& v : a.second) {
93 storage_map.Set(v, storage_map[a.first]);
94 if (storage_map[a.first][Expr()][0] == "global" &&
95 storage_info.accept_textures_.count(v)) {
96 Map<Expr, Array<String>> ent;
97 ent.Set(Expr(), storage_info.accept_textures_[v][Expr()]);
98 storage_map.Set(v, ent);
99 for (const auto& calls : storage_info.accept_textures_[v]) {
100 if (calls.first != Expr()) {
101 if (storage_map.count(a.first)) {
102 Map<Expr, Array<String>> ent_call = storage_map[a.first];
103 ent_call.Set(calls.first, calls.second);
104 storage_map.Set(a.first, ent_call);
105 } else {
106 Map<Expr, Array<String>> ent_call;
107 ent_call.Set(calls.first, calls.second);
108 storage_map.Set(a.first, ent_call);
109 }
110 }
111 }
112 }
113 }
114 }
115 }
116 return storage_map;
117 }
118
119 private:
120 void Visit(const Expr& expr) {
121 // Pre-order traversal to enable upward propagation
122 // of consumer storage scopes to producers when desirable.
123 if (const auto* fn = expr.as<FunctionNode>()) {
124 this->VisitExpr(fn->body);
125 for (const auto& param : fn->params) {
126 this->VisitExpr(param);
127 }
128 } else {
129 this->VisitExpr(expr);
130 }
131 }

137 void DeviceAwareVisitExpr_(const FunctionNode* function_node) final {
138 if (!function_node->HasNonzeroAttr(attr::kPrimitive)) {
139 for (auto&& param : function_node->params) {
140 auto virtual_device = GetVirtualDevice(param);
141 param->virtual_device_ =
142 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
143 virtual_device->target, "global");
144 }
145 }
146 transform::DeviceAwareExprVisitor::DeviceAwareVisitExpr_(function_node);

148
149 void DeviceAwareVisitExpr_(const CallNode* call) final {
150 // Check the contents of this primitive function
151 if (const auto* fn = call->op.as<FunctionNode>()) {
152 if (fn->HasNonzeroAttr(attr::kPrimitive)) {
153 primitive_supports_texture_ = false;
154 Visit(call->op);
155 if (primitive_supports_texture_) {
156 if (call->checked_type().as<TensorTypeNode>()) {
157 std::string scope = "global.texture";
158 if (const auto* ttype = call->checked_type().as<TensorTypeNode>()) {
159 scope = Scope(ttype->shape, GetVirtualDevice(GetRef<Expr>(call)));
160 }
161 storage_scope_[call].push_back(scope);
162 } else {
163 const auto* tuple_type = call->type_as<TupleTypeNode>();
164 ICHECK(tuple_type);
165 // TODO(csullivan): Add support for mixed output storage scope.
166 // In current adreno storage planner all outputs of a
167 // primitive function are assumed to be of the same storage
168 // type. This should be easy to extend in the future.
169 for (size_t i = 0; i < tuple_type->fields.size(); i++) {
170 storage_scope_[call].push_back("global.texture");
171 }
172 }
173 for (size_t i = 0; i < fn->params.size(); i++) {
174 args_to_vars_[call->args[i]].push_back(fn->params[i]);
175 // adding info about arguments if they can be converted to texture
176 for (const auto& ttype : FlattenTupleType(fn->params[i]->checked_type())) {
177 std::string scope = Scope(ttype->shape, GetVirtualDevice(GetRef<Expr>(call)));
178 if (scope.find("global.texture") != std::string::npos) {
179 if (accept_textures_.count(fn->params[i])) {
180 Map<Expr, Array<String>> ent = accept_textures_[fn->params[i]];
181 ent.Set(GetRef<Expr>(call), Array<String>{scope});
182 ent.Set(Expr(), Array<String>{scope});
183 accept_textures_.Set(fn->params[i], ent);
184 } else {
185 Map<Expr, Array<String>> ent;
186 ent.Set(GetRef<Expr>(call), Array<String>{scope});
187 ent.Set(Expr(), Array<String>{scope});
188 accept_textures_.Set(fn->params[i], ent);
189 }
190 }
191 }
192 }
193 }
194 // Add consumer storage scope information for call arguments
195 for (auto& arg : call->args) {
196 if (storage_scope_.count(call)) {
197 ICHECK(!HasMixedStorageOutputs(call))
198 << "Mixed output storage scopes are not currently supported";
199 consumer_storage_scopes_[arg.operator->()].push_back("global.texture");
200 } else {
201 consumer_storage_scopes_[arg.operator->()].push_back("global");
202 }
203 }
204 }
205 }
206
207 primitive_supports_texture_ = SupportsTextureStorage(call);
208
209 for (auto& arg : call->args) {
210 Visit(arg);
211 }
212 // We have all callees filled into storage_scope_ if they support textures
213 // We need to verify if this call expects texture and if it does not, remove from
214 // storage_scope_ since initially storage_scope_ is filled only based on knowledge
215 // that function able to work with textures, but not necessary that this texture is
216 // expected by function callee
217 for (auto& arg : call->args) {
218 if (consumer_storage_scopes_.count(arg.operator->()) &&
219 GetConsumerScope(consumer_storage_scopes_[arg.operator->()]) != "global.texture") {
220 storage_scope_.erase(arg.operator->());
221 }
222 }
223 }

225 /**
226 * Defines the name of the memory scope which can fit the tensor of required shape
227 *
228 * The scope stands for "global" if tensor does not satisfy current flattening rules for textures
229 * (texture currently has to be 5d tensors with value eq 4 in the last dimension)
230 *
231 * The packing layout inside the texture scope (the part after the dash) is defined
232 * during the shape itself. Hardware can have limitations on the texture spatial dimensions
233 * we must not exceed these sizes. In addition to the fitting of h/w limitation we want to
234 * get balanced packing where final spatial sizes of textures will not be too different
235 * @param shape shape to be analyzed
236 * @param vd VirtualDevice for the tensors determined of memory scope
237 * @return string representing memory scope either "global" or "global.texture-layout"
238 */
239 std::string Scope(Array<PrimExpr> shape, const VirtualDevice& vd) {
240 // currently we support only textures been made from 5d tensors
241 // 5d requirement is not limitation of textures in general, it is limitation how
242 // we are representing memory scopes/layout and flattening of textures in tir
243 if (vd != VirtualDevice::FullyUnconstrained() && shape.size() == 5 &&
244 shape[4].as<IntImmNode>()->value == 4) {
245 std::map<int, std::string> diffs;
246 int limit =
247 vd->target->GetAttr<Integer>("texture_spatial_limit").value_or(Integer(16384))->value;
248 int a0 = shape[0].as<IntImmNode>()->value;
249 int a1 = shape[1].as<IntImmNode>()->value;
250 int a2 = shape[2].as<IntImmNode>()->value;
251 int a3 = shape[3].as<IntImmNode>()->value;
252
253 int d3l = a0 * a1 * a2;
254 int d3r = a3;
255 int diff3 = d3l > d3r ? d3l - d3r : d3r - d3l;
256 if (d3l < limit && d3r < limit) diffs[diff3] = "";
257
258 int d2l = a0 * a1;
259 int d2r = a2 * a3;
260 int diff2 = d2l > d2r ? d2l - d2r : d2r - d2l;
261 if (d2l < limit && d2r < limit) diffs[diff2] = "nhwc";
262
263 int d1l = a0;
264 int d1r = a1 * a2 * a3;
265 int diff1 = d1l > d1r ? d1l - d1r : d1r - d1l;
266 if (d1l < limit && d1r < limit) diffs[diff1] = "weight";
267 if (!diffs.empty()) {
268 std::string scope = "global.texture";
269 if (!diffs.begin()->second.empty()) {
270 scope += ("-" + diffs.begin()->second);
271 }
272 return scope;
273 }
274 }
275 return "global";
276 }

278 void ApplyConsumerScopeToInputs(const ExprNode* expr) {
279 std::string scope;
280 auto consumer_scopes_it = consumer_storage_scopes_.find(expr);
281 if (consumer_scopes_it != consumer_storage_scopes_.end()) {
282 std::string consumer_scope = GetConsumerScope(consumer_scopes_it->second);
283 ICHECK(!storage_scope_.count(expr))
284 << "Already propagated consumer scopes to input: " << GetRef<Expr>(expr);
285
286 bool expr_is_rgba_vectorizable = false;
287 if (const auto* ttype = expr->checked_type().as<TensorTypeNode>()) {
288 scope = Scope(ttype->shape, GetVirtualDevice(GetRef<Expr>(expr)));
289 if (scope != "global") {
290 auto inner_dim = ttype->shape.back().as<IntImmNode>();
291 if (inner_dim && inner_dim->value == 4) {
292 expr_is_rgba_vectorizable = true;
293 }
294 }
295 }
296
297 // Only propagate texture scope from consumers to input expr if
298 // the input shape of the input expr is rgba vectorizable.
299 if (consumer_scope.find("global.texture") != std::string::npos) {
300 if (expr_is_rgba_vectorizable) {
301 storage_scope_[expr].push_back(scope);
302 }
303 } else {
304 storage_scope_[expr].push_back(consumer_scope);
305 }
306 }
307 }
308
309 void LegalizeProducerStorage() {
310 for (auto& kv : consumer_storage_scopes_) {
311 const ExprNode* producer = kv.first;
312 std::string legal_scope = GetConsumerScope(kv.second);
313 if (storage_scope_.count(producer)) {
314 ICHECK(!HasMixedStorageOutputs(producer))
315 << "Mixed output storage scopes are not currently supported";
316 if (storage_scope_[producer][0].find(legal_scope) == std::string::npos) {
317 for (size_t i = 0; i < storage_scope_[producer].size(); i++) {
318 // Only support uniform storage scope across all outputs for now
319 storage_scope_[producer][i] = legal_scope;
320 }
321 }
322 }
323 }
324 }
325
326 std::string GetConsumerScope(const std::vector<std::string>& consumer_scopes) const {
327 if (!consumer_scopes.size()) {
328 return "global";
329 }
330 std::string texture_tag = "global.texture";
331 for (auto& consumer_scope : consumer_scopes) {
332 if (consumer_scope.find(texture_tag) == std::string::npos) {
333 return "global";
334 }
335 }
336 return texture_tag;
337 }
338
339 bool CanConsumeTextures(const std::vector<std::string>& consumer_scopes) const {
340 std::string texture_tag = "global.texture";
341 for (auto& consumer_scope : consumer_scopes) {
342 if (consumer_scope.find(texture_tag) == 0) {
343 return true;
344 }
345 }
346 return false;
347 }
348
349 bool HasMixedStorageOutputs(const ExprNode* expr) {
350 if (storage_scope_.count(expr)) {
351 std::string ref_scope = storage_scope_[expr][0];
352 for (std::string& scope : storage_scope_[expr]) {
353 if (scope != ref_scope) {
354 return true;
355 }
356 }
357 }
358 return false;
359 }
360
361 bool SupportsTextureStorage(const CallNode* call) const {
362 bool supports_texture_storage = false;
363 if (auto attrs = call->attrs.as<Conv2DAttrs>()) {
364 if (attrs->data_layout == "NCHW4c" && attrs->kernel_layout == "OIHW4o") {
365 supports_texture_storage = true;
366 } else if (attrs->data_layout == "NHWC4c" &&
367 (attrs->kernel_layout == "HWOI4o" || attrs->kernel_layout == "HWIO4o" ||
368 attrs->kernel_layout == "OIHW4o")) {
369 supports_texture_storage = true;
370 }
371 } else if (auto attrs = call->attrs.as<Conv2DWinogradAttrs>()) {
372 if ((attrs->data_layout == "NCHW4c" || attrs->data_layout == "NHWC4c") &&
373 (attrs->kernel_layout == "OIHW4o" || attrs->kernel_layout == "HWIO4o")) {
374 supports_texture_storage = true;
375 }
376 } else if (auto attrs = call->attrs.as<GlobalPool2DAttrs>()) {
377 if (attrs->layout == "NCHW4c") {
378 supports_texture_storage = true;
379 }
380 } else if (auto attrs = call->attrs.as<MaxPool2DAttrs>()) {
381 if (attrs->layout == "NCHW4c") {
382 supports_texture_storage = true;
383 }
384 } else if (auto attrs = call->attrs.as<AvgPool2DAttrs>()) {
385 if (attrs->layout == "NCHW4c") {
386 supports_texture_storage = true;
387 }
388 } else if (const OpNode* opnode = call->op.as<OpNode>()) {
389 auto fpattern = Op::GetAttrMap<TOpPattern>("TOpPattern");
390 auto pattern = fpattern[GetRef<Op>(opnode)];
391 if (pattern <= kInjective) {
392 if (const auto* ttype = call->checked_type().as<TensorTypeNode>()) {
393 if (ttype->shape.size() == 5) {
394 supports_texture_storage = true;
395 }
396 }
397 }
398 }
399
400 return supports_texture_storage;
401 }

403 /*! \brief Temporary state for marking whether a visited function
404 * primitive supports texture storage scope */
405 bool primitive_supports_texture_ = false;
406 /*! \brief expr storage scope mapping for each output */
407 std::unordered_map<const ExprNode*, std::vector<std::string>> storage_scope_;
408 /*! \brief output storage scopes used by consumers of expr key */
409 std::unordered_map<const ExprNode*, std::vector<std::string>> consumer_storage_scopes_;
410 /*! \brief mapping of arguments to call to function variables*/
411 std::unordered_map<Expr, std::vector<Var>, ObjectPtrHash, ObjectPtrEqual> args_to_vars_;
412 /*! \brief mapping of arguments that can be converted to texture*/
413 Map<Expr, Map<Expr, Array<String>>> accept_textures_;
414};

418/**
419 * @brief rewrite of virtual devices, memory_scope part for expressions defined
420 * by the StorageInfo analysis pass
421 *
422 * Currently this workflow supports analysis and rewriting of VirtualDevice for
423 * Constants and function Variables
424 */
425class RewriteVDStorageScopes : public transform::DeviceAwareExprMutator {
426 using VarMap = std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual>;
427
428 public:
429 explicit RewriteVDStorageScopes(const Map<Expr, Map<Expr, Array<String>>>& storage_scope)
430 : transform::DeviceAwareExprMutator(Optional<IRModule>()), storage_scope_(storage_scope) {}
431
432 Function Rewrite(const Expr& expr) { return Downcast<Function>(Mutate(expr)); }
433
434 Expr VisitExpr_(const VarNode* vn) final {
435 if (storage_scope_.find(GetRef<Expr>(vn)) != storage_scope_.end() &&
436 storage_scope_[GetRef<Expr>(vn)].find(Expr()) != storage_scope_[GetRef<Expr>(vn)].end() &&
437 storage_scope_[GetRef<Expr>(vn)][Expr()][0] != "global") {
438 Var c = Var(vn->vid, vn->type_annotation, vn->span);
439 auto virtual_device = GetVirtualDevice(GetRef<Expr>(vn));
440 c->virtual_device_ =
441 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
442 virtual_device->target, storage_scope_[GetRef<Expr>(vn)][Expr()][0]);
443 return c;
444 }
445 return GetRef<Var>(vn);
446 }
447
448 Expr VisitExpr_(const ConstantNode* vn) final {
449 if (storage_scope_.find(GetRef<Expr>(vn)) != storage_scope_.end() &&
450 storage_scope_[GetRef<Expr>(vn)].find(Expr()) != storage_scope_[GetRef<Expr>(vn)].end()) {
451 Expr c = Constant(vn->data, vn->span);
452 auto virtual_device = GetVirtualDevice(GetRef<Expr>(vn));
453 c = OnDevice(
454 c,
455 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
456 virtual_device->target, storage_scope_[GetRef<Expr>(vn)][Expr()][0]),
457 true);
458 return c;
459 }
460 return GetRef<Constant>(vn);
461 }
462
463 Expr DeviceAwareVisitExpr_(const CallNode* call_node) final {
464 // we need to duplicate ExprMutator::VisitExpr_ to correct argument scopes and
465 // put device_copy
466 auto new_op = this->Mutate(call_node->op);
467
468 tvm::Array<Type> ty_args;
469 ty_args.reserve(call_node->type_args.size());
470
471 for (auto ty_arg : call_node->type_args) {
472 auto new_ty_arg = this->VisitType(ty_arg);
473 ty_args.push_back(new_ty_arg);
474 }
475
476 tvm::Array<Expr> call_args;
477 call_args.reserve(call_node->args.size());
478 for (auto arg : call_node->args) {
479 auto new_arg = this->Mutate(arg);
480 // verification if we need to put device_copy
481 if (storage_scope_.count(arg) && storage_scope_[arg].count(GetRef<Expr>(call_node))) {
482 auto virtual_device = GetVirtualDevice(GetRef<Expr>(call_node));
483 VirtualDevice virtual_device_from =
484 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
485 virtual_device->target, virtual_device->memory_scope);
486 VirtualDevice virtual_device_to =
487 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
488 virtual_device->target, storage_scope_[arg][GetRef<Expr>(call_node)][0]);
489 new_arg = DeviceCopy(new_arg, virtual_device_from, virtual_device_to);
490 new_arg = OnDevice(
491 new_arg,
492 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
493 virtual_device->target, storage_scope_[arg][GetRef<Expr>(call_node)][0]),
494 true);
495 }
496 call_args.push_back(new_arg);
497 }
498
499 auto new_call = WithFields(GetRef<Call>(call_node), new_op, call_args, {}, ty_args);
500
501 auto virtual_device = GetVirtualDevice(GetRef<Expr>(call_node));
502 std::string memory_scope = "";
503 if (storage_scope_.find(GetRef<Expr>(call_node)) != storage_scope_.end() &&
504 storage_scope_[GetRef<Expr>(call_node)].find(Expr()) !=
505 storage_scope_[GetRef<Expr>(call_node)].end()) {
506 memory_scope = storage_scope_[GetRef<Expr>(call_node)][Expr()][0];
507 } else if (virtual_device->memory_scope != "") {
508 memory_scope = virtual_device->memory_scope;
509 } else if (!call_node->op.as<FunctionNode>()) {
510 memory_scope = "";
511 }
512 if (!memory_scope.empty()) {
513 new_call =
514 OnDevice(new_call,
515 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
516 virtual_device->target, memory_scope),
517 true);
518 }
519 return new_call;
520 }
521
522 private:
523 Map<Expr, Map<Expr, Array<String>>> storage_scope_;
524 VarMap new_vars_;
525 Array<String> current_function_scope_;
526};

532/**
533 * @brief Collects all target devices participated in graph
534 */
535class CollectVirtualDevices : public transform::DeviceAwareExprVisitor {
536 public:
537 CollectVirtualDevices() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
538 /**
539 * @brief Get all unique device elements from target of each VirtualDevice
540 *
541 * @param expr - IR
542 * @return set of devices
543 */
544 std::set<std::string> GetDevices(const Expr& expr) {
545 this->Run(expr);
546 return std::move(devices_);
547 }
548
549 void Visit(const Expr& expr) {
550 // Pre-order traversal to enable upward propagation
551 // of consumer storage scopes to producers when desirable.
552 if (const auto* fn = expr.as<FunctionNode>()) {
553 this->VisitExpr(fn->body);
554 for (const auto& param : fn->params) {
555 this->VisitExpr(param);
556 }
557 } else {
558 this->VisitExpr(expr);
559 }
560 }
561
562 void DeviceAwareVisitExpr_(const CallNode* call) final {
563 auto vd = GetVirtualDevice(GetRef<Expr>(call));
564 if (vd != VirtualDevice::FullyUnconstrained()) {
565 if (Optional<String> t_device = vd->target->GetAttr<String>("device")) {
566 devices_.insert(vd->target->kind->name + "." + t_device.value());
567 }
568 }
569 for (auto& arg : call->args) {
570 Visit(arg);
571 }
572 }
573
574 void Run(const Expr& expr) { VisitExpr(expr); }
575 using transform::DeviceAwareExprVisitor::VisitExpr_;
576 std::set<std::string> devices_;
577};

579/*!
580 * \brief Collect the target specific tensor storage info for each expression's output.
581 * \param expr The expression.
582 * \return The device based storage mapping.
583 */
584Map<Expr, Map<Expr, Array<String>>> CollectStorageInfo(const Expr& expr) {
585 std::set<std::string> device_types = CollectVirtualDevices().GetDevices(expr);
586 // TODO(amalyshe): current approach collects all targets withing graph and call the only
587 // function corresponding to all these targets in alphabetic order
588 // this will work reliable only for case of only one device and should be redesigned
589 // to handle common case
590 std::string ftarget_prefix = "relay.backend";
591 for (auto& dev_id : device_types) {
592 ftarget_prefix += (std::string(".") + dev_id);
593 }
594
595 Map<Expr, Map<Expr, Array<String>>> storage_info = {};
596 if (const auto* f = runtime::Registry::Get(ftarget_prefix + "._CollectStorageInfo")) {
597 storage_info = (*f)(expr);
598 }
599 return storage_info;
600}
601
602Expr AnnotateMemoryScopeExpr(const Expr& expr, const IRModule& mod, CompilationConfig config) {
603 auto storage_scope = CollectStorageInfo(expr);
604 if (storage_scope.size()) {
605 return RewriteVDStorageScopes(storage_scope).Rewrite(expr);
606 } else {
607 return expr;
608 }
609}
