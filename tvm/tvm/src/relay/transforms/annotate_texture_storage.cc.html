20/*!
21 * \file annotate_texture_storage.cc
22 * \brief Collection of target specific relay passes which
23 * storage scope related information.
24 *
25 * - CollectStorageInfo returns a mapping from relay expr
26 * to a list of output storage scopes for each output.
27 * These scopes are used during memory planning as well
28 * as downstream when doing codegen and in the graph runtime when doing runtime dataspace
29 * allocations.
30 *
31 * - AnnotateMemoryScope calls *target.CollectStorageInfo for all target been represented
32 * in the graph and rewrites graph modifying or inserting of VirtualDevice with required
33 * memory_scope collected from the CollectStorageInfo
34 */

51/**
52 * @brief Analyzes the graph and returns mapping of expressions vs desired memory scope
53 */
54class StorageInfo : private transform::DeviceAwareExprVisitor {
55 public:
56 StorageInfo() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
57
58 static Map<Expr, Array<String>> GetStorageMap(const Expr& expr) {
59 StorageInfo storage_info;
60 storage_info.VisitExpr(expr);
61 storage_info.LegalizeProducerStorage();
62 Map<Expr, Array<String>> storage_map;
63 for (auto& kv : storage_info.storage_scope_) {
64 std::vector<String> storage_scopes;
65 std::copy(kv.second.begin(), kv.second.end(), std::back_inserter(storage_scopes));
66 storage_map.Set(GetRef<Expr>(kv.first), Array<String>{storage_scopes});
67 }
68
69 // Filling the input arguments by "global" scope to handle PlanDevice algo which propagates
70 // virtual devices from outputs to inputs. At the same time outputs must be unconstrained
71 // to avoid useless device_copy
72 for (const auto& cs : storage_info.consumer_storage_scopes_) {
73 // we have record in consumers that mean that potentially consumer
74 // dealt with textures anyhow, it's safe to mark this expr as global scope
75 // even without verification of the consumer's outputs scope
76 if (storage_info.CanConsumeTextures(cs.second) &&
77 storage_map.find(GetRef<Expr>(cs.first)) == storage_map.end()) {
78 storage_map.Set(GetRef<Expr>(cs.first), Array<String>{"global"});
79 }
80 }
81
82 // initial algo assumes mapping of outputs of the expr that is not enough, need to update
83 // VirtualDevice for function variables to get proper codegen. Adding vars to storage_map
84 for (const auto& a : storage_info.args_to_vars_) {
85 if (storage_map.count(a.first)) {
86 for (const auto& v : a.second) {
87 storage_map.Set(v, storage_map[a.first]);
88 }
89 }
90 }
91 return storage_map;
92 }
93
94 private:
95 void Visit(const Expr& expr) {
96 // Pre-order traversal to enable upward propagation
97 // of consumer storage scopes to producers when desirable.
98 if (const auto* fn = expr.as<FunctionNode>()) {
99 this->VisitExpr(fn->body);
100 for (const auto& param : fn->params) {
101 this->VisitExpr(param);
102 }
103 } else {
104 this->VisitExpr(expr);
105 }
106 }
107
108 void VisitExpr_(const VarNode* vn) final { ApplyConsumerScopeToInputs(vn); }
109
110 void VisitExpr_(const ConstantNode* cn) final { ApplyConsumerScopeToInputs(cn); }
111
112 void DeviceAwareVisitExpr_(const CallNode* call) final {
113 // Check the contents of this primitive function
114 if (const auto* fn = call->op.as<FunctionNode>()) {
115 if (fn->HasNonzeroAttr(attr::kPrimitive)) {
116 primitive_supports_texture_ = false;
117 Visit(call->op);
118 if (primitive_supports_texture_) {
119 if (call->checked_type().as<TensorTypeNode>()) {
120 std::string scope = "global.texture";
121 if (const auto* ttype = call->checked_type().as<TensorTypeNode>()) {
122 scope = Scope(ttype->shape, GetVirtualDevice(GetRef<Expr>(call)));
123 }
124 storage_scope_[call].push_back(scope);
125 } else {
126 const auto* tuple_type = call->type_as<TupleTypeNode>();
127 ICHECK(tuple_type);
128 // TODO(csullivan): Add support for mixed output storage scope.
129 // In current adreno storage planner all outputs of a
130 // primitive function are assumed to be of the same storage
131 // type. This should be easy to extend in the future.
132 for (size_t i = 0; i < tuple_type->fields.size(); i++) {
133 storage_scope_[call].push_back("global.texture");
134 }
135 }
136 for (size_t i = 0; i < fn->params.size(); i++) {
137 args_to_vars_[call->args[i]].push_back(fn->params[i]);
138 }
139 }
140 // Add consumer storage scope information for call arguments
141 for (auto& arg : call->args) {
142 if (storage_scope_.count(call)) {
143 ICHECK(!HasMixedStorageOutputs(call))
144 << "Mixed output storage scopes are not currently supported";
145 consumer_storage_scopes_[arg.operator->()].push_back("global.texture");
146 } else {
147 consumer_storage_scopes_[arg.operator->()].push_back("global");
148 }
149 }
150 }
151 }
152
153 primitive_supports_texture_ = SupportsTextureStorage(call);
154
155 for (auto& arg : call->args) {
156 Visit(arg);
157 }
158 // We have all callees filled into storage_scope_ if they support textures
159 // We need to verify if this call expects texture and if it does not, remove from
160 // storage_scope_ since initially storage_scope_ is filled only based on knowledge
161 // that function able to work with textures, but not necessary that this texture is
162 // expected by function callee
163 for (auto& arg : call->args) {
164 if (consumer_storage_scopes_.count(arg.operator->()) &&
165 GetConsumerScope(consumer_storage_scopes_[arg.operator->()]) != "global.texture") {
166 storage_scope_.erase(arg.operator->());
167 if (const auto* cn = arg.as<CallNode>()) {
168 if (const auto* fn = cn->op.as<FunctionNode>()) {
169 storage_scope_.erase(fn->body.operator->());
170 }
171 }
172 }
173 }
174 }
175
176 /**
177 * Defines the name of the memory scope which can fit the tensor of required shape
178 *
179 * The scope stands for "global" if tensor does not satisfy current flattening rules for textures
180 * (texture currently has to be 5d tensors with value eq 4 in the last dimension)
181 *
182 * The packing layout inside the texture scope (the part after the dash) is defined
183 * during the shape itself. Hardware can have limitations on the texture spatial dimensions
184 * we must not exceed these sizes. In addition to the fitting of h/w limitation we want to
185 * get balanced packing where final spatial sizes of textures will not be too different
186 * @param shape shape to be analyzed
187 * @param vd VirtualDevice for the tensors determined of memory scope
188 * @return string representing memory scope either "global" or "global.texture-layout"
189 */
190 std::string Scope(Array<PrimExpr> shape, const VirtualDevice& vd) {
191 // currently we support only textures been made from 5d tensors
192 // 5d requirement is not limitation of textures in general, it is limitation how
193 // we are representing memory scopes/layout and flattening of textures in tir
194 if (vd != VirtualDevice::FullyUnconstrained() && shape.size() == 5 &&
195 shape[4].as<IntImmNode>()->value == 4) {
196 std::map<int, std::string> diffs;
197 int limit =
198 vd->target->GetAttr<Integer>("texture_spatial_limit").value_or(Integer(16384))->value;
199 int a0 = shape[0].as<IntImmNode>()->value;
200 int a1 = shape[1].as<IntImmNode>()->value;
201 int a2 = shape[2].as<IntImmNode>()->value;
202 int a3 = shape[3].as<IntImmNode>()->value;
203
204 int d3l = a0 * a1 * a2;
205 int d3r = a3;
206 int diff3 = d3l > d3r ? d3l - d3r : d3r - d3l;
207 if (d3l < limit && d3r < limit) diffs[diff3] = "";
208
209 int d2l = a0 * a1;
210 int d2r = a2 * a3;
211 int diff2 = d2l > d2r ? d2l - d2r : d2r - d2l;
212 if (d2l < limit && d2r < limit) diffs[diff2] = "nhwc";
213
214 int d1l = a0;
215 int d1r = a1 * a2 * a3;
216 int diff1 = d1l > d1r ? d1l - d1r : d1r - d1l;
217 if (d1l < limit && d1r < limit) diffs[diff1] = "weight";
218 if (!diffs.empty()) {
219 std::string scope = "global.texture";
220 if (!diffs.begin()->second.empty()) {
221 scope += ("-" + diffs.begin()->second);
222 }
223 return scope;
224 }
225 }
226 return "global";
227 }
228
229 void ApplyConsumerScopeToInputs(const ExprNode* expr) {
230 std::string scope;
231 auto consumer_scopes_it = consumer_storage_scopes_.find(expr);
232 if (consumer_scopes_it != consumer_storage_scopes_.end()) {
233 std::string consumer_scope = GetConsumerScope(consumer_scopes_it->second);
234 ICHECK(!storage_scope_.count(expr))
235 << "Already propagated consumer scopes to input: " << GetRef<Expr>(expr);
236
237 bool expr_is_rgba_vectorizable = false;
238 if (const auto* ttype = expr->checked_type().as<TensorTypeNode>()) {
239 scope = Scope(ttype->shape, GetVirtualDevice(GetRef<Expr>(expr)));
240 if (scope != "global") {
241 auto inner_dim = ttype->shape.back().as<IntImmNode>();
242 if (inner_dim && inner_dim->value == 4) {
243 expr_is_rgba_vectorizable = true;
244 }
245 }
246 }
247
248 // Only propagate texture scope from consumers to input expr if
249 // the input shape of the input expr is rgba vectorizable.
250 if (consumer_scope.find("global.texture") != std::string::npos) {
251 if (expr_is_rgba_vectorizable) {
252 storage_scope_[expr].push_back(scope);
253 }
254 } else {
255 storage_scope_[expr].push_back(consumer_scope);
256 }
257 }
258 }
259
260 void LegalizeProducerStorage() {
261 for (auto& kv : consumer_storage_scopes_) {
262 const ExprNode* producer = kv.first;
263 std::string legal_scope = GetConsumerScope(kv.second);
264 if (storage_scope_.count(producer)) {
265 ICHECK(!HasMixedStorageOutputs(producer))
266 << "Mixed output storage scopes are not currently supported";
267 if (storage_scope_[producer][0].find(legal_scope) == std::string::npos) {
268 for (size_t i = 0; i < storage_scope_[producer].size(); i++) {
269 // Only support uniform storage scope across all outputs for now
270 storage_scope_[producer][i] = legal_scope;
271 }
272 }
273 }
274 }
275 }
276
277 std::string GetConsumerScope(const std::vector<std::string>& consumer_scopes) const {
278 if (!consumer_scopes.size()) {
279 return "global";
280 }
281 std::string texture_tag = "global.texture";
282 for (auto& consumer_scope : consumer_scopes) {
283 if (consumer_scope.find(texture_tag) == std::string::npos) {
284 return "global";
285 }
286 }
287 return texture_tag;
288 }
289
290 bool CanConsumeTextures(const std::vector<std::string>& consumer_scopes) const {
291 std::string texture_tag = "global.texture";
292 for (auto& consumer_scope : consumer_scopes) {
293 if (consumer_scope.find(texture_tag) == 0) {
294 return true;
295 }
296 }
297 return false;
298 }
299
300 bool HasMixedStorageOutputs(const ExprNode* expr) {
301 if (storage_scope_.count(expr)) {
302 std::string ref_scope = storage_scope_[expr][0];
303 for (std::string& scope : storage_scope_[expr]) {
304 if (scope != ref_scope) {
305 return true;
306 }
307 }
308 }
309 return false;
310 }
311
312 bool SupportsTextureStorage(const CallNode* call) const {
313 bool supports_texture_storage = false;
314 if (auto attrs = call->attrs.as<Conv2DAttrs>()) {
315 if (attrs->data_layout == "NCHW4c" && attrs->kernel_layout == "OIHW4o") {
316 supports_texture_storage = true;
317 } else if (attrs->data_layout == "NHWC4c" &&
318 (attrs->kernel_layout == "HWOI4o" || attrs->kernel_layout == "HWIO4o" ||
319 attrs->kernel_layout == "OIHW4o")) {
320 supports_texture_storage = true;
321 }
322 } else if (auto attrs = call->attrs.as<Conv2DWinogradAttrs>()) {
323 if ((attrs->data_layout == "NCHW4c" || attrs->data_layout == "NHWC4c") &&
324 (attrs->kernel_layout == "OIHW4o" || attrs->kernel_layout == "HWIO4o")) {
325 supports_texture_storage = true;
326 }
327 } else if (auto attrs = call->attrs.as<GlobalPool2DAttrs>()) {
328 if (attrs->layout == "NCHW4c") {
329 supports_texture_storage = true;
330 }
331 } else if (auto attrs = call->attrs.as<MaxPool2DAttrs>()) {
332 if (attrs->layout == "NCHW4c") {
333 supports_texture_storage = true;
334 }
335 } else if (auto attrs = call->attrs.as<AvgPool2DAttrs>()) {
336 if (attrs->layout == "NCHW4c") {
337 supports_texture_storage = true;
338 }
339 }
340
341 return supports_texture_storage;
342 }

344 /*! \brief Temporary state for marking whether a visited function
345 * primitive supports texture storage scope */
346 bool primitive_supports_texture_ = false;
347 /*! \brief expr storage scope mapping for each output */
348 std::unordered_map<const ExprNode*, std::vector<std::string>> storage_scope_;
349 /*! \brief output storage scopes used by consumers of expr key */
350 std::unordered_map<const ExprNode*, std::vector<std::string>> consumer_storage_scopes_;
351 /*! \brief mapping of arguments to call to function variables*/
352 std::unordered_map<Expr, std::vector<Var>, ObjectPtrHash, ObjectPtrEqual> args_to_vars_;
353};

357/**
358 * @brief rewrite of virtual devices, memory_scope part for expressions defined
359 * by the StorageInfo analysis pass
360 *
361 * Currently this workflow supports analysis and rewriting of VirtualDevice for
362 * Constants and function Variables
363 */
364class RewriteVDStorageScopes : public transform::DeviceAwareExprMutator {
365 using VarMap = std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual>;
366
367 public:
368 explicit RewriteVDStorageScopes(const Map<Expr, Array<String>>& storage_scope)
369 : transform::DeviceAwareExprMutator(Optional<IRModule>()), storage_scope_(storage_scope) {}
370
371 Function Rewrite(const Expr& expr) { return Downcast<Function>(Mutate(expr)); }
372
373 Expr VisitExpr_(const VarNode* vn) final {
374 if (storage_scope_.find(GetRef<Expr>(vn)) != storage_scope_.end() &&
375 storage_scope_[GetRef<Expr>(vn)][0] != "global") {
376 Var c = Var(vn->vid, vn->type_annotation, vn->span);
377 auto virtual_device = GetVirtualDevice(GetRef<Expr>(vn));
378 c->virtual_device_ =
379 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
380 virtual_device->target, storage_scope_[GetRef<Expr>(vn)][0]);
381 return c;
382 }
383 return GetRef<Var>(vn);
384 }
385
386 Expr VisitExpr_(const ConstantNode* vn) final {
387 if (storage_scope_.find(GetRef<Expr>(vn)) != storage_scope_.end()) {
388 Expr c = Constant(vn->data, vn->span);
389 auto virtual_device = GetVirtualDevice(GetRef<Expr>(vn));
390 c = OnDevice(c,
391 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
392 virtual_device->target, storage_scope_[GetRef<Expr>(vn)][0]),
393 true);
394 return c;
395 }
396 return GetRef<Constant>(vn);
397 }
398
399 Expr DeviceAwareVisitExpr_(const CallNode* call_node) final {
400 auto new_call = transform::DeviceAwareExprMutator::DeviceAwareVisitExpr_(call_node);
401 auto virtual_device = GetVirtualDevice(GetRef<Expr>(call_node));
402 std::string memory_scope = "";
403 if (storage_scope_.find(GetRef<Expr>(call_node)) != storage_scope_.end()) {
404 memory_scope = storage_scope_[GetRef<Expr>(call_node)][0];
405 } else if (virtual_device->memory_scope != "") {
406 memory_scope = virtual_device->memory_scope;
407 } else if (!call_node->op.as<FunctionNode>()) {
408 memory_scope = "";
409 }
410 if (!memory_scope.empty()) {
411 new_call =
412 OnDevice(new_call,
413 VirtualDevice(virtual_device->device_type(), virtual_device->virtual_device_id,
414 virtual_device->target, memory_scope),
415 true);
416 }
417 return new_call;
418 }
419
420 private:
421 Map<Expr, Array<String>> storage_scope_;
422 VarMap new_vars_;
423 Array<String> current_function_scope_;
424};

430/**
431 * @brief Collects all target devices participated in graph
432 */
433class CollectVirtualDevices : public transform::DeviceAwareExprVisitor {
434 public:
435 CollectVirtualDevices() : transform::DeviceAwareExprVisitor(Optional<IRModule>()) {}
436 /**
437 * @brief Get all unique device elements from target of each VirtualDevice
438 *
439 * @param expr - IR
440 * @return set of devices
441 */
442 std::set<std::string> GetDevices(const Expr& expr) {
443 this->Run(expr);
444 return std::move(devices_);
445 }
446
447 void Visit(const Expr& expr) {
448 // Pre-order traversal to enable upward propagation
449 // of consumer storage scopes to producers when desirable.
450 if (const auto* fn = expr.as<FunctionNode>()) {
451 this->VisitExpr(fn->body);
452 for (const auto& param : fn->params) {
453 this->VisitExpr(param);
454 }
455 } else {
456 this->VisitExpr(expr);
457 }
458 }
459
460 void DeviceAwareVisitExpr_(const CallNode* call) final {
461 auto vd = GetVirtualDevice(GetRef<Expr>(call));
462 if (vd != VirtualDevice::FullyUnconstrained()) {
463 if (Optional<String> t_device = vd->target->GetAttr<String>("device")) {
464 devices_.insert(vd->target->kind->name + "." + t_device.value());
465 }
466 }
467 for (auto& arg : call->args) {
468 Visit(arg);
469 }
470 }
471
472 void Run(const Expr& expr) { VisitExpr(expr); }
473 using transform::DeviceAwareExprVisitor::VisitExpr_;
474 std::set<std::string> devices_;
475};

477/*!
478 * \brief Collect the target specific tensor storage info for each expression's output.
479 * \param expr The expression.
480 * \return The device based storage mapping.
481 */
482Map<Expr, Array<String>> CollectStorageInfo(const Expr& expr) {
483 std::set<std::string> device_types = CollectVirtualDevices().GetDevices(expr);
484 // TODO(amalyshe): current approach collects all targets withing graph and call the only
485 // function corresponding to all these targets in alphabetic order
486 // this will work reliable only for case of only one device and should be redesigned
487 // to handle common case
488 std::string ftarget_prefix = "relay.backend";
489 for (auto& dev_id : device_types) {
490 ftarget_prefix += (std::string(".") + dev_id);
491 }
492
493 Map<Expr, Array<String>> storage_info = {};
494 if (const auto* f = runtime::Registry::Get(ftarget_prefix + "._CollectStorageInfo")) {
495 storage_info = (*f)(expr);
496 }
497 return storage_info;
498}
499
500Expr AnnotateMemoryScopeExpr(const Expr& expr, const IRModule& mod, CompilationConfig config) {
501 auto storage_scope = CollectStorageInfo(expr);
502 if (storage_scope.size()) {
503 return RewriteVDStorageScopes(storage_scope).Rewrite(expr);
504 } else {
505 return expr;
506 }
507}

509namespace transform {
510tvm::transform::Pass AnnotateMemoryScope(CompilationConfig config) {
511 runtime::TypedPackedFunc<Function(Function, IRModule, PassContext)> pass_func =
512 [config = std::move(config)](Function f, IRModule m, PassContext pc) {
513 return Downcast<Function>(AnnotateMemoryScopeExpr(f, m, config));
514 };
515 return CreateFunctionPass(pass_func, 2, "AnnotateMemoryScope", {});
516}
