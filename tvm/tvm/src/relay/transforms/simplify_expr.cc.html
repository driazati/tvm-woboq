45/*!
46 * \brief SimplifyReshape matches the pattern of consecutive reshape or reverse_reshape ops,
47 * and merges into one reshape op.
48 */
49class SimplifyReshape : public DFPatternRewrite {
50 public:
51 SimplifyReshape() {
52 x_ = IsWildcard();
53 auto reshape1 = IsOp("reshape") || IsOp("contrib_reverse_reshape");
54 auto reshape2 = IsOp("reshape") || IsOp("contrib_reverse_reshape");
55 pattern_ = reshape1({reshape2({x_})});
56 }
57
58 Expr Callback(const Expr& pre, const Expr& post,
59 const Map<DFPattern, Array<Expr>>& node_map) const override {
60 auto x = node_map[x_][0];
61 bool const_shape = true;
62 Array<Integer> newshape;
63 for (auto dim : Downcast<TensorType>(pre->checked_type())->shape) {
64 if (dim.as<IntImmNode>() == nullptr) {
65 const_shape = false;
66 break;
67 }
68 newshape.push_back(Downcast<Integer>(dim));
69 }
70 if (const_shape) {
71 return MakeReshape(x, newshape);
72 }
73 return post;
74 }
75
76 private:
77 /*! \brief Pattern input */
78 DFPattern x_;
79};

81/*!
82 * \brief SimplifySameCast matches the pattern of cast data to the same dtype.
83 */
84class SimplifySameCast : public DFPatternRewrite {
85 public:
86 SimplifySameCast() {
87 data_pat_ = IsWildcard();
88 like_pat_ = IsWildcard();
89 pattern_ = IsOp("cast_like")({data_pat_, like_pat_}) || IsOp("cast")({data_pat_});
90 }
91
92 Expr Callback(const Expr& pre, const Expr& post,
93 const Map<DFPattern, Array<Expr>>& node_map) const override {
94 const CallNode* call = pre.as<CallNode>();
95 const TensorTypeNode* data_ty = call->args[0]->checked_type().as<TensorTypeNode>();
96 const TensorTypeNode* like_ty = pre->checked_type().as<TensorTypeNode>();
97 if (like_ty->dtype == data_ty->dtype) {
98 return node_map[data_pat_][0];
99 }
100 return post;
101 }
102
103 protected:
104 DFPattern data_pat_;
105 DFPattern like_pat_;

108/*!
109 * \brief SimplifyConsecutiveCast matches the pattern of consecutive cast/cast_like ops
110 */
111class SimplifyConsecutiveCast : public DFPatternRewrite {
112 public:
113 SimplifyConsecutiveCast() {
114 data_ = IsWildcard();
115 cast1_ = IsOp("cast_like")({data_, IsWildcard()}) || IsOp("cast")({data_});
116 pattern_ = IsOp("cast_like")({cast1_, IsWildcard()}) || IsOp("cast")({cast1_});
117 }
118
119 Expr Callback(const Expr& pre, const Expr& post,
120 const Map<DFPattern, Array<Expr>>& node_map) const override {
121 auto data = node_map[data_][0];
122 auto cast1 = Downcast<Call>(node_map[cast1_][0]);
123 auto data_type = Downcast<TensorType>(data->checked_type());
124 DataType cast1_dtype = Downcast<TensorType>(cast1->checked_type())->dtype;
125
126 if (!IsWidenCast(data_type->dtype, cast1_dtype)) {
127 // Cannot remove the narrow cast
128 return post;
129 }
130
131 const CallNode* cast2 = post.as<CallNode>();
132 DataType cast2_dtype = Downcast<TensorType>(cast2->checked_type())->dtype;
133 auto expr = MakeCast(data, cast2_dtype);
134
135 // We need to set the checked type as it may be needed in the next callback
136 expr->checked_type_ = TensorType(data_type->shape, cast2_dtype);
137 return expr;
138 }
139
140 bool IsWidenCast(DataType origin, DataType cast) const {
141 /* Return whether casting from origin to cast results in more or the same precision.*/
142 if (origin.code() == cast.code() && origin.bits() <= cast.bits()) {
143 return true;
144 }
145 if (origin.code() == DataType::kBFloat || cast.code() == DataType::kBFloat) {
146 // BFloat cast cannot be omitted
147 return false;
148 }
149 if (origin.code() < cast.code() && origin.bits() <= cast.bits()) {
150 // Loosely have a hiearchy to datatypes
151 // e.g. int --> uint --> float has increasing range of numbers they can represent
152 return true;
153 }
154 return false;
155 }
156
157 protected:
158 DFPattern data_;
159 DFPattern cast1_;
160};
161
162bool CheckDataTypeMaxMinValue(DataType dtype, double min_value, double max_value) {
163 if (dtype.is_int() || dtype.is_uint()) {
164 double ubound = static_cast<double>(Downcast<IntImm>(tvm::max_value(dtype))->value);
165 double lbound = static_cast<double>(Downcast<IntImm>(tvm::min_value(dtype))->value);
166 return ubound == max_value && lbound == min_value;
167 } else if (dtype.is_float()) {
168 double ubound = Downcast<FloatImm>(tvm::max_value(dtype))->value;
169 double lbound = Downcast<FloatImm>(tvm::min_value(dtype))->value;
170 return ubound == max_value && lbound == min_value;
171 }
172
173 return false;
174}

176/*!
177 * \brief SimplifyClipAndConsecutiveCast matches the pattern clip->cast->cast and remove redundant
178 * casts.
179 * Analysis of "redundancy" is done based on clip min/max values and min/max values of casted data
180 * type.
181 */
182class SimplifyClipAndConsecutiveCast : public DFPatternRewrite {
183 public:
184 SimplifyClipAndConsecutiveCast() {
185 clip_ = IsOp("clip")({IsWildcard()});
186 cast1_ = IsOp("cast")({clip_});
187 pattern_ = IsOp("cast")({cast1_});
188 }
189
190 Expr Callback(const Expr& pre, const Expr& post,
191 const Map<DFPattern, Array<Expr>>& node_map) const override {
192 auto clip = Downcast<Call>(node_map[clip_][0]);
193 const CallNode* clip_node = clip.as<CallNode>();
194 const ClipAttrs* clip_attrs = clip_node->attrs.as<ClipAttrs>();
195 DataType clip_dtype = Downcast<TensorType>(clip->checked_type())->dtype;
196
197 auto cast1 = Downcast<Call>(node_map[cast1_][0]);
198 DataType cast1_dtype = Downcast<TensorType>(cast1->checked_type())->dtype;
199
200 auto cast2 = Downcast<Call>(post);
201 DataType cast2_dtype = Downcast<TensorType>(cast2->checked_type())->dtype;
202
203 if (clip_dtype == cast2_dtype &&
204 CheckDataTypeMaxMinValue(cast1_dtype, clip_attrs->a_min, clip_attrs->a_max)) {
205 // Case 1:
206 // Data type of Clip == target data type of second Cast and min/max value of Clip == min/max
207 // value of first Clip target data type. In this case both Clip ops can be removed.
208 // Example:
209 // %0 == [type=int32]
210 // %1 = clip(%0, a_min=0f, a_max=255f) [type=int32]
211 // %2 = cast(%1, dtype="uint8") [type=uint8]
212 // %3 = cast(%2, dtype="int32") [type=int32]
213 //
214 // Optimized to (both casts can be removed):
215 // %1 = clip(%0, a_min=0f, a_max=255f) [type=int32]
216 return node_map[clip_][0];
217 }
218 return post;
219 }
220
221 protected:
222 DFPattern clip_, cast1_;
223};

225/*!
226 * \brief SimplifyCastClip matches the pattern cast->clip and remove redundant Cast based on Clip
227 * min/max values and min/max values of Cast target data type.
228 *
229 * Example:
230 * %1 = cast(%0, dtype="uint8") [type=uint8]
231 * %2 = clip(%1, a_min=0f, a_max=255f) [type=int8]
232 *
233 * Optimized to (remove Clip):
234 * %1 = cast(%0, dtype="uint8") [type=uint8]
235 */
236class SimplifyCastClip : public DFPatternRewrite {
237 public:
238 SimplifyCastClip() {
239 cast_ = IsOp("cast")({IsWildcard()});
240 pattern_ = IsOp("clip")({cast_});
241 }
242
243 Expr Callback(const Expr& pre, const Expr& post,
244 const Map<DFPattern, Array<Expr>>& node_map) const override {
245 auto cast = Downcast<Call>(node_map[cast_][0]);
246 DataType cast_dtype = Downcast<TensorType>(cast->checked_type())->dtype;
247
248 auto clip = Downcast<Call>(post);
249 const CallNode* clip_node = clip.as<CallNode>();
250 const ClipAttrs* clip_attrs = clip_node->attrs.as<ClipAttrs>();
251
252 if (CheckDataTypeMaxMinValue(cast_dtype, clip_attrs->a_min, clip_attrs->a_max)) {
253 return node_map[cast_][0];
254 }
255 return post;
256 }
257
258 protected:
259 DFPattern clip_, cast_;
260};

262/*!
263 * \brief SimplifyTranspose matches the pattern of consecutive transpose op,
264 * and merges or cancels them.
265 */
266class SimplifyTranspose : public DFPatternRewrite {
267 public:
268 SimplifyTranspose() {
269 x_ = IsWildcard();
270 auto trans1 = IsOp("transpose") || IsOp("layout_transform");
271 auto trans2 = IsOp("transpose") || IsOp("layout_transform");
272 pattern_ = trans1({trans2({x_})});
273 }
274
275 Expr Callback(const Expr& pre, const Expr& post,
276 const Map<DFPattern, Array<Expr>>& node_map) const override {
277 auto x = node_map[x_][0];
278
279 Call trans_call = Downcast<Call>(post);
280
281 // Try to fuse any rank changing layout transformations
282 if (auto layout_trans = FoldRankChangingLayoutTrans(x, trans_call)) {
283 if (auto attr = layout_trans.value()->attrs.as<LayoutTransformAttrs>()) {
284 // Prune any trivial layout transformation
285 if (attr->src_layout == attr->dst_layout) {
286 return x;
287 }
288 }
289 return layout_trans.value();
290 }
291
292 // Initialize axes
293 int ndim = Downcast<TensorType>(pre->checked_type())->shape.size();
294 Array<Integer> axes;
295 for (int i = 0; i < ndim; ++i) {
296 axes.push_back(i);
297 }
298
299 // Collect axes changes from the matched pattern, including two consecutive transposes.
300 std::vector<std::vector<int>> interm_axes;
301 interm_axes.push_back(GetTransposeAxisOrder(trans_call, ndim));
302 trans_call = Downcast<Call>(trans_call->args[0]);
303 interm_axes.push_back(GetTransposeAxisOrder(trans_call, ndim));
304
305 // Calculate the final axes in reverse order (from root to output)
306 auto it = interm_axes.rbegin();
307 while (it != interm_axes.rend()) {
308 auto interm = *it;
309
310 Array<Integer> new_axes;
311 for (int i = 0; i < ndim; ++i) {
312 new_axes.push_back(axes[interm[i]]);
313 }
314 axes = new_axes;
315 it++;
316 }
317
318 // Check if the transpose is still required
319 bool need_transpose = false;
320 for (int i = 0; i < ndim; ++i) {
321 if (axes[i] != i) {
322 need_transpose = true;
323 break;
324 }
325 }
326
327 if (need_transpose) {
328 return MakeTranspose(x, axes);
329 }
330 return x;
331 }
332
333 String PermuteLayout(const String& layout, std::vector<int> axes_order) const {
334 std::string new_layout{};
335 std::string old_layout{layout};
336 ICHECK_EQ(axes_order.size(), layout.size())
337 << "Number of axes must match the number of named axes in the layout to permute: length("
338 << old_layout << ") != " << axes_order.size();
339 std::stringstream order;
340 for (auto axis : axes_order) {
341 new_layout += old_layout[axis];
342 order << axis << ", ";
343 }
344 DLOG(INFO) << "Using transpose axes order {" << order.str()
345 << "} to permute layout: " << old_layout << " to " << new_layout;
346 return new_layout;
347 }
348
349 struct RankChangingLayoutDescriptor {
350 Layout src_layout;
351 Layout dst_layout;
352 // Either a rank changing layout transform or a transpose
353 Call other_transform;
354 };
355
356 std::unique_ptr<RankChangingLayoutDescriptor> GetRankChangeDescriptor(const Call& call) const {
357 std::unique_ptr<RankChangingLayoutDescriptor> desc{nullptr};
358 if (auto attr = call->attrs.as<LayoutTransformAttrs>()) {
359 if (attr->src_layout.length() != attr->dst_layout.length()) {
360 desc = std::make_unique<RankChangingLayoutDescriptor>();
361 desc->src_layout = Layout(attr->src_layout);
362 desc->dst_layout = Layout(attr->dst_layout);
363 desc->other_transform = Downcast<Call>(call->args[0]);
364 }
365 }
366 if (auto attr = Downcast<Call>(call->args[0])->attrs.as<LayoutTransformAttrs>()) {
367 if (attr->src_layout.length() != attr->dst_layout.length()) {
368 if (!desc) {
369 desc = std::make_unique<RankChangingLayoutDescriptor>();
370 desc->src_layout = Layout(attr->src_layout);
371 desc->dst_layout = Layout(attr->dst_layout);
372 desc->other_transform = call;
373 } else {
374 ICHECK(desc->src_layout->name == attr->dst_layout)
375 << "Back-to-back layout transforms must have the same intermediate layout: "
376 << desc->src_layout->name << " != " << attr->dst_layout;
377 desc->src_layout = Layout(attr->src_layout);
378 }
379 }
380 }
381 return desc;
382 }

384 /*
385 * \brief Fuse call and it's argument into a single layout_transform operator
386 * when either call or it's argument is a rang changing layout_transform, e.g.,
387 *
388 * Simplify
389 *
390 * [N, H, W, C] -> Transpose -> [N, C, H, W] -> LayoutTrans -> [N, C, H, W, 4c]
391 *
392 * to,
393 *
394 * [N, H, W, C] -> LayoutTrans -> [N, C, H, W, 4c].
395 *
396 * \param The input expression to the matched pattern
397 * \param The pattern root; the second of two consecutive Transpose/LayoutTransform ops
398 */
399 Optional<Call> FoldRankChangingLayoutTrans(const Expr& data, const Call& call) const {
400 // Check to see if either the first or second call in matched pattern
401 // is a rank changing layout transform. If so, return a descriptor containing
402 // the layouts and any additional transpose or layout transform op.
403 auto desc = GetRankChangeDescriptor(call);
404 if (desc == nullptr) {
405 // No rank changing layout transform
406 return Optional<Call>{nullptr};
407 }
408
409 Optional<Expr> output_layout_trans;
410 // Fuse a rank increasing layout transform and a preceeding transpose
411 if (desc->src_layout->axes.size() < desc->dst_layout->axes.size()) {
412 auto axes = GetTransposeAxisOrder(desc->other_transform, desc->src_layout->axes.size());
413 // Calculate the reverse axis order and apply to the source layout
414 std::vector<int> inverse(axes.size());
415 for (size_t i = 0; i < axes.size(); i++) {
416 inverse[axes[i]] = i;
417 }
418 String new_layout = PermuteLayout(desc->src_layout->name, inverse);
419 output_layout_trans = MakeLayoutTransform(data, new_layout, desc->dst_layout->name);
420 // Fuse a rank descreasing layout transform followed by a transpose
421 } else if (desc->src_layout->axes.size() > desc->dst_layout->axes.size()) {
422 auto axes = GetTransposeAxisOrder(desc->other_transform, desc->dst_layout->axes.size());
423 String new_layout = PermuteLayout(desc->dst_layout->name, axes);
424 output_layout_trans = MakeLayoutTransform(data, desc->src_layout->name, new_layout);
425 // Fuse two back-to-back layout transformations which change rank
426 } else if (desc->other_transform->attrs.as<LayoutTransformAttrs>()) {
427 output_layout_trans =
428 MakeLayoutTransform(data, desc->src_layout->name, desc->dst_layout->name);
429 }
430 return Downcast<Call>(output_layout_trans);
431 }
432
433 std::vector<int> GetTransposeAxisOrder(const Call& call, int ndim) const {
434 std::vector<int> attr_axes;
435 if (auto attr = call->attrs.as<TransposeAttrs>()) {
436 if (attr->axes.defined()) {
437 for (int i = 0; i < ndim; ++i) {
438 int64_t axis = attr->axes[i].IntValue();
439 axis += (axis < 0) ? ndim : 0;
440 attr_axes.push_back(axis);
441 }
442 } else {
443 // Empty axes means reverse
444 for (int i = ndim - 1; i >= 0; --i) {
445 attr_axes.push_back(i);
446 }
447 }
448 } else if (auto attr = call->attrs.as<LayoutTransformAttrs>()) {
449 Layout src_layout(attr->src_layout);
450 Layout dst_layout(attr->dst_layout);
451 for (int i = 0; i < ndim; ++i) {
452 attr_axes.push_back(src_layout.IndexOf(dst_layout[i]));
453 }
454 } else {
455 CHECK(false) << "Expected transpose or layout_transform, but got "
456 << Downcast<Op>(call->op)->name;
457 }
458 return std::move(attr_axes);
459 }
460
461 private:
462 /*! \brief Pattern input */
463 DFPattern x_;
464};

466/*!
467 * \brief FullElementwise finds full like ops followed by broadcasting ops, and eliminates
468 * the full op by directly passing the fill value into the broadcasting op.
469 */
470class FullElementwise : public DFPatternRewrite {
471 public:
472 FullElementwise() {
473 x_ = IsWildcard();
474 data_ = IsWildcard();
475 value_ = IsConstant();
476
477 full_ = IsOp("full")({value_}) || IsOp("full_like")({data_, value_});
478 ones_ = IsOp("ones")({}) || IsOp("ones_like")({data_});
479 zeros_ = IsOp("zeros")({}) || IsOp("zeros_like")({data_});
480
481 Map<String, ObjectRef> attrs;
482 attrs.Set("TOpPattern", Integer(static_cast<int>(kBroadcast)));
483 DFPattern op = IsWildcard().HasAttr(attrs);
484 DFPattern full = full_ || ones_ || zeros_;
485 pattern_ = op({full, x_}) || op({x_, full});
486 }
487
488 Expr Callback(const Expr& pre, const Expr& post,
489 const Map<DFPattern, Array<Expr>>& node_map) const override {
490 const CallNode* call = pre.as<CallNode>();
491 ICHECK(call);
492 Type pre_type = pre->checked_type_;
493 ICHECK(pre_type.as<TensorTypeNode>());
494 auto dtype = pre_type.as<TensorTypeNode>()->dtype;
495 auto x = node_map[x_][0];
496 bool is_left = post.as<CallNode>()->args[1] == x;
497 Type x_type;
498 if (is_left) {
499 x_type = call->args[1]->checked_type_;
500 } else {
501 x_type = call->args[0]->checked_type_;
502 }
503
504 if (StructuralEqual()(x_type, pre_type)) {
505 Expr value;
506 if (node_map.count(full_)) {
507 value = node_map[value_][0];
508 ICHECK(IsConstScalar(value));
509 } else if (node_map.count(ones_)) {
510 value = MakeConstantScalar(dtype, 1);
511 } else if (node_map.count(zeros_)) {
512 value = MakeConstantScalar(dtype, 0);
513 } else {
514 ICHECK(false) << "Didn't find a full op while matching full + elementwise";
515 }
516 if (is_left) {
517 return Call(call->op, {value, x}, call->attrs, call->type_args, call->span);
518 } else {
519 return Call(call->op, {x, value}, call->attrs, call->type_args, call->span);
520 }
521 }
522 return post;
523 }
524
525 private:
526 /*! \brief binary argument */
527 DFPattern x_;
528 /*! \brief data ops get shape from */
529 DFPattern data_;
530 /*! \brief constant input */
531 DFPattern value_;
532 /*! \brief full op */
533 DFPattern full_;
534 /*! \brief ones op */
535 DFPattern ones_;
536 /*! \brief zeros op */
537 DFPattern zeros_;
538};

540/*!
541 * \brief Converts `*_like` operators to their explicit shape equivalent (e.g. `zeros_like(x, y)` to
542 * `zeros(x, y.shape)`), when the target shape is concrete. This removes unnecessary dependencies
543 * and can enable more opportunities for operator fusion.
544 */
545class ConcretizeLikeRewrite : public DFPatternRewrite {
546 public:
547 explicit ConcretizeLikeRewrite(const Op& op) {
548 ICHECK(op->num_inputs == 1 || op->num_inputs == 2)
549 << "ConcretizeLike does not handle operators that aren't unary or binary, got: " << op;
550 like_pat_ = IsWildcard();
551 data_pat_ = IsWildcard();
552 if (op->num_inputs == 1) {
553 pattern_ = IsExpr(op)({like_pat_});
554 } else {
555 pattern_ = IsExpr(op)({data_pat_, like_pat_});
556 }
557 }
558
559 virtual bool Check(const Expr& pre, const Expr& post,
560 const Map<DFPattern, Array<Expr>>& node_map) const {
561 const CallNode* call_node = pre.as<CallNode>();
562 ICHECK(call_node);
563
564 if (!call_node->checked_type().as<TensorTypeNode>()) {
565 return false;
566 }
567
568 return true;
569 }
570
571 virtual Expr Concretize(const Map<DFPattern, Array<Expr>>& node_map, Array<Integer> shape,
572 DataType dtype) const = 0;
573
574 Expr Callback(const Expr& pre, const Expr& post,
575 const Map<DFPattern, Array<Expr>>& node_map) const override {
576 if (!Check(pre, post, node_map)) {
577 return post;
578 }
579
580 const TensorTypeNode* like_ty = pre->checked_type().as<TensorTypeNode>();
581 Array<Integer> cshape;
582 for (const auto& dim : like_ty->shape) {
583 if (const auto* imm = dim.as<IntImmNode>()) {
584 cshape.push_back(Integer(GetRef<IntImm>(imm)));
585 } else {
586 // shape is not static, don't concretize
587 return post;
588 }
589 }
590
591 return Concretize(node_map, cshape, like_ty->dtype);
592 }
593
594 protected:
595 DFPattern data_pat_;
596 DFPattern like_pat_;
597};

640class ConcretizeCollapseSumLikeRewrite : public ConcretizeLikeRewrite {
641 public:
642 ConcretizeCollapseSumLikeRewrite() : ConcretizeLikeRewrite(Op::Get("collapse_sum_like")) {}
643
644 Expr Concretize(const Map<DFPattern, Array<Expr>>& node_map, Array<Integer> shape,
645 DataType dtype) const override {
646 ICHECK_LE(shape.size(), std::numeric_limits<int64_t>::max());
647 static const Op& op = Op::Get("collapse_sum_to");
648 auto attrs = make_object<InitOpAttrs>();
649 attrs->shape = shape;
650 std::vector<int64_t> s;
651 std::transform(shape.begin(), shape.end(), std::back_inserter(s),
652 [](Integer i) { return i.IntValue(); });
653 auto cshape = MakeConstantTensor(DataType::Int(32), {static_cast<int64_t>(shape.size())}, s);
654 return Call(op, {node_map[data_pat_][0], cshape}, Attrs(attrs));
655 }
656};

663 DataType dtype) const override {
664 return MakeBroadCastTo(node_map[data_pat_][0], shape);

698/*! \brief Eliminates expressions that are equivalent to identity. */
699class EliminateIdentityRewrite : public DFPatternRewrite {
700 public:
701 EliminateIdentityRewrite() {
702 x_ = IsWildcard();
703 const_ = IsConstant();
704
705 DFPattern add_op = IsOp("add");
706 DFPattern mul_op = IsOp("multiply");
707 DFPattern zeros_expr = IsOp("zeros")({}) || IsOp("zeros_like")({IsWildcard()}) || const_;
708 DFPattern ones_expr = IsOp("ones")({}) || IsOp("ones_like")({IsWildcard()}) || const_;
709
710 // add and multiply are commutative so we don't need another pattern for reversed args
711 DFPattern add_id = add_op({x_, zeros_expr});
712 DFPattern mul_id = mul_op({x_, ones_expr});
713
714 DFPattern sub_id = IsOp("subtract")({x_, zeros_expr});
715 DFPattern div_id = IsOp("divide")({x_, ones_expr});
716
717 pattern_ = add_id || mul_id || sub_id || div_id;
718 }
719
720 bool CheckConstant(const OpNode* op, const ConstantNode* constant) const {
721 if (!IsScalar(GetRef<Expr>(constant))) {
722 return false;
723 }
724 auto value = TryToScalar(constant->data, 0);
725 if (!value) {
726 // unsupported dtype
727 return false;
728 }
729 if (op->name == "add" || op->name == "subtract") {
730 return value.value() == 0.0;
731 } else if (op->name == "multiply" || op->name == "divide") {
732 return value.value() == 1.0;
733 }
734 return false;
735 }
736
737 Expr Callback(const Expr& pre, const Expr& post,
738 const Map<DFPattern, Array<Expr>>& node_map) const override {
739 const CallNode* call = pre.as<CallNode>();
740 ICHECK(call);
741 Type pre_type = pre->checked_type_;
742 ICHECK(pre_type.as<TensorTypeNode>());
743 auto x = node_map[x_][0];
744 bool is_left = post.as<CallNode>()->args[1] == x;
745 Type x_type;
746 if (is_left) {
747 x_type = call->args[1]->checked_type_;
748 } else {
749 x_type = call->args[0]->checked_type_;
750 }
751
752 if (node_map.count(const_)) {
753 // the other argument is a Constant in this case
754 const ConstantNode* constant = node_map[const_][0].as<ConstantNode>();
755 const OpNode* op = call->op.as<OpNode>();
756 ICHECK(constant);
757 ICHECK(op);
758 if (!CheckConstant(op, constant)) {
759 return post;
760 }
761 }
762
763 if (StructuralEqual()(x_type, pre_type)) {
764 return x;
765 }
766
767 return post;
768 }
769
770 private:
771 DFPattern x_;
772 DFPattern const_;
773};

775/*! \brief Switch adjacent add-mul with constants to mul-add.
776 * As mul-add pattern is more friendly to FoldScaleAxis.
777 */
778class SwitchAddMultiply : public DFPatternRewrite {
779 public:
780 SwitchAddMultiply() {
781 x_ = IsWildcard();
782 c1_ = IsConstant();
783 c2_ = IsConstant();
784 pattern_ = (x_ + c1_) * c2_;
785 }
786
787 Expr Callback(const Expr& pre, const Expr& post,
788 const Map<DFPattern, Array<Expr>>& node_map) const override {
789 auto x = node_map[x_][0];
790 auto c1 = node_map[c1_][0];
791 auto c2 = node_map[c2_][0];
792
793 if (x.as<ConstantNode>()) {
794 return post;
795 }
796
797 Expr const_expr = Call(Op::Get("multiply"), {c1, c2});
798 IRModule const_mod = IRModule::FromExpr(const_expr);
799 const_mod = transform::FoldConstant()(const_mod);
800 GlobalVar const_main = const_mod->GetGlobalVar("main");
801 Expr const_val = Downcast<Function>(const_mod->functions[const_main])->body;
802
803 return Call(Op::Get("add"), {Call(Op::Get("multiply"), {x, c2}), const_val});
804 }
805
806 private:
807 DFPattern x_;
808 DFPattern c1_;
809 DFPattern c2_;
810};

812/*! \brief Simplify two adjacent multiply or add with constants for further constant folding.
813 * The pattern matching supports commutative property.
814 */
815class SimplifyAdjacentMultiplyOrAdd : public DFPatternRewrite {
816 public:
817 SimplifyAdjacentMultiplyOrAdd() {
818 x_ = IsWildcard();
819 c1_ = IsConstant();
820 c2_ = IsConstant();
821 pattern_ = (x_ * c1_ * c2_) || (x_ + c1_ + c2_);
822 }
823
824 Expr Callback(const Expr& pre, const Expr& post,
825 const Map<DFPattern, Array<Expr>>& node_map) const override {
826 const CallNode* call = pre.as<CallNode>();
827 auto x = node_map[x_][0];
828 auto c1 = node_map[c1_][0];
829 auto c2 = node_map[c2_][0];
830
831 if (x.as<ConstantNode>()) {
832 return post;
833 }
834
835 Expr const_expr = Call(call->op, {c1, c2});
836 IRModule const_mod = IRModule::FromExpr(const_expr);
837 const_mod = transform::FoldConstant()(const_mod);
838 GlobalVar const_main = const_mod->GetGlobalVar("main");
839 Expr const_val = Downcast<Function>(const_mod->functions[const_main])->body;
840
841 return Call(call->op, {x, const_val});
842 }
843
844 private:
845 DFPattern x_;
846 DFPattern c1_;
847 DFPattern c2_;
848};

918Expr SimplifyExpr(const Expr& expr, const IRModule& mod) {
919 // the rewrites will be applied in the given order, and repeated until fixed point
920 DFPatternRewriteComposer composer;
921 composer.AddRewrite<ConcretizeZerosLikeRewrite>();
922 composer.AddRewrite<ConcretizeOnesLikeRewrite>();
923 composer.AddRewrite<ConcretizeFullLikeRewrite>();
924 composer.AddRewrite<ConcretizeReshapeLikeRewrite>();
925 composer.AddRewrite<ConcretizeCollapseSumLikeRewrite>();
926 composer.AddRewrite<ConcretizeBroadcastToLikeRewrite>();
927 composer.AddRewrite<ConcretizeCastLikeRewrite>();
928 composer.AddRewrite<SimplifyRSqrt>();
929 composer.AddRewrite<EliminateIdentityRewrite>();
930 composer.AddRewrite<SimplifyReshape>();
931 composer.AddRewrite<SimplifyTranspose>();
932 composer.AddRewrite<SimplifySameCast>();
933 composer.AddRewrite<SimplifyConsecutiveCast>();
934 composer.AddRewrite<FullElementwise>();
935 composer.AddRewrite<SwitchAddMultiply>();
936 composer.AddRewrite<SimplifyAdjacentMultiplyOrAdd>();
937 composer.AddRewrite<SimplifyDQArgMax>();
938 composer.AddRewrite<SimplifyDQArgMin>();
939 composer.AddRewrite<SimplifyDQArgSort>();
940 composer.AddRewrite<SimplifyClipAndConsecutiveCast>();
941 composer.AddRewrite<SimplifyCastClip>();
942 return RewritePatterns(composer.MakeCallbacks(), expr, mod);
943}
