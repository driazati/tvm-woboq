45/*!
46 * \brief SimplifyReshape matches the pattern of consecutive reshape or reverse_reshape ops,
47 * and merges into one reshape op.
48 */
49class SimplifyReshape : public DFPatternRewrite {
50 public:
51 SimplifyReshape() {
52 x_ = IsWildcard();
53 auto reshape1 = IsOp("reshape") || IsOp("contrib_reverse_reshape");
54 auto reshape2 = IsOp("reshape") || IsOp("contrib_reverse_reshape");
55 pattern_ = reshape1({reshape2({x_})});
56 }
57
58 Expr Callback(const Expr& pre, const Expr& post,
59 const Map<DFPattern, Array<Expr>>& node_map) const override {
60 auto x = node_map[x_][0];
61 bool const_shape = true;
62 Array<Integer> newshape;
63 for (auto dim : Downcast<TensorType>(pre->checked_type())->shape) {
64 if (dim.as<IntImmNode>() == nullptr) {
65 const_shape = false;
66 break;
67 }
68 newshape.push_back(Downcast<Integer>(dim));
69 }
70 if (const_shape) {
71 return MakeReshape(x, newshape);
72 }
73 return post;
74 }
75
76 private:
77 /*! \brief Pattern input */
78 DFPattern x_;
79};

81/*!
82 * \brief SimplifySameCast matches the pattern of cast data to the same dtype.
83 */
84class SimplifySameCast : public DFPatternRewrite {
85 public:
86 SimplifySameCast() {
87 data_pat_ = IsWildcard();
88 like_pat_ = IsWildcard();
89 pattern_ = IsOp("cast_like")({data_pat_, like_pat_}) || IsOp("cast")({data_pat_});
90 }
91
92 Expr Callback(const Expr& pre, const Expr& post,
93 const Map<DFPattern, Array<Expr>>& node_map) const override {
94 const CallNode* call = pre.as<CallNode>();
95 const TensorTypeNode* data_ty = call->args[0]->checked_type().as<TensorTypeNode>();
96 const TensorTypeNode* like_ty = pre->checked_type().as<TensorTypeNode>();
97 if (like_ty->dtype == data_ty->dtype) {
98 return node_map[data_pat_][0];
99 }
100 return post;
101 }
102
103 protected:
104 DFPattern data_pat_;
105 DFPattern like_pat_;
106};

108/*!
109 * \brief SimplifyConsecutiveCast matches the pattern of consecutive cast/cast_like ops
110 */
111class SimplifyConsecutiveCast : public DFPatternRewrite {
112 public:
113 SimplifyConsecutiveCast() {
114 data_ = IsWildcard();
115 cast1_ = IsOp("cast_like")({data_, IsWildcard()}) || IsOp("cast")({data_});
116 pattern_ = IsOp("cast_like")({cast1_, IsWildcard()}) || IsOp("cast")({cast1_});
117 }
118
119 Expr Callback(const Expr& pre, const Expr& post,
120 const Map<DFPattern, Array<Expr>>& node_map) const override {
121 auto data = node_map[data_][0];
122 auto cast1 = Downcast<Call>(node_map[cast1_][0]);
123 auto data_type = Downcast<TensorType>(data->checked_type());
124 DataType cast1_dtype = Downcast<TensorType>(cast1->checked_type())->dtype;
125
126 if (!IsWidenCast(data_type->dtype, cast1_dtype)) {
127 // Cannot remove the narrow cast
128 return post;
129 }
130
131 const CallNode* cast2 = post.as<CallNode>();
132 DataType cast2_dtype = Downcast<TensorType>(cast2->checked_type())->dtype;
133 auto expr = MakeCast(data, cast2_dtype);
134
135 // We need to set the checked type as it may be needed in the next callback
136 expr->checked_type_ = TensorType(data_type->shape, cast2_dtype);
137 return expr;
138 }
139
140 bool IsWidenCast(DataType origin, DataType cast) const {
141 /* Return whether casting from origin to cast results in more or the same precision.*/
142 if (origin.code() == cast.code() && origin.bits() <= cast.bits()) {
143 return true;
144 }
145 if (origin.code() == DataType::kBFloat || cast.code() == DataType::kBFloat) {
146 // BFloat cast cannot be omitted
147 return false;
148 }
149 if (origin.code() < cast.code()) {
150 // Loosely have a hiearchy to datatypes
151 // e.g. int --> uint --> float has increasing range of numbers they can represent
152 return true;
153 }
154 return false;
155 }
156
157 protected:
158 DFPattern data_;
159 DFPattern cast1_;
160};

162/*!
163 * \brief SimplifyTranspose matches the pattern of consecutive transpose op,
164 * and merges or cancels them.
165 */
166class SimplifyTranspose : public DFPatternRewrite {
167 public:
168 SimplifyTranspose() {
169 x_ = IsWildcard();
170 auto trans1 = IsOp("transpose") || IsOp("layout_transform");
171 auto trans2 = IsOp("transpose") || IsOp("layout_transform");
172 pattern_ = trans1({trans2({x_})});
173 }
174
175 Expr Callback(const Expr& pre, const Expr& post,
176 const Map<DFPattern, Array<Expr>>& node_map) const override {
177 auto x = node_map[x_][0];
178
179 Call trans_call = Downcast<Call>(post);
180
181 // Try to fuse any rank changing layout transformations
182 if (auto layout_trans = FoldRankChangingLayoutTrans(x, trans_call)) {
183 if (auto attr = layout_trans.value()->attrs.as<LayoutTransformAttrs>()) {
184 // Prune any trivial layout transformation
185 if (attr->src_layout == attr->dst_layout) {
186 return x;
187 }
188 }
189 return layout_trans.value();
190 }
191
192 // Initialize axes
193 int ndim = Downcast<TensorType>(pre->checked_type())->shape.size();
194 Array<Integer> axes;
195 for (int i = 0; i < ndim; ++i) {
196 axes.push_back(i);
197 }
198
199 // Collect axes changes from the matched pattern, including two consecutive transposes.
200 std::vector<std::vector<int>> interm_axes;
201 interm_axes.push_back(GetTransposeAxisOrder(trans_call, ndim));
202 trans_call = Downcast<Call>(trans_call->args[0]);
203 interm_axes.push_back(GetTransposeAxisOrder(trans_call, ndim));
204
205 // Calculate the final axes in reverse order (from root to output)
206 auto it = interm_axes.rbegin();
207 while (it != interm_axes.rend()) {
208 auto interm = *it;
209
210 Array<Integer> new_axes;
211 for (int i = 0; i < ndim; ++i) {
212 new_axes.push_back(axes[interm[i]]);
213 }
214 axes = new_axes;
215 it++;
216 }
217
218 // Check if the transpose is still required
219 bool need_transpose = false;
220 for (int i = 0; i < ndim; ++i) {
221 if (axes[i] != i) {
222 need_transpose = true;
223 break;
224 }
225 }
226
227 if (need_transpose) {
228 return MakeTranspose(x, axes);
229 }
230 return x;
231 }
232
233 String PermuteLayout(const String& layout, std::vector<int> axes_order) const {
234 std::string new_layout{};
235 std::string old_layout{layout};
236 ICHECK_EQ(axes_order.size(), layout.size())
237 << "Number of axes must match the number of named axes in the layout to permute: length("
238 << old_layout << ") != " << axes_order.size();
239 std::stringstream order;
240 for (auto axis : axes_order) {
241 new_layout += old_layout[axis];
242 order << axis << ", ";
243 }
244 DLOG(INFO) << "Using transpose axes order {" << order.str()
245 << "} to permute layout: " << old_layout << " to " << new_layout;
246 return new_layout;
247 }

249 struct RankChangingLayoutDescriptor {
250 Layout src_layout;
251 Layout dst_layout;
252 // Either a rank changing layout transform or a transpose
253 Call other_transform;
254 };
255
256 std::unique_ptr<RankChangingLayoutDescriptor> GetRankChangeDescriptor(const Call& call) const {
257 std::unique_ptr<RankChangingLayoutDescriptor> desc{nullptr};
258 if (auto attr = call->attrs.as<LayoutTransformAttrs>()) {
259 if (attr->src_layout.length() != attr->dst_layout.length()) {
260 desc = std::make_unique<RankChangingLayoutDescriptor>();
261 desc->src_layout = Layout(attr->src_layout);
262 desc->dst_layout = Layout(attr->dst_layout);
263 desc->other_transform = Downcast<Call>(call->args[0]);
264 }
265 }
266 if (auto attr = Downcast<Call>(call->args[0])->attrs.as<LayoutTransformAttrs>()) {
267 if (attr->src_layout.length() != attr->dst_layout.length()) {
268 if (!desc) {
269 desc = std::make_unique<RankChangingLayoutDescriptor>();
270 desc->src_layout = Layout(attr->src_layout);
271 desc->dst_layout = Layout(attr->dst_layout);
272 desc->other_transform = call;
273 } else {
274 ICHECK(desc->src_layout->name == attr->dst_layout)
275 << "Back-to-back layout transforms must have the same intermediate layout: "
276 << desc->src_layout->name << " != " << attr->dst_layout;
277 desc->src_layout = Layout(attr->src_layout);
278 }
279 }
280 }
281 return desc;
282 }

284 /*
285 * \brief Fuse call and it's argument into a single layout_transform operator
286 * when either call or it's argument is a rang changing layout_transform, e.g.,
287 *
288 * Simplify
289 *
290 * [N, H, W, C] -> Transpose -> [N, C, H, W] -> LayoutTrans -> [N, C, H, W, 4c]
291 *
292 * to,
293 *
294 * [N, H, W, C] -> LayoutTrans -> [N, C, H, W, 4c].
295 *
296 * \param The input expression to the matched pattern
297 * \param The pattern root; the second of two consecutive Transpose/LayoutTransform ops
298 */
299 Optional<Call> FoldRankChangingLayoutTrans(const Expr& data, const Call& call) const {
300 // Check to see if either the first or second call in matched pattern
301 // is a rank changing layout transform. If so, return a descriptor containing
302 // the layouts and any additional transpose or layout transform op.
303 auto desc = GetRankChangeDescriptor(call);
304 if (desc == nullptr) {
305 // No rank changing layout transform
306 return Optional<Call>{nullptr};
307 }
308
309 Optional<Expr> output_layout_trans;
310 // Fuse a rank increasing layout transform and a preceeding transpose
311 if (desc->src_layout->axes.size() < desc->dst_layout->axes.size()) {
312 auto axes = GetTransposeAxisOrder(desc->other_transform, desc->src_layout->axes.size());
313 // Calculate the reverse axis order and apply to the source layout
314 std::vector<int> inverse(axes.size());
315 for (size_t i = 0; i < axes.size(); i++) {
316 inverse[axes[i]] = i;
317 }
318 String new_layout = PermuteLayout(desc->src_layout->name, inverse);
319 output_layout_trans = MakeLayoutTransform(data, new_layout, desc->dst_layout->name);
320 // Fuse a rank descreasing layout transform followed by a transpose
321 } else if (desc->src_layout->axes.size() > desc->dst_layout->axes.size()) {
322 auto axes = GetTransposeAxisOrder(desc->other_transform, desc->dst_layout->axes.size());
323 String new_layout = PermuteLayout(desc->dst_layout->name, axes);
324 output_layout_trans = MakeLayoutTransform(data, desc->src_layout->name, new_layout);
325 // Fuse two back-to-back layout transformations which change rank
326 } else if (desc->other_transform->attrs.as<LayoutTransformAttrs>()) {
327 output_layout_trans =
328 MakeLayoutTransform(data, desc->src_layout->name, desc->dst_layout->name);
329 }
330 return Downcast<Call>(output_layout_trans);
331 }
332
333 std::vector<int> GetTransposeAxisOrder(const Call& call, int ndim) const {
334 std::vector<int> attr_axes;
335 if (auto attr = call->attrs.as<TransposeAttrs>()) {
336 if (attr->axes.defined()) {
337 for (int i = 0; i < ndim; ++i) {
338 int64_t axis = attr->axes[i].IntValue();
339 axis += (axis < 0) ? ndim : 0;
340 attr_axes.push_back(axis);
341 }
342 } else {
343 // Empty axes means reverse
344 for (int i = ndim - 1; i >= 0; --i) {
345 attr_axes.push_back(i);
346 }
347 }
348 } else if (auto attr = call->attrs.as<LayoutTransformAttrs>()) {
349 Layout src_layout(attr->src_layout);
350 Layout dst_layout(attr->dst_layout);
351 for (int i = 0; i < ndim; ++i) {
352 attr_axes.push_back(src_layout.IndexOf(dst_layout[i]));
353 }
354 } else {
355 CHECK(false) << "Expected transpose or layout_transform, but got "
356 << Downcast<Op>(call->op)->name;
357 }
358 return std::move(attr_axes);
359 }
360
361 private:
362 /*! \brief Pattern input */
363 DFPattern x_;
364};

366/*!
367 * \brief FullElementwise finds full like ops followed by broadcasting ops, and eliminates
368 * the full op by directly passing the fill value into the broadcasting op.
369 */
370class FullElementwise : public DFPatternRewrite {
371 public:
372 FullElementwise() {
373 x_ = IsWildcard();
374 data_ = IsWildcard();
375 value_ = IsConstant();
376
377 full_ = IsOp("full")({value_}) || IsOp("full_like")({data_, value_});
378 ones_ = IsOp("ones")({}) || IsOp("ones_like")({data_});
379 zeros_ = IsOp("zeros")({}) || IsOp("zeros_like")({data_});
380
381 Map<String, ObjectRef> attrs;
382 attrs.Set("TOpPattern", Integer(static_cast<int>(kBroadcast)));
383 DFPattern op = IsWildcard().HasAttr(attrs);
384 DFPattern full = full_ || ones_ || zeros_;
385 pattern_ = op({full, x_}) || op({x_, full});
386 }
387
388 Expr Callback(const Expr& pre, const Expr& post,
389 const Map<DFPattern, Array<Expr>>& node_map) const override {
390 const CallNode* call = pre.as<CallNode>();
391 ICHECK(call);
392 Type pre_type = pre->checked_type_;
393 ICHECK(pre_type.as<TensorTypeNode>());
394 auto dtype = pre_type.as<TensorTypeNode>()->dtype;
395 auto x = node_map[x_][0];
396 bool is_left = post.as<CallNode>()->args[1] == x;
397 Type x_type;
398 if (is_left) {
399 x_type = call->args[1]->checked_type_;
400 } else {
401 x_type = call->args[0]->checked_type_;
402 }
403
404 if (StructuralEqual()(x_type, pre_type)) {
405 Expr value;
406 if (node_map.count(full_)) {
407 value = node_map[value_][0];
408 ICHECK(IsConstScalar(value));
409 } else if (node_map.count(ones_)) {
410 value = MakeConstantScalar(dtype, 1);
411 } else if (node_map.count(zeros_)) {
412 value = MakeConstantScalar(dtype, 0);
413 } else {
414 ICHECK(false) << "Didn't find a full op while matching full + elementwise";
415 }
416 if (is_left) {
417 return Call(call->op, {value, x}, call->attrs, call->type_args, call->span);
418 } else {
419 return Call(call->op, {x, value}, call->attrs, call->type_args, call->span);
420 }
421 }
422 return post;
423 }
424
425 private:
426 /*! \brief binary argument */
427 DFPattern x_;
428 /*! \brief data ops get shape from */
429 DFPattern data_;
430 /*! \brief constant input */
431 DFPattern value_;
432 /*! \brief full op */
433 DFPattern full_;
434 /*! \brief ones op */
435 DFPattern ones_;
436 /*! \brief zeros op */
437 DFPattern zeros_;
438};

440/*!
441 * \brief Converts `*_like` operators to their explicit shape equivalent (e.g. `zeros_like(x, y)` to
442 * `zeros(x, y.shape)`), when the target shape is concrete. This removes unnecessary dependencies
443 * and can enable more opportunities for operator fusion.
444 */
445class ConcretizeLikeRewrite : public DFPatternRewrite {
446 public:
447 explicit ConcretizeLikeRewrite(const Op& op) {
448 ICHECK(op->num_inputs == 1 || op->num_inputs == 2)
449 << "ConcretizeLike does not handle operators that aren't unary or binary, got: " << op;
450 like_pat_ = IsWildcard();
451 data_pat_ = IsWildcard();
452 if (op->num_inputs == 1) {
453 pattern_ = IsExpr(op)({like_pat_});
454 } else {
455 pattern_ = IsExpr(op)({data_pat_, like_pat_});
456 }
457 }
458
459 virtual bool Check(const Expr& pre, const Expr& post,
460 const Map<DFPattern, Array<Expr>>& node_map) const {
461 const CallNode* call_node = pre.as<CallNode>();
462 ICHECK(call_node);
463
464 if (!call_node->checked_type().as<TensorTypeNode>()) {
465 return false;
466 }
467
468 return true;
469 }
470
471 virtual Expr Concretize(const Map<DFPattern, Array<Expr>>& node_map, Array<Integer> shape,
472 DataType dtype) const = 0;
473
474 Expr Callback(const Expr& pre, const Expr& post,
475 const Map<DFPattern, Array<Expr>>& node_map) const override {
476 if (!Check(pre, post, node_map)) {
477 return post;
478 }
479
480 const TensorTypeNode* like_ty = pre->checked_type().as<TensorTypeNode>();
481 Array<Integer> cshape;
482 for (const auto& dim : like_ty->shape) {
483 if (const auto* imm = dim.as<IntImmNode>()) {
484 cshape.push_back(Integer(GetRef<IntImm>(imm)));
485 } else {
486 // shape is not static, don't concretize
487 return post;
488 }
489 }
490
491 return Concretize(node_map, cshape, like_ty->dtype);
492 }
493
494 protected:
495 DFPattern data_pat_;
496 DFPattern like_pat_;
497};

568/*! \brief Eliminates expressions that are equivalent to identity. */
569class EliminateIdentityRewrite : public DFPatternRewrite {
570 public:
571 EliminateIdentityRewrite() {
572 x_ = IsWildcard();
573 const_ = IsConstant();
574
575 DFPattern add_op = IsOp("add");
576 DFPattern mul_op = IsOp("multiply");
577 DFPattern zeros_expr = IsOp("zeros")({}) || IsOp("zeros_like")({IsWildcard()}) || const_;
578 DFPattern ones_expr = IsOp("ones")({}) || IsOp("ones_like")({IsWildcard()}) || const_;
579
580 // add and multiply are commutative so we don't need another pattern for reversed args
581 DFPattern add_id = add_op({x_, zeros_expr});
582 DFPattern mul_id = mul_op({x_, ones_expr});
583
584 DFPattern sub_id = IsOp("subtract")({x_, zeros_expr});
585 DFPattern div_id = IsOp("divide")({x_, ones_expr});
586
587 pattern_ = add_id || mul_id || sub_id || div_id;
588 }
589
590 bool CheckConstant(const OpNode* op, const ConstantNode* constant) const {
591 if (!IsScalar(GetRef<Expr>(constant))) {
592 return false;
593 }
594 auto value = TryToScalar(constant->data, 0);
595 if (!value) {
596 // unsupported dtype
597 return false;
598 }
599 if (op->name == "add" || op->name == "subtract") {
600 return value.value() == 0.0;
601 } else if (op->name == "multiply" || op->name == "divide") {
602 return value.value() == 1.0;
603 }
604 return false;
605 }
606
607 Expr Callback(const Expr& pre, const Expr& post,
608 const Map<DFPattern, Array<Expr>>& node_map) const override {
609 const CallNode* call = pre.as<CallNode>();
610 ICHECK(call);
611 Type pre_type = pre->checked_type_;
612 ICHECK(pre_type.as<TensorTypeNode>());
613 auto x = node_map[x_][0];
614 bool is_left = post.as<CallNode>()->args[1] == x;
615 Type x_type;
616 if (is_left) {
617 x_type = call->args[1]->checked_type_;
618 } else {
619 x_type = call->args[0]->checked_type_;
620 }
621
622 if (node_map.count(const_)) {
623 // the other argument is a Constant in this case
624 const ConstantNode* constant = node_map[const_][0].as<ConstantNode>();
625 const OpNode* op = call->op.as<OpNode>();
626 ICHECK(constant);
627 ICHECK(op);
628 if (!CheckConstant(op, constant)) {
629 return post;
630 }
631 }
632
633 if (StructuralEqual()(x_type, pre_type)) {
634 return x;
635 }
636
637 return post;
638 }
639
640 private:
641 DFPattern x_;
642 DFPattern const_;
643};

645/*! \brief Make two consecutive add able to be constant_folded.
646 * This pattern matching supports commutative property for addition.
647 */
648class SimplifyConsecutiveAdd : public DFPatternRewrite {
649 public:
650 SimplifyConsecutiveAdd() {
651 x_ = IsWildcard();
652 const1_ = IsConstant();
653 const2_ = IsConstant();
654 DFPattern add_op = IsOp("add");
655 pattern_ = add_op({add_op({x_, const1_}), const2_});
656 }
657
658 Expr Callback(const Expr& pre, const Expr& post,
659 const Map<DFPattern, Array<Expr>>& node_map) const override {
660 const CallNode* call = pre.as<CallNode>();
661 auto x = node_map[x_][0];
662 auto c1 = node_map[const1_][0];
663 auto c2 = node_map[const2_][0];
664
665 auto pre_call = call;
666 // Find the next add call.
667 if (pre_call->args[1].as<ConstantNode>()) {
668 pre_call = pre_call->args[0].as<CallNode>();
669 } else {
670 pre_call = pre_call->args[1].as<CallNode>();
671 }
672 // Do nothing if both inputs are not constants as they will be constant folded already.
673 if (pre_call->args[0].as<ConstantNode>() && pre_call->args[1].as<ConstantNode>()) {
674 return post;
675 } else {
676 auto add_res = Call(call->op, {c1, c2});
677 return Call(call->op, {x, add_res});
678 }
679 return post;
680 }
681
682 private:
683 DFPattern x_;
684 DFPattern const1_;
685 DFPattern const2_;
686};

688Expr SimplifyExpr(const Expr& expr, const IRModule& mod) {
689 // the rewrites will be applied in the given order, and repeated until fixed point
690 DFPatternRewriteComposer composer;
691 composer.AddRewrite<ConcretizeZerosLikeRewrite>();
692 composer.AddRewrite<ConcretizeOnesLikeRewrite>();
693 composer.AddRewrite<ConcretizeFullLikeRewrite>();
694 composer.AddRewrite<ConcretizeReshapeLikeRewrite>();
695 composer.AddRewrite<ConcretizeCollapseSumLikeRewrite>();
696 composer.AddRewrite<ConcretizeBroadcastToLikeRewrite>();
697 composer.AddRewrite<EliminateIdentityRewrite>();
698 composer.AddRewrite<SimplifyReshape>();
699 composer.AddRewrite<SimplifyTranspose>();
700 composer.AddRewrite<SimplifySameCast>();
701 composer.AddRewrite<SimplifyConsecutiveCast>();
702 composer.AddRewrite<FullElementwise>();
703 composer.AddRewrite<SimplifyConsecutiveAdd>();
704 return RewritePatterns(composer.MakeCallbacks(), expr, mod);
705}
