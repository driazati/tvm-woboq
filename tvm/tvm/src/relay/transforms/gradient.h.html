24#ifndef TVM_RELAY_TRANSFORMS_GRADIENT_H_
25#define TVM_RELAY_TRANSFORMS_GRADIENT_H_
26
27#include <tvm/relay/expr.h>
28#include <tvm/relay/function.h>
29
30#include <vector>
31
32namespace tvm {
33namespace relay {
34
35inline Type GradRetType(const Function& f) {
36 // if type annotations are provided, we will construct a ret type;
37 // otherwise, leave it to be inferred
38 if (!f->ret_type.defined()) {
39 return Type();
40 }
41 std::vector<Type> vt;
42 for (const auto& p : f->params) {
43 if (!p->type_annotation.defined()) {
44 return Type();
45 }
46 vt.push_back(p->type_annotation);
47 }
48
49 return TupleType({f->ret_type, TupleType(vt)});
50}
51
52} // namespace relay
53} // namespace tvm
54#endif // TVM_RELAY_TRANSFORMS_GRADIENT_H_
