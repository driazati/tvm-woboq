20/*!
21 *
22 * \file combine_parallel_conv2d.cc
23 * \brief Combine parallel 2d convolutions into a single convolution.
24 *
25 * This pass replaces convolutions that share the same input node and the same
26 * arguments (except that the number of output channels can be different) with a
27 * single convolution. The weight of the new 2d convolution is the concatenation
28 * of the original weights. Elemwise and broadcast ops following conv2d are also
29 * combined if possible.
30 *
31 * This prevents launching multiple kernels in networks with multiple
32 * convolution branches, such as Inception block.
33 */
34

52class ParallelConv2DCombiner : public ParallelOpCombiner {
53 public:
54 explicit ParallelConv2DCombiner(uint64_t min_num_branches)
55 : ParallelOpCombiner("nn.conv2d", min_num_branches) {}
56
57 protected:
58 bool IsSupportedOp(const CallNode* n) { return n->attrs.as<Conv2DAttrs>()->groups == 1; }
59
60 bool CanOpsBeCombined(const CallNode* a, const CallNode* b) {
61 StructuralEqual eq;
62 const Layout kOIHW("OIHW");
63 const auto* attrs_a = a->attrs.as<Conv2DAttrs>();
64 const auto* attrs_b = b->attrs.as<Conv2DAttrs>();
65 ICHECK(attrs_a);
66 ICHECK(attrs_b);
67 const auto* tweight_a = a->args[1]->type_as<TensorTypeNode>();
68 const auto* tweight_b = b->args[1]->type_as<TensorTypeNode>();
69 const auto shape_a =
70 tir::BijectiveLayout(Layout(attrs_a->kernel_layout), kOIHW).ForwardShape(tweight_a->shape);
71 const auto shape_b =
72 tir::BijectiveLayout(Layout(attrs_b->kernel_layout), kOIHW).ForwardShape(tweight_b->shape);
73
74 return eq(attrs_a->strides, attrs_b->strides) && eq(attrs_a->padding, attrs_b->padding) &&
75 eq(attrs_a->dilation, attrs_b->dilation) && eq(attrs_a->groups, attrs_b->groups) &&
76 eq(attrs_a->data_layout, attrs_b->data_layout) &&
77 eq(attrs_a->kernel_layout, attrs_b->kernel_layout) &&
78 eq(attrs_a->out_dtype, attrs_b->out_dtype) &&
79 eq(attrs_a->out_layout, attrs_b->out_layout) && eq(shape_a[2], shape_b[2]) &&
80 eq(shape_a[3], shape_b[3]);
81 }
82
83 Call MakeCombinedOp(const Group& branches) {
84 const Op& conv2d = Op::Get("nn.conv2d");
85 Expr data = branches[0][0]->args[0];
86 Expr new_weight;
87 IndexExpr new_channels;
88 std::tie(new_weight, new_channels) = TransformWeight(branches);
89
90 const CallNode* group_root = branches[0][0];
91 const auto* attrs = group_root->attrs.as<Conv2DAttrs>();
92 ICHECK(attrs);
93 const auto new_attrs = make_object<Conv2DAttrs>();
94 new_attrs->strides = attrs->strides;
95 new_attrs->padding = attrs->padding;
96 new_attrs->dilation = attrs->dilation;
97 new_attrs->groups = attrs->groups;
98 new_attrs->kernel_size = attrs->kernel_size;
99 new_attrs->data_layout = attrs->data_layout;
100 new_attrs->kernel_layout = attrs->kernel_layout;
101 new_attrs->out_layout = attrs->out_layout;
102 new_attrs->out_dtype = attrs->out_dtype;
103 new_attrs->channels = new_channels;
104
105 const std::string& layout =
106 new_attrs->out_layout == "" ? new_attrs->data_layout : new_attrs->out_layout;
107 channel_pos_ = layout.find('C');
108 ICHECK_NE(channel_pos_, std::string::npos);
109
110 return Call(conv2d, {data, new_weight}, Attrs{new_attrs}, {});
111 }
112
113 bool IsArgCompatible(const CallNode* a, const CallNode* b, size_t index) {
114 StructuralEqual eq;
115 auto ta = a->args[index]->type_as<TensorTypeNode>();
116 auto tb = b->args[index]->type_as<TensorTypeNode>();
117 auto toutput_a = a->type_as<TensorTypeNode>();
118 auto toutput_b = b->type_as<TensorTypeNode>();
119
120 if (!eq(ta->dtype, tb->dtype) || ta->shape.size() != tb->shape.size()) return false;
121
122 // Position of the 'C' dimension in the argument
123 size_t arg_channel_pos = channel_pos_ - toutput_a->shape.size() + ta->shape.size();
124
125 // Channel super-dimension shoule be present and not broadcasted
126 if ((arg_channel_pos > channel_pos_) || // size_t overflow
127 !eq(ta->shape[arg_channel_pos], toutput_a->shape[channel_pos_]) ||
128 !eq(tb->shape[arg_channel_pos], toutput_b->shape[channel_pos_]))
129 return false;
130
131 for (size_t i = 0; i < ta->shape.size(); i++) {
132 if (i == arg_channel_pos) continue;
133 if (!eq(ta->shape[i], tb->shape[i])) return false;
134 }
135 return true;
136 }
137
138 Call MakeCombinedCallFromFollowingOps(const Expr& data, const Group& branches, size_t depth,
139 size_t parent_index) {
140 Array<Expr> new_args;
141 const CallNode* call = branches[0][depth];
142 size_t ndim = call->type_as<TensorTypeNode>()->shape.size();
143
144 for (size_t i = 0; i < call->args.size(); i++) {
145 if (i == parent_index) {
146 new_args.push_back(data);
147 continue;
148 }
149
150 size_t arg_ndim = call->args[i]->type_as<TensorTypeNode>()->shape.size();
151 size_t arg_channel_pos = channel_pos_ - ndim + arg_ndim;
152 Array<Expr> tuple;
153 for (const auto& branch : branches) {
154 tuple.push_back(branch[depth]->args[i]);
155 }
156
157 auto concat = MakeConcatenate(Tuple(tuple), arg_channel_pos);
158 new_args.push_back(std::move(concat));
159 }
160
161 return Call(call->op, new_args, call->attrs, {});
162 }
163
164 void UpdateGroupOutput(const Expr& data, const Group& branches, size_t depth,
165 ExprSubstMap* subst_map) {
166 int64_t index = 0;
167
168 for (const auto& branch : branches) {
169 const CallNode* conv2d = branch[0];
170 int64_t channels = GetConv2DSuperChannelsDim(conv2d);
171 Array<Integer> begin;
172 Array<Integer> end;
173 for (size_t i = 0; i < channel_pos_; i++) {
174 begin.push_back(0);
175 end.push_back(-1);
176 }
177 begin.push_back(index);
178 index += channels;
179 end.push_back(channels);
180 Array<Integer> strides(begin.size(), 1);
181 auto slice = MakeStridedSlice(data, begin, end, strides, "size");
182 subst_map->insert({GetRef<Expr>(branch[depth]), slice});
183 }
184 }
185
186 private:
187 /* \brief index of channel dimension */
188 size_t channel_pos_;
189
190 std::tuple<Expr, IndexExpr> TransformWeight(const Group& branches) {
191 int64_t num_filters = 0; // number of filters of the transformed weight
192 Array<Expr> weights;
193 for (const auto& branch : branches) {
194 auto conv2d = branch[0];
195 weights.push_back(conv2d->args[1]);
196 auto channels = GetConv2DSuperChannelsDim(conv2d);
197 num_filters += channels;
198 }
199 auto index =
200 branches[0][0]->attrs.as<Conv2DAttrs>()->kernel_layout.operator std::string().find('O');
201 ICHECK_NE(index, std::string::npos);
202 return std::make_tuple(MakeConcatenate(Tuple(weights), index),
203 tir::make_const(DataType::Int(32), num_filters));
204 }
205};
206
207/*! \brief Combine parallel conv2d if number of branches >= min_num_branches */
208Expr CombineParallelConv2D(const Expr& expr, uint64_t min_num_branches) {
209 return ParallelConv2DCombiner(min_num_branches).Combine(expr);
210}

212namespace transform {
213
214Pass CombineParallelConv2D(uint64_t min_num_branches) {
215 runtime::TypedPackedFunc<Function(Function, IRModule, PassContext)> pass_func =
216 [=](Function f, IRModule m, PassContext pc) {
217 return Downcast<Function>(CombineParallelConv2D(f, min_num_branches));
218 };
219 return CreateFunctionPass(pass_func, 4, "CombineParallelConv2d", {"InferType"});
220}
