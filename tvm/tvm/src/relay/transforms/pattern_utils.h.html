53/*!
54 * \brief Dispatch DataType to the C++ data type
55 * during runtime.
56 */
57#define TVM_DTYPE_DISPATCH(type, DType, ...) \
58 if (type == DataType::Float(64)) { \
59 typedef double DType; \
60 { __VA_ARGS__ } \
61 } else if (type == DataType::Float(32)) { \
62 typedef float DType; \
63 { __VA_ARGS__ } \
64 } else if (type == DataType::Float(16)) { \
65 typedef uint16_t DType; \
66 { __VA_ARGS__ } \
67 } else if (type == DataType::BFloat(16)) { \
68 typedef uint16_t DType; \
69 { __VA_ARGS__ } \
70 } else if (type == DataType::Int(64)) { \
71 typedef int64_t DType; \
72 { __VA_ARGS__ } \
73 } else if (type == DataType::Int(32)) { \
74 typedef int32_t DType; \
75 { __VA_ARGS__ } \
76 } else if (type == DataType::Int(16)) { \
77 typedef int16_t DType; \
78 { __VA_ARGS__ } \
79 } else if (type == DataType::Int(8)) { \
80 typedef int8_t DType; \
81 { __VA_ARGS__ } \
82 } else if (type == DataType::UInt(64)) { \
83 typedef uint64_t DType; \
84 { __VA_ARGS__ } \
85 } else if (type == DataType::UInt(32)) { \
86 typedef uint32_t DType; \
87 { __VA_ARGS__ } \
88 } else if (type == DataType::UInt(16)) { \
89 typedef uint16_t DType; \
90 { __VA_ARGS__ } \
91 } else if (type == DataType::UInt(8)) { \
92 typedef uint8_t DType; \
93 { __VA_ARGS__ } \
94 } else if (type == DataType::Bool()) { \
95 typedef bool DType; \
96 { __VA_ARGS__ } \
97 } else if ((*tvm::runtime::Registry::Get("runtime._datatype_get_type_registered"))( \
98 static_cast<uint8_t>(type.code()))) { \
99 typedef double DType; \
100 { __VA_ARGS__ } \
101 } else { \
102 LOG(FATAL) << "unknown data type " << type; \
103 }

105/*!
106 * \brief Try to match lhs and rhs via broadcasting rule, such that:
107 *
108 * rhs matches the dimension of lhs specified by lhs_axes
109 * rhs's value equals 1 on rest of dimensions.
110 *
111 * \param tlhs The type of left operand (data)
112 * \param trhs The type right operand (bias)
113 * \param lhs_axes The axes on lhs to match.
114 * \param rhs_value A squeezed version of rhs which only contains matched dimension.
115 * \return Whether match is successful.
116 */
117inline bool MatchBroadcastToLeftAxes(const TensorTypeNode* tlhs, const TensorTypeNode* trhs,
118 const Array<Integer>& lhs_axes, Expr* rhs_value = nullptr) {
119 if (tlhs->shape.size() < trhs->shape.size()) return false;
120 StructuralEqual equal;
121 size_t base = tlhs->shape.size() - trhs->shape.size();
122 size_t j = 0;
123
124 ObjectPtr<SqueezeAttrs> squeeze_attrs;
125 if (rhs_value != nullptr) {
126 squeeze_attrs = make_object<SqueezeAttrs>();
127 }
128
129 for (size_t i = 0; i < tlhs->shape.size(); ++i) {
130 if (j < lhs_axes.size() && i == static_cast<size_t>(lhs_axes[j]->value)) {
131 if (i < base || !equal(tlhs->shape[i], trhs->shape[i - base])) {
132 return false;
133 }
134 ++j;
135 } else if (i >= base) {
136 if (!tir::is_const_int(trhs->shape[i - base], 1)) {
137 return false;
138 }
139 if (rhs_value != nullptr) {
140 squeeze_attrs->axis.push_back(static_cast<int>(i - base));
141 }
142 }
143 }
144 if (rhs_value != nullptr && squeeze_attrs->axis.size() != 0) {
145 static const Op& squeeze_op = Op::Get("squeeze");
146 *rhs_value = Call(squeeze_op, {rhs_value[0]}, Attrs(squeeze_attrs), {});
147 }
148 return true;
149}

151/*!
152 * \brief Expand 1D Tensor to match axis.
153 *
154 * The result bias can be used to add or multiply to
155 * the target Tensor on the specified axis via broadcasting rule.
156 *
157 * \param bias The bias.
158 * \param target_ndim Target dimension.
159 * \param axes The axis on the output we want to match on.
160 */
161inline Expr ExpandBiasToMatchAxis(Expr bias, int target_ndim, const Array<Integer>& axes) {
162 static const Op& expand_dims = Op::Get("expand_dims");
163 for (size_t i = axes.size(); i != 0; --i) {
164 if (i == axes.size()) {
165 int64_t num_pad_axis = target_ndim - axes[i - 1]->value - 1;
166 if (num_pad_axis > 0) {
167 auto attrs = make_object<ExpandDimsAttrs>();
168 attrs->axis = i;
169 attrs->num_newaxis = static_cast<int>(num_pad_axis);
170 bias = Call(expand_dims, {bias}, Attrs(attrs), {});
171 }
172 } else {
173 int64_t diff = axes[i]->value - axes[i - 1]->value;
174 ICHECK_GE(diff, 0L);
175 if (diff > 0) {
176 auto attrs = make_object<ExpandDimsAttrs>();
177 attrs->axis = i;
178 attrs->num_newaxis = static_cast<int>(diff);
179 bias = Call(expand_dims, {bias}, Attrs(attrs), {});
180 }
181 }
182 }
183 return bias;
184}

186/*!
187 * \brief Check if the call is depthwise conv3d.
188 *
189 * \param call The conv call.
190 * \param param The conv attributes.
191 * \return Whether it is depthwise_conv3d.
192 */
193template <typename ATTRS>
194inline bool IsDepthwiseConv(const Call& call, ATTRS param, const Layout& kernel_layout) {
195 static const Layout kOIXX =
196 backend::IsOp(call.as<CallNode>(), "nn.conv2d") ? Layout("OIHW") : Layout("OIDHW");
197 const auto bilayout = tir::BijectiveLayout(kernel_layout, kOIXX);
198 auto wshape = bilayout.ForwardShape(call->args[1]->type_as<TensorTypeNode>()->shape);
199 return tir::is_const_int(wshape[0], param->groups) && tir::is_const_int(wshape[1], 1);
200}

202/*!
203 * \brief Get super-dimension of output channels of conv2d
204 * \param call The conv2d call.
205 * \return Super-dimension size of output channels of conv2d.
206 */
207inline int64_t GetConv2DSuperChannelsDim(const CallNode* call) {
208 auto param = call->attrs.as<Conv2DAttrs>();
209 auto tweight = call->args[1]->type_as<TensorTypeNode>();
210 auto index = param->kernel_layout.operator std::string().find('O');
211 ICHECK_NE(index, std::string::npos);
212 auto channels = tir::as_const_int(tweight->shape[index]);
213 return *channels;
214}

216/*!
217 * \brief Is single value tensor (scalar).
218 * \param expr The expr.
219 * \return True if single value tensor.
220 */
221inline bool IsScalar(const Expr& expr) {
222 if (auto tensor_type = expr->checked_type().as<TensorTypeNode>()) {
223 for (auto dim_index_expr : tensor_type->shape) {
224 if (auto dim_index = dim_index_expr.as<IntImmNode>()) {
225 if (dim_index->value != 1) {
226 return false;
227 }
228 } else {
229 return false;
230 }
231 }
232 } else {
233 return false;
234 }
235 return true;
236}

251/*!
252 * \brief Create a Constant with a scalar
253 *
254 * \param dtype The data type.
255 * \param value The value of the scalar.
256 * \return A Constant.
257 */
258template <typename T>
259inline Constant MakeConstantScalar(DataType dtype, T value) {
260 runtime::NDArray arr = runtime::NDArray::Empty({}, dtype, {kDLCPU, 0});
261 TVM_DTYPE_DISPATCH(dtype, DType, {
262 if (dtype == DataType::Float(16)) {
263 // convert to float16
264 // storage is uint16_t
265 *static_cast<DType*>(arr->data) =
266 __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 10>(static_cast<float>(value));
267 } else if (dtype == DataType::BFloat(16)) {
268 // convert to bfloat16
269 // storage is uint16_t
270 *static_cast<DType*>(arr->data) =
271 __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 7>(static_cast<float>(value));
272 } else {
273 *static_cast<DType*>(arr->data) = value;
274 }
275 })
276 return Constant(arr);
277}

279/*!
280 * \brief Create a Constant with a tensor.
281 *
282 * \param dtype The data type.
283 * \param value The vector of the tensor values.
284 * \return A Constant.
285 */
286template <typename T>
287static inline Constant MakeConstantTensor(DataType dtype, std::vector<int64_t> shape,
288 std::vector<T> value) {
289 runtime::NDArray arr = runtime::NDArray::Empty(shape, dtype, {kDLCPU, 0});
290 TVM_DTYPE_DISPATCH(dtype, DType, {
291 for (size_t i = 0; i < value.size(); i++) {
292 if (dtype == DataType::Float(16)) {
293 // convert to float16
294 // storage is uint16_t
295 // Similar handling as that in MakeConstantScalar
296 *(static_cast<DType*>(arr->data) + i) =
297 __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 10>(
298 static_cast<float>(value[i]));
299 } else if (dtype == DataType::BFloat(16)) {
300 // convert to bfloat16
301 // storage is uint16_t
302 *(static_cast<DType*>(arr->data) + i) =
303 __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 7>(
304 static_cast<float>(value[i]));
305 } else {
306 *(static_cast<DType*>(arr->data) + i) = value[i];
307 }
308 }
309 })
310 return Constant(arr);
311}

347/*!
348 * \brief Create a Constant tensor of zeros.
349 *
350 * \param dtype The data type.
351 * \param shape The shape of the output constant tensor.
352 * \return A Constant.
353 */
354static inline Constant MakeConstantZeros(DataType dtype, std::vector<int64_t> shape) {
355 runtime::NDArray arr = runtime::NDArray::Empty(shape, dtype, {kDLCPU, 0});
356 int64_t data_size = 1;
357 for (int64_t dim : shape) {
358 data_size *= dim;
359 }
360 TVM_DTYPE_DISPATCH(dtype, DType, {
361 for (int64_t i = 0; i < data_size; i++) {
362 if (dtype == DataType::Float(16)) {
363 // convert to float16
364 // storage is uint16_t
365 // Similar handling as that in MakeConstantScalar
366 *(static_cast<DType*>(arr->data) + i) =
367 __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 10>(static_cast<float>(0));
368 } else if (dtype == DataType::BFloat(16)) {
369 // convert to bfloat16
370 // storage is uint16_t
371 *(static_cast<DType*>(arr->data) + i) =
372 __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 7>(static_cast<float>(0));
373 } else {
374 *(static_cast<DType*>(arr->data) + i) = 0;
375 }
376 }
377 })
378 return Constant(arr);
379}

381/*!
382 * \brief Check whether a shape is static and create corresponding Constant.
383 Eventually this will be removed and replaced with CheckConstantShapeArrayInteger
384 *
385 * \param shape The Array of the shape values.
386 * \return A Constant.
387 */
388static inline Constant CheckConstantShape(const Array<IndexExpr>& shape) {
389 auto shape_array =
390 runtime::NDArray::Empty({int64_t(shape.size())}, DataType::Int(64), {kDLCPU, 0});
391 auto* shape_data = static_cast<int64_t*>(shape_array->data);
392 for (size_t i = 0; i < shape.size(); ++i) {
393 const auto& dim_val = shape[i].as<IntImmNode>();
394 ICHECK(dim_val) << "Do not support symbolic shape for "
395 "Array format. Pass shape as Expr instead.";
396 shape_data[i] = dim_val->value;
397 }
398 return Constant(shape_array);
399}

421/*!
422 * \brief Check if two expressions are equal scalars.
423 * \param a The expression to be checked.
424 * \param b The expression to be checked
425 * \return Whether two expressions are equal scalars.
426 */
427inline bool IsEqualScalar(const Expr& a, const Expr& b) {
428 const auto* constant_a = a.as<ConstantNode>();
429 const auto* constant_b = b.as<ConstantNode>();
430 if (!constant_a || !constant_b || !constant_a->is_scalar() || !constant_b->is_scalar()) {
431 return false;
432 }
433 return tvm::StructuralEqual()(a, b);
434}

436/*!
437 * \brief Convert an element of a NDArray with type int or float to scalar.
438 * \param array Input NDArray
439 * \param i element index
440 * \return Converted scalar value, or None if conversion failed
441 */
442static inline std::optional<long double> TryToScalar(const runtime::NDArray& array, size_t i = 0) {
443 if (array->dtype.code == kDLInt) {
444 if (array->dtype.bits == 8) {
445 return std::optional<long double>(reinterpret_cast<int8_t*>(array->data)[i]);
446 } else if (array->dtype.bits == 16) {
447 return std::optional<long double>(reinterpret_cast<int16_t*>(array->data)[i]);
448 } else if (array->dtype.bits == 32) {
449 return std::optional<long double>(reinterpret_cast<int32_t*>(array->data)[i]);
450 } else if (array->dtype.bits == 64) {
451 return std::optional<long double>(reinterpret_cast<int64_t*>(array->data)[i]);
452 }
453 } else if (array->dtype.code == kDLUInt) {
454 if (array->dtype.bits == 1) { // bool
455 return std::optional<long double>(reinterpret_cast<uint8_t*>(array->data)[i]);
456 } else if (array->dtype.bits == 8) {
457 return std::optional<long double>(reinterpret_cast<uint8_t*>(array->data)[i]);
458 } else if (array->dtype.bits == 16) {
459 return std::optional<long double>(reinterpret_cast<uint16_t*>(array->data)[i]);
460 } else if (array->dtype.bits == 32) {
461 return std::optional<long double>(reinterpret_cast<uint32_t*>(array->data)[i]);
462 } else if (array->dtype.bits == 64) {
463 return std::optional<long double>(reinterpret_cast<uint64_t*>(array->data)[i]);
464 }
465 } else if (array->dtype.code == kDLFloat) {
466 if (array->dtype.bits == 16) {
467 return std::optional<long double>(
468 __extendXfYf2__<uint16_t, uint16_t, 10, float, uint32_t, 23>(
469 reinterpret_cast<uint16_t*>(array->data)[i]));
470 }
471 if (array->dtype.bits == 32) {
472 return std::optional<long double>(reinterpret_cast<float*>(array->data)[i]);
473 } else if (array->dtype.bits == 64) {
474 return std::optional<long double>(reinterpret_cast<double*>(array->data)[i]);
475 }
476 } else if (array->dtype.code == kDLBfloat) {
477 if (array->dtype.bits == 16) {
478 return std::optional<long double>(__extendXfYf2__<uint16_t, uint16_t, 7, float, uint32_t, 23>(
479 reinterpret_cast<uint16_t*>(array->data)[i]));
480 }
481 }
482 return std::nullopt;
483}

485/*!
486 * \brief Convert an element of a NDArray with type int or float to scalar.
487 * \param array Input NDArray
488 * \param i element index
489 * \return Converted scalar value
490 */
491static inline long double ToScalar(const runtime::NDArray& array, size_t i = 0) {
492 auto try_value = TryToScalar(array, i);
493 ICHECK(try_value) << "Unknown data type: " << tvm::runtime::DLDataType2String(array->dtype);
494 return try_value.value();
495}

497/*!
498 * \brief Convert a NDArray with type int or float to Array<Integer>.
499 * \param array Input NDArray
500 * \return Converted Array.
501 */
502static inline Array<Integer> ToVector(const runtime::NDArray& array) {
503 size_t ndim = array.Shape().size();
504 ICHECK_EQ(ndim, 1) << "This function should only be used for 1D NDArrays";
505 size_t len = array.Shape().front();
506 Array<Integer> out;
507 for (size_t i = 0; i < len; ++i) {
508 long double elem_val = ToScalar(array, i);
509 out.push_back(Integer(IntImm(DataType::Int(32), static_cast<int64_t>(elem_val))));
510 }
511 return out;
512}

514/*!
515 * \brief Convert a NDArray with type int or float to Array<FloatImm>.
516 * \param array Input NDArray
517 * \return Converted Array.
518 */
519static inline Array<FloatImm> ToFloatVector(const runtime::NDArray& array) {
520 size_t ndim = array.Shape().size();
521 ICHECK_EQ(ndim, 1) << "This function should only be used for 1D NDArrays";
522 size_t len = array.Shape().front();
523 Array<FloatImm> out;
524 for (size_t i = 0; i < len; ++i) {
525 long double elem_val = ToScalar(array, i);
526 out.push_back(FloatImm(DataType::Float(32), static_cast<float>(elem_val)));
527 }
528 return out;
529}

531/*!
532 * \brief Convert a NDArray with type int or float to Array<Array<Integer>>.
533 * \param array Input NDArray
534 * \return Converted Array.
535 */
536static inline Array<Array<Integer>> ToMatrix(const runtime::NDArray& array) {
537 size_t ndim = array.Shape().size();
538 ICHECK_EQ(ndim, 2) << "This function should only used for 2D NDArrays";
539 size_t dim1 = array.Shape().at(0);
540 size_t dim2 = array.Shape().at(1);
541
542 Array<Array<Integer>> out;
543
544 for (size_t i = 0; i < dim1; ++i) {
545 Array<Integer> inner_out;
546 for (size_t j = 0; j < dim2; ++j) {
547 double elem_val = ToScalar(array, i * dim2 + j);
548 inner_out.push_back(Integer(static_cast<int>(elem_val)));
549 }
550 out.push_back(inner_out);
551 }
552 return out;
553}

828inline Expr Hardswish(Expr x) {
829 auto three = MakeConstantScalar(DataType::Float(32), 3.0);
830 auto six = MakeConstantScalar(DataType::Float(32), 6.0);
831 auto x2 = Add(x, three);
832 x2 = Clip(x2, 0.0, 6.0);
833 x2 = Multiply(x, x2);
834 x2 = Divide(x2, six);
835 return x2;
836}
