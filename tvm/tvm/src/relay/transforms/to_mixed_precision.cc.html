40// A callable which hashes std::pair
41struct pair_hash {
42 template <class T1, class T2>
43 std::size_t operator()(const std::pair<T1, T2>& pair) const {
44 auto h1 = std::hash<T1>()(pair.first);
45 auto h2 = std::hash<T2>()(pair.second);
46
47 // Use boost's combine_hash strategy
48 return h1 ^ (h1 + 0x9e3779b9 + (h2 << 6) + (h2 >> 2));
49 }
50};

52// MIXED_PRECISION_ALWAYS ops should always be done in lower precision due to the speed and memory
53// savings. MIXED_PRECISION_FOLLOW ops can be done in lower precision but don't have speedups to
54// justify a cast. MIXED_PRECISION_NEVER colored ops should not be done in lower precision due to
55// numerical reasons.
56enum MixedTypeConversionCategory : int {
57 MIXED_PRECISION_ALWAYS = 0,
58 MIXED_PRECISION_FOLLOW = 1,
59 MIXED_PRECISION_NEVER = 2
60};
61
62// A map of a parent node and a wanted dtype to existing nodes casted to the wanted dtype
63using CachedCastNodes = std::unordered_map<std::pair<const ExprNode*, DataType>, Expr, pair_hash>;
64
65// Return array is of type : [MixedTypeConversionCategory (int), String, String]
66// The fields are : [ConversionCategory, accumulation_datatype, output_datatype]
67// Call is a call node, DataType is the mixed precision type
68using FTVMMixedPrecisionConversionType = runtime::TypedPackedFunc<Array<ObjectRef>(
69 const Call& call_node, const std::string& target_dtype_str)>;

71/*! \brief This class transforms the given relay module into a version where
72 * as many operations as possible operate in the target mixed precision dtype.
73 *
74 * Input : A Relay module with operations registered with FTVMMixedPrecisionConversionType
75 * functions. These describe when and how the operations will be transformed
76 * into the target precision dtype.
77 *
78 * Output : A Relay module with some operations transformed according to the below
79 * methodology.
80 *
81 * Methodology :
82 * 1) Each relay Op is either of conversion category ALWAYS, FOLLOW, NEVER
83 * defined by the associated FTVMMixedPrecisionConversionType function.
84 * If an operation is not registered, it by default is assumed to be
85 * FOLLOW.
86 * 2) ALWAYS operations always convert the input floating point args into
87 * the target mixed precision dtype. FOLLOW Ops will convert the input
88 * floating point args back into FP32 unless all floating point args
89 * are in the target mixed precision dtypes. NEVER ops will always cast
90 * inputs back into FP32.
91 * 3) Each ALWAYS Op, and FOLLOW Op with mixed precision dtype arguments
92 * also have an associated accumulation_dtype and output_dtype which
93 * describe whether a larger dtype is used to accumulate the results
94 * of the operation. The output_dtype meanwhile describes the dtype
95 * most Ops should use from this accumulator.
96 */
97class MixedPrecisionPass : public MixedModeMutator {
98 private:
99 /*! \brief A cache of nodes + target dtype to a cast version of the node with target dtype. */
100 CachedCastNodes cast_nodes_cache_;
101
102 /*! \brief The target datatype we want to convert to e.g. FP16 */
103 const DataType mixed_precision_type_;
104
105 /*! \brief Map of Ops with no associated FTVMMixedPrecisionConversionType to the times they were
106 * encountered. Used for emitting warnings on missing ops in the pass.
107 */
108 std::unordered_map<std::string, int> missing_ops_;
109 const RelayExprNode* root_;
110 std::vector<DataType> original_dtype_;
111 bool keep_orig_output_dtype_;
112
113 Attrs GetNewAttrs(const CallNode* call, const DataType& accumulation_dtype) const {
114 /* If the accumulation dtype is in the attributes make a copy and mutate the field. */
115 Attrs cur_attrs = call->attrs;
116 if (cur_attrs.get() != nullptr) {
117 // TODO(AndrewZhaoLuo): Figure out a better way to do this
118 // modify output_dtype attributes (accumulation dtypes for ops)
119 if (auto attrs = cur_attrs.as<Conv1DAttrs>()) {
120 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
121 } else if (auto attrs = cur_attrs.as<Conv1DTransposeAttrs>()) {
122 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
123 } else if (auto attrs = cur_attrs.as<Conv2DAttrs>()) {
124 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
125 } else if (auto attrs = cur_attrs.as<Conv2DTransposeAttrs>()) {
126 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
127 } else if (auto attrs = cur_attrs.as<Conv2DWinogradAttrs>()) {
128 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
129 } else if (auto attrs = cur_attrs.as<Conv2DWinogradNNPACKWeightTransformAttrs>()) {
130 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
131 } else if (auto attrs = cur_attrs.as<DeformableConv2DAttrs>()) {
132 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
133 } else if (auto attrs = cur_attrs.as<Conv3DAttrs>()) {
134 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
135 } else if (auto attrs = cur_attrs.as<Conv3DTransposeAttrs>()) {
136 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
137 } else if (auto attrs = cur_attrs.as<Conv3DWinogradAttrs>()) {
138 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
139 } else if (auto attrs = cur_attrs.as<DenseAttrs>()) {
140 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
141 } else if (auto attrs = cur_attrs.as<BatchMatmulAttrs>()) {
142 return ModifyAttrsOutputDType(attrs, accumulation_dtype);
143 }
144
145 // modify dtype attributes (creating new tensors of type dtype)
146 if (auto attrs = cur_attrs.as<InitOpAttrs>()) {
147 return ModifyAttrsDType(attrs, accumulation_dtype);
148 }
149 }
150
151 return cur_attrs;
152 }
153
154 template <typename T>
155 Attrs ModifyAttrsOutputDType(const T* attrs, const DataType& accumulation_dtype) const {
156 /*
157 Helper template to modify relevant attributes with out_dtype type.
158 These represent accumulation dtypes for some operations e.g.
159 conv2d might take in fp16 and give a fp32 result.
160 Attrs is const because we get it as a const.
161 */
162 DataType cur_type = (attrs->out_dtype);
163 ObjectPtr<T> new_attrs = make_object<T>(*attrs);
164 if (cur_type.is_float() || cur_type.is_void()) new_attrs->out_dtype = accumulation_dtype;
165 return Attrs(new_attrs);
166 }
167
168 template <typename T>
169 Attrs ModifyAttrsDType(const T* attrs, const DataType& accumulation_dtype) const {
170 /*
171 Helper template to modify relevant attributes with dtype type.
172 This determines the output dtype for some ops. For example
173 zeros creates a tensor of zeros of the specified dtype.
174 Attrs is const because we get it as a const.
175 */
176 DataType cur_type = (attrs->dtype);
177 ObjectPtr<T> new_attrs = make_object<T>(*attrs);
178 if (cur_type.is_float() || cur_type.is_void()) new_attrs->dtype = accumulation_dtype;
179 return Attrs(new_attrs);
180 }
181
182 Type GetType(const Expr& expr) const {
183 // The expression has not been changed AND it's existing type
184 // is known to still be valid. (See special handling for tuples etc
185 // below for where we null out checked_type_ when we can not
186 // sure it is still valid.
187 Type checked_type = expr->checked_type_;
188 if (checked_type.defined()) {
189 return checked_type;
190 }
191
192 // This also populates the checked_type_ field for expr
193 return transform::InferTypeLocal(expr);
194 }
195
196 bool IsMixedPrecisionType(const Type& t, bool ignore_non_float = false) const {
197 /* Returns whether t is a type with only target mixed precision type elements.
198 If ignore_non_float, then ignore non-floating types.
199 */
200 if (const TensorTypeNode* tensor_type = t.as<TensorTypeNode>()) {
201 bool is_supported_floating_point_type =
202 (tensor_type->dtype).is_float() || (tensor_type->dtype).is_bfloat16();
203 return (ignore_non_float && !is_supported_floating_point_type) ||
204 tensor_type->dtype == mixed_precision_type_;
205 } else if (const TupleTypeNode* tuple_type = t.as<TupleTypeNode>()) {
206 for (Type t : tuple_type->fields) {
207 if (!IsMixedPrecisionType(t, ignore_non_float)) return false;
208 }
209 return true;
210 } else {
211 LOG(FATAL) << "Unsupported type " << t << " we don't know how to handle";
212 return false;
213 }
214 }
215
216 Expr CachedCast(const Expr& expr, const DataType& expr_dtype, const DataType& wanted_dtype) {
217 /* Cast tensor to the wanted datatype, returning a cached version if it's already been done. */
218
219 // If this is not a floating point type, do not cast. E.g. it might be an integer
220 if (!expr_dtype.is_float()) {
221 return expr;
222 }
223
224 if (expr_dtype == wanted_dtype) {
225 return expr;
226 }
227
228 const ExprNode* expr_node = expr.as<ExprNode>();
229 CHECK(expr_node) << "Non-expression node found in cast: " << expr;
230
231 // Use cached result if possible.
232 auto search = cast_nodes_cache_.find({expr_node, wanted_dtype});
233 if (search != cast_nodes_cache_.end()) {
234 return search->second;
235 }
236
237 Expr result = Cast(expr, wanted_dtype);
238 cast_nodes_cache_[{expr_node, wanted_dtype}] = result;
239
240 // Reverse the cache result, e.g. if we want to reverse the cast simply point to original node
241 const ExprNode* new_expr_node = result.as<ExprNode>();
242 cast_nodes_cache_[{new_expr_node, expr_dtype}] = expr;
243 return result;
244 }
245
246 Expr CastArg(const Expr& expr, const Type& expr_type, const DataType& wanted_dtype) {
247 /* Helper for casting arguments to call_nodes handling all relevant cases. */
248 if (const TensorTypeNode* tensor_type = expr_type.as<TensorTypeNode>()) {
249 return CachedCast(expr, tensor_type->dtype, wanted_dtype);
250 } else if (const TupleTypeNode* tuple_type = expr_type.as<TupleTypeNode>()) {
251 Array<Expr> new_expr;
252 bool all_same = true;
253 for (size_t i = 0; i < (tuple_type->fields).size(); i++) {
254 Expr tuple_element = GetField(expr, i);
255 Type tuple_element_dtype = (tuple_type->fields)[i];
256 Expr casted_element = CastArg(tuple_element, tuple_element_dtype, wanted_dtype);
257 new_expr.push_back(casted_element);
258 all_same &= casted_element.same_as(tuple_element);
259 }
260 return all_same ? expr : Tuple(new_expr);
261 }
262 CHECK(0) << "Unsupported type " << expr_type << " we don't know how to cast for arguments!";
263 return expr;
264 }
265
266 std::pair<Array<Expr>, Array<Type>> CastAllArgs(const Array<Expr>& cur_args,
267 const Array<Type>& cur_arg_types,
268 const DataType& wanted_dtype) {
269 Array<Expr> new_args;
270 Array<Type> new_arg_types;
271 for (size_t i = 0; i < cur_args.size(); i++) {
272 Expr cur_arg = cur_args[i];
273 Type cur_arg_type = cur_arg_types[i];
274 Expr new_arg = CastArg(cur_arg, cur_arg_type, wanted_dtype);
275 Type new_arg_type = GetType(new_arg);
276 new_args.push_back(new_arg);
277 new_arg_types.push_back(new_arg_type);
278 }
279 return {new_args, new_arg_types};
280 }

282 public:
283 using MixedModeMutator::VisitExpr_;
284
285 explicit MixedPrecisionPass(Expr base, bool keep_orig_output_dtype,
286 DataType mixed_precision_type = DataType::Float(16))
287 : MixedModeMutator(),
288 mixed_precision_type_(mixed_precision_type),
289 root_(Downcast<Function>(base)->body.get()),
290 keep_orig_output_dtype_(keep_orig_output_dtype) {
291 if (keep_orig_output_dtype_) {
292 if (root_->IsInstance<tvm::relay::TupleNode>()) {
293 const TupleTypeNode* tuple_type = (root_->checked_type_).as<TupleTypeNode>();
294 for (Type t : tuple_type->fields) {
295 const TensorTypeNode* tensor_type = t.as<TensorTypeNode>();
296 original_dtype_.push_back(tensor_type->dtype);
297 }
298 } else if (root_->IsInstance<tvm::relay::CallNode>()) {
299 original_dtype_.push_back((root_->checked_type_).as<TensorTypeNode>()->dtype);
300 }
301 }
302 if (!mixed_precision_type_.is_float() && !mixed_precision_type_.is_bfloat16()) {
303 LOG(FATAL) << "Only support IEEE floating point mixed precision types and bfloat16, but got "
304 << mixed_precision_type_;
305 }
306 }

308 Expr Rewrite_(const CallNode* pre_call_node, const Expr& post) final {
309 const CallNode* post_call_node = post.as<CallNode>();
310 CHECK(post_call_node) << "Expected a CallNode, but got " << post;
311
312 Expr cur_op = post_call_node->op;
313
314 // TODO(AndrewZhaoLuo): Support ADTs
315 // Relay's algebraic data types are not supported yet.
316 ICHECK(!cur_op.as<GlobalVarNode>() // used to declare functions for recursion
317 && !cur_op.as<ConstructorNode>() // constructing ADT types
318 && !cur_op.as<VarNode>()) // used for calling recursive functions
319 << "Algebraic Data Types (ADT) are not supported yet for mixed precision pass.";
320
321 // Get info on the operation being called:
322 // conversion category (int), accumulation dtype (str), output dtype (str)
323 MixedTypeConversionCategory initial_category;
324 DataType accumulation_dtype, output_dtype;
325 if (cur_op.as<FunctionNode>()) {
326 // Avoid messing with functions to avoid changing signature
327 initial_category = MIXED_PRECISION_NEVER;
328 accumulation_dtype = DataType::Float(32);
329 output_dtype = DataType::Float(32);
330 } else if (cur_op.as<OpNode>()) {
331 static auto attr_map =
332 Op::GetAttrMap<FTVMMixedPrecisionConversionType>("FTVMMixedPrecisionConversionType");
333 Op op = Downcast<Op>(cur_op);
334 if (attr_map.count(op)) {
335 // Calculate the conversion category and dtypes from registered attribute.
336 FTVMMixedPrecisionConversionType func = attr_map[op];
337 Array<ObjectRef> op_descriptor =
338 func(GetRef<Call>(pre_call_node), DLDataType2String(mixed_precision_type_));
339 ICHECK(op_descriptor.size() == 3)
340 << "got the wrong number of returned arguments (expected 3 got " << op_descriptor.size()
341 << ") from FTVMMixedPrecisionConversionType for " << AsText(op, false);
342
343 int64_t op_conversion_type = Downcast<Integer>(op_descriptor[0])->value;
344 initial_category = static_cast<MixedTypeConversionCategory>(op_conversion_type);
345 accumulation_dtype = DataType(String2DLDataType(Downcast<String>(op_descriptor[1])));
346 output_dtype = DataType(String2DLDataType(Downcast<String>(op_descriptor[2])));
347 } else {
348 missing_ops_[op->name] += 1;
349
350 // If not registered, by default assume is a generic FOLLOW operation.
351 initial_category = MIXED_PRECISION_FOLLOW;
352 accumulation_dtype = mixed_precision_type_;
353 output_dtype = mixed_precision_type_;
354 }
355 } else {
356 LOG(FATAL) << "Unsupported op type in CallNode: " << pre_call_node->op;
357 }
358
359 // First check if all the new mutated args are in lower precision form
360 Array<Type> cur_arg_types;
361 bool all_args_mixed_type_compatible = true;
362 for (Expr arg : post_call_node->args) {
363 Type cur_arg_type = GetType(arg);
364 cur_arg_types.push_back(cur_arg_type);
365
366 if (initial_category == MIXED_PRECISION_FOLLOW && all_args_mixed_type_compatible) {
367 // We can cast Vars and Constants to the right types so don't care about the types.
368 bool is_mixed_type_compatible = IsMixedPrecisionType(cur_arg_type, true) ||
369 arg->IsInstance<VarNode>() ||
370 arg->IsInstance<ConstantNode>();
371 all_args_mixed_type_compatible &= is_mixed_type_compatible;
372 }
373 }
374
375 // Determine the final category we want for conversion
376 MixedTypeConversionCategory final_category = initial_category;
377 if (initial_category == MIXED_PRECISION_FOLLOW) {
378 final_category =
379 all_args_mixed_type_compatible ? MIXED_PRECISION_ALWAYS : MIXED_PRECISION_NEVER;
380 }
381
382 // Create the new arguments to the call.
383 DataType wanted_arg_dtypes =
384 final_category == MIXED_PRECISION_ALWAYS ? mixed_precision_type_ : DataType::Float(32);
385 auto call_args_and_types = CastAllArgs(post_call_node->args, cur_arg_types, wanted_arg_dtypes);
386 Array<Expr> new_args = call_args_and_types.first;
387 Array<Type> new_arg_types;
388
389 if (pre_call_node->op.as<FunctionNode>()) {
390 // Function Nodes don't store type info in the Call, it should be a []
391 new_arg_types = pre_call_node->type_args;
392 } else {
393 new_arg_types = call_args_and_types.second;
394 }
395
396 // Finally create the new attributes.
397 if (final_category == MIXED_PRECISION_ALWAYS) {
398 Attrs new_attrs = GetNewAttrs(pre_call_node, accumulation_dtype);
399 Expr output = Call(cur_op, new_args, new_attrs, new_arg_types, pre_call_node->span);
400 if (accumulation_dtype != output_dtype) {
401 output = CastArg(output, GetType(output), output_dtype);
402 }
403 if (pre_call_node == root_ && keep_orig_output_dtype_) {
404 if (original_dtype_[0] != output_dtype) {
405 output = CastArg(output, GetType(output), original_dtype_[0]);
406 }
407 }
408 return output;
409 }
410
411 return Call(cur_op, new_args, pre_call_node->attrs, new_arg_types, pre_call_node->span);
412 }

416 post->checked_type_ = Type(nullptr);

419
420 Expr Rewrite_(const TupleNode* pre, const Expr& post) {
421 // The old checked type in the expression may not be valid so clear it
422 post->checked_type_ = Type(nullptr);
423 if (pre == root_ && keep_orig_output_dtype_) {
424 Array<Expr> new_expr;
425 bool all_same = true;
426 for (size_t i = 0; i < original_dtype_.size(); i++) {
427 Expr output_element = GetField(post, i);
428 Expr casted_element;
429 auto output_element_type = transform::InferTypeLocal(output_element);
430 casted_element = CastArg(output_element, output_element_type, original_dtype_[i]);
431 new_expr.push_back(casted_element);
432 all_same &= casted_element.same_as(output_element);
433 }
434 if (!all_same) {
435 return Tuple(new_expr);
436 }
437 }
438 return post;
439 }

441 Expr VisitExpr_(const FunctionNode* func) final {
442 // Erase the ret_type annotation and let the normal pass recalculate
443 const_cast<FunctionNode*>(func)->ret_type = Type(nullptr);
444 return ExprMutator::VisitExpr_(func);
445 }
446
447 Expr VisitExpr_(const LetNode* op) final {
448 // First convert as much of the bound computation to lower precision as possible
449 Expr value = this->Mutate(op->value);
450
451 // Then rewrite the var type and associated expression
452 Var var = Downcast<Var>(this->Mutate(op->var));
453 VarNode* mutable_var = const_cast<VarNode*>((op->var).as<VarNode>());
454 mutable_var->type_annotation = GetType(value);
455 mutable_var->checked_type_ = mutable_var->type_annotation;
456
457 // Mutate body last as it may depend on previous results
458 Expr body = this->Mutate(op->body);
459 return Let(var, value, body, op->span);
460 }

463 friend Expr ToMixedPrecision(const Expr& expr, bool keep_orig_output_dtype,
464 const DataType& mixed_precision_type, int missing_op_mode);

467Expr ToMixedPrecision(const Expr& expr, bool keep_orig_output_dtype,
468 const DataType& mixed_precision_type, int missing_op_mode) {
469 /*
470 missing_op_mode:
471
472 0: Does not allow any missing ops. Will throw errors and terminate the pass when encountering any.
473 1: Allow missing ops but throw warnings.
474 2: Allow missing ops and silently ignore them.
475 */
476 ICHECK(missing_op_mode >= 0 && missing_op_mode <= 2)
477 << " missing_op_mode must be either 0, 1, or 2 got " << missing_op_mode;
478
479 MixedPrecisionPass converter =
480 MixedPrecisionPass(expr, keep_orig_output_dtype, mixed_precision_type);
481 auto result = converter.Mutate(expr);
482
483 for (auto it = converter.missing_ops_.begin();
484 missing_op_mode != 2 && it != converter.missing_ops_.end(); it++) {
485 std::string op_name = it->first;
486 int appear_count = it->second;
487
488 LOG(WARNING) << "Op \"" << op_name << "\" not registered "
489 << "FTVMMixedPrecisionConversionType appears " << appear_count
490 << " times in graph.";
491 }
492
493 if (converter.missing_ops_.size() != 0 && missing_op_mode == 0) {
494 CHECK(0) << "Missing ops were found!";
495 }
496 return result;
497}

501Pass ToMixedPrecision(DataType mixed_precision_type, int missing_op_mode) {
502 runtime::TypedPackedFunc<Function(Function, IRModule, PassContext)> pass_func =
503 [=](Function f, IRModule m, PassContext pc) {
504 bool keep_orig_output_dtype = false;
505 keep_orig_output_dtype = pc->GetConfig("relay.ToMixedPrecision.keep_orig_output_dtype",
506 Bool(keep_orig_output_dtype))
507 .value();
508 return Downcast<Function>(
509 ToMixedPrecision(f, keep_orig_output_dtype, mixed_precision_type, missing_op_mode));
510 };
511 return CreateFunctionPass(pass_func, 0, "ToMixedPrecision", {});
512}
