42/*
43 Note on Fusing algorithm:
44
45 The main challenge of general fusor is to handle possible diamond shape branches,
46 in the following graph, conv2d can be fused to elemwise add.
47
48 conv2d
49 / | \
50 / | \
51 op op op
52 \ | /
53 \ | /
54 elemwise add
55 |
56
57 However, at the point of conv2d we do not necessarily know that all the future paths
58 will merge at the elemwise add. The fusion algorithm applies post-dominator analysis.
59
60 The immediate post-dominator of a node defined by the closest node where all the future path goes
61 into. In the above case, the elemwise add is the post-dominator of conv2d. The general algorithm
62 is as follows:
63
64 - Construct a DAG of dataflow graph for dominator analysis
65 - Construct a post-dominator tree which gives immediate post dominator of each node.
66 - Run fusion algorithm with the given post-dominator information.
67
68 Note that, because we run analysis on a DAG, we use a single pass post-dominator
69 tree construction algorithm via LCA, which is simpler than the full version that handles cycles.
70
71 The fusion algorithm traverses from each node and checks if it can be fused to its
72 immediate post dominator. It has to check the following things:
73
74 - CheckPath: check all the path between a node and its immediate post-dominator
75 satisfies the fuse condition.
76 - Note that these intermediate node can already be fused with another nodes, the algorithm
77 will still run correctly.
78 - CommitFuse: mark all the nodes between source and post-dominator as the same group.
79 - We use an Union-Find data structure to manage the groups.
80*/

91/*!
92 * \brief Indexed data flow graph in forward direction.
93 * This is a temporary data structure used for operator fusion analysis.
94 *
95 * This data structure only captures the dataflow fragment and
96 * could ignore blocks like let by simply ordering each dataflow block
97 * and mark the output node as extern_ref;
98 */
99class IndexedForwardGraph {
100 public:
101 struct Node;
102 /*!
103 * The forward edge in the dataflow graph.
104 */
105 struct Edge {
106 /*! \brief The corresponding node */
107 Node* node{nullptr};
108 /*! \brief The respective pattern of this op */
109 OpPatternKind pattern{kOpaque};
110 };
111 /*! \brief A node in the graph. */
112 struct Node {
113 /*! \brief weak reference to the corresponding edge. */
114 const tvm::Object* ref{nullptr};
115 /*! \brief The index of the node in topological order. */
116 size_t index{0};
117 /*! \brief Whether this node is referenced by external source */
118 bool extern_ref{false};
119 /*! \brief The general pattern in the node */
120 OpPatternKind pattern{kOpaque};
121 /*! \brief The outputs of the node. */
122 LinkedList<Edge> outputs;
123 };
124 /*! \brief The node map that maps node to graph */
125 std::unordered_map<const tvm::Object*, Node*> node_map;
126 /*! \brief All the nodes in post DFS order */
127 std::vector<Node*> post_dfs_order;
128
129 /*! \brief Dump the graph into string. */
130 void DebugDump() {
131 std::ostringstream os;
132 for (size_t i = 0; i < post_dfs_order.size(); ++i) {
133 Node* node = post_dfs_order[i];
134 os << "node[" << i << "], " << GetRef<ObjectRef>(node->ref) << " outputs=[";
135 for (auto* link = node->outputs.head; link != nullptr; link = link->next) {
136 os << link->value.node->index << ", ";
137 }
138 os << "]\n";
139 }
140 LOG(INFO) << os.str();
141 }
142 /*!
143 * \brief create a indexed forward graph.
144 * \param arena The arena used for data allocation.
145 * \param body The body of the expression to create a graph.
146 */
147 static IndexedForwardGraph Create(support::Arena* arena, const Expr& body);
148
149 private:
150 class Creator;
151};

153// Creator of post dominator tree of the dataflow
154class IndexedForwardGraph::Creator : private ExprVisitor {
155 public:
156 explicit Creator(support::Arena* arena) : arena_(arena) {}
157
158 IndexedForwardGraph Prepare(const Expr& body) {
159 this->Update(body, nullptr, kOpaque);
160 this->VisitExpr(body);
161 return std::move(graph_);
162 }
163
164 private:
165 /*! \brief allocator of all the internal node object */
166 support::Arena* arena_;
167 // The output.
168 IndexedForwardGraph graph_;
169 // attribute equal comparator
170 StructuralEqual attr_equal_;
171 // Update the message stored at the node.
172 void Update(const Expr& node, IndexedForwardGraph::Node* parent, OpPatternKind pattern) {
173 const tvm::Object* key = node.get();
174 IndexedForwardGraph::Node* current;
175 auto it = graph_.node_map.find(key);
176 if (it != graph_.node_map.end()) {
177 current = it->second;
178 } else {
179 current = arena_->make<IndexedForwardGraph::Node>();
180 graph_.node_map[key] = current;
181 }
182 if (parent != nullptr) {
183 auto* link = arena_->make<LinkNode<IndexedForwardGraph::Edge>>();
184 link->value.node = parent;
185 link->value.pattern = pattern;
186 current->outputs.Push(link);
187 } else {
188 current->extern_ref = true;
189 }
190 }
191
192 void AddNode(const tvm::Object* key) {
193 auto it = graph_.node_map.find(key);
194 ICHECK(it != graph_.node_map.end()) << "Cannot find node " << GetRef<ObjectRef>(key);
195 IndexedForwardGraph::Node* node = it->second;
196 ICHECK(node->ref == nullptr);
197 node->ref = key;
198 node->index = graph_.post_dfs_order.size();
199 graph_.post_dfs_order.push_back(node);
200 }
201
202 // Post order tree
203 void VisitExpr_(const FunctionNode* op) final {
204 // Skip the function that should be handled by external codegen.
205 if (op->GetAttr<String>(attr::kCompiler).defined()) return;
206
207 for (auto param : op->params) {
208 this->Update(param, nullptr, kOpaque);
209 }
210 this->Update(op->body, nullptr, kOpaque);
211 ExprVisitor::VisitExpr_(op);
212 }
213
214 void VisitExpr_(const ConstantNode* op) final {
215 this->AddNode(op);
216 Node* node = graph_.node_map.at(op);
217 DataType dtype = DataType(op->data->dtype);
218 // This rule must be consistent with code generator.
219 bool is_simple_const =
220 (dtype == DataType::Int(32) || dtype == DataType::Int(64) || dtype == DataType::Float(32) ||
221 dtype == DataType::Float(64) || dtype == DataType::Bool());
222 if (op->is_scalar() && is_simple_const) {
223 node->pattern = kElemWise;
224 } else {
225 // for now, mark non-scalar constant
226 // as opaque, we will not choose to fuse it.
227 node->pattern = kOpaque;
228 }
229 }
230
231 void VisitExpr_(const CallNode* call) final {
232 ICHECK(graph_.node_map.count(call));
233 Node* node = graph_.node_map.at(call);
234 static auto fpattern = Op::GetAttrMap<TOpPattern>("TOpPattern");
235 // Now we set the pattern of this call.
236 //
237 // If we see a call mentioning an operator we should mark it with its
238 // annotated pattern.
239 //
240 // If the pattern is not annotated we will default to opaque.
241 //
242 // Finally if the operator position is not a call node we will
243 // need to call Update, as it may be an arbitrary expression.
244 OpPatternKind op_pattern = kOpaque;
245 if (const OpNode* opnode = call->op.as<OpNode>()) {
246 auto op = GetRef<Op>(opnode);
247 if (IsDynamic(call->checked_type()) && IsDataDependent(call)) {
248 // output of a shape func can't be fed to a data-dependent shape func
249 op_pattern = kOpaque;
250 } else {
251 op_pattern = static_cast<OpPatternKind>(fpattern[op]);
252 }
253 } else {
254 this->Update(call->op, node, kOpaque);
255 }
256
257 node->pattern = op_pattern;
258 this->Update(call->op, nullptr, kOpaque);
259 const auto* rtype = call->checked_type().as<TensorTypeNode>();
260 // pass the analysis back to all the children it references.
261 for (size_t i = 0; i < call->args.size(); ++i) {
262 const auto* arg_type = call->args[i]->checked_type().as<TensorTypeNode>();
263 // specifically check if result type is the same as arguments type
264 OpPatternKind edge_pattern = op_pattern;
265 if (edge_pattern == kBroadcast && arg_type != nullptr && rtype != nullptr &&
266 attr_equal_(rtype->shape, arg_type->shape)) {
267 edge_pattern = kElemWise;
268 }
269 this->Update(call->args[i], node, edge_pattern);
270 }
271 ExprVisitor::VisitExpr_(call);
272 this->AddNode(call);
273 }
274
275 void VisitExpr_(const TupleNode* op) final {
276 ICHECK(graph_.node_map.count(op));
277 Node* tuple_node = graph_.node_map.at(op);
278 tuple_node->pattern = kTuple;
279 for (const Expr& field : op->fields) {
280 if (field->checked_type().as<TensorTypeNode>()) {
281 this->Update(field, tuple_node, kInjective);
282 } else {
283 this->Update(field, nullptr, kOpaque);
284 }
285 }
286 ExprVisitor::VisitExpr_(op);
287 this->AddNode(op);
288 }
289
290 void VisitExpr_(const TupleGetItemNode* op) final {
291 auto tuple_type = op->tuple->checked_type().as<TupleTypeNode>();
292 ICHECK(tuple_type);
293 // When TVM lowers a fused function, it expects all arguments to be a Tensor or
294 // a tuple containing only Tensors. But this tuple may contain a reference or
295 // another tuple. To avoid modifying codegen logic, we do not allow fusing through this node
296 // if the tuple contains such non Tensor fields. However, all fields will be recursively
297 // visited via call to ExprVisitor::VisitExpr_(op) below and corresponding visitor methods.
298 bool has_non_tensor = false;
299 for (auto ty : tuple_type->fields) {
300 if (!ty.as<TensorTypeNode>()) {
301 has_non_tensor = true;
302 break;
303 }
304 }
305 if (has_non_tensor) {
306 this->Update(op->tuple, nullptr, kOpaque);
307 } else {
308 ICHECK(graph_.node_map.count(op));
309 Node* node = graph_.node_map.at(op);
310 node->pattern = kInjective;
311 this->Update(op->tuple, node, kInjective);
312 }
313 ExprVisitor::VisitExpr_(op);
314 this->AddNode(op);
315 }
316
317 void VisitExpr_(const VarNode* op) final { this->AddNode(op); }
318
319 void VisitExpr_(const LetNode* op) final {
320 // do not fuse through let.
321 auto pre_visit = [this](const LetNode* op) {
322 // Rely on the Memoizer to cache pre-visit values
323 this->Update(op->var, nullptr, kOpaque);
324 this->Update(op->value, nullptr, kOpaque);
325 this->Update(op->body, nullptr, kOpaque);
326 this->VisitExpr(op->var);
327 this->VisitExpr(op->value);
328 };
329 auto post_visit = [this](const LetNode* op) {
330 this->VisitExpr(op->body);
331 this->visit_counter_[op] += 1;
332 this->AddNode(op);
333 };
334 ExpandANormalForm(op, pre_visit, post_visit);
335 }
336
337 void VisitExpr_(const IfNode* op) final {
338 // do not fuse through if.
339 this->Update(op->cond, nullptr, kOpaque);
340 this->Update(op->true_branch, nullptr, kOpaque);
341 this->Update(op->false_branch, nullptr, kOpaque);
342 ExprVisitor::VisitExpr_(op);
343 this->AddNode(op);
344 }
345
346 void VisitExpr_(const RefCreateNode* op) final {
347 this->Update(op->value, nullptr, kOpaque);
348 ExprVisitor::VisitExpr_(op);
349 this->AddNode(op);
350 }
351
352 void VisitExpr_(const RefReadNode* op) final {
353 this->Update(op->ref, nullptr, kOpaque);
354 ExprVisitor::VisitExpr_(op);
355 this->AddNode(op);
356 }
357
358 void VisitExpr_(const RefWriteNode* op) final {
359 this->Update(op->ref, nullptr, kOpaque);
360 this->Update(op->value, nullptr, kOpaque);
361 ExprVisitor::VisitExpr_(op);
362 this->AddNode(op);
363 }
364
365 void VisitExpr_(const MatchNode* op) final {
366 this->Update(op->data, nullptr, kOpaque);
367 for (const Clause& c : op->clauses) {
368 this->Update(c->rhs, nullptr, kOpaque);
369 }
370 ExprVisitor::VisitExpr_(op);
371 this->AddNode(op);
372 }
373};
374
375IndexedForwardGraph IndexedForwardGraph::Create(support::Arena* arena, const Expr& body) {
376 return Creator(arena).Prepare(body);
377}

379/*!
380 * \brief Dominator tree that represent domination or
381 * post domination relation of the node.
382 */
383class DominatorTree {
384 public:
385 /*!
386 * \brief A node in the dominator tree.
387 */
388 struct Node {
389 /*! \brief The node in the tree */
390 IndexedForwardGraph::Node* gnode{nullptr};
391 /*! \brief parent of the tree */
392 Node* parent{nullptr};
393 /*! \brief current depth*/
394 int depth{0};
395 /*! \brief aggregated pattern to parent */
396 OpPatternKind pattern{kOpaque};
397 };
398 // index -> node.
399 std::vector<Node*> nodes;
400 /*!
401 * \brief compute a post dominator relation for a given dataflow graph.
402 * \param arena The arena used for node allocation.
403 * \param graph The graph to be analyzed.
404 * \return The dominator tree of the graph.
405 * \note This algorithm makes use of the fact that graph is DAG,
406 * and runs a single pass algorithm via LCA (Least Common Ancestor)
407 */
408 static DominatorTree PostDom(support::Arena* arena, const IndexedForwardGraph& graph);
409
410 private:
411 // Combine pattern together.
412 static OpPatternKind CombinePattern(OpPatternKind lhs, OpPatternKind rhs) {
413 if (lhs > rhs) return lhs;
414 return rhs;
415 }
416 /*!
417 * \brief Find the least common ancestor of the two nodes.
418 * \param lhs The left node.
419 * \param rhs The right node.
420 * \param edge_pattern
421 * The combined edge pattern across all the parents.
422 * \return The least common ancestor of the two.
423 */
424 static Node* LeastCommonAncestor(Node* lhs, Node* rhs, OpPatternKind* edge_pattern) {
425 while (lhs != rhs) {
426 if (lhs == nullptr) return nullptr;
427 if (rhs == nullptr) return nullptr;
428 if (lhs->depth < rhs->depth) {
429 edge_pattern[0] = CombinePattern(edge_pattern[0], rhs->pattern);
430 rhs = rhs->parent;
431 } else if (rhs->depth < lhs->depth) {
432 edge_pattern[0] = CombinePattern(edge_pattern[0], lhs->pattern);
433 lhs = lhs->parent;
434 } else {
435 edge_pattern[0] = CombinePattern(edge_pattern[0], lhs->pattern);
436 edge_pattern[0] = CombinePattern(edge_pattern[0], rhs->pattern);
437 lhs = lhs->parent;
438 rhs = rhs->parent;
439 }
440 }
441 return lhs;
442 }
443 /*!
444 * \brief Find the least common ancestor of a list of nodes.
445 * \param nodes the nodes.
446 * \param edge_pattern
447 * The combined edge pattern across all the parents.
448 * \return The least common ancestor of all nodes.
449 */
450 Node* LeastCommonAncestor(const LinkedList<IndexedForwardGraph::Edge>& input_nodes,
451 OpPatternKind* edge_pattern) {
452 auto link = input_nodes.head;
453 if (link == nullptr) {
454 return nullptr;
455 }
456 auto get_node = [&](const IndexedForwardGraph::Edge& edge) {
457 size_t oindex = edge.node->index;
458 ICHECK_LT(oindex, nodes.size());
459 Node* onode = nodes[oindex];
460 ICHECK(onode != nullptr);
461 return onode;
462 };
463 Node* parent = get_node(link->value);
464 *edge_pattern = CombinePattern(*edge_pattern, link->value.pattern);
465 link = link->next;
466 for (; link != nullptr; link = link->next) {
467 parent = LeastCommonAncestor(parent, get_node(link->value), edge_pattern);
468 *edge_pattern = CombinePattern(*edge_pattern, link->value.pattern);
469 }
470 return parent;
471 }
472 /*!
473 * \brief Convert the Node from an IndexedForwardGraph Node into DomaintorTree Node.
474 * \param arena The Arena.
475 * \param gnode An IndexedForwardGraph Node.
476 * \return The DominatorTree Node.
477 */
478 Node* GetNode(support::Arena* arena, IndexedForwardGraph::Node* gnode) {
479 Node* tnode = arena->make<Node>();
480 tnode->gnode = gnode;
481 if (gnode->extern_ref) {
482 tnode->depth = 1;
483 tnode->parent = nullptr;
484 tnode->pattern = kOpaque;
485 } else {
486 // find the LCAs of all outputs.
487 OpPatternKind pattern = kElemWise;
488 Node* parent = LeastCommonAncestor(gnode->outputs, &pattern);
489 tnode->depth = parent ? parent->depth + 1 : 1;
490 tnode->parent = parent;
491 tnode->pattern = pattern;
492 }
493 return tnode;
494 }
495};
496
497DominatorTree DominatorTree::PostDom(support::Arena* arena, const IndexedForwardGraph& graph) {
498 DominatorTree tree;
499 tree.nodes.resize(graph.post_dfs_order.size(), nullptr);
500 // reverse topo order
501 for (size_t i = graph.post_dfs_order.size(); i != 0; --i) {
502 size_t index = i - 1;
503 tree.nodes[index] = tree.GetNode(arena, graph.post_dfs_order[index]);
504 }
505 return tree;
506}

508/*!
509 * \brief A partition of the graph marked by union find data structure.
510 */
511class GraphPartitioner {
512 public:
513 explicit GraphPartitioner(support::Arena* arena, int opt_level, size_t max_fuse_depth)
514 : arena_(arena), opt_level_(opt_level), max_fuse_depth_(max_fuse_depth) {}
515 /*!
516 * \brief Group as a union find data structure.
517 */
518 struct Group {
519 /*! \brief The parent in the union find data structure. */
520 Group* parent{nullptr};
521 /*! \brief The pattern of the group */
522 OpPatternKind pattern;
523 /*! \brief reference to the root node. */
524 const tvm::Object* root_ref{nullptr};
525 /*!
526 * \brief Reference to the anchor node,
527 * this field is not nullptr only if pattern is kOutEWiseFusable.
528 */
529 const tvm::Object* anchor_ref{nullptr};
530 /*!
531 * \brief Find the group root, perform path compression
532 * \return The root type node.
533 */
534 Group* FindRoot() {
535 // fast path
536 if (this->parent == nullptr) return this;
537 // slow path with path compression.
538 Group* root = this;
539 while (root->parent != nullptr) {
540 root = root->parent;
541 }
542 for (Group* p = this; p != root;) {
543 Group* parent = p->parent;
544 p->parent = root;
545 p = parent;
546 }
547 return root;
548 }
549
550 /*!
551 * \brief The number of nodes belonging to this group
552 */
553 uint32_t num_nodes{1};
554 };
555 /*!
556 * \brief Partition a graph.
557 * \return group assignments of each node.
558 */
559 std::vector<Group*> Partition(const IndexedForwardGraph& graph);
560
561 private:
562 /*! \brief The internal arena for temporary space. */
563 support::Arena* arena_;
564 /*! \brief optimization level for fuse operation. */
565 int opt_level_;
566 /*! \brief The maximum number of operations in one fused function */
567 size_t max_fuse_depth_;
568 /*! \brief The internal groups. */
569 std::vector<Group*> groups_;
570 /*! \brief internal field used for deduplication */
571 std::unordered_set<IndexedForwardGraph::Node*> visited_;
572 // Internal implelementation of CheckPath
573 template <typename F>
574 bool CheckPath_(IndexedForwardGraph::Node* src, IndexedForwardGraph::Node* sink, F fcond) {
575 if (visited_.count(src)) return true;
576 visited_.insert(src);
577 Group* gnode = groups_[src->index];
578 ICHECK(gnode != nullptr);
579 gnode = gnode->FindRoot();
580 if (!fcond(gnode->pattern, src == sink)) return false;
581 if (src == sink) return true;
582 for (auto link = src->outputs.head; link != nullptr; link = link->next) {
583 if (!CheckPath_(link->value.node, sink, fcond)) return false;
584 }
585 return true;
586 }
587 /*!
588 * \brief Check all the node and edge pattern
589 * between src and sink satisfies fcond.
590 *
591 * src is not checked.
592 *
593 * \param src The source node.
594 * \param sink The termination node.
595 * \param fcond The condition to be checked.
596 * \tparam F the condition function, with signature
597 * \note sink must be a post-dominator of src.
598 */
599 template <typename F>
600 bool CheckPath(IndexedForwardGraph::Node* src, IndexedForwardGraph::Node* sink, F fcond) {
601 ICHECK(!src->extern_ref);
602 visited_.clear();
603 ICHECK(src != sink);
604 for (auto link = src->outputs.head; link != nullptr; link = link->next) {
605 if (!CheckPath_(link->value.node, sink, fcond)) return false;
606 }
607 return true;
608 }
609 // Combine two patterns together.
610 static OpPatternKind CombinePattern(OpPatternKind lhs, OpPatternKind rhs) {
611 if (lhs > kBroadcast && rhs > kBroadcast) {
612 LOG(FATAL) << "Cannot merge two complex group together";
613 }
614 if (lhs > rhs) return lhs;
615 return rhs;
616 }
617 /*!
618 * \brief Merge the child group to the parent.
619 * \param child The child group.
620 * \param parent The parent group.
621 */
622 void MergeFromTo(Group* child, Group* parent) {
623 child = child->FindRoot();
624 parent = parent->FindRoot();
625 if (child == parent) return;
626 // update the number of nodes of the parent group
627 parent->num_nodes += child->num_nodes;
628 child->parent = parent;
629 // update anchor ref and pattern
630 if (child->anchor_ref != nullptr) {
631 ICHECK(parent->anchor_ref == nullptr);
632 parent->anchor_ref = child->anchor_ref;
633 parent->pattern = CombinePattern(child->pattern, parent->pattern);
634 }
635 }
636 // Internal implelementation of CommitFuse
637 void CommitFuse_(IndexedForwardGraph::Node* src, IndexedForwardGraph::Node* sink, Group* target) {
638 if (src == sink) return;
639 if (visited_.count(src)) return;
640 visited_.insert(src);
641 Group* gnode = groups_[src->index];
642 ICHECK(gnode != nullptr);
643 // merge the current group to the parent if possible.
644 MergeFromTo(gnode, target);
645 for (auto link = src->outputs.head; link != nullptr; link = link->next) {
646 CommitFuse_(link->value.node, sink, target);
647 }
648 }
649 /*!
650 * \brief Commit fusion operation.
651 * \param src The source node.
652 * \param sink The termination node.
653 * \note sink must be a post-dominator of src.
654 */
655 void CommitFuse(IndexedForwardGraph::Node* src, IndexedForwardGraph::Node* sink) {
656 Group* target = groups_[sink->index];
657 visited_.clear();
658 ICHECK(src != sink);
659 CommitFuse_(src, sink, target);
660 }
661
662 size_t CountNodesUptoSink_(IndexedForwardGraph::Node* src, IndexedForwardGraph::Node* sink) {
663 if (src == sink || visited_.count(src)) return 0;
664 visited_.insert(src);
665 Group* gnode = groups_[src->index];
666 ICHECK(gnode != nullptr);
667 auto sum = gnode->num_nodes;
668 for (auto link = src->outputs.head; link != nullptr; link = link->next) {
669 sum += CountNodesUptoSink_(link->value.node, sink);
670 }
671 return sum;
672 }
673
674 // Count the number of nodes in a fused subgraph if child is additionaly fused.
675 // dom_parent is already known to be a part of the subgraph.
676 // For a diamond structure, there can be multiple paths connecting child and dom_parent.
677 // All intermediate nodes between child and dom_parent are taken into account.
678 // Since dom_parent can itself be an intermediate node in the subgraph, calling FindRoot()
679 // is important for correct calculation.
680 size_t CountFusedNodesWithNewChild(IndexedForwardGraph::Node* child,
681 IndexedForwardGraph::Node* dom_parent) {
682 Group* target = groups_[dom_parent->index];
683 visited_.clear();
684 ICHECK(child != dom_parent);
685 return target->FindRoot()->num_nodes + CountNodesUptoSink_(child, dom_parent);
686 }
687
688 // Initialize the groups.
689 void InitGroups(const IndexedForwardGraph& graph) {
690 groups_.resize(graph.post_dfs_order.size());
691 for (size_t nid = 0; nid < groups_.size(); ++nid) {
692 const auto* graph_node = graph.post_dfs_order[nid];
693 auto* group_node = arena_->make<Group>();
694 group_node->pattern = graph_node->pattern;
695 group_node->root_ref = graph_node->ref;
696 // set anchor ref if necessary.
697 if (group_node->pattern == kOutEWiseFusable) {
698 group_node->anchor_ref = graph_node->ref;
699 }
700 groups_[nid] = group_node;
701 }
702 }
703
704 // execute the fusion algorithm.
705 void RunFuse(const IndexedForwardGraph& graph, const DominatorTree& post_dom_tree, int phase) {
706 for (size_t nid = 0; nid < groups_.size(); ++nid) {
707 // the group of current node has been specified already.
708 auto* graph_node = graph.post_dfs_order[nid];
709 auto* dom_node = post_dom_tree.nodes[nid];
710 Group* group_node = groups_[nid];
711 ICHECK(group_node != nullptr);
712 // no actions for opaque nodes
713 if (group_node->pattern == kOpaque) continue;
714 // no actions needed if the current node have no dominator
715 if (dom_node->parent == nullptr) continue;
716 ICHECK(!graph_node->extern_ref);
717 size_t dom_parent_gindex = dom_node->parent->gnode->index;
718
719 // refuse the fusion if too many ops are going to be fused together
720 if (CountFusedNodesWithNewChild(graph_node, dom_node->parent->gnode) > max_fuse_depth_)
721 continue;
722
723 if (phase == 2) {
724 // Fuse injective ops into intermediate tuples, if any
725 if (group_node->pattern > kInjective) continue;
726 Group* dom_parent_group = groups_[dom_parent_gindex];
727 Group* dom_root_group = dom_parent_group->FindRoot();
728 // If dom node group has a tuple as its root, we do not fuse tuple fields into it
729 if (dom_root_group->pattern == kTuple) continue;
730 if (dom_parent_group->pattern == kTuple && dom_root_group->pattern <= kInjective) {
731 // Now we know the tuple has been fused into subsequent injective ops
732 auto fcond = [](OpPatternKind kind, bool is_sink) { return kind <= kInjective; };
733 // dom_root_group can also be tuple, as in inception layers
734 // CheckPath is needed to avoid fusing two intermediate tuples
735 if (CheckPath(graph_node, dom_node->parent->gnode, fcond)) {
736 CommitFuse(graph_node, dom_node->parent->gnode);
737 }
738 }
739 continue;
740 }
741
742 // Skip if current node is already fused to the parent.
743 if (groups_[dom_parent_gindex] != nullptr &&
744 group_node->FindRoot() == groups_[dom_parent_gindex]->FindRoot()) {
745 continue;
746 }
747 // Do not fuse into tuple for now
748 if (groups_[dom_parent_gindex]->pattern == kTuple) continue;
749 // Try to fuse current node to its post-dominator.
750 if (group_node->pattern == kOutEWiseFusable) {
751 if (phase != 0) continue;
752 // Path for OutEWiseFusable: conv2d
753 // Check if the dominator relation is elemwise.
754 if (dom_node->parent != nullptr && dom_node->pattern == kElemWise) {
755 ICHECK(dom_node->parent->gnode != nullptr);
756 // The fuse can be executed if all the intermediate ops are still broadcast.
757 auto fcond = [](OpPatternKind kind, bool is_sink) { return kind <= kBroadcast; };
758 if (CheckPath(graph_node, dom_node->parent->gnode, fcond)) {
759 CommitFuse(graph_node, dom_node->parent->gnode);
760 }
761 }
762 } else if (group_node->pattern <= kBroadcast) {
763 // Pre-condition: can only be fused to parent which is injective or reduction.
764 if (dom_node->parent != nullptr &&
765 (dom_node->pattern <= kInjective || dom_node->pattern == kCommReduce)) {
766 // Check if all the intermediate ops are still broadcast.
767 // The final terminal node can already be fused to a OutEWiseFusable group.
768 auto fcond = [](OpPatternKind kind, bool is_sink) {
769 if (!is_sink) {
770 // Elemwise, broadcast, and injective ops on the parallel branches
771 // are allowed be fused to the elemwise/broadcast anchor.
772 return kind <= kInjective;
773 } else {
774 return (kind <= kBroadcast || kind == kCommReduce || kind == kInjective ||
775 kind == kOutEWiseFusable);
776 }
777 };
778 if (CheckPath(graph_node, dom_node->parent->gnode, fcond)) {
779 CommitFuse(graph_node, dom_node->parent->gnode);
780 }
781 }
782 } else if (group_node->pattern == kInjective || group_node->pattern == kTuple) {
783 // defer injective fusion to second phase.
784 // so conv2d always finishes fusing.
785 if (phase != 1) continue;
786 // Check if all path are injective.
787 auto fcond = [](OpPatternKind kind, bool is_sink) { return kind <= kInjective; };
788 if (CheckPath(graph_node, dom_node->parent->gnode, fcond)) {
789 CommitFuse(graph_node, dom_node->parent->gnode);
790 }
791 } else {
792 // do nothing.
793 ICHECK(group_node->pattern == kCommReduce);
794 }
795 }
796 }
797};

799std::vector<GraphPartitioner::Group*> GraphPartitioner::Partition(
800 const IndexedForwardGraph& graph) {
801 this->InitGroups(graph);
802 if (opt_level_ == 0) return std::move(groups_);
803 // get post dominator tree
804 auto post_dom_tree = DominatorTree::PostDom(arena_, graph);
805 // run fusion algorithm.
806 for (int phase = 0; phase < 3; ++phase) {
807 this->RunFuse(graph, post_dom_tree, phase);
808 }
809 return std::move(groups_);
810}

846 /*! \brief Temporary information from each group. */
847 struct GroupInfo {
848 public:
849 // The parameters of the function.
850 Array<Var> params;
851 // The arguments to call the functions.
852 Array<Expr> arguments;
853 // Get a new parameter or allocate an old one
854 Var GetOrAllocParam(const Expr& expr, const Type& type) {
855 // run linear scan as most fused groups contain only a few inputs.
856 for (size_t i = 0; i < arguments.size(); ++i) {
857 if (expr.same_as(arguments[i])) return params[i];
858 }
859 // create a new parameter.
860 std::ostringstream os;
861 os << "p" << params.size();
862 auto var = Var(os.str(), type);
863 params.push_back(var);
864 arguments.push_back(expr);
865 return var;
866 }
867 };
868 /*! \brief Internal arena. */
869 support::Arena arena_;
870 /*! \brief The group assignment map. */
871 std::unordered_map<const Object*, GraphPartitioner::Group*> gmap_;
872 /* \brief Internal group information map. */
873 std::unordered_map<GraphPartitioner::Group*, GroupInfo> ginfo_;

875 // Skip primitive function.
876 Expr VisitExpr_(const FunctionNode* fn_node) {
877 if (fn_node->HasNonzeroAttr(attr::kPrimitive)) {
878 return GetRef<Expr>(fn_node);
879 } else {
880 return ExprMutator::VisitExpr_(fn_node);
881 }
882 }
883
884 // Transform calls.
885 Expr Rewrite_(const CallNode* call, const Expr& post) {
886 if (call->op.as<OpNode>()) {
887 static auto fnoncomputational = Op::GetAttrMap<TNonComputational>("TNonComputational");
888 static auto fqnncanonicalize = Op::GetAttrMap<FTVMLegalize>("FTVMQnnCanonicalize");
889
890 Op op = Downcast<Op>(call->op);
891 if (fnoncomputational.get(op, false) && !fqnncanonicalize.count(op)) {
892 return ExprMutator::VisitExpr_(call);
893 }
894
895 // If it is a primitive op call
896 // then we must have a group assignment for it already.
897 ICHECK(gmap_.count(call));
898 if (call->op == stop_fusion_op) {
899 return ExprMutator::VisitExpr(call->args[0]);
900 }
901 auto* ret_group = gmap_.at(call)->FindRoot();
902 Array<Expr> new_args = GetNewArguments(call->args, ret_group);
903
904 auto new_call = Call(call->op, new_args, call->attrs, call->type_args, call->span);
905
906 if (ret_group->root_ref == call) {
907 // This is the root of the group
908 // create the new call node.
909 return MakeNewFunction(ret_group, call->checked_type(), new_call);
910 } else {
911 // This is an intermediate node of a fused function
912 // simply return the new call.
913 return std::move(new_call);
914 }
915 } else {
916 return ExprMutator::VisitExpr_(call);
917 }
918 }
919
920 Expr Rewrite_(const TupleNode* tuple_node, const Expr& post) {
921 auto* ret_group = gmap_.at(tuple_node)->FindRoot();
922 if (ret_group->root_ref == tuple_node) {
923 return ExprMutator::VisitExpr_(tuple_node);
924 }
925 // This tuple is an intermediate node in the group
926 Array<Expr> new_fields = GetNewArguments(tuple_node->fields, ret_group);
927 return WithFields(GetRef<Tuple>(tuple_node), new_fields);
928 }
929
930 Expr Rewrite_(const TupleGetItemNode* tuple_get, const Expr& post) {
931 auto* ret_group = gmap_.at(tuple_get)->FindRoot();
932 auto new_tuple = GetNewArguments({tuple_get->tuple}, ret_group)[0];
933 auto new_node = TupleGetItem(new_tuple, tuple_get->index);
934 if (ret_group->root_ref == tuple_get) {
935 if (gmap_.at(tuple_get->tuple.get())->FindRoot() != ret_group) {
936 // Isolated. This case occurs when tuple is created by an Opaque op
937 // e.g. multibox_transform_loc
938 return ExprMutator::VisitExpr_(tuple_get);
939 }
940 // A new function whose output is a tuple field access
941 return MakeNewFunction(ret_group, tuple_get->checked_type(), new_node);
942 }
943 // This is an intermediate node in the group
944 return std::move(new_node);
945 }

947 Expr VisitExpr_(const LetNode* op) final {
948 auto pre_visit = [this](const LetNode* op) {
949 // Rely on the Memoizer to cache pre-visit values
950 this->VisitExpr(op->var);
951 this->VisitExpr(op->value);
952 };
953 auto post_visit = [this](const LetNode* op) {
954 // Rely on the Memoizer to cache pre-visit values
955 Var var = Downcast<Var>(this->VisitExpr(op->var));
956 Expr value = this->VisitExpr(op->value);
957 // Visit body and cache the op
958 Expr body = this->VisitExpr(op->body);
959 auto expr = GetRef<Expr>(op);
960 if (var.same_as(op->var) && value.same_as(op->value) && body.same_as(op->body)) {
961 this->memo_[expr] = expr;
962 } else {
963 this->memo_[expr] = Let(var, value, body);
964 }
965 };
966 ExpandANormalForm(op, pre_visit, post_visit);
967 return memo_[GetRef<Expr>(op)];
968 }

970 Expr MakeNewFunction(GraphPartitioner::Group* group, Type ret_type, Expr body) {
971 // Quickly check special properties of the fused function.
972 // A pass to check if the fused op contains only reshape ops.
973 class CheckReshapeOnly : public ExprVisitor {
974 public:
975 void VisitExpr_(const CallNode* cn) final {
976 this->has_call = true;
977 static auto freshape_op = Op::GetAttrMap<TReshapeOp>("TReshapeOp");
978
979 if (!freshape_op.get(cn->op, false)) {
980 this->reshape_only = false;
981 }
982
983 if (!this->reshape_only) return;
984 ExprVisitor::VisitExpr_(cn);
985 }
986
987 void VisitExpr_(const VarNode* vn) final {
988 if (!vn->type_annotation.defined() || !vn->type_annotation->IsInstance<TensorTypeNode>()) {
989 this->reshape_only = false;
990 }
991 }
992
993 bool reshape_only = true;
994 bool has_call = false;
995 } visitor;
996
997 visitor(body);
998 const GroupInfo& ginfo = ginfo_[group];
999 auto func = Function(ginfo.params, body, ret_type, {});
1000 func = WithAttr(std::move(func), attr::kPrimitive, tvm::Integer(visitor.has_call));
1001 // TODO(mbs): "reshape" cleanup.
1002 if (visitor.has_call && visitor.reshape_only) {
1003 func = WithAttr(std::move(func), attr::kReshapeOnly, tvm::Integer(visitor.reshape_only));
1004 }
1005 return Call(func, ginfo.arguments, Attrs());
1006 }

1008 Array<Expr> GetNewArguments(const tvm::Array<Expr>& args,
1009 GraphPartitioner::Group* current_group) {
1010 Array<Expr> new_args;
1011 for (auto arg : args) {
1012 auto* arg_group = gmap_.at(arg.get())->FindRoot();
1013 auto type = arg->checked_type();
1014 Expr new_arg = this->Mutate(arg);
1015 if (current_group != arg_group) {
1016 if (!link_params_ || new_arg.as<ConstantNode>() == nullptr) {
1017 Var param = ginfo_[current_group].GetOrAllocParam(new_arg, type);
1018 new_args.push_back(param);
1019 } else {
1020 new_args.push_back(new_arg);
1021 }
1022 } else {
1023 new_args.push_back(new_arg);
1024 }
1025 }
1026 return new_args;
1027 }
