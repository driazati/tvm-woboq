36class LayoutIndexQueue {
37 public:
38 static LayoutIndexQueue* Global() {
39 static LayoutIndexQueue inst;
40 return &inst;
41 }
42
43 void Clear() {
44 std::lock_guard<std::mutex> lock(mutex_);
45 queue_.clear();
46 }
47
48 private:
49 friend class MetaScheduleLayoutRewriter;
50 std::mutex mutex_;
51 std::deque<tir::IndexMap> queue_;
52};
53
54void MetaScheduleLayoutRewriter::LayoutQueuePush(const tir::IndexMap& index_map) {
55 LayoutIndexQueue* self = LayoutIndexQueue::Global();
56 {
57 std::lock_guard<std::mutex> lock(self->mutex_);
58 self->queue_.push_back(index_map);
59 }
60}

62bool IsSupportedOp(const OpNode* op) {
63 static std::vector<std::string> target_ops{
64 "nn.conv2d", //
65 "nn.contrib_conv2d_winograd_without_weight_transform",
66 "nn.conv3d",
67 "nn.matmul",
68 "nn.dense",
69 "nn.batch_matmul",
70 };
71 return std::find(target_ops.begin(), target_ops.end(), op->name) != target_ops.end();
72}

74#define TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(Attr, AttrType, OriginalShape, Result) \
75 if (const AttrType* ptr = Attr.as<AttrType>()) { \
76 ObjectPtr<AttrType> n = make_object<AttrType>(*ptr); \
77 n->meta_schedule_original_shape = OriginalShape; \
78 Result = Attrs(n); \
79 }

81// Mutate ops in a function
82class MetaScheduleFuncMutator : public ExprMutator {
83 public:
84 explicit MetaScheduleFuncMutator(std::deque<tir::IndexMap>&& layout_queue)
85 : layout_queue_(std::move(layout_queue)) {}
86
87 Expr VisitExpr_(const CallNode* call) {
88 Expr expr = ExprMutator::VisitExpr_(call);
89 if (layout_queue_.empty()) {
90 return expr;
91 }
92 if (const auto* call = expr.as<CallNode>()) {
93 if (const auto* op = call->op.as<OpNode>()) {
94 if (IsSupportedOp(op)) {
95 ICHECK_EQ(call->args.size(), 2);
96 tir::IndexMap index_map = layout_queue_.front();
97 layout_queue_.pop_front();
98 Var var = Downcast<Var>(call->args[1]);
99 Array<PrimExpr> shape = Downcast<TensorType>(var->type_annotation)->shape;
100 Attrs attrs{nullptr};
101 TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(call->attrs, Conv2DAttrs, shape, attrs);
102 TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(call->attrs, Conv2DWinogradAttrs, shape, attrs);
103 TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(call->attrs, Conv3DAttrs, shape, attrs);
104 TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(call->attrs, MatmulAttrs, shape, attrs);
105 TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(call->attrs, DenseAttrs, shape, attrs);
106 TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE(call->attrs, BatchMatmulAttrs, shape, attrs);
107 ICHECK(attrs.defined()) << "TypeError: Unknown attribute: " << call->attrs;
108 expr = Call(call->op,
109 {call->args[0], MakeMetaScheduleLayoutTransform(call->args[1], index_map)},
110 attrs);
111 }
112 }
113 }
114 return expr;
115 }
116
117 private:
118 std::deque<tir::IndexMap> layout_queue_;
119};
120
121#undef TVM_RELAY_LAYOUT_WITH_ORIGINAL_SHAPE

123Expr MetaScheduleLayoutRewriter::VisitExpr_(const CallNode* call) {
124 Expr expr = ExprMutator::VisitExpr_(call);
125 call = expr.as<CallNode>();
126 if (call != nullptr) {
127 if (const auto* func = call->op.as<FunctionNode>()) {
128 LayoutIndexQueue* self = LayoutIndexQueue::Global();
129 self->queue_.clear();
130 tec::PrimFuncFor(GetRef<Function>(func), Target::Current(), GlobalVarSupply(NameSupply("")));
131 if (!self->queue_.empty()) {
132 std::deque<tir::IndexMap> queue = std::move(self->queue_);
133 self->queue_.clear();
134 return MetaScheduleFuncMutator(std::move(queue)).VisitExpr(expr);
135 }
136 }
137 }
138 return expr;
139}

143Pass MetaScheduleLayoutRewrite() {
144 runtime::TypedPackedFunc<Function(Function, IRModule, PassContext)> pass_func =
145 [=](Function f, IRModule m, PassContext pc) -> Function {
146 return Downcast<Function>(MetaScheduleLayoutRewriter().Mutate(std::move(f)));
147 };
148 return CreateFunctionPass(pass_func, 3, "MetaScheduleLayoutRewrite", {"InferType"});
149}

151#define TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(Attrs, AttrType) \
152 if (const auto* p = Attrs.as<AttrType>()) { \
153 return p->meta_schedule_original_shape; \
154 }
155
156TVM_REGISTER_GLOBAL("relay.attrs.get_meta_schedule_original_shape")
157 .set_body_typed([](const Attrs& attrs) -> Array<PrimExpr> {
158 TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(attrs, Conv2DAttrs);
159 TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(attrs, Conv2DWinogradAttrs);
160 TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(attrs, Conv3DAttrs);
161 TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(attrs, MatmulAttrs);
162 TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(attrs, DenseAttrs);
163 TVM_RELAY_META_SCHEDULE_LAYOUT_REWRITE_GET_ORIGINAL_SHAPE(attrs, BatchMatmulAttrs);
164 LOG(FATAL) << "TypeError: Unknown attribute: " << attrs;
165 throw;
166 });
167TVM_REGISTER_GLOBAL("relay._transform.MetaScheduleLayoutRewrite")
168 .set_body_typed(MetaScheduleLayoutRewrite);
