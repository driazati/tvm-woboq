38bool DFPatternMatcher::Match(const DFPattern& pattern, const Expr& expr) {
39 VLOG(1) << "Match " << PrettyPrint(pattern) << " in:" << std::endl << PrettyPrint(expr);
40 memo_.clear();
41 matched_nodes_.clear();
42 return VisitDFPattern(pattern, expr);
43}

45void DFPatternMatcher::ClearMap(size_t watermark) {
46 for (size_t i = watermark; i < matched_nodes_.size(); ++i) {
47 memo_.erase(matched_nodes_[i]);
48 }
49 matched_nodes_.erase(matched_nodes_.begin() + watermark, matched_nodes_.end());
50}

52bool DFPatternMatcher::VisitDFPattern(const DFPattern& pattern, const Expr& expr) {
53 if (memoize_ && memo_.count(pattern)) {
54 ICHECK_EQ(memo_[pattern].size(), 1);
55 return expr.same_as(memo_[pattern][0]);
56 } else {
57 auto watermark = matched_nodes_.size();
58 auto out = DFPatternFunctor::VisitDFPattern(pattern, expr);
59 if (out) {
60 memo_[pattern].push_back(expr);
61 matched_nodes_.push_back(pattern);
62 VLOG(1) << "Matched " << PrettyPrint(pattern) << " at:" << std::endl << PrettyPrint(expr);
63 } else {
64 ClearMap(watermark);
65 }
66 return out;
67 }
68}

70bool DFPatternMatcher::VisitDFPattern_(const AltPatternNode* op, const Expr& expr) {
71 return VisitDFPattern(op->left, expr) || VisitDFPattern(op->right, expr);

74bool MatchRetValue(const ObjectRef& lhs, const TVMRetValue& rhs) {
75 switch (rhs.type_code()) {
76 case kDLInt:
77 if (auto* val = lhs.as<IntImmNode>()) {
78 return val->value == rhs.operator int64_t();
79 }
80 break;
81 case kDLFloat:
82 if (auto* val = lhs.as<FloatImmNode>()) {
83 return val->value == rhs.operator double();
84 }
85 break;
86 case kTVMStr:
87 if (auto* val = lhs.as<tir::StringImmNode>()) {
88 return val->value == rhs.operator std::string();
89 } else if (auto* val = lhs.as<StringObj>()) {
90 return val->data == rhs.operator std::string();
91 }
92 break;
93 case kTVMDataType:
94 if (auto* val = lhs.as<tir::StringImmNode>()) {
95 return rhs.operator std::string() == val->value;
96 } else if (auto* val = lhs.as<StringObj>()) {
97 return rhs.operator std::string() == val->data;
98 } else {
99 ICHECK(false) << "PatternMatcher: Unsupported TVMDataType " << lhs;
100 }
101 break;
102 case kTVMObjectHandle:
103 if (rhs.IsObjectRef<String>()) {
104 if (auto* val = lhs.as<tir::StringImmNode>()) {
105 return rhs.operator String() == val->value;
106 } else if (auto* val = lhs.as<StringObj>()) {
107 return rhs.operator String() == val->data;
108 }
109 } else {
110 // Compare the objects for structural equality
111 static auto* structural_equal = runtime::Registry::Get("node.StructuralEqual");
112 ICHECK(structural_equal) << "node.StructuralEqual is not registered.";
113 if ((*structural_equal)(lhs, GetRef<ObjectRef>(rhs.ptr<Object>()), false, true)) {
114 return true;
115 }
116 }
117 break;
118 default:
119 ICHECK(false) << "Unsupported type code in Pattern Node " << rhs.type_code();
120 }
121 return false;
122}

124bool DFPatternMatcher::VisitDFPattern_(const AttrPatternNode* attr_pattern, const Expr& expr) {
125 bool matches = VisitDFPattern(attr_pattern->pattern, expr);
126 if (!matches) {
127 return matches;
128 }
129 auto attributes = attr_pattern->attrs.as<DictAttrsNode>()->dict;
130 if (const auto* op_node = expr.as<OpNode>()) {
131 Op op = GetRef<Op>(op_node);
132 for (auto kv : attributes) {
133 auto attr_name = kv.first;
134 auto attr_value = kv.second;
135 if (Op::HasAttrMap(attr_name)) {
136 auto op_map = Op::GetAttrMap<TVMRetValue>(attr_name);
137 if (op_map.count(op)) {
138 matches &= MatchRetValue(attr_value, op_map[op]);
139 } else {
140 matches = false;
141 }
142 } else {
143 matches = false;
144 }
145 }
146 } else if (auto* op = expr.as<CallNode>()) {
147 matches = true;
148 // TODO(mbrookhart): When OpNode Attrs move from TVMRetValue to the Object system, remove this
149 // and replace the whole thing with a Visitor-based approach
150 ReflectionVTable* reflection = ReflectionVTable::Global();
151 auto attrs_node = const_cast<BaseAttrsNode*>(op->attrs.get());
152 // attrs may be undefined on non-op calls so we check first
153 std::vector<std::string> attr_names;
154 if (attrs_node) {
155 attr_names = reflection->ListAttrNames(attrs_node);
156 }
157 for (auto kv : attributes) {
158 std::string attr = kv.first;
159 if (matches && std::find(attr_names.begin(), attr_names.end(), attr) != attr_names.end()) {
160 matches &= MatchRetValue(kv.second, reflection->GetAttr(attrs_node, attr));
161 } else {
162 matches = false;
163 break;
164 }
165 }
166 } else if (auto* op = expr.as<FunctionNode>()) {
167 matches = true;
168 for (auto kv : attributes) {
169 if (matches && op->attrs.defined() && op->attrs->dict.count(kv.first)) {
170 matches &= StructuralEqual()(kv.second, op->attrs->dict[kv.first]);
171 } else {
172 matches = false;
173 break;
174 }
175 }
176 } else {
177 matches = false;
178 }
179 return matches;
180}

182Array<DFPattern> reverse(const Array<DFPattern>& args) {
183 Array<DFPattern> new_args;
184 for (auto it = args.rbegin(); it != args.rend(); ++it) {
185 new_args.push_back(*it);
186 }
187 return new_args;

190bool DFPatternMatcher::VisitDFPattern_(const CallPatternNode* op, const Expr& expr) {
191 // utilities
192 auto get_op_node = [](const CallPatternNode* op) -> const tvm::OpNode* {
193 if (op) {
194 if (auto* expr_pattern = op->op.as<ExprPatternNode>()) {
195 return expr_pattern->expr.as<OpNode>();
196 }
197 }
198 return nullptr;
199 };
200 auto is_pattern_op = [&get_op_node](const CallPatternNode* op, std::string op_type) {
201 if (const auto* op_node = get_op_node(op)) {
202 if (op_node->name == op_type) {
203 return true;
204 }
205 }
206 return false;
207 };
208 auto is_expr_op = [](const Expr& expr, std::string op_type) {
209 if (const auto* call_node = expr.as<CallNode>()) {
210 if (const auto* op_node = call_node->op.as<OpNode>()) {
211 if (op_node->name == op_type) {
212 return true;
213 }
214 }
215 }
216 return false;
217 };
218
219 // logic
220 auto watermark = matched_nodes_.size();
221 if (const auto* call_node = expr.as<CallNode>()) {
222 auto matches_op = VisitDFPattern(op->op, call_node->op);
223 if (matches_op) {
224 auto watermark2 = matched_nodes_.size();
225
226 auto match_args = [this, &watermark2](const Array<DFPattern> pattern_args,
227 const Array<Expr> expr_args) {
228 bool matches = true;
229 size_t i = 0;
230 if (pattern_args.defined()) {
231 if (pattern_args.size() == expr_args.size()) {
232 while (matches && i < pattern_args.size()) {
233 matches &= VisitDFPattern(pattern_args[i], expr_args[i]);
234 ++i;
235 }
236 } else {
237 matches = false;
238 }
239 }
240 if (!matches) {
241 ClearMap(watermark2);
242 }
243 return matches;
244 };
245
246 // Standard case
247 if (match_args(op->args, call_node->args)) {
248 return true;
249 }
250 // Commutative Matching
251 if (const OpNode* op_node = get_op_node(op)) {
252 if ((op_node->name == "add") || (op_node->name == "multiply")) {
253 if (match_args(reverse(op->args), call_node->args)) {
254 return true;
255 }
256 }
257 }
258 } else {
259 ClearMap(watermark);
260 // associate divide/multiply
261 if (is_pattern_op(op, "divide")) {
262 if (const auto* arg_node = op->args[0].as<CallPatternNode>()) {
263 if (is_pattern_op(arg_node, "multiply") && is_expr_op(expr, "multiply") &&
264 (is_expr_op(call_node->args[0], "divide") ||
265 is_expr_op(call_node->args[1], "divide"))) {
266 bool out = false;
267 for (size_t arg_id = 0; arg_id < 2; ++arg_id) {
268 auto div = CallPattern(op->op, {arg_node->args[arg_id], op->args[1]});
269 auto mul = CallPattern(arg_node->op, {arg_node->args[(arg_id + 1) % 2], div});
270 out = VisitDFPattern(mul, expr);
271 if (out) {
272 return true;
273 } else {
274 ClearMap(watermark);
275 }
276 }
277 return out;
278 }
279 }
280 }
281 if (is_pattern_op(op, "multiply")) {
282 // associate multiply/divide
283 for (size_t arg_id = 0; arg_id < 2; ++arg_id) {
284 if (auto* arg_node = op->args[arg_id].as<CallPatternNode>()) {
285 if (is_pattern_op(arg_node, "divide") && is_expr_op(expr, "divide") &&
286 (is_expr_op(call_node->args[0], "multiply") ||
287 is_expr_op(call_node->args[1], "multiply"))) {
288 auto mul = CallPattern(op->op, {arg_node->args[0], op->args[(arg_id + 1) % 2]});
289 auto div = CallPattern(arg_node->op, {mul, arg_node->args[1]});
290 return VisitDFPattern(div, expr);
291 }
292 }
293 }
294 }
295 }
296 }
297 return false;
298}

300// Recursively find the Dominator parent along all inputs paths.
301bool DFPatternMatcher::MatchesPath(const DominatorPatternNode* op, const Expr& expr) {
302 auto call_node = expr.as<CallNode>();
303 auto index_node = expr_to_node(expr);
304 for (auto node : index_node->inputs_) {
305 if (!(call_node && node->ref() == call_node->op)) {
306 memoize_ = true;
307 if (VisitDFPattern(op->parent, node->ref())) {
308 return true;
309 } else {
310 memoize_ = false;
311 if (!VisitDFPattern(op->path, node->ref())) {
312 return false;
313 }
314 if (!MatchesPath(op, node->ref())) {
315 return false;
316 }
317 }
318 }
319 }
320 return true;
321}

323// Iteratively ensure that the parent is dominated somewhere by the child or the path
324bool DFPatternMatcher::DominatesParent(const DominatorPatternNode* op, const Expr& expr) {
325 std::stack<Expr> stack;
326 std::unordered_set<const ExprNode*> visited;
327 stack.push(expr);
328 while (!stack.empty()) {
329 Expr current = stack.top();
330 stack.pop();
331 for (auto node : expr_to_node(current)->dominator_children_) {
332 if (visited.count(node->node_ref_) == 0) {
333 if (VisitDFPattern(op->parent, node->ref())) {
334 return true;
335 } else {
336 stack.push(node->ref());
337 }
338 visited.insert(node->node_ref_);
339 }
340 }
341 }
342 return false;
343}

345bool DFPatternMatcher::VisitDFPattern_(const DominatorPatternNode* op, const Expr& expr) {
346 if (VisitDFPattern(op->child, expr)) {
347 bool matches_path = MatchesPath(op, expr);
348 memoize_ = true;
349 if (matches_path) {
350 return DominatesParent(op, expr);
351 }
352 }
353 return false;
354}

360bool DFPatternMatcher::VisitDFPattern_(const FunctionPatternNode* op, const Expr& expr) {
361 bool matches = false;
362 if (const auto* func = expr.as<FunctionNode>()) {
363 matches = true;
364 if (op->params.defined()) {
365 size_t i = 0;
366 if (op->params.size() == func->params.size()) {
367 while (matches && i < op->params.size()) {
368 matches &= VisitDFPattern(op->params[i], func->params[i]);
369 ++i;
370 }
371 } else {
372 matches = false;
373 }
374 }
375 if (matches) {
376 matches &= VisitDFPattern(op->body, func->body);
377 }
378 }
379 return matches;
380}

382bool DFPatternMatcher::VisitDFPattern_(const TupleGetItemPatternNode* op, const Expr& expr) {
383 bool matches = false;
384 if (const auto* tuple_get_item_node = expr.as<TupleGetItemNode>()) {
385 matches = (op->index == -1 || op->index == tuple_get_item_node->index) &&
386 VisitDFPattern(op->tuple, tuple_get_item_node->tuple);
387 }
388 return matches;
389}

391bool DFPatternMatcher::VisitDFPattern_(const TuplePatternNode* op, const Expr& expr) {
392 bool matches = false;
393 if (const auto* tuple_node = expr.as<TupleNode>()) {
394 matches = true;
395 if (op->fields.defined()) {
396 if (op->fields.size() == tuple_node->fields.size()) {
397 size_t i = 0;
398 while (matches && i < op->fields.size()) {
399 matches &= VisitDFPattern(op->fields[i], tuple_node->fields[i]);
400 ++i;
401 }
402 } else {
403 matches = false;
404 }
405 }
406 }
407 return matches;

410bool DFPatternMatcher::VisitDFPattern_(const IfPatternNode* op, const Expr& expr) {
411 if (const auto* if_node = expr.as<IfNode>()) {
412 auto cond = if_node->cond;
413 auto true_branch = if_node->true_branch;
414 auto false_branch = if_node->false_branch;
415 return VisitDFPattern(op->cond, cond) && VisitDFPattern(op->true_branch, true_branch) &&
416 VisitDFPattern(op->false_branch, false_branch);
417 }
418 return false;
419}

421bool DFPatternMatcher::VisitDFPattern_(const LetPatternNode* op, const Expr& expr) {
422 if (const auto* let_node = expr.as<LetNode>()) {
423 return VisitDFPattern(op->var, let_node->var) && VisitDFPattern(op->value, let_node->value) &&
424 VisitDFPattern(op->body, let_node->body);
425 }
426 return false;
427}

429Expr InferType(const Expr& expr) {
430 auto mod = IRModule::FromExpr(expr);
431 mod = transform::InferType()(mod);
432 if (expr.as<FunctionNode>()) {
433 return mod->Lookup("main");
434 } else {
435 return mod->Lookup("main").as<FunctionNode>()->body;
436 }
437}

439Expr InferTypeWithModule(const Expr& expr, const IRModule& m) {
440 IRModule mod(m->functions, m->type_definitions, m->Imports());
441 int idx = 0;
442 std::string gv_name;
443 do {
444 std::ostringstream oss;
445 oss << "_tmp" << idx;
446 gv_name = oss.str();
447 ++idx;
448 } while (mod->ContainGlobalVar(gv_name));
449 GlobalVar gvar(gv_name);
450 BaseFunc func;
451 if (expr.as<FunctionNode>()) {
452 func = Downcast<Function>(expr);
453 } else {
454 func = relay::Function(relay::FreeVars(expr), expr, Type(), relay::FreeTypeVars(expr, mod), {});
455 }
456 mod->Add(gvar, func);
457 mod = transform::InferType()(mod);
458 Expr ret;
459 if (expr.as<FunctionNode>()) {
460 ret = mod->Lookup(gvar);
461 } else {
462 ret = mod->Lookup(gvar).as<FunctionNode>()->body;
463 }
464 return ret;
465}

514/*! \brief Creates a new set of nodes based on Group inputs, used to create functions and perform
515 * group overlap analysis */
516class MatchExtractor : public ExprMutator {
517 public:
518 explicit MatchExtractor(
519 const std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual>& inputs)
520 : inputs_(inputs) {}
521 const std::unordered_map<Expr, Expr, ObjectPtrHash, ObjectPtrEqual>& GetMemo() {
522 return this->memo_;
523 }
524 const std::string& GetName() { return name_; }
525
526 protected:
527 Expr VisitExpr(const Expr& pre) override {
528 if (inputs_.count(pre)) {
529 return inputs_.at(pre);
530 }
531 return ExprMutator::VisitExpr(pre);
532 }
533 Expr VisitExpr_(const TupleNode* op) override {
534 auto out = ExprMutator::VisitExpr_(op);
535 name_ += "Tuple_";
536 return out;
537 };
538 Expr VisitExpr_(const FunctionNode* op) override {
539 auto out = ExprMutator::VisitExpr_(op);
540 name_ += "Function";
541 return out;
542 };
543 Expr VisitExpr_(const CallNode* call_node) override {
544 auto out = ExprMutator::VisitExpr_(call_node);
545 if (auto operation = call_node->op.as<OpNode>()) {
546 name_ += operation->name + "_";
547 } else {
548 name_ += "Call_";
549 }
550 return out;
551 };
552 Expr VisitExpr_(const LetNode* op) override {
553 auto out = ExprMutator::VisitExpr_(op);
554 name_ += "Let_";
555 return out;
556 };
557 Expr VisitExpr_(const IfNode* op) override {
558 auto out = ExprMutator::VisitExpr_(op);
559 name_ += "If_";
560 return out;
561 };
562 Expr VisitExpr_(const TupleGetItemNode* op) override {
563 auto out = ExprMutator::VisitExpr_(op);
564 name_ += "TupleGetItem" + std::to_string(op->index) + "_";
565 return out;
566 };
567 Expr VisitExpr_(const MatchNode* op) override {
568 auto out = ExprMutator::VisitExpr_(op);
569 name_ += "Match_";
570 return out;
571 };
572 std::string name_;
573 const std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual> inputs_;
574};

576/*! \brief Group expressions that match the pattern */
577const std::unordered_map<int, PatternGrouper::Group>& PatternGrouper::GroupMatches(
578 const DFPattern& pattern, const Expr& pre) {
579 groups_.clear();
580 gid_assignments_.clear();
581
582 pattern_ = pattern;
583 pattern_graph_ = CreateIndexedGraph(pattern_);
584 std::unique_ptr<IndexedGraph<Expr>> expr_graph = CreateIndexedGraph(pre);
585 DFPatternMatcher matcher(expr_graph.get());
586 matcher_ = &matcher;
587 this->VisitExprs();
588 return this->groups_;
589}
590
591void PatternGrouper::VisitExprs() {
592 std::unordered_set<Expr, ObjectPtrHash, ObjectPtrEqual> pre_partitioned;
593 for (PostDfsIndex i = matcher_->size(); i != 0; --i) {
594 PostDfsIndex index = i - 1;
595 const auto current = matcher_->index_to_node(index)->ref();
596 if (gid_assignments_.count(current) == 0) { // Don't visit nodes we've already grouped
597 if (auto op = current.as<FunctionNode>()) {
598 if (op->attrs.defined() && op->attrs->dict.count(attr::kPartitionedFromPattern) != 0) {
599 pre_partitioned.insert(current);
600 PostOrderVisit(op->body,
601 [&pre_partitioned](const Expr& expr) { pre_partitioned.insert(expr); });
602 }
603 }
604 if (pre_partitioned.count(current) == 0 && matcher_->Match(pattern_, current)) {
605 CreateGroup(current);
606 }
607 }
608 }
609}

611void PatternGrouper::CreateGroup(const Expr& expr) {
612 VLOG(1) << "Creating group for:" << std::endl << PrettyPrint(expr);
613
614 int var_number = 0;
615
616 auto node_map = matcher_->GetMemo();
617 // Get fuzzy patterns
618 std::unordered_set<Expr, ObjectPtrHash, ObjectPtrEqual> fuzzy_matches;
619 for (PostDfsIndex index = 0; index < pattern_graph_->size(); ++index) {
620 auto node = pattern_graph_->index_to_node(index);
621 // Don't treat fuzzy Dominator patterns input variables for partition
622 if (auto op = node->ref().as<DominatorPatternNode>()) {
623 for (auto fuzzy_op : {op->parent, op->path}) {
624 for (auto match : node_map[fuzzy_op]) {
625 fuzzy_matches.insert(match);
626 }
627 }
628 }
629 // Don't treat Function params or body as input variables for partition
630 if (node->ref().as<FunctionPatternNode>()) {
631 auto matches = node_map[node->ref()];
632 for (auto match : matches) {
633 auto sub_graph = CreateIndexedGraph(match.as<FunctionNode>()->body);
634 for (PostDfsIndex sub_index = 0; sub_index < sub_graph->size(); ++sub_index) {
635 auto sub_node = sub_graph->index_to_node(sub_index);
636 fuzzy_matches.insert(sub_node->ref());
637 }
638 }
639 }
640 }
641
642 // Create input variables
643 Group group;
644 group.root_node = expr;
645 group.matched_nodes = node_map;
646
647 std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual> inputs;
648 Array<Var> params;
649
650 for (PostDfsIndex index = 0; index < pattern_graph_->size(); ++index) {
651 auto node = pattern_graph_->index_to_node(index);
652 auto make_input = [&](const Expr& input) {
653 if (fuzzy_matches.count(input) == 0 && input.as<OpNode>() == nullptr &&
654 input.as<FunctionNode>() == nullptr && !EmbedConst(input, node->ref())) {
655 inputs[input] =
656 Var("FunctionVar_" + std::to_string(graph_number_) + "_" + std::to_string(var_number),
657 NullValue<Type>());
658 group.args.push_back(input);
659 params.push_back(inputs[input]);
660 var_number++;
661 }
662 };
663 auto tuple = node->ref().as<TuplePatternNode>();
664 auto call = node->ref().as<CallPatternNode>();
665 if (tuple && !tuple->fields.defined()) {
666 if (node_map.count(node->ref())) {
667 auto matches = node_map[node->ref()];
668 for (auto match : matches) {
669 for (auto input : match.as<TupleNode>()->fields) {
670 make_input(input);
671 }
672 }
673 }
674 } else if (call && !call->args.defined()) {
675 if (node_map.count(node->ref())) {
676 auto matches = node_map[node->ref()];
677 for (auto match : matches) {
678 for (auto input : match.as<CallNode>()->args) {
679 make_input(input);
680 }
681 }
682 }
683 } else if (node->inputs_.size() == 0) {
684 if (node_map.count(node->ref())) {
685 auto matches = node_map[node->ref()];
686 for (auto match : matches) {
687 make_input(match);
688 }
689 }
690 }
691 }
692
693 graph_number_++;
694
695 // Extract a Function. Used in Partition directly,
696 // used to determine Group overlap in other passes
697 auto extractor = MatchExtractor(inputs);
698 auto body = extractor.Mutate(expr);
699
700 group.function = Function(params, body, NullValue<Type>(), Array<TypeVar>());
701 VLOG(1) << "Candidate extracted function:" << std::endl << PrettyPrint(group.function);
702 group.name = extractor.GetName();
703 // Check to make sure we aren't overlapping with another group or creating an invalid fusion
704 // The MatchExtractor will create a new graph by replacing nodes that match the inputs of the
705 // pattern with the input FunctionVar* Variables. The resulting memoization map will only
706 // contain nodes in the expression that matched the pattern. If a non-input node of the pattern
707 // (i.e., some piece of computation) overlaps with the nodes in a previous group, we'll have a
708 // situation where we try to rewrite the same node twice in the second rewriting or parition
709 // pass. This isn't valid, so we check for it here. We ignore Ops, functions, and constants
710 // because they exist more globally outside of the fusion.
711 // Similiarly, if interior nodes in a group are used outside of the group fusing to a single
712 // output would create an invalid graph tranformation, so we block the creation of such groups.
713 auto memo = extractor.GetMemo();
714 for (auto kv : memo) {
715 VLOG(1) << "matched index " << matcher_->expr_to_node(kv.first)->index_;
716 }
717
718 for (auto kv : memo) {
719 // Check to ensure that this node isn't an input or a global
720 if (inputs.count(kv.first) == 0 && kv.first.as<OpNode>() == nullptr &&
721 kv.first.as<FunctionNode>() == nullptr && kv.first.as<ConstantNode>() == nullptr) {
722 if (gid_assignments_.count(kv.first) != 0) {
723 // check to see if the node is use in other groups
724 // Exit due to overlapping partitions
725 return;
726 } else if (kv.second != body) {
727 // if the node isn't the output of the group
728 auto node = matcher_->expr_to_node(kv.first);
729 for (auto* output : node->outputs_) {
730 if (memo.count(output->ref()) == 0) {
731 // A node inside the matched group contributes an output to nodes outside of the matched
732 // group...
733 auto root = matcher_->expr_to_node(expr);
734 if (!root->Dominates(output)) {
735 // ...and the outside dataflow does not come back to the root of the matched group.
736 // So reject the match since it would create a cycle.
737 VLOG(1) << "Rejecting group since would create a cycle with output " << output->index_
738 << " for root " << root->index_ << " in graph:" << std::endl
739 << matcher_->expr_graph().ToString();
740 return;
741 }
742 // else: We'll allow the output to be included in the matched group.
743 }
744 }
745 }
746 }
747 }
748 // Assign Group Ids
749 group.gid = ++gid_;
750 for (auto kv : extractor.GetMemo()) {
751 gid_assignments_[kv.first] = gid_;
752 }
753
754 // Save Group
755 groups_[group.gid] = std::move(group);
756}

758bool PatternGrouper::EmbedConst(const Expr& expr, const DFPattern pattern) {
759 bool embed = false;
760 if (expr.as<ConstantNode>()) {
761 if (pattern.as<ConstantPatternNode>() != nullptr) {
762 embed = true;
763 } else if (auto expr_pat = pattern.as<ExprPatternNode>()) {
764 if (expr_pat->expr.as<ConstantNode>()) {
765 embed = true;
766 }
767 } else if (auto alt_pat = pattern.as<AltPatternNode>()) {
768 if (matcher_->Match(alt_pat->left, expr)) {
769 embed = EmbedConst(expr, alt_pat->left);
770 } else {
771 embed = EmbedConst(expr, alt_pat->right);
772 }
773 }
774 }
775 return embed;
776}

798Expr PatternRewriter::Rewrite(const Array<DFPatternCallback>& callbacks, const Expr& pre) {
799 VLOG_CONTEXT << "PatternRewriter";
800 VLOG(1) << "rewriting:" << std::endl << PrettyPrint(pre);
801 auto post = pre;
802 auto last = post;
803 // rewrite the graph until it stops changing to make sure all rewrites are complete
804 int count = 0;
805 bool equal = true;
806 static auto* structural_equal = runtime::Registry::Get("node.StructuralEqual");
807 ICHECK(structural_equal) << "node.StructuralEqual is not registered.";
808 do {
809 last = post;
810 for (auto callback : callbacks) {
811 callback_ = callback;
812 if (callback_->require_type) {
813 post = InferTypeWithModule(post, mod_);
814 }
815 auto grouper = PatternGrouper();
816 groups_ = grouper.GroupMatches(callback_->pattern, post);
817 gid_assignments_ = grouper.GetGIDAssignments();
818 memo_.clear();
819 VLOG(1) << "pre rewritten:" << std::endl << PrettyPrint(pre);
820 post = this->VisitExpr(post);
821 VLOG(1) << "post rewritten:" << std::endl << PrettyPrint(post);
822 count++;
823 }
824 equal = (*structural_equal)(last, post, false, true);
825 } while (!equal && count < 100 && !callback_->rewrite_once);
826 if (count >= 100) {
827 LOG(FATAL) << "Observed 100 rewrite passes, possible conflicting passes?";
828 }
829 return post;
830}

832Expr PatternRewriter::DispatchVisitExpr(const Expr& pre) {
833 auto post = MixedModeMutator::DispatchVisitExpr(pre);
834 if (gid_assignments_.count(pre) && pre == groups_[gid_assignments_[pre]].root_node) {
835 // Convert the pre-rewrite node map to a post-rewrite node map
836 auto group = groups_[gid_assignments_[pre]];
837 std::unordered_map<DFPattern, Array<Expr>, ObjectPtrHash, ObjectPtrEqual> node_map;
838 for (auto kv : group.matched_nodes) {
839 Array<Expr> tmp;
840 for (size_t i = 0; i < kv.second.size(); ++i) {
841 tmp.push_back(this->memo_[kv.second[i]]);
842 }
843 node_map.insert({kv.first, tmp});
844 }
845 // run the user callback function
846 return callback_->function(pre, post, Map<DFPattern, Array<Expr>>(node_map));
847 }
848 return post;
849}

857/*!
858 * \brief PatternPartitioner replaces expressions that match a pattern with function call that
859 * perform the same computation but allow for further analysis and lowering.
860 *
861 * The class uses PatternGrouper to support the dominator pattern.
862 */
863class PatternPartitioner : protected MixedModeMutator {
864 public:
865 Expr Partition(const DFPattern& pattern, const Expr& pre, const Map<String, ObjectRef>& attrs,
866 PackedFunc check) {
867 if (pattern.as<FunctionPatternNode>()) {
868 LOG(WARNING) << "Partioning a Function that isn't called doesn't make sense, skipping"
869 << pattern;
870 return pre;
871 }
872 auto grouper = PatternGrouper();
873 groups_ = grouper.GroupMatches(pattern, pre);
874 gid_assignments_ = grouper.GetGIDAssignments();
875 attrs_ = attrs;
876 check_ = check;
877 return this->VisitExpr(pre);
878 }
879
880 protected:
881 Expr RewritePartition(const PatternGrouper::Group& group) {
882 Array<Expr> args;
883 for (size_t i = 0; i < group.args.size(); ++i) {
884 args.push_back(memo_[group.args[i]]);
885 }
886 Function func = WithAttr(group.function, attr::kPartitionedFromPattern, String(group.name));
887 if (!attrs_.empty()) {
888 for (auto kv : attrs_) {
889 func = WithAttr(std::move(func), kv.first, kv.second);
890 }
891 }
892 return Call(func, args);
893 }
894
895 Expr DispatchVisitExpr(const Expr& pre) override {
896 auto post = MixedModeMutator::DispatchVisitExpr(pre);
897 if (gid_assignments_.count(pre) && pre == groups_[gid_assignments_[pre]].root_node &&
898 static_cast<bool>(check_(pre))) {
899 post = RewritePartition(groups_[gid_assignments_[pre]]);
900 }
901 return post;
902 }
903
904 Map<String, ObjectRef> attrs_;
905 std::unordered_map<int, PatternGrouper::Group> groups_;
906 std::unordered_map<Expr, int, ObjectPtrHash, ObjectPtrEqual> gid_assignments_;
907 PackedFunc check_;
908};
909
910Expr PartitionPattern(DFPattern pattern, Expr expr, Map<String, ObjectRef> attrs,
911 PackedFunc check) {
912 return PatternPartitioner().Partition(pattern, expr, attrs, check);
913}
