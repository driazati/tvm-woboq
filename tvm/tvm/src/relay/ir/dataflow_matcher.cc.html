39bool DFPatternMatcher::Match(const DFPattern& pattern, const Expr& expr) {
40 VLOG(1) << "Match " << PrettyPrint(pattern) << " in:" << std::endl << PrettyPrint(expr);
41 memo_.clear();
42 matched_nodes_.clear();
43 return VisitDFPattern(pattern, expr);
44}

46void DFPatternMatcher::ClearMap(size_t watermark) {
47 for (size_t i = watermark; i < matched_nodes_.size(); ++i) {
48 memo_.erase(matched_nodes_[i]);
49 }
50 matched_nodes_.erase(matched_nodes_.begin() + watermark, matched_nodes_.end());
51}

53bool DFPatternMatcher::VisitDFPattern(const DFPattern& pattern, const Expr& expr) {
54 if (memoize_ && memo_.count(pattern)) {
55 ICHECK_EQ(memo_[pattern].size(), 1);
56 return expr.same_as(memo_[pattern][0]);
57 } else {
58 auto watermark = matched_nodes_.size();
59 auto out = DFPatternFunctor::VisitDFPattern(pattern, expr);
60 if (out) {
61 memo_[pattern].push_back(expr);
62 matched_nodes_.push_back(pattern);
63 VLOG(1) << "Matched " << PrettyPrint(pattern) << " at:" << std::endl << PrettyPrint(expr);
64 } else {
65 ClearMap(watermark);
66 }
67 return out;
68 }
69}

75bool MatchRetValue(const ObjectRef& lhs, const TVMRetValue& rhs) {
76 switch (rhs.type_code()) {
77 case kDLInt:
78 if (auto* val = lhs.as<IntImmNode>()) {
79 return val->value == rhs.operator int64_t();
80 }
81 break;
82 case kDLFloat:
83 if (auto* val = lhs.as<FloatImmNode>()) {
84 return val->value == rhs.operator double();
85 }
86 break;
87 case kTVMStr:
88 if (auto* val = lhs.as<tir::StringImmNode>()) {
89 return val->value == rhs.operator std::string();
90 } else if (auto* val = lhs.as<StringObj>()) {
91 return val->data == rhs.operator std::string();
92 }
93 break;
94 case kTVMDataType:
95 if (auto* val = lhs.as<tir::StringImmNode>()) {
96 return rhs.operator std::string() == val->value;
97 } else if (auto* val = lhs.as<StringObj>()) {
98 return rhs.operator std::string() == val->data;
99 } else {
100 ICHECK(false) << "PatternMatcher: Unsupported TVMDataType " << lhs;
101 }
102 break;
103 case kTVMObjectHandle:
104 if (rhs.IsObjectRef<String>()) {
105 if (auto* val = lhs.as<tir::StringImmNode>()) {
106 return rhs.operator String() == val->value;
107 } else if (auto* val = lhs.as<StringObj>()) {
108 return rhs.operator String() == val->data;
109 }
110 } else {
111 // Compare the objects for structural equality
112 static auto* structural_equal = runtime::Registry::Get("node.StructuralEqual");
113 ICHECK(structural_equal) << "node.StructuralEqual is not registered.";
114 if ((*structural_equal)(lhs, GetRef<ObjectRef>(rhs.ptr<Object>()), false, true)) {
115 return true;
116 }
117 }
118 break;
119 default:
120 ICHECK(false) << "Unsupported type code in Pattern Node " << rhs.type_code();
121 }
122 return false;
123}

125bool DFPatternMatcher::VisitDFPattern_(const AttrPatternNode* attr_pattern, const Expr& expr) {
126 bool matches = VisitDFPattern(attr_pattern->pattern, expr);
127 if (!matches) {
128 return matches;
129 }
130 auto attributes = attr_pattern->attrs.as<DictAttrsNode>()->dict;
131 if (const auto* op_node = expr.as<OpNode>()) {
132 Op op = GetRef<Op>(op_node);
133 for (auto kv : attributes) {
134 auto attr_name = kv.first;
135 auto attr_value = kv.second;
136 if (Op::HasAttrMap(attr_name)) {
137 auto op_map = Op::GetAttrMap<TVMRetValue>(attr_name);
138 if (op_map.count(op)) {
139 matches &= MatchRetValue(attr_value, op_map[op]);
140 } else {
141 matches = false;
142 }
143 } else {
144 matches = false;
145 }
146 }
147 } else if (auto* op = expr.as<CallNode>()) {
148 matches = true;
149 // TODO(mbrookhart): When OpNode Attrs move from TVMRetValue to the Object system, remove this
150 // and replace the whole thing with a Visitor-based approach
151 ReflectionVTable* reflection = ReflectionVTable::Global();
152 auto attrs_node = const_cast<BaseAttrsNode*>(op->attrs.get());
153 // attrs may be undefined on non-op calls so we check first
154 std::vector<std::string> attr_names;
155 if (attrs_node) {
156 attr_names = reflection->ListAttrNames(attrs_node);
157 }
158 for (auto kv : attributes) {
159 std::string attr = kv.first;
160 if (matches && std::find(attr_names.begin(), attr_names.end(), attr) != attr_names.end()) {
161 matches &= MatchRetValue(kv.second, reflection->GetAttr(attrs_node, attr));
162 } else {
163 matches = false;
164 break;
165 }
166 }
167 } else if (auto* op = expr.as<FunctionNode>()) {
168 matches = true;
169 for (auto kv : attributes) {
170 if (matches && op->attrs.defined() && op->attrs->dict.count(kv.first)) {
171 matches &= StructuralEqual()(kv.second, op->attrs->dict[kv.first]);
172 } else {
173 matches = false;
174 break;
175 }
176 }
177 } else {
178 matches = false;
179 }
180 return matches;
181}

183Array<DFPattern> reverse(const Array<DFPattern>& args) {
184 Array<DFPattern> new_args;
185 for (auto it = args.rbegin(); it != args.rend(); ++it) {
186 new_args.push_back(*it);
187 }
188 return new_args;
189}

191bool DFPatternMatcher::VisitDFPattern_(const CallPatternNode* op, const Expr& expr) {
192 // utilities
193 auto get_op_node = [](const CallPatternNode* op) -> const tvm::OpNode* {
194 if (op) {
195 if (auto* expr_pattern = op->op.as<ExprPatternNode>()) {
196 return expr_pattern->expr.as<OpNode>();
197 }
198 }
199 return nullptr;
200 };
201 auto is_pattern_op = [&get_op_node](const CallPatternNode* op, std::string op_type) {
202 if (const auto* op_node = get_op_node(op)) {
203 if (op_node->name == op_type) {
204 return true;
205 }
206 }
207 return false;
208 };
209 auto is_expr_op = [](const Expr& expr, std::string op_type) {
210 if (const auto* call_node = expr.as<CallNode>()) {
211 if (const auto* op_node = call_node->op.as<OpNode>()) {
212 if (op_node->name == op_type) {
213 return true;
214 }
215 }
216 }
217 return false;
218 };
219
220 // logic
221 auto watermark = matched_nodes_.size();
222 if (const auto* call_node = expr.as<CallNode>()) {
223 auto matches_op = VisitDFPattern(op->op, call_node->op);
224 if (matches_op) {
225 auto watermark2 = matched_nodes_.size();
226
227 auto match_args = [this, &watermark2](const Array<DFPattern> pattern_args,
228 const Array<Expr> expr_args) {
229 bool matches = true;
230 size_t i = 0;
231 if (pattern_args.defined()) {
232 if (pattern_args.size() == expr_args.size()) {
233 while (matches && i < pattern_args.size()) {
234 matches &= VisitDFPattern(pattern_args[i], expr_args[i]);
235 ++i;
236 }
237 } else {
238 matches = false;
239 }
240 }
241 if (!matches) {
242 ClearMap(watermark2);
243 }
244 return matches;
245 };
246
247 // Standard case
248 if (match_args(op->args, call_node->args)) {
249 return true;
250 }
251 // Commutative Matching
252 if (const OpNode* op_node = get_op_node(op)) {
253 if ((op_node->name == "add") || (op_node->name == "multiply")) {
254 if (match_args(reverse(op->args), call_node->args)) {
255 return true;
256 }
257 }
258 }
259 } else {
260 ClearMap(watermark);
261 // associate divide/multiply
262 if (is_pattern_op(op, "divide")) {
263 if (const auto* arg_node = op->args[0].as<CallPatternNode>()) {
264 if (is_pattern_op(arg_node, "multiply") && is_expr_op(expr, "multiply") &&
265 (is_expr_op(call_node->args[0], "divide") ||
266 is_expr_op(call_node->args[1], "divide"))) {
267 bool out = false;
268 for (size_t arg_id = 0; arg_id < 2; ++arg_id) {
269 auto div = CallPattern(op->op, {arg_node->args[arg_id], op->args[1]});
270 auto mul = CallPattern(arg_node->op, {arg_node->args[(arg_id + 1) % 2], div});
271 out = VisitDFPattern(mul, expr);
272 if (out) {
273 return true;
274 } else {
275 ClearMap(watermark);
276 }
277 }
278 return out;
279 }
280 }
281 }
282 if (is_pattern_op(op, "multiply")) {
283 // associate multiply/divide
284 for (size_t arg_id = 0; arg_id < 2; ++arg_id) {
285 if (auto* arg_node = op->args[arg_id].as<CallPatternNode>()) {
286 if (is_pattern_op(arg_node, "divide") && is_expr_op(expr, "divide") &&
287 (is_expr_op(call_node->args[0], "multiply") ||
288 is_expr_op(call_node->args[1], "multiply"))) {
289 auto mul = CallPattern(op->op, {arg_node->args[0], op->args[(arg_id + 1) % 2]});
290 auto div = CallPattern(arg_node->op, {mul, arg_node->args[1]});
291 return VisitDFPattern(div, expr);
292 }
293 }
294 }
295 }
296 }
297 }
298 return false;
299}

301// Recursively find the Dominator parent along all inputs paths.
302bool DFPatternMatcher::MatchesPath(const DominatorPatternNode* op, const Expr& expr) {
303 auto call_node = expr.as<CallNode>();
304 auto index_node = expr_to_node(expr);
305 for (auto node : index_node->inputs_) {
306 if (!(call_node && node->ref() == call_node->op)) {
307 memoize_ = true;
308 if (VisitDFPattern(op->parent, node->ref())) {
309 return true;
310 } else {
311 memoize_ = false;
312 if (!VisitDFPattern(op->path, node->ref())) {
313 return false;
314 }
315 if (!MatchesPath(op, node->ref())) {
316 return false;
317 }
318 }
319 }
320 }
321 return true;
322}

324// Iteratively ensure that the parent is dominated somewhere by the child or the path
325bool DFPatternMatcher::DominatesParent(const DominatorPatternNode* op, const Expr& expr) {
326 std::stack<Expr> stack;
327 std::unordered_set<const ExprNode*> visited;
328 stack.push(expr);
329 while (!stack.empty()) {
330 Expr current = stack.top();
331 stack.pop();
332 for (auto node : expr_to_node(current)->dominator_children_) {
333 if (visited.count(node->node_ref_) == 0) {
334 if (VisitDFPattern(op->parent, node->ref())) {
335 return true;
336 } else {
337 stack.push(node->ref());
338 }
339 visited.insert(node->node_ref_);
340 }
341 }
342 }
343 return false;
344}

346bool DFPatternMatcher::VisitDFPattern_(const DominatorPatternNode* op, const Expr& expr) {
347 if (VisitDFPattern(op->child, expr)) {
348 bool matches_path = MatchesPath(op, expr);
349 memoize_ = true;
350 if (matches_path) {
351 return DominatesParent(op, expr);
352 }
353 }
354 return false;
355}

361bool DFPatternMatcher::VisitDFPattern_(const FunctionPatternNode* op, const Expr& expr) {
362 bool matches = false;
363 if (const auto* func = expr.as<FunctionNode>()) {
364 matches = true;
365 if (op->params.defined()) {
366 size_t i = 0;
367 if (op->params.size() == func->params.size()) {
368 while (matches && i < op->params.size()) {
369 matches &= VisitDFPattern(op->params[i], func->params[i]);
370 ++i;
371 }
372 } else {
373 matches = false;
374 }
375 }
376 if (matches) {
377 matches &= VisitDFPattern(op->body, func->body);
378 }
379 }
380 return matches;
381}

383bool DFPatternMatcher::VisitDFPattern_(const TupleGetItemPatternNode* op, const Expr& expr) {
384 bool matches = false;
385 if (const auto* tuple_get_item_node = expr.as<TupleGetItemNode>()) {
386 matches = (op->index == -1 || op->index == tuple_get_item_node->index) &&
387 VisitDFPattern(op->tuple, tuple_get_item_node->tuple);
388 }
389 return matches;
390}

392bool DFPatternMatcher::VisitDFPattern_(const TuplePatternNode* op, const Expr& expr) {
393 bool matches = false;
394 if (const auto* tuple_node = expr.as<TupleNode>()) {
395 matches = true;
396 if (op->fields.defined()) {
397 if (op->fields.size() == tuple_node->fields.size()) {
398 size_t i = 0;
399 while (matches && i < op->fields.size()) {
400 matches &= VisitDFPattern(op->fields[i], tuple_node->fields[i]);
401 ++i;
402 }
403 } else {
404 matches = false;
405 }
406 }
407 }
408 return matches;

411bool DFPatternMatcher::VisitDFPattern_(const IfPatternNode* op, const Expr& expr) {
412 if (const auto* if_node = expr.as<IfNode>()) {
413 auto cond = if_node->cond;
414 auto true_branch = if_node->true_branch;
415 auto false_branch = if_node->false_branch;
416 return VisitDFPattern(op->cond, cond) && VisitDFPattern(op->true_branch, true_branch) &&
417 VisitDFPattern(op->false_branch, false_branch);
418 }
419 return false;
420}

422bool DFPatternMatcher::VisitDFPattern_(const LetPatternNode* op, const Expr& expr) {
423 if (const auto* let_node = expr.as<LetNode>()) {
424 return VisitDFPattern(op->var, let_node->var) && VisitDFPattern(op->value, let_node->value) &&
425 VisitDFPattern(op->body, let_node->body);
426 }
427 return false;
428}

430Expr InferType(const Expr& expr) {
431 auto mod = IRModule::FromExpr(expr);
432 mod = transform::InferType()(mod);
433 if (expr.as<FunctionNode>()) {
434 return mod->Lookup("main");
435 } else {
436 return mod->Lookup("main").as<FunctionNode>()->body;
437 }
438}

440Expr InferTypeWithModule(const Expr& expr, const IRModule& m) {
441 IRModule mod(m->functions, m->type_definitions, m->Imports());
442 GlobalVarSupply global_var_supply = GlobalVarSupply(mod);
443 GlobalVar gvar = global_var_supply->FreshGlobal("_tmp", false);
444 BaseFunc func;
445 if (expr.as<FunctionNode>()) {
446 func = Downcast<Function>(expr);
447 } else {
448 func = relay::Function(relay::FreeVars(expr), expr, Type(), relay::FreeTypeVars(expr, mod), {});
449 }
450 mod->Add(gvar, func);
451 mod = transform::InferType()(mod);
452 Expr ret;
453 if (expr.as<FunctionNode>()) {
454 ret = mod->Lookup(gvar);
455 } else {
456 ret = mod->Lookup(gvar).as<FunctionNode>()->body;
457 }
458 return ret;
459}

508/*! \brief Creates a new set of nodes based on Group inputs, used to create functions and perform
509 * group overlap analysis */
510class MatchExtractor : public ExprMutator {
511 public:
512 explicit MatchExtractor(
513 const std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual>& inputs)
514 : inputs_(inputs) {}
515 const std::unordered_map<Expr, Expr, ObjectPtrHash, ObjectPtrEqual>& GetMemo() {
516 return this->memo_;
517 }
518 const std::string& GetName() { return name_; }
519
520 protected:
521 Expr VisitExpr(const Expr& pre) override {
522 if (inputs_.count(pre)) {
523 return inputs_.at(pre);
524 }
525 return ExprMutator::VisitExpr(pre);
526 }
527 Expr VisitExpr_(const TupleNode* op) override {
528 auto out = ExprMutator::VisitExpr_(op);
529 name_ += "Tuple_";
530 return out;
531 };
532 Expr VisitExpr_(const FunctionNode* op) override {
533 auto out = ExprMutator::VisitExpr_(op);
534 name_ += "Function";
535 return out;
536 };
537 Expr VisitExpr_(const CallNode* call_node) override {
538 auto out = ExprMutator::VisitExpr_(call_node);
539 if (auto operation = call_node->op.as<OpNode>()) {
540 name_ += operation->name + "_";
541 } else {
542 name_ += "Call_";
543 }
544 return out;
545 };
546 Expr VisitExpr_(const LetNode* op) override {
547 auto out = ExprMutator::VisitExpr_(op);
548 name_ += "Let_";
549 return out;
550 };
551 Expr VisitExpr_(const IfNode* op) override {
552 auto out = ExprMutator::VisitExpr_(op);
553 name_ += "If_";
554 return out;
555 };
556 Expr VisitExpr_(const TupleGetItemNode* op) override {
557 auto out = ExprMutator::VisitExpr_(op);
558 name_ += "TupleGetItem" + std::to_string(op->index) + "_";
559 return out;
560 };
561 Expr VisitExpr_(const MatchNode* op) override {
562 auto out = ExprMutator::VisitExpr_(op);
563 name_ += "Match_";
564 return out;
565 };
566 std::string name_;
567 const std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual> inputs_;
568};

570/*! \brief Group expressions that match the pattern */
571const std::unordered_map<int, PatternGrouper::Group>& PatternGrouper::GroupMatches(
572 const DFPattern& pattern, const Expr& pre) {
573 groups_.clear();
574 gid_assignments_.clear();
575
576 pattern_ = pattern;
577 pattern_graph_ = CreateIndexedGraph(pattern_);
578 std::unique_ptr<IndexedGraph<Expr>> expr_graph = CreateIndexedGraph(pre);
579 DFPatternMatcher matcher(expr_graph.get());
580 matcher_ = &matcher;
581 this->VisitExprs();
582 return this->groups_;
583}
584
585void PatternGrouper::VisitExprs() {
586 std::unordered_set<Expr, ObjectPtrHash, ObjectPtrEqual> pre_partitioned;
587 for (PostDfsIndex i = matcher_->size(); i != 0; --i) {
588 PostDfsIndex index = i - 1;
589 const auto current = matcher_->index_to_node(index)->ref();
590 if (gid_assignments_.count(current) == 0) { // Don't visit nodes we've already grouped
591 if (auto op = current.as<FunctionNode>()) {
592 if (op->attrs.defined() && op->attrs->dict.count(attr::kPartitionedFromPattern) != 0) {
593 pre_partitioned.insert(current);
594 PostOrderVisit(op->body,
595 [&pre_partitioned](const Expr& expr) { pre_partitioned.insert(expr); });
596 }
597 }
598 if (pre_partitioned.count(current) == 0 && matcher_->Match(pattern_, current)) {
599 CreateGroup(current);
600 }
601 }
602 }
603}

605void PatternGrouper::CreateGroup(const Expr& expr) {
606 VLOG(1) << "Creating group for:" << std::endl << PrettyPrint(expr);
607
608 int var_number = 0;
609
610 auto node_map = matcher_->GetMemo();
611 // Get fuzzy patterns
612 std::unordered_set<Expr, ObjectPtrHash, ObjectPtrEqual> fuzzy_matches;
613 for (PostDfsIndex index = 0; index < pattern_graph_->size(); ++index) {
614 auto node = pattern_graph_->index_to_node(index);
615 // Don't treat fuzzy Dominator patterns input variables for partition
616 if (auto op = node->ref().as<DominatorPatternNode>()) {
617 for (auto fuzzy_op : {op->parent, op->path}) {
618 for (auto match : node_map[fuzzy_op]) {
619 fuzzy_matches.insert(match);
620 }
621 }
622 }
623 // Don't treat Function params or body as input variables for partition
624 if (node->ref().as<FunctionPatternNode>()) {
625 auto matches = node_map[node->ref()];
626 for (auto match : matches) {
627 auto sub_graph = CreateIndexedGraph(match.as<FunctionNode>()->body);
628 for (PostDfsIndex sub_index = 0; sub_index < sub_graph->size(); ++sub_index) {
629 auto sub_node = sub_graph->index_to_node(sub_index);
630 fuzzy_matches.insert(sub_node->ref());
631 }
632 }
633 }
634 }
635
636 // Create input variables
637 Group group;
638 group.root_node = expr;
639 group.matched_nodes = node_map;
640
641 std::unordered_map<Expr, Var, ObjectPtrHash, ObjectPtrEqual> inputs;
642 Array<Var> params;
643
644 for (PostDfsIndex index = 0; index < pattern_graph_->size(); ++index) {
645 auto node = pattern_graph_->index_to_node(index);
646 auto make_input = [&](const Expr& input) {
647 if (fuzzy_matches.count(input) == 0 && input.as<OpNode>() == nullptr &&
648 input.as<FunctionNode>() == nullptr && !EmbedConst(input, node->ref())) {
649 inputs[input] =
650 Var("FunctionVar_" + std::to_string(graph_number_) + "_" + std::to_string(var_number),
651 NullValue<Type>());
652 group.args.push_back(input);
653 params.push_back(inputs[input]);
654 var_number++;
655 }
656 };
657 auto tuple = node->ref().as<TuplePatternNode>();
658 auto call = node->ref().as<CallPatternNode>();
659 if (tuple && !tuple->fields.defined()) {
660 if (node_map.count(node->ref())) {
661 auto matches = node_map[node->ref()];
662 for (auto match : matches) {
663 for (auto input : match.as<TupleNode>()->fields) {
664 make_input(input);
665 }
666 }
667 }
668 } else if (call && !call->args.defined()) {
669 if (node_map.count(node->ref())) {
670 auto matches = node_map[node->ref()];
671 for (auto match : matches) {
672 for (auto input : match.as<CallNode>()->args) {
673 make_input(input);
674 }
675 }
676 }
677 } else if (node->inputs_.size() == 0) {
678 if (node_map.count(node->ref())) {
679 auto matches = node_map[node->ref()];
680 for (auto match : matches) {
681 make_input(match);
682 }
683 }
684 }
685 }
686
687 graph_number_++;
688
689 // Extract a Function. Used in Partition directly,
690 // used to determine Group overlap in other passes
691 auto extractor = MatchExtractor(inputs);
692 auto body = extractor.Mutate(expr);
693
694 group.function = Function(params, body, NullValue<Type>(), Array<TypeVar>());
695 VLOG(1) << "Candidate extracted function:" << std::endl << PrettyPrint(group.function);
696 group.name = extractor.GetName();
697 // Check to make sure we aren't overlapping with another group or creating an invalid fusion
698 // The MatchExtractor will create a new graph by replacing nodes that match the inputs of the
699 // pattern with the input FunctionVar* Variables. The resulting memoization map will only
700 // contain nodes in the expression that matched the pattern. If a non-input node of the pattern
701 // (i.e., some piece of computation) overlaps with the nodes in a previous group, we'll have a
702 // situation where we try to rewrite the same node twice in the second rewriting or parition
703 // pass. This isn't valid, so we check for it here. We ignore Ops, functions, and constants
704 // because they exist more globally outside of the fusion.
705 // Similiarly, if interior nodes in a group are used outside of the group fusing to a single
706 // output would create an invalid graph tranformation, so we block the creation of such groups.
707 auto memo = extractor.GetMemo();
708 for (auto kv : memo) {
709 VLOG(1) << "matched index " << matcher_->expr_to_node(kv.first)->index_;
710 }
711
712 for (auto kv : memo) {
713 // Check to ensure that this node isn't an input or a global
714 if (inputs.count(kv.first) == 0 && kv.first.as<OpNode>() == nullptr &&
715 kv.first.as<FunctionNode>() == nullptr && kv.first.as<ConstantNode>() == nullptr) {
716 if (gid_assignments_.count(kv.first) != 0) {
717 // check to see if the node is use in other groups
718 // Exit due to overlapping partitions
719 return;
720 } else if (kv.second != body) {
721 // if the node isn't the output of the group
722 auto node = matcher_->expr_to_node(kv.first);
723 for (auto* output : node->outputs_) {
724 if (memo.count(output->ref()) == 0) {
725 // A node inside the matched group contributes an output to nodes outside of the matched
726 // group...
727 auto root = matcher_->expr_to_node(expr);
728 if (!root->Dominates(output)) {
729 // ...and the outside dataflow does not come back to the root of the matched group.
730 // So reject the match since it would create a cycle.
731 VLOG(1) << "Rejecting group since would create a cycle with output " << output->index_
732 << " for root " << root->index_ << " in graph:" << std::endl
733 << matcher_->expr_graph().ToString();
734 return;
735 }
736 // else: We'll allow the output to be included in the matched group.
737 }
738 }
739 }
740 }
741 }
742 // Assign Group Ids
743 group.gid = ++gid_;
744 for (auto kv : extractor.GetMemo()) {
745 gid_assignments_[kv.first] = gid_;
746 }
747
748 // Save Group
749 groups_[group.gid] = std::move(group);
750}

752bool PatternGrouper::EmbedConst(const Expr& expr, const DFPattern pattern) {
753 bool embed = false;
754 if (expr.as<ConstantNode>()) {
755 if (pattern.as<ConstantPatternNode>() != nullptr) {
756 embed = true;
757 } else if (auto expr_pat = pattern.as<ExprPatternNode>()) {
758 if (expr_pat->expr.as<ConstantNode>()) {
759 embed = true;
760 }
761 } else if (auto alt_pat = pattern.as<AltPatternNode>()) {
762 if (matcher_->Match(alt_pat->left, expr)) {
763 embed = EmbedConst(expr, alt_pat->left);
764 } else {
765 embed = EmbedConst(expr, alt_pat->right);
766 }
767 }
768 }
769 return embed;
770}

792Expr PatternRewriter::Rewrite(const Array<DFPatternCallback>& callbacks, const Expr& pre) {
793 VLOG_CONTEXT << "PatternRewriter";
794 VLOG(1) << "rewriting:" << std::endl << PrettyPrint(pre);
795 auto post = pre;
796 auto last = post;
797 // rewrite the graph until it stops changing to make sure all rewrites are complete
798 int count = 0;
799 bool equal = true;
800 static auto* structural_equal = runtime::Registry::Get("node.StructuralEqual");
801 ICHECK(structural_equal) << "node.StructuralEqual is not registered.";
802 do {
803 last = post;
804 for (auto callback : callbacks) {
805 callback_ = callback;
806 if (callback_->require_type) {
807 post = InferTypeWithModule(post, mod_);
808 }
809 auto grouper = PatternGrouper();
810 groups_ = grouper.GroupMatches(callback_->pattern, post);
811 gid_assignments_ = grouper.GetGIDAssignments();
812 memo_.clear();
813 VLOG(1) << "pre rewritten:" << std::endl << PrettyPrint(pre);
814 post = this->VisitExpr(post);
815 VLOG(1) << "post rewritten:" << std::endl << PrettyPrint(post);
816 count++;
817 }
818 equal = (*structural_equal)(last, post, false, true);
819 } while (!equal && count < 100 && !callback_->rewrite_once);
820 if (count >= 100) {
821 LOG(FATAL) << "Observed 100 rewrite passes, possible conflicting passes?";
822 }
823 return post;
824}

826Expr PatternRewriter::DispatchVisitExpr(const Expr& pre) {
827 auto post = MixedModeMutator::DispatchVisitExpr(pre);
828 if (gid_assignments_.count(pre) && pre == groups_[gid_assignments_[pre]].root_node) {
829 // Convert the pre-rewrite node map to a post-rewrite node map
830 auto group = groups_[gid_assignments_[pre]];
831 std::unordered_map<DFPattern, Array<Expr>, ObjectPtrHash, ObjectPtrEqual> node_map;
832 for (auto kv : group.matched_nodes) {
833 Array<Expr> tmp;
834 for (size_t i = 0; i < kv.second.size(); ++i) {
835 tmp.push_back(this->memo_[kv.second[i]]);
836 }
837 node_map.insert({kv.first, tmp});
838 }
839 // run the user callback function
840 return callback_->function(pre, post, Map<DFPattern, Array<Expr>>(node_map));
841 }
842 return post;
843}

851/*!
852 * \brief PatternPartitioner replaces expressions that match a pattern with function call that
853 * perform the same computation but allow for further analysis and lowering.
854 *
855 * The class uses PatternGrouper to support the dominator pattern.
856 */
857class PatternPartitioner : protected MixedModeMutator {
858 public:
859 Expr Partition(const DFPattern& pattern, const Expr& pre, const Map<String, ObjectRef>& attrs,
860 PackedFunc check) {
861 if (pattern.as<FunctionPatternNode>()) {
862 LOG(WARNING) << "Partioning a Function that isn't called doesn't make sense, skipping"
863 << pattern;
864 return pre;
865 }
866 auto grouper = PatternGrouper();
867 groups_ = grouper.GroupMatches(pattern, pre);
868 gid_assignments_ = grouper.GetGIDAssignments();
869 attrs_ = attrs;
870 check_ = check;
871 return this->VisitExpr(pre);
872 }
873
874 protected:
875 Expr RewritePartition(const PatternGrouper::Group& group) {
876 Array<Expr> args;
877 for (size_t i = 0; i < group.args.size(); ++i) {
878 args.push_back(memo_[group.args[i]]);
879 }
880 Function func = WithAttr(group.function, attr::kPartitionedFromPattern, String(group.name));
881 if (!attrs_.empty()) {
882 for (auto kv : attrs_) {
883 func = WithAttr(std::move(func), kv.first, kv.second);
884 }
885 }
886 return Call(func, args);
887 }
888
889 Expr DispatchVisitExpr(const Expr& pre) override {
890 auto post = MixedModeMutator::DispatchVisitExpr(pre);
891 if (gid_assignments_.count(pre) && pre == groups_[gid_assignments_[pre]].root_node &&
892 static_cast<bool>(check_(pre))) {
893 post = RewritePartition(groups_[gid_assignments_[pre]]);
894 }
895 return post;
896 }
897
898 Map<String, ObjectRef> attrs_;
899 std::unordered_map<int, PatternGrouper::Group> groups_;
900 std::unordered_map<Expr, int, ObjectPtrHash, ObjectPtrEqual> gid_assignments_;
901 PackedFunc check_;
902};
903
904Expr PartitionPattern(DFPattern pattern, Expr expr, Map<String, ObjectRef> attrs,
905 PackedFunc check) {
906 return PatternPartitioner().Partition(pattern, expr, attrs, check);
907}
