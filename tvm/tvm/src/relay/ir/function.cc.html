29Function::Function(tvm::Array<Var> params, Expr body, Type ret_type,
30 tvm::Array<TypeVar> type_params, DictAttrs attrs, Span span) {
31 ObjectPtr<FunctionNode> n = make_object<FunctionNode>();
32 ICHECK(params.defined());
33 ICHECK(type_params.defined());
34 n->params = std::move(params);
35 n->body = std::move(body);
36 n->ret_type = std::move(ret_type);
37 n->type_params = std::move(type_params);
38 n->attrs = std::move(attrs);
39 n->virtual_device_ = VirtualDevice::FullyUnconstrained();
40 n->span = std::move(span);
41 data_ = std::move(n);
42}

44Function WithFields(Function function, Optional<Array<Var>> opt_params, Optional<Expr> opt_body,
45 Optional<Type> opt_ret_type, Optional<Array<TypeVar>> opt_ty_params,
46 Optional<DictAttrs> opt_attrs, Optional<VirtualDevice> opt_virtual_device,
47 Optional<Span> opt_span) {
48 Array<Var> params = opt_params.value_or(function->params);
49 Expr body = opt_body.value_or(function->body);
50 Type ret_type = opt_ret_type.value_or(function->ret_type);
51 Array<TypeVar> ty_params = opt_ty_params.value_or(function->type_params);
52 DictAttrs attrs = opt_attrs.value_or(function->attrs);
53 VirtualDevice virtual_device = opt_virtual_device.value_or(function->virtual_device());
54 Span span = opt_span.value_or(function->span);
55
56 bool unchanged = body.same_as(function->body) && ret_type.same_as(function->ret_type) &&
57 attrs.same_as(function->attrs) &&
58 virtual_device.same_as(function->virtual_device()) &&
59 span.same_as(function->span);
60
61 // Check that all the type params are unchanged
62 if (unchanged) {
63 bool all_ty_params_unchanged = true;
64 if (ty_params.size() == function->type_params.size()) {
65 for (size_t i = 0; i < ty_params.size(); i++) {
66 all_ty_params_unchanged &= ty_params[i].same_as(function->type_params[i]);
67 }
68 } else {
69 all_ty_params_unchanged = false;
70 }
71 unchanged &= all_ty_params_unchanged;
72 }
73
74 // Check that all the params are unchanged
75 if (unchanged) {
76 bool all_params_unchanged = true;
77 if (params.size() == function->params.size()) {
78 for (size_t i = 0; i < params.size(); i++) {
79 all_params_unchanged &= params[i].same_as(function->params[i]);
80 }
81 } else {
82 all_params_unchanged = false;
83 }
84 unchanged &= all_params_unchanged;
85 }
86
87 if (!unchanged) {
88 FunctionNode* cow_function_node = function.CopyOnWrite();
89 cow_function_node->params = params;
90 cow_function_node->body = body;
91 cow_function_node->ret_type = ret_type;
92 cow_function_node->type_params = ty_params;
93 cow_function_node->attrs = attrs;
94 cow_function_node->virtual_device_ = virtual_device;
95 cow_function_node->span = span;
96 }
97 return function;
98}

100FuncType FunctionNode::func_type_annotation() const {
101 Array<Type> param_types;
102 for (auto param : this->params) {
103 Type param_type =
104 (param->type_annotation.defined()) ? param->type_annotation : IncompleteType(Kind::kType);
105 param_types.push_back(param_type);
106 }
107
108 Type ret_type = (this->ret_type.defined()) ? this->ret_type : IncompleteType(Kind::kType);
109 return FuncType(param_types, ret_type, this->type_params, {});
110}

112const FunctionNode* AsOptimizableFunctionNode(const BaseFunc& base_func) {
113 if (const auto* function_node = base_func.as<FunctionNode>()) {
114 if (!function_node->GetAttr<String>(attr::kCompiler).defined() &&
115 !function_node->HasNonzeroAttr(attr::kExtern) &&
116 !function_node->HasNonzeroAttr(attr::kSkipOptimization)) {
117 return function_node;
118 }
119 }
120 return nullptr;
121}

125TVM_REGISTER_GLOBAL("relay.ir.Function")
126 .set_body_typed([](tvm::Array<Var> params, Expr body, Type ret_type,
127 tvm::Array<TypeVar> ty_params, tvm::DictAttrs attrs) {
128 return Function(params, body, ret_type, ty_params, attrs);
129 });
130TVM_REGISTER_GLOBAL("relay.ir.FunctionWithFields")
131 .set_body_typed([](Function function, Optional<Array<Var>> opt_params, Optional<Expr> opt_body,
132 Optional<Type> opt_ret_type, Optional<Array<TypeVar>> opt_ty_params,
133 Optional<DictAttrs> opt_attrs, Optional<VirtualDevice> opt_virtual_device,
134 Optional<Span> opt_span) {
135 return WithFields(function, opt_params, opt_body, opt_ret_type, opt_ty_params, opt_attrs,
136 opt_virtual_device, opt_span);
137 });
