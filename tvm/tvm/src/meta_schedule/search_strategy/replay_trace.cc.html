24/*! \brief A search strategy that generates measure candidates using trace and random decisions. */
25class ReplayTraceNode : public SearchStrategyNode {
26 public:
27 /*! \brief The state of the search strategy. */
28 struct State {
29 /*! \brief The search strategy itself */
30 ReplayTraceNode* self;
31 /*! \brief The design spaces. */
32 Array<tir::Trace> design_spaces;
33 /*! \brief `[st, ed)` are the indices of the next batch of candidates. */
34 int st;
35 /*! \brief `[st, ed)` are the indices of the next batch of candidates. */
36 int ed;
37
38 /*! \brief The module to be tuned. */
39 Array<IRModule> per_thread_mod_{nullptr};
40
41 explicit State(ReplayTraceNode* self, Array<tir::Trace> design_spaces)
42 : self(self), design_spaces(design_spaces), st(0), ed(self->num_trials_per_iter) {
43 const TuneContextNode* ctx = self->context_;
44 ICHECK(ctx);
45 IRModule mod = ctx->mod.value();
46 this->per_thread_mod_.reserve(ctx->num_threads);
47 for (int i = 0; i < ctx->num_threads; i++) {
48 this->per_thread_mod_.push_back(DeepCopyIRModule(mod));
49 }
50 }
51
52 inline Optional<Array<MeasureCandidate>> GenerateMeasureCandidates();
53 inline void NotifyRunnerResults(const Array<RunnerResult>& results);
54 };
55
56 /*! \brief The number of trials per iteration. */
57 int num_trials_per_iter;
58 /*! \brief The number of total trials. */
59 int max_trials_per_task;
60 /*! \brief The max number of failures during trace replaying. */
61 int max_fail_count;
62
63 /*! \brief The tuning context of the search strategy. */
64 const TuneContextNode* context_{nullptr};
65 /*! \brief The random state. -1 means using random number. */
66 TRandState rand_state_ = -1;
67 /*! \brief The state of the search strategy. */
68 std::unique_ptr<State> state_ = nullptr;
69
70 void VisitAttrs(tvm::AttrVisitor* v) {
71 v->Visit("num_trials_per_iter", &num_trials_per_iter);
72 v->Visit("max_trials_per_task", &max_trials_per_task);
73 v->Visit("max_fail_count", &max_fail_count);
74 // `context_` is not visited.
75 // `rand_state_` is not visited
76 // `state_` is not visited
77 }
78
79 static constexpr const char* _type_key = "meta_schedule.ReplayTrace";
80 TVM_DECLARE_FINAL_OBJECT_INFO(ReplayTraceNode, SearchStrategyNode);
81
82 void InitializeWithTuneContext(const TuneContext& context) final {
83 CHECK(context->mod.defined()) << "ValueError: TuneContext.mod is not defined";
84 this->context_ = context.get();
85 this->rand_state_ = ForkSeed(&context->rand_state);
86 this->state_.reset();
87 }
88
89 void PreTuning(const Array<tir::Schedule>& design_spaces, const Optional<Database>& database,
90 const Optional<CostModel>& cost_model) final {
91 ICHECK(!design_spaces.empty());
92 CHECK(this->context_ != nullptr) << "ValueError: Did you forget to initialize the TuneContext?";
93 if (this->state_ != nullptr) {
94 TVM_PY_LOG(WARNING, this->context_->logging_func) << "RelayTrace is already initialized.";
95 this->state_.reset();
96 }
97 ICHECK(this->state_ == nullptr);
98 Array<tir::Trace> design_space_traces;
99 design_space_traces.reserve(design_spaces.size());
100 for (const tir::Schedule& space : design_spaces) {
101 design_space_traces.push_back(space->trace().value()->Simplified(true));
102 }
103 this->state_ = std::make_unique<State>(this, design_space_traces);
104 }
105
106 void PostTuning() final {
107 ICHECK(this->state_ != nullptr);
108 this->state_.reset();
109 }
110
111 Optional<Array<MeasureCandidate>> GenerateMeasureCandidates() final {
112 ICHECK(this->state_ != nullptr);
113 return this->state_->GenerateMeasureCandidates();
114 }
115
116 void NotifyRunnerResults(const Array<MeasureCandidate>& measure_candidates,
117 const Array<RunnerResult>& results) final {
118 ICHECK(this->state_ != nullptr);
119 this->state_->NotifyRunnerResults(results);
120 }
121};

123inline Optional<Array<MeasureCandidate>> ReplayTraceNode::State::GenerateMeasureCandidates() {
124 if (st >= self->max_trials_per_task) {
125 return NullOpt;
126 }
127 ed = std::min(ed, self->max_trials_per_task);
128 ICHECK_LT(st, ed);
129 const TuneContextNode* ctx = self->context_;
130 ICHECK(ctx);
131 std::vector<TRandState> per_thread_rand_state = ForkSeed(&self->rand_state_, ctx->num_threads);
132 Array<MeasureCandidate> per_task_result(ed - st, MeasureCandidate{nullptr});
133 ThreadedTraceApply pp(ctx->postprocs);
134 auto f_worker = [this, &per_thread_rand_state, &per_task_result, &pp](int thread_id,
135 int task_id) -> void {
136 TRandState& rand_state = per_thread_rand_state[thread_id];
137 IRModule mod = this->per_thread_mod_[thread_id];
138
139 for (int fail_count = 0; fail_count < self->max_fail_count; fail_count++) {
140 int design_space_index = tir::SampleInt(&rand_state, 0, design_spaces.size());
141 tir::Trace trace = design_spaces[design_space_index];
142 tir::Trace new_trace = tir::Trace(trace->insts, {});
143 if (Optional<tir::Schedule> opt_sch = pp.Apply(mod, new_trace, &rand_state)) {
144 tir::Schedule sch = opt_sch.value();
145 Array<ArgInfo> args_info = ArgInfo::FromEntryFunc(sch->mod(), /*remove_preproc=*/true);
146 per_task_result.Set(task_id, MeasureCandidate(sch, args_info));
147 break;
148 }
149 }
150 };
151 support::parallel_for_dynamic(0, ed - st, ctx->num_threads, f_worker);
152 Array<MeasureCandidate> filtered;
153 filtered.reserve(ed - st);
154 for (MeasureCandidate result : per_task_result)
155 if (result.defined()) {
156 filtered.push_back(result);
157 }
158 return filtered;
159}
160
161inline void ReplayTraceNode::State::NotifyRunnerResults(const Array<RunnerResult>& results) {
162 st += self->num_trials_per_iter;
163 ed += self->num_trials_per_iter;
164}

166SearchStrategy SearchStrategy::ReplayTrace(int num_trials_per_iter, int max_trials_per_task,
167 int max_fail_count) {
168 ObjectPtr<ReplayTraceNode> n = make_object<ReplayTraceNode>();
169 n->num_trials_per_iter = num_trials_per_iter;
170 n->max_trials_per_task = max_trials_per_task;
171 n->max_fail_count = max_fail_count;
172 return SearchStrategy(n);
173}
