21namespace tvm {
22namespace meta_schedule {

24/*! \brief A search strategy that generates measure candidates using space generator. */
25class ReplayFuncNode : public SearchStrategyNode {
26 public:
27 /*! \brief The state of the search strategy. */
28 struct State {
29 /*! \brief The search strategy itself */
30 ReplayFuncNode* self;
31 /*! \brief `[st, ed)` are the indices of the next batch of candidates. */
32 int st;
33 /*! \brief `[st, ed)` are the indices of the next batch of candidates. */
34 int ed;
35
36 explicit State(ReplayFuncNode* self) : self(self), st(0), ed(self->num_trials_per_iter) {
37 const TuneContextNode* ctx = self->context_;
38 ICHECK(ctx);
39 }
40
41 inline Optional<Array<MeasureCandidate>> GenerateMeasureCandidates();
42 inline void NotifyRunnerResults(const Array<RunnerResult>& results);
43 };
44
45 /*! \brief The number of trials per iteration. */
46 int num_trials_per_iter;
47 /*! \brief The number of total trials. */
48 int max_trials_per_task;
49
50 /*! \brief The tuning context of the search strategy. */
51 const TuneContextNode* context_{nullptr};
52 /*! \brief The random state. -1 means using random number. */
53 TRandState rand_state_ = -1;
54 /*! \brief The state of the search strategy. */
55 std::unique_ptr<State> state_ = nullptr;
56
57 void VisitAttrs(tvm::AttrVisitor* v) {
58 v->Visit("num_trials_per_iter", &num_trials_per_iter);
59 v->Visit("max_trials_per_task", &max_trials_per_task);
60 // `context_` is not visited.
61 // `rand_state_` is not visited
62 // `state_` is not visited
63 }
64
65 static constexpr const char* _type_key = "meta_schedule.ReplayFunc";
66 TVM_DECLARE_FINAL_OBJECT_INFO(ReplayFuncNode, SearchStrategyNode);
67
68 void InitializeWithTuneContext(const TuneContext& context) final {
69 CHECK(context->space_generator.defined())
70 << "ValueError: TuneContext.space_generator is not defined";
71 CHECK(context->mod.defined()) << "ValueError: TuneContext.mod is not defined";
72 this->context_ = context.get();
73 this->rand_state_ = ForkSeed(&context->rand_state);
74 this->state_.reset();
75 }
76
77 void PreTuning(const Array<tir::Schedule>& design_spaces, const Optional<Database>& database,
78 const Optional<CostModel>& cost_model) final {
79 CHECK(this->context_ != nullptr) << "ValueError: Did you forget to initialize the TuneContext?";
80 if (this->state_ != nullptr) {
81 TVM_PY_LOG(WARNING, this->context_->logging_func) << "ReplayFunc is already initialized.";
82 this->state_.reset();
83 }
84 ICHECK(this->state_ == nullptr);
85 this->state_ = std::make_unique<State>(this);
86 }
87
88 void PostTuning() final {
89 ICHECK(this->state_ != nullptr);
90 this->state_.reset();
91 }
92
93 Optional<Array<MeasureCandidate>> GenerateMeasureCandidates() final {
94 ICHECK(this->state_ != nullptr);
95 return this->state_->GenerateMeasureCandidates();
96 }
97
98 void NotifyRunnerResults(const Array<MeasureCandidate>& measure_candidates,
99 const Array<RunnerResult>& results) final {
100 ICHECK(this->state_ != nullptr);
101 this->state_->NotifyRunnerResults(results);
102 }
103};
104
105inline Optional<Array<MeasureCandidate>> ReplayFuncNode::State::GenerateMeasureCandidates() {
106 if (st >= self->max_trials_per_task) {
107 return NullOpt;
108 }
109 ed = std::min(ed, self->max_trials_per_task);
110 Array<MeasureCandidate> result;
111 const TuneContextNode* ctx = self->context_;
112 ICHECK(ctx);
113 IRModule mod = ctx->mod.value();
114 for (int i = st; i < ed; i++) {
115 for (;;) {
116 Array<tir::Schedule> schs = ctx->space_generator.value()->GenerateDesignSpace(mod);
117 int design_space_index = tir::SampleInt(&self->rand_state_, 0, schs.size());
118 tir::Schedule sch = schs[design_space_index];
119 sch->EnterPostproc();
120 bool failed = false;
121 for (const Postproc& proc : ctx->postprocs) {
122 if (!proc->Apply(sch)) {
123 failed = true;
124 break;
125 }
126 }
127 if (!failed) {
128 Array<ArgInfo> args_info = ArgInfo::FromEntryFunc(sch->mod(), /*remove_preproc=*/true);
129 result.push_back(MeasureCandidate(sch, args_info));
130 break;
131 }
132 }
133 }
134 return result;
135}
136
137inline void ReplayFuncNode::State::NotifyRunnerResults(const Array<RunnerResult>& results) {
138 st += self->num_trials_per_iter;
139 ed += self->num_trials_per_iter;
140}
141
142SearchStrategy SearchStrategy::ReplayFunc(int num_trials_per_iter, int max_trials_per_task) {
143 ObjectPtr<ReplayFuncNode> n = make_object<ReplayFuncNode>();
144 n->num_trials_per_iter = num_trials_per_iter;
145 n->max_trials_per_task = max_trials_per_task;
146 return SearchStrategy(n);
147}
