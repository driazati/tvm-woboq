34/*! \brief An auxiliary data structure to help deduplicate IRModules */
35class IRModuleSet {
36 public:
37 explicit IRModuleSet(const ModuleEquality& mod_eq)
38 : tab_(/*bucket_count*/ 0, ItemHash(), ItemEqual(mod_eq)) {}
39
40 /*! \brief Add an IRModule to the set */
41 void Add(const IRModule& mod, size_t shash) { tab_.insert(Item{mod, shash}); }
42 /*! \brief Check if the IRModule is in the set */
43 bool Has(const IRModule& mod, size_t shash) const { return tab_.count(Item{mod, shash}); }
44
45 private:
46 struct Item {
47 IRModule mod;
48 size_t shash;
49 };
50 struct ItemHash {
51 size_t operator()(const Item& hash) const { return hash.shash; }
52 };
53 struct ItemEqual {
54 explicit ItemEqual(const ModuleEquality& mod_eq) : mod_eq_(mod_eq) {}
55 ItemEqual& operator=(const ItemEqual& other) { return *this; }
56
57 bool operator()(const Item& lhs, const Item& rhs) const {
58 return lhs.shash == rhs.shash && mod_eq_.Equal(lhs.mod, rhs.mod);
59 }
60
61 const ModuleEquality& mod_eq_;
62 };
63
64 std::unordered_set<Item, ItemHash, ItemEqual> tab_;
65};

67/*!
68 * \brief A heap with a size up-limit. If overflow happens, it evicted the worst items.
69 * \note It maintains a min heap in terms of `Item::score`. Therefore, when
70 * overflow happens, the element evicted is the one with the min `Item::score`.
71 * As time goes, the elements in the heap are going to be larger.
72 */
73class SizedHeap {
74 public:
75 struct Item {
76 Schedule sch;
77 double score;
78 bool operator<(const Item& other) const { return score > other.score; }
79 };
80
81 /*!
82 * \brief Constructor
83 * \param size_limit The up-limit of the heap size
84 */
85 explicit SizedHeap(int size_limit) : size_limit(size_limit) { heap.reserve(size_limit); }
86
87 /*!
88 * \brief Push the specific item to the heap if its key did not appears in the heap
89 * \param item The item to be pushed
90 */
91 void Push(Schedule sch, double score) {
92 int size = heap.size();
93 if (size < size_limit) {
94 // Heap is not full, just push
95 heap.emplace_back(Item{sch, score});
96 std::push_heap(heap.begin(), heap.end());
97 } else if (score > heap.front().score) {
98 // if the item is better than the worst one in the heap, we can safely kick it out
99 std::pop_heap(heap.begin(), heap.end());
100 heap.back() = {sch, score};
101 std::push_heap(heap.begin(), heap.end());
102 }
103 // Otherwise, the item is worse than any other element in the heap
104 }
105
106 /*! \brief Up-limit of the heap size */
107 int size_limit;
108 /*! \brief The heap, the worse the topper */
109 std::vector<Item> heap;
110};

112struct PerThreadData {
113 IRModule mod{nullptr};
114 TRandState rand_state{-1};
115 std::function<int32_t()> trace_sampler = nullptr;
116 std::function<Optional<Mutator>()> mutator_sampler = nullptr;
117
118 /*!
119 * \brief Set the value for the trace and mutator samplers per thread.
120 * \param scores The predicted score for the given samples.
121 * \param genetic_mutate_prob The probability of mutation.
122 * \param mutator_probs The probability of each mutator as a dict.
123 */
124 void Set(const std::vector<double>& scores, double genetic_mutate_prob,
125 const Map<Mutator, FloatImm>& mutator_probs) {
126 trace_sampler = tir::MakeMultinomialSampler(&rand_state, scores);
127 mutator_sampler = MakeMutatorSampler(genetic_mutate_prob, mutator_probs, &rand_state);
128 }
129
130 private:
131 /*!
132 * \brief Create a sampler function that picks mutators according to the mass function
133 * \param rand_state The random state for sampling
134 * \return The sampler created
135 */
136 static std::function<Optional<Mutator>()> MakeMutatorSampler(
137 double genetic_mutate_prob, //
138 const Map<Mutator, FloatImm>& mutator_probs, //
139 TRandState* rand_state) {
140 std::vector<Optional<Mutator>> mutators;
141 std::vector<double> masses;
142 mutators.push_back(NullOpt);
143 masses.push_back(1.0 - genetic_mutate_prob);
144 double total_mass_mutator = 0.0;
145 if (genetic_mutate_prob > 0) {
146 for (const auto& kv : mutator_probs) {
147 Mutator mutator = kv.first;
148 double mass = kv.second->value;
149 total_mass_mutator += mass;
150 mutators.push_back(mutator);
151 masses.push_back(mass * genetic_mutate_prob);
152 }
153 }
154 // Normalize the sum to 1.0
155 if (total_mass_mutator == 0.0) {
156 masses[0] = 1.0;
157 for (int i = 1, n = masses.size(); i < n; ++i) {
158 masses[i] = 0.0;
159 }
160 } else if (total_mass_mutator != 1.0) {
161 for (int i = 1, n = masses.size(); i < n; ++i) {
162 masses[i] /= total_mass_mutator;
163 }
164 }
165 return [idx_sampler = tir::MakeMultinomialSampler(rand_state, masses),
166 mutators = std::move(mutators)]() -> Optional<Mutator> {
167 int i = idx_sampler();
168 return mutators[i];
169 };
170 }
171};

173struct ConcurrentBitmask {
174 /*! The bit width. */
175 static constexpr const int kBitWidth = 64;
176 /*! \brief The size of the concurrent bitmask. */
177 int size;
178 /*! \brief The bitmasks. */
179 std::vector<uint64_t> bitmask;
180 /*! \brief The mutexes, one per kBitWidth(64 here) bitmasks. */
181 std::vector<std::mutex> mutexes;
182
183 /*!
184 * \brief Constructor
185 * \param n The total slots managed by the concurrent bitmask.
186 */
187 explicit ConcurrentBitmask(int n)
188 : size((n + kBitWidth - 1) / kBitWidth), bitmask(size, 0), mutexes(size) {}
189 /*!
190 * \brief Query and mark the given index if not visited before.
191 * \param x The index to concurrently check if used. If not, mark as used.
192 * \return Whether the index has been used before.
193 */
194 bool QueryAndMark(int x) {
195 constexpr uint64_t one = 1;
196 std::unique_lock<std::mutex> lock(mutexes[x / kBitWidth]);
197 if (bitmask[x / kBitWidth] & (one << (x % kBitWidth))) {
198 return false;
199 } else {
200 bitmask[x / kBitWidth] |= one << (x % kBitWidth);
201 return true;
202 }
203 }
204};

208/*!
209 * \brief Assemble measure candidates from the given candidate traces.
210 * \param traces The picked candidate traces.
211 * \return The assembled measure candidates.
212 */
213Array<MeasureCandidate> AssembleCandidates(const std::vector<Schedule>& picks) {
214 Array<MeasureCandidate> measure_inputs;
215 measure_inputs.reserve(picks.size());
216 for (const Schedule& sch : picks) {
217 measure_inputs.push_back(
218 MeasureCandidate(sch, ArgInfo::FromEntryFunc(sch->mod(), /*remove_preproc=*/true)));
219 }
220 return measure_inputs;
221}
222
223/*!
224 * \brief Predict the normalized score of each candidate.
225 * \param candidates The candidates for prediction
226 * \param task The search task
227 * \param space The search space
228 * \return The normalized score in the prediction
229 */
230std::vector<double> PredictNormalizedScore(const std::vector<Schedule>& candidates,
231 const TuneContext& context,
232 const CostModel& cost_model) {
233 auto _ = Profiler::TimedScope("EvoSearch/Evolve/PredictNormalizedScore");
234 ICHECK(!candidates.empty()) << "Candidates given for score prediction can not be empty list!";
235 std::vector<double> scores = cost_model->Predict(context, AssembleCandidates(candidates));
236 for (double& score : scores) {
237 score = std::max(0.0, score);
238 }
239 return scores;
240}

244/*!\brief A search strategy that generates measure candidates using evolutionary search. */

249 /*! \brief The search strategy itself */

251 /*! \brief The number of total trials. */
252 int max_trials;
253 /*! \brief The number of trials per iteration. */
254 int num_trials_per_iter;
255 /*! \brief `[st, ed)` are the indices of the next batch of candidates. */
256 int st;
257 /*! \brief `[st, ed)` are the indices of the next batch of candidates. */

259 /*! \brief The counter of returning empty results. */

262 Array<tir::Trace> design_spaces;
263 /*! \brief Pre thread data including module to be tuned and random state. */
264 std::vector<PerThreadData> per_thread_data_;
265 /*!
266 * \brief The workloads that are already measured.
267 * TODO(junrushao1994): add records from the database to avoid re-measuring.
268 * */
269 IRModuleSet measured_workloads_;
270 /*! \brief A Database for selecting useful candidates. */
271 Database database_{nullptr};
272 /*! \brief A cost model helping to explore the search space */
273 CostModel cost_model_{nullptr};
274 /*! \brief The token registered for the given workload in database. */
275 Workload token_{nullptr};

277 explicit State(EvolutionarySearchNode* self, int max_trials, int num_trials_per_iter,
278 Array<Schedule> design_space_schedules, Database database, CostModel cost_model)
279 : self(self),
280 max_trials(max_trials),
281 num_trials_per_iter(num_trials_per_iter),
282 st(0),
283 ed(num_trials_per_iter),
284 num_empty_iters(0),
285 measured_workloads_(database->GetModuleEquality()) {
286 design_spaces.reserve(design_spaces.size());
287 for (const Schedule& space : design_space_schedules) {
288 design_spaces.push_back(space->trace().value()->Simplified(true));
289 }
290 const TuneContextNode* ctx = self->ctx_;
291 IRModule mod = ctx->mod.value();
292 this->per_thread_data_.resize(ctx->num_threads);
293 for (PerThreadData& data : this->per_thread_data_) {
294 data.mod = DeepCopyIRModule(mod);
295 data.rand_state = ForkSeed(&self->rand_state_);
296 }
297 this->database_ = database;
298 this->cost_model_ = cost_model;
299 this->token_ = database->CommitWorkload(mod);
300 }

411 this->mutator_probs_ = ctx->space_generator.value()->mutator_probs.value();
412 this->rand_state_ = ForkSeed(&ctx->rand_state);
413 this->state_.reset();
414 }

416 void PreTuning(int max_trials, int num_trials_per_iter, const Array<Schedule>& design_spaces,
417 const Optional<Database>& database, const Optional<CostModel>& cost_model) final {
418 ICHECK(!design_spaces.empty());
419 CHECK(this->ctx_ != nullptr) << "ValueError: Did you forget to initialize the TuneContext?";
420 CHECK(database.defined())
421 << "ValueError: Database is not supplied in PreTuning. Evolutionary"
422 "search algorithm requires a database to be present, so that it "
423 "could sample from previously-explored population. If you do not "
424 "intent to store data on disk, please use `tvm.meta_schedule.database.MemoryDatabase`";
425 CHECK(cost_model.defined())
426 << "ValueError: CostModel is not supplied in PreTuning. Evolutionary search "
427 "algorithm expects a cost model to filter out potentially less efficient kernels. If "
428 "you do not expect a cost model to help, please use "
429 "`tvm.meta_schedule.cost_model.RandomModel`";
430 CHECK(this->state_ == nullptr)
431 << "ValueError: `PreTuning` is already invoked without corresponding `PostTuning`.";
432 this->state_ = std::make_unique<State>(this, max_trials, num_trials_per_iter, design_spaces,
433 database.value(), cost_model.value());
434 }

470std::vector<Schedule> EvolutionarySearchNode::State::PickBestFromDatabase(int num) {
471 auto _ = Profiler::TimedScope("EvoSearch/PickBestFromDatabase");
472 std::vector<tir::Trace> measured_traces;
473 measured_traces.reserve(num);
474 Array<TuningRecord> top_records = this->database_->GetTopK(this->token_, num);
475 for (TuningRecord record : top_records) {
476 measured_traces.push_back(record->trace);
477 }
478 int actual_num = measured_traces.size();
479 ThreadedTraceApply pp(self->postprocs_);
480 std::vector<Schedule> results(actual_num, Schedule{nullptr});
481 auto f_proc_measured = [this, &measured_traces, &results, &pp](int thread_id,
482 int trace_id) -> void {
483 PerThreadData& data = this->per_thread_data_.at(thread_id);
484 TRandState* rand_state = &data.rand_state;
485 const IRModule& mod = data.mod;
486 tir::Trace trace = measured_traces.at(trace_id);
487 Schedule& result = results.at(trace_id);
488 ICHECK(!result.defined());
489 if (Optional<Schedule> sch = pp.Apply(mod, trace, rand_state)) {
490 result = sch.value();
491 } else {
492 LOG(FATAL) << "ValueError: Cannot postprocess the trace:\n" << trace;
493 throw;
494 }
495 };
496 support::parallel_for_dynamic(0, actual_num, self->ctx_->num_threads, f_proc_measured);
497 return results;
498}

500std::vector<Schedule> EvolutionarySearchNode::State::SampleInitPopulation(int num) {
501 auto _ = Profiler::TimedScope("EvoSearch/SampleInitPopulation");
502 ThreadedTraceApply pp(self->postprocs_);
503 std::vector<Schedule> out_schs;
504 while (static_cast<int>(out_schs.size()) < self->init_min_unmeasured) {
505 std::vector<Schedule> results(num, Schedule{nullptr});
506 auto f_proc_unmeasured = [this, &results, &pp](int thread_id, int trace_id) -> void {
507 PerThreadData& data = this->per_thread_data_.at(thread_id);
508 TRandState* rand_state = &data.rand_state;
509 const IRModule& mod = data.mod;
510 Schedule& result = results.at(trace_id);
511 ICHECK(!result.defined());
512 int design_space_index = tir::SampleInt(rand_state, 0, design_spaces.size());
513 tir::Trace trace(design_spaces[design_space_index]->insts, {});
514 if (Optional<Schedule> sch = pp.Apply(mod, trace, rand_state)) {
515 result = sch.value();
516 }
517 };
518 support::parallel_for_dynamic(0, num, self->ctx_->num_threads, f_proc_unmeasured);
519 for (int i = 0; i < num; i++) {
520 if (results[i].defined()) {
521 out_schs.push_back(results[i]);
522 }
523 }
524 TVM_PY_LOG(INFO, self->ctx_->logger) << "Sample-Init-Population summary:\n"
525 << pp.SummarizeFailures();
526 }
527 return out_schs;
528}

530std::vector<Schedule> EvolutionarySearchNode::State::EvolveWithCostModel(
531 std::vector<Schedule> population, int num) {
532 IRModuleSet exists(database_->GetModuleEquality());
533 {
534 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Misc/CopyMeasuredWorkloads");
535 ICHECK_GT(num, 0);
536 // The heap to record best schedule, we do not consider schedules that are already measured
537 exists = this->measured_workloads_;
538 }
539 SizedHeap heap(num);
540 for (int iter = 0;; ++iter) {
541 // Predict normalized score with the cost model,
542 std::vector<double> scores =
543 PredictNormalizedScore(population, GetRef<TuneContext>(self->ctx_), this->cost_model_);
544
545 {
546 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Misc");
547 ICHECK_EQ(scores.size(), population.size());
548 for (int i = 0, n = population.size(); i < n; ++i) {
549 Schedule sch = population.at(i);
550 IRModule mod = sch->mod();
551 size_t shash = ModuleHash(mod);
552 double score = scores.at(i);
553 if (!exists.Has(mod, shash)) {
554 exists.Add(mod, shash);
555 heap.Push(sch, score);
556 }
557 }
558 // Discontinue once it reaches end of search
559 if (iter == self->genetic_num_iters) {
560 break;
561 }
562 // Set threaded samplers, with probability from predicated normalized throughput
563 for (PerThreadData& data : this->per_thread_data_) {
564 data.Set(scores, self->genetic_mutate_prob, self->mutator_probs_);
565 }
566 }
567 {
568 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Mutation");
569 ThreadedTraceApply pp(self->postprocs_);
570 ConcurrentBitmask cbmask(self->population_size);
571 std::vector<Schedule> next_population(self->population_size, Schedule{nullptr});
572 // The worker function
573 auto f_find_candidate = [&cbmask, &population, &next_population, &pp, this](int thread_id,
574 int trace_id) {
575 // Prepare samplers
576 PerThreadData& data = this->per_thread_data_.at(thread_id);
577 TRandState* rand_state = &data.rand_state;
578 const IRModule& mod = data.mod;
579 std::function<int()>& trace_sampler = data.trace_sampler;
580 std::function<Optional<Mutator>()>& mutator_sampler = data.mutator_sampler;
581 Schedule& result = next_population.at(trace_id);
582 int sampled_trace_id = -1;
583 // Loop until success
584 for (int fail_count = 0; fail_count <= self->genetic_max_fail_count; ++fail_count) {
585 sampled_trace_id = trace_sampler();
586 tir::Trace trace = population.at(sampled_trace_id)->trace().value();
587 if (Optional<Mutator> opt_mutator = mutator_sampler()) {
588 // Decision: mutate
589 Mutator mutator = opt_mutator.value();
590 if (Optional<tir::Trace> new_trace = mutator->Apply(trace, rand_state)) {
591 if (Optional<Schedule> sch = pp.Apply(mod, new_trace.value(), rand_state)) {
592 // note that sch's trace is different from new_trace
593 // because it contains post-processing information
594 result = sch.value();
595 break;
596 }
597 }
598 } else if (cbmask.QueryAndMark(sampled_trace_id)) {
599 // Decision: do not mutate
600 break;
601 }
602 }
603 // if retry count exceeds the limit, reuse an old sample
604 if (!result.defined()) {
605 result = population.at(sampled_trace_id);
606 }
607 };
608 support::parallel_for_dynamic(0, self->population_size, self->ctx_->num_threads,
609 f_find_candidate);
610
611 population.swap(next_population);
612 TVM_PY_LOG(INFO, self->ctx_->logger) << "Evolve iter #" << iter << " done. Summary:\n"
613 << pp.SummarizeFailures();
614 }
615 }
616 // Return the best states from the heap, sorting from higher score to lower ones
617 {
618 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Misc");
619 std::sort(heap.heap.begin(), heap.heap.end());
620 std::vector<Schedule> results;
621 results.reserve(num);
622 for (const SizedHeap::Item& item : heap.heap) {
623 results.push_back(item.sch);
624 }
625
626 constexpr int kNumScoresPerLine = 16;
627 std::ostringstream os;
628 int n = heap.heap.size();
629 for (int st = 0; st < n; st += kNumScoresPerLine) {
630 os << std::endl;
631 int ed = std::min(st + kNumScoresPerLine, n);
632 os << "[" << (st + 1) << " : " << ed << "]:\t";
633 for (int i = st; i < ed; ++i) {
634 if (i != st) {
635 os << " ";
636 }
637 os << std::fixed << std::setprecision(4) << heap.heap.at(i).score;
638 }
639 }
640 TVM_PY_LOG(INFO, self->ctx_->logger)
641 << "Scores of the best " << n << " candidates:" << os.str();
642 return results;
643 }
644}

646std::vector<Schedule> EvolutionarySearchNode::State::PickWithEpsGreedy(
647 const std::vector<Schedule>& unmeasured, const std::vector<Schedule>& bests, int num) {
648 auto _ = Profiler::TimedScope("EvoSearch/PickWithEpsGreedy");
649 int num_rands = num * self->eps_greedy;
650 int num_bests = num - num_rands;
651 std::vector<int> rands =
652 tir::SampleWithoutReplacement(&self->rand_state_, unmeasured.size(), unmeasured.size());
653 std::vector<Schedule> results;
654 results.reserve(num);
655 IRModuleSet& measured_workloads = this->measured_workloads_;
656 for (int i = 0, i_bests = 0, i_rands = 0; i < num; ++i) {
657 bool has_best = i_bests < static_cast<int>(bests.size());
658 bool has_rand = i_rands < static_cast<int>(rands.size());
659 // Pick a schedule
660 Schedule sch{nullptr};
661 // If needs `bests`, then prefer `bests`
662 if (i < num_bests) {
663 if (has_best) {
664 sch = bests[i_bests++];
665 } else if (has_rand) {
666 sch = unmeasured[rands[i_rands++]];
667 } else {
668 break;
669 }
670 } else {
671 // Else prefer `rands`
672 if (has_rand) {
673 sch = unmeasured[rands[i_rands++]];
674 } else if (has_best) {
675 sch = bests[i_bests++];
676 } else {
677 break;
678 }
679 }
680 IRModule mod = sch->mod();
681 size_t shash = ModuleHash(mod);
682 if (!measured_workloads.Has(mod, shash)) {
683 measured_workloads.Add(mod, shash);
684 results.push_back(sch);
685 }
686 }
687 return results;
688}

690Optional<Array<MeasureCandidate>> EvolutionarySearchNode::State::GenerateMeasureCandidates() {
691 if (st >= max_trials) {
692 return NullOpt;
693 }
694 int sample_num = num_trials_per_iter;
695 if (ed > max_trials) {
696 sample_num = max_trials - st;
697 ed = max_trials;
698 }
699 ICHECK_LT(st, ed);
700 int pop = self->population_size;
701 std::vector<Schedule> inits;
702 inits.reserve(pop);
703
704 TVM_PY_LOG(INFO, self->ctx_->logger) << "Generating candidates......";
705 std::vector<Schedule> measured = PickBestFromDatabase(pop * self->init_measured_ratio);
706 TVM_PY_LOG(INFO, self->ctx_->logger)
707 << "Picked top " << measured.size() << " candidate(s) from database";
708 std::vector<Schedule> unmeasured = SampleInitPopulation(pop - measured.size());
709 TVM_PY_LOG(INFO, self->ctx_->logger) << "Sampled " << unmeasured.size() << " candidate(s)";
710 inits.insert(inits.end(), measured.begin(), measured.end());
711 inits.insert(inits.end(), unmeasured.begin(), unmeasured.end());
712 std::vector<Schedule> bests = EvolveWithCostModel(inits, sample_num);
713 TVM_PY_LOG(INFO, self->ctx_->logger)
714 << "Got " << bests.size() << " candidate(s) with evolutionary search";
715 std::vector<Schedule> picks = PickWithEpsGreedy(unmeasured, bests, sample_num);
716 TVM_PY_LOG(INFO, self->ctx_->logger)
717 << "Sending " << picks.size() << " candidates(s) for measurement";
718 if (picks.empty()) {
719 ++this->num_empty_iters;
720 if (this->num_empty_iters >= self->num_empty_iters_before_early_stop) {
721 return NullOpt;
722 }
723 }
724 return AssembleCandidates(picks);
725}

770Array<Schedule> EvolutionarySearchEvolveWithCostModel(EvolutionarySearch self,
771 Array<Schedule> population, int num) {
772 Array<Schedule> result;
773 std::vector<Schedule> population_vec =
774 std::vector<Schedule>(population.begin(), population.end());
775 std::vector<Schedule> schs = self->state_->EvolveWithCostModel(population_vec, num);
776 for (Schedule sch : schs) {
777 IRModule mod = sch->mod();
778 size_t shash = self->state_->ModuleHash(mod);
779 if (!self->state_->measured_workloads_.Has(mod, shash)) {
780 self->state_->measured_workloads_.Add(mod, shash);
781 result.push_back(sch);
782 }
783 }
784 return result;
