33/*! \brief An auxiliary data structure to help deduplicate IRModules */
34class IRModuleSet {
35 public:
36 /*! \brief Add an IRModule to the set */
37 void Add(const IRModule& mod, size_t shash) { tab_.insert(Item{mod, shash}); }
38 /*! \brief Check if the IRModule is in the set */
39 bool Has(const IRModule& mod, size_t shash) const { return tab_.count(Item{mod, shash}); }
40
41 private:
42 struct Item {
43 IRModule mod;
44 size_t shash;
45 };
46 struct ItemHash {
47 size_t operator()(const Item& hash) const { return hash.shash; }
48 };
49 struct ItemEqual {
50 bool operator()(const Item& lhs, const Item& rhs) const {
51 return lhs.shash == rhs.shash && StructuralEqual()(lhs.mod, rhs.mod);
52 }
53 };
54 std::unordered_set<Item, ItemHash, ItemEqual> tab_;
55};

57/*!
58 * \brief A heap with a size up-limit. If overflow happens, it evicted the worst items.
59 * \note It maintains a min heap in terms of `Item::score`. Therefore, when
60 * overflow happens, the element evicted is the one with the min `Item::score`.
61 * As time goes, the elements in the heap are going to be larger.
62 */
63class SizedHeap {
64 public:
65 struct Item {
66 Schedule sch;
67 double score;
68 bool operator<(const Item& other) const { return score > other.score; }
69 };
70
71 /*!
72 * \brief Constructor
73 * \param size_limit The up-limit of the heap size
74 */
75 explicit SizedHeap(int size_limit) : size_limit(size_limit) { heap.reserve(size_limit); }
76
77 /*!
78 * \brief Push the specific item to the heap if its key did not appears in the heap
79 * \param item The item to be pushed
80 */
81 void Push(Schedule sch, double score) {
82 int size = heap.size();
83 if (size < size_limit) {
84 // Heap is not full, just push
85 heap.emplace_back(Item{sch, score});
86 std::push_heap(heap.begin(), heap.end());
87 } else if (score > heap.front().score) {
88 // if the item is better than the worst one in the heap, we can safely kick it out
89 std::pop_heap(heap.begin(), heap.end());
90 heap.back() = {sch, score};
91 std::push_heap(heap.begin(), heap.end());
92 }
93 // Otherwise, the item is worse than any other element in the heap
94 }
95
96 /*! \brief Up-limit of the heap size */
97 int size_limit;
98 /*! \brief The heap, the worse the topper */
99 std::vector<Item> heap;
100};

102struct PerThreadData {
103 IRModule mod{nullptr};
104 TRandState rand_state{-1};
105 std::function<int32_t()> trace_sampler = nullptr;
106 std::function<Optional<Mutator>()> mutator_sampler = nullptr;
107
108 /*!
109 * \brief Set the value for the trace and mutator samplers per thread.
110 * \param scores The predicted score for the given samples.
111 * \param genetic_mutate_prob The probability of mutation.
112 * \param mutator_probs The probability of each mutator as a dict.
113 */
114 void Set(const std::vector<double>& scores, double genetic_mutate_prob,
115 const Map<Mutator, FloatImm>& mutator_probs) {
116 trace_sampler = tir::MakeMultinomialSampler(&rand_state, scores);
117 mutator_sampler = MakeMutatorSampler(genetic_mutate_prob, mutator_probs, &rand_state);
118 }
119
120 private:
121 /*!
122 * \brief Create a sampler function that picks mutators according to the mass function
123 * \param rand_state The random state for sampling
124 * \return The sampler created
125 */
126 static std::function<Optional<Mutator>()> MakeMutatorSampler(
127 double genetic_mutate_prob, //
128 const Map<Mutator, FloatImm>& mutator_probs, //
129 TRandState* rand_state) {
130 std::vector<Optional<Mutator>> mutators;
131 std::vector<double> masses;
132 mutators.push_back(NullOpt);
133 masses.push_back(1.0 - genetic_mutate_prob);
134 double total_mass_mutator = 0.0;
135 if (genetic_mutate_prob > 0) {
136 for (const auto& kv : mutator_probs) {
137 Mutator mutator = kv.first;
138 double mass = kv.second->value;
139 total_mass_mutator += mass;
140 mutators.push_back(mutator);
141 masses.push_back(mass * genetic_mutate_prob);
142 }
143 }
144 // Normalize the sum to 1.0
145 if (total_mass_mutator == 0.0) {
146 masses[0] = 1.0;
147 for (int i = 1, n = masses.size(); i < n; ++i) {
148 masses[i] = 0.0;
149 }
150 } else if (total_mass_mutator != 1.0) {
151 for (int i = 1, n = masses.size(); i < n; ++i) {
152 masses[i] /= total_mass_mutator;
153 }
154 }
155 return [idx_sampler = tir::MakeMultinomialSampler(rand_state, masses),
156 mutators = std::move(mutators)]() -> Optional<Mutator> {
157 int i = idx_sampler();
158 return mutators[i];
159 };
160 }
161};

163struct ConcurrentBitmask {
164 /*! The bit width. */
165 static constexpr const int kBitWidth = 64;
166 /*! \brief The size of the concurrent bitmask. */
167 int size;
168 /*! \brief The bitmasks. */
169 std::vector<uint64_t> bitmask;
170 /*! \brief The mutexes, one per kBitWidth(64 here) bitmasks. */
171 std::vector<std::mutex> mutexes;
172
173 /*!
174 * \brief Constructor
175 * \param n The total slots managed by the concurrent bitmask.
176 */
177 explicit ConcurrentBitmask(int n)
178 : size((n + kBitWidth - 1) / kBitWidth), bitmask(size, 0), mutexes(size) {}
179 /*!
180 * \brief Query and mark the given index if not visited before.
181 * \param x The index to concurrently check if used. If not, mark as used.
182 * \return Whether the index has been used before.
183 */
184 bool QueryAndMark(int x) {
185 constexpr uint64_t one = 1;
186 std::unique_lock<std::mutex> lock(mutexes[x / kBitWidth]);
187 if (bitmask[x / kBitWidth] & (one << (x % kBitWidth))) {
188 return false;
189 } else {
190 bitmask[x / kBitWidth] |= one << (x % kBitWidth);
191 return true;
192 }
193 }
194};

198/*!
199 * \brief Assemble measure candidates from the given candidate traces.
200 * \param traces The picked candidate traces.
201 * \return The assembled measure candidates.
202 */
203Array<MeasureCandidate> AssembleCandidates(const std::vector<Schedule>& picks) {
204 Array<MeasureCandidate> measure_inputs;
205 measure_inputs.reserve(picks.size());
206 for (const Schedule& sch : picks) {
207 measure_inputs.push_back(
208 MeasureCandidate(sch, ArgInfo::FromEntryFunc(sch->mod(), /*remove_preproc=*/true)));
209 }
210 return measure_inputs;
211}

213/*!
214 * \brief Predict the normalized score of each candidate.
215 * \param candidates The candidates for prediction
216 * \param task The search task
217 * \param space The search space
218 * \return The normalized score in the prediction
219 */
220std::vector<double> PredictNormalizedScore(const std::vector<Schedule>& candidates,
221 const TuneContext& context,
222 const CostModel& cost_model) {
223 auto _ = Profiler::TimedScope("EvoSearch/Evolve/PredictNormalizedScore");
224 ICHECK(!candidates.empty()) << "Candidates given for score prediction can not be empty list!";
225 std::vector<double> scores = cost_model->Predict(context, AssembleCandidates(candidates));
226 for (double& score : scores) {
227 score = std::max(0.0, score);
228 }
229 return scores;
230}

263 explicit State(EvolutionarySearchNode* self, Array<tir::Trace> design_spaces, Database database,

377 void InitializeWithTuneContext(const TuneContext& context) final {
378 CHECK(context.defined()) << "TuneContext must be defined!";
379 CHECK(context->num_threads > 0) << "Number of threads has to be larger than 0.";
380 CHECK(context->target.defined()) << "Target must be defined!";
381 this->context_ = context.get();
382 this->rand_state_ = ForkSeed(&context->rand_state);
383 for (const auto& kv : context->mutator_probs) {
384 double mass = kv.second->value;
385 TVM_META_SCHEDULE_CHECK_PROB_RANGE(mass, "mutator_probs");
386 }
387 this->state_.reset();
388 }

390 void PreTuning(const Array<Schedule>& design_spaces, const Optional<Database>& database,
391 const Optional<CostModel>& cost_model) final {
392 ICHECK(!design_spaces.empty());
393 CHECK(this->context_ != nullptr) << "ValueError: Did you forget to initialize the TuneContext?";
394 CHECK(database.defined())
395 << "ValueError: Database is not supplied in PreTuning. Evolutionary"
396 "search algorithm requires a database to be present, so that it "
397 "could sample from previously-explored population. If you do not "
398 "intent to store data on disk, please use `tvm.meta_schedule.database.MemoryDatabase`";
399 CHECK(cost_model.defined())
400 << "ValueError: CostModel is not supplied in PreTuning. Evolutionary search "
401 "algorithm expects a cost model to filter out potentially less efficient kernels. If "
402 "you do not expect a cost model to help, please use "
403 "`tvm.meta_schedule.cost_model.RandomModel`";
404 if (this->state_ != nullptr) {
405 TVM_PY_LOG(WARNING, this->context_->logging_func)
406 << "EvolutionarySearch is already initialized.";
407 this->state_.reset();
408 }
409 ICHECK(this->state_ == nullptr);
410 Array<tir::Trace> design_space_traces;
411 design_space_traces.reserve(design_spaces.size());
412 for (const Schedule& space : design_spaces) {
413 design_space_traces.push_back(space->trace().value()->Simplified(true));
414 }
415 this->state_ =
416 std::make_unique<State>(this, design_space_traces, database.value(), cost_model.value());
417 }
418
419 void PostTuning() final {
420 ICHECK(this->state_ != nullptr);
421 this->state_.reset();
422 }
423
424 Optional<Array<MeasureCandidate>> GenerateMeasureCandidates() final {
425 ICHECK(this->state_ != nullptr);
426 return this->state_->GenerateMeasureCandidates();
427 }
428
429 void NotifyRunnerResults(const Array<MeasureCandidate>& measure_candidates,
430 const Array<RunnerResult>& results) final {
431 ICHECK(this->state_ != nullptr);
432 this->state_->NotifyRunnerResults(measure_candidates, results);
433 }

436std::vector<Schedule> EvolutionarySearchNode::State::PickBestFromDatabase(int num) {
437 auto _ = Profiler::TimedScope("EvoSearch/PickBestFromDatabase");
438 std::vector<tir::Trace> measured_traces;
439 measured_traces.reserve(num);
440 Array<TuningRecord> top_records = this->database_->GetTopK(this->token_, num);
441 for (TuningRecord record : top_records) {
442 measured_traces.push_back(record->trace);
443 }
444 int actual_num = measured_traces.size();
445 ThreadedTraceApply pp(self->context_->postprocs);
446 std::vector<Schedule> results(actual_num, Schedule{nullptr});
447 auto f_proc_measured = [this, &measured_traces, &results, &pp](int thread_id,
448 int trace_id) -> void {
449 PerThreadData& data = this->per_thread_data_.at(thread_id);
450 TRandState* rand_state = &data.rand_state;
451 const IRModule& mod = data.mod;
452 tir::Trace trace = measured_traces.at(trace_id);
453 Schedule& result = results.at(trace_id);
454 ICHECK(!result.defined());
455 if (Optional<Schedule> sch = pp.Apply(mod, trace, rand_state)) {
456 result = sch.value();
457 } else {
458 LOG(FATAL) << "ValueError: Cannot postprocess the trace:\n" << trace;
459 throw;
460 }
461 };
462 support::parallel_for_dynamic(0, actual_num, self->context_->num_threads, f_proc_measured);
463 return results;
464}

466std::vector<Schedule> EvolutionarySearchNode::State::SampleInitPopulation(int num) {
467 auto _ = Profiler::TimedScope("EvoSearch/SampleInitPopulation");
468 ThreadedTraceApply pp(self->context_->postprocs);
469 std::vector<Schedule> out_schs;
470 while (static_cast<int>(out_schs.size()) < self->init_min_unmeasured) {
471 std::vector<Schedule> results(num, Schedule{nullptr});
472 auto f_proc_unmeasured = [this, &results, &pp](int thread_id, int trace_id) -> void {
473 PerThreadData& data = this->per_thread_data_.at(thread_id);
474 TRandState* rand_state = &data.rand_state;
475 const IRModule& mod = data.mod;
476 Schedule& result = results.at(trace_id);
477 ICHECK(!result.defined());
478 int design_space_index = tir::SampleInt(rand_state, 0, design_spaces.size());
479 tir::Trace trace(design_spaces[design_space_index]->insts, {});
480 if (Optional<Schedule> sch = pp.Apply(mod, trace, rand_state)) {
481 result = sch.value();
482 }
483 };
484 support::parallel_for_dynamic(0, num, self->context_->num_threads, f_proc_unmeasured);
485 for (int i = 0; i < num; i++) {
486 if (results[i].defined()) {
487 out_schs.push_back(results[i]);
488 }
489 }
490 TVM_PY_LOG(INFO, self->context_->logging_func) << "Sample-Init-Population summary:\n"
491 << pp.SummarizeFailures();
492 }
493 return out_schs;
494}

496std::vector<Schedule> EvolutionarySearchNode::State::EvolveWithCostModel(
497 std::vector<Schedule> population, int num) {
498 IRModuleSet exists;
499 {
500 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Misc/CopyMeasuredWorkloads");
501 ICHECK_GT(num, 0);
502 // The heap to record best schedule, we do not consider schedules that are already measured
503 exists = this->measured_workloads_;
504 }
505 SizedHeap heap(num);
506 for (int iter = 0;; ++iter) {
507 // Predict normalized score with the cost model,
508 std::vector<double> scores =
509 PredictNormalizedScore(population, GetRef<TuneContext>(self->context_), this->cost_model_);
510
511 {
512 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Misc");
513 ICHECK_EQ(scores.size(), population.size());
514 for (int i = 0, n = population.size(); i < n; ++i) {
515 Schedule sch = population.at(i);
516 IRModule mod = sch->mod();
517 size_t shash = StructuralHash()(mod);
518 double score = scores.at(i);
519 if (!exists.Has(mod, shash)) {
520 exists.Add(mod, shash);
521 heap.Push(sch, score);
522 }
523 }
524 // Discontinue once it reaches end of search
525 if (iter == self->genetic_num_iters) {
526 break;
527 }
528 // Set threaded samplers, with probability from predicated normalized throughput
529 for (PerThreadData& data : this->per_thread_data_) {
530 data.Set(scores, self->genetic_mutate_prob, self->context_->mutator_probs);
531 }
532 }
533 {
534 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Mutation");
535 ThreadedTraceApply pp(self->context_->postprocs);
536 ConcurrentBitmask cbmask(self->population_size);
537 std::vector<Schedule> next_population(self->population_size, Schedule{nullptr});
538 // The worker function
539 auto f_find_candidate = [&cbmask, &population, &next_population, &pp, this](int thread_id,
540 int trace_id) {
541 // Prepare samplers
542 PerThreadData& data = this->per_thread_data_.at(thread_id);
543 TRandState* rand_state = &data.rand_state;
544 const IRModule& mod = data.mod;
545 std::function<int()>& trace_sampler = data.trace_sampler;
546 std::function<Optional<Mutator>()>& mutator_sampler = data.mutator_sampler;
547 Schedule& result = next_population.at(trace_id);
548 int sampled_trace_id = -1;
549 // Loop until success
550 for (int fail_count = 0; fail_count <= self->genetic_max_fail_count; ++fail_count) {
551 sampled_trace_id = trace_sampler();
552 tir::Trace trace = population.at(sampled_trace_id)->trace().value();
553 if (Optional<Mutator> opt_mutator = mutator_sampler()) {
554 // Decision: mutate
555 Mutator mutator = opt_mutator.value();
556 if (Optional<tir::Trace> new_trace = mutator->Apply(trace, rand_state)) {
557 if (Optional<Schedule> sch = pp.Apply(mod, new_trace.value(), rand_state)) {
558 // note that sch's trace is different from new_trace
559 // because it contains post-processing information
560 result = sch.value();
561 break;
562 }
563 }
564 } else if (cbmask.QueryAndMark(sampled_trace_id)) {
565 // Decision: do not mutate
566 break;
567 }
568 }
569 // if retry count exceeds the limit, reuse an old sample
570 if (!result.defined()) {
571 result = population.at(sampled_trace_id);
572 }
573 };
574 support::parallel_for_dynamic(0, self->population_size, self->context_->num_threads,
575 f_find_candidate);
576
577 population.swap(next_population);
578 TVM_PY_LOG(INFO, self->context_->logging_func)
579 << "Evolve iter #" << iter << " done. Summary:\n"
580 << pp.SummarizeFailures();
581 }
582 }
583 // Return the best states from the heap, sorting from higher score to lower ones
584 {
585 auto _ = Profiler::TimedScope("EvoSearch/Evolve/Misc");
586 std::sort(heap.heap.begin(), heap.heap.end());
587 std::vector<Schedule> results;
588 results.reserve(num);
589 for (const SizedHeap::Item& item : heap.heap) {
590 results.push_back(item.sch);
591 }
592
593 constexpr int kNumScoresPerLine = 16;
594 std::ostringstream os;
595 int n = heap.heap.size();
596 for (int st = 0; st < n; st += kNumScoresPerLine) {
597 os << std::endl;
598 int ed = std::min(st + kNumScoresPerLine, n);
599 os << "[" << (st + 1) << " : " << ed << "]:\t";
600 for (int i = st; i < ed; ++i) {
601 if (i != st) {
602 os << " ";
603 }
604 os << std::fixed << std::setprecision(4) << heap.heap.at(i).score;
605 }
606 }
607 TVM_PY_LOG(INFO, self->context_->logging_func)
608 << "Scores of the best " << n << " candidates:" << os.str();
609 return results;
610 }
611}

613std::vector<Schedule> EvolutionarySearchNode::State::PickWithEpsGreedy(
614 const std::vector<Schedule>& unmeasured, const std::vector<Schedule>& bests, int num) {
615 auto _ = Profiler::TimedScope("EvoSearch/PickWithEpsGreedy");
616 int num_rands = num * self->eps_greedy;
617 int num_bests = num - num_rands;
618 std::vector<int> rands =
619 tir::SampleWithoutReplacement(&self->rand_state_, unmeasured.size(), unmeasured.size());
620 std::vector<Schedule> results;
621 results.reserve(num);
622 IRModuleSet& measured_workloads = this->measured_workloads_;
623 for (int i = 0, i_bests = 0, i_rands = 0; i < num; ++i) {
624 bool has_best = i_bests < static_cast<int>(bests.size());
625 bool has_rand = i_rands < static_cast<int>(rands.size());
626 // Pick a schedule
627 Schedule sch{nullptr};
628 // If needs `bests`, then prefer `bests`
629 if (i < num_bests) {
630 if (has_best) {
631 sch = bests[i_bests++];
632 } else if (has_rand) {
633 sch = unmeasured[rands[i_rands++]];
634 } else {
635 break;
636 }
637 } else {
638 // Else prefer `rands`
639 if (has_rand) {
640 sch = unmeasured[rands[i_rands++]];
641 } else if (has_best) {
642 sch = bests[i_bests++];
643 } else {
644 break;
645 }
646 }
647 IRModule mod = sch->mod();
648 size_t shash = StructuralHash()(mod);
649 if (!measured_workloads.Has(mod, shash)) {
650 measured_workloads.Add(mod, shash);
651 results.push_back(sch);
652 }
653 }
654 return results;
655}

657Optional<Array<MeasureCandidate>> EvolutionarySearchNode::State::GenerateMeasureCandidates() {
658 if (st >= self->max_trials_per_task) {
659 return NullOpt;
660 }
661 int sample_num = self->num_trials_per_iter;
662 if (ed > self->max_trials_per_task) {
663 sample_num = self->max_trials_per_task - st;
664 ed = self->max_trials_per_task;
665 }
666 ICHECK_LT(st, ed);
667 int pop = self->population_size;
668 std::vector<Schedule> inits;
669 inits.reserve(pop);
670
671 TVM_PY_LOG(INFO, self->context_->logging_func) << "Generating candidates......";
672 std::vector<Schedule> measured = PickBestFromDatabase(pop * self->init_measured_ratio);
673 TVM_PY_LOG(INFO, self->context_->logging_func)
674 << "Picked top " << measured.size() << " candidate(s) from database";
675 std::vector<Schedule> unmeasured = SampleInitPopulation(pop - measured.size());
676 TVM_PY_LOG(INFO, self->context_->logging_func)
677 << "Sampled " << unmeasured.size() << " candidate(s)";
678 inits.insert(inits.end(), measured.begin(), measured.end());
679 inits.insert(inits.end(), unmeasured.begin(), unmeasured.end());
680 std::vector<Schedule> bests = EvolveWithCostModel(inits, sample_num);
681 TVM_PY_LOG(INFO, self->context_->logging_func)
682 << "Got " << bests.size() << " candidate(s) with evolutionary search";
683 std::vector<Schedule> picks = PickWithEpsGreedy(unmeasured, bests, sample_num);
684 TVM_PY_LOG(INFO, self->context_->logging_func)
685 << "Sending " << picks.size() << " candidates(s) for measurement";
686 if (picks.empty()) {
687 ++this->num_empty_iters;
688 if (this->num_empty_iters >= self->num_empty_iters_before_early_stop) {
689 return NullOpt;
690 }
691 }
692 return AssembleCandidates(picks);
693}

738Array<Schedule> EvolutionarySearchEvolveWithCostModel(EvolutionarySearch self,
739 Array<Schedule> population, int num) {
740 Array<Schedule> result;
741 std::vector<Schedule> population_vec =
742 std::vector<Schedule>(population.begin(), population.end());
743 std::vector<Schedule> schs = self->state_->EvolveWithCostModel(population_vec, num);
744 for (Schedule sch : schs) {

746 size_t shash = StructuralHash()(mod);
747 if (!self->state_->measured_workloads_.Has(mod, shash)) {
748 self->state_->measured_workloads_.Add(mod, shash);
749 result.push_back(sch);
750 }
751 }
752 return result;
