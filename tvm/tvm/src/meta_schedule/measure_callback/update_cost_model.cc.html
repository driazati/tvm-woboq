19#include "../utils.h"

21namespace tvm {
22namespace meta_schedule {
23
24class UpdateCostModelNode : public MeasureCallbackNode {
25 public:
26 void Apply(const TaskScheduler& task_scheduler, int task_id,
27 const Array<MeasureCandidate>& measure_candidates,
28 const Array<BuilderResult>& builder_results,
29 const Array<RunnerResult>& runner_results) final {
30 auto _ = Profiler::TimedScope("MeasureCallback/UpdateCostModel");
31 TuneContext task = task_scheduler->tasks[task_id];
32 ICHECK(task_scheduler->cost_model.defined())
33 << "Cost model must be defined for the task scheduler!";
34 ICHECK(task->measure_candidates.defined()) << "Task's measure candidates must be present!";
35 CostModel cost_model = task_scheduler->cost_model.value();
36 ICHECK_EQ(measure_candidates.size(), builder_results.size());
37 ICHECK_EQ(runner_results.size(), builder_results.size());
38 int n = builder_results.size();
39 Array<MeasureCandidate> pruned_candidate;
40 Array<RunnerResult> pruned_runner_result;
41 pruned_candidate.reserve(n);
42 pruned_runner_result.reserve(n);
43 for (int i = 0; i < n; i++) {
44 if (!builder_results[i]->error_msg.defined()) {
45 pruned_candidate.push_back(measure_candidates[i]);
46 pruned_runner_result.push_back(runner_results[i]);
47 }
48 }
49 cost_model->Update(task, pruned_candidate, pruned_runner_result);
50 }
51
52 static constexpr const char* _type_key = "meta_schedule.UpdateCostModel";
53 TVM_DECLARE_FINAL_OBJECT_INFO(UpdateCostModelNode, MeasureCallbackNode);
54};
55
56MeasureCallback MeasureCallback::UpdateCostModel() {
57 ObjectPtr<UpdateCostModelNode> n = make_object<UpdateCostModelNode>();
58 return MeasureCallback(n);
59}
60
61TVM_REGISTER_NODE_TYPE(UpdateCostModelNode);
62TVM_REGISTER_GLOBAL("meta_schedule.MeasureCallbackUpdateCostModel")
63 .set_body_typed(MeasureCallback::UpdateCostModel);
