26constexpr const double kMaxTime = 1e10;
27
28std::string GetTaskName(const TuneContext& task, int task_id) {
29 std::ostringstream os;
30 os << "Task #" << task_id << ": " << task->task_name;
31 return os.str();
32}
33
34struct TaskInfo {
35 std::string name;
36 double flop = 0.0;
37 int trials = -1;
38 int best_round = -1;
39 double best_ms = kMaxTime;
40 double best_gflops = 0.0;
41 int error_count = 0;
42 PackedFunc logging_func;
43
44 explicit TaskInfo(const String& name, PackedFunc logging_func)
45 : name(name), logging_func(logging_func) {}
46
47 void Update(double run_ms) {
48 ++trials;
49 if (run_ms < best_ms) {
50 best_ms = run_ms;
51 best_round = trials;
52 best_gflops = flop / run_ms / 1e6;
53 }
54 TVM_PY_LOG(INFO, logging_func) << "[" << name << "] Trial #" << trials //
55 << std::fixed << std::setprecision(4) //
56 << ": GFLOPs: " << (flop / run_ms / 1e6) //
57 << ". Time: " << run_ms << " ms" //
58 << ". Best GFLOPs: " << best_gflops;
59 }
60
61 void UpdateError(std::string err, const MeasureCandidate& candidate) {
62 static const auto* f_proc = runtime::Registry::Get("meta_schedule._process_error_message");
63 ICHECK(f_proc != nullptr);
64 err = (*f_proc)(err).operator std::string();
65 ++error_count;
66 ++trials;
67 TVM_PY_LOG(INFO, logging_func)
68 << "[" << name << "] Trial #" << trials //
69 << std::fixed << std::setprecision(4) //
70 << ": Error in building: " << err << "\n"
71 << tir::AsTVMScript(candidate->sch->mod()) << "\n"
72 << Concat(candidate->sch->trace().value()->AsPython(false), "\n");
73 }
74};
75
76class EchoStatisticsNode : public MeasureCallbackNode {
77 public:
78 void Apply(const TaskScheduler& task_scheduler, int task_id,
79 const Array<MeasureCandidate>& measure_candidates,
80 const Array<BuilderResult>& builder_results,
81 const Array<RunnerResult>& runner_results) final {
82 auto _ = Profiler::TimedScope("MeasureCallback/EchoStatistics");
83 if (this->task_info.empty()) {
84 SetupTaskInfo(task_scheduler->tasks);
85 }
86 ICHECK_EQ(measure_candidates.size(), builder_results.size());
87 ICHECK_EQ(measure_candidates.size(), runner_results.size());
88 int n = measure_candidates.size();
89 TuneContext task = task_scheduler->tasks[task_id];
90 TaskInfo& info = this->task_info[task_id];
91 std::string task_name = GetTaskName(task, task_id);
92 for (int i = 0; i < n; ++i) {
93 MeasureCandidate candidate = measure_candidates[i];
94 BuilderResult builder_result = builder_results[i];
95 RunnerResult runner_result = runner_results[i];
96 if (Optional<String> err = builder_result->error_msg) {
97 info.UpdateError(err.value(), candidate);
98 } else if (Optional<String> err = runner_result->error_msg) {
99 info.UpdateError(err.value(), candidate);
100 } else {
101 ICHECK(runner_result->run_secs.defined());
102 info.Update(GetRunMsMedian(runner_result));
103 }
104 }
105 }
106
107 void SetupTaskInfo(const Array<TuneContext>& tasks) {
108 task_info.reserve(tasks.size());
109 int task_id = 0;
110 for (const TuneContext& task : tasks) {
111 task_info.push_back(TaskInfo(GetTaskName(task, task_id), task->logging_func));
112 TaskInfo& info = task_info.back();
113 info.flop = tir::EstimateTIRFlops(task->mod.value());
114 ++task_id;
115 }
116 }
117
118 std::vector<TaskInfo> task_info;
119
120 static constexpr const char* _type_key = "meta_schedule.EchoStatistics";
121 TVM_DECLARE_FINAL_OBJECT_INFO(EchoStatisticsNode, MeasureCallbackNode);
122};
123
124MeasureCallback MeasureCallback::EchoStatistics() {
125 ObjectPtr<EchoStatisticsNode> n = make_object<EchoStatisticsNode>();
126 return MeasureCallback(n);
127}

129TVM_REGISTER_NODE_TYPE(EchoStatisticsNode);
130TVM_REGISTER_GLOBAL("meta_schedule.MeasureCallbackEchoStatistics")
131 .set_body_typed(MeasureCallback::EchoStatistics);
