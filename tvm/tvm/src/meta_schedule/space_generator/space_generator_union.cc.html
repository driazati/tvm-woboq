19#include "../utils.h"

21namespace tvm {
22namespace meta_schedule {
23
24/*! \brief The union of design space generators. */
25class SpaceGeneratorUnionNode : public SpaceGeneratorNode {
26 public:
27 /*! \brief The array of design space generators unioned, could be recursive. */
28 Array<SpaceGenerator> space_generators;
29
30 void VisitAttrs(tvm::AttrVisitor* v) { v->Visit("space_generators", &space_generators); }
31
32 void InitializeWithTuneContext(const TuneContext& context) final {
33 // Initialize each space generator.
34 for (const SpaceGenerator& space_generator : space_generators) {
35 space_generator->InitializeWithTuneContext(context);
36 }
37 }
38
39 Array<tir::Schedule> GenerateDesignSpace(const IRModule& mod) final {
40 Array<tir::Schedule> design_spaces;
41 for (const SpaceGenerator& space_generator : space_generators) {
42 // Generate partial design spaces from each design space generator.
43 Array<tir::Schedule> partial = space_generator->GenerateDesignSpace(mod);
44 // Merge the partial design spaces.
45 design_spaces.insert(design_spaces.end(), partial.begin(), partial.end());
46 }
47 return design_spaces;
48 }
49
50 static constexpr const char* _type_key = "meta_schedule.SpaceGeneratorUnion";
51 TVM_DECLARE_FINAL_OBJECT_INFO(SpaceGeneratorUnionNode, SpaceGeneratorNode);
52};
53
54/*!
55 * \brief Create a design space generator as union of given design space generators.
56 * \param space_generators Array of the design space generators to be unioned.
57 * \return The design space generator created.
58 */
59SpaceGenerator SpaceGenerator::SpaceGeneratorUnion(Array<SpaceGenerator> space_generators) {
60 ObjectPtr<SpaceGeneratorUnionNode> n = make_object<SpaceGeneratorUnionNode>();
61 n->space_generators = std::move(space_generators);
62 return SpaceGenerator(n);
63}

65TVM_REGISTER_NODE_TYPE(SpaceGeneratorUnionNode);
66TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorSpaceGeneratorUnion")
67 .set_body_typed(SpaceGenerator::SpaceGeneratorUnion);
