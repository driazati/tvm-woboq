24/*! \brief Collecting all the blocks */
25class BlockCollector : public tir::StmtVisitor {
26 public:
27 static Array<tir::BlockRV> Collect(const tir::Schedule& sch,
28 const runtime::PackedFunc f_block_filter = nullptr) { //
29 return BlockCollector(sch, f_block_filter).Run();
30 }
31
32 private:
33 /*! \brief Entry point */
34 Array<tir::BlockRV> Run() {
35 std::vector<tir::BlockRV> results;
36 for (const auto& kv : sch_->mod()->functions) {
37 const GlobalVar& gv = kv.first; // `gv->name_hint` is the name of the function
38 const BaseFunc& base_func = kv.second; // this can be PrimFunc or relay::Function
39 if (const auto* func = base_func.as<tir::PrimFuncNode>()) {
40 func_name_ = gv->name_hint;
41 block_names_.clear();
42 blocks_to_collect_.clear();
43 VisitStmt(func->body);
44 for (const String& name : blocks_to_collect_) {
45 results.push_back(sch_->GetBlock(name, func_name_));
46 }
47 }
48 }
49 return results;
50 }
51 /*! \brief Constructor */
52 explicit BlockCollector(const tir::Schedule& sch,
53 const runtime::PackedFunc f_block_filter = nullptr)
54 : sch_(sch), f_block_filter_(f_block_filter) {}
55 /*! \brief Override the Stmt visiting behaviour */
56 void VisitStmt_(const tir::BlockNode* block) override {
57 tir::StmtVisitor::VisitStmt_(block);
58 CHECK(block_names_.count(block->name_hint) == 0)
59 << "Duplicated block name " << block->name_hint << " in function " << func_name_
60 << " not supported!";
61 block_names_.insert(block->name_hint);
62
63 // If filter function is provided, use it to selectively collect blocks.
64 // Otherwise collect all blocks.
65 Bool collect_block = Bool(true);
66 if (f_block_filter_ != nullptr) {
67 collect_block = f_block_filter_(GetRef<tir::Block>(block));
68 }
69 if (collect_block) {
70 blocks_to_collect_.push_back(block->name_hint);
71 }
72 }
73
74 /*! \brief The schedule to be collected */
75 const tir::Schedule& sch_;
76 /*! \brief An optional packed func that allows only certain blocks to be collected. */
77 const runtime::PackedFunc f_block_filter_;
78 /*! \brief The set of func name and block name pair */
79 std::unordered_set<String> block_names_;
80 /* \brief The list of blocks to collect in order */
81 Array<String> blocks_to_collect_;
82 /*! \brief Name of the current PrimFunc */
83 String func_name_;
84};

86/*!
87 * \brief Design Space Generator that generates design spaces by applying schedule rules to blocks
88 * in post-DFS order.
89 * */
90class PostOrderApplyNode : public SpaceGeneratorNode {
91 public:
92 /*! \brief The random state. -1 means using random number. */
93 TRandState rand_state_ = -1;
94 /*! \brief The schedule rules to be applied in order. */
95 Array<ScheduleRule> sch_rules_{nullptr};
96 /*! \brief The logging function to use. */
97 PackedFunc logging_func;
98 /*! \brief Optional block names to target. If not specified all blocks will have spaces generated.
99 */
100 runtime::PackedFunc f_block_filter_ = nullptr;
101
102 void VisitAttrs(tvm::AttrVisitor* v) {
103 // `rand_state_` is not visited
104 // `sch_rules_` is not visited
105 }
106
107 void InitializeWithTuneContext(const TuneContext& context) final {
108 this->rand_state_ = ForkSeed(&context->rand_state);
109 CHECK(context->sch_rules.defined())
110 << "ValueError: Schedules rules not given in PostOrderApply!";
111 this->sch_rules_ = context->sch_rules;
112 this->logging_func = context->logging_func;
113 }
114
115 Array<tir::Schedule> GenerateDesignSpace(const IRModule& mod_) final {
116 using ScheduleAndUnvisitedBlocks = std::pair<tir::Schedule, Array<tir::BlockRV>>;
117 tir::Schedule sch = tir::Schedule::Traced(
118 /*mod=*/mod_,
119 /*rand_state=*/ForkSeed(&this->rand_state_),
120 /*debug_mode=*/0,
121 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kDetail);
122
123 std::vector<ScheduleAndUnvisitedBlocks> stack;
124 Array<tir::Schedule> result{sch};
125 // Enumerate the schedule rules first because you can
126 // always concat multiple schedule rules as one
127 Array<tir::BlockRV> all_blocks = BlockCollector::Collect(sch, f_block_filter_);
128 Array<Optional<ScheduleRule>> rules{NullOpt};
129 rules.insert(rules.end(), sch_rules_.begin(), sch_rules_.end());
130 for (Optional<ScheduleRule> sch_rule : rules) {
131 if (sch_rule.defined()) {
132 for (const tir::Schedule& sch : result) {
133 stack.emplace_back(sch, all_blocks);
134 }
135 } else {
136 for (const tir::Schedule& sch : result) {
137 stack.emplace_back(sch, Array<tir::BlockRV>{all_blocks.rbegin(), all_blocks.rend()});
138 }
139 }
140 result.clear();
141 while (!stack.empty()) {
142 // get the stack.top()
143 tir::Schedule sch;
144 Array<tir::BlockRV> blocks;
145 std::tie(sch, blocks) = stack.back();
146 stack.pop_back();
147 // if all blocks are visited
148 if (blocks.empty()) {
149 result.push_back(sch);
150 continue;
151 }
152 // otherwise, get the last block that is not visited
153 tir::BlockRV block_rv = blocks.back();
154 blocks.pop_back();
155 if (!sch->HasBlock(block_rv)) {
156 stack.emplace_back(sch, blocks);
157 continue;
158 }
159
160 Optional<String> ann = tir::GetAnn<String>(sch->GetSRef(block_rv), "schedule_rule");
161 const runtime::PackedFunc* custom_schedule_fn =
162 ann.defined() ? runtime::Registry::Get(ann.value()) : nullptr;
163 const bool has_schedule_rule = custom_schedule_fn != nullptr;
164
165 if (ann.defined() && ann.value() != "None" && !has_schedule_rule) {
166 LOG(WARNING) << "Custom schedule rule not found, ignoring schedule_rule annotation: "
167 << ann.value();
168 }
169
170 if ((has_schedule_rule && sch_rule.defined()) ||
171 (!has_schedule_rule && !sch_rule.defined()) ||
172 (ann.defined() && ann.value() == "None")) {
173 stack.emplace_back(sch, blocks);
174 continue;
175 }
176
177 Array<tir::Schedule> applied{nullptr};
178 if (sch_rule.defined()) {
179 applied = sch_rule.value()->Apply(sch, /*block=*/block_rv);
180 } else {
181 ICHECK(custom_schedule_fn)
182 << "ValueError: Custom schedule rule not found: " << ann.value();
183 applied = (*custom_schedule_fn)(sch, block_rv);
184 }
185
186 for (const tir::Schedule& sch : applied) {
187 stack.emplace_back(sch, blocks);
188 }
189 }
190 }
191 return result;
192 }
193 static constexpr const char* _type_key = "meta_schedule.PostOrderApply";
194 TVM_DECLARE_FINAL_OBJECT_INFO(PostOrderApplyNode, SpaceGeneratorNode);
195};
196
197SpaceGenerator SpaceGenerator::PostOrderApply(runtime::PackedFunc f_block_filter) {
198 ObjectPtr<PostOrderApplyNode> n = make_object<PostOrderApplyNode>();
199 n->f_block_filter_ = f_block_filter;
200 return SpaceGenerator(n);
201}
