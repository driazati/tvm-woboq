24String GetRuleKindFromTarget(const Target& target) {
25 if (target->kind->name == "llvm") {
26 return "llvm";
27 }
28 if (target->kind->name == "hexagon") {
29 return "hexagon";
30 }
31 if (target->kind->name == "cuda") {
32 if (Optional<String> opt_sm = target->GetAttr<String>("arch")) {
33 std::string sm = opt_sm.value();
34 if (support::StartsWith(sm, "sm_")) {
35 sm = sm.substr(3);
36 try {
37 if (std::stoi(sm) >= 75) {
38 return "cuda_tensorcore";
39 }
40 } catch (const std::invalid_argument& e) {
41 LOG(WARNING) << "ValueError: Unable to parse `target.arch`: " << sm
42 << ". Details: " << e.what();
43 }
44 }
45 }
46 return "cuda";
47 }
48
49 if (IsGPUTarget(target->kind->name)) {
50 return "cuda";
51 }
52
53 LOG(FATAL) << "Unsupported target: " << target;
54 throw;
55}

57void SpaceGeneratorNode::InitializeWithTuneContext(const TuneContext& context) {
58 if (context->target.defined() && //
59 !(sch_rules.defined() && //
60 postprocs.defined() && //
61 mutator_probs.defined())) {
62 String kind = GetRuleKindFromTarget(context->target.value());
63 Array<ScheduleRule> default_sch_rules;
64 Array<Postproc> default_postprocs;
65 Map<Mutator, FloatImm> default_mutator_probs;
66 if (kind == "llvm") {
67 default_sch_rules = ScheduleRule::DefaultLLVM();
68 default_postprocs = Postproc::DefaultLLVM();
69 default_mutator_probs = Mutator::DefaultLLVM();
70 } else if (kind == "cuda") {
71 default_sch_rules = ScheduleRule::DefaultCUDA();
72 default_postprocs = Postproc::DefaultCUDA();
73 default_mutator_probs = Mutator::DefaultCUDA();
74 } else if (kind == "cuda_tensorcore") {
75 default_sch_rules = ScheduleRule::DefaultCUDATensorCore();
76 default_postprocs = Postproc::DefaultCUDATensorCore();
77 default_mutator_probs = Mutator::DefaultCUDATensorCore();
78 } else if (kind == "hexagon") {
79 default_sch_rules = ScheduleRule::DefaultHexagon();
80 default_postprocs = Postproc::DefaultHexagon();
81 default_mutator_probs = Mutator::DefaultHexagon();
82 } else {
83 LOG(FATAL) << "Unsupported kind: " << kind;
84 throw;
85 }
86 if (!sch_rules.defined()) {
87 sch_rules = default_sch_rules;
88 }
89 if (!postprocs.defined()) {
90 postprocs = default_postprocs;
91 }
92 if (!mutator_probs.defined()) {
93 mutator_probs = default_mutator_probs;
94 }
95 }
96 if (sch_rules.defined()) {
97 for (ScheduleRule i : sch_rules.value()) {
98 i->InitializeWithTuneContext(context);
99 }
100 }
101 if (postprocs.defined()) {
102 for (Postproc i : postprocs.value()) {
103 i->InitializeWithTuneContext(context);
104 }
105 }
106 if (mutator_probs.defined()) {
107 for (const auto& kv : mutator_probs.value()) {
108 Mutator mutator = kv.first;
109 mutator->InitializeWithTuneContext(context);
110 }
111 }
112}

114void PySpaceGeneratorNode::InitializeWithTuneContext(const TuneContext& context) {
115 ICHECK(f_initialize_with_tune_context != nullptr)
116 << "PySpaceGenerator's InitializeWithTuneContext method not implemented!";
117 f_initialize_with_tune_context(context);
118}

120Array<tir::Schedule> PySpaceGeneratorNode::GenerateDesignSpace(const IRModule& mod) {
121 ICHECK(f_generate_design_space != nullptr)
122 << "PySpaceGenerator's GenerateDesignSpace method not implemented!";
123 return f_generate_design_space(mod);
124}

126SpaceGenerator PySpaceGeneratorNode::Clone() const {
127 ICHECK(f_clone != nullptr) << "PySpaceGenerator's Clone method not implemented!";
128 return f_clone();
129}

131SpaceGenerator SpaceGenerator::PySpaceGenerator(
132 Optional<Array<ScheduleRule>> sch_rules, Optional<Array<Postproc>> postprocs,
133 Optional<Map<Mutator, FloatImm>> mutator_probs,
134 FInitializeWithTuneContext f_initialize_with_tune_context,
135 FGenerateDesignSpace f_generate_design_space, FClone f_clone) {
136 ObjectPtr<PySpaceGeneratorNode> n = make_object<PySpaceGeneratorNode>();
137 n->sch_rules = sch_rules;
138 n->postprocs = postprocs;
139 n->mutator_probs = mutator_probs;
140 n->f_initialize_with_tune_context = std::move(f_initialize_with_tune_context);
141 n->f_generate_design_space = std::move(f_generate_design_space);
142 n->f_clone = std::move(f_clone);
143 return SpaceGenerator(n);
144}
