21namespace tvm {
22namespace meta_schedule {
23
24void PySpaceGeneratorNode::InitializeWithTuneContext(const TuneContext& context) {
25 ICHECK(f_initialize_with_tune_context != nullptr)
26 << "PySpaceGenerator's InitializeWithTuneContext method not implemented!";
27 f_initialize_with_tune_context(context);
28}
29
30Array<tir::Schedule> PySpaceGeneratorNode::GenerateDesignSpace(const IRModule& mod) {
31 ICHECK(f_generate_design_space != nullptr)
32 << "PySpaceGenerator's GenerateDesignSpace method not implemented!";
33 return f_generate_design_space(mod);
34}
35
36SpaceGenerator PySpaceGeneratorNode::Clone() const {
37 ICHECK(f_clone != nullptr) << "PySpaceGenerator's Clone method not implemented!";
38 return f_clone();
39}
40
41SpaceGenerator SpaceGenerator::PySpaceGenerator(
42 FInitializeWithTuneContext f_initialize_with_tune_context,
43 FGenerateDesignSpace f_generate_design_space, FClone f_clone) {
44 ObjectPtr<PySpaceGeneratorNode> n = make_object<PySpaceGeneratorNode>();
45 n->f_initialize_with_tune_context = std::move(f_initialize_with_tune_context);
46 n->f_generate_design_space = std::move(f_generate_design_space);
47 n->f_clone = std::move(f_clone);
48 return SpaceGenerator(n);
49}

54TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorInitializeWithTuneContext")

57 .set_body_method<SpaceGenerator>(&SpaceGeneratorNode::GenerateDesignSpace);
58TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorPySpaceGenerator")
59 .set_body_typed(SpaceGenerator::PySpaceGenerator);
60TVM_REGISTER_GLOBAL("meta_schedule.SpaceGeneratorClone")
61 .set_body_method<SpaceGenerator>(&SpaceGeneratorNode::Clone);
