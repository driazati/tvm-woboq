69/*!
70 * \brief Class to accumulate an log message on the python side. Do not use directly, instead use
71 * TVM_PY_LOG(DEBUG), TVM_PY_LOG(INFO), TVM_PY_LOG(WARNING), TVM_PY_ERROR(ERROR).
72 * \sa TVM_PY_LOG
73 * \sa TVM_PY_LOG_CLEAR_SCREEN
74 */
75class PyLogMessage {
76 public:
77 enum class Level : int32_t {
78 CLEAR = -10,
79 DEBUG = 10,
80 INFO = 20,
81 WARNING = 30,
82 ERROR = 40,
83 // FATAL not included
84 };
85
86 explicit PyLogMessage(const char* filename, int lineno, PackedFunc logger, Level logging_level)
87 : filename_(filename), lineno_(lineno), logger_(logger), logging_level_(logging_level) {}
88
89 TVM_NO_INLINE ~PyLogMessage() {
90 ICHECK(logging_level_ != Level::CLEAR)
91 << "Cannot use CLEAR as logging level in TVM_PY_LOG, please use TVM_PY_LOG_CLEAR_SCREEN.";
92 if (this->logger_ != nullptr) {
93 logger_(static_cast<int>(logging_level_), std::string(filename_), lineno_, stream_.str());
94 } else {
95 if (logging_level_ == Level::INFO) {
96 runtime::detail::LogMessage(filename_, lineno_, TVM_LOG_LEVEL_INFO).stream()
97 << stream_.str();
98 } else if (logging_level_ == Level::WARNING) {
99 runtime::detail::LogMessage(filename_, lineno_, TVM_LOG_LEVEL_WARNING).stream()
100 << stream_.str();
101 } else if (logging_level_ == Level::ERROR) {
102 runtime::detail::LogMessage(filename_, lineno_, TVM_LOG_LEVEL_ERROR).stream()
103 << stream_.str();
104 } else if (logging_level_ == Level::DEBUG) {
105 runtime::detail::LogMessage(filename_, lineno_, TVM_LOG_LEVEL_DEBUG).stream()
106 << stream_.str();
107 } else {
108 runtime::detail::LogFatal(filename_, lineno_).stream() << stream_.str();
109 }
110 }
111 }
112 std::ostringstream& stream() { return stream_; }
113
114 private:
115 const char* filename_;
116 int lineno_;
117 std::ostringstream stream_;
118 PackedFunc logger_;
119 Level logging_level_;
120};

148 * \brief A helper function to clear logging output for ipython kernel and console.
149 * \param file The file name.
150 * \param lineno The line number.

152 */
153inline void clear_logging(const char* file, int lineno, PackedFunc logging_func) {

241/*!
242 * \brief Fork a random state into another ones, i.e. PRNG splitting.
243 * The given random state is also mutated.
244 * \param rand_state The random state to be forked
245 * \param n The number of forks
246 * \return The forked random states
247 */
248inline std::vector<support::LinearCongruentialEngine::TRandState> ForkSeed(
249 support::LinearCongruentialEngine::TRandState* rand_state, int n) {
250 std::vector<support::LinearCongruentialEngine::TRandState> results;
251 results.reserve(n);
252 for (int i = 0; i < n; ++i) {
253 results.push_back(support::LinearCongruentialEngine(rand_state).ForkSeed());
254 }
255 return results;
256}

267/*!

269 * \param strs The strings to concatenate
270 * \param delim The delimiter
271 * \return The concatenated string
272 */
273inline std::string Concat(const Array<String>& strs, const std::string& delim) {
274 if (strs.empty()) {
275 return "";
276 }
277 std::ostringstream os;
278 os << strs[0];
279 for (int i = 1, n = strs.size(); i < n; ++i) {
280 os << delim << strs[i];
281 }
282 return os.str();
283}

298/*!
299 * \brief A helper data structure that replays a trace and collects failure counts
300 * for each postprocessor
301 */
302struct ThreadedTraceApply {
303 /*! \brief Constructor */
304 explicit ThreadedTraceApply(const Array<Postproc>& postprocs)
305 : n_(postprocs.size()), items_(new Item[n_]) {
306 for (int i = 0; i < n_; ++i) {
307 items_[i].postproc = postprocs[i];
308 items_[i].fail_counter = 0;
309 }
310 }
311
312 /*! \brief Destructor */
313 ~ThreadedTraceApply() { delete[] items_; }
314
315 /*!
316 * \brief Apply the trace and postprocessors to an IRModule
317 * \param mod The IRModule to be applied
318 * \param trace The trace to apply to the IRModule
319 * \param rand_state The random seed
320 * \return The schedule created, or NullOpt if any postprocessor fails
321 */
322 Optional<tir::Schedule> Apply(const IRModule& mod, const tir::Trace& trace,
323 TRandState* rand_state) {
324 tir::Schedule sch =
325 tir::Schedule::Traced(mod,
326 /*rand_state=*/ForkSeed(rand_state),
327 /*debug_mode=*/0,
328 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kNone);
329
330 trace->ApplyToSchedule(sch, /*remove_postproc=*/true);
331 sch->EnterPostproc();
332
333 for (int i = 0; i < n_; ++i) {
334 Item& item = items_[i];
335 if (!item.postproc->Apply(sch)) {
336 item.fail_counter++;
337 return NullOpt;
338 }
339 }
340 return sch;
341 }
342
343 /*! \brief Returns a string summarizing the failures on each postprocessor */
344 std::string SummarizeFailures() const {
345 std::ostringstream os;
346 for (int i = 0; i < n_; ++i) {
347 const Item& item = items_[i];
348 os << "Postproc #" << i << " [" << item.postproc //
349 << "]: " << item.fail_counter.load() << " failure(s)";
350 if (i != n_ - 1) {
351 os << "\n";
352 }
353 }
354 return os.str();
355 }
356
357 private:
358 /*! \brief A helper data structure that stores the fail count for each postprocessor. */
359 struct Item {
360 /*! \brief The postprocessor. */
361 Postproc postproc{nullptr};
362 /*! \brief The thread-safe postprocessor failure counter. */
363 std::atomic<int> fail_counter{0};
364 };
365
366 /*! \brief The number of total postprocessors. */
367 int n_;
368 /*! \brief The pointer to the list of postprocessor items. */
369 Item* items_;
370};

372/*!
373 * \brief Get the number of cores in CPU
374 * \param target The target
375 * \return The number of cores.
376 */
377inline int GetTargetNumCores(const Target& target) {
378 int num_cores = target->GetAttr<Integer>("num-cores").value_or(-1).IntValue();
379 if (num_cores == -1) {
380 static const auto* f_cpu_count = runtime::Registry::Get("meta_schedule.cpu_count");
381 ICHECK(f_cpu_count)
382 << "ValueError: Cannot find the packed function \"meta_schedule._cpu_count\"";
383 num_cores = (*f_cpu_count)(false);
384 LOG(FATAL)
385 << "Target does not have attribute \"num-cores\", physical core number must be "
386 "defined! For example, on the local machine, the target must be \"llvm -num-cores "
387 << num_cores << "\"";
388 }
389 return num_cores;
390}

392/*!
393 * \brief Get the median of the running time from RunnerResult in millisecond
394 * \param results The results from RunnerResult
395 * \return The median of the running time in millisecond
396 */
397inline double GetRunMsMedian(const RunnerResult& runner_result) {
398 Array<FloatImm> run_secs = runner_result->run_secs.value();
399 ICHECK(!run_secs.empty());
400 std::vector<double> v;
401 v.reserve(run_secs.size());
402 std::transform(run_secs.begin(), run_secs.end(), std::back_inserter(v),
403 [](const FloatImm& f) -> double { return f->value; });
404 std::sort(v.begin(), v.end());
405 int n = v.size();
406 if (n % 2 == 0) {
407 return (v[n / 2 - 1] + v[n / 2]) * 0.5 * 1000.0;
408 } else {
409 return v[n / 2] * 1000.0;
410 }
411}

413/*!
414 * \brief Convert the given object to an array of floating point numbers
415 * \param obj The object to be converted
416 * \return The array of floating point numbers
417 */
418inline Array<FloatImm> AsFloatArray(const ObjectRef& obj) {
419 const ArrayNode* arr = obj.as<ArrayNode>();
420 ICHECK(arr) << "TypeError: Expect an array, but gets: " << obj->GetTypeKey();
421 Array<FloatImm> results;
422 results.reserve(arr->size());
423 for (const ObjectRef& elem : *arr) {
424 if (const auto* int_imm = elem.as<IntImmNode>()) {
425 results.push_back(FloatImm(DataType::Float(32), int_imm->value));
426 } else if (const auto* float_imm = elem.as<FloatImmNode>()) {
427 results.push_back(FloatImm(DataType::Float(32), float_imm->value));
428 } else {
429 LOG(FATAL) << "TypeError: Expect an array of float or int, but gets: " << elem->GetTypeKey();
430 }
431 }
432 return results;
433}

455/*! \brief The struct defining comparison function of sorting by mean run seconds. */
456struct SortTuningRecordByMeanRunSecs {
457 static const constexpr double kMaxMeanTime = 1e10;
458
459 static double Mean(const Array<FloatImm>& a) {
460 if (a.empty()) {
461 return kMaxMeanTime;
462 }
463 double sum = 0.0;
464 for (const FloatImm& i : a) {
465 sum += i->value;
466 }
467 return sum / a.size();
468 }
469
470 bool operator()(const TuningRecord& a, const TuningRecord& b) const {
471 double a_time = Mean(a->run_secs.value_or({}));
472 double b_time = Mean(b->run_secs.value_or({}));
473 return a_time < b_time;
474 }
475};

477/*!
478 * \brief The helper function to clone schedule rules, postprocessors, and mutators.
479 * \param src The source space generator.
480 * \param dst The destination space generator.
481 */
482inline void CloneRules(const SpaceGeneratorNode* src, SpaceGeneratorNode* dst) {
483 if (src->sch_rules.defined()) {
484 Array<ScheduleRule> original = src->sch_rules.value();
485 Array<ScheduleRule> sch_rules;
486 sch_rules.reserve(original.size());
487 for (const ScheduleRule& sch_rule : original) {
488 sch_rules.push_back(sch_rule->Clone());
489 }
490 dst->sch_rules = std::move(sch_rules);
491 }
492 if (src->postprocs.defined()) {
493 Array<Postproc> original = src->postprocs.value();
494 Array<Postproc> postprocs;
495 postprocs.reserve(original.size());
496 for (const Postproc& postproc : original) {
497 postprocs.push_back(postproc->Clone());
498 }
499 dst->postprocs = std::move(postprocs);
500 }
501 if (src->mutator_probs.defined()) {
502 Map<Mutator, FloatImm> original = src->mutator_probs.value();
503 Map<Mutator, FloatImm> mutator_probs;
504 for (const auto& kv : original) {
505 mutator_probs.Set(kv.first->Clone(), kv.second);
506 }
507 dst->mutator_probs = std::move(mutator_probs);
508 }
509}

517/*!
518 * \brief Create an AutoInline schedule rule for the given target.
519 * \param target_name The name of the target ("llvm", "cuda", etc.)
520 * \return The AutoInline schedule rule for the given target.
521 */
522inline ScheduleRule GetDefaultAutoInline(const std::string& target_name) {
523 Array<ScheduleRule> rules{nullptr};
524 if (target_name == "llvm") {
525 rules = ScheduleRule::DefaultLLVM();
526 } else if (target_name == "hexagon") {
527 rules = ScheduleRule::DefaultHexagon();
528 } else if (IsGPUTarget(target_name)) {
529 rules = ScheduleRule::DefaultCUDA();
530 } else {
531 LOG(FATAL) << "ValueError: Unsupported target: " << target_name;
532 }
533 for (const ScheduleRule& rule : rules) {
534 if (rule->GetTypeKey() == "meta_schedule.AutoInline") {
535 return rule;
536 }
537 }
538 LOG(FATAL) << "ValueError: AutoInline rule is not found in the default rules for target: "
539 << target_name;
540 throw;
541}
