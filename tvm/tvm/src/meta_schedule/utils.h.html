67/*!
68 * \brief Class to accumulate an log message on the python side. Do not use directly, instead use
69 * TVM_PY_LOG(DEBUG), TVM_PY_LOG(INFO), TVM_PY_LOG(WARNING), TVM_PY_ERROR(ERROR).
70 */
71class PyLogMessage {
72 public:
73 enum class Level : int32_t {
74 DEBUG = 10,
75 INFO = 20,
76 WARNING = 30,
77 ERROR = 40,
78 // FATAL not included
79 };
80
81 PyLogMessage(const std::string& file, int lineno, PackedFunc logging_func, Level logging_level) {
82 this->logging_func = logging_func;
83 this->logging_level = logging_level;
84 }
85 TVM_NO_INLINE ~PyLogMessage() {
86 if (this->logging_func.defined()) {
87 logging_func(static_cast<int>(logging_level), stream_.str());
88 } else {
89 if (logging_level == Level::INFO) {
90 LOG(INFO) << stream_.str();
91 } else if (logging_level == Level::WARNING) {
92 LOG(WARNING) << stream_.str();
93 } else if (logging_level == Level::ERROR) {
94 LOG(ERROR) << stream_.str();
95 } else if (logging_level == Level::DEBUG) {
96 DLOG(INFO) << stream_.str();
97 } else {
98 LOG(FATAL) << stream_.str();
99 }
100 }
101 }
102 std::ostringstream& stream() { return stream_; }
103
104 private:
105 std::ostringstream stream_;
106 PackedFunc logging_func;
107 Level logging_level;
108};

113/*!
114 * \brief Get the base64 encoded result of a string.
115 * \param str The string to encode.
116 * \return The base64 encoded string.
117 */
118inline std::string Base64Encode(std::string str) {
119 std::string result;
120 dmlc::MemoryStringStream m_stream(&result);
121 support::Base64OutStream b64stream(&m_stream);
122 static_cast<dmlc::Stream*>(&b64stream)->Write(str);
123 b64stream.Finish();
124 return result;
125}
126
127/*!
128 * \brief Get the base64 decoded result of a string.
129 * \param str The string to decode.
130 * \return The base64 decoded string.
131 */
132inline std::string Base64Decode(std::string str) {
133 std::string result;
134 dmlc::MemoryStringStream m_stream(&str);
135 support::Base64InStream b64stream(&m_stream);
136 b64stream.InitPosition();
137 static_cast<dmlc::Stream*>(&b64stream)->Read(&result);
138 return result;
139}

177/*!
178 * \brief Fork a random state into another, i.e. PRNG splitting.
179 * The given random state is also mutated.
180 * \param rand_state The random state to be forked
181 * \return The forked random state
182 */
183inline support::LinearCongruentialEngine::TRandState ForkSeed(
184 support::LinearCongruentialEngine::TRandState* rand_state) {
185 return support::LinearCongruentialEngine(rand_state).ForkSeed();
186}
187
188/*!
189 * \brief Fork a random state into another ones, i.e. PRNG splitting.
190 * The given random state is also mutated.
191 * \param rand_state The random state to be forked
192 * \param n The number of forks
193 * \return The forked random states
194 */
195inline std::vector<support::LinearCongruentialEngine::TRandState> ForkSeed(
196 support::LinearCongruentialEngine::TRandState* rand_state, int n) {
197 std::vector<support::LinearCongruentialEngine::TRandState> results;
198 results.reserve(n);
199 for (int i = 0; i < n; ++i) {
200 results.push_back(support::LinearCongruentialEngine(rand_state).ForkSeed());
201 }
202 return results;
203}

214/*!
215 * \brief Concatenate strings
216 * \param strs The strings to concatenate
217 * \param delim The delimiter
218 * \return The concatenated string
219 */
220inline std::string Concat(const Array<String>& strs, const std::string& delim) {
221 if (strs.empty()) {
222 return "";
223 }
224 std::ostringstream os;
225 os << strs[0];
226 for (int i = 1, n = strs.size(); i < n; ++i) {
227 os << delim << strs[i];
228 }
229 return os.str();
230}

245/*!
246 * \brief A helper data structure that replays a trace and collects failure counts
247 * for each postprocessor
248 */
249struct ThreadedTraceApply {
250 /*! \brief Constructor */
251 explicit ThreadedTraceApply(const Array<Postproc>& postprocs)
252 : n_(postprocs.size()), items_(new Item[n_]) {
253 for (int i = 0; i < n_; ++i) {
254 items_[i].postproc = postprocs[i];
255 items_[i].fail_counter = 0;
256 }
257 }
258
259 /*! \brief Destructor */
260 ~ThreadedTraceApply() { delete[] items_; }
261
262 /*!
263 * \brief Apply the trace and postprocessors to an IRModule
264 * \param mod The IRModule to be applied
265 * \param trace The trace to apply to the IRModule
266 * \param rand_state The random seed
267 * \return The schedule created, or NullOpt if any postprocessor fails
268 */
269 Optional<tir::Schedule> Apply(const IRModule& mod, const tir::Trace& trace,
270 TRandState* rand_state) {
271 tir::Schedule sch =
272 tir::Schedule::Traced(mod,
273 /*rand_state=*/ForkSeed(rand_state),
274 /*debug_mode=*/0,
275 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kNone);
276
277 trace->ApplyToSchedule(sch, /*remove_postproc=*/true);
278 sch->EnterPostproc();
279
280 for (int i = 0; i < n_; ++i) {
281 Item& item = items_[i];
282 try {
283 if (!item.postproc->Apply(sch)) {
284 ++item.fail_counter;
285 return NullOpt;
286 }
287 } catch (const std::exception& e) {
288 // Used in multi-thread, only output to screen but failure summary sent to logging
289 LOG(WARNING) << "ThreadedTraceApply::Apply failed with error " << e.what();
290 return NullOpt;
291 }
292 }
293 return sch;
294 }
295
296 /*! \brief Returns a string summarizing the failures on each postprocessor */
297 std::string SummarizeFailures() const {
298 std::ostringstream os;
299 for (int i = 0; i < n_; ++i) {
300 const Item& item = items_[i];
301 os << "Postproc #" << i << " [" << item.postproc //
302 << "]: " << item.fail_counter.load() << " failure(s)";
303 if (i != n_ - 1) {
304 os << "\n";
305 }
306 }
307 return os.str();
308 }
309
310 private:
311 /*! \brief A helper data structure that stores the fail count for each postprocessor. */
312 struct Item {
313 /*! \brief The postprocessor. */
314 Postproc postproc{nullptr};
315 /*! \brief The thread-safe postprocessor failure counter. */
316 std::atomic<int> fail_counter{0};
317 };
318
319 /*! \brief The number of total postprocessors. */
320 int n_;
321 /*! \brief The pointer to the list of postprocessor items. */
322 Item* items_;
323};

325/*!
326 * \brief Get the number of cores in CPU
327 * \param target The target
328 * \return The number of cores.
329 */
330inline int GetTargetNumCores(const Target& target) {
331 int num_cores = target->GetAttr<Integer>("num-cores").value_or(-1).IntValue();
332 if (num_cores == -1) {
333 static const auto* f_cpu_count = runtime::Registry::Get("meta_schedule.cpu_count");
334 ICHECK(f_cpu_count)
335 << "ValueError: Cannot find the packed function \"meta_schedule._cpu_count\"";
336 num_cores = (*f_cpu_count)(false);
337 LOG(FATAL)
338 << "Target does not have attribute \"num-cores\", physical core number must be "
339 "defined! For example, on the local machine, the target must be \"llvm -num-cores "
340 << num_cores << "\"";
341 }
342 return num_cores;
343}

345/*!
346 * \brief Get the median of the running time from RunnerResult in millisecond
347 * \param results The results from RunnerResult
348 * \return The median of the running time in millisecond
349 */
350inline double GetRunMsMedian(const RunnerResult& runner_result) {
351 Array<FloatImm> run_secs = runner_result->run_secs.value();
352 ICHECK(!run_secs.empty());
353 std::vector<double> v;
354 v.reserve(run_secs.size());
355 std::transform(run_secs.begin(), run_secs.end(), std::back_inserter(v),
356 [](const FloatImm& f) -> double { return f->value; });
357 std::sort(v.begin(), v.end());
358 int n = v.size();
359 if (n % 2 == 0) {
360 return (v[n / 2 - 1] + v[n / 2]) * 0.5 * 1000.0;
361 } else {
362 return v[n / 2] * 1000.0;
363 }
364}

366/*!
367 * \brief Convert the given object to an array of floating point numbers
368 * \param obj The object to be converted
369 * \return The array of floating point numbers
370 */
371inline Array<FloatImm> AsFloatArray(const ObjectRef& obj) {
372 const ArrayNode* arr = obj.as<ArrayNode>();
373 ICHECK(arr) << "TypeError: Expect an array, but gets: " << obj->GetTypeKey();
374 Array<FloatImm> results;
375 results.reserve(arr->size());
376 for (const ObjectRef& elem : *arr) {
377 if (const auto* int_imm = elem.as<IntImmNode>()) {
378 results.push_back(FloatImm(DataType::Float(32), int_imm->value));
379 } else if (const auto* float_imm = elem.as<FloatImmNode>()) {
380 results.push_back(FloatImm(DataType::Float(32), float_imm->value));
381 } else {
382 LOG(FATAL) << "TypeError: Expect an array of float or int, but gets: " << elem->GetTypeKey();
383 }
384 }
385 return results;
386}

388/*!
389 * \brief Convert the given object to an array of integers
390 * \param obj The object to be converted
391 * \return The array of integers
392 */
393inline Array<Integer> AsIntArray(const ObjectRef& obj) {
394 const ArrayNode* arr = obj.as<ArrayNode>();
395 ICHECK(arr) << "TypeError: Expect an array, but gets: " << obj->GetTypeKey();
396 Array<Integer> results;
397 results.reserve(arr->size());
398 for (const ObjectRef& elem : *arr) {
399 if (const auto* int_imm = elem.as<IntImmNode>()) {
400 results.push_back(Integer(int_imm->value));
401 } else {
402 LOG(FATAL) << "TypeError: Expect an array of integers, but gets: " << elem->GetTypeKey();
403 }
404 }
405 return results;
