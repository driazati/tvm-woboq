29void JSONDumps(ObjectRef json_obj, std::ostringstream& os) {
30 if (!json_obj.defined()) {
31 os << "null";
32 } else if (const auto* int_imm = json_obj.as<IntImmNode>()) {
33 if (int_imm->dtype == DataType::Bool()) {
34 if (int_imm->value) {
35 os << "true";
36 } else {
37 os << "false";
38 }
39 } else {
40 os << int_imm->value;
41 }
42 } else if (const auto* float_imm = json_obj.as<FloatImmNode>()) {
43 os << std::setprecision(20) << float_imm->value;
44 } else if (const auto* str = json_obj.as<runtime::StringObj>()) {
45 os << '"' << support::StrEscape(str->data, str->size) << '"';
46 } else if (const auto* array = json_obj.as<runtime::ArrayNode>()) {
47 os << "[";
48 int n = array->size();
49 for (int i = 0; i < n; ++i) {
50 if (i != 0) {
51 os << ",";
52 }
53 JSONDumps(array->at(i), os);
54 }
55 os << "]";
56 } else if (const auto* dict = json_obj.as<runtime::MapNode>()) {
57 int n = dict->size();
58 std::vector<std::pair<String, ObjectRef>> key_values;
59 key_values.reserve(n);
60 for (const auto& kv : *dict) {
61 if (const auto* k = kv.first.as<StringObj>()) {
62 key_values.emplace_back(GetRef<String>(k), kv.second);
63 } else {
64 LOG(FATAL) << "TypeError: Only string keys are supported in JSON dumps, but got: "
65 << kv.first->GetTypeKey();
66 }
67 }
68 std::sort(key_values.begin(), key_values.end());
69 os << "{";
70 for (int i = 0; i < n; ++i) {
71 const auto& kv = key_values[i];
72 if (i != 0) {
73 os << ",";
74 }
75 os << '"' << support::StrEscape(kv.first->data, kv.first->size) << '"';
76 os << ":";
77 JSONDumps(kv.second, os);
78 }
79 os << "}";
80 } else {
81 LOG(FATAL) << "TypeError: Unsupported type in JSON object: " << json_obj->GetTypeKey();
82 }
83}
84
85std::string JSONDumps(ObjectRef json_obj) {
86 std::ostringstream os;
87 JSONDumps(json_obj, os);
88 return os.str();
89}

91class JSONTokenizer {
92 public:
93 enum class TokenType : int32_t {
94 kEOF = 0, // end of file
95 kNull = 1, // null
96 kTrue = 2, // true
97 kFalse = 3, // false
98 kLeftSquare = 4, // [
99 kRightSquare = 5, // ]
100 kLeftCurly = 6, // {
101 kRightCurly = 7, // }
102 kComma = 8, // ,
103 kColon = 9, // :
104 kInteger = 10, // integers
105 kFloat = 11, // floating point numbers
106 kString = 12, // string
107 };
108
109 struct Token {
110 TokenType type;
111 ObjectRef value{nullptr};
112 };
113
114 explicit JSONTokenizer(const char* st, const char* ed) : cur_(st), end_(ed) {}
115
116 Token Next() {
117 for (; cur_ != end_ && std::isspace(*cur_); ++cur_) {
118 }
119 if (cur_ == end_) return Token{TokenType::kEOF};
120 if (NextLeftSquare()) return Token{TokenType::kLeftSquare};
121 if (NextRightSquare()) return Token{TokenType::kRightSquare};
122 if (NextLeftCurly()) return Token{TokenType::kLeftCurly};
123 if (NextRightCurly()) return Token{TokenType::kRightCurly};
124 if (NextComma()) return Token{TokenType::kComma};
125 if (NextColon()) return Token{TokenType::kColon};
126 if (NextNull()) return Token{TokenType::kNull};
127 if (NextTrue()) return Token{TokenType::kTrue};
128 if (NextFalse()) return Token{TokenType::kFalse};
129 Token token;
130 if (NextString(&token)) return token;
131 if (NextNumber(&token)) return token;
132 LOG(FATAL) << "ValueError: Cannot tokenize: " << std::string(cur_, end_);
133 throw;
134 }
135
136 private:
137 bool NextLeftSquare() { return NextLiteral('['); }
138 bool NextRightSquare() { return NextLiteral(']'); }
139 bool NextLeftCurly() { return NextLiteral('{'); }
140 bool NextRightCurly() { return NextLiteral('}'); }
141 bool NextComma() { return NextLiteral(','); }
142 bool NextColon() { return NextLiteral(':'); }
143 bool NextNull() { return NextLiteral("null", 4); }
144 bool NextTrue() { return NextLiteral("true", 4); }
145 bool NextFalse() { return NextLiteral("false", 5); }
146
147 bool NextNumber(Token* token) {
148 using runtime::DataType;
149 bool is_float = false;
150 const char* st = cur_;
151 for (; cur_ != end_; ++cur_) {
152 if (std::isdigit(*cur_) || *cur_ == '+' || *cur_ == '-') {
153 continue;
154 } else if (*cur_ == '.' || *cur_ == 'e' || *cur_ == 'E') {
155 is_float = true;
156 } else {
157 break;
158 }
159 }
160 if (st == cur_) {
161 return false;
162 }
163 std::string to_parse(st, cur_);
164 if (!is_float) {
165 try {
166 *token = Token{TokenType::kInteger, IntImm(DataType::Int(64), std::stoll(to_parse))};
167 } catch (const std::invalid_argument& e) {
168 LOG(WARNING) << "ValueError: Invalid argument to std::stoll: " << to_parse
169 << ". Details: " << e.what() << ". Switching to std::stod now.";
170 is_float = true;
171 } catch (const std::out_of_range& e) {
172 LOG(WARNING) << "ValueError: Out-of-range for std::stoll: " << to_parse
173 << ". Details: " << e.what() << ". Switching to std::stod now.";
174 is_float = true;
175 }
176 }
177 if (is_float) {
178 try {
179 *token = Token{TokenType::kFloat, FloatImm(DataType::Float(64), std::stod(to_parse))};
180 } catch (const std::invalid_argument& e) {
181 LOG(INFO) << "ValueError: Invalid argument to std::stod: " << to_parse
182 << ". Details: " << e.what();
183 } catch (const std::out_of_range& e) {
184 LOG(INFO) << "ValueError: Out-of-range for std::stod: " << to_parse
185 << ". Details: " << e.what();
186 }
187 }
188 return true;
189 }
190
191 bool NextString(Token* token) {
192 if (cur_ == end_ || *cur_ != '"') return false;
193 ++cur_;
194 std::string str;
195 for (; cur_ != end_ && *cur_ != '\"'; ++cur_) {
196 if (*cur_ != '\\') {
197 str.push_back(*cur_);
198 continue;
199 }
200 ++cur_;
201 if (cur_ == end_) {
202 LOG(FATAL) << "ValueError: Unexpected end of string: \\";
203 throw;
204 }
205 switch (*cur_) {
206 case '\"':
207 str.push_back('\"');
208 break;
209 case '\\':
210 str.push_back('\\');
211 break;
212 case '/':
213 str.push_back('/');
214 break;
215 case 'b':
216 str.push_back('\b');
217 break;
218 case 'f':
219 str.push_back('\f');
220 break;
221 case 'n':
222 str.push_back('\n');
223 break;
224 case 'r':
225 str.push_back('\r');
226 break;
227 case 't':
228 str.push_back('\t');
229 break;
230 default:
231 LOG(FATAL) << "ValueError: Unsupported escape sequence: \\" << *cur_;
232 }
233 }
234 if (cur_ == end_) {
235 LOG(FATAL) << "ValueError: Unexpected end of string";
236 }
237 ++cur_;
238 *token = Token{TokenType::kString, String(str)};
239 return true;
240 }
241
242 bool NextLiteral(char c) {
243 if (cur_ != end_ && *cur_ == c) {
244 ++cur_;
245 return true;
246 }
247 return false;
248 }
249
250 bool NextLiteral(const char* str, int len) {
251 if (cur_ + len <= end_ && std::strncmp(cur_, str, len) == 0) {
252 cur_ += len;
253 return true;
254 }
255 return false;
256 }
257 /*! \brief The current pointer */
258 const char* cur_;
259 /*! \brief End of the string */
260 const char* end_;
261
262 friend class JSONParser;
263};
264
265class JSONParser {
266 public:
267 using TokenType = JSONTokenizer::TokenType;
268 using Token = JSONTokenizer::Token;
269
270 explicit JSONParser(const char* st, const char* ed) : tokenizer_(st, ed) {}
271
272 ObjectRef Get() {
273 Token token = tokenizer_.Next();
274 if (token.type == TokenType::kEOF) {
275 return ObjectRef(nullptr);
276 }
277 return ParseObject(std::move(token));
278 }
279
280 private:
281 ObjectRef ParseObject(Token token) {
282 switch (token.type) {
283 case TokenType::kNull:
284 return ObjectRef(nullptr);
285 case TokenType::kTrue:
286 return Bool(true);
287 case TokenType::kFalse:
288 return Bool(false);
289 case TokenType::kLeftSquare:
290 return ParseArray();
291 case TokenType::kLeftCurly:
292 return ParseDict();
293 case TokenType::kString:
294 case TokenType::kInteger:
295 case TokenType::kFloat:
296 return token.value;
297 case TokenType::kRightSquare:
298 LOG(FATAL) << "ValueError: Unexpected token: ]";
299 case TokenType::kRightCurly:
300 LOG(FATAL) << "ValueError: Unexpected token: }";
301 case TokenType::kComma:
302 LOG(FATAL) << "ValueError: Unexpected token: ,";
303 case TokenType::kColon:
304 LOG(FATAL) << "ValueError: Unexpected token: :";
305 case TokenType::kEOF:
306 LOG(FATAL) << "ValueError: Unexpected EOF";
307 default:
308 throw;
309 }
310 }
311
312 Array<ObjectRef> ParseArray() {
313 bool is_first = true;
314 Array<ObjectRef> results;
315 for (;;) {
316 Token token;
317 if (is_first) {
318 is_first = false;
319 token = Token{TokenType::kComma};
320 } else {
321 token = tokenizer_.Next();
322 }
323 // Three cases overall:
324 // - Case 1. 1 token: "]"
325 // - Case 2. 2 tokens: ",", "]"
326 // - Case 3. 2 tokens: ",", "obj"
327 if (token.type == TokenType::kRightSquare) { // Case 1
328 break;
329 } else if (token.type == TokenType::kComma) {
330 token = tokenizer_.Next();
331 if (token.type == TokenType::kRightSquare) { // Case 2
332 break;
333 }
334 // Case 3
335 results.push_back(ParseObject(std::move(token)));
336 continue;
337 } else {
338 LOG(FATAL) << "ValueError: Unexpected token before: " << tokenizer_.cur_;
339 }
340 }
341 return results;
342 }
343
344 Map<String, ObjectRef> ParseDict() {
345 bool is_first = true;
346 Map<String, ObjectRef> results;
347 for (;;) {
348 Token token;
349 if (is_first) {
350 is_first = false;
351 token = Token{TokenType::kComma};
352 } else {
353 token = tokenizer_.Next();
354 }
355 // Three cases overall:
356 // - Case 1. 1 token: "}"
357 // - Case 2. 2 tokens: ",", "}"
358 // - Case 3. 2 tokens: ",", "key", ":", "value"
359 if (token.type == TokenType::kRightCurly) { // Case 1
360 break;
361 } else if (token.type == TokenType::kComma) {
362 token = tokenizer_.Next();
363 if (token.type == TokenType::kRightCurly) { // Case 2
364 break;
365 }
366 // Case 3
367 ObjectRef key = ParseObject(std::move(token));
368 ICHECK(key->IsInstance<StringObj>())
369 << "ValueError: key must be a string, but gets: " << key;
370 token = tokenizer_.Next();
371 CHECK(token.type == TokenType::kColon)
372 << "ValueError: Unexpected token before: " << tokenizer_.cur_;
373 ObjectRef value = ParseObject(tokenizer_.Next());
374 results.Set(Downcast<String>(key), value);
375 continue;
376 } else {
377 LOG(FATAL) << "ValueError: Unexpected token before: " << tokenizer_.cur_;
378 }
379 }
380 return results;
381 }
382
383 JSONTokenizer tokenizer_;
384};
385
386ObjectRef JSONLoads(std::string str) {
387 const char* st = str.c_str();
388 const char* ed = st + str.length();
389 return JSONParser(st, ed).Get();
390}
