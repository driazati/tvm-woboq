19#include "../utils.h"

21namespace tvm {
22namespace meta_schedule {
23
24class MemoryDatabaseNode : public DatabaseNode {
25 public:
26 Array<TuningRecord> records;
27 Array<Workload> workloads;
28
29 void VisitAttrs(AttrVisitor* v) {
30 v->Visit("records", &records);
31 v->Visit("workloads", &workloads);
32 }
33
34 static constexpr const char* _type_key = "meta_schedule.MemoryDatabase";
35 TVM_DECLARE_FINAL_OBJECT_INFO(MemoryDatabaseNode, DatabaseNode);
36
37 public:
38 bool HasWorkload(const IRModule& mod) final {
39 for (const auto& workload : workloads) {
40 if (StructuralEqual()(workload->mod, mod)) {
41 return true;
42 }
43 }
44 return false;
45 }
46
47 Workload CommitWorkload(const IRModule& mod) {
48 for (const auto& workload : workloads) {
49 if (StructuralEqual()(workload->mod, mod)) {
50 return workload;
51 }
52 }
53 Workload workload(mod, StructuralHash()(mod));
54 workloads.push_back(workload);
55 return workload;
56 }
57
58 void CommitTuningRecord(const TuningRecord& record) { records.push_back(record); }
59
60 Array<TuningRecord> GetTopK(const Workload& workload, int top_k) {
61 std::vector<std::pair<double, TuningRecord>> results;
62 results.reserve(this->records.size());
63 for (const TuningRecord& record : records) {
64 if (!record->run_secs.defined()) {
65 continue;
66 }
67 Array<FloatImm> run_secs = record->run_secs.value();
68 if (run_secs.empty()) {
69 continue;
70 }
71 if (record->workload.same_as(workload)) {
72 double sum = 0.0;
73 for (const FloatImm& i : run_secs) {
74 sum += i->value;
75 }
76 results.emplace_back(sum / run_secs.size(), record);
77 }
78 }
79 std::sort(results.begin(), results.end());
80 auto begin = results.begin();
81 auto end = results.end();
82 if (static_cast<int>(results.size()) > top_k) {
83 end = begin + top_k;
84 }
85 Array<TuningRecord> ret;
86 ret.reserve(end - begin);
87 while (begin != end) {
88 ret.push_back(begin->second);
89 ++begin;
90 }
91 return ret;
92 }
93
94 Array<TuningRecord> GetAllTuningRecords() { return records; }
95
96 int64_t Size() { return records.size(); }
97};
98
99Database Database::MemoryDatabase() {
100 ObjectPtr<MemoryDatabaseNode> n = make_object<MemoryDatabaseNode>();
101 n->records.clear();
102 n->workloads.clear();
103 return Database(n);
104}
105
106TVM_REGISTER_NODE_TYPE(MemoryDatabaseNode);
107TVM_REGISTER_GLOBAL("meta_schedule.DatabaseMemoryDatabase")
108 .set_body_typed(Database::MemoryDatabase);
