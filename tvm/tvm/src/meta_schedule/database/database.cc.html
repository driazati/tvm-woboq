26Workload::Workload(IRModule mod) {
27 ObjectPtr<WorkloadNode> n = runtime::make_object<WorkloadNode>();
28 n->shash = tvm::StructuralHash()(mod);
29 n->mod = mod;
30 data_ = std::move(n);
31}

33Workload::Workload(IRModule mod, Workload::THashCode shash) {
34 ObjectPtr<WorkloadNode> n = runtime::make_object<WorkloadNode>();
35 n->mod = mod;
36 n->shash = shash;
37 data_ = std::move(n);
38}

40ObjectRef WorkloadNode::AsJSON() const {
41 // Convert `this->mod` to JSON
42 std::string json_mod = tvm::SaveJSON(this->mod);
43 // Dump the JSON string to base64
44 std::string b64_mod = Base64Encode(json_mod);
45 // Output
46 return Array<ObjectRef>{SHash2Str(this->shash), String(b64_mod)};
47}
48
49Workload Workload::FromJSON(const ObjectRef& json_obj) {
50 IRModule mod{nullptr};
51 THashCode shash = 0;
52 try {
53 const ArrayNode* json_array = json_obj.as<ArrayNode>();
54 CHECK(json_array && json_array->size() == 2);
55 // Load json[0] => shash
56 String str_shash = Downcast<String>(json_array->at(0));
57 // Load json[1] => mod
58 {
59 String b64_mod = Downcast<String>(json_array->at(1));
60 std::string json_mod = Base64Decode(b64_mod);
61 mod = Downcast<IRModule>(LoadJSON(json_mod));
62 }
63 // Verify SHash(mod) == shash
64 shash = tvm::StructuralHash()(mod);
65 String recalc_shash = SHash2Str(shash);
66 CHECK_EQ(recalc_shash, str_shash) << "ValueError: Structural hash changed. Given: " << str_shash
67 << "; Recalculated: " << recalc_shash;
68 } catch (const std::runtime_error& e) { // includes tvm::Error and dmlc::Error
69 LOG(FATAL) << "ValueError: Unable to parse the JSON object: " << json_obj
70 << "\nThe error is: " << e.what();
71 }
72 return Workload(mod, shash);
73}

77TuningRecord::TuningRecord(tir::Trace trace, Workload workload, Optional<Array<FloatImm>> run_secs,
78 Optional<Target> target, Optional<Array<ArgInfo>> args_info) {
79 ObjectPtr<TuningRecordNode> n = make_object<TuningRecordNode>();
80 n->trace = trace;
81 n->workload = workload;
82 n->run_secs = run_secs;
83 n->target = target;
84 n->args_info = args_info;
85 this->data_ = n;
86}

88MeasureCandidate TuningRecordNode::AsMeasureCandidate() const {
89 tir::Schedule sch =
90 tir::Schedule::Traced(workload->mod, -1, 0, tir::ScheduleErrorRenderLevel::kDetail);
91 trace->ApplyToSchedule(sch, false, nullptr);
92 return MeasureCandidate(sch, ArgInfo::FromEntryFunc(sch->mod(), /*remove_preproc=*/true));
93}
94
95ObjectRef TuningRecordNode::AsJSON() const {
96 Optional<Array<ObjectRef>> json_args_info{nullptr};
97 Optional<ObjectRef> json_target{nullptr};
98 if (args_info.defined()) {
99 Array<ObjectRef> info;
100 info.reserve(args_info.value().size());
101 for (const ArgInfo& arg_info : args_info.value()) {
102 info.push_back(arg_info->AsJSON());
103 }
104 json_args_info = info;
105 }
106 if (target.defined()) {
107 json_target = target.value()->Export();
108 }
109 return Array<ObjectRef>{trace->AsJSON(false), //
110 run_secs, //
111 json_target, //
112 json_args_info};
113}

115TuningRecord TuningRecord::FromJSON(const ObjectRef& json_obj, const Workload& workload) {
116 tir::Trace trace{nullptr};
117 Optional<Array<FloatImm>> run_secs{nullptr};
118 Optional<Target> target{nullptr};
119 Optional<Array<ArgInfo>> args_info{nullptr};
120 try {
121 const ArrayNode* json_array = json_obj.as<ArrayNode>();
122 CHECK(json_array && json_array->size() == 4);
123 // Load json[1] => run_secs
124 if (json_array->at(1).defined()) {
125 run_secs = AsFloatArray(json_array->at(1));
126 }
127 // Load json[2] => target
128 if (json_array->at(2).defined()) {
129 target = Target(Downcast<Map<String, ObjectRef>>(json_array->at(2)));
130 }
131 // Load json[3] => args_info
132 if (json_array->at(3).defined()) {
133 const ArrayNode* json_args_info = json_array->at(3).as<ArrayNode>();
134 Array<ArgInfo> info;
135 info.reserve(json_args_info->size());
136 for (const ObjectRef& json_arg_info : *json_args_info) {
137 info.push_back(ArgInfo::FromJSON(json_arg_info));
138 }
139 args_info = info;
140 }
141 // Load json[0] => trace
142 {
143 const ObjectRef& json_trace = json_array->at(0);
144 tir::Schedule sch =
145 tir::Schedule::Traced(workload->mod, /*seed=*/-1, /*debug_mask=*/0,
146 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kNone);
147 tir::Trace::ApplyJSONToSchedule(json_trace, sch);
148 trace = sch->trace().value();
149 }
150 } catch (const std::runtime_error& e) { // includes tvm::Error and dmlc::Error
151 LOG(FATAL) << "ValueError: Unable to parse the JSON object: " << json_obj
152 << "\nThe error is: " << e.what();
153 }
154 return TuningRecord(trace, workload, run_secs, target, args_info);
155}

159Optional<TuningRecord> DatabaseNode::QueryTuningRecord(const IRModule& mod, const Target& target,
160 const String& workload_name) {
161 if (!this->HasWorkload(mod)) {
162 return NullOpt;
163 }
164 Array<TuningRecord> records = this->GetTopK(this->CommitWorkload(mod), 1);
165 if (records.empty()) {
166 return NullOpt;
167 }
168 ICHECK_EQ(records.size(), 1);
169 return records[0];
170}
171
172Optional<tir::Schedule> DatabaseNode::QuerySchedule(const IRModule& mod, const Target& target,
173 const String& workload_name) {
174 if (Optional<TuningRecord> opt_record = this->QueryTuningRecord(mod, target, workload_name)) {
175 TuningRecord record = opt_record.value();
176 tir::Schedule sch =
177 tir::Schedule::Traced(record->workload->mod, /*seed=*/-1, /*debug_mask=*/0,
178 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kDetail);
179 record->trace->ApplyToSchedule(sch, false);
180 return sch;
181 } else {
182 return NullOpt;
183 }
184}
185
186Optional<IRModule> DatabaseNode::QueryIRModule(const IRModule& mod, const Target& target,
187 const String& workload_name) {
188 if (Optional<tir::Schedule> opt_sch = this->QuerySchedule(mod, target, workload_name)) {
189 return opt_sch.value()->mod();
190 } else {
191 return NullOpt;
192 }
193}

195std::vector<Database>* ThreadLocalDatabases() {
196 static thread_local std::vector<Database> tls;
197 return &tls;
198}
199
200void Database::EnterWithScope() { ThreadLocalDatabases()->push_back(*this); }
201
202void Database::ExitWithScope() { ThreadLocalDatabases()->pop_back(); }
203
204Optional<Database> Database::Current() {
205 std::vector<Database>* tls = ThreadLocalDatabases();
206 if (tls->empty()) {
207 return NullOpt;
208 } else {
209 return tls->back();
210 }
211}
