19#include "../utils.h"

21namespace tvm {
22namespace meta_schedule {
23
24class ScheduleFnDatabaseNode : public DatabaseNode {
25 public:
26 runtime::TypedPackedFunc<bool(tir::Schedule)> schedule_fn;
27
28 void VisitAttrs(AttrVisitor* v) {
29 // `schedule_fn` is not visited.
30 }
31
32 static constexpr const char* _type_key = "meta_schedule.ScheduleFnDatabase";
33 TVM_DECLARE_FINAL_OBJECT_INFO(ScheduleFnDatabaseNode, DatabaseNode);
34
35 public:
36 Optional<TuningRecord> QueryTuningRecord(const IRModule& mod, const Target& target,
37 const String& workload_name) final {
38 if (Optional<tir::Schedule> sch = this->QuerySchedule(mod, target, workload_name)) {
39 return TuningRecord(sch.value()->trace().value(),
40 /*workload=*/Workload(mod, 0), //
41 /*run_secs=*/NullOpt, //
42 /*target=*/target, //
43 /*arg_info=*/NullOpt);
44 }
45 return NullOpt;
46 }
47
48 Optional<tir::Schedule> QuerySchedule(const IRModule& mod, const Target& target,
49 const String& workload_name) final {
50 tir::Schedule sch =
51 tir::Schedule::Traced(WithAttr<IRModule>(mod, "task_name", workload_name),
52 /*rand_state=*/-1,
53 /*debug_mode=*/0,
54 /*error_render_level=*/tir::ScheduleErrorRenderLevel::kDetail);
55 if (!schedule_fn(sch)) {
56 return NullOpt;
57 }
58 return sch;
59 }
60
61 bool HasWorkload(const IRModule& mod) final {
62 LOG(FATAL) << "NotImplementedError: ScheduleFnDatabase.HasWorkload";
63 throw;
64 }
65
66 Workload CommitWorkload(const IRModule& mod) final {
67 LOG(FATAL) << "NotImplementedError: ScheduleFnDatabase.CommitWorkload";
68 throw;
69 }
70
71 void CommitTuningRecord(const TuningRecord& record) final {
72 LOG(FATAL) << "NotImplementedError: ScheduleFnDatabase.CommitTuningRecord";
73 throw;
74 }
75
76 Array<TuningRecord> GetTopK(const Workload& workload, int top_k) final {
77 LOG(FATAL) << "NotImplementedError: ScheduleFnDatabase.GetTopK";
78 throw;
79 }
80
81 Array<TuningRecord> GetAllTuningRecords() final {
82 LOG(FATAL) << "NotImplementedError: ScheduleFnDatabase.GetAllTuningRecords";
83 throw;
84 }
85
86 int64_t Size() final {
87 LOG(FATAL) << "NotImplementedError: ScheduleFnDatabase.size";
88 throw;
89 }
90};
91
92Database Database::ScheduleFnDatabase(runtime::TypedPackedFunc<bool(tir::Schedule)> schedule_fn) {
93 ObjectPtr<ScheduleFnDatabaseNode> n = make_object<ScheduleFnDatabaseNode>();
94 n->schedule_fn = std::move(schedule_fn);
95 return Database(n);
96}
97
98TVM_REGISTER_NODE_TYPE(ScheduleFnDatabaseNode);
99TVM_REGISTER_GLOBAL("meta_schedule.DatabaseScheduleFnDatabase")
100 .set_body_typed(Database::ScheduleFnDatabase);
