26TuneContext::TuneContext(Optional<IRModule> mod, //
27 Optional<Target> target, //
28 Optional<SpaceGenerator> space_generator, //
29 Optional<SearchStrategy> search_strategy, //
30 Optional<Array<ScheduleRule>> sch_rules, //
31 Optional<Array<Postproc>> postprocs, //
32 Optional<Map<Mutator, FloatImm>> mutator_probs, //
33 Optional<String> task_name, //
34 PackedFunc logging_func, //
35 support::LinearCongruentialEngine::TRandState rand_state, //
36 int num_threads) {
37 ObjectPtr<TuneContextNode> n = make_object<TuneContextNode>();
38 n->mod = mod;
39 n->target = target;
40 n->space_generator = space_generator;
41 n->search_strategy = search_strategy;
42 n->sch_rules = sch_rules.value_or({});
43 n->postprocs = postprocs.value_or({});
44 n->mutator_probs = mutator_probs.value_or({});
45 n->task_name = task_name;
46 n->logging_func = logging_func;
47 support::LinearCongruentialEngine(&n->rand_state).Seed(rand_state);
48 n->num_threads = num_threads;
49 n->is_terminated = false;
50 n->runner_futures = NullOpt;
51 n->measure_candidates = NullOpt;
52 data_ = std::move(n);
53}
54
55TuneContext TuneContextNode::Clone() const {
56 ObjectPtr<TuneContextNode> n = make_object<TuneContextNode>(*this);
57 if (this->sch_rules.defined()) {
58 n->sch_rules = Array<ScheduleRule>();
59 for (const ScheduleRule& sch_rule : this->sch_rules) {
60 n->sch_rules.push_back(sch_rule->Clone());
61 }
62 }
63 if (this->postprocs.defined()) {
64 n->postprocs = Array<Postproc>();
65 for (const Postproc& postproc : this->postprocs) {
66 n->postprocs.push_back(postproc->Clone());
67 }
68 }
69 if (this->mutator_probs.defined()) {
70 n->mutator_probs = Map<Mutator, FloatImm>();
71 for (const auto& kv : this->mutator_probs) {
72 n->mutator_probs.Set(kv.first->Clone(), kv.second);
73 }
74 }
75 if (this->space_generator.defined()) n->space_generator = this->space_generator.value()->Clone();
76 if (this->search_strategy.defined()) n->search_strategy = this->search_strategy.value()->Clone();
77 n->rand_state = support::LinearCongruentialEngine(&n->rand_state).ForkSeed();
78 n->Initialize();
79 return TuneContext(n);
80}

82void TuneContextNode::Initialize() {
83 if (this->space_generator.defined()) {
84 this->space_generator.value()->InitializeWithTuneContext(GetRef<TuneContext>(this));
85 }
86 if (this->search_strategy.defined()) {
87 this->search_strategy.value()->InitializeWithTuneContext(GetRef<TuneContext>(this));
88 }
89 for (const ScheduleRule& sch_rule : sch_rules) {
90 sch_rule->InitializeWithTuneContext(GetRef<TuneContext>(this));
91 }
92 for (const Postproc& postproc : postprocs) {
93 postproc->InitializeWithTuneContext(GetRef<TuneContext>(this));
94 }
95 for (const auto& kv : mutator_probs) {
96 kv.first->InitializeWithTuneContext(GetRef<TuneContext>(this));
97 }
98}

104void TuneContextNode::_SendToBuilder(const Builder& builder) {
105 auto _ = Profiler::TimedScope("SendToBuilder");
106 Array<MeasureCandidate> candidates = this->measure_candidates.value();
107 Target target = this->target.value();
108 Array<BuilderInput> inputs;
109 inputs.reserve(candidates.size());
110 for (const MeasureCandidate& candidate : candidates) {
111 inputs.push_back(BuilderInput(candidate->sch->mod(), target));
112 }
113 this->builder_results = builder->Build(inputs);
114}
115
116void TuneContextNode::_SendToRunner(const Runner& runner) {
117 auto _ = Profiler::TimedScope("SendToRunner");
118 Array<MeasureCandidate> candidates = this->measure_candidates.value();
119 Array<BuilderResult> builder_results = this->builder_results.value();
120 Target target = this->target.value();
121 ICHECK_EQ(candidates.size(), builder_results.size());
122 int n = candidates.size();
123 int n_build_errors = 0;
124 Array<RunnerInput> inputs;
125 inputs.reserve(n);
126 for (int i = 0; i < n; ++i) {
127 const MeasureCandidate& candidate = candidates[i];
128 const BuilderResult& builder_result = builder_results[i];
129 if (builder_result->error_msg.defined()) {
130 ++n_build_errors;
131 continue;
132 }
133 inputs.push_back(RunnerInput(/*artifact_path=*/builder_result->artifact_path.value(),
134 /*device_type=*/target->kind->name,
135 /*args_info=*/candidate->args_info));
136 }
137 Array<RunnerFuture> futures = runner->Run(inputs);
138 if (n_build_errors == 0) {
139 this->runner_futures = futures;
140 return;
141 }
142 Array<RunnerFuture> results;
143 results.reserve(n);
144 for (int i = 0, j = 0; i < n; ++i) {
145 const BuilderResult& builder_result = builder_results[i];
146 if (builder_result->error_msg.defined()) {
147 results.push_back(RunnerFuture(
148 /*f_done=*/[]() -> bool { return true; },
149 /*f_result=*/
150 [msg = builder_result->error_msg]() -> RunnerResult {
151 return RunnerResult(NullOpt, msg);
152 }));
153 } else {
154 results.push_back(futures[j++]);
155 }
156 }
157 this->runner_futures = results;
158}
159
160Array<RunnerResult> TuneContextNode::_Join() {
161 ICHECK(this->runner_futures.defined());
162 Array<RunnerFuture> futures = this->runner_futures.value();
163 int n = futures.size();
164 Array<RunnerResult> results;
165 {
166 auto _ = Profiler::TimedScope("JoinRunnerFutures");
167 results.reserve(n);
168 for (RunnerFuture future : futures) {
169 results.push_back(future->Result());
170 }
171 }
172 if (this->search_strategy.defined()) {
173 this->search_strategy.value()->NotifyRunnerResults(this->measure_candidates.value(), results);
174 }
175 ICHECK(this->measure_candidates.defined());
176 ICHECK(this->builder_results.defined());
177 ICHECK_EQ(results.size(), this->measure_candidates.value().size());
178 ICHECK_EQ(results.size(), this->builder_results.value().size());
179 return results;
180}
181
182void TuneContextNode::_ClearMeasureState() {
183 this->measure_candidates = NullOpt;
184 this->builder_results = NullOpt;
185 this->runner_futures = NullOpt;
186}
