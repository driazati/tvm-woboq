30ExtractedTask::ExtractedTask(String task_name, IRModule mod, Target target,
31 Array<IRModule> dispatched, int weight) {
32 ObjectPtr<ExtractedTaskNode> n = make_object<ExtractedTaskNode>();
33 n->task_name = task_name;
34 n->mod = mod;
35 n->target = target;
36 n->dispatched = dispatched;
37 n->weight = weight;
38 data_ = n;
39}

41Optional<tir::PrimFunc> DefaultTaskFilterImpl(const Array<te::Tensor>& args,
42 const Array<runtime::NDArray>& constants,
43 bool allow_extern_op) {
44 using namespace ::tvm::te;
45 std::vector<Tensor> stack;
46 std::unordered_set<const TensorNode*> visited;
47 for (const Tensor& v : args) {
48 for (const PrimExpr& e : v->shape) {
49 // Dynamic shape is not supported for now
50 if (!e->IsInstance<IntImmNode>()) {
51 return NullOpt;
52 }
53 }
54 if (!visited.count(v.get())) {
55 visited.insert(v.get());
56 stack.push_back(v);
57 }
58 }
59 while (!stack.empty()) {
60 Tensor tensor = stack.back();
61 stack.pop_back();
62 if (tensor->op->IsInstance<PlaceholderOpNode>()) {
63 // do nothing
64 } else if (tensor->op->IsInstance<ComputeOpNode>() ||
65 (allow_extern_op && tensor->op->IsInstance<ExternOpNode>())) {
66 Array<Tensor> inputs = tensor->op->InputTensors();
67 for (const Tensor& v : inputs) {
68 if (!visited.count(v.get())) {
69 visited.insert(v.get());
70 stack.push_back(v);
71 }
72 }
73 } else {
74 return NullOpt;
75 }
76 }
77 PrimFunc func = te::CreatePrimFuncWithConstants(args, constants);
78 bool dynamic_loop_extent = false;
79 PostOrderVisit(func->body, [&dynamic_loop_extent](const ObjectRef& obj) -> void {
80 if (const auto* loop = obj.as<tir::ForNode>()) {
81 if (!loop->extent->IsInstance<IntImmNode>()) {
82 dynamic_loop_extent = true;
83 }
84 }
85 });
86 if (dynamic_loop_extent) {
87 return NullOpt;
88 }
89 return func;
90}
91
92Optional<tir::PrimFunc> DefaultTaskFilter(const Array<te::Tensor>& args,
93 const Array<runtime::NDArray>& constants) {
94 return DefaultTaskFilterImpl(args, constants, false);
95}
96
97Optional<tir::PrimFunc> DefaultTaskFilterAllowExtern(const Array<te::Tensor>& args,
98 const Array<runtime::NDArray>& constants) {
99 return DefaultTaskFilterImpl(args, constants, true);
100}
