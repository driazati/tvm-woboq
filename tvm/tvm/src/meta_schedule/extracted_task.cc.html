30ExtractedTask::ExtractedTask(String task_name, IRModule mod, Target target,
31 Array<IRModule> dispatched, int weight) {
32 ObjectPtr<ExtractedTaskNode> n = make_object<ExtractedTaskNode>();
33 n->task_name = task_name;
34 n->mod = mod;
35 n->target = target;
36 n->dispatched = dispatched;
37 n->weight = weight;
38 data_ = n;
39}
40
41TVM_REGISTER_NODE_TYPE(ExtractedTaskNode);
42TVM_REGISTER_GLOBAL("meta_schedule.ExtractedTask")
43 .set_body_typed([](String task_name, IRModule mod, Target target, Array<IRModule> dispatched,
44 int weight) -> ExtractedTask {
45 return ExtractedTask(task_name, mod, target, dispatched, weight);
46 });
