26class ThreadExtentChecker : private StmtVisitor {
27 public:
28 static bool Check(const Stmt& stmt, int thread_warp_size) {
29 try {
30 ICHECK(thread_warp_size > 0);
31 ThreadExtentChecker checker(thread_warp_size);
32 checker.VisitStmt(stmt);
33 return true;
34 } catch (const dmlc::Error& e) {
35 return false;
36 }
37 }
38
39 private:
40 explicit ThreadExtentChecker(int thread_warp_size) : thread_warp_size_(thread_warp_size) {}
41
42 void VisitStmt_(const ForNode* loop) {
43 runtime::ThreadScope thread_scope = GetThreadScope(loop);
44 if (IsThreadIdx(thread_scope)) {
45 if (const int64_t* p_ext = GetLoopIntExtent(loop)) {
46 int64_t ext = *p_ext;
47 if (thread_scope.dim_index == 0) {
48 std::swap(thread_idx_x, ext);
49 StmtVisitor::VisitStmt_(loop);
50 std::swap(thread_idx_x, ext);
51 } else if (thread_scope.dim_index == 1) {
52 std::swap(thread_idx_y, ext);
53 StmtVisitor::VisitStmt_(loop);
54 std::swap(thread_idx_y, ext);
55 } else if (thread_scope.dim_index == 2) {
56 std::swap(thread_idx_z, ext);
57 StmtVisitor::VisitStmt_(loop);
58 std::swap(thread_idx_z, ext);
59 } else {
60 StmtVisitor::VisitStmt_(loop);
61 }
62 return;
63 } else {
64 throw dmlc::Error("Dynamic thread extent");
65 }
66 }
67 StmtVisitor::VisitStmt_(loop);
68 }
69
70 void VisitStmt_(const BlockNode* block) {
71 int old_thread_idx_x = thread_idx_x;
72 if (block->annotations.count(attr::warp_execution)) {
73 thread_idx_x = thread_warp_size_;
74 }
75 if (Optional<Integer> low_inclusive =
76 GetAnn<Integer>(block, attr::meta_schedule_thread_extent_low_inclusive)) {
77 if (Optional<Integer> high_inclusive =
78 GetAnn<Integer>(block, attr::meta_schedule_thread_extent_high_inclusive)) {
79 int64_t low = low_inclusive.value()->value;
80 int64_t high = high_inclusive.value()->value;
81 int64_t thread_extent_product = thread_idx_x * thread_idx_y * thread_idx_z;
82 if (!(low <= thread_extent_product && thread_extent_product <= high)) {
83 throw dmlc::Error("Thread extent");
84 }
85 }
86 }
87 StmtVisitor::VisitStmt_(block);
88 thread_idx_x = old_thread_idx_x;
89 }
90
91 int64_t thread_idx_x = 1;
92 int64_t thread_idx_y = 1;
93 int64_t thread_idx_z = 1;
94 int thread_warp_size_ = -1;
95};

103/*! \brief Extract attribute from a target. */
104Integer Extract(const Target& target, const char* name) {
105 ICHECK(target.defined());
106 if (Optional<Integer> v = target->GetAttr<Integer>(name)) {
107 return v.value();
108 }
109 LOG(FATAL) << "AttributedError: \"" << name << "\" is not defined in the target";
110 throw;
111}

113/*! \brief Verify the correctness of the generated GPU code. */
114class VerifyGPUCodeNode : public PostprocNode {
115 public:
116 Map<String, PrimExpr> target_constraints_{nullptr};
117 int thread_warp_size_ = -1;
118
119 void InitializeWithTuneContext(const TuneContext& context) final {
120 ICHECK(context->target.defined());
121 Target target = context->target.value();
122 this->target_constraints_ = Map<String, PrimExpr>{
123 {"max_shared_memory_per_block", Extract(target, "max_shared_memory_per_block")},
124 {"max_threads_per_block", Extract(target, "max_threads_per_block")},
125 {"max_vthread", Integer(8)},
126 {"max_vector_bytes", Integer(16)},
127 };
128 thread_warp_size_ = Extract(target, "thread_warp_size").IntValue();
129 }
130
131 bool Verify(const IRModule& mod) const {
132 for (const auto& kv : mod->functions) {
133 if (const auto* prim_func = kv.second.as<tir::PrimFuncNode>()) {
134 if (!tir::VerifyGPUCode(GetRef<tir::PrimFunc>(prim_func), this->target_constraints_)) {
135 return false;
136 }
137 }
138 }
139 return true;
140 }
141
142 bool Apply(const tir::Schedule& sch) final {
143 IRModule mod = sch->mod();
144 for (const auto& kv : mod->functions) {
145 const GlobalVar& g_var = kv.first;
146 const BaseFunc& base_func = kv.second;
147 if (const auto* prim_func = base_func.as<tir::PrimFuncNode>()) {
148 if (!tir::ThreadExtentChecker::Check(prim_func->body, thread_warp_size_)) {
149 return false;
150 }
151 IRModule lowered{nullptr};
152 try {
153 auto pass_list = Array<tvm::transform::Pass>();
154 // Phase 1
155 // First three passes are not needed in TIR schedule.
156 // pass_list.push_back(tir::transform::InjectPrefetch());
157 // pass_list.push_back(tir::transform::TextureFlatten());
158 // pass_list.push_back(tir::transform::StorageFlatten(64, instrument_bound_checkers));
159 pass_list.push_back(tir::transform::LowerCrossThreadReduction());
160 pass_list.push_back(tir::transform::LowerInitBlock());
161 pass_list.push_back(tir::transform::PlanAndUpdateBufferAllocationLocation());
162 pass_list.push_back(tir::transform::ConvertBlocksToOpaque());
163 pass_list.push_back(tir::transform::UnifyThreadBinding());
164 pass_list.push_back(tir::transform::CompactBufferAllocation());
165 pass_list.push_back(tir::transform::LowerMatchBuffer());
166 pass_list.push_back(tir::transform::InjectSoftwarePipeline());
167 pass_list.push_back(tir::transform::LowerOpaqueBlock());
168 pass_list.push_back(tir::transform::FlattenBuffer());
169 pass_list.push_back(tir::transform::BF16Legalize());
170 pass_list.push_back(tir::transform::NarrowDataType(32));
171 pass_list.push_back(tir::transform::Simplify());
172 // Phase 2
173 pass_list.push_back(tir::transform::VectorizeLoop(true));
174 pass_list.push_back(tir::transform::InjectVirtualThread());
175 pass_list.push_back(tir::transform::InjectDoubleBuffer());
176 pass_list.push_back(tir::transform::StorageRewrite());
177 pass_list.push_back(tir::transform::MergeDynamicSharedMemoryAllocations());
178 // Convert Function to IRModule
179 transform::PassContext pass_ctx = transform::PassContext::Current();
180 tir::PrimFunc f = WithAttr(GetRef<tir::PrimFunc>(prim_func), "global_symbol",
181 runtime::String(g_var->name_hint));
182 bool noalias = pass_ctx->GetConfig<Bool>("tir.noalias", Bool(true)).value();
183 if (noalias) {
184 f = WithAttr(std::move(f), "tir.noalias", Bool(true));
185 }
186 IRModule mod = IRModule(Map<GlobalVar, BaseFunc>({{GlobalVar(g_var->name_hint), f}}));
187 lowered = tvm::transform::Sequential(pass_list)(std::move(mod));
188 } catch (const dmlc::Error& e) {
189 return false;
190 }
191 if (!Verify(lowered)) {
192 return false;
193 }
194 }
195 }
196 return true;
197 }
198
199 Postproc Clone() const {
200 ObjectPtr<VerifyGPUCodeNode> n = make_object<VerifyGPUCodeNode>(*this);
201 n->target_constraints_ = this->target_constraints_;
202 return Postproc(n);
203 }
204
205 static constexpr const char* _type_key = "meta_schedule.VerifyGPUCode";
206 TVM_DECLARE_FINAL_OBJECT_INFO(VerifyGPUCodeNode, PostprocNode);
207};

209Postproc Postproc::VerifyGPUCode() {
210 ObjectPtr<VerifyGPUCodeNode> n = make_object<VerifyGPUCodeNode>();
211 return Postproc(n);
212}
