33inline BlockRV GetOnlyProducer(Schedule sch, BlockRV block) {
34 Array<BlockRV> producers = sch->GetProducers(block);
35 ICHECK_EQ(producers.size(), 1);

45inline LoopRV ScheduleDataPack(Schedule sch, BlockRV block) {
46 Array<ExprRV> factors{nullptr};
47 Array<LoopRV> loops = sch->GetLoops(block);
48 ICHECK_EQ(loops.size(), 6);
49
50 factors = sch->SamplePerfectTile(loops[2], /*n=*/2, /*max_innermost_factor=*/64);
51 Array<LoopRV> t0 = sch->Split(loops[2], {factors.begin(), factors.end()});
52 ICHECK_EQ(t0.size(), 2);
53
54 factors = sch->SamplePerfectTile(loops[3], /*n=*/2, /*max_innermost_factor=*/64);
55 Array<LoopRV> t1 = sch->Split(loops[3], {factors.begin(), factors.end()});
56 ICHECK_EQ(t1.size(), 2);
57
58 if (const int64_t* i = tir::GetLoopIntExtent(sch->GetSRef(loops[0]))) {
59 if (*i <= 16) {
60 sch->Unroll(loops[0]);
61 }
62 }
63 if (const int64_t* i = tir::GetLoopIntExtent(sch->GetSRef(loops[1]))) {
64 if (*i <= 16) {
65 sch->Unroll(loops[1]);
66 }
67 }
68 sch->Unroll(loops[4]);
69 sch->Unroll(loops[5]);
70 sch->Reorder({
71 t0[0],
72 t1[0],
73 t0[1],
74 t1[1],
75 loops[0],
76 loops[1],
77 loops[4],
78 loops[5],
79 });
80 return t1[1];
81}

83inline LoopRV ScheduleDataPackNCHW(Schedule sch, BlockRV block) {
84 Array<LoopRV> loops = sch->GetLoops(block);
85 ICHECK_EQ(loops.size(), 6);
86
87 if (const int64_t* i = tir::GetLoopIntExtent(sch->GetSRef(loops[0]))) {
88 if (*i <= 16) {
89 sch->Unroll(loops[0]);
90 }
91 }
92 if (const int64_t* i = tir::GetLoopIntExtent(sch->GetSRef(loops[1]))) {
93 if (*i <= 16) {
94 sch->Unroll(loops[1]);
95 }
96 }
97 sch->Unroll(loops[4]);
98 sch->Unroll(loops[5]);
99
100 Array<ExprRV> factors = sch->SamplePerfectTile(loops[3], /*n=*/2, /*max_innermost_factor=*/64);
101 Array<LoopRV> split =
102 sch->Split(loops[3], /*factors=*/{factors[0], factors[1]}, /*preserve_unit_loops=*/true);
103
104 LoopRV fused = sch->Fuse({loops[2], split[0]});
105 sch->Reorder({fused, split[1], loops[0], loops[1]});
106 return split[1];
107}

115TVM_REGISTER_GLOBAL("meta_schedule.winograd_data_pack.llvm")
116 .set_body_typed([](Schedule sch, BlockRV data_pack) -> Array<Schedule> {
117 BlockRV input_tile = GetOnlyProducer(sch, data_pack);
118 BlockRV data_pad = GetOnlyProducer(sch, input_tile);
119 ScheduleDataPack(sch, data_pack);
120 sch->ComputeAt(input_tile, /*loop_rv=*/sch->SampleComputeLocation(input_tile),
121 /*preserve_unit_loops=*/true);
122 sch->ComputeAt(data_pad, /*loop_rv=*/sch->SampleComputeLocation(data_pad),
123 /*preserve_unit_loops=*/true);
124 return {sch};
125 });

127TVM_REGISTER_GLOBAL("meta_schedule.winograd_output.nchw.cuda")
128 .set_body_typed([](Schedule sch, BlockRV output) -> Array<Schedule> {
129 // get loops
130 Array<LoopRV> loops = sch->GetLoops(output);
131 ICHECK_EQ(loops.size(), 4);
132
133 BlockRV OL{nullptr};
134
135 // tile
136 Optional<PrimExpr> tile_size =
137 tir::GetAnn<PrimExpr>(sch->GetSRef(output), "winograd_tile_size");
138 ICHECK(tile_size.defined()) << "Winograd tile size is not defined in block annotation!";
139 Array<LoopRV> split0 = sch->Split(loops[2], {NullOpt, tile_size.value()});
140 Array<LoopRV> split1 = sch->Split(loops[3], {NullOpt, tile_size.value()});
141 sch->Reorder({split0[0], split1[0], split0[1], split1[1]});
142
143 // compute_at
144 BlockRV inverse = GetOnlyProducer(sch, output);
145 sch->ComputeAt(inverse, /*loop_rv=*/split1[0],
146 /*preserve_unit_loops=*/true);
147
148 // fuse
149 LoopRV fused = sch->Fuse({loops[0], loops[1], split0[0], split1[0]});
150
151 int64_t max_threadblocks = 256;
152 int64_t max_threads_per_block = 1024;
153 auto get_factor = MakeFactorSampler(sch, {32, 64, 128, 256, 512, 1024});
154 BindBlockThreadIdx(sch, output, max_threadblocks, max_threads_per_block, get_factor);
155 return {sch};
156 });

169 .set_body_typed([](Schedule sch, BlockRV inverse) -> Array<Schedule> {
170 sch->SetScope(inverse, /*buffer_index=*/0, /*storage_scope=*/"local");
171 Array<LoopRV> loops = sch->GetLoops(inverse);
172 ICHECK_EQ(loops.size(), 6);
173 if (const int64_t* i = tir::GetLoopIntExtent(sch->GetSRef(loops[2]))) {
174 if (*i <= 16) {
175 sch->Unroll(loops[2]);
176 }
177 }
178 if (const int64_t* i = tir::GetLoopIntExtent(sch->GetSRef(loops[3]))) {
179 if (*i <= 16) {
180 sch->Unroll(loops[3]);
181 }
182 }
183 sch->Unroll(loops[4]);
184 sch->Unroll(loops[5]);
185 return {sch};

188TVM_REGISTER_GLOBAL("meta_schedule.winograd_data_pack.cuda")
189 .set_body_typed([](Schedule sch, BlockRV data_pack) -> Array<Schedule> {
190 BlockRV input_tile = GetOnlyProducer(sch, data_pack);
191 BlockRV data_pad = GetOnlyProducer(sch, input_tile);
192 LoopRV loop = ScheduleDataPack(sch, data_pack);
193 sch->ComputeAt(input_tile, /*loop_rv=*/loop, /*preserve_unit_loops=*/true);
194 sch->SetScope(input_tile, /*buffer_index=*/0, /*storage_scope=*/"local");
195 sch->ComputeInline(data_pad);
196 int64_t max_threadblocks = 256;
197 int64_t max_threads_per_block = 1024;
198 auto get_factor = MakeFactorSampler(sch, {32, 64, 128, 256, 512, 1024});
199 BindBlockThreadIdx(sch, data_pack, max_threadblocks, max_threads_per_block, get_factor);
200 return {sch};
201 });

203TVM_REGISTER_GLOBAL("meta_schedule.winograd_data_pack.nchw.cuda")
204 .set_body_typed([](Schedule sch, BlockRV data_pack) -> Array<Schedule> {
205 BlockRV input_tile = GetOnlyProducer(sch, data_pack);
206 BlockRV data_pad = GetOnlyProducer(sch, input_tile);
207
208 BlockRV data_l = sch->CacheWrite(data_pack, /*buffer_index=*/0, /*storage_scope=*/"local");
209 LoopRV loop = ScheduleDataPackNCHW(sch, data_pack);
210 sch->ReverseComputeAt(data_l, loop, /*preserve_unit_loops=*/true);
211 sch->ComputeAt(input_tile, /*loop_rv=*/loop, /*preserve_unit_loops=*/true);
212 sch->ComputeInline(data_pad);
213
214 int64_t max_threadblocks = 256;
215 int64_t max_threads_per_block = 1024;
216 auto get_factor = MakeFactorSampler(sch, {32, 64, 128, 256, 512, 1024});
217 BindBlockThreadIdx(sch, data_pack, max_threadblocks, max_threads_per_block, get_factor);
218 return {sch};
219 });
