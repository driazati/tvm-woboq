34/*!
35 * \brief Configuration of data reuse type:
36 * 0) kNoReuse: no reuse is allowed, then no cache_read/write is performed.
37 * 1) kMayReuse: reuse is allowed, but no reuse is explored.
38 * 2) kMustReuse: reuse is allowed and no reuse is not explored.
39 */
40enum class ReuseType : int32_t {
41 kNoReuse = 0,
42 kMayReuse = 1,
43 kMustReuse = 2,
44};
45
46/*!
47 * \brief Converts a string to ReuseType.
48 * \param str The string to be converted.
49 * \return The converted ReuseType.
50 */
51inline ReuseType Str2ReuseType(const String& str) {
52 if (str == "no") {
53 return ReuseType::kNoReuse;
54 } else if (str == "may") {
55 return ReuseType::kMayReuse;
56 } else if (str == "must") {
57 return ReuseType::kMustReuse;
58 } else {
59 LOG(FATAL) << "ValueError: Unknown ReuseType: " << str;
60 throw;
61 }
62}
63
64/*! \brief Configuration of data reuse patterns */
65struct ReuseConfig {
66 /*! \brief Type of data reuse: no-reuse, may-reuse or must-reuse */
67 ReuseType req;
68 /*! \brief Which levels are caching stage inserted at */
69 std::vector<int> levels;
70 /*! \brief The storage scope */
71 String scope;
72
73 /*! \brief Default constructor: no data reuse */
74 ReuseConfig() : req(ReuseType::kNoReuse) {}
75
76 /*! \brief Construct from a configuration dictionary */
77 explicit ReuseConfig(const Map<String, ObjectRef>& config)
78 : req(Str2ReuseType(Downcast<String>(config.at("req")))),
79 levels(support::AsVector<Integer, int>(Downcast<Array<Integer>>(config.at("levels")))),
80 scope(Downcast<String>(config.at("scope"))) {
81 ICHECK_EQ(config.size(), 3);
82 }
83};

120/*!
121 * \brief Helper to apply a sub-rule to a list of auto scheduling states
122 * \tparam FLambda The type of the sub-rule functor
123 * \param states The list of states to be applied
124 * \return The list of states after applying the sub-rule
125 */
126template <class FLambda>
127std::vector<State> SubRule(std::vector<State> states, FLambda sub_rule) {
128 std::vector<State> results;
129 for (auto&& state : states) {
130 std::vector<State> next = sub_rule(std::move(state));
131 results.insert(results.end(), //
132 std::make_move_iterator(next.begin()), //
133 std::make_move_iterator(next.end()));
134 }
135 return results;
136}

138/*!
139 * \brief The mega rule: multi-level tiling with data reuse
140 */
141class MultiLevelTilingNode : public ScheduleRuleNode {
142 public:
143 virtual ~MultiLevelTilingNode() = default;
144
145 // SubRule 1. add write cache
146 std::vector<State> AddWriteReuse(State state) const;
147 // SubRule 2. tile the loop nest
148 std::vector<State> TileLoopNest(State state) const;
149 // SubRule 3. add read cache
150 std::vector<State> AddReadReuse(State state) const;
151
152 // Do nothing; Inherited from ScheduleRuleNode
153 void InitializeWithTuneContext(const TuneContext& context) final;
154
155 // Entry of the mega rule; Inherited from ScheduleRuleNode
156 Array<tir::Schedule> Apply(const tir::Schedule& sch, const tir::BlockRV& block_rv) override;
157
158 protected:
159 virtual std::vector<State> ApplySubRules(std::vector<State> states);
160
161 // Annotate a block to use cooperative fetching
162 void AnnotateCooperativeFetching(tir::Schedule* sch, const tir::BlockRV& block) const;
163
164 public:
165 /*!
166 * \brief The tiling structure. Recommended:
167 * - 'SSRSRS' on CPU
168 * - 'SSSRRSRS' on GPU
169 */
170 String structure;
171 /*! \brief For each level of tiles, which thread axis it is bound to */
172 Array<String> tile_binds;
173 /*! \brief The maximum size of the innermost factor */
174 int max_innermost_factor;
175 /*! \brief The length of vector lane in vectorized cooperative fetching */
176 std::vector<int> vector_load_lens;
177 /*! \brief Data reuse configuration for reading */
178 ReuseConfig reuse_read_;
179 /*! \brief Data reuse configuration for writing */
180 ReuseConfig reuse_write_;
181 /*! \brief The indices of spatial tiles in `structure` */
182 std::vector<int> s_indices_;
183 /*! \brief The indices of reduction tiles in `structure` */
184 std::vector<int> r_indices_;
185 /*! \brief The size of the thread warp */
186 int thread_warp_size_;
187 /*! \brief The maximum number of threads to be used size of a thread warp */
188 int max_threads_per_block_;
189
190 void VisitAttrs(tvm::AttrVisitor* v) {
191 v->Visit("structure", &structure);
192 v->Visit("tile_binds", &tile_binds);
193 v->Visit("max_innermost_factor", &max_innermost_factor);
194 // `vector_load_lens` is not visited
195 // `reuse_read_` is not visited
196 // `reuse_write_` is not visited
197 // `s_indices_` is not visited
198 // `r_indices_` is not visited
199 // `thread_warp_size_` is not visited
200 // `max_threads_per_block` is not visited
201 }
202
203 static constexpr const char* _type_key = "meta_schedule.MultiLevelTiling";
204 TVM_DECLARE_BASE_OBJECT_INFO(MultiLevelTilingNode, ScheduleRuleNode);
205};
206
207template <typename NodeType>
208ObjectPtr<NodeType> MultiLevelTilingInitCommon(String structure, Optional<Array<String>> tile_binds,
209 Optional<Integer> max_innermost_factor,
210 Optional<Array<Integer>> vector_load_lens,
211 Optional<Map<String, ObjectRef>> reuse_read,
212 Optional<Map<String, ObjectRef>> reuse_write) {
213 ObjectPtr<NodeType> n = make_object<NodeType>();
214 n->structure = structure;
215 n->tile_binds = tile_binds.value_or({});
216 n->max_innermost_factor = max_innermost_factor.value_or(Integer(-1))->value;
217 n->vector_load_lens = vector_load_lens.defined()
218 ? support::AsVector<Integer, int>(vector_load_lens.value())
219 : std::vector<int>();
220 n->reuse_read_ = reuse_read.defined() ? ReuseConfig(reuse_read.value()) : ReuseConfig();
221 n->reuse_write_ = reuse_write.defined() ? ReuseConfig(reuse_write.value()) : ReuseConfig();
222 for (int i = 0, len = structure.size(); i < len; ++i) {
223 char c = structure.data()[i];
224 if (c == 'S') {
225 n->s_indices_.push_back(i);
226 } else if (c == 'R') {
227 n->r_indices_.push_back(i);
228 } else {
229 LOG(FATAL) << "ValueError: Invalid tiling structure: " << structure;
230 }
231 }
232 n->thread_warp_size_ = -1;
233 n->max_threads_per_block_ = -1;
234 return n;
235}
