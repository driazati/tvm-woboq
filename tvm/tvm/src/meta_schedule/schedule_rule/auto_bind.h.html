19#ifndef TVM_META_SCHEDULE_SCHEDULE_RULE_AUTO_BIND_H_

22#include "../utils.h"
23
24namespace tvm {
25namespace meta_schedule {
26
27/*!
28 * \brief Bind the given block if it is not bound to blockIdx or threadIdx.
29 * \param sch The schedule.
30 * \param block The block to be bound.
31 * \param max_threadblocks The maximum number of threadblocks allowed.
32 * \param max_threads The maximum number of threads allowed.
33 * \param get_factor A function that returns the tiling factor.
34 */
35void BindBlockThreadIdx(const tir::Schedule& sch, const tir::BlockRV& block,
36 int64_t max_threadblocks, int64_t max_threads_per_block,
37 std::function<tir::ExprRV(int64_t max_extent)> get_factor);
38
39/*!
40 * \brief Given candidates of thread_extents, make a sampler that use `sch->SampleCategorical`
41 * to return a random thread extent.
42 * \param sch The schedule
43 * \param thread_extents The candidate thread extents.
44 * \return A sampler that returns a random thread extent.
45 */
46std::function<tir::ExprRV(int64_t max_extent)> MakeFactorSampler(tir::Schedule sch,
47 Array<Integer> thread_extents);
48
49} // namespace meta_schedule
50} // namespace tvm
