31/*!
32 * \brief Get the buffer dimensions for all the read buffers of a block, but marks the reduction
33 * buffers' dimensions as -1
34 * \param block_sref The block to be processed
35 * \return The buffer dimensions for all the read buffers of a block, except for reduction buffers
36 * \note The method is not designed for generic analysis and relies on assumptions in the scenario
37 * of multi-level tiling, so it's intentionally kept inside this file not in the analysis header
38 */
39std::vector<int> GetReadBufferNDims(const StmtSRef& block_sref) {
40 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
41 const BufferNode* write_buffer = block->writes[0]->buffer.get();
42 int n = block->reads.size();
43 std::vector<int> results(n, -1);
44 for (int i = 0; i < n; ++i) {
45 const BufferNode* read_buffer = block->reads[i]->buffer.get();
46 if (read_buffer != write_buffer) {
47 results[i] = read_buffer->shape.size();
48 }
49 }
50 return results;
51}

93// Entry of the mega rule; Inherited from ScheduleRuleNode
94Array<Schedule> MultiLevelTilingNode::Apply(const Schedule& sch, const BlockRV& block_rv) {
95 if (!NeedsMultiLevelTiling(sch->state(), sch->GetSRef(block_rv))) {
96 return {sch};
97 }
98 sch->Annotate(block_rv, tir::attr::meta_schedule_tiling_structure, structure);
99
100 Array<Schedule> results;
101 for (auto&& state : ApplySubRules({State(sch, block_rv)})) {
102 results.push_back(std::move(state->sch));
103 }
104 return results;
105}

107std::vector<State> MultiLevelTilingNode::ApplySubRules(std::vector<State> states) {
108 states = SubRule(std::move(states), [&](State state) { return TileLoopNest(std::move(state)); });
109 states = SubRule(std::move(states), [&](State state) { return AddWriteReuse(std::move(state)); });
110 states = SubRule(std::move(states), [&](State state) { return AddReadReuse(std::move(state)); });
111 return states;
112}

114std::vector<State> MultiLevelTilingNode::AddWriteReuse(State state) const {
115 const ReuseConfig& config = this->reuse_write_;
116 if (config.req == ReuseType::kNoReuse) {
117 return {std::move(state)};
118 }
119 std::vector<int> levels = config.levels;
120 ReuseType req = config.req;
121 if (Optional<Array<Integer>> ann = tir::GetAnn<Array<Integer>>(
122 state->sch->GetSRef(state->block_rv), "meta_schedule.write_cache_level")) {
123 req = ReuseType::kMustReuse;
124 levels.clear();
125 std::transform(ann.value().begin(), ann.value().end(), std::back_inserter(levels),
126 [](auto&& v) { return v.IntValue(); });
127 }
128 std::vector<State> results;
129 if (req == ReuseType::kMayReuse) {
130 // Case 1. If the write cache is already there, we don't need to add another.
131 Array<BlockRV> consumer_rvs = state->sch->GetConsumers(state->block_rv);
132 if (consumer_rvs.size() == 1 && IsWriteCache(state->sch->GetSRef(consumer_rvs[0]))) {
133 for (int level : levels) {
134 State new_state = state->Copy();
135 const LoopRV& loop_rv = new_state->tiles[level - 1].back();
136 new_state->sch->ReverseComputeAt(consumer_rvs[0], loop_rv, true);
137 results.push_back(std::move(new_state));
138 }
139 state->write_reuse.emplace(0, consumer_rvs[0]);
140 results.push_back(state);
141 return results;
142 } else {
143 // Case 2. No write cache is added
144 State new_state = state->Copy();
145 results.emplace_back(std::move(new_state));
146 }
147 }
148
149 // Case 3. Add one write cache
150 BlockRV write_cache =
151 state->sch->CacheWrite(/*block_rv=*/state->block_rv, /*read_buffer_index=*/0,
152 /*storage_scope=*/config.scope);
153 state->write_reuse.emplace(0, write_cache);
154 for (int level : levels) {
155 State new_state = state->Copy();
156 const LoopRV& loop_rv = new_state->tiles[level - 1].back();
157 new_state->sch->ReverseComputeAt(write_cache, loop_rv, true);
158 results.push_back(std::move(new_state));
159 }
160 return results;
161}

163std::vector<State> MultiLevelTilingNode::TileLoopNest(State state) const {
164 Schedule& sch = state->sch;
165 const BlockRV& block_rv = state->block_rv;
166 // Step 1. Assuming trivial binding, pair the loops and their iter-var-types
167 Array<LoopRV> loops = sch->GetLoops(block_rv);
168 std::vector<IterVarType> iter_types = GetBlockVarTypes(sch->GetSRef(state->block_rv));
169 ICHECK_EQ(loops.size(), iter_types.size());
170 // Step 2. For each loop axis, tile it
171 int64_t spatial_loop_product = 1;
172 std::vector<Array<LoopRV>> tiles(s_indices_.size() + r_indices_.size());
173 for (int i = 0, n = loops.size(); i < n; ++i) {
174 LoopRV loop = loops[i];
175 const std::vector<int>* idx = nullptr;
176 if (iter_types[i] == IterVarType::kDataPar) {
177 idx = &s_indices_;
178 if (spatial_loop_product != -1) {
179 if (const int64_t* extent = tir::GetLoopIntExtent(sch->Get(loop).get())) {
180 spatial_loop_product *= *extent;
181 } else {
182 spatial_loop_product = -1;
183 }
184 }
185 } else if (iter_types[i] == IterVarType::kCommReduce) {
186 idx = &r_indices_;
187 } else {
188 continue;
189 }
190 // Do the split
191 int n_tiles = idx->size();
192 Array<tir::ExprRV> factors = sch->SamplePerfectTile(
193 /*loop=*/loop,
194 /*n=*/n_tiles,
195 /*max_innermost_factor=*/max_innermost_factor);
196 Array<tir::LoopRV> splits = sch->Split(/*loop=*/loop,
197 /*factors=*/{factors.begin(), factors.end()});
198 // Put every tile to its slot
199 for (int j = 0; j < n_tiles; ++j) {
200 tiles[idx->at(j)].push_back(splits[j]);
201 }
202 }
203 // Step 3. Reorder to organize the tiles
204 sch->Reorder(support::ConcatArrayList<LoopRV>(tiles.begin(), tiles.end()));
205 // Step 4. Bind the tiles to threads
206 int n_binds = std::min(tile_binds.size(), tiles.size());
207 for (int i = 0; i < n_binds; ++i) {
208 LoopRV fused = sch->Fuse(tiles[i]);
209 sch->Bind(fused, tile_binds[i]);
210 tiles[i] = {fused};
211 }
212 state->tiles = Array<Array<LoopRV>>{tiles.begin(), tiles.end()};
213 if (this->thread_warp_size_ != -1) {
214 int64_t low_inclusive = 1;
215 int64_t high_inclusive = this->max_threads_per_block_;
216 if (spatial_loop_product > 2 * this->thread_warp_size_) {
217 low_inclusive = this->thread_warp_size_;
218 }
219 sch->Annotate(block_rv, tir::attr::meta_schedule_thread_extent_low_inclusive,
220 Integer(low_inclusive));
221 sch->Annotate(block_rv, tir::attr::meta_schedule_thread_extent_high_inclusive,
222 Integer(high_inclusive));
223 }
224 return {state};
225}

227std::vector<State> MultiLevelTilingNode::AddReadReuse(State state) const {
228 const ReuseConfig& config = this->reuse_read_;
229 if (config.req == ReuseType::kNoReuse) {
230 return {std::move(state)};
231 }
232 ICHECK(config.req != ReuseType::kMayReuse);
233 const BlockRV& block_rv = state->block_rv;
234 std::vector<State> results;
235 results.reserve(config.levels.size());
236 for (int level : config.levels) {
237 State new_state = state->Copy();
238 Schedule& sch = new_state->sch;
239 const LoopRV& loop_rv = state->tiles[level - 1].back();
240 // Enumerate all buffers that are read but not written
241 std::vector<int> read_buffer_ndims = tir::GetReadBufferNDims(sch->GetSRef(block_rv));
242 for (int i = 0, n_reads = read_buffer_ndims.size(); i < n_reads; ++i) {
243 int buffer_ndim = read_buffer_ndims[i];
244 if (buffer_ndim == -1) {
245 continue;
246 }
247 // Do cache_read
248 BlockRV cache_read_block = sch->CacheRead(block_rv, i, config.scope);
249 // Insert cache_read block to the proper place
250 sch->ComputeAt(cache_read_block, loop_rv, true);
251 // Fuse the iterators of the cache_read
252 Array<LoopRV> buffer_loops = sch->GetLoops(cache_read_block);
253 LoopRV fused = sch->Fuse(Array<LoopRV>{buffer_loops.end() - buffer_ndim, //
254 buffer_loops.end()});
255 AnnotateCooperativeFetching(&sch, cache_read_block);
256 new_state->read_reuse.emplace(i, cache_read_block);
257 }
258 results.push_back(std::move(new_state));
259 }
260 return results;
261}

263void MultiLevelTilingNode::AnnotateCooperativeFetching(Schedule* sch,
264 const tir::BlockRV& block) const {
265 // Filter out invalid vector lanes according to the data type.
266 const tir::BlockNode* block_node = (*sch)->GetSRef(block)->StmtAs<tir::BlockNode>();
267 ICHECK_EQ(block_node->writes.size(), 1);
268 const runtime::DataType dtype = block_node->writes[0]->buffer->dtype;
269 std::function<bool(int)> f_filter = nullptr;
270 if (dtype == runtime::DataType::Float(32)) {
271 f_filter = [&](int vector_len) { return vector_len <= 4; };
272 } else if (dtype == runtime::DataType::Float(16)) {
273 f_filter = [&](int vector_len) {
274 return (vector_len == 1 || vector_len % 2 == 0) && vector_len <= 8;
275 };
276 } else if (dtype == runtime::DataType::Int(8)) {
277 f_filter = [&](int vector_len) { return vector_len <= 16; };
278 }
279 std::vector<int> valid_vector_lens;
280 valid_vector_lens.reserve(vector_load_lens.size());
281 if (f_filter != nullptr) {
282 std::copy_if(vector_load_lens.begin(), vector_load_lens.end(),
283 std::back_inserter(valid_vector_lens), f_filter);
284 } else {
285 valid_vector_lens = vector_load_lens;
286 }
287
288 if (!valid_vector_lens.empty()) {
289 int n = valid_vector_lens.size();
290 double prob = 1.0 / n;
291 tir::ExprRV vector_load_len =
292 (*sch)->SampleCategorical(support::AsArray<int, Integer>(valid_vector_lens),
293 Array<FloatImm>(n, FloatImm(DataType::Float(64), prob)));
294 (*sch)->Annotate(block, tir::attr::meta_schedule_cooperative_fetch, vector_load_len);
295 }
