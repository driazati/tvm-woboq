31/*!
32 * \brief Get the buffer dimensions for all the read buffers of a block, but marks the reduction
33 * buffers' dimensions as -1
34 * \param block_sref The block to be processed
35 * \return The buffer dimensions for all the read buffers of a block, except for reduction buffers
36 * \note The method is not designed for generic analysis and relies on assumptions in the scenario
37 * of multi-level tiling, so it's intentionally kept inside this file not in the analysis header
38 */
39std::vector<int> GetReadBufferNDims(const StmtSRef& block_sref) {
40 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
41 const BufferNode* write_buffer = block->writes[0]->buffer.get();
42 int n = block->reads.size();
43 std::vector<int> results(n, -1);
44 for (int i = 0; i < n; ++i) {
45 const BufferNode* read_buffer = block->reads[i]->buffer.get();
46 if (read_buffer != write_buffer) {
47 results[i] = read_buffer->shape.size();
48 }
49 }
50 return results;
51}

93// Entry of the mega rule; Inherited from ScheduleRuleNode
94Array<Schedule> MultiLevelTilingNode::Apply(const Schedule& sch, const BlockRV& block_rv) {
95 if (!NeedsMultiLevelTiling(sch->state(), sch->GetSRef(block_rv))) {
96 return {sch};
97 }
98 sch->Annotate(block_rv, tir::attr::meta_schedule_tiling_structure, structure);
99
100 Array<Schedule> results;
101 for (auto&& state : ApplySubRules({State(sch, block_rv)})) {
102 results.push_back(std::move(state->sch));
103 }
104 return results;
105}

113std::vector<State> MultiLevelTilingNode::ApplySubRules(std::vector<State> states) {
114 states = SubRule(std::move(states), [&](State state) { return TileLoopNest(std::move(state)); });
115 states = SubRule(std::move(states), [&](State state) { return AddWriteReuse(std::move(state)); });
116 states = SubRule(std::move(states), [&](State state) { return AddReadReuse(std::move(state)); });
117 return states;
118}

120std::vector<State> MultiLevelTilingNode::AddWriteReuse(State state) const {
121 const ReuseConfig& config = this->reuse_write_;
122 if (config.req == ReuseType::kNoReuse) {
123 return {std::move(state)};
124 }
125 std::vector<int> levels = config.levels;
126 ReuseType req = config.req;
127 if (Optional<Array<Integer>> ann = tir::GetAnn<Array<Integer>>(
128 state->sch->GetSRef(state->block_rv), "meta_schedule.write_cache_level")) {
129 req = ReuseType::kMustReuse;
130 levels.clear();
131 std::transform(ann.value().begin(), ann.value().end(), std::back_inserter(levels),
132 [](auto&& v) { return v.IntValue(); });
133 }
134 std::vector<State> results;
135 if (req == ReuseType::kMayReuse) {
136 // Case 1. If the write cache is already there, we don't need to add another.
137 Array<BlockRV> consumer_rvs = state->sch->GetConsumers(state->block_rv);
138 if (consumer_rvs.size() == 1 && IsWriteCache(state->sch->GetSRef(consumer_rvs[0]))) {
139 for (int level : levels) {
140 State new_state = state->Copy();
141 const LoopRV& loop_rv = new_state->tiles[level - 1].back();
142 new_state->sch->ReverseComputeAt(consumer_rvs[0], loop_rv, true);
143 results.push_back(std::move(new_state));
144 }
145 state->write_reuse.emplace(0, consumer_rvs[0]);
146 results.push_back(state);
147 return results;
148 } else {
149 // Case 2. No write cache is added
150 State new_state = state->Copy();
151 results.emplace_back(std::move(new_state));
152 }
153 }
154
155 // Case 3. Add one write cache
156 BlockRV write_cache =
157 state->sch->CacheWrite(/*block_rv=*/state->block_rv, /*read_buffer_index=*/0,
158 /*storage_scope=*/config.scope);
159 state->write_reuse.emplace(0, write_cache);
160 for (int level : levels) {
161 State new_state = state->Copy();
162 const LoopRV& loop_rv = new_state->tiles[level - 1].back();
163 new_state->sch->ReverseComputeAt(write_cache, loop_rv, true);
164 results.push_back(std::move(new_state));
165 }
166 return results;
167}

169Array<tir::LoopRV> MultiLevelTilingNode::SplitLoop(const Schedule& sch, BlockRV block, LoopRV loop,
170 int n_tiles) const {
171 Array<tir::ExprRV> factors = sch->SamplePerfectTile(
172 /*loop=*/loop,
173 /*n=*/n_tiles,
174 /*max_innermost_factor=*/max_innermost_factor);
175 Array<tir::LoopRV> splits = sch->Split(/*loop=*/loop,
176 /*factors=*/{factors.begin(), factors.end()});
177 return splits;
178}

180std::vector<State> MultiLevelTilingNode::TileLoopNest(State state) const {
181 Schedule& sch = state->sch;
182 const BlockRV& block_rv = state->block_rv;
183 // Step 1. Assuming trivial binding, pair the loops and their iter-var-types
184 Array<LoopRV> loops = sch->GetLoops(block_rv);
185 std::vector<IterVarType> iter_types = GetBlockVarTypes(sch->GetSRef(state->block_rv));
186 ICHECK_EQ(loops.size(), iter_types.size());
187 // Step 2. For each loop axis, tile it
188 int64_t spatial_loop_product = 1;
189 std::vector<Array<LoopRV>> tiles(s_indices_.size() + r_indices_.size());
190 for (int i = 0, n = loops.size(); i < n; ++i) {
191 LoopRV loop = loops[i];
192 const std::vector<int>* idx = nullptr;
193
194 if (iter_types[i] == IterVarType::kDataPar) {
195 idx = &s_indices_;
196 if (spatial_loop_product != -1) {
197 if (const int64_t* extent = tir::GetLoopIntExtent(sch->Get(loop).get())) {
198 spatial_loop_product *= *extent;
199 } else {
200 spatial_loop_product = -1;
201 }
202 }
203 } else if (iter_types[i] == IterVarType::kCommReduce) {
204 idx = &r_indices_;
205 } else {
206 continue;
207 }
208
209 const int n_tiles = idx->size();
210
211 if (n_tiles == 1) {
212 tiles[idx->at(0)].push_back(loop);
213 } else {
214 auto splits = SplitLoop(sch, block_rv, loop, n_tiles);
215
216 // Put every tile to its slot
217 for (int j = 0; j < n_tiles; ++j) {
218 tiles[idx->at(j)].push_back(splits[j]);
219 }
220 }
221 }
222 // Step 3. Reorder to organize the tiles
223 sch->Reorder(support::ConcatArrayList<LoopRV>(tiles.begin(), tiles.end()));
224 // Step 4. Bind the tiles to threads
225 int n_binds = std::min(tile_binds.size(), tiles.size());
226 for (int i = 0; i < n_binds; ++i) {
227 LoopRV fused = sch->Fuse(tiles[i]);
228 sch->Bind(fused, tile_binds[i]);
229 tiles[i] = {fused};
230 }
231 state->tiles = Array<Array<LoopRV>>{tiles.begin(), tiles.end()};
232 if (this->thread_warp_size_ != -1) {
233 int64_t low_inclusive = 1;
234 int64_t high_inclusive = this->max_threads_per_block_;
235 if (spatial_loop_product > 2 * this->thread_warp_size_) {
236 low_inclusive = this->thread_warp_size_;
237 }
238 sch->Annotate(block_rv, tir::attr::meta_schedule_thread_extent_low_inclusive,
239 Integer(low_inclusive));
240 sch->Annotate(block_rv, tir::attr::meta_schedule_thread_extent_high_inclusive,
241 Integer(high_inclusive));
242 }
243 return {state};
244}

246std::vector<State> MultiLevelTilingNode::AddReadReuse(State state) const {
247 const ReuseConfig& config = this->reuse_read_;
248 if (config.req == ReuseType::kNoReuse) {
249 return {std::move(state)};
250 }
251 ICHECK(config.req != ReuseType::kMayReuse);
252 const BlockRV& block_rv = state->block_rv;
253 std::vector<State> results;
254 results.reserve(config.levels.size());
255 for (int level : config.levels) {
256 State new_state = state->Copy();
257 Schedule& sch = new_state->sch;
258 const LoopRV& loop_rv = state->tiles[level - 1].back();
259 // Enumerate all buffers that are read but not written
260 std::vector<int> read_buffer_ndims = tir::GetReadBufferNDims(sch->GetSRef(block_rv));
261 for (int i = 0, n_reads = read_buffer_ndims.size(); i < n_reads; ++i) {
262 int buffer_ndim = read_buffer_ndims[i];
263 if (buffer_ndim == -1) {
264 continue;
265 }
266 // Do cache_read
267 BlockRV cache_read_block = sch->CacheRead(block_rv, i, config.scope);
268 // Insert cache_read block to the proper place
269 sch->ComputeAt(cache_read_block, loop_rv, true);
270 // Fuse the iterators of the cache_read
271 Array<LoopRV> buffer_loops = sch->GetLoops(cache_read_block);
272 LoopRV fused = sch->Fuse(Array<LoopRV>{buffer_loops.end() - buffer_ndim, //
273 buffer_loops.end()});
274 AnnotateCooperativeFetching(&sch, cache_read_block);
275 new_state->read_reuse.emplace(i, cache_read_block);
276 }
277 results.push_back(std::move(new_state));
278 }
279 return results;
280}

282void MultiLevelTilingNode::AnnotateCooperativeFetching(Schedule* sch,
283 const tir::BlockRV& block) const {
284 // Filter out invalid vector lanes according to the data type.
285 const tir::BlockNode* block_node = (*sch)->GetSRef(block)->StmtAs<tir::BlockNode>();
286 ICHECK_EQ(block_node->writes.size(), 1);
287 const runtime::DataType dtype = block_node->writes[0]->buffer->dtype;
288 std::function<bool(int)> f_filter = nullptr;
289 if (dtype == runtime::DataType::Float(32)) {
290 f_filter = [&](int vector_len) { return vector_len <= 4; };
291 } else if (dtype == runtime::DataType::Float(16)) {
292 f_filter = [&](int vector_len) {
293 return (vector_len == 1 || vector_len % 2 == 0) && vector_len <= 8;
294 };
295 } else if (dtype == runtime::DataType::Int(8)) {
296 f_filter = [&](int vector_len) { return vector_len <= 16; };
297 }
298 std::vector<int> valid_vector_lens;
299 valid_vector_lens.reserve(vector_load_lens.size());
300 if (f_filter != nullptr) {
301 std::copy_if(vector_load_lens.begin(), vector_load_lens.end(),
302 std::back_inserter(valid_vector_lens), f_filter);
303 } else {
304 valid_vector_lens = vector_load_lens;
305 }
306
307 if (!valid_vector_lens.empty()) {
308 int n = valid_vector_lens.size();
309 double prob = 1.0 / n;
310 tir::ExprRV vector_load_len =
311 (*sch)->SampleCategorical(support::AsArray<int, Integer>(valid_vector_lens),
312 Array<FloatImm>(n, FloatImm(DataType::Float(64), prob)));
313 (*sch)->Annotate(block, tir::attr::meta_schedule_cooperative_fetch, vector_load_len);
314 }
