29void BindBlockThreadIdx(const tir::Schedule& sch, const tir::BlockRV& block_rv,
30 int64_t max_threadblocks, int64_t max_threads_per_block,
31 std::function<tir::ExprRV(int64_t)> get_factor) {
32 using namespace tvm::tir;
33 StmtSRef block_sref = sch->GetSRef(block_rv);
34 if (block_sref->parent == nullptr) {
35 return;
36 }
37 if (tir::HasBeenMultiLevelTiled(block_sref)) {
38 return;
39 }
40 Array<StmtSRef> loops = tir::GetLoops(block_sref);
41 int n = loops.size();
42 int i_block_idx = -1;
43 int i_thread_idx = -1;
44 int i_multi_child = -1;
45 int i_spatial_loop = -1;
46 for (int i = 0; i < n; ++i) {
47 const StmtSRef& loop_sref = loops[i];
48 const ForNode* loop = TVM_SREF_TO_FOR(loop_sref);
49 runtime::ThreadScope thread_scope = GetThreadScope(loop);
50 if (IsBlockIdx(thread_scope)) {
51 if (i_block_idx == -1) {
52 i_block_idx = i;
53 }
54 }
55 if (IsThreadIdx(thread_scope)) {
56 if (i_thread_idx == -1) {
57 i_thread_idx = i;
58 }
59 }
60 if (loop->kind != ForKind::kSerial) {
61 if (i_multi_child == -1) {
62 i_multi_child = i;
63 }
64 }
65 if (!IsSingleStmt(loop->body)) {
66 if (i_multi_child == -1) {
67 i_multi_child = i + 1;
68 }
69 }
70 if (GetLoopIterType(loop_sref) == IterVarType::kDataPar) {
71 if (i_spatial_loop == i - 1) {
72 ++i_spatial_loop;
73 }
74 }
75 }
76 if (i_multi_child == -1) {
77 i_multi_child = n;
78 }
79 if (i_block_idx != -1 && i_thread_idx != -1) {
80 return;
81 }
82 if (i_block_idx != -1 && i_thread_idx == -1) {
83 ICHECK(false) << "Unsupported case, where blockIdx is bound but threadIdx is not";
84 throw;
85 }
86 LoopRV loop_rv{nullptr};
87 {
88 Array<LoopRV> loop_rvs = sch->GetLoops(block_rv);
89 if (i_spatial_loop == -1) {
90 LoopRV spatial_loop_rv{nullptr};
91 if (loop_rvs.empty()) {
92 spatial_loop_rv = sch->AddUnitLoop(block_rv);
93 } else {
94 spatial_loop_rv = sch->AddUnitLoop(loop_rvs[0]);
95 }
96 loop_rvs.insert(loop_rvs.begin(), spatial_loop_rv);
97 i_spatial_loop = 0;
98 if (i_block_idx != -1) {
99 i_block_idx += 1;
100 }
101 if (i_thread_idx != -1) {
102 i_thread_idx += 1;
103 }
104 if (i_multi_child != -1) {
105 i_multi_child += 1;
106 }
107 }
108 if (i_block_idx == -1 && i_thread_idx != -1) {
109 int num_fuse = std::min(std::min(i_multi_child, i_thread_idx), i_spatial_loop + 1);
110 Array<LoopRV> loop_rvs = sch->GetLoops(block_rv);
111 loop_rv = sch->Fuse({loop_rvs.begin(), loop_rvs.begin() + num_fuse});
112 sch->Bind(loop_rv, "blockIdx.x");
113 return;
114 } else { // i_block_idx == -1 && i_thread_idx == -1
115 int num_fuse = std::min(i_multi_child, i_spatial_loop + 1);
116 loop_rv = sch->Fuse({loop_rvs.begin(), loop_rvs.begin() + num_fuse});
117 }
118 }
119 int64_t extent = -1;
120 if (const int64_t* e = GetLoopIntExtent(sch->Get(loop_rv).get())) {
121 extent = *e;
122 } else {
123 extent = std::numeric_limits<int64_t>::max();
124 }
125 if (extent <= max_threadblocks * max_threads_per_block) {
126 ExprRV factor = get_factor(std::min(extent, max_threads_per_block));
127 Array<LoopRV> splits = sch->Split(loop_rv, {NullOpt, factor});
128 ICHECK_EQ(splits.size(), 2);
129 sch->Bind(splits[0], "blockIdx.x");
130 sch->Bind(splits[1], "threadIdx.x");
131 } else {
132 Array<LoopRV> splits = sch->Split(loop_rv, {NullOpt,
133 Integer(max_threadblocks), //
134 Integer(max_threads_per_block)});
135 ICHECK_EQ(splits.size(), 3);
136 sch->Reorder({splits[1], splits[2], splits[0]});
137 sch->Bind(splits[1], "blockIdx.x");
138 sch->Bind(splits[2], "threadIdx.x");
139 }
140}
141
142std::function<tir::ExprRV(int64_t)> MakeFactorSampler(tir::Schedule sch,
143 Array<Integer> thread_extents) {
144 return [sch = std::move(sch),
145 thread_extents = std::move(thread_extents)](int64_t max_extent) -> tir::ExprRV {
146 Array<Integer> extents;
147 extents.reserve(thread_extents.size());
148 for (const Integer extent : thread_extents) {
149 if (extent->value <= max_extent) {
150 extents.push_back(extent);
151 }
152 }
153 int n = extents.size();
154 if (n == 0) {
155 return Integer(max_extent);
156 }
157 if (n == 1) {
158 return Integer(extents[0]);
159 }
160 Array<FloatImm> probs(n, FloatImm(DataType::Float(64), 1.0 / n));
161 return sch->SampleCategorical(extents, probs);
162 };

165class AutoBindNode : public ScheduleRuleNode {
166 public:
167 // Inherited from ScheduleRuleNode
168 void InitializeWithTuneContext(const TuneContext& context) final {
169 CHECK(context->target.defined()) << "ValueError: target is not defined";
170 Optional<Integer> max_threads_per_block =
171 context->target.value()->GetAttr<Integer>("max_threads_per_block");
172 CHECK(max_threads_per_block.defined())
173 << "ValueError: missing attribute `max_threads_per_block` in the target";
174 this->max_threads_per_block_ = max_threads_per_block.value().IntValue();
175 }
176
177 // Inherited from ScheduleRuleNode
178 Array<tir::Schedule> Apply(const tir::Schedule& sch, const tir::BlockRV& block_rv) final;
179
180 // Inherited from ScheduleRuleNode
181 ScheduleRule Clone() const final {
182 ObjectPtr<AutoBindNode> n = make_object<AutoBindNode>(*this);
183 return ScheduleRule(n);
184 }
185
186 public:
187 /*! \brief The max number of threads per block from Target */
188 int64_t max_threads_per_block_ = -1;
189 /*! \brief The max number of threadblocks in the cuda device */
190 int64_t max_threadblocks_ = -1;
191 /*! \brief thread_extents Candidates of thread axis extent. */
192 Array<Integer> thread_extents_;
193
194 void VisitAttrs(tvm::AttrVisitor* v) {
195 // `max_threads_per_block_` is not visited
196 // `max_threadblocks_` is not visited
197 // `thread_extents_` is not visited
198 }
199
200 static constexpr const char* _type_key = "meta_schedule.AutoBind";
201 TVM_DECLARE_FINAL_OBJECT_INFO(AutoBindNode, ScheduleRuleNode);
202};
203
204Array<tir::Schedule> AutoBindNode::Apply(const tir::Schedule& sch, const tir::BlockRV& block_rv) {
205 ICHECK_NE(this->max_threads_per_block_, -1);
206 auto get_factor = MakeFactorSampler(sch, this->thread_extents_);
207 BindBlockThreadIdx(sch, block_rv, max_threadblocks_, max_threads_per_block_, get_factor);
208 return {sch};
209}
210
211ScheduleRule ScheduleRule::AutoBind(int max_threadblocks, Array<Integer> thread_extents,
212 int max_threads_per_block) {
213 ObjectPtr<AutoBindNode> n = make_object<AutoBindNode>();
214 n->max_threadblocks_ = max_threadblocks;
215 n->max_threads_per_block_ = max_threads_per_block;
216 n->thread_extents_ = std::move(thread_extents);
217 return ScheduleRule(n);
218}
