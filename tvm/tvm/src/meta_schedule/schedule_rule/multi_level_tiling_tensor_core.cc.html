35struct TensorCoreIntrinGroup {
36 String init_intrin;
37 String load_a_intrin;
38 String load_b_intrin;
39 String compute_intrin;
40 String store_intrin;
41
42 /*! \brief Create TensorCoreIntrinGroup from config in a map. The map should contains the
43 * following keys:
44 * - init
45 * - load_a
46 * - load_b
47 * - compute
48 * - store
49 * The values of the keys should be the names of the corresponding intrinsics and should be
50 * registered via TensorIntrin.Register beforehand.
51 */
52 static TensorCoreIntrinGroup FromConfig(const Map<String, String>& config);
53};
54
55TensorCoreIntrinGroup TensorCoreIntrinGroup::FromConfig(const Map<String, String>& config) {
56 auto f_initialize_intrin = [&config](String key_name, String* intrin_name) {
57 CHECK(config.count(key_name)) << "ValueError: " << key_name << " is not set.";
58 *intrin_name = config.at(key_name);
59 // Check the existence of the intrin
60 tir::TensorIntrin::Get(*intrin_name);
61 };
62 TensorCoreIntrinGroup intrin_group;
63 f_initialize_intrin("init", &intrin_group.init_intrin);
64 f_initialize_intrin("load_a", &intrin_group.load_a_intrin);
65 f_initialize_intrin("load_b", &intrin_group.load_b_intrin);
66 f_initialize_intrin("compute", &intrin_group.compute_intrin);
67 f_initialize_intrin("store", &intrin_group.store_intrin);
68 return intrin_group;
69}

119/*!
120 * \brief Extension of MultiLevelTiling for auto-tensorizing with a single group of tensor core
121 * intrinsics.
122 */
123class MultiLevelTilingTensorCoreNode : public MultiLevelTilingNode {
124 private:
125 // SubRule: Add tensorization-related transformations
126 inline std::vector<State> TransformForTensorization(TensorCoreState state) const;
127 // Subrule: Add tensorized load
128 inline std::vector<State> AddReadReuseTensorCore(TensorCoreState state) const;
129 // Subrule: Add tensorized store
130 inline std::vector<State> AddWriteReuseTensorCore(TensorCoreState state) const;
131 // Subrule: Add software pipeline
132 inline std::vector<State> AddSoftwarePipeline(TensorCoreState state) const;
133
134 // Override ApplySubRules to apply tensorization-specific sub-rules
135 std::vector<State> ApplySubRules(std::vector<State> states) final;
136
137 // Override Apply to apply tensorization-specific analysis before applying sub-rules
138 Array<Schedule> Apply(const Schedule& sch, const BlockRV& block_rv) final;
139
140 // Inherited from ScheduleRuleNode
141 ScheduleRule Clone() const final {
142 ObjectPtr<MultiLevelTilingTensorCoreNode> n =
143 make_object<MultiLevelTilingTensorCoreNode>(*this);
144 return ScheduleRule(n);
145 }

147 /*!
148 * \brief Transform and tensorize with the given tensor intrin
149 * \param state The state of the meta schedule rule
150 * \param intrin_name The name of the tensor intrin
151 * \return The loop to be tensorized. NullOpt if the workload can't be tensorized.
152 */
153 Optional<LoopRV> TransformWithTensorIntrin(TensorCoreStateNode* state,
154 const String& intrin_name) const;

156 /*!
157 * \brief Tile, blockize and annotate for tensorization with the given intrin
158 * \param block_rv The block to be tensorized
159 * \param intrin_name The name of the tensor intrin
160 */
161 void TileAndAnnotateTensorize(Schedule* sch, const BlockRV& block_rv,
162 const String& intrin_name) const;

164 public:
165 /*! \brief The candidate tensor core intrin groups to apply */
166 std::vector<TensorCoreIntrinGroup> intrin_groups;
167 /*! \brief Whether to use software pipeline */
168 bool use_software_pipeline = false;
169 static constexpr const char* _type_key = "meta_schedule.MultiLevelTilingTensorCore";
170 TVM_DECLARE_FINAL_OBJECT_INFO(MultiLevelTilingTensorCoreNode, MultiLevelTilingNode);

172 private:
173};

175// Entry of the mega rule; Inherited from ScheduleRuleNode
176Array<Schedule> MultiLevelTilingTensorCoreNode::Apply(const Schedule& sch,
177 const BlockRV& block_rv) {
178 if (!NeedsMultiLevelTiling(sch->state(), sch->GetSRef(block_rv))) {
179 return {sch};
180 }
181
182 std::unordered_map<int, tir::AutoTensorizeMappingInfo> intrin_group_to_mapping_info;
183 for (int i = 0, n = intrin_groups.size(); i < n; ++i) {
184 TensorCoreIntrinGroup intrin_group = intrin_groups[i];
185 Optional<tir::AutoTensorizeMappingInfo> mapping_info = tir::GetAutoTensorizeMappingInfo(
186 sch->state(), sch->GetSRef(block_rv),
187 tir::TensorIntrin::Get(intrin_groups[i].compute_intrin).value()->desc);
188 if (mapping_info.defined()) {
189 intrin_group_to_mapping_info.emplace(i, mapping_info.value());
190 }
191 }
192
193 if (intrin_group_to_mapping_info.empty()) {
194 // No tensor intrinsics can be applied.
195 return {sch};
196 }
197
198 // Save the original schedule so that we can roll back transformations if tensorization
199 // fail.
200 Schedule original_sch = sch;
201
202 std::vector<State> initial_states;
203 for (const auto& kv : intrin_group_to_mapping_info) {
204 const TensorCoreIntrinGroup& intrin_group = intrin_groups[kv.first];
205 const tir::AutoTensorizeMappingInfo& mapping_info = kv.second;
206 Schedule new_sch = sch->Copy();
207 new_sch->Annotate(block_rv, tir::attr::meta_schedule_tiling_structure, structure);
208 initial_states.push_back(TensorCoreState(intrin_group, mapping_info, new_sch, block_rv));
209 }
210 Array<Schedule> results;
211 for (auto&& state : ApplySubRules(initial_states)) {
212 TVM_PY_LOG(INFO, logging_func) << "Sketch " << results.size() << ": tensorizing with "
213 << state.as<TensorCoreStateNode>()->intrin_group.compute_intrin;
214 results.push_back(std::move(state->sch));
215 }
216 if (results.empty()) {
217 TVM_PY_LOG(INFO, logging_func) << "The workload cannot be tensorized.";
218 return {original_sch};
219 }
220 return results;
221}

223std::vector<State> MultiLevelTilingTensorCoreNode::ApplySubRules(std::vector<State> states) {
224 states = SubRule(std::move(states), [&](State state) {
225 return TransformForTensorization(Downcast<TensorCoreState>(state));
226 });
227 states = SubRule(std::move(states), [&](State state) { return TileLoopNest(state); });
228 states = SubRule(std::move(states), [&](State state) { return AddWriteReuse(state); });
229 states = SubRule(std::move(states), [&](State state) {
230 return AddWriteReuseTensorCore(Downcast<TensorCoreState>(state));
231 });
232 states = SubRule(std::move(states), [&](State state) { return AddReadReuse(state); });
233 states = SubRule(std::move(states), [&](State state) {
234 return AddReadReuseTensorCore(Downcast<TensorCoreState>(state));
235 });
236 states = SubRule(std::move(states), [&](State state) {
237 return AddSoftwarePipeline(Downcast<TensorCoreState>(state));
238 });
239 return states;
240}

242void MultiLevelTilingTensorCoreNode::TileAndAnnotateTensorize(Schedule* sch,
243 const BlockRV& block_rv,
244 const String& intrin_name) const {
245 Optional<LoopRV> loop = TileWithTensorIntrin(*sch, block_rv, intrin_name).value();
246 ICHECK(loop.defined());
247 BlockRV blockized_outer = (*sch)->Blockize(loop.value());
248 (*sch)->Annotate(blockized_outer, tir::attr::meta_schedule_auto_tensorize, intrin_name);
249}

251std::vector<State> MultiLevelTilingTensorCoreNode::AddWriteReuseTensorCore(
252 TensorCoreState state) const {
253 // Add the cache write stage for Tensor Core
254 int level = r_indices_.front() - 1;
255 const LoopRV& loop = state->tiles[level].back();
256 Schedule& sch = state->sch;
257 auto cache_write = sch->CacheWrite(state->block_rv, 0, "wmma.accumulator");
258 sch->ReverseComputeAt(cache_write, loop, true);
259
260 if (state->write_reuse.count(0)) {
261 AnnotateCooperativeFetching(&sch, state->write_reuse[0]);
262 }
263 sch->ReverseComputeInline(state->tensor_core_reindex_store);
264 TileAndAnnotateTensorize(&sch, cache_write, state->intrin_group.store_intrin);
265 return {state};
266}

268std::vector<State> MultiLevelTilingTensorCoreNode::AddReadReuseTensorCore(
269 TensorCoreState state) const {
270 const Array<LoopRV>& r_tiles = state->tiles[r_indices_[1]];
271 Schedule& sch = state->sch;
272 ICHECK(!r_tiles.empty()) << "ValueError: Cannot find the suitable reduction loop in the block";
273
274 auto f_tensorize_load = [&](int read_index, String scope, String intrin_name) {
275 auto cache_read = sch->CacheRead(state->block_rv, read_index, scope);
276 state->sch->ComputeAt(cache_read, r_tiles.back(), true);
277 TileAndAnnotateTensorize(&sch, cache_read, intrin_name);
278 };
279
280 f_tensorize_load(0, "wmma.matrix_a", state->intrin_group.load_a_intrin);
281 f_tensorize_load(1, "wmma.matrix_b", state->intrin_group.load_b_intrin);
282 sch->ComputeInline(state->tensor_core_reindex_A);
283 sch->ComputeInline(state->tensor_core_reindex_B);
284
285 for (int i = 0; i < 2; ++i) {
286 const tir::BlockRV cache_read = state->read_reuse.at(i);
287 const tir::BlockNode* cache_read_block = sch->GetSRef(cache_read)->StmtAs<tir::BlockNode>();
288 tir::Buffer cache_read_buffer = tir::GetNthAccessBuffer(
289 sch->state(), GetRef<tir::Block>(cache_read_block), 0, tir::BufferIndexType::kWrite);
290 const DataType& dtype = cache_read_buffer->dtype;
291 if (dtype.is_float16()) {
292 sch->StorageAlign(cache_read, 0, -2, 32, 8);
293 } else if (dtype.is_int() && dtype.bits() == 8) {
294 sch->StorageAlign(cache_read, 0, -2, 32, 16);
295 } else {
296 TVM_PY_LOG(WARNING, logging_func) << "StorageAlign is not applied for data type " << dtype
297 << ", shared memory accesses might be inefficient.";
298 }
299 }
300 return {state};
301}

303std::vector<State> MultiLevelTilingTensorCoreNode::AddSoftwarePipeline(
304 TensorCoreState state) const {
305 if (!use_software_pipeline) {
306 return {state};
307 }
308 // The current config is not suitable for software pipelining.
309 if (r_indices_.size() < 2) {
310 return {state};
311 }
312
313 Schedule& sch = state->sch;
314 // Check reduction length after blockize.
315 int64_t reduction_length = 1;
316 for (int r_index : r_indices_) {
317 const Array<LoopRV>& tiles = state->tiles[r_index];
318 for (const LoopRV& tile : tiles) {
319 const auto* extent = sch->Get(tile)->extent.as<IntImmNode>();
320 ICHECK(extent != nullptr) << "Dynamic extent is not supported.";
321 reduction_length *= extent->value;
322 }
323 }
324 if (reduction_length <= 1) {
325 return {state};
326 }
327
328 // Add local stage and double buffering
329 for (int i = 0; i < 2; ++i) {
330 const tir::BlockRV cache_read = state->read_reuse.at(i);
331 sch->Annotate(cache_read, tir::attr::manifest_shared_memory_local_stage, Integer(1));
332 sch->Annotate(cache_read, tir::attr::double_buffer_scope, Integer(0));
333 }
334
335 // Add annotations of software pipeline
336 //
337 // Before pipelining, the original loop can be expressed as the pseudo code below:
338 //
339 // for k0 in [0, K0):
340 // load tile k0 to registers
341 // load tile k0 from registers to shared memory
342 //
343 // for k1 in [0, K1):
344 // load fragment k1 of tile k0
345 // compute matmul with fragment k1
346 //
347
348 // Inner software pipeline: Prefetch to tensor core fragment by one iteration
349 // The following annotation for the inner loop is equivalent the pesudo code below:
350 //
351 // Pipelined inner loop:
352 //
353 // prologue:
354 // load fragment 0
355 // body:
356 // for k1 in [0, K1 - 1):
357 // load fragment k1 + 1
358 // compute matmul with fragment k1
359 // epilogue:
360 // compute matmul with fragment K1 - 1
361 //
362 sch->Annotate(state->tiles[r_indices_[1]].back(), tir::attr::software_pipeline_stage,
363 Array<Integer>{0, 0, 1});
364 sch->Annotate(state->tiles[r_indices_[1]].back(), tir::attr::software_pipeline_order,
365 Array<Integer>{0, 1, 2});
366 // Outer software pipeline: Interleave the outer loop with the (pipelined) inner loop.
367 // The prefetching stage of the inner pipeline is executed by one iteration in the outer loop.
368 // The following annotation for the outer loop is equivalent the pesudo code below:
369 //
370 // Pipelined outer loop with nested inner pipeline:
371 //
372 // prologue:
373 // load tile 0 to registers
374 // load tile 0 from registers to shared memory
375 //
376 // // prologue of the inner pipeline
377 // load fragment 0 of tile 0
378 //
379 // body:
380 // for k0 in [0, K0 - 1):
381 // load tile k0 + 1 to registers
382 //
383 // // body of the inner pipeline
384 // for k1 in [0, K1 - 1):
385 // load fragment k1 + 1 of tile k0
386 // compute matmul with fragment k1 of tile k0
387 //
388 // load tile k0 + 1 from registers to shared memory
389 //
390 // // prologue of the inner pipeline
391 // load fragment 0 of tile k0 + 1
392 //
393 // // epilogue of the inner pipeline
394 // compute matmul with fragment K1 - 1 of tile k0
395 //
396 // epilogue:
397 //
398 // // body of the inner pipeline
399 // for k1 in [0, K1 - 1):
400 // load fragment k1 + 1 of tile K0 - 1
401 // compute matmul with fragment k1 of tile K0 - 1
402 //
403 // // epilogue of the inner pipeline
404 // compute matmul with fragment K1 - 1 of tile K0 - 1
405 //
406 sch->Annotate(state->tiles[r_indices_[0]].back(), tir::attr::software_pipeline_stage,
407 Array<Integer>{0, 0, 0, 0, 0, 1, 1});
408 sch->Annotate(state->tiles[r_indices_[0]].back(), tir::attr::software_pipeline_order,
409 Array<Integer>{0, 3, 1, 4, 5, 2, 6});
410
411 return {state};
412}

414Optional<LoopRV> MultiLevelTilingTensorCoreNode::TransformWithTensorIntrin(
415 TensorCoreStateNode* state, const String& intrin_name) const {
416 BlockRV block_rv = state->block_rv;
417 const tir::AutoTensorizeMappingInfo& mapping_info = state->mapping_info;
418 tir::StmtSRef block_sref = state->sch->GetSRef(state->block_rv);
419
420 // Add reindex stages
421 const tir::BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
422 // Hold the reference of the block before reindex
423 const tir::Block block_before_reindex = GetRef<tir::Block>(block);
424 if (block->reads.size() != 2 || block->writes.size() != 1) {
425 // only matmul-like computation is allowed
426 return NullOpt;
427 }
428 state->tensor_core_reindex_store =
429 state->sch->ReIndex(state->block_rv, 0, tir::BufferIndexType::kWrite);
430 state->tensor_core_reindex_A =
431 state->sch->ReIndex(state->block_rv, 0, tir::BufferIndexType::kRead);
432 state->tensor_core_reindex_B =
433 state->sch->ReIndex(state->block_rv, 1, tir::BufferIndexType::kRead);
434
435 // Transform the layout of reindex buffers accordingly.
436 // The index map defines the mapping for the computation block. We need to extract the sub index
437 // map to transform the load and store block.
438 ICHECK_EQ(mapping_info->mappings.size(), 1U); // assume only one mapping is present
439 const tir::IndexMap& index_map = mapping_info->mappings[0];
440
441 // Find the correspondence between block iters and the iters in the index map.
442 std::unordered_map<tir::Var, tir::Var, ObjectPtrHash, ObjectPtrEqual> lhs_to_index_map_src;
443 std::unordered_map<tir::Var, PrimExpr, ObjectPtrHash, ObjectPtrEqual> rhs_to_index_map_tgt;
444 std::unordered_set<tir::Var, ObjectPtrHash, ObjectPtrEqual> unmapped_index_map_src;
445 ICHECK_EQ(mapping_info->lhs_iters.size(), index_map->initial_indices.size());
446 for (int i = 0; i < static_cast<int>(mapping_info->lhs_iters.size()); ++i) {
447 lhs_to_index_map_src[mapping_info->lhs_iters[i]->var] = index_map->initial_indices[i];
448 }
449 // The number of result iters in the index map is equal or more than the number of rhs (the
450 // tensor intrin) iters. When there are extra iters, these iters represent unmapped iters from
451 // the lhs. They will be skipped during pattern matching for tensorization. An example of such
452 // case is batch matmul, the batch dimension is kept after layout transformations and it will be
453 // kept as a outer loop after tensorization.
454 int offset = static_cast<int>(index_map->final_indices.size()) -
455 static_cast<int>(mapping_info->rhs_iters.size());
456 ICHECK_GE(offset, 0);
457 for (int i = 0; i < offset; ++i) {
458 const tir::VarNode* var_ptr = index_map->final_indices[i].as<tir::VarNode>();
459 ICHECK(var_ptr != nullptr);
460 unmapped_index_map_src.insert(GetRef<tir::Var>(var_ptr));
461 }
462 for (int i = offset; i < static_cast<int>(index_map->final_indices.size()); ++i) {
463 rhs_to_index_map_tgt[mapping_info->rhs_iters[i - offset]->var] = index_map->final_indices[i];
464 }
465
466 auto f_get_sub_index_map = [&](const tir::Buffer& lhs_buffer, const tir::Region& lhs_region) {
467 std::vector<tir::Var> sub_index_map_src;
468 std::vector<PrimExpr> sub_index_map_tgt;
469 const tir::Buffer& rhs_buffer = mapping_info->lhs_buffer_map[lhs_buffer];
470 for (const Range& range : lhs_region) {
471 ICHECK(tir::is_one(range->extent));
472 const tir::VarNode* var_ptr = range->min.as<tir::VarNode>();
473 ICHECK(var_ptr != nullptr);
474 const tir::Var& lhs_representer = lhs_to_index_map_src[GetRef<tir::Var>(var_ptr)];
475 sub_index_map_src.push_back(lhs_representer);
476 if (unmapped_index_map_src.count(lhs_representer)) {
477 sub_index_map_tgt.push_back(lhs_representer);
478 }
479 }
480 for (size_t i = 0; i < mapping_info->rhs_buffer_indices[rhs_buffer].size(); ++i) {
481 const tir::VarNode* var = mapping_info->rhs_buffer_indices[rhs_buffer][i].as<tir::VarNode>();
482 ICHECK(var != nullptr);
483 sub_index_map_tgt.push_back(rhs_to_index_map_tgt[GetRef<tir::Var>(var)]);
484 }
485 return tir::IndexMap(sub_index_map_src, sub_index_map_tgt);
486 };
487
488 std::unordered_set<tir::Buffer, ObjectPtrHash, ObjectPtrEqual> visited_buffers;
489
490 auto f_transform_buffer_layout = [&](tir::BufferIndexType index_type, int buffer_index) {
491 const tir::Buffer& lhs_buffer = tir::GetNthAccessBuffer(
492 state->sch->state(), block_before_reindex, buffer_index, index_type);
493 if (visited_buffers.count(lhs_buffer)) {
494 return;
495 }
496 visited_buffers.insert(lhs_buffer);
497 // Refresh block pointer (block sref is not invalidated)
498 block = TVM_SREF_TO_BLOCK(block_sref);
499 const tir::BufferRegion& reindexed_buffer_region = tir::GetNthAccessBufferRegion(
500 state->sch->state(), GetRef<tir::Block>(block), buffer_index, index_type);
501 auto sub_index_map = f_get_sub_index_map(lhs_buffer, reindexed_buffer_region->region);
502 state->sch->TransformLayout(state->block_rv, buffer_index, index_type, sub_index_map, NullOpt);
503 };
504
505 for (int i = 0, n = block_before_reindex->reads.size(); i < n; ++i) {
506 f_transform_buffer_layout(tir::BufferIndexType::kRead, i);
507 }
508 for (int i = 0, n = block_before_reindex->writes.size(); i < n; ++i) {
509 f_transform_buffer_layout(tir::BufferIndexType::kWrite, i);
510 }
511
512 // Transform the layout of current block and reindex blocks
513 state->sch->TransformBlockLayout(state->tensor_core_reindex_store, index_map);
514 state->sch->TransformBlockLayout(state->tensor_core_reindex_A, index_map);
515 state->sch->TransformBlockLayout(state->tensor_core_reindex_B, index_map);
516 state->sch->TransformBlockLayout(state->block_rv, index_map);
517
518 return tir::TileWithTensorIntrin(state->sch, state->block_rv, intrin_name,
519 /*allow_padding=*/true);
520}
521
522inline std::vector<State> MultiLevelTilingTensorCoreNode::TransformForTensorization(
523 TensorCoreState state) const {
524 // Do reindex and layout transformations.
525 Optional<LoopRV> transformed_loop_rv =
526 TransformWithTensorIntrin(state.operator->(), state->intrin_group.compute_intrin);
527 if (!transformed_loop_rv.defined()) {
528 // The workload can't be tensorized.
529 return {};
530 }
531
532 // Do blockize
533 state->block_rv = state->sch->Blockize(transformed_loop_rv.value());
534
535 // Add annotations for post processors.
536 state->sch->Annotate(state->block_rv, tir::attr::meta_schedule_auto_tensorize,
537 state->intrin_group.compute_intrin);
538 state->sch->Annotate(state->block_rv, tir::attr::meta_schedule_auto_tensorize_init,
539 state->intrin_group.init_intrin);
540 state->sch->Annotate(state->block_rv, tir::attr::warp_execution, Integer(1));
541 return {std::move(state)};
542}

544ScheduleRule ScheduleRule::MultiLevelTilingTensorCore(
545 Array<Map<String, String>> intrin_groups, String structure, Optional<Array<String>> tile_binds,
546 Optional<Integer> max_innermost_factor, Optional<Array<Integer>> vector_load_lens,
547 Optional<Map<String, ObjectRef>> reuse_read, Optional<Map<String, ObjectRef>> reuse_write,
548 bool use_software_pipeline) {
549 auto node = MultiLevelTilingInitCommon<MultiLevelTilingTensorCoreNode>(
550 structure, tile_binds, max_innermost_factor, vector_load_lens, reuse_read, reuse_write);
551
552 node->intrin_groups.reserve(intrin_groups.size());
553 for (const auto& intrin_group_config : intrin_groups) {
554 node->intrin_groups.emplace_back(TensorCoreIntrinGroup::FromConfig(intrin_group_config));
555 }
556 node->use_software_pipeline = use_software_pipeline;
557 return ScheduleRule(node);
558}
