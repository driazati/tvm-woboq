24/*! \brief The gradient based task scheduler. */
25class GradientBasedNode final : public TaskSchedulerNode {
26 public:
27 double alpha;
28 int window_size;
29 support::LinearCongruentialEngine::TRandState rand_state;
30
31 int round_robin_rounds_;
32 std::vector<std::vector<double>> best_latency_history_;
33
34 void VisitAttrs(tvm::AttrVisitor* v) {
35 TaskSchedulerNode::VisitAttrs(v);
36 v->Visit("alpha", &alpha);
37 v->Visit("window_size", &window_size);
38 // `rand_state` is not visited.
39 // `num_rounds_already_` is not visited.
40 // `best_latency_history_` is not visited.
41 }
42
43 static constexpr const char* _type_key = "meta_schedule.GradientBased";
44 TVM_DECLARE_FINAL_OBJECT_INFO(GradientBasedNode, TaskSchedulerNode);
45
46 public:
47 void Tune(Array<TuneContext> tasks, Array<FloatImm> task_weights, int max_trials_global,
48 int max_trials_per_task, int num_trials_per_iter, Builder builder, Runner runner,
49 Array<MeasureCallback> measure_callbacks, Optional<Database> database,
50 Optional<CostModel> cost_model) final {
51 int n_tasks = tasks.size();
52 round_robin_rounds_ = 0;
53 best_latency_history_.resize(n_tasks, std::vector<double>());
54 TaskSchedulerNode::Tune(tasks, task_weights, max_trials_global, max_trials_per_task,
55 num_trials_per_iter, builder, runner, measure_callbacks, database,
56 cost_model);
57 }
58
59 int NextTaskId() final {
60 int n_tasks = this->tasks_.size();
61 // Step 1. Check if it's in round robin mode.
62 if (round_robin_rounds_ == 0) {
63 TVM_PY_LOG_CLEAR_SCREEN(this->logger);
64 this->PrintTuningStatistics();
65 }
66 if (round_robin_rounds_ < n_tasks) {
67 return round_robin_rounds_++;
68 }
69 if (round_robin_rounds_ == n_tasks) {
70 for (int i = 0; i < n_tasks; ++i) {
71 this->JoinRunningTask(i);
72 }
73 ++round_robin_rounds_;
74 }
75 // Step 2. Collect the tasks that are not terminated yet
76 std::vector<int> tasks_alive;
77 {
78 tasks_alive.reserve(n_tasks);
79 for (int i = 0; i < n_tasks; ++i) {
80 this->TouchTask(i);
81 if (!this->tasks_[i]->is_terminated) {
82 tasks_alive.push_back(i);
83 }
84 }
85 if (tasks_alive.empty()) {
86 return -1;
87 }
88 }
89 // Step 3. Calculate the gradient of each task alive
90 std::vector<double> grad;
91 grad.reserve(n_tasks);
92 for (int task_id : tasks_alive) {
93 const std::vector<double>& best_latency = this->best_latency_history_.at(task_id);
94 int n = best_latency.size();
95 ICHECK_GE(n, 1);
96 double task_weight = this->tasks_[task_id]->task_weight;
97 int w = this->window_size;
98 double best = best_latency[n - 1];
99 if (best < 1e9) {
100 double g1 = (n >= 1 + w) ? (best_latency[n - 1 - w] - best) / w : 0.0;
101 double g2 = best / n;
102 double g = alpha * g1 + (1 - alpha) * g2;
103 grad.push_back(g * task_weight);
104 } else {
105 // If the best time cost is unavailable, it means some task is not valid. Skip it.
106 grad.push_back(-1e9);
107 }
108 }
109 // Step 4. Select the task with the largest gradient
110 auto max_grad = std::max_element(grad.begin(), grad.end());
111 auto min_grad = std::min_element(grad.begin(), grad.end());
112 int task_id = -1;
113 if (*max_grad == *min_grad) {
114 task_id = tasks_alive[tir::SampleInt(&this->rand_state, 0, tasks_alive.size())];
115 } else {
116 task_id = tasks_alive[std::distance(grad.begin(), max_grad)];
117 }
118 if (this->tasks_[task_id]->runner_futures.defined()) {
119 JoinRunningTask(task_id);
120 }
121 return task_id;
122 }
123
124 Array<RunnerResult> JoinRunningTask(int task_id) final {
125 Array<RunnerResult> results = TaskSchedulerNode::JoinRunningTask(task_id);
126 TaskRecordNode* task = this->tasks_[task_id].get();
127 this->best_latency_history_.at(task_id).push_back(
128 *std::min_element(task->latency_ms.begin(), //
129 task->latency_ms.end()));
130 return results;
131 }
132};
133
134TaskScheduler TaskScheduler::GradientBased(PackedFunc logger, double alpha, int window_size,
135 support::LinearCongruentialEngine::TRandState seed) {
136 ObjectPtr<GradientBasedNode> n = make_object<GradientBasedNode>();
137 n->logger = logger;
138 n->alpha = alpha;
139 n->window_size = window_size;
140 n->rand_state = support::LinearCongruentialEngine::NormalizeSeed(seed);
141 return TaskScheduler(n);
142}
