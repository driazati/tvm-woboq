24struct TaskRecord {
25 TuneContext task;
26 double weight;
27 double flop;
28 std::vector<double> best_time_cost_history; // in ms
29 int trials;
30};
31
32/*! \brief The gradient based task scheduler. */
33class GradientBasedNode final : public TaskSchedulerNode {
34 public:
35 // Parameters used in gradient computation
36 double alpha;
37 int window_size;
38
39 std::vector<TaskRecord> task_records_;
40 std::vector<double> best_time_cost_per_task_; // in ms
41 int num_rounds_already_;
42 support::LinearCongruentialEngine::TRandState rand_state_;
43
44 void VisitAttrs(tvm::AttrVisitor* v) {
45 TaskSchedulerNode::VisitAttrs(v);
46 v->Visit("alpha", &alpha);
47 v->Visit("window_size", &window_size);
48 // `task_records_` is not visited.
49 // `best_time_cost_per_task_` is not visited.
50 // `num_rounds_already_` is not visited.
51 // `rand_state_` is not visited.
52 }
53
54 static constexpr const char* _type_key = "meta_schedule.GradientBased";
55 TVM_DECLARE_FINAL_OBJECT_INFO(GradientBasedNode, TaskSchedulerNode);
56
57 public:
58 std::string TuningStatistics() const {
59 std::ostringstream os;
60 int n_tasks = task_records_.size();
61 int total_trials = 0;
62 double total_latency = 0.0;
63 support::TablePrinter p;
64 p.Row() << "ID"
65 << "Name"
66 << "FLOP"
67 << "Weight"
68 << "Speed (GFLOPS)"
69 << "Latency (us)"
70 << "Weighted Latency (us)"
71 << "Trials"
72 << "Terminated";
73 p.Separator();
74 for (int i = 0; i < n_tasks; ++i) {
75 const TaskRecord& record = task_records_[i];
76 auto row = p.Row();
77 int trials = record.trials;
78 row << /*id=*/i //
79 << /*name=*/record.task->task_name.value() //
80 << /*flops=*/static_cast<int64_t>(record.flop) //
81 << /*weight=*/static_cast<int>(record.weight);
82 double latency = 1e9;
83 if (trials > 0) {
84 latency = record.best_time_cost_history.back();
85 }
86 if (latency >= 1e9) {
87 row << /*speed=*/"N/A" << /*latency=*/"N/A" << /*weighted_latency=*/"N/A";
88 } else {
89 latency *= 1000.0;
90 double speed = record.flop / latency / 1000.0;
91 double weighted_latency = latency * record.weight;
92 row << /*speed=*/speed << /*latency=*/latency << /*weighted_latency=*/weighted_latency;
93 total_latency += weighted_latency;
94 total_trials += trials;
95 }
96 row << trials;
97 if (tasks[i]->is_terminated) {
98 row << "Y";
99 } else {
100 row << "";
101 }
102 }
103 p.Separator();
104 os << p.AsStr() //
105 << "\nTotal trials: " << total_trials //
106 << "\nTotal latency (us): " << total_latency //
107 << "\n";
108 return os.str();
109 }
110
111 int NextTaskId() final {
112 int n_tasks = task_records_.size();
113 // Round robin
114 if (num_rounds_already_ == 0) {
115 TVM_PY_LOG(INFO, this->logging_func) << "\n" << this->TuningStatistics();
116 }
117 if (num_rounds_already_ < n_tasks) {
118 return num_rounds_already_++;
119 }
120 if (num_rounds_already_ == n_tasks) {
121 for (int i = 0; i < n_tasks; ++i) {
122 this->JoinRunningTask(i);
123 }
124 }
125 ++num_rounds_already_;
126 // Check running tasks
127 std::vector<int> tasks_alive;
128 tasks_alive.reserve(n_tasks);
129 for (int i = 0; i < n_tasks; ++i) {
130 this->TouchTask(i);
131 if (!tasks[i]->is_terminated) {
132 tasks_alive.push_back(i);
133 }
134 }
135 if (tasks_alive.empty()) {
136 return -1;
137 }
138 std::vector<double> grad;
139 grad.reserve(n_tasks);
140 for (int task_id : tasks_alive) {
141 const TaskRecord& record = task_records_[task_id];
142 const int w = this->window_size;
143 int n = record.best_time_cost_history.size();
144 ICHECK_GE(n, 1);
145 double best = record.best_time_cost_history[n - 1];
146 if (best < 1e9) {
147 double g1 = (n >= 1 + w) ? (record.best_time_cost_history[n - 1 - w] - best) / w : 0.0;
148 double g2 = best / n;
149 double g = alpha * g1 + (1 - alpha) * g2;
150 grad.push_back(g * record.weight);
151 } else {
152 // If the best time cost is unavailable, it means some task is not valid. Skip it.
153 grad.push_back(-1e9);
154 }
155 }
156 auto max_grad = std::max_element(grad.begin(), grad.end());
157 auto min_grad = std::min_element(grad.begin(), grad.end());
158 int task_id = -1;
159 if (*max_grad == *min_grad) {
160 task_id = tasks_alive[tir::SampleInt(&rand_state_, 0, tasks_alive.size())];
161 } else {
162 task_id = tasks_alive[std::distance(grad.begin(), max_grad)];
163 }
164 if (tasks[task_id]->runner_futures.defined()) {
165 JoinRunningTask(task_id);
166 }
167 return task_id;
168 }
169
170 Array<RunnerResult> JoinRunningTask(int task_id) final {
171 TaskRecord& record = task_records_[task_id];
172 Array<RunnerResult> results = TaskSchedulerNode::JoinRunningTask(task_id);
173 double& best_time_cost = this->best_time_cost_per_task_[task_id];
174 for (const RunnerResult& result : results) {
175 if (!result->error_msg.defined()) {
176 best_time_cost = std::min(best_time_cost, GetRunMsMedian(result));
177 }
178 }
179 record.best_time_cost_history.push_back(best_time_cost);
180 record.trials += results.size();
181 TVM_PY_LOG(INFO, this->logging_func)
182 << "[Updated] Task #" << task_id << ": " << record.task->task_name << "\n"
183 << this->TuningStatistics();
184 return results;
185 }
186};

188TaskScheduler TaskScheduler::GradientBased(Array<TuneContext> tasks, //
189 Array<FloatImm> task_weights, //
190 Builder builder, //
191 Runner runner, //
192 Optional<Database> database, //
193 Optional<CostModel> cost_model, //
194 Optional<Array<MeasureCallback>> measure_callbacks, //
195 int max_trials, //
196 PackedFunc logging_func, //
197 double alpha, //
198 int window_size, //
199 support::LinearCongruentialEngine::TRandState seed) {
200 CHECK_EQ(tasks.size(), task_weights.size())
201 << "The size of `tasks` should have the same as `task_weights`.";
202 int n_tasks = tasks.size();
203 std::vector<TaskRecord> task_records;
204 task_records.reserve(n_tasks);
205 for (int i = 0; i < n_tasks; ++i) {
206 task_records.push_back(TaskRecord{
207 /*task=*/tasks[i],
208 /*weights=*/task_weights[i]->value,
209 /*flop=*/std::max(1.0, tir::EstimateTIRFlops(tasks[i]->mod.value())),
210 /*best_time_cost_history=*/{},
211 /*trials=*/0,
212 });
213 }
214 ObjectPtr<GradientBasedNode> n = make_object<GradientBasedNode>();
215 n->tasks = tasks;
216 n->builder = builder;
217 n->runner = runner;
218 n->database = database;
219 n->max_trials = max_trials;
220 n->cost_model = cost_model;
221 n->measure_callbacks = measure_callbacks.value_or({});
222 n->logging_func = logging_func;
223 n->num_trials_already = 0;
224 n->alpha = alpha;
225 n->window_size = window_size;
226 n->task_records_ = std::move(task_records);
227 n->best_time_cost_per_task_ = std::vector<double>(n_tasks, 1e100);
228 n->num_rounds_already_ = 0;
229 support::LinearCongruentialEngine(&n->rand_state_).Seed(seed);
230 return TaskScheduler(n);
231}
