21namespace tvm {
22namespace meta_schedule {

24/*! \brief The round-robin style task scheduler. */
25class RoundRobinNode final : public TaskSchedulerNode {
26 public:
27 /*! \brief The current task id processed. */
28 int task_id = -1;
29
30 void VisitAttrs(tvm::AttrVisitor* v) {
31 TaskSchedulerNode::VisitAttrs(v);
32 v->Visit("task_id", &task_id);
33 }
34
35 static constexpr const char* _type_key = "meta_schedule.RoundRobin";
36 TVM_DECLARE_FINAL_OBJECT_INFO(RoundRobinNode, TaskSchedulerNode);
37
38 protected:
39 int NextTaskId() final {
40 int n_tasks = this->tasks.size();
41 for (int i = 0; i < n_tasks; ++i) {
42 this->TouchTask(i);
43 }
44 for (int i = 0; i < n_tasks; ++i) {
45 task_id = (task_id + 1) % n_tasks;
46 TuneContext task = tasks[task_id];
47 if (!task->is_terminated) {
48 if (task->runner_futures.defined()) {
49 JoinRunningTask(task_id);
50 }
51 return task_id;
52 }
53 }
54 return -1;
55 }
56};
57
58TaskScheduler TaskScheduler::RoundRobin(Array<TuneContext> tasks, //
59 Builder builder, //
60 Runner runner, //
61 Optional<Database> database, //
62 Optional<CostModel> cost_model, //
63 Optional<Array<MeasureCallback>> measure_callbacks, //
64 int max_trials, //
65 PackedFunc logging_func) {
66 ObjectPtr<RoundRobinNode> n = make_object<RoundRobinNode>();
67 n->tasks = tasks;
68 n->builder = builder;
69 n->runner = runner;
70 n->database = database;
71 n->max_trials = max_trials;
72 n->cost_model = cost_model;
73 n->measure_callbacks = measure_callbacks.value_or({});
74 n->logging_func = logging_func;
75 n->num_trials_already = 0;
76 n->task_id = -1;
77 return TaskScheduler(n);
78}

80TVM_REGISTER_NODE_TYPE(RoundRobinNode);
81TVM_REGISTER_GLOBAL("meta_schedule.TaskSchedulerRoundRobin")
82 .set_body_typed(TaskScheduler::RoundRobin);
