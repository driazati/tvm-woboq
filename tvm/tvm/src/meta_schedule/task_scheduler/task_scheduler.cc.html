24void TaskSchedulerNode::InitializeTask(int task_id) {
25 auto _ = Profiler::TimedScope("InitializeTask");
26 TuneContext task = this->tasks[task_id];
27 TVM_PY_LOG(INFO, this->logging_func)
28 << "Initializing Task #" << task_id << ": " << task->task_name;
29 TVM_PY_LOG(INFO, task->logging_func)
30 << "Initializing Task #" << task_id << ": " << task->task_name;
31 CHECK(task->mod.defined()) << "ValueError: Require `context.mod`, but it is not defined";
32 CHECK(task->space_generator.defined())
33 << "ValueError: Require `context.space_generator`, but it is not defined";
34 CHECK(task->search_strategy.defined())
35 << "ValueError: Require `context.search_strategy`, but it is not defined";
36 TVM_PY_LOG(INFO, task->logging_func) << "\n" << tir::AsTVMScript(task->mod);
37 task->Initialize();
38 Array<tir::Schedule> design_spaces =
39 task->space_generator.value()->GenerateDesignSpace(task->mod.value());
40 TVM_PY_LOG(INFO, task->logging_func)
41 << "Total " << design_spaces.size() << " design space(s) generated";
42 for (int i = 0, n = design_spaces.size(); i < n; ++i) {
43 tir::Schedule sch = design_spaces[i];
44 tir::Trace trace = sch->trace().value();
45 trace = trace->Simplified(true);
46 TVM_PY_LOG(INFO, task->logging_func) << "Design space #" << i << ":\n"
47 << tir::AsTVMScript(sch->mod()) << "\n"
48 << Concat(trace->AsPython(false), "\n");
49 }
50 task->search_strategy.value()->PreTuning(design_spaces, database, cost_model);
51}

53void TaskSchedulerNode::Tune() {
54 int n_tasks = this->tasks.size();
55 for (int task_id = 0; task_id < n_tasks; ++task_id) {
56 InitializeTask(task_id);
57 }
58 int running_tasks = tasks.size();
59 for (int task_id; num_trials_already < max_trials && (task_id = NextTaskId()) != -1;) {
60 TVM_PY_LOG(INFO, this->logging_func)
61 << "Scheduler picks Task #" << task_id << ": " << tasks[task_id]->task_name;
62 TuneContext task = tasks[task_id];
63 ICHECK(!task->is_terminated);
64 ICHECK(!task->runner_futures.defined());
65 if (Optional<Array<MeasureCandidate>> candidates =
66 task->search_strategy.value()->GenerateMeasureCandidates()) {
67 int num_candidates = candidates.value().size();
68 task->_SetMeasureCandidates(candidates.value());
69 num_trials_already += num_candidates;
70 TVM_PY_LOG(INFO, this->logging_func)
71 << "Sending " << num_candidates << " sample(s) to builder";
72 task->_SendToBuilder(this->builder);
73 TVM_PY_LOG(INFO, this->logging_func)
74 << "Sending " << num_candidates << " sample(s) to runner";
75 task->_SendToRunner(this->runner);
76 } else {
77 ICHECK(!task->is_terminated);
78 task->is_terminated = true;
79 --running_tasks;
80 TVM_PY_LOG(INFO, this->logging_func)
81 << "Task #" << task_id << " has finished. Remaining task(s): " << running_tasks;
82 }
83 }
84 for (int task_id = 0; task_id < n_tasks; ++task_id) {
85 TuneContext task = tasks[task_id];
86 if (!task->is_terminated) {
87 if (task->runner_futures.defined()) {
88 JoinRunningTask(task_id);
89 }
90 task->is_terminated = true;
91 --running_tasks;
92 TVM_PY_LOG(INFO, this->logging_func)
93 << "Task #" << task_id << " has finished. Remaining task(s): " << running_tasks;
94 }
95 task->search_strategy.value()->PostTuning();
96 }
97}

99void TaskSchedulerNode::TouchTask(int task_id) {
100 TuneContext task = tasks[task_id];
101 if (!task->is_terminated && task->runner_futures.defined()) {
102 for (const RunnerFuture future : task->runner_futures.value()) {
103 if (!future->Done()) {
104 return;
105 }
106 }
107 this->JoinRunningTask(task_id);
108 }
109}

111Array<RunnerResult> TaskSchedulerNode::JoinRunningTask(int task_id) {
112 TuneContext task = tasks[task_id];
113 Array<RunnerResult> results = task->_Join();
114 for (const MeasureCallback& callback : this->measure_callbacks) {
115 callback->Apply(GetRef<TaskScheduler>(this), task_id, task->measure_candidates.value(),
116 task->builder_results.value(), results);
117 }
118 task->_ClearMeasureState();
119 return results;
120}

159TaskScheduler TaskScheduler::PyTaskScheduler(
160 Array<TuneContext> tasks, //
161 Builder builder, //
162 Runner runner, //
163 Optional<Database> database, //
164 Optional<CostModel> cost_model, //
165 Optional<Array<MeasureCallback>> measure_callbacks, //
166 int max_trials, //
167 PackedFunc logging_func, //
168 PyTaskSchedulerNode::FTune f_tune, //
169 PyTaskSchedulerNode::FInitializeTask f_initialize_task, //
170 PyTaskSchedulerNode::FTouchTask f_touch_task, //
171 PyTaskSchedulerNode::FJoinRunningTask f_join_running_task, //
172 PyTaskSchedulerNode::FNextTaskId f_next_task_id) {
173 ObjectPtr<PyTaskSchedulerNode> n = make_object<PyTaskSchedulerNode>();
174 n->tasks = tasks;
175 n->builder = builder;
176 n->runner = runner;
177 n->database = database;
178 n->max_trials = max_trials;
179 n->cost_model = cost_model;
180 if (measure_callbacks.defined()) {
181 n->measure_callbacks = measure_callbacks.value();
182 } else {
183 n->measure_callbacks = {};
184 }
185 n->logging_func = logging_func;
186 n->num_trials_already = 0;
187 n->f_tune = f_tune;
188 n->f_initialize_task = f_initialize_task;
189 n->f_touch_task = f_touch_task;
190 n->f_join_running_task = f_join_running_task;
191 n->f_next_task_id = f_next_task_id;
192 return TaskScheduler(n);
193}
