19#include "module_equality.h"
20
21#include <tvm/ir/module.h>
22#include <tvm/node/structural_equal.h>
23#include <tvm/node/structural_hash.h>
24
25#include <memory>
26
27namespace tvm {
28namespace meta_schedule {
29
30class ModuleEqualityStructural : public ModuleEquality {
31 public:
32 size_t Hash(IRModule mod) const { return tvm::StructuralHash()(mod); }
33 bool Equal(IRModule lhs, IRModule rhs) const { return tvm::StructuralEqual()(lhs, rhs); }
34};
35
36std::unique_ptr<ModuleEquality> ModuleEquality::Create(const std::string& mod_eq_name) {
37 if (mod_eq_name == "structural") {
38 return std::make_unique<ModuleEqualityStructural>();
39 }
40 LOG(FATAL) << "Unknown module equality " << mod_eq_name;
41 return nullptr;
42}
