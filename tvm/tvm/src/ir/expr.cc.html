44PrimExpr PrimExpr::FromObject_(ObjectRef ref) {
45 using runtime::ObjectTypeChecker;
46 if (auto* ptr = ref.as<tir::IterVarNode>()) {
47 return GetRef<tir::IterVar>(ptr)->var;
48 }
49 if (auto* ptr = ref.as<te::TensorNode>()) {
50 return GetRef<te::Tensor>(ptr)();
51 }
52 if (auto* ptr = ref.as<runtime::StringObj>()) {
53 return tir::StringImm(GetRef<runtime::String>(ptr));
54 }
55 if (const auto* buffer_region = ref.as<tir::BufferRegionNode>()) {
56 Array<PrimExpr> indices;
57 indices.reserve(buffer_region->region.size());
58 for (const Range& r : buffer_region->region) {
59 if (tvm::tir::is_one(r->extent)) {
60 indices.push_back(r->min);
61 } else if (const auto* extent = r->extent.as<IntImmNode>()) {
62 indices.push_back(tir::Ramp(r->min, tvm::tir::make_const(r->min->dtype, 1), extent->value));
63 } else {
64 LOG(FATAL) << "ValueError: Cannot convert to BufferLoad: " << ref;
65 }
66 }
67 return tir::BufferLoad(buffer_region->buffer, indices);
68 }
69 Optional<String> actual_type = ObjectTypeChecker<PrimExpr>::CheckAndGetMismatch(ref.get());
70 ICHECK(!actual_type.defined()) << "Expected type " << ObjectTypeChecker<PrimExpr>::TypeName()
71 << " but got " << actual_type.value();
72 return Downcast<PrimExpr>(ref);
73}

75IntImm::IntImm(DataType dtype, int64_t value, Span span) {
76 ICHECK(dtype.is_scalar()) << "ValueError: IntImm can only take scalar, but " << dtype
77 << " was supplied.";
78 ICHECK(dtype.is_int() || dtype.is_uint())
79 << "ValueError: IntImm supports only int or uint type, but " << dtype << " was supplied.";
80 if (dtype.is_uint()) {
81 ICHECK_GE(value, 0U) << "ValueError: Literal value " << value
82 << " is negative for unsigned integer type " << dtype;
83 if (dtype.bits() < 64) {
84 ICHECK_LT(value, 1LL << dtype.bits())
85 << "ValueError: Literal value " << value << " exceeds maximum of " << dtype;
86 }
87 } else if (dtype.bits() == 1) {
88 // int(1)
89 ICHECK(value == 0 || value == 1) << "ValueError: " << value << " exceeds range of " << dtype;
90 } else if (dtype.bits() < 64) {
91 ICHECK_GE(value, -(1LL << (dtype.bits() - 1)))
92 << "ValueError: Literal value " << value << " exceeds minimum of " << dtype;
93 ICHECK_LT(value, 1LL << (dtype.bits() - 1))
94 << "ValueError: Literal value " << value << " exceeds maximum of " << dtype;
95 }
96 ObjectPtr<IntImmNode> node = make_object<IntImmNode>();
97 node->dtype = dtype;
98 node->value = value;
99 node->span = span;
100 data_ = std::move(node);
101}

119FloatImm::FloatImm(DataType dtype, double value, Span span) {
120 ICHECK_EQ(dtype.lanes(), 1) << "ValueError: FloatImm can only take scalar.";
121
122 ICHECK(dtype.is_float() || dtype.is_bfloat16() || dtype.code() >= DataType::kCustomBegin)
123 << "ValueError: FloatImm supports only float, but " << dtype << " was supplied.";
124
125 // check range for float32 and float16 since they have specified range.
126 if (!std::isinf(value) && !std::isnan(value)) {
127 if (dtype.bits() == 32) {
128 ICHECK_GE(value, std::numeric_limits<float>::lowest())
129 << "ValueError: Literal value " << value << " exceeds minimum of " << dtype;
130 ICHECK_LE(value, std::numeric_limits<float>::max())
131 << "ValueError: Literal value " << value << " exceeds maximum of " << dtype;
132 } else if (dtype.is_float16()) {
133 ICHECK_GE(value, -support::kMaxFloat16)
134 << "ValueError: Literal value " << value << " exceeds minimum of " << dtype;
135 ICHECK_LE(value, support::kMaxFloat16)
136 << "ValueError: Literal value " << value << " exceeds maximum of " << dtype;
137 }
138 }
139 ObjectPtr<FloatImmNode> node = make_object<FloatImmNode>();
140 node->dtype = dtype;
141 node->value = value;
142 node->span = span;
143 data_ = std::move(node);
144}

147 return FloatImm(dtype, value, span);

150TVM_REGISTER_NODE_TYPE(FloatImmNode);
151
152TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
153 .set_dispatch<FloatImmNode>([](const ObjectRef& node, ReprPrinter* p) {
154 auto* op = static_cast<const FloatImmNode*>(node.get());
155 auto& stream = p->stream;
156 switch (op->dtype.bits()) {
157 case 64:
158 stream << op->value;
159 break;
160 case 32:
161 stream << op->value << 'f';
162 break;
163 case 16:
164 stream << op->value << 'h';
165 break;
166 default:
167 LOG(FATAL) << "Unknown float type bits=" << op->dtype.bits();
168 }
169 });

171Range::Range(PrimExpr begin, PrimExpr end, Span span)
172 : Range(make_object<RangeNode>(begin, tir::is_zero(begin) ? end : (end - begin), span)) {}
