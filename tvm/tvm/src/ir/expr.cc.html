42PrimExpr PrimExpr::FromObject_(ObjectRef ref) {
43 using runtime::ObjectTypeChecker;
44 if (auto* ptr = ref.as<tir::IterVarNode>()) {
45 return GetRef<tir::IterVar>(ptr)->var;
46 }
47 if (auto* ptr = ref.as<te::TensorNode>()) {
48 return GetRef<te::Tensor>(ptr)();
49 }
50 if (auto* ptr = ref.as<runtime::StringObj>()) {
51 return tir::StringImm(GetRef<runtime::String>(ptr));
52 }
53 if (const auto* buffer_region = ref.as<tir::BufferRegionNode>()) {
54 Array<PrimExpr> indices;
55 indices.reserve(buffer_region->region.size());
56 for (const Range& r : buffer_region->region) {
57 if (tvm::tir::is_one(r->extent)) {
58 indices.push_back(r->min);
59 } else if (const auto* extent = r->extent.as<IntImmNode>()) {
60 indices.push_back(tir::Ramp(r->min, tvm::tir::make_const(r->min->dtype, 1), extent->value));
61 } else {
62 LOG(FATAL) << "ValueError: Cannot convert to BufferLoad: " << ref;
63 }
64 }
65 return tir::BufferLoad(buffer_region->buffer, indices);
66 }
67 Optional<String> actual_type = ObjectTypeChecker<PrimExpr>::CheckAndGetMismatch(ref.get());
68 ICHECK(!actual_type.defined()) << "Expected type " << ObjectTypeChecker<PrimExpr>::TypeName()
69 << " but got " << actual_type.value();
70 return Downcast<PrimExpr>(ref);
71}

73IntImm::IntImm(DataType dtype, int64_t value, Span span) {
74 ICHECK(dtype.is_scalar()) << "ValueError: IntImm can only take scalar, but " << dtype
75 << " was supplied.";
76 ICHECK(dtype.is_int() || dtype.is_uint())
77 << "ValueError: IntImm supports only int or uint type, but " << dtype << " was supplied.";
78 if (dtype.is_uint()) {
79 ICHECK_GE(value, 0U);
80 }
81 ObjectPtr<IntImmNode> node = make_object<IntImmNode>();
82 node->dtype = dtype;
83 node->value = value;
84 node->span = span;
85 data_ = std::move(node);
86}
87
88TVM_REGISTER_GLOBAL("ir.IntImm").set_body_typed([](DataType dtype, int64_t value, Span span) {
89 return IntImm(dtype, value, span);
90});

92TVM_REGISTER_NODE_TYPE(IntImmNode);
93
94TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
95 .set_dispatch<IntImmNode>([](const ObjectRef& node, ReprPrinter* p) {
96 auto* op = static_cast<const IntImmNode*>(node.get());
97 if (op->dtype == DataType::Int(32)) {
98 p->stream << op->value;
99 } else {
100 p->stream << "(" << op->dtype << ")" << op->value;
101 }
102 });

104FloatImm::FloatImm(DataType dtype, double value, Span span) {
105 ICHECK_EQ(dtype.lanes(), 1) << "ValueError: FloatImm can only take scalar.";
106 ObjectPtr<FloatImmNode> node = make_object<FloatImmNode>();
107 node->dtype = dtype;
108 node->value = value;
109 node->span = span;
110 data_ = std::move(node);
111}
112
113TVM_REGISTER_GLOBAL("ir.FloatImm").set_body_typed([](DataType dtype, double value, Span span) {
114 return FloatImm(dtype, value, span);
115});
116
117TVM_REGISTER_NODE_TYPE(FloatImmNode);
118
119TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
120 .set_dispatch<FloatImmNode>([](const ObjectRef& node, ReprPrinter* p) {
121 auto* op = static_cast<const FloatImmNode*>(node.get());
122 auto& stream = p->stream;
123 switch (op->dtype.bits()) {
124 case 64:
125 stream << op->value;
126 break;
127 case 32:
128 stream << op->value << 'f';
129 break;
130 case 16:
131 stream << op->value << 'h';
132 break;
133 default:
134 LOG(FATAL) << "Unknown float type bits=" << op->dtype.bits();
135 }
136 });

138Range::Range(PrimExpr begin, PrimExpr end, Span span)
139 : Range(make_object<RangeNode>(begin, tir::is_zero(begin) ? end : (end - begin), span)) {}
140
141Range Range::FromMinExtent(PrimExpr min, PrimExpr extent, Span span) {
142 return Range(make_object<RangeNode>(min, extent, span));
143}
144
145TVM_REGISTER_GLOBAL("ir.Range_from_min_extent").set_body_typed(Range::FromMinExtent);
146
147TVM_REGISTER_GLOBAL("ir.Range").set_body([](TVMArgs args, TVMRetValue* ret) {
148 *ret = Range(args[0], args[1], args[2]);
149});
150
151TVM_REGISTER_NODE_TYPE(RangeNode);
152
153TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
154 .set_dispatch<RangeNode>([](const ObjectRef& node, ReprPrinter* p) {
155 auto* op = static_cast<const RangeNode*>(node.get());
156 p->stream << "range(min=" << op->min << ", ext=" << op->extent << ')';
157 });

159GlobalVar::GlobalVar(String name_hint, Type type, Span span) {
160 ObjectPtr<GlobalVarNode> n = make_object<GlobalVarNode>();
161 n->name_hint = std::move(name_hint);
162 n->checked_type_ = std::move(type);
163 n->span = std::move(span);
164 data_ = std::move(n);
165}
166
167TVM_REGISTER_NODE_TYPE(GlobalVarNode);
168
169TVM_REGISTER_GLOBAL("ir.GlobalVar").set_body_typed([](String name, Type type) {
170 return GlobalVar(name, type);
171});
172
173TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
174 .set_dispatch<GlobalVarNode>([](const ObjectRef& ref, ReprPrinter* p) {
175 auto* node = static_cast<const GlobalVarNode*>(ref.get());
176 p->stream << "GlobalVar(" << node->name_hint << ")";
177 });
