24#include <tvm/relay/adt.h>
25#include <tvm/relay/type.h>
26
27namespace tvm {
28
29Constructor::Constructor(String name_hint, tvm::Array<Type> inputs, GlobalTypeVar belong_to) {
30 ObjectPtr<ConstructorNode> n = make_object<ConstructorNode>();
31 n->name_hint = std::move(name_hint);
32 n->inputs = std::move(inputs);
33 n->belong_to = std::move(belong_to);
34 data_ = std::move(n);
35}
36
37TVM_REGISTER_NODE_TYPE(ConstructorNode);
38
39TVM_REGISTER_GLOBAL("ir.Constructor")
40 .set_body_typed([](String name_hint, tvm::Array<Type> inputs, GlobalTypeVar belong_to) {
41 return Constructor(name_hint, inputs, belong_to);
42 });
43
44TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
45 .set_dispatch<ConstructorNode>([](const ObjectRef& ref, ReprPrinter* p) {
46 auto* node = static_cast<const ConstructorNode*>(ref.get());
47 p->stream << "ConstructorNode(" << node->name_hint << ", " << node->inputs << ", "
48 << node->belong_to << ")";
49 });
50
51TypeData::TypeData(GlobalTypeVar header, tvm::Array<TypeVar> type_vars,
52 tvm::Array<Constructor> constructors) {
53 ObjectPtr<TypeDataNode> n = make_object<TypeDataNode>();
54 n->header = std::move(header);
55 n->type_vars = std::move(type_vars);
56 n->constructors = std::move(constructors);
57 data_ = std::move(n);
58}
59
60TVM_REGISTER_NODE_TYPE(TypeDataNode);
61
62TVM_REGISTER_GLOBAL("ir.TypeData")
63 .set_body_typed([](GlobalTypeVar header, tvm::Array<TypeVar> type_vars,
64 tvm::Array<Constructor> constructors) {
65 return TypeData(header, type_vars, constructors);
66 });
67
68TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
69 .set_dispatch<TypeDataNode>([](const ObjectRef& ref, ReprPrinter* p) {
70 auto* node = static_cast<const TypeDataNode*>(ref.get());
71 p->stream << "TypeDataNode(" << node->header << ", " << node->type_vars << ", "
72 << node->constructors << ")";
73 });
74
