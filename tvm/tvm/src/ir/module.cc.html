46IRModule::IRModule(tvm::Map<GlobalVar, BaseFunc> functions,
47 tvm::Map<GlobalTypeVar, TypeData> type_definitions,
48 std::unordered_set<String> import_set, parser::SourceMap source_map,
49 DictAttrs attrs) {
50 auto n = make_object<IRModuleNode>();
51 n->functions = std::move(functions);
52 n->type_definitions = std::move(type_definitions);
53 n->global_type_var_map_ = {};
54 n->global_var_map_ = {};
55 n->constructor_tag_map_ = {};
56 n->import_set_ = std::move(import_set);
57 n->source_map = source_map;
58 n->attrs = std::move(attrs);
59
60 for (const auto& kv : n->functions) {
61 // set global var map
62 ICHECK(n->global_var_map_.count(kv.first->name_hint) == 0)
63 << "Duplicate global function name " << kv.first->name_hint;
64 n->global_var_map_.Set(kv.first->name_hint, kv.first);
65 }
66
67 for (const auto& kv : n->type_definitions) {
68 // set global typevar map
69 ICHECK(n->global_type_var_map_.count(kv.first->name_hint) == 0)
70 << "Duplicate global type definition name " << kv.first->name_hint;
71 n->global_type_var_map_.Set(kv.first->name_hint, kv.first);
72 n->RegisterConstructors(kv.first, kv.second);
73 }
74 data_ = std::move(n);
75}

77bool IRModuleNode::SEqualReduce(const IRModuleNode* other, SEqualReducer equal) const {
78 if (functions.size() != other->functions.size()) return false;
79 if (!equal(this->attrs, other->attrs)) return false;
80 for (const auto& kv : this->functions) {
81 if (!other->ContainGlobalVar(kv.first->name_hint)) return false;
82 if (!equal(kv.second, other->Lookup(kv.first->name_hint))) return false;
83 }
84 if (type_definitions.size() != other->type_definitions.size()) return false;
85 for (const auto& kv : this->type_definitions) {
86 if (!other->ContainGlobalTypeVar(kv.first->name_hint)) return false;
87 if (!equal(kv.second, other->LookupTypeDef(kv.first->name_hint))) return false;
88 }
89 return true;
90}

92void IRModuleNode::SHashReduce(SHashReducer hash_reduce) const {
93 using KV = std::pair<std::string, ObjectRef>;
94 // hash the functions.
95 std::vector<KV> temp;
96
97 auto reduce_temp = [&]() {
98 // sort by the hash key of the keys.
99 std::sort(temp.begin(), temp.end(),
100 [](const KV& lhs, const KV& rhs) { return lhs.first < rhs.first; });
101
102 hash_reduce(static_cast<uint64_t>(temp.size()));
103 // hash the content
104 for (size_t i = 0; i < temp.size(); ++i) {
105 hash_reduce(temp[i].first);
106 hash_reduce(temp[i].second);
107 }
108 };
109
110 for (const auto& kv : this->functions) {
111 temp.emplace_back(kv.first->name_hint, kv.second);
112 }
113 reduce_temp();
114
115 temp.clear();
116 for (const auto& kv : this->type_definitions) {
117 temp.emplace_back(kv.first->name_hint, kv.second);
118 }
119 reduce_temp();
120 hash_reduce(this->attrs);
121}

123bool IRModuleNode::ContainGlobalVar(const String& name) const {
124 return global_var_map_.find(name) != global_var_map_.end();

127bool IRModuleNode::ContainGlobalTypeVar(const String& name) const {

131GlobalVar IRModuleNode::GetGlobalVar(const String& name) const {
132 auto it = global_var_map_.find(name);
133 if (it == global_var_map_.end()) {
134 std::ostringstream msg;
135 msg << "ValueError: Cannot find global var \"" << name << "\" in the Module\n"
136 << "candidates are: [";
137 int counter = 0;
138 for (auto kv : global_var_map_) {
139 if (counter++ != 0) {
140 msg << ", ";
141 }
142 msg << "\"" << kv.first << "\"";
143 }
144 msg << "]";
145 LOG(FATAL) << msg.str();
146 }
147 return (*it).second;
148}

150tvm::Array<GlobalVar> IRModuleNode::GetGlobalVars() const {
151 std::vector<GlobalVar> global_vars;
152 for (const auto& pair : global_var_map_) {
153 global_vars.push_back(pair.second);

158GlobalTypeVar IRModuleNode::GetGlobalTypeVar(const String& name) const {
159 ICHECK(global_type_var_map_.defined());
160 auto it = global_type_var_map_.find(name);
161 ICHECK(it != global_type_var_map_.end())
162 << "Cannot find global type var " << name << " in the Module";
163 return (*it).second;
164}

166Constructor IRModuleNode::GetConstructor(const String& adt, const String& cons) const {
167 TypeData typeDef = this->LookupTypeDef(adt);
168 for (Constructor c : typeDef->constructors) {
169 if (cons.compare(c->name_hint) == 0) {
170 return c;
171 }
172 }
173
174 LOG(FATAL) << adt << " does not contain constructor " << cons;
175 return {};
176}

178tvm::Array<GlobalTypeVar> IRModuleNode::GetGlobalTypeVars() const {
179 std::vector<GlobalTypeVar> global_type_vars;
180 for (const auto& pair : global_type_var_map_) {
181 global_type_vars.push_back(pair.second);
182 }
183 return tvm::Array<GlobalTypeVar>(global_type_vars);
184}

186void WarnIfMalformed(const IRModule& mod, relay::Function func) {
187 func = Downcast<relay::Function>(relay::DeDup(func));
188 // Type check the item before we add it to the module.
189 auto fv = relay::FreeVars(func);
190 auto ftv = relay::FreeTypeVars(func, mod);
191 // TODO(@jroesch): refactor to use diagnostic context
192 ICHECK_EQ(fv.size(), 0) << "Function:" << std::endl
193 << PrettyPrint(func) << std::endl
194 << "contains free variables: " << fv;
195 ICHECK_EQ(ftv.size(), 0) << "Function:" << std::endl
196 << PrettyPrint(func) << std::endl
197 << "contains free type variables: " << fv;
198}

200void IRModuleNode::Add(const GlobalVar& var, const BaseFunc& f, bool update) {
201 BaseFunc checked_func = f;
202 if (auto* ptr = f.as<relay::FunctionNode>()) {
203 WarnIfMalformed(GetRef<IRModule>(this), GetRef<relay::Function>(ptr));
204 }
205
206 AddUnchecked(var, checked_func);
207}

209void IRModuleNode::AddUnchecked(const GlobalVar& var, const BaseFunc& func) {
210 this->functions.Set(var, func);
211
212 auto it = global_var_map_.find(var->name_hint);
213 if (it != global_var_map_.end()) {
214 ICHECK_EQ((*it).second, var);
215 } else {
216 ICHECK(global_var_map_.count(var->name_hint) == 0)
217 << "Duplicate global function name " << PrettyPrint(var);
218 }
219
220 global_var_map_.Set(var->name_hint, var);
221}

223void IRModuleNode::RegisterConstructors(const GlobalTypeVar& var, const TypeData& type) {
224 // We hash the global type var name to use as a globally unique prefix for tags.
225 // The hash will be used as the most significant byte of the tag, with the index of
226 // the constructor in the less significant bytes
227 size_t hash = std::hash<std::string>()(var->name_hint);
228 int32_t prefix = static_cast<int32_t>(hash & 0xff) << 24;
229 for (size_t i = 0; i < type->constructors.size(); ++i) {
230 type->constructors[i]->tag = prefix | static_cast<int32_t>(i);
231 constructor_tag_map_[type->constructors[i]->tag] = type->constructors[i];
232 }
233}

235void IRModuleNode::AddTypeDef(const GlobalTypeVar& var, const TypeData& type, bool update) {
236 // TODO(@jroesch): we have temporarily removed kind checking here, and will consolidate
237 // to the type checker in follow up PR.
238 AddTypeDefUnchecked(var, type, update);
239}

241void IRModuleNode::AddTypeDefUnchecked(const GlobalTypeVar& var, const TypeData& type,
242 bool update) {
243 this->type_definitions.Set(var, type);
244 if (!update) {
245 // set global type var map
246 ICHECK(global_type_var_map_.count(var->name_hint) == 0)
247 << "Duplicate global type definition name " << PrettyPrint(var);
248 }
249 global_type_var_map_.Set(var->name_hint, var);
250 RegisterConstructors(var, type);
251}

261void IRModuleNode::Remove(const GlobalVar& var) {
262 auto functions_node = this->functions.CopyOnWrite();
263 functions_node->erase(var);
264 auto gvar_node = global_var_map_.CopyOnWrite();
265 gvar_node->erase(var->name_hint);
266}

268BaseFunc IRModuleNode::Lookup(const GlobalVar& var) const {
269 auto it = functions.find(var);
270 ICHECK(it != functions.end()) << "There is no definition of " << PrettyPrint(var);
271 return (*it).second;
272}

274BaseFunc IRModuleNode::Lookup(const String& name) const {

296/*!
297 * \brief Renames global type/term variables to prefer the GlobalTypeVar/GlobalVar in the lhs
298 * ('one') side above the rhs ('two').
299 */
300struct Renamer : relay::ExprMutator, TypeMutator {
301 Map<String, GlobalVar> defs;
302 Map<String, GlobalTypeVar> types;
303 std::unordered_map<int32_t, Constructor> ctors;
304
305 Renamer(Map<String, GlobalVar> defs_one, Map<String, GlobalVar> defs_two,
306 Map<String, GlobalTypeVar> types_one, Map<String, GlobalTypeVar> types_two,
307 std::unordered_map<int32_t, Constructor> ctors_one,
308 std::unordered_map<int32_t, Constructor> ctor_two) {
309 for (auto pair : defs_one) {
310 defs.Set(pair.first, pair.second);
311 }
312
313 for (auto pair : defs_two) {
314 auto it = defs.find(pair.first);
315 if (it == defs.end()) {
316 defs.Set(pair.first, pair.second);
317 }
318 }
319
320 for (auto pair : types_one) {
321 types.Set(pair.first, pair.second);
322 }
323
324 for (auto pair : types_two) {
325 auto it = types.find(pair.first);
326 if (it == types.end()) {
327 types.Set(pair.first, pair.second);
328 }
329 }
330 }
331
332 relay::Expr VisitExpr_(const GlobalVarNode* node) override { return defs.at(node->name_hint); }
333
334 Type VisitType_(const GlobalTypeVarNode* node) override { return types.at(node->name_hint); }
335};

337void IRModuleNode::Update(const IRModule& mod) {
338 Renamer renamer(this->global_var_map_, mod->global_var_map_, this->global_type_var_map_,
339 mod->global_type_var_map_, this->constructor_tag_map_, mod->constructor_tag_map_);
340
341 this->global_var_map_ = renamer.defs;
342 this->global_type_var_map_ = renamer.types;
343 this->constructor_tag_map_ = renamer.ctors;
344
345 for (auto pair : mod->type_definitions) {
346 auto tvar = renamer.types.at(pair.first->name_hint);
347 auto ty = renamer.ExprMutator::VisitType(pair.second);
348 this->AddTypeDefUnchecked(tvar, Downcast<TypeData>(ty), true);
349 }
350
351 for (auto pair : mod->functions) {
352 if (auto rfn = pair.second.as<relay::FunctionNode>()) {
353 auto gvar = renamer.defs.at(pair.first->name_hint);
354 auto fn = renamer.VisitExpr(GetRef<relay::Function>(rfn));
355 this->AddUnchecked(gvar, Downcast<BaseFunc>(fn));
356 } else {
357 // TODO(@jroesch): rename into IRModule.
358 this->AddUnchecked(pair.first, pair.second);
359 }
360 }
361}

368std::pair<IRModule, GlobalVar> IRModule::FromExprInContext(
369 const RelayExpr& expr, const tvm::Map<GlobalVar, BaseFunc>& global_funcs,
370 const tvm::Map<GlobalTypeVar, TypeData>& type_definitions,
371 std::unordered_set<String> import_set) {
372 auto mod = IRModule(global_funcs, type_definitions, std::move(import_set));
373 String gv_name;
374
375 // All global definitions must be functions.
376 BaseFunc func;
377 if (auto* func_node = expr.as<BaseFuncNode>()) {
378 func = GetRef<BaseFunc>(func_node);
379 if (auto opt = func->GetAttr<String>(tvm::attr::kGlobalSymbol)) {
380 // Function literal has been annotated with it's required global symbol.
381 gv_name = opt.value();
382 }
383 } else {
384 func = relay::Function(relay::FreeVars(expr), expr, Type(), relay::FreeTypeVars(expr, mod), {});
385 }
386
387 GlobalVar main_gv;
388 auto global_var_supply = GlobalVarSupply(mod);
389 if (gv_name.empty()) {
390 // Bind function to 'main' (though rename if would clash with existing 'main').
391 main_gv = global_var_supply->FreshGlobal("main", false);
392 } else {
393 main_gv = global_var_supply->UniqueGlobalFor(gv_name, false);
394 }
395 mod->Add(main_gv, func);
396 return {mod, main_gv};
397}

399IRModule IRModule::FromExpr(const RelayExpr& expr, const Map<GlobalVar, BaseFunc>& global_funcs,
400 const Map<GlobalTypeVar, TypeData>& type_definitions) {
401 return FromExprInContext(expr, global_funcs, type_definitions).first;
402}

404void IRModuleNode::Import(const String& path) {
405 if (this->import_set_.count(path) == 0) {
406 this->import_set_.insert(path);
407 std::fstream src_file(path, std::fstream::in);
408 std::string file_contents{std::istreambuf_iterator<char>(src_file),
409 std::istreambuf_iterator<char>()};
410 auto mod_to_import = parser::ParseModule(path, file_contents, GetRef<IRModule>(this));
411 Update(mod_to_import);
412 }
413}

444TVM_REGISTER_GLOBAL("ir.Module_Add").set_body([](TVMArgs args, TVMRetValue* ret) {
445 IRModule mod = args[0];
446 GlobalVar var = args[1];
447 ObjectRef val = args[2];
448 bool update = args[3];
449 ICHECK(val->IsInstance<RelayExprNode>());
450
451 if (val->IsInstance<BaseFuncNode>()) {
452 mod->Add(var, Downcast<BaseFunc>(val), update);
453 } else if (val->IsInstance<GlobalVarNode>()) {
454 GlobalVar gv = Downcast<GlobalVar>(val);
455 auto mod_copy = IRModule(make_object<IRModuleNode>(*mod.operator->()));
456 mod_copy = relay::transform::EtaExpand(
457 /* expand_constructor */ false,
458 /* expand_global_var */ true)(mod_copy);
459 auto func = mod_copy->Lookup(gv->name_hint);
460 mod->Add(var, Downcast<relay::Function>(func), update);
461 } else {
462 auto func = relay::Function({}, Downcast<RelayExpr>(val), Type(nullptr), {});
463 mod->Add(var, func, update);
464 }
465 *ret = mod;
466});
