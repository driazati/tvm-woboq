30ObjectPtr<Object> GetSourceNameNode(const String& name) {
31 // always return pointer as the reference can change as map re-allocate.
32 // or use another level of indirection by creating a unique_ptr
33 static std::unordered_map<String, ObjectPtr<SourceNameNode>> source_map;
34
35 auto sn = source_map.find(name);
36 if (sn == source_map.end()) {
37 ObjectPtr<SourceNameNode> n = make_object<SourceNameNode>();
38 source_map[name] = n;
39 n->name = std::move(name);
40 return n;
41 } else {
42 return sn->second;
43 }
44}
45
46ObjectPtr<Object> GetSourceNameNodeByStr(const std::string& name) {
47 return GetSourceNameNode(name);
48}
49
50SourceName SourceName::Get(const String& name) { return SourceName(GetSourceNameNode(name)); }
51
52TVM_REGISTER_GLOBAL("ir.SourceName").set_body_typed(SourceName::Get);
53
54TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
55 .set_dispatch<SourceNameNode>([](const ObjectRef& ref, ReprPrinter* p) {
56 auto* node = static_cast<const SourceNameNode*>(ref.get());
57 p->stream << "SourceName(" << node->name << ", " << node << ")";
58 });
59
60TVM_REGISTER_NODE_TYPE(SourceNameNode)
61 .set_creator(GetSourceNameNodeByStr)
62 .set_repr_bytes([](const Object* n) -> std::string {
63 return static_cast<const SourceNameNode*>(n)->name;
64 });

66Span::Span(SourceName source_name, int line, int end_line, int column, int end_column) {
67 auto n = make_object<SpanNode>();
68 n->source_name = std::move(source_name);
69 n->line = line;
70 n->end_line = end_line;
71 n->column = column;
72 n->end_column = end_column;
73 data_ = std::move(n);
74}
75
76Span Span::Merge(const Span& other) const {
77 ICHECK(this->defined() && other.defined()) << "Span::Merge: both spans must be defined";
78
79 ICHECK((*this)->source_name == other->source_name);
80 return Span((*this)->source_name, std::min((*this)->line, other->line),
81 std::max((*this)->end_line, other->end_line),
82 std::min((*this)->column, other->column),
83 std::max((*this)->end_column, other->end_column));
84}
85
86TVM_REGISTER_NODE_TYPE(SpanNode);
87
88TVM_REGISTER_GLOBAL("ir.Span").set_body_typed([](SourceName source_name, int line, int end_line,
89 int column, int end_column) {
90 return Span(source_name, line, end_line, column, end_column);
91});
