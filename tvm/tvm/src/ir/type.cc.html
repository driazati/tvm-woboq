28PrimType::PrimType(runtime::DataType dtype) {
29 ObjectPtr<PrimTypeNode> n = make_object<PrimTypeNode>();
30 n->dtype = dtype;
31 data_ = std::move(n);
32}
33
34TVM_REGISTER_NODE_TYPE(PrimTypeNode);
35
36TVM_REGISTER_GLOBAL("ir.PrimType").set_body_typed([](runtime::DataType dtype) {
37 return PrimType(dtype);
38});
39
40TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
41 .set_dispatch<PrimTypeNode>([](const ObjectRef& ref, ReprPrinter* p) {
42 auto* node = static_cast<const PrimTypeNode*>(ref.get());
43 p->stream << node->dtype;
44 });

46PointerType::PointerType(Type element_type, String storage_scope) {
47 ObjectPtr<PointerTypeNode> n = make_object<PointerTypeNode>();
48 n->element_type = std::move(element_type);
49 n->storage_scope = std::move(storage_scope);
50 data_ = std::move(n);
51}
52
53TVM_REGISTER_NODE_TYPE(PointerTypeNode);
54
55TVM_REGISTER_GLOBAL("ir.PointerType")
56 .set_body_typed([](Type element_type, String storage_scope = "") {
57 return PointerType(element_type, storage_scope);
58 });
59
60TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
61 .set_dispatch<PointerTypeNode>([](const ObjectRef& ref, ReprPrinter* p) {
62 auto* node = static_cast<const PointerTypeNode*>(ref.get());
63 if (!node->storage_scope.empty()) {
64 p->stream << node->storage_scope << " ";
65 }
66 p->Print(node->element_type);
67 p->stream << '*';
68 });

110FuncType::FuncType(tvm::Array<Type> arg_types, Type ret_type, tvm::Array<TypeVar> type_params,
111 tvm::Array<TypeConstraint> type_constraints, Span span) {
112 ObjectPtr<FuncTypeNode> n = make_object<FuncTypeNode>();
113 n->arg_types = std::move(arg_types);
114 n->ret_type = std::move(ret_type);
115 n->type_params = std::move(type_params);
116 n->type_constraints = std::move(type_constraints);
117 n->span = std::move(span);
118 data_ = std::move(n);
119}
120
121TVM_REGISTER_NODE_TYPE(FuncTypeNode);
122
123TVM_REGISTER_GLOBAL("ir.FuncType")
124 .set_body_typed([](tvm::Array<Type> arg_types, Type ret_type, tvm::Array<TypeVar> type_params,
125 tvm::Array<TypeConstraint> type_constraints) {
126 return FuncType(arg_types, ret_type, type_params, type_constraints);
127 });
128
129TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
130 .set_dispatch<FuncTypeNode>([](const ObjectRef& ref, ReprPrinter* p) {

132 p->stream << "FuncType(" << node->type_params << ", " << node->arg_types << ", "
133 << node->ret_type << ", " << node->type_constraints << ")";
