42template <typename T, typename U>
43using NodeMap = std::unordered_map<T, U, ObjectPtrHash, ObjectPtrEqual>;
44
45void ErrorReporter::RenderErrors(const IRModule& module, bool use_color) {
46 // First we pick an error reporting strategy for each error.
47 // TODO(@jroesch): Spanned errors are currently not supported.
48 for (auto err : this->errors_) {
49 ICHECK(!err.span.defined()) << "attempting to use spanned errors, currently not supported";
50 }
51
52 NodeMap<GlobalVar, NodeMap<ObjectRef, std::string>> error_maps;
53
54 // Set control mode in order to produce colors;
55 if (use_color) {
56 rang::setControlMode(rang::control::Force);
57 }
58
59 for (auto pair : this->node_to_gv_) {
60 auto node = pair.first;
61 auto global = Downcast<GlobalVar>(pair.second);
62
63 auto has_errs = this->node_to_error_.find(node);
64
65 ICHECK(has_errs != this->node_to_error_.end());
66
67 const auto& error_indices = has_errs->second;
68
69 std::stringstream err_msg;
70
71 err_msg << rang::fg::red;
72 err_msg << " ";
73 for (auto index : error_indices) {
74 err_msg << this->errors_[index].what() << "; ";
75 }
76 err_msg << rang::fg::reset;
77
78 // Setup error map.
79 auto it = error_maps.find(global);
80 if (it != error_maps.end()) {
81 it->second.insert({node, err_msg.str()});
82 } else {
83 error_maps.insert({global, {{node, err_msg.str()}}});
84 }
85 }
86
87 // Now we will construct the fully-annotated program to display to
88 // the user.
89 std::stringstream annotated_prog;
90
91 // First we output a header for the errors.
92 annotated_prog << rang::style::bold << std::endl
93 << "Error(s) have occurred. The program has been annotated with them:" << std::endl
94 << std::endl
95 << rang::style::reset;
96
97 // For each global function which contains errors, we will
98 // construct an annotated function.
99 for (auto pair : error_maps) {
100 auto global = pair.first;
101 auto err_map = pair.second;
102 auto func = module->Lookup(global);
103
104 // We output the name of the function before displaying
105 // the annotated program.
106 annotated_prog << rang::style::bold << "In `" << global->name_hint << "`: " << std::endl
107 << rang::style::reset;
108
109 // We then call into the Relay printer to generate the program.
110 //
111 // The annotation callback will annotate the error messages
112 // contained in the map.
113 annotated_prog << AsText(func, false, [&err_map](const ObjectRef& expr) {
114 auto it = err_map.find(expr);
115 if (it != err_map.end()) {
116 ICHECK_NE(it->second.size(), 0);
117 return it->second;
118 } else {
119 return std::string("");
120 }
121 });
122 }
123
124 auto msg = annotated_prog.str();
125
126 if (use_color) {
127 rang::setControlMode(rang::control::Auto);
128 }
129
130 // Finally we report the error, currently we do so to LOG(FATAL),
131 // it may be good to instead report it to std::cout.
132 LOG(FATAL) << annotated_prog.str() << std::endl;
133}
134
135void ErrorReporter::ReportAt(const GlobalVar& global, const ObjectRef& node,
136 const CompileError& err) {
137 size_t index_to_insert = this->errors_.size();
138 this->errors_.push_back(err);
139 auto it = this->node_to_error_.find(node);
140 if (it != this->node_to_error_.end()) {
141 it->second.push_back(index_to_insert);
142 } else {
143 this->node_to_error_.insert({node, {index_to_insert}});
144 }
145 this->node_to_gv_.insert({node, global});
146}
