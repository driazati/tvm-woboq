121 /*! \brief The internal map. */
122 std::unordered_map<std::string, T> name_map;
123};

125/*! \brief A stack of scopes.
126 *
127 * In order to properly handle scoping we must maintain a stack of scopes.
128 *
129 * A stack allows users to write programs which contain repeated variable
130 * names and to properly handle both nested scopes and removal of variables
131 * when they go out of scope.
132 *
133 * This is the classic approach to lexical scoping.
134 */
135template <typename T>
136class ScopeStack {
137 private:
138 std::vector<Scope<T>> scope_stack;
139 std::unordered_map<std::string, T> free_vars;
140
141 public:
142 /*! \brief Adds a variable binding to the current scope. */
143 void Add(const std::string& name, const T& value) {
144 if (!this->scope_stack.size()) {
145 LOG(FATAL) << "internal issue";
146 }
147 this->scope_stack.back().name_map.insert({name, value});
148 }
149
150 void AddFreeVar(const std::string& name, const T& value) { free_vars.insert({name, value}); }
151
152 /*! \brief Looks up a variable name in the scope stack returning the matching variable
153 * in most recent scope. */
154 T Lookup(const std::string& name) {
155 for (auto scope = this->scope_stack.rbegin(); scope != this->scope_stack.rend(); ++scope) {
156 auto it = scope->name_map.find(name);
157 if (it != scope->name_map.end()) {
158 return it->second;
159 }
160 }
161
162 // Check if we bound a free variable declaration.
163 auto it = free_vars.find(name);
164 if (it != free_vars.end()) {
165 return it->second;
166 }
167
168 return T();
169 }
170
171 /*! \brief Adds a fresh scope. */
172 void PushStack() { this->scope_stack.push_back(Scope<T>()); }
173
174 /*! \brief Removes the most recent scope. */
175 void PopStack() { this->scope_stack.pop_back(); }
176};

182/*! \brief A table of interning strings as global function and type names. */
183template <typename T>
184struct InternTable {
185 /*! \brief The internal table mapping strings to a unique allocation. */
186 std::unordered_map<std::string, T> table;
187 DiagnosticContext* ctx;
188
189 /*! \brief Add the unique allocation. */
190 void Add(const std::string& name, const T& t) {
191 auto it = table.find(name);
192 if (it != table.end()) {
193 throw DuplicateKeyError("duplicate key name in intern table");
194 } else {
195 table.insert({name, t});
196 }
197 }
198
199 /*! \brief Return the unique allocation. */
200 Optional<T> Get(const std::string& name) const {
201 auto it = table.find(name);
202 if (it != table.end()) {
203 return Optional<T>(it->second);
204 } else {
205 return Optional<T>();
206 }
207 }
208};
209
210GlobalVar AddOrGet(InternTable<GlobalVar>* table, const std::string& name) {
211 auto var = table->Get(name);
212 if (var) {
213 return var.value();
214 } else {
215 auto gvar = GlobalVar(name);
216 table->Add(name, gvar);
217 return gvar;
218 }
219}
220
221GlobalTypeVar AddOrGet(InternTable<GlobalTypeVar>* table, const std::string& name,
222 TypeKind kind = TypeKind::kType) {
223 auto var = table->Get(name);
224 if (var) {
225 auto tvar = var.value();
226 TypeKind& tvar_kind = const_cast<TypeKind&>(tvar->kind);
227 tvar_kind = kind;
228 return tvar;
229 } else {
230 auto gvar = GlobalTypeVar(name, kind);
231 table->Add(name, gvar);
232 return gvar;
233 }
234}

236/*! \brief The parser class is the main interface to the parser.
237 * the parser is not currently exposed beyond this .cc file.
238 *
239 * The parser is initialized with a diagnostic context, an
240 * operator table, and a token stream.
241 *
242 * The rest of the internal state is used to map the human readable
243 * form to in-memory IR representation.
244 *
245 * The main entry point to the parser are a set of parsing methods
246 * such as `ParseModule` and `ParseExpr`.
247 *
248 * As with traditional recursive descent parsers the parsing methods
249 * are factored recursively just as one would do with a formal language
250 * grammar.
251 *
252 * You can view a recursive descent parser as a human friendly way to specify
253 * a state machine, and thus this factoring is necessary as the 'state' of this
254 * machine is the combination of the current parsing method and the next token.
255 *
256 * Parsing proceeds by matching a token and then dispatching to the appropriate
257 * method to parse the next tokens in the stream.
258 *
259 * For example if we are parsing a type and encounter a "Tensor" token we switch
260 * into a mode for parsing `[`, a shape, a comma, a data type and then a `]`.
261 *
262 * Certain matches like this are unambiguous and proceed in a straight line fashion
263 * once the initial token is found. Other parsing is more complex and requires some
264 * tricks to correctly parse.
265 *
266 * For example when we find a '(' in an expression context, it may be part of
267 * a tuple, the arguments to a call, or a parenthesized expression. The below code
268 * disambiguate these cases by factoring expression parsing into a series of methods
269 * which encode the parsing context and thus how to interpret the parenthesis.
270 *
271 * For more information one should be able to read the code in order starting with
272 * `ParseModule` or `ParseExpr`.
273 */

352 /*! \brief Examine the next token in the stream, the current parser is configured to be
353 * whitespace insensitive so we will skip all whitespace or comment tokens. */
354 Token Peek() {
355 // For now we ignore all whitespace tokens and comments.
356 // We can tweak this behavior later to enable white space sensitivity in the parser.
357 while (pos < static_cast<int64_t>(tokens.size()) && ignore_whitespace &&
358 (tokens.at(pos)->token_type == TokenType::kWhitespace ||
359 tokens.at(pos)->token_type == TokenType::kNewline ||
360 tokens.at(pos)->token_type == TokenType::kLineComment ||
361 tokens.at(pos)->token_type == TokenType::kComment)) {
362 pos++;
363 }
364
365 if (pos < static_cast<int64_t>(tokens.size())) {
366 return Token(this->tokens.at(pos));
367 } else {
368 return Token::Null();
369 }
370 }

372 /*! \brief Lookahead by N tokens.
373 * \param n The number of tokens to lookahead.
374 * \return The Nth token.
375 */
376 Token Lookahead(int n) {
377 ICHECK_GE(n, 1) << "lookahead is only valid when n >= 1";
378
379 // We intend to skip n - 1 tokens, then return the nth.
380 auto old_pos = pos;
381 for (int i = 0; i < n - 1; i++) {
382 Peek();
383 pos++;
384 }
385
386 auto tok = Peek();
387 pos = old_pos;
388 return tok;
389 }
390
391 /*! \brief Consume a token, this method is the lowest level way to consume a token
392 * and will not ignore white space or look ahead in anyway.
393 *
394 * /param token_type The token type to match.
395 */
396 void Consume(const TokenType& token_type) {
397 if (tokens[pos]->token_type != token_type) {
398 this->diag_ctx.EmitFatal(Diagnostic::Error(tokens[pos]->span)
399 << "expected a " << Pretty(token_type) << " found "
400 << Pretty(Peek()->token_type));
401 }
402 pos++;
403 }
404
405 /*! Match a token in the stream, this will first invoke Peek, ignoring tokens such
406 * as whitespace or comments returning the first meaningful token.
407 *
408 * We then try and consume the requested token, this will trigger an error if the
409 * current token does not match the token_type.
410 */
411 Token Match(const TokenType& token_type) {
412 auto tok = Peek();
413 Consume(token_type);
414 return tok;
415 }
416
417 /*! Conditionally consume a token when it matches, this will never trigger an error
418 * as we guard against consuming the token before we do.
419 *
420 * Useful for matching optional tokens, effectively looksahead by one.
421 */
422 bool WhenMatch(const TokenType& token_type) {
423 VLOG(9) << "Parser::WhenMatch: Peek() == " << Peek();
424 if (Peek()->token_type == token_type) {
425 Consume(token_type);
426 return true;
427 } else {
428 return false;
429 }
430 }

573 template <typename R>
574 R WithSpan(std::function<R()> parser) {
575 auto start_span = Peek()->span;
576 VLOG(9) << "WithSpan: start_span = " << start_span;
577 R ast = parser();
578 if (ast.defined()) {
579 // The token at the head of the stream is now 1 past where we parsed. So we find its start
580 // position as its start and end, so that when we merge we only grow the spanned region
581 // to the start of the current stream.
582 auto span_pos = pos - 1;
583 while ((tokens.at(span_pos)->token_type == TokenType::kWhitespace ||
584 tokens.at(span_pos)->token_type == TokenType::kNewline ||
585 tokens.at(span_pos)->token_type == TokenType::kLineComment ||
586 tokens.at(span_pos)->token_type == TokenType::kComment)) {
587 span_pos--;
588 }
589 auto end_token = tokens.at(span_pos);
590 VLOG(9) << "WithSpan: end_span = " << end_token->span;
591 ast->span = start_span.Merge(end_token->span);
592 }
593 return ast;
594 }

596 struct MetaRef {
597 std::string type_key;
598 uint64_t node_index;
599 Span span;
600 MetaRef(std::string type_key, uint64_t node_index, Span span)
601 : type_key(type_key), node_index(node_index), span(span) {}
602 };
603
604 MetaRef MetaRefFromToken(const Token& tok) {
605 Call ref = Downcast<Call>(tok->data);
606 auto attrs = ref->attrs.as<MetaRefAttrs>();
607 auto type_key = attrs->node_type_key;
608 auto index = attrs->node_index;
609 return MetaRef(type_key, index, ref->span);
610 }
611
612 /*! \brief Parse a meta reference of the form `meta[type_key][node_index]`.
613 * For example `meta[relay.Constant][0]` references the first constant, `meta[relay.Constant][1]`
614 * the second, and so on.
615 */
616 ObjectRef ParseMetaRef() {
617 auto meta_ref_tok = Match(TokenType::kMetaReference);
618 auto meta_ref = MetaRefFromToken(meta_ref_tok);
619 auto it = this->meta_table.find(meta_ref.type_key);
620 if (it != this->meta_table.end()) {
621 auto nodes = (*it).second;
622 if (meta_ref.node_index < nodes.size()) {
623 return nodes[meta_ref.node_index];
624 } else {
625 this->diag_ctx.Emit(Diagnostic::Error(meta_ref.span)
626 << "the node index `" << meta_ref.node_index
627 << "` is out of bounds for `" << meta_ref.type_key << "`");
628 return ObjectRef();
629 }
630 } else {
631 this->diag_ctx.Emit(Diagnostic::Error(meta_ref.span)
632 << "no entry in the meta table for `" << meta_ref.type_key << "`");
633 return ObjectRef();
634 }
635 }
636 /*! \brief Parses a sequence beginning with a start token, separated by a seperator token, and
637 * ending with a stop token.
638 *
639 * The simple form being <start> (<parse()> <seperator>)* <stop>.
640 *
641 * This also provides a fourth argument which is allowed to run when the sequence which matches
642 * the inner sequence can not proceed.
643 *
644 * This is useful for parsing things like attributes which don't match the standard expression
645 * parsers but are contained within the stop token.
646 */
647 template <typename T>
648 Array<T> ParseSequence(TokenType start, TokenType sep, TokenType stop, std::function<T()> parse,
649 std::function<bool()> before_stop = nullptr) {
650 VLOG(9) << "Parser::ParseSequence: start=" << ToString(start) << " sep=" << ToString(sep)
651 << " stop=" << ToString(stop);
652 Match(start);
653
654 // This is for the empty arguments list case, if we have <start> <leftovers> <stop> token stream
655 // we must parse leftovers, then match a stop token.
656 if (before_stop) {
657 auto did_parse = before_stop();
658 if (did_parse) {
659 Match(stop);
660 return {};
661 }
662 }
663
664 // This is the case in which we find an empty arguments lists and no leftovers.
665 if (WhenMatch(stop)) {
666 return Array<T>();
667 } else {
668 VLOG(9) << "Parser::ParseSequence: parse first";
669 auto data = parse();
670 Array<T> elements = {data};
671
672 if (WhenMatch(stop)) {
673 return elements;
674 // parse '( expr ',' * ')'
675 } else if (WhenMatch(sep)) {
676 while (true) {
677 VLOG(9) << "Parser::ParseSequence: parse element";
678 if (WhenMatch(stop)) {
679 break;
680 } else {
681 // If before stop is
682 if (before_stop) {
683 auto did_parse = before_stop();
684 if (did_parse) {
685 Match(stop);
686 return elements;
687 }
688 }
689 auto data = parse();
690 WhenMatch(sep);
691 elements.push_back(data);
692 }
693 }
694 return elements;
695 } else {
696 auto next = Peek();
697 this->diag_ctx.EmitFatal(Diagnostic::Error(next->span)
698 << "expected a " << Pretty(stop) << " found "
699 << Pretty(next->token_type));
700 return Array<T>(nullptr);
701 }
702 }

705 /*! \brief Parse a full IRModule. */
706 IRModule ParseModule() {
707 // Parse the semver header at the top of the module.
708 this->version = ParseSemVer();
709 // Parse the definitions.
710 auto defs = ParseDefinitions();
711 // Parse the metadata section at the end.
712 auto metadata = ParseMetadata();
713
714 Match(TokenType::kEndOfFile);
715
716 for (auto type_def : defs.types) {
717 module->AddTypeDef(type_def->header, type_def);
718 }
719
720 for (auto func : defs.funcs) {
721 module->Add(func.global, func.function, true);
722 }
723
724 return module;
725 }

727 /*! \brief Parse the semantic versioning header. */
728 SemVer ParseSemVer(bool required = true) {
729 if (Peek()->token_type == TokenType::kVersion) {
730 auto version = Match(TokenType::kVersion);
731 // TODO(@jroesch): we currently only support 0.0.5.

783 /*! \brief Parse zero or more Relay type definitions. */
784 TypeData ParseTypeDef() {
785 // Match the `type` keyword.
786 Match(TokenType::kTypeDef);
787 // Parse the type's identifier.
788 auto type_tok = Match(TokenType::kIdentifier);
789 auto type_id = type_tok.ToString();
790 auto type_global = AddOrGet(&type_names, type_id, TypeKind::kAdtHandle);
791
792 Array<TypeVar> generics;
793
794 bool should_pop = false;
795 if (Peek()->token_type == TokenType::kLSquare) {
796 // If we have generics we need to add a type scope.
797 PushTypeScope();
798 should_pop = true;
799 generics = ParseSequence<TypeVar>(
800 TokenType::kLSquare, TokenType::kComma, TokenType::kRSquare, [&]() {
801 auto type_var_name = Match(TokenType::kIdentifier).ToString();
802 return BindTypeVar(type_var_name, TypeKind::kType);
803 });
804 }
805
806 Array<tvm::Constructor> ctors;
807 if (Peek()->token_type == TokenType::kLCurly) {
808 // Parse the list of constructors.
809 ctors = ParseSequence<tvm::Constructor>(
810 TokenType::kLCurly, TokenType::kComma, TokenType::kRCurly, [&]() {
811 // First match the name of the constructor.
812 auto ctor_tok = Match(TokenType::kIdentifier);
813 auto ctor_name = ctor_tok.ToString();
814
815 Constructor ctor;
816 // Match the optional field list.
817 if (Peek()->token_type != TokenType::kOpenParen) {
818 ctor = tvm::Constructor(ctor_name, {}, type_global);
819 } else {
820 auto arg_types =
821 ParseSequence<Type>(TokenType::kOpenParen, TokenType::kComma,
822 TokenType::kCloseParen, [&]() { return ParseType(); });
823 ctor = tvm::Constructor(ctor_name, arg_types, type_global);
824 }
825
826 ICHECK(ctor.defined());
827
828 try {
829 this->ctors.Add(ctor_name, ctor);
830 } catch (const DuplicateKeyError& e) {
831 this->diag_ctx.EmitFatal(Diagnostic::Error(ctor_tok->span)
832 << "a constructor with the name "
833 << "`" << ctor_name << "` "
834 << "was previously defined");
835 }
836
837 return ctor;
838 });
839 }
840
841 // Now pop the type scope.
842 if (should_pop) {
843 PopTypeScopes(1);
844 }
845
846 return TypeData(type_global, generics, ctors);
847 }

873 /*! \brief Parse a single Relay expression. */
874 Expr ParseExpr() {
875 VLOG(9) << "Parser::ParseExpr";
876 return WithSpan<Expr>([this] {
877 std::vector<Expr> exprs;
878
879 while (true) {
880 VLOG(9) << "Parser::ParseExpr: parsing a single expression";
881 auto next = Peek();
882 switch (next->token_type) {
883 // For graph or let, match first rhs, then invoke ParseBindingExpr
884 // ParseBindingExpression then parse_lhs() parse_rhs() ';' continue
885 case TokenType::kLCurly: {
886 // NB: Might need to optimize to remove deep recursion.
887 // Stack should only grow proportionally to the number of
888 // nested scopes.
889 // Parses `{` expression `}`.
890 auto block = WithSpan<Expr>([&]() {
891 return Bracket<Expr>(TokenType::kLCurly, TokenType::kRCurly, [&]() {
892 PushScope();
893 auto expr = ParseExpr();
894 PopScopes(1);
895 return expr;
896 });
897 });
898 exprs.push_back(block);
899 break;
900 }
901 case TokenType::kFreeVar: {
902 Consume(TokenType::kFreeVar);
903 auto var_token = Match(TokenType::kLocal);
904
905 Type type;
906 if (WhenMatch(TokenType::kColon)) {
907 type = ParseType();
908 } else {
909 type = IncompleteType();
910 }
911
912 BindFreeVar(var_token.ToString(), type);
913 break;
914 }
915 // Parses `let ...`;
916 case TokenType::kLet:
917 exprs.push_back(ParseBindingExpr());
918 break;
919 case TokenType::kMatch:
920 case TokenType::kPartialMatch: {
921 bool is_total = next->token_type == TokenType::kMatch;
922 Consume(next->token_type);
923 exprs.push_back(ParseMatch(is_total));
924 break;
925 }
926
927 // %x ...
928 case TokenType::kGraph:
929 if (Lookahead(2)->token_type == TokenType::kEqual) {
930 exprs.push_back(ParseBindingExpr());
931 break;
932 }
933 // intentional fall through here.
934 default: {
935 exprs.push_back(ParseExprBinOp());
936 break;
937 }
938 }
939
940 if (!WhenMatch(TokenType::kSemicolon)) {
941 break;
942 }
943 }
944
945 ICHECK_GE(exprs.size(), 1);
946
947 if (exprs.size() == 1) {
948 // ICHECK(exprs[0].defined() && exprs[0]->span.defined())
949 // << "parser must set expression spans.\n"
950 // << exprs[0];
951 return exprs[0];
952 } else {
953 auto body = exprs.back();
954 exprs.pop_back();
955 while (exprs.size()) {
956 auto value = exprs.back();
957 ICHECK(value->span.defined()) << "parser must set expression spans.";
958 exprs.pop_back();
959 body = relay::Let(Var("", IncompleteType()), value, body, value->span.Merge(body->span));
960 }
961 ICHECK(body->span.defined()) << "parser must set expression spans.";
962 return body;
963 }
964 });
965 }

967 /*! \brief Parse a "binding expression"; an expression where
968 * a graph or let variable is bound.
969 *
970 * In order to avoid stack overflow this is implemented in a special
971 * iterative way to keep stack depth constant in a long chain of bindings.
972 */
973 Expr ParseBindingExpr() {
974 // We use a loop here so that the stack depth
975 // does not grow linearly with a sequence of
976 // graph or let bindings.
977 //
978 // Assuming we start at call depth k, we will
979 // enter k + c call frames to parse the RHS
980 // of the bindings where `c` is the depth
981 // of recursion needed by RHS.
982 //
983 // If RHS is a call expresssion the c=1.
984 //
985 // Once we have parsed the RHS we will be
986 // back at depth K, and will return to
987 // this loop header to parse another
988 // graph or let binding.
989 //
990 // This ensures for n sequential bindings
991 // the call depth will be the same before
992 // and after parsing the n bindings.
993 VLOG(9) << "Parser::ParseBindingExpr";
994 std::vector<std::tuple<Var, Expr, Span>> bindings;
995 int scopes = 0;
996
997 while (true) {
998 auto next = Peek();
999 if (next->token_type == TokenType::kGraph && Lookahead(2)->token_type == TokenType::kEqual) {
1000 Match(TokenType::kGraph);
1001 Match(TokenType::kEqual);
1002 auto val = this->ParseExprBinOp();
1003 Match(TokenType::kSemicolon);
1004 AddGraphBinding(next, val);
1005 } else if (next->token_type == TokenType::kLet) {
1006 auto span = next->span;
1007 // Parse the 'let'.
1008 Consume(TokenType::kLet);
1009
1010 // Parse the local '%<id>'.
1011 auto local_tok = Match(TokenType::kLocal);
1012 auto string = local_tok.ToString();
1013
1014 // Parse the optional type annotation (':' <type>).
1015 Type type;
1016 if (WhenMatch(TokenType::kColon)) {
1017 type = ParseType();
1018 }
1019
1020 auto var = BindVar(string, type);
1021
1022 // Parse the '=';
1023 Match(TokenType::kEqual);
1024
1025 // Parse the body, and the ';'.
1026 auto val = this->ParseExprBinOp();
1027 Consume(TokenType::kSemicolon);
1028
1029 // Add the bindings to the local data structure.
1030 std::tuple<relay::Var, relay::Expr, Span> tuple(var, val, span);
1031 bindings.push_back(tuple);
1032 scopes++;
1033 PushScope();
1034 } else {
1035 // This is the only case we will increase the stack
1036 // depth.
1037 //
1038 // If we parse a program which is a sequence of N bindings
1039 // followed by a single body expression we will end up with
1040 // a call depth of 3, the first call to ParseExpr, then
1041 // ParseBindingExpr, then finally ParseExpr once more.
1042
1043 auto body = this->ParseExpr();
1044
1045 // Remove the same number of scopes we added.
1046 PopScopes(scopes);
1047
1048 if (bindings.size() == 0) {
1049 return body;
1050 } else {
1051 // We can now build the let binding up backwards.
1052 for (auto binding = bindings.rbegin(); binding != bindings.rend(); binding++) {
1053 auto span = body->span.Merge(std::get<2>(*binding));
1054 body = relay::Let(std::get<0>(*binding), std::get<1>(*binding), body, span);
1055 }
1056 return body;
1057 }
1058 }
1059 }
1060 }

1157 });

1264 Expr ParseExprBinOp() {
1265 VLOG(9) << "Parser::ParseExprBinOp";
1266 return WithSpan<Expr>([this] {
1267 // We must parse at least one expression, the default
1268 // case is that there is no operator and we will fall
1269 // through.
1270 std::vector<Expr> exprs;
1271 Expr expr = WithSpan<Expr>([this] { return ParseCallExpr(); });
1272
1273 exprs.push_back(expr);
1274
1275 // Now we parse an optional op.
1276 std::vector<Rule> ops;
1277
1278 // We will now parse 0 or more operator occurrences.
1279 while (true) {
1280 auto opt_op = ParseOp();
1281
1282 // If we didn't parse one we done.
1283 if (opt_op.size() == 0) {
1284 break;
1285 }
1286
1287 // Read the operation we parsed;
1288 auto op = opt_op[0];
1289
1290 Expr right = WithSpan<Expr>([this] { return ParseCallExpr(); });
1291 ICHECK(right->span.defined());
1292
1293 // If the operator stack is empty
1294 // we parse an operator and expression
1295 // and push them to stacks, then
1296 // continue.
1297 if (ops.size() == 0) {
1298 ops.push_back(op);
1299 exprs.push_back(right);
1300 continue;
1301 }
1302
1303 if (op.precedence > ops.back().precedence ||
1304 (op.precedence == ops.back().precedence && op.left_assoc == false)) {
1305 ops.push_back(op);
1306 exprs.push_back(right);
1307 continue;
1308 }
1309
1310 while (ops.size() && (op.precedence < ops.back().precedence ||
1311 (op.precedence == ops.back().precedence && op.left_assoc == true))) {
1312 Rule new_op = ops.back();
1313 ops.pop_back();
1314 Expr right = exprs.back();
1315 exprs.pop_back();
1316 Expr left = exprs.back();
1317 exprs.pop_back();
1318 ICHECK(new_op.op.defined()) << "a call op must be set " << new_op.op;
1319 exprs.push_back(
1320 relay::Call(new_op.op, {left, right}, Attrs(), {}, left->span.Merge(right->span)));
1321 }
1322
1323 exprs.push_back(right);
1324 ops.push_back(op);
1325 }
1326
1327 while (ops.size()) {
1328 Rule new_op = ops.back();
1329 ops.pop_back();
1330 Expr right = exprs.back();
1331 exprs.pop_back();
1332 Expr left = exprs.back();
1333 exprs.pop_back();
1334 ICHECK(new_op.op.defined()) << "a call op must be set " << new_op.op;
1335 exprs.push_back(
1336 relay::Call(new_op.op, {left, right}, Attrs(), {}, left->span.Merge(right->span)));
1337 }
1338
1339 ICHECK_EQ(ops.size(), 0) << "No operations should be left on the operation stack.";
1340
1341 ICHECK_EQ(exprs.size(), 1)
1342 << "Only a single expression should be left on the expression stack.";
1343
1344 return exprs[0];
1345 });
1346 }

1407 Expr ParseCallArgs(Expr op) {

1409
1410 VLOG(9) << "Parser::ParseCallArgs";

1412 std::string op_key;

1424 [&] {
1425 auto is_ident = Lookahead(1)->token_type == TokenType::kIdentifier;
1426 auto next_is_equal = Lookahead(2)->token_type == TokenType::kEqual;
1427 auto is_pretty_attrs = is_ident && next_is_equal;
1428 auto is_meta_next = Lookahead(1)->token_type == TokenType::kMetaReference;
1429 // TODO(@jroesch): might not handle trailing comma
1430 auto last_meta = Lookahead(2)->token_type == TokenType::kCloseParen;
1431 auto is_meta_attrs = is_meta_next && last_meta;
1432
1433 if (is_pretty_attrs || is_meta_attrs) {
1434 if (is_meta_attrs) {
1435 auto meta_ref = ParseMetaRef();
1436 if (meta_ref.as<BaseAttrsNode>()) {
1437 attrs = Downcast<Attrs>(meta_ref);
1438 } else {
1439 // Not awesome parsing code here.
1440 this->pos--;
1441 return false;
1442 }
1443 } else {
1444 auto raw_attrs = ParseAttrs();
1445 if (is_op && op_key.size()) {
1446 auto attr_obj = tvm::ReflectionVTable::Global()->CreateObject(op_key, raw_attrs);
1447 ICHECK(attr_obj.defined());
1448 attrs = Downcast<Attrs>(attr_obj);
1449 } else if (raw_attrs.count("attrs_type_key")) {
1450 String attr_key = Downcast<String>(raw_attrs["attrs_type_key"]);
1451 if (attr_key.size()) {
1452 raw_attrs.erase("attrs_type_key");
1453 auto attr_obj =
1454 tvm::ReflectionVTable::Global()->CreateObject(attr_key, raw_attrs);
1455 ICHECK(attr_obj.defined());
1456 attrs = Downcast<Attrs>(attr_obj);
1457 }
1458 } else {
1459 this->diag_ctx.EmitFatal(Diagnostic::Error(op->span)
1460 << "unable to determine the 'attrs_type_key' with which "
1461 "to represent the call attributes for this operator");
1462 }
1463 }
1464 return true;
1465 }
1466 return false;
1467 });
1468
1469 if (!attrs.defined()) {
1470 if (is_op && op_key.size()) {
1471 auto attr_obj = tvm::ReflectionVTable::Global()->CreateObject(op_key, {});
1472 ICHECK(attr_obj.defined());
1473 attrs = Downcast<Attrs>(attr_obj);
1474 }
1475 }
1476
1477 // TODO(@jroesch): in a secondary pass adjust spans.
1478 return Expr(Call(op, args, attrs, {}));
1479 } else {
1480 return Expr();
1481 }
1482
1483 return Expr();
1484 }

1486 Expr ParseCallExpr() {
1487 VLOG(9) << "Parser::ParseCallExpr";
1488 return WithSpan<Expr>([this] {
1489 Expr expr = ParseAtomicExpr();
1490 // Parse as many call args as possible, building up expression
1491 //
1492 // NB(@jroesch): this seems like a hack but in order to parse curried functions
1493 // and avoid complex grammar we will parse multiple call lists in a row.
1494 while (Peek()->token_type == TokenType::kOpenParen) {
1495 auto new_expr = ParseCallArgs(expr);
1496
1497 if (new_expr.defined()) {
1498 expr = new_expr;
1499 } else {
1500 break;
1501 }
1502 }
1503
1504 // We need a zero-arity case for constructors.
1505 if (auto ctor_node = expr.as<ConstructorNode>()) {
1506 if (ctor_node->inputs.size() == 0) {
1507 return Expr(Call(expr, {}));
1508 }
1509 }
1510
1511 return expr;
1512 });
1513 }

1528 Expr ParseAtomicExpr() {
1529 VLOG(9) << "Parser::ParseAtomicExpr";
1530 Expr expr = WithSpan<Expr>([this] {
1531 auto next = Peek();
1532 switch (next->token_type) {
1533 case TokenType::kInteger:
1534 case TokenType::kFloat: {
1535 Consume(next->token_type);
1536 auto number = NumberToNDArray(next);
1537 Expr e = Constant(number, next->span);
1538 ICHECK(e->span.defined()) << "constant spans must be defined";
1539 return e;
1540 }
1541 case TokenType::kBoolean: {
1542 Consume(TokenType::kBoolean);
1543 int64_t value = Downcast<tvm::Integer>(next->data).IntValue();
1544 Expr e = Constant(support::BoolToNDArray(value), next->span);
1545 ICHECK(e->span.defined()) << "constant spans must be defined";
1546 return e;
1547 }
1548 // Parse a local of the form `%x`.
1549 case TokenType::kLocal: {
1550 Consume(TokenType::kLocal);
1551 return Expr(LookupLocal(next));
1552 }
1553 // Parse a local of the form `@x`.
1554 case TokenType::kGlobal: {
1555 auto global_name = next.ToString();
1556 Consume(TokenType::kGlobal);
1557 auto global = AddOrGet(&global_names, global_name);
1558 return Expr(global);
1559 }
1560 // Parse a local of the form `x`.
1561 // Right now we fail to parse `x.y`.
1562 case TokenType::kIdentifier: {
1563 auto ctor = ctors.Get(next.ToString());
1564 if (ctor) {
1565 Consume(TokenType::kIdentifier);
1566 return Expr(ctor.value());
1567 } else {
1568 auto spanned_idents = ParseHierarchicalName();
1569 auto idents = spanned_idents.data;
1570 auto span = spanned_idents.span;
1571 return GetOp(GetHierarchicalName(idents), span);
1572 }
1573 }
1574 case TokenType::kGraph: {
1575 Consume(TokenType::kGraph);
1576 return LookupGraphBinding(next);
1577 }
1578 case TokenType::kMetaReference: {
1579 return Downcast<Expr>(ParseMetaRef());
1580 }
1581 case TokenType::kFn: {
1582 Consume(TokenType::kFn);
1583 Expr e = ParseFunctionDef();
1584 ICHECK(e->span.defined()) << "function spans must be defined.\n" << e;
1585 return e;
1586 }
1587 case TokenType::kIf: {
1588 Expr e = ParseIf();
1589 return e;
1590 }
1591 case TokenType::kRef: {
1592 Consume(TokenType::kRef);
1593 Match(TokenType::kOpenParen);
1594 auto ref_value = ParseExpr();
1595 Match(TokenType::kCloseParen);
1596 return static_cast<Expr>(RefCreate(ref_value));
1597 }
1598 case TokenType::kRefRead: {
1599 return WithSpan<Expr>([&]() {
1600 Consume(TokenType::kRefRead);
1601 Match(TokenType::kOpenParen);
1602 auto ref = ParseExpr();
1603 Match(TokenType::kCloseParen);
1604 return static_cast<Expr>(RefRead(ref));
1605 });
1606 }
1607 case TokenType::kRefWrite: {
1608 return WithSpan<Expr>([&]() {
1609 Consume(TokenType::kRefWrite);
1610 Match(TokenType::kOpenParen);
1611 auto ref = ParseExpr();
1612 Match(TokenType::kComma);
1613 auto value = ParseExpr();
1614 Match(TokenType::kCloseParen);
1615 return static_cast<Expr>(RefWrite(ref, value));
1616 });
1617 }
1618 case TokenType::kOpenParen: {
1619 Span sp = next->span;
1620 Consume(TokenType::kOpenParen);
1621 // parse '(' ')'
1622 if (WhenMatch(TokenType::kCloseParen)) {
1623 return Expr(Tuple(Array<Expr>()));
1624 } else {
1625 Expr subexpr = ParseExpr();
1626 // parse '(' expr ')'
1627 if (WhenMatch(TokenType::kCloseParen)) {
1628 return subexpr;
1629 // parse '( expr ',' * ')'
1630 } else if (WhenMatch(TokenType::kComma)) {
1631 Array<Expr> exprs = {subexpr};
1632 while (true) {
1633 if (WhenMatch(TokenType::kCloseParen)) {
1634 break;
1635 } else {
1636 auto element = ParseExpr();
1637 auto comma = Peek();
1638 if (WhenMatch(TokenType::kComma)) {
1639 sp = sp.Merge(element->span.Merge(comma->span));
1640 } else {
1641 sp = sp.Merge(element->span);
1642 }
1643 exprs.push_back(element);
1644 }
1645 }
1646 Expr tuple = Tuple(exprs, sp);
1647 ICHECK(tuple->span.defined()) << "tuple span should be defined";
1648 return tuple;
1649 }
1650 }
1651 }
1652 default: {
1653 this->diag_ctx.EmitFatal(Diagnostic::Error(next->span)
1654 << "expected an expression found " << Pretty(next->token_type));
1655 return Expr();
1656 }
1657 }
1658 });
1659
1660 if (WhenMatch(TokenType::kPeriod)) {
1661 auto token = Match(TokenType::kInteger);
1662 auto index = token.ToNumber();
1663 auto span = token->span.Merge(expr->span);
1664 VLOG(9) << "Parser::ParseAtomicExpr: tuple get item";
1665 return relay::TupleGetItem(expr, index, span);
1666 } else {
1667 return expr;
1668 }
1669 }

1671 /*! \brief Parse a hierarchical name.
1672 *
1673 * The tokenizer produces a token stream of <id1> . <id2>
1674 * and so on for names of the form `nn.conv2d`.
1675 * Currently we only use string names everywhere instead
1676 * of a notion of a hierarchical name.
1677 *
1678 * The below utility reassembles a token stream into a
1679 * single stream inserting the required periods needed
1680 * to look up registered names.
1681 */
1682 Spanned<Array<String>> ParseHierarchicalName() {
1683 Array<String> idents;
1684 Span span;
1685 while (Peek()->token_type == TokenType::kIdentifier) {
1686 auto token = Peek();
1687
1688 if (span.defined()) {
1689 span = span.Merge(token->span);
1690 } else {
1691 span = token->span;
1692 }
1693
1694 auto name = token.ToString();
1695 idents.push_back(name);
1696 Consume(TokenType::kIdentifier);
1697
1698 // Keep parsing while we see a trailing period.
1699 if (Peek()->token_type == TokenType::kPeriod) {
1700 Consume(TokenType::kPeriod);
1701 continue;
1702 } else {
1703 // No more periods means we are done!
1704 break;
1705 }
1706 }
1707
1708 return Spanned<Array<String>>(idents, span);
1709 }
1710
1711 std::string GetHierarchicalName(Array<String> idents) {
1712 ICHECK_NE(idents.size(), 0);
1713 std::stringstream hierarchical_name;
1714 int i = 0;
1715 int periods = idents.size() - 1;
1716 for (auto ident : idents) {
1717 hierarchical_name << ident;
1718 if (i < periods) {
1719 hierarchical_name << ".";
1720 i++;
1721 }
1722 }
1723 return hierarchical_name.str();
1724 }

1790 /*! \brief Parses a TVM type.
1791 *
1792 * This matches either a `Tensor[shape, dtype]`, a user defined ADT, a tuple type,
1793 * a scalar type or an incomplete type `_`.
1794 */
1795 Type ParseType() {
1796 return WithSpan<Type>([&]() -> Type {
1797 auto tok = Peek();
1798
1799 if (tok->token_type == TokenType::kOpenParen) {
1800 auto tys =
1801 ParseSequence<relay::Type>(TokenType::kOpenParen, TokenType::kComma,
1802 TokenType::kCloseParen, [&]() { return ParseType(); });
1803 return relay::TupleType(tys);
1804 } else if (WhenMatch(TokenType::kFn)) {
1805 return ParseFunctionType();
1806 } else if (WhenMatch(TokenType::kIdentifier)) {
1807 auto id = tok.ToString();
1808 if (id == "Tensor") {
1809 Match(TokenType::kLSquare);
1810 auto shape = ParseShape();
1811 Match(TokenType::kComma);
1812 auto dtype_tok = Match(TokenType::kIdentifier);
1813 auto dtype = DataType(String2DLDataType(dtype_tok.ToString()));
1814 Match(TokenType::kRSquare);
1815 return TensorType(shape, dtype);
1816 } else {
1817 auto ty = tok.ToString();
1818 if (ty.rfind("int", 0) == 0 || ty.find("float", 0) == 0 || ty.find("uint", 0) == 0 ||
1819 ty.find("bool", 0) == 0) {
1820 // Need to do better error handling here.
1821 auto dtype = DataType(String2DLDataType(tok.ToString()));
1822 return TensorType({}, dtype);
1823 } else {
1824 return ParseNonPrimitiveType(tok);
1825 }
1826 }
1827 } else if (WhenMatch(TokenType::kUnderscore)) {
1828 return IncompleteType();
1829 } else {
1830 this->diag_ctx.EmitFatal(Diagnostic::Error(tok->span)
1831 << "failed to parse type found " << tok);
1832 return Type();
1833 }
1834 });
1835 }

1883Parser InitParser(const std::string& file_name, const std::string& file_content,
1884 const Optional<IRModule>& init_module, const MetaTable& init_meta_table) {
1885 VLOG(9) << "InitParser: file_name: " << file_name << "file_content_size: " << file_content.size();
1886 SourceName src_name = SourceName::Get(file_name);
1887 Source source(src_name, file_content);
1888
1889 IRModule module;
1890 if (!init_module) {
1891 SourceMap source_map;
1892 module = IRModule({}, {}, {}, source_map);
1893 } else {
1894 module = init_module.value();
1895 }
1896
1897 module->source_map.Add(source);
1898
1899 auto diag_ctx = DiagnosticContext::Default(module);
1900 auto tokens_and_table = Tokenize(diag_ctx, source);
1901
1902 auto tokens = tokens_and_table.first;
1903 MetaTable meta_data_table = tokens_and_table.second.ToMetadata();
1904
1905 // Merge any entries in init_meta_table into anything captured in the #[metadata] section
1906 // of the file_content. Metadata references within file_content must use indexes which account
1907 // for this ordering.
1908 for (const auto& pair : init_meta_table) {
1909 Array<ObjectRef> items;
1910 if (meta_data_table.count(pair.first)) {
1911 items = meta_data_table[pair.first];
1912 }
1913 for (const auto& obj : pair.second) {
1914 items.push_back(obj);
1915 }
1916 meta_data_table.Set(pair.first, items);
1917 }
1918
1919 return Parser(module, diag_ctx, source, tokens, DefaultOpTable(), std::move(meta_data_table));
1920}

1923 const Optional<IRModule>& init_module, const MetaTable& init_meta_table) {
1924 VLOG_CONTEXT << "ParseModule";
