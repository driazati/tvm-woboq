32namespace tvm {

35using tvm::relay::transform::CreateFunctionPass;
36using tvm::transform::PassContext;
37
38/* Set to arbitrary high number, since we should never schedule in normal pass manager flow. */
39static int kMetaExpandOptLevel = 1337;
40
41TVM_REGISTER_NODE_TYPE(MetaRefAttrs);
42
43bool MetaRefRel(const Array<Type>& types, int num_inputs, const Attrs& attrs,
44 const TypeReporter& reporter) {
45 LOG(FATAL) << "need to expand before type checking";
46 return true;
47}
48
49RELAY_REGISTER_OP("parser.MetaRef")
50 .describe(R"code(A reference into the meta table.)code" TVM_ADD_FILELINE)
51 .set_attrs_type<MetaRefAttrs>()
52 .set_num_inputs(0)
53 .set_support_level(10)
54 .add_type_rel("MetaRef", MetaRefRel)
55 .set_attr<TOpIsStateful>("TOpIsStateful", false)
56 .set_attr<TNonComputational>("TNonComputational", true);
57
58Expr MetaRef(std::string type_key, uint64_t node_index) {
59 static const Op& op = Op::Get("parser.MetaRef");
60 auto attrs = make_object<MetaRefAttrs>();
61 attrs->node_type_key = tvm::String(type_key);
62 attrs->node_index = node_index;
63 return Call(op, {}, Attrs(attrs), {});
64}
65
66struct MetaRefExpander : public ExprMutator {
67 MetaTable table;
68
69 explicit MetaRefExpander(const MetaTable& table) : table(table) {}
70
71 Expr VisitExpr_(const CallNode* call) final {
72 if (auto op_node = call->op.as<OpNode>()) {
73 if (op_node->name == "parser.MetaRef") {
74 auto meta_attrs = call->attrs.as<MetaRefAttrs>();
75 ICHECK(meta_attrs) << "an internal error has occurred";
76 auto nodes = table.at(meta_attrs->node_type_key);
77 ICHECK_LT(meta_attrs->node_index, nodes.size());
78 return Downcast<Expr>(nodes[meta_attrs->node_index]);
79 }
80 }
81
82 return ExprMutator::VisitExpr_(call);
83 }
84};
85
86Function ExpandMetaRefs(const MetaTable& meta_table, const relay::Function& func) {
87 MetaRefExpander expander(meta_table);
88 return Downcast<Function>(expander.VisitExpr(func));
89}
90
91IRModule ExpandMetaRefs(const MetaTable& meta_table, const IRModule& mod) {
92 auto pass = CreateFunctionPass([&](Function func, IRModule module,
93 PassContext ctx) { return ExpandMetaRefs(meta_table, func); },
94 kMetaExpandOptLevel, "ExpandMetaRefs", {});
95
96 return pass(mod, PassContext::Create());
97}
