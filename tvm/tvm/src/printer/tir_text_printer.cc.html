47Doc TIRTextPrinter::Print(const ObjectRef& node) {
48 if (!node.defined()) return Doc::Text("(nullptr)");
49 if (node->IsInstance<StmtNode>()) {
50 return VisitStmt(Downcast<Stmt>(node));
51 } else if (node->IsInstance<AnyNode>()) {
52 return Doc::Text("?");
53 } else if (node->IsInstance<PrimExprNode>()) {
54 return VisitExpr(Downcast<PrimExpr>(node));
55 } else if (node->IsInstance<TypeNode>()) {
56 return VisitType(Downcast<Type>(node));
57 } else if (node->IsInstance<PrimFuncNode>()) {
58 return PrintPrimFunc(Downcast<PrimFunc>(node));
59 } else if (node->IsInstance<IRModuleNode>()) {
60 return PrintIRModule(Downcast<IRModule>(node));
61 } else if (node->IsInstance<ArrayNode>()) {
62 return PrintArray(node.as<ArrayNode>());
63 } else if (node->IsInstance<IterVarNode>()) {
64 return PrintIterVar(node.as<IterVarNode>());
65 } else if (node->IsInstance<RangeNode>()) {
66 return PrintRange(node.as<RangeNode>());
67 } else if (node->IsInstance<BufferNode>()) {
68 return PrintBuffer(node.as<BufferNode>());
69 } else if (node->IsInstance<DataProducerNode>()) {
70 return PrintProducer(node.as<DataProducerNode>());
71 } else if (node->IsInstance<StringObj>()) {
72 return PrintString(node.as<StringObj>());
73 } else if (node->IsInstance<BufferRegionNode>()) {
74 return PrintBufferRegion(node.as<BufferRegionNode>());
75 } else if (node->IsInstance<TargetNode>()) {
76 return Doc::Text(node.as<TargetNode>()->ToDebugString());
77 } else {
78 return this->meta_->GetMetaNode(node);
79 }
80}

82Doc TIRTextPrinter::PrintPrimFunc(const PrimFunc& prim_func) {
83 const auto* op = prim_func.operator->();
84 const auto& signature = op->func_type_annotation();
85 // collect Meta in DictAttr
86 if (prim_func->attrs.defined()) {
87 for (const auto& it : prim_func->attrs->dict) {
88 meta_collector_.Collect(it.second);
89 }
90 }
91 // collect buffers in buffer_map
92 memo_var_.clear();
93 memo_buf_.clear();
94
95 // ordered vars associated with buffers, for consistent printing
96 std::vector<Var> buffer_vars_ordered;
97
98 for (Var v : op->params) {
99 auto buffer_map_find = op->buffer_map.find(v);
100 if (buffer_map_find != op->buffer_map.end()) {
101 auto map_data = *buffer_map_find;
102 buffer_vars_ordered.push_back(map_data.first);
103 memo_buf_[map_data.second] = AllocBuf(map_data.second);
104 }
105 }
106
107 // print PrimFunc
108 Doc doc;
109 doc << "primfn"
110 << "(";
111 // print params and its type annotation
112 std::vector<Doc> params;
113 for (const auto& param : op->params) {
114 params.push_back(Print(param));
115 }
116 Doc sep;
117 doc << PrintSep(params, Doc::Indent(9, Doc::Text(", "))) << ")";
118 // print return type
119 doc << " -> " << Print(signature->ret_type);
120 // print attr
121 Doc attr_doc;
122 std::vector<Doc> attr_docs;
123 if (prim_func->attrs.defined()) {
124 for (const auto& it : op->attrs->dict) {
125 attr_docs.push_back(Doc::StrLiteral(it.first) << ": " << Print(it.second));
126 }
127 attr_doc << Doc::NewLine() << "attr = {" << PrintSep(attr_docs, Doc::Text(", ")) << "}";
128 doc << Doc::Indent(2, attr_doc);
129 }
130
131 // print all the buffers in the tree
132 if (memo_buf_.size() != 0) {
133 Doc buffer_doc;
134 std::vector<Doc> buffer_docs;
135 for (const Var& v : buffer_vars_ordered) {
136 const Buffer buf = op->buffer_map[v];
137 buffer_docs.push_back(BufferNode2Doc(buf.get(), Print(buf)));
138 }
139 buffer_doc << Doc::NewLine() << "buffers = {";
140 buffer_doc << PrintSep(buffer_docs, Doc::Indent(11, Doc::Text(",") << Doc::NewLine()));
141 doc << Doc::Indent(2, buffer_doc) << "}";
142 }
143
144 if (op->buffer_map.size() != 0) {
145 // print buffer_map
146 std::vector<Doc> buffer_map_doc;
147 for (const Var& v : buffer_vars_ordered) {
148 const Buffer buf = op->buffer_map[v];
149 buffer_map_doc.push_back(Print(v) << ": " << Print(buf));
150 }
151 doc << Doc::Indent(
152 2, Doc::NewLine() << "buffer_map = {" << PrintSep(buffer_map_doc, Doc::Text(", ")) << "}");
153 }
154
155 doc << PrintBody(op->body);
156 return doc;
157}

159Doc TIRTextPrinter::PrintIRModule(const IRModule& module) {
160 const auto* op = module.operator->();
161 Doc doc;
162
163 Doc body;
164 body << Doc::NewLine();
165 std::vector<Doc> functions;
166 for (auto it = op->functions.begin(); it != op->functions.end(); ++it) {
167 if ((*it).second.as<PrimFuncNode>()) {
168 functions.push_back(Print((*it).second));
169 }
170 }
171 body << TIRTextPrinter::PrintSep(functions, Doc::NewLine() << Doc::NewLine());
172 doc << Doc::Indent(0, body);
173 return doc;
174}

176Doc TIRTextPrinter::PrintArray(const ArrayNode* op) {
177 Doc doc;
178 doc << '[';
179 for (size_t i = 0; i < op->size(); ++i) {
180 if (i != 0) {
181 doc << ", ";
182 }
183 doc << Print(op->at(i));
184 }
185 doc << ']';
186 return doc;
187}

206Doc TIRTextPrinter::PrintBuffer(const BufferNode* op) {
207 const Buffer& buffer = GetRef<Buffer>(op);
208
209 if (meta_->InMeta(buffer)) {
210 return meta_->GetMetaNode(buffer);
211 } else if (memo_buf_.count(buffer)) {
212 return memo_buf_[buffer];
213 } else {
214 memo_buf_[buffer] = AllocBuf(buffer);
215 return BufferNode2Doc(op, memo_buf_[buffer]);
216 }

232Doc TIRTextPrinter::BufferNode2Doc(const BufferNode* buf, Doc doc) {
233 doc << Doc::Text(": Buffer(") << Print(buf->data) << ", " << PrintDType(buf->dtype) << ", "
234 << Print(buf->shape) << ", " << Print(buf->strides);
235 if (!is_zero(buf->elem_offset)) {
236 doc << ", elem_offset=" << Print(buf->elem_offset);
237 }
238 if (buf->axis_separators.size()) {
239 doc << ", axis_separators=" << Print(buf->axis_separators);
240 }
241 if (GetRef<Buffer>(buf).scope() != "global") {
242 doc << ", scope=" << Doc::StrLiteral(GetRef<Buffer>(buf).scope());
243 }
244 if (buf->data_alignment != runtime::kAllocAlignment) {
245 doc << ", align=" << buf->data_alignment;
246 }
247 if (buf->offset_factor != 1) {
248 doc << ", offset_factor=" << buf->offset_factor;
249 }
250 if (buf->buffer_type != 1) {
251 doc << ", type=" << Doc::StrLiteral("auto");
252 }
253 return doc << ")";
254}

261Doc TIRTextPrinter::PrintBufferRegion(const BufferRegionNode* op) {
262 Doc doc;
263 doc << Print(op->buffer) << "[";
264 for (size_t i = 0; i < op->region.size(); ++i) {
265 if (i != 0) {
266 doc << ", ";
267 }
268 const auto& range = op->region[i];
269 if (!is_one(range->extent)) {
270 doc << Print(range->min) << ":" << Print(range->min + range->extent);
271 } else {
272 doc << Print(range->min);
273 }
274 }
275 doc << "]";
276 return doc;
277}

408Doc TIRTextPrinter::VisitExpr_(const CallNode* op) {
409 Doc doc;
410 std::vector<Doc> func_args;
411 if (auto* ptr_op = op->op.as<OpNode>()) {
412 doc << "@" << Doc::Text(ptr_op->name) << "(";
413 if (ptr_op->name == "tir.call_llvm_pure_intrin") {
414 auto f = tvm::runtime::Registry::Get("target.llvm_get_intrinsic_name");
415 ICHECK(f != nullptr)
416 << "Cannot find target.llvm_get_intrinsic_name. Compile with USE_LLVM=On";
417 func_args.push_back(Print((*f)(Downcast<IntImm>(op->args[0])->value)));
418 for (size_t i = 1; i < op->args.size(); i++) {
419 func_args.push_back(Print(op->args[i]));
420 }
421 } else {
422 for (const auto& arg : op->args) {
423 func_args.push_back(Print(arg));
424 }
425 }
426 } else {
427 // TODO(bohan): Print out the name by he global var in the module.
428 auto* op_gvar = op->op.as<GlobalVarNode>();
429 ICHECK(op_gvar != nullptr);
430 doc << "@" << Doc::Text(op_gvar->name_hint) << "(";
431 for (const auto& arg : op->args) {
432 func_args.push_back(Print(arg));
433 }
434 }
435 doc << PrintSep(func_args, Doc::Text(", ")) << ", dtype=" << PrintDType(op->dtype) << ")";
436 return doc;
437}

611Doc TIRTextPrinter::VisitStmt_(const BlockRealizeNode* op) {
612 const auto* block_op = op->block.as<BlockNode>();
613 // print block name and block vars
614 Doc doc;
615 doc << "block([";
616 std::vector<Doc> block_var_docs;
617 for (const auto& iter_var : block_op->iter_vars) {
618 Doc block_var_doc;
619 if (is_zero(iter_var->dom->min) && iter_var->iter_type == kDataPar) {
620 block_var_doc << Print(iter_var->dom->extent);
621 } else {
622 block_var_doc << "tir.";
623 switch (iter_var->iter_type) {
624 case kDataPar:
625 block_var_doc << "range";
626 break;
627 case kCommReduce:
628 block_var_doc << "reduce_axis";
629 break;
630 case kOrdered:
631 block_var_doc << "scan_axis";
632 break;
633 case kOpaque:
634 block_var_doc << "opaque_axis";
635 break;
636 default:
637 LOG(FATAL) << "Unknown block var iter type";
638 break;
639 }
640 block_var_doc << "(" << Print(iter_var->dom->min) << ", "
641 << Print(iter_var->dom->min + iter_var->dom->extent) << ")";
642 }
643 block_var_docs.push_back(block_var_doc);
644 }
645 doc << PrintSep(block_var_docs, Doc::Text(", ")) << "], ";
646 doc << Doc::StrLiteral(block_op->name_hint) << ")";
647 std::vector<Doc> block_var_names;
648 for (const auto& iter_var : block_op->iter_vars) {
649 Doc block_var_name;
650 AllocVar(iter_var->var);
651 block_var_names.push_back(Print(iter_var->var));
652 }
653 if (!block_var_names.empty()) {
654 doc << " as [" << PrintSep(block_var_names, Doc::Text(", ")) << "]";
655 }
656 doc << " {";
657 Doc block_attr_doc;
658 // print predicate, binding, read/write tensor region, annotations
659 if (!is_one(op->predicate)) {
660 block_attr_doc << Doc::NewLine() << "where(" << Print(op->predicate) << ")";
661 }
662 for (size_t i = 0; i < block_op->iter_vars.size(); ++i)
663 block_attr_doc << Doc::NewLine() << "bind(" << Print(block_op->iter_vars[i]->var) << ", "
664 << Print(op->iter_values[i]) << ")";
665 block_attr_doc << Doc::NewLine() << "tir.reads(" << Print(block_op->reads) << ")";
666 block_attr_doc << Doc::NewLine() << "tir.writes(" << Print(block_op->writes) << ")";
667 if (!block_op->annotations.empty()) {
668 std::vector<Doc> attr_docs;
669 for (const auto& it : block_op->annotations) {
670 attr_docs.push_back(Doc::StrLiteral(it.first) << ": " << Print(it.second));
671 }
672 block_attr_doc << Doc::NewLine() << "tir.attrs({" << PrintSep(attr_docs, Doc::Text(", "))
673 << "})";
674 }
675 // print body
676 Doc body;
677 body << Doc::NewLine();
678 for (const auto& alloc_buf : block_op->alloc_buffers) {
679 body << AllocBuf(alloc_buf) << " = alloc_buffer(" << PrintDType(alloc_buf->dtype)
680 << Print(alloc_buf->shape) << ")" << Doc::NewLine();
681 }
682 for (const auto& match_buf : block_op->match_buffers) {
683 body << AllocBuf(match_buf->buffer) << " = match_buffer(" << Print(match_buf->source) << ")"
684 << Doc::NewLine();
685 }
686 if (block_op->init.defined()) {
687 Doc init_block;
688 init_block << "with init()";
689 init_block << PrintBody(block_op->init.value());
690 body << init_block << Doc::NewLine();
691 }
692 body << Print(block_op->body);
693 doc << Doc::Indent(2, block_attr_doc << body);
694 return doc;
695}

731template <typename T>
732Doc TIRTextPrinter::PrintConstScalar(DataType dtype, const T& data) {
733 Doc doc;
734 std::ostringstream os;
735 os << data;
736 if (dtype == DataType::Int(32)) {
737 doc << Doc::Text(os.str());
738 } else {
739 if (dtype.bits() == 1 && dtype.lanes() == 1 && dtype.code() == kDLUInt) {
740 doc << ((data == 1) ? "True" : "False");
741 return doc;
742 }
743 doc << Doc::Text(os.str());
744 switch (dtype.code()) {
745 case kDLInt:
746 doc << "i";
747 break;
748 case kDLUInt:
749 doc << "u";
750 break;
751 case kDLFloat:
752 doc << "f";
753 break;
754 }
755 doc << Doc::Text(std::to_string(dtype.bits()));
756 if (dtype.lanes() != 1) doc << "x" << Doc::Text(std::to_string(dtype.lanes()));
757 }
758 return doc;
759}

773Doc TIRTextPrinter::AllocVar(const Var& var) {
774 const auto& it = memo_var_.find(var);
775 if (it != memo_var_.end()) {
776 return it->second;
777 }
778 std::string name = var->name_hint.operator std::string();
779 if (name.length() == 0 || !std::isalpha(name[0])) {
780 name = "v" + name;
781 }
782 Doc val = GetUniqueName(name);
783 memo_var_[var] = val;
784 return val << ": " << Print(GetType(var));
785}
