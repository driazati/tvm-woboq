20/*!
21 * \file relay_text_printer.cc
22 * \brief Printer to print out the IR text format
23 * that can be parsed by a parser.
24 *
25 * Supports ANF, GNF in relay and metadata.
26 *
27 * Inlining heuristics:
28 * - Always inline:
29 * - GlobalVar
30 * - Constant
31 * - Op
32 * - Var
33 * - Otherwise, inline if the node is at the end of a scope and is used at most once.
34 */

55/*!
56 * \brief Print additional info about expr in comment.
57 * \param expr The expression.
58 */
59Doc RelayTextPrinter::PrintOptionalInfo(const Expr& expr) {
60 Doc doc;
61 if (!opt_info_memo_.insert(expr).second) {
62 return doc;
63 }
64 // default annotations
65 if (annotate_ == nullptr) {
66 if ((expr.as<ConstantNode>() || expr.as<CallNode>() || expr.as<VarNode>() ||
67 expr.as<FunctionNode>() || expr.as<TupleNode>() || expr.as<TupleGetItemNode>()) &&
68 (expr->checked_type_.defined() || expr->span.defined())) {
69 doc << " /*";
70 if (expr->checked_type_.defined()) {
71 doc << " ty=" << Print(expr->checked_type());
72 }
73 if (expr->span.defined()) {
74 doc << " span=" << PrintSpan(expr->span);
75 }
76 doc << " */";
77 }
78 } else {
79 std::string annotated_expr = annotate_(expr);
80 if (annotated_expr != "") {
81 doc << annotated_expr;
82 }
83 }
84 return doc;
85}

87// indent a new body
88Doc RelayTextPrinter::PrintBody(const ObjectRef& node, int indent) {
89 Doc doc;
90 Doc body;
91 doc << "{";
92 doc << Doc::Indent(indent, body << Doc::NewLine() << PrintScope(node)) << Doc::NewLine();
93 doc << "}";
94 return doc;

97// create a new scope by creating a new printer object. This allows temp var
98// numbers to be reused and prevents hoisted vars from escaping too far
99Doc RelayTextPrinter::PrintScope(const ObjectRef& node) {
100 // print in a new scope
101 doc_stack_.push_back(Doc());
102 // must print first so doc_stack_.back() reference doesn't become stale
103 Doc doc = Print(node, false, true);
104 doc = doc_stack_.back() << doc;
105 doc_stack_.pop_back();
106 return doc;
107}

109Doc RelayTextPrinter::PrintFinal(const ObjectRef& node) {
110 if (node.defined() && node->IsInstance<BaseFuncNode>() &&
111 !node->IsInstance<relay::FunctionNode>()) {
112 // Temporarily skip non-relay functions.
113 // TODO(tvm-team) enhance the code to work for all functions
114 } else if (node.as<ExprNode>()) {
115 Expr expr = Downcast<Expr>(node);
116 dg_ = DependencyGraph::Create(&arena_, expr);
117 }
118
119 Doc doc;
120 doc << PrintScope(node);
121 return doc;
122}
123
124Doc RelayTextPrinter::Print(const ObjectRef& node, bool meta, bool try_inline) {
125 bool is_non_relay_func = node.defined() && node->IsInstance<BaseFuncNode>() &&
126 !node->IsInstance<relay::FunctionNode>();
127 if (node.as<ExprNode>() && !is_non_relay_func) {
128 return PrintExpr(Downcast<Expr>(node), meta, try_inline);
129 } else if (node.as<TypeNode>()) {
130 return PrintType(Downcast<Type>(node), meta);
131 } else if (node.as<PatternNode>()) {
132 return PrintPattern(Downcast<Pattern>(node), meta);
133 } else if (node.as<IRModuleNode>()) {
134 return PrintMod(Downcast<IRModule>(node));
135 } else {
136 // default module.
137 std::ostringstream os;
138 os << node;
139 return Doc::RawText(os.str());
140 }
141}

150/*!
151 * \brief get a unique name with the corresponding prefix
152 * \param prefix The prefix of the name
153 * \return The returned name.
154 */
155Doc RelayTextPrinter::GetUniqueName(const std::string& prefix) {
156 std::string unique_prefix = prefix;
157 auto it = name_alloc_map_.find(prefix);
158 if (it != name_alloc_map_.end()) {
159 while (true) {
160 std::ostringstream os;
161 os << prefix << (++it->second);
162 std::string name = os.str();
163 if (name_alloc_map_.count(name) == 0) {
164 unique_prefix = name;
165 break;
166 }
167 }
168 }
169 name_alloc_map_[unique_prefix] = 0;
170 return Doc::Text(unique_prefix);
171}

173Doc RelayTextPrinter::Print(Kind k) {
174 switch (k) {
175 case kType:
176 return Doc::Text("Type");
177 case kShapeVar:
178 return Doc::Text("Shape");
179 case kBaseType:
180 return Doc::Text("BaseType");
181 case kConstraint:
182 return Doc::Text("Constraint");
183 case kAdtHandle:
184 return Doc::Text("AdtHandle");
185 case kTypeData:
186 return Doc::Text("TypeData");
187 default:
188 LOG(ERROR) << "Unknown Kind";
189 throw;
190 }
191}
192/*!
193 * \brief Allocate name to a type variable.
194 * \param var The input type variable.
195 * \return The corresponding name.
196 */
197Doc RelayTextPrinter::AllocTypeVar(const TypeVar& var) {
198 if (memo_type_.count(var)) {
199 Doc val = memo_type_[var];
200 val << "-malformed-ir";
201 return val;
202 }
203 std::string name = var->name_hint;
204 if (name.length() == 0 || !std::isalpha(name[0])) {
205 name = "t" + name;
206 }
207 Doc val = GetUniqueName(name);
208 memo_type_[var] = val;
209 if (var->kind != kType) {
210 val << ": " << Print(var->kind);
211 }
212 return val;
213}

215/*!
216 * \brief Allocate name to a variable.
217 * \param var The input variable.
218 * \return The corresponding name.
219 */
220Doc RelayTextPrinter::AllocVar(const Var& var) {
221 // still print if ir is malformed, but show the error.
222 if (memo_.count(var)) {
223 Doc val = memo_[var];
224 val << "-malformed-ir";
225 return val;
226 }
227 std::string name = var->name_hint();
228 // always make sure first name is alpha
229 if (name.length() == 0 || !std::isalpha(name[0])) {
230 name = "v" + name;
231 }
232 Doc val = GetUniqueName("%" + name);
233 memo_[var] = val; // Referential occurrences will not include the following.
234 if (!var->virtual_device()->IsFullyUnconstrained()) {
235 val << " {" << kVirtualDevice << "=" << PrintAttributeValue(var->virtual_device()) << "}";
236 }
237 if (var->type_annotation.defined()) {
238 val << ": " << Print(var->type_annotation);
239 }
240
241 val << PrintOptionalInfo(var);
242 return val;
243}

245bool RelayTextPrinter::IsUnique(const Expr& expr) {
246 auto it = dg_.expr_node.find(expr);
247 if (it == dg_.expr_node.end()) {
248 return true;
249 } else {
250 return !(it->second->parents.head && it->second->parents.head->next);
251 }
252}

254bool RelayTextPrinter::AlwaysInline(const Expr& expr) {
255 return expr.as<GlobalVarNode>() || expr.as<ConstantNode>() || expr.as<OpNode>() ||
256 expr.as<VarNode>() || expr.as<ConstructorNode>();
257}

259Doc RelayTextPrinter::VisitLeaf(const Expr& expr) {
260 if (!CheckVisited(expr)) {
261 Doc result = ExprFunctor<Doc(const Expr&)>::VisitExpr(expr);
262 // Add if not added after visiting
263 if (!CheckVisited(expr)) {
264 memo_[expr] = result;
265 } else {
266 result_memo_[expr] = result;
267 }
268 return result;
269 }
270 return memo_[expr];
271}
272
273bool RelayTextPrinter::CheckVisited(const Expr& expr) { return (memo_.count(expr)); }
274
275Doc RelayTextPrinter::VisitExpr(const Expr& expr) {
276 auto fcheck_visited = [this](const Expr& expr) { return this->CheckVisited(expr); };
277 auto fvisit_leaf = [this](const Expr& expr) { return this->VisitLeaf(expr); };
278
279 if (fcheck_visited(expr)) {
280 return memo_[expr];
281 } else {
282 ExpandDataflow(expr, fcheck_visited, fvisit_leaf);
283 return memo_[expr];
284 }
285}

290Doc RelayTextPrinter::PrintExpr(const Expr& expr, bool meta, bool try_inline, bool optional_info) {
291 // Exploit memoization to print GNF.
292 // The first time we visit an expression, we need to allocate a temp var
293 // for it. Every subsequent time we can just use its assigned variable.
294 // This works since hashing uses pointer equality.
295
296 // determine whether to inline
297 bool inline_expr = AlwaysInline(expr);
298
299 if (try_inline) {
300 inline_expr |= IsUnique(expr);
301 }
302
303 Doc printed_expr;
304
305 if (meta) {
306 printed_expr = meta_->GetMetaNode(GetRef<ObjectRef>(expr.get()));
307 } else if (!inline_expr && expr.as<LetNode>()) {
308 // wrap GNFed let in brackets
309 Doc body;
310 printed_expr << "(";
311 printed_expr << Doc::Indent(2, body << Doc::NewLine() << VisitExpr(expr)) << Doc::NewLine();
312 printed_expr << ")";
313 } else {
314 printed_expr = VisitExpr(expr);
315 }
316
317 if (optional_info) {
318 printed_expr << PrintOptionalInfo(expr);
319 }
320
321 // add expr to doc
322 if (expr.as<VarNode>()) {
323 // This is our first time visiting the var and we hit the VarNode case
324 // in the visitor. Thus the variable is free.
325 if (var_memo_.insert(expr).second && result_memo_.count(expr)) {
326 doc_stack_.back() << "free_var " << result_memo_[expr] << ";" << Doc::NewLine();
327 }
328 // Memoization is done in AllocVar.
329 return memo_[expr];
330 } else if (inline_expr) {
331 memo_[expr] = printed_expr;
332 return printed_expr;
333 } else {
334 // Already exists. Reuse
335 if (!var_memo_.insert(expr).second) {
336 return memo_[expr];
337 }
338 Doc temp_var = AllocTemp();
339 memo_[expr] = temp_var;
340 doc_stack_.back() << temp_var << " = " << printed_expr << ";" << Doc::NewLine();
341 return temp_var;
342 }
343}

349Doc RelayTextPrinter::VisitExpr_(const ConstantNode* op) {
350 // Print out simple scalars directly.
351 if (support::IsSimpleScalar(op)) {
352 return Doc::Text(support::NDArrayScalarToString(op->data));
353 }
354 // Fallbock: record it as a meta node.
355 Doc doc;
356 // Don't append optional_info. Because the entry function is Print,
357 // and it will append the optional_info afterwards.
358 return doc << PrintExpr(GetRef<Expr>(op), /*meta=*/true, /*try_inline=*/false,
359 /*optional_info=*/false);
360}

390Doc RelayTextPrinter::VisitExpr_(const LetNode* op) {
391 int n = 0;
392 Expr let = GetRef<Let>(op);
393 while (auto let_node = let.as<LetNode>()) {
394 Doc doc;
395 doc << "let " << AllocVar(let_node->var) << " = " << Print(let_node->value, false, true) << ";"
396 << Doc::NewLine();
397 doc_stack_.push_back(doc);
398 let = let_node->body;
399 ++n;
400 }
401 Doc doc = PrintScope(let);
402 for (int i = 0; i < n; ++i) {
403 doc = doc_stack_.back() << doc;
404 doc_stack_.pop_back();
405 }
406 return doc;
407}

409Doc RelayTextPrinter::PrintFunc(const Doc& prefix, const relay::Function& fn) {
410 Doc doc;
411 doc << prefix;
412 if (fn->type_params.size() > 0) {
413 doc << "[";
414 std::vector<Doc> type_params;
415 for (const TypeVar& tv : fn->type_params) {
416 type_params.push_back(Doc::Text(tv->name_hint));
417 }
418 doc << Doc::Concat(type_params);
419 doc << "]";
420 }
421 doc << "(";
422 std::vector<Doc> params;
423 for (Var param : fn->params) {
424 params.push_back(AllocVar(param));
425 }
426 for (const Doc& d : PrintDictAttrs(fn->attrs)) {
427 params.push_back(d);
428 }
429 if (!fn->virtual_device()->IsFullyUnconstrained()) {
430 Doc vid_doc;
431 vid_doc << kVirtualDevice << "=" << PrintAttributeValue(fn->virtual_device());
432 params.push_back(vid_doc);
433 }
434 doc << Doc::Concat(params) << ") ";
435 if (fn->ret_type.defined()) {
436 doc << "-> " << Print(fn->ret_type) << " ";
437 }
438 doc << PrintBody(fn->body);
439 return doc;
440}

442Doc RelayTextPrinter::PrintFunc(const Doc& prefix, const BaseFunc& base_func) {
443 if (auto* n = base_func.as<relay::FunctionNode>()) {
444 return PrintFunc(prefix, GetRef<relay::Function>(n));
445 } else if (auto* n = base_func.as<tir::PrimFuncNode>()) {
446 std::ostringstream os;
447 os << GetRef<tir::PrimFunc>(n);
448 return Doc::RawText(os.str());
449 } else {
450 // def @xyz = meta['ExternalFunc'][id]
451 Doc doc;
452 doc << prefix << " = " << meta_->GetMetaNode(base_func);
453 return doc;
454 }

457Doc RelayTextPrinter::PrintMod(const IRModule& mod) {
458 Doc doc;
459 int counter = 0;
460 // type definitions
461 for (const auto& kv : mod->type_definitions) {
462 if (counter++ != 0) {
463 doc << Doc::NewLine();
464 }
465 doc << Print(kv.second);
466 doc << Doc::NewLine();
467 }
468 // functions
469 for (const auto& kv : mod->functions) {
470 if (kv.second.as<relay::FunctionNode>()) {
471 dg_ = DependencyGraph::Create(&arena_, kv.second);
472 }
473 if (counter++ != 0) {
474 doc << Doc::NewLine();
475 }
476 std::ostringstream os;
477 os << "def @" << kv.first->name_hint;
478 doc << PrintFunc(Doc::Text(os.str()), kv.second);
479 doc << Doc::NewLine();
480 }
481 return doc;
482}

496Doc RelayTextPrinter::VisitExpr_(const CallNode* op) {
497 Doc doc;
498 // visit args first so they are lifted before the op
499 // this places op closer to its call site
500 std::vector<Doc> args;
501 for (const Expr& arg : op->args) {
502 args.push_back(Print(arg));
503 }
504
505 for (const Doc& d : PrintCallAttrs(op->attrs, op->op)) {
506 args.push_back(d);
507 }
508 const auto* cons_node = op->op.as<ConstructorNode>();
509 if (cons_node) {
510 doc << cons_node->name_hint;
511 } else {
512 doc << Print(op->op);
513 }
514
515 if (cons_node && cons_node->inputs.size() == 0) {
516 // don't print as a call if it's a 0-arity cons
517 return doc;
518 } else {
519 doc << "(" << Doc::Concat(args) << ")";
520 return doc;
521 }
522}

539Doc RelayTextPrinter::VisitExpr_(const MatchNode* op) {
540 // TODO(jmp): Lots of code duplication here because PrintBody and PrintScope don't accept Docs.
541 Doc doc;
542 Doc body;
543 doc << "match";
544 if (!op->complete) {
545 doc << "?";
546 }
547 doc << " (" << Print(op->data) << ") {";
548 std::vector<Doc> clause_docs;
549 for (const auto& clause : op->clauses) {
550 Doc clause_doc;
551 clause_doc << PrintPattern(clause->lhs, false) << " => ";
552 Doc rhs_doc = PrintScope(clause->rhs);
553 // TODO(@jroesch): This is unsound right now, and we need to revist it.
554 // if (clause->rhs.as<LetNode>()) {
555 // only add braces if there are multiple lines on the rhs
556 rhs_doc = Doc::Brace("{", rhs_doc, "}");
557 // }
558 clause_doc << rhs_doc << ",";
559 clause_docs.push_back(clause_doc);
560 }
561 doc << Doc::Indent(2, body << Doc::NewLine() << Doc::Concat(clause_docs, Doc::NewLine()))
562 << Doc::NewLine() << "}";
563 return doc;
564}

720Doc RelayTextPrinter::VisitType_(const TypeDataNode* node) {
721 in_adt_def_ = true;
722 Doc doc;
723 doc << "type " << Print(node->header);
724
725 // type vars
726 if (node->type_vars.size() != 0) {
727 doc << "[";
728 std::vector<Doc> type_vars;
729 for (Type type_var : node->type_vars) {
730 type_vars.push_back(Print(type_var));
731 }
732 doc << Doc::Concat(type_vars) << "]";
733 }
734 doc << " ";
735
736 std::vector<Doc> constructor_docs;
737 for (Constructor constructor : node->constructors) {
738 constructor_docs.push_back(Print(constructor, /* meta */ false, /* try_inline */ true));
739 }
740 Doc separator;
741 separator << "," << Doc::NewLine();
742 Doc adt_body;
743 adt_body << Doc::Concat(constructor_docs, separator);
744 // add trailing comma if there are any constructors
745 if (!constructor_docs.empty()) {
746 adt_body << ",";
747 }
748 doc << Doc::Brace("{", adt_body, "}");
749 in_adt_def_ = false;
750 return doc;
751}

797/*!
798 * \brief Attribute printer which prints the attributes in the call.
799 */
800class RelayTextPrinter::AttrPrinter : public AttrVisitor {
801 public:
802 AttrPrinter(std::vector<Doc>* doc, RelayTextPrinter* parent) : docs(doc), parent_(parent) {}
803
804 template <typename T>
805 void PrintKV(const char* key, const T& value) {
806 Doc doc;
807 doc << key << "=" << value;
808 docs->push_back(doc);
809 }
810
811 void Visit(const char* key, double* value) final {
812 Doc doc;
813 doc << key << "=" << *value << "f";
814 docs->push_back(doc);
815 }
816
817 void Visit(const char* key, int64_t* value) final { PrintKV(key, *value); }
818 void Visit(const char* key, uint64_t* value) final { PrintKV(key, *value); }
819 void Visit(const char* key, int* value) final { PrintKV(key, *value); }
820 void Visit(const char* key, bool* value) final { PrintKV(key, Doc::PyBoolLiteral(*value)); }
821 void Visit(const char* key, std::string* value) final { PrintKV(key, Doc::StrLiteral(*value)); }
822 void Visit(const char* key, void** value) final { LOG(FATAL) << "do not allow void as argument"; }
823 void Visit(const char* key, DataType* value) final {
824 PrintKV(key, Doc::StrLiteral(runtime::DLDataType2String(*value)));
825 }
826 void Visit(const char* key, runtime::NDArray* value) final {
827 LOG(FATAL) << "do not allow NDarray as argument";
828 }
829 void Visit(const char* key, runtime::ObjectRef* obj) final {
830 PrintKV(key, parent_->PrintAttributeValue(*obj));
831 }
832
833 private:
834 std::vector<Doc>* docs;
835 RelayTextPrinter* parent_;
836};

838void RelayTextPrinter::AppendGenericAttrs(std::vector<Doc>* docs, const Attrs& attrs,
839 bool include_type_key) {
840 if (!attrs.defined()) {
841 return;
842 }
843 AttrPrinter printer(docs, this);
844 // Need to drop cost cast since in general VisitNonDefaultAttrs can mutate, but in this
845 // case we are read-only.
846 const_cast<BaseAttrsNode*>(attrs.get())->VisitNonDefaultAttrs(&printer);
847 if (include_type_key) {
848 std::string s = attrs->GetTypeKey();
849 printer.Visit("attrs_type_key", &s);
850 }
851}

853std::vector<Doc> RelayTextPrinter::PrintCallAttrs(const Attrs& attrs, const Expr& op) {
854 std::vector<Doc> docs;
855 if (!attrs.defined()) {
856 return docs;
857 }
858 const auto* op_node = op.as<OpNode>();
859 if (show_meta_data_ && op_node && (attrs->type_index() != op_node->attrs_type_index)) {
860 // The parser can only understand calls with attributes if they match the operator's
861 // declared attribute type. If that's not the case fall back to the meta[...] representation.
862 docs.push_back(meta_->GetMetaNode(attrs));
863 } else {
864 AppendGenericAttrs(&docs, attrs, /*include_type_key=*/!op_node);
865 }
866 return docs;
867}

889Doc RelayTextPrinter::PrintAttributeValue(const ObjectRef& value, bool force_meta) {
890 if (value.defined()) {
891 Doc printed_attr;
892 if (value.as<tvm::tir::AnyNode>()) {
893 printed_attr << "?";
894 } else if (auto str_obj = value.as<tvm::StringObj>()) {
895 printed_attr << Doc::StrLiteral(GetRef<String>(str_obj));
896 } else if (force_meta) {
897 printed_attr = meta_->GetMetaNode(Downcast<ObjectRef>(value));
898 } else if (const auto* virtual_device_node = value.as<VirtualDeviceNode>()) {
899 if (show_meta_data_) {
900 printed_attr = meta_->GetMetaNode(GetRef<ObjectRef>(virtual_device_node));
901 } else {
902 // Special case: The ReprPrinter for VirtualDeviceNodes is much easier to work with while
903 // debugging.
904 std::ostringstream os;
905 os << GetRef<VirtualDevice>(virtual_device_node);
906 return Doc::Text(os.str());
907 }
908 } else if (const auto* base_attr_node = value.as<BaseAttrsNode>()) {
909 if (show_meta_data_) {
910 printed_attr = meta_->GetMetaNode(GetRef<ObjectRef>(base_attr_node));
911 } else {
912 // Special case: The non-meta form for attributes are much easier to work with while
913 // debugging.
914 printed_attr = PrintAttrsAsAttributeValue(GetRef<Attrs>(base_attr_node));
915 }
916 } else if (const auto* base_map_node = value.as<MapNode>()) {
917 if (show_meta_data_) {
918 printed_attr = meta_->GetMetaNode(GetRef<ObjectRef>(base_map_node));
919 } else {
920 // Special case: Show maps fields as key=value pairs to help debugging.
921 printed_attr << PrintMapAsAttributeValue(GetRef<Map<ObjectRef, ObjectRef>>(base_map_node));
922 }
923 } else if (const auto* global_var_node = value.as<GlobalVarNode>()) {
924 if (show_meta_data_) {
925 printed_attr = meta_->GetMetaNode(GetRef<ObjectRef>(global_var_node));
926 } else {
927 printed_attr << "'" << global_var_node->name_hint << "'";
928 }
929 } else {
930 printed_attr = VisitAttr(value);
931 }
932 return printed_attr;
933 } else {
934 return Doc::Text("None");
935 }
936}

938Doc RelayTextPrinter::PrintAttrsAsAttributeValue(const Attrs& attrs) {
939 std::vector<Doc> docs;
940 AppendGenericAttrs(&docs, attrs, /*include_type_key=*/false);
941 Doc doc;
942 doc << "{" << Doc::Concat(docs) << "}";
943 return doc;
944}

946Doc RelayTextPrinter::PrintMapAsAttributeValue(const Map<ObjectRef, ObjectRef>& map) {
947 std::vector<Doc> docs;
948 for (const auto& k : map) {
949 Doc doc;
950 doc << PrintAttributeValue(k.first);
951 doc << "=";
952 doc << PrintAttributeValue(k.second);
953 docs.push_back(doc);
954 }
955 Doc doc;
956 doc << "{" << Doc::Concat(docs) << "}";
957 return doc;
958}
