99 ICHECK_EQ(args.size(), 5) << "topi.take expects 4 or 5 arguments";

105});

149TVM_REGISTER_GLOBAL("topi.matmul").set_body([](TVMArgs args, TVMRetValue* rv) {
150 switch (args.size()) {
151 case 2:
152 *rv = matmul(args[0], args[1]);
153 break;
154 case 3:
155 *rv = matmul(args[0], args[1], args[2]);
156 break;
157 case 4:
158 *rv = matmul(args[0], args[1], args[2], args[3]);
159 break;
160 default:
161 ICHECK(0) << "topi.matmul expects 2, 3 or 4 arguments";
162 }
163});

180TVM_REGISTER_GLOBAL("topi.strided_slice").set_body([](TVMArgs args, TVMRetValue* rv) {
181 Tensor x = args[0];
182 Array<PrimExpr> begin = args[1];
183 Array<PrimExpr> end = args[2];
184 Array<PrimExpr> strides = args[3];
185 if (IsConstIntArray(begin) && IsConstIntArray(end) && IsConstIntArray(strides)) {
186 Array<Integer> begin_static = args[1];
187 Array<Integer> end_static = args[2];
188 Array<Integer> strides_static = args[3];
189 Array<Integer> axes = args[4];
190 std::string slice_mode = args[5];
191 if (axes.size() > 0) {
192 *rv = strided_slice_with_axes(x, begin_static, end_static, strides_static, axes, slice_mode);
193 } else {
194 *rv = strided_slice(x, begin_static, end_static, strides_static, slice_mode);
195 }
196 } else {
197 *rv = dynamic_strided_slice(x, begin, end, strides);
198 }
199});
