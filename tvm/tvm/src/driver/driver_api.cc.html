66bool ShouldAnnotateEntryFunc(const IRModule mod) {
67 Optional<tvm::relay::Executor> executor = mod->GetAttr<tvm::relay::Executor>("executor");
68 const bool aot_executor = executor.defined() && executor.value()->name == "aot";
69 const bool single_entry_func = (mod->functions.size() == 1);
70 return single_entry_func && !aot_executor;
71}

73/*! \return The default host target for a given device target */
74Target DefaultTargetHost(Target target) {
75 if (target.defined() && target->kind->device_type == kDLCPU) {
76 return target;
77 } else {
78 if (LLVMEnabled()) {
79 return Target("llvm");
80 } else {
81 return Target("stackvm");
82 }
83 }

86void GetBinds(const Array<ObjectRef>& args, bool compact,
87 const std::unordered_map<te::Tensor, tir::Buffer>& binds,
88 Map<te::Tensor, tir::Buffer>* out_binds, Array<ObjectRef>* out_arg_list) {
89 *out_binds = binds;
90
91 for (const ObjectRef& x : args) {
92 if (const te::TensorNode* tensor_node = x.as<te::TensorNode>()) {
93 te::Tensor x_ref = GetRef<te::Tensor>(tensor_node);
94 if (out_binds->find(x_ref) == out_binds->end()) {
95 tir::Buffer buf = tir::BufferWithOffsetAlignment(x_ref->shape, x_ref->dtype,
96 x_ref->op->name, -1, 0, compact);
97 out_binds->Set(x_ref, buf);
98 out_arg_list->push_back(buf);
99 } else {
100 out_arg_list->push_back((*out_binds)[x_ref]);
101 }
102 } else if (x.as<te::BufferNode>() || x.as<tir::VarNode>()) {
103 out_arg_list->push_back(x);
104 } else {
105 LOG(FATAL)
106 << "Expected type of the elements of args to be te::Tensor, te::Buffer or tir::Var, "
107 << "but got a " << x->GetTypeKey();
108 }
109 }
110}

113 const std::unordered_map<te::Tensor, tir::Buffer>& binds,
114 Map<te::Tensor, tir::Buffer>* out_binds, Array<ObjectRef>* out_arg_list) {
115 Array<ObjectRef> ref_args;

117 ref_args.push_back(x);

123 .set_body_typed([](const Array<ObjectRef>& args, bool compact,
124 const Map<te::Tensor, tir::Buffer>& binds) {
125 std::unordered_map<te::Tensor, tir::Buffer> c_binds;

139 return out_arr;

142Array<tvm::transform::Pass> CreatePassList(bool disable_loop_partition) {
143 transform::PassContext pass_ctx = transform::PassContext::Current();
144
145 bool disable_vectorize = pass_ctx->GetConfig<Bool>("tir.disable_vectorize", Bool(false)).value();
146 bool disable_storage_rewrite =
147 pass_ctx->GetConfig<Bool>("tir.disable_storage_rewrite", Bool(false)).value();
148 bool instrument_bound_checkers =
149 pass_ctx->GetConfig<Bool>("tir.instrument_bound_checkers", Bool(false)).value();
150 bool disable_cse_tir = pass_ctx->GetConfig<Bool>("tir.disable_cse_tir", Bool(false)).value();
151 bool enable_equiv_terms_in_cse_tir =
152 pass_ctx->GetConfig<Bool>("tir.enable_equiv_terms_in_cse_tir", Bool(false)).value();
153
154 // Get any user-added passes
155 Array<Array<ObjectRef>> add_lower_pass =
156 pass_ctx->GetConfig<Array<Array<ObjectRef>>>("tir.add_lower_pass", Array<Array<ObjectRef>>())
157 .value();
158
159 Array<transform::Pass> user_lower_phase0 = Array<transform::Pass>();
160 Array<transform::Pass> user_lower_phase1 = Array<transform::Pass>();
161 Array<transform::Pass> user_lower_phase2 = Array<transform::Pass>();
162 Array<transform::Pass> user_lower_phase3 = Array<transform::Pass>();
163
164 // phase passes is of the form
165 // [[phase_number, pass], [phase_number, pass]... ]
166 for (Array<ObjectRef> phase_pass : add_lower_pass) {
167 const IntImmNode* phase_num = phase_pass[0].as<IntImmNode>();
168 ICHECK(phase_num)
169 << "Expected the first entry in the inner Array of tir.add_lower_pass to be an integer";
170 int phase_num_val = phase_num->value;
171
172 CHECK_GE(phase_num_val, 0);
173
174 const tvm::transform::PassNode* pass_node = phase_pass[1].as<tvm::transform::PassNode>();
175 tvm::transform::Pass pass = GetRef<tvm::transform::Pass>(pass_node);
176 // Copy the pass into the correct phase
177 if (phase_num_val == 0) {
178 user_lower_phase0.push_back(pass);
179 } else if (phase_num_val == 1) {
180 user_lower_phase1.push_back(pass);
181 } else if (phase_num_val == 2) {
182 user_lower_phase2.push_back(pass);
183 } else if (phase_num_val >= 3) {
184 user_lower_phase3.push_back(pass);
185 }
186 }
187
188 // Construct the pass list, inserting the user provided passes at the end of the phase
189
190 // PHASE 0
191 Array<tvm::transform::Pass> pass_list = user_lower_phase0;
192
193 // PHASE 1
194 pass_list.push_back(tir::transform::InjectPrefetch());
195 pass_list.push_back(tir::transform::TextureFlatten());
196 pass_list.push_back(tir::transform::StorageFlatten(64, instrument_bound_checkers));
197 pass_list.push_back(tir::transform::LowerCrossThreadReduction());
198 pass_list.push_back(tir::transform::LowerInitBlock());
199 pass_list.push_back(tir::transform::PlanAndUpdateBufferAllocationLocation());
200 pass_list.push_back(tir::transform::ConvertBlocksToOpaque());
201 pass_list.push_back(tir::transform::UnifyThreadBinding());
202 pass_list.push_back(tir::transform::ManifestSharedMemoryLocalStage());
203 pass_list.push_back(tir::transform::CompactBufferAllocation());
204 pass_list.push_back(tir::transform::LowerMatchBuffer());
205 pass_list.push_back(tir::transform::InjectSoftwarePipeline());
206 pass_list.push_back(tir::transform::LowerOpaqueBlock());
207 pass_list.push_back(tir::transform::FlattenBuffer());
208 pass_list.push_back(tir::transform::BF16Legalize());
209 pass_list.push_back(tir::transform::NarrowDataType(32));
210 pass_list.push_back(tir::transform::Simplify());
211
212 // Add user-defined phase-1 passes
213 pass_list.insert(pass_list.end(), user_lower_phase1.begin(), user_lower_phase1.end());
214
215 // PHASE 2
216 if (!disable_loop_partition) {
217 pass_list.push_back(tir::transform::LoopPartition());
218 }
219
220 pass_list.push_back(tir::transform::VectorizeLoop(!disable_vectorize));
221 pass_list.push_back(tir::transform::InjectVirtualThread());
222 pass_list.push_back(tir::transform::InjectDoubleBuffer());
223 if (!disable_storage_rewrite) {
224 pass_list.push_back(tir::transform::StorageRewrite());
225 }
226 // LowerVtcmAlloc must occur after any transformations that modify memory allocation locations
227 pass_list.push_back(tir::transform::LowerVtcmAlloc());
228 pass_list.push_back(tir::transform::UnrollLoop());
229
230 // Add user-defined phase-2 passes
231 pass_list.insert(pass_list.end(), user_lower_phase2.begin(), user_lower_phase2.end());
232
233 // PHASE 3
234 pass_list.push_back(tir::transform::RenormalizeSplitPattern());
235 pass_list.push_back(tir::transform::Simplify());
236 pass_list.push_back(tir::transform::RemoveNoOp());
237 pass_list.push_back(tir::transform::RewriteUnsafeSelect());
238 pass_list.push_back(tir::transform::HoistIfThenElse());
239
240 // Add user-defined phase-3 passes
241 pass_list.insert(pass_list.end(), user_lower_phase3.begin(), user_lower_phase3.end());
242
243 if (instrument_bound_checkers) {
244 pass_list.push_back(tir::transform::InstrumentBoundCheckers());
245 }
246
247 pass_list.push_back(
248 tir::transform::CommonSubexprElimTIR(!disable_cse_tir, enable_equiv_terms_in_cse_tir));
249
250 return pass_list;
251}

264// Convert te schedule to IRModule
265IRModule ScheduleToModule(te::Schedule sch, const Array<ObjectRef>& args, const std::string& name,
266 const std::unordered_map<te::Tensor, tir::Buffer>& binds,
267 GlobalVarSupply global_var_supply) {
268 sch = sch.normalize();
269
270 transform::PassContext pass_ctx = transform::PassContext::Current();
271 bool debug_keep_trivial_loop =
272 pass_ctx->GetConfig<Bool>("tir.debug_keep_trivial_loop", Bool(false)).value();
273
274 // Before TIR transformation.
275 tir::Stmt stmt = te::ScheduleOps(sch, te::InferBound(sch), debug_keep_trivial_loop);
276 bool compact = te::VerifyCompactBuffer(stmt);
277
278 Map<te::Tensor, tir::Buffer> out_binds;
279 Array<ObjectRef> out_arg_list;
280 GetBinds(args, compact, binds, &out_binds, &out_arg_list);
281
282 // Build the function, converting from te::Tensor to tir::Buffer
283 tir::PrimFunc f = te::SchedulePostProcToPrimFunc(out_arg_list, std::move(stmt), out_binds);
284 f = WithAttr(std::move(f), "global_symbol", runtime::String(name));
285
286 // Mark this schedule as being converted from an TE schedule. Makes sure that
287 // the correct TE passes are run.
288 f = WithAttr(std::move(f), "from_legacy_te_schedule", Bool(true));
289
290 bool noalias = pass_ctx->GetConfig<Bool>("tir.noalias", Bool(true)).value();
291
292 if (noalias) {
293 f = WithAttr(std::move(f), "tir.noalias", Bool(true));
294 }
295 GlobalVar global_var = global_var_supply->UniqueGlobalFor(name, false);
296 return IRModule(Map<GlobalVar, BaseFunc>({{global_var, f}}));
297}

307 }

309 IRModule mod =
310 ScheduleToModule(std::move(sch), args, name, c_binds, GlobalVarSupply(NameSupply("")));
311 return mod;
312 });

377/**
378 * This function takes the input module that contains both the device and host opts.
379 * Then, it applies transformation on the original module before splitting into separate modules for
380 * device and host. Then it also applies transformations on the new splitted modules.
381 */
382std::pair<IRModule, IRModule> SplitMixedModule(IRModule mod_mixed, const Target& target_arg,
383 const Target& target_host_arg) {
384 Target target = target_arg, target_host = target_host_arg;
385 CheckAndUpdateHostConsistency(&target, &target_host);
386
387 ICHECK(mod_mixed.defined()) << "This module must be defined";
388
389 mod_mixed = ApplyPasses(mod_mixed, MixedModulePassManager(mod_mixed, target));
390
391 IRModule host_mod = ApplyPasses(mod_mixed, HostModulePassManager(mod_mixed, target_host));
392
393 IRModule device_mod = ApplyPasses(mod_mixed, DeviceModulePassManager(mod_mixed, target));
394
395 auto keys = target->GetKeys();
396
397 CheckAndUpdateHostConsistency(&target, &target_host);
398
399 bool target_is_gpu = std::find(keys.begin(), keys.end(), "gpu") != keys.end();
400 if (target_is_gpu && device_mod->functions.size() == 0) {
401 DLOG(WARNING) << "Specified target " << target->str()
402 << " but cannot find device code. Did you forget to bind?";
403 }
404
405 return {host_mod, device_mod};
406}

408runtime::Module TIRToRuntime(const Map<Target, IRModule>& inputs_arg,
409 const Target& target_host_arg) {
410 auto pass_ctx = transform::PassContext::Current();
411
412 std::vector<runtime::Module> device_modules;
413 Map<Target, IRModule> inputs = inputs_arg;
414 Target target_host = target_host_arg;
415
416 // Fetch previous defined target host in targets
417 CheckAndUpdateHostConsistency(&inputs, &target_host);
418
419 if (!target_host.defined()) {
420 for (const auto& it : inputs) {
421 if (it.first->kind->device_type == kDLCPU || it.first->kind->device_type == kDLMicroDev) {
422 target_host = it.first;
423 break;
424 }
425 }
426 }
427
428 if (!target_host.defined()) {
429 target_host = DefaultTargetHost(target_host);
430 }
431
432 // Update target host for all targets
433 CheckAndUpdateHostConsistency(&inputs, &target_host);
434
435 // Take the attrs from the first module so the eventual modules have them.
436 // Ideally this would just be one unified module all the way through;
437 IRModule first_module = (*inputs.begin()).second;
438 IRModule mhost_all = IRModule(Map<GlobalVar, BaseFunc>(), {}, {}, {}, first_module->attrs);
439
440 ICHECK(mhost_all.defined()) << "The host module must be defined";
441
442 for (const auto& it : inputs) {
443 if (it.second.defined()) {
444 const Target& target = it.first;
445 const IRModule& ir_module = it.second;
446 auto pair = SplitMixedModule(ir_module, target, target_host);
447 auto& host_mod = pair.first;
448 auto& device_mod = pair.second;
449
450 ICHECK(host_mod.defined()) << "The split host module must be defined";
451
452 ICHECK(mhost_all.defined()) << "The host module must be defined";
453
454 // We don't want library modules going back into host codegen
455 // unless they're supposed to. Here if we overrode the target host
456 // to allow lowering previously we check that it's meant to be placed
457 // back into the host Module.
458 bool overrides_host_target = target->kind->device_type == target_host->kind->device_type;
459 bool non_host_target_kind = target->kind != target_host->kind;
460 if (overrides_host_target && non_host_target_kind) {
461 device_modules.push_back(codegen::Build(host_mod, it.first));
462 } else {
463 mhost_all->Update(host_mod);
464 }
465
466 if (device_mod->functions.size() != 0) {
467 device_modules.push_back(codegen::Build(device_mod, it.first));
468 }
469 }
470 }
471
472 runtime::Module mhost = codegen::Build(mhost_all, target_host);
473 for (const auto& it : device_modules) {
474 if (it.operator->()) {
475 mhost.Import(it);
476 }
477 }
478
479 return mhost;
480}

494// Build for heterogeneous execution when target is a string.
495runtime::Module build(const Map<String, IRModule>& inputs_arg, const Target& target_host_arg) {
496 Map<Target, IRModule> updated_inputs;
497 Target target_host = target_host_arg;
498 for (const auto& it : inputs_arg) {
499 Target target = Target(it.first);
500 CheckAndUpdateHostConsistency(&target, &target_host);
501 Optional<String> device = target->GetAttr<String>("device");
502 if (device.defined() && device.value() == "vta") {
503 target = Target("ext_dev");
504 }
505 updated_inputs.Set(target, it.second);
506 }
507 return TIRToRuntime(updated_inputs, target_host);
508}

520transform::Sequential MixedModulePassManager(IRModule mixed_mod, Target target) {
521 transform::PassContext pass_ctx = transform::PassContext::Current();
522
523 Array<Pass> mixed_pass_list;
524
525 mixed_pass_list.push_back(tir::transform::BindTarget(target));
526
527 mixed_pass_list.push_back(tir::transform::VerifyMemory());
528
529 if (ShouldAnnotateEntryFunc(mixed_mod)) {
530 mixed_pass_list.push_back(tir::transform::AnnotateEntryFunc());
531 }
532
533 bool detect_global_barrier =
534 pass_ctx->GetConfig<Bool>("tir.detect_global_barrier", Bool(false)).value();
535 if (detect_global_barrier) {
536 mixed_pass_list.push_back(tir::transform::ThreadSync("global"));
537 }
538
539 mixed_pass_list.push_back(tir::transform::ThreadSync("shared"));
540 mixed_pass_list.push_back(tir::transform::ThreadSync("shared.dyn"));
541 mixed_pass_list.push_back(tir::transform::MergeDynamicSharedMemoryAllocations());
542 mixed_pass_list.push_back(tir::transform::ThreadSync("warp"));
543 mixed_pass_list.push_back(tir::transform::InferFragment());
544 mixed_pass_list.push_back(tir::transform::LowerThreadAllreduce());
545
546 bool use_ptx_async_copy =
547 pass_ctx->GetConfig<Bool>("tir.use_ptx_async_copy", Bool(false)).value();
548
549 if (use_ptx_async_copy) {
550 mixed_pass_list.push_back(tir::transform::InjectPTXAsyncCopy());
551 }
552
553 bool unpacked_api = mixed_mod->GetAttr<relay::Executor>(tvm::attr::kExecutor)
554 .value_or(relay::Executor::Create("graph", {}))
555 ->GetAttr<Bool>("unpacked-api")
556 .value_or(Bool(false));
557 if (unpacked_api) {
558 mixed_pass_list.push_back(tir::transform::MakeUnpackedAPI());
559 } else {
560 mixed_pass_list.push_back(tir::transform::MakePackedAPI(-1));
561 }
562 mixed_pass_list.push_back(tir::transform::SplitHostDevice());
563
564 return transform::Sequential(mixed_pass_list);
565}

572transform::Sequential HostModulePassManager(IRModule mixed_mod, Target target_host) {
573 Array<tvm::transform::Pass> host_pass_list;
574
