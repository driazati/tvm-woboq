31Buffer BufferDecl(Array<PrimExpr> shape, DataType dtype, String buffer_name, Optional<Var> data,
32 Optional<Array<PrimExpr>> strides, Optional<PrimExpr> elem_offset,
33 String storage_scope, int align, int offset_factor, String buffer_type,
34 Optional<Array<IntImm>> axis_separators) {
35 Var buffer_data;
36 if (!data.defined()) {
37 DataType storage_dtype = dtype;
38 if (storage_dtype == DataType::Bool()) {
39 storage_dtype = DataType::Int(8);
40 }
41 buffer_data = tvm::tir::Var(buffer_name, PointerType(PrimType(storage_dtype), storage_scope));
42 } else {
43 buffer_data = data.value();
44 }
45 if (!elem_offset.defined() && offset_factor) {
46 DataType shape_dtype = shape.empty() ? DataType::Int(32) : shape[0]->dtype;
47 elem_offset = tvm::tir::Var("elem_offset", shape_dtype);
48 }
49 return Buffer(buffer_data, dtype, shape, strides.value_or(Array<PrimExpr>()),
50 elem_offset.value_or(PrimExpr()), buffer_name, align, offset_factor,
51 (buffer_type == "auto_broadcast") ? tvm::tir::kAutoBroadcast : tvm::tir::kDefault,
52 axis_separators.value_or(Array<IntImm>()));
53}

55PrimFuncFrame PrimFunc() {
56 ObjectPtr<PrimFuncFrameNode> n = make_object<PrimFuncFrameNode>();
57 n->name = NullOpt;
58 n->args.clear();
59 n->ret_type = NullOpt;
60 n->buffer_map.clear();
61 n->preflattened_buffer_map.clear();
62 n->attrs = NullOpt;
63 n->env_threads.clear();
64 n->root_alloc_buffers.clear();
65 return PrimFuncFrame(n);
66}

75Buffer Arg(String name, Buffer buffer) {

77 details::Namer::Name(buffer, name);
78 Var handle(buffer->name + "_handle", DataType::Handle());
79 frame->args.push_back(handle);
80 frame->buffer_map.Set(handle, buffer);

111Buffer MatchBuffer(ObjectRef param, Array<PrimExpr> shape, DataType dtype, Optional<Var> data,
112 Array<PrimExpr> strides, PrimExpr elem_offset, String storage_scope, int align,
113 int offset_factor, String buffer_type_str, Array<IntImm> axis_separators) {
114 Buffer buffer = BufferDecl(shape, dtype, "", data, strides, elem_offset, storage_scope, align,
115 offset_factor, buffer_type_str, axis_separators);
116 if (const auto* var = param.as<tvm::tir::VarNode>()) {
117 PrimFuncFrame frame = FindPrimFuncFrame("T.match_buffer");
118 Var v = GetRef<Var>(var);
119 for (auto const& arg : frame->args) {
120 if (arg.same_as(v)) {
121 frame->buffer_map.Set(v, buffer);
122 return buffer;
123 }
124 }
125 LOG(FATAL) << "ValueError: Can not bind non-input param to buffer.";
126 } else if (const auto* buffer_load = param.as<tvm::tir::BufferLoadNode>()) {
127 BlockFrame frame = FindBlockFrame("T.match_buffer");
128 frame->match_buffers.push_back(tvm::tir::MatchBufferRegion(
129 buffer, BufferRegionFromLoad(GetRef<tvm::tir::BufferLoad>(buffer_load))));
130 } else if (const auto* buffer_region = param.as<tvm::tir::BufferRegionNode>()) {
131 BlockFrame frame = FindBlockFrame("T.match_buffer");
132 frame->match_buffers.push_back(
133 tvm::tir::MatchBufferRegion(buffer, GetRef<tvm::tir::BufferRegion>(buffer_region)));
134 } else {
135 LOG(FATAL) << "ValueError: Unexpected type for TIR MatchBuffer.";
136 }
137 return buffer;
138}

140void PreflattenedBuffer(Buffer postflattened_buffer, Array<PrimExpr> shape, DataType dtype,
141 Optional<Var> data, Array<PrimExpr> strides, PrimExpr elem_offset,
142 String storage_scope, int align, int offset_factor, String buffer_type_str,
143 Array<IntImm> axis_separators) {
144 PrimFuncFrame frame = FindPrimFuncFrame("T.preflattened_buffer");
145 for (auto const& p : frame->buffer_map) {
146 if (p.second.same_as(postflattened_buffer)) {
147 String buffer_name(postflattened_buffer->name + "_preflatten");
148 Buffer buffer =
149 BufferDecl(shape, dtype, buffer_name, data.value_or(p.second->data), strides, elem_offset,
150 storage_scope, align, offset_factor, buffer_type_str, axis_separators);
151 details::Namer::Name(buffer, buffer_name);
152 frame->preflattened_buffer_map.Set(p.first, buffer);
153 return;
154 }
155 }
156 LOG(FATAL) << "ValueError: postflattened buffer " << postflattened_buffer->name
157 << " does not exist.";
158}

160BlockFrame Block(String name, bool no_realize) {
161 ObjectPtr<BlockFrameNode> n = make_object<BlockFrameNode>();
162 n->name = name;
163 n->iter_vars.clear();
164 n->reads = NullOpt;
165 n->writes = NullOpt;
166 n->init = NullOpt;
167 n->alloc_buffers.clear();
168 n->match_buffers.clear();
169 n->annotations = NullOpt;
170 n->iter_values.clear();

173 return BlockFrame(n);

176#define TVM_TIR_IR_BUILDER_FOR_FRAME(Method, Kind) \
177 ForFrame Method(PrimExpr start, PrimExpr stop, Optional<Map<String, ObjectRef>> annotations) { \
178 PrimExpr min = start; \
179 PrimExpr extent = arith::Analyzer().Simplify(stop - start); \
180 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>(); \
181 int bits = std::max(min.dtype().bits(), extent.dtype().bits()); \
182 n->vars = {Var("v", DataType::Int(bits))}; \
183 n->doms = {Range::FromMinExtent(min, extent)}; \
184 n->f_make_for_loop = [annotations](Array<Var> vars, Array<Range> doms, tvm::tir::Stmt body) { \
185 ICHECK_EQ(vars.size(), 1); \
186 ICHECK_EQ(doms.size(), 1); \
187 return tvm::tir::For(vars[0], doms[0]->min, doms[0]->extent, Kind, body, NullOpt, \
188 annotations.value_or(Map<String, ObjectRef>())); \
189 }; \
190 return ForFrame(n); \
191 }
192
193TVM_TIR_IR_BUILDER_FOR_FRAME(Serial, tvm::tir::ForKind::kSerial);
194TVM_TIR_IR_BUILDER_FOR_FRAME(Parallel, tvm::tir::ForKind::kParallel);
195TVM_TIR_IR_BUILDER_FOR_FRAME(Vectorized, tvm::tir::ForKind::kVectorized);
196TVM_TIR_IR_BUILDER_FOR_FRAME(Unroll, tvm::tir::ForKind::kUnrolled);
197
198#undef TVM_TIR_IR_BUILDER_FOR_FRAME

200ForFrame ThreadBinding(PrimExpr start, PrimExpr stop, String thread,
201 Optional<Map<String, ObjectRef>> annotations) {
202 using namespace tvm::tir;
203 PrimExpr min = start;
204 PrimExpr extent = arith::Analyzer().Simplify(stop - start);
205 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>();
206 int bits = std::max(min.dtype().bits(), extent.dtype().bits());
207 n->vars = {Var("v", DataType::Int(bits))};
208 n->doms = {Range::FromMinExtent(min, extent)};
209 n->f_make_for_loop = [annotations, thread](Array<Var> vars, Array<Range> doms, Stmt body) -> For {
210 ICHECK_EQ(vars.size(), 1);
211 ICHECK_EQ(doms.size(), 1);
212 IterVar iter_var(Range(nullptr), Var("iter", DataType::Int(32)), IterVarType::kThreadIndex,
213 thread);
214 return For(vars[0], doms[0]->min, doms[0]->extent, ForKind::kThreadBinding, body, iter_var,
215 annotations.value_or(Map<String, ObjectRef>()));
216 };
217 return ForFrame(n);
218}

220ForFrame Grid(Array<PrimExpr> extents) {
221 using namespace tvm::tir;
222 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>();
223 n->vars.reserve(extents.size());
224 n->doms.reserve(extents.size());
225 for (const auto& extent : extents) {
226 DataType dtype = extent.dtype();
227 n->vars.push_back(Var("v", extent.dtype()));
228 n->doms.push_back(Range(make_const(dtype, 0), extent));
229 }
230 n->f_make_for_loop = [](Array<Var> vars, Array<Range> doms, Stmt body) -> Stmt {
231 ICHECK_EQ(vars.size(), doms.size());
232 int n = vars.size();
233 for (int i = n - 1; i >= 0; --i) {
234 Range dom = doms[i];
235 Var var = vars[i];
236 body = For(var, dom->min, dom->extent, ForKind::kSerial, std::move(body),
237 /*thread_binding=*/NullOpt, /*annotations=*/{});
238 }
239 return body;
240 };
241 return ForFrame(n);
242}

248TVM_STATIC_IR_FUNCTOR(Namer, vtable)
249 .set_dispatch<tvm::tir::BufferNode>([](const ObjectRef& node, String name) -> void {
250 tvm::tir::BufferNode* buffer =
251 const_cast<tvm::tir::BufferNode*>(node.as<tvm::tir::BufferNode>());
252 buffer->name = name;
253 Namer::Name(buffer->data, name);
254 int n = buffer->strides.size();
255 for (int i = 0; i < n; ++i) {
256 PrimExpr e = buffer->strides[i];
257 if (const tvm::tir::VarNode* v = e.as<tvm::tir::VarNode>()) {
258 Namer::Name(GetRef<tvm::tir::Var>(v), name + "_s" + std::to_string(i));
259 }
260 }
261 });
