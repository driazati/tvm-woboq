29using tvm::tir::IterVar;

31PrimFuncFrame PrimFunc() {
32 ObjectPtr<PrimFuncFrameNode> n = make_object<PrimFuncFrameNode>();
33 n->name = NullOpt;
34 n->args.clear();
35 n->ret_type = NullOpt;
36 n->buffer_map.clear();
37 n->preflattened_buffer_map.clear();
38 n->attrs = NullOpt;
39 n->env_threads.clear();
40 n->root_alloc_buffers.clear();
41 return PrimFuncFrame(n);
42}
43
44BlockFrame Block(String name, bool no_realize) {
45 ObjectPtr<BlockFrameNode> n = make_object<BlockFrameNode>();
46 n->name = name;
47 n->iter_vars.clear();
48 n->reads = NullOpt;
49 n->writes = NullOpt;
50 n->init = NullOpt;
51 n->alloc_buffers.clear();
52 n->match_buffers.clear();
53 n->annotations = NullOpt;
54 n->iter_values.clear();
55 n->predicate = NullOpt;
56 n->no_realize = no_realize;
57 return BlockFrame(n);
58}
59
60void Evaluate(PrimExpr value) { AddToParent(tvm::tir::Evaluate(value)); }
61TVM_REGISTER_GLOBAL("script.ir_builder.tir.PrimFunc").set_body_typed(PrimFunc);
62TVM_REGISTER_GLOBAL("script.ir_builder.tir.Block").set_body_typed(Block);
63TVM_REGISTER_GLOBAL("script.ir_builder.tir.Evaluate").set_body_typed(Evaluate);
