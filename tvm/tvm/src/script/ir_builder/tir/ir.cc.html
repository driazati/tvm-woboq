31Buffer BufferDecl(Array<PrimExpr> shape, DataType dtype, String buffer_name, Optional<Var> data,
32 Optional<Array<PrimExpr>> strides, Optional<PrimExpr> elem_offset,
33 String storage_scope, int align, int offset_factor, String buffer_type,
34 Optional<Array<IntImm>> axis_separators) {
35 Var buffer_data;
36 if (!data.defined()) {
37 DataType storage_dtype = dtype;
38 if (storage_dtype == DataType::Bool()) {
39 storage_dtype = DataType::Int(8);
40 }
41 buffer_data = tvm::tir::Var(buffer_name, PointerType(PrimType(storage_dtype), storage_scope));
42 } else {
43 buffer_data = data.value();
44 }
45 if (!elem_offset.defined() && offset_factor) {
46 DataType shape_dtype = shape.empty() ? DataType::Int(32) : shape[0]->dtype;
47 elem_offset = tvm::tir::Var("elem_offset", shape_dtype);
48 }
49 return Buffer(buffer_data, dtype, shape, strides.value_or(Array<PrimExpr>()),
50 elem_offset.value_or(PrimExpr()), buffer_name, align, offset_factor,
51 (buffer_type == "auto_broadcast") ? tvm::tir::kAutoBroadcast : tvm::tir::kDefault,
52 axis_separators.value_or(Array<IntImm>()));
53}

111Buffer MatchBuffer(ObjectRef param, Array<PrimExpr> shape, DataType dtype, Optional<Var> data,
112 Array<PrimExpr> strides, PrimExpr elem_offset, String storage_scope, int align,
113 int offset_factor, String buffer_type_str, Array<IntImm> axis_separators) {
114 Buffer buffer = BufferDecl(shape, dtype, "", data, strides, elem_offset, storage_scope, align,
115 offset_factor, buffer_type_str, axis_separators);
116 if (const auto* var = param.as<tvm::tir::VarNode>()) {
117 PrimFuncFrame frame = FindPrimFuncFrame("T.match_buffer");
118 Var v = GetRef<Var>(var);
119 for (auto const& arg : frame->args) {
120 if (arg.same_as(v)) {
121 frame->buffer_map.Set(v, buffer);
122 return buffer;
123 }
124 }
125 LOG(FATAL) << "ValueError: Can not bind non-input param to buffer.";
126 } else if (const auto* buffer_load = param.as<tvm::tir::BufferLoadNode>()) {
127 BlockFrame frame = FindBlockFrame("T.match_buffer");
128 frame->match_buffers.push_back(tvm::tir::MatchBufferRegion(
129 buffer, BufferRegionFromLoad(GetRef<tvm::tir::BufferLoad>(buffer_load))));
130 } else if (const auto* buffer_region = param.as<tvm::tir::BufferRegionNode>()) {
131 BlockFrame frame = FindBlockFrame("T.match_buffer");
132 frame->match_buffers.push_back(
133 tvm::tir::MatchBufferRegion(buffer, GetRef<tvm::tir::BufferRegion>(buffer_region)));
134 } else {
135 LOG(FATAL) << "ValueError: Unexpected type for TIR MatchBuffer.";
136 }
137 return buffer;
138}

140void PreflattenedBuffer(Buffer postflattened_buffer, Array<PrimExpr> shape, DataType dtype,
141 Optional<Var> data, Array<PrimExpr> strides, PrimExpr elem_offset,
142 String storage_scope, int align, int offset_factor, String buffer_type_str,
143 Array<IntImm> axis_separators) {
144 PrimFuncFrame frame = FindPrimFuncFrame("T.preflattened_buffer");
145 for (auto const& p : frame->buffer_map) {
146 if (p.second.same_as(postflattened_buffer)) {
147 String buffer_name(postflattened_buffer->name + "_preflatten");
148 Buffer buffer =
149 BufferDecl(shape, dtype, buffer_name, data.value_or(p.second->data), strides, elem_offset,
150 storage_scope, align, offset_factor, buffer_type_str, axis_separators);
151 details::Namer::Name(buffer, buffer_name);
152 frame->preflattened_buffer_map.Set(p.first, buffer);
153 return;
154 }
155 }
156 LOG(FATAL) << "ValueError: postflattened buffer " << postflattened_buffer->name
157 << " does not exist.";
158}

187void Reads(Array<ObjectRef> buffer_slices) {
188 using namespace tvm::tir;
189 BlockFrame frame = FindBlockFrame("T.reads");
190 if (frame->reads.defined()) {
191 LOG(FATAL) << "ValueError: Duplicate read region declaration, previous one is " << frame->reads;
192 }
193 Array<BufferRegion> reads;
194 for (const ObjectRef& obj : buffer_slices) {
195 if (const auto* buffer_region = obj.as<BufferRegionNode>()) {
196 reads.push_back(GetRef<BufferRegion>(buffer_region));
197 } else if (const auto* buffer_load = obj.as<BufferLoadNode>()) {
198 reads.push_back(BufferRegionFromLoad(GetRef<BufferLoad>(buffer_load)));
199 } else {
200 LOG(FATAL) << "Invalid type for buffer reads.";
201 }
202 }
203 frame->reads = reads;
204}

234Buffer AllocBuffer(Array<PrimExpr> shape, DataType dtype, Optional<Var> data,
235 Array<PrimExpr> strides, PrimExpr elem_offset, String storage_scope, int align,
236 int offset_factor, String buffer_type_str, Array<IntImm> axis_separators) {
237 Buffer buffer = BufferDecl(shape, dtype, "", data, strides, elem_offset, storage_scope, align,
238 offset_factor, buffer_type_str, axis_separators);
239 IRBuilder builder = IRBuilder::Current();
240 if (Optional<BlockFrame> frame = builder->GetLastFrame<BlockFrame>()) {
241 frame.value()->alloc_buffers.push_back(buffer);
242 } else if (Optional<PrimFuncFrame> frame = builder->GetLastFrame<PrimFuncFrame>()) {
243 frame.value()->root_alloc_buffers.push_back(buffer);
244 } else {
245 LOG(FATAL) << "ValueError: Block frame or PrimFunc frame not find. Please ensure "
246 "'T.alloc_buffer' is called under T.block() or T.prim_func()";
247 }
248 return buffer;
249}

252IterVar PushBlockVar(IterVar iter_var, PrimExpr binding) {
253 if (Optional<BlockFrame> opt_frame = IRBuilder::Current()->GetLastFrame<BlockFrame>()) {
254 BlockFrame frame = opt_frame.value();
255 frame->iter_vars.push_back(iter_var);
256 frame->iter_values.push_back(binding);
257 } else {
258 LOG(FATAL) << "TypeError: The last frame is not BlockFrame";
259 }
260 return iter_var;
261}

263#define TVM_TIR_IR_BUILDER_AXIS(Method, Kind, Name) \
264 Var Method(Range dom, PrimExpr binding, DataType dtype) { \
265 ICHECK(dom.defined()) << Name << " axis must have a domain"; \
266 int bits = std::max({dom->min.dtype().bits(), dom->extent.dtype().bits(), dtype.bits()}); \
267 return PushBlockVar(IterVar(/*dom=*/dom, /*var=*/Var("", dtype.with_bits(bits)), \
268 /*iter_type=*/Kind, /*thread_tag=*/""), \
269 binding) \
270 ->var; \
271 }
272TVM_TIR_IR_BUILDER_AXIS(Spatial, tvm::tir::IterVarType::kDataPar, "Spatial");
273TVM_TIR_IR_BUILDER_AXIS(Reduce, tvm::tir::IterVarType::kCommReduce, "Reduction");
274TVM_TIR_IR_BUILDER_AXIS(Scan, tvm::tir::IterVarType::kOrdered, "Scan");
275TVM_TIR_IR_BUILDER_AXIS(Opaque, tvm::tir::IterVarType::kOpaque, "Opaque");
276#undef TVM_TIR_IR_BUILDER_AXIS

278Array<Var> Remap(String kinds, Array<PrimExpr> bindings, DataType dtype) {
279 using namespace tvm::tir;
280 Array<Var> results;
281 ICHECK_EQ(kinds.size(), bindings.size());
282 int n = bindings.size();
283 results.reserve(n);
284 for (int i = 0; i < n; ++i) {
285 char c = kinds.c_str()[i];
286 PrimExpr e = bindings[i];
287 const VarNode* v = e.as<VarNode>();
288 ICHECK(v) << "TypeError: Only Var is supported in T.axis.remap";
289 Range dom{nullptr};
290 for (const auto& frame : IRBuilder::Current()->frames) {
291 if (const auto* for_frame = frame.as<ForFrameNode>()) {
292 ICHECK_EQ(for_frame->doms.size(), for_frame->vars.size());
293 int n = for_frame->doms.size();
294 for (int i = 0; i < n; ++i) {
295 if (for_frame->vars[i].get() == v) {
296 dom = for_frame->doms[i];
297 break;
298 }
299 }
300 if (dom.defined()) {
301 break;
302 }
303 }
304 }
305 ICHECK(dom.defined()) << "TypeError: Variable is not in the loop: " << GetRef<Var>(v);
306 DataType dtype = v->dtype;
307 if (c == 'S') {
308 results.push_back(PushBlockVar(IterVar(/*dom=*/dom,
309 /*var=*/Var("", dtype),
310 /*iter_type=*/IterVarType::kDataPar,
311 /*thread_tag=*/""),
312 e)
313 ->var);
314 } else if (c == 'R') {
315 results.push_back(PushBlockVar(IterVar(/*dom=*/dom,
316 /*var=*/Var("", dtype),
317 /*iter_type=*/IterVarType::kCommReduce,
318 /*thread_tag=*/""),
319 e)
320 ->var);
321 } else {
322 LOG(FATAL) << "Unknown axis kind: " << c;
323 }
324 }
325 return results;
326}

330#define TVM_TIR_IR_BUILDER_FOR_FRAME(Method, Kind) \
331 ForFrame Method(PrimExpr start, PrimExpr stop, Optional<Map<String, ObjectRef>> annotations) { \
332 PrimExpr min = start; \
333 PrimExpr extent = arith::Analyzer().Simplify(stop - start); \
334 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>(); \
335 int bits = std::max(min.dtype().bits(), extent.dtype().bits()); \
336 n->vars = {Var("v", DataType::Int(bits))}; \
337 n->doms = {Range::FromMinExtent(min, extent)}; \
338 n->f_make_for_loop = [annotations](Array<Var> vars, Array<Range> doms, tvm::tir::Stmt body) { \
339 ICHECK_EQ(vars.size(), 1); \
340 ICHECK_EQ(doms.size(), 1); \
341 return tvm::tir::For(vars[0], doms[0]->min, doms[0]->extent, Kind, body, NullOpt, \
342 annotations.value_or(Map<String, ObjectRef>())); \
343 }; \
344 return ForFrame(n); \
345 }
346
347TVM_TIR_IR_BUILDER_FOR_FRAME(Serial, tvm::tir::ForKind::kSerial);
348TVM_TIR_IR_BUILDER_FOR_FRAME(Parallel, tvm::tir::ForKind::kParallel);
349TVM_TIR_IR_BUILDER_FOR_FRAME(Vectorized, tvm::tir::ForKind::kVectorized);
350TVM_TIR_IR_BUILDER_FOR_FRAME(Unroll, tvm::tir::ForKind::kUnrolled);
351
352#undef TVM_TIR_IR_BUILDER_FOR_FRAME

354ForFrame ThreadBinding(PrimExpr start, PrimExpr stop, String thread,
355 Optional<Map<String, ObjectRef>> annotations) {
356 using namespace tvm::tir;
357 PrimExpr min = start;
358 PrimExpr extent = arith::Analyzer().Simplify(stop - start);
359 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>();
360 int bits = std::max(min.dtype().bits(), extent.dtype().bits());
361 n->vars = {Var("v", DataType::Int(bits))};
362 n->doms = {Range::FromMinExtent(min, extent)};
363 n->f_make_for_loop = [annotations, thread](Array<Var> vars, Array<Range> doms, Stmt body) -> For {
364 ICHECK_EQ(vars.size(), 1);
365 ICHECK_EQ(doms.size(), 1);
366 IterVar iter_var(Range(nullptr), Var("iter", DataType::Int(32)), IterVarType::kThreadIndex,
367 thread);
368 return For(vars[0], doms[0]->min, doms[0]->extent, ForKind::kThreadBinding, body, iter_var,
369 annotations.value_or(Map<String, ObjectRef>()));
370 };
371 return ForFrame(n);
372}
373
374ForFrame Grid(Array<PrimExpr> extents) {
375 using namespace tvm::tir;
376 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>();
377 n->vars.reserve(extents.size());
378 n->doms.reserve(extents.size());
379 for (const auto& extent : extents) {
380 DataType dtype = extent.dtype();
381 n->vars.push_back(Var("v", extent.dtype()));
382 n->doms.push_back(Range(make_const(dtype, 0), extent));
383 }
384 n->f_make_for_loop = [](Array<Var> vars, Array<Range> doms, Stmt body) -> Stmt {
385 ICHECK_EQ(vars.size(), doms.size());
386 int n = vars.size();
387 for (int i = n - 1; i >= 0; --i) {
388 Range dom = doms[i];
389 Var var = vars[i];
390 body = For(var, dom->min, dom->extent, ForKind::kSerial, std::move(body),
391 /*thread_binding=*/NullOpt, /*annotations=*/{});
392 }
393 return body;
394 };
395 return ForFrame(n);
396}

412LaunchThreadFrame LaunchThread(Var var, PrimExpr extent) {
413 IterVar iter_var{nullptr};
414
415 if (Optional<PrimFuncFrame> opt_frame = IRBuilder::Current()->FindFrame<PrimFuncFrame>()) {
416 if (Optional<IterVar> opt_iter_var = opt_frame.value()->env_threads.Get(var)) {
417 iter_var = opt_iter_var.value();
418 } else {
419 LOG(FATAL) << "ValueError: " << var->name_hint
420 << " is not an env_thread created using T.env_thread.";
421 }
422 } else {
423 LOG(FATAL) << "LaunchThread can only be used inside a PrimFunc";
424 }
425 ObjectPtr<LaunchThreadFrameNode> n = make_object<LaunchThreadFrameNode>();
426 if (!iter_var->dom.defined()) {
427 const_cast<tvm::tir::IterVarNode*>(iter_var.get())->dom = Range(0, extent);
428 } else if (!arith::Analyzer().CanProveEqual(iter_var->dom->extent, extent)) {
429 LOG(FATAL) << "ValueError: Inconsistent extents of environment thread. "
430 << iter_var->dom->extent << " vs " << extent;
431 }
432 n->iter_var = iter_var;
433 n->extent = extent;
434 n->attr_key = iter_var->thread_tag == "vthread" ? "virtual_thread" : "thread_extent";
435 return LaunchThreadFrame(n);
436}
