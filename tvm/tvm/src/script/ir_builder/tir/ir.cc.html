31Buffer BufferDecl(Array<PrimExpr> shape, DataType dtype, String buffer_name, Optional<Var> data,
32 Optional<Array<PrimExpr>> strides, Optional<PrimExpr> elem_offset,
33 String storage_scope, int align, int offset_factor, String buffer_type,
34 Optional<Array<IntImm>> axis_separators) {
35 Var buffer_data;
36 if (!data.defined()) {
37 DataType storage_dtype = dtype;
38 if (storage_dtype == DataType::Bool()) {
39 storage_dtype = DataType::Int(8);
40 }
41 buffer_data = tvm::tir::Var(buffer_name, PointerType(PrimType(storage_dtype), storage_scope));
42 } else {
43 buffer_data = data.value();
44 }
45 if (!elem_offset.defined() && offset_factor) {
46 DataType shape_dtype = shape.empty() ? DataType::Int(32) : shape[0]->dtype;
47 elem_offset = tvm::tir::Var("elem_offset", shape_dtype);
48 }
49 return Buffer(buffer_data, dtype, shape, strides.value_or(Array<PrimExpr>()),
50 elem_offset.value_or(PrimExpr()), buffer_name, align, offset_factor,
51 (buffer_type == "auto_broadcast") ? tvm::tir::kAutoBroadcast : tvm::tir::kDefault,
52 axis_separators.value_or(Array<IntImm>()));
53}

111Buffer MatchBuffer(ObjectRef param, Array<PrimExpr> shape, DataType dtype, Optional<Var> data,
112 Array<PrimExpr> strides, PrimExpr elem_offset, String storage_scope, int align,
113 int offset_factor, String buffer_type_str, Array<IntImm> axis_separators) {
114 Buffer buffer = BufferDecl(shape, dtype, "", data, strides, elem_offset, storage_scope, align,
115 offset_factor, buffer_type_str, axis_separators);
116 if (const auto* var = param.as<tvm::tir::VarNode>()) {
117 PrimFuncFrame frame = FindPrimFuncFrame("T.match_buffer");
118 Var v = GetRef<Var>(var);
119 for (auto const& arg : frame->args) {
120 if (arg.same_as(v)) {
121 frame->buffer_map.Set(v, buffer);
122 return buffer;
123 }
124 }
125 LOG(FATAL) << "ValueError: Can not bind non-input param to buffer.";
126 } else if (const auto* buffer_load = param.as<tvm::tir::BufferLoadNode>()) {
127 BlockFrame frame = FindBlockFrame("T.match_buffer");
128 frame->match_buffers.push_back(tvm::tir::MatchBufferRegion(
129 buffer, BufferRegionFromLoad(GetRef<tvm::tir::BufferLoad>(buffer_load))));
130 } else if (const auto* buffer_region = param.as<tvm::tir::BufferRegionNode>()) {
131 BlockFrame frame = FindBlockFrame("T.match_buffer");
132 frame->match_buffers.push_back(
133 tvm::tir::MatchBufferRegion(buffer, GetRef<tvm::tir::BufferRegion>(buffer_region)));
134 } else {
135 LOG(FATAL) << "ValueError: Unexpected type for TIR MatchBuffer.";
136 }
137 return buffer;
138}

140void PreflattenedBuffer(Buffer postflattened_buffer, Array<PrimExpr> shape, DataType dtype,
141 Optional<Var> data, Array<PrimExpr> strides, PrimExpr elem_offset,
142 String storage_scope, int align, int offset_factor, String buffer_type_str,
143 Array<IntImm> axis_separators) {
144 PrimFuncFrame frame = FindPrimFuncFrame("T.preflattened_buffer");
145 for (auto const& p : frame->buffer_map) {
146 if (p.second.same_as(postflattened_buffer)) {
147 String buffer_name(postflattened_buffer->name + "_preflatten");
148 Buffer buffer =
149 BufferDecl(shape, dtype, buffer_name, data.value_or(p.second->data), strides, elem_offset,
150 storage_scope, align, offset_factor, buffer_type_str, axis_separators);
151 details::Namer::Name(buffer, buffer_name);
152 frame->preflattened_buffer_map.Set(p.first, buffer);
153 return;
154 }
155 }
156 LOG(FATAL) << "ValueError: postflattened buffer " << postflattened_buffer->name
157 << " does not exist.";
158}

178IterVar PushBlockVar(IterVar iter_var, PrimExpr binding) {
179 if (Optional<BlockFrame> opt_frame = IRBuilder::Current()->GetLastFrame<BlockFrame>()) {
180 BlockFrame frame = opt_frame.value();
181 frame->iter_vars.push_back(iter_var);
182 frame->iter_values.push_back(binding);
183 } else {
184 LOG(FATAL) << "TypeError: The last frame is not BlockFrame";
185 }
186 return iter_var;
187}

189#define TVM_TIR_IR_BUILDER_AXIS(Method, Kind, Name) \
190 Var Method(Range dom, PrimExpr binding, DataType dtype) { \
191 ICHECK(dom.defined()) << Name << " axis must have a domain"; \
192 int bits = std::max({dom->min.dtype().bits(), dom->extent.dtype().bits(), dtype.bits()}); \
193 return PushBlockVar(IterVar(/*dom=*/dom, /*var=*/Var("", dtype.with_bits(bits)), \
194 /*iter_type=*/Kind, /*thread_tag=*/""), \
195 binding) \
196 ->var; \
197 }
198TVM_TIR_IR_BUILDER_AXIS(Spatial, tvm::tir::IterVarType::kDataPar, "Spatial");
199TVM_TIR_IR_BUILDER_AXIS(Reduce, tvm::tir::IterVarType::kCommReduce, "Reduction");
200TVM_TIR_IR_BUILDER_AXIS(Scan, tvm::tir::IterVarType::kOrdered, "Scan");
201TVM_TIR_IR_BUILDER_AXIS(Opaque, tvm::tir::IterVarType::kOpaque, "Opaque");
202#undef TVM_TIR_IR_BUILDER_AXIS

204Array<Var> Remap(String kinds, Array<PrimExpr> bindings, DataType dtype) {
205 using namespace tvm::tir;
206 Array<Var> results;
207 ICHECK_EQ(kinds.size(), bindings.size());
208 int n = bindings.size();
209 results.reserve(n);
210 for (int i = 0; i < n; ++i) {
211 char c = kinds.c_str()[i];
212 PrimExpr e = bindings[i];
213 const VarNode* v = e.as<VarNode>();
214 ICHECK(v) << "TypeError: Only Var is supported in T.axis.remap";
215 Range dom{nullptr};
216 for (const auto& frame : IRBuilder::Current()->frames) {
217 if (const auto* for_frame = frame.as<ForFrameNode>()) {
218 ICHECK_EQ(for_frame->doms.size(), for_frame->vars.size());
219 int n = for_frame->doms.size();
220 for (int i = 0; i < n; ++i) {
221 if (for_frame->vars[i].get() == v) {
222 dom = for_frame->doms[i];
223 break;
224 }
225 }
226 if (dom.defined()) {
227 break;
228 }
229 }
230 }
231 ICHECK(dom.defined()) << "TypeError: Variable is not in the loop: " << GetRef<Var>(v);
232 DataType dtype = v->dtype;
233 if (c == 'S') {
234 results.push_back(PushBlockVar(IterVar(/*dom=*/dom,
235 /*var=*/Var("", dtype),
236 /*iter_type=*/IterVarType::kDataPar,
237 /*thread_tag=*/""),
238 e)
239 ->var);
240 } else if (c == 'R') {
241 results.push_back(PushBlockVar(IterVar(/*dom=*/dom,
242 /*var=*/Var("", dtype),
243 /*iter_type=*/IterVarType::kCommReduce,
244 /*thread_tag=*/""),
245 e)
246 ->var);
247 } else {
248 LOG(FATAL) << "Unknown axis kind: " << c;
249 }
250 }
251 return results;
252}

256#define TVM_TIR_IR_BUILDER_FOR_FRAME(Method, Kind) \
257 ForFrame Method(PrimExpr start, PrimExpr stop, Optional<Map<String, ObjectRef>> annotations) { \
258 PrimExpr min = start; \
259 PrimExpr extent = arith::Analyzer().Simplify(stop - start); \
260 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>(); \
261 int bits = std::max(min.dtype().bits(), extent.dtype().bits()); \
262 n->vars = {Var("v", DataType::Int(bits))}; \
263 n->doms = {Range::FromMinExtent(min, extent)}; \
264 n->f_make_for_loop = [annotations](Array<Var> vars, Array<Range> doms, tvm::tir::Stmt body) { \
265 ICHECK_EQ(vars.size(), 1); \
266 ICHECK_EQ(doms.size(), 1); \
267 return tvm::tir::For(vars[0], doms[0]->min, doms[0]->extent, Kind, body, NullOpt, \
268 annotations.value_or(Map<String, ObjectRef>())); \
269 }; \
270 return ForFrame(n); \
271 }
272
273TVM_TIR_IR_BUILDER_FOR_FRAME(Serial, tvm::tir::ForKind::kSerial);
274TVM_TIR_IR_BUILDER_FOR_FRAME(Parallel, tvm::tir::ForKind::kParallel);
275TVM_TIR_IR_BUILDER_FOR_FRAME(Vectorized, tvm::tir::ForKind::kVectorized);
276TVM_TIR_IR_BUILDER_FOR_FRAME(Unroll, tvm::tir::ForKind::kUnrolled);
277
278#undef TVM_TIR_IR_BUILDER_FOR_FRAME

280ForFrame ThreadBinding(PrimExpr start, PrimExpr stop, String thread,
281 Optional<Map<String, ObjectRef>> annotations) {
282 using namespace tvm::tir;
283 PrimExpr min = start;
284 PrimExpr extent = arith::Analyzer().Simplify(stop - start);
285 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>();
286 int bits = std::max(min.dtype().bits(), extent.dtype().bits());
287 n->vars = {Var("v", DataType::Int(bits))};
288 n->doms = {Range::FromMinExtent(min, extent)};
289 n->f_make_for_loop = [annotations, thread](Array<Var> vars, Array<Range> doms, Stmt body) -> For {
290 ICHECK_EQ(vars.size(), 1);
291 ICHECK_EQ(doms.size(), 1);
292 IterVar iter_var(Range(nullptr), Var("iter", DataType::Int(32)), IterVarType::kThreadIndex,
293 thread);
294 return For(vars[0], doms[0]->min, doms[0]->extent, ForKind::kThreadBinding, body, iter_var,
295 annotations.value_or(Map<String, ObjectRef>()));
296 };
297 return ForFrame(n);
298}

300ForFrame Grid(Array<PrimExpr> extents) {
301 using namespace tvm::tir;
302 ObjectPtr<ForFrameNode> n = make_object<ForFrameNode>();
303 n->vars.reserve(extents.size());
304 n->doms.reserve(extents.size());
305 for (const auto& extent : extents) {
306 DataType dtype = extent.dtype();
307 n->vars.push_back(Var("v", extent.dtype()));
308 n->doms.push_back(Range(make_const(dtype, 0), extent));
309 }
310 n->f_make_for_loop = [](Array<Var> vars, Array<Range> doms, Stmt body) -> Stmt {
311 ICHECK_EQ(vars.size(), doms.size());
312 int n = vars.size();
313 for (int i = n - 1; i >= 0; --i) {
314 Range dom = doms[i];
315 Var var = vars[i];
316 body = For(var, dom->min, dom->extent, ForKind::kSerial, std::move(body),
317 /*thread_binding=*/NullOpt, /*annotations=*/{});
318 }
319 return body;
320 };
321 return ForFrame(n);
322}
