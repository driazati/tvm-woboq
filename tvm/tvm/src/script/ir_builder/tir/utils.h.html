31inline void AddToParent(tvm::tir::Stmt stmt) {
32 IRBuilder builder = IRBuilder::Current();
33 if (builder->frames.empty()) {
34 ICHECK(!builder->result.defined()) << "ValueError: Builder.result has already been set";
35 builder->result = stmt;
36 } else if (const auto* tir_frame = builder->frames.back().as<TIRFrameNode>()) {
37 GetRef<TIRFrame>(tir_frame)->stmts.push_back(stmt);
38 } else {
39 LOG(FATAL) << "TypeError: Unsupported frame type: " << builder->frames.back();
40 }
41}
42
43inline tvm::tir::Stmt AsStmt(const Array<tvm::tir::Stmt>& stmt) {
44 using namespace tvm::tir;
45 if (stmt.empty()) {
46 return tvm::tir::Evaluate(0);
47 } else if (stmt.size() == 1) {
48 return stmt[0];
49 } else {
50 return SeqStmt(stmt);
51 }
52}
53
54inline PrimFuncFrame FindPrimFuncFrame(const String& method) {
55 if (Optional<PrimFuncFrame> frame = IRBuilder::Current()->GetLastFrame<PrimFuncFrame>()) {
56 return frame.value();
57 }
58 LOG(FATAL) << "ValueError: PrimFunc frame not find. Please ensure '" << method
59 << "' is called under T.prim_func()";
60 throw;
61}
62
63inline BlockFrame FindBlockFrame(const String& method) {
64 if (Optional<BlockFrame> frame = IRBuilder::Current()->GetLastFrame<BlockFrame>()) {
65 return frame.value();
66 }
67 LOG(FATAL) << "ValueError: Block frame not find. Please ensure '" << method
68 << "' is called under T.block()";
69 throw;
70}
