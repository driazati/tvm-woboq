31/*!
32 * \brief Add tir Stmt to the top frame in IRBuilder frame stack.
33 * \param stmt The Stmt.
34 */
35inline void AddToParent(tvm::tir::Stmt stmt) {
36 IRBuilder builder = IRBuilder::Current();
37 if (builder->frames.empty()) {
38 ICHECK(!builder->result.defined()) << "ValueError: Builder.result has already been set";
39 builder->result = stmt;
40 } else if (const auto* tir_frame = builder->frames.back().as<TIRFrameNode>()) {
41 GetRef<TIRFrame>(tir_frame)->stmts.push_back(stmt);
42 } else {
43 LOG(FATAL) << "TypeError: Unsupported frame type: " << builder->frames.back();
44 }
45}

47/*!
48 * \brief Convert array of tir Stmt to single Stmt.
49 * \param stmt The array of Stmt.
50 * \return The SeqStmt.
51 */
52inline tvm::tir::Stmt AsStmt(const Array<tvm::tir::Stmt>& stmt) {
53 using namespace tvm::tir;
54 if (stmt.empty()) {
55 return tvm::tir::Evaluate(0);
56 } else if (stmt.size() == 1) {
57 return stmt[0];
58 } else {
59 return SeqStmt(stmt);
60 }
61}

63/*!
64 * \brief Check whether the top frame in IRBuilder frame stack is PrimFuncFrame.
65 * \param method The method name to be printed when throwing exception.
66 * \return The top frame of PrimFuncFrame.
67 */
68inline PrimFuncFrame FindPrimFuncFrame(const String& method) {
69 if (Optional<PrimFuncFrame> frame = IRBuilder::Current()->GetLastFrame<PrimFuncFrame>()) {
70 return frame.value();
71 }
72 LOG(FATAL) << "ValueError: PrimFunc frame not find. Please ensure '" << method
73 << "' is called under T.prim_func()";
74 throw;
75}

106/*!
107 * \brief Convert BufferLoad to BufferRegion.
108 * \param buffer_load The BufferLoad.
109 * \return The converted BufferRegion.
110 */
111inline tvm::tir::BufferRegion BufferRegionFromLoad(tvm::tir::BufferLoad buffer_load) {
112 Array<Range> ranges;
113 for (const PrimExpr& index : buffer_load->indices) {
114 ranges.push_back(Range::FromMinExtent(index, IntImm(index->dtype, 1)));
115 }
116 return tvm::tir::BufferRegion(buffer_load->buffer, ranges);
117}
