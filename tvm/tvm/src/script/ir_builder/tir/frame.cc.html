30void PrimFuncFrameNode::ExitWithScope() {
31 TIRFrameNode::ExitWithScope();
32 tvm::tir::PrimFunc func(
33 /*params=*/args,
34 /*body=*/AsStmt(stmts),
35 /*ret_type=*/ret_type.value_or(TupleType::Empty()),
36 /*buffer_map=*/buffer_map,
37 /*preflattened_buffer_map=*/preflattened_buffer_map,
38 /*attrs=*/attrs.defined() ? DictAttrs(attrs.value()) : NullValue<DictAttrs>());
39 func = tvm::tir::ScriptComplete(func, root_alloc_buffers);
40 IRBuilder builder = IRBuilder::Current();
41 if (builder->frames.empty()) {
42 ICHECK(!builder->result.defined()) << "ValueError: Builder.result has already been set";
43 builder->result = func;
44 } else if (Optional<ir::IRModuleFrame> opt_frame = builder->FindFrame<ir::IRModuleFrame>()) {
45 ir::IRModuleFrame frame = opt_frame.value();
46 frame->global_vars.push_back(GlobalVar(name.value_or("")));
47 frame->functions.push_back(func);
48 } else {
49 LOG(FATAL) << "ValueError: Cannot find where to insert PrimFunc";
50 }
51}

53void BlockFrameNode::ExitWithScope() {
54 TIRFrameNode::ExitWithScope();
55 Array<tvm::tir::Buffer> tir_alloc_buffers;
56 for (const tvm::tir::Buffer& buffer : alloc_buffers) {
57 tir_alloc_buffers.push_back(buffer);
58 }
59 Map<String, ObjectRef> attrs = annotations.value_or({});
60 if (int detect_access = (!reads.defined()) | (!writes.defined() << 1)) {
61 attrs.Set("tir.script_parsing_detect_access", tvm::IntImm(DataType::Int(64), detect_access));
62 }
63 tvm::tir::Block block(iter_vars, reads.value_or(Array<tvm::tir::BufferRegion>()),
64 writes.value_or(Array<tvm::tir::BufferRegion>()), name, AsStmt(stmts), init,
65 tir_alloc_buffers, match_buffers, attrs);
66 if (no_realize) {
67 CHECK(iter_values.empty())
68 << "ValueError: Block bindings are not allowed when `no_realize=True`";
69 CHECK(!predicate.defined()) << "ValueError: `T.where` is not allowed when `no_realize=True`";
70 AddToParent(block);
71 } else {
72 AddToParent(tvm::tir::BlockRealize(iter_values, predicate.value_or(Bool(true)), block));
73 }
74}

76void BlockInitFrameNode::EnterWithScope() {
77 BlockFrame frame = FindBlockFrame("T.init");
78 if (frame->init.defined()) {
79 LOG(FATAL) << "ValueError: Duplicate block init declaration";
80 }
81 TIRFrameNode::EnterWithScope();
82}
83
84void BlockInitFrameNode::ExitWithScope() {
85 TIRFrameNode::ExitWithScope();
86 BlockFrame frame = FindBlockFrame("T.init");
87 frame->init = AsStmt(stmts);
88}

139void IfFrameNode::ExitWithScope() {
140 TIRFrameNode::ExitWithScope();
141 if (!stmts.empty()) {
142 LOG(FATAL) << "stmt within IfThenElse frame should be either in ThenFrame or ElseFrame";
143 }
144 if (!then_stmts.defined()) {
145 LOG(FATAL) << "IfThenElse frame should have at least one then branch";
146 }
147 AddToParent(tvm::tir::IfThenElse(
148 condition, AsStmt(then_stmts.value()),
149 else_stmts.defined() ? AsStmt(else_stmts.value()) : tvm::tir::Stmt(nullptr)));
150}
151
152void ThenFrameNode::EnterWithScope() {
153 IfFrame frame = FindIfFrame("T.then_");
154 if (frame->then_stmts.defined()) {
155 LOG(FATAL) << "ValueError: Duplicate then branch declaration, previous one is "
156 << frame->then_stmts.value();
157 }
158 TIRFrameNode::EnterWithScope();
159}
160
161void ThenFrameNode::ExitWithScope() {
162 TIRFrameNode::ExitWithScope();
163 FindIfFrame("T.then_")->then_stmts = stmts;
164}
165
166void ElseFrameNode::EnterWithScope() {
167 IfFrame frame = FindIfFrame("T.else_");
168 if (!frame->then_stmts.defined()) {
169 LOG(FATAL) << "The else branch should follow then branch";
170 }
171 if (frame->else_stmts.defined()) {
172 LOG(FATAL) << "ValueError: Duplicate else branch declaration, previous one is "
173 << frame->else_stmts.value();
174 }
175 TIRFrameNode::EnterWithScope();
176}
177
178void ElseFrameNode::ExitWithScope() {
179 TIRFrameNode::ExitWithScope();
180 FindIfFrame("T.else_")->else_stmts = stmts;
181}

183void DeclBufferFrameNode::ExitWithScope() {

186}
