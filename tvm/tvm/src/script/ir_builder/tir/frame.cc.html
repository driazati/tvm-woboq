30void PrimFuncFrameNode::ExitWithScope() {
31 TIRFrameNode::ExitWithScope();
32 tvm::tir::PrimFunc func(
33 /*params=*/args,
34 /*body=*/AsStmt(stmts),
35 /*ret_type=*/ret_type.value_or(TupleType::Empty()),
36 /*buffer_map=*/buffer_map,
37 /*preflattened_buffer_map=*/preflattened_buffer_map,
38 /*attrs=*/attrs.defined() ? DictAttrs(attrs.value()) : NullValue<DictAttrs>());
39 func = tvm::tir::ScriptComplete(func, root_alloc_buffers);
40 IRBuilder builder = IRBuilder::Current();
41 if (builder->frames.empty()) {
42 ICHECK(!builder->result.defined()) << "ValueError: Builder.result has already been set";
43 builder->result = func;
44 } else if (Optional<ir::IRModuleFrame> opt_frame = builder->FindFrame<ir::IRModuleFrame>()) {
45 ir::IRModuleFrame frame = opt_frame.value();
46 frame->global_vars.push_back(GlobalVar(name.value_or("")));
47 frame->functions.push_back(func);
48 } else {
49 LOG(FATAL) << "ValueError: Cannot find where to insert PrimFunc";
50 }
51}
52
53void BlockFrameNode::ExitWithScope() {
54 TIRFrameNode::ExitWithScope();
55 Array<tvm::tir::Buffer> tir_alloc_buffers;
56 for (const tvm::tir::Buffer& buffer : alloc_buffers) {
57 tir_alloc_buffers.push_back(buffer);
58 }
59 Map<String, ObjectRef> attrs = annotations.value_or({});
60 if (int detect_access = (!reads.defined()) | (!writes.defined() << 1)) {
61 attrs.Set("tir.script_parsing_detect_access", tvm::IntImm(DataType::Int(64), detect_access));
62 }
63 tvm::tir::Block block(iter_vars, reads.value_or(Array<tvm::tir::BufferRegion>()),
64 writes.value_or(Array<tvm::tir::BufferRegion>()), name, AsStmt(stmts), init,
65 tir_alloc_buffers, match_buffers, attrs);
66 if (no_realize) {
67 CHECK(iter_values.empty())
68 << "ValueError: Block bindings are not allowed when `no_realize=True`";
69 CHECK(!predicate.defined()) << "ValueError: `T.where` is not allowed when `no_realize=True`";
70 AddToParent(block);
71 } else {
72 AddToParent(tvm::tir::BlockRealize(iter_values, predicate.value_or(Bool(true)), block));
73 }
74}
75
76void ForFrameNode::ExitWithScope() {
77 TIRFrameNode::ExitWithScope();
78 AddToParent(this->f_make_for_loop(vars, doms, AsStmt(stmts)));
79}
