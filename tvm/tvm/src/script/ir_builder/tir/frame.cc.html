30void PrimFuncFrameNode::ExitWithScope() {
31 TIRFrameNode::ExitWithScope();
32 tvm::tir::PrimFunc func(
33 /*params=*/args,
34 /*body=*/AsStmt(stmts),
35 /*ret_type=*/ret_type.value_or(TupleType::Empty()),
36 /*buffer_map=*/buffer_map,
37 /*preflattened_buffer_map=*/preflattened_buffer_map,
38 /*attrs=*/attrs.defined() ? DictAttrs(attrs.value()) : NullValue<DictAttrs>());
39 func = tvm::tir::ScriptComplete(func, root_alloc_buffers);
40 IRBuilder builder = IRBuilder::Current();
41 if (builder->frames.empty()) {
42 ICHECK(!builder->result.defined()) << "ValueError: Builder.result has already been set";
43 builder->result = func;
44 } else if (Optional<ir::IRModuleFrame> opt_frame = builder->FindFrame<ir::IRModuleFrame>()) {
45 ir::IRModuleFrame frame = opt_frame.value();
46 frame->global_vars.push_back(GlobalVar(name.value_or("")));
47 frame->functions.push_back(func);
48 } else {
49 LOG(FATAL) << "ValueError: Cannot find where to insert PrimFunc";
50 }
51}
