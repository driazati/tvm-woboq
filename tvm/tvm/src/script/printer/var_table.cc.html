30String GenerateUniqueName(const String& name_hint, std::unordered_set<String>* defined_names) {
31 String name = name_hint;
32 for (int i = 1; !defined_names->insert(name).second; ++i) {
33 name = name_hint + "_" + std::to_string(i);
34 }
35 return name;
36}
37
38IdDoc VarTableNode::Define(const ObjectRef& obj, const String& name_hint,
39 const ObjectPath& object_path, const Frame& frame) {
40 String name = GenerateUniqueName(name_hint, &this->defined_names);
41 DocFactory doc_factory = [name]() { return IdDoc(name); };
42
43 auto result = obj2info.insert({obj, VariableInfo{std::move(doc_factory), name}});
44 ICHECK(result.second) << "Duplicated object: " << obj;
45
46 IdDoc def_doc(name);
47 def_doc->source_paths.push_back(object_path);
48
49 frame->AddExitCallback([this, obj]() { this->RemoveVar(obj); });
50
51 return def_doc;
52}
53
54void VarTableNode::DefineByDoc(const ObjectRef& obj, DocFactory doc_factory, const Frame& frame) {
55 ICHECK(obj2info.find(obj) == obj2info.end()) << "Duplicated object: " << obj;
56
57 ICHECK(!doc_factory()->IsInstance<IdDocNode>())
58 << "VarTableNode::Define cannot be used for variable that's mapped to IdDoc.";
59
60 obj2info.insert({obj, VariableInfo{std::move(doc_factory), NullOpt}});
61
62 frame->AddExitCallback([this, obj]() { this->RemoveVar(obj); });
63}
64
65Optional<ExprDoc> VarTableNode::GetVarDoc(const ObjectRef& obj,
66 const ObjectPath& object_path) const {
67 auto it = obj2info.find(obj);
68 if (it == obj2info.end()) {
69 return NullOpt;
70 }
71 ExprDoc doc = it->second.doc_factory();
72 doc->source_paths.push_back(object_path);
73 return doc;
74}
75
76bool VarTableNode::IsVarDefined(const ObjectRef& obj) const { return obj2info.count(obj); }
77
78void VarTableNode::RemoveVar(const ObjectRef& obj) {
79 auto it = obj2info.find(obj);
80 ICHECK(it != obj2info.end()) << "No such object: " << obj;
81
82 if (it->second.name.defined()) {
83 defined_names.erase(it->second.name.value());
84 }
85 obj2info.erase(it);
86}

88VarTable::VarTable() { data_ = make_object<VarTableNode>(); }

90TVM_REGISTER_NODE_TYPE(VarTableNode);
91TVM_REGISTER_GLOBAL("script.printer.VarTable").set_body_typed([]() { return VarTable(); });
92TVM_REGISTER_GLOBAL("script.printer.VarTableDefine")
93 .set_body_method<VarTable, VarTableNode, IdDoc, const ObjectRef&, const String&,
94 const ObjectPath&, const Frame&>(&VarTableNode::Define);
95TVM_REGISTER_GLOBAL("script.printer.VarTableDefineByDoc")
96 .set_body_typed([](VarTable var_table, const ObjectRef& obj, runtime::PackedFunc factory,
97 Frame frame) {
98 var_table->DefineByDoc(
99 obj, [f = std::move(factory)]() { return f(); }, frame);
100 });
101TVM_REGISTER_GLOBAL("script.printer.VarTableGetVarDoc")
102 .set_body_method<VarTable, VarTableNode, Optional<ExprDoc>, const ObjectRef&,
103 const ObjectPath&>(&VarTableNode::GetVarDoc);
104TVM_REGISTER_GLOBAL("script.printer.VarTableIsVarDefined")
105 .set_body_method<VarTable>(&VarTableNode::IsVarDefined);
