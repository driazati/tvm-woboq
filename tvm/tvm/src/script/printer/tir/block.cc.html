25Doc PrintBlock(IRDocsifier d, tir::Block block, ObjectPath block_p, //
26 Optional<tir::BlockRealize> opt_realize, Optional<ObjectPath> opt_realize_p) {
27 With<TIRFrame> frame(d, block);
28 ICHECK_EQ(opt_realize.defined(), opt_realize_p.defined());
29 const tir::BlockRealizeNode* realize = opt_realize.value().get();
30 const ObjectPathNode* realize_p = opt_realize_p.get();
31 // Step 1. Handle block var and block bindings
32 int n_vars = block->iter_vars.size();
33 for (int i = 0; i < n_vars; ++i) {
34 tir::IterVar iter_var = block->iter_vars[i];
35 ObjectPath iter_var_p = block_p->Attr("iter_var")->ArrayIndex(i);
36 ExprDoc rhs = TIR(d)->Attr("axis");
37 if (iter_var->iter_type == tir::IterVarType::kDataPar) {
38 rhs = rhs->Attr("spatial");
39 } else if (iter_var->iter_type == tir::IterVarType::kCommReduce) {
40 rhs = rhs->Attr("reduce");
41 } else if (iter_var->iter_type == tir::IterVarType::kOrdered) {
42 rhs = rhs->Attr("scan");
43 } else if (iter_var->iter_type == tir::IterVarType::kOpaque) {
44 rhs = rhs->Attr("opaque");
45 } else {
46 LOG(FATAL) << "ValueError: Unknown IterVarType in block signature: "
47 << tir::IterVarType2String(iter_var->iter_type);
48 }
49 ExprDoc dom{nullptr};
50 if (tir::is_zero(iter_var->dom->min)) {
51 ExprDoc extent = d->AsDoc<ExprDoc>(iter_var->dom->extent, //
52 iter_var_p->Attr("dom")->Attr("extent"));
53 dom = extent;
54 } else {
55 ExprDoc min = d->AsDoc<ExprDoc>(iter_var->dom->min, iter_var_p->Attr("dom")->Attr("min"));
56 ExprDoc max = d->AsDoc<ExprDoc>(iter_var->dom->min + iter_var->dom->extent,
57 iter_var_p->Attr("dom")->Attr("extent"));
58 dom = TupleDoc({min, max});
59 }
60 if (realize) {
61 ExprDoc binding = d->AsDoc<ExprDoc>(realize->iter_values[i], //
62 realize_p->Attr("iter_values")->ArrayIndex(i));
63 rhs = rhs->Call({dom, binding});
64 } else {
65 rhs = rhs->Call({dom});
66 }
67 (*frame)->stmts.push_back(AssignDoc(DefineVar(iter_var->var, *frame, d), rhs, NullOpt));
68 }
69 // Step 2. Handle block predicate
70 if (realize) {
71 ICHECK(realize->predicate.defined() && realize->predicate->dtype.is_bool());
72 if (!tir::is_one(realize->predicate)) {
73 (*frame)->stmts.push_back(ExprStmtDoc(TIR(d)->Attr("where")->Call(
74 {d->AsDoc<ExprDoc>(realize->predicate, realize_p->Attr("predicate"))})));
75 }
76 }
77 // Step 3. Handle block read/write regions
78 {
79 Array<ExprDoc> reads;
80 for (int i = 0, n = block->reads.size(); i < n; ++i) {
81 reads.push_back(d->AsDoc<ExprDoc>(block->reads[i], block_p->Attr("reads")->ArrayIndex(i)));
82 }
83 (*frame)->stmts.push_back(ExprStmtDoc(TIR(d)->Attr("reads")->Call(reads)));
84 Array<ExprDoc> writes;
85 for (int i = 0, n = block->writes.size(); i < n; ++i) {
86 writes.push_back(d->AsDoc<ExprDoc>(block->writes[i], block_p->Attr("writes")->ArrayIndex(i)));
87 }
88 (*frame)->stmts.push_back(ExprStmtDoc(TIR(d)->Attr("writes")->Call(writes)));
89 }
90 // Step 4. Handle block attributes
91 if (!block->annotations.empty()) {
92 (*frame)->stmts.push_back(ExprStmtDoc(
93 TIR(d)
94 ->Attr("block_attr")
95 ->Call({d->AsDoc<ExprDoc>(block->annotations, block_p->Attr("annotations"))})));
96 }
97 // Step 5. Handle `alloc_buffer`
98 for (int i = 0, n = block->alloc_buffers.size(); i < n; ++i) {
99 tir::Buffer buffer = block->alloc_buffers[i];
100 ObjectPath buffer_p = block_p->Attr("alloc_buffers")->ArrayIndex(i);
101 IdDoc lhs = DefineBuffer(buffer, *frame, d);
102 ExprDoc rhs = BufferDecl(buffer, "alloc_buffer", {}, buffer_p, *frame, d);
103 (*frame)->stmts.push_back(AssignDoc(lhs, rhs, NullOpt));
104 }
105 // Step 6. Handle `match_buffer`
106 for (int i = 0, n = block->match_buffers.size(); i < n; ++i) {
107 tir::MatchBufferRegion buffer_region = block->match_buffers[i];
108 ObjectPath buffer_region_p = block_p->Attr("match_buffers")->ArrayIndex(i);
109 StmtDoc doc = d->AsDoc<StmtDoc>(buffer_region, buffer_region_p);
110 (*frame)->stmts.push_back(doc);
111 }
112 // Step 7. Handle init block
113 if (block->init.defined()) {
114 tir::Stmt init = block->init.value();
115 With<TIRFrame> init_frame(d, init);
116 AsDocBody(init, block_p->Attr("init"), init_frame->get(), d);
117 (*frame)->stmts.push_back(
118 ScopeDoc(NullOpt, TIR(d)->Attr("init")->Call({}), (*init_frame)->stmts));
119 }
120 // Step 8. Handle block body
121 AsDocBody(block->body, block_p->Attr("body"), frame->get(), d);
122 return ScopeDoc(NullOpt, TIR(d)->Attr("block")->Call({LiteralDoc::Str(block->name_hint)}),
123 (*frame)->stmts);
124}
125
126TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
127 .set_dispatch<tir::BlockRealize>(
128 "", [](tir::BlockRealize realize, ObjectPath p, IRDocsifier d) -> Doc {
129 return PrintBlock(d, realize->block, p->Attr("block"), realize, p);
130 });
131
132TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
133 .set_dispatch<tir::Block>("", [](tir::Block block, ObjectPath p, IRDocsifier d) -> Doc {
134 return PrintBlock(d, block, p, NullOpt, NullOpt);
135 });
136
137TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
138 .set_dispatch<tir::MatchBufferRegion>(
139 "", [](tir::MatchBufferRegion stmt, ObjectPath p, IRDocsifier d) -> Doc {
140 Frame frame = d->frames.back();
141 ExprDoc lhs = DefineBuffer(stmt->buffer, frame, d);
142 ExprDoc src_buffer = d->AsDoc<ExprDoc>(stmt->source, p->Attr("source"));
143 ExprDoc rhs = BufferDecl(stmt->buffer, "match_buffer", {src_buffer}, p->Attr("buffer"),
144 d->frames.back(), d);
145 return AssignDoc(lhs, rhs, NullOpt);
146 });
