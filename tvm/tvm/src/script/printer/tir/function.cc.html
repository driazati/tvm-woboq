25String FindFunctionName(const IRDocsifier& d, const tir::PrimFunc& f) {
26 if (!d->mod.defined()) {
27 return "main";
28 }
29 for (const auto& kv : d->mod.value()->functions) {
30 if (kv.second.same_as(f)) {
31 return kv.first->name_hint;
32 }
33 }
34 return "main";
35}
36
37TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
38 .set_dispatch<tir::PrimFunc>("", [](tir::PrimFunc func, ObjectPath p, IRDocsifier d) -> Doc {
39 d->SetCommonPrefix(func, [](const ObjectRef& obj) {
40 return obj->IsInstance<tir::VarNode>() || obj->IsInstance<tir::BufferNode>();
41 });
42 With<TIRFrame> frame(d, func);
43 (*frame)->AddDispatchToken(d, "tir");
44 int n_args = func->params.size();
45 // Step 1. Handle `func->params`
46 Array<AssignDoc> args;
47 args.reserve(n_args);
48 for (int i = 0; i < n_args; ++i) {
49 tir::Var var = func->params[i];
50 ObjectPath var_p = p->Attr("params")->ArrayIndex(i);
51 ExprDoc a = d->AsDoc<ExprDoc>(var->type_annotation, var_p->Attr("type_annotation"));
52 args.push_back(AssignDoc(DefineVar(var, *frame, d), NullOpt, a));
53 }
54 // Step 2. Handle `func->attrs`
55 if (func->attrs.defined() && !func->attrs->dict.empty()) {
56 (*frame)->stmts.push_back(
57 ExprStmtDoc(TIR(d)
58 ->Attr("func_attr") //
59 ->Call({d->AsDoc<ExprDoc>(func->attrs, p->Attr("attrs"))})));
60 }
61 // Step 3. Handle `func->buffer_map`
62 for (int i = 0; i < n_args; ++i) {
63 tir::Var param = func->params[i];
64 if (func->buffer_map.count(param)) {
65 tir::Buffer buffer = func->buffer_map[param];
66 ExprDoc param = args[i]->lhs;
67 ObjectPath buffer_p = p->Attr("buffer_map")->MapValue(param);
68 ExprDoc lhs =
69 DefineBuffer(buffer, *frame, d); // TODO(@junrushao): switch `lhs` and `rhs`
70 ExprDoc rhs = BufferDecl(buffer, "match_buffer", {param}, buffer_p, *frame, d);
71 (*frame)->stmts.push_back(AssignDoc(lhs, rhs, NullOpt));
72 }
73 }
74 // Step 4. Handle `func->body`
75 AsDocBody(func->body, p->Attr("body"), frame->get(), d);
76 return FunctionDoc(
77 /*name=*/IdDoc(FindFunctionName(d, func)),
78 /*args=*/args,
79 /*decorators=*/{TIR(d)->Attr("prim_func")},
80 /*return_type=*/d->AsDoc<ExprDoc>(func->ret_type, p->Attr("ret_type")),
81 /*body=*/(*frame)->stmts);
82 });
