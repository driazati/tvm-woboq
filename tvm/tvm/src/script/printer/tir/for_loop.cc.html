25TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
26 .set_dispatch<tir::For>("", [](tir::For loop, ObjectPath p, IRDocsifier d) -> Doc {
27 // Step 1. Check syntactic sugar: `T.grid`
28 std::vector<const tir::ForNode*> grid;
29 std::unordered_set<const tir::VarNode*> grid_loop_vars;
30 auto f_var_dep = [&grid_loop_vars](const PrimExpr& e) -> bool {
31 return tir::UsesVar(e, [&grid_loop_vars](const tir::VarNode* v) -> bool { //
32 return grid_loop_vars.count(v);
33 });
34 };
35 for (const tir::ForNode* l = loop.get(); l != nullptr; l = l->body.as<tir::ForNode>()) {
36 ICHECK(l->loop_var->dtype == l->min->dtype);
37 ICHECK(l->loop_var->dtype == l->extent->dtype);
38 if (l->kind != tir::ForKind::kSerial || //
39 !tir::is_zero(l->min) || //
40 !l->annotations.empty() || //
41 f_var_dep(l->extent)) {
42 break;
43 }
44 grid.push_back(l);
45 grid_loop_vars.insert(l->loop_var.get());
46 }
47 With<TIRFrame> f(d, loop);
48 // Step 2. Construct `T.grid`
49 if (grid.size() > 1) {
50 int n = grid.size();
51 Array<ExprDoc> lhs;
52 Array<ExprDoc> rhs;
53 lhs.reserve(n);
54 rhs.reserve(n);
55 for (int i = 0; i < n; ++i) {
56 const tir::ForNode* loop = grid[i];
57 lhs.push_back(DefineVar(loop->loop_var, *f, d));
58 rhs.push_back(d->AsDoc<ExprDoc>(loop->extent, p->Attr("extent")));
59 p = p->Attr("body");
60 }
61 AsDocBody(grid.back()->body, p, (*f).get(), d);
62 return ForDoc(TupleDoc(lhs), TIR(d)->Attr("grid")->Call(rhs), (*f)->stmts);
63 }
64 // Step 3. If not `T.grid`, print loop kind accordingly
65 IdDoc lhs = DefineVar(loop->loop_var, *f, d);
66 Optional<ExprDoc> min = NullOpt;
67 Optional<ExprDoc> max = NullOpt;
68 Optional<ExprDoc> annotations = NullOpt;
69 Optional<ExprDoc> thread = NullOpt;
70 if (tir::is_zero(loop->min)) {
71 max = d->AsDoc<ExprDoc>(loop->extent, p->Attr("extent"));
72 } else {
73 min = d->AsDoc<ExprDoc>(loop->min, p->Attr("min"));
74 max = d->AsDoc<ExprDoc>(loop->min + loop->extent, p->Attr("extent"));
75 }
76 if (!loop->annotations.empty()) {
77 annotations = d->AsDoc<ExprDoc>(loop->annotations, p->Attr("annotations"));
78 }
79 ExprDoc prefix = TIR(d);
80 if (loop->kind == tir::ForKind::kSerial) {
81 if (loop->annotations.empty()) {
82 prefix = IdDoc("range");
83 } else {
84 prefix = prefix->Attr("serial");
85 }
86 } else if (loop->kind == tir::ForKind::kParallel) {
87 prefix = prefix->Attr("parallel");
88 } else if (loop->kind == tir::ForKind::kUnrolled) {
89 prefix = prefix->Attr("unroll");
90 } else if (loop->kind == tir::ForKind::kVectorized) {
91 prefix = prefix->Attr("vectorized");
92 } else if (loop->kind == tir::ForKind::kThreadBinding) {
93 prefix = prefix->Attr("thread_binding");
94 thread = LiteralDoc::Str(loop->thread_binding.value()->thread_tag);
95 } else {
96 LOG(FATAL) << "ValueError: Unknown ForKind: " << tir::ForKind2String(loop->kind);
97 }
98 Array<ExprDoc> args;
99 Array<String> kwargs_keys;
100 Array<ExprDoc> kwargs_values;
101 if (min.defined()) {
102 args.push_back(min.value());
103 }
104 if (max.defined()) {
105 args.push_back(max.value());
106 }
107 if (thread.defined()) {
108 kwargs_keys.push_back("thread");
109 kwargs_values.push_back(thread.value());
110 }
111 if (annotations.defined()) {
112 kwargs_keys.push_back("annotations");
113 kwargs_values.push_back(annotations.value());
114 }
115 ExprDoc rhs = prefix->Call(args, kwargs_keys, kwargs_values);
116 AsDocBody(loop->body, p, (*f).get(), d);
117 return ForDoc(lhs, rhs, (*f)->stmts);
118 });
