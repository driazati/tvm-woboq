26Doc DoConciseScoping(const Optional<ExprDoc>& lhs, const ExprDoc& rhs, Array<StmtDoc>* stmts,
27 bool concise_scoping) {
28 if (concise_scoping) {
29 if (lhs.defined()) {
30 stmts->insert(stmts->begin(), AssignDoc(lhs.value(), rhs, NullOpt));
31 } else {
32 stmts->insert(stmts->begin(), ExprStmtDoc(rhs));
33 }
34 return StmtBlockDoc(*stmts);
35 } else {
36 return ScopeDoc(lhs, rhs, *stmts);
37 }
38}
39
40bool AllowConciseScoping(const IRDocsifier& d) {
41 ICHECK(!d->frames.empty());
42 if (const auto* f = d->frames.back().as<TIRFrameNode>()) {
43 return f->allow_concise_scoping;
44 }
45 LOG(FATAL) << "NotImplementedError: fragment printing";
46}

49 .set_dispatch<tir::Evaluate>("", [](tir::Evaluate eval, ObjectPath p, IRDocsifier d) -> Doc {
50 ExprDoc value = d->AsDoc<ExprDoc>(eval->value, p->Attr("value"));
51 if (eval->value->IsInstance<tir::CallNode>()) {
52 return ExprStmtDoc(value);
53 }
54 return ExprStmtDoc(TIR(d)->Attr("evaluate")->Call({value}));
55 });

57TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
58 .set_dispatch<tir::LetStmt>("", [](tir::LetStmt stmt, ObjectPath p, IRDocsifier d) -> Doc {
59 bool concise = AllowConciseScoping(d);
60 ExprDoc rhs = d->AsDoc<ExprDoc>(stmt->value, p->Attr("value"));
61 With<TIRFrame> f(d, stmt);
62 ExprDoc lhs = d->IsVarDefined(stmt->var) ? d->GetVarDoc(stmt->var).value()
63 : DefineVar(stmt->var, *f, d);
64 AsDocBody(stmt->body, p->Attr("body"), f->get(), d);
65 Array<StmtDoc>* stmts = &(*f)->stmts;
66 if (concise) {
67 Type type = stmt->var->type_annotation;
68 Optional<ExprDoc> type_doc =
69 d->AsDoc<ExprDoc>(type, p->Attr("var")->Attr("type_annotation"));
70 if (const auto* tuple_type = type.as<TupleTypeNode>()) {
71 if (tuple_type->fields.empty()) {
72 type_doc = NullOpt;
73 }
74 }
75 stmts->insert(stmts->begin(), AssignDoc(lhs, rhs, type_doc));
76 return StmtBlockDoc(*stmts);
77 } else {
78 rhs = TIR(d)->Attr("let")->Call({lhs, rhs});
79 return ScopeDoc(NullOpt, rhs, *stmts);
80 }
81 });

83TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
84 .set_dispatch<tir::AssertStmt>(
85 "", [](tir::AssertStmt stmt, ObjectPath p, IRDocsifier d) -> Doc {
86 bool concise = AllowConciseScoping(d);
87 ExprDoc cond = d->AsDoc<ExprDoc>(stmt->condition, p->Attr("condition"));
88 ExprDoc msg = d->AsDoc<ExprDoc>(stmt->message, p->Attr("message"));
89 With<TIRFrame> f(d, stmt);
90 AsDocBody(stmt->body, p->Attr("body"), f->get(), d);
91 if (concise) {
92 Array<StmtDoc>* stmts = &(*f)->stmts;
93 stmts->insert(stmts->begin(), AssertDoc(cond, msg));
94 return StmtBlockDoc(*stmts);
95 }
96 return ScopeDoc(NullOpt, TIR(d)->Attr("Assert")->Call({cond, msg}), (*f)->stmts);
97 });

109 "", [](tir::DeclBuffer stmt, ObjectPath p, IRDocsifier d) -> Doc {
110 bool concise = AllowConciseScoping(d);
111 ExprDoc rhs =
112 BufferDecl(stmt->buffer, "decl_buffer", {}, p->Attr("buffer"), d->frames.back(), d);

116 return DoConciseScoping(lhs, rhs, &(*f)->stmts, concise);

119TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
120 .set_dispatch<tir::IfThenElse>( //
121 "", [](tir::IfThenElse stmt, ObjectPath p, IRDocsifier d) -> Doc {
122 ExprDoc cond = d->AsDoc<ExprDoc>(stmt->condition, p->Attr("condition"));
123 Array<StmtDoc> then_branch;
124 Array<StmtDoc> else_branch;
125 if (stmt->then_case.defined()) {
126 With<TIRFrame> f(d, stmt->then_case);
127 AsDocBody(stmt->then_case, p->Attr("then_case"), f->get(), d);
128 then_branch = (*f)->stmts;
129 }
130 if (stmt->else_case.defined()) {
131 With<TIRFrame> f(d, stmt->else_case);
132 AsDocBody(stmt->else_case.value(), p->Attr("else_case"), f->get(), d);
133 else_branch = (*f)->stmts;
134 }
135 return IfDoc(cond, then_branch, else_branch);
136 });

157TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
158 .set_dispatch<tir::Allocate>( //
159 "", [](tir::Allocate stmt, ObjectPath p, IRDocsifier d) -> Doc {
160 bool concise = AllowConciseScoping(d);
161 String storage_scope = tir::GetPtrStorageScope(stmt->buffer_var);
162 Array<ExprDoc> args;
163 Array<String> kwargs_keys;
164 Array<ExprDoc> kwargs_values;
165 args.push_back(d->AsDoc<ExprDoc>(stmt->extents, p->Attr("extents")));
166 args.push_back(LiteralDoc::DataType(stmt->dtype));
167 args.push_back(LiteralDoc::Str(storage_scope));
168 if (!tir::is_one(stmt->condition)) {
169 args.push_back(d->AsDoc<ExprDoc>(stmt->condition, p->Attr("condition")));
170 }
171 if (!stmt->annotations.empty()) {
172 kwargs_keys.push_back("annotations");
173 kwargs_values.push_back(d->AsDoc<ExprDoc>(stmt->annotations, p->Attr("annotations")));
174 }
175 ExprDoc lhs = DefineVar(stmt->buffer_var, d->frames.back(), d);
176 With<TIRFrame> f(d, stmt);
177 ExprDoc rhs = TIR(d)->Attr("allocate")->Call(args, kwargs_keys, kwargs_values);
178 AsDocBody(stmt->body, p->Attr("body"), f->get(), d);
179 return DoConciseScoping(lhs, rhs, &(*f)->stmts, concise);
180 });

182template <typename T>
183ExprDoc PrintNDArray(::tvm::runtime::NDArray arr) {
184 // FIXME(@junrushao): this is a hack and can be wrong in most of the cases
185 constexpr int NUM_PRINT = 200;
186 int ndim = arr->ndim;
187 int tot_dim = 1;
188 for (int i = 0; i < ndim; i++) {
189 tot_dim *= arr->shape[i];
190 }
191 Array<ExprDoc> result;
192 T* data_ptr = reinterpret_cast<T*>(arr->data);
193 runtime::DataType dtype = arr.DataType();
194 for (int i = 0; i < tot_dim; i++) {
195 if (dtype.is_float()) {
196 result.push_back(LiteralDoc::Float(data_ptr[i]));
197 } else {
198 result.push_back(LiteralDoc::Int(data_ptr[i]));
199 }
200 if (i == NUM_PRINT) {
201 break;
202 }
203 }
204 return ListDoc(result);
205}

207TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
208 .set_dispatch<tir::AllocateConst>(
209 "", [](tir::AllocateConst stmt, ObjectPath p, IRDocsifier d) -> Doc {
210 bool concise = AllowConciseScoping(d);
211 String storage_scope = tir::GetPtrStorageScope(stmt->buffer_var);
212 Array<ExprDoc> args;
213 Array<String> kwargs_keys;
214 Array<ExprDoc> kwargs_values;
215 ExprDoc data_doc{nullptr};
216 if (stmt->dtype.is_int()) {
217 if (stmt->dtype.bits() == 8) {
218 data_doc = PrintNDArray<int8_t>(stmt->data.value());
219 } else if (stmt->dtype.bits() == 16) {
220 data_doc = PrintNDArray<int16_t>(stmt->data.value());
221 } else if (stmt->dtype.bits() == 32) {
222 data_doc = PrintNDArray<int32_t>(stmt->data.value());
223 } else if (stmt->dtype.bits() == 64) {
224 data_doc = PrintNDArray<int64_t>(stmt->data.value());
225 } else {
226 LOG(FATAL) << "DataType not supported";
227 }
228 } else if (stmt->dtype.is_uint()) {
229 if (stmt->dtype.bits() == 8) {
230 data_doc = PrintNDArray<uint8_t>(stmt->data.value());
231 } else if (stmt->dtype.bits() == 16) {
232 data_doc = PrintNDArray<uint16_t>(stmt->data.value());
233 } else if (stmt->dtype.bits() == 32) {
234 data_doc = PrintNDArray<uint32_t>(stmt->data.value());
235 } else if (stmt->dtype.bits() == 64) {
236 data_doc = PrintNDArray<uint64_t>(stmt->data.value());
237 } else {
238 LOG(FATAL) << "DataType not supported";
239 }
240 } else if (stmt->dtype.is_float()) {
241 if (stmt->dtype.bits() == 16) {
242 data_doc = PrintNDArray<int16_t>(stmt->data.value());
243 } else if (stmt->dtype.bits() == 32) {
244 data_doc = PrintNDArray<float>(stmt->data.value());
245 } else if (stmt->dtype.bits() == 64) {
246 data_doc = PrintNDArray<double>(stmt->data.value());
247 } else {
248 LOG(FATAL) << "DataType not supported";
249 }
250 } else {
251 LOG(FATAL) << "DataType not supported";
252 }
253 args.push_back(data_doc);
254 args.push_back(LiteralDoc::DataType(stmt->dtype));
255 args.push_back(d->AsDoc<ExprDoc>(stmt->extents, p->Attr("extents")));
256 ExprDoc rhs = TIR(d)->Attr("allocate_const")->Call(args, kwargs_keys, kwargs_values);
257 With<TIRFrame> f(d, stmt);
258 ExprDoc lhs = DefineVar(stmt->buffer_var, *f, d);
259 AsDocBody(stmt->body, p->Attr("body"), f->get(), d);
260 return DoConciseScoping(lhs, rhs, &(*f)->stmts, concise);
261 });

263ExprDoc DocsifyBufferRealize(const tir::BufferRealizeNode* stmt, Optional<ExprDoc> value, //
264 ObjectPath p, IRDocsifier d) {
265 ExprDoc buffer = d->AsDoc<ExprDoc>(stmt->buffer, p->Attr("buffer"));
266 {
267 Array<Doc> bounds;
268 bounds.reserve(stmt->bounds.size());
269 for (int i = 0, n = stmt->bounds.size(); i < n; ++i) {
270 Range range = stmt->bounds[i];
271 ObjectPath range_p = p->Attr("bounds")->ArrayIndex(i);
272 bounds.push_back(
273 SliceDoc(d->AsDoc<ExprDoc>(range->min, range_p->Attr("min")),
274 d->AsDoc<ExprDoc>(range->min + range->extent, range_p->Attr("extent")), //
275 NullOpt));
276 }
277 buffer = buffer[bounds];
278 }
279 Array<ExprDoc> args{buffer};
280 Array<String> kwargs_keys;
281 Array<ExprDoc> kwargs_values;
282 if (value.defined()) {
283 args.push_back(value.value());
284 }
285 if (!tir::is_one(stmt->condition)) {
286 kwargs_keys.push_back("condition");
287 kwargs_values.push_back(d->AsDoc<ExprDoc>(stmt->condition, p->Attr("condition")));
288 }
289 return TIR(d)->Attr("realize")->Call(args, kwargs_keys, kwargs_values);
290}

302TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
303 .set_dispatch<tir::AttrStmt>( //
304 "", [](tir::AttrStmt stmt, ObjectPath p, IRDocsifier d) -> Doc {
305 bool concise = AllowConciseScoping(d);
306 Optional<ExprDoc> rhs = NullOpt;
307 tir::Stmt body = stmt->body;
308 ObjectPath body_p = p->Attr("body");
309 if (stmt->attr_key == "realize_scope") {
310 if (const auto* realize = stmt->body.as<tir::BufferRealizeNode>()) {
311 if (realize->buffer.same_as(stmt->node)) {
312 rhs =
313 DocsifyBufferRealize(realize,
314 /*value=*/d->AsDoc<ExprDoc>(stmt->value, p->Attr("value")),
315 /*p=*/p->Attr("body"), d);
316 body = realize->body;
317 body_p = body_p->Attr("body");
318 }
319 }
320 }
321 if (stmt->attr_key == "thread_extent" || stmt->attr_key == "virtual_thread") {
322 if (const auto* iter_var = stmt->node.as<tir::IterVarNode>()) {
323 if (!d->IsVarDefined(iter_var->var)) {
324 // `DefineVar` is not used here because a more specific name is desirable
325 Frame f = FindLowestVarDef(iter_var->var, d).value();
326 DefineVar(iter_var->var, f, d);
327 f->stmts.push_back(
328 AssignDoc(d->AsDoc<ExprDoc>(iter_var->var, p->Attr("node")->Attr("var")),
329 TIR(d) //
330 ->Attr("env_thread")
331 ->Call({LiteralDoc::Str(iter_var->thread_tag)}), //
332 NullOpt));
333 }
334 rhs = TIR(d)
335 ->Attr("launch_thread")
336 ->Call({
337 d->AsDoc<ExprDoc>(iter_var->var, p->Attr("node")),
338 d->AsDoc<ExprDoc>(stmt->value, p->Attr("value")),
339 });
340 }
341 }
342 if (!rhs.defined()) {
343 rhs = TIR(d)->Attr("attr")->Call({
344 d->AsDoc<ExprDoc>(stmt->node, p->Attr("node")),
345 LiteralDoc::Str(stmt->attr_key),
346 d->AsDoc<ExprDoc>(stmt->value, p->Attr("value")),
347 });
348 }
349 With<TIRFrame> f(d, stmt);
350 AsDocBody(body, body_p, f->get(), d);
351 return DoConciseScoping(NullOpt, rhs.value(), &(*f)->stmts, concise);
352 });
