27Map<String, ExprDoc> BufferAttrs(const tir::Buffer& buffer, const ObjectPath& p, const Frame& frame,
28 const IRDocsifier& d) {
29 Map<String, ExprDoc> kwargs;
30 auto implicit_var_def = [&](const PrimExpr& e, const ObjectPath& p, const String& key) {
31 if (Optional<ExprDoc> doc = d->GetVarDoc(e)) {
32 kwargs.Set(key, doc.value());
33 return false;
34 }
35 if (e->IsInstance<tir::VarNode>()) {
36 d->Define(e, frame, [=]() { return d->AsDoc<IdDoc>(buffer, p)->Attr(key); });
37 return true;
38 }
39 kwargs.Set(key, d->AsDoc<ExprDoc>(e, p));
40 return false;
41 };
42 auto array_out_line_var_def = [&](const Array<PrimExpr>& array, const ObjectPath& p,
43 const String& key) {
44 int n = array.size();
45 Array<ExprDoc> results;
46 results.reserve(n);
47 for (int i = 0; i < n; ++i) {
48 PrimExpr s = array[i];
49 ObjectPath s_path = p->ArrayIndex(i);
50 // Add out-of-line definition for a new Var in shape
51 results.push_back(d->AsDoc<ExprDoc>(s, s_path));
52 }
53 kwargs.Set(key, TupleDoc(results));
54 };
55 // Step 1. Handle `buffer.shape`
56 array_out_line_var_def(buffer->shape, p->Attr("shape"), "shape");
57 // Step 2. Handle `buffer.dtype`
58 if (buffer->dtype != Default::BufferDType()) {
59 kwargs.Set("dtype", LiteralDoc::DataType(buffer->dtype));
60 }
61 // Step 3. Handle `buffer.data`
62 implicit_var_def(buffer->data, p->Attr("data"), "data");
63 // Step 4. Handle `buffer.strides`
64 if (!buffer->strides.empty()) {
65 array_out_line_var_def(buffer->strides, p->Attr("strides"), "strides");
66 }
67 // Step 5. Handle `buffer.elem_offset`
68 bool needs_print_factor = false;
69 if (const auto* int_imm = buffer->elem_offset.as<IntImmNode>()) {
70 if (int_imm->value != 0) {
71 kwargs.Set("elem_offset", d->AsDoc<ExprDoc>(buffer->elem_offset, p->Attr("elem_offset")));
72 }
73 } else {
74 needs_print_factor =
75 implicit_var_def(buffer->elem_offset, p->Attr("elem_offset"), "elem_offset");
76 }
77 // Step 6. Handle `buffer.scope`
78 {
79 String scope = buffer.scope();
80 if (scope != "global") {
81 kwargs.Set("scope", LiteralDoc::Str(scope));
82 }
83 }
84 // Step 7. Handle `buffer.data_alignment`
85 if (buffer->data_alignment != runtime::kAllocAlignment) {
86 kwargs.Set("align", LiteralDoc::Int(buffer->data_alignment));
87 }
88 // Step 8. Handle `buffer.offset_factor`
89 if (needs_print_factor || buffer->offset_factor != 1) {
90 kwargs.Set("offset_factor", LiteralDoc::Int(buffer->offset_factor));
91 }
92 // Step 9. Handle `buffer.buffer_type`
93 if (buffer->buffer_type != tir::BufferType::kDefault) {
94 kwargs.Set("type", LiteralDoc::Str("auto"));
95 }
96 // Step 10. Handle `buffer.axis_separator`
97 if (!buffer->axis_separators.empty()) {
98 kwargs.Set("axis_separators",
99 d->AsDoc<ExprDoc>(buffer->axis_separators, p->Attr("axis_separators")));
100 }
101 return kwargs;
102}

104ExprDoc BufferCall(const ExprDoc& prefix, const Map<String, ExprDoc>& attrs, Array<ExprDoc> args) {
105 Array<String> kwargs_keys;
106 Array<ExprDoc> kwargs_values;
107 for (String s : {"shape", "dtype"}) {
108 if (Optional<ExprDoc> doc = attrs.Get(s)) {
109 args.push_back(doc.value());
110 }
111 }
112 for (String s : {"data", "strides", "elem_offset", "scope", "align", "offset_factor", "type",
113 "axis_separators"}) {
114 if (Optional<ExprDoc> doc = attrs.Get(s)) {
115 kwargs_keys.push_back(s);
116 kwargs_values.push_back(doc.value());
117 }
118 }
119 return prefix->Call(args, kwargs_keys, kwargs_values);
120}

122ExprDoc BufferDecl(const tir::Buffer& buffer, const String& method, const Array<ExprDoc>& args,
123 const ObjectPath& p, const Frame& frame, const IRDocsifier& d) {
124 return BufferCall(/*prefix=*/TIR(d)->Attr(method),
125 /*attrs=*/BufferAttrs(buffer, p, frame, d),
126 /*args=*/args);
127}

129Doc BufferIndex(const PrimExpr& index, const ObjectPath& p, const IRDocsifier& d) {
130 if (const auto* ramp = index.as<tir::RampNode>()) {
131 if (const auto* stride = ramp->stride.as<IntImmNode>()) {
132 ExprDoc start = d->AsDoc<ExprDoc>(ramp->base, p->Attr("base"));
133 ExprDoc stop = d->AsDoc<ExprDoc>(ramp->base + ramp->lanes * ramp->stride, p->Attr("lanes"));
134 Optional<ExprDoc> step = NullOpt;
135 if (stride->value != 1) {
136 step = d->AsDoc<ExprDoc>(ramp->stride, p->Attr("stride"));
137 }
138 return SliceDoc(start, stop, step);
139 }
140 }
141 return d->AsDoc<ExprDoc>(index, p);
142}

144ExprDoc BufferIndices(const tir::Buffer& buffer, const Array<PrimExpr>& indices,
145 const ObjectPath& p, const IRDocsifier& d) {
146 int n = indices.size();
147 Array<Doc> indices_doc;
148 indices_doc.reserve(n);
149 for (int i = 0; i < n; ++i) {
150 indices_doc.push_back(BufferIndex(indices[i], p->Attr("indices")->ArrayIndex(i), d));
151 }
152 return d->AsDoc<ExprDoc>(buffer, p->Attr("buffer"))[indices_doc];
153}

155TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
156 .set_dispatch<tir::BufferRegion>(
157 "", [](tir::BufferRegion buffer_region, ObjectPath p, IRDocsifier d) -> Doc {
158 ExprDoc prefix = d->AsDoc<ExprDoc>(buffer_region->buffer, p->Attr("buffer"));
159 p = p->Attr("region");
160 Array<Range> region = buffer_region->region;
161 int n = region.size();
162 Array<Doc> indices;
163 indices.reserve(n);
164 for (int i = 0; i < n; ++i) {
165 Range range = region[i];
166 ExprDoc min = d->AsDoc<ExprDoc>(range->min, p->ArrayIndex(i)->Attr("min"));
167 if (tir::is_one(range->extent)) {
168 indices.push_back(min);
169 } else {
170 ExprDoc max =
171 d->AsDoc<ExprDoc>(range->min + range->extent, p->ArrayIndex(i)->Attr("extent"));
172 indices.push_back(SliceDoc(min, max, NullOpt));
173 }
174 }
175 return prefix[indices];
176 });

179 .set_dispatch<tir::BufferStore>( //
180 "", [](tir::BufferStore store, ObjectPath p, IRDocsifier d) -> Doc {
181 return AssignDoc(/*lhs=*/BufferIndices(store->buffer, store->indices, p, d),
182 /*rhs=*/d->AsDoc<ExprDoc>(store->value, p->Attr("value")), NullOpt);
183 });
