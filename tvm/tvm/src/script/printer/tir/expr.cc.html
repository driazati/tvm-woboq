27Doc PrintVar(const tir::Var& var, const ObjectPath& p, const IRDocsifier& d) {
28 if (!d->IsVarDefined(var)) {
29 if (Optional<Frame> opt_f = FindLowestVarDef(var, d)) {
30 ExprDoc lhs = DefineVar(var, opt_f.value(), d);
31 Type type = var->type_annotation;
32 if (const auto* ptr_type = type.as<PointerTypeNode>()) {
33 ICHECK(ptr_type->element_type->IsInstance<PrimTypeNode>());
34 ExprDoc rhs = d->AsDoc<ExprDoc>(type, p->Attr("type_annotation"));
35 opt_f.value()->stmts.push_back(AssignDoc(lhs, rhs, NullOpt));
36 } else {
37 ExprDoc rhs = TIR(d)->Attr("var")->Call({LiteralDoc::DataType(var->dtype)});
38 opt_f.value()->stmts.push_back(AssignDoc(lhs, rhs, NullOpt));
39 }
40 }
41 }
42 if (Optional<ExprDoc> doc = d->GetVarDoc(var)) {
43 return doc.value();
44 }
45 LOG(FATAL) << "IndexError: Variable is not defined in the environment: " << var;
46}

144TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
145 .set_dispatch<tir::CommReducer>( //
146 "", [](tir::CommReducer r, ObjectPath p, IRDocsifier d) -> Doc {
147 ICHECK_EQ(r->lhs.size(), r->rhs.size());
148 LambdaDoc lambda{nullptr};
149 {
150 With<TIRFrame> f(d, r);
151 int n_vars = r->lhs.size();
152 Array<IdDoc> vars;
153 vars.reserve(n_vars + n_vars);
154 for (int i = 0; i < n_vars; ++i) {
155 vars.push_back(DefineVar(r->lhs[i], *f, d));
156 }
157 for (int i = 0; i < n_vars; ++i) {
158 vars.push_back(DefineVar(r->rhs[i], *f, d));
159 }
160 int n_results = r->result.size();
161 Array<ExprDoc> results;
162 results.reserve(n_results);
163 for (int i = 0; i < n_results; ++i) {
164 results.push_back(d->AsDoc<ExprDoc>(r->result[i], p->Attr("result")->ArrayIndex(i)));
165 }
166 if (results.size() == 1) {
167 lambda = LambdaDoc(vars, results[0]);
168 } else {
169 lambda = LambdaDoc(vars, TupleDoc(results));
170 }
171 }
172 ExprDoc id = d->AsDoc<ExprDoc>(r->identity_element, p->Attr("identity_element"));
173 return TIR(d)->Attr("comm_reducer")->Call({lambda, id});
174 });

185TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
186 .set_dispatch<tir::Call>("", [](tir::Call call, ObjectPath p, IRDocsifier d) -> Doc {
187 static const OpAttrMap<tir::TScriptPrinterName>& op_names =
188 Op::GetAttrMap<tir::TScriptPrinterName>("TScriptPrinterName");
189 static const std::unordered_set<const Object*> dtype_first_arg = {
190 tir::builtin::reinterpret().get(),
191 tir::builtin::call_extern().get(),
192 tir::builtin::call_llvm_intrin().get(), //
193 tir::builtin::call_llvm_pure_intrin().get(), //
194 tir::builtin::call_pure_extern().get(), //
195 tir::builtin::ptx_mma().get(),
196 tir::builtin::ptx_mma_sp().get(),
197 tir::builtin::ptx_ldmatrix().get(),
198 tir::builtin::ptx_cp_async().get(),
199 tir::builtin::mma_store().get(),
200 tir::builtin::mma_fill().get(),
201 tir::builtin::vectorlow().get(),
202 tir::builtin::vectorhigh().get(),
203 tir::builtin::vectorcombine().get(),
204 Op::Get("tir.type_annotation").get(),
205 };
206 static const std::unordered_set<const Object*> dtype_last_arg = {
207 tir::builtin::tvm_struct_get().get(),
208 };
209 ExprDoc prefix{nullptr};
210 if (const auto* op = call->op.as<OpNode>()) {
211 String name = op_names[GetRef<Op>(op)];
212 prefix = TIR(d)->Attr(name);
213 } else if (const auto* gv = call->op.as<GlobalVarNode>()) {
214 prefix = LiteralDoc::Str(gv->name_hint);
215 } else {
216 LOG(FATAL) << "call: " << call;
217 }
218 Array<ExprDoc> args;
219 int n_args = call->args.size();
220 args.reserve(n_args + 1);
221 if (dtype_first_arg.count(call->op.get())) {
222 args.push_back(LiteralDoc::DataType(call->dtype));
223 }
224 for (int i = 0; i < n_args; ++i) {
225 args.push_back(d->AsDoc<ExprDoc>(call->args[i], p->Attr("args")->ArrayIndex(i)));
226 }
227 if (dtype_last_arg.count(call->op.get())) {
228 args.push_back(LiteralDoc::DataType(call->dtype));
229 }
230 return prefix->Call(args);
231 });

254#define TVM_SCRIPT_PRINTER_DEF_BINARY(NodeType, OpString) \
255 TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable) \
256 .set_dispatch<tir::NodeType>("", \
257 [](tir::NodeType node, ObjectPath p, IRDocsifier d) -> Doc { \
258 ExprDoc a = d->AsDoc<ExprDoc>(node->a, p->Attr("a")); \
259 ExprDoc b = d->AsDoc<ExprDoc>(node->b, p->Attr("b")); \
260 return TIR(d)->Attr(OpString)->Call({a, b}); \
261 });
262
263#define TVM_SCRIPT_PRINTER_DEF_BINARY_WITH_SUGAR(NodeType, OpString, OpKind) \
264 TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable) \
265 .set_dispatch<tir::NodeType>( \
266 "", [](tir::NodeType node, ObjectPath p, IRDocsifier d) -> Doc { \
267 ExprDoc a = d->AsDoc<ExprDoc>(node->a, p->Attr("a")); \
268 ExprDoc b = d->AsDoc<ExprDoc>(node->b, p->Attr("b")); \
269 if (a->IsInstance<LiteralDocNode>() && b->IsInstance<LiteralDocNode>()) { \
270 return TIR(d)->Attr(OpString)->Call({a, b}); \
271 } \
272 return OperationDoc(OperationDocNode::Kind::OpKind, {a, b}); \
273 });
