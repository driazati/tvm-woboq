23namespace tvm {
24namespace script {
25namespace printer {

27TVM_REGISTER_NODE_TYPE(TIRFrameNode);
28
29TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
30 .set_dispatch<IntImm>("", [](IntImm imm, ObjectPath p, IRDocsifier d) -> Doc {
31 DataType dtype = imm->dtype;
32 if (dtype == Default::IntDType()) {
33 return LiteralDoc::Int(imm->value);
34 } else if (dtype == DataType::Bool()) {
35 return LiteralDoc::Boolean(imm->value);
36 } else {
37 return TIR(d) //
38 ->Attr(runtime::DLDataType2String(dtype))
39 ->Call({LiteralDoc::Int(imm->value)});
40 }
41 });
42
43TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
44 .set_dispatch<FloatImm>("", [](FloatImm imm, ObjectPath p, IRDocsifier d) -> Doc {
45 DataType dtype = imm->dtype;
46 if (dtype == Default::FloatDType()) {
47 return LiteralDoc::Float(imm->value);
48 } else {
49 return TIR(d)
50 ->Attr(runtime::DLDataType2String(dtype))
51 ->Call({LiteralDoc::Float(imm->value)});
52 }
53 });
54
55TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
56 .set_dispatch<Range>("", [](Range range, ObjectPath p, IRDocsifier d) -> Doc {
57 return TIR(d)->Attr("Range")->Call({
58 d->AsDoc<ExprDoc>(range->min, p->Attr("min")),
59 d->AsDoc<ExprDoc>(range->extent, p->Attr("extent")),
60 });
61 });
62
63TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
64 .set_dispatch<PrimType>("", [](PrimType ty, ObjectPath p, IRDocsifier d) -> Doc {
65 std::string dtype = ty->dtype.is_void() ? "void" : runtime::DLDataType2String(ty->dtype);
66 return TIR(d)->Attr(dtype);
67 });
68
69TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
70 .set_dispatch<PointerType>("", [](PointerType ty, ObjectPath p, IRDocsifier d) -> Doc {
71 ExprDoc element_type = d->AsDoc<ExprDoc>(ty->element_type, p->Attr("element_type"));
72 if (ty->storage_scope == "") {
73 return TIR(d)->Attr("Ptr")->Call({element_type});
74 } else {
75 return TIR(d)->Attr("Ptr")->Call({element_type, LiteralDoc::Str(ty->storage_scope)});
76 }
77 });
78
79TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
80 .set_dispatch<TupleType>("", [](TupleType ty, ObjectPath p, IRDocsifier d) -> Doc {
81 if (ty->fields.empty()) {
82 return LiteralDoc::None();
83 }
84 return TIR(d) //
85 ->Attr("Tuple")
86 ->Call(d->AsDoc<ListDoc>(ty->fields, p->Attr("fields"))->elements);
87 });
88
89TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
90 .set_dispatch<Target>("", [](Target target, ObjectPath p, IRDocsifier d) -> Doc {
91 Map<String, ObjectRef> config = target->Export();
92 return TIR(d)->Attr("target")->Call({d->AsDoc<ExprDoc>(config, p)});
93 });
