39/*! \brief A printer frame for TIR fragment */
40class TIRFrameNode : public FrameNode {
41 public:
42 /*! \brief The TIR fragment the frame corresponds to */
43 ObjectRef tir;
44 /*! \brief Whether or not the frame allows concise scoping */
45 bool allow_concise_scoping{false};
46
47 void VisitAttrs(AttrVisitor* v) {
48 FrameNode::VisitAttrs(v);
49 v->Visit("tir", &tir);
50 v->Visit("allow_concise_scoping", &allow_concise_scoping);
51 }
52
53 static constexpr const char* _type_key = "script.printer.TIRFrame";
54 TVM_DECLARE_FINAL_OBJECT_INFO(TIRFrameNode, FrameNode);
55};
56
57/*! \brief Managed reference to TIRFrameNode */
58class TIRFrame : public Frame {
59 public:
60 /*! \brief Constructor */
61 explicit TIRFrame(const IRDocsifier& d, const ObjectRef& tir) {
62 ObjectPtr<TIRFrameNode> n = make_object<TIRFrameNode>();
63 n->stmts.clear();
64 n->d = d.get();
65 n->tir = tir;
66 data_ = std::move(n);
67 }
68
69 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(TIRFrame, Frame, TIRFrameNode);
70};

77/*!
78 * \brief Defines a variable in the IRDocsifier at the given frame,
79 * and returns the corresponding IdDoc
80 * \param var The variable to define
81 * \param d The IRDocsifier
82 * \param frame The frame to define the variable in
83 * \return The IdDoc corresponding to the variable
84 */
85inline IdDoc DefineVar(const tir::Var& var, const Frame& frame, const IRDocsifier& d) {
86 return d->Define(var, frame, var->name_hint.empty() ? "v" : var->name_hint);
87}

89/*!
90 * \brief Defines a buffer in the IRDocsifier at the given frame,
91 * and returns the corresponding IdDoc
92 * \param buffer The buffer to define
93 * \param frame The frame to define the buffer in
94 * \param d The IRDocsifier
95 * \return The IdDoc corresponding to the buffer
96 */
97inline IdDoc DefineBuffer(const tir::Buffer& buffer, const Frame& frame, const IRDocsifier& d) {
98 return d->Define(buffer, frame, buffer->name.empty() ? "buffer" : buffer->name);

101/*!
102 * \brief Recursively process the body statements of a TIR fragment represented by a frame
103 * \param stmt The body statement to process
104 * \param p The object path
105 * \param f The frame
106 * \param d The IRDocsifier
107 */
108inline void AsDocBody(const tir::Stmt& stmt, ObjectPath p, TIRFrameNode* f, const IRDocsifier& d) {
109 if (const auto* seq_stmt = stmt.as<tir::SeqStmtNode>()) {
110 Array<tir::Stmt> body = seq_stmt->seq;
111 p = p->Attr("seq");
112 for (int i = 0, n = body.size(); i < n; ++i) {
113 f->allow_concise_scoping = (i == n - 1);
114 Doc doc = d->AsDoc(body[i], p->ArrayIndex(i));
115 if (const auto* block = doc.as<StmtBlockDocNode>()) {
116 f->stmts.insert(f->stmts.end(), block->stmts.begin(), block->stmts.end());
117 } else {
118 f->stmts.push_back(Downcast<StmtDoc>(doc));
119 }
120 }
121 } else {
122 f->allow_concise_scoping = true;
123 Doc doc = d->AsDoc(stmt, p);
124 if (const auto* block = doc.as<StmtBlockDocNode>()) {
125 f->stmts.insert(f->stmts.end(), block->stmts.begin(), block->stmts.end());
126 } else {
127 f->stmts.push_back(Downcast<StmtDoc>(doc));
128 }
129 }
130}

132/*!
133 * \brief Find the top frame in the stack that could place a var definition
134 * \param var The var to be defined
135 * \param d The IRDocsifier
136 * \return The frame that could place the var definition
137 */
138inline Optional<Frame> FindLowestVarDef(const ObjectRef& var, const IRDocsifier& d) {
139 if (!d->common_prefix.count(var.get())) {
140 return NullOpt;
141 }
142 int n_frames = d->frames.size();
143 std::unordered_map<const Object*, const FrameNode*> tir_to_frame;
144 tir_to_frame.reserve(n_frames);
145 for (int i = n_frames - 1; i >= 0; --i) {
146 if (const auto* f = d->frames[i].as<TIRFrameNode>()) {
147 tir_to_frame[f->tir.get()] = f;
148 }
149 }
150 const std::vector<const Object*>& path = d->common_prefix.at(var.get());
151 for (auto it = path.rbegin(); it != path.rend(); ++it) {
152 if (tir_to_frame.count(*it)) {
153 return GetRef<Frame>(tir_to_frame.at(*it));
154 }
155 }
156 return NullOpt;
157}
