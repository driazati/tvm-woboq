20#include <tvm/script/printer/traced_object_functor.h>

22namespace tvm {
23namespace script {
24namespace printer {
25
26const runtime::PackedFunc* GetDispatchFunctionForToken(const DispatchTable& table,
27 const String& token, uint32_t type_index) {
28 auto it = table.find(token);
29 if (it == table.end()) {
30 return nullptr;
31 }
32 const std::vector<runtime::PackedFunc>& tab = it->second;
33 if (type_index >= tab.size()) {
34 return nullptr;
35 }
36 const PackedFunc* f = &tab[type_index];
37 if (f->defined()) {
38 return f;
39 } else {
40 return nullptr;
41 }
42}
43
44const runtime::PackedFunc& GetDispatchFunction(const DispatchTable& dispatch_table,
45 const String& token, uint32_t type_index) {
46 if (const runtime::PackedFunc* pf =
47 GetDispatchFunctionForToken(dispatch_table, token, type_index)) {
48 return *pf;
49 } else if (const runtime::PackedFunc* pf =
50 GetDispatchFunctionForToken(dispatch_table, kDefaultDispatchToken, type_index)) {
51 // Fallback to function with the default dispatch token
52 return *pf;
53 } else {
54 ICHECK(false) << "ObjectFunctor calls un-registered function on type: "
55 << runtime::Object::TypeIndex2Key(type_index) << " (token: " << token << ")";
56 throw;
57 }
58}
59
60void SetDispatchFunction(DispatchTable* dispatch_table, const String& token, uint32_t type_index,
61 runtime::PackedFunc f) {
62 std::vector<runtime::PackedFunc>* table = &(*dispatch_table)[token];
63 if (table->size() <= type_index) {
64 table->resize(type_index + 1, nullptr);
65 }
66 runtime::PackedFunc& slot = (*table)[type_index];
67 if (slot != nullptr) {
68 ICHECK(false) << "Dispatch for type is already registered: "
69 << runtime::Object::TypeIndex2Key(type_index);
70 }
71 slot = f;
72}
73
74void RemoveDispatchFunction(DispatchTable* dispatch_table, const String& token,
75 uint32_t type_index) {
76 std::vector<runtime::PackedFunc>* table = &(*dispatch_table)[token];
77 if (table->size() <= type_index) {
78 return;
79 }
80 (*table)[type_index] = nullptr;
81}
