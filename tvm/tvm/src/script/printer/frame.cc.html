22namespace tvm {
23namespace script {
24namespace printer {

26MetadataFrame::MetadataFrame() : MetadataFrame(make_object<MetadataFrameNode>()) {}
27
28VarDefFrame::VarDefFrame() : VarDefFrame(make_object<VarDefFrameNode>()) {}
29
30TVM_REGISTER_NODE_TYPE(FrameNode);
31TVM_REGISTER_GLOBAL("script.printer.FrameAddExitCallback")
32 .set_body_typed([](Frame frame, runtime::TypedPackedFunc<void()> callback) {
33 frame->AddExitCallback(callback);
34 });
35TVM_REGISTER_GLOBAL("script.printer.FrameEnterWithScope")
36 .set_body_method<Frame>(&FrameNode::EnterWithScope);
37TVM_REGISTER_GLOBAL("script.printer.FrameExitWithScope")
38 .set_body_method<Frame>(&FrameNode::ExitWithScope);
39
40TVM_REGISTER_NODE_TYPE(MetadataFrameNode);
41TVM_REGISTER_GLOBAL("script.printer.MetadataFrame").set_body_typed([]() {
42 return MetadataFrame();
43});
44
45TVM_REGISTER_NODE_TYPE(VarDefFrameNode);
46TVM_REGISTER_GLOBAL("script.printer.VarDefFrame").set_body_typed([]() { return VarDefFrame(); });
