32namespace tvm {
33namespace script {
34namespace printer {
35
36/*! \brief Range of byte offsets in a string */
37using ByteSpan = std::pair<size_t, size_t>;
38
39/*! \brief Options to customize DocPrinter's output */
40struct DocPrinterOptions {
41 /*! \brief Number of spaces for one level of indentation */
42 int indent_spaces = 4;
43
44 /*! \brief Whether to print the line numbers */
45 bool print_line_numbers = false;
46
47 /*!
48 * \brief Number of context lines to print around the underlined text.
49 *
50 * If set to a non-default value `n`, only print `n` context lines before and after
51 * the underlined pieces of text.
52 */
53 size_t num_context_lines = std::numeric_limits<size_t>::max();
54};
55
56/*!
57 * \brief DocPrinter is responsible for printing Doc tree into text format
58 * \details This is the base class for translating Doc into string.
59 * Each target language needs to have its subclass of DocPrinter
60 * to define the actual logic of printing Doc.
61 *
62 * \sa Doc
63 */
64class DocPrinter {
65 public:
66 /*!
67 * \brief The constructor of DocPrinter
68 *
69 * \param options the option for printer
70 */
71 explicit DocPrinter(const DocPrinterOptions& options);
72 virtual ~DocPrinter() = default;
73
74 /*!
75 * \brief Append a doc into the final content
76 *
77 * \param doc the Doc to be printed
78 *
79 * \sa GetString
80 */
81 void Append(const Doc& doc);
82
83 /*!
84 * \brief Append a doc to the final content
85 *
86 * \param doc Doc to be printed
87 * \param path_to_underline Object path to be underlined
88 *
89 * \sa GetString
90 */
91 void Append(const Doc& doc, Optional<ObjectPath> path_to_underline);
92
93 /*!
94 * \brief Get the printed string of all Doc appended
95 *
96 * The content of each Doc in the returned string will
97 * appear in the same order as they are appended.
98 *
99 * \sa Append
100 */
101 String GetString() const;
102
103 protected:
104 /*!
105 * \brief Get the printed string
106 *
107 * It will dispatch to the PrintTypedDoc method based on
108 * the actual type of Doc.
109 *
110 * \sa PrintTypedDoc
111 */
112 void PrintDoc(const Doc& doc);
113
114 /*!
115 * \brief Virtual method to print a LiteralDoc
116 */
117 virtual void PrintTypedDoc(const LiteralDoc& doc) = 0;
118
119 /*!
120 * \brief Virtual method to print an IdDoc
121 */
122 virtual void PrintTypedDoc(const IdDoc& doc) = 0;
123
124 /*!
125 * \brief Virtual method to print an AttrAccessDoc
126 */
127 virtual void PrintTypedDoc(const AttrAccessDoc& doc) = 0;
128
129 /*!
130 * \brief Virtual method to print an IndexDoc
131 */
132 virtual void PrintTypedDoc(const IndexDoc& doc) = 0;
133
134 /*!
135 * \brief Virtual method to print an OperationDoc
136 */
137 virtual void PrintTypedDoc(const OperationDoc& doc) = 0;
138
139 /*!
140 * \brief Virtual method to print a CallDoc
141 */
142 virtual void PrintTypedDoc(const CallDoc& doc) = 0;
143
144 /*!
145 * \brief Virtual method to print a LambdaDoc
146 */
147 virtual void PrintTypedDoc(const LambdaDoc& doc) = 0;
148
149 /*!
150 * \brief Virtual method to print a ListDoc
151 */
152 virtual void PrintTypedDoc(const ListDoc& doc) = 0;
153
154 /*!
155 * \brief Virtual method to print a TupleDoc
156 */
157 virtual void PrintTypedDoc(const TupleDoc& doc) = 0;
158
159 /*!
160 * \brief Virtual method to print a DictDoc
161 */
162 virtual void PrintTypedDoc(const DictDoc& doc) = 0;
163
164 /*!
165 * \brief Virtual method to print a SliceDoc
166 */
167 virtual void PrintTypedDoc(const SliceDoc& doc) = 0;
168
169 /*!
170 * \brief Virtual method to print a StmtBlockDoc
171 */
172 virtual void PrintTypedDoc(const StmtBlockDoc& doc) = 0;
173
174 /*!
175 * \brief Virtual method to print an AssignDoc
176 */
177 virtual void PrintTypedDoc(const AssignDoc& doc) = 0;
178
179 /*!
180 * \brief Virtual method to print an IfDoc
181 */
182 virtual void PrintTypedDoc(const IfDoc& doc) = 0;
183
184 /*!
185 * \brief Virtual method to print a WhileDoc
186 */
187 virtual void PrintTypedDoc(const WhileDoc& doc) = 0;
188
189 /*!
190 * \brief Virtual method to print a ForDoc
191 */
192 virtual void PrintTypedDoc(const ForDoc& doc) = 0;
193
194 /*!
195 * \brief Virtual method to print a ScopeDoc
196 */
197 virtual void PrintTypedDoc(const ScopeDoc& doc) = 0;
198
199 /*!
200 * \brief Virtual method to print an ExprStmtDoc
201 */
202 virtual void PrintTypedDoc(const ExprStmtDoc& doc) = 0;
203
204 /*!
205 * \brief Virtual method to print an AssertDoc
206 */
207 virtual void PrintTypedDoc(const AssertDoc& doc) = 0;
208
209 /*!
210 * \brief Virtual method to print a ReturnDoc
211 */
212 virtual void PrintTypedDoc(const ReturnDoc& doc) = 0;
213
214 /*!
215 * \brief Virtual method to print a FunctionDoc
216 */
217 virtual void PrintTypedDoc(const FunctionDoc& doc) = 0;
218
219 /*!
220 * \brief Virtual method to print a ClassDoc
221 */
222 virtual void PrintTypedDoc(const ClassDoc& doc) = 0;
223
224 /*!
225 * \brief Increase the indent level of any content to be
226 * printed after this call
227 */
228 void IncreaseIndent() { indent_ += options_.indent_spaces; }
229
230 /*!
231 * \brief Decrease the indent level of any content to be
232 * printed after this call
233 */
234 void DecreaseIndent() { indent_ -= options_.indent_spaces; }
235
236 /*!
237 * \brief Add a new line into the output stream
238 *
239 * \sa output_
240 */
241 std::ostream& NewLine() {
242 output_ << "\n";
243 line_starts_.push_back(output_.tellp());
244 output_ << std::string(indent_, ' ');
245 return output_;
246 }
247
248 /*!
249 * \brief The output stream of printer
250 *
251 * All printed content will be stored in this stream and returned
252 * when GetString is called.
253 *
254 * \sa GetString
255 */
256 std::ostringstream output_;
257
258 private:
259 void MarkSpan(const ByteSpan& span, const ObjectPath& path);
260
261 /*! \brief Options to customize certain aspects of the output */
262 DocPrinterOptions options_;
263
264 /*! \brief the current level of indent */
265 int indent_ = 0;
266
267 /*! \brief For each line in the output_, byte offset of its first character */
268 std::vector<size_t> line_starts_;
269
270 /*! \brief Path of the object that we would like to underline */
271 Optional<ObjectPath> path_to_underline_;
272
273 /*!
274 * \brief Candidate spans to be underlined, until we find a better match.
275 * (A better match is an object with a longer path that is still a prefix of path_to_underline_.)
276 */
277 std::vector<ByteSpan> current_underline_candidates_;
278
279 /*! \brief Path length of the objects that are current candidates for underlining. */
280 int current_max_path_length_;
281
282 /*! \brief Spans that we have already committed to underline. */
283 std::vector<ByteSpan> underlines_;
284};

286} // namespace printer
