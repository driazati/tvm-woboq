21namespace tvm {

23namespace printer {

25TVM_REGISTER_NODE_TYPE(IRFrameNode);
26
27TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
28 .set_dispatch<IRModule>("", [](IRModule mod, ObjectPath p, IRDocsifier d) -> Doc {
29 std::vector<std::pair<GlobalVar, BaseFunc>> functions{mod->functions.begin(),
30 mod->functions.end()};
31 // print "main" first
32 std::sort(functions.begin(), functions.end(), [](const auto& lhs, const auto& rhs) {
33 String lhs_name = lhs.first->name_hint;
34 String rhs_name = rhs.first->name_hint;
35 if (lhs_name == "main") {
36 lhs_name = "";
37 }
38 if (rhs_name == "main") {
39 rhs_name = "";
40 }
41 return lhs_name < rhs_name;
42 });
43 ICHECK(!d->mod.defined());
44 d->mod = mod;
45 {
46 With<IRFrame> f(d);
47 (*f)->AddDispatchToken(d, "ir");
48 for (const auto& kv : functions) {
49 GlobalVar gv = kv.first;
50 BaseFunc func = kv.second;
51 (*f)->stmts.push_back(d->AsDoc<FunctionDoc>(func, p->Attr("functions")->MapValue(gv)));
52 }
53 return ClassDoc(IdDoc("Module"), {IR(d)}, (*f)->stmts);
54 }
55 });
56
57TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
58 .set_dispatch<DictAttrs>("", [](DictAttrs attrs, ObjectPath p, IRDocsifier d) -> Doc {
59 return d->AsDoc(attrs->dict, p->Attr("dict"));
60 });
61
62TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
63 .set_dispatch<GlobalVar>("", [](GlobalVar gv, ObjectPath p, IRDocsifier d) -> Doc {
64 return IdDoc("GlobalVar")->Call({LiteralDoc::Str(gv->name_hint)});
65 });
66
67TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
68 .set_dispatch<Op>("", [](Op op, ObjectPath p, IRDocsifier d) -> Doc {
69 return IdDoc("Op")->Call({LiteralDoc::Str(op->name)});
70 });
