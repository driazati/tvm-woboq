25TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
26 .set_dispatch<String>("", [](String s, ObjectPath p, IRDocsifier d) -> Doc {
27 return LiteralDoc::Str(s);
28 });
29
30TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
31 .set_dispatch<Array<ObjectRef>>( //
32 "", [](Array<ObjectRef> array, ObjectPath p, IRDocsifier d) -> Doc {
33 int n = array.size();
34 Array<ExprDoc> results;
35 results.reserve(n);
36 for (int i = 0; i < n; ++i) {
37 results.push_back(d->AsDoc<ExprDoc>(array[i], p->ArrayIndex(i)));
38 }
39 return ListDoc(results);
40 });
41
42TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
43 .set_dispatch<Map<ObjectRef, ObjectRef>>( //
44 "", [](Map<ObjectRef, ObjectRef> dict, ObjectPath p, IRDocsifier d) -> Doc {
45 using POO = std::pair<ObjectRef, ObjectRef>;
46 std::vector<POO> items{dict.begin(), dict.end()};
47 bool is_str_map = true;
48 for (const auto& kv : items) {
49 if (!kv.first.as<runtime::StringObj>()) {
50 is_str_map = false;
51 break;
52 }
53 }
54 if (is_str_map) {
55 std::sort(items.begin(), items.end(), [](const POO& lhs, const POO& rhs) {
56 return Downcast<String>(lhs.first) < Downcast<String>(rhs.first);
57 });
58 } else {
59 std::sort(items.begin(), items.end(), [](const POO& lhs, const POO& rhs) {
60 return lhs.first.get() < rhs.first.get();
61 });
62 }
63 int n = dict.size();
64 Array<ExprDoc> ks;
65 Array<ExprDoc> vs;
66 ks.reserve(n);
67 vs.reserve(n);
68 for (int i = 0; i < n; ++i) {
69 ks.push_back(d->AsDoc<ExprDoc>(items[i].first, p->MissingMapEntry()));
70 vs.push_back(d->AsDoc<ExprDoc>(items[i].second, p->MapValue(items[i].first)));
71 }
72 return DictDoc(ks, vs);
73 });
