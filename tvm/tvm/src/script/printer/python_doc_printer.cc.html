34/*!
35 * \brief Operator precedence
36 *
37 * This is based on
38 * https://docs.python.org/3/reference/expressions.html#operator-precedence
39 */
40enum class ExprPrecedence : int32_t {
41 /*! \brief Unknown precedence */
42 kUnkown = 0,
43 /*! \brief Lambda Expression */
44 kLambda = 1,
45 /*! \brief Conditional Expression */
46 kIfThenElse = 2,
47 /*! \brief Boolean OR */
48 kBooleanOr = 3,
49 /*! \brief Boolean AND */
50 kBooleanAnd = 4,
51 /*! \brief Boolean NOT */
52 kBooleanNot = 5,
53 /*! \brief Comparisons */
54 kComparison = 6,
55 /*! \brief Bitwise OR */
56 kBitwiseOr = 7,
57 /*! \brief Bitwise XOR */
58 kBitwiseXor = 8,
59 /*! \brief Bitwise AND */
60 kBitwiseAnd = 9,
61 /*! \brief Shift Operators */
62 kShift = 10,
63 /*! \brief Addition and subtraction */
64 kAdd = 11,
65 /*! \brief Multiplication, division, floor division, remainder */
66 kMult = 12,
67 /*! \brief Positive negative and bitwise NOT */
68 kUnary = 13,
69 /*! \brief Exponentiation */
70 kExp = 14,
71 /*! \brief Index access, attribute access, call and atom expression */
72 kIdentity = 15,
73};

75ExprPrecedence GetExprPrecedence(const ExprDoc& doc) {
76 // Key is the value of OperationDocNode::Kind
77 static const std::vector<ExprPrecedence> op_kind_precedence = []() {
78 using OpKind = OperationDocNode::Kind;
79 std::map<OpKind, ExprPrecedence> raw_table = {
80 {OpKind::kUSub, ExprPrecedence::kUnary},
81 {OpKind::kInvert, ExprPrecedence::kUnary},
82 {OpKind::kNot, ExprPrecedence::kBooleanNot},
83 {OpKind::kAdd, ExprPrecedence::kAdd},
84 {OpKind::kSub, ExprPrecedence::kAdd},
85 {OpKind::kMult, ExprPrecedence::kMult},
86 {OpKind::kDiv, ExprPrecedence::kMult},
87 {OpKind::kFloorDiv, ExprPrecedence::kMult},
88 {OpKind::kMod, ExprPrecedence::kMult},
89 {OpKind::kPow, ExprPrecedence::kExp},
90 {OpKind::kLShift, ExprPrecedence::kShift},
91 {OpKind::kRShift, ExprPrecedence::kShift},
92 {OpKind::kBitAnd, ExprPrecedence::kBitwiseAnd},
93 {OpKind::kBitOr, ExprPrecedence::kBitwiseOr},
94 {OpKind::kBitXor, ExprPrecedence::kBitwiseXor},
95 {OpKind::kLt, ExprPrecedence::kComparison},
96 {OpKind::kLtE, ExprPrecedence::kComparison},
97 {OpKind::kEq, ExprPrecedence::kComparison},
98 {OpKind::kNotEq, ExprPrecedence::kComparison},
99 {OpKind::kGt, ExprPrecedence::kComparison},
100 {OpKind::kGtE, ExprPrecedence::kComparison},
101 {OpKind::kAnd, ExprPrecedence::kBooleanAnd},
102 {OpKind::kOr, ExprPrecedence::kBooleanOr},
103 {OpKind::kIfThenElse, ExprPrecedence::kIfThenElse},
104 };
105 int n = static_cast<int>(OpKind::kSpecialEnd);
106 std::vector<ExprPrecedence> table(n + 1, ExprPrecedence::kUnkown);
107 for (const auto& kv : raw_table) {
108 table[static_cast<int>(kv.first)] = kv.second;
109 }
110 return table;
111 }();
112
113 // Key is the type index of Doc
114 static const std::unordered_map<uint32_t, ExprPrecedence> doc_type_precedence = {
115 {LiteralDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
116 {IdDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
117 {AttrAccessDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
118 {IndexDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
119 {CallDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
120 {LambdaDocNode::RuntimeTypeIndex(), ExprPrecedence::kLambda},
121 {TupleDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
122 {ListDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
123 {DictDocNode::RuntimeTypeIndex(), ExprPrecedence::kIdentity},
124 };
125
126 if (const auto* op_doc = doc.as<OperationDocNode>()) {
127 size_t kind = static_cast<int>(op_doc->kind);
128 ICHECK_LT(kind, op_kind_precedence.size()) << "ValueError: Invalid operation: " << kind;
129 ExprPrecedence precedence = op_kind_precedence[kind];
130 ICHECK(precedence != ExprPrecedence::kUnkown)
131 << "Precedence for operator " << static_cast<int>(op_doc->kind) << " is unknown";
132 return precedence;
133 }
134 auto it = doc_type_precedence.find(doc->type_index());
135 if (it != doc_type_precedence.end()) {
136 return it->second;
137 }
138 ICHECK(false) << "Precedence for doc type " << doc->GetTypeKey() << " is unknown";
139 throw;
140}

175 template <typename DocType>
176 void PrintJoinedDocs(const Array<DocType>& docs, const std::string& separator) {
177 bool is_first = true;
178 for (auto& doc : docs) {
179 if (is_first) {
180 is_first = false;
181 } else {
182 output_ << separator;
183 }
184 PrintDoc(doc);
185 }
186 }

188 void PrintIndentedBlock(const Array<StmtDoc>& docs) {
189 IncreaseIndent();
190 for (const StmtDoc& d : docs) {
191 NewLine();
192 PrintDoc(d);
193 }
194 if (docs.empty()) {
195 NewLine();
196 output_ << "pass";
197 }
198 DecreaseIndent();
199 }

201 void PrintDecorators(const Array<ExprDoc>& decorators) {
202 for (const ExprDoc& decorator : decorators) {
203 output_ << "@";
204 PrintDoc(decorator);
205 NewLine();
206 }
207 }

209 /*!
210 * \brief Print expression and add parenthesis if needed.
211 */
212 void PrintChildExpr(const ExprDoc& doc, ExprPrecedence parent_precedence,
213 bool parenthesis_for_same_precedence = false) {
214 ExprPrecedence doc_precedence = GetExprPrecedence(doc);
215 if (doc_precedence < parent_precedence ||
216 (parenthesis_for_same_precedence && doc_precedence == parent_precedence)) {
217 output_ << "(";
218 PrintDoc(doc);
219 output_ << ")";
220 } else {
221 PrintDoc(doc);
222 }
223 }
224
225 /*!
226 * \brief Print expression and add parenthesis if doc has lower precedence than parent.
227 */
228 void PrintChildExpr(const ExprDoc& doc, const ExprDoc& parent,
229 bool parenthesis_for_same_precedence = false) {
230 ExprPrecedence parent_precedence = GetExprPrecedence(parent);
231 return PrintChildExpr(doc, parent_precedence, parenthesis_for_same_precedence);
232 }
233
234 /*!
235 * \brief Print expression and add parenthesis if doc doesn't have higher precedence than parent.
236 *
237 * This function should be used to print an child expression that needs to be wrapped
238 * by parenthesis even if it has the same precedence as its parent, e.g., the `b` in `a + b`
239 * and the `b` and `c` in `a if b else c`.
240 */
241 void PrintChildExprConservatively(const ExprDoc& doc, const ExprDoc& parent) {
242 PrintChildExpr(doc, parent, /*parenthesis_for_same_precedence=*/true);
243 }
244
245 void MaybePrintCommentInline(const StmtDoc& stmt) {
246 if (stmt->comment.defined()) {
247 const std::string& comment = stmt->comment.value();
248 bool has_newline = std::find(comment.begin(), comment.end(), '\n') != comment.end();
249 CHECK(!has_newline) << "ValueError: the comment string of " << stmt->GetTypeKey()
250 << " cannot have newline.";
251 output_ << " # " << comment;
252 }
253 }
254
255 void MaybePrintCommentWithNewLine(const StmtDoc& stmt) {
256 if (stmt->comment.defined()) {
257 std::vector<std::string> comment_lines = support::Split(stmt->comment.value(), '\n');
258 for (const std::string& line : comment_lines) {
259 output_ << "# " << line;
260 NewLine();
261 }
262 }
263 }
264
265 void PrintBlockComment(const String& comment) {
266 IncreaseIndent();
267 NewLine() << "\"\"\"";
268
269 std::vector<std::string> comment_lines = support::Split(comment, '\n');
270 for (const std::string& line : comment_lines) {
271 if (line.empty()) {
272 // No indentation on empty line
273 output_ << "\n";
274 } else {
275 NewLine() << line;
276 }
277 }
278
279 NewLine() << "\"\"\"";
280 DecreaseIndent();
281 }
282};

284void PythonDocPrinter::PrintTypedDoc(const LiteralDoc& doc) {
285 const ObjectRef& value = doc->value;
286 if (!value.defined()) {
287 output_ << "None";
288 } else if (const auto* int_imm = value.as<IntImmNode>()) {
289 if (int_imm->dtype.is_bool()) {
290 output_ << (int_imm->value ? "True" : "False");
291 } else {
292 output_ << int_imm->value;
293 }
294 } else if (const auto* float_imm = value.as<FloatImmNode>()) {
295 // TODO(yelite): Make float number printing roundtrippable
296 output_.precision(17);
297 output_ << float_imm->value;
298 } else if (const auto* string_obj = value.as<StringObj>()) {
299 output_ << "\"" << support::StrEscape(string_obj->data, string_obj->size) << "\"";
300 } else {
301 LOG(FATAL) << "TypeError: Unsupported literal value type: " << value->GetTypeKey();
302 }
303}

305void PythonDocPrinter::PrintTypedDoc(const IdDoc& doc) { output_ << doc->name; }
306
307void PythonDocPrinter::PrintTypedDoc(const AttrAccessDoc& doc) {
308 PrintChildExpr(doc->value, doc);
309 output_ << "." << doc->name;
310}

312void PythonDocPrinter::PrintTypedDoc(const IndexDoc& doc) {
313 PrintChildExpr(doc->value, doc);
314 if (doc->indices.size() == 0) {
315 output_ << "[()]";
316 } else {
317 output_ << "[";
318 PrintJoinedDocs(doc->indices, ", ");
319 output_ << "]";
320 }

323const std::string OperatorToString(OperationDocNode::Kind operation_kind) {
324 static const std::vector<std::string> op_kind2str = []() {
325 using OpKind = OperationDocNode::Kind;
326 std::map<OpKind, std::string> raw_table = {
327 {OpKind::kUSub, "-"}, //
328 {OpKind::kInvert, "~"}, //
329 {OpKind::kNot, "not "}, //
330 {OpKind::kAdd, "+"}, //
331 {OpKind::kSub, "-"}, //
332 {OpKind::kMult, "*"}, //
333 {OpKind::kDiv, "/"}, //
334 {OpKind::kFloorDiv, "//"}, //
335 {OpKind::kMod, "%"}, //
336 {OpKind::kPow, "**"}, //
337 {OpKind::kLShift, "<<"}, //
338 {OpKind::kRShift, ">>"}, //
339 {OpKind::kBitAnd, "&"}, //
340 {OpKind::kBitOr, "|"}, //
341 {OpKind::kBitXor, "^"}, //
342 {OpKind::kLt, "<"}, //
343 {OpKind::kLtE, "<="}, //
344 {OpKind::kEq, "=="}, //
345 {OpKind::kNotEq, "!="}, //
346 {OpKind::kGt, ">"}, //
347 {OpKind::kGtE, ">="}, //
348 {OpKind::kAnd, "and"}, //
349 {OpKind::kOr, "or"}, //
350 };
351
352 std::vector<std::string> table;
353 table.resize(static_cast<int>(OperationDocNode::Kind::kSpecialEnd) + 1);
354
355 for (const auto& kv : raw_table) {
356 table[static_cast<int>(kv.first)] = kv.second;
357 }
358
359 return table;
360 }();
361
362 auto op_index = static_cast<int>(operation_kind);
363 ICHECK_LT(op_index, op_kind2str.size());
364 const std::string str = op_kind2str[op_index];
365 ICHECK(!str.empty()) << "OperationDocNode::Kind " << static_cast<int>(operation_kind)
366 << " cannot be converted to operator token in Python directly.";
367 return str;
368}

370void PythonDocPrinter::PrintTypedDoc(const OperationDoc& doc) {
371 using OpKind = OperationDocNode::Kind;
372 if (doc->kind < OpKind::kUnaryEnd) {
373 // Unary Operators
374 ICHECK_EQ(doc->operands.size(), 1);
375 output_ << OperatorToString(doc->kind);
376 PrintChildExpr(doc->operands[0], doc);
377 } else if (doc->kind == OpKind::kPow) {
378 // Power operator is different than other binary operators
379 // It's right-associative and binds less tightly than unary operator on its right.
380 // https://docs.python.org/3/reference/expressions.html#the-power-operator
381 // https://docs.python.org/3/reference/expressions.html#operator-precedence
382 ICHECK_EQ(doc->operands.size(), 2);
383 PrintChildExprConservatively(doc->operands[0], doc);
384 output_ << " ** ";
385 PrintChildExpr(doc->operands[1], ExprPrecedence::kUnary);
386 } else if (doc->kind < OpKind::kBinaryEnd) {
387 // Binary Operator
388 ICHECK_EQ(doc->operands.size(), 2);
389 PrintChildExpr(doc->operands[0], doc);
390 output_ << " " << OperatorToString(doc->kind) << " ";
391 PrintChildExprConservatively(doc->operands[1], doc);
392 } else if (doc->kind == OpKind::kIfThenElse) {
393 ICHECK_EQ(doc->operands.size(), 3)
394 << "ValueError: IfThenElse requires 3 operands, but got " << doc->operands.size();
395 PrintChildExpr(doc->operands[1], doc);
396 output_ << " if ";
397 PrintChildExprConservatively(doc->operands[0], doc);
398 output_ << " else ";
399 PrintChildExprConservatively(doc->operands[2], doc);
400 } else {
401 LOG(FATAL) << "Unknown OperationDocNode::Kind " << static_cast<int>(doc->kind);
402 throw;
403 }
404}

406void PythonDocPrinter::PrintTypedDoc(const CallDoc& doc) {
407 PrintChildExpr(doc->callee, doc);
408
409 output_ << "(";
410
411 // Print positional args
412 bool is_first = true;
413 for (const ExprDoc& arg : doc->args) {
414 if (is_first) {
415 is_first = false;
416 } else {
417 output_ << ", ";
418 }
419 PrintDoc(arg);
420 }
421
422 // Print keyword args
423 ICHECK_EQ(doc->kwargs_keys.size(), doc->kwargs_values.size())
424 << "CallDoc should have equal number of elements in kwargs_keys and kwargs_values.";
425 for (size_t i = 0; i < doc->kwargs_keys.size(); i++) {
426 if (is_first) {
427 is_first = false;
428 } else {
429 output_ << ", ";
430 }
431 const String& keyword = doc->kwargs_keys[i];
432 output_ << keyword;
433 output_ << "=";
434 PrintDoc(doc->kwargs_values[i]);
435 }
436
437 output_ << ")";
438}

464void PythonDocPrinter::PrintTypedDoc(const DictDoc& doc) {
465 ICHECK_EQ(doc->keys.size(), doc->values.size())
466 << "DictDoc should have equal number of elements in keys and values.";
467 output_ << "{";
468 size_t idx = 0;
469 for (const ExprDoc& key : doc->keys) {
470 if (idx > 0) {
471 output_ << ", ";
472 }
473 PrintDoc(key);
474 output_ << ": ";
475 PrintDoc(doc->values[idx]);
476 idx++;
477 }
478 output_ << "}";
479}

502void PythonDocPrinter::PrintTypedDoc(const AssignDoc& doc) {
503 if (const auto* tuple_doc = doc->lhs.as<TupleDocNode>()) {
504 PrintJoinedDocs(tuple_doc->elements, ", ");
505 } else {
506 PrintDoc(doc->lhs);
507 }
508
509 if (doc->annotation) {
510 output_ << ": ";
511 PrintDoc(doc->annotation.value());
512 }
513 if (doc->rhs) {
514 output_ << " = ";
515 PrintDoc(doc->rhs.value());
516 }
517 MaybePrintCommentInline(doc);
518}

520void PythonDocPrinter::PrintTypedDoc(const IfDoc& doc) {
521 MaybePrintCommentWithNewLine(doc);
522 output_ << "if ";
523 PrintDoc(doc->predicate);
524 output_ << ":";

527
528 if (!doc->else_branch.empty()) {

530 output_ << "else:";

589void PythonDocPrinter::PrintTypedDoc(const FunctionDoc& doc) {
590 for (const AssignDoc& arg_doc : doc->args) {
591 ICHECK(arg_doc->comment == nullptr) << "Function arg cannot have comment attached to them.";
592 }
593
594 PrintDecorators(doc->decorators);
595
596 output_ << "def ";
597 PrintDoc(doc->name);
598
599 output_ << "(";
600 PrintJoinedDocs(doc->args, ", ");
601 output_ << ")";
602
603 if (doc->return_type.defined()) {
604 output_ << " -> ";
605 PrintDoc(doc->return_type.value());
606 }
607
608 output_ << ":";
609
610 if (doc->comment.defined()) {
611 PrintBlockComment(doc->comment.value());
612 }
613 PrintIndentedBlock(doc->body);
614 NewLineWithoutIndent();
615}

617void PythonDocPrinter::PrintTypedDoc(const ClassDoc& doc) {
618 PrintDecorators(doc->decorators);
619
620 output_ << "class ";
621 PrintDoc(doc->name);
622 output_ << ":";
623
624 if (doc->comment.defined()) {
625 PrintBlockComment(doc->comment.value());
626 }
627 PrintIndentedBlock(doc->body);
628 NewLineWithoutIndent();
629}

631String DocToPythonScript(Doc doc, int indent_spaces, bool print_line_numbers, int num_context_lines,
632 Optional<ObjectPath> path_to_underline) {
633 DocPrinterOptions options;
634 options.indent_spaces = indent_spaces;
635 options.print_line_numbers = print_line_numbers;
636 if (num_context_lines >= 0) {
637 options.num_context_lines = num_context_lines;
638 }
639
640 PythonDocPrinter printer(options);
641 printer.Append(doc, path_to_underline);
642 return printer.GetString();
643}
