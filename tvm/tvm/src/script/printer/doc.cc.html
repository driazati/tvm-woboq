28ExprDoc ExprDocNode::Attr(String attr) const { return AttrAccessDoc(GetRef<ExprDoc>(this), attr); }
29
30ExprDoc ExprDocNode::Attr(TracedObject<String> attr) const {
31 auto doc = AttrAccessDoc(GetRef<ExprDoc>(this), attr.Get());
32 doc->source_paths.push_back(attr.GetPath());
33 return std::move(doc);
34}
35
36ExprDoc ExprDocNode::operator[](Array<Doc> indices) const {
37 return IndexDoc(GetRef<ExprDoc>(this), indices);
38}
39
40ExprDoc ExprDocNode::Call(Array<ExprDoc, void> args) const {
41 return CallDoc(GetRef<ExprDoc>(this), args, {}, {});
42}
43
44ExprDoc ExprDocNode::Call(Array<ExprDoc, void> args, Array<String, void> kwargs_keys,
45 Array<ExprDoc, void> kwargs_values) const {
46 return CallDoc(GetRef<ExprDoc>(this), args, kwargs_keys, kwargs_values);
47}

141AssignDoc::AssignDoc(ExprDoc lhs, Optional<ExprDoc> rhs, Optional<ExprDoc> annotation) {
142 CHECK(rhs.defined() || annotation.defined())
143 << "ValueError: At least one of rhs and annotation needs to be non-null for AssignDoc.";
144 CHECK(lhs->IsInstance<IdDocNode>() || annotation == nullptr)
145 << "ValueError: annotation can only be nonnull if lhs is an identifier.";
146
147 ObjectPtr<AssignDocNode> n = make_object<AssignDocNode>();
148 n->lhs = lhs;
149 n->rhs = rhs;
150 n->annotation = annotation;
151 this->data_ = std::move(n);
152}
153
154IfDoc::IfDoc(ExprDoc predicate, Array<StmtDoc> then_branch, Array<StmtDoc> else_branch) {
155 CHECK(!then_branch.empty() || !else_branch.empty())
156 << "ValueError: At least one of the then branch or else branch needs to be non-empty.";
157
158 ObjectPtr<IfDocNode> n = make_object<IfDocNode>();
159 n->predicate = predicate;
160 n->then_branch = then_branch;
161 n->else_branch = else_branch;
162 this->data_ = std::move(n);
163}
