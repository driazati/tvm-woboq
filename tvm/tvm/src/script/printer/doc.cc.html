28ExprDoc ExprDocNode::Attr(String attr) const { return AttrAccessDoc(GetRef<ExprDoc>(this), attr); }

30ExprDoc ExprDocNode::operator[](Array<Doc> indices) const {
31 return IndexDoc(GetRef<ExprDoc>(this), indices);

34ExprDoc ExprDocNode::Call(Array<ExprDoc, void> args) const {
35 return CallDoc(GetRef<ExprDoc>(this), args, Array<String>(), Array<ExprDoc>());
36}

38ExprDoc ExprDocNode::Call(Array<ExprDoc, void> args, Array<String, void> kwargs_keys,
39 Array<ExprDoc, void> kwargs_values) const {
40 return CallDoc(GetRef<ExprDoc>(this), args, kwargs_keys, kwargs_values);
41}

135AssignDoc::AssignDoc(ExprDoc lhs, Optional<ExprDoc> rhs, Optional<ExprDoc> annotation) {
136 CHECK(rhs.defined() || annotation.defined())
137 << "ValueError: At least one of rhs and annotation needs to be non-null for AssignDoc.";
138 CHECK(lhs->IsInstance<IdDocNode>() || annotation == nullptr)
139 << "ValueError: annotation can only be nonnull if lhs is an identifier.";
140
141 ObjectPtr<AssignDocNode> n = make_object<AssignDocNode>();
142 n->lhs = lhs;
143 n->rhs = rhs;
144 n->annotation = annotation;
145 this->data_ = std::move(n);
146}
147
148IfDoc::IfDoc(ExprDoc predicate, Array<StmtDoc> then_branch, Array<StmtDoc> else_branch) {
149 CHECK(!then_branch.empty() || !else_branch.empty())
150 << "ValueError: At least one of the then branch or else branch needs to be non-empty.";
151
152 ObjectPtr<IfDocNode> n = make_object<IfDocNode>();
153 n->predicate = predicate;
154 n->then_branch = then_branch;
155 n->else_branch = else_branch;
156 this->data_ = std::move(n);
157}
