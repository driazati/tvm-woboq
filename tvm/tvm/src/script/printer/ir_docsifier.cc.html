30Doc IRDocsifierNode::AsDocImpl(const TracedObject<ObjectRef>& obj) const {
31 return IRDocsifier::vtable()(dispatch_tokens.back(), obj, GetRef<IRDocsifier>(this));
32}

34IRDocsifier::IRDocsifier(Map<String, String> ir_prefix) {
35 auto n = make_object<IRDocsifierNode>();
36 n->ir_prefix = std::move(ir_prefix);
37 n->dispatch_tokens.push_back(kDefaultDispatchToken);
38 data_ = std::move(n);
39}

41IRDocsifier::FType& IRDocsifier::vtable() {
42 static IRDocsifier::FType inst;
43 return inst;
44}

46RootNodeContainer::RootNodeContainer(ObjectRef root_node) {
47 auto n = make_object<RootNodeContainerNode>();
48 n->root_node = std::move(root_node);

52// Add a default dispatch for the RootNodeContainer to throw error.
53// To add implementation for a new IR, RootNodeContainer needs to be
54// registered under the dispatch token of that IR, like:
55// \code
56// TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
57// .set_dispatch("relax", [](TracedObject<RootNodeContainer> obj, IRDocsifier p) {
58// const ObjectRef& root_node = obj.Get()->root_node;
59// \\ More specialized logic for your IR.
60// return p->AsDoc<Doc>(MakeTraced(root_node));
61// });
62// \endcode
63TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
64 .set_dispatch<RootNodeContainer>([](TracedObject<RootNodeContainer> obj, IRDocsifier p) -> Doc {
65 String top_dispatch_token = p->dispatch_tokens.back();
66 ICHECK_NE(top_dispatch_token, "");
67 ICHECK(false) << "Printing IR " << top_dispatch_token << " is not implemented.";
68 throw;
69 });
70
71TVM_REGISTER_NODE_TYPE(IRDocsifierNode);
72TVM_REGISTER_GLOBAL("script.printer.IRDocsifier").set_body_typed([](Map<String, String> ir_prefix) {
73 return IRDocsifier(ir_prefix);
74});
75TVM_REGISTER_GLOBAL("script.printer.IRDocsifierAsDoc")
76 .set_body_typed([](IRDocsifier p, ObjectRef obj, ObjectPath obj_path) {
77 return p->AsDoc<Doc>(MakeTraced(obj, obj_path));
78 });
79
80TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPushDispatchToken")
81 .set_body_typed([](IRDocsifier p, String token) { p->dispatch_tokens.push_back(token); });
82TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPopDispatchToken").set_body_typed([](IRDocsifier p) {
83 p->dispatch_tokens.pop_back();
84});
85
86TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPushFrame")
87 .set_body_typed([](IRDocsifier p, Frame frame) { p->frames.push_back(frame); });
88TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPopFrame").set_body_typed([](IRDocsifier p) {
89 p->frames.pop_back();
90});
91
92TVM_REGISTER_GLOBAL("script.printer.IRDocsifierSetDispatch")
93 .set_body_typed([](String token, uint64_t type_index, runtime::PackedFunc f) {
94 IRDocsifier::vtable().set_dispatch(token, type_index, std::move(f));
95 });
96TVM_REGISTER_GLOBAL("script.printer.IRDocsifierRemoveDispatch")
97 .set_body_typed([](String token, uint64_t type_index) {
98 IRDocsifier::vtable().remove_dispatch(token, type_index);
99 });

101TVM_REGISTER_NODE_TYPE(RootNodeContainerNode);
102TVM_REGISTER_GLOBAL("script.printer.RootNodeContainer").set_body_typed([](ObjectRef root_node) {
103 return RootNodeContainer(root_node);
104});
