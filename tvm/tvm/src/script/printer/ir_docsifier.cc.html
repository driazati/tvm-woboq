28String GenerateUniqueName(std::string name_hint, std::unordered_set<String>* defined_names) {
29 for (char& c : name_hint) {
30 if (c != 'c' && !std::isalnum(c)) {
31 c = '_';
32 }
33 }
34 std::string name = name_hint;
35 for (int i = 1; !defined_names->insert(name).second; ++i) {
36 name = name_hint + "_" + std::to_string(i);
37 }
38 return name;
39}
40
41IdDoc IRDocsifierNode::Define(const ObjectRef& obj, const Frame& frame, const String& name_hint) {
42 String name = GenerateUniqueName(name_hint, &this->defined_names);
43 DocCreator doc_factory = [name]() { return IdDoc(name); };
44 auto result = obj2info.insert({obj, VariableInfo{std::move(doc_factory), name}});
45 ICHECK(result.second) << "Duplicated object: " << obj;
46 IdDoc def_doc(name);
47 frame->AddExitCallback([this, obj]() { this->RemoveVar(obj); });
48 return def_doc;
49}
50
51void IRDocsifierNode::Define(const ObjectRef& obj, const Frame& frame, DocCreator doc_factory) {
52 ICHECK(obj2info.find(obj) == obj2info.end()) << "Duplicated object: " << obj;
53 ICHECK(!doc_factory()->IsInstance<IdDocNode>())
54 << "IRDocsifierNode::Define cannot be used for variable that's mapped to IdDoc.";
55 obj2info.insert({obj, VariableInfo{std::move(doc_factory), NullOpt}});
56 frame->AddExitCallback([this, obj]() { this->RemoveVar(obj); });
57}
58
59Optional<ExprDoc> IRDocsifierNode::GetVarDoc(const ObjectRef& obj) const {
60 auto it = obj2info.find(obj);
61 if (it == obj2info.end()) {
62 return NullOpt;
63 }
64 return it->second.creator();
65}
66
67bool IRDocsifierNode::IsVarDefined(const ObjectRef& obj) const { return obj2info.count(obj); }
68
69void IRDocsifierNode::RemoveVar(const ObjectRef& obj) {
70 auto it = obj2info.find(obj);
71 ICHECK(it != obj2info.end()) << "No such object: " << obj;
72 if (it->second.name.defined()) {
73 defined_names.erase(it->second.name.value());
74 }
75 obj2info.erase(it);
76}

78void IRDocsifierNode::SetCommonPrefix(const ObjectRef& root,
79 runtime::TypedPackedFunc<bool(ObjectRef)> is_var) {
80 class Visitor : public AttrVisitor {
81 public:
82 inline void operator()(ObjectRef obj) { Visit("", &obj); }
83
84 private:
85 void Visit(const char* key, double* value) final {}
86 void Visit(const char* key, int64_t* value) final {}
87 void Visit(const char* key, uint64_t* value) final {}
88 void Visit(const char* key, int* value) final {}
89 void Visit(const char* key, bool* value) final {}
90 void Visit(const char* key, std::string* value) final {}
91 void Visit(const char* key, void** value) final {}
92 void Visit(const char* key, DataType* value) final {}
93 void Visit(const char* key, runtime::NDArray* value) final {}
94 void Visit(const char* key, ObjectRef* value) final {
95 const Object* obj = value->get();
96 if (obj == nullptr) {
97 return;
98 }
99 stack_.push_back(obj);
100 if (obj->IsInstance<ArrayNode>()) {
101 const ArrayNode* array = static_cast<const ArrayNode*>(obj);
102 for (ObjectRef element : *array) {
103 this->Visit("", &element);
104 }
105 } else if (obj->IsInstance<MapNode>()) {
106 const MapNode* map = static_cast<const MapNode*>(obj);
107 for (std::pair<ObjectRef, ObjectRef> kv : *map) {
108 this->Visit("", &kv.first);
109 this->Visit("", &kv.second);
110 }
111 } else {
112 vtable_->VisitAttrs(const_cast<Object*>(obj), this);
113 }
114 if (is_var(GetRef<ObjectRef>(obj))) {
115 HandleVar(obj);
116 }
117 stack_.pop_back();
118 }
119
120 void HandleVar(const Object* var) {
121 if (common_prefix.count(var) == 0) {
122 common_prefix[var] = stack_;
123 return;
124 }
125 std::vector<const Object*>& a = common_prefix[var];
126 std::vector<const Object*>& b = stack_;
127 int n = std::min(a.size(), b.size());
128 for (int i = 0; i < n; ++i) {
129 if (a[i] != b[i]) {
130 a.resize(i);
131 break;
132 }
133 }
134 }
135
136 ReflectionVTable* vtable_ = ReflectionVTable::Global();
137 std::vector<const Object*> stack_;
138
139 public:
140 runtime::TypedPackedFunc<bool(ObjectRef)> is_var;
141 std::unordered_map<const Object*, std::vector<const Object*>> common_prefix;
142 };
143 Visitor visitor;
144 visitor.is_var = is_var;
145 visitor(root);
146 this->common_prefix = std::move(visitor.common_prefix);
147}

149IRDocsifier::IRDocsifier(Map<String, String> ir_prefix) {
150 auto n = make_object<IRDocsifierNode>();
151 n->ir_prefix = std::move(ir_prefix);
152 n->dispatch_tokens.push_back("");
153 data_ = std::move(n);
154}
