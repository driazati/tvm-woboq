25namespace tvm {
26namespace script {
27namespace printer {
28
29Doc IRDocsifierNode::AsDocImpl(const TracedObject<ObjectRef>& obj) const {
30 return IRDocsifier::vtable()(dispatch_tokens.back(), obj, GetRef<IRDocsifier>(this));
31}
32
33IRDocsifier::IRDocsifier(Map<String, String> ir_prefix) {
34 auto n = make_object<IRDocsifierNode>();
35 n->ir_prefix = std::move(ir_prefix);
36 n->dispatch_tokens.push_back(kDefaultDispatchToken);
37 data_ = std::move(n);
38}
39
40IRDocsifier::FType& IRDocsifier::vtable() {
41 static IRDocsifier::FType inst;
42 return inst;
43}
44
45TVM_REGISTER_NODE_TYPE(IRDocsifierNode);
46TVM_REGISTER_GLOBAL("script.printer.IRDocsifier").set_body_typed([](Map<String, String> ir_prefix) {
47 return IRDocsifier(ir_prefix);
48});
49TVM_REGISTER_GLOBAL("script.printer.IRDocsifierAsDoc")
50 .set_body_typed([](IRDocsifier p, ObjectRef obj, ObjectPath obj_path) {
51 return p->AsDoc<Doc>(MakeTraced(obj, obj_path));
52 });
53
54TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPushDispatchToken")
55 .set_body_typed([](IRDocsifier p, String token) { p->dispatch_tokens.push_back(token); });
56TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPopDispatchToken").set_body_typed([](IRDocsifier p) {
57 p->dispatch_tokens.pop_back();
58});
59
60TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPushFrame")
61 .set_body_typed([](IRDocsifier p, Frame frame) { p->frames.push_back(frame); });
62TVM_REGISTER_GLOBAL("script.printer.IRDocsifierPopFrame").set_body_typed([](IRDocsifier p) {
63 p->frames.pop_back();
64});
65
66TVM_REGISTER_GLOBAL("script.printer.IRDocsifierSetDispatch")
67 .set_body_typed([](String token, uint64_t type_index, runtime::PackedFunc f) {
68 IRDocsifier::vtable().set_dispatch(token, type_index, std::move(f));
69 });
70TVM_REGISTER_GLOBAL("script.printer.IRDocsifierRemoveDispatch")
71 .set_body_typed([](String token, uint64_t type_index) {
72 IRDocsifier::vtable().remove_dispatch(token, type_index);
73 });
