20#include <tvm/runtime/registry.h>
21#include <tvm/script/printer.h>
22#include <tvm/script/printer/doc.h>
23#include <tvm/script/printer/doc_printer.h>
24#include <tvm/script/printer/frame.h>
25#include <tvm/script/printer/ir_docsifier.h>
26
27namespace tvm {
28namespace script {
29namespace printer {
30
31String Script( //
32 const ObjectRef& root_node, //
33 String ir_name, //
34 Map<String, String> ir_prefix, //
35 int indent_spaces, //
36 bool print_line_numbers, //
37 int num_context_lines, //
38 Optional<ObjectPath> path_to_underline //
39) {
40 IRDocsifier ir_docsifier(ir_prefix);
41
42 auto dispatch_ctx = ir_docsifier->WithDispatchToken(ir_name);
43
44 Doc doc = ir_docsifier->AsDoc<Doc>(MakeTraced(RootNodeContainer(root_node)));
45
46 return DocToPythonScript(doc, indent_spaces, print_line_numbers, num_context_lines,
47 path_to_underline);
48}
49
50TVM_REGISTER_GLOBAL("script.printer.Script").set_body_typed(&Script);
51
52} // namespace printer
53} // namespace script
54} // namespace tvm
