71/*! \brief Argsort. Order: largest to smallest */
72template <typename T>
73inline std::vector<int> Argsort(const std::vector<T>& scores) {
74 std::vector<int> index;
75 index.reserve(scores.size());
76 for (size_t i = 0; i < scores.size(); ++i) {
77 index.push_back(i);
78 }
79 auto cmp = [&scores](int l, int r) { return scores[l] > scores[r]; };
80 std::sort(index.begin(), index.end(), cmp);
81 return index;
82}

84/*! \brief Convert operation to stage id. */
85inline int OperationToStage(const te::Operation& op, const State& state) {
86 for (size_t i = 0; i < state->stages.size(); ++i) {
87 if (op == state->stages[i]->op) {
88 return i;
89 }
90 }
91 LOG(FATAL) << "Cannot find op: " << op;
92 return -1;
93}

170/*! \brief Get all consumers for a stage. This function propagates the relation for inlined ops. */
171inline std::set<int> GetConsumers(const SearchTask& task, const State& state, int stage_id) {
172 std::unordered_set<te::Operation, ObjectHash, ObjectEqual> consumers;
173 std::set<int> ret;
174
175 if (state->current_compute_dag) {
176 consumers = state->current_compute_dag.as<ComputeDAGNode>()->access_analyzer.GetConsumers(
177 state, state->stages[stage_id]->op);
178 } else {
179 consumers = task->compute_dag->access_analyzer.GetConsumers(state, state->stages[stage_id]->op);
180 }
181
182 for (const auto& op : consumers) {
183 ret.insert(OperationToStage(op, state));
184 }
185 return ret;
186}

188/*! \brief Check if a stage has single consumer or all of its consumers share a common root, return
189 * the target consumer root or -1. */
190inline int GetSingleConsumerId(const SearchTask& task, const State& state, int stage_id) {
191 const std::set<int>& consumers = GetConsumers(task, state, stage_id);
192 if (consumers.empty()) {
193 return -1;
194 }
195
196 if (consumers.size() == 1) {
197 return *consumers.begin();
198 } else {
199 // Check all consumers share a common root
200 int common_root_id = -1;
201 bool mismatch = false;
202 for (const auto& consumer_stage_id : consumers) {
203 int root_id = -1;
204 if (state->stages[consumer_stage_id]->compute_at == ComputeAtKind::kRoot) {
205 root_id = consumer_stage_id;
206 } else if (state->stages[consumer_stage_id]->compute_at == ComputeAtKind::kIter) {
207 root_id = state->attach_map->stage_to_attach_iter.at(consumer_stage_id).first;
208 } else {
209 LOG(FATAL) << "Invalid case";
210 }
211
212 if (common_root_id == -1) {
213 common_root_id = root_id;
214 } else {
215 if (common_root_id != root_id) {
216 mismatch = true;
217 break;
218 }
219 }
220 }
221
222 return mismatch ? -1 : common_root_id;
223 }
224}

302/*! \brief Compute the product of lengths of all space iters and all reduce iters, respectively. */
303inline std::pair<int64_t, int64_t> GetCumulativeSpaceAndReductionLength(const Stage& stage) {
304 int64_t cum_space_len = 1, cum_reduce_len = 1;
305 for (const auto& iter : stage->iters) {
306 if (iter->iter_kind == IteratorKind::kSpatial) {
307 cum_space_len *= GetExtent(iter);
308 } else if (iter->iter_kind == IteratorKind::kReduction) {
309 cum_reduce_len *= GetExtent(iter);
310 }
311 }
312 return std::make_pair(cum_space_len, cum_reduce_len);
313}

315/*! \brief Return whether this stage needs rfactor. */
316inline bool NeedsRfactor(const SearchTask& task, const State& state, int stage_id) {
317 const auto& op = state->stages[stage_id]->op;
318 if (op->IsInstance<te::ComputeOpNode>()) {
319 // Compute the product of lengths of all space iters and all reduce iters
320 int cum_space_len, cum_reduce_len;
321 std::tie(cum_space_len, cum_reduce_len) =
322 GetCumulativeSpaceAndReductionLength(state->stages[stage_id]);
323
324 if (NeedsMultilevelTiling(task, state, stage_id)) {
325 // Do not use rfactor if we have enough parallelism on space iters
326 if (cum_space_len > cum_reduce_len || cum_space_len > task->hardware_params->num_cores * 16) {
327 return false;
328 } else {
329 return true;
330 }
331 } else if (cum_reduce_len > 1) {
332 // Always try rfactor for reduction ops
333 return cum_reduce_len > task->hardware_params->num_cores;
334 }
335 }
336
337 return false;
338}

360/*! \brief Return whether the stage has only one consumer and they are elementwise-matched. */
361inline bool HasSingleElementwiseMatchedConsumer(const SearchTask& task, const State& state,
362 int stage_id, int* target_stage_id = nullptr) {
363 // Temporal object to be used if the input pointer is nullptr
364 int temp_target_stage_id;
365 if (target_stage_id == nullptr) {
366 target_stage_id = &temp_target_stage_id;
367 }
368 const std::set<int>& consumers = GetConsumers(task, state, stage_id);
369 if (consumers.size() == 1) {
370 *target_stage_id = *consumers.begin();
371 if (ElementwiseMatch(task, state, stage_id, *target_stage_id) &&
372 (!(HasReduceIter(state->stages[stage_id]) &&
373 HasReduceIter(state->stages[*target_stage_id]))) &&
374 (!StrEndsWith(state->stages[*target_stage_id]->op->name, ".shared"))) {
375 return true;
376 }
377 }
378 return false;
379}

441/*! \brief Return whether the stage does cross thread reduction. */
442inline bool HasCrossThreadReduction(const State& state, int stage_id) {
443 std::function<bool(const Stage&)> check_stage = [](const Stage& in_stage) {
444 for (const auto& iter : in_stage->iters) {
445 if (iter->annotation == IteratorAnnotation::kThreadX &&
446 iter->iter_kind == IteratorKind::kReduction) {
447 return true;
448 }
449 }
450 return false;
451 };
452
453 // Check the stage itself
454 if (check_stage(state->stages[stage_id])) {
455 return true;
456 }
457
458 // Check the attached stages
459 for (size_t iter_id = 0; iter_id < state->stages[stage_id]->iters.size(); iter_id++) {
460 const auto& res =
461 state->attach_map->iter_to_attached_stages.find(std::make_pair(stage_id, iter_id));
462 if (res != state->attach_map->iter_to_attached_stages.end()) {
463 for (int attached_stage_id : res->second) {
464 if (check_stage(state->stages[attached_stage_id])) {
465 return true;
466 }
467 }
468 }
469 }
470
471 return false;
472}

481/*! \brief Extract primitive iterators from a nested fused or splitted iterator's name. */
482inline void ExtractOriginalIterators(const std::string& name, std::set<std::string>* rets) {
483 size_t last_pos = 0;
484 for (size_t i = 0; i < name.size(); ++i) {
485 if (name[i] == '@' || name[i] == '.') { // '@' for fuse and '.' for split
486 if (!isdigit(name[last_pos]) && name[last_pos] != '@' && name[last_pos] != '.') {
487 rets->insert(name.substr(last_pos, i - last_pos));
488 }
489 last_pos = i + 1;
490 }
491 }
492
493 if (last_pos < name.size() && !isdigit(name[last_pos]) && name[last_pos] != '@' &&
494 name[last_pos] != '.') {
495 rets->insert(name.substr(last_pos, name.size() - last_pos));
496 }
497}

499/*! \brief Get the last reduce iterator in the outermost reduce tile. */
500inline Iterator GetLastReduceIteratorInOutermostReduceTile(const Stage& stage) {
501 auto pop = stage->op.as<te::ComputeOpNode>();
502 ICHECK(pop != nullptr);
503 std::set<std::string> original_names;
504
505 const std::set<std::string>& no_split_at_inner_name_set =
506 stage->op->attrs.count(SearchPolicyKey::no_split_at_inner)
507 ? GetIterNameSetParam(stage->op->attrs, SearchPolicyKey::no_split_at_inner)
508 : std::set<std::string>();
509 size_t reduce_axis_size = 0;
510 for (const auto axis : pop->reduce_axis) {
511 if (!no_split_at_inner_name_set.count(axis->var->name_hint)) {
512 reduce_axis_size++;
513 }
514 }
515 if (reduce_axis_size) {
516 for (const auto& iter : stage->iters) {
517 if (iter->iter_kind == IteratorKind::kReduction) {
518 ExtractOriginalIterators(iter->name, &original_names);
519 if (original_names.size() == reduce_axis_size) {
520 return iter;
521 }
522 }
523 }
524 } else {
525 // Return the first reduce iterator
526 for (const auto& iter : stage->iters) {
527 if (iter->iter_kind == IteratorKind::kReduction) {
528 return iter;
529 }
530 }
531 }
532
533 LOG(FATAL) << "Cannot find the iterator.";
534 return stage->iters[0];
535}

537/*! \brief Get the target stage id of a history step in the new state.
538 * We need this because the stage_id in the history may be stale due to later steps */
539inline int GetTargetStageIDInState(const State& s, int step_id) {
540 int stage_inc = 0;
541
542 for (size_t i = step_id + 1; i < s->transform_steps.size(); ++i) {
543 if (IsStageNumberChangingStep(s->transform_steps[i])) {
544 if (s->transform_steps[i]->stage_id <= s->transform_steps[step_id]->stage_id + stage_inc)
545 stage_inc++;
546 }
547 }
548 return s->transform_steps[step_id]->stage_id + stage_inc;
549}

551/*! \brief Get all split steps for one stage. */
552inline void GetSplitStepIds(const State& s, int stage_id, std::vector<int>* split_step_ids) {
553 for (int i = static_cast<int>(s->transform_steps.size()) - 1; i >= 0; --i) {

563 }
564 }
565 }
566}

568/*! \brief Fuse all reduction iterators. */
569inline State FuseAllReductionIterators(const State& state, int stage_id, Iterator* fused_iter,
570 Array<Iterator>* space_iters,
571 Array<Iterator>* reduce_iters) {
572 space_iters->clear();
573 reduce_iters->clear();
574
575 for (const auto& iter : state->stages[stage_id]->iters) {
576 if (iter->iter_kind == IteratorKind::kSpatial) {
577 space_iters->push_back(iter);
578 } else if (iter->iter_kind == IteratorKind::kReduction) {
579 reduce_iters->push_back(iter);
580 }
581 }
582
583 ICHECK(!reduce_iters->empty());
584 State tmp_s = state;
585 if (reduce_iters->size() > 1) {
586 *fused_iter = tmp_s.fuse(stage_id, *reduce_iters);
587 } else {
588 *fused_iter = (*reduce_iters)[0];
589 }
590 return tmp_s;
591}
592
593/*! \brief Fuse all outer level space iterators. */
594inline State FuseAllOuterSpaceIterators(const State& state, int stage_id, Iterator* fused_iter) {
595 std::vector<Iterator> to_fuse;
596 for (size_t iter_id = 0; iter_id < state->stages[stage_id]->iters.size(); ++iter_id) {
597 const auto& it = state->stages[stage_id]->iters[iter_id];
598 // Stop at reduce iterator or annotated iterator
599 if (it->iter_kind == IteratorKind::kReduction || it->annotation != IteratorAnnotation::kNone) {
600 break;
601 }
602 // Stop at compute_at attach point
603 if (state->attach_map->iter_to_attached_stages.count(std::make_pair(stage_id, iter_id - 1))) {
604 break;
605 }
606 to_fuse.push_back(it);
607 }
608
609 State tmp_s = state;
610 if (to_fuse.size() == 1) {
611 *fused_iter = to_fuse[0];
612 } else {
613 *fused_iter = tmp_s.fuse(stage_id, to_fuse);
614 }
615 return tmp_s;
616}

618/*! \brief Random sample states. */
619inline Array<State> RandomSampleStates(const Array<State>& in_states, std::mt19937* random_gen,
620 size_t out_size) {
621 Array<State> out_states;

628/*! \brief Compute prefix-sum probabiilty based on the given weights */
629inline void ComputePrefixSumProb(const std::vector<float>& weights,
630 std::vector<double>* prefix_sum_probs) {
631 // Compute selection probabilities.
632 float sum = 0.0;
633 prefix_sum_probs->resize(weights.size());
634 for (size_t i = 0; i < weights.size(); ++i) {
635 sum += std::max(weights[i], 0.0f);
636 (*prefix_sum_probs)[i] = sum;
637 }
638 for (size_t i = 0; i < weights.size(); ++i) {
639 (*prefix_sum_probs)[i] /= sum;
640 }
641}
642
643/*! \brief Random choose an index according to a prefix sum probability. */
644inline int RandomChoose(const std::vector<double>& prefix_sum_probs, std::mt19937* random_gen) {
645 std::uniform_real_distribution<> dis(0.0, 1.0);
646 double x = dis(*random_gen);
647
648 ICHECK(!prefix_sum_probs.empty());
649
650 return std::lower_bound(prefix_sum_probs.begin(), prefix_sum_probs.end(), x) -
651 prefix_sum_probs.begin();
652}
