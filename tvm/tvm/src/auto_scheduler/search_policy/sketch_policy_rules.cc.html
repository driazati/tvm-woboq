56/********** RuleAlwaysInline **********/
57inline bool ShouldAlwaysBeInlined(const SketchPolicyNode& policy, const State& state,
58 int stage_id) {
59 const SearchTask& task = policy.search_task;
60 const Stage& stage = state->stages[stage_id];
61
62 // Check the inline limitation of TE
63 if (stage->op_type == StageKind::kPlaceholder || IsOutputOp(task, state, stage_id) ||
64 HasReduceIter(stage)) {
65 return false;
66 }
67
68 if (IsGPUTask(task)) { // Greedily inline all inlinable ops on gpu
69 return true;
70 } else {
71 // Only always-inline strict-inlinable ops on cpu.
72 // The computation location of other ops will be tuned by InitChangeComputeLocation
73 // and MutateComputeLocation.
74 return IsStrictlyInlineable(task, state, stage_id);
75 }
76}
77
78SketchGenerationRule::ConditionKind RuleAlwaysInline::MeetCondition(const SketchPolicyNode& policy,
79 const State& state,
80 int stage_id) const {
81 return ShouldAlwaysBeInlined(policy, state, stage_id) ? ConditionKind::kApplyAndSkipRest
82 : ConditionKind::kSkip;
83}
84
85std::vector<std::pair<State, int>> RuleAlwaysInline::Apply(const SketchPolicyNode& policy,
86 const State& state, int stage_id) const {
87 State tmp_s = state;
88 tmp_s.compute_inline(stage_id);
89 return {std::make_pair(std::move(tmp_s), stage_id - 1)};
90}

112/********** RuleMultiLevelTilingWithFusion **********/
113
114SketchGenerationRule::ConditionKind RuleMultiLevelTilingWithFusion::MeetCondition(
115 const SketchPolicyNode& policy, const State& state, int stage_id) const {
116 if (NeedsMultilevelTiling(policy.search_task, state, stage_id) &&
117 HasSingleElementwiseMatchedConsumer(policy.search_task, state, stage_id)) {
118 // Always do fusion for stage with cache_write or is in GPU policy
119 return HasCacheWriteStage(state, stage_id) || IsGPUTask(policy.search_task)
120 ? ConditionKind::kApplyAndSkipRest
121 : ConditionKind::kApply;
122 }
123 return ConditionKind::kSkip;
124}
125
126std::vector<std::pair<State, int>> RuleMultiLevelTilingWithFusion::Apply(
127 const SketchPolicyNode& policy, const State& state, int stage_id) const {
128 int target_stage_id;
129 ICHECK(
130 HasSingleElementwiseMatchedConsumer(policy.search_task, state, stage_id, &target_stage_id));
131 const std::string& multi_level_tiling_structure =
132 IsGPUTask(policy.search_task)
133 ? GetStringParam(policy.params, SketchParamKey::MultiLevelTiling::gpu_structure)
134 : GetStringParam(policy.params, SketchParamKey::MultiLevelTiling::cpu_structure);
135 std::vector<int> spatial_split_step_ids;
136 State base_state =
137 DoMultiLevelTiling(state, stage_id, multi_level_tiling_structure, &spatial_split_step_ids);
138
139 std::vector<std::pair<State, int>> ret;
140 std::vector<int> follow_tiling_levels =
141 IsGPUTask(policy.search_task) ? std::vector<int>{3} : std::vector<int>{1, 2};
142 for (int level : follow_tiling_levels) {
143 if (tolower(multi_level_tiling_structure[level - 1]) != 's') {
144 continue;
145 }
146 State tmp_s = base_state;
147 tmp_s = FollowTiling(tmp_s, target_stage_id, spatial_split_step_ids, level);
148 const Iterator& target_iter =
149 tmp_s->stages[target_stage_id]->iters[level * spatial_split_step_ids.size() - 1];
150 tmp_s.compute_at(stage_id, target_stage_id, target_iter);
151 ret.emplace_back(std::move(tmp_s), stage_id - 1);
152 }
153
154 return ret;
155}

157/********** RuleAddCacheRead **********/
158
159SketchGenerationRule::ConditionKind RuleAddCacheRead::MeetCondition(const SketchPolicyNode& policy,
160 const State& state,
161 int stage_id) const {
162 const SearchTask& task = policy.search_task;
163
164 // Don't cache_read a stage if it has multiple consumers
165 const std::set<int>& consumers = GetConsumers(task, state, stage_id);
166
167 if (consumers.size() == 0) return ConditionKind::kSkip;
168 // Don't cache_read a stage if its consumer does not need multi-level tiling
169 int target_stage_id = *consumers.begin();
170 if (!NeedsMultilevelTiling(task, state, target_stage_id)) {
171 return ConditionKind::kSkip;
172 }
173
174 // Don't cache_read a stage if its consumer does cross-thread reduction
175 if (HasCrossThreadReduction(state, target_stage_id)) {
176 return ConditionKind::kSkip;
177 }
178
179 // Only direct producers can be cache read
180 const std::set<int>& producers = GetDirectProducers(task, state, target_stage_id);
181 if (producers.find(stage_id) == producers.end()) {
182 return ConditionKind::kSkip;
183 }
184
185 return ConditionKind::kApplyAndSkipRest;
186}
187
188std::vector<std::pair<State, int>> RuleAddCacheRead::Apply(const SketchPolicyNode& policy,
189 const State& state, int stage_id) const {
190 const SearchTask& task = policy.search_task;
191 const std::set<int>& consumers = GetConsumers(task, state, stage_id);
192 State tmp_s = state;
193
194 int target_stage_id_offset = 0;
195 for (int orig_target_stage_id : consumers) {
196 int target_stage_id = orig_target_stage_id + target_stage_id_offset;
197
198 // Cache read add shared memory
199 int added_stage_id = tmp_s.cache_read(stage_id, "shared", {target_stage_id}, task->compute_dag);
200 target_stage_id_offset++;
201 target_stage_id++;
202
203 const auto& share_read_pos =
204 GetLastReduceIteratorInOutermostReduceTile(tmp_s->stages[target_stage_id]);
205 tmp_s.compute_at(added_stage_id, target_stage_id, share_read_pos);
206 }
207
208 return {std::make_pair(tmp_s, stage_id)};
209}

211/********** RuleAddCacheWrite **********/
212
213SketchGenerationRule::ConditionKind RuleAddCacheWrite::MeetCondition(const SketchPolicyNode& policy,
214 const State& state,
215 int stage_id) const {
216 // Add cache write if a stage needs multi-level tiling, but does not have a element-wise
217 // matched consumer
218 if (NeedsMultilevelTiling(policy.search_task, state, stage_id) &&
219 !HasSingleElementwiseMatchedConsumer(policy.search_task, state, stage_id)) {
220 // An apply and skip rule will be handled in RuleMultiLevelTilingWithFusion
221 return IsGPUTask(policy.search_task) ? ConditionKind::kApplyAndSkipRest : ConditionKind::kApply;
222 }
223
224 return ConditionKind::kSkip;
225}

227std::vector<std::pair<State, int>> RuleAddCacheWrite::Apply(const SketchPolicyNode& policy,
228 const State& state,
229 int stage_id) const {
230 State tmp_s = state;
231 tmp_s.cache_write(stage_id, "local", policy.search_task->compute_dag);
232 return {std::make_pair(std::move(tmp_s), stage_id)};
233}

235/********** RuleAddRfactor **********/
236
237SketchGenerationRule::ConditionKind RuleAddRfactor::MeetCondition(const SketchPolicyNode& policy,
238 const State& state,
239 int stage_id) const {
240 return (NeedsRfactor(policy.search_task, state, stage_id) && !HasCacheWriteStage(state, stage_id))
241 ? ConditionKind::kApply
242 : ConditionKind::kSkip;
243}
244
245std::vector<std::pair<State, int>> RuleAddRfactor::Apply(const SketchPolicyNode& policy,
246 const State& state, int stage_id) const {
247 // Fuse all reduction iters
248 Array<Iterator> space_iters, reduce_iters;
249 Iterator fused_reduce_iter;
250 State base_state =
251 FuseAllReductionIterators(state, stage_id, &fused_reduce_iter, &space_iters, &reduce_iters);
252
253 // TODO(merrymercy): We can do more analysis here to generate less and more efficient sketches.
254 // In some cases, we only need rfactor for more parallel
255 // In some cases, we only need rfactor for vectorization.
256 // Now we will generate two versions and let the search figure out the bette one.
257
258 // Split reduction iters
259 const auto& split_res = base_state.split(stage_id, fused_reduce_iter, {Integer(1)});
260 int factor_axis_id = static_cast<int>(space_iters.size());
261 std::vector<std::pair<State, int>> ret;
262 for (const auto& split_iter : split_res) {
263 State tmp_s = base_state;
264 int rstage_id =
265 tmp_s.rfactor(stage_id, split_iter, factor_axis_id, policy.search_task->compute_dag);
266
267 // reorder the space iterator to innermost for vectorization
268 if (split_iter == split_res[1]) {
269 Array<Iterator> new_order;
270 for (size_t i = 0; i < tmp_s->stages[rstage_id]->iters.size(); ++i) {
271 if (i != space_iters.size()) {
272 new_order.push_back(tmp_s->stages[rstage_id]->iters[i]);
273 }
274 }
275 new_order.push_back(tmp_s->stages[rstage_id]->iters[space_iters.size()]);
276 tmp_s.reorder(rstage_id, new_order);
277 }
278
279 ret.emplace_back(std::move(tmp_s), rstage_id - 1);
280 }
281
282 return ret;
283}

285/********** RuleSimplifyComputeWithConstTensor **********/
286
287SketchGenerationRule::ConditionKind RuleSimplifyComputeWithConstTensor::MeetCondition(
288 const SketchPolicyNode& policy, const State& state, int stage_id) const {
289 return state->stages[stage_id]->op->attrs.count(SearchPolicyKey::simplify_const_tensor_indices)
290 ? ConditionKind::kApplyAndSkipRest
291 : ConditionKind::kSkip;
292}
293
294std::vector<std::pair<State, int>> RuleSimplifyComputeWithConstTensor::Apply(
295 const SketchPolicyNode& policy, const State& state, int stage_id) const {
296 std::set<std::string> const_tensor_indices = GetIterNameSetParam(
297 state->stages[stage_id]->op->attrs, SearchPolicyKey::simplify_const_tensor_indices);
298
299 State tmp_s = state;
300 Array<Array<Iterator>> tiled_outer_iters;
301 Array<Iterator> unrolled_inner_iters;
302
303 // Currently set to 2
304 size_t tile_level = 2;
305
306 for (const auto& iter : state->stages[stage_id]->iters) {
307 if (const_tensor_indices.count(iter->name)) {
308 // unroll indices of const tensors
309 unrolled_inner_iters.push_back(tmp_s.unroll(stage_id, iter));
310 } else {
311 // tile other space indices
312 ICHECK(iter->iter_kind == IteratorKind::kSpatial);
313 tiled_outer_iters.push_back(
314 tmp_s.split(stage_id, iter, Array<Optional<Integer>>(tile_level - 1, NullOpt)));
315 }
316 }
317
318 // reorder them
319 Array<Iterator> new_order;
320 for (size_t i = 0; i < tile_level; ++i) {
321 for (size_t j = 0; j < tiled_outer_iters.size(); ++j) {
322 new_order.push_back(tiled_outer_iters[j][i]);
323 }
324 }
325 new_order.insert(new_order.end(), unrolled_inner_iters.begin(), unrolled_inner_iters.end());
326 tmp_s.reorder(stage_id, new_order);
327
328 return {std::make_pair(tmp_s, stage_id - 1)};
329}

331/********** RuleCrossThreadReduction **********/
332
333SketchGenerationRule::ConditionKind RuleCrossThreadReduction::MeetCondition(
334 const SketchPolicyNode& policy, const State& state, int stage_id) const {
335 ICHECK(IsGPUTask(policy.search_task));
336
337 // If it is an intermediate state created by RuleAddCacheWrite,
338 // we just skip it.
339 if (HasCacheWriteStage(state, stage_id)) {
340 return ConditionKind::kSkip;
341 }
342
343 const auto& op = state->stages[stage_id]->op;
344 if (op->IsInstance<te::ComputeOpNode>()) {
345 // Compute the product of lengths of all space iters and all reduce iters
346 int cum_space_len, cum_reduce_len;
347 std::tie(cum_space_len, cum_reduce_len) =
348 GetCumulativeSpaceAndReductionLength(state->stages[stage_id]);
349
350 if (NeedsMultilevelTiling(policy.search_task, state, stage_id)) {
351 // Avoid rfactor if we have enough parallelism on space iters
352 if (cum_space_len > policy.search_task->hardware_params->max_threads_per_block) {
353 return ConditionKind::kSkip;
354 }
355
356 return cum_space_len < cum_reduce_len ? ConditionKind::kApply : ConditionKind::kSkip;
357 } else if (cum_reduce_len > 1) {
358 // Try rfactor for other reduction operators
359 return cum_reduce_len > policy.search_task->hardware_params->warp_size ? ConditionKind::kApply
360 : ConditionKind::kSkip;
361 }
362 }
363
364 return ConditionKind::kSkip;
365}
366
367std::vector<std::pair<State, int>> RuleCrossThreadReduction::Apply(const SketchPolicyNode& policy,
368 const State& state,
369 int stage_id) const {
370 const SearchTask& task = policy.search_task;
371 State tmp_s = state;
372
373 // fuse all reduction iters
374 Array<Iterator> space_iters, reduce_iters;
375 Iterator fused_reduce_iter;
376 tmp_s =
377 FuseAllReductionIterators(tmp_s, stage_id, &fused_reduce_iter, &space_iters, &reduce_iters);
378
379 // Check the opportunity for kernel fusion
380 bool fusible = false;
381 int target_stage_id = GetSingleConsumerId(policy.search_task, tmp_s, stage_id);
382 int num_common_outer = -1;
383 if (target_stage_id >= 0) {
384 num_common_outer =
385 GetNumCommonOuterIterator(policy.search_task, tmp_s, stage_id, target_stage_id);
386 if (num_common_outer > 0 &&
387 !NeedsMultilevelTiling(policy.search_task, state, target_stage_id)) {
388 fusible = true;
389 }
390 }
391
392 if (fusible) {
393 const Stage& target_stage = state->stages[target_stage_id];
394 std::vector<int> split_step_ids;
395
396 GetSplitStepIds(tmp_s, target_stage_id, &split_step_ids);
397
398 if (split_step_ids.size() == 0) {
399 // If the target stage does not have split step,
400 // it must be a simple stage without reduce iters.
401 // We then should do a split for it.
402 ICHECK(!HasReduceIter(target_stage));
403 const auto& split_res = tmp_s.split(target_stage_id, target_stage->iters.back(),
404 {Integer(task->hardware_params->warp_size)});
405 tmp_s.bind(target_stage_id, split_res[1], IteratorAnnotation::kThreadX);
406 split_step_ids.push_back(tmp_s->transform_steps.size() - 2);
407 }
408
409 ICHECK_EQ(split_step_ids.size(), 1);
410
411 const Iterator& target_iter = tmp_s->stages[target_stage_id]->iters[num_common_outer - 1];
412 const auto& split_res = tmp_s.follow_split(stage_id, fused_reduce_iter, split_step_ids[0], 1);
413 tmp_s.bind(stage_id, split_res[1], IteratorAnnotation::kThreadX);
414 tmp_s.compute_at(stage_id, target_stage_id, target_iter);
415 } else {
416 const auto& split_res =
417 tmp_s.split(stage_id, fused_reduce_iter, {Integer(task->hardware_params->warp_size)});
418 tmp_s.bind(stage_id, split_res[1], IteratorAnnotation::kThreadX);
419 }
420
421 return {std::make_pair(std::move(tmp_s), stage_id - 1)};
422}

424/********** RuleSpecialComputeLocationGPU **********/
425
426SketchGenerationRule::ConditionKind RuleSpecialComputeLocationGPU::MeetCondition(
427 const SketchPolicyNode& policy, const State& state, int stage_id) const {
428 if (GetProducers(policy.search_task, state, stage_id).empty()) {
429 return ConditionKind::kSkip;
430 }
431
432 if (!ShouldAlwaysBeInlined(policy, state, stage_id)) {
433 return ConditionKind::kSkip;
434 }
435
436 const std::set<int>& consumers = GetConsumers(policy.search_task, state, stage_id);
437 if (consumers.size() == 1 && state->stages[*consumers.begin()]->op->attrs.count(
438 SearchPolicyKey::simplify_const_tensor_indices)) {
439 return ConditionKind::kApplyAndSkipRest;
440 }
441
442 return ConditionKind::kSkip;
443}
444
445std::vector<std::pair<State, int>> RuleSpecialComputeLocationGPU::Apply(
446 const SketchPolicyNode& policy, const State& state, int stage_id) const {
447 State tmp_s = state;
448 const std::set<int>& consumers = GetConsumers(policy.search_task, state, stage_id);
449 ICHECK_EQ(consumers.size(), 1);
450
451 // Get the last outer space iterator that is not unrolled.
452 const Stage& target_stage = state->stages[*consumers.begin()];
453 for (size_t i = 0; i < target_stage->iters.size(); ++i) {
454 if (target_stage->iters[i]->annotation == IteratorAnnotation::kUnroll) {
455 ICHECK_GT(i, 0);
456
457 tmp_s.compute_at(stage_id, *consumers.begin(), target_stage->iters[i - 1]);
458 break;
459 }
460 }
461
462 return {std::make_pair(std::move(tmp_s), stage_id - 1)};
463}

466
467SketchGenerationRule::ConditionKind RuleCustomSketch::MeetCondition(const SketchPolicyNode& policy,
468 const State& state,
469 int stage_id) const {
470 auto ret = meet_condition_func_(tvm::runtime::GetRef<SketchPolicy>(&policy), state, stage_id);

472 return ConditionKind(static_cast<int>(ret));
473 } else {
474 LOG(WARNING) << "Wrong rule condition value. Apply the rule and skip the rest";
475 return ConditionKind::kApplyAndSkipRest;

478
479std::vector<std::pair<State, int>> RuleCustomSketch::Apply(const SketchPolicyNode& policy,
480 const State& state, int stage_id) const {
481 Array<Array<ObjectRef>> apply_ret =
482 apply_func_(tvm::runtime::GetRef<SketchPolicy>(&policy), state, stage_id);
483 std::vector<std::pair<State, int>> ret;
484 for (const auto& item : apply_ret) {
485 CHECK_EQ(item.size(), 2);
486 auto next = item[1].as<IntImmNode>();
487 ret.emplace_back(Downcast<State>(item[0]), next->value);
488 }
489 return ret;
490}

492/********** Init Population **********/
493
494PopulationGenerationRule::ResultKind InitFillTileSize::Apply(SketchPolicyNode* policy, State* state,
495 std::mt19937* rand_gen) const {
496 SplitFactorizationMemo split_memo;
497 int max_innermost_split_factor =
498 GetIntParam(policy->params, SketchParamKey::max_innermost_split_factor);
499
500 StateNode* pstate = state->CopyOnWrite();
501 // Scan the transformation history and randomly fill tiles size for all SplitStep
502 for (size_t step_id = 0; step_id < (*state)->transform_steps.size(); ++step_id) {
503 if (auto ps = (*state)->transform_steps[step_id].as<SplitStepNode>()) {
504 bool all_defined = true;
505 for (const auto& len : ps->lengths) {
506 if (!len) {
507 all_defined = false;
508 break;
509 }
510 }
511 if (all_defined) {
512 continue;
513 }
514
515 ICHECK(ps->extent);
516 int extent = GetIntImm(ps->extent.value());
517 const auto& candidate_lens = split_memo.GetFactorizationSchemes(extent, ps->lengths.size(),
518 max_innermost_split_factor);
519 ICHECK(!candidate_lens.empty());
520 const auto& candidate_lengths = candidate_lens[(*rand_gen)() % candidate_lens.size()];
521
522 pstate->transform_steps.Set(
523 step_id,
524 SplitStep(ps->stage_id, ps->iter_id, ps->extent,
525 Array<Optional<Integer>>(candidate_lengths.begin(), candidate_lengths.end()),
526 ps->inner_to_outer));
527 }
528 }
529 pstate->concrete = true;
530
531 return ResultKind::kValid;
532}
533
534PopulationGenerationRule::ResultKind InitChangeComputeLocation::Apply(
535 SketchPolicyNode* policy, State* state, std::mt19937* rand_gen) const {
536 if (GetIntParam(policy->params, SketchParamKey::disable_change_compute_location)) {
537 return ResultKind::kValid;
538 }
539
540 for (int stage_id = static_cast<int>((*state)->stages.size()) - 1; stage_id >= 0; stage_id--) {
541 const Stage& stage = (*state)->stages[stage_id];
542 // Skip the inlined stages and placeholders
543 if (stage->op_type == StageKind::kPlaceholder || stage->compute_at == ComputeAtKind::kInlined) {
544 continue;
545 }
546 // Skip the tiled stages
547 if (IsTiled(stage) || NeedsMultilevelTiling(policy->search_task, *state, stage_id)) {
548 continue;
549 }
550
551 std::vector<std::pair<int, int>> candidates =
552 GetComputeLocationCandidates(policy->search_task, *state, stage_id);
553
554 int choice = (*rand_gen)() % (candidates.size() + 2);
555
556 if (choice == 0) {
557 if (!HasReduceIter(stage)) {
558 const auto& stage_to_attach_iter = (*state)->attach_map->stage_to_attach_iter;
559 if (stage_to_attach_iter.find(stage_id) != stage_to_attach_iter.end()) {
560 state->compute_inline(stage_id);
561 }
562 }
563 } else if (choice == 1) {
564 state->compute_root(stage_id);
565 } else {
566 choice = choice - 2;
567 const Stage& stage = (*state)->stages[candidates[choice].first];
568 state->compute_at(stage_id, candidates[choice].first,
569 stage->iters[candidates[choice].second]);
570 }
571 }
572
573 try {
574 *state = policy->search_task->compute_dag.InferBound(*state);
575 } catch (std::exception& e) {
576 return ResultKind::kInvalid;
577 }
578 return ResultKind::kValid;
579}
580
581PopulationGenerationRule::ResultKind InitParallel::Apply(SketchPolicyNode* policy, State* state,
582 std::mt19937* rand_gen) const {
583 std::function<void(const SketchPolicyNode&, State*, int stage_id, int iter_offset)>
584 annotate_parallel;
585 annotate_parallel = [&annotate_parallel](const SketchPolicyNode& policy, State* state,
586 int stage_id, int iter_offset) {
587 const Stage& stage = (*state)->stages[stage_id];
588
589 Array<Iterator> to_fuse;
590 int64_t parallel_degree = 1;
591
592 // Try to fuse and parallel the outermost n iterators
593 // Stop if we meet reduce iterator or we have enough parallel degree
594 size_t iter_id = iter_offset;
595 for (; iter_id < stage->iters.size(); ++iter_id) {
596 const Iterator& it = stage->iters[iter_id];
597 if (it->iter_kind == IteratorKind::kReduction ||
598 it->annotation != IteratorAnnotation::kNone) {
599 break;
600 }
601 to_fuse.push_back(it);
602 parallel_degree *= GetExtent(it);
603
604 if (parallel_degree > policy.search_task->hardware_params->num_cores * 16) {
605 break;
606 }
607
608 if ((*state)->attach_map->iter_to_attached_stages.count(std::make_pair(stage_id, iter_id))) {
609 break;
610 }
611 }
612
613 if (parallel_degree == 1) {
614 auto res =
615 (*state)->attach_map->iter_to_attached_stages.find(std::make_pair(stage_id, iter_id));
616 if (res != (*state)->attach_map->iter_to_attached_stages.end()) {
617 for (int attached_stage_id : res->second) {
618 annotate_parallel(policy, state, attached_stage_id, 0);
619 }
620 annotate_parallel(policy, state, stage_id, iter_id + 1);
621 }
622 }
623
624 if (!to_fuse.empty()) {
625 if (to_fuse.size() == 1) {
626 state->parallel(stage_id, to_fuse[0]);
627 } else {
628 Iterator fused_iter = state->fuse(stage_id, to_fuse);
629 state->parallel(stage_id, fused_iter);
630 }
631 }
632 };
633
634 for (size_t stage_id = 0; stage_id < (*state)->stages.size(); ++stage_id) {
635 const Stage& stage = (*state)->stages[stage_id];
636 if (stage->compute_at != ComputeAtKind::kRoot || stage->op_type == StageKind::kPlaceholder) {
637 continue;
638 }
639
640 annotate_parallel(*policy, state, stage_id, 0);
641 }
642
643 return ResultKind::kValid;
644}

646PopulationGenerationRule::ResultKind InitUnroll::Apply(SketchPolicyNode* policy, State* state,
647 std::mt19937* rand_gen) const {
648 std::vector<int>& auto_unroll_configs =
649 IsGPUTask(policy->search_task) ? auto_unroll_configs_gpu : auto_unroll_configs_cpu;
650 for (size_t stage_id = 0; stage_id < (*state)->stages.size(); ++stage_id) {
651 const Stage& stage = (*state)->stages[stage_id];
652 // Skip the inlined stage and placeholder stage
653 if (stage->compute_at == ComputeAtKind::kInlined || stage->op_type == StageKind::kPlaceholder) {
654 continue;
655 }
656
657 // Handle always_unroll_inner attr
658 if (stage->op->attrs.count(SearchPolicyKey::always_unroll_inner)) {
659 const auto& to_unroll_name_set =
660 GetIterNameSetParam(stage->op->attrs, SearchPolicyKey::always_unroll_inner);
661
662 // Unroll the space iterators and reduce iterators listed in the attrs in the innermost
663 // tile
664 std::set<std::string> visited_names;
665 for (int n = static_cast<int>(stage->iters.size()) - 1; n >= 0; n--) {
666 const Iterator& it = stage->iters[n];
667
668 // If we meet two iterators that come from a same original iterator,
669 // then we are out of the innermost tile
670 size_t size_before = visited_names.size();
671 ExtractOriginalIterators(it->name, &visited_names);
672 if (size_before == visited_names.size()) {
673 break;
674 }
675
676 std::set<std::string> name;
677 ExtractOriginalIterators(it->name, &name);
678 if (name.size() == 1 && to_unroll_name_set.count(*name.begin())) {
679 if (it->annotation == IteratorAnnotation::kNone) {
680 state->unroll(stage_id, it);
681 }
682 }
683 }
684 }
685
686 if (HasReduceIter(stage)) {
687 // Use auto unroll for multi level tiled stage
688 int value = auto_unroll_configs[(*rand_gen)() % auto_unroll_configs.size()];
689 state->pragma(stage_id, (*state)->stages[stage_id]->iters[0],
690 std::string("auto_unroll_max_step") + "$" + std::to_string(value));
691 }
692 }
693
694 return ResultKind::kValid;
695}

697PopulationGenerationRule::ResultKind InitVectorization::Apply(SketchPolicyNode* policy,
698 State* state,
699 std::mt19937* rand_gen) const {
700 for (size_t stage_id = 0; stage_id < (*state)->stages.size(); ++stage_id) {
701 const Stage& stage = (*state)->stages[stage_id];
702 // Skip the inlined stage and placeholder stage
703 if (stage->compute_at == ComputeAtKind::kInlined || stage->op_type == StageKind::kPlaceholder) {
704 continue;
705 }
706
707 // Try to fuse and vectorize the space iterators in the inner most tile
708 int64_t cum_length_prod = 1;
709
710 int num_fusible = 0;
711 while (num_fusible < static_cast<int>(stage->iters.size())) {
712 int iter_id = static_cast<int>(stage->iters.size()) - 1 - num_fusible;
713 // Stop if this iterator has been a compute at attach point
714 if ((*state)->attach_map->iter_to_attached_stages.count(std::make_pair(stage_id, iter_id))) {
715 break;
716 }
717
718 const Iterator& it = stage->iters[iter_id];
719 // Stop if we meet a reduce iterator or annotated iterator
720 if (it->iter_kind == IteratorKind::kReduction ||
721 it->annotation != IteratorAnnotation::kNone) {
722 break;
723 }
724
725 // Stop if the memory access is not continuous (vectorizable)
726 // Note: The check is too hard, so we use heuristic here
727 if (IsTiled(stage) && num_fusible != 0) {
728 // If the stage is tiled, then the memory access must not be continuous
729 // for the innermost two iterators
730 break;
731 }
732
733 cum_length_prod *= GetExtent(it);
734 if (cum_length_prod > GetIntParam(policy->params, SketchParamKey::max_vectorize_size)) {
735 break;
736 }
737
738 num_fusible++;
739 }
740
741 if (num_fusible > 1) {
742 // Select a random range to fuse
743 num_fusible = 1 + (*rand_gen)() % (num_fusible - 1);
744 }
745
746 if (num_fusible == 1) {
747 state->vectorize(stage_id, stage->iters.back());
748 } else if (num_fusible > 1) {
749 Array<Iterator> to_fuse(stage->iters.end() + (-num_fusible), stage->iters.end());
750 state->vectorize(stage_id, state->fuse(stage_id, to_fuse));
751 }
752 }
753
754 return ResultKind::kValid;
755}

757PopulationGenerationRule::ResultKind InitThreadBind::Apply(SketchPolicyNode* policy, State* state,
758 std::mt19937* rand_gen) const {
759 // Collect all stages that are roots of stages that perform multi-level tiling.
760 std::set<int> multi_level_tiling_root_set;
761 for (size_t stage_id = 0; stage_id < (*state)->stages.size(); ++stage_id) {
762 if (NeedsMultilevelTiling(policy->search_task, *state, stage_id)) {
763 const Stage& stage = (*state)->stages[stage_id];
764 if (stage->compute_at == ComputeAtKind::kInlined) {
765 continue;
766 } else if (stage->compute_at != ComputeAtKind::kIter) {
767 // This stage is not multi-level tiled,
768 // so it must be produced by RuleCrossThreadReduction.
769 ICHECK(HasCrossThreadReduction(*state, stage_id));
770 } else {
771 const auto res = (*state)->attach_map->stage_to_attach_iter.find(stage_id);
772 ICHECK(res != (*state)->attach_map->stage_to_attach_iter.end());
773 multi_level_tiling_root_set.insert(res->second.first);
774 }
775 }
776 }
777
778 *state = policy->search_task->compute_dag.InferBound(*state);
779
780 for (int stage_id = (*state)->stages.size() - 1; stage_id >= 0; --stage_id) {
781 const Stage& stage = (*state)->stages[stage_id];
782
783 if (stage->compute_at == ComputeAtKind::kInlined || stage->op_type == StageKind::kPlaceholder) {
784 continue;
785 }
786
787 // Deal with the cross-thread reduction generated by RuleCrossThreadReduction
788 if (HasCrossThreadReduction(*state, stage_id)) {
789 if (stage->compute_at != ComputeAtKind::kRoot) {
790 continue;
791 }
792
793 Iterator fused_it;
794 *state = std::move(FuseAllOuterSpaceIterators(*state, stage_id, &fused_it));
795 state->bind(stage_id, fused_it, IteratorAnnotation::kBlockX);
796 continue;
797 }
798
799 // Skip if this stage has already been annotaed with threadIdx.x
800 if (HasAnnotatedIter(stage, IteratorAnnotation::kThreadX)) {
801 continue;
802 }
803
804 if (stage->compute_at == ComputeAtKind::kRoot) {
805 // This stage has not been tiled, but in GPU schedule, we must tile the root stage
806 // to do thread binding
807 if (!multi_level_tiling_root_set.count(stage_id)) {
808 Iterator fused_it;
809 *state = FuseAllOuterSpaceIterators(*state, stage_id, &fused_it);
810
811 if (GetExtent(fused_it) <= policy->search_task->hardware_params->warp_size) {
812 state->bind(stage_id, fused_it, IteratorAnnotation::kThreadX);
813 } else {
814 // Set threadIdx.x = default_warp_size by default.
815 // The later EvolutionarySearch will try more possibility
816 const auto& split_its = state->split(
817 stage_id, fused_it, {Integer(policy->search_task->hardware_params->warp_size)});
818 state->bind(stage_id, split_its[0], IteratorAnnotation::kBlockX);
819 state->bind(stage_id, split_its[1], IteratorAnnotation::kThreadX);
820 }
821 continue;
822 }
823
824 // Otherwise, this is a tiled root stage, we assume it should be tiled with 3 space level
825 // in the outer iterators.
826 // The remaining part deals with the thread binding for multi-level tiled stages
827 auto pop = stage->op.as<te::ComputeOpNode>();
828 std::vector<Iterator> to_fuse;
829 int total_space_extent = 1;
830 for (const auto& i : pop->root_iter_vars()) {
831 ICHECK(i->dom.defined());
832 const auto& pint = i->dom->extent.as<IntImmNode>();
833 ICHECK(pint);
834 total_space_extent *= pint->value;
835 }
836
837 bool check_min_thread_extent = true;
838 // If the total space extent is too small, disable the check of minimal thread extent
839 if (total_space_extent <= policy->search_task->hardware_params->warp_size * 2) {
840 check_min_thread_extent = false;
841 }
842
843 // Fuse the outermost space tile as blockIdx
844 for (size_t i = 0; i < pop->axis.size(); i++) {
845 const auto& it = (*state)->stages[stage_id]->iters[i];
846 // There may be some iterators that are marked with no split, stop if reaches next
847 // tiling level
848 if (!StrEndsWith(it->name, ".0")) {
849 break;
850 }
851 to_fuse.push_back(it);
852 }
853 const auto& blockidx_it = state->fuse(stage_id, to_fuse);
854 state->bind(stage_id, blockidx_it, IteratorAnnotation::kBlockX);
855
856 // Fuse the second outermost space tile as vthread
857 to_fuse.clear();
858 for (size_t i = 1; i < pop->axis.size() + 1; i++) {
859 const auto& it = (*state)->stages[stage_id]->iters[i];
860 // There may be some iterators that are marked with no split, stop if reaches next
861 // tiling level
862 if (!StrEndsWith(it->name, ".1")) {
863 break;
864 }
865 to_fuse.push_back((*state)->stages[stage_id]->iters[i]);
866 }
867 const auto& vthread_it = state->fuse(stage_id, to_fuse);
868 if (GetExtent(vthread_it) > policy->search_task->hardware_params->max_vthread_extent) {
869 return ResultKind::kInvalid;
870 }
871 state->bind(stage_id, vthread_it, IteratorAnnotation::kVThread);
872
873 // Fuse the third outermost space tile as threadIdx
874 to_fuse.clear();
875 for (size_t i = 2; i < pop->axis.size() + 2; i++) {
876 const auto& it = (*state)->stages[stage_id]->iters[i];
877 // There may be some iterators that are marked with no split, stop if reaches next
878 // tiling level
879 if (!StrEndsWith(it->name, ".2")) {
880 break;
881 }
882 to_fuse.push_back((*state)->stages[stage_id]->iters[i]);
883 }
884 const auto& threadidx_it = state->fuse(stage_id, to_fuse);
885 if (check_min_thread_extent &&
886 GetExtent(threadidx_it) < policy->search_task->hardware_params->warp_size) {
887 return ResultKind::kInvalid;
888 }
889 state->bind(stage_id, threadidx_it, IteratorAnnotation::kThreadX);
890 } else if (stage->compute_at == ComputeAtKind::kIter &&
891 StrEndsWith(stage->op->name, ".shared")) {
892 // Do cooperative fetching for the cache read stage.
893 // Get spatial_split_step_ids from the root stage
894 const auto& it = (*state)->attach_map->stage_to_attach_iter.find(stage_id);
895 ICHECK(it != (*state)->attach_map->stage_to_attach_iter.end());
896 Array<Integer> spatial_split_step_ids = GetSpatialSplitStepIds(*state, it->second.first);
897
898 // Fuse all iterators to do cooperative fetching
899 Iterator fused = state->fuse(stage_id, (*state)->stages[stage_id]->iters);
900 // Split out an extra iterator for vectorization
901 // The later EvolutionarySearch will try more possibility
902 const auto& iters0 = state->split(stage_id, fused, {Integer(1)});
903 state->vectorize(stage_id, iters0[1]);
904 // Follow split to keep a same thread extent with the root stage
905 const auto& iters1 =
906 state->follow_fused_split(stage_id, iters0[0], spatial_split_step_ids, 1, true);
907 state->bind(stage_id, iters1[1], IteratorAnnotation::kThreadX);
908 }
909 }
910 return ResultKind::kValid;
911}

913PopulationGenerationRule::ResultKind MutateTileSize::Apply(SketchPolicyNode* policy, State* state,
914 std::mt19937* rand_gen) const {
915 int max_innermost_split_factor =
916 GetIntParam(policy->params, SketchParamKey::max_innermost_split_factor);
917
918 // Extract all SplitStep
919 std::vector<size_t> split_step_ids;
920 for (size_t i = 0; i < (*state)->transform_steps.size(); ++i) {
921 if (auto ps = (*state)->transform_steps[i].as<SplitStepNode>()) {
922 if (!ps->extent.defined() || !ps->extent.value()->IsInstance<IntImmNode>()) {
923 continue;
924 }
925 auto innermost_factor = ps->lengths.back().value_or(max_innermost_split_factor + 1);
926 if (GetIntImm(innermost_factor) <= max_innermost_split_factor) {
927 split_step_ids.push_back(i);
928 }
929 }
930 }
931 if (split_step_ids.empty()) {
932 // No tile size could be mutated.
933 return ResultKind::kInvalid;
934 }
935
936 // Select a SplitStep with extent larger than one to mutate.
937 int retry_ct = 0;
938 int64_t extent = 1;
939 int step_id;
940 const SplitStepNode* ps;
941
942 do {
943 step_id = split_step_ids[(*rand_gen)() % split_step_ids.size()];
944 ps = (*state)->transform_steps[step_id].as<SplitStepNode>();
945 ICHECK(ps != nullptr);
946 extent = GetIntImm(ps->extent.value());
947 retry_ct += 1;
948 } while (retry_ct < static_cast<int>(split_step_ids.size()) << 2 && (extent == 1 || extent == 0));
949
950 if (extent <= 1) {
951 // Cannot find a step with extent larger than one.
952 return ResultKind::kInvalid;
953 }
954
955 // Fetch the current tile sizes.
956 std::vector<int> lengths(ps->lengths.size() + 1, 1);
957 for (int i = 0; i < static_cast<int>(ps->lengths.size()); ++i) {
958 lengths[i + 1] = GetIntImm(ps->lengths[i].value());
959 }
960 lengths[0] = extent / ElementProduct(lengths);
961
962 // Random permute the tile size order.
963 std::vector<int> random_perm;
964 RandomPermutation(lengths.size(), &random_perm, rand_gen);
965
966 // Try to divide a factor from one tile size and multiple it to another.
967 for (size_t i = 0; i < random_perm.size(); ++i) {
968 size_t src_idx = random_perm[i];
969 int length = lengths[src_idx];
970 if (length <= 1) {
971 continue;
972 }
973
974 // Divide one factor from lengths[src_idx] and multiply it to lengths[dst_idx]
975 size_t dst_idx = random_perm[(i + 1) % random_perm.size()];
976 const std::vector<int>& factors = policy->split_memo.GetFactors(length);
977 ICHECK_GE(factors.size(), 1);
978
979 int divide_factor;
980 if (dst_idx == lengths.size() - 1) {
981 // Maintain the restriction of hardware_params.max_innermost_split_factor.
982 int max_factor_index = static_cast<int>(factors.size()) - 1;
983 for (; max_factor_index >= 1; max_factor_index--) {
984 if (factors[max_factor_index] * lengths[dst_idx] <= max_innermost_split_factor) {
985 break;
986 }
987 }
988 if (max_factor_index == 0) {
989 // Failed on this dst_idx, try next one.
990 continue;
991 }
992 divide_factor = factors[1 + (*rand_gen)() % (max_factor_index)];
993 } else {
994 divide_factor = factors[1 + (*rand_gen)() % (factors.size() - 1)];
995 }
996
997 // Divide one factor from lengths[src_idx] and multiply it to lengths[dst_idx].
998 Array<Integer> new_lengths;
999 for (size_t j = 1; j < lengths.size(); ++j) {
1000 if (j == src_idx) {
1001 new_lengths.push_back(Integer(lengths[j] / divide_factor));
1002 } else if (j == dst_idx) {
1003 new_lengths.push_back(Integer(lengths[j] * divide_factor));
1004 } else {
1005 new_lengths.push_back(Integer(lengths[j]));
1006 }
1007 }
1008
1009 ICHECK_LE(GetIntImm(new_lengths.back()), max_innermost_split_factor);
1010
1011 StateNode* pstate = state->CopyOnWrite();
1012 pstate->transform_steps.Set(
1013 step_id, SplitStep(ps->stage_id, ps->iter_id, ps->extent,
1014 Array<Optional<Integer>>(new_lengths.begin(), new_lengths.end()),
1015 ps->inner_to_outer));
1016 return ResultKind::kValid;
1017 }
1018 return ResultKind::kInvalid;
1019}

1022 std::mt19937* rand_gen) const {
1023 // Extract all auto_unroll_max_step pragma steps.

1026 if (auto ps = (*state)->transform_steps[i].as<PragmaStepNode>()) {

1041 auto ps = (*state)->transform_steps[step_id].as<PragmaStepNode>();

1056PopulationGenerationRule::ResultKind MutateComputeLocation::Apply(SketchPolicyNode* policy,
1057 State* state,
1058 std::mt19937* rand_gen) const {
1059 if (GetIntParam(policy->params, SketchParamKey::disable_change_compute_location)) {
1060 return ResultKind::kInvalid;
1061 }
1062
1063 // Extract all compute_at steps.
1064 std::vector<int> compute_at_steps;
1065 for (size_t s = 0; s < (*state)->transform_steps.size(); ++s) {
1066 if (auto ps = (*state)->transform_steps[s].as<ComputeAtStepNode>()) {
1067 int stage_inc = GetTargetStageIDInState(*state, s) - ps->stage_id;
1068
1069 if (IsTiled((*state)->stages[ps->stage_id + stage_inc])) {
1070 continue;
1071 }
1072
1073 if (NeedsMultilevelTiling(policy->search_task, *state, ps->stage_id + stage_inc)) {
1074 continue;
1075 }
1076 compute_at_steps.push_back(s);
1077 }
1078 }
1079 if (compute_at_steps.empty()) {
1080 return ResultKind::kInvalid;
1081 }
1082
1083 // Randomly pick one step
1084 size_t step_id = compute_at_steps[(*rand_gen)() % compute_at_steps.size()];
1085 auto ps = (*state)->transform_steps[step_id].as<ComputeAtStepNode>();
1086 int stage_inc = GetTargetStageIDInState(*state, step_id) - ps->stage_id;
1087 ICHECK(ps != nullptr);
1088
1089 // Randomly pick a new computation location
1090 std::vector<std::pair<int, int>> candidates =
1091 GetComputeLocationCandidates(policy->search_task, *state, ps->stage_id + stage_inc);
1092 if (candidates.empty()) {
1093 return ResultKind::kInvalid;
1094 }
1095 int choice = (*rand_gen)() % (candidates.size());
1096 int new_compute_at_stage_id = candidates[choice].first;
1097 int new_compute_at_iter_id = candidates[choice].second;
1098
1099 // Replay a new state.
1100 State tmp_s = policy->search_task->compute_dag->init_state;
1101 for (size_t s = 0; s < (*state)->transform_steps.size(); ++s) {
1102 if (s == step_id) {
1103 tmp_s.CopyOnWrite()->transform_steps.push_back(
1104 ComputeAtStep(ps->stage_id, new_compute_at_stage_id - stage_inc, new_compute_at_iter_id));
1105 } else {
1106 tmp_s.CopyOnWrite()->transform_steps.push_back((*state)->transform_steps[s]);
1107 }
1108 try {
1109 StepApplyToState(tmp_s->transform_steps.back(), &tmp_s, policy->search_task->compute_dag);
1110 } catch (Error& e) {
1111 return ResultKind::kInvalid;
1112 }
1113 }
1114
1115 *state = tmp_s;
1116 return ResultKind::kValid;
1117}

1119PopulationGenerationRule::ResultKind MutateParallel::Apply(SketchPolicyNode* policy, State* state,
1120 std::mt19937* rand_gen) const {
1121 // This mutation rule only focuses on a case that parallel was added to
1122 // the outermost loop and the loop is generated by fusing other loops.
1123 // In short, we mutate the fusion step before the parallel step.
1124
1125 // Extract all parallel steps.
1126 std::vector<int> parallel_steps;
1127 for (size_t s = 0; s < (*state)->transform_steps.size(); ++s) {
1128 auto ps = (*state)->transform_steps[s].as<AnnotationStepNode>();
1129 if (!ps || ps->annotation != IteratorAnnotation::kParallel) {
1130 continue;
1131 }
1132
1133 // Skip non-outermost loop or the parallel step without fusion beforehand.
1134 if (ps->iter_id != 0 || s == 0 || !(*state)->transform_steps[s - 1].as<FuseStepNode>()) {
1135 continue;
1136 }
1137 auto fuse_step = (*state)->transform_steps[s - 1].as<FuseStepNode>();
1138 if (fuse_step->fused_ids[0] != 0) {
1139 continue;
1140 }
1141
1142 parallel_steps.push_back(s);
1143 }
1144 if (parallel_steps.empty()) {
1145 return ResultKind::kInvalid;
1146 }
1147
1148 // Randomly pick one parallel step.
1149 size_t step_id = parallel_steps[(*rand_gen)() % parallel_steps.size()];
1150
1151 // Replay a new state until the picked fuse step.
1152 State tmp_s = policy->search_task->compute_dag->init_state;
1153 for (size_t s = 0; s < step_id - 1; ++s) {
1154 const auto& step = (*state)->transform_steps[s];
1155 tmp_s.CopyOnWrite()->transform_steps.push_back(step);
1156 StepApplyToState(step, &tmp_s, policy->search_task->compute_dag);
1157 }
1158
1159 // Compute all possible fusion granularities
1160 auto fuse_step = (*state)->transform_steps[step_id - 1].as<FuseStepNode>();
1161 int stage_id = fuse_step->stage_id;
1162 const Stage& stage = tmp_s->stages[stage_id];
1163 size_t max_fusable_iter_id;
1164 for (max_fusable_iter_id = 0; max_fusable_iter_id < stage->iters.size(); ++max_fusable_iter_id) {
1165 const Iterator& it = stage->iters[max_fusable_iter_id];
1166 if (it->iter_kind == IteratorKind::kReduction || it->annotation != IteratorAnnotation::kNone) {
1167 break;
1168 }
1169
1170 if (tmp_s->attach_map->iter_to_attached_stages.count(
1171 std::make_pair(stage_id, max_fusable_iter_id))) {
1172 break;
1173 }
1174 }
1175
1176 if (max_fusable_iter_id == 0) {
1177 return ResultKind::kInvalid;
1178 }
1179
1180 // Randomly pick one granularity
1181 int fuse_to_iter_id = (*rand_gen)() % max_fusable_iter_id + 1;
1182 Array<Integer> fused_ids;
1183 for (int i = 0; i < fuse_to_iter_id; ++i) {
1184 fused_ids.push_back(i);
1185 }
1186 int iter_offset = fuse_step->fused_ids.back()->value - fused_ids.back()->value;
1187 if (iter_offset == 0) {
1188 return ResultKind::kInvalid;
1189 }
1190
1191 // Replay the mutated fused and annotation step.
1192 auto new_fuse_step = FuseStep(stage_id, fused_ids);
1193 tmp_s.CopyOnWrite()->transform_steps.push_back(new_fuse_step);
1194 StepApplyToState(new_fuse_step, &tmp_s, policy->search_task->compute_dag);
1195 tmp_s.CopyOnWrite()->transform_steps.push_back((*state)->transform_steps[step_id]);
1196 StepApplyToState((*state)->transform_steps[step_id], &tmp_s, policy->search_task->compute_dag);
1197
1198 // Replay the rest steps.
1199 for (size_t s = step_id + 1; s < (*state)->transform_steps.size(); ++s) {
1200 auto step = (*state)->transform_steps[s];
1201 if (step->stage_id == stage_id) {
1202 // Since we changed the loop structure, iter ID in later steps to the same stage
1203 // has to be adjusted.
1204 if (auto ps = step.as<AnnotationStepNode>()) {
1205 if (ps->iter_id == 0) {
1206 step = AnnotationStep(ps->stage_id, 0, ps->annotation);
1207 } else {
1208 ICHECK_LE(ps->iter_id + iter_offset, tmp_s->stages[stage_id]->iters.size());
1209 step = AnnotationStep(ps->stage_id, ps->iter_id + iter_offset, ps->annotation);
1210 }
1211 } else if (auto ps = step.as<PragmaStepNode>()) {
1212 if (ps->iter_id == 0) {
1213 step = PragmaStep(ps->stage_id, 0, ps->pragma_type);
1214 } else {
1215 ICHECK_LE(ps->iter_id + iter_offset, tmp_s->stages[stage_id]->iters.size());
1216 step = PragmaStep(ps->stage_id, ps->iter_id + iter_offset, ps->pragma_type);
1217 }
1218 } else {
1219 return ResultKind::kInvalid;
1220 }
1221 }
1222 if (IsStageNumberChangingStep(step)) {
1223 // For these steps, we have to update stage_id because these steps will make stage_id
1224 // out-dated. But here we just simply give up this mutation for simplicity.
1225 // This is not an issue because this will never happend in normal cases where all these steps
1226 // are before parallel steps.
1227 return ResultKind::kInvalid;
1228 }
1229 tmp_s.CopyOnWrite()->transform_steps.push_back(step);
1230 try {
1231 StepApplyToState(tmp_s->transform_steps.back(), &tmp_s, policy->search_task->compute_dag);
1232 } catch (Error& e) {
1233 return ResultKind::kInvalid;
1234 }
1235 }
1236
1237 *state = tmp_s;
1238 return ResultKind::kValid;
1239}
