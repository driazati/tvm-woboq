211inline Block ConcreteScheduleNode::Get(const BlockRV& block_rv) const {
212 StmtSRef sref = this->GetSRef(block_rv);
213 const BlockNode* block = TVM_SREF_TO_BLOCK(sref);
214 return GetRef<Block>(block);
215}

217inline For ConcreteScheduleNode::Get(const LoopRV& loop_rv) const {
218 StmtSRef sref = this->GetSRef(loop_rv);
219 const ForNode* loop = TVM_SREF_TO_FOR(sref);
220 return GetRef<For>(loop);
221}

223inline PrimExpr ConcreteScheduleNode::Get(const ExprRV& expr_rv) const {
224 PrimExpr transformed = Substitute(expr_rv, [this](const Var& var) -> Optional<PrimExpr> {
225 auto it = this->symbol_table_.find(var);
226 if (it == this->symbol_table_.end()) {
227 LOG(FATAL) << "IndexError: Cannot find corresponding ExprRV: " << var;
228 }
229 const ObjectRef& obj = (*it).second;
230 const auto* int_imm = TVM_TYPE_AS(obj, IntImmNode);
231 return Integer(int_imm->value);
232 });
233 return this->analyzer_->Simplify(transformed);
234}
235
236inline bool ConcreteScheduleNode::HasBlock(const BlockRV& block_rv) const {
237 auto it = this->symbol_table_.find(block_rv);
238 if (it == this->symbol_table_.end()) {
239 return false;
240 }
241 const ObjectRef& obj = (*it).second;
242 const auto* sref = obj.as<StmtSRefNode>();
243 if (sref == nullptr || sref->stmt == nullptr) {
244 return false;
245 }
246 return true;
247}
248
249inline StmtSRef ConcreteScheduleNode::GetSRef(const BlockRV& block_rv) const {
250 auto it = this->symbol_table_.find(block_rv);
251 if (it == this->symbol_table_.end()) {
252 LOG(FATAL) << "IndexError: Cannot find corresponding BlockRV: " << block_rv;
253 }
254 const ObjectRef& obj = (*it).second;
255 const auto* sref = obj.as<StmtSRefNode>();
256 if (sref == nullptr) {
257 LOG(FATAL) << "ValueError: BlockRV's corresponding type is invalid: "
258 << (obj.defined() ? obj->GetTypeKey() : "None");
259 }
260 if (sref->stmt == nullptr) {
261 LOG(FATAL) << "ValueError: The block no longer exists in the IRModule";
262 }
263 return GetRef<StmtSRef>(sref);
264}
265
266inline StmtSRef ConcreteScheduleNode::GetSRef(const LoopRV& loop_rv) const {
267 static StmtSRef inline_mark = StmtSRef::InlineMark();
268 static StmtSRef root_mark = StmtSRef::RootMark();
269 auto it = this->symbol_table_.find(loop_rv);
270 if (it == this->symbol_table_.end()) {
271 LOG(FATAL) << "IndexError: Cannot find corresponding LoopRV: " << loop_rv;
272 }
273 const ObjectRef& obj = (*it).second;
274 if (obj.same_as(inline_mark)) {
275 return inline_mark;
276 }
277 if (obj.same_as(root_mark)) {
278 return root_mark;
279 }
280 const auto* sref = obj.as<StmtSRefNode>();
281 if (sref == nullptr) {
282 LOG(FATAL) << "ValueError: LoopRV's corresponding type is invalid: "
283 << (obj.defined() ? obj->GetTypeKey() : "None");
284 }
285 if (sref->stmt == nullptr) {
286 LOG(FATAL) << "ValueError: The loop no longer exists in the IRModule";
287 }
288 return GetRef<StmtSRef>(sref);
289}

291template <class T>
292inline Array<StmtSRef> GetSRefsHelper(const ConcreteScheduleNode* sch, const Array<T>& rvs) {
293 Array<StmtSRef> result;
294 result.reserve(rvs.size());
295 for (const T& rv : rvs) {
296 result.push_back(sch->GetSRef(rv));
297 }
298 return result;
299}
300
301inline Array<StmtSRef> ConcreteScheduleNode::GetSRefs(const Array<BlockRV>& rvs) const {
302 return GetSRefsHelper(this, rvs);
303}
304
305inline Array<StmtSRef> ConcreteScheduleNode::GetSRefs(const Array<LoopRV>& rvs) const {
306 return GetSRefsHelper(this, rvs);
307}

309/******** Adding/Removing elements in the symbol table ********/
310
311template <class T>
312inline Array<T> ConcreteScheduleNode::CreateRV(const Array<StmtSRef>& srefs) {
313 Array<T> result;
314 result.reserve(srefs.size());
315 for (const StmtSRef& sref : srefs) {
316 T rv;
317 this->symbol_table_.Set(rv, sref);
318 result.push_back(rv);
319 }
320 return result;
321}
322
323template <class T>
324inline T ConcreteScheduleNode::CreateRV(const StmtSRef& sref) {
325 T rv;
326 this->symbol_table_.Set(rv, sref);
327 return std::move(rv);
328}
329
330inline ExprRV ConcreteScheduleNode::CreateRV(int64_t value) {
331 Var rv("v" + std::to_string(this->symbol_table_.size() + 1), DataType::Int(32));
332 this->symbol_table_.Set(rv, Integer(static_cast<int32_t>(value)));
333 return std::move(rv);
334}
335
336inline Array<ExprRV> ConcreteScheduleNode::CreateRV(const std::vector<int64_t>& value) {
337 Array<ExprRV> results;
338 results.reserve(value.size());
339 for (int64_t v : value) {
340 results.push_back(CreateRV(v));
341 }
342 return results;
343}
344
345inline void ConcreteScheduleNode::RemoveFromSymbolTable(const ObjectRef& obj) {
346 auto it = this->symbol_table_.find(obj);
347 if (it != this->symbol_table_.end()) {
348 this->symbol_table_.erase(obj);
349 } else {
350 LOG(FATAL) << "IndexError: Cannot find the object in the symbol table: " << obj;
351 throw;
352 }
353}
