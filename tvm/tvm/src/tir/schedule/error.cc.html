24String ScheduleError::RenderReport(const String& primitive) const {
25 IRModule mod = this->mod();
26 std::ostringstream os;
27
28 // get locations of interest
29 Array<ObjectRef> locs = LocationsOfInterest();
30 std::unordered_map<ObjectRef, String, ObjectPtrHash, ObjectPtrEqual> loc_obj_to_name;
31 int n_locs = locs.size();
32 std::string msg = DetailRenderTemplate();
33 if (n_locs > 0) {
34 for (int i = 0; i < n_locs; ++i) {
35 std::string name = locs[i]->GetTypeKey() + '#' + std::to_string(i);
36 std::string src = "{" + std::to_string(i) + "}";
37 for (size_t pos; (pos = msg.find(src)) != std::string::npos;) {
38 msg.replace(pos, src.length(), name);
39 }
40 loc_obj_to_name.emplace(locs[i], std::move(name));
41 }
42 }
43
44 // print IR module
45 runtime::TypedPackedFunc<std::string(Stmt)> annotate =
46 runtime::TypedPackedFunc<std::string(Stmt)>(
47 [&loc_obj_to_name](const Stmt& expr) -> std::string {
48 auto it = loc_obj_to_name.find(Downcast<ObjectRef>(expr));
49 if (it == loc_obj_to_name.end()) {
50 return "";
51 }
52 return it->second;
53 });
54
55 os << "ScheduleError: An error occurred in the schedule primitive '" << primitive
56 << "'.\n\nThe IR with diagnostic is:\n"
57 << AsTVMScriptWithDiagnostic(mod, "T", false, annotate);
58
59 // print error message
60 os << "Error message: " << msg;
61 return os.str();
62}
