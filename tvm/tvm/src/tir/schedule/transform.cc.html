27Block WithAnnotation(const BlockNode* block, const String& attr_key, const ObjectRef& attr_value) {
28 Map<String, ObjectRef> annotations = block->annotations;
29 annotations.Set(attr_key, attr_value);
30 ObjectPtr<BlockNode> new_block = make_object<BlockNode>(*block);
31 new_block->annotations = std::move(annotations);
32 return Block(new_block);
33}

35/******** Buffer Related ********/
36Buffer WithScope(const Buffer& buffer, const String& scope) {
37 ObjectPtr<BufferNode> new_buffer = make_object<BufferNode>(*buffer.get());
38 ObjectPtr<VarNode> new_var = make_object<VarNode>(*buffer->data.get());
39 const auto* ptr_type = TVM_TYPE_AS(ptr_type, buffer->data->type_annotation, PointerTypeNode);
40 new_var->type_annotation = PointerType(ptr_type->element_type, scope);
41 new_buffer->data = Var(new_var->name_hint + "_" + scope, new_var->type_annotation);
42 new_buffer->name = buffer->name + "_" + scope;
43 return Buffer(new_buffer);
44}

46Array<BufferRegion> ReplaceBuffer(Array<BufferRegion> regions, const Buffer& source,
47 const Buffer& target) {
48 regions.MutateByApply([&source, &target](BufferRegion region) -> BufferRegion {
49 if (region->buffer.same_as(source)) {
50 ObjectPtr<BufferRegionNode> n = make_object<BufferRegionNode>(*region.get());
51 n->buffer = target;
52 return BufferRegion(n);
53 }
54 return region;
55 });
56 return regions;
57}

59Array<MatchBufferRegion> ReplaceBuffer(Array<MatchBufferRegion> match_buffers, const Buffer& source,
60 const Buffer& target) {
61 match_buffers.MutateByApply([&source,
62 &target](MatchBufferRegion match_buffer) -> MatchBufferRegion {
63 if (match_buffer->source->buffer.same_as(source)) {
64 ObjectPtr<MatchBufferRegionNode> n = make_object<MatchBufferRegionNode>(*match_buffer.get());
65 n->source = BufferRegion(target, n->source->region);
66 return MatchBufferRegion(n);
67 }
68 return match_buffer;
69 });
70 return match_buffers;
71}

73Array<BufferRegion> ReplaceBufferRegion(Array<BufferRegion> regions, const Buffer& source_buffer,
74 const BufferRegion& target) {
75 regions.MutateByApply([&source_buffer, &target](const BufferRegion& region) -> BufferRegion {
76 if (region->buffer.same_as(source_buffer)) {
77 return target;
78 }
79 return region;
80 });
81 return regions;
82}

84Array<MatchBufferRegion> ReplaceBufferRegion(Array<MatchBufferRegion> match_buffers,
85 const Buffer& source_buffer,
86 const BufferRegion& target) {
87 match_buffers.MutateByApply([&source_buffer, &target](
88 const MatchBufferRegion& match_buffer) -> MatchBufferRegion {
89 if (match_buffer->source->buffer.same_as(source_buffer)) {
90 ObjectPtr<MatchBufferRegionNode> n = make_object<MatchBufferRegionNode>(*match_buffer.get());
91 n->source = target;
92 return MatchBufferRegion(n);
93 }
94 return match_buffer;
95 });
96 return match_buffers;
97}

121MatchBufferRegion ReplaceBufferMutator::VisitMatchBufferRegion(
122 const MatchBufferRegion& match_buffer) {
123 auto it = buffer_var_map_.find(match_buffer->source->buffer->data.get());
124 if (it != buffer_var_map_.end()) {
125 return MatchBufferRegion(match_buffer->buffer,
126 BufferRegion(it->second, match_buffer->source->region));
127 } else {
128 return match_buffer;
129 }
130}
131
132Stmt ReplaceBufferMutator::VisitStmt_(const BlockNode* block) {
133 // To reduce the number of blocks in block sref reuse map, we check whether the block is really
134 // mutated (i.e., the old buffer appears in the block). If so, we return the block after
135 // mutation. Otherwise we just return the original block.
136
137 auto f_mutate_match_buffer = [this](const MatchBufferRegion& match_buffer) {
138 return this->VisitMatchBufferRegion(match_buffer);
139 };
140 auto f_mutate_read_write_region = [this](const BufferRegion& buffer_region) {
141 auto it = buffer_var_map_.find(buffer_region->buffer->data.get());
142 return it == buffer_var_map_.end() ? buffer_region
143 : BufferRegion(it->second, buffer_region->region);
144 };
145 auto f_mutate_alloc_buffers = [this](const Buffer& buffer) {
146 auto it = buffer_var_map_.find(buffer->data.get());
147 return it == buffer_var_map_.end() ? buffer : it->second;
148 };
149
150 // Step 1. Mutate `match_buffers`. If an old buffer appears as a source of MatchBufferRegion,
151 Array<MatchBufferRegion> match_buffers = MutateArray(block->match_buffers, f_mutate_match_buffer);
152 // Step 2. Mutate the read/write region.
153 Array<BufferRegion> reads = MutateArray(block->reads, f_mutate_read_write_region);
154 Array<BufferRegion> writes = MutateArray(block->writes, f_mutate_read_write_region);
155 // Step 3. Mutate `alloc_buffers` for the old buffer allocated in this block.
156 Array<Buffer> alloc_buffers = MutateArray(block->alloc_buffers, f_mutate_alloc_buffers);
157 // Step 4. Recursively mutate the block.
158 Block mutated_block = Downcast<Block>(StmtMutator::VisitStmt_(block));
159
160 if (mutated_block.get() == block && reads.same_as(mutated_block->reads) &&
161 writes.same_as(mutated_block->writes) &&
162 alloc_buffers.same_as(mutated_block->alloc_buffers) &&
163 match_buffers.same_as(mutated_block->match_buffers)) {
164 return GetRef<Block>(block);
165 } else {
166 ObjectPtr<BlockNode> n = CopyOnWrite(mutated_block.get());
167 n->reads = std::move(reads);
168 n->writes = std::move(writes);
169 n->alloc_buffers = std::move(alloc_buffers);
170 n->match_buffers = std::move(match_buffers);
171
172 Block new_block(n);
173 if (block_sref_reuse_ != nullptr) {
174 block_sref_reuse_->Set(GetRef<Block>(block), new_block);
175 }
176 return std::move(new_block);
177 }
178}

180/******** Block Removal ********/
181
182void LeafBlockRemovalPlan(const ScheduleState& self, const StmtSRef& leaf_block_sref,
183 Stmt* src_stmt, Stmt* tgt_stmt) {
184 class OnlyLeafError : public ScheduleError {
185 public:
186 explicit OnlyLeafError(IRModule mod, Block leaf_block, Block scope_root)
187 : mod_(mod), leaf_block_(leaf_block), scope_root_(scope_root) {}
188
189 String FastErrorString() const final {
190 return "ScheduleError: Cannot remove the only leaf in the scope";
191 }
192
193 String DetailRenderTemplate() const final {
194 return "Block {0} is the only leaf in the scope {1}, which cannot be removed; Otherwise the "
195 "scope will be empty.";
196 }
197
198 IRModule mod() const final { return mod_; }
199 Array<ObjectRef> LocationsOfInterest() const final { return {leaf_block_, scope_root_}; }
200
201 IRModule mod_;
202 Block leaf_block_;
203 Block scope_root_;
204 };
205
206 // Go upwards until find an ancestor with more than one child
207 const StmtNode* last_stmt = leaf_block_sref->stmt;
208 StmtSRefNode* sref = leaf_block_sref->parent;
209 for (;; last_stmt = sref->stmt, sref = sref->parent) {
210 if (const auto* loop = sref->StmtAs<ForNode>()) {
211 if (const auto* seq = loop->body.as<SeqStmtNode>()) {
212 if (seq->size() > 1) {
213 break;
214 }
215 }
216 } else {
217 // Removal is not done beyond scope-level.
218 // When encountering a block, i.e. the scope root, we simply stop
219 break;
220 }
221 }
222 if (const auto* block = sref->StmtAs<BlockNode>()) {
223 if (const auto* seq = block->body.as<SeqStmtNode>()) {
224 ObjectPtr<BlockNode> n = make_object<BlockNode>(*block);
225 n->body = RemoveFromSeqStmt(GetRef<SeqStmt>(seq), GetRef<Stmt>(last_stmt));
226 *src_stmt = GetRef<Stmt>(block);
227 *tgt_stmt = Stmt(std::move(n));
228 return;
229 }
230 }
231 if (const auto* loop = sref->StmtAs<ForNode>()) {
232 if (const auto* seq = loop->body.as<SeqStmtNode>()) {
233 ObjectPtr<ForNode> n = make_object<ForNode>(*loop);
234 n->body = RemoveFromSeqStmt(GetRef<SeqStmt>(seq), GetRef<Stmt>(last_stmt));
235 *src_stmt = GetRef<Stmt>(loop);
236 *tgt_stmt = Stmt(std::move(n));
237 return;
238 }
239 }
240 ICHECK(sref != nullptr && sref->stmt != nullptr);
241 const auto* leaf_block = TVM_SREF_TO_BLOCK(leaf_block, leaf_block_sref);
242 const auto* scope_block = TVM_SREF_TO_BLOCK(scope_block, sref);
243 throw OnlyLeafError(self->mod, GetRef<Block>(leaf_block), GetRef<Block>(scope_block));
244}

246Optional<LoopRV> TileWithTensorIntrin(const tir::Schedule& sch, const tir::BlockRV& block_rv,
247 const String& intrin_name) {
248 Optional<tir::TensorizeInfo> opt_tensorize_info = GetTensorizeLoopMapping(
249 sch->state(), sch->GetSRef(block_rv), tir::TensorIntrin::Get(intrin_name)->desc);
250 if (!opt_tensorize_info) return NullOpt;
251 const tir::TensorizeInfoNode* info = opt_tensorize_info.value().get();
252 // Construct a mapping from tir loops back to LoopRVs
253 Map<tir::StmtSRef, LoopRV> loop2rv;
254 {
255 Array<LoopRV> loop_rvs = sch->GetLoops(block_rv);
256 for (const LoopRV& loop_rv : loop_rvs) {
257 loop2rv.Set(sch->GetSRef(loop_rv), loop_rv);
258 }
259 }
260 // Split the loops
261 arith::Analyzer analyzer;
262 std::unordered_set<const tir::StmtSRefNode*> inner_loops;
263 std::vector<LoopRV> reorder_suffix;
264 reorder_suffix.resize(info->loop_map.size());
265 for (const auto& kv : info->loop_map) {
266 // Extract mapping (block_loop => desc_loop)
267 const tir::StmtSRef& block_loop_sref = kv.first;
268 const tir::ForNode* block_loop = block_loop_sref->StmtAs<tir::ForNode>();
269 const tir::ForNode* desc_loop = kv.second.get();
270 ICHECK(block_loop != nullptr && desc_loop != nullptr);
271 // Extract the loop extent
272 PrimExpr block_extent = analyzer.Simplify(block_loop->extent);
273 PrimExpr desc_extent = analyzer.Simplify(desc_loop->extent);
274 const auto* int_block_extent = block_extent.as<IntImmNode>();
275 const auto* int_desc_extent = desc_extent.as<IntImmNode>();
276 ICHECK(int_block_extent != nullptr && int_desc_extent != nullptr);
277 // Check divisibility
278 int64_t total = int_block_extent->value;
279 int64_t inner = int_desc_extent->value;
280 ICHECK_EQ(total % inner, 0);
281 // Do the split. Leave the outer extent as NullOpt (unspecified) so that the split factors
282 // can be used for different extents (needed during tuning).
283 Array<LoopRV> split = sch->Split(loop2rv.at(block_loop_sref), {NullOpt, Integer(inner)});
284 ICHECK_EQ(split.size(), 2);
285 inner_loops.insert(sch->GetSRef(split[1]).operator->());
286 // The inner split will be reordered to the loop domain that is tensorized
287 int desc_loop_index = info->desc_loop_indexer.at(GetRef<tir::For>(desc_loop)).IntValue();
288 reorder_suffix[desc_loop_index] = split[1];
289 }
290 // Reorder the loops
291 std::vector<LoopRV> reorder_list;
292 bool meet = false;
293 Array<LoopRV> all_loops = sch->GetLoops(block_rv);
294 for (const LoopRV& loop : all_loops) {
295 if (inner_loops.count(sch->GetSRef(loop).operator->())) {
296 meet = true;
297 } else if (meet) {
298 reorder_list.push_back(loop);
299 }
300 }
301 reorder_list.insert(reorder_list.end(), reorder_suffix.begin(), reorder_suffix.end());
302 sch->Reorder(reorder_list);
303 ICHECK(!reorder_suffix.empty());
304 return reorder_suffix[0];
305}

309/******** BlockBufferAccessSimplifier ********/
310void BlockBufferAccessSimplifier::SimplifyAccessRegion(Array<BufferRegion>* old_access_regions) {
311 auto fmutate = [this](const BufferRegion& buffer_region) {
312 std::vector<Range> new_buffer_region;
313 for (const auto& range : buffer_region->region) {
314 new_buffer_region.push_back(Range::FromMinExtent(analyzer_->Simplify(range->min),
315 analyzer_->Simplify(range->extent)));
316 }
317 return BufferRegion(buffer_region->buffer, new_buffer_region);
318 };
319 (*old_access_regions).MutateByApply(fmutate);
320}
321
322Stmt BlockBufferAccessSimplifier::VisitStmt_(const BlockNode* op) {
323 Block block = Downcast<Block>(arith::IRMutatorWithAnalyzer::VisitStmt_(op));
324 auto* n = block.CopyOnWrite();
325 SimplifyAccessRegion(&n->reads);
326 SimplifyAccessRegion(&n->writes);
327 return std::move(block);
328}
