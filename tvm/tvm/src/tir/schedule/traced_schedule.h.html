19#ifndef TVM_TIR_SCHEDULE_TRACED_SCHEDULE_H_
20#define TVM_TIR_SCHEDULE_TRACED_SCHEDULE_H_
21
22#include "./concrete_schedule.h"
23
24namespace tvm {
25namespace tir {
26
27class TracedScheduleNode : public ConcreteScheduleNode {
28 friend class Schedule;
29
30 protected:
31 Trace trace_;
32
33 public:
34 void VisitAttrs(tvm::AttrVisitor* v) {
35 // `state_` is not visited
36 // `error_render_level_` is not visited
37 // `symbol_table_` is not visited
38 // `analyzer_` is not visitied
39 // `trace_` is not visited
40 }
41
42 ~TracedScheduleNode() = default;
43
44 public:
45 Optional<Trace> trace() const final { return trace_; }
46 Schedule Copy() final;
47
48 public:
49 /******** Schedule: Sampling ********/
50 ExprRV SampleCategorical(const Array<Integer>& candidates, const Array<FloatImm>& probs,
51 Optional<Integer> decision = NullOpt) final;
52 Array<ExprRV> SamplePerfectTile(const LoopRV& loop_rv, int n, int max_innermost_factor,
53 Optional<Array<Integer>> decision = NullOpt) final;
54 LoopRV SampleComputeLocation(const BlockRV& block_rv, Optional<Integer> decision = NullOpt) final;
55 /******** Schedule: Get blocks & loops ********/
56 BlockRV GetBlock(const String& name, const Optional<String>& func_name) final;
57 Array<LoopRV> GetLoops(const BlockRV& block_rv) final;
58 Array<BlockRV> GetChildBlocks(const BlockRV& block_rv) final;
59 Array<BlockRV> GetChildBlocks(const LoopRV& loop_rv) final;
60 Array<BlockRV> GetProducers(const BlockRV& block_rv) final;
61 Array<BlockRV> GetConsumers(const BlockRV& block_rv) final;
62 /******** Schedule: Transform loops ********/
63 LoopRV Fuse(const Array<LoopRV>& loop_rvs, bool preserve_unit_iters) final;
64 Array<LoopRV> Split(const LoopRV& loop_rv, const Array<Optional<ExprRV>>& factor_rvs,
65 bool preserve_unit_iters) final;
66 void Reorder(const Array<LoopRV>& ordered_loop_rvs) final;
67 LoopRV AddUnitLoop(const BlockRV& block_rv) final;
68 LoopRV AddUnitLoop(const LoopRV& loop_rv) final;
69 /******** Schedule: Manipulate ForKind ********/
70 void Parallel(const LoopRV& loop_rv) final;
71 void Vectorize(const LoopRV& loop_rv) final;
72 void Bind(const LoopRV& loop_rv, const String& thread_axis) final;
73 void Unroll(const LoopRV& loop_rv) final;
74 /******** Schedule: Insert cache stages ********/
75 BlockRV CacheRead(const BlockRV& block_rv, int read_buffer_index, const String& storage_scope,
76 const Array<BlockRV> consumer_blocks = {}) final;
77 BlockRV CacheWrite(const BlockRV& block_rv, int write_buffer_index,
78 const String& storage_scope) final;
79 BlockRV ReIndex(const BlockRV& block_rv, int buffer_index,
80 BufferIndexType buffer_index_type) final;
81 /******** Schedule: Compute location ********/
82 void ComputeAt(const BlockRV& block_rv, const LoopRV& loop_rv, bool preserve_unit_loops,
83 int index = -1) final;
84 void ReverseComputeAt(const BlockRV& block_rv, const LoopRV& loop_rv, bool preserve_unit_loops,
85 int index = -1) final;
86 void ComputeInline(const BlockRV& block_rv) final;
87 void ReverseComputeInline(const BlockRV& block_rv) final;
88 /******** Schedule: Reduction ********/
89 BlockRV DecomposeReduction(const BlockRV& block_rv, const LoopRV& loop_rv) final;
90 BlockRV RFactor(const LoopRV& loop_rv, int factor_axis) final;
91 /******** Schedule: Block annotation ********/
92 void StorageAlign(const BlockRV& block_rv, int buffer_index, int axis, int factor,
93 int offset) final;
94 void SetScope(const BlockRV& block_rv, int buffer_index, const String& storage_scope) final;
95 /******** Schedule: Blockize & Tensorize ********/
96 BlockRV Blockize(const LoopRV& loop_rv) final;
97 void Tensorize(const BlockRV& block_rv, const String& intrin) final;
98 void Tensorize(const LoopRV& loop_rv, const String& intrin) final;
99 /******** Schedule: Annotation ********/
100 void Annotate(const LoopRV& loop_rv, const String& ann_key, const ObjectRef& ann_val) override;
101 void Unannotate(const LoopRV& loop_rv, const String& ann_key) override;
102 void Annotate(const BlockRV& block_rv, const String& ann_key, const ObjectRef& ann_val) override;
103 void Unannotate(const BlockRV& block_rv, const String& ann_key) override;
104 /******** Schedule: Layout transformation ********/
105 void TransformLayout(const BlockRV& block_rv, int buffer_index, BufferIndexType buffer_index_type,
106 const IndexMap& index_map) override;
107 void TransformBlockLayout(const BlockRV& block_rv, const IndexMap& index_map) override;
108 void SetAxisSeparator(const BlockRV& block_rv, int buffer_index,
109 BufferIndexType buffer_index_type,
110 const Array<IntImm>& axis_separators) final;
111 /******** Schedule: Padding ********/
112 BlockRV DecomposePadding(const BlockRV& block_rv, const LoopRV& loop_rv) final;
113 void PadEinsum(const BlockRV& block_rv, const Array<Integer>& padding) final;
114 /******** Schedule: Misc ********/
115 void EnterPostproc() final;
116};
117
118} // namespace tir
119} // namespace tvm
120
121#endif // TVM_TIR_SCHEDULE_TRACED_SCHEDULE_H_
