52/*!
53 * \brief A helper macro to convert an sref to the statement it points to,
54 * then check if the downcasting succeeded.
55 * \param Result The result variable, used for checking
56 * \param SRef The SRef to be cast
57 * \param Type The type to be cast to, can be Block or For
58 */
59#define TVM_SREF_AS_OR_ERR(Result, SRef, Type) \
60 SRef->StmtAs<Type>(); \
61 ICHECK(Result)
62
63/*!
64 * \brief A helper macro to convert an sref to the block it points to,
65 * throwing an internal error if downcasting fails
66 * \param Result The result variable, used for checking
67 * \param SRef The SRef to be cast
68 */
69#define TVM_SREF_TO_BLOCK(Result, SRef) \
70 TVM_SREF_AS_OR_ERR(Result, SRef, ::tvm::tir::BlockNode) \
71 << "TypeError: Expects StmtSRef `" << #SRef \
72 << "` points to `Block`, but gets: " << (SRef->stmt ? SRef->stmt->GetTypeKey() : "None")
73
74/*!
75 * \brief A helper macro to convert an sref to the for-loop it points to,
76 * throwing an internal error if downcasting fails
77 * \param Result The name of the result variable, used for checking
78 * \param SRef The SRef to be cast
79 */
80#define TVM_SREF_TO_FOR(Result, SRef) \
81 TVM_SREF_AS_OR_ERR(Result, SRef, ::tvm::tir::ForNode) \
82 << "TypeError: Expects StmtSRef `" << #SRef \
83 << "` points to `Loop`, but gets: " << (SRef->stmt ? SRef->stmt->GetTypeKey() : "None")
84
85/*!
86 * \brief Downcast a TVM ObjectRef to its corresponding container using `ObjectRef::as<Type>`,
87 * then check if the downcasting succeeded.
88 * \param Result The result variable, used for checking
89 * \param From The ObjectRef to be downcast
90 * \param Type The type to be downcast to
91 */
92#define TVM_TYPE_AS_OR_ERR(Result, From, Type) \
93 From.as<Type>(); \
94 ICHECK(Result)

96/*!
97 * \brief Downcast a TVM ObjectRef to its corresponding container using `ObjectRef::as<Type>`,
98 * throwing an internal error if downcast fails.
99 * \param Result The result variable, used for checking
100 * \param From The ObjectRef to be downcast
101 * \param Type The type to be downcast to

103#define TVM_TYPE_AS(Result, From, Type) \
104 TVM_TYPE_AS_OR_ERR(Result, From, Type) \

108/*!
109 * \brief Convert an array of loop StmtSRefs to an array of loops
110 * \param loop_srefs The loop StmtSRefs to be converted
111 * \return The conversion result loops
112 */
113inline Array<For> LoopSRefs2Loops(const Array<StmtSRef>& loop_srefs) {
114 Array<For> loops;
115 loops.reserve(loop_srefs.size());
116 for (StmtSRef loop_sref : loop_srefs) {
117 const ForNode* loop = TVM_SREF_TO_FOR(loop, loop_sref);
118 loops.push_back(GetRef<For>(loop));
119 }
120 return loops;
121}

123/*!
124 * \brief Convert an array of block rvs to an array of block StmtSRefs
125 * \param sch The schedule used to evaluate the random variables
126 * \param block_rvs The random variables to be converted
127 * \return The conversion result srefs
128 */
129inline Array<StmtSRef> BlockRVs2StmtSRefs(const Schedule& sch, const Array<BlockRV>& block_rvs) {
130 Array<StmtSRef> block_srefs;
131 block_srefs.reserve(block_rvs.size());
132 for (const BlockRV& block_rv : block_rvs) {
133 block_srefs.push_back(sch->GetSRef(block_rv));
134 }
135 return block_srefs;
136}

138/******** Storage scope ********/
139
140/*!
141 * \brief Determine if iterators of a storage scope should be relaxed
142 * under a specific thread scope
143 * \param storage_scope The storage scope that the iterators are on
144 * \param thread_scope The thread scope to be relaxed
145 * \return A boolean indicating the result
146 */
147inline bool CanRelaxStorageUnderThread(const runtime::StorageScope& storage_scope,
148 const runtime::ThreadScope& thread_scope) {
149 if (storage_scope.rank == runtime::StorageRank::kWarp) {
150 // for warp memory, we only relax threadIdx.x
151 return thread_scope.rank == 1 && thread_scope.dim_index == 0;
152 }
153 return static_cast<int>(storage_scope.rank) <= static_cast<int>(thread_scope.rank);
154}

158/*!
159 * \brief Remove a specific Stmt from a SeqStmt. If a SeqStmt contains a BlockRealize,
160 * whose block is the Stmt to be removed, then remove that BlockRealize too.
161 * \param seq The SeqStmt to be removed from
162 * \param to_remove The Stmt to be removed
163 * \return The removal result
164 */
165inline Stmt RemoveFromSeqStmt(const SeqStmt& seq, const Stmt& to_remove) {
166 ICHECK_GT(seq->size(), 1);
167 Array<Stmt> new_stmts;
168 new_stmts.reserve(seq->size());
169 for (const Stmt& stmt : seq->seq) {
170 if (to_remove.same_as(stmt)) {
171 continue;
172 }
173 if (const auto* realize = stmt.as<BlockRealizeNode>()) {
174 if (to_remove.same_as(realize->block)) {
175 continue;
176 }
177 }
178 new_stmts.push_back(stmt);
179 }
180 return SeqStmt::Flatten(new_stmts);
181}

183/*!
184 * \brief Convert a Stmt to an Array.
185 * \param stmt The Stmt to be converted to
186 * \return If the Stmt is SeqStmt, then returns the sequence;
187 * Otherwise, returns a single-element Array with the Stmt inside.
188 */
189inline Array<Stmt> AsArray(const Stmt& stmt) {
190 if (const auto* seq_stmt = stmt.as<SeqStmtNode>()) {
191 return seq_stmt->seq;
192 }
193 return {stmt};
194}

196/*!
197 * \brief Checks of a statement is a SeqStmt that contains multiple statements
198 * \param stmt The statement to be checked
199 * \return A boolean indicating the result
200 */
201inline bool IsSingleStmt(const Stmt& stmt) {
202 if (const auto* seq_stmt = stmt.as<SeqStmtNode>()) {
203 return seq_stmt->seq.size() == 1;
204 }
205 return true;
206}

210/*!
211 * \brief Create a new IterVar for the input For loop, with specified name and type
212 * \param loop The loop to be created from
213 * \param name The name of the new IterVar
214 * \param iter_var_type The type of the new IterVar
215 * \return The newly created IterVar
216 */
217inline IterVar IterVarFromLoop(const For& loop, String name, IterVarType iter_var_type) {
218 return IterVar(Range::FromMinExtent(loop->min, loop->extent),
219 Var(std::move(name), loop->loop_var.dtype()), iter_var_type);

222/*!
223 * \brief Get the thread scope bound to the specific loop
224 * \param loop The loop to be inspected
225 * \return The thread scope bound to the loop
226 */
227inline runtime::ThreadScope GetThreadScope(const ForNode* loop) {
228 if (loop->kind == ForKind::kThreadBinding) {
229 return runtime::ThreadScope::Create(loop->thread_binding.value()->thread_tag);
230 }
231 return runtime::ThreadScope{-1, -1};
232}

234/*!
235 * \brief Check if the thread scope is blockIdx
236 * \param thread_scope The thread scope to be checked
237 * \return True if the thread scope is blockIdx
238 */
239inline bool IsBlockIdx(const runtime::ThreadScope& thread_scope) {
240 return thread_scope.rank == 0; // The rank of blockIdx is 0
241}

243/*!
244 * \brief Check if the thread scope is threadIdx
245 * \param thread_scope The thread scope to be checked
246 * \return True if the thread scope is threadIdx
247 */
248inline bool IsThreadIdx(const runtime::ThreadScope& thread_scope) {
249 return thread_scope.rank == 1 && thread_scope.dim_index >= 0;
250}

252/******** Integer set ********/

254/*!
255 * \brief Converts the Ranges to IntSets
256 * \param var_dom The ranges of variables
257 * \return The integer sets of the variables
258 */
259inline Map<Var, arith::IntSet> AsIntSet(const Map<Var, Range>& var_dom) {
260 std::unordered_map<Var, arith::IntSet, ObjectPtrHash, ObjectPtrEqual> result;
261 result.reserve(var_dom.size());
262 for (auto kv : var_dom) {
263 Var& var = kv.first;
264 Range& range = kv.second;
265 result.emplace(std::move(var), arith::IntSet::FromRange(std::move(range)));
266 }
267 return {result.begin(), result.end()};
268}

289/*!
290 * \brief Check if an expression consists of a single variable,
291 * or a variable plus/minus an constant integer shift
292 * \param expr The expression to be checked
293 * \return The single variable in the expression, or NullOpt if the expression is neither a variable
294 * or a constant shift from a variable
295 */
296inline Optional<Var> AnalyzeVarWithShift(const PrimExpr& expr, Optional<IntImm>* constant) {
297 if (const auto* var = expr.as<VarNode>()) {
298 *constant = NullOpt;
299 return GetRef<Var>(var);
300 }
301 arith::PVar<Var> var;
302 arith::PVar<IntImm> shift;
303 // match: "var + shift"
304 if ((var + shift).Match(expr) || (shift + var).Match(expr)) {
305 *constant = shift.Eval();
306 return var.Eval();
307 }
308 // match: "var - shift"
309 if ((var - shift).Match(expr)) {
310 IntImm result = shift.Eval();
311 *constant = IntImm(result->dtype, -result->value);
312 return var.Eval();
313 }
314 return NullOpt;
315}

319/*!
320 * \brief Get the annotation on a Block/For
321 * \tparam TObjectRef The type of the annotation value
322 * \param sref The sref to the block or the for loop
323 * \param ann_key The annotation key to be looked up
324 * \return NullOpt if not found; otherwise the annotation value
325 */
326template <class TObjectRef, class TStmtNode>
327inline Optional<TObjectRef> GetAnn(const TStmtNode* stmt, const String& ann_key) {
328 const Map<String, ObjectRef>* annotations = &stmt->annotations;
329 for (const auto& ann : *annotations) {
330 if (ann.first == ann_key) {
331 return Downcast<TObjectRef>(ann.second);
332 }
333 }
334 return NullOpt;
335}
336
337/*!
338 * \brief Get the annotation on a Block/For
339 * \tparam TObjectRef The type of the annotation value
340 * \param sref The sref to the block or the for loop
341 * \param ann_key The annotation key to be looked up
342 * \return NullOpt if not found; otherwise the annotation value
343 */
344template <class TObjectRef>
345inline Optional<TObjectRef> GetAnn(const StmtSRef& sref, const String& ann_key) {
346 if (const auto* loop = sref->StmtAs<ForNode>()) {
347 return GetAnn<TObjectRef, ForNode>(loop, ann_key);
348 } else if (const auto* block = sref->StmtAs<BlockNode>()) {
349 return GetAnn<TObjectRef, BlockNode>(block, ann_key);
350 } else {
351 LOG(FATAL) << "TypeError: Unknown type of sref: " << sref->stmt->GetTypeKey();
352 throw;
353 }
354}
355
356/*!
357 * \brief Check if a Block/For has a specific pair of annotation key and values
358 * \param sref The sref to the block or the for loop
359 * \param ann_key The annotation key to be checked
360 * \param ann_val The annotation value to be checked
361 * \return Whether a Block/For has a specific pair of annotation key and values
362 */
363inline bool HasAnn(const StmtSRef& sref, const String& ann_key, const String& ann_val) {
364 Optional<String> result = GetAnn<String>(sref, ann_key);
365 return result.defined() && result.value() == ann_val;
366}
367
368/*!
369 * \brief Check if a Block/For has a specific pair of annotation key and values
370 * \param sref The sref to the block or the for loop
371 * \param ann_key The annotation key to be checked
372 * \param ann_val The boolean annotation value to be checked
373 * \return Whether a Block/For has a specific pair of annotation key and values
374 */
375inline bool HasAnn(const StmtSRef& sref, const String& ann_key, bool ann_val) {
376 Optional<Bool> result = GetAnn<Bool>(sref, ann_key);
377 return result.defined() && result.value() == ann_val;
378}

380/********** Helper Functions for RuleAddRFactor and RuleCrossThreadReduction **********/
381
382/*!
383 * \brief Reorder the reduction loops to innermost positions if needed.
384 * \param sch The schedule
385 * \param block_rv The block where to apply the reorder
386 * \param fused_reduce_loop The fusion-generated loop to return.
387 * \param num_spatial_loops The number of spatial loops to return.
388 * \note Before invoking this helper function, make sure that the block has only spatial and
389 * reduction loop axes.
390 */
391inline void ReorderAndFuseReductionLoops(const tir::Schedule& sch, const tir::BlockRV& block_rv,
392 tir::LoopRV* fused_reduce_loop,
393 size_t* num_spatial_loops) {
394 Array<tir::LoopRV> loops = sch->GetLoops(block_rv);
395 Array<tir::StmtSRef> loop_srefs;
396 for (const tir::LoopRV& loop_rv : loops) {
397 loop_srefs.push_back(sch->GetSRef(loop_rv));
398 }
399
400 Array<tir::LoopRV> new_order;
401 // Step 1. Add spatial loops.
402 *num_spatial_loops = 0;
403 for (size_t i = 0; i < loops.size(); ++i) {
404 if (GetLoopIterType(loop_srefs[i]) == tir::kDataPar) {
405 new_order.push_back(loops[i]);
406 (*num_spatial_loops)++;
407 }
408 }
409 // Step 2. Add reduction loops.
410 Array<tir::LoopRV> reduction_loops;
411 for (size_t i = 0; i < loops.size(); ++i) {
412 if (GetLoopIterType(loop_srefs[i]) == tir::kCommReduce) {
413 new_order.push_back(loops[i]);
414 reduction_loops.push_back(loops[i]);
415 }
416 }
417 // Step 3. Apply reordering if new_order differs from the original order.
418 ICHECK_EQ(new_order.size(), loops.size());
419 for (size_t i = 0; i < loops.size(); ++i) {
420 if (!new_order[i].same_as(loops[i])) {
421 sch->Reorder(new_order);
422 break;
423 }
424 }
425 // Step 4. Fuse all the reduction loops if there are multiple reduction loops.
426 CHECK(!reduction_loops.empty()) << "ValueError: There should be at least one reduction loop";
427 if (reduction_loops.size() > 1) {
428 *fused_reduce_loop = sch->Fuse(reduction_loops);
429 } else {
430 *fused_reduce_loop = reduction_loops[0];
431 }
432}
