27class TensorIntrinMismatchError : public ScheduleError {
28 public:
29 explicit TensorIntrinMismatchError(IRModule lhs_mod, Stmt lhs_stmt, Stmt rhs_stmt,
30 std::vector<std::string> error_messages)
31 : lhs_mod_(std::move(lhs_mod)),
32 lhs_stmt_(std::move(lhs_stmt)),
33 rhs_stmt_(std::move(rhs_stmt)),
34 error_messages_(std::move(error_messages)) {
35 ICHECK(lhs_stmt_->IsInstance<ForNode>() || lhs_stmt_->IsInstance<BlockNode>());
36 }
37
38 String FastErrorString() const final {
39 return "ScheduleError: The stmt doesn't match the tensor intrin.";
40 }
41
42 String DetailRenderTemplate() const final {
43 std::ostringstream os;
44 os << "The stmt {0} doesn't match the tensor intrin\n " << rhs_stmt_;
45 for (const auto& msg : error_messages_) {
46 os << msg << std::endl;
47 }
48 return os.str();
49 }
50
51 IRModule mod() const final { return lhs_mod_; }
52
53 Array<ObjectRef> LocationsOfInterest() const final { return {lhs_stmt_}; }
54
55 private:
56 IRModule lhs_mod_;
57 Stmt lhs_stmt_;
58 Stmt rhs_stmt_;
59 std::vector<std::string> error_messages_;
60};

62/* Override the dispatcher to make sure RHS is always valid */
63bool TensorizeComparator::VisitStmt(const Stmt& n, const Stmt& other) {
64 bool equal = n.same_as(other) ||
65 ((n->type_index() == other->type_index()) && StmtComparator::VisitStmt(n, other));
66 if (!equal && assert_mode_ && (n->IsInstance<ForNode>() || n->IsInstance<BlockNode>())) {
67 throw TensorIntrinMismatchError(lhs_mod_, n, other, std::move(error_messages_));
68 }
69 return equal;
70}
71
72bool TensorizeComparator::VisitExpr(const PrimExpr& n, const PrimExpr& other) {
73 bool equal =
74 n.same_as(other) || ((n->type_index() == other->type_index()) && n->dtype == other->dtype &&
75 ExprComparator::VisitExpr(n, other));
76 if (!equal && assert_mode_) {
77 std::ostringstream os;
78 os << "Expression mismatch: " << n << " vs " << other;
79 EmitError(os.str());
80 }
81 return equal;
82}

84bool TensorizeComparator::VisitStmt_(const ForNode* op, const Stmt& other) {
85 const auto* rhs = other.as<ForNode>();
86 if (!DefEqual(op->loop_var, rhs->loop_var)) return false;
87 if (!VisitExpr(op->min, rhs->min)) return false;
88 if (!VisitExpr(op->extent, rhs->extent)) return false;
89 if (op->thread_binding.defined() != rhs->thread_binding.defined()) return false;
90 if (op->thread_binding.defined() &&
91 !VisitExpr(op->thread_binding.value(), rhs->thread_binding.value())) {
92 return false;
93 }
94 if (op->kind != rhs->kind) return false;
95 if (!CompareAnnotationMap(op->annotations, rhs->annotations)) return false;
96 return VisitStmt(op->body, rhs->body);
97}

100 const auto* rhs = other.as<SeqStmtNode>();
101 return CompareArray(op->seq, rhs->seq, &TensorizeComparator::VisitStmt);

104bool TensorizeComparator::VisitStmt_(const BufferStoreNode* op, const Stmt& other) {
105 const auto* rhs = other.as<BufferStoreNode>();
106 return CompareBufferAccess(op, rhs) && VisitExpr(op->value, rhs->value);

109bool TensorizeComparator::VisitStmt_(const BlockRealizeNode* op, const Stmt& other) {
110 const auto* rhs = other.as<BlockRealizeNode>();
111 if (!is_scope_block) {
112 if (!CompareArray(op->iter_values, rhs->iter_values, &TensorizeComparator::VisitExpr)) {
113 return false;
114 }
115 }
116 return VisitExpr(op->predicate, rhs->predicate) && VisitStmt(op->block, rhs->block);
117}
118
119bool TensorizeComparator::VisitStmt_(const BlockNode* op, const Stmt& other) {
120 const auto* rhs = other.as<BlockNode>();
121 // Check block equality.
122 // All iter vars and buffer regions including the order should match.
123 // When checking iter vars, DefEqual is used to remap variables.
124 if (!is_scope_block) {
125 if (!CompareArray(op->iter_vars, rhs->iter_vars, &TensorizeComparator::CompareIterVar)) {
126 return false;
127 }
128 if (!CompareArray(op->alloc_buffers, rhs->alloc_buffers, &TensorizeComparator::CompareBuffer)) {
129 return false;
130 }
131 }
132 if (!CompareArray(op->writes, rhs->writes, &TensorizeComparator::CompareBufferRegion)) {
133 return false;
134 }
135 if (!CompareArray(op->reads, rhs->reads, &TensorizeComparator::CompareBufferRegion)) {
136 return false;
137 }
138 is_scope_block = false;
139 return VisitStmt(op->body, rhs->body);
140}

182bool TensorizeComparator::VisitExpr_(const VarNode* op, const PrimExpr& other) {
183 const auto* rhs = other.as<VarNode>();
184 auto lhs = GetRef<Var>(op);
185 if (lhs.same_as(other)) return true;
186 if (op->dtype != rhs->dtype) return false;
187 auto it = equal_map_.find(lhs);
188 return it != equal_map_.end() && it->second.same_as(other);
189}

191bool TensorizeComparator::VisitExpr_(const BufferLoadNode* op, const PrimExpr& other) {
192 const auto* rhs = other.as<BufferLoadNode>();
193 return CompareBufferAccess(op, rhs);
194}

196bool TensorizeComparator::VisitExpr_(const SelectNode* op, const PrimExpr& other) {
197 const auto* rhs = other.as<SelectNode>();
198 return VisitExpr(op->condition, rhs->condition) && VisitExpr(op->true_value, rhs->true_value) &&
199 VisitExpr(op->false_value, rhs->false_value);
200}

202bool TensorizeComparator::DefEqual(const Var& lhs, const Var& rhs) {
203 if (lhs.same_as(rhs)) return true;
204 auto it = equal_map_.find(lhs);
205 // If there is already a mapping
206 if (it != equal_map_.end()) return it->second.same_as(rhs);
207 // Otherwise remap lhs to rhs
208 equal_map_[lhs] = rhs;
209 analyzer_.Bind(lhs, rhs);
210 return true;
211}

213bool TensorizeComparator::CompareAnnotation(const std::pair<String, ObjectRef>& lhs,
214 const std::pair<String, ObjectRef>& rhs) {
215 if (lhs.first != rhs.first) return false;
216 return VisitExpr(Downcast<PrimExpr>(lhs.second), Downcast<PrimExpr>(rhs.second));
217}
218
219bool TensorizeComparator::CompareAnnotationMap(const Map<String, ObjectRef>& lhs,
220 const Map<String, ObjectRef>& rhs) {
221 if (lhs.same_as(rhs)) return true;
222 if (lhs.size() != rhs.size()) return false;
223
224 auto sort_map =
225 [](const Map<String, ObjectRef>& map) -> std::vector<std::pair<String, ObjectRef>> {
226 std::vector<std::pair<String, ObjectRef>> ret(map.begin(), map.end());
227 sort(ret.begin(), ret.end());
228 return ret;
229 };
230
231 std::vector<std::pair<String, ObjectRef>> lhs_array = sort_map(lhs);
232 std::vector<std::pair<String, ObjectRef>> rhs_array = sort_map(rhs);
233
234 for (size_t i = 0; i < lhs.size(); ++i) {
235 if (!CompareAnnotation(lhs_array[i], rhs_array[i])) return false;
236 }
237 return true;
238}

240bool TensorizeComparator::CompareBuffer(const Buffer& lhs, const Buffer& rhs) {
241 if (lhs.same_as(rhs)) return true;
242 auto it = rhs_buffer_map_.find(rhs);
243 bool equal;
244 if (it != rhs_buffer_map_.end()) {
245 equal = (*it).second.same_as(lhs);
246 } else {
247 // Remap both buffer itself and buffer data, skip buffer shape
248 equal =
249 DefEqual(lhs->data, rhs->data) && lhs->dtype == rhs->dtype && lhs.scope() == rhs.scope();
250 if (equal) {
251 rhs_buffer_map_[rhs] = lhs;
252 }
253 }
254 return equal;
255}

257bool TensorizeComparator::CompareBufferRegion(const BufferRegion& lhs, const BufferRegion& rhs) {
258 if (!CompareBuffer(lhs->buffer, rhs->buffer)) {
259 if (assert_mode_) {
260 std::ostringstream os;
261 os << "Buffer mismatch: " << lhs->buffer << " vs " << rhs->buffer;
262 EmitError(os.str());
263 }
264 return false;
265 }
266 int offset = static_cast<int>(lhs->region.size()) - static_cast<int>(rhs->region.size());
267 // Number of indices in RHS (desc of the tensor intrinsic) must be smaller than it in LHS
268 if (offset < 0) return false;
269
270 auto it = buffer_indices_.find(lhs->buffer);
271 if (it == buffer_indices_.end()) {
272 // Update base indices for the buffer, this can only happen if it is visiting the scope block.
273 ICHECK(is_scope_block);
274 std::vector<PrimExpr> indices_base;
275 indices_base.reserve(lhs->region.size());
276 for (int i = 0; i < offset; i++) {
277 // High-dim region must be element-wise
278 if (!is_one(lhs->region[i]->extent)) return false;
279 indices_base.emplace_back(lhs->region[i]->min);
280 }
281 for (size_t i = 0; i < rhs->region.size(); i++) {
282 // save base index
283 indices_base.emplace_back(lhs->region[i + offset]->min);
284 // check extent match
285 if (!analyzer_.CanProveEqual(lhs->region[i + offset]->extent, rhs->region[i]->extent)) {
286 return false;
287 }
288 }
289 buffer_indices_.emplace(lhs->buffer, std::move(indices_base));
290 } else {
291 // Check the base indices are consistent.
292 const std::vector<PrimExpr>& indices_base = it->second;
293 for (int i = 0; i < offset; i++) {
294 // High-dim region must be element-wise
295 if (!is_one(lhs->region[i]->extent)) return false;
296 if (!analyzer_.CanProveEqual(indices_base[i], lhs->region[i]->min)) return false;
297 }
298 for (size_t i = 0; i < rhs->region.size(); i++) {
299 // check extent match
300 if (!analyzer_.CanProveEqual(lhs->region[i + offset]->extent, rhs->region[i]->extent)) {
301 return false;
302 }
303 PrimExpr normalized_lhs_min = (lhs->region[i + offset]->min - indices_base[i + offset]);
304 if (!analyzer_.CanProveEqual(normalized_lhs_min, rhs->region[i]->min)) {
305 return false;
306 }
307 }
308 }
309 return true;
310}

312// Comparator for BufferStoreNode and BufferLoadNode
313template <typename T>
314bool TensorizeComparator::CompareBufferAccess(const T* lhs, const T* rhs) {
315 if (!CompareBuffer(lhs->buffer, rhs->buffer)) return false;
316 int offset = static_cast<int>(lhs->indices.size()) - static_cast<int>(rhs->indices.size());
317 if (offset < 0) return false;
318 auto it = buffer_indices_.find(lhs->buffer);
319 ICHECK(it != buffer_indices_.end());
320 const std::vector<PrimExpr>& indices_base = (*it).second;
321 ICHECK_EQ(indices_base.size(), rhs->indices.size() + offset);
322 for (size_t i = 0; i < rhs->indices.size(); i++) {
323 PrimExpr normalized_lhs_index = lhs->indices[i + offset] - indices_base[i + offset];
324 if (!analyzer_.CanProveEqual(normalized_lhs_index, rhs->indices[i])) {
325 if (assert_mode_) {
326 std::ostringstream os;
327 os << "Buffer indices mismatch: " << lhs->indices[i + offset] << " vs " << rhs->indices[i];
328 EmitError(os.str());
329 }
330 return false;
331 }
332 }
333 return true;
334}

336template <typename T, typename Self, typename F>
337bool TensorizeComparator::CompareArray(const Array<T>& lhs, const Array<T>& rhs, F Self::*cmp) {
338 if (lhs.same_as(rhs)) return true;
339 if (lhs.size() != rhs.size()) return false;
340 for (size_t i = 0; i < lhs.size(); ++i) {
341 if (!(static_cast<Self*>(this)->*cmp)(lhs[i], rhs[i])) return false;
342 }
343 return true;
344}

368bool AutoTensorizeComparator::VisitStmt_(const BlockNode* op, const Stmt& other) {
369 const auto* rhs = other.as<BlockNode>();
370 // Check block equality.
371 // All iter vars and buffer regions including the order should match.
372 // When checking iter vars, DefEqual is used to remap variables.
373 if (!is_scope_block) {
374 if (!CompareArray(op->iter_vars, rhs->iter_vars, &AutoTensorizeComparator::CompareIterVar)) {
375 return false;
376 }
377 if (!CompareAnnotationMap(op->annotations, rhs->annotations)) {
378 return false;
379 }
380 if (!CompareArray(op->alloc_buffers, rhs->alloc_buffers,
381 &AutoTensorizeComparator::CompareBuffer)) {
382 return false;
383 }
384 for (const IterVar& block_iter : op->iter_vars) {
385 inner_iter_dom_map_.Set(block_iter->var, arith::IntSet::FromRange(block_iter->dom));
386 }
387 } else {
388 auto collect_iter = [&](const BlockNode* op, std::vector<IterVar>& iters) -> bool {
389 for (const auto& iter : op->iter_vars) {
390 analyzer_.Bind(iter->var, iter->dom);
391 if (iter->iter_type == IterVarType::kDataPar ||
392 iter->iter_type == IterVarType::kCommReduce) {
393 iters.push_back(iter);
394 } else {
395 return false;
396 }
397 }
398 return true;
399 };
400 if (!collect_iter(op, lhs_iters_)) {
401 return false;
402 }
403 if (!collect_iter(rhs, rhs_iters_)) {
404 return false;
405 }
406 }
407 is_scope_block = false;
408 return VisitStmt(op->body, rhs->body);
409}

411bool AutoTensorizeComparator::CompareBuffer(const Buffer& lhs, const Buffer& rhs) {
412 if (lhs.same_as(rhs)) return true;
413 auto it = rhs_buffer_map_.find(rhs);
414 bool equal;
415 if (it != rhs_buffer_map_.end()) {
416 equal = (*it).second.same_as(lhs);
417 } else {
418 // Remap both buffer itself and buffer data, skip buffer shape and scope
419 equal = DefEqual(lhs->data, rhs->data) && lhs->dtype == rhs->dtype;
420 if (equal) {
421 rhs_buffer_map_[rhs] = lhs;
422 lhs_buffer_map_[lhs] = rhs;
423 }
424 }
425 return equal;
426}

438template <typename T>
439bool AutoTensorizeComparator::CompareBufferAccess(const T* lhs, const T* rhs) {
440 if (!CompareBuffer(lhs->buffer, rhs->buffer)) return false;
441 auto it_lhs = lhs_buffer_indices_map_.find(lhs->buffer);
442 if (it_lhs == lhs_buffer_indices_map_.end()) {
443 if (rhs_buffer_indices_map_.find(rhs->buffer) != rhs_buffer_indices_map_.end()) {
444 return false;
445 }
446 std::vector<PrimExpr> lhs_indices;
447 for (const auto& index : lhs->indices) {
448 lhs_indices.push_back(analyzer_.Simplify(index));
449 }
450
451 auto is_scalar_access = [](const Array<PrimExpr>& indices, PrimExpr index) {
452 // Check if the indexing is of the form C[0]
453 if (indices.size() > 1) return false;
454 auto int_imm = index.template as<IntImmNode>();
455 if (int_imm && int_imm->value == 0) return true;
456 return false;
457 };
458
459 for (const auto& index : rhs->indices) {
460 if (!index.template as<VarNode>() && !is_scalar_access(rhs->indices, index)) return false;
461 }
462 lhs_buffer_indices_map_[lhs->buffer] = lhs_indices;
463 rhs_buffer_indices_map_[rhs->buffer] = rhs->indices;
464 } else {
465 auto it_rhs = rhs_buffer_indices_map_.find(rhs->buffer);
466 if (it_rhs == rhs_buffer_indices_map_.end()) {
467 return false;
468 }
469 auto indices_check = [&](const Array<PrimExpr>& indices,
470 const Array<PrimExpr>& old_indices) -> bool {
471 if (indices.size() != old_indices.size()) {
472 return false;
473 }
474 for (size_t i = 0; i < indices.size(); ++i) {
475 if (!analyzer_.CanProveEqual(indices[i], old_indices[i])) {
476 return false;
477 }
478 }
479 return true;
480 };
481 if (!indices_check(lhs->indices, it_lhs->second)) return false;
482 if (!indices_check(rhs->indices, it_rhs->second)) return false;
483 }
484 return true;
485}
