37int GetNumValidInstructions(const Array<Instruction>& insts, bool remove_postproc) {
38 if (!remove_postproc) {
39 return insts.size();
40 }
41 int n_insts = 0;
42 for (const Instruction& inst : insts) {
43 if (!inst->kind->IsPostproc()) {
44 ++n_insts;
45 } else {
46 break;
47 }
48 }
49 return n_insts;
50}

52/**************** TranslateInputRVs ****************/
53
54Array<ObjectRef> TranslateInputRVs(const Array<ObjectRef>& inputs,
55 const std::unordered_map<const Object*, const Object*>& rv_map) {
56 Array<ObjectRef> result;
57 result.reserve(inputs.size());
58 for (const ObjectRef& input : inputs) {
59 if (!input.defined() || // constant: nullptr
60 input->IsInstance<StringObj>() || // constant: string
61 input->IsInstance<IntImmNode>() || // constant: integer
62 input->IsInstance<FloatImmNode>()) { // constant: float
63 result.push_back(input);
64 } else if (input->IsInstance<BlockRVNode>() || // RV: block
65 input->IsInstance<LoopRVNode>() || // RV: loop
66 input->IsInstance<VarNode>()) { // RV: var
67 auto it = rv_map.find(input.get());
68 ICHECK(it != rv_map.end()) << "IndexError: Random variable doesn't exist: " << input;
69 result.push_back(GetRef<ObjectRef>(it->second));
70 } else if (const auto* expr = input.as<PrimExprNode>()) { // RV: Expr
71 result.push_back(
72 Substitute(GetRef<PrimExpr>(expr), [&rv_map](const Var& var) -> Optional<PrimExpr> {
73 auto it = rv_map.find(var.get());
74 if (it == rv_map.end()) {
75 return NullOpt;
76 }
77 const Object* dst = it->second;
78 ICHECK(dst->IsInstance<VarNode>())
79 << "TypeError: Expect 'tir.Var', but gets: " << dst->GetTypeKey();
80 return GetRef<Var>(static_cast<const VarNode*>(dst));
81 }));
82 } else if (input->IsInstance<ArrayNode>()) {
83 // Recursively convert elements of the array into a new list of ObjectRefs.
84 result.push_back(TranslateInputRVs(Downcast<Array<ObjectRef>>(input), rv_map));
85 } else {
86 ICHECK(false) << "TypeError: Cannot recognize the type of an input random variable: "
87 << input->GetTypeKey();
88 throw;
89 }
90 }
91 return result;
92}
93
94Array<ObjectRef> TranslateInputRVs(
95 const Array<ObjectRef>& inputs,
96 const std::unordered_map<ObjectRef, String, ObjectPtrHash, ObjectPtrEqual>& rv_names) {
97 Array<ObjectRef> results;
98 results.reserve(inputs.size());
99 for (const ObjectRef& input : inputs) {
100 if (!input.defined()) {
101 // Case 0. nullptr => None
102 results.push_back(String("None"));
103 continue;
104 }
105 auto it = rv_names.find(input);
106 if (it != rv_names.end()) {
107 // Case 1. BlockRV, LoopRV, VarRV
108 results.push_back(it->second);
109 } else if (const auto* str_obj = input.as<StringObj>()) {
110 // Case 2. string => "content"
111 results.push_back(String('"' + std::string(str_obj->data) + '"'));
112 } else if (input->IsInstance<IntImmNode>() || input->IsInstance<FloatImmNode>()) {
113 // Case 3. integer or floating-point number
114 results.push_back(input);
115 } else if (input->IsInstance<ArrayNode>()) {
116 // Case 4: array
117 results.push_back(TranslateInputRVs(Downcast<Array<ObjectRef>>(input), rv_names));
118 } else if (input->IsInstance<MapNode>()) {
119 // Case 5: dict
120 results.push_back(input);
121 } else if (input->IsInstance<BlockRVNode>() || inputs->IsInstance<LoopRVNode>() ||
122 inputs->IsInstance<VarNode>()) {
123 LOG(FATAL) << "IndexError: Random variable is not defined " << input;
124 throw;
125 } else {
126 LOG(FATAL) << "TypeError: Stringifying is not supported for type: " << input->GetTypeKey();
127 throw;
128 }
129 }
130 return results;
131}
132
133Array<ObjectRef> TranslateInputRVs(const Array<ObjectRef>& inputs,
134 const std::unordered_map<std::string, ObjectRef>& named_rvs) {
135 Array<ObjectRef> results;
136 results.reserve(inputs.size());
137 for (const ObjectRef& input : inputs) {
138 // Case 3. integer or floating-point number
139 if (input->IsInstance<IntImmNode>() || input->IsInstance<FloatImmNode>()) {
140 results.push_back(input);
141 continue;
142 }
143 // Case 4. array
144 if (input->IsInstance<ArrayNode>()) {
145 results.push_back(TranslateInputRVs(Downcast<Array<ObjectRef>>(input), named_rvs));
146 continue;
147 }
148 // Case 5. dict
149 if (input->IsInstance<MapNode>()) {
150 results.push_back(input);
151 continue;
152 }
153 const auto* str = input.as<StringObj>();
154 CHECK(str) << "TypeError: Expect String, but gets: " << input->GetTypeKey();
155 CHECK_GT(str->size, 0) << "ValueError: Empty string is not allowed in input names";
156 const char* name = str->data;
157 int64_t size = str->size;
158 // Case 2. string
159 if (size >= 2 && name[0] == '"' && name[size - 1] == '"') {
160 results.push_back(String(std::string(name + 1, size - 2)));
161 continue;
162 }
163 // Case 0 & 1. None, BlockRV, LoopRV, VarRV
164 auto it = named_rvs.find(name);
165 CHECK(it != named_rvs.end()) << "ValueError: The random variable is not defined: " << name;
166 results.push_back(it->second);
167 }
168 return results;
169}

171/**************** TranslateAddOutputRVs ****************/
172
173void TranslateAddOutputRVs(const Array<ObjectRef>& old_outputs, const Array<ObjectRef>& new_outputs,
174 std::unordered_map<const Object*, const Object*>* rv_map) {
175 ICHECK_EQ(old_outputs.size(), new_outputs.size());
176 int n = old_outputs.size();
177 const ObjectRef* p_old = old_outputs.GetArrayNode()->begin();
178 const ObjectRef* p_new = new_outputs.GetArrayNode()->begin();
179 for (int i = 0; i < n; ++i) {
180 (*rv_map)[p_old[i].get()] = p_new[i].get();
181 }
182}
183
184Array<String> TranslateAddOutputRVs(
185 const Array<ObjectRef>& outputs,
186 std::unordered_map<ObjectRef, String, ObjectPtrHash, ObjectPtrEqual>* rv_names) {
187 Array<String> results;
188 results.reserve(outputs.size());
189 for (const ObjectRef& output : outputs) {
190 int i = rv_names->size();
191 ICHECK(!rv_names->count(output))
192 << "ValueError: The random variable has been produced once: " << rv_names->at(output);
193 String result{ObjectPtr<StringObj>{nullptr}};
194 if (output->IsInstance<BlockRVNode>()) {
195 result = "b" + std::to_string(i);
196 } else if (output->IsInstance<LoopRVNode>()) {
197 result = "l" + std::to_string(i);
198 } else if (output->IsInstance<VarNode>()) {
199 result = "v" + std::to_string(i);
200 } else {
201 LOG(FATAL) << "TypeError: Cannot recognize the type of the random variable: "
202 << output->GetTypeKey();
203 throw;
204 }
205 results.push_back(result);
206 rv_names->emplace(output, std::move(result));
207 }
208 return results;
209}
210
211void TranslateAddOutputRVs(const Array<String>& old_outputs, const Array<ObjectRef>& new_outputs,
212 std::unordered_map<std::string, ObjectRef>* named_rvs) {
213 ICHECK_EQ(old_outputs.size(), new_outputs.size());
214 int n = old_outputs.size();
215 const ObjectRef* p_old = old_outputs.GetArrayNode()->begin();
216 const ObjectRef* p_new = new_outputs.GetArrayNode()->begin();
217 for (int i = 0; i < n; ++i) {
218 const auto* name = static_cast<const StringObj*>(p_old[i].get());
219 named_rvs->emplace(std::string(name->data, name->size), p_new[i]);
220 }
221}

224
225Optional<ObjectRef> TraceNode::GetDecision(const Instruction& inst) const {
226 auto it = this->decisions.find(inst);
227 return it == this->decisions.end() ? Optional<ObjectRef>(NullOpt) : (*it).second;
228}
229
230void TraceNode::Append(Instruction inst) { insts.push_back(std::move(inst)); }
231
232void TraceNode::Append(Instruction inst, ObjectRef decision) {
233 decisions.Set(inst, std::move(decision));
234 insts.push_back(std::move(inst));
235}
236
237Optional<Instruction> TraceNode::Pop() {
238 if (insts.empty()) {
239 return NullOpt;
240 }
241 Instruction inst = insts.back();
242 insts.pop_back();
243 if (decisions.count(inst)) {
244 decisions.erase(inst);
245 }
246 return inst;
247}

249/**************** Interfacing with InstructionKind ****************/
250
251void TraceNode::ApplyToSchedule(
252 Schedule sch, bool remove_postproc,
253 runtime::TypedPackedFunc<ObjectRef(const Instruction& inst, const Array<ObjectRef>& inputs, //
254 const Array<ObjectRef>& attrs, //
255 const Optional<ObjectRef>& decision)>
256 decision_provider) const {
257 std::unordered_map<const Object*, const Object*> rv_map;
258 for (const Instruction& inst : this->insts) {
259 if (remove_postproc && inst->kind->IsPostproc()) {
260 break;
261 }
262 Array<ObjectRef> inputs = TranslateInputRVs(inst->inputs, rv_map);
263 Array<ObjectRef> attrs = inst->attrs;
264 Optional<ObjectRef> decision = this->GetDecision(inst);
265 if (decision_provider != nullptr) {
266 decision = decision_provider(inst, inputs, attrs, decision);
267 }
268 Array<ObjectRef> outputs = inst->kind->f_apply_to_schedule(sch, inputs, attrs, decision);
269 TranslateAddOutputRVs(inst->outputs, outputs, &rv_map);
270 }
271}

273ObjectRef TraceNode::AsJSON(bool remove_postproc) const {
274 std::unordered_map<ObjectRef, String, ObjectPtrHash, ObjectPtrEqual> rv_names;
275 Array<ObjectRef> json_insts;
276 Array<ObjectRef> json_decisions;
277 json_insts.reserve(this->insts.size());
278 json_decisions.reserve(this->insts.size());
279
280 int i = 0;
281 for (const Instruction& inst : this->insts) {
282 const InstructionKind& kind = inst->kind;
283 if (remove_postproc && kind->IsPostproc()) {
284 break;
285 }
286 json_insts.push_back(Array<ObjectRef>{
287 /* 0: inst name */ kind->name,
288 /* 1: inputs */ TranslateInputRVs(inst->inputs, rv_names),
289 /* 2: attrs */ kind->f_attrs_as_json != nullptr ? kind->f_attrs_as_json(inst->attrs)
290 : ObjectRef(inst->attrs),
291 /* 3: outputs */ TranslateAddOutputRVs(inst->outputs, &rv_names),
292 });
293 if (Optional<ObjectRef> decision = this->GetDecision(inst)) {
294 json_decisions.push_back(Array<ObjectRef>{
295 /* 0: index */ Integer(i),
296 /* 1: decision */ decision.value(),
297 });
298 }
299 ++i;
300 }
301 return Array<ObjectRef>{
302 /* 0: trace */ std::move(json_insts),
303 /* 1: decision */ std::move(json_decisions),
304 };
305}

307Array<String> TraceNode::AsPython(bool remove_postproc) const {
308 std::unordered_map<ObjectRef, String, ObjectPtrHash, ObjectPtrEqual> rv_names;
309 Array<String> py_trace;
310 py_trace.reserve(this->insts.size());
311 for (const Instruction& inst : this->insts) {
312 if (remove_postproc && inst->kind->IsPostproc()) {
313 break;
314 }
315 Array<ObjectRef> attrs;
316 attrs.reserve(inst->attrs.size());
317 for (const ObjectRef& obj : inst->attrs) {
318 if (const auto* str = obj.as<StringObj>()) {
319 attrs.push_back(String('"' + std::string(str->data) + '"'));
320 } else {
321 attrs.push_back(obj);
322 }
323 }
324 py_trace.push_back(
325 inst->kind->f_as_python(/*inputs=*/TranslateInputRVs(inst->inputs, rv_names),
326 /*attrs=*/attrs,
327 /*decision=*/this->GetDecision(inst),
328 /*outputs=*/TranslateAddOutputRVs(inst->outputs, &rv_names)));
329 }
330 return py_trace;
331}

333void Trace::ApplyJSONToSchedule(ObjectRef json, Schedule sch) {
334 Array<ObjectRef> json_insts{nullptr};
335 Array<ObjectRef> json_decisions{nullptr};
336 // Parse `json` into `json_insts` and `json_decisions`
337 try {
338 const ArrayNode* arr = json.as<ArrayNode>();
339 ICHECK(arr && arr->size() == 2);
340 const auto* arr0 = arr->at(0).as<ArrayNode>();
341 const auto* arr1 = arr->at(1).as<ArrayNode>();
342 ICHECK(arr0 && arr1);
343 json_insts = GetRef<Array<ObjectRef>>(arr0);
344 json_decisions = GetRef<Array<ObjectRef>>(arr1);
345 } catch (const tvm::Error& e) {
346 LOG(FATAL) << "ValueError: The json entry of a trace should contain two arrays, an array of "
347 "instructions and an array of decisions, but gets: "
348 << json;
349 throw;
350 }
351 // Parse `json_decisions`
352 std::vector<Optional<ObjectRef>> decisions(json_insts.size(), NullOpt);
353 for (const ObjectRef& decision_entry : json_decisions) {
354 int index = -1;
355 ObjectRef decision{nullptr};
356 try {
357 const ArrayNode* arr = decision_entry.as<ArrayNode>();
358 ICHECK(arr && arr->size() == 2);
359 const IntImmNode* arr0 = arr->at(0).as<IntImmNode>();
360 ICHECK(arr0);
361 index = arr0->value;
362 decision = arr->at(1);
363 } catch (const tvm::Error& e) {
364 LOG(FATAL) << "ValueError: Each entry of a json decision should be a tuple [index, "
365 "decision], but gets: "
366 << decision_entry;
367 throw;
368 }
369 decisions[index] = std::move(decision);
370 }
371 // Parse `json_insts`
372 std::unordered_map<std::string, ObjectRef> named_rvs{{"None", ObjectRef{nullptr}}};
373 int i = 0;
374 for (const ObjectRef& inst_entry : json_insts) {
375 InstructionKind kind{nullptr};
376 Array<ObjectRef> inputs{nullptr};
377 Array<ObjectRef> attrs{nullptr};
378 Array<String> outputs{ObjectPtr<Object>{nullptr}};
379 // Parse the entry
380 try {
381 const auto* arr = inst_entry.as<ArrayNode>();
382 ICHECK(arr && arr->size() == 4);
383 const auto* arr0 = arr->at(0).as<StringObj>();
384 const auto* arr1 = arr->at(1).as<ArrayNode>();
385 const auto* arr2 = arr->at(2).as<ArrayNode>();
386 const auto* arr3 = arr->at(3).as<ArrayNode>();
387 ICHECK(arr0 && arr1 && arr2 && arr3);
388 for (const ObjectRef& str : *arr3) {
389 ICHECK(str->IsInstance<StringObj>());
390 }
391 kind = InstructionKind::Get(arr0->data);
392 inputs = GetRef<Array<ObjectRef>>(arr1);
393 attrs = GetRef<Array<ObjectRef>>(arr2);
394 outputs = GetRef<Array<String>>(arr3);
395 } catch (const tvm::Error& e) {
396 LOG(FATAL) << "ValueError: Each entry of a json instruction should be a tuple [inst_name, "
397 "inputs, attrs, outputs], but gets: "
398 << inst_entry;
399 throw;
400 }
401 // Parse inputs
402 inputs = TranslateInputRVs(inputs, named_rvs);
403 // Parse attrs
404 if (kind->f_attrs_from_json != nullptr) {
405 attrs = kind->f_attrs_from_json(attrs);
406 }
407 // Apply to the schedule
408 Array<ObjectRef> new_outputs = kind->f_apply_to_schedule(sch, inputs, attrs, decisions[i]);
409 // Parse outputs
410 TranslateAddOutputRVs(outputs, new_outputs, &named_rvs);
411 ++i;
412 }
413}

417Trace TraceNode::WithDecision(Instruction inst, ObjectRef decision, bool remove_postproc) const {
418 int n_insts = GetNumValidInstructions(this->insts, remove_postproc);
419 Array<Instruction> new_insts =
420 Array<Instruction>{this->insts.begin(), this->insts.begin() + n_insts};
421 Map<Instruction, ObjectRef> new_decisions{this->decisions.begin(), this->decisions.end()};
422 new_decisions.Set(std::move(inst), std::move(decision));
423 return Trace(new_insts, new_decisions);
424}
425
426Trace TraceNode::Simplified(bool remove_postproc) const {
427 int n_insts = GetNumValidInstructions(this->insts, remove_postproc);
428 std::unordered_set<const Object*> used_rvs;
429 std::vector<Instruction> new_insts;
430 std::unordered_map<Instruction, ObjectRef, ObjectPtrHash, ObjectPtrEqual> new_decisions;
431 new_insts.reserve(n_insts);
432 new_decisions.reserve(this->decisions.size());
433 for (int inst_idx = n_insts - 1; inst_idx >= 0; --inst_idx) {
434 const Instruction& inst = this->insts[inst_idx];
435 // Check if all the variables the instruction defined are dead
436 // If so, and the instruction is pure, we can safely remove this instruction
437 bool all_defs_dead = inst->kind->is_pure;
438 if (all_defs_dead) {
439 for (const ObjectRef& obj : inst->outputs) {
440 if (used_rvs.count(obj.get())) {
441 all_defs_dead = false;
442 break;
443 }
444 }
445 }
446 // Remove this instruction
447 if (all_defs_dead) {
448 continue;
449 }
450 // Otherwise this instruction is not dead
451 new_insts.push_back(inst);
452 if (Optional<ObjectRef> decision = this->GetDecision(inst)) {
453 new_decisions.emplace(inst, std::move(decision));
454 }
455 // Add its inputs as "used" ones
456 for (const ObjectRef& obj : inst->inputs) {
457 if (!obj.defined()) {
458 continue;
459 } else if (obj->IsInstance<BlockRVNode>() || obj->IsInstance<LoopRVNode>() ||
460 obj->IsInstance<VarNode>()) {
461 used_rvs.insert(obj.get());
462 continue;
463 } else if (obj->IsInstance<PrimExprNode>()) {
464 PostOrderVisit(obj, [&used_rvs](const ObjectRef& obj) -> void {
465 if (obj->IsInstance<VarNode>()) {
466 used_rvs.insert(obj.get());
467 }
468 });
469 }
470 }
471 }
472 return Trace(Array<Instruction>(new_insts.rbegin(), new_insts.rend()),
473 Map<Instruction, ObjectRef>(new_decisions));
474}
