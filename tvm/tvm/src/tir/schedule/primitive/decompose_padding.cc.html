25/*! \brief Information used to create new padding block */
26struct PaddingBlockInfo {
27 /*! \brief In-bound block iter regions, wrt loop vars. */
28 Array<Range> in_bound_region;
29 /*! \brief In-bound value, wrt block iter vars. */
30 PrimExpr in_bound_value;
31 /*! \brief Condition of in-bound write, wrt loop vars. */
32 PrimExpr in_bound_predicate;
33 /*! \brief Padding value, should be a constant. */
34 PrimExpr pad_value;
35};

37class PaddingPatternMatchError : public ScheduleError {

39 PaddingPatternMatchError(IRModule mod, Block block, const std::string& error_msg)
40 : mod_(std::move(mod)), block_(std::move(block)), error_msg_(error_msg) {}

62/*!
63 * \brief Helper class to analyze and check the padding pattern of the block,
64 * then return the padding information.
65 */
66class PaddingInfoAnalyzer {
67 public:
68 static PaddingBlockInfo CheckAndGetPaddingInfo(IRModule mod, const BlockRealizeNode* realize,
69 const Map<Var, Range>& dom_map,
70 arith::Analyzer* analyzer) {
71 PaddingInfoAnalyzer padding_analyzer(analyzer);
72 if (!padding_analyzer.MatchPadding(realize, dom_map)) {
73 throw PaddingPatternMatchError(mod, realize->block, padding_analyzer.error_msg_);
74 }
75 return padding_analyzer.info_;
76 }
77
78 private:
79 explicit PaddingInfoAnalyzer(arith::Analyzer* analyzer) : analyzer_(analyzer) {}
80
81 /*! \brief Detect padding pattern and update result. */
82 bool MatchPadding(const BlockRealizeNode* realize, const Map<Var, Range>& dom_map) {
83 // Step 1. Check match padding computation pattern.
84 // A[...] = T.if_then_else(predicate, B[...], imm)
85 Block block = realize->block;
86 std::unordered_map<const VarNode*, PrimExpr> iter_values;
87 for (size_t i = 0; i < realize->iter_values.size(); ++i) {
88 Var block_var = block->iter_vars[i]->var;
89 iter_values[block_var.get()] = realize->iter_values[i];
90 }
91 const BufferStoreNode* store = block->body.as<BufferStoreNode>();
92 if (!store) {
93 SetError("Block body expect a BufferStore to the write buffer");
94 return false;
95 }
96 const CallNode* if_then_else = store->value.as<CallNode>();
97 if (!if_then_else || !if_then_else->op.same_as(tir::builtin::if_then_else())) {
98 SetError("Value of BufferStore expect to be constrained by a padding predicate");
99 return false;
100 }
101 PrimExpr pad_predicate = Substitute(if_then_else->args[0], iter_values);
102 PrimExpr in_bound_value = if_then_else->args[1];
103 PrimExpr pad_value = if_then_else->args[2];
104 if (!is_const_number(pad_value)) {
105 SetError("Pad value should be constant");
106 return false;
107 }
108
109 // Step 2. Check in-bound computation to be effectiveless.
110 if (SideEffect(if_then_else->args[1]) > CallEffectKind::kReadState) {
111 SetError("Inbound computation should not have side-effect");
112 return false;
113 }
114
115 // Step 3. Analyze in-bound write region.
116 PrimExpr in_bound_predicate = RewritePredicate(pad_predicate && realize->predicate);
117 Array<Range> in_bound_region = this->EstimateInBoundRegion(
118 /*iter_values=*/realize->iter_values, /*dom_map=*/dom_map,
119 /*in_bound_predicate=*/in_bound_predicate);
120 if (in_bound_region.empty()) {
121 return false;
122 }
123
124 // Step 4. Update result information.
125 info_.in_bound_value = if_then_else->args[1];
126 info_.in_bound_region = in_bound_region;
127 info_.in_bound_predicate = in_bound_predicate;
128 info_.pad_value = pad_value;
129 return true;
130 }
131
132 /*! \brief Rewrite predicate to left recursive conjunction, drop likely annotation. */
133 PrimExpr RewritePredicate(const PrimExpr& predicate) {
134 PrimExpr res = const_true();
135 std::function<void(PrimExpr)> update = [&res, &update](PrimExpr e) {
136 arith::PVar<PrimExpr> a, b;
137 if ((a && b).Match(e)) {
138 update(a.Eval());
139 update(b.Eval());
140 } else {
141 if (const CallNode* call = e.as<CallNode>()) {
142 if (call->op.same_as(builtin::likely())) {
143 e = call->args[0];
144 }
145 }
146 res = res && e;
147 }
148 };
149 update(predicate);
150 return analyzer_->Simplify(res);
151 }
152
153 /*! \brief Return iteration region of block vars where the padding predicate evals to true. */
154 Array<Range> EstimateInBoundRegion(const Array<PrimExpr>& iter_values,
155 const Map<Var, Range>& dom_map,
156 const PrimExpr& in_bound_predicate) {
157 Array<Range> region;
158
159 auto res = arith::DetectIterMap(iter_values, dom_map, in_bound_predicate,
160 arith::IterMapLevel::Surjective, analyzer_);
161 if (res->indices.empty()) {
162 SetError("Block iters are not independent wrt padding condition");
163 return {};
164 }
165 for (const arith::IterSumExpr& sum : res->indices) {
166 if (sum->args.empty()) {
167 region.push_back(Range::FromMinExtent(sum->base, 1));
168 } else {
169 ICHECK_EQ(sum->args.size(), 1U);
170 if (!analyzer_->CanProveEqual(sum->args[0]->scale, 1)) {
171 SetError("Strided iteration is not supported");
172 return {};
173 }
174 region.push_back(Range::FromMinExtent(sum->base, sum->args[0]->extent));
175 }
176 }
177 return region;
178 }
179
180 void SetError(const std::string& msg) { error_msg_ = msg; }
181
182 /*! \brief padding info analyse result. */
183 PaddingBlockInfo info_;
184 /*! \brief current error message. */
185 std::string error_msg_;
186 /*! \brief arithmetic analyzer. */
187 arith::Analyzer* analyzer_;
188};

190/*! \brief Create block to fill constant pad values into full region */
191static std::pair<Stmt, BlockRealize> CreateConstBlock(const BlockRealizeNode* realize,
192 const PaddingBlockInfo& info,
193 const Array<For>& loops,
194 const Stmt& highest_pos_inclusive,
195 arith::Analyzer* analyzer) {
196 const Block& block = realize->block;
197 Array<IterVar> new_iter_vars;
198 Map<Var, PrimExpr> repl_dict;
199
200 // create new block itervars
201 for (size_t i = 0; i < block->iter_vars.size(); ++i) {
202 const IterVar& origin_iter = block->iter_vars[i];
203 Var new_var = origin_iter->var.copy_with_suffix("");
204 new_iter_vars.push_back(IterVar(origin_iter->dom, new_var, IterVarType::kDataPar));
205 repl_dict.Set(origin_iter->var, new_var);
206 }
207
208 // rewrite expr helper
209 auto rewrite_expr = [&repl_dict, analyzer](const PrimExpr& e) {
210 return analyzer->Simplify(Substitute(e, repl_dict));
211 };
212
213 // create new write region
214 ICHECK_EQ(block->writes.size(), 1U);
215 BufferRegion write_region =
216 BufferRegion(block->writes[0]->buffer,
217 MutateArray(block->writes[0]->region, [rewrite_expr](const Range& r) {
218 return Range::FromMinExtent(rewrite_expr(r->min), rewrite_expr(r->extent));
219 }));
220
221 // create block to fill const pad values
222 BufferStore store = Downcast<BufferStore>(block->body);
223 store.CopyOnWrite()->value = info.pad_value;
224 store.CopyOnWrite()->indices = MutateArray(store->indices, rewrite_expr);
225 Block new_block(/*iter_vars=*/new_iter_vars, /*reads=*/{}, /*writes=*/{write_region},
226 /*name_hint=*/block->name_hint + "_pad_const", /*body=*/std::move(store));
227
228 // create new loop vars
229 Array<Var> new_loop_vars;
230 for (const For& loop : loops) {
231 Var new_var = loop->loop_var.copy_with_suffix("");
232 new_loop_vars.push_back(new_var);
233 repl_dict.Set(loop->loop_var, new_var);
234 if (loop.same_as(highest_pos_inclusive)) {
235 break;
236 }
237 }
238
239 // create new block realize node
240 Array<PrimExpr> new_iter_values;
241 for (size_t i = 0; i < realize->iter_values.size(); ++i) {
242 new_iter_values.push_back(rewrite_expr(realize->iter_values[i]));
243 }
244 BlockRealize new_realize(/*iter_values=*/new_iter_values,
245 /*predicate=*/rewrite_expr(realize->predicate),
246 /*block=*/new_block);
247
248 // create new loops
249 Stmt nest_stmt_root = new_realize;
250 for (size_t i = 0; i < new_loop_vars.size(); ++i) {
251 For loop = loops[i];
252 nest_stmt_root =
253 For(new_loop_vars[i], loop->min, loop->extent, ForKind::kSerial, nest_stmt_root);
254 }
255
256 return {nest_stmt_root, new_realize};
257}

259/*! \brief Create block to fill in-bound region values. */
260static std::pair<Stmt, BlockRealize> CreateInBoundBlock(const BlockRealizeNode* realize,
261 const PaddingBlockInfo& info,
262
263 const Array<For>& loops,
264 const Stmt& highest_pos_inclusive,
265 arith::Analyzer* analyzer) {
266 const Block& block = realize->block;
267 Array<IterVar> new_iter_vars;
268 Map<Var, PrimExpr> repl_dict;
269
270 // record loop ranges to be mutated
271 Map<Var, Range> new_loop_ranges;
272 for (const For& loop : loops) {
273 new_loop_ranges.Set(loop->loop_var, Range::FromMinExtent(loop->min, loop->extent));
274 if (loop.same_as(highest_pos_inclusive)) {
275 break;
276 }
277 }
278
279 // create new block iter vars and iter bindings
280 Array<PrimExpr> new_iter_binding;
281 for (size_t i = 0; i < info.in_bound_region.size(); ++i) {
282 // add new block itervar
283 const IterVar& origin_itervar = block->iter_vars[i];
284 Var new_var = origin_itervar->var.copy_with_suffix("");
285 Range new_range =
286 Range::FromMinExtent(make_const(new_var->dtype, 0), info.in_bound_region[i]->extent);
287 new_iter_vars.push_back(IterVar(new_range, new_var, IterVarType::kDataPar));
288 repl_dict.Set(origin_itervar->var, new_var + info.in_bound_region[i]->min);
289
290 // update new loop range
291 Var loop_var = GetRef<Var>(realize->iter_values[i].as<VarNode>());
292 if (loop_var.defined() && new_loop_ranges.count(loop_var)) {
293 // if the block binding is the loop var with single child, mutate loop range
294 // instead of insert extra block predicate
295 new_loop_ranges.Set(loop_var, new_range);
296 new_iter_binding.push_back(realize->iter_values[i]);
297 repl_dict.Set(loop_var, loop_var + info.in_bound_region[i]->min);
298 analyzer->Bind(loop_var, new_range, /*allow_override=*/true);
299 } else {
300 new_iter_binding.push_back(
301 analyzer->Simplify(realize->iter_values[i] - info.in_bound_region[i]->min));
302 }
303 }
304
305 // rewrite helpers
306 auto rewrite_expr = [&repl_dict, analyzer](const PrimExpr& e) {
307 return analyzer->Simplify(Substitute(e, repl_dict));
308 };
309 auto rewrite_region = [rewrite_expr](const Region& region) {
310 return MutateArray(region, [rewrite_expr](const Range& r) {
311 return Range::FromMinExtent(rewrite_expr(r->min), rewrite_expr(r->extent));
312 });
313 };
314
315 // create new read/write region for in-bound accesses
316 Array<BufferRegion> reads, writes;
317 for (const BufferRegion& read : block->reads) {
318 reads.push_back(BufferRegion(read->buffer, rewrite_region(read->region)));
319 }
320 for (const BufferRegion& write : block->writes) {
321 writes.push_back(BufferRegion(write->buffer, rewrite_region(write->region)));
322 }
323
324 // create new block realize node
325 BufferStore store = Downcast<BufferStore>(block->body);
326 store.CopyOnWrite()->value = rewrite_expr(info.in_bound_value);
327 store.CopyOnWrite()->indices = MutateArray(store->indices, rewrite_expr);
328 Block new_block(/*iter_vars=*/new_iter_vars, /*reads=*/reads, /*writes=*/writes,
329 /*name_hint=*/block->name_hint, /*body=*/std::move(store));
330 PrimExpr new_predicate = rewrite_expr(info.in_bound_predicate);
331 BlockRealize new_realize(/*iter_values=*/new_iter_binding, /*predicate=*/new_predicate,
332 /*block=*/new_block);
333
334 // create new loops
335 Stmt nest_stmt_root = new_realize;
336 for (const For& loop : loops) {
337 auto it = new_loop_ranges.find(loop->loop_var);
338 PrimExpr min = it == new_loop_ranges.end() ? loop->min : (*it).second->min;
339 PrimExpr extent = it == new_loop_ranges.end() ? loop->extent : (*it).second->extent;
340 nest_stmt_root = For(loop->loop_var, min, extent, loop->kind, nest_stmt_root,
341 loop->thread_binding, loop->annotations, loop->span);
342 if (loop.same_as(highest_pos_inclusive)) {
343 break;
344 }
345 }
346 return {nest_stmt_root, new_realize};
347}

349/*!
350 * \brief A helper class to create a new scope that contains decomposed padding blocks.
351 */
352class DecomposePaddingBlockReplacer : public StmtMutator {
353 public:
354 /*! \brief Replacement information */
355 struct ReplaceDesc {
356 /*! \brief loop above which to insert const pad value filling code. */
357 For const_filling_pos;
358 /*! \brief loop under which to insert in bound value filling code. */
359 For in_bound_filling_pos;
360 /*! \brief const pad value filling loop. */
361 Stmt const_filling_loop;
362 /*! \brief highest in bound value filling loop with single child. */
363 Stmt in_bound_filling_loop;
364 /*! \brief const pad value filling block. */
365 BlockRealize const_filling_block;
366 /*! \brief in bound value filling block. */
367 BlockRealize in_bound_filling_block;
368 };
369
370 static Block Replace(Block scope_root, const ReplaceDesc& desc) {
371 DecomposePaddingBlockReplacer replacer(desc);
372 return Downcast<Block>(replacer(std::move(scope_root)));
373 }
374
375 private:
376 explicit DecomposePaddingBlockReplacer(const ReplaceDesc& desc) : desc_(desc) {}
377
378 Stmt VisitStmt_(const ForNode* op) final {
379 Stmt new_loop;
380 if (op == desc_.in_bound_filling_pos.get()) {
381 // position to rewrite inbound filling code
382 new_loop = desc_.in_bound_filling_loop;
383 } else {
384 new_loop = StmtMutator::VisitStmt_(op);
385 }
386 if (op == desc_.const_filling_pos.get()) {
387 // position to insert pad value filling code
388 return std::move(SeqStmt({desc_.const_filling_loop, new_loop}));
389 }
390 return std::move(new_loop);
391 }
392
393 Stmt VisitStmt_(const SeqStmtNode* seq) final {
394 Array<Stmt> new_stmts;
395 new_stmts.reserve(seq->seq.size());
396 for (const Stmt& old_stmt : seq->seq) {
397 new_stmts.push_back(VisitStmt(old_stmt));
398 }
399 return SeqStmt::Flatten(new_stmts);
400 }
401
402 private:
403 const ReplaceDesc& desc_;
404};

406StmtSRef DecomposePaddingImpl(ScheduleState self, const StmtSRef& block_sref,
407 const StmtSRef& loop_sref, bool check_only) {
408 /*!
409 * Check
410 * - the block is a compact block
411 * - the loop is an ancester of the block
412 * - the block match padding pattern
413 * Mutate
414 * - generate new block to fill padding values
415 * - trim original block to write non-padding part only
416 */
417 // Condition Checks and Information Collection
418 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
419 const BlockRealizeNode* realize = GetBlockRealize(self, block_sref).get();
420 Map<Var, Range> dom_map;
421 arith::Analyzer analyzer;
422
423 // Check 1. check the block is complete.
424 StmtSRef scope_root_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
425 CheckCompleteBlock(self, block_sref, scope_root_sref);
426
427 // Check 2. Check loop_sref is an ancestor of block_sref. Also collect
428 // - the highest loop position (inclusive) to insert const pad value filling code above.
429 // - the highest loop position (inclusive) to replace with in-bound value filling code.
430 Array<StmtSRef> loop_srefs = GetLoops(block_sref);
431 Array<For> loops;
432 bool found_const_filling_pos = false;
433 bool found_in_bound_filling_pos = false;
434 For const_filling_pos = GetRef<For>(loop_sref->StmtAs<ForNode>());
435 For in_bound_filling_pos{nullptr};
436 for (auto it = loop_srefs.rbegin(); it != loop_srefs.rend(); ++it) {
437 For cur_loop = GetRef<For>((*it)->StmtAs<ForNode>());
438 Range range = Range::FromMinExtent(cur_loop->min, cur_loop->extent);
439 dom_map.Set(cur_loop->loop_var, range);
440 analyzer.Bind(cur_loop->loop_var, range);
441 loops.push_back(cur_loop);
442
443 if (!found_const_filling_pos) {
444 if (cur_loop.same_as(const_filling_pos)) {
445 found_const_filling_pos = true;
446 }
447 }
448
449 if (!found_in_bound_filling_pos) {
450 if (!cur_loop->body->IsInstance<ForNode>() &&
451 !cur_loop->body->IsInstance<BlockRealizeNode>()) {
452 found_in_bound_filling_pos = true;
453 } else {
454 in_bound_filling_pos = cur_loop;
455 }
456 }
457 }
458 ICHECK(in_bound_filling_pos.defined());
459 if (!found_const_filling_pos) {
460 throw LoopPositionError(self->mod, const_filling_pos, GetRef<Block>(block),
461 "decompose_padding");
462 }
463
464 // Check 3. match padding pattern and return padding operation info.
465 PaddingBlockInfo info =
466 PaddingInfoAnalyzer::CheckAndGetPaddingInfo(self->mod, realize, dom_map, &analyzer);
467
468 // IR Manipulation
469 // Step 1. Create const pad value filling part and in-bound value filling part.
470 DecomposePaddingBlockReplacer::ReplaceDesc replace_desc;
471 replace_desc.const_filling_pos = const_filling_pos;
472 replace_desc.in_bound_filling_pos = in_bound_filling_pos;
473 std::tie(replace_desc.const_filling_loop, replace_desc.const_filling_block) =
474 CreateConstBlock(realize, info, loops, const_filling_pos, &analyzer);
475 std::tie(replace_desc.in_bound_filling_loop, replace_desc.in_bound_filling_block) =
476 CreateInBoundBlock(realize, info, loops, in_bound_filling_pos, &analyzer);
477
478 // Step 2. Execute IR replacement.
479 Block old_scope_root_block = GetRef<Block>(scope_root_sref->StmtAs<BlockNode>());
480 Block new_scope_root = DecomposePaddingBlockReplacer::Replace(old_scope_root_block, replace_desc);
481 if (check_only) {
482 return block_sref;
483 }
484
485 // Step 3. Update schedule states.
486 self->Replace(scope_root_sref, new_scope_root,
487 {{old_scope_root_block, new_scope_root},
488 {GetRef<Block>(block), replace_desc.in_bound_filling_block->block}});
489 auto new_block_sref = self->stmt2ref.at(replace_desc.const_filling_block->block.get());
490
491 // Set block info of created const pad value filling block
492 BlockInfo& block_info = self->block_info[new_block_sref];
493 block_info.affine_binding = true;
494 block_info.region_cover = true;
495 block_info.scope->stage_pipeline = true;
496
497 // If the const pad value filling block is lifted out of the original subtree,
498 // set the region_cover flag as false since region_cover is the property under the subtree.
499 bool preserve_stage_pipeline = true;
500 for (const StmtSRef& consumer_sref : GetConsumers(self, block_sref)) {
501 StmtSRef lca = GetSRefLowestCommonAncestor({consumer_sref, block_sref});
502 const StmtSRefNode* parent = new_block_sref->parent;
503 bool is_under_lca = false;
504 while (parent) {
505 if (parent == lca.get()) {
506 is_under_lca = true;
507 break;
508 }
509 parent = parent->parent;
510 }
511 if (!is_under_lca) {
512 preserve_stage_pipeline = false;
513 self->block_info[consumer_sref].region_cover = false;
514 }
515 }
516 if (!preserve_stage_pipeline) {
517 self->block_info[scope_root_sref].scope->stage_pipeline = false;
518 }
519 return new_block_sref;
520}
521
522StmtSRef DecomposePadding(ScheduleState self, const StmtSRef& block_sref,
523 const StmtSRef& loop_sref) {
524 return DecomposePaddingImpl(self, block_sref, loop_sref, false);
525}
526
527bool CanDecomposePadding(ScheduleState self, const StmtSRef& block_sref,
528 const StmtSRef& loop_sref) {
529 try {
530 DecomposePaddingImpl(self, block_sref, loop_sref, true);
531 } catch (const tvm::runtime::Error& e) {
532 return false;
533 }
534 return true;
535}
