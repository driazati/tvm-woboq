41/*!
42 * \brief ScheduleError that the bindings of the inner block are not divisible by the subspace
43 * represented by the outer loops.
44 */
45class SubspaceNotDivisibleError : public ScheduleError {
46 public:
47 explicit SubspaceNotDivisibleError(IRModule mod, For scope_loop, Block inner_block)
48 : mod_(std::move(mod)),
49 scope_loop_(std::move(scope_loop)),
50 inner_block_(std::move(inner_block)) {}
51
52 String FastErrorString() const final {
53 return "ScheduleError: The bindings of the inner block can not be blockized.";
54 }
55
56 String DetailRenderTemplate() const final {
57 return "ScheduleError: The bindings of the inner block {0} can not be blockized by the loops "
58 "starting at {1}.";
59 }
60
61 IRModule mod() const final { return mod_; }
62
63 Array<ObjectRef> LocationsOfInterest() const final { return {inner_block_, scope_loop_}; }
64
65 private:
66 IRModule mod_;
67 For scope_loop_;
68 Block inner_block_;
69};

71/*!
72 * \brief Detect if bindings are a trivial case of the subspace division where we can divide the
73 * block iter bindings into two categories:
74 * 1. The binding covers no inner loop vars.
75 * 2. The binding covers only inner loop vars.
76 *
77 * The bindings are not required to be quasi-affine.
78 *
79 * \param iter_vars The input iterators
80 * \param bindings The values of iter_vars
81 * \param predicate The predicate constraint on the input iterators.
82 * \param outer_iters The iters of the outer space
83 * \param inner_iters The iters of the inner space
84 * \return The result of the subspace division.
85 */
86Array<Array<arith::IterMark>> TrivialSubspaceDivision(const Array<IterVar>& iter_vars,
87 const Array<PrimExpr>& bindings,
88 const PrimExpr& predicate,
89 const Array<Var>& outer_iters,
90 const Array<Var>& inner_iters) {
91 if (!is_one(predicate)) return {};
92 Array<Array<arith::IterMark>> res;
93 std::unordered_set<const VarNode*> outer_loop_vars;
94 std::unordered_set<const VarNode*> inner_loop_vars;
95
96 auto make_uses_var = [](const Array<Var>& vars) -> std::function<bool(const PrimExpr& expr)> {
97 std::unordered_set<const VarNode*> var_set;
98 var_set.reserve(vars.size());
99 for (const Var& var : vars) {
100 var_set.insert(var.get());
101 }
102 return [var_set = std::move(var_set)](const PrimExpr& expr) -> bool {
103 return UsesVar(expr, [&var_set](const VarNode* var) {
104 return var_set.count(var); //
105 });
106 };
107 };
108 auto use_outer_loop_vars = make_uses_var(outer_iters);
109 auto use_inner_loop_vars = make_uses_var(inner_iters);
110 arith::IterMark unit_iter_mark(arith::IterSumExpr({}, 0), 1);
111
112 for (int i = 0, n = bindings.size(); i < n; ++i) {
113 bool outer = use_outer_loop_vars(bindings[i]);
114 bool inner = use_inner_loop_vars(bindings[i]);
115 arith::IterMark iter_mark;
116 if (bindings[i]->IsInstance<VarNode>()) {
117 iter_mark = arith::IterMark(
118 arith::IterSplitExpr(arith::IterMark(bindings[i], iter_vars[i]->dom->extent)),
119 iter_vars[i]->dom->extent);
120 } else {
121 iter_mark = arith::IterMark(arith::IterSumExpr({}, bindings[i]), iter_vars[i]->dom->extent);
122 }
123 if (outer && !inner) {
124 res.push_back({/*outer_iter=*/iter_mark, /*inner_iter=*/unit_iter_mark});
125 } else if (inner && !outer) {
126 res.push_back({/*outer_iter=*/unit_iter_mark, /*inner_iter=*/iter_mark});
127 } else if (!outer && !inner) {
128 res.push_back({/*outer_iter=*/unit_iter_mark, /*inner_iter=*/unit_iter_mark});
129 } else {
130 return {};
131 }
132 }
133 res.push_back({arith::IterMark(arith::IterSumExpr({}, 0), Bool(true)),
134 arith::IterMark(arith::IterSumExpr({}, 0), Bool(true))});
135 return res;
136}

138/*!
139 * \brief Subspace division. The space is divided into two subspaces:
140 * 1. The subspace represented by the outer loops above `loop_sref` (exclusive).
141 * 2. The subspace represented by the inner loops below `loop_sref` (inclusive).
142 * \param realize The inner block
143 * \param block_sref The sref to the inner block
144 * \param loop_sref The loop that is the root of the second subspace.
145 * \param loops The loops that represents the second part of the subspace.
146 * \param analyzer The arithmetic analyzer to use.
147 */
148Array<Array<arith::IterMark>> SubspaceDivide(const BlockRealize& realize,
149 const StmtSRef& block_sref, //
150 const StmtSRef& loop_sref, //
151 std::vector<const ForNode*>* loops,
152 arith::Analyzer* analyzer) {
153 Array<Var> inner_vars;
154 Array<Var> outer_vars;
155 Map<Var, Range> loop_var_domain;
156 bool inner = true;
157 for (StmtSRefNode* sref = block_sref->parent; //
158 sref && sref->stmt->IsInstance<ForNode>(); //
159 sref = sref->parent) {
160 const ForNode* loop = static_cast<const ForNode*>(sref->stmt);
161 if (inner) {
162 loops->push_back(loop);
163 inner_vars.push_back(loop->loop_var);
164 } else {
165 outer_vars.push_back(loop->loop_var);
166 }
167 loop_var_domain.Set(loop->loop_var, Range::FromMinExtent(loop->min, loop->extent));
168 if (sref == loop_sref.get()) {
169 inner = false;
170 }
171 }
172 Array<Array<arith::IterMark>> result =
173 arith::SubspaceDivide(realize->iter_values, loop_var_domain, inner_vars, realize->predicate,
174 arith::IterMapLevel::Surjective, analyzer);
175 if (!result.empty()) {
176 return result;
177 }
178 return TrivialSubspaceDivision(realize->block->iter_vars,
179 realize->iter_values, //
180 realize->predicate, //
181 outer_vars, inner_vars);
182}

184/*!
185 * \brief Derive the block bindings for both inner and outer block
186 * \param iter_vars The original block iterators to the inner block
187 * \param division The subspace division.
188 * \param outer_iter_vars The outer block iterators.
189 * \param outer_bindings The outer block bindings.
190 * \param inner_iter_vars The inner block iterators.
191 * \param inner_bindings The inner block bindings.
192 * \return A substitution plan to the iterators in the original inner block.
193 */
194Map<Var, PrimExpr> DeriveBlockBinding(const Array<IterVar>& iter_vars, //
195 const Array<Array<arith::IterMark>>& division, //
196 Array<IterVar>* outer_iter_vars, //
197 Array<PrimExpr>* outer_bindings, //
198 Array<IterVar>* inner_iter_vars, //
199 Array<PrimExpr>* inner_bindings) {
200 using arith::IterMapExpr;
201 using arith::IterMapExprNode;
202 using arith::NormalizeIterMapToExpr;
203 Map<Var, PrimExpr> block_var_subst;
204 ICHECK_EQ(iter_vars.size() + 1, division.size());
205 for (int i = 0, n = iter_vars.size(); i < n; ++i) {
206 const IterVar& iter_var = iter_vars[i];
207 arith::IterMark outer_mark = division[i][0];
208 arith::IterMark inner_mark = division[i][1];
209 IterMapExpr outer_binding = Downcast<IterMapExpr>(outer_mark->source);
210 IterMapExpr inner_binding = Downcast<IterMapExpr>(inner_mark->source);
211 // After computing the subspace division, bindings[i] can be written as
212 // outer_binding * inner_binding->extent + inner_binding
213 // The outer block will have binding: iter_outer -> outer_binding
214 // The inner block will have binding: iter_inner -> inner_binding
215 // The iter in the original block will be substituted with base + iter_inner where
216 // base == iter_outer * iter_inner_extent
217 if (is_one(inner_mark->extent)) { // IsOuter
218 // extract this iter var to outer block directly
219 outer_bindings->push_back(NormalizeIterMapToExpr(outer_binding));
220 outer_iter_vars->push_back(iter_var);
221 continue;
222 }
223 // create iter var for the outer block
224 IterVar outer_iter(/*dom=*/RangeFromExtent(outer_mark->extent),
225 /*var=*/iter_var->var.copy_with_suffix("_o"),
226 /*iter_type=*/iter_var->iter_type);
227 outer_bindings->push_back(NormalizeIterMapToExpr(outer_binding));
228 outer_iter_vars->push_back(outer_iter);
229 // create iter var for the inner block
230 IterVar inner_iter(/*dom=*/RangeFromExtent(inner_mark->extent),
231 /*var=*/iter_var->var.copy_with_suffix("_i"),
232 /*iter_type=*/iter_var->iter_type);
233 inner_bindings->push_back(NormalizeIterMapToExpr(inner_binding));
234 inner_iter_vars->push_back(inner_iter);
235 // substitution
236 PrimExpr sub{nullptr};
237 if (is_one(outer_mark->extent)) {
238 sub = inner_iter->var;
239 } else {
240 sub = outer_iter * inner_mark->extent + inner_iter->var;
241 }
242 block_var_subst.Set(iter_var->var, sub);
243 }
244 return block_var_subst;
245}

247/*!
248 * \brief Generate the inner block for blockization
249 * \param is_write_reduction Whether the write regions of the inner block are actually reduction.
250 * \param iter_vars IterVars used in the inner block.
251 * \param iter_values IterVar bindings used in the inner block.
252 * \param predicate The predicate of the inner block.
253 * \param block The inner block as a template to be created from. This method will modify its
254 * `iter_vars`, `init` and `reads` fields.
255 * \return The inner block created.
256 */
257BlockRealize GenerateInner(bool is_write_reduction,
258 const Array<IterVar>& iter_vars, //
259 const Array<PrimExpr>& iter_values, //
260 const PrimExpr& predicate, //
261 Block block) {
262 BlockNode* n = block.CopyOnWrite();
263 n->iter_vars = iter_vars;
264 n->init = NullOpt;
265 if (is_write_reduction) {
266 Array<BufferRegion> reads;
267 reads.reserve(block->writes.size() + block->reads.size());
268 reads.insert(reads.end(), block->writes.begin(), block->writes.end());
269 reads.insert(reads.end(), block->reads.begin(), block->reads.end());
270 n->reads = std::move(reads);
271 }
272 return BlockRealize(/*iter_values=*/iter_values, /*predicate=*/predicate,
273 /*block=*/block);
274}

276/*!
277 * \brief Generate the init stmt for the outer block
278 * \param block The original block with init.
279 * \param inner_realize The block realize of the inner block after blockize.
280 * \param loops The inner loops after blockize.
281 * \return The subtree of the init block and its outer loops.
282 */
283Stmt GenerateOuterInit(const Stmt& block_init, const BlockRealize& inner_realize,
284 const std::vector<const ForNode*>& loops, String block_name) {
285 const Block& inner_block = inner_realize->block;
286 Map<Var, PrimExpr> subst_map;
287 // Step 1: Create new block vars for the block inside the init stmt of outer block
288 // A iter is used in the block if
289 // 1) It is data parallel
290 // 2) It is used in the original init block
291 Array<IterVar> iter_vars;
292 Array<PrimExpr> iter_values;
293 ICHECK_EQ(inner_block->iter_vars.size(), inner_realize->iter_values.size());
294 int n = inner_block->iter_vars.size();
295 iter_vars.reserve(n);
296 iter_values.reserve(n);
297 for (int i = 0; i < n; ++i) {
298 const IterVar& old_iter_var = inner_block->iter_vars[i];
299 const PrimExpr& iter_value = inner_realize->iter_values[i];
300 if (old_iter_var->iter_type == IterVarType::kDataPar &&
301 UsesVar(block_init, old_iter_var->var)) {
302 ObjectPtr<IterVarNode> new_iter_var = make_object<IterVarNode>(*old_iter_var.get());
303 new_iter_var->var = new_iter_var->var.copy_with_suffix("_init");
304 subst_map.Set(old_iter_var->var, new_iter_var->var);
305 iter_vars.push_back(IterVar(new_iter_var));
306 iter_values.push_back(iter_value);
307 }
308 }
309 // Step 2: Generate the block inside init stmt of outer block
310 Stmt stmt = BlockRealize(
311 /*iter_values=*/iter_values,
312 /*predicate=*/inner_realize->predicate,
313 /*block=*/
314 Block(/*iter_vars=*/iter_vars,
315 /*reads=*/{},
316 /*writes=*/inner_block->writes,
317 /*name_hint=*/block_name,
318 /*body=*/block_init,
319 /*init=*/NullOpt));
320 // Step 3. Create the loop nest on top of the block
321 for (const ForNode* loop : loops) {
322 bool is_init_loop = false;
323 for (const PrimExpr& init_binding : iter_values) {
324 if (UsesVar(init_binding, loop->loop_var)) {
325 is_init_loop = true;
326 break;
327 }
328 }
329 if (is_init_loop) {
330 ObjectPtr<ForNode> new_loop = make_object<ForNode>(*loop);
331 new_loop->loop_var = loop->loop_var.copy_with_suffix("");
332 new_loop->body = std::move(stmt);
333 subst_map.Set(loop->loop_var, new_loop->loop_var);
334 stmt = For(new_loop);
335 }
336 }
337 // Step 4: Substitute the iter vars and loop vars
338 return Substitute(stmt, subst_map);
339}

341/*!
342 * \brief Substitute variables in the stmt, do simplification and track block substitution
343 * \param stmt The stmt to be substituted.
344 * \param sub The substitution map.
345 * \param block_sref_reuse The block substitution happens during the substitution.
346 * \param analyzer The analyzer for arithmetic simplification.
347 * \return The substituted stmt.
348 */
349Stmt Substitute(const Stmt& stmt, const Map<Var, PrimExpr>& sub,
350 Map<Block, Block>* block_sref_reuse, arith::Analyzer* analyzer) {
351 struct Replacer : public StmtExprMutator {
352 explicit Replacer(const Map<Var, PrimExpr>& sub, Map<Block, Block>* block_sref_reuse,
353 arith::Analyzer* analyzer)
354 : sub_(sub), block_sref_reuse_(block_sref_reuse), analyzer_(analyzer) {}
355
356 PrimExpr VisitExpr(const PrimExpr& op) final {
357 PrimExpr result = StmtExprMutator::VisitExpr(op);
358 if (!result.same_as(op)) {
359 return analyzer_->Simplify(result);
360 }
361 return result;
362 }
363
364 PrimExpr VisitExpr_(const VarNode* op) final {
365 if (Optional<PrimExpr> e = sub_.Get(GetRef<Var>(op))) {
366 return e.value();
367 }
368 return StmtExprMutator::VisitExpr_(op);
369 }
370
371 Stmt VisitStmt_(const BlockNode* op) final {
372 Block src = GetRef<Block>(op);
373 Block tgt = Downcast<Block>(StmtExprMutator::VisitStmt_(op));
374 if (!src.same_as(tgt)) {
375 block_sref_reuse_->Set(src, tgt);
376 }
377 return std::move(tgt);
378 }
379
380 const Map<Var, PrimExpr>& sub_;
381 Map<Block, Block>* block_sref_reuse_;
382 arith::Analyzer* analyzer_;
383 };
384 return Replacer(sub, block_sref_reuse, analyzer)(stmt);
385}

387/*!
388 * \brief Relax the variables for the given regions
389 * \param regions The regions to be relaxed.
390 * \param dom_map The variables to be relaxed
391 * \return The relaxed regions
392 */
393Array<BufferRegion> EvalSetRegions(const Array<BufferRegion>& regions,
394 const Map<Var, arith::IntSet>& dom_map) {
395 Array<BufferRegion> results;
396 results.reserve(regions.size());
397 for (const BufferRegion& buffer_region : regions) {
398 const Buffer& buffer = buffer_region->buffer;
399 Array<arith::IntSet> relaxed = arith::EvalSet(buffer_region->region, dom_map);
400 ICHECK_EQ(relaxed.size(), buffer->shape.size());
401 int ndim = buffer->shape.size();
402 Array<Range> new_region;
403 new_region.reserve(ndim);
404 for (int i = 0; i < ndim; ++i) {
405 new_region.push_back(relaxed[i].CoverRange(RangeFromExtent(buffer->shape[i])));
406 }
407 results.push_back(BufferRegion(buffer, new_region));
408 }
409 return results;
410}

412/*!
413 * \brief Create the loop nest on top of the given stmt.
414 * \param stmt The stmt to be wrapped.
415 * \param loops The loop nests
416 * \return The wrapped stmt.
417 */
418Stmt MakeLoopNest(Stmt stmt, const std::vector<const ForNode*>& loops) {
419 for (const ForNode* loop : loops) {
420 ObjectPtr<ForNode> new_loop = make_object<ForNode>(*loop);
421 new_loop->body = std::move(stmt);
422 stmt = For(new_loop);
423 }
424 return stmt;
425}

427BlockRealize BlockizeImpl(const ScheduleState& self, const StmtSRef& loop_sref,
428 Map<Block, Block>* block_sref_reuse, arith::Analyzer* analyzer) {
429 TVM_SREF_TO_FOR(loop_sref);
430 // Step 1: Check and get the only block under `loop`.
431 BlockRealize block_realize = CheckGetSingleChildBlockRealizeOnSRefTree(self, loop_sref);
432 Block block = block_realize->block;
433 StmtSRef block_sref = self->stmt2ref.at(block.get());
434 // Step 2: Derive subspace division
435 std::vector<const ForNode*> loops;
436 Array<Array<arith::IterMark>> division =
437 SubspaceDivide(block_realize, block_sref, loop_sref, &loops, analyzer);
438 if (division.empty()) {
439 throw SubspaceNotDivisibleError(self->mod, GetRef<For>(loops.back()), block);
440 }
441 PrimExpr outer_predicate = division.back()[0]->extent;
442 PrimExpr inner_predicate = division.back()[1]->extent;
443 // Step 3. Derive block bindings for both outer and inner block.
444 Array<IterVar> outer_iter_vars;
445 Array<IterVar> inner_iter_vars;
446 Array<PrimExpr> outer_bindings;
447 Array<PrimExpr> inner_bindings;
448 Map<Var, PrimExpr> block_var_subst = //
449 DeriveBlockBinding(block->iter_vars, division, //
450 &outer_iter_vars, &outer_bindings, //
451 &inner_iter_vars, &inner_bindings);
452 // Step 4: Do var substitution to adjust to the new block bindings
453 Map<Var, arith::IntSet> inner_iter_dom;
454 for (const IterVar& iter : inner_iter_vars) {
455 inner_iter_dom.Set(iter->var, arith::IntSet::FromRange(iter->dom));
456 analyzer->Bind(iter->var, iter->dom);
457 }
458 Block block_subst =
459 Downcast<Block>(Substitute(block, block_var_subst, block_sref_reuse, analyzer));
460 // Step 5: Generate the inner block. The write regions of the inner blocks will be reduction if
461 // 1. The original block has init stmt.
462 // 2. There are outer reduction iter vars.
463 bool has_outer_reduction = false;
464 if (block_subst->init.defined()) {
465 for (const IterVar& iter_var : outer_iter_vars) {
466 if (iter_var->iter_type == kCommReduce) {
467 has_outer_reduction = true;
468 break;
469 }
470 }
471 }
472 BlockRealize inner_realize = GenerateInner(/*is_write_reduction=*/has_outer_reduction,
473 /*iter_vars=*/inner_iter_vars,
474 /*iter_values*/ inner_bindings,
475 /*predicate=*/inner_predicate,
476 /*block=*/block_subst);
477 block_sref_reuse->Set(block, inner_realize->block);
478 // Step 6: Generate the outer block.
479 return BlockRealize(
480 /*iter_values=*/std::move(outer_bindings),
481 /*predicate=*/std::move(outer_predicate),
482 /*block=*/
483 Block(/*iter_vars=*/std::move(outer_iter_vars),
484 /*reads=*/EvalSetRegions(block_subst->reads, inner_iter_dom),
485 /*writes=*/EvalSetRegions(block_subst->writes, inner_iter_dom),
486 /*name_hint=*/block_subst->name_hint + "_o",
487 /*body=*/MakeLoopNest(inner_realize, loops),
488 /*init=*/
489 block_subst->init.defined() //
490 ? GenerateOuterInit(block_subst->init.value(), inner_realize, loops,
491 block_subst->name_hint + "_init")
492 : Optional<Stmt>(NullOpt)));
493}

495StmtSRef Blockize(ScheduleState self, const StmtSRef& loop_sref) {
496 arith::Analyzer analyzer;
497 Map<Block, Block> block_sref_reuse;
498 BlockRealize blockized = BlockizeImpl(self, loop_sref, &block_sref_reuse, &analyzer);
499 self->Replace(loop_sref, blockized, block_sref_reuse);
500 StmtSRef result = self->stmt2ref.at(blockized->block.get());
501 StmtSRef scope_root = tir::GetScopeRoot(self, result, /*require_stage_pipeline=*/false);
502 bool scope_block_affine_binding = self->IsAffineBlockBinding(scope_root);
503 self->UpdateScopeBlockInfo(tir::GetBlockRealize(self, scope_root));
504 self->block_info[scope_root].affine_binding = scope_block_affine_binding;
505 return result;
506}

508void Tensorize(ScheduleState self, const StmtSRef& sref, const TensorIntrin& intrin) {
509 // Step 1: Blockize the subtree rooted at the given loop if needed
510 BlockRealize block_realize{nullptr};
511 Optional<Block> old_block = NullOpt;
512 if (sref->stmt->IsInstance<BlockNode>()) {
513 block_realize = GetBlockRealize(self, sref);
514 old_block = block_realize->block;
515 } else if (sref->stmt->IsInstance<ForNode>()) {
516 arith::Analyzer analyzer;
517 Map<Block, Block> block_sref_reuse;
518 block_realize = BlockizeImpl(self, sref, &block_sref_reuse, &analyzer);
519 } else {
520 LOG(FATAL) << "TypeError: Tensorize only support For or Block, but gets: "
521 << GetRef<Stmt>(sref->stmt);
522 throw;
523 }
524 PrimFunc intrin_desc = intrin->desc;
525 PrimFunc intrin_impl = DeepCopy(intrin->impl);
526 // Step 2: Structural pattern matching
527 TensorizeComparator comparator(self->mod, /*assert_mode=*/true);
528 comparator.VisitStmt(block_realize, intrin_desc->body);
529 // Step 3: Prepare necessary mapping
530 // 1) Buffer mapping from intrin impl buffers to intrin desc buffers.
531 // 2) Buffer mapping from intrin impl buffers to buffers in the current AST.
532 // 3) Mapping impl buffers to their accessed regions.
533 std::unordered_map<Buffer, Buffer, ObjectPtrHash, ObjectPtrEqual> impl2desc;
534 ICHECK_EQ(intrin_desc->params.size(), intrin_impl->params.size());
535 for (int i = 0, n = intrin_desc->params.size(); i < n; ++i) {
536 const Buffer& desc = intrin_desc->buffer_map[intrin_desc->params[i]];
537 const Buffer& impl = intrin_impl->buffer_map[intrin_impl->params[i]];
538 impl2desc[impl] = desc;
539 }
540 std::unordered_map<Buffer, Buffer, ObjectPtrHash, ObjectPtrEqual> impl2cur;
541 for (const auto& pair : impl2desc) {
542 const Buffer& impl = pair.first;
543 const Buffer& desc = pair.second;
544 ICHECK(comparator.rhs_buffer_map_.count(desc));
545 impl2cur[impl] = comparator.rhs_buffer_map_[desc];
546 }
547 std::unordered_map<Buffer, Array<Range>, ObjectPtrHash, ObjectPtrEqual> impl2region;
548 Block impl_block = Downcast<BlockRealize>(intrin_impl->body)->block;
549 for (const BufferRegion& read : impl_block->reads) {
550 impl2region.emplace(read->buffer, read->region);
551 }
552 for (const BufferRegion& write : impl_block->writes) {
553 impl2region.emplace(write->buffer, write->region);
554 }
555 // Step 4: Create MatchBufferRegion for the params of the impl function of the tensor
556 // intrin to make them subregions of the buffer in the original IR.
557 Array<MatchBufferRegion> match_buffer_regions;
558 match_buffer_regions.reserve(intrin_impl->params.size());
559 for (int i = 0, n = intrin_impl->params.size(); i < n; ++i) {
560 const Buffer& impl = intrin_impl->buffer_map.at(intrin_impl->params[i]);
561 const Buffer& cur = impl2cur.at(impl);
562 const Array<Range>& old_region = impl2region.at(impl);
563 const std::vector<PrimExpr>& indices_base = comparator.buffer_indices_.at(cur);
564 int offset = static_cast<int>(indices_base.size()) - static_cast<int>(old_region.size());
565 ICHECK(offset >= 0);
566 Array<Range> new_region;
567 new_region.reserve(cur->shape.size());
568 for (int i = 0; i < offset; i++) {
569 PrimExpr min = indices_base[i];
570 PrimExpr extent = make_const(min.dtype(), 1);
571 new_region.push_back(Range::FromMinExtent(min, extent));
572 }
573 for (int i = 0; i < static_cast<int>(old_region.size()); i++) {
574 PrimExpr min = indices_base[i + offset];
575 PrimExpr extent = old_region[i]->extent;
576 new_region.push_back(Range::FromMinExtent(min, extent));
577 }
578 match_buffer_regions.push_back(MatchBufferRegion(impl, BufferRegion(cur, new_region)));
579 }
580 // Step 5: Replace the subtree in the original IR with the tensor intrin impl.
581 {
582 BlockNode* block = block_realize.CopyOnWrite()->block.CopyOnWrite();
583 block->body = impl_block->body;
584 block->match_buffers = std::move(match_buffer_regions);
585 }
586 if (old_block.defined()) {
587 self->Replace(sref, block_realize->block, {{old_block.value(), block_realize->block}});
588 } else {
589 self->Replace(sref, block_realize, {});
590 }
591 // Step 6: Update the cached flags.
592 StmtSRef result = self->stmt2ref.at(block_realize->block.get());
593 StmtSRef scope_root = tir::GetScopeRoot(self, result, /*require_stage_pipeline=*/false);
594 self->UpdateScopeBlockInfo(scope_root->StmtAs<BlockNode>()->body);
595}
