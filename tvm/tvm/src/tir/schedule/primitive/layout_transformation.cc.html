25class TransformLayoutRewriter : private arith::IRMutatorWithAnalyzer {
26 public:
27 /*!
28 * \brief Rewrite the access to the buffer after the transformation
29 * \param scope_stmt The parent statement that contains all accesses to the target buffer
30 * \param old_buffer The target buffer before transformation
31 * \param new_buffer The new buffer after transformation
32 * \param index_map The transformation applied to the buffer
33 * \return The new AST rooting at the original parent scope and the map from the old block to the
34 * new block
35 */
36 static std::pair<Stmt, Map<Block, Block>> Rewrite(const Stmt& scope_stmt,
37 const Buffer& old_buffer,
38 const Buffer& new_buffer,
39 const IndexMap& index_map) {
40 arith::Analyzer analyzer;
41 TransformLayoutRewriter rewriter(old_buffer, new_buffer, index_map, &analyzer);
42 Stmt result = rewriter(scope_stmt);
43 return {result, rewriter.block_sref_reuse_};
44 }
45
46 private:
47 TransformLayoutRewriter(const Buffer& old_buffer, const Buffer& new_buffer,
48 const IndexMap& index_map, arith::Analyzer* analyzer)
49 : IRMutatorWithAnalyzer(analyzer),
50 old_buffer_(old_buffer),
51 new_buffer_(new_buffer),
52 index_map_(index_map),
53 buffer_data_to_buffer_{{new_buffer->data, new_buffer}} {}
54
55 void RewriteBufferAccess(Buffer* buffer, Array<PrimExpr>* indices) {
56 *buffer = new_buffer_;
57 *indices = index_map_->MapIndices(*indices, analyzer_);
58 }
59
60 using Parent = arith::IRMutatorWithAnalyzer;
61 using Parent::VisitExpr_;
62 using Parent::VisitStmt_;
63
64 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
65 BufferLoad buffer_load = Downcast<BufferLoad>(Parent::VisitExpr_(op));
66 if (buffer_load->buffer.same_as(old_buffer_)) {
67 auto* n = buffer_load.CopyOnWrite();
68 RewriteBufferAccess(&n->buffer, &n->indices);
69 }
70 return std::move(buffer_load);
71 }
72
73 Stmt VisitStmt_(const BufferStoreNode* op) final {
74 BufferStore buffer_store = Downcast<BufferStore>(Parent::VisitStmt_(op));
75 if (buffer_store->buffer.same_as(old_buffer_)) {
76 auto* n = buffer_store.CopyOnWrite();
77 RewriteBufferAccess(&n->buffer, &n->indices);
78 }
79 return std::move(buffer_store);
80 }
81
82 void RewriteAccessRegion(Array<BufferRegion>* old_access_regions,
83 const Array<BufferRegion>& infered_access_regions) {
84 auto fmutate = [this, &infered_access_regions](const BufferRegion& buffer_region) {
85 if (buffer_region->buffer.same_as(old_buffer_)) {
86 ICHECK(infered_access_regions.size() == 1);
87 return infered_access_regions[0];
88 }
89 return buffer_region;
90 };
91 (*old_access_regions).MutateByApply(fmutate);
92 }
93
94 Stmt VisitStmt_(const BlockNode* op) final {
95 Block block = Downcast<Block>(Parent::VisitStmt_(op));
96 auto infered_access_regions = GetBlockReadWriteRegion(block, buffer_data_to_buffer_);
97 auto* n = block.CopyOnWrite();
98 RewriteAccessRegion(&n->reads, infered_access_regions[0]);
99 RewriteAccessRegion(&n->writes, infered_access_regions[1]);
100 block_sref_reuse_.Set(GetRef<Block>(op), block);
101 return std::move(block);
102 }
103
104 const Buffer& old_buffer_;
105 const Buffer& new_buffer_;
106 const IndexMap& index_map_;
107 Map<Var, Buffer> buffer_data_to_buffer_;
108 Map<Block, Block> block_sref_reuse_;
109};

135void TransformLayout(ScheduleState self, const StmtSRef& block_sref, int buffer_index,
136 BufferIndexType buffer_index_type, const IndexMap& index_map) {
137 const BlockNode* block_ptr = TVM_SREF_TO_BLOCK(block_ptr, block_sref);
138 Buffer old_buffer =
139 GetNthAccessBuffer(self, GetRef<Block>(block_ptr), buffer_index, buffer_index_type);
140 Optional<StmtSRef> defining_site_sref;
141 bool is_alloc;
142 std::tie(defining_site_sref, is_alloc) = GetBufferDefiningSite(block_sref, old_buffer);
143 if (defining_site_sref.defined() && !is_alloc) {
144 throw BufferIsSubregionError(self->mod, old_buffer);
145 }
146
147 StmtSRef scope_sref = defining_site_sref.defined()
148 ? defining_site_sref.value()
149 : GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
150 const BlockNode* scope_block = TVM_SREF_TO_BLOCK(scope_block, scope_sref);
151
152 // Step 1: Infer the shape of the new buffer
153 ObjectPtr<BufferNode> new_buffer_node = make_object<BufferNode>(*(old_buffer.get()));
154 new_buffer_node->shape = index_map->MapShape(old_buffer->shape);
155 Buffer new_buffer{new_buffer_node};
156
157 // Step 2: Rewrite access indices and regions of the buffer
158 Stmt new_stmt;
159 Map<Block, Block> block_sref_reuse;
160 std::tie(new_stmt, block_sref_reuse) = TransformLayoutRewriter::Rewrite(
161 GetRef<Block>(scope_block), old_buffer, new_buffer, index_map);
162 Block new_scope_block = Downcast<Block>(new_stmt);
163
164 // Step 3: Rewrite alloc_buffer of the block or buffer_map of the PrimFunc.
165 if (defining_site_sref.defined()) {
166 auto* n = new_scope_block.CopyOnWrite();
167 n->alloc_buffers.MutateByApply([&old_buffer, &new_buffer](const Buffer& buffer) {
168 if (buffer.same_as(old_buffer)) {
169 return new_buffer;
170 }
171 return buffer;
172 });
173 block_sref_reuse.Set(GetRef<Block>(scope_block), new_scope_block);
174 } else {
175 GlobalVar g_var;
176 GetRootPrimFunc(self->mod, scope_block, &g_var);
177 IRModuleNode* new_mod = self->mod.CopyOnWrite();
178 MapNode* new_map = new_mod->functions.CopyOnWrite();
179 PrimFunc ref_new_func = Downcast<PrimFunc>(std::move(new_map->at(g_var)));
180 PrimFuncNode* new_func = ref_new_func.CopyOnWrite();
181 MapNode* new_buffer_map = new_func->buffer_map.CopyOnWrite();
182 for (auto it = new_buffer_map->begin(); it != new_buffer_map->end(); ++it) {
183 if ((*it).second.same_as(old_buffer)) {
184 (*it).second = new_buffer;
185 }
186 }
187 new_map->at(g_var) = std::move(ref_new_func);
188 }
189
190 // Step 4: Replace the scope block with the new block
191 self->Replace(scope_sref, new_scope_block, block_sref_reuse);
192}

194/*!
195 * \brief Detect the block iter type assoicated with the expression
196 *
197 * This function collects block iters in the expression and check if the block iters have the same
198 * iter type. The detected iter type is the iter type of the block iters in the expression
199 * if they have the same iter type, otherwise the detected iter type will be kOpaque.
200 *
201 * \param expr The expression
202 * \param block_iter_type_map The mapping from block iter to iter type
203 * \return The detected block iter type
204 */
205IterVarType DetectNewBlockIterType(
206 const PrimExpr& expr,
207 const std::unordered_map<const VarNode*, IterVarType>& block_iter_type_map) {
208 IterVarType result{kOpaque};
209 bool found = false;
210 PostOrderVisit(expr, [&](const ObjectRef& obj) {
211 if (const VarNode* var = obj.as<VarNode>()) {
212 auto it = block_iter_type_map.find(var);
213 if (it != block_iter_type_map.end()) {
214 if (!found) {
215 found = true;
216 result = it->second;
217 } else if (result != it->second) {
218 result = kOpaque;
219 return false;
220 }
221 }
222 }
223 return true;
224 });
225 return result;
226}

345void TransformBlockLayout(ScheduleState self, const StmtSRef& block_sref,
346 const IndexMap& index_map) {
347 const BlockNode* block_ptr = TVM_SREF_TO_BLOCK(block_ptr, block_sref);
348 const Block& block = GetRef<Block>(block_ptr);
349 arith::Analyzer analyzer;
350
351 // Step 1: Collect outer loops and loop vars
352 Array<StmtSRef> loops = GetLoops(block_sref); // outer loops of the block
353 std::unordered_set<const VarNode*> loop_vars; // loop vars of the outer loops
354 for (const StmtSRef& loop_sref : loops) {
355 CheckLoopStartsWithZero(self, loop_sref, &analyzer);
356 loop_vars.emplace(loop_sref->StmtAs<ForNode>()->loop_var.get());
357 }
358
359 // Step 2: Check the all outer loops have a single child and the block bindings are trivial (all
360 // binding values are loop vars)
361 StmtSRef scope_sref{nullptr}; // the scope statement for replacement
362 if (!loops.empty()) {
363 scope_sref = loops.front();
364 CheckGetSingleChildBlockRealizeOnSRefTree(self, loops.front());
365 } else {
366 scope_sref = block_sref;
367 }
368
369 BlockRealize block_realize = GetBlockRealize(self, block_sref);
370 NotTrivialBindingError::CheckBlockHasTrivialBinding(self->mod, block_realize, loop_vars);
371
372 // Step 3: Collect information of block iter vars
373 Array<PrimExpr> block_vars; // iter_var->var of each block iter
374 Map<Var, Range> block_iter_dom; // domain of block iter
375 std::unordered_map<const VarNode*, IterVarType> block_iter_type; // iter type of block iter
376
377 Array<PrimExpr>
378 block_iter_range_array; // array of block iter extents in the same order as block iters
379 for (const auto& iter_var : block->iter_vars) {
380 block_vars.push_back(iter_var->var);
381 block_iter_dom.Set(iter_var->var, iter_var->dom);
382 block_iter_type[iter_var->var.get()] = iter_var->iter_type;
383 ICHECK(is_zero(iter_var->dom->min));
384 block_iter_range_array.push_back(iter_var->dom->extent);
385 }
386
387 // Step 4: Apply the IndexMap to block iters.
388 IndexMapNotApplicableToBlockIterError::Check(self->mod, block, index_map);
389 Array<PrimExpr> transformed_block_iters = index_map->MapIndices(block_vars);
390 Array<PrimExpr> new_block_iter_range = index_map->MapShape(block_iter_range_array);
391
392 auto iter_map = arith::DetectIterMap(
393 /*indices=*/transformed_block_iters, /*input_iters=*/block_iter_dom, /*predicate=*/Bool(true),
394 /*check_level=*/arith::IterMapLevel::Bijective, &analyzer,
395 /*simplify_trivial_iterators=*/true);
396 if (iter_map->indices.empty()) {
397 throw NotBijectiveAffineIndexMapError(self->mod, index_map);
398 }
399
400 // Step 5: Create the new block after transformation.
401
402 // Step 5.1: Create new block iters. After applying the IndexMap f to block iters ax_0, ..., ax_n,
403 // create block iter each expression in f(ax_0, ..., ax_n).
404 Array<IterVar> new_block_iters; // new block iters
405 Array<PrimExpr> new_block_vars; // iter_var->var of new block iters
406 for (size_t i = 0; i < index_map->final_indices.size(); ++i) {
407 Var new_block_var{"v" + std::to_string(i), DataType::Int(32)};
408 new_block_vars.push_back(new_block_var);
409 IterVarType iter_type = DetectNewBlockIterType(transformed_block_iters[i], block_iter_type);
410 if (iter_type == kOpaque) {
411 throw OpaqueNewIterTypeError(self->mod, GetRef<Block>(block_ptr), transformed_block_iters[i]);
412 }
413 new_block_iters.push_back(IterVar(/*dom=*/Range::FromMinExtent(0, new_block_iter_range[i]),
414 /*var=*/std::move(new_block_var), /*iter_type=*/iter_type));
415 }
416
417 // Step 5.2: Update the block body. Use the inverse map f^{-1} to replace the original block iters
418 // in the body.
419
420 auto inverse_map = arith::InverseAffineIterMap(iter_map->indices, new_block_vars);
421 // Trivial block iters will be simplified in DetectIterMap, they should be mapped to constant
422 // zero.
423 for (const auto& iter_var : block_ptr->iter_vars) {
424 if (inverse_map.find(iter_var->var) == inverse_map.end()) {
425 ICHECK(is_one(iter_var->dom->extent));
426 inverse_map.Set(iter_var->var, 0);
427 }
428 }
429
430 Block new_block = Downcast<Block>(Substitute(GetRef<Block>(block_ptr), inverse_map));
431 new_block.CopyOnWrite()->iter_vars = new_block_iters;
432 new_block = Downcast<Block>(BlockBufferAccessSimplifier::Simplify(new_block, &analyzer));
433
434 // Step 5.3: Create outer loops for each new block iter.
435
436 // Make new loop vars
437 Array<PrimExpr> new_loop_vars;
438 for (int i = 0; i < static_cast<int>(new_block_iters.size()); ++i) {
439 new_loop_vars.push_back(Var("ax" + std::to_string(i), DataType::Int(32)));
440 }
441
442 // Make new block realize
443 BlockRealizeNode* new_block_realize = block_realize.CopyOnWrite();
444 new_block_realize->iter_values = new_loop_vars;
445 new_block_realize->block = new_block;
446
447 // Generate outer loops
448 Stmt body = GetRef<Stmt>(new_block_realize);
449 for (int i = static_cast<int>(new_loop_vars.size()) - 1; i >= 0; --i) {
450 body = For(Downcast<Var>(new_loop_vars[i]), 0, new_block_iter_range[i], ForKind::kSerial,
451 std::move(body));
452 }
453
454 // Step 6: Do the actual replacement
455 self->Replace(scope_sref, body, {{block, new_block}});
456}

458class BufferAxisSeparatorMutator : private ReplaceBufferMutator {
459 public:
460 static Block Mutate(const Block& scope_block, const Buffer& old_buffer, Buffer new_buffer,
461 Map<Block, Block>* block_sref_reuse) {
462 BufferAxisSeparatorMutator mutator(old_buffer, std::move(new_buffer), block_sref_reuse);
463 return Downcast<Block>(mutator.VisitStmt(scope_block));
464 }
465
466 private:
467 BufferAxisSeparatorMutator(const Buffer& old_buffer, Buffer new_buffer,
468 Map<Block, Block>* block_sref_reuse)
469 : ReplaceBufferMutator(old_buffer, new_buffer, block_sref_reuse) {}
470
471 MatchBufferRegion VisitMatchBufferRegion(const MatchBufferRegion& match_buffer) final {
472 auto it = buffer_var_map_.find(match_buffer->source->buffer->data.get());
473 if (it != buffer_var_map_.end()) {
474 const Buffer& new_source_buffer = it->second;
475 Buffer new_target_buffer = match_buffer->buffer;
476 new_target_buffer.CopyOnWrite()->axis_separators = new_source_buffer->axis_separators;
477 if (new_target_buffer->shape.size() != new_source_buffer->shape.size()) {
478 LOG(WARNING)
479 << "Target buffer in match_buffer doesn't have the same dimensionality as its source "
480 "buffer. `axis_separators` for the target buffer might be incorrect.";
481 }
482 buffer_var_map_[new_target_buffer->data.get()] = new_target_buffer;
483 return MatchBufferRegion(new_target_buffer,
484 BufferRegion(new_source_buffer, match_buffer->source->region));
485 }
486 return match_buffer;
487 }
488};

490void SetAxisSeparator(ScheduleState self, const StmtSRef& block_sref, int buffer_index,
491 BufferIndexType buffer_index_type, const Array<IntImm>& axis_separators) {
492 const BlockNode* block_ptr = TVM_SREF_TO_BLOCK(block_ptr, block_sref);
493 Buffer old_buffer =
494 GetNthAccessBuffer(self, GetRef<Block>(block_ptr), buffer_index, buffer_index_type);
495 Optional<StmtSRef> defining_site_sref;
496 bool is_alloc;
497 std::tie(defining_site_sref, is_alloc) = GetBufferDefiningSite(block_sref, old_buffer);
498 if (defining_site_sref.defined() && !is_alloc) {
499 throw BufferIsSubregionError(self->mod, old_buffer);
500 }
501
502 StmtSRef scope_sref = defining_site_sref.defined()
503 ? defining_site_sref.value()
504 : GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
505 const BlockNode* scope_block = TVM_SREF_TO_BLOCK(scope_block, scope_sref);
506
507 // Step 1: Check and update axis_separators of the buffer.
508 Buffer new_buffer = old_buffer;
509 new_buffer.CopyOnWrite()->axis_separators = axis_separators;
510
511 Map<Block, Block> block_sref_reuse;
512
513 // Step 2: Rewrite alloc_buffer of the block or buffer_map of the PrimFunc.
514 Block new_scope_block = BufferAxisSeparatorMutator::Mutate(GetRef<Block>(scope_block), old_buffer,
515 new_buffer, &block_sref_reuse);
516 if (!defining_site_sref.defined()) {
517 // mutate buffer_map of the PrimFunc
518 GlobalVar g_var;
519 GetRootPrimFunc(self->mod, scope_block, &g_var);
520 IRModuleNode* new_mod = self->mod.CopyOnWrite();
521 MapNode* new_map = new_mod->functions.CopyOnWrite();
522 PrimFunc ref_new_func = Downcast<PrimFunc>(std::move(new_map->at(g_var)));
523 PrimFuncNode* new_func = ref_new_func.CopyOnWrite();
524 MapNode* new_buffer_map = new_func->buffer_map.CopyOnWrite();
525 for (auto it = new_buffer_map->begin(); it != new_buffer_map->end(); ++it) {
526 if ((*it).second.same_as(old_buffer)) {
527 (*it).second = new_buffer;
528 }
529 }
530 new_map->at(g_var) = std::move(ref_new_func);
531 }
532
533 // Step 4: Replace the scope block with the new block
534 self->Replace(scope_sref, new_scope_block, block_sref_reuse);
535}
