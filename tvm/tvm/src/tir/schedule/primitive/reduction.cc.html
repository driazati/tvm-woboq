24/*!
25 * \brief A helper class to create a new scope that contains decomposed init body
26 * and replaced old reduction block.
27 */
28class DecomposeReductionBlockReplacer : public StmtMutator {
29 public:
30 /*!
31 * \brief The open interface to users to call the helper class
32 * \param old_scope_root The original block scope before decomposition
33 * \param target_loop The loop we insert the decomposed init body before
34 * \param decompose_body The decomposed init body
35 * \param old_reduction_block The reduction block we want to decompose
36 * \return The new block scope and the updated reduction block
37 */
38 static std::pair<Block, Block> Replace(Block old_scope_root, For target_loop,
39 Stmt decomposed_body, Block old_reduction_block) {
40 DecomposeReductionBlockReplacer replacer(std::move(target_loop), std::move(decomposed_body),
41 std::move(old_reduction_block));
42 return std::make_pair(Downcast<Block>(replacer(std::move(old_scope_root))),
43 replacer.new_reduction_block_);
44 }
45
46 private:
47 explicit DecomposeReductionBlockReplacer(For target_loop, Stmt decomposed_body,
48 Block old_reduction_block)
49 : target_loop_(std::move(target_loop)),
50 decomposed_body_(std::move(decomposed_body)),
51 old_reduction_block_(std::move(old_reduction_block)) {}
52
53 Stmt VisitStmt_(const ForNode* loop) final {
54 Stmt mutated_stmt = StmtMutator::VisitStmt_(loop);
55 if (loop == target_loop_.get()) {
56 return SeqStmt({decomposed_body_, mutated_stmt});
57 } else {
58 return mutated_stmt;
59 }
60 }
61
62 Stmt VisitStmt_(const BlockNode* block) final {
63 if (block == old_reduction_block_.get()) {
64 ObjectPtr<BlockNode> p_new_block = CopyOnWrite(block);
65 p_new_block->name_hint = p_new_block->name_hint + "_update";
66 p_new_block->init = NullOpt;
67 // Add write regions back to read regions in update block.
68 Array<BufferRegion> new_reads;
69 std::unordered_set<const BufferNode*> read_bufs;
70 for (const BufferRegion& read_access : block->reads) {
71 read_bufs.insert(read_access->buffer.get());
72 }
73 for (const BufferRegion& write_access : block->writes) {
74 if (read_bufs.find(write_access->buffer.get()) == read_bufs.end()) {
75 new_reads.push_back(write_access);
76 }
77 }
78 for (const BufferRegion& read_access : block->reads) {
79 new_reads.push_back(read_access);
80 }
81 p_new_block->reads = new_reads;
82 new_reduction_block_ = Block(p_new_block);
83 return new_reduction_block_;
84 } else {
85 return StmtMutator::VisitStmt_(block);
86 }
87 }
88
89 Stmt VisitStmt_(const SeqStmtNode* seq) final {
90 Array<Stmt> new_stmts;
91 new_stmts.reserve(seq->seq.size());
92 for (const Stmt& old_stmt : seq->seq) {
93 new_stmts.push_back(VisitStmt(old_stmt));
94 }
95 return SeqStmt::Flatten(new_stmts);
96 }
97
98 private:
99 For target_loop_;
100 Stmt decomposed_body_;
101 Block old_reduction_block_;
102 Block new_reduction_block_;
103};

105class LoopHeightError : public ScheduleError {
106 public:
107 static void CheckLoopHigherThanReduceLoops(const IRModule& mod, const BlockNode* block,
108 const BlockRealizeNode* realize,
109 const Array<StmtSRef>& loops,
110 const StmtSRef& loop_sref) {
111 for (int i = 0, n = block->iter_vars.size(); i < n; ++i) {
112 // For each block var of type kCommReduce, check its binding
113 const IterVar& iter_var = block->iter_vars[i];
114 const PrimExpr& binding = realize->iter_values[i];
115 if (iter_var->iter_type != IterVarType::kCommReduce) {
116 continue;
117 }
118 for (const StmtSRef& higher_loop : loops) {
119 // Only check loops not lower than the target loop
120 if (higher_loop.same_as(loop_sref)) {
121 break;
122 }
123 // loop_var of a higher loop shouldn't contain loop var
124 const Var& loop_var = higher_loop->StmtAs<ForNode>()->loop_var;
125 if (UsesVar(binding, [v = loop_var.get()](const VarNode* var) { return var == v; })) {
126 const ForNode* loop = TVM_SREF_TO_FOR(loop_sref);
127 throw LoopHeightError(mod, GetRef<For>(loop), GetRef<Block>(block));
128 }
129 }
130 }
131 }
132
133 explicit LoopHeightError(IRModule mod, For loop, Block block)
134 : mod_(std::move(mod)), loop_(std::move(loop)), block_(std::move(block)) {}
135
136 String FastErrorString() const final {
137 return "ScheduleError: decompose_reduction expect the loop to be higher than all the loops "
138 "related to reduce block var";
139 }
140
141 String DetailRenderTemplate() const final {
142 std::ostringstream os;
143 os << "ScheduleError: decompose_reduction expect the loop {0} to be higher than all the loops "
144 "related to reduce block var of block {1}";
145 return os.str();
146 }
147
148 IRModule mod() const final { return mod_; }
149 Array<ObjectRef> LocationsOfInterest() const final { return {loop_, block_}; }
150
151 IRModule mod_;
152 For loop_;
153 Block block_;
154};

156PrimExpr RemakePredicate(PrimExpr pred, const std::unordered_set<const VarNode*>& discarded_loops) {
157 if (is_one(pred)) return Bool(true);
158 PrimExpr new_pred = Bool(true);
159 auto f = [&](const VarNode* var) { return discarded_loops.count(var); };
160 arith::PVar<PrimExpr> lhs, rhs, rest;
161 for (;;) {
162 if ((rest && (lhs < rhs)).Match(pred)) {
163 if (!UsesVar(lhs.Eval(), f)) new_pred = new_pred && (lhs.Eval() < rhs.Eval());
164 pred = rest.Eval();
165 } else if ((lhs < rhs).Match(pred)) {
166 if (!UsesVar(lhs.Eval(), f)) new_pred = new_pred && (lhs.Eval() < rhs.Eval());
167 break;
168 } else {
169 ICHECK(false) << "Unexpected predicate for reduction block";
170 }
171 }
172 return new_pred;
173}

175StmtSRef DecomposeReduction(ScheduleState self, const StmtSRef& block_sref,
176 const StmtSRef& loop_sref) {
177 /*!
178 * Check
179 * - block is a reduction block
180 * - loop is not lower than all the loops related to reduce block var
181 * Mutate
182 * - generate loops related to data par block vars
183 * - generate corresponding init block and update block
184 */
185 // Condition Checks and Information Collection
186 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
187 const ForNode* loop = TVM_SREF_TO_FOR(loop_sref);
188 // Get the outer loops from high to low
189 Array<StmtSRef> loops = GetLoops(block_sref);
190 const BlockRealizeNode* realize = GetBlockRealize(self, block_sref).get();
191 // Cond 0. Check loop_sref is an ancestor of block_sref
192 if (std::find(loops.begin(), loops.end(), loop_sref) == loops.end()) {
193 throw LoopPositionError(self->mod, GetRef<For>(loop), GetRef<Block>(block),
194 "decompose_reduction");
195 }
196 // Cond 1. Check block is reduction
197 StmtSRef scope_root_sref = GetScopeRoot(self, block_sref,
198 /*require_stage_pipeline=*/false);
199 CheckReductionBlock(self, block_sref, scope_root_sref);
200 // Cond 2. Check 'loop' is higher than all the loops related to block var of type reduction
201 LoopHeightError::CheckLoopHigherThanReduceLoops(self->mod, block, realize, loops, loop_sref);
202 // IR Manipulation
203 ObjectPtr<BlockNode> init_block = make_object<BlockNode>();
204 ObjectPtr<BlockRealizeNode> init_realize = make_object<BlockRealizeNode>();
205 init_block->name_hint = block->name_hint + "_init";
206 init_block->annotations = block->annotations;
207 init_realize->iter_values = {};
208 init_realize->block = Block(init_block);
209 // Step 1. Create new block vars and their bindings
210 // Maps an old block var to the new corresponding block var
211 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectPtrEqual> block_var_map;
212 block_var_map.reserve(block->iter_vars.size());
213 for (int i = 0, n = block->iter_vars.size(); i < n; ++i) {
214 const IterVar& iter_var = block->iter_vars[i];
215 const PrimExpr& binding = realize->iter_values[i];
216 // Only process data parallel block vars
217 if (iter_var->iter_type != IterVarType::kDataPar) {
218 continue;
219 }
220 // Create a new block var
221 IterVar new_iter_var(/*dom=*/iter_var->dom,
222 /*var=*/iter_var->var.copy_with_suffix(""),
223 /*iter_type=*/iter_var->iter_type,
224 /*thread_tag=*/iter_var->thread_tag);
225 // Add a block var and its binding
226 init_block->iter_vars.push_back(new_iter_var);
227 init_realize->iter_values.push_back(binding);
228 // Add a mapping from old block vars to new block vars
229 block_var_map[iter_var->var] = new_iter_var->var;
230 }
231 // Step 2. After copying block vars, substitute them in init block
232 init_block->body = Substitute(block->init.value(), block_var_map);
233 for (const BufferRegion& write : block->writes) {
234 init_block->writes.push_back(
235 BufferRegion(write->buffer, Substitute(write->region, block_var_map)));
236 }
237 // Step 3. Scan loops not higher than the specified loop above the reduction block.
238 // If the loop is used in the init block binding, then it is chosen.
239 // Otherwise, it is discarded.
240 std::unordered_set<const VarNode*> discarded_loops;
241 std::vector<int> chosen_loops;
242 for (int i = static_cast<int>(loops.size()) - 1; i >= 0; --i) {
243 const VarNode* loop_var = loops[i]->StmtAs<ForNode>()->loop_var.get();
244 bool discarded = true;
245 for (const PrimExpr& expr : init_realize->iter_values) {
246 if (!UsesVar(expr, [v = loop_var](const VarNode* var) { return var == v; })) {
247 continue;
248 }
249 // The loop is related to init block bindings;
250 chosen_loops.push_back(i);
251 discarded = false;
252 break;
253 }
254 if (discarded) discarded_loops.insert(loop_var);
255 // Only scan loops not higher than the given loop
256 if (loops[i].same_as(loop_sref)) {
257 break;
258 }
259 }
260 // Step 4. After scanning loops, make a new predicate in the init block realize
261 // We discard predicate that is related to discarded loops
262 init_realize->predicate = RemakePredicate(realize->predicate, discarded_loops);
263 // Step 5. Create new loops above init block
264 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectPtrEqual> loop_var_map;
265 Stmt body = BlockRealize(init_realize);
266 for (int i : chosen_loops) {
267 const ForNode* old_loop = TVM_SREF_TO_FOR(loops[i]);
268 // Create a new equivalent to the chosen loop
269 Var old_loop_var = old_loop->loop_var;
270 Var new_loop_var = old_loop_var.copy_with_suffix("_init");
271 loop_var_map[old_loop_var] = new_loop_var;
272 body = For(/*loop_var=*/new_loop_var,
273 /*min=*/old_loop->min,
274 /*extent=*/old_loop->extent,
275 /*kind=*/old_loop->kind,
276 /*body=*/body);
277 }
278 body = Substitute(body, loop_var_map);
279 // Step 6. Mutate IR
280 const BlockNode* old_scope_root = TVM_SREF_TO_BLOCK(scope_root_sref);
281 auto [new_scope_root, new_reduction_block] = DecomposeReductionBlockReplacer::Replace(
282 GetRef<Block>(old_scope_root), GetRef<For>(loop), body, GetRef<Block>(block));
283 self->Replace(scope_root_sref, new_scope_root,
284 {{GetRef<Block>(old_scope_root), new_scope_root},
285 {GetRef<Block>(block), new_reduction_block}});
286 self->UpdateScopeBlockInfo(new_scope_root);
287 return self->stmt2ref.at(init_block.get());
288}

292/*!
293 * \brief A structure used for registering new commutative reducers, and store all the registered
294 * reducers. The reducers are preserved in a list, in the form of "reducer-getter function". When
295 * invoking a reducer-getter function with a specific datatype, the reducer-getter will return the
296 * CommReducer of the corresponding reduction pattern and the specific datatype
297 */
298struct ReducerRegistry {
299 ReducerRegistry()
300 : reducer_getters{CreateReducerGetter([](const Var& x, const Var& y) { return x + y; },
301 [](DataType dtype) { return make_const(dtype, 0); }),
302 CreateReducerGetter([](const Var& x, const Var& y) { return x * y; },
303 [](DataType dtype) { return make_const(dtype, 1); }),
304 CreateReducerGetter([](const Var& x, const Var& y) { return min(x, y); },
305 [](DataType dtype) { return max_value(dtype); }),
306 CreateReducerGetter([](const Var& x, const Var& y) { return max(x, y); },
307 [](DataType dtype) { return min_value(dtype); })} {}
308
309 static void RegisterReducer(TypedPackedFunc<PrimExpr(Var, Var)> combiner_getter,
310 TypedPackedFunc<PrimExpr(DataType)> identity_getter) {
311 ReducerRegistry::Global()->reducer_getters.push_back(ReducerRegistry::CreateReducerGetter(
312 std::move(combiner_getter), std::move(identity_getter)));
313 }
314
315 static TypedPackedFunc<CommReducer(DataType)> CreateReducerGetter(
316 TypedPackedFunc<PrimExpr(Var, Var)> combiner_getter,
317 TypedPackedFunc<PrimExpr(DataType)> identity_getter) {
318 return [combiner_getter = std::move(combiner_getter),
319 identity_getter = std::move(identity_getter)](DataType dtype) -> CommReducer {
320 Var lhs("x", dtype);
321 Var rhs("y", dtype);
322 return CommReducer({lhs}, {rhs}, {combiner_getter(lhs, rhs)}, {identity_getter(dtype)});
323 };
324 }
325
326 static ReducerRegistry* Global() {
327 static ReducerRegistry instance;
328 return &instance;
329 }
330
331 std::vector<TypedPackedFunc<CommReducer(DataType)>> reducer_getters;
332};
333
334std::vector<TypedPackedFunc<CommReducer(DataType)>> GetReducerGetters() {
335 return ReducerRegistry::Global()->reducer_getters;
336}

376 os << "The write buffer " << buffer_->name << " has " << ndim

380 return os.str();
381 }

383 IRModule mod() const final { return mod_; }

385
386 static int CheckAndUpdate(const IRModule& mod, const Buffer& buffer, int factor_axis) {
387 int ndim = static_cast<int>(buffer->shape.size());
388 if (factor_axis < -(ndim + 1) || factor_axis > ndim) {
389 throw FactorAxisOutOfRangeError(mod, buffer, factor_axis);
390 }
391 // If factor_axis is negative, convert it to a non-negative one.
392 if (factor_axis < 0) {
393 factor_axis += ndim + 1;
394 }
395 return factor_axis;
396 }
397
398 IRModule mod_;
399 Buffer buffer_;
400 int factor_axis_;
401};

403class LoopPropertyError : public ScheduleError {
404 public:
405 enum ErrorType {
406 kDataParIterTouchRFactorLoop = 0,
407 kLoopTouchedByBothKindsOfBlockIters = 1,
408 kNotFirstChildBlockOfOutermostLoop = 2,
409 kUnboundLoopUnderReductionLoop = 3
410 };
411
412 explicit LoopPropertyError(IRModule mod, For loop, ErrorType error_type)
413 : mod_(std::move(mod)), loop_(std::move(loop)), error_type_(error_type) {}
414
415 String FastErrorString() const final {
416 switch (error_type_) {
417 case kDataParIterTouchRFactorLoop:
418 return "ScheduleError: The loop to be applied rfactor is required not to be touched by any "
419 "data parallel block iter of the block";
420 case kLoopTouchedByBothKindsOfBlockIters:
421 return "ScheduleError: The loops outside of the reduction block are required not to be "
422 "touched by both data parallel block iters and reduction block iters";
423 case kNotFirstChildBlockOfOutermostLoop:
424 return "ScheduleError: The reduction block should be the first child block of the "
425 "outermost loop outside of it";
426 case kUnboundLoopUnderReductionLoop:
427 return "ScheduleError: A loop who has extent greater than one and is not bound to any "
428 "block iter should not appear under a reduction loop";
429 }
430 ICHECK(false) << "Unreachable";
431 throw;
432 }
433
434 String DetailRenderTemplate() const final {
435 switch (error_type_) {
436 case kDataParIterTouchRFactorLoop:
437 return "The loop to be applied rfactor is {0}, which is required not to be touched by any "
438 "data parallel block iter of the block below. However, some of the block's data "
439 "parallel block iters touch this loop";
440 case kLoopTouchedByBothKindsOfBlockIters:
441 return "It is not allowed that the loop {0} is touched by both some data parallel block "
442 "iters and some reduction block iters";
443 case kNotFirstChildBlockOfOutermostLoop:
444 return "The first child block of the outermost loop {0} is not the reduction block.";
445 case kUnboundLoopUnderReductionLoop:
446 return "The loop {0} has extent greater than one, and is not bound to any block iter. "
447 "Therefore it shouldn't appear under a reduction loop";
448 }
449 ICHECK(false) << "Unreachable";
450 throw;
451 }
452
453 IRModule mod() const final { return mod_; }
454 Array<ObjectRef> LocationsOfInterest() const final { return {loop_}; }
455
456 static void CheckLoopProperty(const ScheduleState& self, const Array<For>& loops,
457 const ForNode* rf_loop, const Block& block,
458 const std::unordered_set<const VarNode*>& data_par_loop_vars,
459 const std::unordered_set<const VarNode*>& reduce_loop_vars) {
460 Array<BlockRealize> children_of_outermost_loop =
461 GetChildBlockRealizeOnSRefTree(self->stmt2ref.at(loops[0].get()));
462 if (!children_of_outermost_loop[0]->block.same_as(block)) {
463 throw LoopPropertyError(self->mod, loops[0], kNotFirstChildBlockOfOutermostLoop);
464 }
465
466 bool meet_reduction_loop = false;
467 for (const For& loop : loops) {
468 bool data_par_touched = data_par_loop_vars.count(loop->loop_var.get());
469 bool reduction_touched = reduce_loop_vars.count(loop->loop_var.get());
470
471 if (data_par_touched && reduction_touched) {
472 throw LoopPropertyError(self->mod, loop, kLoopTouchedByBothKindsOfBlockIters);
473 } else if (data_par_touched) {
474 if (loop.get() == rf_loop) {
475 throw LoopPropertyError(self->mod, loop, kDataParIterTouchRFactorLoop);
476 }
477 continue;
478 } else if (reduction_touched) {
479 if (!meet_reduction_loop) {
480 CheckGetSingleChildBlockRealizeOnSRefTree(self, self->stmt2ref.at(loop.get()));
481 meet_reduction_loop = true;
482 }
483 continue;
484 } else if (meet_reduction_loop && !is_one(loop->extent)) {
485 throw LoopPropertyError(self->mod, loop, kUnboundLoopUnderReductionLoop);
486 }
487 }
488 }
489
490 IRModule mod_;
491 For loop_;
492 ErrorType error_type_;
493};

496 * \brief For each loop in the given array of loop, associate its loop var with the loop itself
497 * using a mapping
498 * \param loops The loops to be analyzed

529/*!
530 * \brief The base class of the rfactor/write-back block creator, which creates the blocks in four
531 * steps:
532 * 1) Create the new block iters and the their iter bindings
533 * 2) Create the reduction update of the new block
534 * 3) Create the read/write regions of the new block
535 * 4) Create the new block and the new block-realize
536 */
537class BaseBlockCreator {
538 public:
539 explicit BaseBlockCreator(BlockRealize old_block_realize, For rf_loop,
540 BufferStore old_reduction_update, CommReducer reducer, Buffer rf_buffer,
541 bool is_rf_block)
542 : old_block_realize_(std::move(old_block_realize)),
543 rf_loop_(std::move(rf_loop)),
544 old_reduction_update_(std::move(old_reduction_update)),
545 reducer_(std::move(reducer)),
546 rf_buffer_(std::move(rf_buffer)),
547 is_rf_block_(is_rf_block) {
548 n_block_iters_ = static_cast<int>(old_block_realize_->iter_values.size());
549 }
550
551 void CreateBlock() {
552 CreateAdditionalIter();
553 for (int i = 0; i < n_block_iters_; ++i) {
554 CreateNormalIters(i);
555 }
556 bool has_reduce_iter = false;
557 for (const IterVar& iter_var : iter_vars_) {
558 if (iter_var->iter_type == IterVarType::kCommReduce) {
559 has_reduce_iter = true;
560 break;
561 }
562 }
563 CreateReductionUpdate(has_reduce_iter);
564 CreateReadWriteRegions();
565
566 String new_block_name = old_block_realize_->block->name_hint;
567 PrimExpr predicate = const_true();
568 if (is_rf_block_) {
569 new_block_name = new_block_name + "_rf";
570 predicate = old_block_realize_->predicate;
571 }
572 Optional<Stmt> init_block =
573 has_reduce_iter ? BufferStore(new_reduction_update_->buffer, reducer_->identity_element[0],
574 new_reduction_update_->indices)
575 : Optional<Stmt>(NullOpt);
576 new_block_ = Block(
577 /*iter_vars=*/iter_vars_,
578 /*reads=*/read_regions_,
579 /*writes=*/write_regions_,
580 /*name_hint=*/new_block_name,
581 /*body=*/new_reduction_update_,
582 /*init=*/init_block,
583 /*alloc_buffers=*/{},
584 /*match_buffers=*/{},
585 /*annotations=*/old_block_realize_->block->annotations);
586 new_block_realize_ = BlockRealize(iter_values_, predicate, new_block_);
587 }
588
589 private:
590 virtual void CreateAdditionalIter() = 0;
591 virtual void CreateNormalIters(int idx) = 0;
592 virtual void CreateReductionUpdate(bool has_reduce_iter) = 0;
593 virtual void CreateReadWriteRegions() = 0;
594
595 public:

636/*!
637 * \brief The derived class of the rfactor block creator, which implements all virtual methods in
638 * the base creator
639 * \details Start constructing the rfactor block. The main difficulty to construct the rfactor block
640 * is to create its block iters. So here we introduce the algorithm to create the block iters.
641 * 1. Create a block iter for the rfactor loop. The block binding of this iter is the loop var, and
642 * the block iter is data parallel.
643 * 2. For all the old block's block iters, there are two cases:
644 * (a) If it is data parallel block iter, or a reduction block iter which doesn't touch the
645 * rfactor loop, we keep it and its block binding in the rfactor block.
646 * (b) Otherwise it is a reduction block iter which touches the rfactor loop. In this case, we
647 * "split" the block iter into one or more new block iters and do not keep the old block
648 * var. More specifically, we create a new reduction block iter for each loop var that
649 * appears in the reduction block iter's binding (except for the rfactor loop), and the
650 * binding of the new block iter is exactly the loop var. (Note that for each loop var, we
651 * create at most one block iter, even if there are multiple old block iters which touch
652 * both this loop and the rfactor loop).
653 * Then we substitute the appearances of the old block iter with the new created block
654 * iters by recording two mappings: one maps loops vars to new created block iters which
655 * is used for binding substitution, and another maps old block iters to new expressions
656 * which is used for substitutions of the old block iters.
657 */
658class RFactorBlockCreator : public BaseBlockCreator {
659 public:
660 explicit RFactorBlockCreator(BlockRealize old_block_realize, For rf_loop,
661 BufferStore old_reduction_update, CommReducer reducer,
662 Buffer rf_buffer,
663 std::unordered_map<const VarNode*, For> loop_vars2loop,
664 int factor_axis, PrimExpr combiner_rhs)
665 : BaseBlockCreator(std::move(old_block_realize), std::move(rf_loop),
666 std::move(old_reduction_update), std::move(reducer), std::move(rf_buffer),
667 true),
668 loop_vars2loop_(std::move(loop_vars2loop)),
669 factor_axis_(factor_axis),
670 combiner_rhs_(std::move(combiner_rhs)) {}
671
672 private:
673 void CreateAdditionalIter() final {
674 // Create a new data parallel block iter for the rfactor loop.
675 additional_iter_ =
676 IterVarFromLoop(rf_loop_, "v" + rf_loop_->loop_var->name_hint, IterVarType::kDataPar);
677 loop_var2block_binding_[rf_loop_->loop_var.get()] = additional_iter_->var;
678 iter_vars_.push_back(additional_iter_);
679 iter_values_.push_back(rf_loop_->loop_var);
680 }
681
682 void CreateNormalIters(int idx) final {
683 IterVar old_iter = old_block_realize_->block->iter_vars[idx];
684 PrimExpr old_binding = old_block_realize_->iter_values[idx];
685 if (old_iter->iter_type == IterVarType::kDataPar ||
686 !UsesVar(old_binding,
687 [v = rf_loop_->loop_var.get()](const VarNode* var) { return var == v; })) {
688 // The old block iter is either a data parallel block iter, or a reduction block iter that
689 // doesn't touch the rfactor loop. In this case reuse the old reduction block iter and its
690 // corresponding binding.
691 iter_vars_.push_back(old_iter);
692 iter_values_.push_back(old_binding);
693 return;
694 }
695 ICHECK(old_iter->iter_type == kCommReduce);
696 // This block iter is a reduction block iter that touches the rfactor loop. So next we try to
697 // create a new block iter for all loop vars that appear in the old binding.
698 Array<Var> vars_in_old_binding = UndefinedVars(old_binding);
699 for (const Var& var : vars_in_old_binding) {
700 auto it = loop_vars2loop_.find(var.get());
701 if (it == loop_vars2loop_.end()) {
702 // `var` is not a loop var. So skip.
703 continue;
704 }
705 const For& loop = it->second;
706 if (loop_var2block_binding_.find(var.get()) == loop_var2block_binding_.end()) {
707 // We haven't created the new block iter for `var`. So here we create it, append it
708 // and its binding to `rf_block_iter_vars` and `rf_block_iter_values` respectively.
709 IterVar new_iter_var =
710 IterVarFromLoop(loop, "v" + loop->loop_var->name_hint, IterVarType::kCommReduce);
711 loop_var2block_binding_[var.get()] = new_iter_var->var;
712 iter_vars_.push_back(new_iter_var);
713 iter_values_.push_back(var);
714 }
715 }
716 // Substitute the original binding with new block iters. Store the result expression
717 // in `rf_var_map` for future substitution.
718 var_map_.Set(old_iter->var, Substitute(old_binding, loop_var2block_binding_));
719 }
720
721 void CreateReductionUpdate(bool has_reduce_iter) final {
722 rf_buf_access_indices_ = old_reduction_update_->indices;
723 rf_buf_access_indices_.insert(rf_buf_access_indices_.begin() + factor_axis_,
724 additional_iter_->var);
725 PrimExpr rhs{nullptr};
726 if (has_reduce_iter) {
727 rhs = (*reducer_.get())({BufferLoad(rf_buffer_, rf_buf_access_indices_)}, {combiner_rhs_})[0];
728 } else {
729 rhs = combiner_rhs_;
730 }
731 new_reduction_update_ = BufferStore(rf_buffer_, rhs, rf_buf_access_indices_);
732 new_reduction_update_ = Downcast<BufferStore>(Substitute(new_reduction_update_, var_map_));
733 }
734
735 void CreateReadWriteRegions() final {
736 const Block& old_block = old_block_realize_->block;
737 read_regions_ = CreateRegions(old_block->reads);
738 write_regions_ = CreateRegions(old_block->writes);
739 }
740
741 Array<BufferRegion> CreateRegions(const Array<BufferRegion>& old_regions) {
742 Array<BufferRegion> new_regions;
743 new_regions.reserve(old_regions.size());
744 for (const BufferRegion& buffer_region : old_regions) {
745 if (buffer_region->buffer.same_as(old_reduction_update_->buffer)) {
746 Array<Range> region = buffer_region->region;
747 region.insert(region.begin() + factor_axis_,
748 Range::FromMinExtent(additional_iter_->var, 1));
749 new_regions.push_back(BufferRegion(rf_buffer_, Substitute(region, var_map_)));
750 } else {
751 new_regions.push_back(
752 BufferRegion(buffer_region->buffer, Substitute(buffer_region->region, var_map_)));
753 }
754 }
755 return new_regions;
756 }
757
758 public:
759 /*! \brief The generated additional block iter in rfactor block for the rfactor loop */
760 IterVar additional_iter_;
761
762 private:
763 /*!
764 * \brief A mapping which maps a loop var to its corresponding For loop for all the reduction
765 * block's outer loops
766 */
767 std::unordered_map<const VarNode*, For> loop_vars2loop_;
768 /*! \brief The factor_axis specified for rfactor */
769 int factor_axis_;
770 /*! \brief The rhs of the combiner in the reduction update of the old block */
771 PrimExpr combiner_rhs_;
772 /*!
773 * \brief A mapping which maps loop vars to new created block iters. This map is used to
774 * substitute the loop vars which appear in the bindings of some old block iters with the new
775 * created block iters
776 */
777 std::unordered_map<const VarNode*, PrimExpr> loop_var2block_binding_;
778};

781 * \brief The derived class of the write-back block creator, which implements all virtual methods in
782 * the base creator
783 */
784class WriteBackBlockCreator : public BaseBlockCreator {
785 public:
786 explicit WriteBackBlockCreator(BlockRealize old_block_realize, For rf_loop,
787 BufferStore old_reduction_update, CommReducer reducer,
788 Buffer rf_buffer, IterVar rf_additional_iter,
789 PrimExpr combiner_lhs, Array<PrimExpr> rf_buf_access_indices)
790 : BaseBlockCreator(std::move(old_block_realize), std::move(rf_loop),

793 rf_additional_iter_(std::move(rf_additional_iter)),
794 combiner_lhs_(std::move(combiner_lhs)) {
795 iter_vars_.reserve(n_block_iters_);
796 iter_values_.reserve(n_block_iters_);
797 rf_buf_access_indices_ = std::move(rf_buf_access_indices);
798 }
799
800 private:
801 void CreateAdditionalIter() final {
802 // Create a new reduction block iter for the rfactor loop.
803 IterVar wb_new_block_iter =
804 IterVarFromLoop(rf_loop_, "v" + rf_loop_->loop_var->name_hint, kCommReduce);
805 iter_vars_.push_back(wb_new_block_iter);
806 iter_values_.push_back(rf_loop_->loop_var);
807 var_map_.Set(rf_additional_iter_->var, wb_new_block_iter->var);
808 }
809
810 void CreateNormalIters(int idx) final {
811 IterVar old_block_iter = old_block_realize_->block->iter_vars[idx];
812 if (old_block_iter->iter_type == IterVarType::kDataPar) {
813 iter_vars_.emplace_back(old_block_iter->dom, old_block_iter->var.copy_with_suffix(""),
814 kDataPar);
815 iter_values_.push_back(old_block_realize_->iter_values[idx]);
816 var_map_.Set(old_block_iter->var, iter_vars_.back());
817 }
818 }
819
820 void CreateReductionUpdate(bool has_reduce_iter) final {
821 wb_lhs_ = Downcast<BufferLoad>(Substitute(combiner_lhs_, var_map_));
822 wb_rhs_ =
823 Downcast<BufferLoad>(Substitute(BufferLoad(rf_buffer_, rf_buf_access_indices_), var_map_));
824 new_reduction_update_ =
825 BufferStore(old_reduction_update_->buffer, (*reducer_.get())({wb_lhs_}, {wb_rhs_})[0],
826 old_reduction_update_->indices);
827 new_reduction_update_ = Downcast<BufferStore>(Substitute(new_reduction_update_, var_map_));
828 }
829
830 void CreateReadWriteRegions() final {
831 read_regions_.push_back(CreateRegion(wb_rhs_));
832 write_regions_.push_back(CreateRegion(wb_lhs_));
833 }

837 region.reserve(load->indices.size());

839 region.push_back(Range::FromMinExtent(index, 1));
840 }
841 return BufferRegion(load->buffer, std::move(region));
842 }

846 IterVar rf_additional_iter_;
847 /*! \brief The lhs of the combiner in the reduction update of the old block */
848 PrimExpr combiner_lhs_;
849 /*! \brief The lhs of the combiner of the write-back block */
850 BufferLoad wb_lhs_;
851 /*! \brief The rhs of the combiner of the write-back block */
852 BufferLoad wb_rhs_;

855/*!
856 * \brief Create new outer loops for the rfactor block, meanwhile update the rfactor block's iter
857 * bindings to use the new created loop vars
858 * \param rf_block_realize The BlockRealize of the rfactor block
859 * \param loops The loops to be wrapped over the rfactor block
860 * \return A Stmt which is the wrapping result
861 */
862Stmt CreateLoopOutsideRfactorBlock(BlockRealize rf_block_realize, const Array<For>& loops) {
863 int n_loops = static_cast<int>(loops.size());
864
865 // Step 1. Create new loop vars.
866 Array<For> new_loops;
867 std::unordered_map<const VarNode*, PrimExpr> new_loop_var_map;
868 new_loops.reserve(n_loops);
869 new_loop_var_map.reserve(n_loops);
870 for (const For& old_loop : loops) {
871 Var new_loop_var = old_loop->loop_var.copy_with_suffix("");
872 new_loop_var_map[old_loop->loop_var.get()] = new_loop_var;
873 }
874
875 // Step 2. Update the iter bindings and predicate of the rfactor block.
876 Array<PrimExpr> new_bindings;
877 new_bindings.reserve(rf_block_realize->iter_values.size());
878 for (const PrimExpr& old_binding : rf_block_realize->iter_values) {
879 new_bindings.push_back(Substitute(old_binding, new_loop_var_map));
880 }
881 {
882 BlockRealizeNode* p_rf_block_realize = rf_block_realize.CopyOnWrite();
883 p_rf_block_realize->iter_values = new_bindings;
884 p_rf_block_realize->predicate = Substitute(rf_block_realize->predicate, new_loop_var_map);
885 }
886
887 // Step 3. Wrap `rf_block_realize` with outer loops.
888 Stmt rf_body = rf_block_realize;
889 for (int i = n_loops - 1; i >= 0; --i) {
890 ObjectPtr<ForNode> p_loop = make_object<ForNode>(*loops[i].get());
891 p_loop->loop_var = Downcast<Var>(new_loop_var_map[loops[i]->loop_var.get()]);
892 p_loop->body = rf_body;
893 rf_body = For(std::move(p_loop));
894 }
895
896 return rf_body;
897}

899class BlockReplacer : public StmtMutator {
900 public:
901 /*!
902 * \brief The replace takes the old scope root block as input, and does four things:
903 * 1) replace the reduction block with the write-back block,
904 * 2) remove loops outside the write-back block that are touched by reduction block iters, except
905 * for the rfactor loop
906 * 3) combine the rfactor block (wrapped with outer loops) and the transformed outermost loop
907 * into a SeqStmt, and
908 * 4) insert the rfactor buffer into the scope root block's `alloc_buffers`
909 * After transformation, the function returns the new scope root block
910 * \param scope_root_block The old scope root block
911 * \param rf_body The rfactor block, which is already wrapped with outer loops
912 * \param outermost_loop The loop that is outermost among all loops outside the reduction block
913 * \param wb_block_realize The new created BlockRealize of the write-back block
914 * \param old_block_realize The BlockRealize of the reduction block
915 * \param rf_loop The rfactor loop, which should be kept outside the write-back block
916 * \param reduce_loop_vars The loops that are touched by reduction block iters, used to remove
917 * loops outside the write-back block
918 * \param loop_vars2loop The mapping from loop vars to loops that are outside the reduction block,
919 * which is used to reduce redundant recursive visits
920 * \param rf_buffer The rfactor buffer to be added into the scope root's `alloc_buffers`
921 * \return The transformed new scope root block
922 */
923 static Block Replace(Block scope_root_block, Stmt rf_body, For outermost_loop,
924 BlockRealize wb_block_realize, BlockRealize old_block_realize, For rf_loop,
925 std::unordered_set<const VarNode*> reduce_loop_vars,
926 std::unordered_map<const VarNode*, For> loop_vars2loop,
927 const Buffer& rf_buffer) {
928 BlockReplacer replacer(std::move(rf_body), std::move(outermost_loop),
929 std::move(wb_block_realize), std::move(old_block_realize),
930 std::move(rf_loop), std::move(reduce_loop_vars),
931 std::move(loop_vars2loop));
932 Block new_scope_root = Downcast<Block>(replacer(std::move(scope_root_block)));
933 BlockNode* p = new_scope_root.CopyOnWrite();
934 p->alloc_buffers.push_back(rf_buffer);
935 return new_scope_root;
936 }
937
938 private:
939 explicit BlockReplacer(Stmt rf_body, For outermost_loop, BlockRealize wb_block_realize,
940 BlockRealize old_block_realize, For rf_loop,
941 std::unordered_set<const VarNode*> reduce_loop_vars,
942 std::unordered_map<const VarNode*, For> loop_vars2loop)
943 : rf_body_(std::move(rf_body)),
944 outermost_loop_(std::move(outermost_loop)),
945 wb_block_realize_(std::move(wb_block_realize)),
946 old_block_realize_(std::move(old_block_realize)),
947 rf_loop_(std::move(rf_loop)),
948 reduce_loop_vars_(std::move(reduce_loop_vars)),
949 loop_vars2loop_(std::move(loop_vars2loop)) {}
950
951 Stmt VisitStmt_(const ForNode* loop) final {
952 // Step 1. Check whether this loop is outside the reduction block. Given that we've made sure
953 // that the scope root block has stage-pipeline property, if this loop is not outside the
954 // reduction block, there's no need to recursively mutate.
955 if (!loop_vars2loop_.count(loop->loop_var.get())) {
956 return GetRef<For>(loop);
957 }
958
959 // Step 2. Recursively mutate.
960 Stmt body = StmtMutator::VisitStmt(loop->body);
961
962 // Step 3. If this loop is the rfactor loop and isn't touched by any reduction block iter, it
963 // should be kept outside the write-back block. Otherwise it shouldn't.
964 if (loop == rf_loop_.get() || !reduce_loop_vars_.count(loop->loop_var.get())) {
965 ObjectPtr<ForNode> p_loop = CopyOnWrite(loop);
966 p_loop->body = body;
967 body = Stmt(p_loop);
968 }
969
970 // Step 4. If this loop is the outermost loop of the reduction block, return the combination of
971 // `rf_body_` and the mutation result `body`. Otherwise return the mutation result.
972 return loop == outermost_loop_.get() ? SeqStmt({rf_body_, body}) : body;
973 }
974
975 Stmt VisitStmt_(const BlockRealizeNode* block_realize) final {
976 // Due to the visitor's behavior on ForNode, this block-realize must be the reduction block's
977 // block-realize. And we directly return the new `wb_block_realize`.
978 ICHECK_EQ(block_realize, old_block_realize_.get());
979 return wb_block_realize_;
980 }
981
982 Stmt VisitStmt_(const SeqStmtNode* seq) final {
983 Array<Stmt> new_stmts;
984 new_stmts.reserve(static_cast<int>(seq->seq.size()));
985
986 for (const Stmt old_stmt : seq->seq) {
987 new_stmts.push_back(VisitStmt(old_stmt));
988 }
989 return SeqStmt::Flatten(new_stmts);
990 }
991
992 private:
993 Stmt rf_body_;
994 For outermost_loop_;
995 BlockRealize wb_block_realize_;
996 BlockRealize old_block_realize_;
997 For rf_loop_;
998 std::unordered_set<const VarNode*> reduce_loop_vars_;
999 std::unordered_map<const VarNode*, For> loop_vars2loop_;
1000};

1002StmtSRef RFactor(ScheduleState self, const StmtSRef& rf_loop_sref, int factor_axis) {
1003 // *****************************************************
1004 // * Condition Checks and Information Collection *
1005 // *****************************************************
1006
1007 // Step 1. Check some basic conditions for rfactor. Get the block and block-realize.
1008 BlockRealize block_realize = CheckGetSingleChildBlockRealizeOnSRefTree(self, rf_loop_sref);
1009 const StmtSRef& block_sref = self->stmt2ref.at(block_realize->block.get());
1010 const Block& block = block_realize->block;
1011 StmtSRef scope_root = GetScopeRoot(self, block_sref, //
1012 /*require_stage_pipeline=*/true);
1013 CheckReductionBlock(self, block_sref, scope_root);
1014 const ForNode* rf_loop = TVM_SREF_TO_FOR(rf_loop_sref);
1015 if (rf_loop->kind != ForKind::kSerial) {
1016 throw NotSerialLoopKindError(self->mod, GetRef<For>(rf_loop));
1017 }
1018
1019 // Step 2. Collect loop vars that are touched by data parallel block iters and reduction block
1020 // iters, respectively.
1021 std::unordered_set<const VarNode*> data_par_loop_vars;
1022 std::unordered_set<const VarNode*> reduce_loop_vars;
1023 GetVarsTouchedByBlockIters(block_realize, &data_par_loop_vars, &reduce_loop_vars);
1024
1025 // Step 3. Collect the loops of the reduction block. Construct a mapping from loops to
1026 // corresponding loop vars.
1027 Array<For> loops = LoopSRefs2Loops(GetLoops(block_sref));
1028 std::unordered_map<const VarNode*, For> loop_vars2loop = GetLoopVar2LoopMap(loops);
1029
1030 // Step 4. Check four properties that the loops should have:
1031 // - the rfactor loop cannot be touched by any data parallel block iter;
1032 // - all the loops cannot be touched by both data parallel block iters and reduction block iters;
1033 // - the outermost loop should have the reduction block as its first child block;
1034 // - the outermost loop that is touched by some reduction block iters can only have one child
1035 // block.
1036 LoopPropertyError::CheckLoopProperty(self, loops, rf_loop, block, data_par_loop_vars,
1037 reduce_loop_vars);
1038
1039 // Step 5. Get the `init` identity and the `update` combiner of the reduction. Extract the
1040 // commutative reducer, combiner lhs and combiner rhs from the reduction identity and the
1041 // reduction combiner. The lhs will be used when constructing the write-back block, and the rhs
1042 // will be used when constructing the rfactor block.
1043 auto [init, update] = GetBufferStoresFromReductionBlock(self, block);
1044 auto [reducer, combiner_lhs, combiner_rhs] =
1045 GetReducerAndCombinerLhsRhs(self, init->value, update);
1046
1047 // Step 6. Check whether `factor_axis` is in a correct range, and convert it to non-negative if it
1048 // is negative.
1049 factor_axis = FactorAxisOutOfRangeError::CheckAndUpdate(self->mod, update->buffer, factor_axis);
1050
1051 // *****************************************************
1052 // * IR Manipulation *
1053 // *****************************************************
1054 // Since rfactor splits the reduction block into two, we call the first one "rfactor block", and
1055 // the latter one "write-back block", and the intermediate buffer is called "rfactor buffer".
1056
1057 // Step 1. Create the intermediate buffer (a.k.a. rfactor buffer), which has an additional
1058 // dimension that specified by `factor_axis` and `rf_loop`.
1059 Buffer rf_buffer = CreateRFactorBuffer(update->buffer, factor_axis, rf_loop);
1060
1061 // Step 2. Create the rfactor block.
1062 RFactorBlockCreator rf_block_creator(block_realize, GetRef<For>(rf_loop), update, reducer,
1063 rf_buffer, loop_vars2loop, factor_axis,
1064 std::move(combiner_rhs));
1065 rf_block_creator.CreateBlock();
1066
1067 // Step 3. Create the write-back block.
1068 WriteBackBlockCreator wb_block_creator(block_realize, GetRef<For>(rf_loop), update, reducer,
1069 rf_buffer, std::move(rf_block_creator.additional_iter_),
1070 std::move(combiner_lhs),
1071 std::move(rf_block_creator.rf_buf_access_indices_));
1072 wb_block_creator.CreateBlock();
1073
1074 // Step 4. Wrap the rfactor block with loops.
1075 Stmt rf_body = CreateLoopOutsideRfactorBlock(rf_block_creator.new_block_realize_, loops);
1076
1077 // *****************************************************
1078 // * Schedule Replacement & Update *
1079 // *****************************************************
1080
1081 // Step 1. Substitute the old scope root block with the new scope root block.
1082 Block old_scope_root_block = GetRef<Block>(scope_root->StmtAs<BlockNode>());
1083 Block new_scope_root_block = BlockReplacer::Replace(
1084 old_scope_root_block, rf_body, loops[0], wb_block_creator.new_block_realize_, block_realize,
1085 GetRef<For>(rf_loop), reduce_loop_vars, loop_vars2loop, rf_buffer);
1086 self->Replace(
1087 scope_root, new_scope_root_block,
1088 {{old_scope_root_block, new_scope_root_block}, {block, wb_block_creator.new_block_}});
1089
1090 // Step 2. Update scope information.
1091 std::vector<StmtSRef> new_block_srefs{self->stmt2ref.at(rf_block_creator.new_block_.get()),
1092 self->stmt2ref.at(wb_block_creator.new_block_.get())};
1093 for (const StmtSRef& new_block_sref : new_block_srefs) {
1094 BlockInfo& info = self->block_info[new_block_sref];
1095 info.affine_binding = true;
1096 info.region_cover = true;
1097 info.scope->stage_pipeline = true;
1098 }
1099 return new_block_srefs[0];
1100}
