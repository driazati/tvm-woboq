59class NotSingleReadWriteBuffer : public ScheduleError {
60 public:
61 explicit NotSingleReadWriteBuffer(IRModule mod, bool is_read, Block block)
62 : mod_(mod), is_read_(is_read), block_(std::move(block)) {}
63
64 String FastErrorString() const final {
65 return is_read_ ? "ScheduleError: The block is allowed to read only a single buffer region"
66 : "ScheduleError: The block is allowed to write only a single buffer region";
67 }
68
69 String DetailRenderTemplate() const final {
70 if (is_read_) {
71 int k = block_->reads.size();
72 return "The block is only allowed to read a single buffer region, but it reads " +
73 std::to_string(k) + " region(s): {0}";
74 } else {
75 int k = block_->writes.size();
76 return "The block is only allowed to write a single buffer region, but it writes " +
77 std::to_string(k) + " region(s): {0}";
78 }
79 }
80
81 IRModule mod() const final { return mod_; }
82 Array<ObjectRef> LocationsOfInterest() const final { return {block_}; }
83
84 IRModule mod_;
85 bool is_read_;
86 Block block_;
87
88 static Buffer GetSingleRead(const ScheduleState& self, const Block& block,
89 const StmtSRef& scope_root_sref) {
90 const std::unordered_map<Buffer, Array<StmtSRef>, ObjectPtrHash, ObjectPtrEqual>&
91 buffer_writers = self->block_info.at(scope_root_sref).scope->buffer_writers;
92 const BufferNode* read_buffer = nullptr;
93 for (const BufferRegion& read_region : block->reads) {
94 const BufferNode* buffer = read_region->buffer.get();
95 if (buffer == read_buffer) {
96 continue;
97 }
98 if (buffer_writers.count(GetRef<Buffer>(buffer)) > 0) {
99 if (read_buffer != nullptr) {
100 throw NotSingleReadWriteBuffer(self->mod, true, block);
101 }
102 read_buffer = buffer;
103 }
104 }
105 if (read_buffer == nullptr) {
106 throw NotSingleReadWriteBuffer(self->mod, true, block);
107 }
108 return GetRef<Buffer>(read_buffer);
109 }
110
111 static Buffer GetSingleWrite(const ScheduleState& self, const Block& block) {
112 if (block->writes.size() != 1) {
113 throw NotSingleReadWriteBuffer(self->mod, false, block);
114 }
115 return block->writes[0]->buffer;
116 }

164 static void Check(const ScheduleState& self, const StmtSRef& consumer_block_sref,
165 const StmtSRef& scope_root_sref) {
166 BlockScope scope = self->GetBlockScope(scope_root_sref);
167 Array<Dependency> producers = scope->GetDepsByDst(consumer_block_sref);
168 if (producers.size() == 1 && producers[0]->kind == DepKind::kRAW) {
169 const StmtSRef& producer_block_sref = producers[0]->src;
170 if (IsCompleteBlock(self, producer_block_sref, scope_root_sref)) {
171 Array<Dependency> consumers = scope->GetDepsBySrc(producer_block_sref);
172 if (consumers.size() == 1) {
173 return;
174 }
175 }
176 }
177 const BlockNode* block = TVM_SREF_TO_BLOCK(consumer_block_sref);
178 throw NonSingleProducerError(self->mod, GetRef<Block>(block));
179 }

207/*!
208 * \brief The base class of the inliner, which handles:
209 * 1) Substitute a subtree with the specific block being inlined
210 * 2) Update the block signature to reflect the changes of read/write/allocated buffers
211 * 3) Maintain a list of index variables and their substitution of the buffer being inlined
212 */
213class BaseInliner : public StmtExprMutator {
214 protected:
215 explicit BaseInliner(const Buffer& inlined_buffer, const Block& inlined_block,
216 const StmtSRef& scope_root_sref)
217 : inlined_buffer_(inlined_buffer),
218 inlined_store_(inlined_block->body.as<BufferStoreNode>()),
219 scope_root_sref_(scope_root_sref) {
220 AddBuffersInBlockSignature(inlined_block.get());
221 }
222
223 PrimExpr VisitExpr_(const VarNode* var) final {
224 CheckOpaqueAccess(var);
225 return StmtExprMutator::VisitExpr_(var);
226 }
227
228 PrimExpr VisitExpr_(const LoadNode* op) final {
229 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
230 return PrimExpr();
231 }
232
233 Stmt VisitStmt_(const StoreNode* op) final {
234 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
235 return Stmt();
236 }
237
238 Stmt VisitStmt_(const ForNode* loop) final {
239 if (src_stmt.get() == loop) {
240 loop = tgt_stmt.as<ForNode>();
241 ICHECK(loop != nullptr);
242 }
243 return StmtExprMutator::VisitStmt_(loop);
244 }
245
246 Stmt VisitStmt_(const BlockNode* block) final {
247 CheckMatchBufferRegion(block);
248 AddBuffersInBlockSignature(block);
249 Block src_block = GetRef<Block>(block);
250 if (src_block.same_as(src_stmt)) {
251 block = tgt_stmt.as<BlockNode>();
252 ICHECK(block != nullptr);
253 }
254 Block tgt_block = Downcast<Block>(StmtExprMutator::VisitStmt_(block));
255 bool is_scope_root = src_block.get() == scope_root_sref_->stmt;
256 tgt_block = UpdateBuffersInBlockSignature(std::move(tgt_block), is_scope_root);
257 block_reuse.Set(src_block, tgt_block);
258 return std::move(tgt_block);
259 }

261 /*!
262 * \brief Count the number of undefined variables that are not used
263 * as buffer objects.
264 *
265 * This is used to determine whether inlining or reverse inlining is
266 * possible. The only undefined variables present should be the
267 * load/store indices, or buffer access based on those indices.
268 *
269 * \param stmt The statement in which to count undefined variables
270 */
271 static int GetNumUndefinedNonpointerVars(const Stmt& stmt) {
272 auto undefined_vars = UndefinedVars(stmt, {});
273 // Buffer pointers and the inlined indices are allowed, but no
274 // other variables may appear in the inlined block.
275 int num_nonpointer_vars = 0;
276 for (const auto& var : undefined_vars) {
277 bool is_pointer = var->dtype.is_handle() && var->type_annotation.defined() &&
278 var->type_annotation.as<PointerTypeNode>();
279 if (!is_pointer) {
280 num_nonpointer_vars++;
281 }
282 }
283 return num_nonpointer_vars;
284 }
285
286 private:
287 /*!
288 * \brief Add the buffers in the block signature to the `buffer_var_map_`,
289 * which is used for auto-completion of a block's read/write region
290 * \param block The block whose signature to be added
291 */
292 void AddBuffersInBlockSignature(const BlockNode* block) {
293 for (const BufferRegion& buffer_region : block->reads) {
294 const Buffer& buffer = buffer_region->buffer;
295 buffer_var_map_.Set(buffer->data, buffer);
296 }
297 for (const BufferRegion& buffer_region : block->writes) {
298 const Buffer& buffer = buffer_region->buffer;
299 buffer_var_map_.Set(buffer->data, buffer);
300 }
301 for (const Buffer& buffer : block->alloc_buffers) {
302 buffer_var_map_.Set(buffer->data, buffer);
303 }
304 }
305
306 /*!
307 * \brief Update the following block signature:
308 * 1) T.alloc_buffer, if the block is scope root
309 * 2) T.reads, if the block is not scope root
310 * 3) T.writes, if the block is not scope root
311 * \param block The block to be updated
312 * \param is_scope_root A flag indicating if a block is the scope root of the block to be inlined
313 * \return The updated block
314 */
315 Block UpdateBuffersInBlockSignature(Block block, bool is_scope_root) {
316 // Step 1. Update `BlockNode::alloc_buffers`
317 Array<Buffer> alloc_buffers;
318 if (is_scope_root) {
319 alloc_buffers.reserve(block->alloc_buffers.size());
320 for (const Buffer& alloc_buffer : block->alloc_buffers) {
321 if (!alloc_buffer.same_as(inlined_buffer_)) {
322 alloc_buffers.push_back(alloc_buffer);
323 }
324 }
325 } else {
326 alloc_buffers = std::move(block->alloc_buffers);
327 }
328 // Step 2. Update `BlockNode::reads` and `BlockNode::writes`
329 Array<BufferRegion> reads = std::move(block->reads);
330 Array<BufferRegion> writes = std::move(block->writes);
331 auto f_access_inline_buffer = [this](const BufferRegion& access) {
332 return access->buffer.same_as(this->inlined_buffer_);
333 };
334 if (!is_scope_root && (std::any_of(reads.begin(), reads.end(), f_access_inline_buffer) ||
335 std::any_of(writes.begin(), writes.end(), f_access_inline_buffer))) {
336 Array<Array<BufferRegion>> inspected = GetBlockReadWriteRegion(block, buffer_var_map_);
337 reads = std::move(inspected[0]);
338 writes = std::move(inspected[1]);
339 }
340 // Step 3. Assemble the result
341 BlockNode* n = block.CopyOnWrite();
342 n->reads = std::move(reads);
343 n->writes = std::move(writes);
344 n->alloc_buffers = std::move(alloc_buffers);
345 return block;
346 }

348 /*!
349 * \brief Opaque access to the buffer to be inlined is disallowed.
350 * This method checks if a buffer var belongs to the buffer
351 * \param buffer_var The buffer var to be checked
352 */
353 void CheckOpaqueAccess(const VarNode* buffer_var) {
354 if (inlined_buffer_->data.get() == buffer_var) {
355 this->has_opaque_access = true;
356 }
357 }

359 /*!
360 * \brief The buffer to be inlined is not allowed to be region matched.
361 * This method checks if a block has the disallowed behavior of buffer region match.
362 * \param block The block to be checked
363 */
364 void CheckMatchBufferRegion(const BlockNode* block) {
365 for (const MatchBufferRegion& match_buffer_region : block->match_buffers) {
366 const Buffer& matched = match_buffer_region->source->buffer;
367 if (matched.same_as(inlined_buffer_)) {
368 this->has_opaque_access = true;
369 }
370 }
371 }

373 protected:
374 /*! \brief The buffer to be inlined */
375 Buffer inlined_buffer_{nullptr};
376 /*! \brief The body of the block to be inlined */
377 const BufferStoreNode* inlined_store_{nullptr};
378 /*! \brief The scope root */
379 StmtSRef scope_root_sref_{nullptr};
380 /*! \brief Maps a buffer's data field to itself */
381 Map<Var, Buffer> buffer_var_map_;

383 std::vector<const VarNode*> idx_vars_;

385 std::unordered_map<const VarNode*, PrimExpr> idx_sub_;

387 public:
388 /*!
389 * \brief The Stmt to be replaced when removing the leaf block
390 * \note The pair (src_stmt, tgt_stmt) are produced by LeafBlockRemovalPlan to indicate a
391 * transformation on top of the input AST. We take this approach to avoid changing the AST twice
392 */
393 Stmt src_stmt{nullptr};
394 /*! \brief The Stmt to be replaced to when removing the leaf block */
395 Stmt tgt_stmt{nullptr};
396 /*! \brief The reuse mapping of block srefs */
397 Map<Block, Block> block_reuse;
398 /*! \brief Indicates if there is any opaque access of the inlined buffer */
399 bool has_opaque_access{false};

402/*!
403 * \brief Helper to inline the producer block into its consumer(s)
404 * The derived class implements the following functionalities:
405 * 1) Substitute `BufferLoad` on the buffer to be inlined
406 * to its value calculation in the producer block
407 * 2) Analyze the producer block to determine the remapping of index variables
408 */
409class ComputeInliner : public BaseInliner {
410 public:
411 explicit ComputeInliner(const Buffer& inlined_buffer, const Block& producer_block,
412 const StmtSRef& scope_root_sref)
413 : BaseInliner(inlined_buffer, producer_block, scope_root_sref) {}
414
415 bool BodyPatternAllowInline(const Block& producer_block) {
416 if (inlined_store_ == nullptr) {
417 return false;
418 }
419
420 int n_vars = GetNumUndefinedNonpointerVars(GetRef<Stmt>(inlined_store_));
421 if (!UpdateAndCheckIndexVars(inlined_store_->indices, n_vars)) {
422 return false;
423 }
424 return true;
425 }
426
427 private:
428 using BaseInliner::VisitExpr_;
429 using BaseInliner::VisitStmt_;
430
431 PrimExpr VisitExpr_(const BufferLoadNode* _load) final {
432 BufferLoad load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(_load));
433 if (!load->buffer.same_as(inlined_buffer_)) {
434 return std::move(load);
435 }
436 return ReplaceInlinedBuffer(std::move(load));
437 }
438
439 PrimExpr ReplaceInlinedBuffer(BufferLoad load) {
440 SetIndexSubstitution(load->indices);
441 return Substitute(inlined_store_->value, idx_sub_);
442 }
443
444 /*!
445 * \brief Check if the indices are atomic distinct variables and the access is n-dimensional.
446 * If so, set `self->idx_vars_` properly.
447 * \param indices The indices to be extracted
448 * \param expected_ndim The expected ndim of the access
449 * \return A boolean flag indicating if the check is successful
450 */
451 bool UpdateAndCheckIndexVars(const Array<PrimExpr>& indices, int expected_ndim) {
452 int n = indices.size();
453 if (n != expected_ndim) {
454 // Failure: dimension mismatch
455 return false;
456 }
457 std::vector<const VarNode*> result;
458 result.reserve(n);
459 for (const PrimExpr& i : indices) {
460 if (const auto* var = i.as<VarNode>()) {
461 result.push_back(var);
462 } else {
463 // Failure: indexing expression is not a variable
464 return false;
465 }
466 }
467 using DistinctSet = std::unordered_set<const VarNode*>;
468 int n_distinct = DistinctSet(result.begin(), result.end()).size();
469 if (n != n_distinct) {
470 // Failure: indexing variables are not distinct
471 return false;
472 }
473 if (idx_vars_.empty()) {
474 idx_vars_ = std::move(result);
475 } else if (!support::ArrayWithSameContent(idx_vars_, result)) {
476 // Failure: indexing variables are not consitent in different BufferLoads
477 return false;
478 }
479 return true;
480 }
481
482 /*!
483 * \brief Set the mapping of index substitution `self->idx_sub_`
484 * \param indices The expressions that the corresponding index variables are replaced to
485 */
486 void SetIndexSubstitution(const Array<PrimExpr>& indices) {
487 ICHECK_EQ(indices.size(), idx_vars_.size());
488 int n = idx_vars_.size();
489 idx_sub_.reserve(n);
490 for (int i = 0; i < n; ++i) {
491 idx_sub_[idx_vars_[i]] = indices[i];
492 }
493 }
494};

496/*!
497 * \brief Helper to inline the consumer block into its producer
498 * The derived class implements the following functionalities:
499 * 1) Analyze the consumer block to determine the remapping of index variables
500 * 2) Substitute `BufferStore` of the buffer to be inlined,
501 * replacing it with direct writing to the buffer that consumer writes
502 */
503class ReverseComputeInliner : public BaseInliner {
504 class Substituter : public StmtExprMutator {
505 public:
506 explicit Substituter(ReverseComputeInliner* self) : self_(self) {}
507
508 private:
509 PrimExpr VisitExpr_(const VarNode* var) final {
510 auto it = self_->idx_sub_.find(var);
511 ICHECK(it != self_->idx_sub_.end());
512 return (*it).second;
513 }
514
515 PrimExpr VisitExpr_(const BufferLoadNode* _load) final {
516 BufferLoad load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(_load));
517 return load->buffer.same_as(self_->inlined_buffer_) ? self_->producer_rhs_ : load;
518 }
519
520 ReverseComputeInliner* self_;
521 };
522
523 public:
524 explicit ReverseComputeInliner(const Buffer& inlined_buffer, const Block& consumer_block,
525 const StmtSRef& scope_root_sref)
526 : BaseInliner(inlined_buffer, consumer_block, scope_root_sref) {}
527
528 bool BodyPatternAllowInline(const Block& consumer_block) {
529 if (inlined_store_ == nullptr) {
530 // Failure: block body is not BufferStore
531 return false;
532 }
533 std::vector<const BufferLoadNode*> loads = ExtractBufferLoad(inlined_buffer_, inlined_store_);
534 if (loads.size() == 0) {
535 // Failure: no BufferLoad from the `inlined_buffer_`
536 return false;
537 }
538
539 // Collect block iter domains and update the substition map
540 Map<Var, Range> consumer_iter_doms;
541 for (const auto& iter_var : consumer_block->iter_vars) {
542 consumer_iter_doms.Set(iter_var->var, iter_var->dom);
543 // Set default mapping for unit iters
544 if (is_const_int(iter_var->dom->extent, 1) && is_const_int(iter_var->dom->min)) {
545 idx_sub_[iter_var->var.get()] = iter_var->dom->min;
546 }
547 }
548
549 for (const BufferLoadNode* load : loads) {
550 if (!UpdateAndCheckIndexExprs(load->indices)) {
551 return false;
552 }
553 }
554
555 auto res = arith::DetectIterMap(
556 /*indices=*/buffer_load_indices_,
557 /*input_iters=*/consumer_iter_doms,
558 /*predicate=*/true,
559 /*check_level=*/arith::IterMapLevel::Bijective,
560 /*analyzer=*/&analyzer,
561 /*simplify_trivial_iterators=*/false);
562 buffer_load_iter_map_ = res->indices;
563 if (buffer_load_iter_map_.empty()) {
564 // Failure: indices of BufferLoad are not bijective affine
565 return false;
566 }
567 return true;
568 }
569
570 private:
571 using BaseInliner::VisitExpr_;
572 using BaseInliner::VisitStmt_;
573
574 Stmt VisitStmt_(const BufferStoreNode* _store) final {
575 BufferStore store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(_store));
576 if (!store->buffer.same_as(inlined_buffer_)) {
577 return std::move(store);
578 }
579 return ReplaceInlinedBuffer(std::move(store));
580 }
581
582 /*!
583 * \brief Apply the inverse of `buffer_load_iter_map_` to producer indices. Update `idx_sub_` with
584 * the result. It will be later used to transform the BufferStore indices of the producer.
585 * \param producer_indices The BufferStore indices of the producer.
586 */
587 void CreateInverseMapping(const Array<PrimExpr> producer_indices) {
588 auto inverse_iter_map = arith::InverseAffineIterMap(buffer_load_iter_map_, producer_indices);
589 for (const auto& pair : inverse_iter_map) {
590 idx_sub_[pair.first.get()] = pair.second;
591 }
592 }
593
594 Stmt ReplaceInlinedBuffer(BufferStore producer) {
595 CreateInverseMapping(producer->indices);
596 producer_rhs_ = producer->value;
597 return Substituter(this)(GetRef<BufferStore>(inlined_store_));
598 }

600 /*!
601 * \brief Extracts expressions that loads a specific buffer
602 * \param buffer The buffer to be loaded from
603 * \param from The BufferStore statement to be extracted from
604 * \return A list of `BufferLoad` expressions
605 */
606 static std::vector<const BufferLoadNode*> ExtractBufferLoad(const Buffer& buffer,
607 const BufferStoreNode* from) {
608 struct Extractor : public ExprVisitor {
609 void VisitExpr_(const BufferLoadNode* load) final {
610 if (load->buffer.get() == buffer) {
611 result.push_back(load);
612 }
613 ExprVisitor::VisitExpr_(load);
614 }
615 const BufferNode* buffer;
616 std::vector<const BufferLoadNode*> result;
617 } extractor;
618 extractor.buffer = buffer.get();
619 for (const PrimExpr& expr : from->indices) {
620 extractor(expr);
621 }
622 extractor(from->value);
623 return std::move(extractor.result);
624 }
625
626 /*!
627 * \brief Update `buffer_load_indices_` with the given indices. If `buffer_load_indices_` is
628 * already non-empty, check it is consistent with the given indices.
629 * \param indices The indices
630 * \param expected_ndim The expected ndim of the access
631 * \return A boolean flag indicating if the check is successful
632 */
633 bool UpdateAndCheckIndexExprs(const Array<PrimExpr>& indices) {
634 if (buffer_load_indices_.empty()) {
635 buffer_load_indices_ = indices;
636 } else if (!std::equal(buffer_load_indices_.begin(), buffer_load_indices_.end(),
637 indices.begin(), indices.end(), ExprDeepEqual())) {
638 // Failure: indices are not consistent in different BufferLoads
639 return false;
640 }
641 return true;
642 }
643
644 /*! \brief The RHS value of the producer's BufferStore statement */
645 PrimExpr producer_rhs_{nullptr};
646 /*! \brief The indices of the consumer's BufferLoad */
647 Array<PrimExpr> buffer_load_indices_;
648 /*! \brief The IterMap representing the indices of the consumer's BufferLoad */
649 Array<arith::IterSumExpr> buffer_load_iter_map_{nullptr};
650 /*! \brief The arithmetic analyzer */
651 arith::Analyzer analyzer;
652};

654void ComputeInlineImpl(ScheduleState self, const StmtSRef& producer_block_sref,
655 bool check_only = false) {
656 const BlockNode* _producer_block = TVM_SREF_TO_BLOCK(producer_block_sref);
657 Block producer_block = GetRef<Block>(_producer_block);
658 HasInitBlock::Check(self->mod, producer_block);
659 Buffer inlined_buffer = NotSingleReadWriteBuffer::GetSingleWrite(self, producer_block);
660 // Step 1. Get the scope block
661 StmtSRef scope_root_sref = GetScopeRoot(self, producer_block_sref,
662 /*require_stage_pipeline=*/true);
663 // Step 2. Check completeness
664 CheckNotOutputBlock(self, producer_block_sref, scope_root_sref);
665 CheckCompleteBlock(self, producer_block_sref, scope_root_sref);
666 // Step 3. Analyze the block body
667 ComputeInliner inliner(inlined_buffer, producer_block, scope_root_sref);
668 if (!inliner.BodyPatternAllowInline(producer_block)) {
669 throw BodyAnalysisError(false, self->mod, producer_block);
670 }
671 // Step 4. Create a plan that removes the leaf block to be inlined
672 LeafBlockRemovalPlan(self, producer_block_sref, &inliner.src_stmt, &inliner.tgt_stmt);
673 // Step 5. Create an AST where the leaf `producer_block_sref` points to is removed,
674 // and update other blocks who read from the removed block
675 Stmt tgt_stmt = inliner(GetRef<Stmt>(scope_root_sref->stmt));
676 if (inliner.has_opaque_access) {
677 throw OpaqueAccessError(self->mod, scope_root_sref);
678 }
679 // Step 6. Do the real mutation on the AST and the sref tree in the schedule state
680 if (check_only) {
681 return;
682 }
683 self->Replace(scope_root_sref, tgt_stmt, inliner.block_reuse);
684}

686void ComputeInline(ScheduleState self, const StmtSRef& producer_block_sref) {
687 ComputeInlineImpl(self, producer_block_sref);
688}

693 } catch (const tvm::runtime::Error& e) {
694 return false;

699void ReverseComputeInlineImpl(ScheduleState self, const StmtSRef& consumer_block_sref,
700 bool check_only = false) {
701 const BlockNode* _consumer_block = TVM_SREF_TO_BLOCK(consumer_block_sref);
702 Block consumer_block = GetRef<Block>(_consumer_block);
703 HasInitBlock::Check(self->mod, consumer_block);
704 // Step 1. Get the scope block
705 StmtSRef scope_root_sref = GetScopeRoot(self, consumer_block_sref, //
706 /*require_stage_pipeline=*/true);
707 Buffer inlined_buffer =
708 NotSingleReadWriteBuffer::GetSingleRead(self, consumer_block, scope_root_sref);
709 // Step 2. Check completeness
710 CheckCompleteBlock(self, consumer_block_sref, scope_root_sref);
711 // Step 3. Check if the consumer has a single complete producer
712 NonSingleProducerError::Check(self, consumer_block_sref, scope_root_sref);
713 // Step 4. Analyze the block body
714 ReverseComputeInliner inliner(inlined_buffer, consumer_block, scope_root_sref);
715 if (!inliner.BodyPatternAllowInline(consumer_block)) {
716 throw BodyAnalysisError(true, self->mod, consumer_block);
717 }
718 // Step 5. Create a plan that removes the leaf block to be inlined
719 LeafBlockRemovalPlan(self, consumer_block_sref, &inliner.src_stmt, &inliner.tgt_stmt);
720 // Step 6. Create an AST where the leaf `consumer_block_sref` points to is removed,
721 // and update other blocks who read from the removed block
722 Stmt tgt_stmt = inliner(GetRef<Stmt>(scope_root_sref->stmt));
723 if (inliner.has_opaque_access) {
724 throw OpaqueAccessError(self->mod, scope_root_sref);
725 }
726 // Step 7. Do the real mutation on the AST and the sref tree in the schedule state
727 if (check_only) {
728 return;
729 }
730 self->Replace(scope_root_sref, tgt_stmt, inliner.block_reuse);
731}
