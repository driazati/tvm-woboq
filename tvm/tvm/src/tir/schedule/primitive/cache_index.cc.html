28/*! \brief The auxiliary info used for the insertion point and content of the cache stage. */
29struct IndexInfo {
30 /*! \brief The target buffer to cache the index. */
31 Buffer target_buffer;
32 /*! \brief The cache buffer to store the precomputed index */
33 std::vector<Buffer> cache_buffer;
34 /*! \brief The expr to be precomputed */
35 std::vector<PrimExpr> index_exprs;
36 /*! \brief The range of the loop vars relating to index computation */
37 Map<Var, Range> range_map;
38 /*! \brief The binding table of the block var and the loop var */
39 Map<Var, PrimExpr> var_binding;
40 /*! \brief The block var of the target block */
41 std::vector<Array<Var>> origin_block_vars;
42 /*! \brief The index to insert the cache stage. */
43 size_t loc_pos;
44 /*! \brief The cache stage to be inserted. */
45 Stmt cache_stage;
46 /*! \brief The map used for ScheduleStateNode::Replace. */
47 Map<Block, Block> block_reuse;
48};
49
50/*!
51 * \brief Determine the data type base on the integer range.
52 * \param range The range of the integer.
53 * \returns A data type that covers the input range.
54 */
55DataType DetermineDatatype(const arith::IntSet& range) {
56 arith::Analyzer ana;
57 if (ana.CanProve(range.min() >= INT32_MIN && range.max() <= INT32_MAX)) {
58 return DataType::Int(32);
59 } else {
60 ICHECK(ana.CanProve(range.min() >= make_const(DataType::Int(64), INT64_MIN) &&
61 range.max() <= make_const(DataType::Int(64), INT64_MAX)));
62 return DataType::Int(64);
63 }
64}

66/*! \brief Collect the index info to be cached */
67class IndexInfoCollector : public StmtExprVisitor {
68 public:
69 /*!
70 * \brief Collect the index info for cache_index and write into the IndexInfo
71 * \param self The state of the schedule \param block_sref The sref of the target
72 * block of the target buffer being applied cache_index \param scope_sref The sref
73 * of the scope block of the target block \param info The index info.
74 */
75 static void Collect(const ScheduleState& self, const StmtSRef& block_sref,
76 const StmtSRef& scope_sref, IndexInfo* info) {
77 IndexInfoCollector collector(self, block_sref, scope_sref, info->target_buffer);
78 collector(GetRef<Stmt>(scope_sref->stmt));
79 // info->loc_sref = collector.loc_sref_;
80 info->loc_pos = collector.loc_pos_;
81 info->index_exprs = collector.exprs_;
82 info->range_map = collector.range_map_;
83 }
84
85 private:
86 /*!
87 * \brief Constructor
88 * \param self The state of the schedule
89 * \param block_sref The sref of the target block of the buffer being applied cache_index
90 * \param scope_sref The sref of the scope block of the target block
91 * \param buffer The target buffer to cache the indexs
92 */
93 IndexInfoCollector(const ScheduleState self, const StmtSRef& block_sref,
94 const StmtSRef& scope_sref, const Buffer& buffer)
95 : self_(self), block_sref_(block_sref), scope_sref_(scope_sref), buffer_(buffer) {}
96
97 void VisitStmt_(const SeqStmtNode* seq_stmt) final {
98 for (size_t i = 0; i < seq_stmt->size(); ++i) {
99 if (loc_pos_ != -1) {
100 break;
101 }
102 VisitStmt(seq_stmt->seq[i]);
103 // `pos` can be assigned only once when we visited `block_sref`
104 if (visited_block_ && loc_pos_ == -1 && update_seq_pos_) {
105 // The offset of insert position from the block
106 loc_pos_ = i;
107 return;
108 }
109 }
110 }
111
112 void VisitStmt_(const BlockNode* block) final {
113 // Only visit the target's parent block
114 StmtVisitor::VisitStmt_(block);
115 if (block == scope_sref_->stmt) {
116 // The block vistied is the current parent scope
117 // Handling cases when no SeqStmt in the scope
118 if (visited_block_ && loc_pos_ == -1) {
119 loc_pos_ = 0;
120 }
121 } else if (block_sref_->stmt == block) {
122 visited_block_ = true;
123 }
124 // Update seq pos only at top scope
125 if (visited_block_ && self_->stmt2ref.at(block)->parent == scope_sref_.get()) {
126 update_seq_pos_ = true;
127 }
128 }
129
130 void VisitStmt_(const ForNode* loop) final {
131 range_map_.Set(loop->loop_var, Range::FromMinExtent(loop->min, loop->extent));
132 StmtVisitor::VisitStmt_(loop);
133 // Update seq pos only at top scope
134 if (visited_block_ && self_->stmt2ref.at(loop)->parent == scope_sref_.get()) {
135 update_seq_pos_ = true;
136 }
137 }
138
139 void VisitExpr_(const BufferLoadNode* load) final {
140 if (load->buffer.same_as(buffer_)) {
141 for (const PrimExpr& it : load->indices) {
142 if (!it->IsInstance<VarNode>()) {
143 exprs_.push_back(it);
144 }
145 }
146 }
147 ExprVisitor::VisitExpr_(load);
148 }
149
150 /*! \brief The schedule class */
151 const ScheduleState self_;
152 /*! \brief The target block that read the target buffer */
153 const StmtSRef& block_sref_;
154 /*! \brief The parent scope of the target block */
155 const StmtSRef& scope_sref_;
156 /*! \brief The target buffer to cache the index */
157 const Buffer& buffer_;
158 /*! \brief The calculation expr to be precomputed */
159 std::vector<PrimExpr> exprs_;
160 /*! \brief The flag whether we have visited the target block */
161 bool visited_block_{false};
162 /*! \brief The index to insert the cache_index stage */
163 int loc_pos_{-1};
164 /*! \brief The flag indicating the right scope to update seq pos */
165 bool update_seq_pos_{false};
166 /*! \brief Record the ranges of iter vars */
167 Map<Var, Range> range_map_;
168};

170/*!
171 * \brief Create a loop nest that writes precomputed index into index buffer.
172 * \param cache_region The cached copy region.
173 * \param info The cache stage information, which will be updated in the function.
174 * \param storage_scope The storage scope of the cached buffer (only used in naming here)
175 * \returns A block indicating the body of the loop nesting.
176 */
177Array<Block> MakeIndexCacheStage(IndexInfo* info) {
178 Array<Block> blocks;
179 Array<Stmt> bodies;
180 bodies.reserve(info->index_exprs.size());
181 info->cache_buffer.reserve(info->index_exprs.size());
182 const String& storage_scope = info->target_buffer.scope();
183
184 // For each index calculation, create a block to pre-compute.
185 for (size_t expr_index = 0; expr_index < info->index_exprs.size(); expr_index++) {
186 const PrimExpr& index_expr = info->index_exprs[expr_index];
187
188 // Collect the block vars in original index computation
189 info->origin_block_vars.push_back({});
190 PostOrderVisit(index_expr, [&info, &expr_index](const ObjectRef& node) {
191 if (node->IsInstance<VarNode>()) {
192 Var iter_var = Downcast<Var>(node);
193 const Array<Var>& origin_block_var = info->origin_block_vars[expr_index];
194 auto find_result = std::find_if(origin_block_var.begin(), origin_block_var.end(),
195 [&](Var it) { return it.get() == iter_var.get(); });
196 if (find_result == origin_block_var.end()) {
197 info->origin_block_vars[expr_index].push_back(iter_var);
198 }
199 }
200 });
201
202 // Collect the loop vars corresponding to collected block vars,
203 // which will be used to create new loop vars
204 std::vector<Var> iter_vars;
205 for (const Var& it : info->origin_block_vars[expr_index]) {
206 PostOrderVisit(info->var_binding.at(it), [/*&info,*/ &iter_vars](const ObjectRef& node) {
207 if (node->IsInstance<VarNode>()) {
208 Var iter_var = Downcast<Var>(node);
209 if (std::find_if(iter_vars.begin(), iter_vars.end(),
210 [&](Var it) { return it.get() == iter_var.get(); }) == iter_vars.end()) {
211 iter_vars.push_back(iter_var);
212 }
213 }
214 });
215 }
216
217 // Inference the shape and create cache buffer
218 arith::IntSet val_range =
219 arith::EvalSet(Substitute(index_expr, info->var_binding), arith::AsIntSet(info->range_map));
220 DataType data_type = DetermineDatatype(val_range);
221 Var index_buffer_var("index_var_" + std::to_string(expr_index),
222 PointerType(PrimType(data_type), storage_scope));
223 Array<PrimExpr> buffer_shape;
224 for (const Var& it : info->origin_block_vars[expr_index]) {
225 buffer_shape.push_back(
226 arith::EvalSet(info->var_binding.at(it), arith::AsIntSet(info->range_map)).max() + 1);
227 }
228 info->cache_buffer.push_back(Buffer(index_buffer_var, data_type, buffer_shape, {1}, {0},
229 index_buffer_var->name_hint, 0, 0, kDefault));
230
231 // Create loop vars and block vars' binding_value
232 std::vector<Var> loop_vars;
233 Map<Var, PrimExpr> replace_table;
234 for (const Var& it : iter_vars) {
235 DataType data_type = DetermineDatatype(arith::IntSet::FromRange(info->range_map.at(it)));
236 Var loop_var("ax" + std::to_string(replace_table.size()), data_type);
237 loop_vars.push_back(loop_var);
238 replace_table.Set(it, loop_var);
239 }
240 // Create iter_values from the original block.
241 std::vector<PrimExpr> iter_values;
242 for (const Var& it : info->origin_block_vars[expr_index]) {
243 iter_values.push_back(Substitute(info->var_binding.at(it), replace_table));
244 }
245 // block variables
246 Array<IterVar> block_vars;
247 // block access region for write buffers
248 Region access_region;
249 // indices used in block body
250 Array<PrimExpr> access_indices;
251 Map<Var, PrimExpr> block_var_map;
252 // Create block vars, block's accessed region and accessing indices
253 for (size_t i = 0; i < info->origin_block_vars[expr_index].size(); i++) {
254 const Var& block_var = info->origin_block_vars[expr_index][i];
255 Var var("v" + std::to_string(access_indices.size()), block_var.dtype());
256 Range range = Range::FromMinExtent(make_zero(block_var.dtype()),
257 info->range_map.at(iter_vars[i])->extent);
258 block_vars.push_back(IterVar(/*dom=*/range,
259 /*var=*/var,
260 /*IterVarType=*/kDataPar));
261
262 access_indices.push_back(var);
263 access_region.push_back(Range::FromMinExtent(var, make_const(var.dtype(), 1)));
264 block_var_map.Set(block_var, var);
265 }
266
267 // Create the index computing block
268 PrimExpr new_expr = Substitute(index_expr, block_var_map);
269 Block block(
270 /*iter_vars=*/std::move(block_vars),
271 /*reads=*/{},
272 /*writes=*/{BufferRegion(info->cache_buffer[expr_index], access_region)},
273 /*name_hint=*/"index_" + std::to_string(expr_index),
274 /*body=*/
275 BufferStore(info->cache_buffer[expr_index], new_expr, access_indices),
276 /*init=*/NullOpt,
277 /*alloc_buffers=*/{},
278 /*match_buffers=*/{},
279 /*annotations=*/{});
280 blocks.push_back(block);
281 // Create the block realize node
282 Stmt body = BlockRealize(/*values=*/iter_values,
283 /*predicate=*/const_true(),
284 /*block=*/block);
285 // Create surrounding loops
286 for (size_t i = loop_vars.size(); i >= 1; --i) {
287 body = For(/*loop_var=*/loop_vars[i - 1],
288 /*min=*/0,
289 /*extent=*/info->range_map.at(iter_vars[i - 1])->extent,
290 /*kind=*/ForKind::kSerial,
291 /*body=*/body);
292 }
293 bodies.push_back(body);
294 }
295
296 info->cache_stage = SeqStmt(bodies);
297 return blocks;
298}

300/*!
301 * \brief Insert the cache stages into the specific position
302 * \param stmt A sequence of statements or a single statement that the new stage is inserted in
303 * \param pos The position where the cache stage is inserted
304 * \param stage The stage to be inserted
305 * \return A SeqStmt, the result after insertion
306 */
307Stmt InsertIndexStage(const Stmt& stmt, int pos, const Stmt& stage) {
308 if (const auto* seq_stmt = stmt.as<SeqStmtNode>()) {
309 ObjectPtr<SeqStmtNode> result = make_object<SeqStmtNode>(*seq_stmt);
310 result->seq.insert(result->seq.begin() + pos, stage);
311 return SeqStmt(result);
312 }
313 if (pos == 0) {
314 return SeqStmt::Flatten<Array<Stmt>>({stage, stmt});
315 }
316 ICHECK_EQ(pos, 1);
317 return SeqStmt::Flatten<Array<Stmt>>({stmt, stage});
318}

320/*! \brief Mutator for CacheIndex. */
321class CacheIndexRewriter : public StmtExprMutator {
322 public:
323 /*!
324 * \brief Rewrite the AST and add stages of writting precomputed index
325 * \param scope_sref The parent scope of this mutation
326 * \param info The index information
327 * \return The new AST rooting at the original parent scope
328 */
329 static Stmt Rewrite(const StmtSRef& scope_sref, IndexInfo* info) {
330 CacheIndexRewriter rewriter(scope_sref, info);
331 return rewriter(GetRef<Stmt>(scope_sref->stmt));
332 }
333
334 private:
335 explicit CacheIndexRewriter(const StmtSRef& scope_sref, IndexInfo* info)
336 : scope_sref_(scope_sref), info_(info) {
337 cache_indices_.reserve(info_->origin_block_vars.size());
338 for (const Array<Var>& group_it : info_->origin_block_vars) {
339 cache_indices_.push_back({});
340 for (const Var& it : group_it) {
341 cache_indices_.back().push_back(it);
342 }
343 }
344 }
345
346 Stmt VisitStmt_(const BlockNode* block) final {
347 Block old_stmt = GetRef<Block>(block);
348 // Mutate the body
349 Block stmt = Downcast<Block>(StmtMutator::VisitStmt_(block));
350
351 // Check if it is the block corresponding to the parent scope
352 if (block == scope_sref_->stmt) {
353 // If so, put buffer allocation and insert cache stages on the parent scope
354 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
355 n->body = InsertIndexStage(n->body, info_->loc_pos, info_->cache_stage);
356 for (const Buffer& it : info_->cache_buffer) {
357 n->alloc_buffers.push_back(it);
358 }
359 stmt = Block(n);
360 }
361 info_->block_reuse.Set(old_stmt, stmt);
362 return std::move(stmt);
363 }
364
365 PrimExpr VisitExpr_(const BufferLoadNode* load) final {
366 if (load->buffer.same_as(info_->target_buffer)) {
367 // Rewrite the target buffer load
368 Array<PrimExpr> new_indices;
369 for (const PrimExpr& index : load->indices) {
370 auto it = std::find_if(info_->index_exprs.begin(), info_->index_exprs.end(),
371 [&](PrimExpr& e) { return e.get() == index.get(); });
372 if (it == info_->index_exprs.end()) {
373 new_indices.push_back(index);
374 } else {
375 // Replace load index with cached index
376 auto offset = std::distance(info_->index_exprs.begin(), it);
377 new_indices.push_back(BufferLoad(info_->cache_buffer[offset], cache_indices_[offset]));
378 }
379 }
380 return BufferLoad(load->buffer, new_indices);
381 }
382 return ExprMutator::VisitExpr_(load);
383 }
384
385 PrimExpr VisitExpr_(const LoadNode* op) final {
386 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
387 return PrimExpr();
388 }
389
390 private:
391 /*! \brief The parent scope of the insertion */
392 const StmtSRef& scope_sref_;
393 /*! \brief The info for inserting cache stage */
394 IndexInfo* info_;
395 /*! \brief The indices for the cache buffer */
396 std::vector<Array<PrimExpr>> cache_indices_;
397};
398
399Array<StmtSRef> CacheIndex(ScheduleState self, const StmtSRef& block_sref, int buffer_index) {
400 /*!
401 * Check:
402 * - The index is in the array of block reading region
403 *
404 * Mutate:
405 * - Allocate new cache buffers under the current scope.
406 * - Precompute the index and store it in cache buffers.
407 */
408
409 // Step 0. Checking index, getting the target buffer and the parent scope
410 IndexInfo info;
411 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
412 info.target_buffer =
413 GetNthAccessBuffer(self, GetRef<Block>(block), buffer_index, BufferIndexType::kRead);
414 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
415
416 // Step 1. Collect the indexing info of target buffer.
417 IndexInfoCollector::Collect(self, block_sref, scope_sref, &info);
418
419 // Step 2. Create cache stages and rewrite the stmt.
420 BlockRealize realize = GetBlockRealize(self, block_sref);
421 info.var_binding = GetBindings(realize);
422 Array<Block> cache_stages = MakeIndexCacheStage(&info);
423 Stmt new_scope = CacheIndexRewriter::Rewrite(/*scope_sref=*/scope_sref, /*info=*/&info);
424
425 bool old_stage_pipeline = self->block_info[block_sref].scope->stage_pipeline;
426
427 // Step 3. Replacing and updating flags.
428 self->Replace(scope_sref, new_scope, info.block_reuse);
429 Array<StmtSRef> result_block_srefs;
430 for (const Block& it : cache_stages) {
431 StmtSRef result_block_sref = self->stmt2ref.at(it.get());
432 result_block_srefs.push_back(result_block_sref);
433 BlockInfo& block_info = self->block_info[result_block_sref];
434
435 bool affine_binding = false;
436 if (result_block_sref->parent == nullptr) {
437 affine_binding = true;
438 } else {
439 arith::Analyzer analyzer;
440 StmtSRef parent_sref = GetRef<StmtSRef>(result_block_sref->parent);
441 affine_binding = IsAffineBinding(/*realize=*/GetBlockRealize(self, result_block_sref),
442 /*loop_var_ranges=*/LoopDomainOfSRefTreePath(parent_sref),
443 /*analyzer=*/&analyzer);
444 }
445
446 block_info.affine_binding = affine_binding;
447 block_info.region_cover = true;
448 block_info.scope->stage_pipeline = old_stage_pipeline;
449 }
450
451 return result_block_srefs;
452}
