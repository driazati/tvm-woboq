27class NotSingleWriteBlock : public ScheduleError {
28 public:
29 explicit NotSingleWriteBlock(IRModule mod, Buffer buffer, Array<StmtSRef> write_blocks)
30 : mod_(std::move(mod)), buffer_(std::move(buffer)) {
31 ICHECK_GT(write_blocks.size(), 1);
32 write_blocks_.reserve(write_blocks.size());
33 for (const StmtSRef& block_sref : write_blocks) {
34 const BlockNode* block = TVM_SREF_TO_BLOCK(block, block_sref);
35 write_blocks_.push_back(GetRef<Block>(block));
36 }
37 }
38
39 String FastErrorString() const final {
40 return "ScheduleError: The buffer is allowed to be written by single block.";
41 }
42
43 String DetailRenderTemplate() const final {
44 size_t k = write_blocks_.size();
45 return "The buffer " + buffer_->name + " is expected to be written by single block, but got " +
46 std::to_string(k) + " blocks who write it.";
47 }
48
49 IRModule mod() const final { return mod_; }
50 Array<ObjectRef> LocationsOfInterest() const final {
51 return {write_blocks_.begin(), write_blocks_.end()};
52 }
53
54 private:
55 IRModule mod_;
56 Buffer buffer_;
57 Array<Block> write_blocks_;
58};

62/*! \brief The auxiliary info used for the insertion point and content of the cache stage. */
63struct CacheStageInfo {
64 /*! \brief The buffer to be read. */
65 Buffer read_buffer;
66 /*! \brief The buffer to be written. */
67 Buffer write_buffer;
68 /*! \brief The buffer allocation to be inserted into the block signature. */
69 Buffer alloc;
70 /*! \brief The AST node whose body is where the cache stage should be inserted. */
71 StmtSRef loc_sref;
72 /*! \brief The index to insert the cache_read/cache_write stage. */
73 size_t loc_pos;
74 /*! \brief The cache_read/cache_write stage to be inserted. */
75 Stmt cache_stage;
76 /*! \brief The map used for ScheduleStateNode::Replace. */
77 Map<Block, Block> block_reuse;
78};

80/*! \brief Return the buffer region realted with the buffer */
81Optional<BufferRegion> GetBufferRegionFromBuffer(const Array<BufferRegion>& buffer_regions,
82 const Buffer& buffer) {
83 Optional<BufferRegion> res = NullOpt;
84 for (const auto& region : buffer_regions) {
85 if (region->buffer.same_as(buffer)) {
86 ICHECK(!res.defined());
87 res = region;
88 }
89 }
90 return res;
91}

93/*!
94 * \brief Create a loop nest that represents cache copy (cache_read / cache_write) from read buffer
95 * to write buffer.
96 * \note This function will store the stmt with loop nesting to the CacheStageInfo, but only return
97 * the inside block.
98 * \param cache_region The cached copy region.
99 * \param info The cache stage information, which will be updated in the function.
100 * \param storage_scope The storage scope of the cached buffer (only used in naming here)
101 * \returns A block indicating the body of the loop nesting.
102 */
103Block MakeCacheStage(const BufferRegion& cache_region, CacheStageInfo* info,
104 const String& storage_scope) {
105 // loop variables
106 std::vector<Var> loop_vars;
107 // bindings in block realize
108 std::vector<PrimExpr> iter_values;
109 // Create loop vars and block vars' binding_value
110 for (const Range& axis_range : cache_region->region) {
111 Var loop_var("ax" + std::to_string(loop_vars.size()), axis_range->extent.dtype());
112 loop_vars.push_back(loop_var);
113 iter_values.push_back(axis_range->min + loop_var);
114 }
115 // block variables
116 Array<IterVar> block_vars;
117 // block access region for read/write buffers
118 Region access_region;
119 // indices used in block body
120 Array<PrimExpr> access_indices;
121 // Create block vars, block's accessed region and accessing indices
122 for (const PrimExpr& dim : cache_region->buffer->shape) {
123 Var var("v" + std::to_string(access_indices.size()), dim.dtype());
124 block_vars.push_back(IterVar(/*dom=*/Range::FromMinExtent(make_zero(dim->dtype), dim),
125 /*var=*/var,
126 /*IterVarType=*/kDataPar));
127 access_indices.push_back(var);
128 access_region.push_back(Range::FromMinExtent(var, make_const(var.dtype(), 1)));
129 }
130
131 // Create the body block:
132 // reads = [read_buffer[access_region]]
133 // writes = [write_buffer[access_region]]
134 // write_buffer[access_indices] = read_buffer[access_indices]
135 Block block(
136 /*iter_vars=*/std::move(block_vars),
137 /*reads=*/{BufferRegion(info->read_buffer, access_region)},
138 /*writes=*/{BufferRegion(info->write_buffer, access_region)},
139 /*name_hint=*/cache_region->buffer->name + "_" + storage_scope,
140 /*body=*/
141 BufferStore(info->write_buffer, BufferLoad(info->read_buffer, access_indices),
142 access_indices),
143 /*init=*/NullOpt,
144 /*alloc_buffers=*/{},
145 /*match_buffers=*/{},
146 /*annotations=*/{});
147 // Create the block realize node
148 Stmt body = BlockRealize(/*values=*/iter_values,
149 /*predicate=*/const_true(),
150 /*block=*/block);
151 // Create surrounding loops
152 for (size_t i = loop_vars.size(); i >= 1; --i) {
153 body = For(/*loop_var=*/loop_vars[i - 1],
154 /*min=*/0,
155 /*extent=*/cache_region->region[i - 1]->extent,
156 /*kind=*/ForKind::kSerial,
157 /*body=*/body);
158 }
159 info->cache_stage = std::move(body);
160 return block;
161}

163/*!
164 * \brief Create the reindex block and generate the corresponding outer loops.
165 * \details The reindex block is a data copy block between the reindex buffer (the intermediate
166 * buffer), and the target buffer.
167 If buffer_index_type == kWrite, copy from the reindex buffer to the target buffer.
168 If buffer_index_type == kRead, copy from the target buffer to the reindex buffer.
169 The reindex block has the same block iters and the surrounding loops as the input block.
170 However, if a block iter is not used in the indices of the target buffer being reindexed, the
171 domain of the block iter, and the corresponding outer loop, will become constant value one, making
172 it a trivial iter.
173 * \param block The block to be reindexed
174 * \param info The cache info
175 * \param covered The set of block iter vars covered in the buffer access indices
176 * \param original_indices The original buffer access indices
177 * \param buffer_index The index of the target buffer
178 * \param buffer_index_type The type of buffer index
179 * \return The reindex block.
180 */
181Block MakeReIndexStage(const Block& block, CacheStageInfo* info,
182 const std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual>& covered,
183 const Array<PrimExpr>& original_indices, int buffer_index,
184 BufferIndexType buffer_index_type) {
185 // iters of the reindex block
186 Array<IterVar> new_block_iters;
187 // the substition map from the original block iter to the iters of the reindex block
188 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectEqual> block_var_replace_map;
189 // block access region of reindexed buffer and target buffer
190 Region reindex_region, target_region;
191 // indices to access the reindex buffer and the target buffer
192 Array<PrimExpr> reindex_indices, target_indices;
193
194 // Step 1: Create block iters, access regions of the reindex block, and accessing indices to the
195 // reindex buffer.
196 for (const IterVar& iter : block->iter_vars) {
197 Var var("v" + std::to_string(new_block_iters.size()));
198 bool used = covered.count(iter->var);
199 new_block_iters.push_back(IterVar(/*dom=*/used ? iter->dom : Range::FromMinExtent(0, 1),
200 /*var=*/var,
201 /*IterVarType=*/kDataPar));
202 if (used) {
203 reindex_indices.push_back(var);
204 reindex_region.push_back(Range::FromMinExtent(var, 1));
205 }
206 block_var_replace_map[iter->var] = var;
207 }
208
209 // Step 2: Replace the original block iters with the new block iters
210 BufferRegion buffer_region = buffer_index_type == BufferIndexType::kWrite
211 ? block->writes[buffer_index]
212 : block->reads[buffer_index];
213 target_region = Substitute(buffer_region->region, block_var_replace_map);
214 for (const PrimExpr& index : original_indices) {
215 target_indices.push_back(Substitute(index, block_var_replace_map));
216 }
217
218 // Step 3: Create the reindex block
219
220 // The src and the dst region and indices of the data copy
221 Region src_region{nullptr};
222 Region dst_region{nullptr};
223 Array<PrimExpr> src_indices{nullptr};
224 Array<PrimExpr> dst_indices{nullptr};
225
226 if (buffer_index_type == BufferIndexType::kWrite) {
227 src_region = reindex_region;
228 dst_region = target_region;
229 src_indices = reindex_indices;
230 dst_indices = target_indices;
231 } else {
232 src_region = target_region;
233 dst_region = reindex_region;
234 src_indices = target_indices;
235 dst_indices = reindex_indices;
236 }
237
238 // Create the body block
239 Block new_block(
240 /*iter_vars=*/new_block_iters,
241 /*reads=*/
242 {BufferRegion(info->read_buffer, src_region)},
243 /*writes=*/
244 {BufferRegion(info->write_buffer, dst_region)},
245 /*name_hint=*/buffer_region->buffer->name + "_reindex",
246 /*body=*/
247 BufferStore(info->write_buffer, BufferLoad(info->read_buffer, src_indices), dst_indices));
248
249 // Step 4: Create surrounding loops
250
251 // Create loop vars and bindings for block iters
252 std::vector<Var> loop_vars; // loop variables
253 std::vector<PrimExpr> iter_values; // bindings in block realize
254 for (int i = 0; i < static_cast<int>(block->iter_vars.size()); ++i) {
255 Var loop_var("ax" + std::to_string(loop_vars.size()));
256 loop_vars.push_back(loop_var);
257 iter_values.push_back(loop_var);
258 }
259
260 // Create the block realize node
261 Stmt body = BlockRealize(/*values=*/iter_values,
262 /*predicate=*/const_true(),
263 /*block=*/new_block);
264
265 // Create the chain of loops
266 for (int i = static_cast<int>(new_block_iters.size()) - 1; i >= 0; --i) {
267 body = For(/*loop_var=*/loop_vars[i],
268 /*min=*/new_block_iters[i]->dom->min,
269 /*extent=*/new_block_iters[i]->dom->extent,
270 /*kind=*/ForKind::kSerial,
271 /*body=*/std::move(body));
272 }
273 // Update cache info, which will be used in the later rewriting.
274 info->cache_stage = std::move(body);
275 return new_block;
276}

278/*!
279 * \brief Recalculate the `affine_binding` flag of a specifc block
280 * \param block_sref The sref to the specific block
281 */
282bool CalculateAffineFlag(const ScheduleState& self, const StmtSRef& block_sref) {
283 if (block_sref->parent == nullptr) {
284 return true;
285 }
286 arith::Analyzer analyzer;
287 StmtSRef parent_sref = GetRef<StmtSRef>(block_sref->parent);
288 return IsAffineBinding(/*realize=*/GetBlockRealize(self, block_sref),
289 /*loop_var_ranges=*/LoopDomainOfSRefTreePath(parent_sref),
290 /*analyzer=*/&analyzer);
291}

293/*!
294 * \brief Insert the cache_read/cache_write stage into the specific position
295 * \param stmt A sequence of statements or a single statement that the new stage is inserted in
296 * \param pos The position where the cache stage is inserted
297 * \param stage The stage to be inserted
298 * \return A SeqStmt, the result after insertion
299 */
300SeqStmt InsertCacheStage(const Stmt& stmt, int pos, const Stmt& stage) {
301 if (const auto* seq_stmt = stmt.as<SeqStmtNode>()) {
302 ObjectPtr<SeqStmtNode> result = make_object<SeqStmtNode>(*seq_stmt);
303 result->seq.insert(result->seq.begin() + pos, stage);
304 return SeqStmt(result);
305 }
306 if (pos == 0) {
307 return SeqStmt({stage, stmt});
308 }
309 ICHECK_EQ(pos, 1);
310 return SeqStmt({stmt, stage});
311}

313/*!
314 * \brief Get the only writer block of the input buffer in a given scope block.
315 * \param self The state of the schedule
316 * \param scope_sref The scope block where the write is considered
317 * \param buffer The queried buffer
318 * \return The sref of the only writer of the input buffer in the given scope,
319 * or `NullOpt` if no block writes it in the scope.
320 * \throw NotSingleWriteBlock if there are more than one intrested block.
321 */
322Optional<StmtSRef> GetOnlyWriteBlock(ScheduleState self, const StmtSRef& scope_sref,
323 const Buffer& buffer) {
324 BlockScope scope = self->GetBlockScope(scope_sref);
325 auto it = scope->buffer_writers.find(buffer);
326 if (it == scope->buffer_writers.end()) {
327 return NullOpt;
328 } else {
329 const Array<StmtSRef>& block_srefs = it->second;
330 ICHECK(!block_srefs.empty());
331 if (block_srefs.size() > 1) {
332 throw NotSingleWriteBlock(self->mod, buffer, block_srefs);
333 }
334 return block_srefs[0];
335 }
336}

338/*!
339 * \brief Get the buffer region under the sref tree path [dom_low_inclusive, dom_high_exclusive)
340 * \param self The state of the schedule.
341 * \param buffer_region The buffer region to be analyzed.
342 * \param block_sref The sref of the block related to the region.
343 * \param dom_low_inclusive The lowest node in the sref tree path.
344 * \param dom_high_exclusive The highest node in the sref tree path.
345 * \return The relaxed buffer region.
346 */
347BufferRegion RelaxBufferRegion(ScheduleState self, const BufferRegion& buffer_region,
348 const StmtSRef& block_sref, const StmtSRef& dom_low_inclusive,
349 const StmtSRef& dom_high_exclusive) {
350 BlockRealize realize = GetBlockRealize(self, block_sref);
351 Map<Var, PrimExpr> binding = GetBindings(realize);
352 const Buffer& buffer = buffer_region->buffer;
353 arith::Analyzer analyzer;
354 BufferRegion subst_region = BufferRegion(buffer, Substitute(buffer_region->region, binding));
355 Array<arith::IntSet> int_sets = AnalyzeRegionUpperBound(
356 /*region=*/subst_region,
357 /*predicate=*/realize->predicate,
358 /*dom_low_inclusive=*/dom_low_inclusive,
359 /*dom_high_exclusive=*/dom_high_exclusive,
360 /*analyzer=*/&analyzer);
361 ICHECK_EQ(buffer_region->region.size(), int_sets.size());
362
363 Region region;
364 region.reserve(int_sets.size());
365 for (size_t i = 0; i < int_sets.size(); ++i) {
366 region.push_back(int_sets[i].CoverRange(Range::FromMinExtent(0, buffer->shape[i])));
367 }
368 return BufferRegion(buffer, region);
369}

371/*! \brief Detect the insertion position of the new cache stage */
372class CacheLocDetector : public StmtVisitor {
373 public:
374 /*!
375 * \brief Detect the insertion position of the cache stage, and write the position into the
376 * CacheStageInfo \param self The state of the schedule \param block_sref The sref of the unique
377 * writer block of the buffer being applied cache_read or cache_write \param scope_sref The sref
378 * of the scope block of the cached block \param info The cache stage info.
379 */
380 static void Detect(const ScheduleState& self, const StmtSRef& block_sref,
381 const StmtSRef& scope_sref, CacheStageInfo* info) {
382 std::vector<StmtSRef> related_blocks;
383 for (const Dependency& def : self->GetBlockScope(scope_sref)->GetDepsBySrc(block_sref)) {
384 if (def->kind == DepKind::kRAW) {
385 related_blocks.push_back(def->dst);
386 }
387 }
388 if (!related_blocks.empty()) {
389 CacheLocDetector detector(self, block_sref, scope_sref, related_blocks);
390 detector(GetRef<Stmt>(scope_sref->stmt));
391 info->loc_sref = detector.loc_sref_;
392 info->loc_pos = detector.loc_pos_;
393 } else {
394 info->loc_sref = scope_sref;
395 const auto* body = scope_sref->StmtAs<BlockNode>()->body.as<SeqStmtNode>();
396 info->loc_pos = body == nullptr ? 1 : body->size();
397 }
398 }
399
400 private:
401 /*!
402 * \brief Constructor
403 * \param self The state of the schedule
404 * \param block_sref The sref of the unique writer block of the buffer being applied cache_read or
405 * cache_write \param scope_sref The sref of the scope block of the cached block \param
406 * related_blocks Producer blocks for cache_write, or consumer blocks for cache_read
407 */
408 CacheLocDetector(const ScheduleState self, const StmtSRef& block_sref, const StmtSRef& scope_sref,
409 const std::vector<StmtSRef>& related_blocks)
410 : self_(self),
411 block_sref_(block_sref),
412 scope_sref_(scope_sref),
413 related_blocks_(related_blocks) {}
414
415 void VisitStmt_(const SeqStmtNode* seq_stmt) final {
416 bool previous_visited_block = visited_block_;
417 bool previous_visited_related = visited_related_;
418 visited_block_ = visited_related_ = false;
419
420 int pos = -1;
421 for (size_t i = 0; i < seq_stmt->size(); ++i) {
422 if (loc_pos_ != -1) {
423 break;
424 }
425 VisitStmt(seq_stmt->seq[i]);
426 // `pos` can be assigned only once when we visited `block_sref`
427 if (visited_block_ && visited_related_ && pos == -1) {
428 // The offset of insert position from the block
429 pos = i;
430 }
431 }
432 visited_block_ = visited_block_ || previous_visited_block;
433 visited_related_ = visited_related_ || previous_visited_related;
434 // Only we visited the writing block and any one of the related blocks
435 // That means that we have found the lowest ancestor
436 // of the block and any one of the related ones
437 if (visited_block_ && visited_related_ && loc_pos_ == -1) {
438 loc_pos_ = pos;
439 }
440 }
441
442 void VisitStmt_(const BlockNode* block) final {
443 // Only visit the current scope under buffer writer's parent block
444 if (block == scope_sref_->stmt) {
445 // The block vistied is the current parent scope
446 StmtVisitor::VisitStmt_(block);
447 // Handling cache_read for input buffer
448 if (visited_block_ && visited_related_ && !loc_sref_.defined()) {
449 loc_sref_ = self_->stmt2ref.at(block);
450 if (loc_pos_ == -1) {
451 loc_pos_ = 1;
452 }
453 }
454 return;
455 }
456 // Update `visited_block`
457 if (block_sref_->stmt == block) {
458 visited_block_ = true;
459 return;
460 }
461 // Update `visited_related`
462 for (const StmtSRef& related_block : related_blocks_) {
463 if (related_block->stmt == block) {
464 visited_related_ = true;
465 return;
466 }
467 }
468 }
469
470 void VisitStmt_(const ForNode* loop) final {
471 StmtVisitor::VisitStmt_(loop);
472 if (visited_block_ && visited_related_ && !loc_sref_.defined() && loc_pos_ != -1) {
473 loc_sref_ = self_->stmt2ref.at(loop);
474 }
475 }
476
477 private:
478 /*! \brief The schedule class */
479 const ScheduleState self_;
480 /*! \brief The dominate block which write the buffer */
481 const StmtSRef& block_sref_;
482 /*! \brief The parent scope of the dominate block */
483 const StmtSRef& scope_sref_;
484 /*! \brief Producer blocks for cache_write and consumer blocks for cache_read */
485 const std::vector<StmtSRef>& related_blocks_;
486 /*! \brief The flag whether we have visited the dominate block */
487 bool visited_block_{false};
488 /*! \brief The flag whether we have visited at least one related blocks */
489 bool visited_related_{false};
490 /*! \brief The AST node whose body is where the cache stage should be inserted */
491 StmtSRef loc_sref_{nullptr};
492 /*! \brief The index to insert the cache_read/cache_write stage */
493 int loc_pos_{-1};
494};

496/*! \brief Mutator for CacheRead. */
497class CacheReadRewriter : public StmtExprMutator {
498 public:
499 /*!
500 * \brief Rewrite the AST and add a cache_read stage with the information provided
501 * \param scope_sref The parent scope of this mutation
502 * \param info The cache stage information
503 * \return The new AST rooting at the original parent scope
504 */
505 static Stmt Rewrite(const StmtSRef& scope_sref, CacheStageInfo* info) {
506 CacheReadRewriter rewriter(scope_sref, info);
507 return rewriter(GetRef<Stmt>(scope_sref->stmt));
508 }
509
510 private:
511 explicit CacheReadRewriter(const StmtSRef& scope_sref, CacheStageInfo* info)
512 : scope_sref_(scope_sref), info_(info) {}
513
514 Stmt VisitStmt_(const ForNode* loop) final {
515 Stmt stmt = StmtMutator::VisitStmt_(loop);
516 // Check the insertion point
517 if (loop == info_->loc_sref->stmt) {
518 // Insert cache stage into the loop if it is the right place
519 ObjectPtr<ForNode> n = make_object<ForNode>(*stmt.as<ForNode>());
520 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
521 stmt = Stmt(n);
522 }
523 return stmt;
524 }
525
526 Stmt VisitStmt_(const BlockNode* block) final {
527 Block old_stmt = GetRef<Block>(block);
528 // We don't mutate the block which generates info->read_buffer
529 if (block != scope_sref_->stmt &&
530 GetBufferRegionFromBuffer(block->writes, info_->read_buffer).defined()) {
531 return std::move(old_stmt);
532 }
533 // Mutate the body
534 Block stmt = Downcast<Block>(StmtMutator::VisitStmt_(block));
535 // Check the insertion point
536 if (block == info_->loc_sref->stmt) {
537 // Insert cache stage into the block if it is the right place
538 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
539 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
540 stmt = Block(n);
541 }
542 // Check if it is the block corresponding to the parent scope
543 if (block == scope_sref_->stmt) {
544 // If so, put buffer allocation on the parent scope
545 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
546 n->alloc_buffers.push_back(info_->alloc);
547 stmt = Block(n);
548 } else {
549 // Otherwise, update read regions and match_buffers
550 Array<BufferRegion> reads =
551 ReplaceBuffer(block->reads, info_->read_buffer, info_->write_buffer);
552 Array<MatchBufferRegion> match_buffers =
553 ReplaceBuffer(block->match_buffers, info_->read_buffer, info_->write_buffer);
554 if (!reads.same_as(block->reads) || !match_buffers.same_as(block->match_buffers)) {
555 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
556 n->reads = std::move(reads);
557 n->match_buffers = std::move(match_buffers);
558 stmt = Block(n);
559 }
560 }
561 info_->block_reuse.Set(old_stmt, stmt);
562 return std::move(stmt);
563 }
564
565 PrimExpr VisitExpr_(const BufferLoadNode* load) final {
566 if (load->buffer.same_as(info_->read_buffer)) {
567 ObjectPtr<BufferLoadNode> n = make_object<BufferLoadNode>(*load);
568 n->buffer = info_->write_buffer;
569 return PrimExpr(n);
570 }
571 return ExprMutator::VisitExpr_(load);
572 }
573
574 PrimExpr VisitExpr_(const LoadNode* op) final {
575 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
576 return PrimExpr();
577 }
578
579 PrimExpr VisitExpr_(const VarNode* op) final {
580 if (op == info_->read_buffer->data.get()) {
581 return info_->write_buffer->data;
582 }
583 return GetRef<PrimExpr>(op);
584 }
585
586 private:
587 /*! \brief The parent scope of the insertion */
588 const StmtSRef& scope_sref_;
589 /*! \brief The info for inserting cache stage */
590 CacheStageInfo* info_;
591};

593/*! \brief Mutator for CacheWrite */
594class CacheWriteRewriter : public StmtExprMutator {
595 public:
596 /*!
597 * \brief Rewrite the AST and add a cache_write stage with the information provided.
598 * \param scope_sref The parent scope of this mutation.
599 * \param writer_block_sref The only writer block in the scope.
600 * \param info The cache stage information.
601 * \return The new AST rooting at the original parent scope.
602 */
603 static Stmt Rewrite(const StmtSRef& scope_sref, const StmtSRef& writer_block_sref,
604 CacheStageInfo* info) {
605 CacheWriteRewriter rewriter(scope_sref, writer_block_sref, info);
606 return rewriter(GetRef<Stmt>(scope_sref->stmt));
607 }
608
609 private:
610 explicit CacheWriteRewriter(const StmtSRef& scope_sref, const StmtSRef& writer_block_sref,
611 CacheStageInfo* info)
612 : scope_sref_(scope_sref), writer_block_sref_(writer_block_sref), info_(info) {}
613
614 Stmt VisitStmt_(const ForNode* loop) final {
615 Stmt stmt = StmtMutator::VisitStmt_(loop);
616 // Check the insertion point
617 if (loop == info_->loc_sref->stmt) {
618 // Insert cache stage into the loop if it is the right place
619 ObjectPtr<ForNode> n = make_object<ForNode>(*stmt.as<ForNode>());
620 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
621 stmt = Stmt(n);
622 }
623 return stmt;
624 }
625
626 Stmt VisitStmt_(const BlockNode* block) final {
627 Block old_stmt = GetRef<Block>(block);
628 // We only mutate the block which generates info->write_buffer
629 if (block != writer_block_sref_->stmt && block != scope_sref_->stmt && !under_writer_block_) {
630 return std::move(old_stmt);
631 }
632
633 // Mutate the body
634 bool under_scope = under_writer_block_ || block == writer_block_sref_->stmt;
635 std::swap(under_scope, under_writer_block_);
636 Block stmt = Downcast<Block>(StmtMutator::VisitStmt_(block));
637 std::swap(under_scope, under_writer_block_);
638
639 // Find the insertion point
640 if (block == info_->loc_sref->stmt) {
641 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
642 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
643 stmt = Block(n);
644 }
645 // Put buffer allocation on the parent scope
646 if (block == scope_sref_->stmt) {
647 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
648 n->alloc_buffers.push_back(info_->alloc);
649 stmt = Block(n);
650 } else {
651 // Since cache_write changes the block, we need to update the buffer it writes
652 auto writes = ReplaceBuffer(block->writes, info_->write_buffer, info_->read_buffer);
653 auto reads = ReplaceBuffer(block->reads, info_->write_buffer, info_->read_buffer);
654 auto match_buffers =
655 ReplaceBuffer(block->match_buffers, info_->write_buffer, info_->read_buffer);
656 if (!writes.same_as(block->writes) || !reads.same_as(block->reads) ||
657 !match_buffers.same_as(block->match_buffers)) {
658 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
659 n->writes = std::move(writes);
660 n->reads = std::move(reads);
661 n->match_buffers = std::move(match_buffers);
662 stmt = Block(n);
663 }
664 }
665 info_->block_reuse.Set(old_stmt, stmt);
666 return std::move(stmt);
667 }
668
669 Stmt VisitStmt_(const BufferStoreNode* store) final {
670 BufferStore stmt = Downcast<BufferStore>(StmtMutator::VisitStmt_(store));
671 if (stmt->buffer.same_as(info_->write_buffer)) {
672 auto n = CopyOnWrite(stmt.get());
673 n->buffer = info_->read_buffer;
674 return Stmt(n);
675 } else {
676 return std::move(stmt);
677 }
678 }
679
680 PrimExpr VisitExpr_(const BufferLoadNode* load) final {
681 if (load->buffer.same_as(info_->write_buffer)) {
682 ObjectPtr<BufferLoadNode> n = make_object<BufferLoadNode>(*load);
683 n->buffer = info_->read_buffer;
684 return PrimExpr(n);
685 }
686 return ExprMutator::VisitExpr_(load);
687 }
688
689 PrimExpr VisitExpr_(const LoadNode* op) final {
690 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
691 return PrimExpr();
692 }
693
694 Stmt VisitStmt_(const StoreNode* op) final {
695 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
696 return Stmt();
697 }
698
699 PrimExpr VisitExpr_(const VarNode* op) final {
700 if (op == info_->write_buffer->data.get()) {
701 return info_->read_buffer->data;
702 }
703 return GetRef<PrimExpr>(op);
704 }

706 private:
707 /*! \brief The parent scope of the insertion. */
708 const StmtSRef& scope_sref_;
709 /*! \brief The parent scope of the insertion. */
710 const StmtSRef& writer_block_sref_;
711 /*! \brief The info for inserting cache stage. */
712 CacheStageInfo* info_;
713 /*! \brief Whether the current node is under the given block. */
714 bool under_writer_block_{false};
715};

717/*!
718 * \brief Create a new buffer by change the shape with block iters to be used as the reindex buffer
719 * \param buffer The given buffer.
720 * \param block_iters The block iters.
721 * \param covered Set of block iter vars covered by the buffer access indices
722 * \return The new buffer with target shape.
723 */
724Buffer CreateReindexBuffer(const Buffer& buffer, const Array<IterVar>& block_iters,
725 const std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual>& covered) {
726 ObjectPtr<BufferNode> new_buffer = make_object<BufferNode>(*buffer.get());
727 ObjectPtr<VarNode> new_var = make_object<VarNode>(*buffer->data.get());
728 std::vector<PrimExpr> new_shape;
729 std::vector<PrimExpr> new_strides;
730 for (const auto& iter : block_iters) {
731 if (covered.count(iter->var)) {
732 new_shape.push_back(iter->dom->min + iter->dom->extent);
733 }
734 }
735 new_strides.clear();
736 new_buffer->shape = new_shape;
737 new_buffer->strides = new_strides;
738 new_buffer->data = buffer->data.copy_with_suffix("_reindex");
739 new_buffer->name = buffer->name + "_reindex";
740 return Buffer(new_buffer);
741}

863/*! \brief Mutator of ReIndex */
864class ReIndexRewriter : public StmtExprMutator {
865 public:
866 static Stmt Rewrite(const StmtSRef& scope_sref, const StmtSRef& block_sref, CacheStageInfo* info,
867 const std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual>& covered) {
868 ReIndexRewriter rewriter(block_sref, info, covered);
869 return rewriter(GetRef<Stmt>(scope_sref->stmt));
870 }
871
872 private:
873 explicit ReIndexRewriter(const StmtSRef& block_sref, CacheStageInfo* info,
874 const std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual>& covered)
875 : block_sref_(block_sref), info_(info), covered_(covered) {
876 new_buffer_ = info->alloc;
877 old_buffer_ = info->read_buffer.same_as(new_buffer_) ? info->write_buffer : info->read_buffer;
878 }
879
880 Stmt VisitStmt_(const BlockNode* block) final {
881 Block old_stmt = GetRef<Block>(block);
882 if (is_scope_) {
883 is_scope_ = false;
884 Block stmt = Downcast<Block>(StmtExprMutator::VisitStmt_(block));
885 // Insert cache stage into the loop
886 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
887 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
888 n->alloc_buffers.push_back(info_->alloc);
889 stmt = Block(n);
890 info_->block_reuse.Set(old_stmt, stmt);
891 return std::move(stmt);
892 }
893
894 // Visiting the blokc being reindexed
895 if (block == block_sref_->stmt) {
896 // Collect the updated indices and regions
897 for (const IterVar& iter : block->iter_vars) {
898 if (covered_.count(iter->var)) {
899 indices_.push_back(iter->var);
900 region_.push_back(Range::FromMinExtent(iter->var, 1));
901 }
902 }
903 Block stmt = Downcast<Block>(StmtExprMutator::VisitStmt_(block));
904 // Update block reads/writes to use the intermediate reindex buffer
905 auto writes =
906 ReplaceBufferRegion(block->writes, old_buffer_, BufferRegion{new_buffer_, region_});
907 auto reads =
908 ReplaceBufferRegion(block->reads, old_buffer_, BufferRegion{new_buffer_, region_});
909 auto match_buffers = ReplaceBufferRegion(block->match_buffers, old_buffer_,
910 BufferRegion{new_buffer_, region_});
911 if (!writes.same_as(block->writes) || !reads.same_as(block->reads) ||
912 !match_buffers.same_as(block->match_buffers)) {
913 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
914 n->writes = std::move(writes);
915 n->reads = std::move(reads);
916 n->match_buffers = std::move(match_buffers);
917 stmt = Block(n);
918 }
919 info_->block_reuse.Set(old_stmt, stmt);
920 return std::move(stmt);
921 }
922 return std::move(old_stmt);
923 }
924
925 template <typename Node>
926 Node VisitBufferAccess(Node node) {
927 if (node->buffer.same_as(old_buffer_)) {
928 auto* n = node.CopyOnWrite();
929 n->buffer = new_buffer_;
930 n->indices = indices_;
931 }
932 return node;
933 }
934 Stmt VisitStmt_(const BufferStoreNode* op) final {
935 BufferStore buffer_store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
936 return VisitBufferAccess(std::move(buffer_store));
937 }
938
939 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
940 BufferLoad buffer_load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
941 return VisitBufferAccess(std::move(buffer_load));
942 }
943
944 private:
945 /*! \brief The parent scope of the insertion. */
946 const StmtSRef& block_sref_;
947 /*! \brief The info for inserting reindex stage. */
948 CacheStageInfo* info_;
949 /*! \brief Whether old block var is covered in the indices */
950 const std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual>& covered_;
951 /*! \brief Whether the current block is scope block */
952 bool is_scope_{true};
953 /*! \brief The buffer to be replaced */
954 Buffer old_buffer_;
955 /*! \brief The reindex buffer */
956 Buffer new_buffer_;
957 /*! \brief The new indices */
958 Array<PrimExpr> indices_;
959 /*! \brief The new region */
960 Region region_;
961};

963/******** Implementation ********/
964
965StmtSRef CacheRead(ScheduleState self, const StmtSRef& block_sref, int read_buffer_index,
966 const String& storage_scope) {
967 /*!
968 * Check:
969 * - The index is in the array of block reading region
970 * - There is at most one block who write the buffer in the scope
971 *
972 * Mutate:
973 * - Allocate new cache buffer under the current scope.
974 * - Find the lowest ancestor of the block and ANY ONE of the consumers blocks.
975 * - Copy the buffer with the consumed region.
976 */
977
978 // Step 0. Check the input storage scope.
979 CheckStorageScope(self, storage_scope);
980
981 // Step 1. Check index, getting the target buffer and the parent scope
982 const BlockNode* block = TVM_SREF_TO_BLOCK(block, block_sref);
983 Buffer read_buffer =
984 GetNthAccessBuffer(self, GetRef<Block>(block), read_buffer_index, BufferIndexType::kRead);
985 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/true);
986 const BlockNode* scope_block = TVM_SREF_TO_BLOCK(scope_block, scope_sref);
987
988 // Step 2. Create CacheStageInfo
989 CacheStageInfo info;
990 info.read_buffer = read_buffer;
991 // Create the corresponding buffer to be written, i.e. result of cache_read
992 info.write_buffer = WithScope(read_buffer, storage_scope);
993 // Create the corresponding buffer allocation
994 info.alloc = info.write_buffer;
995
996 // Step 3. Update cache stage info.
997 BufferRegion cache_region{nullptr};
998 if (Optional<StmtSRef> _write_block_sref = GetOnlyWriteBlock(self, scope_sref, read_buffer)) {
999 // Case 1. The buffer is written inside the block.
1000 StmtSRef write_block_sref = _write_block_sref.value();
1001 const BlockNode* write_block = TVM_SREF_TO_BLOCK(write_block, write_block_sref);
1002 // Find the producing region
1003 BufferRegion region = GetBufferRegionFromBuffer(write_block->writes, read_buffer).value();
1004 StmtSRef parent_sref = GetRef<StmtSRef>(write_block_sref->parent);
1005
1006 // Detect insert position
1007 CacheLocDetector::Detect(self, write_block_sref, scope_sref, &info);
1008 cache_region = RelaxBufferRegion(self, region, write_block_sref, parent_sref, info.loc_sref);
1009 } else {
1010 // Case 2. The buffer is the input block for the scope.
1011 info.loc_sref = scope_sref;
1012 info.loc_pos = 0;
1013 if (Optional<BufferRegion> region =
1014 GetBufferRegionFromBuffer(scope_block->reads, read_buffer)) {
1015 cache_region = region.value();
1016 } else {
1017 cache_region = BufferRegion::FullRegion(read_buffer);
1018 }
1019 }
1020
1021 // Step 4. Making new cache stage block and rewrite readers.
1022 Block cache_read_stage = MakeCacheStage(/*cache_region=*/cache_region, /*info=*/&info,
1023 /*storage_scope=*/storage_scope);
1024 Stmt new_scope = CacheReadRewriter::Rewrite(/*scope_sref=*/scope_sref, /*info=*/&info);
1025
1026 // Step 5. Replacing and updating flags.
1027 self->Replace(scope_sref, new_scope, info.block_reuse);
1028 StmtSRef result_block_sref = self->stmt2ref.at(cache_read_stage.get());
1029 BlockInfo& block_info = self->block_info[result_block_sref];
1030 block_info.affine_binding = CalculateAffineFlag(self, result_block_sref);
1031 block_info.region_cover = true;
1032 block_info.scope->stage_pipeline = true;
1033 return result_block_sref;
1034}
1035
1036StmtSRef CacheWrite(ScheduleState self, const StmtSRef& block_sref, int write_buffer_index,
1037 const String& storage_scope) {
1038 /*!
1039 * Check:
1040 * - The index is in the array of block reading region
1041 * - There is only one block who write the buffer in the scope
1042 *
1043 * Mutate:
1044 * - Allocate new cache buffer under the current scope.
1045 * - Find the lowest ancestor of the block and ANY ONE of the producer blocks.
1046 * - Copy the buffer with the consumed region.
1047 */
1048
1049 // Step 0. Check the input storage scope.
1050 CheckStorageScope(self, storage_scope);
1051
1052 // Step 1. Checking index, getting the target buffer and the parent scope
1053 const BlockNode* block = TVM_SREF_TO_BLOCK(block, block_sref);
1054 Buffer write_buffer =
1055 GetNthAccessBuffer(self, GetRef<Block>(block), write_buffer_index, BufferIndexType::kWrite);
1056 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/true);
1057
1058 // Step 2. Creating CacheStageInfo
1059 CacheStageInfo info;
1060 info.read_buffer = WithScope(write_buffer, storage_scope);
1061 // Create the corresponding buffer to be written, i.e. result of cache_write
1062 info.write_buffer = write_buffer;
1063 // Create the corresponding buffer allocation
1064 info.alloc = info.read_buffer;
1065
1066 // Step 3. Check the only writer block.
1067 ICHECK_EQ(block_sref.get(), GetOnlyWriteBlock(self, scope_sref, write_buffer).get());
1068
1069 // Step 4. Find the producing region and insert position
1070 BufferRegion region = GetBufferRegionFromBuffer(block->writes, write_buffer).value();
1071 StmtSRef parent_sref = GetRef<StmtSRef>(block_sref->parent);
1072 // Detect insert position
1073 CacheLocDetector::Detect(self, block_sref, scope_sref, &info);
1074 BufferRegion cache_region =
1075 RelaxBufferRegion(self, region, block_sref, parent_sref, info.loc_sref);
1076
1077 // Step 5. Making new cache stage block and rewrite readers.
1078 Block cache_write_stage = MakeCacheStage(/*cache_region=*/cache_region, /*info=*/&info,
1079 /*storage_scope=*/storage_scope);
1080 Stmt new_scope = CacheWriteRewriter::Rewrite(/*scope_sref=*/scope_sref,
1081 /*writer_block_sref=*/block_sref, /*info=*/&info);
1082
1083 // Step 6. Replacing and updating flags.
1084 self->Replace(scope_sref, new_scope, info.block_reuse);
1085 StmtSRef result_block_sref = self->stmt2ref.at(cache_write_stage.get());
1086 BlockInfo& block_info = self->block_info[result_block_sref];
1087 block_info.affine_binding = CalculateAffineFlag(self, result_block_sref);
1088 block_info.region_cover = true;
1089 block_info.scope->stage_pipeline = true;
1090 return result_block_sref;
1091}

1093StmtSRef ReIndex(ScheduleState self, const StmtSRef& block_sref, int buffer_index,
1094 BufferIndexType buffer_index_type) {
1095 const BlockNode* block_ptr = TVM_SREF_TO_BLOCK(block_ptr, block_sref);
1096 Block block = GetRef<Block>(block_ptr);
1097 Buffer buffer = GetNthAccessBuffer(self, block, buffer_index, buffer_index_type);
1098 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/true);
1099 arith::Analyzer analyzer;
1100
1101 // Step 1. Collect the original indices and check there's only single pattern of related
1102 // Load/Store and the buffer is not accessed opaquely
1103 Array<PrimExpr> original_indices = ReIndexCollector::Collect(self->mod, buffer, block);
1104 // Simplify the indices if possible
1105 for (const IterVar& iter : block->iter_vars) {
1106 analyzer.Bind(iter->var, iter->dom);
1107 }
1108 original_indices.MutateByApply(
1109 [&analyzer](const PrimExpr& expr) { return analyzer.Simplify(expr); });
1110
1111 // Collect block iters appearing in the original_indices
1112 std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual> covered;
1113 for (const PrimExpr& index : original_indices) {
1114 PreOrderVisit(index, [&](const ObjectRef& obj) -> bool {
1115 if (const VarNode* var = obj.as<VarNode>()) {
1116 covered.insert(GetRef<Var>(var));
1117 }
1118 return true;
1119 });
1120 }
1121
1122 // Step 2. Creating CacheStageInfo
1123 CacheStageInfo info;
1124 // Create the corresponding buffer to be read(write), i.e. the result of reindex read(write)
1125 if (buffer_index_type == BufferIndexType::kWrite) {
1126 info.read_buffer = CreateReindexBuffer(buffer, block->iter_vars, covered);
1127 info.write_buffer = buffer;
1128 info.alloc = info.read_buffer;
1129 } else {
1130 info.read_buffer = buffer;
1131 info.write_buffer = CreateReindexBuffer(buffer, block->iter_vars, covered);
1132 info.alloc = info.write_buffer;
1133 }
1134
1135 // Step 3. Check the block belongs to a chain loop nesting under the scope,
1136 // and get the insert location
1137 const StmtSRefNode* loop;
1138 for (loop = block_sref->parent; loop->parent != scope_sref.get();) {
1139 const ForNode* outer = loop->parent->StmtAs<ForNode>();
1140 const ForNode* inner = loop->StmtAs<ForNode>();
1141 ICHECK(outer != nullptr && inner != nullptr);
1142 ICHECK(outer->body.get() == inner);
1143 loop = loop->parent;
1144 }
1145
1146 info.loc_pos = loop->seq_index == -1 ? 0 : loop->seq_index;
1147 if (buffer_index_type == BufferIndexType::kWrite) {
1148 info.loc_pos++;
1149 }
1150
1151 // Step 4. Making new reindex stage block and rewrite
1152 Block reindex_stage =
1153 MakeReIndexStage(block, &info, covered, original_indices, buffer_index, buffer_index_type);
1154 Stmt new_scope = ReIndexRewriter::Rewrite(scope_sref, block_sref, &info, covered);
1155
1156 // Step 5. Replacing and updating flags
1157 self->Replace(scope_sref, new_scope, info.block_reuse);
1158 StmtSRef result_block_sref = self->stmt2ref.at(reindex_stage.get());
1159 BlockInfo& block_info = self->block_info[result_block_sref];
1160 block_info.affine_binding = CalculateAffineFlag(self, result_block_sref);
1161 block_info.region_cover = true;
1162 block_info.scope->stage_pipeline = true;
1163 return result_block_sref;
1164}
