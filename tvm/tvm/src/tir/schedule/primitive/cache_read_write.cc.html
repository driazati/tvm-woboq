29class NotSingleWriteBlock : public ScheduleError {
30 public:
31 explicit NotSingleWriteBlock(IRModule mod, Buffer buffer, Array<StmtSRef> write_blocks)
32 : mod_(std::move(mod)), buffer_(std::move(buffer)) {
33 ICHECK_GT(write_blocks.size(), 1);
34 write_blocks_.reserve(write_blocks.size());
35 for (const StmtSRef& block_sref : write_blocks) {
36 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
37 write_blocks_.push_back(GetRef<Block>(block));
38 }
39 }
40
41 String FastErrorString() const final {
42 return "ScheduleError: The buffer is allowed to be written by single block.";
43 }
44
45 String DetailRenderTemplate() const final {
46 size_t k = write_blocks_.size();
47 return "The buffer " + buffer_->name + " is expected to be written by single block, but got " +
48 std::to_string(k) + " blocks who write it.";
49 }
50
51 IRModule mod() const final { return mod_; }
52 Array<ObjectRef> LocationsOfInterest() const final {
53 return {write_blocks_.begin(), write_blocks_.end()};
54 }
55
56 private:
57 IRModule mod_;
58 Buffer buffer_;
59 Array<Block> write_blocks_;
60};

64/*! \brief The auxiliary info used for the insertion point and content of the cache stage. */
65struct CacheStageInfo {
66 /*! \brief The buffer to be read. */
67 Buffer read_buffer;
68 /*! \brief The buffer to be written. */
69 Buffer write_buffer;
70 /*! \brief The buffer allocation to be inserted into the block signature. */
71 Optional<Buffer> alloc;
72 /*! \brief The AST node whose body is where the cache stage should be inserted. */
73 StmtSRef loc_sref;
74 /*! \brief The index to insert the cache_read/cache_write stage. */
75 size_t loc_pos;
76 /*! \brief The cache_read/cache_write stage to be inserted. */
77 Stmt cache_stage;
78 /*! \brief The map used for ScheduleStateNode::Replace. */
79 Map<Block, Block> block_reuse;
80 /*! \brief A set of blocks that will consume the new cache. */
81 std::unordered_set<StmtSRef, ObjectHash, ObjectEqual> consumer_blocks;
82};

84/*! \brief Return the buffer region realted with the buffer */
85Optional<BufferRegion> GetBufferRegionFromBuffer(const Array<BufferRegion>& buffer_regions,
86 const Buffer& buffer) {
87 Optional<BufferRegion> res = NullOpt;
88 for (const auto& region : buffer_regions) {
89 if (region->buffer.same_as(buffer)) {
90 ICHECK(!res.defined());

92 }

97/*!
98 * \brief Create a loop nest that represents cache copy (cache_read / cache_write) from read buffer
99 * to write buffer.
100 * \note This function will store the stmt with loop nesting to the CacheStageInfo, but only return
101 * the inside block.
102 * \param cache_region The cached copy region.
103 * \param info The cache stage information, which will be updated in the function.
104 * \param storage_scope The storage scope of the cached buffer (only used in naming here)
105 * \returns A block indicating the body of the loop nesting.
106 */
107Block MakeCacheStage(const BufferRegion& cache_region, CacheStageInfo* info,
108 const String& storage_scope) {
109 // loop variables
110 std::vector<Var> loop_vars;
111 // bindings in block realize
112 std::vector<PrimExpr> iter_values;
113 // Create loop vars and block vars' binding_value
114 for (const Range& axis_range : cache_region->region) {
115 Var loop_var("ax" + std::to_string(loop_vars.size()), axis_range->extent.dtype());
116 loop_vars.push_back(loop_var);
117 iter_values.push_back(axis_range->min + loop_var);
118 }
119 // block variables
120 Array<IterVar> block_vars;
121 // block access region for read/write buffers
122 Region access_region;
123 // indices used in block body
124 Array<PrimExpr> access_indices;
125 // Create block vars, block's accessed region and accessing indices
126 for (const PrimExpr& dim : cache_region->buffer->shape) {
127 Var var("v" + std::to_string(access_indices.size()), dim.dtype());
128 block_vars.push_back(IterVar(/*dom=*/Range::FromMinExtent(make_zero(dim->dtype), dim),
129 /*var=*/var,
130 /*IterVarType=*/kDataPar));
131 access_indices.push_back(var);
132 access_region.push_back(Range::FromMinExtent(var, make_const(var.dtype(), 1)));
133 }
134
135 // Create the body block:
136 // reads = [read_buffer[access_region]]
137 // writes = [write_buffer[access_region]]
138 // write_buffer[access_indices] = read_buffer[access_indices]
139 Block block(
140 /*iter_vars=*/std::move(block_vars),
141 /*reads=*/{BufferRegion(info->read_buffer, access_region)},
142 /*writes=*/{BufferRegion(info->write_buffer, access_region)},
143 /*name_hint=*/cache_region->buffer->name + "_" + storage_scope,
144 /*body=*/
145 BufferStore(info->write_buffer, BufferLoad(info->read_buffer, access_indices),
146 access_indices),
147 /*init=*/NullOpt,
148 /*alloc_buffers=*/{},
149 /*match_buffers=*/{},
150 /*annotations=*/{});
151 // Create the block realize node
152 Stmt body = BlockRealize(/*values=*/iter_values,
153 /*predicate=*/const_true(),
154 /*block=*/block);
155 // Create surrounding loops
156 for (size_t i = loop_vars.size(); i >= 1; --i) {
157 body = For(/*loop_var=*/loop_vars[i - 1],
158 /*min=*/0,
159 /*extent=*/cache_region->region[i - 1]->extent,
160 /*kind=*/ForKind::kSerial,
161 /*body=*/body);
162 }
163 info->cache_stage = std::move(body);
164 return block;
165}

167/*!
168 * \brief Create the reindex block and generate the corresponding outer loops.
169 * \details The reindex block is a data copy block between the reindex buffer (the intermediate
170 * buffer), and the target buffer.
171 If buffer_index_type == kWrite, copy from the reindex buffer to the target buffer.
172 If buffer_index_type == kRead, copy from the target buffer to the reindex buffer.
173 The reindex block has the same block iters and the surrounding loops as the input block.
174 However, if a block iter is not used in the indices of the target buffer being reindexed, the
175 domain of the block iter, and the corresponding outer loop, will become constant value one, making
176 it a trivial iter.
177 * \param block The block to be reindexed
178 * \param info The cache info
179 * \param covered The set of block iter vars covered in the buffer access indices
180 * \param original_indices The original buffer access indices
181 * \param buffer_index The index of the target buffer
182 * \param buffer_index_type The type of buffer index
183 * \return The reindex block.
184 */
185Block MakeReIndexStage(const Block& block, CacheStageInfo* info,
186 const std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual>& covered,
187 const Array<PrimExpr>& original_indices, int buffer_index,
188 BufferIndexType buffer_index_type) {
189 // iters of the reindex block
190 Array<IterVar> new_block_iters;
191 // the substition map from the original block iter to the iters of the reindex block
192 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectEqual> block_var_replace_map;
193 // indices to access the reindex buffer and the target buffer
194 Array<PrimExpr> reindex_indices, target_indices;
195
196 // Step 1: Create block iters, access regions of the reindex block, and accessing indices to the
197 // reindex buffer.
198 std::unordered_set<int> skipped_block_iters;
199 for (int i = 0, n = block->iter_vars.size(); i < n; ++i) {
200 const IterVar& iter = block->iter_vars[i];
201 Var var("v" + std::to_string(new_block_iters.size()), iter->var->dtype);
202 bool used = covered.count(iter->var);
203 if (used) {
204 new_block_iters.push_back(IterVar(/*dom=*/iter->dom,
205 /*var=*/var,
206 /*IterVarType=*/kDataPar));
207 } else {
208 skipped_block_iters.insert(i);
209 }
210 if (used) {
211 reindex_indices.push_back(var);
212 }
213 block_var_replace_map[iter->var] = var;
214 }
215
216 // Step 2: Replace the original block iters with the new block iters
217 for (const PrimExpr& index : original_indices) {
218 target_indices.push_back(Substitute(index, block_var_replace_map));
219 }
220
221 // Step 3: Create the reindex block
222
223 // The src and the dst region and indices of the data copy
224 Region src_region{nullptr};
225 Region dst_region{nullptr};
226 Array<PrimExpr> src_indices{nullptr};
227 Array<PrimExpr> dst_indices{nullptr};
228
229 if (buffer_index_type == BufferIndexType::kWrite) {
230 src_indices = reindex_indices;
231 dst_indices = target_indices;
232 } else {
233 src_indices = target_indices;
234 dst_indices = reindex_indices;
235 }
236
237 // Create the body block
238 Block new_block(
239 /*iter_vars=*/new_block_iters,
240 /*reads=*/{BufferRegion::FromPoint(info->read_buffer, src_indices)},
241 /*writes=*/{BufferRegion::FromPoint(info->write_buffer, dst_indices)},
242 /*name_hint=*/info->write_buffer->name + "_reindex",
243 /*body=*/
244 BufferStore(info->write_buffer, BufferLoad(info->read_buffer, src_indices), dst_indices));
245
246 // Step 4: Create surrounding loops
247
248 // Create loop vars and bindings for block iters
249 std::vector<Var> loop_vars; // loop variables
250 std::vector<PrimExpr> iter_values; // bindings in block realize
251 for (int i = 0; i < static_cast<int>(block->iter_vars.size()); ++i) {
252 if (skipped_block_iters.count(i)) {
253 continue;
254 }
255 Var loop_var("ax" + std::to_string(loop_vars.size()), block->iter_vars[i]->var->dtype);
256 loop_vars.push_back(loop_var);
257 iter_values.push_back(loop_var);
258 }
259
260 // Create the block realize node
261 Stmt body = BlockRealize(/*values=*/iter_values,
262 /*predicate=*/const_true(),
263 /*block=*/new_block);
264
265 // Create the chain of loops
266 for (int i = static_cast<int>(new_block_iters.size()) - 1; i >= 0; --i) {
267 body = For(/*loop_var=*/loop_vars[i],
268 /*min=*/new_block_iters[i]->dom->min,
269 /*extent=*/new_block_iters[i]->dom->extent,
270 /*kind=*/ForKind::kSerial,
271 /*body=*/std::move(body));
272 }
273 // Update cache info, which will be used in the later rewriting.
274 info->cache_stage = std::move(body);
275 return new_block;
276}

293/*!
294 * \brief Insert the cache_read/cache_write stage into the specific position
295 * \param stmt A sequence of statements or a single statement that the new stage is inserted in
296 * \param pos The position where the cache stage is inserted
297 * \param stage The stage to be inserted
298 * \return A SeqStmt, the result after insertion
299 */
300Stmt InsertCacheStage(const Stmt& stmt, int pos, const Stmt& stage) {
301 if (const auto* alloc = stmt.as<AllocateConstNode>()) {
302 auto seq_stmt = InsertCacheStage(alloc->body, pos, stage);
303 return AllocateConst(alloc->buffer_var, alloc->dtype, alloc->extents, alloc->data, seq_stmt,
304 alloc->annotations, alloc->span);
305 }
306 if (const auto* seq_stmt = stmt.as<SeqStmtNode>()) {
307 ObjectPtr<SeqStmtNode> result = make_object<SeqStmtNode>(*seq_stmt);
308 result->seq.insert(result->seq.begin() + pos, stage);
309 return SeqStmt(result);
310 }
311 if (pos == 0) {
312 return SeqStmt({stage, stmt});
313 }
314 ICHECK_EQ(pos, 1);
315 return SeqStmt({stmt, stage});
316}

318/*!
319 * \brief Get the only writer block of the input buffer in a given scope block.
320 * \param self The state of the schedule
321 * \param scope_sref The scope block where the write is considered
322 * \param buffer The queried buffer
323 * \return The sref of the only writer of the input buffer in the given scope,
324 * or `NullOpt` if no block writes it in the scope.
325 * \throw NotSingleWriteBlock if there are more than one intrested block.
326 */
327Optional<StmtSRef> GetOnlyWriteBlock(ScheduleState self, const StmtSRef& scope_sref,
328 const Buffer& buffer) {
329 BlockScope scope = self->GetBlockScope(scope_sref);
330 auto it = scope->buffer_writers.find(buffer);
331 if (it == scope->buffer_writers.end()) {
332 return NullOpt;
333 } else {
334 const Array<StmtSRef>& block_srefs = it->second;
335 ICHECK(!block_srefs.empty());
336 if (block_srefs.size() > 1) {
337 throw NotSingleWriteBlock(self->mod, buffer, block_srefs);
338 }
339 return block_srefs[0];
340 }
341}

343/*!
344 * \brief Get the buffer region under the sref tree path [dom_low_inclusive, dom_high_exclusive)
345 * \param self The state of the schedule.
346 * \param buffer_region The buffer region to be analyzed.
347 * \param block_sref The sref of the block related to the region.
348 * \param dom_low_inclusive The lowest node in the sref tree path.
349 * \param dom_high_exclusive The highest node in the sref tree path.
350 * \return The relaxed buffer region.
351 */
352BufferRegion RelaxBufferRegion(ScheduleState self, const BufferRegion& buffer_region,
353 const StmtSRef& block_sref, const StmtSRef& dom_low_inclusive,
354 const StmtSRef& dom_high_exclusive) {
355 BlockRealize realize = GetBlockRealize(self, block_sref);
356 Map<Var, PrimExpr> binding = GetBindings(realize);
357 const Buffer& buffer = buffer_region->buffer;
358 arith::Analyzer analyzer;
359 BufferRegion subst_region = BufferRegion(buffer, Substitute(buffer_region->region, binding));
360 Array<arith::IntSet> int_sets = AnalyzeRegionUpperBound(
361 /*region=*/subst_region,
362 /*predicate=*/realize->predicate,
363 /*dom_low_inclusive=*/dom_low_inclusive,
364 /*dom_high_exclusive=*/dom_high_exclusive,
365 /*analyzer=*/&analyzer);
366 ICHECK_EQ(buffer_region->region.size(), int_sets.size());
367
368 Region region;
369 region.reserve(int_sets.size());
370 for (size_t i = 0; i < int_sets.size(); ++i) {
371 region.push_back(int_sets[i].CoverRange(Range::FromMinExtent(0, buffer->shape[i])));
372 }
373 return BufferRegion(buffer, region);
374}

376/*! \brief Detect the insertion position of the new cache stage */
377class CacheLocDetector : public StmtVisitor {
378 public:
379 /*!
380 * \brief Detect the insertion position of the cache stage, and write the position into the
381 * CacheStageInfo \param self The state of the schedule \param block_sref The sref of the unique
382 * writer block of the buffer being applied cache_read or cache_write \param scope_sref The sref
383 * of the scope block of the cached block \param info The cache stage info.
384 */
385 template <bool is_cache_read>
386 static void Detect(const ScheduleState& self, const StmtSRef& block_sref,
387 const StmtSRef& scope_sref, CacheStageInfo* info) {
388 std::vector<StmtSRef> related_blocks;
389 // If consumer is specified, skip detecting the others
390 if (is_cache_read) {
391 if (info->consumer_blocks.size() > 0) {
392 for (StmtSRef consumer : info->consumer_blocks) {
393 related_blocks.emplace_back(consumer);
394 }
395 } else {
396 for (const Dependency& def : self->GetBlockScope(scope_sref)->GetDepsBySrc(block_sref)) {
397 if (def->kind == DepKind::kRAW) {
398 related_blocks.push_back(def->dst);
399 }
400 }
401 }
402 } else {
403 for (const Dependency& def : self->GetBlockScope(scope_sref)->GetDepsBySrc(block_sref)) {
404 if (def->kind == DepKind::kRAW) {
405 if (info->consumer_blocks.count(def->dst)) {
406 continue;
407 }
408 related_blocks.push_back(def->dst);
409 }
410 }
411 }
412
413 if (!related_blocks.empty()) {
414 CacheLocDetector detector(self, block_sref, scope_sref, related_blocks);
415 detector(GetRef<Stmt>(scope_sref->stmt));
416 info->loc_sref = detector.loc_sref_;
417 info->loc_pos = detector.loc_pos_;
418 } else {
419 info->loc_sref = scope_sref;
420 const auto* body = scope_sref->StmtAs<BlockNode>()->body.as<SeqStmtNode>();
421 info->loc_pos = body == nullptr ? 1 : body->size();
422 }
423 }
424
425 private:
426 /*!
427 * \brief Constructor
428 * \param self The state of the schedule
429 * \param block_sref The sref of the unique writer block of the buffer being applied cache_read or
430 * cache_write \param scope_sref The sref of the scope block of the cached block \param
431 * related_blocks Producer blocks for cache_write, or consumer blocks for cache_read
432 */
433 CacheLocDetector(const ScheduleState self, const StmtSRef& block_sref, const StmtSRef& scope_sref,
434 const std::vector<StmtSRef>& related_blocks)
435 : self_(self),
436 block_sref_(block_sref),
437 scope_sref_(scope_sref),
438 related_blocks_(related_blocks) {}
439
440 void VisitStmt_(const SeqStmtNode* seq_stmt) final {
441 bool previous_visited_block = visited_block_;
442 visited_block_ = false;
443
444 for (size_t i = 0; i < seq_stmt->size(); ++i) {
445 if (loc_pos_ != -1) {
446 break;
447 }
448 VisitStmt(seq_stmt->seq[i]);
449 // `pos` can be assigned only once when we visited `block_sref`
450 if (visited_block_ && visited_related_ && loc_pos_ == -1) {
451 // The offset of insert position from the block
452 loc_pos_ = i;
453 break;
454 } else if (visited_related_) {
455 // If meet the target consumer, stop searching
456 break;
457 }
458 }
459 visited_block_ = visited_block_ || previous_visited_block;
460 }
461
462 void VisitStmt_(const BlockNode* block) final {
463 // Only visit the current scope under buffer writer's parent block
464 if (block == scope_sref_->stmt) {
465 // The block vistied is the current parent scope
466 StmtVisitor::VisitStmt_(block);
467 // Handling cases when insert outside any loop or cache_read for input buffer
468 if (visited_related_ && !loc_sref_.defined()) {
469 loc_sref_ = self_->stmt2ref.at(block);
470 // Handling cache_read for input buffer
471 if (visited_block_ == false && loc_pos_ == -1) {
472 loc_pos_ = 0;
473 }
474 }
475 return;
476 }
477 // Update `visited_block`
478 if (block_sref_->stmt == block) {
479 visited_block_ = true;
480 return;
481 }
482 // Update `visited_related`
483 for (const StmtSRef& related_block : related_blocks_) {
484 if (related_block->stmt == block) {
485 visited_related_ = true;
486 return;
487 }
488 }
489 }
490
491 void VisitStmt_(const ForNode* loop) final {
492 StmtVisitor::VisitStmt_(loop);
493 if (visited_block_ && visited_related_ && !loc_sref_.defined() && loc_pos_ != -1) {
494 loc_sref_ = self_->stmt2ref.at(loop);
495 }
496 }
497
498 private:
499 /*! \brief The schedule class */
500 const ScheduleState self_;
501 /*! \brief The dominate block which write the buffer */
502 const StmtSRef& block_sref_;
503 /*! \brief The parent scope of the dominate block */
504 const StmtSRef& scope_sref_;
505 /*! \brief Producer blocks for cache_write and consumer blocks for cache_read */
506 const std::vector<StmtSRef>& related_blocks_;
507 /*! \brief The flag whether we have visited the dominate block */
508 bool visited_block_{false};
509 /*! \brief The flag whether we have visited at least one related blocks */
510 bool visited_related_{false};
511 /*! \brief The AST node whose body is where the cache stage should be inserted */
512 StmtSRef loc_sref_{nullptr};
513 /*! \brief The index to insert the cache_read/cache_write stage */
514 int loc_pos_{-1};
515};

517/*! \brief Detect the insertion position of the new cache stage */
518class CacheInplaceLocDetector : public StmtVisitor {
519 public:
520 /*!
521 * \brief Detect the insertion position of the cache stage, and write the position into the
522 * CacheStageInfo \param self The state of the schedule \param block_sref The sref of the unique

528 CacheInplaceLocDetector detector(self, block_sref, scope_sref);

535 /*!
536 * \brief Constructor
537 * \param self The state of the schedule
538 * \param block_sref The sref of the unique writer block of the buffer being applied cache_inplace
539 * \param scope_sref The sref of the scope block of the cached block
540 */
541 CacheInplaceLocDetector(const ScheduleState self, const StmtSRef& block_sref,
542 const StmtSRef& scope_sref)
543 : self_(self), block_sref_(block_sref), scope_sref_(scope_sref) {}

557 }

561 // Only visit the current scope under buffer writer's parent block

574 visited_block_ = true;
575 }

579 StmtVisitor::VisitStmt_(loop);
580 if (visited_block_ && !loc_sref_.defined()) {

584 }
585 }
586 }

603/*! \brief Mutator for CacheRead. */
604class CacheReadRewriter : public StmtExprMutator {
605 public:
606 /*!
607 * \brief Rewrite the AST and add a cache_read stage with the information provided
608 * \param scope_sref The parent scope of this mutation
609 * \param info The cache stage information
610 * \return The new AST rooting at the original parent scope
611 */
612 static Stmt Rewrite(const StmtSRef& scope_sref, CacheStageInfo* info) {
613 CacheReadRewriter rewriter(scope_sref, info);
614 return rewriter(GetRef<Stmt>(scope_sref->stmt));
615 }
616
617 private:
618 explicit CacheReadRewriter(const StmtSRef& scope_sref, CacheStageInfo* info)
619 : scope_sref_(scope_sref), info_(info) {}
620
621 Stmt VisitStmt_(const ForNode* loop) final {
622 Stmt stmt = StmtMutator::VisitStmt_(loop);
623 // Check the insertion point
624 if (loop == info_->loc_sref->stmt) {
625 // Insert cache stage into the loop if it is the right place
626 ObjectPtr<ForNode> n = make_object<ForNode>(*stmt.as<ForNode>());
627 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
628 stmt = Stmt(n);
629 }
630 return stmt;
631 }
632
633 Stmt VisitStmt_(const BlockNode* block) final {
634 Block old_stmt = GetRef<Block>(block);
635 // Check if this block is one of the specified consumers.
636 // If no consumer blocks are specified, all blocks should be considered consumers.
637 bool is_consumer = info_->consumer_blocks.empty();
638 // Otherwise check if this is one of the specified blocks.
639 for (StmtSRef consumer_sref : info_->consumer_blocks) {
640 const BlockNode* consumer_node = TVM_SREF_TO_BLOCK(consumer_sref);
641 Block consumer_block = GetRef<Block>(consumer_node);
642 if (old_stmt.same_as(consumer_block)) {
643 is_consumer = true;
644 }
645 }
646 // Keep track of this blocks status. We'll use this when rewriting loads.
647 current_block_consumes = is_consumer;
648 // We don't mutate the block which generates info->read_buffer.
649 if (block != scope_sref_->stmt &&
650 GetBufferRegionFromBuffer(block->writes, info_->read_buffer).defined()) {
651 return std::move(old_stmt);
652 }
653 // Mutate the body
654 Block stmt = Downcast<Block>(StmtMutator::VisitStmt_(block));
655 // Check the insertion point
656 if (block == info_->loc_sref->stmt) {
657 // Insert cache stage into the block if it is the right place
658 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
659 n->body = InsertCacheStage(n->body, info_->loc_pos, info_->cache_stage);
660 stmt = Block(n);
661 }
662 // Check if it is the block corresponding to the parent scope
663 if (block == scope_sref_->stmt) {
664 // If so, put buffer allocation on the parent scope
665 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
666 // In cache_inplace case, alloc_buffer may be already exits.
667 if (info_->alloc.defined()) {
668 n->alloc_buffers.push_back(info_->alloc.value());
669 stmt = Block(n);
670 }
671 } else {
672 // Otherwise, update read regions and match_buffers
673 // Only make this change if the block is one of the specified consumers.
674 if (is_consumer) {
675 Array<BufferRegion> reads =
676 ReplaceBuffer(block->reads, info_->read_buffer, info_->write_buffer);
677 Array<MatchBufferRegion> match_buffers =
678 ReplaceBuffer(block->match_buffers, info_->read_buffer, info_->write_buffer);
679 if (!reads.same_as(block->reads) || !match_buffers.same_as(block->match_buffers)) {
680 ObjectPtr<BlockNode> n = make_object<BlockNode>(*stmt.as<BlockNode>());
681 n->reads = std::move(reads);
682 n->match_buffers = std::move(match_buffers);
683 stmt = Block(n);
684 }
685 }
686 }
687 info_->block_reuse.Set(old_stmt, stmt);
688 return std::move(stmt);
689 }
690
691 PrimExpr VisitExpr_(const BufferLoadNode* load) final {
692 if (load->buffer.same_as(info_->read_buffer) && current_block_consumes) {
693 ObjectPtr<BufferLoadNode> n = make_object<BufferLoadNode>(*load);
694 n->buffer = info_->write_buffer;
695 return PrimExpr(n);
696 }
697 return ExprMutator::VisitExpr_(load);
698 }
699
700 PrimExpr VisitExpr_(const LoadNode* op) final {
701 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
702 }
703
704 PrimExpr VisitExpr_(const VarNode* op) final {
705 if (op == info_->read_buffer->data.get()) {
706 return info_->write_buffer->data;
707 }
708 return GetRef<PrimExpr>(op);
709 }
710
711 private:
712 /*! \brief The parent scope of the insertion */
713 const StmtSRef& scope_sref_;
714 /*! \brief The info for inserting cache stage */
715 CacheStageInfo* info_;
716 /*! \brief Whether the most recently visited block is a specified consumer. */
717 bool current_block_consumes;
718};

1147
1148StmtSRef CacheRead(ScheduleState self, const StmtSRef& block_sref, int read_buffer_index,
1149 const String& storage_scope, const Array<StmtSRef> consumer_blocks) {
1150 /*!
1151 * Check:
1152 * - The index is in the array of block reading region
1153 * - There is at most one block who write the buffer in the scope
1154 *
1155 * Mutate:
1156 * - Allocate new cache buffer under the current scope.
1157 * - Find the lowest ancestor of the block and ANY ONE of the consumers blocks.
1158 * - Copy the buffer with the consumed region.
1159 */
1160
1161 // Step 0. Check the input storage scope.
1162 CheckStorageScope(self, storage_scope);
1163
1164 // Step 1. Check index, getting the target buffer and the parent scope
1165 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
1166 Buffer read_buffer =
1167 GetNthAccessBuffer(self, GetRef<Block>(block), read_buffer_index, BufferIndexType::kRead);
1168 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
1169 // Check required region cover for cache_read
1170 CheckRegionCover(self, scope_sref, read_buffer);
1171 const BlockNode* scope_block = TVM_SREF_TO_BLOCK(scope_sref);
1172
1173 // Step 2. Create CacheStageInfo
1174 CacheStageInfo info;
1175 info.read_buffer = read_buffer;
1176 // Create the corresponding buffer to be written, i.e. result of cache_read
1177 info.write_buffer = WithScope(read_buffer, storage_scope);
1178 // Create the corresponding buffer allocation
1179 info.alloc = info.write_buffer;
1180
1181 // info.consumer_blocks indicates which buffers should consume the cache.
1182 for (auto consumer : consumer_blocks) {
1183 info.consumer_blocks.insert(consumer);
1184 for (auto child : tir::GetChildBlocks(self, consumer)) {
1185 info.consumer_blocks.insert(child);
1186 }
1187 }
1188
1189 // Step 3. Update cache stage info.
1190 BufferRegion cache_region{nullptr};
1191 if (Optional<StmtSRef> _write_block_sref = GetOnlyWriteBlock(self, scope_sref, read_buffer)) {
1192 // Case 1. The buffer is written inside the block.
1193 StmtSRef write_block_sref = _write_block_sref.value();
1194 const BlockNode* write_block = TVM_SREF_TO_BLOCK(write_block_sref);
1195 // Find the producing region
1196 BufferRegion region = GetBufferRegionFromBuffer(write_block->writes, read_buffer).value();
1197 StmtSRef parent_sref = GetRef<StmtSRef>(write_block_sref->parent);
1198
1199 // Detect insert position
1200 CacheLocDetector::Detect</*is_cache_read=*/true>(self, write_block_sref, scope_sref, &info);
1201 cache_region = RelaxBufferRegion(self, region, write_block_sref, parent_sref, info.loc_sref);
1202 } else {
1203 // Case 2. The buffer is the input block for the scope.
1204 info.loc_sref = scope_sref;
1205 info.loc_pos = 0;
1206 if (Optional<BufferRegion> region =
1207 GetBufferRegionFromBuffer(scope_block->reads, read_buffer)) {
1208 cache_region = region.value();
1209 } else {
1210 cache_region = BufferRegion::FullRegion(read_buffer);
1211 }
1212 }
1213
1214 // Step 4. Making new cache stage block and rewrite readers.
1215 Block cache_read_stage = MakeCacheStage(/*cache_region=*/cache_region, /*info=*/&info,
1216 /*storage_scope=*/storage_scope);
1217 Stmt new_scope = CacheReadRewriter::Rewrite(/*scope_sref=*/scope_sref, /*info=*/&info);
1218
1219 // Step 5. Replacing and updating flags.
1220 self->Replace(scope_sref, new_scope, info.block_reuse);
1221 StmtSRef result_block_sref = self->stmt2ref.at(cache_read_stage.get());
1222 BlockInfo& block_info = self->block_info[result_block_sref];
1223 block_info.affine_binding = CalculateAffineFlag(self, result_block_sref);
1224 block_info.region_cover = true;
1225 block_info.scope->stage_pipeline = true;
1226 return result_block_sref;

1229StmtSRef CacheWrite(ScheduleState self, const StmtSRef& block_sref, int write_buffer_index,
1230 const String& storage_scope, const Array<StmtSRef> consumer_blocks) {
1231 /*!
1232 * Check:
1233 * - The index is in the array of block reading region
1234 * - There is only one block who write the buffer in the scope
1235 *
1236 * Mutate:
1237 * - Allocate new cache buffer under the current scope.
1238 * - Find the lowest ancestor of the block and ANY ONE of the producer blocks.
1239 * - Copy the buffer with the consumed region.
1240 */
1241
1242 // Step 0. Check the input storage scope.
1243 CheckStorageScope(self, storage_scope);
1244
1245 // Step 1. Checking index, getting the target buffer and the parent scope
1246 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
1247 Buffer write_buffer =
1248 GetNthAccessBuffer(self, GetRef<Block>(block), write_buffer_index, BufferIndexType::kWrite);
1249 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
1250
1251 // Step 2. Creating CacheStageInfo
1252 CacheStageInfo info;
1253 info.read_buffer = WithScope(write_buffer, storage_scope);
1254 // Create the corresponding buffer to be written, i.e. result of cache_write
1255 info.write_buffer = write_buffer;
1256 // Create the corresponding buffer allocation
1257 info.alloc = info.read_buffer;
1258
1259 // info.consumer_blocks indicates which buffers should consume the cache.
1260 for (auto consumer : consumer_blocks) {
1261 info.consumer_blocks.insert(consumer);
1262 for (auto child : tir::GetChildBlocks(self, consumer)) {
1263 info.consumer_blocks.insert(child);
1264 }
1265 }
1266
1267 // Step 3. Check the only writer block.
1268 ICHECK_EQ(block_sref.get(), GetOnlyWriteBlock(self, scope_sref, write_buffer).get());
1269
1270 // Step 4. Find the producing region and insert position
1271 BufferRegion region = GetBufferRegionFromBuffer(block->writes, write_buffer).value();
1272 StmtSRef parent_sref = GetRef<StmtSRef>(block_sref->parent);
1273 // Detect insert position
1274 CacheLocDetector::Detect</*is_cache_read=*/false>(self, block_sref, scope_sref, &info);
1275 BufferRegion cache_region =
1276 RelaxBufferRegion(self, region, block_sref, parent_sref, info.loc_sref);
1277
1278 // Step 5. Making new cache stage block and rewrite readers.
1279 Block cache_write_stage = MakeCacheStage(/*cache_region=*/cache_region, /*info=*/&info,
1280 /*storage_scope=*/storage_scope);
1281 Stmt new_scope = CacheWriteRewriter::Rewrite(/*scope_sref=*/scope_sref,
1282 /*writer_block_sref=*/block_sref, /*info=*/&info);
1283
1284 // Step 6. Replacing and updating flags.
1285 self->Replace(scope_sref, new_scope, info.block_reuse);
1286 StmtSRef result_block_sref = self->stmt2ref.at(cache_write_stage.get());
1287 BlockInfo& block_info = self->block_info[result_block_sref];
1288 block_info.affine_binding = CalculateAffineFlag(self, result_block_sref);
1289 block_info.region_cover = true;
1290 block_info.scope->stage_pipeline = true;
1291 return result_block_sref;
1292}

1314Array<StmtSRef> CacheInplace(ScheduleState self, const StmtSRef& block_sref, int read_buffer_index,
1315 const String& storage_scope) {
1316 /*!
1317 * Do cache read then cache write
1318 */
1319
1320 // Check 0. Check the input storage scope.
1321 CheckStorageScope(self, storage_scope);
1322
1323 // Check 1. Check index, get the target buffer and the parent scope
1324 const BlockNode* block = TVM_SREF_TO_BLOCK(block_sref);
1325 Buffer buffer =
1326 GetNthAccessBuffer(self, GetRef<Block>(block), read_buffer_index, BufferIndexType::kRead);
1327 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
1328
1329 // Check 3. Check required region cover for cache_read
1330 CheckRegionCover(self, scope_sref, buffer);
1331
1332 // Check 4. Check if target block both read & write target buffer.
1333 const BlockNode* rw_block = TVM_SREF_TO_BLOCK(block_sref);
1334 Optional<BufferRegion> read_region = GetBufferRegionFromBuffer(rw_block->reads, buffer);
1335 Optional<BufferRegion> write_region = GetBufferRegionFromBuffer(rw_block->writes, buffer);
1336 if (!read_region.defined() || !write_region.defined()) {
1337 throw NotReadWriteError(self->mod, GetRef<Block>(rw_block), buffer);
1338 }
1339
1340 Array<StmtSRef> results_block_sref;
1341 Buffer new_buffer = WithScope(buffer, storage_scope);
1342
1343 // Do cache read
1344 // Cache read step 0. Create CacheStageInfo
1345 CacheStageInfo info;
1346 info.read_buffer = buffer;
1347 // Create the corresponding buffer to be written for cache_read
1348 info.write_buffer = new_buffer;
1349 // Create the corresponding buffer allocation
1350 info.alloc = info.write_buffer;
1351 // Indicate which buffers should consume the cache.
1352 info.consumer_blocks.insert(block_sref);
1353
1354 // Cache read step 1. Detect insert position
1355 CacheInplaceLocDetector::Detect(self, block_sref, scope_sref, &info);
1356
1357 // Cache read step 2. Making new cache stage block and rewrite readers.
1358 Block cache_read_stage = MakeCacheStage(/*cache_region=*/read_region.value(), /*info=*/&info,
1359 /*storage_scope=*/storage_scope);
1360 Stmt new_scope = CacheReadRewriter::Rewrite(/*scope_sref=*/scope_sref, /*info=*/&info);
1361
1362 // Cache read step 3. Replacing and updating flags for cache read.
1363 self->Replace(scope_sref, new_scope, info.block_reuse);
1364 StmtSRef result_block_sref = self->stmt2ref.at(cache_read_stage.get());
1365 BlockInfo& block_info_read = self->block_info[result_block_sref];
1366 block_info_read.affine_binding = CalculateAffineFlag(self, result_block_sref);
1367 block_info_read.region_cover = true;
1368 block_info_read.scope->stage_pipeline = false;
1369 results_block_sref.push_back(result_block_sref);
1370
1371 // Do cache write
1372 // Cache write step 0. Update cache stage info for cache_read.
1373 info.read_buffer = new_buffer;
1374 // Create the corresponding buffer to be written, i.e. result of cache_write
1375 info.write_buffer = buffer;
1376 // Create the corresponding buffer allocation
1377 info.alloc = nullptr;
1378 info.consumer_blocks.clear();
1379
1380 // Cache write step 1. Detect insert position
1381 CacheInplaceLocDetector::Detect(self, block_sref, scope_sref, &info);
1382 // insert after target block for cache write
1383 info.loc_pos += 1;
1384
1385 // Cache write step 2. Making new cache stage block and rewrite readers.
1386 Block cache_write_stage = MakeCacheStage(/*cache_region=*/write_region.value(), /*info=*/&info,
1387 /*storage_scope=*/storage_scope);
1388 new_scope = CacheWriteRewriter::Rewrite(/*scope_sref=*/scope_sref,
1389 /*writer_block_sref=*/block_sref, /*info=*/&info);
1390
1391 // Cache write step 4. Replacing and updating flags for cache write.
1392 self->Replace(scope_sref, new_scope, info.block_reuse);
1393 result_block_sref = self->stmt2ref.at(cache_write_stage.get());
1394 BlockInfo& block_info_write = self->block_info[result_block_sref];
1395 block_info_write.affine_binding = CalculateAffineFlag(self, result_block_sref);
1396 block_info_write.region_cover = true;
1397 block_info_write.scope->stage_pipeline = false;
1398 results_block_sref.push_back(result_block_sref);
1399
1400 return results_block_sref;
1401}

1403StmtSRef ReIndex(ScheduleState self, const StmtSRef& block_sref, int buffer_index,
1404 BufferIndexType buffer_index_type) {
1405 const BlockNode* block_ptr = TVM_SREF_TO_BLOCK(block_sref);
1406 Block block = GetRef<Block>(block_ptr);
1407 Buffer buffer = GetNthAccessBuffer(self, block, buffer_index, buffer_index_type);
1408 StmtSRef scope_sref = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/true);
1409 arith::Analyzer analyzer;
1410
1411 // Step 1. Collect the original indices and check there's only single pattern of related
1412 // Load/Store and the buffer is not accessed opaquely
1413 Array<PrimExpr> original_indices = ReIndexCollector::Collect(self->mod, buffer, block);
1414 // Simplify the indices if possible
1415 for (const IterVar& iter : block->iter_vars) {
1416 analyzer.Bind(iter->var, iter->dom);
1417 }
1418 original_indices.MutateByApply(
1419 [&analyzer](const PrimExpr& expr) { return SimplifyNonTrivialExpr(expr, &analyzer); });
1420
1421 // Collect block iters appearing in the original_indices
1422 std::unordered_set<Var, ObjectPtrHash, ObjectPtrEqual> covered;
1423 for (const PrimExpr& index : original_indices) {
1424 PreOrderVisit(index, [&](const ObjectRef& obj) -> bool {
1425 if (const VarNode* var = obj.as<VarNode>()) {
1426 covered.insert(GetRef<Var>(var));
1427 }
1428 return true;
1429 });
1430 }
1431
1432 // Step 2. Creating CacheStageInfo
1433 CacheStageInfo info;
1434 // Create the corresponding buffer to be read(write), i.e. the result of reindex read(write)
1435 if (buffer_index_type == BufferIndexType::kWrite) {
1436 info.read_buffer = CreateReindexBuffer(buffer, block->iter_vars, covered);
1437 info.write_buffer = buffer;
1438 info.alloc = info.read_buffer;
1439 } else {
1440 info.read_buffer = buffer;
1441 info.write_buffer = CreateReindexBuffer(buffer, block->iter_vars, covered);
1442 info.alloc = info.write_buffer;
1443 }
1444
1445 // Step 3. Check the block belongs to a chain loop nesting under the scope,
1446 // and get the insert location
1447 const StmtSRefNode* loop;
1448 for (loop = block_sref->parent; loop->parent != scope_sref.get();) {
1449 const ForNode* outer = loop->parent->StmtAs<ForNode>();
1450 const ForNode* inner = loop->StmtAs<ForNode>();
1451 ICHECK(outer != nullptr && inner != nullptr);
1452 ICHECK(outer->body.get() == inner);
1453 loop = loop->parent;
1454 }
1455
1456 info.loc_pos = loop->seq_index == -1 ? 0 : loop->seq_index;
1457 if (buffer_index_type == BufferIndexType::kWrite) {
1458 info.loc_pos++;
1459 }
1460
1461 // Step 4. Making new reindex stage block and rewrite
1462 Block reindex_stage =
1463 MakeReIndexStage(block, &info, covered, original_indices, buffer_index, buffer_index_type);
1464 Stmt new_scope = ReIndexRewriter::Rewrite(scope_sref, block_sref, &info, covered);
1465
1466 // Step 5. Replacing and updating flags
1467 self->Replace(scope_sref, new_scope, info.block_reuse);
1468 StmtSRef result_block_sref = self->stmt2ref.at(reindex_stage.get());
1469 BlockInfo& block_info = self->block_info[result_block_sref];
1470 block_info.affine_binding = CalculateAffineFlag(self, result_block_sref);
1471 block_info.region_cover = true;
1472 block_info.scope->stage_pipeline = true;
1473 return result_block_sref;
1474}
