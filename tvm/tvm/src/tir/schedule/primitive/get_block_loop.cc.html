24Array<StmtSRef> GetBlocks(const ScheduleState& self, const String& name, const GlobalVar& gv) {
25 struct Finder : public StmtVisitor {
26 explicit Finder(const ScheduleState& self, const String& name) : self_(self), name_(name) {}
27
28 void VisitStmt_(const BlockNode* block) override {
29 if (block->name_hint == name_) {
30 auto it = self_->stmt2ref.find(block);
31 ICHECK(it != self_->stmt2ref.end());
32 results_.push_back(it->second);
33 }
34 StmtVisitor::VisitStmt_(block);
35 }
36
37 const ScheduleState& self_;
38 const String& name_;
39 Array<StmtSRef> results_;
40 };
41
42 BaseFunc func = self->mod->Lookup(gv);
43 const auto* prim_func = TVM_TYPE_AS(func, PrimFuncNode);
44 Finder finder(self, name);
45 finder(prim_func->body);
46 return std::move(finder.results_);
47}

49Array<StmtSRef> GetLoops(const StmtSRef& block_sref) {
50 std::vector<StmtSRef> result;
51 for (StmtSRefNode* parent = block_sref->parent; parent && parent->stmt->IsInstance<ForNode>();
52 parent = parent->parent) {
53 result.push_back(GetRef<StmtSRef>(parent));
54 }
55 return {result.rbegin(), result.rend()};
56}

58Array<StmtSRef> GetChildBlocks(const ScheduleState& self, const StmtSRef& parent_sref) {
59 struct Collector : public StmtVisitor {
60 private:
61 void VisitStmt_(const BlockNode* block) final { result.push_back(self->stmt2ref.at(block)); }
62
63 public:
64 explicit Collector(const ScheduleState& self) : self(self) {}
65
66 const ScheduleState& self;
67 Array<StmtSRef> result;
68 };
69 Collector collector(self);
70 if (parent_sref->stmt->IsInstance<ForNode>()) {
71 const auto* loop = static_cast<const ForNode*>(parent_sref->stmt);
72 collector(loop->body);
73 } else if (parent_sref->stmt->IsInstance<BlockNode>()) {
74 const auto* block = static_cast<const BlockNode*>(parent_sref->stmt);
75 collector(block->body);
76 }
77 return std::move(collector.result);
78}

80Array<StmtSRef> GetProducers(const ScheduleState& self, const StmtSRef& block_sref) {
81 StmtSRef scope_root = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
82 Array<Dependency> edges = self->GetBlockScope(scope_root)->GetDepsByDst(block_sref);
83 Array<StmtSRef> results;
84 std::unordered_set<StmtSRef, ObjectPtrHash, ObjectPtrEqual> result_set;
85 results.reserve(edges.size());
86 for (const Dependency& edge : edges) {
87 if ((edge->kind == DepKind::kRAW || edge->kind == DepKind::kWAW) &&
88 !result_set.count(edge->src)) {
89 results.push_back(edge->src);
90 result_set.emplace(edge->src);
91 }
92 }
93 return results;
94}
95
96Array<StmtSRef> GetConsumers(const ScheduleState& self, const StmtSRef& block_sref) {
97 StmtSRef scope_root = GetScopeRoot(self, block_sref, /*require_stage_pipeline=*/false);
98 Array<Dependency> edges = self->GetBlockScope(scope_root)->GetDepsBySrc(block_sref);
99 Array<StmtSRef> results;
100 std::unordered_set<StmtSRef, ObjectPtrHash, ObjectPtrEqual> result_set;
101 results.reserve(edges.size());
102 for (const Dependency& edge : edges) {
103 if ((edge->kind == DepKind::kRAW || edge->kind == DepKind::kWAW) &&
104 !result_set.count(edge->dst)) {
105 results.push_back(edge->dst);
106 result_set.emplace(edge->dst);
107 }
108 }
109 return results;
110}

163struct GetChildBlocksTraits : public UnpackedInstTraits<GetChildBlocksTraits> {
164 static constexpr const char* kName = "GetChildBlocks";
165 static constexpr bool kIsPure = true;
166
167 private:
168 static constexpr size_t kNumInputs = 1;
169 static constexpr size_t kNumAttrs = 0;
170 static constexpr size_t kNumDecisions = 0;
171
172 static Array<BlockRV> UnpackedApplyToSchedule(Schedule sch, ObjectRef block_or_loop_rv) {
173 if (const auto* block = block_or_loop_rv.as<BlockRVNode>()) {
174 return sch->GetChildBlocks(GetRef<BlockRV>(block));
175 }
176 if (const auto* loop = block_or_loop_rv.as<LoopRVNode>()) {
177 return sch->GetChildBlocks(GetRef<LoopRV>(loop));
178 }
179 LOG(FATAL) << "TypeError: Expected Block or Loop, but gets: " << block_or_loop_rv->GetTypeKey();
180 throw;
181 }
182
183 static String UnpackedAsPython(Array<String> outputs, String block_or_loop_rv) {
184 PythonAPICall py("get_child_blocks");
185 py.Input("", block_or_loop_rv);
186 py.OutputList(outputs);
187 return py.Str();
188 }

191 friend struct ::tvm::tir::UnpackedInstTraits;
