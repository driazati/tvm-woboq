54bool HasBufferLoad(PrimExpr expr) {
55 struct Visitor : public ExprVisitor {
56 void VisitExpr_(const BufferLoadNode* node) override { found_buffer_load = true; }
57 bool found_buffer_load{false};
58 };
59
60 Visitor visitor;
61 visitor(expr);
62 return visitor.found_buffer_load;
63}

65Optional<PrimExpr> SubstituteParamValues(const Array<Var>& param_vars,
66 const Array<PrimExpr>& param_values,
67 const PrimExpr& expr) {
68 ICHECK_EQ(param_vars.size(), param_values.size())
69 << "Expression was defined as having " << param_vars.size() << " parameters, but received "
70 << param_values.size() << " arguments.";
71
72 Map<tir::Var, PrimExpr> var_map;
73 for (size_t i = 0; i < param_values.size(); i++) {
74 var_map.Set(param_vars[i], param_values[i]);
75 }
76
77 return Substitute(expr, var_map);
78}

81PrimExpr BufferTouch::BeforeLoopIteration() const {
82 PrimExpr loop_predicate = Bool(true);
83 for (auto it = loop_var_expressions.rbegin(); it != loop_var_expressions.rend(); it++) {
84 const Var& loop_var = it->first;
85 const PrimExpr& loop_expr = it->second;
86 loop_predicate = (loop_var <= loop_expr) || ((loop_var == loop_expr) && loop_predicate);
87 }
88 return loop_predicate;
89}
90
91PrimExpr BufferTouch::AtLoopIteration() const {
92 PrimExpr loop_predicate = Bool(true);
93 for (auto it = loop_var_expressions.rbegin(); it != loop_var_expressions.rend(); it++) {
94 const Var& loop_var = it->first;
95 const PrimExpr& loop_expr = it->second;
96 loop_predicate = (loop_var == loop_expr) && loop_predicate;
97 }
98 return loop_predicate;
99}

101PrimExpr BufferTouch::AfterLoopIteration() const {
102 PrimExpr loop_predicate = Bool(true);
103 for (auto it = loop_var_expressions.rbegin(); it != loop_var_expressions.rend(); it++) {
104 const Var& loop_var = it->first;
105 const PrimExpr& loop_expr = it->second;
106 loop_predicate = (loop_var >= loop_expr) || ((loop_var == loop_expr) && loop_predicate);
107 }
108 return loop_predicate;
109}

149class BufferConstraintApply : public IRMutatorWithAnalyzer {
150 public:
151 using Parent = IRMutatorWithAnalyzer;
152
153 BufferConstraintApply(const Map<Buffer, Array<Var>>& axis_var_lookup,
154 const std::vector<BufferTouch>& knowns, Analyzer* analyzer)
155 : Parent(analyzer), axis_var_lookup_(axis_var_lookup), knowns_(knowns) {}
156
157 using Parent::VisitExpr_;
158
159 PrimExpr VisitExpr_(const BufferLoadNode* op) override {
160 for (const auto& known : knowns_) {
161 if (!op->buffer.same_as(known.buffer)) {
162 continue;
163 }
164
165 Optional<Var> lane_var = NullOpt;
166 IntImm num_lanes;
167
168 Array<PrimExpr> indices = op->indices.Map([&](const auto& index) {
169 if (index.dtype().lanes() == 1) {
170 return index;
171 } else {
172 ICHECK(!lane_var) << "Multiple indices found with non-scalar values";
173 lane_var = Var("lane", index.dtype().element_of());
174 num_lanes = IntImm(index.dtype().element_of(), index.dtype().lanes());
175 return UnwrapVectorExpr(index, lane_var.value());
176 }
177 });
178
179 auto axis_vars = axis_var_lookup_.at(op->buffer);
180 PrimExpr predicate = SubstituteParamValues(axis_vars, indices, known.predicate).value();
181
182 std::optional<With<ConstraintContext>> context;
183 if (lane_var.defined()) {
184 Var lanes = lane_var.value();
185 PrimExpr known = (IntImm(lanes.dtype(), 0) <= lanes) && (lanes < num_lanes);
186 context.emplace(analyzer_, known);
187 }
188
189 if (analyzer_->CanProve(predicate)) {
190 return SubstituteParamValues(axis_vars, op->indices, known.value).value();
191 }
192 }
193
194 return GetRef<PrimExpr>(op);
195 }
196
197 private:
198 const Map<Buffer, Array<Var>>& axis_var_lookup_;
199 const std::vector<BufferTouch>& knowns_;
200};

221 void VisitStmt(const Stmt& stmt) override {
222 // Update the lookup table to determine which control-flow block
223 // contains the start of the specified statement. This is used
224 // later to determine which set of known values should be used to
225 // simplify a statement.
226 out_->control_flow_lookup_[stmt.get()] = CurrentControlBlock();
227 Stmt prev_stmt = current_stmt_;
228 current_stmt_ = stmt;
229 Parent::VisitStmt(stmt);
230 current_stmt_ = prev_stmt;
231 }

233 void VisitStmt_(const EvaluateNode* op) override {
234 if (auto* call = op->value.as<CallNode>()) {
235 if (call->op.same_as(builtin::assume())) {
236 Assume(call->args[0], true);
237 return;
238 }
239 }

241 Parent::VisitStmt_(op);

244 void Assume(PrimExpr assumption, bool from_assume_statement) {
245 for (const auto& expr : ExtractConstraints(assumption, false)) {
246 AssumeConstraintComponent(expr, from_assume_statement);
247 }

249
250 void AssumeConstraintComponent(PrimExpr assumption, bool from_assume_statement) {
251 PrimExpr additional_predicate = Bool(true);
252
253 std::vector<PrimExpr> buffer_exprs;
254 for (const auto& expr : ExtractComponents(assumption)) {
255 auto side_effect = tir::SideEffect(expr);
256 if (side_effect <= tir::CallEffectKind::kPure) {
257 // Pulling out portions of the assumption that do not depend
258 // on a buffer value allows the following two forms to be
259 // treated identically.
260 //
261 // Option 1: if i < 3: T.assume(buf[i] == value)
262 // Option 2: T.assume(i>=3 or buf[i] == value)
263 additional_predicate = additional_predicate && logical_not(expr);
264 } else if (side_effect == tir::CallEffectKind::kReadState) {
265 buffer_exprs.push_back(expr);
266 } else {
267 LOG(FATAL) << "Assumption must be pure or read-only, but contained expression " << expr
268 << " with side-effect \'" << side_effect << "\'";
269 }
270 }
271
272 if (buffer_exprs.empty()) {
273 out_->non_buffer_assumptions_.push_back(!CurrentScopePredicate() || assumption);
274 return;
275 }
276
277 CHECK_EQ(buffer_exprs.size(), 1) << "T.assume must contain only a single buffer expression";
278
279 auto* as_equal_node = buffer_exprs[0].as<tir::EQNode>();
280 CHECK(as_equal_node || !from_assume_statement)
281 << "T.assume buffer constraint must be of the form 'buffer[indices] == "
282 "value', but received "
283 << assumption;
284 if (!as_equal_node) {
285 // This assumption is an inequality on a data-dependent
286 // conditional. Not an error for this to occur, but also not
287 // something that is currently supported.
288 return;
289 }
290
291 tir::BufferLoad load;
292 PrimExpr value;
293 if (auto* as_load = as_equal_node->a.as<tir::BufferLoadNode>()) {
294 load = GetRef<tir::BufferLoad>(as_load);
295 value = as_equal_node->b;
296 } else if (auto* as_load = as_equal_node->b.as<tir::BufferLoadNode>()) {
297 load = GetRef<tir::BufferLoad>(as_load);
298 value = as_equal_node->a;
299 } else if (!from_assume_statement) {
300 return;
301 } else {
302 LOG(FATAL) << "T.assume buffer constraint must be of the form 'buffer[indices] == value'";
303 }
304
305 auto has_side_effect = tir::SideEffect(value) > tir::CallEffectKind::kPure;
306 CHECK(!has_side_effect || !from_assume_statement)
307 << "Buffer value in constraint must be pure expression, but was " << value;
308 if (has_side_effect) {
309 return;
310 }
311
312 {
313 InternalConstraintContext context(this, additional_predicate);
314 VisitAccess(load, BufferTouch::AccessType::Assume, value);
315 }
316 // Appending a control block ensures that all control blocks have
317 // at most one statement that changes the known buffer contents.
318 auto prev_block = CurrentControlBlock();
319 auto new_block = AppendControlBlock();
320 MarkControlFlow(prev_block, new_block);
321 }

340 Parent::VisitExpr_(op);

345 void VisitStmt_(const BufferStoreNode* op) override {
346 Parent::VisitStmt_(op);
347 VisitAccess(GetRef<BufferStore>(op), BufferTouch::AccessType::Write, op->value);
348 // Appending a control block ensures that all control blocks have
349 // at most one statement that changes the buffer contents.
350 auto prev_block = CurrentControlBlock();
351 auto new_block = AppendControlBlock();
352 MarkControlFlow(prev_block, new_block);
353 }
354
355 void VisitStmt_(const ForNode* op) override {
356 out_->iterator_ranges_.Set(op->loop_var, Range::FromMinExtent(op->min, op->extent));
357
358 auto before_loop = CurrentControlBlock();
359 size_t loop_start = -1;
360
361 {
362 BindActiveLoopVar binding(this, op->loop_var, op->min, op->extent);
363 loop_start = AppendControlBlock();
364 Parent::VisitStmt_(op);
365 }
366
367 auto loop_end = CurrentControlBlock();
368 auto after_loop = AppendControlBlock();
369 PrimExpr max_iterator_value = analyzer_.Simplify(op->min + op->extent - 1);
370 {
371 auto [forward, backward] = MarkControlFlow(before_loop, loop_start);
372 backward.post_condition = (op->loop_var == op->min);
373 forward.var_remap = {{op->loop_var, op->min}};
374 }
375 {
376 auto [forward, backward] = MarkControlFlow(loop_end, after_loop);
377 backward.var_remap = {{op->loop_var, max_iterator_value}};
378 forward.post_condition = (op->loop_var == max_iterator_value);
379 }
380 {
381 auto [forward, backward] = MarkControlFlow(loop_end, loop_start);
382 backward.var_remap = {{op->loop_var, op->loop_var - 1}};
383 forward.var_remap = {{op->loop_var, op->loop_var + 1}};
384 backward.post_condition = (op->loop_var > op->min);
385 forward.post_condition = (op->loop_var < max_iterator_value);
386 }
387 }
388
389 void VisitStmt_(const IfThenElseNode* op) override {
390 this->VisitExpr(op->condition);
391
392 PrimExpr real_condition = ExtractRealCondition(op->condition);
393
394 auto before_branching = CurrentControlBlock();
395
396 auto branch_start = AppendControlBlock();
397 MarkControlFlow(before_branching, branch_start);
398
399 {
400 InternalConstraintContext context(this, real_condition);
401 auto then_start = AppendControlBlock();
402 if (context.assume.defined()) {
403 Assume(context.assume.value(), false);
404 }
405 auto [forward, backward] = MarkControlFlow(branch_start, then_start);
406 backward.post_condition = real_condition;
407 forward.post_condition = real_condition;
408 this->VisitStmt(op->then_case);
409 }
410 auto then_end = CurrentControlBlock();
411
412 auto negation = analyzer_.rewrite_simplify(!real_condition);
413 {
414 InternalConstraintContext context(this, negation);
415 auto else_start = AppendControlBlock();
416 if (context.assume.defined()) {
417 Assume(context.assume.value(), false);
418 }
419 auto [forward, backward] = MarkControlFlow(branch_start, else_start);
420 backward.post_condition = negation;
421 forward.post_condition = negation;
422
423 if (op->else_case.defined()) {
424 this->VisitStmt(op->else_case.value());
425 }
426 }
427
428 auto else_end = CurrentControlBlock();
429 auto after_branching = AppendControlBlock();
430
431 if (HasBufferLoad(real_condition)) {
432 // The buffer value may have changed during the body of the
433 // condition, so we can't provide it as a post-condition.
434 MarkControlFlow(then_end, after_branching);
435 MarkControlFlow(else_end, after_branching);
436 } else {
437 {
438 auto [forward, backward] = MarkControlFlow(then_end, after_branching);
439 backward.post_condition = real_condition;
440 forward.post_condition = real_condition;
441 }
442 {
443 auto [forward, backward] = MarkControlFlow(else_end, after_branching);
444 backward.post_condition = negation;
445 forward.post_condition = negation;
446 }
447 }

459 /*! \brief Record the interaction with the buffer.
460 *
461 * \param node The TIR node that accesses the buffer. Should be
462 * either a BufferLoad or BufferStore node.
463 *
464 * \param touch_type The type of buffer access being performed. A
465 * BufferStore should always use AccessType::Write. A BufferLoad
466 * may use either AccessType::Read or AccessType::Assume, depending
467 * on whether the BufferLoad occurs within `builtin::assume`.
468 *
469 * \param known_value_expr The value in the buffer following the access.
470 */
471 template <typename BufferAccess>
472 void VisitAccess(const BufferAccess& node, BufferTouch::AccessType touch_type,
473 PrimExpr known_value_expr) {
474 auto& current_block = out_->control_flow_.back();
475 BufferTouch buffer_touch = current_block.MakeBufferTouch(out_, node->buffer, node->indices,
476 touch_type, known_value_expr);
477 current_block.touch_points.push_back(buffer_touch);
478 }

480 /*! \brief Return a predicate for having reached the current
481 * control-flow block
482 *
483 * For example, while inside an IfThenElse, will return the
484 * IfThenElse's condition.
485 */
486 PrimExpr CurrentScopePredicate() const {
487 PrimExpr predicate = Bool(true);
488 for (const auto& condition : conditions_) {
489 predicate = predicate && condition;
490 }
491 return predicate;
492 }

494 /* \brief Add a new control block, returning its index */
495 size_t AppendControlBlock() {
496 size_t index = out_->control_flow_.size();
497 auto& block = out_->control_flow_.emplace_back();
498 block.active_loop_iterators = active_loop_iterators_;
499 block.let_bindings_using_loop = let_bindings_using_loop_;
500 block.scope_predicate = CurrentScopePredicate();
501 return index;
502 }

504 /* \brief The index of the current control block */
505 size_t CurrentControlBlock() { return out_->control_flow_.size() - 1; }

507 /* \brief Mark a possible control from one block to another
508 *
509 * \param from_block The block from which control leaves
510 *
511 * \param to_block The block to which control enters
512 *
513 * \param var_remap Variable replacements that should be made in
514 * known expression while traversing this edge. For example,
515 * replacing `i` with `i-1` when entering the next loop iteration,
516 * or replacing `i` with `n-1` when concluding a loop.
517 */
518 std::pair<ControlFlowGraph::ControlFlowEdge&, ControlFlowGraph::ControlFlowEdge&> MarkControlFlow(
519 size_t from_block, size_t to_block) {
520 ICHECK_LE(from_block, out_->control_flow_.size());
521 ICHECK_LE(to_block, out_->control_flow_.size());
522
523 auto& forward = out_->control_flow_[from_block].successors.emplace_back(
524 ControlFlowGraph::ControlFlowEdge{to_block, {}, NullOpt});
525 auto& backward = out_->control_flow_[to_block].predecessors.emplace_back(
526 ControlFlowGraph::ControlFlowEdge{from_block, {}, NullOpt});
527 return {forward, backward};
528 }

530 // Internal utility, context manager for entering/leaving a scoped constraint
531 struct InternalConstraintContext {
532 InternalConstraintContext(ControlFlowGraphBuilder* self, PrimExpr constraint)
533 : self(self), analyzer_context(&self->analyzer_, constraint) {
534 old_num_constraints = self->conditions_.size();
535
536 auto side_effect = tir::SideEffect(constraint);
537 if (side_effect <= tir::CallEffectKind::kPure) {
538 self->conditions_.push_back(constraint);
539 } else if (side_effect <= tir::CallEffectKind::kReadState) {
540 assume = constraint;
541 }
542
543 new_num_constraints = self->conditions_.size();
544 }
545 ~InternalConstraintContext() {
546 ICHECK_EQ(self->conditions_.size(), new_num_constraints)
547 << "Internal error: Each condition should only be popped once.";
548 self->conditions_.erase(self->conditions_.begin() + old_num_constraints,
549 self->conditions_.end());
550 }
551
552 ControlFlowGraphBuilder* self{nullptr};
553 With<ConstraintContext> analyzer_context;
554 size_t old_num_constraints{0};
555 size_t new_num_constraints{0};
556 Optional<PrimExpr> assume{NullOpt};
557
558 // Disable default-generated copy/move assignment and constructors
559 InternalConstraintContext(const InternalConstraintContext&) = delete;
560 InternalConstraintContext& operator=(const InternalConstraintContext&) = delete;
561 InternalConstraintContext(InternalConstraintContext&&) = delete;
562 InternalConstraintContext& operator=(InternalConstraintContext&&) = delete;
563 };

637std::pair<BufferTouch, Map<Var, Range>> ControlFlowGraph::ControlFlowBlock::MakeBufferTouch(
638 const tir::Buffer& buf, Array<Var> index_variables, Array<PrimExpr> indices,
639 BufferTouch::AccessType touch_type, PrimExpr known_value_expr) const {
640 const auto& current_block = *this;
641
642 Analyzer local_analyzer;
643
644 Optional<Var> lane_var = NullOpt;
645 IntImm num_lanes;
646
647 Array<PrimExpr> index_expressions = indices.Map([&](const auto& index) {
648 if (index.dtype().lanes() == 1) {
649 return index;
650 } else {
651 ICHECK(!lane_var) << "Multiple indices found with non-scalar values";
652 lane_var = Var("lane", index.dtype().element_of());
653 num_lanes = IntImm(index.dtype().element_of(), index.dtype().lanes());
654 return UnwrapVectorExpr(index, lane_var.value());
655 }
656 });
657
658 Array<Var> loop_vars;
659
660 Map<Var, Range> loop_ranges;
661 for (const auto& loop_entry : current_block.active_loop_iterators) {
662 loop_vars.push_back(loop_entry.loop_var);
663 loop_ranges.Set(loop_entry.loop_var, loop_entry.loop_range);
664 }
665
666 // If the indices contain multiple lanes, treat the lane variable
667 // as an additional loop iterator to be solved for and substituted
668 // out.
669 if (lane_var) {
670 loop_vars.push_back(lane_var.value());
671 loop_ranges.Set(lane_var.value(), Range::FromMinExtent(0, num_lanes));
672 }
673
674 IntConstraintsTransform transform = [&]() {
675 ICHECK_EQ(index_variables.size(), index_expressions.size());
676
677 Array<PrimExpr> relations;
678
679 for (size_t i = 0; i < index_expressions.size(); i++) {
680 PrimExpr expr = index_expressions[i];
681 Var var = index_variables[i];
682
683 expr = Substitute(expr, current_block.let_bindings_using_loop);
684 relations.push_back(var == expr);
685 }
686
687 IntConstraints system(loop_vars, loop_ranges, relations);
688 return arith::SolveLinearEquations(system);
689 }();
690
691 Map<Var, PrimExpr> loop_var_to_axis_var = transform->src_to_dst;
692 Map<Var, Range> free_params = transform->dst->ranges;
693 PrimExpr transform_predicate =
694 std::accumulate(transform->dst->relations.begin(), transform->dst->relations.end(),
695 PrimExpr(Bool(true)), [](PrimExpr a, PrimExpr b) { return a && b; });
696
697 transform_predicate = SimplifyAsAndOfOrs(transform_predicate, &local_analyzer);
698
699 auto find_removable_params = [&]() -> Map<Var, PrimExpr> {
700 Map<Var, PrimExpr> removable_params;
701
702 // The arith::SolveLinearEquations is more general than the
703 // utilities in iter_affine_map.h, but can introduce free
704 // parameters that could later be determined with the known
705 // constraints. This step removes all such free parameters.
706 for (const auto& expr : ExtractConstraints(transform_predicate)) {
707 if (auto* as_equal = expr.as<EQNode>()) {
708 auto check_expr = [&](const PrimExpr& a, const PrimExpr& b) {
709 auto* var_ptr = a.as<VarNode>();
710 if (!var_ptr) {
711 return;
712 }
713
714 Var var = GetRef<Var>(var_ptr);
715 if (free_params.count(var) == 0) {
716 return;
717 }
718
719 bool uses_free_param =
720 UsesVar(b, [&](const VarNode* v) { return free_params.count(GetRef<Var>(v)) > 0; });
721 if (uses_free_param) {
722 return;
723 }
724 removable_params.Set(var, b);
725 };
726 check_expr(as_equal->a, as_equal->b);
727 check_expr(as_equal->b, as_equal->a);
728 }
729 }
730
731 // In addition, the arith::SolveLinearEquation can introduce
732 // free parameters with an extent of one. Filtering them out here
733 // avoids needing to track them through later simplifications.
734 for (const auto [var, range] : free_params) {
735 if (is_one(range->extent)) {
736 removable_params.Set(var, range->min);
737 }
738 }
739
740 return removable_params;
741 };
742 for (auto removable_params = find_removable_params(); removable_params.size() > 0;
743 removable_params = find_removable_params()) {
744 auto update = [&](const PrimExpr& expr) {
745 return local_analyzer.Simplify(Substitute(expr, removable_params));
746 };
747
748 Map<Var, PrimExpr> new_map;
749 for (const auto [loop_var, expr] : loop_var_to_axis_var) {
750 static_cast<void>(expr); // gcc 7.x bug, https://gcc.gnu.org/bugzilla/show_bug.cgi?id=81767
751 new_map.Set(loop_var, update(expr));
752 }
753 loop_var_to_axis_var = new_map;
754
755 transform_predicate = update(transform_predicate);
756
757 for (const auto [var, expr] : removable_params) {
758 static_cast<void>(expr); // gcc 7.x bug, https://gcc.gnu.org/bugzilla/show_bug.cgi?id=81767
759 free_params.erase(var);
760 }
761 }
762
763 // Normalization function, applied to both the predicate and the
764 // known value. Converts from an expression in terms of loop
765 // iterators to an expression in terms of buffer indices.
766 auto normalize_expr = [&](PrimExpr expr) -> PrimExpr {
767 expr = Substitute(expr, current_block.let_bindings_using_loop);
768
769 if (lane_var) {
770 expr = UnwrapVectorExpr(expr, lane_var.value());
771 }
772 expr = Substitute(expr, loop_var_to_axis_var);
773
774 return expr;
775 };
776
777 // Collect the current loop variables, along with an expression for
778 // the loop variables in terms of the buffer axis variables. This
779 // is used during forward/backward propagation to generate predicate
780 // tracking whether a loop iteration has been reached.
781 std::vector<std::pair<Var, PrimExpr>> loop_var_expressions;
782 for (const auto& entry : current_block.active_loop_iterators) {
783 auto expr_it = loop_var_to_axis_var.find(entry.loop_var);
784 ICHECK(expr_it != loop_var_to_axis_var.end());
785 loop_var_expressions.push_back({entry.loop_var, (*expr_it).second});
786 }
787
788 // The full predicate is composed of the values required to reach
789 // the scope of the BufferStore or builtin::assume(), any bounds
790 // implied by solving for the axis variables, and any additional
791 // statements resulting from unpacking the expression contained in
792 // builtin::assume().
793 PrimExpr scope_predicate = normalize_expr(current_block.scope_predicate);
794 transform_predicate = normalize_expr(transform_predicate);
795
796 known_value_expr = local_analyzer.Simplify(normalize_expr(known_value_expr));
797
798 // Deliberately use an analyzer without scope-based information,
799 // to avoid simplifying `scope_predicate` to True.
800 PrimExpr predicate_expr = local_analyzer.Simplify(transform_predicate && scope_predicate);
801
802 BufferTouch buffer_touch = {buf, predicate_expr, known_value_expr, loop_var_expressions,
803 touch_type};
804
805 return {buffer_touch, free_params};
806}

892bool BufferTouch::IsEquivalentTo(const BufferTouch& other, Analyzer* analyzer) const {
893 // Constraints must apply to the same buffer to be equivalent
894 if (!buffer.same_as(other.buffer) || touch_type != other.touch_type) {
895 return false;
896 }
897
898 ExprDeepEqual deep_equal;
899
900 auto implies = [&](const PrimExpr& a, const PrimExpr& b) -> bool {
901 With<ConstraintContext> context(analyzer, a);
902 return analyzer->CanProve(b);
903 };
904
905 // Predicates must be equivalent expressions, or must both be undefined
906 bool equivalent_predicates =
907 deep_equal(predicate, other.predicate) ||
908 (implies(predicate, other.predicate) && implies(other.predicate, predicate));
909 if (!equivalent_predicates) {
910 return false;
911 }
912
913 // The known value must be equal
914 if (!deep_equal(value, other.value) && !analyzer->CanProveEqual(value, other.value)) {
915 return false;
916 }
917
918 return true;
919}

977void BufferState::Intersection(const BufferState& b, Analyzer* analyzer) {
978 // For a constraint to be in the output, it must be present in both
979 // inputs.
980
981 std::vector<BufferTouch> new_constraints;
982 for (const auto& ai : constraints_) {
983 for (const auto& bi : b.constraints_) {
984 if (ai.buffer.same_as(bi.buffer)) {
985 PrimExpr predicate = SimplifyAsAndOfOrs(ai.predicate && bi.predicate, analyzer);
986 if (!is_zero(predicate)) {
987 With<ConstraintContext> context(analyzer, predicate);
988 PrimExpr known_value_a = ai.value;
989 PrimExpr known_value_b = bi.value;
990
991 bool is_consistent = analyzer->CanProveEqual(known_value_a, known_value_b);
992 if (is_consistent) {
993 new_constraints.push_back({ai.buffer, predicate, known_value_a});
994 }
995 }
996 }
997 }
998 }
999
1000 constraints_ = std::move(new_constraints);

1003class BufferRegionCollector : public ExprVisitor {
1004 public:
1005 struct Region {
1006 PrimExpr region_predicate;
1007 std::unordered_map<const BufferLoadNode*, Optional<PrimExpr>> known_values;
1008 };
1009
1010 static std::vector<Region> Collect(const Map<Buffer, Array<Var>>& axis_var_lookup,
1011 const std::vector<BufferTouch>& knowns,
1012 const std::vector<Optional<PrimExpr>>& exprs,
1013 Analyzer* analyzer) {
1014 BufferRegionCollector collector(axis_var_lookup, knowns, analyzer);
1015 for (const auto& expr : exprs) {
1016 if (expr) {
1017 collector(expr.value());
1018 }
1019 }
1020
1021 return collector.regions_;
1022 }
1023
1024 private:
1025 using Parent = ExprVisitor;
1026
1027 BufferRegionCollector(const Map<Buffer, Array<Var>>& axis_var_lookup,
1028 const std::vector<BufferTouch>& knowns, Analyzer* analyzer)
1029 : analyzer_(analyzer), axis_var_lookup_(axis_var_lookup), knowns_(knowns) {
1030 regions_.push_back(Region{Bool(true), {}});
1031 }
1032
1033 using Parent::VisitExpr_;
1034
1035 void VisitExpr_(const BufferLoadNode* op) override {
1036 // Helper struct for the known values of this BufferLoad
1037 struct Known {
1038 PrimExpr predicate;
1039 Optional<PrimExpr> value;
1040 };
1041
1042 std::vector<Known> new_regions;
1043
1044 PrimExpr unknown_region = Bool(true);
1045
1046 for (const BufferTouch& constraint : knowns_) {
1047 if (!op->buffer.same_as(constraint.buffer)) {
1048 // This is a different buffer, so continue searching.
1049 continue;
1050 }
1051
1052 auto axis_vars = axis_var_lookup_.at(op->buffer);
1053 PrimExpr touch_predicate =
1054 SubstituteParamValues(axis_vars, op->indices, constraint.predicate).value();
1055 touch_predicate = SimplifyAsAndOfOrs(touch_predicate, analyzer_);
1056
1057 if (!is_zero(touch_predicate)) {
1058 Optional<PrimExpr> known_value =
1059 SubstituteParamValues(axis_vars, op->indices, constraint.value);
1060 new_regions.push_back(Known{touch_predicate, known_value});
1061
1062 unknown_region = unknown_region && !touch_predicate;
1063 unknown_region = SimplifyAsAndOfOrs(unknown_region, analyzer_);
1064 }
1065 }
1066
1067 if (new_regions.size()) {
1068 Analyzer local_analyzer;
1069
1070 if (!is_zero(unknown_region)) {
1071 new_regions.insert(new_regions.begin(), Known{unknown_region, NullOpt});
1072 }
1073
1074 std::vector<Region> updated_regions;
1075 for (const auto& prev_region : regions_) {
1076 for (const auto& new_region : new_regions) {
1077 PrimExpr intersection =
1078 SimplifyAsAndOfOrs(prev_region.region_predicate && new_region.predicate, analyzer_);
1079
1080 if (!is_zero(intersection)) {
1081 Region merged{intersection, prev_region.known_values};
1082 merged.known_values[op] = new_region.value;
1083 updated_regions.push_back(std::move(merged));
1084 }
1085 }
1086 }
1087 regions_ = updated_regions;
1088 }
1089 }
1090
1091 Analyzer* analyzer_;
1092 std::vector<Region> regions_;
1093 const Map<Buffer, Array<Var>>& axis_var_lookup_;
1094 const std::vector<BufferTouch>& knowns_;
1095};

1133void BufferState::ApplyTouches(const Map<Buffer, Array<Var>>& axis_var_lookup,
1134 const std::vector<BufferTouch>& touch_points, Analyzer* analyzer) {
1135 std::vector<BufferTouch> new_knowns;
1136 Map<Buffer, PrimExpr> keep_prior_known_at;
1137
1138 for (auto& touch : touch_points) {
1139 if (touch.touch_type == BufferTouch::AccessType::Read) {
1140 continue;
1141 }
1142
1143 PrimExpr known_value = touch.value;
1144
1145 PrimExpr predicate = touch.predicate && touch.AfterLoopIteration();
1146 auto regions = BufferRegionCollector::Collect(axis_var_lookup, constraints_,
1147 {predicate, touch.value}, analyzer);
1148
1149 for (const auto& region : regions) {
1150 PrimExpr updated_predicate = BufferRegionValueReplacer::Apply(
1151 region.known_values, region.region_predicate && predicate, analyzer);
1152
1153 updated_predicate = SimplifyAsAndOfOrs(updated_predicate, analyzer);
1154 PrimExpr updated_value =
1155 BufferRegionValueReplacer::Apply(region.known_values, known_value, analyzer);
1156
1157 if (!is_zero(updated_predicate)) {
1158 if (auto it = keep_prior_known_at.find(touch.buffer); it != keep_prior_known_at.end()) {
1159 keep_prior_known_at.Set(touch.buffer, (*it).second && !updated_predicate);
1160 } else {
1161 keep_prior_known_at.Set(touch.buffer, !updated_predicate);
1162 }
1163
1164 if (!HasBufferLoad(updated_value)) {
1165 BufferTouch new_constraint{touch.buffer, updated_predicate, updated_value};
1166 new_knowns.push_back(new_constraint);
1167 }
1168 }
1169 }
1170 }
1171
1172 if (keep_prior_known_at.size()) {
1173 for (auto& constraint : constraints_) {
1174 if (auto it = keep_prior_known_at.find(constraint.buffer); it != keep_prior_known_at.end()) {
1175 constraint.predicate = SimplifyAsAndOfOrs(constraint.predicate && (*it).second, analyzer);
1176 }
1177 }
1178 }
1179
1180 if (new_knowns.size()) {
1181 std::vector<bool> used(new_knowns.size(), false);
1182
1183 for (auto& constraint : constraints_) {
1184 PrimExpr expand_known_at = Bool(false);
1185
1186 PrimExpr prev_value = constraint.value;
1187
1188 for (size_t i = 0; i < new_knowns.size(); i++) {
1189 if (new_knowns[i].buffer.same_as(constraint.buffer)) {
1190 Optional<PrimExpr> overwritten_with = new_knowns[i].value;
1191 if (overwritten_with && analyzer->CanProveEqual(prev_value, overwritten_with.value())) {
1192 expand_known_at =
1193 SimplifyAsAndOfOrs(expand_known_at || new_knowns[i].predicate, analyzer);
1194 used[i] = true;
1195 }
1196 }
1197 }
1198
1199 if (!is_zero(expand_known_at)) {
1200 constraint.predicate =
1201 SimplifyAsAndOfOrs(constraint.predicate || expand_known_at, analyzer);
1202 }
1203 }
1204
1205 for (size_t i = 0; i < new_knowns.size(); i++) {
1206 if (!used[i]) {
1207 constraints_.push_back(new_knowns[i]);
1208 }
1209 }
1210 }
1211
1212 constraints_.erase(
1213 std::remove_if(constraints_.begin(), constraints_.end(),
1214 [&](const auto& constraint) { return is_zero(constraint.predicate); }),
1215 constraints_.end());
1216}

1218void BufferState::BackpropUnusedIndices(const Map<Buffer, Array<Var>>& axis_var_lookup,
1219 const std::vector<BufferTouch>& touch_points,
1220 Analyzer* analyzer) {
1221 std::vector<BufferTouch> new_knowns;
1222 Map<Buffer, PrimExpr> keep_prior_known_at;
1223
1224 Map<Buffer, PrimExpr> regions_written;
1225 Map<Buffer, PrimExpr> regions_read;
1226 for (auto it = touch_points.rbegin(); it != touch_points.rend(); it++) {
1227 const auto& touch = *it;
1228
1229 Map<Buffer, PrimExpr>* to_update{nullptr};
1230 if (touch.touch_type == BufferTouch::AccessType::Write) {
1231 to_update = &regions_written;
1232
1233 } else if (touch.touch_type == BufferTouch::AccessType::Read) {
1234 to_update = &regions_read;
1235 } else {
1236 continue;
1237 }
1238
1239 PrimExpr prev = to_update->Get(touch.buffer).value_or(Bool(false));
1240 PrimExpr new_predicate = touch.predicate && touch.BeforeLoopIteration();
1241 to_update->Set(touch.buffer, prev || new_predicate);
1242 }
1243
1244 auto update_map = [&](auto& map) {
1245 Map<Buffer, PrimExpr> new_map;
1246 for (auto [buffer, predicate] : map) {
1247 new_map.Set(buffer, SimplifyAsAndOfOrs(predicate, analyzer));
1248 }
1249 map = std::move(new_map);
1250 };
1251 update_map(regions_written);
1252 update_map(regions_read);
1253
1254 // If buffer is already in used, widen the predicate
1255 for (auto& prev_unused : constraints_) {
1256 if (auto opt_predicate = regions_written.Get(prev_unused.buffer)) {
1257 PrimExpr new_predicate = prev_unused.predicate || opt_predicate.value();
1258 prev_unused.predicate = SimplifyAsAndOfOrs(new_predicate, analyzer);
1259 regions_written.erase(prev_unused.buffer);
1260 }
1261 }
1262
1263 // Otherwise, add new "touch" to represent the unused values
1264 for (auto [buffer, predicate] : regions_written) {
1265 constraints_.push_back(
1266 BufferTouch{buffer, predicate, tir::Call(buffer->dtype, builtin::undef(), {})});
1267 }
1268
1269 // If buffer is read out, narrow the predicate
1270 for (auto& prev_unused : constraints_) {
1271 if (auto opt_pred = regions_read.Get(prev_unused.buffer)) {
1272 PrimExpr predicate = opt_pred.value();
1273 prev_unused.predicate = SimplifyAsAndOfOrs(prev_unused.predicate && !predicate, analyzer);
1274 }
1275 }
1276
1277 // Clean-up and remove any empty constraints
1278 constraints_.erase(
1279 std::remove_if(constraints_.begin(), constraints_.end(),
1280 [](const auto& constraint) { return is_zero(constraint.predicate); }),
1281 constraints_.end());
1282}

1330void ControlFlowGraph::ForwardPropagateKnownValues(size_t max_revisits) {
1331 // Values to visit when searching. Using a std::set to
1332 // preferentially visit nodes near the start of the control flow.
1333 std::set<size_t> to_visit;
1334
1335 // Map from a block's index
1336 std::unordered_map<size_t, size_t> visit_count_lookup;
1337
1338 // Initiatize the locations to search from, propagating values
1339 // forward from all locations that have a known value.
1340 for (size_t i = 0; i < control_flow_.size(); i++) {
1341 bool has_known_value = false;
1342 for (const auto& touch : control_flow_[i].touch_points) {
1343 if (!HasBufferLoad(touch.value)) {
1344 has_known_value = true;
1345 break;
1346 }
1347 }
1348
1349 if (has_known_value) {
1350 to_visit.insert(i);
1351 }
1352 }
1353
1354 Analyzer analyzer;
1355 analyzer.rewrite_simplify.SetEnabledExtensions(arith::RewriteSimplifier::Extension(
1356 arith::RewriteSimplifier::kTransitivelyProveInequalities |
1357 arith::RewriteSimplifier::kApplyConstraintsToBooleanBranches));
1358
1359 analyzer.Bind(iterator_ranges_);
1360 analyzer.Bind(free_predicate_parameters_);
1361
1362 while (to_visit.size()) {
1363 size_t visiting = *to_visit.begin();
1364 to_visit.erase(visiting);
1365
1366 size_t num_previous_visits = visit_count_lookup[visiting]++;
1367
1368 ControlFlowBlock& block = control_flow_[visiting];
1369
1370 // Step 1: Collect known values provided from each predecessor
1371 block.known_at_block_start = [&]() -> BufferState {
1372 if (num_previous_visits >= max_revisits) {
1373 return BufferState();
1374 }
1375
1376 // Validate internal constraint. This should be true by
1377 // construction, as ControlFlowGraphBuilder only builds graphs
1378 // that have two or fewer predecessors.
1379 ICHECK_LE(block.predecessors.size(), 2)
1380 << "InternalError: Each block should have at most two predecessors. "
1381 << "Graph constructed in ControlFlowGraphBuilder did not satisfy this constraint.";
1382
1383 std::vector<BufferState> states;
1384 for (const auto& pred : block.predecessors) {
1385 const auto& pred_block = control_flow_[pred.index];
1386 BufferState state = pred_block.known_at_block_end;
1387 state.Substitute(pred.var_remap, &analyzer);
1388 states.push_back(state);
1389 }
1390
1391 if (std::all_of(block.predecessors.begin(), block.predecessors.end(),
1392 [&](const auto& pred) { return visit_count_lookup[pred.index] == 0; })) {
1393 // Predecessors, if any, are unvisited.
1394 return {};
1395 } else if (block.predecessors.size() == 1) {
1396 // Block has only a single predecessor
1397 return states[0];
1398 }
1399
1400 const auto& pred_a = block.predecessors[0];
1401 const auto& pred_b = block.predecessors[1];
1402
1403 auto& priors_a = states[0];
1404 auto& priors_b = states[1];
1405
1406 // During the first visit of a block, predecessor blocks may be
1407 // unvisited, even though we preferentially visit earlier blocks
1408 // first. (e.g. During the first visit of the start of a For
1409 // loop, the end of the For loop has not yet been visited.) If
1410 // this is the case, assume the best-case scenario that all
1411 // knowns are consistent, and rely on a later visit to
1412 // resolve/remove any conflicts.
1413 if (visit_count_lookup[pred_a.index] == 0) {
1414 return priors_b;
1415 } else if (visit_count_lookup[pred_b.index] == 0) {
1416 return priors_a;
1417 }
1418
1419 if (pred_a.post_condition && pred_b.post_condition) {
1420 // The predicate can identify which predecessor block applies
1421 // (e.g. i==0 for the first loop iteration, i>0 for remaining
1422 // loop iterations). Therefore, we can use all buffer
1423 // constraints, conditional on having come from the
1424 // predecessor that provides it.
1425 priors_a.AddCondition(pred_a.post_condition.value());
1426 priors_b.AddCondition(pred_b.post_condition.value());
1427 priors_a.Union(priors_b, &analyzer);
1428 return priors_a;
1429 } else {
1430 // We don't know which predecessor applies. Therefore, the
1431 // only buffer constraints that can be used are those that
1432 // appear in both predecessors.
1433 priors_a.Intersection(priors_b, &analyzer);
1434 return priors_a;
1435 }
1436 }();
1437
1438 // Step 2: Collect knowns provided as a result of executing this block
1439 auto post_state = [&]() {
1440 if (num_previous_visits >= max_revisits) {
1441 return BufferState();
1442 }
1443 auto post_state = block.known_at_block_start;
1444 post_state.ApplyTouches(axis_var_lookup_, block.touch_points, &analyzer);
1445 post_state.RemoveFreeParameters(free_predicate_parameters_, &analyzer);
1446 return post_state;
1447 }();
1448
1449 // Step 3: If any changes are made to the post knowns since the
1450 // previous time we visited this block, mark the successor block
1451 // as needing to be visited.
1452 if (num_previous_visits == 0 ||
1453 !post_state.IsEquivalentTo(block.known_at_block_end, &analyzer)) {
1454 block.known_at_block_end = std::move(post_state);
1455 for (const auto& successor : block.successors) {
1456 to_visit.insert(successor.index);
1457 }
1458 }
1459 }
1460}

1462void ControlFlowGraph::BackwardPropagateUnusedValues(size_t max_revisits) {
1463 // Values to visit when searching. Using a std::set to
1464 // preferentially visit nodes near the end of the control flow.
1465 std::set<size_t> to_visit;
1466
1467 // Map from a block's index
1468 std::unordered_map<size_t, size_t> visit_count_lookup;
1469
1470 // Initiatize the locations to search from, propagating values
1471 // backward from anywhere that performs a write.
1472 for (size_t i = 0; i < control_flow_.size(); i++) {
1473 const auto& touch_points = control_flow_[i].touch_points;
1474 bool performs_write = std::any_of(
1475 touch_points.begin(), touch_points.end(),
1476 [](const auto& touch) { return touch.touch_type == BufferTouch::AccessType::Write; });
1477 if (performs_write) {
1478 to_visit.insert(i);
1479 }
1480 }
1481
1482 Analyzer analyzer;
1483 analyzer.rewrite_simplify.SetEnabledExtensions(
1484 arith::RewriteSimplifier::kTransitivelyProveInequalities);
1485
1486 analyzer.Bind(iterator_ranges_);
1487 analyzer.Bind(free_predicate_parameters_);
1488
1489 while (to_visit.size()) {
1490 size_t visiting = *to_visit.rbegin();
1491 to_visit.erase(visiting);
1492
1493 size_t num_previous_visits = visit_count_lookup[visiting]++;
1494
1495 ControlFlowBlock& block = control_flow_[visiting];
1496
1497 // Step 1: Collect known unused indices provided by each successor
1498 block.unused_at_block_end = [&]() -> BufferState {
1499 if (num_previous_visits >= max_revisits) {
1500 return BufferState();
1501 }
1502 ICHECK_LE(block.successors.size(), 2)
1503 << "Each block should have at most two successors, but block " << visiting
1504 << " breaks this requirement";
1505
1506 std::vector<BufferState> states;
1507 for (const auto& successor : block.successors) {
1508 const auto& successor_block = control_flow_[successor.index];
1509 BufferState state = successor_block.unused_at_block_start;
1510 state.Substitute(successor.var_remap, &analyzer);
1511 states.push_back(state);
1512 }
1513
1514 if (std::all_of(block.successors.begin(), block.successors.end(), [&](const auto& successor) {
1515 return visit_count_lookup[successor.index] == 0;
1516 })) {
1517 // Successors, if any, are unvisited.
1518 return {};
1519 } else if (block.successors.size() == 1) {
1520 // Block has only a single successor
1521 return states[0];
1522 }
1523
1524 const auto& successor_a = block.successors[0];
1525 const auto& successor_b = block.successors[1];
1526
1527 auto& post_a = states[0];
1528 auto& post_b = states[1];
1529
1530 // During the first visit of a block, successor blocks may be
1531 // unvisited, even though we preferentially visit later blocks
1532 // first. (e.g. During the first visit of the end of a For
1533 // loop, the start of the For loop has not yet been visited.)
1534 // If this is the case, assume the best-case scenario that all
1535 // knowns are consistent, and rely on a later visit to
1536 // resolve/remove any conflicts.
1537 if (visit_count_lookup[successor_a.index] == 0) {
1538 return post_b;
1539 } else if (visit_count_lookup[successor_b.index] == 0) {
1540 return post_a;
1541 }
1542
1543 if (successor_a.post_condition && successor_b.post_condition) {
1544 // The predicate can identify which successor block applies
1545 // (e.g. i==n-1 for the last loop iteration, i<n-1 for earlier
1546 // loop iterations). Therefore, we can use all buffer
1547 // constraints, conditional on having come from the
1548 // successor that provides it.
1549 post_a.AddCondition(successor_a.post_condition.value());
1550 post_b.AddCondition(successor_b.post_condition.value());
1551 post_a.Union(post_b, &analyzer);
1552 return post_a;
1553 } else {
1554 // We don't know which successor applies. Therefore, the
1555 // only buffer constraints that can be used are those that
1556 // appear in both successors.
1557 post_a.Intersection(post_b, &analyzer);
1558 return post_a;
1559 }
1560 }();
1561
1562 // Step 2: Collect knowns provided as a result of executing this block
1563 auto unused_at_block_start = [&]() {
1564 if (num_previous_visits >= max_revisits) {
1565 return BufferState();
1566 }
1567 auto prior_state = block.unused_at_block_end;
1568 prior_state.BackpropUnusedIndices(axis_var_lookup_, block.touch_points, &analyzer);
1569 prior_state.RemoveFreeParameters(free_predicate_parameters_, &analyzer);
1570 return prior_state;
1571 }();
1572
1573 // Step 3: If any changes are made to the post knowns since the
1574 // previous time we visited this block, mark the successor block
1575 // as needing to be visited.
1576 if (num_previous_visits == 0 ||
1577 !unused_at_block_start.IsEquivalentTo(block.unused_at_block_start, &analyzer)) {
1578 block.unused_at_block_start = std::move(unused_at_block_start);
1579 for (const auto& pred : block.predecessors) {
1580 to_visit.insert(pred.index);
1581 }
1582 }
1583 }

1586bool ControlFlowGraph::IsOverwrittenWithoutEffect(const tir::BufferStore& store,
1587 const Stmt& context) const {
1588 Optional<Array<Var>> index_variables = GetIndexVariables(store->buffer);
1589 if (!index_variables) {
1590 return false;
1591 }
1592
1593 auto it = control_flow_lookup_.find(context.get());
1594 ICHECK(it != control_flow_lookup_.end())
1595 << "Context " << PrettyPrint(context) << " did not occur within analyzed statement";
1596 const auto& context_block = control_flow_[it->second];
1597
1598 auto [store_touch, free_params] = context_block.MakeBufferTouch(
1599 store->buffer, index_variables.value(), store->indices, BufferTouch::AccessType::Write,
1600 BufferLoad(store->buffer, store->indices));
1601
1602 Analyzer local_analyzer;
1603 local_analyzer.Bind(free_predicate_parameters_);
1604 local_analyzer.Bind(iterator_ranges_);
1605 local_analyzer.Bind(free_params);
1606 local_analyzer.rewrite_simplify.SetEnabledExtensions(
1607 RewriteSimplifier::kTransitivelyProveInequalities);
1608
1609 PrimExpr predicate = store_touch.predicate && store_touch.AtLoopIteration();
1610
1611 predicate = SimplifyAsAndOfOrs(predicate, &local_analyzer);
1612
1613 for (const auto& unused : context_block.unused_at_block_end.constraints_) {
1614 if (store_touch.buffer.same_as(unused.buffer)) {
1615 PrimExpr difference = SimplifyAsAndOfOrs(predicate && !unused.predicate, &local_analyzer);
1616 if (is_zero(difference)) {
1617 return true;
1618 }
1619 }
1620 }
1621 return false;
1622}

1624PrimExpr ControlFlowGraph::SimplifyInContext(PrimExpr expr, const tir::Stmt& context,
1625 Analyzer* analyzer) const {
1626 size_t context_index = [&]() {
1627 auto it = control_flow_lookup_.find(context.get());
1628 ICHECK(it != control_flow_lookup_.end())
1629 << "Context did not occur in the Stmt provided to BufferTouchPattern's constructor";
1630 return it->second;
1631 }();
1632
1633 PrimExpr constraint = Bool(true);
1634 for (const auto& known : non_buffer_assumptions_) {
1635 constraint = constraint && known;
1636 }
1637 With<ConstraintContext> constraint_context(analyzer, constraint);
1638
1639 expr = control_flow_[context_index].known_at_block_start.SubstituteKnownBufferValues(
1640 std::move(expr), axis_var_lookup_, analyzer);
1641
1642 expr = analyzer->Simplify(std::move(expr));
1643 return expr;
1644}
