30namespace tvm {
31namespace tir {
32namespace transform {
33struct OOBLocation {
34 Buffer buf;
35 size_t dimension;
36 ObjectRef index;
37 arith::IntSet index_bounds;
38 arith::IntSet shape_bounds;
39};
40
41class OOBError : public ScheduleError {
42 public:
43 OOBError(IRModule mod, std::vector<OOBLocation> locations) : mod_(mod), locations_(locations) {}
44 String FastErrorString() const final { return "Out of bound memory access"; }
45
46 String DetailRenderTemplate() const final {
47 std::stringstream s;
48 for (const auto& oob : locations_) {
49 s << "Out of bounds memory access on buffer " << oob.buf->name << " dimension "
50 << oob.dimension << ".";
51 s << " index " << oob.index << " with bounds [" << oob.index_bounds.min() << ", "
52 << oob.index_bounds.max() << "] is outside the range [0, " << oob.shape_bounds.min()
53 << "].";
54 s << "\n";
55 }
56 return s.str();
57 }
58 IRModule mod() const final { return mod_; }
59 Array<ObjectRef> LocationsOfInterest() const final {
60 std::vector<ObjectRef> locs;
61 for (auto loc : locations_) {
62 locs.push_back(loc.index);
63 }
64 return locs;
65 }
66
67 private:
68 IRModule mod_;
69 std::vector<OOBLocation> locations_;
70};
71class OOBCheckerVisitor final : public arith::IRVisitorWithAnalyzer {
72 using IRVisitorWithAnalyzer::VisitExpr_;
73 using IRVisitorWithAnalyzer::VisitStmt_;
74
75 public:
76 void VisitStmt_(const BufferStoreNode* node) final {
77 for (size_t i = 0; i < node->buffer->shape.size(); i++) {
78 CheckBounds(node, i);
79 }
80 IRVisitorWithAnalyzer::VisitStmt_(node);
81 }
82 void VisitExpr_(const BufferLoadNode* node) final {
83 for (size_t i = 0; i < node->buffer->shape.size(); i++) {
84 CheckBounds(node, i);
85 }
86 IRVisitorWithAnalyzer::VisitExpr_(node);
87 }
88
89 template <class T>
90 void CheckBounds(const T* node, size_t i) {
91 auto ind_bounds = analyzer_.int_set(node->indices[i]);
92 auto shape_bounds = analyzer_.int_set(node->buffer->shape[i]);
93 // We would expect that
94 // `analyzer_.CanProve(node->indices[i] < 0 || node->indices[i] >= node->buffer->shape[i])`
95 // would be the way to check if any out of bounds access occurs here, but `CanProve` checks if
96 // the statement is true for all possible values (universal quantification). For a mix of in
97 // bounds and out of bounds access, no out of bounds access would be reported. We instead want
98 // to check if there is any value for which the access is out of bounds (existential
99 // quantification).
100 // An solution would be to check that the index is in bounds for every possible value. This
101 // has the problem that some valid access patterns maybe be valid but not provably valid. We
102 // prefer that this analysis is conservative and only shows errors that are provable. This leads
103 // us to the following check: are the bounds of the index outside the bounds of the shape.
104 if (analyzer_.CanProve(ind_bounds.max() >= shape_bounds.min()) ||
105 analyzer_.CanProve(ind_bounds.min() < 0)) {
106 errors.push_back({node->buffer, i, node->indices[i], ind_bounds, shape_bounds});
107 }
108 }
109
110 std::vector<OOBLocation> errors;
111};
112
113transform::Pass OOBChecker() {
114 auto pass_func = [=](tir::PrimFunc func, IRModule mod, transform::PassContext ctx) {
115 OOBCheckerVisitor checker;
116 checker(func->body);
117 if (checker.errors.size() > 0) {
118 // mod doesn't contain our function, so we construct a new mod with out function
119 IRModule func_mod({{GlobalVar("main"), func}});
120 LOG(FATAL) << OOBError(func_mod, checker.errors).RenderReport("Out of bounds checker");
121 }
122 return func;
123 };
124 return transform::CreatePrimFuncPass(pass_func, 0, "tir.analysis.OOBChecker", {});
125}
