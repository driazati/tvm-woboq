20/*!
21 * \file tir/analysis/buffer_access_lca_detector.cc
22 * \brief Detect the lowest common ancestor(LCA) of buffer access
23 */
24
25#include <tvm/tir/analysis.h>
26#include <tvm/tir/stmt_functor.h>
27
28#include "../../support/arena.h"
29
30namespace tvm {
31namespace tir {
32
33/*!
34 * \brief Detect the lowest common ancestor(LCA) position of Buffer access.
35 * \note Only consider BlockNode and ForNode to be the LCA nodes.
36 */
37class LCADetector : public StmtExprVisitor {
38 public:
39 static Map<Buffer, Optional<Stmt>> Detect(const PrimFunc& func) {
40 LCADetector detector;
41 for (const auto& kv : func->buffer_map) {
42 const Buffer& buffer = kv.second;
43 detector.buffer_var_map_.emplace(buffer->data.get(), buffer.get());
44 }
45
46 // The root node must be explicitly present in the list of
47 // ancestor_scopes_. We cannot use nullptr to represent the root
48 // node, as that is also used to represent a scope that hasn't
49 // been observed before.
50 ScopeInfo root(nullptr, nullptr, 0);
51 detector.ancestor_scopes_.push_back(&root);
52
53 detector(func->body);
54 // Prepare the return
55 Map<Buffer, Optional<Stmt>> buffer_lca;
56 for (const auto& kv : detector.buffer_lca_) {
57 const Buffer& buffer = GetRef<Buffer>(kv.first);
58 const Optional<Stmt> stmt = kv.second ? GetRef<Optional<Stmt>>(kv.second->stmt) : NullOpt;
59 buffer_lca.Set(buffer, stmt);
60 }
61 return buffer_lca;
62 }
63
64 private:
65 /*!
66 * \brief The AST node information for querying LCA.
67 * \note Only BlockNode and ForNode are considered, since they are the only statements whose
68 * body can be a SeqStmt (the LCA of buffer access) in TensorIR.
69 */
70 struct ScopeInfo {
71 // The parent scope info
72 const ScopeInfo* parent_scope_info;
73 // The parent scope stmt node
74 const StmtNode* stmt;
75 // The scope depth in the AST
76 int depth;
77 ScopeInfo(const ScopeInfo* parent_info, const StmtNode* stmt, int depth)
78 : parent_scope_info(parent_info), stmt(stmt), depth(depth) {}
79 };
80
81 void VisitStmt_(const ForNode* op) final {
82 int n = ancestor_scopes_.size();
83 const ScopeInfo* parent_scope = ancestor_scopes_.back();
84 auto* current_scope = arena_.make<ScopeInfo>(parent_scope, op, n);
85 ancestor_scopes_.push_back(current_scope);
86 StmtExprVisitor::VisitStmt_(op);
87 ancestor_scopes_.pop_back();
88 }
89
90 void VisitStmt_(const BlockNode* op) final {
91 int n = ancestor_scopes_.size();
92 for (const Buffer& buf : op->alloc_buffers) {
93 buffer_var_map_.emplace(buf->data.get(), buf.get());
94 }
95
96 const ScopeInfo* parent_scope = ancestor_scopes_.back();
97 auto* current_scope = arena_.make<ScopeInfo>(parent_scope, op, n);
98
99 ancestor_scopes_.push_back(current_scope);
100 // Update match_buffers
101 for (const MatchBufferRegion& match_buffer : op->match_buffers) {
102 UpdateBufferLCA(match_buffer->source->buffer.get());
103 match_buffers_.insert(match_buffer->buffer.get());
104 }
105
106 StmtExprVisitor::VisitStmt_(op);
107 ancestor_scopes_.pop_back();
108 }
109
110 void VisitExpr_(const BufferLoadNode* op) final {
111 UpdateBufferLCA(op->buffer.get());
112 StmtExprVisitor::VisitExpr_(op);
113 }
114
115 void VisitStmt_(const BufferStoreNode* op) final {
116 UpdateBufferLCA(op->buffer.get());
117 StmtExprVisitor::VisitStmt_(op);
118 }
119
120 void VisitStmt_(const BufferRealizeNode* op) final {
121 buffer_var_map_.emplace(op->buffer->data.get(), op->buffer.get());
122 StmtExprVisitor::VisitStmt_(op);
123 }
124
125 // Works for Load/Store and opaque access.
126 void VisitExpr_(const VarNode* op) final { VisitBufferVar(op); }
127
128 // Explict to visit buffer data in Load and Store node.
129 void VisitExpr_(const LoadNode* op) final {
130 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
131 }
132
133 void VisitStmt_(const StoreNode* op) final {
134 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
135 }
136
137 void VisitBufferVar(const VarNode* op) {
138 auto it = buffer_var_map_.find(op);
139 if (it != buffer_var_map_.end()) {
140 UpdateBufferLCA(it->second);
141 }
142 }
143
144 void UpdateBufferLCA(const BufferNode* buffer) {
145 buffer_var_map_.emplace(buffer->data.get(), buffer);
146 if (match_buffers_.find(buffer) == match_buffers_.end()) {
147 // Ingore buffer created by block match_buffer
148 const ScopeInfo*& lca = buffer_lca_[buffer];
149 lca = LowestCommonAncestor(lca, ancestor_scopes_.back());
150 }
151 }
152
153 static const ScopeInfo* LowestCommonAncestor(const ScopeInfo* lhs, const ScopeInfo* rhs) {
154 if (lhs == nullptr) return rhs;
155 if (rhs == nullptr) return lhs;
156 while (lhs->parent_scope_info != nullptr && //
157 rhs->parent_scope_info != nullptr && //
158 lhs != rhs) {
159 if (lhs->depth == rhs->depth) {
160 lhs = lhs->parent_scope_info;
161 rhs = rhs->parent_scope_info;
162 } else if (lhs->depth < rhs->depth) {
163 rhs = rhs->parent_scope_info;
164 } else {
165 lhs = lhs->parent_scope_info;
166 }
167 }
168 if (lhs->parent_scope_info == nullptr) {
169 return lhs;
170 }
171 if (rhs->parent_scope_info == nullptr) {
172 return rhs;
173 }
174 ICHECK(lhs == rhs);
175 return lhs;
176 }
177
178 /*! \brief The ancestor scope stacks info (Block and For). The
179 * first element is initialized in LCADetector::Detect to represent
180 * the root scope.
181 */
182 std::vector<const ScopeInfo*> ancestor_scopes_ = {};
183 /*! \brief The map from Buffer to its LCA ForNode/BlockNode. */
184 std::unordered_map<const BufferNode*, const ScopeInfo*> buffer_lca_ = {};
185 /*! \brief The map from Buffer data to the Buffer. */
186 std::unordered_map<const VarNode*, const BufferNode*> buffer_var_map_ = {};
187 /*! \brief The match buffers inside blocks. */
188 std::unordered_set<const BufferNode*> match_buffers_ = {};
189 /*! \brief Internal arena. */
190 support::Arena arena_;
191};
192
193Map<Buffer, Optional<Stmt>> DetectBufferAccessLCA(const PrimFunc& func) {
194 return LCADetector::Detect(func);
195}
196
197TVM_REGISTER_GLOBAL("tir.analysis.detect_buffer_access_lca").set_body_typed(DetectBufferAccessLCA);
198} // namespace tir
