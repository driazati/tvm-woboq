42/*! \brief Represents an interaction with a buffer */
43struct BufferTouch {
44 enum class AccessType {
45 /*! \brief Buffer access occurs in BufferLoad */
46 Read,
47
48 /*! \brief Buffer access occurs in BufferStore */
49 Write,
50
51 /*! \brief Buffer access occurs in tir::builtin::assume() */
52 Assume,
53 };
54
55 BufferTouch(Buffer buffer, PrimExpr predicate, PrimExpr value)
56 : buffer(buffer),
57 predicate(predicate),
58 value(value),
59 loop_var_expressions({}),
60 touch_type(AccessType::Assume) {}
61
62 BufferTouch(Buffer buffer, PrimExpr predicate, PrimExpr value,
63 std::vector<std::pair<Var, PrimExpr>> loop_var_expressions, AccessType touch_type)
64 : buffer(buffer),
65 predicate(predicate),
66 value(value),
67 loop_var_expressions(loop_var_expressions),
68 touch_type(touch_type) {}
69
70 /*! \brief The buffer being touched */
71 Buffer buffer;
72
73 /*! \brief A predicate that is true when this touch applies
74 *
75 * May be in terms of axis variables to indicate touches that impact
76 * only a portion of a buffer.
77 */
78 PrimExpr predicate;
79
80 /*! \brief The value in this buffer after the touch
81 *
82 * May be in terms of axis variables to indicate a known
83 * non-constant value. May be in terms of a BufferLoad to indicate
84 * an unknown value.
85 */
86 PrimExpr value;
87
88 /*! \brief Active loops during the buffer touch
89 *
90 * The vector contains one entry for each loop that contains the
91 * buffer touch. The `Var` item in each entry is the loop variable
92 * itself. The `PrimExpr` item is an expression for the loop
93 * variable in terms of the buffer axis variables in
94 * `ControlFlowGraph::axis_var_lookup_`.
95 *
96 * Used to construct boolean expressions indicating whether the loop
97 * iteration that performs this touch has been reached.
98 */
99 std::vector<std::pair<Var, PrimExpr>> loop_var_expressions;
100
101 /*! \brief How the buffer was interacted with
102 *
103 * When used as a constraint (e.g. in BufferState), should use
104 * Assume.
105 */
106 AccessType touch_type{AccessType::Assume};
107
108 /*! \brief Generate a boolean expression that is true for indices
109 * accessed by this touch during this iteration or a previous
110 * loop iteration.
111 *
112 * Used during forward propagation, to track known values that were
113 * written in the current loop iteration, or in a preceding loop
114 * iteration.
115 */
116 PrimExpr BeforeLoopIteration() const;
117
118 /*! \brief Generate a boolean expression that is true for indices
119 * accessed by this touch during this loop iteration.
120 *
121 * Used during speculative no-op insertion checks, to specify which
122 * indices must be later overwritten for a store to have no impact
123 * on final results.
124 */
125 PrimExpr AtLoopIteration() const;
126
127 /*! \brief Generate a boolean expression that is true for indices
128 * accessed by this touch during this loop iteration or a
129 * subsequent loop iteration.
130 *
131 * Used during backward propagation, to track indices that that are
132 * overwritten in the current loop iteration or in a later loop
133 * iteration.
134 */
135 PrimExpr AfterLoopIteration() const;
136
137 /* \brief Checks if this touch affects a subset of indices of another
138 *
139 * Returns true if the indices accessed by this touch are a subset
140 * of predicate is true can be proven to be a subset of the other
141 * subset. Returns false if it cannot be proven to be a subset of
142 * ther other subset.
143 */
144 bool IsSubsetOf(const BufferTouch& other, arith::Analyzer* analyzer) const;
145
146 /* \brief Checks if this touch affects distinct indices from another
147 *
148 * Returns true if it can be proven that the two predicates cannot
149 * be simultaneously true. Returns false if it cannot be proven
150 * that the two predicates are distinct.
151 */
152 bool IsDistinctFrom(const BufferTouch& other, arith::Analyzer* analyzer) const;
153
154 /* \brief Checks if this touch affects distinct indices from another
155 *
156 * Returns true if it can be proven that the two predicates cannot
157 * be simultaneously true. Returns false if it cannot be proven
158 * that the two predicates are distinct.
159 */
160 bool IsEquivalentTo(const BufferTouch& other, arith::Analyzer* analyzer) const;
161
162 friend std::ostream& operator<<(std::ostream& os, const BufferTouch& expr);
163};

165/*! \brief Represents the known state of buffers at a specific point */
166class BufferState {
167 public:
168 /*! Default constructor
169 *
170 * Initialize the buffer state with no known information.
171 */
172 BufferState() {}
173
174 /*! \brief Replace BufferLoad instances with known values
175 *
176 * \param expr The expression to be updated.
177 *
178 * \param axis_var_lookup A map from buffer to the variables
179 * representing positions along the buffer's axes.
180 *
181 * \param analyzer The analyzer to use when validating a
182 * constraint's predicate.
183 *
184 * \returns The modified expression. If no substitutions are made,
185 * the original expression is returned.
186 */
187 PrimExpr SubstituteKnownBufferValues(PrimExpr expr,
188 const Map<Buffer, Array<Var>>& axis_var_lookup,
189 arith::Analyzer* analyzer) const;
190
191 /*! \brief Apply a condition to all known constraints
192 *
193 * For example, when propagating pre-loop constraints into the body
194 * of a loop, add a condition that the loop iterator is zero.
195 *
196 * \param condition The condition to apply
197 */
198 void AddCondition(const PrimExpr& condition);
199
200 /*! \brief Perform a variable substitution for all constraints
201 *
202 * For example, when propagating constraints from the end of a loop
203 * to the beginning, replace `i` with `i-1`.
204 *
205 * \param var_remap The variable remapping to apply.
206 */
207 void Substitute(const Map<Var, PrimExpr>& var_remap, arith::Analyzer* analyzer);
208
209 /*! \brief Simplify the predicate of all constraints
210 *
211 * \param analyzer The analyzer with which to simplify
212 */
213 void Simplify(arith::Analyzer* analyzer);
214
215 /*! \brief Update the known buffer values based on buffer touches
216 *
217 * For any Write or Assume touches, update the known values. For
218 * any Read touches, ignore. Used to determine known values at the
219 * end of a control flow block, given the known values at the start.
220 *
221 * \param axis_var_lookup A map from buffer to the variables
222 * representing positions along the buffer's axes.
223 *
224 * \param touch_points The buffer touch points to apply
225 *
226 * \param analyzer The analyzer to use for simplifications
227 */
228 void ApplyTouches(const Map<Buffer, Array<Var>>& axis_var_lookup,
229 const std::vector<BufferTouch>& touch_points, arith::Analyzer* analyzer);
230
231 /*! \brief Update unused buffer locations based on buffer touches
232 *
233 * For any Write, mark the written-to indices as unused. (That is,
234 * immediately prior to assigning `buf[i] = expr`, the value stored
235 * at `buf[i]` is irrelevant.) For any Read, mark the read-from
236 * indices as used. This method is used to determine unused buffer
237 * indices at the start of a control flow block, given the unused
238 * buffer indices values at the end.
239 *
240 * \param axis_var_lookup A map from buffer to the variables
241 * representing positions along the buffer's axes.
242 *
243 * \param touch_points The buffer touch points to apply
244 *
245 * \param analyzer The analyzer to use for simplifications
246 */
247 void BackpropUnusedIndices(const Map<Buffer, Array<Var>>& axis_var_lookup,
248 const std::vector<BufferTouch>& touch_points,
249 arith::Analyzer* analyzer);
250
251 /*! \brief Remove free parameters from the constraints
252 *
253 * \param free_predicate_parameters
254 *
255 * \param analyzer The analyzer with which to simplify after removal
256 */
257 void RemoveFreeParameters(const Map<Var, Range>& free_predicate_parameters,
258 arith::Analyzer* analyzer);
259
260 /*! \brief Check if two buffer states are equivalent
261 *
262 * \param other
263 *
264 * \param analyzer The analyzer used to check equality of PrimExpr
265 *
266 * \return True if the two states are provably equivalent, false otherwise.
267 */
268 bool IsEquivalentTo(const BufferState& other, arith::Analyzer* analyzer) const;
269
270 /* \brief Add known values provided by another state
271 *
272 * \param other The state with which to merge constraints
273 *
274 * \param analyzer The analyzer with which to simplify the result
275 */
276 void Union(const BufferState& other, arith::Analyzer* analyzer);
277
278 /* \brief Remove all known values not consistent with another state
279 *
280 * \param other The state with which to merge constraints
281 *
282 * \param analyzer The analyzer with which to simplify the result
283 */
284 void Intersection(const BufferState& other, arith::Analyzer* analyzer);
285
286 friend std::ostream& operator<<(std::ostream& os, const BufferState&);
287
288 private:
289 friend class ControlFlowGraph;
290 /*! \brief The known constraints */
291 std::vector<BufferTouch> constraints_;
292};

294/*! \brief Represents the flow of control through a `tir::Stmt`
295 *
296 * This class contains an internal representation of the possible
297 * control flow that may occur during execution of a `tir::Stmt`. It
298 * consists of a collection of ControlFlowBlock objects, each of which
299 * represents a subset of operations performed during execution, along
300 * with edges that represent allowed transitions between
301 * `ControlFlowBlock`.
302 *
303 * In addition, the following restrictions are used.
304 *
305 * 1. Each block may have at most two predecessors, and at most two
306 * successors.
307 *
308 * 2. Within each block, values stored in a buffer do not change.
309 * That is, encountering a `BufferStore` node requires creating a
310 * new block.
311 *
312 * For example, consider the following PrimFunc
313 *
314 * ```python
315 * @T.prim_func
316 * def func(T.Buffer[16, "float32"]):
317 * for i in T.serial(16):
318 * if i < 8:
319 * B[i] = i
320 * else:
321 * B[i] = i-8
322 * ```
323 *
324 * The control flow graph would have eight control blocks.
325 *
326 * 1. function_entry, from the start of the function through the
327 * evaluation of the loop's extent.
328 *
329 * Predecessors: n/a
330 * Successors: loop_start
331 *
332 * 2. loop_start, after entering the body of the loop, through the
333 * evaluation of the conditional `i < 8`
334 *
335 * Predecessors: function_entry, after_conditional
336 * Successors: then_clause_start, else_clause_start
337 *
338 * 3. then_clause_start, after entering the then_clause of `i < 8`,
339 * through evaluation of the value `i`.
340 *
341 * Predecessors: loop_start
342 * Successors: then_clause_end
343 *
344 * 4. then_clause_end, after storing to `B[i]` prior to exiting the
345 * then_clause.
346 *
347 * Predecessors: then_clause_start
348 * Successors: after_conditional
349 *
350 * 5. else_clause_start, after entering the else_clause of `i < 8`,
351 * through evaluation of the value `i-8`.
352 *
353 * Predecessors: loop_start
354 * Successors: else_clause_end
355 *
356 * 6. else_clause_end, after storing to `B[i]` prior to exiting the
357 * else_clause.
358 *
359 * Predecessors: else_clause_start
360 * Successors: after_conditional
361 *
362 * 7. after_conditional, after the end of the if/then/else, before the
363 * end of the loop body
364 *
365 * Predecessors: then_clause_end, else_clause_end
366 * Successors: loop_start, after_loop
367 *
368 * 8. after_loop, after the loop
369 *
370 * Predecessors: after_conditional
371 * Successors: n/a
372 *
373 *
374 * By identifying `BufferStore` nodes whose value does not depend on
375 * values stored in input buffers (e.g. initializing `buf[i] = 0.0`),
376 * or whose values are provided using `builtin::assume()`
377 * (e.g. `T.assume(buf[i] == 0.0)`), the value stored in a buffer at
378 * those indices may be known for a given control block. These known
379 * values can then be propagated forward to successor blocks, to be
380 * used in context-dependent simplifications.
381 *
382 * In addition to the allowed transitions between control-flow
383 * blocks, each block also tracks the buffer touch points; which
384 * indices are read from a buffer, which values are written to which
385 * indices of a buffer, and assumptions are provided using
386 * `builtin::assume()`; that occur during the control-flow block.
387 *
388 * Note: The current implementation only tracks the values of
389 * buffers that are constrained to a specific value, and does not
390 * track inequalities that may partially constrain buffer values.
391 * That is, entering a scoped context with a data-dependent equality
392 * condition (e.g. `if buf[i] == value`) is tracked, but entering a
393 * scoped context with a data-dependent inequality condition
394 * (e.g. `if buf[i] > value`) is not tracked.
395 */
396class ControlFlowGraph {
397 public:
398 /* \brief Extract the touch pattern from a TIR statement
399 */
400 explicit ControlFlowGraph(const Stmt& stmt, size_t max_revisits = 5);
401
402 /* \brief Check if a write is overwritten without impacting final results
403 *
404 * \param store The store to be examined
405 *
406 * \param context The context in which the buffer store occurs, used
407 * to identify the control-flow block in which the store occurs. In
408 * most cases, this will be the same object as the `store` itself.
409 *
410 * \param analyzer The analyzer to be used for simplifications
411 *
412 * \return True if the specified store can be proven to be
413 * overwritten without contributing to any later statements.
414 * Returns false otherwise.
415 */
416 bool IsOverwrittenWithoutEffect(const BufferStore& store, const Stmt& context) const;
417
418 /* \brief Simplify the expression, assuming it occurs within the given context
419 *
420 * \param expr The expression to be simplified. Does not need to
421 * have occurred within the statement used to construct this
422 * BufferTouchPattern.
423 *
424 * \param context The statement where this expression occurred, or
425 * is to be inserted. Must occur within the statement used to
426 * construct this BufferTouchPattern.
427 *
428 * \param analyzer The analyzer to be used for simplifications
429 *
430 * \returns The simplified statement
431 */
432 PrimExpr SimplifyInContext(PrimExpr expr, const Stmt& context, arith::Analyzer* analyzer) const;
433
434 /*! \brief Remove the specified BufferStore from the control-flow
435 * graph
436 *
437 * Removing the specified store, which may reflow known values.
438 * This is necessary when simplifying sequential stores of the same
439 * value. Otherwise, the first could be removed as a no-op because
440 * it is overwritten by the second, and the second could be removed
441 * as a no-op because it is the same value as the first.
442 *
443 * \param store The store to remove
444 */
445 void RemoveStore(const tir::BufferStore& store);
446
447 friend std::ostream& operator<<(std::ostream& os, const ControlFlowGraph& pattern);
448
449 private:
450 /*! \brief Return index variables representing locations within a
451 * buffer.
452 *
453 * For a given buffer, will always return the same set of variables.
454 *
455 * \param buf The buffer being accessed
456 *
457 * \param indices The indices at which the buffer is being accessed.
458 * These are used to set the dtype of the buffer axis variables.
459 *
460 * \returns Variables representing a position along the buffer's axis.
461 */
462 Array<Var> GetIndexVariables(const Buffer& buf, const Array<PrimExpr>& indices);
463
464 /*! \brief Return index variables representing locations within a
465 * buffer, if they have been generated before.
466 *
467 * For a given buffer, will always return the same set of variables.
468 *
469 * \param buf The buffer being accessed
470 *
471 * \returns Variables representing a position along the buffer's axis.
472 */
473 Optional<Array<Var>> GetIndexVariables(const Buffer& buf) const;
474
475 /*! \brief Propagate known values from known BufferStore/assume
476 * subsequent control flow blocks
477 */
478 void ForwardPropagateKnownValues(size_t max_revisits);
479
480 /*! \brief Propagate overwritten/unused indices to preceding control
481 * flow blocks
482 */
483 void BackwardPropagateUnusedValues(size_t max_revisits);
484
485 struct ControlFlowEdge {
486 /* \brief The source block of the control flow edge
487 *
488 * Lookup index into `control_flow_`
489 */
490 size_t index;
491
492 /*! \brief Variable remaps
493 *
494 * e.g. Replacing loop iterator `i` with `i-1` when following an
495 * edge from the end of a loop to the beginning of the loop.
496 */
497 Map<Var, PrimExpr> var_remap;
498
499 /*! \brief Condition that must to true after following this edge
500 *
501 * This is applied after variable remapping. For example, `i >
502 * loop_min` when following the an edge from the end of a loop to
503 * the beginning of the loop.
504 */
505 Optional<PrimExpr> post_condition;
506 };
507 friend std::ostream& operator<<(std::ostream& os, const ControlFlowEdge& edge);
508
509 struct ControlFlowBlock {
510 struct LoopEntry {
511 Var loop_var;
512 PrimExpr loop_min;
513 PrimExpr loop_max;
514 Range loop_range;
515 };
516
517 /*! \brief Loop iterators that are active during this block */
518 std::vector<LoopEntry> active_loop_iterators;
519
520 /*! \brief Loop-dependent Let bindings that may appear within the block */
521 Map<Var, PrimExpr> let_bindings_using_loop;
522
523 /*! \brief Predicate that must be true to have reached this block */
524 PrimExpr scope_predicate{Bool(true)};
525
526 /*! \brief All known values prior to executing the block */
527 BufferState known_at_block_start;
528
529 /*! \brief All known values after executing the block */
530 BufferState known_at_block_end;
531
532 /*! \brief Indices whose value at the start of the block is known to be unused */
533 BufferState unused_at_block_start;
534
535 /*! \brief Indices whose value at the end of the block is known to be unused */
536 BufferState unused_at_block_end;
537
538 /* \brief Buffer touches that occur within the block
539 *
540 * All buffer touches within a block can be treated as occurring
541 * simultaneously.
542 */
543 std::vector<BufferTouch> touch_points;
544
545 /* \brief The blocks that occur after this block
546 *
547 * Lookup index into `control_flow_`
548 */
549 std::vector<ControlFlowEdge> successors;
550
551 /* \brief The blocks that occur before this block */
552 std::vector<ControlFlowEdge> predecessors;
553
554 /* \brief Construct a BufferTouch instance within this
555 * ControlFlowBlock
556 *
557 * \param graph The mutable ControlFlowGraph that owns the buffer
558 * touch. Any free parameters used in the BufferTouch's predicate
559 * will be tracked by the ControlFlowGraph.
560 *
561 * \param buf The Buffer being accessed
562 *
563 * \param indices The indices at which the buffer is accessed, in
564 * terms of the loop variables.
565 *
566 * \param touch_type The type of touch being generated
567 *
568 * \param known_expr_value The value being written to the buffer
569 *
570 * \returns The newly generated BufferTouch
571 */
572 BufferTouch MakeBufferTouch(ControlFlowGraph* graph, const Buffer& buf,
573 const Array<PrimExpr>& indices, BufferTouch::AccessType touch_type,
574 PrimExpr known_value_expr) const;
575
576 /* \brief Construct a BufferTouch instance as if it occurred in
577 * this ControlFlowBlock
578 *
579 * Used when speculative checking if a BufferStore could be
580 * inserted.
581 *
582 * \param buf The Buffer being accessed
583 *
584 * \param index_variables The variables representing location
585 * within a buffer, with one variable for each axis of the buffer.
586 *
587 * \param indices The indices at which the buffer is accessed, in
588 * terms of the loop variables.
589 *
590 * \param touch_type The type of touch being generated
591 *
592 * \param known_expr_value The value being written to the buffer
593 *
594 * \returns The newly generated BufferTouch, and a map specifying
595 * all free parameters that may occur in the BufferTouch's
596 * predicate.
597 */
598 std::pair<BufferTouch, Map<Var, Range>> MakeBufferTouch(const Buffer& buf,
599 Array<Var> index_variables,
600 Array<PrimExpr> indices,
601 BufferTouch::AccessType touch_type,
602 PrimExpr known_value_expr) const;
603 };
604 friend std::ostream& operator<<(std::ostream& os, const ControlFlowBlock& pattern);
605
606 /* \brief The control flow that occurs within the analyzed statement */
607 std::vector<ControlFlowBlock> control_flow_;
608
609 /* \brief A lookup into control_flow_
610 *
611 * A map to look up the control flow block that contains the
612 * statement.
613 */
614 std::unordered_map<const StmtNode*, size_t> control_flow_lookup_;
615
616 /*! \brief A map from free parameters to their range
617 *
618 * A BufferStore/BufferLoad has indices in terms of loop iterators,
619 * while the internal BufferTouch must have predicate in terms of
620 * the buffer's axes. While converting to the internal BufferTouch,
621 * reduction axes show up as free parameters. Tracking the range of
622 * the free parameters allows them to be removed later, by requiring
623 * a predicate to be true for all values of the free parameters.
624 */
625 Map<Var, Range> free_predicate_parameters_;
626
627 /*! \brief Ranges of iterators found in the analyzed statement */
628 Map<Var, Range> iterator_ranges_;
629
630 /* \brief A map from buffer to the variables representing positions
631 * along the buffer's axes.
632 *
633 * This is stored here, rather than as part of the BufferState or
634 * BufferTouch, to ensure that all access of a buffer use the same
635 * variables to represent the buffer's axes, reducing the amount of
636 * variable substitution required.
637 */
638 Map<Buffer, Array<Var>> axis_var_lookup_;
639
640 /* \brief Assumptions that do not depend on buffer values
641 *
642 * These may be collected as part of the handling of `builtin::assume()`, and do not depend on any
643 * buffer. Since TIR only allows mutable values as part of buffers, these assumptions may be used
644 * anywhere the
645 */
646 std::vector<PrimExpr> non_buffer_assumptions_;
647
648 friend class ControlFlowGraphBuilder;
649};
