43/*! \brief Represents an interaction with a buffer */
44struct BufferTouch {
45 enum class AccessType {
46 /*! \brief Buffer access occurs in BufferLoad */
47 Read,
48
49 /*! \brief Buffer access occurs in BufferStore */
50 Write,
51
52 /*! \brief Buffer access occurs in tir::builtin::assume() */
53 Assume,
54 };
55
56 BufferTouch(Buffer buffer, PrimExpr predicate, PrimExpr value)
57 : buffer(buffer),
58 predicate(predicate),
59 value(value),
60 loop_var_expressions({}),
61 touch_type(AccessType::Assume) {}
62
63 BufferTouch(Buffer buffer, PrimExpr predicate, PrimExpr value,
64 std::vector<std::pair<Var, PrimExpr>> loop_var_expressions, AccessType touch_type)
65 : buffer(buffer),
66 predicate(predicate),
67 value(value),
68 loop_var_expressions(loop_var_expressions),
69 touch_type(touch_type) {}
70
71 /*! \brief The buffer being touched */
72 Buffer buffer;
73
74 /*! \brief A predicate that is true when this touch applies
75 *
76 * May be in terms of axis variables to indicate touches that impact
77 * only a portion of a buffer.
78 */
79 PrimExpr predicate;
80
81 /*! \brief The value in this buffer after the touch
82 *
83 * May be in terms of axis variables to indicate a known
84 * non-constant value. May be in terms of a BufferLoad to indicate
85 * an unknown value.
86 */
87 PrimExpr value;
88
89 /*! \brief Active loops during the buffer touch
90 *
91 * The vector contains one entry for each loop that contains the
92 * buffer touch. The `Var` item in each entry is the loop variable
93 * itself. The `PrimExpr` item is an expression for the loop
94 * variable in terms of the buffer axis variables in
95 * `ControlFlowGraph::axis_var_lookup_`.
96 *
97 * Used to construct boolean expressions indicating whether the loop
98 * iteration that performs this touch has been reached.
99 */
100 std::vector<std::pair<Var, PrimExpr>> loop_var_expressions;
101
102 /*! \brief How the buffer was interacted with
103 *
104 * When used as a constraint (e.g. in BufferState), should use
105 * Assume.
106 */
107 AccessType touch_type{AccessType::Assume};
108
109 /*! \brief Generate a boolean expression that is true for indices
110 * accessed by this touch during this iteration or a previous
111 * loop iteration.
112 *
113 * Used during forward propagation, to track known values that were
114 * written in the current loop iteration, or in a preceding loop
115 * iteration.
116 */
117 PrimExpr BeforeLoopIteration() const;
118
119 /*! \brief Generate a boolean expression that is true for indices
120 * accessed by this touch during this loop iteration.
121 *
122 * Used during speculative no-op insertion checks, to specify which
123 * indices must be later overwritten for a store to have no impact
124 * on final results.
125 */
126 PrimExpr AtLoopIteration() const;
127
128 /*! \brief Generate a boolean expression that is true for indices
129 * accessed by this touch during this loop iteration or a
130 * subsequent loop iteration.
131 *
132 * Used during backward propagation, to track indices that that are
133 * overwritten in the current loop iteration or in a later loop
134 * iteration.
135 */
136 PrimExpr AfterLoopIteration() const;
137
138 /* \brief Checks if this touch affects a subset of indices of another
139 *
140 * Returns true if the indices accessed by this touch are a subset
141 * of predicate is true can be proven to be a subset of the other
142 * subset. Returns false if it cannot be proven to be a subset of
143 * ther other subset.
144 */
145 bool IsSubsetOf(const BufferTouch& other, arith::Analyzer* analyzer) const;
146
147 /* \brief Checks if this touch affects distinct indices from another
148 *
149 * Returns true if it can be proven that the two predicates cannot
150 * be simultaneously true. Returns false if it cannot be proven
151 * that the two predicates are distinct.
152 */
153 bool IsDistinctFrom(const BufferTouch& other, arith::Analyzer* analyzer) const;
154
155 /* \brief Checks if this touch affects distinct indices from another
156 *
157 * Returns true if it can be proven that the two predicates cannot
158 * be simultaneously true. Returns false if it cannot be proven
159 * that the two predicates are distinct.
160 */
161 bool IsEquivalentTo(const BufferTouch& other, arith::Analyzer* analyzer) const;
162
163 friend std::ostream& operator<<(std::ostream& os, const BufferTouch& expr);
164};

166/*! \brief Represents the known state of buffers at a specific point */
167class BufferState {
168 public:
169 /*! Default constructor
170 *
171 * Initialize the buffer state with no known information.
172 */
173 BufferState() {}
174
175 /*! \brief Replace BufferLoad instances with known values
176 *
177 * \param expr The expression to be updated.
178 *
179 * \param axis_var_lookup A map from buffer to the variables
180 * representing positions along the buffer's axes.
181 *
182 * \param analyzer The analyzer to use when validating a
183 * constraint's predicate.
184 *
185 * \returns The modified expression. If no substitutions are made,
186 * the original expression is returned.
187 */
188 PrimExpr SubstituteKnownBufferValues(PrimExpr expr,
189 const Map<Buffer, Array<Var>>& axis_var_lookup,
190 arith::Analyzer* analyzer) const;
191
192 /*! \brief Apply a condition to all known constraints
193 *
194 * For example, when propagating pre-loop constraints into the body
195 * of a loop, add a condition that the loop iterator is zero.
196 *
197 * \param condition The condition to apply
198 */
199 void AddCondition(const PrimExpr& condition);
200
201 /*! \brief Perform a variable substitution for all constraints
202 *
203 * For example, when propagating constraints from the end of a loop
204 * to the beginning, replace `i` with `i-1`.
205 *
206 * \param var_remap The variable remapping to apply.
207 */
208 void Substitute(const Map<Var, PrimExpr>& var_remap, arith::Analyzer* analyzer);
209
210 /*! \brief Simplify the predicate of all constraints
211 *
212 * \param analyzer The analyzer with which to simplify
213 */
214 void Simplify(arith::Analyzer* analyzer);
215
216 /*! \brief Update the known buffer values based on buffer touches
217 *
218 * For any Write or Assume touches, update the known values. For
219 * any Read touches, ignore. Used to determine known values at the
220 * end of a control flow block, given the known values at the start.
221 *
222 * \param axis_var_lookup A map from buffer to the variables
223 * representing positions along the buffer's axes.
224 *
225 * \param touch_points The buffer touch points to apply
226 *
227 * \param analyzer The analyzer to use for simplifications
228 */
229 void ApplyTouches(const Map<Buffer, Array<Var>>& axis_var_lookup,
230 const std::vector<BufferTouch>& touch_points, arith::Analyzer* analyzer);
231
232 /*! \brief Update unused buffer locations based on buffer touches
233 *
234 * For any Write, mark the written-to indices as unused. (That is,
235 * immediately prior to assigning `buf[i] = expr`, the value stored
236 * at `buf[i]` is irrelevant.) For any Read, mark the read-from
237 * indices as used. This method is used to determine unused buffer
238 * indices at the start of a control flow block, given the unused
239 * buffer indices values at the end.
240 *
241 * \param axis_var_lookup A map from buffer to the variables
242 * representing positions along the buffer's axes.
243 *
244 * \param touch_points The buffer touch points to apply
245 *
246 * \param analyzer The analyzer to use for simplifications
247 */
248 void BackpropUnusedIndices(const Map<Buffer, Array<Var>>& axis_var_lookup,
249 const std::vector<BufferTouch>& touch_points,
250 arith::Analyzer* analyzer);
251
252 /*! \brief Remove free parameters from the constraints
253 *
254 * \param free_predicate_parameters
255 *
256 * \param analyzer The analyzer with which to simplify after removal
257 */
258 void RemoveFreeParameters(const Map<Var, Range>& free_predicate_parameters,
259 arith::Analyzer* analyzer);
260
261 /*! \brief Check if two buffer states are equivalent
262 *
263 * \param other
264 *
265 * \param analyzer The analyzer used to check equality of PrimExpr
266 *
267 * \return True if the two states are provably equivalent, false otherwise.
268 */
269 bool IsEquivalentTo(const BufferState& other, arith::Analyzer* analyzer) const;
270
271 /* \brief Add known values provided by another state
272 *
273 * \param other The state with which to merge constraints
274 *
275 * \param analyzer The analyzer with which to simplify the result
276 */
277 void Union(const BufferState& other, arith::Analyzer* analyzer);
278
279 /* \brief Remove all known values not consistent with another state
280 *
281 * \param other The state with which to merge constraints
282 *
283 * \param analyzer The analyzer with which to simplify the result
284 */
285 void Intersection(const BufferState& other, arith::Analyzer* analyzer);
286
287 friend std::ostream& operator<<(std::ostream& os, const BufferState&);
288
289 private:
290 friend class ControlFlowGraph;
291 /*! \brief The known constraints */
292 std::vector<BufferTouch> constraints_;
293};

295/*! \brief Represents the flow of control through a `tir::Stmt`
296 *
297 * This class contains an internal representation of the possible
298 * control flow that may occur during execution of a `tir::Stmt`. It
299 * consists of a collection of ControlFlowBlock objects, each of which
300 * represents a subset of operations performed during execution, along
301 * with edges that represent allowed transitions between
302 * `ControlFlowBlock`.
303 *
304 * In addition, the following restrictions are used.
305 *
306 * 1. Each block may have at most two predecessors, and at most two
307 * successors.
308 *
309 * 2. Within each block, values stored in a buffer do not change.
310 * That is, encountering a `BufferStore` node requires creating a
311 * new block.
312 *
313 * For example, consider the following PrimFunc
314 *
315 * ```python
316 * @T.prim_func
317 * def func(T.Buffer[16, "float32"]):
318 * for i in T.serial(16):
319 * if i < 8:
320 * B[i] = i
321 * else:
322 * B[i] = i-8
323 * ```
324 *
325 * The control flow graph would have eight control blocks.
326 *
327 * 1. function_entry, from the start of the function through the
328 * evaluation of the loop's extent.
329 *
330 * Predecessors: n/a
331 * Successors: loop_start
332 *
333 * 2. loop_start, after entering the body of the loop, through the
334 * evaluation of the conditional `i < 8`
335 *
336 * Predecessors: function_entry, after_conditional
337 * Successors: then_clause_start, else_clause_start
338 *
339 * 3. then_clause_start, after entering the then_clause of `i < 8`,
340 * through evaluation of the value `i`.
341 *
342 * Predecessors: loop_start
343 * Successors: then_clause_end
344 *
345 * 4. then_clause_end, after storing to `B[i]` prior to exiting the
346 * then_clause.
347 *
348 * Predecessors: then_clause_start
349 * Successors: after_conditional
350 *
351 * 5. else_clause_start, after entering the else_clause of `i < 8`,
352 * through evaluation of the value `i-8`.
353 *
354 * Predecessors: loop_start
355 * Successors: else_clause_end
356 *
357 * 6. else_clause_end, after storing to `B[i]` prior to exiting the
358 * else_clause.
359 *
360 * Predecessors: else_clause_start
361 * Successors: after_conditional
362 *
363 * 7. after_conditional, after the end of the if/then/else, before the
364 * end of the loop body
365 *
366 * Predecessors: then_clause_end, else_clause_end
367 * Successors: loop_start, after_loop
368 *
369 * 8. after_loop, after the loop
370 *
371 * Predecessors: after_conditional
372 * Successors: n/a
373 *
374 *
375 * By identifying `BufferStore` nodes whose value does not depend on
376 * values stored in input buffers (e.g. initializing `buf[i] = 0.0`),
377 * or whose values are provided using `builtin::assume()`
378 * (e.g. `T.assume(buf[i] == 0.0)`), the value stored in a buffer at
379 * those indices may be known for a given control block. These known
380 * values can then be propagated forward to successor blocks, to be
381 * used in context-dependent simplifications.
382 *
383 * In addition to the allowed transitions between control-flow
384 * blocks, each block also tracks the buffer touch points; which
385 * indices are read from a buffer, which values are written to which
386 * indices of a buffer, and assumptions are provided using
387 * `builtin::assume()`; that occur during the control-flow block.
388 *
389 * Note: The current implementation only tracks the values of
390 * buffers that are constrained to a specific value, and does not
391 * track inequalities that may partially constrain buffer values.
392 * That is, entering a scoped context with a data-dependent equality
393 * condition (e.g. `if buf[i] == value`) is tracked, but entering a
394 * scoped context with a data-dependent inequality condition
395 * (e.g. `if buf[i] > value`) is not tracked.
396 */
397class ControlFlowGraph {
398 public:
399 /* \brief Extract the touch pattern from a TIR statement
400 */
401 explicit ControlFlowGraph(const Stmt& stmt, size_t max_revisits = 5);
402
403 /* \brief Check if a write is overwritten without impacting final results
404 *
405 * \param store The store to be examined
406 *
407 * \param context The context in which the buffer store occurs, used
408 * to identify the control-flow block in which the store occurs. In
409 * most cases, this will be the same object as the `store` itself.
410 *
411 * \param analyzer The analyzer to be used for simplifications
412 *
413 * \return True if the specified store can be proven to be
414 * overwritten without contributing to any later statements.
415 * Returns false otherwise.
416 */
417 bool IsOverwrittenWithoutEffect(const BufferStore& store, const Stmt& context) const;
418
419 /* \brief Simplify the expression, assuming it occurs within the given context
420 *
421 * \param expr The expression to be simplified. Does not need to
422 * have occurred within the statement used to construct this
423 * BufferTouchPattern.
424 *
425 * \param context The statement where this expression occurred, or
426 * is to be inserted. Must occur within the statement used to
427 * construct this BufferTouchPattern.
428 *
429 * \param analyzer The analyzer to be used for simplifications
430 *
431 * \returns The simplified statement
432 */
433 PrimExpr SimplifyInContext(PrimExpr expr, const Stmt& context, arith::Analyzer* analyzer) const;
434
435 /*! \brief Remove the specified BufferStore from the control-flow
436 * graph
437 *
438 * Removing the specified store, which may reflow known values.
439 * This is necessary when simplifying sequential stores of the same
440 * value. Otherwise, the first could be removed as a no-op because
441 * it is overwritten by the second, and the second could be removed
442 * as a no-op because it is the same value as the first.
443 *
444 * \param store The store to remove
445 */
446 void RemoveStore(const tir::BufferStore& store);
447
448 friend std::ostream& operator<<(std::ostream& os, const ControlFlowGraph& pattern);
449
450 private:
451 /*! \brief Return index variables representing locations within a
452 * buffer.
453 *
454 * For a given buffer, will always return the same set of variables.
455 *
456 * \param buf The buffer being accessed
457 *
458 * \param indices The indices at which the buffer is being accessed.
459 * These are used to set the dtype of the buffer axis variables.
460 *
461 * \returns Variables representing a position along the buffer's axis.
462 */
463 Array<Var> GetIndexVariables(const Buffer& buf, const Array<PrimExpr>& indices);
464
465 /*! \brief Return index variables representing locations within a
466 * buffer, if they have been generated before.
467 *
468 * For a given buffer, will always return the same set of variables.
469 *
470 * \param buf The buffer being accessed
471 *
472 * \returns Variables representing a position along the buffer's axis.
473 */
474 Optional<Array<Var>> GetIndexVariables(const Buffer& buf) const;
475
476 /*! \brief Propagate known values from known BufferStore/assume
477 * subsequent control flow blocks
478 *
479 * \param flow_from If specified, re-flow only from that block.
480 */
481 void ForwardPropagateKnownValues(std::optional<size_t> flow_from = std::nullopt);
482
483 /*! \brief Propagate overwritten/unused indices to preceding control
484 * flow blocks
485 *
486 * \param flow_from If specified, re-flow only from that block.
487 */
488 void BackwardPropagateUnusedValues(std::optional<size_t> flow_from = std::nullopt);
489
490 struct ControlFlowEdge {
491 /* \brief The source block of the control flow edge
492 *
493 * Lookup index into `control_flow_`
494 */
495 size_t index;
496
497 /*! \brief Variable remaps
498 *
499 * e.g. Replacing loop iterator `i` with `i-1` when following an
500 * edge from the end of a loop to the beginning of the loop.
501 */
502 Map<Var, PrimExpr> var_remap;
503
504 /*! \brief Condition that must to true after following this edge
505 *
506 * This is applied after variable remapping. For example, `i >
507 * loop_min` when following the an edge from the end of a loop to
508 * the beginning of the loop.
509 */
510 Optional<PrimExpr> post_condition;
511 };
512 friend std::ostream& operator<<(std::ostream& os, const ControlFlowEdge& edge);
513
514 struct ControlFlowBlock {
515 struct LoopEntry {
516 Var loop_var;
517 PrimExpr loop_min;
518 PrimExpr loop_max;
519 Range loop_range;
520 };
521
522 /*! \brief Loop iterators that are active during this block */
523 std::vector<LoopEntry> active_loop_iterators;
524
525 /*! \brief Loop-dependent Let bindings that may appear within the block */
526 Map<Var, PrimExpr> let_bindings_using_loop;
527
528 /*! \brief Predicate that must be true to have reached this block */
529 PrimExpr scope_predicate{Bool(true)};
530
531 /*! \brief All known values prior to executing the block */
532 BufferState known_at_block_start;
533
534 /*! \brief All known values after executing the block */
535 BufferState known_at_block_end;
536
537 /*! \brief Indices whose value at the start of the block is known to be unused */
538 BufferState unused_at_block_start;
539
540 /*! \brief Indices whose value at the end of the block is known to be unused */
541 BufferState unused_at_block_end;
542
543 /* \brief Buffer touches that occur within the block
544 *
545 * All buffer touches within a block can be treated as occurring
546 * simultaneously.
547 */
548 std::vector<BufferTouch> touch_points;
549
550 /* \brief The blocks that occur after this block
551 *
552 * Lookup index into `control_flow_`
553 */
554 std::vector<ControlFlowEdge> successors;
555
556 /* \brief The blocks that occur before this block */
557 std::vector<ControlFlowEdge> predecessors;
558
559 /* \brief Construct a BufferTouch instance within this
560 * ControlFlowBlock
561 *
562 * \param graph The mutable ControlFlowGraph that owns the buffer
563 * touch. Any free parameters used in the BufferTouch's predicate
564 * will be tracked by the ControlFlowGraph.
565 *
566 * \param buf The Buffer being accessed
567 *
568 * \param indices The indices at which the buffer is accessed, in
569 * terms of the loop variables.
570 *
571 * \param touch_type The type of touch being generated
572 *
573 * \param known_expr_value The value being written to the buffer
574 *
575 * \returns The newly generated BufferTouch
576 */
577 BufferTouch MakeBufferTouch(ControlFlowGraph* graph, const Buffer& buf,
578 const Array<PrimExpr>& indices, BufferTouch::AccessType touch_type,
579 PrimExpr known_value_expr) const;
580
581 /* \brief Construct a BufferTouch instance as if it occurred in
582 * this ControlFlowBlock
583 *
584 * Used when speculative checking if a BufferStore could be
585 * inserted.
586 *
587 * \param buf The Buffer being accessed
588 *
589 * \param index_variables The variables representing location
590 * within a buffer, with one variable for each axis of the buffer.
591 *
592 * \param indices The indices at which the buffer is accessed, in
593 * terms of the loop variables.
594 *
595 * \param touch_type The type of touch being generated
596 *
597 * \param known_expr_value The value being written to the buffer
598 *
599 * \returns The newly generated BufferTouch, and a map specifying
600 * all free parameters that may occur in the BufferTouch's
601 * predicate.
602 */
603 std::pair<BufferTouch, Map<Var, Range>> MakeBufferTouch(const Buffer& buf,
604 Array<Var> index_variables,
605 Array<PrimExpr> indices,
606 BufferTouch::AccessType touch_type,
607 PrimExpr known_value_expr) const;
608 };
609 friend std::ostream& operator<<(std::ostream& os, const ControlFlowBlock& pattern);
610
611 /* \brief The control flow that occurs within the analyzed statement */
612 std::vector<ControlFlowBlock> control_flow_;
613
614 /* \brief A lookup into control_flow_
615 *
616 * A map to look up the control flow block that contains the
617 * statement.
618 */
619 std::unordered_map<const StmtNode*, size_t> control_flow_lookup_;
620
621 /*! \brief A map from free parameters to their range
622 *
623 * A BufferStore/BufferLoad has indices in terms of loop iterators,
624 * while the internal BufferTouch must have predicate in terms of
625 * the buffer's axes. While converting to the internal BufferTouch,
626 * reduction axes show up as free parameters. Tracking the range of
627 * the free parameters allows them to be removed later, by requiring
628 * a predicate to be true for all values of the free parameters.
629 */
630 Map<Var, Range> free_predicate_parameters_;
631
632 /*! \brief Ranges of iterators found in the analyzed statement */
633 Map<Var, Range> iterator_ranges_;
634
635 /* \brief A map from buffer to the variables representing positions
636 * along the buffer's axes.
637 *
638 * This is stored here, rather than as part of the BufferState or
639 * BufferTouch, to ensure that all access of a buffer use the same
640 * variables to represent the buffer's axes, reducing the amount of
641 * variable substitution required.
642 */
643 Map<Buffer, Array<Var>> axis_var_lookup_;
644
645 /* \brief Assumptions that do not depend on buffer values
646 *
647 * These may be collected as part of the handling of `builtin::assume()`, and do not depend on any
648 * buffer. Since TIR only allows mutable values as part of buffers, these assumptions may be used
649 * anywhere the
650 */
651 std::vector<PrimExpr> non_buffer_assumptions_;
652
653 friend class ControlFlowGraphBuilder;
654
655 /*! \brief The maximum number of revisits while flowing constraints */
656 size_t max_revisits_;
657};
