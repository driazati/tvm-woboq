22 * \brief Deep equality checking.

24#include <tvm/node/object_path.h>
25#include <tvm/node/reflection.h>
26#include <tvm/node/structural_equal.h>
27#include <tvm/runtime/registry.h>
28#include <tvm/tir/analysis.h>

30namespace tvm {
31namespace tir {
32
33class DeepCmpSEqualHandler : public SEqualReducer::Handler {
34 public:
35 // use direct recursion.
36 bool SEqualReduce(const ObjectRef& lhs, const ObjectRef& rhs, bool map_free_vars,
37 const Optional<ObjectPathPair>&) final {
38 if (lhs.same_as(rhs)) return true;
39 if (!lhs.defined() && rhs.defined()) return false;
40 if (!rhs.defined() && lhs.defined()) return false;
41 if (lhs->type_index() != rhs->type_index()) return false;
42 return vtable_->SEqualReduce(lhs.get(), rhs.get(), SEqualReducer(this, nullptr, false)) &&
43 !fail_;
44 }
45
46 void DeferFail(const ObjectPathPair&) final { fail_ = true; }
47
48 ObjectRef MapLhsToRhs(const ObjectRef& lhs) final { return ObjectRef(nullptr); }
49 void MarkGraphNode() final {}
50
51 private:
52 // reflection vtable
53 ReflectionVTable* vtable_ = ReflectionVTable::Global();
54 bool fail_ = false;
55};
56
57bool ExprDeepEqual::operator()(const PrimExpr& lhs, const PrimExpr& rhs) const {
58 // quick path
59 if (lhs.same_as(rhs)) return true;
60 if (!lhs.defined() && rhs.defined()) return false;
61 if (!rhs.defined() && lhs.defined()) return false;
62 if (lhs->type_index() != rhs->type_index()) return false;
63 if (auto* plhs = lhs.as<IntImmNode>()) {
64 auto* prhs = rhs.as<IntImmNode>();
65 return plhs->dtype == prhs->dtype && plhs->value == prhs->value;
66 }
67 if (lhs.as<AnyNode>()) {
68 return false;
69 }
70 return DeepCmpSEqualHandler().SEqualReduce(lhs, rhs, false, NullOpt);
71}
72
73TVM_REGISTER_GLOBAL("tir.analysis.expr_deep_equal")
74 .set_body_typed([](const PrimExpr& lhs, const PrimExpr& rhs) {
75 return ExprDeepEqual()(lhs, rhs);
76 });
