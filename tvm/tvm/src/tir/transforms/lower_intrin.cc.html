38class IntrinInjecter : public tvm::arith::IRMutatorWithAnalyzer {
39 public:
40 using IRMutatorWithAnalyzer::VisitExpr_;
41 using IRMutatorWithAnalyzer::VisitStmt_;
42 using FLowerGeneral = runtime::TypedPackedFunc<PrimExpr(PrimExpr)>;
43
44 IntrinInjecter(arith::Analyzer* analyzer, std::string target, std::string mtriple = "")
45 : IRMutatorWithAnalyzer(analyzer) {
46 std::vector<std::string> patterns;
47 patterns.push_back(target + ".FLowerIntrinsic");
48 patterns.push_back(target + ".FLegalize");
49 bool is_llvm_aarch64 = (mtriple.find("aarch64") != std::string::npos);
50 if (is_llvm_aarch64) {
51 patterns.push_back(target + ".aarch64.FLowerIntrinsic");
52 patterns.push_back(target + ".aarch64.FLegalize");
53 }
54 patterns.push_back("default.FLowerIntrinsic");
55 patterns.push_back("default.FLegalize");
56
57 for (const std::string& pattern : patterns)
58 if (Op::HasAttrMap(pattern)) {
59 attr_maps_.push_back(Op::GetAttrMap<FLowerGeneral>(pattern));
60 if (fma_ == nullptr) {
61 fma_ = (*attr_maps_.rbegin()).get(Op::Get("tir.fma"), nullptr);
62 }
63 }
64 }
65
66 PrimExpr VisitExpr_(const CallNode* op) final {
67 if (auto* ptr_op = op->op.as<OpNode>()) {
68 for (const auto& f_attr_map : attr_maps_) {
69 FLowerGeneral f = f_attr_map.get(GetRef<Op>(ptr_op), nullptr);
70 if (f != nullptr) {
71 PrimExpr e = GetRef<PrimExpr>(op);
72 PrimExpr r = f(e);
73 ICHECK(r.defined()) << "intrinsic rule must always return valid Expr";
74 if (!r.same_as(e)) {
75 r = this->VisitExpr(r);
76 if (r.defined()) {
77 return r;
78 }
79 }
80 }
81 }
82 }
83 return IRMutatorWithAnalyzer::VisitExpr_(op);
84 }
85
86 PrimExpr VisitExpr_(const AddNode* op) final {
87 if (const MulNode* mb = op->b.as<MulNode>()) {
88 return MakeFMA(mb->a, mb->b, op->a, op);
89 } else if (const MulNode* ma = op->a.as<MulNode>()) {
90 return MakeFMA(ma->a, ma->b, op->b, op);
91 }
92 return IRMutatorWithAnalyzer::VisitExpr_(op);
93 }
94
95 // We use floordiv for integer analysis,
96 // but will need to lower them to native truncdiv instructions
97 PrimExpr VisitExpr_(const FloorDivNode* op) final {
98 auto e = GetRef<PrimExpr>(op);
99 PrimExpr ret = IRMutatorWithAnalyzer::VisitExpr_(op);
100 op = ret.as<FloorDivNode>();
101 if (op == nullptr) return ret;
102 int shift;
103 const DataType& dtype = op->dtype;
104 ICHECK(dtype.is_int() || dtype.is_uint());
105
106 if (support_bitwise_op_ && is_const_power_of_two_integer(op->b, &shift)) {
107 // lower to right shift if possible.
108 return op->a >> make_const(dtype, shift);
109 }
110
111 if (analyzer_->CanProveGreaterEqual(op->b, 0)) {
112 // Common path, positive divisor
113 if (analyzer_->CanProveGreaterEqual(op->a, 0) || analyzer_->CanProveGreaterEqual(e, 0)) {
114 return truncdiv(op->a, op->b);
115 } else {
116 DLOG(INFO) << "LowerFloorDiv: Cannot decide the sign of divident";
117 PrimExpr rdiv = truncdiv(op->a, op->b);
118 PrimExpr rmod = truncmod(op->a, op->b);
119 // condition on b >= 0.
120 // truncmod(a, b) < 0 will implies ceildiv,
121 // So we need to correct these cases.
122 if ((dtype == DataType::Int(32) || dtype == DataType::Int(64)) && support_bitwise_op_) {
123 // equivalent to rdiv + (rmod >= 0 ? 0: -1);
124 return rdiv + (rmod >> make_const(dtype, dtype.bits() - 1));
125 } else {
126 return tir::Select(rmod >= 0, rdiv, rdiv - make_const(dtype, 1));
127 }
128 }
129 } else {
130 if (dtype.is_float()) {
131 // floor(a / b)
132 return VisitExpr_(tvm::floor(op->a / op->b).as<CallNode>());
133 } else {
134 // uncommon case
135 DLOG(INFO) << "LowerFloorDiv: Cannot decide the sign of divisor";
136 auto rmod = tir::Var("rmod", dtype);
137 auto rdiv = tir::Var("rdiv", dtype);
138 // b >= 0 => (rmod >=0 ? rdiv : rdiv - 1)
139 // b < 0 => (rmod <= 0 ? rdiv : rdiv - 1)
140 PrimExpr let_rdiv =
141 tir::Let(rdiv, truncdiv(op->a, op->b),
142 tir::Select((op->b >= 0 && rmod >= 0) || (op->b < 0 && rmod <= 0), rdiv,
143 rdiv - make_const(dtype, 1)));
144 return Let(rmod, truncmod(op->a, op->b), let_rdiv);
145 }
146 }
147 }
148
149 PrimExpr VisitExpr_(const FloorModNode* op) final {
150 PrimExpr ret = IRMutatorWithAnalyzer::VisitExpr_(op);
151 op = ret.as<FloorModNode>();
152 if (op == nullptr) return ret;
153 // Lower floordiv to native truncdiv.
154 int shift;
155 const DataType& dtype = op->dtype;
156 ICHECK(dtype.is_int() || dtype.is_uint());
157
158 if (support_bitwise_op_ && is_const_power_of_two_integer(op->b, &shift)) {
159 // lower to masking if possible.
160 int64_t mask = (static_cast<int64_t>(1) << static_cast<int64_t>(shift)) - 1;
161 return op->a & make_const(dtype, mask);
162 }
163
164 if (analyzer_->CanProveGreaterEqual(op->b, 0)) {
165 // Common pass, positive divisor
166 if (analyzer_->CanProveGreaterEqual(op->a, 0)) {
167 return truncmod(op->a, op->b);
168 } else {
169 DLOG(INFO) << "LowerFloorMod: Cannot decide the sign of divident";
170 // NOTE:condition on b >= 0.
171 // mod(a, b) < 0 will imply we are doing ceildiv,
172 // So we need to correct these cases.
173 PrimExpr rmod = truncmod(op->a, op->b);
174 if ((dtype == DataType::Int(32) || dtype == DataType::Int(64)) && support_bitwise_op_) {
175 // (rmod >> shift) & b
176 // -> (rmod >= 0 ? 0: -1) & b
177 // -> rmod >= 0 ? 0 : b
178 return rmod + (op->b & (rmod >> make_const(dtype, dtype.bits() - 1)));
179 } else {
180 return tir::Select(rmod >= 0, rmod, rmod + op->b);
181 }
182 }
183 } else {
184 if (dtype.is_float()) {
185 // a - floor(a / b) * b
186 return op->a - (VisitExpr_(tvm::floor(op->a / op->b).as<CallNode>()) * op->b);
187 } else {
188 // uncommon case
189 DLOG(INFO) << "LowerFloorMod: Cannot decide the sign of divsor and divident";
190 auto rmod = tir::Var("rmod", dtype);
191 // b > 0 && rmod >= 0 -> rmod
192 // b > 0 && rmod < 0 -> rmod + b
193 // b < 0 && rmod < 0 -> rmod
194 // b < 0 && rmod > 0 -> rmod + b
195 return Let(
196 rmod, truncmod(op->a, op->b),
197 Select((op->b >= 0 && rmod >= 0) || (op->b < 0 && rmod <= 0), rmod, rmod + op->b));
198 }
199 }
200 }
201
202 PrimExpr VisitExpr_(const MaxNode* op) final {
203 using namespace arith;
204 PVar<PrimExpr> x, y;
205 PVar<IntImm> c;
206 auto e = GetRef<PrimExpr>(op);
207 if (max(floordiv(x, y), c).Match(e) && c.Eval()->value >= 0 &&
208 analyzer_->CanProveGreaterEqual(y.Eval(), 0)) {
209 return max(VisitExpr(truncdiv(x, y).Eval()), c.Eval());
210 }
211 return IRMutatorWithAnalyzer::VisitExpr_(op);
212 }
213
214 PrimExpr VisitExpr_(const EQNode* op) final {
215 using namespace arith;
216 PVar<PrimExpr> x, y;
217 auto e = GetRef<PrimExpr>(op);
218 if ((floormod(x, y) == 0).Match(e)) {
219 return VisitExpr((truncmod(x, y) == 0).Eval());
220 }
221 return IRMutatorWithAnalyzer::VisitExpr_(op);
222 }
223
224 PrimExpr VisitExpr_(const NENode* op) final {
225 using namespace arith;
226 PVar<PrimExpr> x, y;
227 auto e = GetRef<PrimExpr>(op);
228 if ((floormod(x, y) != 0).Match(e)) {
229 return VisitExpr((truncmod(x, y) != 0).Eval());
230 }
231 return IRMutatorWithAnalyzer::VisitExpr_(op);
232 }
233
234 private:
235 PrimExpr SwapBroadcastCast(const PrimExpr& e) {
236 // Try to change broadcast(cast(x)) to cast(broadcast(x))
237 // For some targets, LLVM will generate more efficient FMA
238 // instruction with the latter. For example, vmla vs. vmlal
239 // on ARM.
240 if (const BroadcastNode* bcast = e.as<BroadcastNode>()) {
241 if (const CastNode* cast = bcast->value.as<CastNode>()) {
242 auto should_swap = [&]() {
243 // Maintain behaviour (int8 -> int16, fp16 -> fp32).
244 if (cast->dtype.bits() == cast->value.dtype().bits() * 2) {
245 return true;
246 }
247 // Check both operands are integer-like.
248 if (!cast->dtype.is_uint() && !cast->dtype.is_int()) {
249 return false;
250 }
251 if (!cast->value.dtype().is_uint() && !cast->value.dtype().is_int()) {
252 return false;
253 }
254 // If both are integer-like, swap if we have a widening cast.
255 return cast->dtype.bits() > cast->value.dtype().bits();
256 };
257
258 if (should_swap()) {
259 PrimExpr new_bcast = Broadcast(cast->value, bcast->lanes);
260 return Cast(bcast->dtype, new_bcast);
261 }
262 }
263 }
264 return e;
265 }
266
267 PrimExpr MakeFMA(const PrimExpr& a, const PrimExpr& b, const PrimExpr& c, const AddNode* op) {
268 // emit fma instruction: a * b + c
269 PrimExpr lhs = SwapBroadcastCast(a);
270 PrimExpr rhs = SwapBroadcastCast(b);
271
272 if (fma_ != nullptr && op->dtype.is_float()) {
273 PrimExpr r = fma_(Call(op->dtype, builtin::fma(), {lhs, rhs, c}));
274 if (r.defined()) return this->VisitExpr(r);
275 } else {
276 if (!lhs.same_as(a) || !rhs.same_as(b)) {
277 PrimExpr mul = this->VisitExpr(Mul(lhs, rhs));
278 return Add(mul, this->VisitExpr(c));
279 }
280 }
281 return IRMutatorWithAnalyzer::VisitExpr_(op);
282 }
283
284 // attribute maps, shared only when FLegalize == FLowerIntrinsic
285 std::vector<OpAttrMap<FLowerGeneral>> attr_maps_;
286 FLowerGeneral fma_{nullptr};
287 bool support_bitwise_op_{true};
288};
289
290Stmt LowerIntrinStmt(Stmt stmt, const std::string& target) {
291 arith::Analyzer analyzer;
292 return IntrinInjecter(&analyzer, target)(std::move(stmt));
293}

295namespace transform {
296
297Pass LowerIntrin() {
298 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
299 auto* n = f.CopyOnWrite();
300 auto target = f->GetAttr<Target>(tvm::attr::kTarget);
301 ICHECK(target.defined()) << "LowerIntrin: Require the target attribute";
302 arith::Analyzer analyzer;
303 auto mtriple = target.value()->GetAttr<runtime::String>("mtriple", "");
304 n->body =
305 IntrinInjecter(&analyzer, target.value()->kind->name, mtriple.value())(std::move(n->body));
306 return f;
307 };
308 return CreatePrimFuncPass(pass_func, 0, "tir.LowerIntrin", {});
309}
