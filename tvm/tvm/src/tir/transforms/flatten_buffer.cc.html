32/*!
33 * \brief Transform multi-dimension BufferLoad/BufferStore into device-supported dimension
34 * for the TIR not contains opaque block.
35 */
36class BufferFlattener : public StmtExprMutator {
37 public:
38 static PrimFunc Flatten(PrimFunc func) {
39 Map<Var, Buffer> preflattened_buffer_map =
40 Merge(func->buffer_map, func->preflattened_buffer_map);
41 auto pass = BufferFlattener(func->buffer_map);
42 auto writer = func.CopyOnWrite();
43 writer->body = pass.VisitStmt(func->body);
44 writer->preflattened_buffer_map = preflattened_buffer_map;
45 writer->buffer_map = pass.updated_extern_buffer_map_;
46 return func;
47 }
48
49 private:
50 explicit BufferFlattener(const Map<Var, Buffer>& extern_buffer_map) {
51 for (const auto& kv : extern_buffer_map) {
52 updated_extern_buffer_map_.Set(kv.first, GetFlattenedBuffer(kv.second));
53 }
54 }
55
56 Stmt VisitStmt_(const AllocateNode* op) final {
57 Allocate alloc = Downcast<Allocate>(StmtExprMutator::VisitStmt_(op));
58 // TODO(Lunderberg): Move the handling of boolean into a
59 // dedicated pass.
60 if (alloc->dtype == DataType::Bool()) {
61 auto writer = alloc.CopyOnWrite();
62 writer->dtype = DataType::Int(8);
63 }
64 // Handle multi-dimension allocations
65 if (alloc->extents.size() == 1) {
66 return std::move(alloc);
67 } else {
68 Array<PrimExpr> flat_extent(static_cast<size_t>(1), 1);
69 for (size_t i = 0; i < alloc->extents.size(); i++) {
70 flat_extent.Set(0, flat_extent[0] * alloc->extents[i]);
71 }
72 auto n = alloc.CopyOnWrite();
73 n->extents = flat_extent;
74 return std::move(alloc);
75 }
76 }
77
78 Buffer GetFlattenedBuffer(Buffer buf) {
79 auto it = buffer_remap_.find(buf);
80 if (it != buffer_remap_.end()) {
81 return it->second;
82 }
83 auto flattened = buf.GetFlattenedBuffer();
84
85 // TODO(Lunderberg): Move the handling of boolean into a
86 // dedicated pass.
87 if (flattened->dtype == DataType::Bool()) {
88 auto writer = flattened.CopyOnWrite();
89 writer->dtype = DataType::Int(8);
90 }
91
92 buffer_remap_[buf] = flattened;
93 return flattened;
94 }
95
96 Stmt VisitStmt_(const BufferStoreNode* op) final {
97 BufferStore store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
98 bool store_returns_bool = (op->value.dtype() == DataType::Bool());
99 store = VisitBufferAccess(store);
100
101 // Handle casts from the value's dtype to the dtype of the
102 // backing array.
103 // TODO(Lunderberg): Move the handling of boolean into a
104 // dedicated pass.
105 if (store_returns_bool) {
106 ICHECK_EQ(store->buffer->dtype, DataType::Int(8))
107 << "Expected int8 backing array for boolean tensor";
108 auto writer = store.CopyOnWrite();
109 writer->value = tvm::cast(DataType::Int(8), store->value);
110 return std::move(store);
111 }
112 return std::move(store);
113 }
114
115 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
116 bool load_returns_bool = (op->dtype == DataType::Bool());
117 BufferLoad load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
118 load = VisitBufferAccess(load);
119 // Handle casts from dtype of the backing array to value's dtype.
120 // TODO(Lunderberg): Move the handling of boolean into a
121 // dedicated pass.
122 if (load_returns_bool) {
123 ICHECK_EQ(load->buffer->dtype, DataType::Int(8))
124 << "Expected int8 backing array for boolean tensor";
125 load.CopyOnWrite()->dtype = DataType::Int(8);
126 return tvm::cast(DataType::Bool(), load);
127 } else {
128 return std::move(load);
129 }
130 }
131
132 template <typename Node>
133 Node VisitBufferAccess(Node node) {
134 ICHECK(node->buffer.defined());
135 auto flattened_indices = node->buffer->ElemOffset(node->indices);
136 Buffer flattened_buffer = GetFlattenedBuffer(node->buffer);
137
138 auto writer = node.CopyOnWrite();
139 writer->buffer = flattened_buffer;
140 writer->indices = flattened_indices;
141 return node;
142 }
143
144 /*! \brief Map of buffers being remapped. */
145 std::unordered_map<Buffer, Buffer, ObjectPtrHash, ObjectPtrEqual> buffer_remap_;
146
147 /*! \brief The updated external buffer map. */
148 Map<Var, Buffer> updated_extern_buffer_map_;
149};
150
151PrimFunc FlattenBuffer(PrimFunc f) {
152 // Only apply this pass to TIR that is not from TE schedules
153 if (!IsFromLegacyTESchedule(f)) {
154 return BufferFlattener::Flatten(f);
155 } else {
156 return f;
157 }
158}

160namespace transform {
161
162Pass FlattenBuffer() {
163 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
164 return FlattenBuffer(std::move(f));
165 };
166 return CreatePrimFuncPass(pass_func, 0, "tir.FlattenBuffer", {});
167}
