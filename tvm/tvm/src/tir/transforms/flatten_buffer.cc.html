33/*!
34 * \brief Transform multi-dimension BufferLoad/BufferStore into device-supported dimension
35 * for the TIR not contains opaque block.
36 */
37class BufferFlattener : public StmtExprMutator {
38 public:
39 static PrimFunc Flatten(PrimFunc func) {
40 Map<Var, Buffer> preflattened_buffer_map =
41 Merge(func->buffer_map, func->preflattened_buffer_map);
42 auto pass = BufferFlattener(func->buffer_map);
43 auto writer = func.CopyOnWrite();
44 writer->body = pass.VisitStmt(func->body);
45 writer->preflattened_buffer_map = preflattened_buffer_map;
46 writer->buffer_map = pass.updated_extern_buffer_map_;
47 return func;
48 }
49
50 private:
51 explicit BufferFlattener(const Map<Var, Buffer>& extern_buffer_map) {
52 for (const auto& kv : extern_buffer_map) {
53 updated_extern_buffer_map_.Set(kv.first, GetFlattenedBuffer(kv.second));
54 }
55 }
56
57 Stmt VisitStmt_(const BlockNode* op) final {
58 ICHECK_EQ(op->match_buffers.size(), 0)
59 << "Unexpected MatchBufferRegion found during tir.transform.FlattenBuffer. "
60 << "All MatchBufferRegion should be removed in tir.transform.LowerMatchBuffer.";
61
62 Block block = GetRef<Block>(op);
63
64 Array<Buffer> alloc_buffers = op->alloc_buffers;
65 alloc_buffers.MutateByApply([this](Buffer buf) { return GetFlattenedBuffer(buf); });
66 if (!alloc_buffers.same_as(op->alloc_buffers)) {
67 block.CopyOnWrite()->alloc_buffers = alloc_buffers;
68 }
69
70 Array<BufferRegion> reads = op->reads;
71 reads.MutateByApply([this](BufferRegion region) { return MutateBufferRegion(region); });
72 if (!reads.same_as(op->reads)) {
73 block.CopyOnWrite()->reads = reads;
74 }
75
76 Array<BufferRegion> writes = op->writes;
77 writes.MutateByApply([this](BufferRegion region) { return MutateBufferRegion(region); });
78 if (!writes.same_as(op->writes)) {
79 block.CopyOnWrite()->writes = writes;
80 }
81
82 return StmtExprMutator::VisitStmt_(block.get());
83 }
84
85 Stmt VisitStmt_(const AllocateNode* op) final {
86 Allocate alloc = Downcast<Allocate>(StmtExprMutator::VisitStmt_(op));
87 // TODO(Lunderberg): Move the handling of boolean into a
88 // dedicated pass.
89 if (alloc->dtype == DataType::Bool()) {
90 auto writer = alloc.CopyOnWrite();
91 writer->dtype = DataType::Int(8);
92 }
93
94 if (alloc->extents.size() == 1) {
95 // No flattening required for buffers that are already flat
96
97 // TODO(rfc-70): Keep the DeclBuffer node as-is. Stripping it
98 // out in the current implementation as not all lowering passes
99 // support DeclBuffer.
100 if (auto* decl_buffer = alloc->body.as<DeclBufferNode>()) {
101 alloc.CopyOnWrite()->body = std::move(decl_buffer->body);
102 }
103
104 return std::move(alloc);
105 }
106
107 if (auto* decl_buffer = alloc->body.as<DeclBufferNode>();
108 decl_buffer && decl_buffer->buffer->data.same_as(alloc->buffer_var)) {
109 // N-d buffer, use the DeclBuffer inside to determine how it
110 // should be flattened.
111 auto& buffer = decl_buffer->buffer;
112 bool matching_buffer = [&]() {
113 if (alloc->dtype != buffer->dtype) {
114 return false;
115 }
116 if (alloc->extents.size() != buffer->shape.size()) {
117 return false;
118 }
119 ExprDeepEqual expr_equal;
120 for (size_t i = 0; i < alloc->extents.size(); i++) {
121 if (!expr_equal(alloc->extents[i], buffer->shape[i])) {
122 return false;
123 }
124 }
125 return true;
126 }();
127
128 if (matching_buffer) {
129 Buffer flattened = GetFlattenedBuffer(buffer);
130
131 auto n = alloc.CopyOnWrite();
132 // TODO(rfc-70): Update the DeclBuffer node instead of
133 // stripping it out. Stripping it out in the current
134 // implementation as not all lowering passes support
135 // DeclBuffer.
136 //
137 // n->body = DeclBuffer(flattened, std::move(decl_buffer->body));
138 n->body = std::move(decl_buffer->body);
139 n->extents = flattened->shape;
140 return std::move(alloc);
141 } else {
142 ICHECK(decl_buffer->buffer->axis_separators.empty())
143 << "DeclBuffer node doesn't match Allocate extents, but also shouldn't be "
144 "flattened to 1-d physical memory";
145 }
146 }
147
148 // Fallback, this is an allocation without a matching DeclBuffer
149 PrimExpr flat_extent = 1;
150 for (const auto& dim : alloc->extents) {
151 flat_extent *= dim;
152 }
153
154 auto n = alloc.CopyOnWrite();
155 n->extents = {flat_extent};
156 return std::move(alloc);
157 }
158
159 Buffer GetFlattenedBuffer(Buffer buf) {
160 auto it = buffer_remap_.find(buf);
161 if (it != buffer_remap_.end()) {
162 return it->second;
163 }
164 auto flattened = buf.GetFlattenedBuffer();
165
166 // TODO(Lunderberg): Move the handling of boolean into a
167 // dedicated pass.
168 if (flattened->dtype == DataType::Bool()) {
169 auto writer = flattened.CopyOnWrite();
170 writer->dtype = DataType::Int(8);
171 }
172
173 buffer_remap_[buf] = flattened;
174 return flattened;
175 }
176
177 Stmt VisitStmt_(const BufferStoreNode* op) final {
178 BufferStore store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
179 bool store_returns_bool = (op->value.dtype() == DataType::Bool());
180 store = VisitBufferAccess(store);
181
182 // Handle casts from the value's dtype to the dtype of the
183 // backing array.
184 // TODO(Lunderberg): Move the handling of boolean into a
185 // dedicated pass.
186 if (store_returns_bool) {
187 ICHECK_EQ(store->buffer->dtype, DataType::Int(8))
188 << "Expected int8 backing array for boolean tensor";
189 auto writer = store.CopyOnWrite();
190 writer->value = tvm::cast(DataType::Int(8), store->value);
191 return std::move(store);
192 }
193 return std::move(store);
194 }
195
196 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
197 bool load_returns_bool = (op->dtype == DataType::Bool());
198 BufferLoad load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
199 load = VisitBufferAccess(load);
200 // Handle casts from dtype of the backing array to value's dtype.
201 // TODO(Lunderberg): Move the handling of boolean into a
202 // dedicated pass.
203 if (load_returns_bool) {
204 ICHECK_EQ(load->buffer->dtype, DataType::Int(8))
205 << "Expected int8 backing array for boolean tensor";
206 load.CopyOnWrite()->dtype = DataType::Int(8);
207 return tvm::cast(DataType::Bool(), load);
208 } else {
209 return std::move(load);
210 }
211 }
212
213 template <typename Node>
214 Node VisitBufferAccess(Node node) {
215 ICHECK(node->buffer.defined());
216 auto flattened_indices = node->buffer->ElemOffset(node->indices);
217 Buffer flattened_buffer = GetFlattenedBuffer(node->buffer);
218
219 auto writer = node.CopyOnWrite();
220 writer->buffer = flattened_buffer;
221 writer->indices = flattened_indices;
222 return node;
223 }
224
225 BufferRegion MutateBufferRegion(BufferRegion region) {
226 Buffer orig_buf = region->buffer;
227 Buffer flattened_buf = GetFlattenedBuffer(orig_buf);
228 if (flattened_buf.same_as(orig_buf)) {
229 return region;
230 }
231
232 Array<PrimExpr> min_values;
233 Array<PrimExpr> max_values;
234 for (const auto& range : region->region) {
235 min_values.push_back(range->min);
236 max_values.push_back(range->min + range->extent - 1);
237 }
238
239 Array<PrimExpr> flattened_min = orig_buf->ElemOffset(min_values);
240 Array<PrimExpr> flattened_max = orig_buf->ElemOffset(max_values);
241
242 Array<Range> flattened_ranges;
243 ICHECK_EQ(flattened_min.size(), flattened_max.size());
244 for (size_t i = 0; i < flattened_min.size(); i++) {
245 flattened_ranges.push_back(Range(flattened_min[i], flattened_max[i] + 1));
246 }
247
248 return BufferRegion(flattened_buf, flattened_ranges);
249 }
250
251 /*! \brief Map of buffers being remapped. */
252 std::unordered_map<Buffer, Buffer, ObjectPtrHash, ObjectPtrEqual> buffer_remap_;
253
254 /*! \brief The updated external buffer map. */
255 Map<Var, Buffer> updated_extern_buffer_map_;
256};
257
258PrimFunc FlattenBuffer(PrimFunc f) {
259 // Only apply this pass to TIR that is not from TE schedules
260 if (!IsFromLegacyTESchedule(f)) {
261 return BufferFlattener::Flatten(f);
262 } else {
263 return f;
264 }
265}

269Pass FlattenBuffer() {
270 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
271 return FlattenBuffer(std::move(f));
272 };
273 return CreatePrimFuncPass(pass_func, 0, "tir.FlattenBuffer", {});
274}
