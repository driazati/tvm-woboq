37Stmt MergeNest(const std::vector<Stmt>& nest, Stmt body) {
38 // use reverse iteration
39 for (auto ri = nest.rbegin(); ri != nest.rend(); ++ri) {
40 Stmt s = *ri;
41 if (const auto* for_ = s.as<ForNode>()) {
42 auto n = make_object<ForNode>(*for_);
43 ICHECK(is_no_op(n->body));
44 n->body = body;
45 body = Stmt(n);
46 } else if (const auto* let = s.as<LetStmtNode>()) {
47 auto n = make_object<LetStmtNode>(*let);
48 ICHECK(is_no_op(n->body));
49 n->body = body;
50 body = Stmt(n);
51 } else if (const auto* attr = s.as<AttrStmtNode>()) {
52 auto n = make_object<AttrStmtNode>(*attr);
53 ICHECK(is_no_op(n->body));
54 n->body = body;
55 body = Stmt(n);
56 } else if (const auto* ite = s.as<IfThenElseNode>()) {
57 auto n = make_object<IfThenElseNode>(*ite);
58 ICHECK(is_no_op(n->then_case));
59 ICHECK(!n->else_case.defined());
60 n->then_case = body;
61 body = Stmt(n);
62 } else if (const auto* seq = s.as<SeqStmtNode>()) {
63 auto n = make_object<SeqStmtNode>(*seq);
64 ICHECK(n->size() != 0 && is_no_op(n->seq[n->size() - 1]));
65 n->seq.Set(n->size() - 1, body);
66 body = Stmt(n);
67 } else if (const auto* assert_ = s.as<AssertStmtNode>()) {
68 auto n = make_object<AssertStmtNode>(*assert_);
69 ICHECK(is_no_op(n->body));
70 n->body = body;
71 body = Stmt(n);
72 } else if (const auto* alloc = s.as<AllocateNode>()) {
73 auto n = make_object<AllocateNode>(*alloc);
74 ICHECK(is_no_op(n->body));
75 n->body = body;
76 body = Stmt(n);
77 } else {
78 LOG(FATAL) << "not supported nest type";
79 }
80 }
81 return body;
82}
83
84Stmt MergeNest(const std::vector<std::vector<Stmt>>& nest, Stmt body) {
85 for (auto ri = nest.rbegin(); ri != nest.rend(); ++ri) {
86 body = MergeNest(*ri, body);
87 }
88 return body;
89}

91class IRConvertSSA final : public StmtExprMutator {
92 public:
93 PrimExpr VisitExpr_(const VarNode* op) final {
94 if (scope_.count(op) && !scope_[op].empty()) {
95 return scope_[op].back();
96 } else {
97 return GetRef<PrimExpr>(op);
98 }
99 }
100 PrimExpr VisitExpr_(const LetNode* op) final {
101 const Var& v = op->var;
102 if (defined_.count(v.get())) {
103 PrimExpr value = this->VisitExpr(op->value);
104 ScopedRedefine redefine(this, v);
105 PrimExpr body = this->VisitExpr(op->body);
106 return Let(redefine.new_var, value, body);
107 } else {
108 defined_.insert(v.get());
109 return StmtExprMutator::VisitExpr_(op);
110 }
111 }
112
113 PrimExpr VisitExpr_(const LoadNode* op) final {
114 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
115 return PrimExpr();
116 }
117
118 Stmt VisitStmt_(const StoreNode* op) final {
119 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
120 return Stmt();
121 }
122
123 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
124 auto node = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
125 auto output = VisitBufferAccess(std::move(node));
126 return std::move(output);
127 }
128
129 Stmt VisitStmt_(const BufferStoreNode* op) final {
130 auto node = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
131 auto output = VisitBufferAccess(std::move(node));
132 return std::move(output);
133 }
134
135 Stmt VisitStmt_(const DeclBufferNode* op) final {
136 DeclBuffer decl = Downcast<DeclBuffer>(StmtExprMutator::VisitStmt_(op));
137 Buffer new_buffer = GetRemappedBuffer(decl->buffer);
138 if (!new_buffer.same_as(decl->buffer)) {
139 decl.CopyOnWrite()->buffer = std::move(new_buffer);
140 }
141 return std::move(decl);
142 }
143
144 template <typename Node>
145 Node VisitBufferAccess(Node node) {
146 Buffer new_buf = GetRemappedBuffer(node->buffer);
147 if (!new_buf.same_as(node->buffer)) {
148 auto writer = node.CopyOnWrite();
149 writer->buffer = new_buf;
150 }
151
152 return node;
153 }
154
155 Buffer GetRemappedBuffer(Buffer buf) {
156 // Determine the buffer var that should be in the updated buffer,
157 // given the current scope. If no redefines are present, then the
158 // buffer var is unchanged.
159 Var new_buffer_var = buf->data;
160 auto var_it = scope_.find(buf->data.get());
161 if (var_it != scope_.end() && !var_it->second.empty()) {
162 new_buffer_var = var_it->second.back();
163 }
164
165 // If no mapping is required, return the original buffer.
166 if (new_buffer_var.same_as(buf->data)) {
167 return buf;
168 }
169
170 // If the current scope already has a mapping of this buffer, use
171 // the mapped buffer.
172 auto key = buf.get();
173 std::vector<Buffer>& buffers = buf_remap_[key];
174 if (buffers.size() && buffers.back()->data.same_as(new_buffer_var)) {
175 return buffers.back();
176 }
177
178 // Otherwise, make and return a new buffer object that uses the
179 // new buffer, pushing it onto the scoped stack of existing
180 // buffers. This will be popped when the new_buffer_var
181 // redefinition is popped.
182 Buffer new_buf(new_buffer_var, buf->dtype, buf->shape, buf->strides, buf->elem_offset,
183 buf->name, buf->data_alignment, buf->offset_factor, buf->buffer_type,
184 buf->axis_separators, buf->span);
185 buffers.push_back(new_buf);
186 return new_buf;
187 }

189 Stmt VisitStmt_(const LetStmtNode* op) final {
190 const Var& v = op->var;
191 if (defined_.count(v.get())) {
192 PrimExpr value = this->VisitExpr(op->value);
193 ScopedRedefine redefine(this, v);
194 Stmt body = this->VisitStmt(op->body);
195 return LetStmt(redefine.new_var, value, body);
196 } else {
197 defined_.insert(v.get());
198 return StmtExprMutator::VisitStmt_(op);
199 }
200 }
201 Stmt VisitStmt_(const ForNode* op) final {
202 const Var& v = op->loop_var;
203 if (defined_.count(v.get())) {
204 ScopedRedefine redefine(this, v);
205 Stmt stmt = StmtExprMutator::VisitStmt_(op);
206 op = stmt.as<ForNode>();
207 return For(redefine.new_var, op->min, op->extent, op->kind, op->body, op->thread_binding,
208 op->annotations);
209 } else {
210 defined_.insert(v.get());
211 return StmtExprMutator::VisitStmt_(op);
212 }
213 }
214 Stmt VisitStmt_(const AllocateNode* op) final {
215 const Var& v = op->buffer_var;
216 if (defined_.count(v.get())) {
217 ScopedRedefine redefine(this, v);
218 Stmt stmt = StmtExprMutator::VisitStmt_(op);
219 op = stmt.as<AllocateNode>();
220 return Allocate(redefine.new_var, op->dtype, op->extents, op->condition, op->body);
221 } else {
222 defined_.insert(v.get());
223 return StmtExprMutator::VisitStmt_(op);
224 }
225 }
226 Stmt VisitStmt_(const AttrStmtNode* op) final {
227 if (const VarNode* v = op->node.as<VarNode>()) {
228 Stmt stmt = StmtExprMutator::VisitStmt_(op);
229 op = stmt.as<AttrStmtNode>();
230 if (scope_.count(v) && scope_[v].size() != 0) {
231 return AttrStmt(scope_[v].back(), op->attr_key, op->value, op->body);
232 } else {
233 return stmt;
234 }
235 } else {
236 return StmtExprMutator::VisitStmt_(op);
237 }
238 }
239
240 private:
241 struct ScopedRedefine {
242 ScopedRedefine(IRConvertSSA* parent, Var old_var) : parent(parent), old_var(old_var) {
243 if (old_var->type_annotation.defined()) {
244 new_var = Var(old_var->name_hint, old_var->type_annotation);
245 } else {
246 new_var = Var(old_var->name_hint, old_var->dtype);
247 }
248 parent->scope_[old_var.get()].push_back(new_var);
249 }
250
251 ~ScopedRedefine() {
252 parent->scope_[old_var.get()].pop_back();
253 for (auto& kv : parent->buf_remap_) {
254 std::vector<Buffer>& buffers = kv.second;
255 if (buffers.size() && (buffers.back()->data.get() == new_var.get())) {
256 buffers.pop_back();
257 }
258 }
259 }
260
261 IRConvertSSA* parent;
262 Var old_var;
263 Var new_var;
264 };
265
266 std::unordered_map<const VarNode*, std::vector<Var>> scope_;
267 std::unordered_set<const VarNode*> defined_;
268 std::unordered_map<const BufferNode*, std::vector<Buffer>> buf_remap_;
269};

279Array<PrimExpr> ConvertIndices(const MatchBufferRegion& match_buffer,
280 const Array<PrimExpr>& indices) {
281 const Buffer& target = match_buffer->buffer;
282 const BufferRegion& source = match_buffer->source;
283 ICHECK_EQ(indices.size(), target->shape.size());
284
285 arith::Analyzer analyzer;
286 Array<PrimExpr> result;
287 result.reserve(source->region.size());
288 size_t offset = source->region.size() - indices.size();
289 for (size_t i = 0; i < offset; ++i) {
290 const Range& range = source->region[i];
291 ICHECK(analyzer.CanProve(range->extent == 1));
292 result.push_back(range->min);
293 }
294 for (size_t i = 0; i < indices.size(); ++i) {
295 const Range& range = source->region[i + offset];
296 const PrimExpr& index = indices[i];
297 result.push_back(range->min + index);
298 }
299 return result;
300}
301
302Region ConvertRegion(const MatchBufferRegion& match_buffer, const Region& region) {
303 const Buffer& target = match_buffer->buffer;
304 const BufferRegion& source = match_buffer->source;
305 ICHECK_EQ(region.size(), target->shape.size());
306
307 arith::Analyzer analyzer;
308 Region result;
309 result.reserve(source->region.size());
310 size_t offset = source->region.size() - region.size();
311 for (size_t i = 0; i < offset; ++i) {
312 const Range& source_range = source->region[i];
313 ICHECK(analyzer.CanProve(source_range->extent == 1));
314 result.push_back(Range::FromMinExtent(source_range->min, 1));
315 }
316 for (size_t i = 0; i < region.size(); ++i) {
317 const Range& source_range = source->region[i + offset];
318 const Range& target_range = region[i];
319 result.push_back(
320 Range::FromMinExtent(source_range->min + target_range->min, target_range->extent));
321 }
322 return result;
323}

330Map<Var, Range> ConditionalBoundsContext::GetVarBoundsFromCondition() {
331 // extract equations and related vars from condition expression.
332 // currently only extract simple integral equations which could be solvable.
333 arith::Analyzer analyzer;
334 PrimExpr condition = is_true_branch_ ? condition_ : analyzer.Simplify(!condition_);
335 Array<PrimExpr> equations;
336 Array<Var> vars;
337 std::function<void(const PrimExpr&)> fvisit = [&equations, &vars, &fvisit](const PrimExpr& e) {
338 if (e->IsInstance<GENode>() || e->IsInstance<GTNode>() || e->IsInstance<LENode>() ||
339 e->IsInstance<LTNode>() || e->IsInstance<EQNode>() || e->IsInstance<NENode>()) {
340 bool is_simple = true;
341 std::vector<Var> cand_vars;
342 PostOrderVisit(e, [&cand_vars, &is_simple, &e](const ObjectRef& obj) {
343 if (obj.same_as(e)) {
344 return;
345 } else if (const VarNode* var = obj.as<VarNode>()) {
346 if (var->dtype.is_int() || var->dtype.is_uint()) {
347 cand_vars.push_back(GetRef<Var>(var));
348 }
349 } else {
350 is_simple &= obj->IsInstance<AddNode>() || obj->IsInstance<SubNode>() ||
351 obj->IsInstance<MulNode>() || obj->IsInstance<FloorDivNode>() ||
352 obj->IsInstance<FloorModNode>() || obj->IsInstance<IntImmNode>();
353 }
354 });
355 if (is_simple && !cand_vars.empty()) {
356 for (const Var& new_var : cand_vars) {
357 if (!std::any_of(vars.begin(), vars.end(),
358 [&new_var](const Var& v) { return v.same_as(new_var); })) {
359 vars.push_back(new_var);
360 }
361 }
362 equations.push_back(Downcast<PrimExpr>(e));
363 }
364 } else if (e->IsInstance<AndNode>()) {
365 And op = Downcast<And>(e);
366 fvisit(op->a);
367 fvisit(op->b);
368 } else if (e->IsInstance<CallNode>()) {
369 Call op = Downcast<Call>(e);
370 if (op->op.same_as(builtin::likely())) {
371 fvisit(op->args[0]);
372 }
373 }
374 };
375 fvisit(condition);
376 if (equations.empty() || vars.empty()) {
377 return Map<Var, Range>();
378 }
379 // build dom ranges for related vars
380 Map<Var, Range> ranges;
381 for (const Var& v : vars) {
382 arith::IntSet dom;
383 auto relax_it = relax_map_->find(v.get());
384 if (relax_it != relax_map_->end()) {
385 dom = relax_it->second;
386 } else {
387 auto hint_it = hint_map_->find(v.get());
388 if (hint_it != hint_map_->end()) {
389 dom = hint_it->second;
390 }
391 }
392 if (dom.defined()) {
393 ranges.Set(v, Range::FromMinExtent(dom.min(), analyzer.Simplify(dom.max() - dom.min() + 1)));
394 }
395 }
396 // solve constraints
397 arith::IntConstraints constraint(vars, ranges, equations);
398 auto result = arith::SolveInequalitiesToRange(constraint);
399 return result->ranges;
400}

402ConditionalBoundsContext::ConditionalBoundsContext(
403 const PrimExpr& condition, std::unordered_map<const VarNode*, arith::IntSet>* relax_map,
404 std::unordered_map<const VarNode*, arith::IntSet>* hint_map, bool is_true_branch)
405 : condition_(condition),
406 relax_map_(relax_map),
407 hint_map_(hint_map),
408 is_true_branch_(is_true_branch) {}
409
410void ConditionalBoundsContext::EnterWithScope() {
411 for (const auto& p : GetVarBoundsFromCondition()) {
412 const auto* var = p.first.get();
413 arith::IntSet new_dom = arith::IntSet::FromRange(p.second);
414 auto relax_it = relax_map_->find(var);
415 if (relax_it != relax_map_->end()) {
416 // this is a bound for relaxed var
417 origin_map_.emplace(var, relax_it->second);
418 relax_it->second = arith::Intersect({relax_it->second, new_dom});
419 } else {
420 // this is a bound for free var
421 auto hint_it = hint_map_->find(var);
422 if (hint_it != hint_map_->end()) {
423 origin_map_.emplace(var, hint_it->second);
424 hint_it->second = arith::Intersect({hint_it->second, new_dom});
425 } else {
426 origin_map_.emplace(var, arith::IntSet::Nothing());
427 hint_map_->insert(hint_it, {var, new_dom});
428 }
429 }
430 }
431}
432
433void ConditionalBoundsContext::ExitWithScope() {
434 for (const auto& p : origin_map_) {
435 const auto* var = p.first;
436 auto relax_it = relax_map_->find(var);
437 if (relax_it != relax_map_->end()) {
438 // recover bound for relaxed var
439 relax_it->second = p.second;
440 } else {
441 // recover bound for free var
442 auto hint_it = hint_map_->find(var);
443 ICHECK(hint_it != hint_map_->end());
444 if (p.second.IsNothing()) {
445 hint_map_->erase(hint_it);
446 } else {
447 hint_it->second = p.second;
448 }
449 }
450 }
451}

453std::pair<PrimExpr, PrimExpr> GetAsyncWaitAttributes(const AttrStmtNode* op) {
454 ICHECK(op && op->attr_key == tir::attr::async_wait_queue_scope);
455 auto inner = op->body.as<AttrStmtNode>();
456 ICHECK(inner && inner->attr_key == tir::attr::async_wait_inflight_count);
457 return std::make_pair(op->value, inner->value);
458}
