38// This pass narrows indexing expressions (like StoreNode::Index)
39// that trivially fit into i32/i16 (denoted by `target_bits_`) to
40// i32/i16. Considering that i32/i16 indices may be more
41// efficient on some backends (while i64 may be more efficient
42// on others, like llvm), we may want this pass when i32/i16
43// indices are more efficient.
44//
45// For Var v, we determine its dtype by examining all the PrimExpr
46// that contains v, denoted by E = {e_0 = v, e_1, e_2, ..., e_k}.
47// If all expressions in E fit into i32/i16, then we think v can be narrowed
48// to i32/i16.
49//
50// To make an indexing expression i32/i16, we must make sure that every
51// component of that expression is of dtype i32/i16. So besides Var, we
52// rewrite the following inside an indexing expression
53// - Var
54// - IntImm
55// - Cast
56//
57// Algorithm:
58// - Use DataTypeVisitor to determine whether a Var can be narrowed or not.
59// - Use DataTypeRewritter to rewrite the components of an indexing expression.
60
61using arith::Analyzer;
62using arith::ConstIntBound;
63using arith::IRMutatorWithAnalyzer;
64
65// Determine the result dtype for Var, IntImm and Cast,
66// which will be stored in `vmap` eventually.
67//
68// Algorithm:
69// We propogate the dtypes of all the Exprs that contain Var `var` into `vmap[var]`.
70// To be more specific, if for each Expr `e` which contains `var`
71// (`var` is a child node of `e` in AST), `e` fits into `target_bits_`,
72// then we narrow `var` into `target_bits_`. That is,
73// `vmap[var] = min(target_bits_, var.dtype.bits())`
74// Otherwise, `var` is not narrowed, that is, `vmap[var] = var.dtype.bits()`
75class DataTypeVisitor final : public StmtExprVisitor {
76 public:
77 explicit DataTypeVisitor(int target_bits) : bits_(target_bits), target_bits_(target_bits) {}
78
79 void VisitExpr(const PrimExpr& e) {
80 if (e.dtype().is_int()) {
81 int bits = max_bits_;
82 if (bound_.find(e) == bound_.end()) {
83 analyzer_.const_int_bound(e, &bound_);
84 }
85 ConstIntBound bound = bound_[e];
86 int64_t ubound = Downcast<IntImm>(max_value(DataType::Int(target_bits_)))->value;
87 int64_t lbound = Downcast<IntImm>(min_value(DataType::Int(target_bits_)))->value;
88 if (e.dtype().bits() <= target_bits_ ||
89 (bound->max_value <= ubound && bound->min_value >= lbound)) {
90 bits = target_bits_;
91 }
92 int tmp = bits > bits_ ? bits : bits_;
93 std::swap(bits_, tmp);
94 StmtExprVisitor::VisitExpr(e);
95 std::swap(bits_, tmp);
96 } else {
97 StmtExprVisitor::VisitExpr(e);
98 }
99 }
100
101 void VisitStmt_(const ForNode* op) {
102 analyzer_.Bind(op->loop_var, Range::FromMinExtent(op->min, op->extent));
103 vextent_[op->loop_var.as<VarNode>()] = op->extent.dtype();
104 return StmtExprVisitor::VisitStmt_(op);
105 }
106
107 void VisitStmt_(const BlockNode* op) {
108 for (const IterVar& iter : op->iter_vars) {
109 analyzer_.Bind(iter->var, Range::FromMinExtent(iter->dom->min, iter->dom->extent));
110 vextent_[iter->var.as<VarNode>()] = iter->dom->extent.dtype();
111 }
112 StmtExprVisitor::VisitStmt_(op);
113 }
114
115 void VisitStmt_(const AttrStmtNode* op) {
116 if (op->attr_key == attr::thread_extent || op->attr_key == attr::virtual_thread) {
117 IterVar iv = Downcast<IterVar>(op->node);
118 ICHECK_NE(iv->thread_tag.length(), 0U);
119 analyzer_.Bind(iv->var, Range::FromMinExtent(0, op->value));
120 vextent_[iv->var.as<VarNode>()] = op->value.dtype();
121 StmtExprVisitor::VisitStmt_(op);
122 } else {
123 StmtExprVisitor::VisitStmt_(op);
124 }
125 }
126
127 void VisitExpr_(const ReduceNode* op) {
128 // Setup the domain information before simplification.
129 for (const IterVar& iv : op->axis) {
130 analyzer_.Bind(iv->var, iv->dom);
131 vextent_[iv->var.as<VarNode>()] = iv->dom->extent.dtype();
132 }
133 // Recursively call simplification when necessary.
134 StmtExprVisitor::VisitExpr_(op);
135 }
136
137 void VisitExpr_(const VarNode* op) {
138 if (vextent_.find(op) != vextent_.end()) {
139 // We only narrow and never promote, so the result dtype
140 // is upperbounded by its original dtype before rewrite.
141 int bits = std::min(vextent_[op].bits(), bits_);
142 if (vmap.find(op) == vmap.end()) {
143 vmap[op] = op->dtype.with_bits(bits);
144 } else {
145 // We take maximum bits for all the possible Expr where a var occurs
146 vmap[op] = op->dtype.with_bits(std::max(vmap[op].bits(), bits));
147 }
148 }
149 StmtExprVisitor::VisitExpr_(op);
150 }
151
152 void VisitExpr_(const IntImmNode* op) {
153 if (op->dtype.is_int()) {
154 // We only narrow and never promote, so the result dtype
155 // is upperbounded by its original dtype before rewrite.
156 int bits = std::min(op->dtype.bits(), bits_);
157 if (vmap.find(op) == vmap.end()) {
158 vmap[op] = op->dtype.with_bits(bits);
159 } else {
160 vmap[op] = op->dtype.with_bits(std::max(vmap[op].bits(), bits));
161 }
162 }
163 StmtExprVisitor::VisitExpr_(op);
164 }
165
166 void VisitExpr_(const CastNode* op) {
167 if (op->dtype.is_int()) {
168 // We only narrow and never promote, so the result dtype
169 // is upperbounded by its original dtype before rewrite.
170 int bits = std::min(op->dtype.bits(), bits_);
171 if (vmap.find(op) == vmap.end()) {
172 vmap[op] = op->dtype.with_bits(bits);
173 } else {
174 vmap[op] = op->dtype.with_bits(std::max(vmap[op].bits(), bits));
175 }
176 }
177 StmtExprVisitor::VisitExpr_(op);
178 }
179
180 // the narrowed datatype of Var and IntImm
181 std::unordered_map<const PrimExprNode*, DataType> vmap;
182
183 protected:
184 // internal analyzer
185 arith::Analyzer analyzer_;
186
187 private:
188 // the maximum possible bits, which serves as an init value
189 static constexpr const int max_bits_ = 64;
190 // the maximum possible bit of the current expression's return dtype
191 int bits_;
192 // the target bits
193 int target_bits_;
194 // the extent of vars to be rewritten
195 std::unordered_map<const VarNode*, DataType> vextent_;
196 // the memorized bound generated by ConstIntBoundAnalyzer
197 arith::ConstIntBoundAnalyzer::BoundMapType bound_;
198};
199
200class NarrowDataTypeRewriter : public IndexDataTypeRewriter {
201 public:
202 using Parent = IndexDataTypeRewriter;
203 explicit NarrowDataTypeRewriter(int target_bits) : visitor_(target_bits) {}
204
205 Stmt operator()(Stmt s) {
206 visitor_(s);
207 for (auto i = visitor_.vmap.begin(), last = visitor_.vmap.end(); i != last;) {
208 PrimExpr e = GetRef<PrimExpr>(i->first);
209 if (e.dtype() == i->second) {
210 i = visitor_.vmap.erase(i);
211 } else {
212 ++i;
213 }
214 }
215 return VisitStmt(s);
216 }
217
218 protected:
219 // This class adds some overrides of `VisitStmt_` and `VisitExpr_` that
220 // are *not* present in the parent class.
221 // These `using` statements ensure that all of the *other* overrides
222 // provided by the parent class are fully visible to users of this class.
223 // (Discussed further in https://github.com/apache/tvm/pull/13267)
224 using Parent::VisitExpr_;
225 using Parent::VisitStmt_;
226
227 Stmt VisitStmt_(const StoreNode* op) final {
228 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
229 }
230
231 PrimExpr VisitExpr_(const LoadNode* op) final {
232 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
233 }
234
235 PrimExpr VisitExpr_(const VarNode* op) final {
236 if (auto it = var_remap_.find(GetRef<Var>(op)); it != var_remap_.end()) {
237 return (*it).second;
238 } else if (visitor_.vmap.find(op) != visitor_.vmap.end()) {
239 Var v = Var(op->name_hint, visitor_.vmap[op]);
240 var_remap_.Set(GetRef<Var>(op), v);
241 return v;
242 }
243 return Parent::VisitExpr_(op);
244 }
245
246 PrimExpr VisitExpr_(const IntImmNode* op) final {
247 if (is_enabled_) {
248 if (visitor_.vmap.find(op) != visitor_.vmap.end()) {
249 return IntImm(visitor_.vmap[op], op->value);
250 }
251 }
252 return Parent::VisitExpr_(op);
253 }
254
255 PrimExpr VisitExpr_(const CastNode* op) final {
256 if (is_enabled_ && visitor_.vmap.find(op) != visitor_.vmap.end()) {
257 PrimExpr e = Parent::VisitExpr_(op);
258 const CastNode* new_op = e.as<CastNode>();
259 ICHECK(new_op != nullptr) << "Expected type to be CastNode"
260 << ", but get " << e->GetTypeKey();
261 return Cast(visitor_.vmap[op], new_op->value);
262 }
263 return Parent::VisitExpr_(op);
264 }
265
266 private:
267 // the internal visitor to deduce the narrowed dtype
268 DataTypeVisitor visitor_;
269 // a map from Var before rewrite to that after rewrite,
270 // ensures one old Var maps to exactly one new Var
271 std::unordered_map<const VarNode*, Var> vmap_;
272};
273
274Stmt NarrowDataType(Stmt stmt, int target_bits) {
275 return NarrowDataTypeRewriter(target_bits)(stmt);
276}

278namespace transform {
279
280Pass NarrowDataType(int target_bits) {
281 auto pass_func = [target_bits](PrimFunc f, IRModule m, PassContext ctx) {
282 auto* n = f.CopyOnWrite();
283 n->body = NarrowDataTypeRewriter(target_bits)(std::move(n->body));
284 return f;
285 };
286 return CreatePrimFuncPass(pass_func, 0, "tir.NarrowDataType", {});
287}
