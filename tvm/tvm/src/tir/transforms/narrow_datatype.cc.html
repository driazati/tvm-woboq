36// This pass narrows indexing expressions (like StoreNode::Index)
37// that trivially fit into i32/i16 (denoted by `target_bits_`) to
38// i32/i16. Considering that i32/i16 indices may be more
39// efficient on some backends (while i64 may be more efficient
40// on others, like llvm), we may want this pass when i32/i16
41// indices are more efficient.
42//
43// For Var v, we determine its dtype by examining all the PrimExpr
44// that contains v, denoted by E = {e_0 = v, e_1, e_2, ..., e_k}.
45// If all expressions in E fit into i32/i16, then we think v can be narrowed
46// to i32/i16.
47//
48// To make an indexing expression i32/i16, we must make sure that every
49// component of that expression is of dtype i32/i16. So besides Var, we
50// rewrite the following inside an indexing expression
51// - Var
52// - IntImm
53// - Cast
54//
55// Algorithm:
56// - Use DataTypeVisitor to determine whether a Var can be narrowed or not.
57// - Use DataTypeRewritter to rewrite the components of an indexing expression.
58
59using arith::Analyzer;
60using arith::ConstIntBound;
61using arith::IRMutatorWithAnalyzer;
62
63// Determine the result dtype for Var, IntImm and Cast,
64// which will be stored in `vmap` eventually.
65//
66// Algorithm:
67// We propogate the dtypes of all the Exprs that contain Var `var` into `vmap[var]`.
68// To be more specific, if for each Expr `e` which contains `var`
69// (`var` is a child node of `e` in AST), `e` fits into `target_bits_`,
70// then we narrow `var` into `target_bits_`. That is,
71// `vmap[var] = min(target_bits_, var.dtype.bits())`
72// Otherwise, `var` is not narrowed, that is, `vmap[var] = var.dtype.bits()`
73class DataTypeVisitor final : public StmtExprVisitor {
74 public:
75 explicit DataTypeVisitor(int target_bits) : bits_(target_bits), target_bits_(target_bits) {}
76
77 void VisitExpr(const PrimExpr& e) {
78 if (e.dtype().is_int()) {
79 int bits = max_bits_;
80 if (bound_.find(e) == bound_.end()) {
81 analyzer_.const_int_bound(e, &bound_);
82 }
83 ConstIntBound bound = bound_[e];
84 int64_t ubound = Downcast<IntImm>(max_value(DataType::Int(target_bits_)))->value;
85 int64_t lbound = Downcast<IntImm>(min_value(DataType::Int(target_bits_)))->value;
86 if (e.dtype().bits() <= target_bits_ ||
87 (bound->max_value <= ubound && bound->min_value >= lbound)) {
88 bits = target_bits_;
89 }
90 int tmp = bits > bits_ ? bits : bits_;
91 std::swap(bits_, tmp);
92 StmtExprVisitor::VisitExpr(e);
93 std::swap(bits_, tmp);
94 } else {
95 StmtExprVisitor::VisitExpr(e);
96 }
97 }
98
99 void VisitStmt_(const ForNode* op) {
100 analyzer_.Bind(op->loop_var, Range::FromMinExtent(op->min, op->extent));
101 vextent_[op->loop_var.as<VarNode>()] = op->extent.dtype();
102 return StmtExprVisitor::VisitStmt_(op);
103 }
104
105 void VisitStmt_(const AttrStmtNode* op) {
106 if (op->attr_key == attr::thread_extent || op->attr_key == attr::virtual_thread) {
107 IterVar iv = Downcast<IterVar>(op->node);
108 ICHECK_NE(iv->thread_tag.length(), 0U);
109 analyzer_.Bind(iv->var, Range::FromMinExtent(0, op->value));
110 vextent_[iv->var.as<VarNode>()] = op->value.dtype();
111 StmtExprVisitor::VisitStmt_(op);
112 } else {
113 StmtExprVisitor::VisitStmt_(op);
114 }
115 }
116
117 void VisitExpr_(const ReduceNode* op) {
118 // Setup the domain information before simplification.
119 for (const IterVar& iv : op->axis) {
120 analyzer_.Bind(iv->var, iv->dom);
121 vextent_[iv->var.as<VarNode>()] = iv->dom->extent.dtype();
122 }
123 // Recursively call simplification when necessary.
124 StmtExprVisitor::VisitExpr_(op);
125 }
126
127 void VisitExpr_(const VarNode* op) {
128 if (vextent_.find(op) != vextent_.end()) {
129 // We only narrow and never promote, so the result dtype
130 // is upperbounded by its original dtype before rewrite.
131 int bits = std::min(vextent_[op].bits(), bits_);
132 if (vmap.find(op) == vmap.end()) {
133 vmap[op] = op->dtype.with_bits(bits);
134 } else {
135 // We take maximum bits for all the possible Expr where a var occurs
136 vmap[op] = op->dtype.with_bits(std::max(vmap[op].bits(), bits));
137 }
138 }
139 StmtExprVisitor::VisitExpr_(op);
140 }
141
142 void VisitExpr_(const IntImmNode* op) {
143 if (op->dtype.is_int()) {
144 // We only narrow and never promote, so the result dtype
145 // is upperbounded by its original dtype before rewrite.
146 int bits = std::min(op->dtype.bits(), bits_);
147 if (vmap.find(op) == vmap.end()) {
148 vmap[op] = op->dtype.with_bits(bits);
149 } else {
150 vmap[op] = op->dtype.with_bits(std::max(vmap[op].bits(), bits));
151 }
152 }
153 StmtExprVisitor::VisitExpr_(op);
154 }
155
156 void VisitExpr_(const CastNode* op) {
157 if (op->dtype.is_int()) {
158 // We only narrow and never promote, so the result dtype
159 // is upperbounded by its original dtype before rewrite.
160 int bits = std::min(op->dtype.bits(), bits_);
161 if (vmap.find(op) == vmap.end()) {
162 vmap[op] = op->dtype.with_bits(bits);
163 } else {
164 vmap[op] = op->dtype.with_bits(std::max(vmap[op].bits(), bits));
165 }
166 }
167 StmtExprVisitor::VisitExpr_(op);
168 }
169
170 // the narrowed datatype of Var and IntImm
171 std::unordered_map<const PrimExprNode*, DataType> vmap;
172
173 protected:
174 // internal analyzer
175 arith::Analyzer analyzer_;
176
177 private:
178 // the maximum possible bits, which serves as an init value
179 static constexpr const int max_bits_ = 64;
180 // the maximum possible bit of the current expression's return dtype
181 int bits_;
182 // the target bits
183 int target_bits_;
184 // the extent of vars to be rewritten
185 std::unordered_map<const VarNode*, DataType> vextent_;
186 // the memorized bound generated by ConstIntBoundAnalyzer
187 arith::ConstIntBoundAnalyzer::BoundMapType bound_;
188};

190class DataTypeRewriter : public StmtExprMutator {
191 public:
192 explicit DataTypeRewriter(int target_bits) : visitor_(target_bits) {}
193
194 Stmt operator()(Stmt s) {
195 visitor_(s);
196 for (auto i = visitor_.vmap.begin(), last = visitor_.vmap.end(); i != last;) {
197 PrimExpr e = GetRef<PrimExpr>(i->first);
198 if (e.dtype() == i->second) {
199 i = visitor_.vmap.erase(i);
200 } else {
201 ++i;
202 }
203 }
204 return VisitStmt(s);
205 }
206
207 Stmt VisitStmt_(const StoreNode* op) final {
208 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
209 return Stmt();
210 }
211
212 PrimExpr VisitExpr_(const LoadNode* op) final {
213 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
214 return PrimExpr();
215 }
216
217 Stmt VisitStmt_(const BufferStoreNode* op) final {
218 BufferStore store = GetRef<BufferStore>(op);
219
220 auto value = this->VisitExpr(op->value);
221 auto indices = VisitIndices(op->indices);
222
223 if (!value.same_as(op->value) || !indices.same_as(op->indices)) {
224 auto writer = store.CopyOnWrite();
225 writer->value = value;
226 writer->indices = indices;
227 }
228
229 return std::move(store);
230 }
231
232 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
233 BufferLoad load = GetRef<BufferLoad>(op);
234
235 auto indices = VisitIndices(op->indices);
236
237 if (!indices.same_as(op->indices)) {
238 auto writer = load.CopyOnWrite();
239 writer->indices = indices;
240 }
241
242 return std::move(load);
243 }
244
245 Array<PrimExpr> VisitIndices(Array<PrimExpr> indices) {
246 is_index_ = true;
247
248 auto fmutate = [this](const PrimExpr& index) { return this->VisitExpr(index); };
249 indices.MutateByApply(fmutate);
250
251 is_index_ = false;
252
253 return indices;
254 }
255
256 Stmt VisitStmt_(const ForNode* op) final {
257 Stmt s = StmtExprMutator::VisitStmt_(op);
258 op = s.as<ForNode>();
259 ICHECK(op != nullptr) << "Expected type to be ForNode"
260 << ", but get " << s->GetTypeKey();
261 PrimExpr e = VisitExpr(op->loop_var);
262 Var var = Downcast<Var>(e);
263 return For(var, cast(var.dtype(), op->min), cast(var.dtype(), op->extent), op->kind, op->body,
264 op->thread_binding, op->annotations);
265 }
266
267 Stmt VisitStmt_(const IfThenElseNode* op) final {
268 IfThenElse updated = Downcast<IfThenElse>(StmtExprMutator::VisitStmt_(op));
269 is_condition_ = true;
270 PrimExpr cond = VisitExpr(op->condition);
271 is_condition_ = false;
272 if (!cond.same_as(op->condition)) {
273 return std::move(IfThenElse(cond, updated->then_case, updated->else_case));
274 }
275 return std::move(updated);
276 }
277
278 Stmt VisitStmt_(const AttrStmtNode* op) final {
279 if (op->attr_key == attr::thread_extent || op->attr_key == attr::virtual_thread) {
280 Stmt s = StmtExprMutator::VisitStmt_(op);
281 op = s.as<AttrStmtNode>();
282 ICHECK(op != nullptr) << "Expected type to be AttrStmtNode"
283 << ", but get " << s->GetTypeKey();
284 const IterVarNode* iv = op->node.as<IterVarNode>();
285 ICHECK(iv != nullptr) << "Expected type to be IterVarNode"
286 << ", but get " << op->node->GetTypeKey();
287 PrimExpr e = VisitExpr(iv->var);
288 Var var = Downcast<Var>(e);
289 if (ivmap_.find(iv) == ivmap_.end()) {
290 Range dom = iv->dom;
291 if (dom.defined()) {
292 PrimExpr extend = dom->extent;
293 if (extend.dtype().is_int() && var.dtype().is_int() &&
294 var.dtype().bits() != extend.dtype().bits()) {
295 DataType dtype = var.dtype();
296 dom = Range(cast(dtype, dom->min), cast(dtype, extend), dom->span);
297 }
298 }
299 ivmap_[iv] = IterVar(dom, var, iv->iter_type, iv->thread_tag);
300 }
301 return AttrStmt(ivmap_[iv], op->attr_key, cast(var.dtype(), op->value), op->body);
302 }
303 return StmtExprMutator::VisitStmt_(op);
304 }
305
306 PrimExpr VisitExpr_(const VarNode* op) final {
307 if (visitor_.vmap.find(op) != visitor_.vmap.end()) {
308 if (vmap_.find(op) == vmap_.end()) {
309 vmap_[op] = Var(op->name_hint, visitor_.vmap[op]);
310 }
311 return vmap_[op];
312 }
313 return StmtExprMutator::VisitExpr_(op);
314 }
315
316 PrimExpr VisitExpr_(const SelectNode* op) final {
317 PrimExpr condition = this->VisitExpr(op->condition);
318 PrimExpr true_value = this->VisitExpr(op->true_value);
319 PrimExpr false_value = this->VisitExpr(op->false_value);
320 if (condition.same_as(op->condition) && true_value.same_as(op->true_value) &&
321 false_value.same_as(op->false_value)) {
322 return GetRef<PrimExpr>(op);
323 } else {
324 if (op->true_value.dtype().is_int() && op->false_value.dtype().is_int()) {
325 int bits = std::max(true_value.dtype().bits(), false_value.dtype().bits());
326 DataType dtype = true_value.dtype().with_bits(bits);
327 if (true_value.dtype() != dtype) true_value = cast(dtype, true_value);
328 if (false_value.dtype() != dtype) false_value = cast(dtype, false_value);
329 }
330 return Select(condition, true_value, false_value);
331 }
332 }
333
334 PrimExpr VisitExpr_(const RampNode* op) final {
335 PrimExpr base = VisitExpr(op->base);
336 PrimExpr stride = VisitExpr(op->stride);
337 if (base.same_as(op->base) && stride.same_as(op->stride)) {
338 return GetRef<PrimExpr>(op);
339 } else {
340 if (base.dtype().is_int()) {
341 ICHECK(stride.dtype().is_int()) << "Ramp base is int but stride is " << stride.dtype();
342 int bits = std::max(base.dtype().bits(), stride.dtype().bits());
343 DataType dtype = base.dtype().with_bits(bits);
344 if (base.dtype() != dtype) base = cast(dtype, base);
345 if (stride.dtype() != dtype) stride = cast(dtype, stride);
346 }
347 return Ramp(base, stride, op->lanes);
348 }
349 }
350
351 PrimExpr VisitExpr_(const SizeVarNode* op) final {
352 if (visitor_.vmap.find(op) != visitor_.vmap.end()) {
353 if (vmap_.find(op) == vmap_.end()) {
354 vmap_[op] = SizeVar(op->name_hint, visitor_.vmap[op]);
355 }
356 return vmap_[op];
357 }
358 return StmtExprMutator::VisitExpr_(op);
359 }
360
361 PrimExpr VisitExpr_(const IntImmNode* op) final {
362 if (is_index_) {
363 if (visitor_.vmap.find(op) != visitor_.vmap.end()) {
364 return IntImm(visitor_.vmap[op], op->value);
365 }
366 }
367 return StmtExprMutator::VisitExpr_(op);
368 }
369
370 PrimExpr VisitExpr_(const CastNode* op) final {
371 if (is_index_ && visitor_.vmap.find(op) != visitor_.vmap.end()) {
372 PrimExpr e = StmtExprMutator::VisitExpr_(op);
373 const CastNode* new_op = e.as<CastNode>();
374 ICHECK(new_op != nullptr) << "Expected type to be CastNode"
375 << ", but get " << e->GetTypeKey();
376 return Cast(visitor_.vmap[op], new_op->value);
377 }
378 return StmtExprMutator::VisitExpr_(op);
379 }
380
381 PrimExpr VisitExpr_(const AddNode* op) final;
382 PrimExpr VisitExpr_(const SubNode* op) final;
383 PrimExpr VisitExpr_(const MulNode* op) final;
384 PrimExpr VisitExpr_(const DivNode* op) final;
385 PrimExpr VisitExpr_(const ModNode* op) final;
386 PrimExpr VisitExpr_(const FloorDivNode* op) final;
387 PrimExpr VisitExpr_(const FloorModNode* op) final;
388 PrimExpr VisitExpr_(const MinNode* op) final;
389 PrimExpr VisitExpr_(const MaxNode* op) final;
390 PrimExpr VisitExpr_(const EQNode* op) final;
391 PrimExpr VisitExpr_(const NENode* op) final;
392 PrimExpr VisitExpr_(const LTNode* op) final;
393 PrimExpr VisitExpr_(const LENode* op) final;
394 PrimExpr VisitExpr_(const GTNode* op) final;
395 PrimExpr VisitExpr_(const GENode* op) final;
396 PrimExpr VisitExpr_(const CallNode* op) final;
397
398 private:
399 // the internal visitor to deduce the narrowed dtype
400 DataTypeVisitor visitor_;
401 // a map from Var before rewrite to that after rewrite,
402 // ensures one old Var maps to exactly one new Var
403 std::unordered_map<const VarNode*, Var> vmap_;
404 // a map from IterVar before rewrite to that after rewrite,
405 // ensures one old IterVar maps to exactly one new IterVar
406 std::unordered_map<const IterVarNode*, IterVar> ivmap_;
407 // indicator of index expr to rewrite
408 bool is_index_{false};
409 // indicator of condition
410 bool is_condition_{false};
411 // cached ops
412 const Op& builtin_pow_ = Op::Get("tir.pow");
413};

415#define DEFINE_BIOP_EXPR_MUTATE_WITH_TYPE_MATCH(OP, FUNC) \
416 PrimExpr DataTypeRewriter::VisitExpr_(const OP* op) { \
417 PrimExpr a = this->VisitExpr(op->a); \
418 PrimExpr b = this->VisitExpr(op->b); \
419 if (a.same_as(op->a) && b.same_as(op->b)) { \
420 return GetRef<PrimExpr>(op); \
421 } else { \
422 return FUNC(a, b); \
423 } \
424 }

426#define DEFINE_CMPOP_EXPR_MUTATE_WITH_TYPE_MATCH(OP, FUNC) \
427 PrimExpr DataTypeRewriter::VisitExpr_(const OP* op) { \
428 bool is_index = is_index_; \
429 bool rewrite = is_condition_ && op->a->dtype.is_int() && op->b->dtype.is_int(); \
430 if (rewrite) { \
431 is_index_ = true; \
432 } \
433 PrimExpr a = this->VisitExpr(op->a); \
434 PrimExpr b = this->VisitExpr(op->b); \
435 is_index_ = is_index; \
436 if (a.same_as(op->a) && b.same_as(op->b)) { \
437 return GetRef<PrimExpr>(op); \
438 } else { \
439 return FUNC(a, b); \
440 } \
441 }

459PrimExpr DataTypeRewriter::VisitExpr_(const CallNode* op) {
460 // handle if_then_else condition
461 if (op->op.same_as(builtin::if_then_else())) {
462 bool is_condition = is_condition_;
463 is_condition_ = true;
464 PrimExpr cond = VisitExpr(op->args[0]);
465 is_condition_ = is_condition;
466 return if_then_else(cond, VisitExpr(op->args[1]), VisitExpr(op->args[2]));
467 }
468
469 PrimExpr e = StmtExprMutator::VisitExpr_(op);
470 op = e.as<CallNode>();
471 ICHECK(op != nullptr) << "Expected type to be CallNode"
472 << ", but get " << e->GetTypeKey();
473 if (op->op.same_as(builtin::shift_right())) {
474 return op->args[0] >> op->args[1];
475 } else if (op->op.same_as(builtin::shift_left())) {
476 return op->args[0] << op->args[1];
477 } else if (op->op.same_as(builtin::bitwise_and())) {
478 return op->args[0] & op->args[1];
479 } else if (op->op.same_as(builtin::bitwise_or())) {
480 return op->args[0] | op->args[1];
481 } else if (op->op.same_as(builtin::bitwise_xor())) {
482 return op->args[0] ^ op->args[1];
483 } else if (op->op.same_as(builtin_pow_)) {
484 return pow(op->args[0], op->args[1]);
485 }
486
487 return e;
488}

490Stmt NarrowDataType(Stmt stmt, int target_bits) { return DataTypeRewriter(target_bits)(stmt); }

492namespace transform {

494Pass NarrowDataType(int target_bits) {
495 auto pass_func = [target_bits](PrimFunc f, IRModule m, PassContext ctx) {
496 auto* n = f.CopyOnWrite();
497 n->body = DataTypeRewriter(target_bits)(std::move(n->body));
498 return f;
499 };
500 return CreatePrimFuncPass(pass_func, 0, "tir.NarrowDataType", {});
501}
