41class UpdatePointerStorageScopeAllReduce final : public UpdatePointerStorageScope {
42 public:
43 explicit UpdatePointerStorageScopeAllReduce(
44 const std::unordered_map<const VarNode*, String>& new_storage_scopes)
45 : UpdatePointerStorageScope(new_storage_scopes) {}
46
47 Stmt VisitStmt_(const AllocateNode* op) final {
48 auto remapped = Downcast<Var>(StmtExprMutator::VisitExpr(op->buffer_var));
49 auto new_scope = GetPtrStorageScope(remapped);
50 if (new_scope != GetPtrStorageScope(op->buffer_var)) {
51 Stmt body = StmtExprMutator::VisitStmt(op->body);
52 if (new_scope == "shared") {
53 // use volatile access to shared buffer.
54 body = AttrStmt(remapped, attr::volatile_scope, 1, body);
55 }
56 return Allocate(remapped, op->dtype, op->extents, op->condition, body);
57 }
58 return StmtExprMutator::VisitStmt_(op);
59 }
60};

62class ThreadAllreduceBuilder final : public StmtExprMutator {
63 public:
64 explicit ThreadAllreduceBuilder(const TargetNode* target)
65 : target_(target),
66 warp_size_(target->GetAttr<Integer>("thread_warp_size", 1).value().IntValue()) {}
67
68 Stmt VisitStmt_(const AttrStmtNode* op) final {
69 if (op->attr_key == attr::thread_extent) {
70 thread_extents_.push_back(op);
71 Stmt ret = StmtExprMutator::VisitStmt_(op);
72 thread_extents_.pop_back();
73 return ret;
74 } else if (op->attr_key == attr::reduce_scope) {
75 const CommReducerNode* combiner = op->node.as<CommReducerNode>();
76 ICHECK(combiner);
77 reduce_combiner_.push_back(combiner);
78 Stmt ret = StmtExprMutator::VisitStmt_(op);
79 reduce_combiner_.pop_back();
80 return ret;
81 } else {
82 return StmtExprMutator::VisitStmt_(op);
83 }
84 }
85 Stmt VisitStmt_(const EvaluateNode* op) final {
86 Stmt stmt = StmtExprMutator::VisitStmt_(op);
87 op = stmt.as<EvaluateNode>();
88 const CallNode* call = op->value.as<CallNode>();
89 if (call && call->op.same_as(builtin::tvm_thread_allreduce())) {
90 return MakeAllreduce(call);
91 } else {
92 return stmt;
93 }
94 }
95 Stmt VisitStmt_(const AllocateNode* op) final {
96 Stmt stmt = StmtExprMutator::VisitStmt_(op);
97 op = stmt.as<AllocateNode>();
98 auto it = alloc_remap_.find(op->buffer_var.get());
99 if (it != alloc_remap_.end()) {
100 const AllocateNode* repl = it->second.as<AllocateNode>();
101 if (warp_allocs_.count(repl)) {
102 new_storage_scopes_[repl->buffer_var.get()] = "local";
103 } else {
104 new_storage_scopes_[repl->buffer_var.get()] = "shared";
105 }
106 return Allocate(repl->buffer_var, repl->dtype, repl->extents, repl->condition, op->body);
107 } else {
108 return stmt;
109 }
110 }
111
112 PrimExpr VisitExpr_(const LoadNode* op) final {
113 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
114 return PrimExpr();
115 }
116
117 Stmt VisitStmt_(const StoreNode* op) final {
118 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
119 return Stmt();
120 }
121
122 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
123 {
124 auto it = load_remap_.find(op->buffer->data.get());
125 if (it != load_remap_.end()) {
126 for (const auto& index : op->indices) {
127 ICHECK(is_zero(index));
128 }
129 return it->second;
130 }
131 }
132
133 BufferLoad load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
134 op = load.get();
135
136 {
137 auto it = buf_remap_.find(op->buffer.get());
138 if (it != buf_remap_.end()) {
139 return BufferLoad(it->second, op->indices, op->span);
140 }
141 }
142
143 {
144 auto it = var_remap_.find(op->buffer->data.get());
145 if (it != var_remap_.end()) {
146 Buffer remapped_buffer(it->second, op->buffer->dtype, op->buffer->shape,
147 op->buffer->strides, op->buffer->elem_offset, op->buffer->name,
148 op->buffer->data_alignment, op->buffer->offset_factor,
149 op->buffer->buffer_type, op->buffer->axis_separators,
150 op->buffer->span);
151 buf_remap_[op->buffer.get()] = remapped_buffer;
152 return BufferLoad(remapped_buffer, op->indices, op->span);
153 }
154 }
155 return StmtExprMutator::VisitExpr_(op);
156 }
157
158 Stmt VisitStmt_(const BufferStoreNode* op) final {
159 BufferStore store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
160
161 auto it = store_remap_.find(store->buffer.get());
162 if (it != store_remap_.end()) {
163 for (const auto& index : op->indices) {
164 ICHECK(is_zero(index));
165 }
166
167 auto writer = store.CopyOnWrite();
168 writer->buffer = it->second;
169 return std::move(store);
170 }
171
172 {
173 auto it = buf_remap_.find(store->buffer.get());
174 if (it != buf_remap_.end()) {
175 return BufferStore(it->second, store->value, store->indices, store->span);
176 }
177 }
178
179 {
180 auto it = var_remap_.find(store->buffer->data.get());
181 if (it != var_remap_.end()) {
182 Buffer remapped_buffer(it->second, store->buffer->dtype, store->buffer->shape,
183 store->buffer->strides, store->buffer->elem_offset,
184 store->buffer->name, store->buffer->data_alignment,
185 store->buffer->offset_factor, store->buffer->buffer_type,
186 store->buffer->axis_separators, store->buffer->span);
187 buf_remap_[store->buffer.get()] = remapped_buffer;
188 return BufferStore(remapped_buffer, store->value, store->indices, store->span);
189 }
190 }
191
192 return std::move(store);
193 }

198 // Thread entry
199 struct ThreadEntry {
200 runtime::ThreadScope scope;
201 IterVar iv;
202 int extent;
203 // comparator
204 bool operator<(const ThreadEntry& other) const {
205 return scope.dim_index < other.scope.dim_index;
206 }
207 };

209 // make allreduce.
210 Stmt MakeAllreduce(const CallNode* call) {
211 ICHECK(!reduce_combiner_.empty());
212 const CommReducerNode* combiner = reduce_combiner_.back();
213 size_t size = combiner->result.size();
214
215 const IntImmNode* size_of_args = call->args[0].as<IntImmNode>();
216 ICHECK(size_of_args) << call->args[0]->GetTypeKey();
217 ICHECK_EQ(size, size_of_args->value);
218 Array<PrimExpr> inits = combiner->identity_element;
219 std::vector<PrimExpr> values(size);
220 std::vector<DataType> types(size);
221 PrimExpr cond = call->args[size + 1];
222 for (size_t idx = 0; idx < size; ++idx) {
223 values[idx] = call->args[1 + idx];
224 if (!is_one(cond)) {
225 values[idx] = Select(cond, values[idx], inits[idx]);
226 }
227 types[idx] = values[idx].dtype();
228 }
229 std::vector<Buffer> buffers(size);
230 for (size_t idx = 0; idx < size; ++idx) {
231 PrimExpr arg = call->args[2 + size + idx];
232 // Loads from boolean buffers may have cast nodes inserted by
233 // earlier passes.
234 if (auto cast = arg.as<CastNode>()) {
235 arg = cast->value;
236 }
237 buffers[idx] = Downcast<BufferLoad>(arg)->buffer;
238 }
239
240 std::unordered_set<const VarNode*> reduce_set;
241 for (size_t i = 2 + 2 * size; i < call->args.size(); ++i) {
242 const VarNode* v = call->args[i].as<VarNode>();
243 // The simply optimization replace a iteration variable with a constant
244 // when extent of the iteration is 1. As threaded IterVar always started from 0,
245 // we can just ignore this variable in this case.
246 if (v) {
247 reduce_set.insert(v);
248 } else {
249 ICHECK(call->args[i].as<IntImmNode>() && call->args[i].as<IntImmNode>()->value == 0)
250 << "arg" << i << "should be a VarNode or IntImmNode";
251 }
252 }
253
254 size_t nmatch = 0;
255 std::vector<ThreadEntry> vred, vpar;
256 for (const AttrStmtNode* attr : thread_extents_) {
257 ThreadEntry e;
258 IterVar iv = Downcast<IterVar>(attr->node);
259 e.scope = runtime::ThreadScope::Create(iv->thread_tag);
260 e.iv = iv;
261 ICHECK_LE(e.scope.rank, 1);
262 ICHECK_GE(e.scope.dim_index, 0) << "vthread do not work with cross thread reduction";
263 if (e.scope.rank == 1) {
264 const auto* ptr = attr->value.as<IntImmNode>();
265 ICHECK(ptr) << "Need constant extent for reduce set " << iv;
266 e.extent = static_cast<int>(ptr->value);
267 // ignore variables equal to 0
268 if (e.extent == 1) {
269 continue;
270 }
271
272 if (reduce_set.count(iv->var.get())) {
273 vred.push_back(e);
274 ++nmatch;
275 } else {
276 vpar.push_back(e);
277 }
278 }
279 }
280 ICHECK_EQ(nmatch, reduce_set.size()) << "Not all reduce index are presented in the context";
281 std::sort(vred.begin(), vred.end());
282 std::sort(vpar.begin(), vpar.end());
283 // the size of each index.
284 int reduce_extent, group_extent;
285 PrimExpr reduce_index = FlattenThread(vred, &reduce_extent);
286 PrimExpr group_index = FlattenThread(vpar, &group_extent);
287
288 // the longest contiguous reduce extent after flattening
289 int contiguous_reduce_extent = 1;
290 std::vector<std::tuple<int, int, bool>> block_threads; // tuple(dim_index, extent, is_reduce)
291 for (const ThreadEntry& thr : vred) {
292 if (thr.scope.rank == 1) { // threadIdx
293 block_threads.emplace_back(thr.scope.dim_index, thr.extent, true);
294 }
295 }
296 for (const ThreadEntry& thr : vpar) {
297 if (thr.scope.rank == 1) { // threadIdx
298 block_threads.emplace_back(thr.scope.dim_index, thr.extent, false);
299 }
300 }
301 // sort according to dim_index
302 std::sort(block_threads.begin(), block_threads.end());
303 for (auto&& thr_attr : block_threads) {
304 auto [dim_index, extent, is_reduce] = thr_attr;
305 (void)dim_index; // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=81767
306 if (is_reduce) {
307 contiguous_reduce_extent *= extent;
308 } else {
309 break;
310 }
311 }
312
313 std::vector<Stmt> seq;
314 std::vector<Var> shared_buffer_vars(size);
315 std::vector<Buffer> shared_bufs(size);
316 std::vector<Buffer> local_bufs;
317 //
318 // This is an optimization. For small reduction sizes, it may be beneficial
319 // for a single warp to performance the entire reduction. No trips to shared
320 // memory and no cross warp synchronizations are required.
321 // The following code emits the reduction as follows:
322 //
323 // Allocate reduction vars v[i], i = 0..size-1
324 //
325 // for offset from WARP_SIZE to 1 by 2
326 //
327 // a <- load(v[i])
328 // b <- shuffle_down(load(v[i], offset))
329 // v[i] <- reduction(a, b)
330 //
331 // broadcast results from lane 0 to all other lanes and store
332 // the final reduction result to the proper location.
333 //
334 if (is_warp_reduction(types, group_extent, reduce_extent, contiguous_reduce_extent)) {
335 ICHECK_LE(reduce_extent, warp_size_) << "not a warp reduction";
336 //
337 // This is the index to the reduction variable, one reduction
338 // variable per warp. Local scope seems easier to reason without
339 // relying on a pattern match pass to fix it later.
340 Array<PrimExpr> zero_indices = {0};
341
342 for (size_t idx = 0; idx < size; ++idx) {
343 Array<PrimExpr> shape = {1};
344
345 Buffer buffer = decl_buffer(shape, types[idx], "red_buf" + std::to_string(idx));
346 Var buffer_var = buffer->data;
347
348 shared_buffer_vars[idx] = buffer_var;
349 shared_bufs[idx] = buffer;
350
351 PrimExpr pred = const_true(types[idx].lanes());
352 seq.emplace_back(BufferStore(shared_bufs[idx], values[idx], zero_indices));
353
354 // Uses a local variable to store the shuffled data. Later
355 // on, an allocation will be built for this local variable.
356 local_bufs.push_back(decl_buffer(shape, types[idx], "t" + std::to_string(idx)));
357 }
358
359 // The mask for this reducer, as this reducer may sit inside
360 // a divergent control flow. Here it uses a variable to cache the current
361 // active channels.
362 //
363 DataType mask_dtype = DataType::UInt(32);
364 Buffer mask_buffer = decl_buffer({1}, mask_dtype, "mask");
365 {
366 PrimExpr mask = Call(mask_dtype, builtin::tvm_warp_activemask(), {});
367 if (group_extent > 1) {
368 mask = mask & (((1 << reduce_extent) - 1) << (reduce_extent * group_index));
369 }
370 seq.emplace_back(BufferStore(mask_buffer, mask, zero_indices));
371 // Push the buffer description. Later this will have an
372 // allocation built for it.
373 local_bufs.push_back(mask_buffer);
374 }
375
376 // Emit reductions within a warp.
377 int start_offset = 1;
378 while (start_offset * 2 < reduce_extent) {
379 start_offset *= 2;
380 }
381 for (int offset = start_offset; offset > 0; offset /= 2) {
382 // Load reduction values, no synchronization needed.
383 Array<PrimExpr> a, b;
384 for (size_t i = 0; i < size; ++i) {
385 Buffer shared_buf = shared_bufs[i];
386 BufferLoad val(shared_buf, zero_indices);
387 ICHECK_EQ(val->dtype, types[i]);
388 a.push_back(val);
389
390 // __shfl_*sync calls shall not appear in if_then_else expressions
391 // as this is causing extra divergency. E.g.
392 //
393 // v1 = (v2 < v3) ? v3 : __shfl_sync(mask, v1, 0);
394 //
395 // behaves differently from
396 //
397 // int t = __shfl_sync(mask, v1, 0);
398 // v1 = (v2 < v3) ? v3 : t;
399 //
400 // The former may cause dead lock as there is a divergent
401 // branch with a warp sync call inside.
402 //
403 PrimExpr other = WarpShuffle(builtin::tvm_warp_shuffle_down(), mask_buffer, val, offset);
404 Buffer local_buf = local_bufs[i];
405 Stmt s = BufferStore(local_buf, other, zero_indices);
406 seq.push_back(s);
407
408 BufferLoad load = BufferLoad(local_buf, zero_indices);
409 ICHECK_EQ(load->dtype, types[i]);
410 b.push_back(load);
411 }
412
413 // Do reductions.
414 Array<PrimExpr> ret = (*combiner)(a, b);
415
416 // Store the reduction result to itself.
417 std::vector<Stmt> stores(size);
418 for (size_t i = 0; i < size; ++i) {
419 Buffer buf = shared_bufs[i];
420 stores[i] = BufferStore(buf, ret[i], zero_indices);
421 }
422
423 // During the sub-warp reduction, values from inactive threads could be read,
424 // which is an undefined behavior according to the cuda document.
425 //
426 // In practise, the return value are usually 0, which does no harm to sum reduction.
427 // However, the result can be incorrect in max or prod reduction.
428 // Therefore an additional range check has to be performed to ensure the correctness.
429 if (offset * 2 > reduce_extent) {
430 PrimExpr cond = reduce_index + offset < reduce_extent;
431 seq.push_back(IfThenElse(cond, SeqStmt::Flatten(stores)));
432 } else {
433 seq.push_back(SeqStmt::Flatten(stores));
434 }
435 }
436
437 // Broadcast the reduction result from lane 0 to all other lanes.
438 // This avoids to emit predicated stores, as all threads are
439 // uniformly writting the same result.
440 //
441 for (size_t i = 0; i < size; ++i) {
442 Buffer buf = shared_bufs[i];
443 PrimExpr val = BufferLoad(buf, zero_indices);
444 ICHECK_EQ(val->dtype, types[i]);
445 PrimExpr splat =
446 WarpShuffle(builtin::tvm_warp_shuffle(), mask_buffer, val, reduce_extent * group_index);
447 seq.push_back(BufferStore(buf, splat, zero_indices));
448 }
449
450 // Update existing allocations.
451 for (size_t i = 0; i < size; ++i) {
452 ICHECK(!load_remap_.count(buffers[i]->data.get()));
453 PrimExpr pred = const_true(types[i].lanes());
454 Buffer buf = shared_bufs[i];
455 PrimExpr val = BufferLoad(buf, zero_indices);
456 ICHECK_EQ(val->dtype, types[i]);
457 load_remap_[buffers[i]->data.get()] = val;
458 store_remap_[buffers[i].get()] = buf;
459 Array<PrimExpr> extents{PrimExpr(1)};
460 auto node = Allocate(buf->data, types[i], extents, pred, Evaluate(0));
461 alloc_remap_[buffers[i]->data.get()] = node;
462 var_remap_[buffers[i]->data.get()] = buf->data;
463 warp_allocs_.insert(node.get());
464 }
465 } else {
466 if (reduce_extent == 1) {
467 // special case, no reduction is needed.
468 std::vector<Stmt> stores;
469 for (size_t i = 0; i < size; ++i) {
470 stores.push_back(BufferStore(buffers[i], values[i], {0}));
471 }
472 return SeqStmt::Flatten(stores);
473 }
474 // This sync is necessary because there might be incomplete read of
475 // previous iteration on the same buffer.
476 seq.emplace_back(SyncThread("shared"));
477 for (size_t idx = 0; idx < size; ++idx) {
478 Buffer buffer = decl_buffer({1}, types[idx], "red_buf" + std::to_string(idx));
479
480 shared_bufs[idx] = buffer;
481 shared_buffer_vars[idx] = buffer->data;
482
483 PrimExpr pred = const_true(types[idx].lanes());
484 seq.emplace_back(BufferStore(shared_bufs[idx], values[idx],
485 {BufIndex(reduce_index, group_index, reduce_extent)}));
486 }
487 seq.emplace_back(SyncThread("shared"));
488 seq.emplace_back(MakeBufAllreduce(combiner, types, shared_bufs, reduce_index, group_index,
489 reduce_extent, group_extent, contiguous_reduce_extent));
490 for (size_t idx = 0; idx < size; ++idx) {
491 ICHECK(!load_remap_.count(buffers[idx]->data.get()));
492 PrimExpr pred = const_true(types[idx].lanes());
493 BufferLoad load(shared_bufs[idx],
494 {BufIndex(make_zero(reduce_index.dtype()), group_index, reduce_extent)});
495 ICHECK_EQ(load->dtype, types[idx]);
496 load_remap_[buffers[idx]->data.get()] = load;
497 alloc_remap_[buffers[idx]->data.get()] =
498 Allocate(shared_bufs[idx]->data, types[idx],
499 {PrimExpr(group_extent), PrimExpr(reduce_extent)}, pred, Evaluate(0));
500 var_remap_[buffers[idx]->data.get()] = shared_bufs[idx]->data;
501 store_remap_[buffers[idx].get()] = shared_bufs[idx];
502 }
503 }
504
505 // Fix all local allocations as all statements are built.
506 Stmt body = SeqStmt::Flatten(seq);
507 for (Buffer buf : local_bufs) {
508 body = Allocate(buf->data, buf->dtype, buf->shape, const_true(buf->dtype.lanes()), body);
509 new_storage_scopes_[buf->data.get()] = "local";
510 }
511
512 return body;
513 }
514
515 // make allreduce.
516 Stmt MakeBufAllreduce(const CommReducerNode* combiner, const std::vector<DataType>& types,
517 const Array<Buffer>& shared_bufs, PrimExpr reduce_index,
518 PrimExpr group_index, int reduce_extent, int group_extent,
519 int contiguous_reduce_extent) {
520 // Get next power of two
521 int reduce_align = 1;
522 while (reduce_extent > reduce_align) {
523 reduce_align = reduce_align << 1;
524 }
525 ICHECK_GT(reduce_align, 1);
526 std::vector<Stmt> seq;
527
528 size_t size = shared_bufs.size();
529 PrimExpr buf_index = BufIndex(reduce_index, group_index, reduce_extent);
530 // make reduction
531 auto fload = [&](int offset) {
532 Array<PrimExpr> a, b;
533 for (size_t i = 0; i < size; ++i) {
534 BufferLoad b_load(shared_bufs[i],
535 {BufIndex(reduce_index + offset, group_index, reduce_extent)});
536 ICHECK_EQ(b_load->dtype, types[i]);
537 b.push_back(b_load);
538
539 BufferLoad a_load(shared_bufs[i], {buf_index});
540 ICHECK_EQ(a_load->dtype, types[i]);
541 a.push_back(a_load);
542 }
543 Array<PrimExpr> ret = (*combiner)(a, b);
544 return ret;
545 };
546 auto fstore = [&](const Array<PrimExpr>& ret) {
547 std::vector<Stmt> stores(size);
548 for (size_t i = 0; i < size; ++i) {
549 stores[i] = BufferStore(shared_bufs[i], ret[i], {buf_index});
550 }
551 return SeqStmt::Flatten(stores);
552 };
553 auto freduce = [&](int offset) {
554 auto ret = fload(offset);
555 return fstore(ret);
556 };
557 // Step one, check for
558 if (reduce_align > reduce_extent) {
559 // reduction with the boundary condition
560 reduce_align = reduce_align >> 1;
561 PrimExpr cond = reduce_index < (reduce_extent - reduce_align);
562 seq.emplace_back(IfThenElse(cond, freduce(reduce_align)));
563 seq.emplace_back(SyncThread("shared"));
564 }
565
566 // normal synchronization
567 bool warp_align = group_extent == 1 || contiguous_reduce_extent % warp_size_ == 0;
568 while (reduce_align > contiguous_reduce_extent || reduce_align > warp_size_ || !warp_align) {
569 if (reduce_align == 1) {
570 break;
571 }
572 reduce_align = reduce_align >> 1;
573 PrimExpr cond = reduce_index < reduce_align;
574 seq.emplace_back(IfThenElse(cond, freduce(reduce_align)));
575 seq.emplace_back(SyncThread("shared"));
576 }
577 // in warp synchronization.
578 if (reduce_align > 1) {
579 PrimExpr in_warp_cond = reduce_index < (reduce_align >> 1);
580
581 std::vector<Stmt> in_warp_seq;
582
583 while (reduce_align > 1) {
584 reduce_align = reduce_align >> 1;
585
586 // freduce can read/write to the same memory location. For
587 // example, with reduce_align of 4, threadIdx 3 reads from
588 // memory location 7 as threadIdx 7 is writing to it.
589 // Therefore, we need to separate out the load from the store
590 // with a memory barrier in-between. This isn't necessary for
591 // the earlier normal synchronization, because those are each
592 // protected by an if-statement. The if-statement is avoided
593 // here to reduce thread divergence.
594 auto loads = fload(reduce_align);
595
596 Array<Var> in_warp_local_vars;
597 for (auto expr : loads) {
598 Var var(
599 "w_" + std::to_string(reduce_align) + "_" + std::to_string(in_warp_local_vars.size()),
600 expr->dtype);
601 in_warp_local_vars.push_back(var);
602 }
603
604 std::vector<Stmt> in_let_statement;
605 in_let_statement.emplace_back(SyncThread("warp"));
606 in_let_statement.emplace_back(
607 fstore({in_warp_local_vars.begin(), in_warp_local_vars.end()}));
608 in_let_statement.emplace_back(SyncThread("warp"));
609
610 Stmt body = SeqStmt::Flatten(in_let_statement);
611 for (size_t i = 0; i < size; i++) {
612 body = LetStmt(in_warp_local_vars[i], loads[i], body);
613 }
614 in_warp_seq.push_back(body);
615 }
616
617 Stmt warp_body = SeqStmt::Flatten(in_warp_seq);
618
619 seq.emplace_back(IfThenElse(in_warp_cond, warp_body));
620 seq.emplace_back(SyncThread("shared"));
621 }
622 return SeqStmt::Flatten(seq);
623 }
624 // Flatten the thread index.
625 // Also return a warp number,
626 PrimExpr FlattenThread(const std::vector<ThreadEntry>& tvec, int* out_total_extent) {
627 int& total_extent = *out_total_extent;
628 total_extent = 1;
629 if (tvec.size() == 0) {
630 return make_zero(DataType::Int(32));
631 }
632
633 PrimExpr ret;
634 for (const ThreadEntry& e : tvec) {
635 if (ret.defined()) {
636 ret = ret + e.iv->var * total_extent;
637 } else {
638 ICHECK_EQ(total_extent, 1);
639 ret = e.iv->var;
640 }
641 total_extent *= e.extent;
642 }
643 return ret;
644 }
645 // The local buffer index.
646 PrimExpr BufIndex(PrimExpr reduce_index, PrimExpr group_index, int reduce_extent) {
647 if (!is_zero(group_index)) {
648 return analyzer_.Simplify(group_index * reduce_extent + reduce_index);
649 } else {
650 return reduce_index;
651 }
652 }
653 // sync thread op.
654 static Stmt SyncThread(const std::string& sync) {
655 return Evaluate(Call(DataType::Int(32), builtin::tvm_storage_sync(), {StringImm(sync)}));
656 }

658 // Emit warp shuffle calls.
659 PrimExpr WarpShuffle(const Op& op, Buffer mask_buffer, PrimExpr val, PrimExpr delta_or_lane) {
660 Array<PrimExpr> indices = {0};
661 PrimExpr mask = BufferLoad(mask_buffer, indices);
662 PrimExpr width = IntImm(DataType::Int(32), warp_size_);
663 Array<PrimExpr> args{mask, val, delta_or_lane, width, width};
664 return Call(val.dtype(), op, args);
665 }

667 // Check if we can use warp level reduction.
668 //
669 // Note: The ROCm backend will only have warp reductions for now.
670 // Also, the warp/wavefront size differs (64 on rocm, 32 on cuda).
671 bool is_warp_reduction(const std::vector<DataType>& types, int group_extent, int reduce_extent,
672 int contiguous_reduce_extent) const {
673 // Only cuda target supports warp reductions.
674 if ((target_->kind->name != "cuda") && (target_->kind->name != "rocm")) return false;
675
676 // rocm only supports 32 bit operands for shuffling at the moment
677 if ((target_->kind->name == "rocm") &&
678 (std::any_of(types.begin(), types.end(), [](DataType ty) {
679 if (ty.is_vector()) return true;
680 return ty.bits() != 32;
681 }))) {
682 return false;
683 }
684
685 // Supported types:
686 // {u}int, {u}long, {u}long long, float, double, half/half2
687 if (std::any_of(types.begin(), types.end(), [](DataType ty) {
688 if (ty.is_float16()) return ty.lanes() > 2;
689 if (ty.is_vector()) return true;
690 return ty.bytes() < 4 || ty.bytes() > 8;
691 })) {
692 return false;
693 }
694 if (thread_extents_.empty()) {
695 return false;
696 }
697
698 // reduce region must be contiguous.
699 if (contiguous_reduce_extent != reduce_extent) {
700 return false;
701 }
702
703 // whether reduce_extent and group_extent are vaild for warp reduction.
704 if (target_->kind->name == "rocm") {
705 return reduce_extent == warp_size_;
706 } else { // target_->kind->name == "cuda"
707 if (reduce_extent == 1) {
708 return false; // no need to warp reduce
709 } else {
710 if (warp_size_ % reduce_extent == 0) {
711 return true; // warp size is multiple of reduce extent
712 } else {
713 return group_extent == 1 && reduce_extent <= warp_size_;
714 }
715 }
716 }
717 }
718
719 // The target.
720 const TargetNode* target_ = nullptr;
721
722 // The warp size of the device.
723 int warp_size_{1};
724
725 // surrounding scope of thread extent.
726 std::vector<const AttrStmtNode*> thread_extents_;
727 std::vector<const CommReducerNode*> reduce_combiner_;
728 // The load remap
729 std::unordered_map<const VarNode*, PrimExpr> load_remap_;
730 // The store remap
731 std::unordered_map<const BufferNode*, Buffer> store_remap_;
732 // Allocate remap
733 std::unordered_map<const VarNode*, Stmt> alloc_remap_;
734 // BufferVar remap
735 std::unordered_map<const VarNode*, Var> var_remap_;
736 // Buffer remap
737 std::unordered_map<const BufferNode*, Buffer> buf_remap_;
738 // Allocate from warp reductions
739 std::unordered_set<const void*> warp_allocs_;
740 // Internal analyzer
741 arith::Analyzer analyzer_;
742};
