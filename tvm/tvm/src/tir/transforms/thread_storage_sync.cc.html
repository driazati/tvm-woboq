40class ThreadSyncPlanner : public StorageAccessVisitor {
41 public:
42 explicit ThreadSyncPlanner(StorageScope sync_scope) : sync_scope_(sync_scope) {}
43
44 // The syncs inserted before each statement
45 std::unordered_set<const Object*> syncs_inserted_;
46
47 protected:
48 bool Enabled(const VarNode* buf, const StorageScope& scope) const final {
49 return in_device_env() && scope == sync_scope_;
50 }
51 // Plan the sync
52 std::vector<AccessEntry> Summarize(std::vector<StmtEntry> seq, const ForNode* loop) final {
53 // Unsynced reads and writes
54 std::vector<AccessEntry> reads;
55 std::vector<AccessEntry> writes;
56 // if it is a loop, rotate two times to consider effect of loop.
57 // simulation based approach to find dependenceies
58 for (size_t i = 0; i < seq.size(); ++i) {
59 const StmtEntry& s = seq[i];
60 // check if sync before statement is needed.
61 bool sync_before_stmt = (syncs_inserted_.count(s.stmt) != 0);
62 // Apply the syncs added already.
63 if (sync_before_stmt) {
64 reads.clear();
65 writes.clear();
66 }
67 for (const AccessEntry& acc : s.access) {
68 if (acc.type == kRead) {
69 if (FindConflict(writes, acc, false)) {
70 sync_before_stmt = true;
71 break;
72 }
73 } else if (acc.type == kWrite) {
74 if (FindConflict(reads, acc, false)) {
75 sync_before_stmt = true;
76 break;
77 }
78 } else if (acc.type == kSync) {
79 reads.clear();
80 writes.clear();
81 }
82 }
83 // If sync is inserted. remove the irrelevant things.
84 if (sync_before_stmt) {
85 reads.clear();
86 writes.clear();
87 }
88 // Add the read/write of current statement
89 for (const AccessEntry& acc : s.access) {
90 if (acc.type == kRead) {
91 reads.push_back(acc);
92 } else if (acc.type == kWrite) {
93 writes.push_back(acc);
94 } else if (acc.type == kSync) {
95 reads.clear();
96 writes.clear();
97 }
98 }
99 if (sync_before_stmt) {
100 ICHECK_EQ(condition_counter(), 0) << "Cannot insert syncs inside condition";
101 syncs_inserted_.insert(s.stmt);
102 }
103 }
104 if (loop != nullptr) {
105 for (size_t i = 0; i < seq.size(); ++i) {
106 const StmtEntry& s = seq[i];
107 if (syncs_inserted_.count(s.stmt) != 0) break;
108 if (reads.empty() && writes.empty()) break;
109 bool sync_before_stmt = false;
110 for (const AccessEntry& acc : s.access) {
111 if (acc.type == kRead) {
112 if (FindConflict(writes, acc, true)) {
113 sync_before_stmt = true;
114 break;
115 }
116 } else if (acc.type == kWrite) {
117 if (FindConflict(reads, acc, true)) {
118 sync_before_stmt = true;
119 break;
120 }
121 } else if (acc.type == kSync) {
122 reads.clear();
123 writes.clear();
124 }
125 }
126 if (sync_before_stmt) {
127 ICHECK_EQ(condition_counter(), 0) << "Cannot insert syncs inside condition";
128 syncs_inserted_.insert(s.stmt);
129 break;
130 }
131 }
132 }
133 // return the exposed entries, remove unecessary ones.
134 int sync_count = 0;
135 // head are before first sync, tail are after last sync
136 std::vector<AccessEntry> head, tail;
137 AccessEntry esync;
138 esync.threads = this->env_threads();
139 esync.type = kSync;
140 esync.scope = sync_scope_;
141
142 for (const StmtEntry& s : seq) {
143 if (syncs_inserted_.count(s.stmt)) {
144 if (sync_count != 0) {
145 tail.clear();
146 } else {
147 head.push_back(esync);
148 }
149 ++sync_count;
150 }
151 for (const AccessEntry& acc : s.access) {
152 if (acc.type == kSync) {
153 if (sync_count != 0) {
154 tail.clear();
155 } else {
156 head.push_back(esync);
157 }
158 ++sync_count;
159 } else {
160 if (sync_count != 0) {
161 tail.push_back(acc);
162 } else {
163 head.push_back(acc);
164 }
165 }
166 }
167 }
168 head.insert(head.end(), tail.begin(), tail.end());
169 if (loop != nullptr) {
170 // clear double buffer flag after a loop is finished.
171 for (AccessEntry& e : head) {
172 e.double_buffer_write = false;
173 }
174 }
175 return head;
176 }
177
178 private:
179 // find conflicting entry in vec.
180 bool FindConflict(const std::vector<AccessEntry>& prev, const AccessEntry& curr,
181 bool loop_carry) {
182 for (const AccessEntry& x : prev) {
183 if (FindConflict(x, curr, loop_carry)) {
184 return true;
185 }
186 }
187 return false;
188 }
189
190 bool FindConflict(const AccessEntry& prev, const AccessEntry& curr, bool loop_carry) {
191 // Access to different buffers does not conflict.
192 if (!prev.buffer.same_as(curr.buffer)) {
193 return false;
194 }
195
196 // Assumes no race between threads
197 // Same index value means no conflicts
198 // TODO(tqchen) more standard set based testing.
199 bool has_same_index = true;
200 for (size_t i = 0; i < prev.touched.size(); i++) {
201 const auto& prev_intset = prev.touched[i];
202 const auto& curr_intset = curr.touched[i];
203
204 bool provably_same_index =
205 prev_intset.IsSinglePoint() && curr_intset.IsSinglePoint() &&
206 ExprDeepEqual()(prev_intset.PointValue(), curr_intset.PointValue());
207
208 if (!provably_same_index) {
209 has_same_index = false;
210 break;
211 }
212 }
213 if (has_same_index) {
214 return false;
215 }
216
217 // If this is a read into a double buffer that was previously
218 // swapped out, then it doesn't conflict.
219 if (prev.double_buffer_write && curr.type == kRead && !loop_carry) {
220 return false;
221 }
222
223 // If nothing else allows sharing the same buffer, then they are
224 // in conflict.
225 return true;
226 }
227
228 private:
229 // synchronization scope
230 StorageScope sync_scope_;
231};
232
233// There are cases where necessary syncthreads is not inserted by ThreadSyncInserter.
234// For example, syncthreads is needed after async_wait_queue in the second loop below,
235// but since ThreadSyncInserter is not aware of the asynchronous semantics, it cannot tell
236// that the syncthreads is needed there.
237//
238// // Pipeline prologue
239// for i in range(125):
240// async_commit_queue(0):
241// async_scope:
242// shared[(i + 3) % 4] = ...
243// ...
244//
245// // Pipeline Epilogue
246// for i in range(3):
247// async_wait_queue(0, 2 - i):
248// local[...] = shared[(i + 125) % 4]
249
250// This class adds syncthreads after all async_wait_queue. That includes syncthreads that
251// can be inserted by ThreadSyncInserter as well, but ThreadSyncInserter will not insert
252// duplicate syncthreads if it finds an existing one at the synchronization point.
253class ThreadSyncAfterWaitQueueInserter : public StmtExprMutator {
254 public:
255 explicit ThreadSyncAfterWaitQueueInserter(StorageScope sync_scope) : sync_scope_(sync_scope) {}
256
257 Stmt VisitStmt_(const AttrStmtNode* op) final {
258 if (op->attr_key == attr::async_wait_queue_scope) {
259 auto sync = Evaluate(Call(DataType::Int(32), builtin::tvm_storage_sync(),
260 {StringImm(sync_scope_.to_string())}));
261 auto inner = op->body.as<AttrStmtNode>();
262 ICHECK(inner && inner->attr_key == tir::attr::async_wait_inflight_count);
263 auto zero = make_zero(DataType::Int(32));
264 auto new_body = SeqStmt({sync, inner->body});
265 return AttrStmt(zero, tir::attr::async_wait_queue_scope, op->value,
266 AttrStmt(zero, tir::attr::async_wait_inflight_count, inner->value, new_body));
267 }
268 return StmtExprMutator::VisitStmt_(op);
269 }
270
271 private:
272 StorageScope sync_scope_;
273};

275class ThreadSyncInserter : public StmtExprMutator {
276 public:
277 ThreadSyncInserter(StorageScope sync_scope, const std::unordered_set<const Object*>& syncs)
278 : sync_scope_(sync_scope), syncs_(syncs) {}
279
280 Stmt VisitStmt(const Stmt& stmt) final {
281 if (syncs_.size() == 0) return stmt;
282 if (syncs_.count(stmt.get())) {
283 Stmt barrier;
284 if (sync_scope_.rank == StorageRank::kGlobal) {
285 barrier = MakeGlobalBarrier();
286 } else {
287 barrier = Evaluate(Call(DataType::Int(32), builtin::tvm_storage_sync(),
288 {StringImm(sync_scope_.to_string())}));
289 }
290 // Mutate after query, to avoid stmt change.
291 auto ret = StmtExprMutator::VisitStmt(stmt);
292 ret = SeqStmt({barrier, ret});
293 return ret;
294 } else {
295 return StmtExprMutator::VisitStmt(stmt);
296 }
297 }
298 PrimExpr VisitExpr_(const LoadNode* op) final {
299 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
300 return PrimExpr();
301 }
302
303 Stmt VisitStmt_(const StoreNode* op) final {
304 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
305 return Stmt();
306 }
307 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
308 if (sync_scope_.rank == StorageRank::kGlobal &&
309 GetScope(op->buffer->data).rank == StorageRank::kGlobal) {
310 ++rw_stats_[op->buffer->data].read_count;
311 }
312 return StmtExprMutator::VisitExpr_(op);
313 }
314 Stmt VisitStmt_(const BufferStoreNode* op) final {
315 if (sync_scope_.rank == StorageRank::kGlobal &&
316 GetScope(op->buffer->data).rank == StorageRank::kGlobal) {
317 ++rw_stats_[op->buffer->data].write_count;
318 }
319 return StmtExprMutator::VisitStmt_(op);
320 }
321 Stmt VisitStmt_(const AttrStmtNode* op) final {
322 if (op->attr_key == attr::thread_extent) {
323 bool temp = true;
324 std::swap(temp, in_thread_env_);
325 thread_extents_.push_back(op);
326 Stmt ret = StmtExprMutator::VisitStmt_(op);
327 thread_extents_.pop_back();
328 std::swap(temp, in_thread_env_);
329 // first thread scope.
330 if (!in_thread_env_ && sync_scope_.rank == StorageRank::kGlobal) {
331 ret = InitGlobalBarrier(ret.as<AttrStmtNode>());
332 num_blocks_ = PrimExpr();
333 is_lead_ = PrimExpr();
334 }
335 return ret;
336 } else {
337 return StmtExprMutator::VisitStmt_(op);
338 }
339 }
340
341 PrimExpr VisitExpr_(const CallNode* op) final {
342 if (op->op.same_as(builtin::tvm_access_ptr())) {
343 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
344 op = expr.as<CallNode>();
345 ICHECK_EQ(op->args.size(), 5U);
346 Var buffer_var(GetRef<Var>(op->args[1].as<VarNode>()));
347 const IntImmNode* flag = op->args[4].as<IntImmNode>();
348 if ((flag->value & 1) && sync_scope_.rank == StorageRank::kGlobal &&
349 GetScope(buffer_var).rank == StorageRank::kGlobal) {
350 ++rw_stats_[buffer_var].read_count;
351 }
352 if (flag->value & 2 && sync_scope_.rank == StorageRank::kGlobal &&
353 GetScope(buffer_var).rank == StorageRank::kGlobal) {
354 ++rw_stats_[buffer_var].write_count;
355 }
356 return expr;
357 } else {
358 return StmtExprMutator::VisitExpr_(op);
359 }
360 }
361
362 private:
363 // RW statistics about data
364 struct Entry {
365 int read_count{0};
366 int write_count{0};
367 };
368
369 // Get current storage scope.
370 StorageScope GetScope(Var buffer_var) const {
371 return StorageScope::Create(GetPtrStorageScope(buffer_var));
372 }
373
374 // private functions.
375 Stmt InitGlobalBarrier(const AttrStmtNode* op) {
376 ICHECK(op != nullptr);
377 Array<PrimExpr> pargs = {StringImm(runtime::symbol::tvm_prepare_global_barrier)};
378 Stmt prep = Evaluate(Call(DataType::Int(32), builtin::tvm_call_packed(), pargs));
379 Stmt body = op->body;
380 for (const auto& kv : rw_stats_) {
381 const auto& e = kv.second;
382 if (e.read_count != 0 && e.write_count != 0) {
383 body = AttrStmt(kv.first, attr::volatile_scope, 1, body);
384 }
385 }
386 rw_stats_.clear();
387 Stmt kinit = Evaluate(Call(DataType::Int(32), builtin::tvm_global_barrier_kinit(), {}));
388 body = SeqStmt({kinit, body});
389 body = AttrStmt(op->node, op->attr_key, op->value, body);
390 return SeqStmt({prep, body});
391 }
392 Stmt MakeGlobalBarrier() {
393 ICHECK(sync_scope_.rank == StorageRank::kGlobal);
394 if (!num_blocks_.defined()) {
395 ICHECK(!is_lead_.defined());
396 num_work_dim_ = thread_extents_.size();
397 for (const AttrStmtNode* attr : thread_extents_) {
398 IterVar iv = Downcast<IterVar>(attr->node);
399 runtime::ThreadScope s = runtime::ThreadScope::Create(iv->thread_tag);
400 if (s.rank == 0) {
401 num_blocks_ = (num_blocks_.defined() ? attr->value * num_blocks_ : attr->value);
402 } else if (s.rank == 1) {
403 PrimExpr cond = iv->var == make_zero(iv->var.dtype());
404 is_lead_ = is_lead_.defined() ? (is_lead_ && cond) : cond;
405 }
406 }
407 } else {
408 ICHECK_EQ(num_work_dim_, thread_extents_.size());
409 }
410 return Evaluate(Call(DataType::Int(32), builtin::tvm_storage_sync(),
411 {StringImm(sync_scope_.to_string()), is_lead_, num_blocks_}));
412 }
413 // data structure.
414 StorageScope sync_scope_;
415 const std::unordered_set<const Object*>& syncs_;
416 // The read write statistics of storage
417 std::unordered_map<Var, Entry, ObjectPtrHash, ObjectPtrEqual> rw_stats_;
418 // The statistics for global barrier
419 bool in_thread_env_{false};
420 // memorized results
421 std::vector<const AttrStmtNode*> thread_extents_;
422 size_t num_work_dim_{0};
423 PrimExpr num_blocks_;
424 PrimExpr is_lead_;
425};
426
427Stmt ThreadSync(Stmt stmt, std::string storage_scope) {
428 StorageScope sync_scope = StorageScope::Create(storage_scope);
429 if (sync_scope.rank == StorageRank::kShared && sync_scope.tag == "") {
430 stmt = ThreadSyncAfterWaitQueueInserter(sync_scope)(stmt);
431 }
432 ThreadSyncPlanner planner(sync_scope);
433 planner(stmt);
434 return ThreadSyncInserter(sync_scope, planner.syncs_inserted_)(std::move(stmt));
435}

437namespace transform {
438
439Pass ThreadSync(String storage_scope) {
440 auto pass_func = [storage_scope](PrimFunc f, IRModule m, PassContext ctx) {
441 auto* n = f.CopyOnWrite();
442 n->body = ThreadSync(std::move(n->body), storage_scope);
443 return f;
444 };
445 return CreatePrimFuncPass(pass_func, 0, "tir.ThreadSync", {});
446}
