43// use/def analysis, also delete unreferenced lets
44class VarUseDefAnalysis : public StmtExprMutator {
45 public:
46 Stmt VisitStmt_(const AttrStmtNode* op) final {
47 if (op->attr_key == attr::thread_extent) {
48 IterVar iv = Downcast<IterVar>(op->node);
49 ICHECK_NE(iv->thread_tag.length(), 0U);
50 // thread_extent can appear multiple times
51 // use the first appearance as def.
52 if (!use_count_.count(iv->var.get())) {
53 this->HandleDef(iv->var.get());
54 thread_axis_.push_back(iv);
55 thread_extent_.push_back(op->value);
56 }
57
58 PrimExpr value = op->value;
59 if (visit_thread_extent_) {
60 value = this->VisitExpr(value);
61 }
62 Stmt body = this->VisitStmt(op->body);
63 if (value.same_as(op->value) && body.same_as(op->body)) {
64 return GetRef<Stmt>(op);
65 }
66 return AttrStmt(op->node, op->attr_key, value, body);
67 } else {
68 return StmtExprMutator::VisitStmt_(op);
69 }
70 }
71
72 Stmt VisitStmt_(const LetStmtNode* op) final {
73 this->HandleDef(op->var.get());
74 Stmt body = this->VisitStmt(op->body);
75 // eliminate unreferenced let
76 if (use_count_.at(op->var.get()) == 0 && SideEffect(op->value) <= CallEffectKind::kReadState &&
77 simplify_let_) {
78 return body;
79 } else {
80 PrimExpr value = this->VisitExpr(op->value);
81 if (body.same_as(op->body) && value.same_as(op->value)) {
82 return GetRef<Stmt>(op);
83 } else {
84 return LetStmt(op->var, value, body);
85 }
86 }
87 }
88
89 Stmt VisitStmt_(const ForNode* op) final {
90 this->HandleDef(op->loop_var.get());
91 return StmtExprMutator::VisitStmt_(op);
92 }
93
94 Stmt VisitStmt_(const AllocateNode* op) final {
95 this->HandleDef(op->buffer_var.get());
96 auto storage_scope = runtime::StorageScope::Create(GetPtrStorageScope(op->buffer_var));
97 if (storage_scope.rank == runtime::StorageRank::kShared && storage_scope.tag == ".dyn") {
98 ICHECK_EQ(use_dyn_shmem_, false) << "Only one dynamic shared memory allocation is allowed.";
99 ICHECK_GT(op->extents.size(), 0);
100 dyn_shmem_size_ = op->extents[0];
101 for (size_t i = 1; i < op->extents.size(); ++i) {
102 dyn_shmem_size_ *= op->extents[i];
103 }
104 dyn_shmem_size_ = dyn_shmem_size_ * (op->dtype.bytes());
105 use_dyn_shmem_ = true;
106 }
107 return StmtExprMutator::VisitStmt_(op);
108 }
109
110 Stmt VisitStmt_(const AllocateConstNode* op) final {
111 this->HandleDef(op->buffer_var.get());
112 return StmtExprMutator::VisitStmt_(op);
113 }
114
115 Stmt VisitStmt_(const StoreNode* op) final {
116 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
117 return Stmt();
118 }
119
120 Stmt VisitStmt_(const BufferStoreNode* op) final {
121 VisitBuffer(op->buffer);
122 return StmtExprMutator::VisitStmt_(op);
123 }
124
125 PrimExpr VisitExpr_(const LetNode* op) final {
126 // Weaker SSA condition
127 // A single var can be binded in multiple lets
128 // but they have to bind to the same value.
129 // This is used to allow cases when we reuse a single let
130 // expression to construct a nested expr.
131 // (let x = 1 in x + 1) * (let x = 1 in x + 1)
132 auto it = let_binding_.find(op->var);
133 PrimExpr value = this->VisitExpr(op->value);
134 if (it != let_binding_.end()) {
135 ICHECK(deep_equal_(it->second->value, value))
136 << "Let cannot bind the same var to two different values";
137 return GetRef<PrimExpr>(it->second);
138 } else {
139 this->HandleDef(op->var.get());
140 let_binding_[op->var] = op;
141 }
142 PrimExpr body = this->VisitExpr(op->body);
143 // eliminate unreferenced let
144 if (use_count_.at(op->var.get()) == 0 && SideEffect(op->value) <= CallEffectKind::kReadState &&
145 simplify_let_) {
146 return body;
147 } else {
148 if (body.same_as(op->body) && value.same_as(op->value)) {
149 return GetRef<PrimExpr>(op);
150 } else {
151 return Let(op->var, value, body);
152 }
153 }
154 }
155
156 PrimExpr VisitExpr_(const VarNode* op) final {
157 this->HandleUse(GetRef<PrimExpr>(op));
158 return StmtExprMutator::VisitExpr_(op);
159 }
160
161 PrimExpr VisitExpr_(const ReduceNode* op) final {
162 for (const auto& iv : op->axis) {
163 this->HandleDef(iv->var.get());
164 }
165 return StmtExprMutator::VisitExpr_(op);
166 }
167
168 PrimExpr VisitExpr_(const LoadNode* op) final {
169 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
170 return PrimExpr();
171 }
172
173 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
174 VisitBuffer(op->buffer);
175 return StmtExprMutator::VisitExpr_(op);
176 }
177
178 void VisitBuffer(Buffer buffer) {
179 this->HandleUse(buffer->data);
180 auto visit_arr = [&](Array<PrimExpr> arr) {
181 for (const auto& element : arr) {
182 this->VisitExpr(element);
183 }
184 };
185
186 visit_arr(buffer->shape);
187 visit_arr(buffer->strides);
188 }
189
190 void HandleDef(const VarNode* v) {
191 ICHECK(!def_count_.count(v)) << "variable " << v->name_hint
192 << " has already been defined, the Stmt is not SSA";
193 ICHECK(!use_count_.count(v)) << "variable " << v->name_hint
194 << " has been used before definition!";
195 use_count_[v] = 0;
196 def_count_[v] = 1;
197 }
198
199 void HandleUse(const PrimExpr& v) {
200 ICHECK(v.as<VarNode>());
201 Var var = Downcast<Var>(v);
202 auto it = use_count_.find(var.get());
203 if (it != use_count_.end()) {
204 if (it->second >= 0) {
205 ++it->second;
206 }
207 } else {
208 undefined_.push_back(var);
209 use_count_[var.get()] = -1;
210 }
211 }
212
213 // The fields are publically readible to
214 // be accessible to the users.
215 bool visit_thread_extent_{true};
216 bool simplify_let_{true};
217 Array<Var> undefined_;
218 Array<IterVar> thread_axis_;
219 Array<PrimExpr> thread_extent_;
220 PrimExpr dyn_shmem_size_{0};
221 bool use_dyn_shmem_{false};
222 std::unordered_map<const VarNode*, int> use_count_;
223 std::unordered_map<const VarNode*, int> def_count_;
224
225 private:
226 ExprDeepEqual deep_equal_;
227 std::unordered_map<Var, const LetNode*, ObjectPtrHash, ObjectPtrEqual> let_binding_;
228};
229
230Array<Var> UndefinedVars(const Stmt& stmt, const Array<Var>& args) {
231 VarUseDefAnalysis m;
232 m.simplify_let_ = false;
233 for (Var arg : args) {
234 m.use_count_[arg.get()] = 0;
235 }
236 m(stmt);
237 return m.undefined_;
238}
239
240Array<Var> UndefinedVars(const PrimExpr& expr) {
241 VarUseDefAnalysis m;
242 m.simplify_let_ = false;
243 m(expr);
244 return m.undefined_;
245}

247class HostDeviceSplitter : public StmtMutator {
248 public:
249 explicit HostDeviceSplitter(IRModule* device_mod, Target device_target, std::string name_prefix)
250 : device_mod_(device_mod), device_target_(device_target), name_prefix_(name_prefix) {}
251
252 Stmt VisitStmt_(const AllocateNode* op) final {
253 handle_data_type_[op->buffer_var.get()] = make_const(op->dtype, 0);
254 return StmtMutator::VisitStmt_(op);
255 }
256
257 Stmt VisitStmt_(const AttrStmtNode* op) final {
258 if (op->attr_key == attr::thread_extent || op->attr_key == attr::pipeline_exec_scope ||
259 op->attr_key == attr::device_scope) {
260 return SplitDeviceFunc(GetRef<Stmt>(op));
261 }
262 return StmtMutator::VisitStmt_(op);
263 }
264
265 private:
266 Stmt SplitDeviceFunc(Stmt body) {
267 std::ostringstream os;
268 os << name_prefix_ << "_kernel" << device_func_counter_++;
269 std::string kernel_symbol = os.str();
270 // isolate the device function.
271 VarUseDefAnalysis m;
272 m.visit_thread_extent_ = false;
273 body = m(std::move(body));
274
275 Array<Var> params;
276 Array<PrimExpr> arguments;
277 Map<tir::Var, PrimExpr> remap_vars;
278
279 // Strictly order the arguments: Var pointers, positional arguments.
280 for (Var var : m.undefined_) {
281 if (var.dtype().is_handle()) {
282 // Create a new version of v.
283 auto it = handle_data_type_.find(var.get());
284 if (it != handle_data_type_.end()) {
285 String storage_scope;
286 if (auto* ptr_type = var->type_annotation.as<PointerTypeNode>()) {
287 storage_scope = ptr_type->storage_scope;
288 }
289 tir::Var new_var(var->name_hint,
290 PointerType(PrimType((*it).second->dtype), storage_scope));
291 params.push_back(new_var);
292 remap_vars.Set(var, new_var);
293 } else {
294 params.push_back(var);
295 }
296 arguments.push_back(var);
297 }
298 }
299 // positional arguments
300 for (Var var : m.undefined_) {
301 if (!var.dtype().is_handle()) {
302 params.push_back(var);
303 arguments.push_back(var);
304 }
305 }
306 GlobalVarSupply global_var_supply = GlobalVarSupply(*device_mod_);
307 GlobalVar kernel_symbol_global = global_var_supply->FreshGlobal(kernel_symbol, false);
308
309 PrimFunc device_func(params, Substitute(body, remap_vars));
310 device_func = WithAttr(std::move(device_func), tir::attr::kDeviceThreadAxis, m.thread_axis_);
311 device_func = WithAttr(std::move(device_func), tvm::attr::kCallingConv,
312 Integer(CallingConv::kDeviceKernelLaunch));
313 device_func = WithAttr(std::move(device_func), tvm::attr::kGlobalSymbol,
314 runtime::String(kernel_symbol_global->name_hint));
315 device_func = WithAttr(std::move(device_func), tir::attr::kNoAlias, Integer(1));
316 device_func = WithAttr(std::move(device_func), tvm::attr::kTarget, device_target_);
317 device_func = WithAttr(std::move(device_func), tir::attr::kIsGlobalFunc, Integer(1));
318 if (m.use_dyn_shmem_) {
319 device_func =
320 WithAttr(std::move(device_func), tir::attr::kDeviceUseDynSharedMemory, Integer(1));
321 }
322 (*device_mod_)->Add(kernel_symbol_global, device_func);
323
324 // generate calls to the device function
325 Array<PrimExpr> call_args;
326 call_args.push_back(StringImm(kernel_symbol_global->name_hint));
327 for (PrimExpr arg : arguments) {
328 call_args.push_back(arg);
329 }
330 for (PrimExpr ext : m.thread_extent_) {
331 call_args.push_back(ext);
332 }
333 if (m.use_dyn_shmem_) {
334 call_args.push_back(m.dyn_shmem_size_);
335 }
336 return Evaluate(Call(DataType::Int(32), builtin::tvm_call_packed(), call_args));
337 }
338
339 // target ir module
340 IRModule* device_mod_;
341 // Device target
342 Target device_target_;
343 // function name hint
344 std::string name_prefix_;
345 // Number of device functions.
346 int device_func_counter_{0};
347 std::unordered_map<const VarNode*, PrimExpr> handle_data_type_;
348};
349
350PrimFunc SplitHostDevice(PrimFunc&& func, IRModule* device_mod) {
351 auto target = func->GetAttr<Target>(tvm::attr::kTarget);
352 ICHECK(target.defined()) << "SplitHostDevice: Require the target attribute";
353 auto global_symbol = func->GetAttr<String>(tvm::attr::kGlobalSymbol);
354 ICHECK(global_symbol.defined())
355 << "SplitHostDevice: Expect PrimFunc to have the global_symbol attribute";
356
357 HostDeviceSplitter splitter(device_mod, target.value(),
358 static_cast<std::string>(global_symbol.value()));
359
360 auto* n = func.CopyOnWrite();
361 n->body = splitter(std::move(n->body));
362 // set the host target to None.
363 func = WithAttr(std::move(func), tvm::attr::kTarget, Target(nullptr));
364 return std::move(func);
365}

367namespace transform {
368
369Pass SplitHostDevice() {
370 auto pass_func = [](IRModule mod, PassContext ctx) {
371 IRModuleNode* mod_ptr = mod.CopyOnWrite();
372 auto* func_dict = mod_ptr->functions.CopyOnWrite();
373 IRModule device_mod = IRModule(Map<GlobalVar, BaseFunc>({}));
374
375 for (auto& kv : *func_dict) {
376 if (kv.second->IsInstance<PrimFuncNode>()) {
377 PrimFunc func = Downcast<PrimFunc>(std::move(kv.second));
378 ICHECK(device_mod.defined()) << "The device module must be defined.";
379 kv.second = SplitHostDevice(std::move(func), &device_mod);
380 }
381 }
382 mod->Update(device_mod);
383 return mod;
384 };
385
386 return tvm::transform::CreateModulePass(pass_func, 0, "tir.SplitHostDevice", {});
387}
