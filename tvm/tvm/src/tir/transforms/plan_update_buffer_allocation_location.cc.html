34class CollectUnmanagedAllocations : public StmtExprVisitor {
35 public:
36 void VisitStmt_(const AllocateNode* op) final {
37 unmanaged_allocations.insert(op->buffer_var.get());
38 StmtExprVisitor::VisitStmt_(op);
39 }
40
41 void VisitStmt_(const AllocateConstNode* op) final {
42 unmanaged_allocations.insert(op->buffer_var.get());
43 StmtExprVisitor::VisitStmt_(op);
44 }
45
46 /*! \brief Buffers that are allocated outside of the BlockNode, and should not be moved by
47 * BufferAllocationLocator. */
48 std::unordered_set<const VarNode*> unmanaged_allocations;
49};
50
51/*! \brief Collect the allocate buffer order. */
52class BufferAllocateOrderCollector : public StmtExprVisitor {
53 public:
54 static Array<Buffer> Collect(const PrimFunc& func) {
55 BufferAllocateOrderCollector collector;
56 for (const auto& kv : func->buffer_map) {
57 collector.buffer_alloc_recorder_.push_back(kv.second);
58 }
59 collector(func->body);
60 return std::move(collector.buffer_alloc_recorder_);
61 }
62
63 private:
64 void VisitStmt_(const BlockNode* op) final {
65 for (const Buffer& buffer : op->alloc_buffers) {
66 buffer_alloc_recorder_.push_back(buffer);
67 }
68 StmtExprVisitor::VisitStmt_(op);
69 }
70
71 void VisitExpr_(const BufferLoadNode* op) final {
72 if (std::find(buffer_alloc_recorder_.begin(), buffer_alloc_recorder_.end(), op->buffer) ==
73 buffer_alloc_recorder_.end()) {
74 buffer_alloc_recorder_.push_back(op->buffer);
75 }
76 StmtExprVisitor::VisitExpr_(op);
77 }
78
79 void VisitStmt_(const BufferStoreNode* op) final {
80 if (std::find(buffer_alloc_recorder_.begin(), buffer_alloc_recorder_.end(), op->buffer) ==
81 buffer_alloc_recorder_.end()) {
82 buffer_alloc_recorder_.push_back(op->buffer);
83 }
84 StmtExprVisitor::VisitStmt_(op);
85 }
86
87 /*! \brief The buffer allocated order recorder. */
88 Array<Buffer> buffer_alloc_recorder_;
89};
90
91class BufferAllocationLocator : public StmtExprMutator {
92 public:
93 explicit BufferAllocationLocator(const PrimFunc& func) {
94 Map<Buffer, Optional<Stmt>> buffer_lca = DetectBufferAccessLCA(func);
95 // The buffer_alloc_recorder Array is used to keep the buffer allocation order
96 // since the buffer_lca Map is unordered.
97 Array<Buffer> buffer_alloc_recorder = BufferAllocateOrderCollector::Collect(func);
98 std::unordered_set<const VarNode*> arg_buffer_vars;
99 CollectUnmanagedAllocations collector;
100 collector(func->body);
101 unmanaged_allocations_ = collector.unmanaged_allocations;
102
103 for (const auto& kv : func->buffer_map) {
104 const Buffer& buffer = kv.second;
105 arg_buffer_vars.emplace(buffer->data.get());
106 buffer_data_to_buffer_.Set(buffer->data, buffer);
107 }
108 // create buffers to be allocated at each stmts
109 for (const auto& buffer : buffer_alloc_recorder) {
110 auto it = buffer_lca.find(buffer);
111 if (it != buffer_lca.end()) {
112 const StmtNode* stmt = (*it).second.get();
113 if (arg_buffer_vars.count(buffer->data.get())) {
114 continue;
115 }
116 if (!unmanaged_allocations_.count(buffer->data.get())) {
117 alloc_buffers_[stmt].push_back(buffer);
118 }
119 buffer_data_to_buffer_.Set(buffer->data, buffer);
120 }
121 }
122 }
123
124 private:
125 Stmt VisitStmt_(const ForNode* op) final {
126 auto it = alloc_buffers_.find(op);
127 if (it == alloc_buffers_.end()) {
128 return StmtMutator::VisitStmt_(op);
129 }
130 for (const Buffer& buf : it->second) {
131 buffer_data_to_buffer_.Set(buf->data, buf);
132 }
133 auto node = Downcast<For>(StmtMutator::VisitStmt_(op));
134
135 Array<Buffer> new_block_alloc_bufs;
136 for (const Buffer& buf : it->second) {
137 if (!unmanaged_allocations_.count(buf->data.get())) {
138 buffer_data_to_buffer_.erase(buf->data);
139 new_block_alloc_bufs.push_back(buf);
140 }
141 }
142
143 if (new_block_alloc_bufs.size()) {
144 node.CopyOnWrite()->body = InjectOpaqueBlock(node->body, new_block_alloc_bufs);
145 }
146
147 return std::move(node);
148 }
149
150 Stmt VisitStmt_(const BlockNode* op) final {
151 ICHECK(!op->init.defined());
152 Array<Buffer> alloc_buffers;
153 auto it = alloc_buffers_.find(op);
154 if (it != alloc_buffers_.end()) {
155 alloc_buffers = it->second;
156 for (const Buffer& buf : it->second) {
157 buffer_data_to_buffer_.Set(buf->data, buf);
158 }
159 }
160 for (const MatchBufferRegion match_buffer : op->match_buffers) {
161 const Var& target_var = match_buffer->buffer->data;
162 const Var& source_var = match_buffer->source->buffer->data;
163 ICHECK(buffer_data_to_buffer_.count(source_var));
164 buffer_data_to_buffer_.Set(target_var, match_buffer->buffer);
165 }
166 Stmt stmt = StmtMutator::VisitStmt_(op);
167 op = stmt.as<BlockNode>();
168 ICHECK(op != nullptr);
169
170 // No longer consider buffers created by match_buffer inside the block when updating access
171 // region.
172 for (const MatchBufferRegion match_buffer : op->match_buffers) {
173 const Var& target_var = match_buffer->buffer->data;
174 buffer_data_to_buffer_.erase(target_var);
175 }
176 // No longer consider buffers allocated inside the block when updating access region.
177 if (it != alloc_buffers_.end()) {
178 for (const Buffer& buf : it->second) {
179 buffer_data_to_buffer_.erase(buf->data);
180 }
181 }
182
183 ObjectPtr<BlockNode> n = CopyOnWrite(op);
184 n->alloc_buffers = std::move(alloc_buffers);
185 // Erase buffer allocated inside the block from access region.
186 n->reads = RemoveRedundantBufferRegion(n->reads);
187 n->writes = RemoveRedundantBufferRegion(n->writes);
188 return Stmt(n);
189 }
190
191 Stmt VisitStmt_(const BufferRealizeNode* op) final {
192 ICHECK(false) << "Internal Error: BufferRealizeNode is not allowed in TensorIR.";
193 throw;
194 }
195
196 Stmt InjectOpaqueBlock(Stmt body, const Array<Buffer>& alloc_buffers) {
197 ICHECK(!alloc_buffers.empty());
198 Block opaque_block(/*iter_vars=*/{},
199 /*reads=*/{},
200 /*writes=*/{},
201 /*name_hint=*/"",
202 /*body=*/std::move(body),
203 /*init=*/NullOpt,
204 /*alloc_buffers=*/alloc_buffers);
205 ObjectPtr<BlockNode> n = CopyOnWrite(opaque_block.get());
206 Array<Array<BufferRegion>> access =
207 GetBlockReadWriteRegion(opaque_block, buffer_data_to_buffer_);
208 n->reads = access[0];
209 n->writes = access[1];
210 BlockRealize realize({}, Bool(true), Block(n));
211 return std::move(realize);
212 }
213
214 Array<BufferRegion> RemoveRedundantBufferRegion(const Array<BufferRegion>& region) const {
215 Array<BufferRegion> result;
216 for (const BufferRegion& buffer_region : region) {
217 if (buffer_data_to_buffer_.count(buffer_region->buffer->data)) {
218 result.push_back(buffer_region);
219 }
220 }
221 return result;
222 }
223
224 /*! \brief The map from stmt to the buffers to be allocated under it. */
225 std::unordered_map<const StmtNode*, Array<Buffer>> alloc_buffers_;
226 /*! \brief The buffer already allocated during recursive visiting. */
227 Map<Var, Buffer> buffer_data_to_buffer_;
228 /*! \brief Buffers that are allocated outside of the BlockNode, and should not be moved. */
229 std::unordered_set<const VarNode*> unmanaged_allocations_;
230};
231
232PrimFunc PlanAndUpdateBufferAllocationLocation(PrimFunc func) {
233 // Only apply this pass to TIR that is not from TE schedules
234 if (!IsFromLegacyTESchedule(func)) {
235 auto fptr = func.CopyOnWrite();
236 BufferAllocationLocator locator(func);
237 fptr->body = locator(fptr->body);
238 return func;
239 } else {
240 return func;
241 }
242}

246Pass PlanAndUpdateBufferAllocationLocation() {
247 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
248 return PlanAndUpdateBufferAllocationLocation(std::move(f));
249 };
250 return CreatePrimFuncPass(pass_func, 0, "tir.PlanAndUpdateBufferAllocationLocation", {});
251}
