34class CollectUnmanagedAllocations : public StmtExprVisitor {
35 public:
36 void VisitStmt_(const AllocateNode* op) final {
37 unmanaged_allocations.insert(op->buffer_var.get());
38 StmtExprVisitor::VisitStmt_(op);
39 }
40
41 void VisitStmt_(const AllocateConstNode* op) final {
42 unmanaged_allocations.insert(op->buffer_var.get());
43 StmtExprVisitor::VisitStmt_(op);
44 }
45
46 /*! \brief Buffers that are allocated outside of the BlockNode, and should not be moved by
47 * BufferAllocationLocator. */
48 std::unordered_set<const VarNode*> unmanaged_allocations;
49};
50
51class BufferAllocationLocator : public StmtExprMutator {
52 public:
53 explicit BufferAllocationLocator(const PrimFunc& func) {
54 Map<Buffer, Optional<Stmt>> buffer_lca = DetectBufferAccessLCA(func);
55 std::unordered_set<const VarNode*> arg_buffer_vars;
56 CollectUnmanagedAllocations collector;
57 collector(func->body);
58 unmanaged_allocations_ = collector.unmanaged_allocations;
59
60 for (const auto& kv : func->buffer_map) {
61 const Buffer& buffer = kv.second;
62 arg_buffer_vars.emplace(buffer->data.get());
63 buffer_data_to_buffer_.Set(buffer->data, buffer);
64 }
65 // create buffers to be allocated at each stmts
66 for (const auto& kv : buffer_lca) {
67 const Buffer& buffer = kv.first;
68 const StmtNode* stmt = kv.second.get();
69 if (arg_buffer_vars.count(buffer->data.get())) {
70 continue;
71 }
72 if (!unmanaged_allocations_.count(buffer->data.get())) {
73 alloc_buffers_[stmt].push_back(buffer);
74 }
75 buffer_data_to_buffer_.Set(buffer->data, buffer);
76 }
77 }
78
79 private:
80 Stmt VisitStmt_(const ForNode* op) final {
81 auto it = alloc_buffers_.find(op);
82 if (it == alloc_buffers_.end()) {
83 return StmtMutator::VisitStmt_(op);
84 }
85 for (const Buffer& buf : it->second) {
86 buffer_data_to_buffer_.Set(buf->data, buf);
87 }
88 auto node = Downcast<For>(StmtMutator::VisitStmt_(op));
89
90 Array<Buffer> new_block_alloc_bufs;
91 for (const Buffer& buf : it->second) {
92 if (!unmanaged_allocations_.count(buf->data.get())) {
93 buffer_data_to_buffer_.erase(buf->data);
94 new_block_alloc_bufs.push_back(buf);
95 }
96 }
97
98 if (new_block_alloc_bufs.size()) {
99 node.CopyOnWrite()->body = InjectOpaqueBlock(node->body, new_block_alloc_bufs);
100 }
101
102 return std::move(node);
103 }
104
105 Stmt VisitStmt_(const BlockNode* op) final {
106 ICHECK(!op->init.defined());
107 Array<Buffer> alloc_buffers;
108 auto it = alloc_buffers_.find(op);
109 if (it != alloc_buffers_.end()) {
110 alloc_buffers = it->second;
111 for (const Buffer& buf : it->second) {
112 buffer_data_to_buffer_.Set(buf->data, buf);
113 }
114 }
115 for (const MatchBufferRegion match_buffer : op->match_buffers) {
116 const Var& target_var = match_buffer->buffer->data;
117 const Var& source_var = match_buffer->source->buffer->data;
118 ICHECK(buffer_data_to_buffer_.count(source_var));
119 buffer_data_to_buffer_.Set(target_var, match_buffer->buffer);
120 }
121 Stmt stmt = StmtMutator::VisitStmt_(op);
122 op = stmt.as<BlockNode>();
123 ICHECK(op != nullptr);
124
125 // No longer consider buffers created by match_buffer inside the block when updating access
126 // region.
127 for (const MatchBufferRegion match_buffer : op->match_buffers) {
128 const Var& target_var = match_buffer->buffer->data;
129 buffer_data_to_buffer_.erase(target_var);
130 }
131 // No longer consider buffers allocated inside the block when updating access region.
132 if (it != alloc_buffers_.end()) {
133 for (const Buffer& buf : it->second) {
134 buffer_data_to_buffer_.erase(buf->data);
135 }
136 }
137
138 ObjectPtr<BlockNode> n = CopyOnWrite(op);
139 n->alloc_buffers = std::move(alloc_buffers);
140 // Erase buffer allocated inside the block from access region.
141 n->reads = RemoveRedundantBufferRegion(n->reads);
142 n->writes = RemoveRedundantBufferRegion(n->writes);
143 return Stmt(n);
144 }
145
146 Stmt VisitStmt_(const BufferRealizeNode* op) final {
147 ICHECK(false) << "Internal Error: BufferRealizeNode is not allowed in TensorIR.";
148 throw;
149 }
150
151 Stmt InjectOpaqueBlock(Stmt body, const Array<Buffer>& alloc_buffers) {
152 ICHECK(!alloc_buffers.empty());
153 Block opaque_block(/*iter_vars=*/{},
154 /*reads=*/{},
155 /*writes=*/{},
156 /*name_hint=*/"",
157 /*body=*/std::move(body),
158 /*init=*/NullOpt,
159 /*alloc_buffers=*/alloc_buffers);
160 ObjectPtr<BlockNode> n = CopyOnWrite(opaque_block.get());
161 Array<Array<BufferRegion>> access =
162 GetBlockReadWriteRegion(opaque_block, buffer_data_to_buffer_);
163 n->reads = access[0];
164 n->writes = access[1];
165 BlockRealize realize({}, Bool(true), Block(n));
166 return std::move(realize);
167 }
168
169 Array<BufferRegion> RemoveRedundantBufferRegion(const Array<BufferRegion>& region) const {
170 Array<BufferRegion> result;
171 for (const BufferRegion& buffer_region : region) {
172 if (buffer_data_to_buffer_.count(buffer_region->buffer->data)) {
173 result.push_back(buffer_region);
174 }
175 }
176 return result;
177 }
178
179 /*! \brief The map from stmt to the buffers to be allocated under it. */
180 std::unordered_map<const StmtNode*, Array<Buffer>> alloc_buffers_;
181 /*! \brief The buffer already allocated during recursive visiting. */
182 Map<Var, Buffer> buffer_data_to_buffer_;
183 /*! \brief Buffers that are allocated outside of the BlockNode, and should not be moved. */
184 std::unordered_set<const VarNode*> unmanaged_allocations_;
185};
186
187PrimFunc PlanAndUpdateBufferAllocationLocation(PrimFunc func) {
188 // Only apply this pass to TIR that is not from TE schedules
189 if (!IsFromLegacyTESchedule(func)) {
190 auto fptr = func.CopyOnWrite();
191 BufferAllocationLocator locator(func);
192 fptr->body = locator(fptr->body);
193 return func;
194 } else {
195 return func;
196 }
197}

201Pass PlanAndUpdateBufferAllocationLocation() {
202 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
203 return PlanAndUpdateBufferAllocationLocation(std::move(f));
204 };
205 return CreatePrimFuncPass(pass_func, 0, "tir.PlanAndUpdateBufferAllocationLocation", {});
206}
