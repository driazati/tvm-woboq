20/*!
21 * \file remove_weight_layout_rewrite_block.cc
22 * \brief Remove weight layout rewrite block before benchmark
23 */
24
25#include <tvm/tir/op.h>
26#include <tvm/tir/stmt_functor.h>
27#include <tvm/tir/transform.h>
28
29namespace tvm {
30namespace tir {
31
32class WeightLayoutRewriteBlockRemover : public StmtMutator {
33 public:
34 static PrimFunc Remove(PrimFunc f) {
35 WeightLayoutRewriteBlockRemover remover;
36 PrimFuncNode* n = f.CopyOnWrite();
37 n->body = remover(std::move(n->body));
38 Map<tir::Var, Buffer> buffer_map;
39 for (const auto& kv : f->buffer_map) {
40 Var param = kv.first;
41 Buffer buffer = kv.second;
42 auto it = remover.buf_map_.find(buffer);
43 if (it != remover.buf_map_.end()) {
44 buffer_map.Set(param, (*it).second);
45 } else {
46 buffer_map.Set(param, buffer);
47 }
48 }
49 n->buffer_map = std::move(buffer_map);
50 return f;
51 }
52
53 private:
54 Stmt VisitStmt_(const BlockNode* op) final {
55 Block block = Downcast<Block>(StmtMutator::VisitStmt_(op));
56
57 auto it = block->annotations.find(attr::meta_schedule_layout_rewrite_preproc);
58 if (it == block->annotations.end() || !is_one(Downcast<PrimExpr>((*it).second))) {
59 // The block is not a weight layout block
60 // Remove allocates if needed
61 Array<Buffer> alloc_buffers;
62 for (const Buffer& buffer : block->alloc_buffers) {
63 if (!rewritten_buffers_.count(buffer)) {
64 alloc_buffers.push_back(buffer);
65 }
66 }
67 if (alloc_buffers.size() < block->alloc_buffers.size()) {
68 auto n = CopyOnWrite(block.get());
69 n->alloc_buffers = std::move(alloc_buffers);
70 return Stmt(n);
71 } else {
72 return std::move(block);
73 }
74 }
75
76 // Step 0. Checking block attrs
77 ICHECK(block->alloc_buffers.empty());
78 ICHECK(block->match_buffers.empty());
79
80 // Step 1. Checking the body is a BufferStore
81 const auto* store = block->body.as<BufferStoreNode>();
82 ICHECK(store);
83
84 // Step 2. Checking the rhs of buffer store is a BufferLoad
85 const auto* load = store->value.as<BufferLoadNode>();
86 ICHECK(load);
87
88 // Step 3. Update Buffer
89 buf_map_.Set(load->buffer, store->buffer);
90 rewritten_buffers_.insert(store->buffer);
91
92 // Step 4. Set block body as no_op
93 auto n = CopyOnWrite(block.get());
94 n->body = std::move(Evaluate(0));
95 n->reads = {};
96 n->writes = {};
97 return Stmt(n);
98 }
99
100 private:
101 /*! \brief The buffer map from original layout buffer to rewritten buffer */
102 Map<Buffer, Buffer> buf_map_;
103 /*! \brief The buffer map from original layout buffer to rewritten buffer */
104 std::unordered_set<Buffer, ObjectPtrHash, ObjectPtrEqual> rewritten_buffers_;
105};
106namespace transform {
107
108Pass RemoveWeightLayoutRewriteBlock() {
109 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
110 return WeightLayoutRewriteBlockRemover::Remove(std::move(f));
111 };
112 return CreatePrimFuncPass(pass_func, 0, "tir.RemoveWeightLayoutRewriteBlock", {});
113}
114
115TVM_REGISTER_GLOBAL("tir.transform.RemoveWeightLayoutRewriteBlock")
116 .set_body_typed(RemoveWeightLayoutRewriteBlock);
117
118} // namespace transform
119
120} // namespace tir
121} // namespace tvm
