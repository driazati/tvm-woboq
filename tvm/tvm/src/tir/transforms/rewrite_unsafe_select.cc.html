34// For now, rewrite unsafe select expression to if_then_else
35// TODO(tqchen) pattern matching to support masked load
36class UnsafeExprDetector : public ExprFunctor<bool(const PrimExpr& n)> {
37 public:
38 // select itself is always considered safe if condition is safe
39 // Because we will issue guard to make sure it is.
40 bool VisitExpr_(const SelectNode* op) { return VisitExpr(op->condition); }
41 bool VisitExpr_(const CallNode* op) {
42 if (op->op.same_as(builtin::if_then_else())) {
43 return VisitExpr(op->args[0]);
44 } else if (op->op.same_as(builtin::address_of())) {
45 const BufferLoadNode* load = op->args[0].as<BufferLoadNode>();
46 for (const auto& index : load->indices) {
47 if (VisitExpr(index)) {
48 return true;
49 }
50 }
51 return false;
52 } else if (auto* ptr_op = op->op.as<OpNode>()) {
53 auto effect_kind = op_call_effect_[GetRef<Op>(ptr_op)];
54 if (effect_kind == CallEffectKind::kPure || effect_kind == CallEffectKind::kExprAnnotation) {
55 for (PrimExpr e : op->args) {
56 if (VisitExpr(e)) return true;
57 }
58 return false;
59 } else {
60 return true;
61 }
62 } else {
63 return true;
64 }
65 }
66 bool VisitExpr_(const BufferLoadNode* op) {
67 // Load is considered unsafe.
68 return true;
69 }
70 bool VisitExpr_(const LoadNode* op) {
71 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
72 return true;
73 }
74 bool VisitExpr_(const AddNode* op) final { return BinaryOp(op); }
75 bool VisitExpr_(const SubNode* op) final { return BinaryOp(op); }
76 bool VisitExpr_(const MulNode* op) final { return BinaryOp(op); }
77 bool VisitExpr_(const DivNode* op) final { return BinaryOp(op); }
78 bool VisitExpr_(const ModNode* op) final { return BinaryOp(op); }
79 bool VisitExpr_(const FloorDivNode* op) final { return BinaryOp(op); }
80 bool VisitExpr_(const FloorModNode* op) final { return BinaryOp(op); }
81 bool VisitExpr_(const MinNode* op) final { return BinaryOp(op); }
82 bool VisitExpr_(const MaxNode* op) final { return BinaryOp(op); }
83 bool VisitExpr_(const EQNode* op) final { return BinaryOp(op); }
84 bool VisitExpr_(const NENode* op) final { return BinaryOp(op); }
85 bool VisitExpr_(const LTNode* op) final { return BinaryOp(op); }
86 bool VisitExpr_(const LENode* op) final { return BinaryOp(op); }
87 bool VisitExpr_(const GTNode* op) final { return BinaryOp(op); }
88 bool VisitExpr_(const GENode* op) final { return BinaryOp(op); }
89 bool VisitExpr_(const AndNode* op) final { return BinaryOp(op); }
90 bool VisitExpr_(const OrNode* op) final { return BinaryOp(op); }
91 bool VisitExpr_(const NotNode* op) final { return VisitExpr(op->a); }
92 bool VisitExpr_(const LetNode* op) final { return VisitExpr(op->body) || VisitExpr(op->value); }
93 bool VisitExpr_(const CastNode* op) final { return VisitExpr(op->value); }
94 bool VisitExpr_(const BroadcastNode* op) final { return VisitExpr(op->value); }
95 bool VisitExpr_(const RampNode* op) final { return VisitExpr(op->base) && VisitExpr(op->stride); }
96 bool VisitExpr_(const ShuffleNode* op) final {
97 for (PrimExpr e : op->vectors) {
98 if (VisitExpr(e)) return true;
99 }
100 return false;
101 }
102 bool VisitExpr_(const VarNode* op) final { return false; }
103 bool VisitExpr_(const IntImmNode* op) final { return false; }
104 bool VisitExpr_(const FloatImmNode* op) final { return false; }
105 bool VisitExpr_(const StringImmNode* op) final { return false; }
106
107 private:
108 template <typename T>
109 bool BinaryOp(const T* op) {
110 return VisitExpr(op->a) || VisitExpr(op->b);
111 }
112
113 OpAttrMap<TCallEffectKind> op_call_effect_ = Op::GetAttrMap<TCallEffectKind>("TCallEffectKind");
114};
115
116class UnsafeSelectRewriter : public StmtExprMutator {
117 public:
118 PrimExpr VisitExpr_(const SelectNode* op) {
119 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
120 op = expr.as<SelectNode>();
121 UnsafeExprDetector unsafe;
122 bool cond_is_scalar_bool = op->condition.dtype().is_bool() && op->condition.dtype().is_scalar();
123 if ((unsafe.VisitExpr(op->true_value) || unsafe.VisitExpr(op->false_value)) &&
124 cond_is_scalar_bool) {
125 return Call(op->dtype, builtin::if_then_else(),
126 {op->condition, op->true_value, op->false_value});
127 } else {
128 return expr;
129 }
130 }
131};
132
133Stmt RewriteUnsafeSelect(Stmt stmt) { return UnsafeSelectRewriter()(std::move(stmt)); }
134
135namespace transform {
136
137Pass RewriteUnsafeSelect() {
138 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
139 auto* n = f.CopyOnWrite();
140 n->body = UnsafeSelectRewriter()(std::move(n->body));
141 return f;
142 };
143 return CreatePrimFuncPass(pass_func, 0, "tir.RewriteUnsafeSelect", {});
144}

146TVM_REGISTER_GLOBAL("tir.transform.RewriteUnsafeSelect").set_body_typed(RewriteUnsafeSelect);
