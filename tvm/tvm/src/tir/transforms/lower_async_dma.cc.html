33class AsyncDMALowerer : public StmtExprMutator {
34 public:
35 explicit AsyncDMALowerer(bool dma_bypass_cache) : dma_bypass_cache_(dma_bypass_cache) {}
36
37 Stmt VisitStmt_(const AttrStmtNode* op) final {
38 // Convert this, for example:
39 // attr [0] "async_wait_queue_scope" = 0;
40 // attr [0] "async_wait_inflight_count" = 0;
41 //
42 // To this:
43 // @tir.dma_wait(
44 // 0, /* queue id */
45 // 0, /* in flight count */
46 // dtype=int32
47 // )
48 if (op->attr_key == tir::attr::async_wait_queue_scope) {
49 // get queue ID
50 auto queue_id_node = op->value.as<IntImmNode>();
51 ICHECK(queue_id_node);
52 int queue_id = queue_id_node->value;
53
54 // abort if we have not seen this queue ID in `copy` transform
55 if (queue_ids_.find(queue_id) == queue_ids_.end()) {
56 DLOG(INFO) << "AsyncDMALowerer exiting because the queue ID observed in the "
57 "`async_wait_queue_scope` transform has not been previously observed in the "
58 "`async_commit_queue_scope` transform";
59 return StmtExprMutator::VisitStmt_(op);
60 }
61
62 auto async_wait = op->body.as<AttrStmtNode>();
63 if (!async_wait || async_wait->attr_key != tir::attr::async_wait_inflight_count) {
64 DLOG(INFO) << "AsyncDMALowerer exiting because the body of the `AttrStmtNode` with key "
65 "`async_wait_queue_scope` does not contain an `AttrStmtNode` with key "
66 "`async_wait_inflight_count`";
67 return StmtExprMutator::VisitStmt_(op);
68 }
69
70 auto call_dma_wait =
71 Evaluate(Call(DataType::Int(32), builtin::dma_wait(), {queue_id, async_wait->value}));
72
73 // concatenate the call with the body and return
74 return SeqStmt({call_dma_wait, StmtExprMutator::VisitStmt(async_wait->body)});
75
76 // Convert this, for example:
77 // attr [0] "async_commit_queue_scope" = 0;
78 // attr [0] "async_scope" = 1;
79 // for (ax0: int32, 0, 128) {
80 // A_global[ax0] = A[ax0]
81 // }
82 //
83 // To this:
84 // @tir.dma_copy(
85 // 0, /* queue id */
86 // @tir.address_of(A_global[0], dtype=handle),
87 // @tir.address_of(A[0], dtype=handle),
88 // 128, /* size */
89 // dtype=int32
90 // )
91 } else if (op->attr_key == tir::attr::async_commit_queue_scope) {
92 // get queue ID
93 auto queue_id_node = op->value.as<IntImmNode>();
94 ICHECK(queue_id_node);
95 int queue_id = queue_id_node->value;
96
97 // walk the graph to verify this is a mem copy ...
98 // 1) async_commit_queue_scope contains async_scope
99 auto async_scope = op->body.as<AttrStmtNode>();
100 if (!async_scope || async_scope->attr_key != tir::attr::async_scope) {
101 DLOG(INFO) << "AsyncDMALowerer exiting because the body of the `AttrStmtNode` with key "
102 "`async_commit_queue_scope` does not contain an `AttrStmtNode` with key "
103 "`async_scope`";
104 return StmtExprMutator::VisitStmt_(op);
105 }
106
107 // 2) async_scope contains single for loop
108 auto for_loop = async_scope->body.as<ForNode>();
109 if (!for_loop) {
110 DLOG(INFO) << "AsyncDMALowerer exiting because the body of the `AttrStmtNode` with key "
111 "`async_scope` does not contain a single `ForNode`";
112 return StmtExprMutator::VisitStmt_(op);
113 }
114
115 // 3) for loop contains buffer store with single index
116 auto bufferstorenode = for_loop->body.as<BufferStoreNode>();
117 if (!bufferstorenode || bufferstorenode->indices.size() != 1) {
118 DLOG(INFO)
119 << "AsyncDMALowerer exiting because the body of the `ForNode` does not contain a "
120 "single `BufferStoreNode` with a single index variable";
121 return StmtExprMutator::VisitStmt_(op);
122 }
123
124 // 4) buffer store value is a buffer load with single index
125 auto bufferloadnode = bufferstorenode->value.as<BufferLoadNode>();
126 if (!bufferloadnode || bufferloadnode->indices.size() != 1) {
127 DLOG(INFO) << "AsyncDMALowerer exiting because the value of the `BufferStoreNode` is not a "
128 "single `BufferLoadNode` with a single index variable";
129 return StmtExprMutator::VisitStmt_(op);
130 }
131
132 // get store buffer; assert it exists and is contiguous given it uses a single index
133 auto bufferstore = bufferstorenode->buffer.as<BufferNode>();
134 ICHECK(bufferstore && bufferstore->strides.empty());
135
136 // get load buffer; assert it exists and is contiguous given it uses a single index
137 auto bufferload = bufferloadnode->buffer.as<BufferNode>();
138 ICHECK(bufferload && bufferload->strides.empty());
139
140 // we will be replacing the entire for loop including its index
141 // with a DMA copy instrinsic that spans the entire index space of the for loop
142 // so we will need to replace the for loop index with value zero in the buffer indices
143 // thus we eliminate the index from the expression so the DMA copy receives the buffer range
144 // base address
145 Map<Var, PrimExpr> loop_var_remap = {{for_loop->loop_var, IntImm(DataType::Int(32), 0)}};
146
147 // map loop variable to zero for the store index & simplify
148 Array<PrimExpr> store_index = bufferstorenode->indices;
149 store_index.MutateByApply([&](PrimExpr expr) {
150 arith::Analyzer analyzer;
151 return analyzer.Simplify(Substitute(std::move(expr), loop_var_remap));
152 });
153
154 // map loop variable to zero for the load index & simplify
155 Array<PrimExpr> load_index = bufferloadnode->indices;
156 load_index.MutateByApply([&](PrimExpr expr) {
157 arith::Analyzer analyzer;
158 return analyzer.Simplify(Substitute(std::move(expr), loop_var_remap));
159 });
160
161 // now that we are about to perform the `copy` transform
162 // save queue ID for inspection in `wait` transform
163 queue_ids_.insert(queue_id);
164
165 return Evaluate(Call(DataType::Int(32), builtin::dma_copy(),
166 {queue_id,
167 Call(DataType::Handle(), builtin::address_of(),
168 {BufferLoad(bufferstorenode->buffer, store_index)}),
169 Call(DataType::Handle(), builtin::address_of(),
170 {BufferLoad(bufferloadnode->buffer, load_index)}),
171 for_loop->extent * bufferloadnode->dtype.bytes(), dma_bypass_cache_}));
172 }
173 return StmtExprMutator::VisitStmt_(op);
174 }
175
176 private:
177 std::set<int> queue_ids_;
178 bool dma_bypass_cache_;
179};
180
181namespace transform {
182
183Pass LowerAsyncDMA() {
184 auto pass_func = [=](PrimFunc f, IRModule m, PassContext ctx) {
185 auto fptr = f.CopyOnWrite();
186 bool dma_bypass_cache = ctx->GetConfig<Bool>("tir.dma_bypass_cache", Bool(false)).value();
187 fptr->body = AsyncDMALowerer(dma_bypass_cache)(std::move(fptr->body));
188 return f;
189 };
190 return CreatePrimFuncPass(pass_func, 0, "tir.LowerAsyncDMA", {});
191}
