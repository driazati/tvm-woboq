41inline PrimExpr BroadcastTo(PrimExpr e, int lanes) {
42 if (e.dtype().lanes() == lanes) return e;
43 if (const BroadcastNode* op = e.as<BroadcastNode>()) {
44 if (lanes % op->lanes == 0) {
45 return Broadcast(op->value, lanes);
46 }
47 }
48 ICHECK_EQ(e.dtype().lanes(), 1) << "Cannot broadcast lane=" << e.dtype().lanes() << " to "
49 << lanes;
50 return Broadcast(e, lanes);
51}

53// Rewrite vectorized allocation access
54// This is necessary for making each vector component containing its own workspace.
55// Originates from Halide's loop vectorizer
56//
57// s[i] = s[i * lanes + var]
58//
59// The same principle applies when using one thread to simulate multiple context.
60//
61class VecAllocAccess : public StmtExprMutator {
62 public:
63 VecAllocAccess(const VarNode* buf, Var var, int var_lanes)
64 : buf_(buf), var_(var), var_lanes_(var_lanes) {}
65
66 PrimExpr VisitExpr_(const LoadNode* op) final {
67 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
68 return PrimExpr();
69 }
70
71 Stmt VisitStmt_(const StoreNode* op) final {
72 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
73 return Stmt();
74 }
75
76 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
77 auto load = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
78 return UpdateBufferAccess(load);
79 }
80
81 Stmt VisitStmt_(const BufferStoreNode* op) final {
82 auto store = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
83 return UpdateBufferAccess(store);
84 }
85
86 private:
87 template <typename Node>
88 Node UpdateBufferAccess(Node node) {
89 // Only update the buffer that's being replaced.
90 if (node->buffer->data.get() != buf_) {
91 return node;
92 }
93
94 // Find/make a Buffer object with the correct updated shape.
95 Buffer buf;
96 auto it = buffer_map_.find(node->buffer.get());
97 if (it != buffer_map_.end()) {
98 buf = it->second;
99 } else {
100 // Extend the least significant dimension by a factor of
101 // var_lanes_. Typically, this will be a 1-d index into a flat
102 // memory space.
103 Array<PrimExpr> shape = node->buffer->shape;
104 shape.Set(shape.size() - 1, analyzer_.Simplify(shape[shape.size() - 1] * var_lanes_));
105
106 // TODO(Lunderberg): Move this pass to be prior to
107 // StorageFlatten/FlattenBuffer, implement by appending a
108 // dimension to the buffer. Since it is currently after the
109 // flattening, the strides are not technically necessary, but
110 // are updated for consistency.
111
112 // Update strides if defined.
113 Array<PrimExpr> strides;
114 for (size_t i = 0; i < strides.size(); i++) {
115 PrimExpr stride = strides[i];
116 if (i != strides.size() - 1) {
117 stride *= var_lanes_;
118 }
119 strides.push_back(analyzer_.Simplify(stride));
120 }
121
122 // Copy everything into the new buffer.
123 buf = node->buffer;
124 auto buf_writer = buf.CopyOnWrite();
125 buf_writer->shape = shape;
126 buf_writer->strides = strides;
127 buffer_map_[buf.get()] = buf;
128 }
129
130 // Extend the last index by the number of lanes in the vectorized
131 // variable.
132 Array<PrimExpr> indices = node->indices;
133 indices.Set(indices.size() - 1,
134 analyzer_.Simplify(indices[indices.size() - 1] * var_lanes_ + var_));
135
136 auto writer = node.CopyOnWrite();
137 writer->buffer = buf;
138 writer->indices = indices;
139 return node;
140 }
141
142 // buffer var
143 const VarNode* buf_;
144 // Updated buffer objects.
145 std::unordered_map<const BufferNode*, Buffer> buffer_map_;
146 // variable to be replaced
147 Var var_;
148 // the lanes.
149 int var_lanes_;
150 // Analyzer for simplifications
151 arith::Analyzer analyzer_;
152};

187 PrimExpr VisitExpr_(const MulNode* op) final {
188 PrimExpr a = this->VisitExpr(op->a);
189 PrimExpr b = this->VisitExpr(op->b);
190 if (a.same_as(op->a) && b.same_as(op->b)) {
191 return GetRef<PrimExpr>(op);
192 } else {
193 int lanes = std::max(a.dtype().lanes(), b.dtype().lanes());
194 if (lanes != 1) {
195 const RampNode* b_ramp = b.as<RampNode>();
196 const RampNode* a_ramp = a.as<RampNode>();
197 if (a_ramp && b.dtype().lanes() == 1 && analyzer_.CanProve(b > 0)) {
198 return Ramp(a_ramp->base * b, a_ramp->stride * b, a_ramp->lanes);
199 }
200 if (b_ramp && a.dtype().lanes() == 1 && analyzer_.CanProve(a > 0)) {
201 return Ramp(b_ramp->base * a, b_ramp->stride * a, b_ramp->lanes);
202 }
203 }
204 return Mul(BroadcastTo(a, lanes), BroadcastTo(b, lanes));
205 }
206 return BinaryVec<Mul>(op);
207 }

232 PrimExpr VisitExpr_(const RampNode* op) final {
233 PrimExpr base = this->VisitExpr(op->base);
234 PrimExpr stride = this->VisitExpr(op->stride);
235 if (base.dtype().lanes() > 1 && stride.dtype().lanes() == 1) {
236 const RampNode* base_ramp = base.as<RampNode>();
237 if (analyzer_.CanProve(base_ramp->stride == stride * make_const(stride.dtype(), op->lanes))) {
238 return Ramp(base_ramp->base, stride, op->lanes * base_ramp->lanes);
239 }
240 }
241 int lanes = std::max(base.dtype().lanes(), stride.dtype().lanes());
242 base = BroadcastTo(base, lanes);
243 stride = BroadcastTo(stride, lanes);
244 Array<PrimExpr> elems;
245 for (int i = 0; i < lanes; ++i) {
246 elems.push_back(
247 Ramp(Shuffle::ExtractElement(base, i), Shuffle::ExtractElement(stride, i), op->lanes));
248 }
249 return Shuffle::Concat(elems);
250 }

252 PrimExpr VisitExpr_(const BroadcastNode* op) final {
253 PrimExpr value = this->VisitExpr(op->value);
254 if (value.dtype().lanes() != 1) {
255 need_scalarize_ = true;
256 return GetRef<PrimExpr>(op);
257 }
258 if (value.same_as(op->value)) {
259 return GetRef<PrimExpr>(op);
260 } else {
261 return Broadcast(op->value, op->lanes);
262 }
263 }

265 PrimExpr VisitExpr_(const SelectNode* op) final {
266 PrimExpr cond = this->VisitExpr(op->condition);
267 PrimExpr t = this->VisitExpr(op->true_value);
268 PrimExpr f = this->VisitExpr(op->false_value);
269 if (cond.same_as(op->condition) && t.same_as(op->true_value) && f.same_as(op->false_value)) {
270 return GetRef<PrimExpr>(op);
271 } else {
272 int lanes = std::max(std::max(cond.dtype().lanes(), t.dtype().lanes()), f.dtype().lanes());
273 return Select(cond, BroadcastTo(t, lanes), BroadcastTo(f, lanes));
274 }

279 return GetRef<PrimExpr>(op);

281 return Cast(op->dtype.with_lanes(value.dtype().lanes()), value);
282 }

291 // Variable
292 PrimExpr VisitExpr_(const VarNode* op) final {
293 Var var = GetRef<Var>(op);
294
295 if (var.same_as(var_)) {
296 return ramp_;
297 }

305 // IfThenElse expr
306 PrimExpr MutateIfThenElseExpr_(const CallNode* op) {
307 PrimExpr cond = this->VisitExpr(op->args[0]);
308 if (cond.dtype().is_vector()) {
309 need_scalarize_ = true;
310 return GetRef<PrimExpr>(op);
311 }
312 PrimExpr t = this->VisitExpr(op->args[1]);
313 PrimExpr f = this->VisitExpr(op->args[2]);
314 if (cond.same_as(op->args[0]) && t.same_as(op->args[1]) && f.same_as(op->args[2])) {
315 return GetRef<PrimExpr>(op);
316 } else {
317 int lanes = std::max(t.dtype().lanes(), f.dtype().lanes());
318 t = BroadcastTo(t, lanes);
319 f = BroadcastTo(f, lanes);
320 return Call(op->dtype.with_lanes(lanes), op->op, {cond, t, f});
321 }
322 }
323 // Call
324 PrimExpr VisitExpr_(const CallNode* op) final {
325 if (op->op.same_as(builtin::if_then_else())) {
326 return MutateIfThenElseExpr_(op);
327 } else if (op->op.same_as(builtin::texture2d_load())) {
328 int lane = 0;
329 Array<PrimExpr> fcd = MutateArray({op->args.back()}, &lane);
330 auto new_args = op->args;
331 new_args.pop_back();
332 new_args.push_back(fcd[0]);
333 return Call(op->dtype.with_lanes(4), op->op, new_args);
334 } else if (op->op.same_as(builtin::texture2d_store())) {
335 int lane = 0;
336 // Vectorize the value to store
337 Array<PrimExpr> value{op->args.back()};
338 Array<PrimExpr> mutated_value = MutateArray(value, &lane);
339 Array<PrimExpr> new_args{op->args[0], op->args[1], op->args[2], mutated_value[0]};
340 return Call(op->dtype.with_lanes(lane), op->op, new_args);
341 }
342 auto* op_ptr = op->op.as<OpNode>();
343 bool vectorizable = op_ptr && op_vectorizable_.get(GetRef<Op>(op_ptr), false);
344
345 if (!vectorizable) {
346 // Cannot vectorize this op
347 Array<PrimExpr> new_args;
348 for (auto arg : op->args) {
349 auto new_arg = this->VisitExpr(arg);
350 if (new_arg.dtype().is_vector()) {
351 need_scalarize_ = true;
352 return GetRef<PrimExpr>(op);
353 }
354 new_args.push_back(new_arg);
355 }
356 if (op->args.same_as(new_args)) {
357 return GetRef<PrimExpr>(op);
358 } else {
359 return Call(op->dtype, op->op, new_args);
360 }
361 } else {
362 int lane = 0;
363 Array<PrimExpr> new_args = MutateArray(op->args, &lane);
364 // normal code path.
365 if (op->args.same_as(new_args)) {
366 return GetRef<PrimExpr>(op);
367 } else {
368 return Call(op->dtype.with_lanes(lane), op->op, new_args);
369 }
370 }
371 }
372 // Load
373 PrimExpr VisitExpr_(const LoadNode* op) final {

375 return PrimExpr();

377 // BufferLoad
378 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
379 auto load = GetRef<BufferLoad>(op);
380
381 auto fmutate = [this](const PrimExpr& index) { return this->VisitExpr(index); };
382 Array<PrimExpr> indices = op->indices.Map(fmutate);
383
384 if (!indices.same_as(op->indices)) {
385 auto writer = load.CopyOnWrite();
386 writer->indices = indices;
387 writer->LegalizeDType();
388 }
389
390 return std::move(load);
391 }
392 // Let
393 PrimExpr VisitExpr_(const LetNode* op) final {
394 PrimExpr value = this->VisitExpr(op->value);
395 // Weaker SSA condition
396 // A single var can be binded in multiple lets
397 // but they have to bind to the same value.
398 // This is used to allow cases when we reuse a single let
399 // expression to cosntruct a nested expr.
400 // (let x = 1 in x + 1) * (let x = 1 in x + 1)
401 auto it = let_binding_.find(op->var);
402 if (it != let_binding_.end()) {
403 ICHECK(deep_equal_(it->second, value))
404 << "Let cannot bind the same var to two different values";
405 }
406 if (value.dtype().lanes() != op->value.dtype().lanes()) {
407 Var new_var(op->var->name_hint, value.dtype());
408 let_binding_[op->var] = new_var;
409 return Let(new_var, value, this->VisitExpr(op->body));
410 } else {
411 let_binding_[op->var] = op->var;
412 PrimExpr body = this->VisitExpr(op->body);
413 if (value.same_as(op->value) && body.same_as(op->body)) {
414 return GetRef<PrimExpr>(op);
415 } else {
416 return Let(op->var, value, body);
417 }
418 }
419 }
420 // Store
421 Stmt VisitStmt_(const StoreNode* op) final {
422 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
423 return Stmt();
424 }
425 // BufferStore
426 Stmt VisitStmt_(const BufferStoreNode* op) final {
427 auto store = GetRef<BufferStore>(op);
428
429 auto fmutate = [this](const PrimExpr& index) { return this->VisitExpr(index); };
430 Array<PrimExpr> indices = op->indices.Map(fmutate);
431
432 PrimExpr value = this->VisitExpr(op->value);
433
434 if (!indices.same_as(op->indices) || !value.same_as(op->value)) {
435 // How many lanes of indexing are present in the index and
436 // buffer element type, excluding the last index. T
437 int other_index_lanes = op->buffer->dtype.lanes();
438 for (size_t i = 0; i < indices.size() - 1; i++) {
439 other_index_lanes *= indices[i].dtype().lanes();
440 }
441
442 // The total number of lanes of indexing, including the last index.
443 int index_lanes = other_index_lanes * indices[indices.size() - 1].dtype().lanes();
444
445 // The total number of lanes in this store operation. Either
446 // the index or the value will be broadcast out to this number
447 // of lanes, depending on which has more lanes.
448 int total_lanes = std::max(index_lanes, value.dtype().lanes());
449
450 ICHECK_EQ(total_lanes % other_index_lanes, 0)
451 << "When storing to buffer " << op->buffer->name << ", cannot produce " << total_lanes
452 << " lanes of storage location by changing the last index.";
453 int last_index_lanes = total_lanes / other_index_lanes;
454
455 // Broadcast the last index such that the total number of index
456 // lanes matches the desired number.
457 indices.Set(indices.size() - 1, BroadcastTo(indices[indices.size() - 1], last_index_lanes));
458
459 auto writer = store.CopyOnWrite();
460 writer->indices = indices;
461 writer->value = BroadcastTo(value, total_lanes);
462 }
463
464 return std::move(store);
465 }
466 // For
467 Stmt VisitStmt_(const ForNode* op) final {
468 if (op->kind == ForKind::kVectorized) {
469 LOG(WARNING) << "Detect vectorize inside vectorized loop, ignoring...";
470 }
471 ICHECK(is_zero(op->min));
472 ICHECK(!op->extent.dtype().is_vector());
473 PrimExpr extent = this->VisitExpr(op->extent);
474 if (extent.dtype().is_vector()) {
475 return Scalarize(GetRef<Stmt>(op));
476 }
477 Stmt body = this->VisitStmt(op->body);
478 if (extent.same_as(op->extent) && body.same_as(op->body)) {
479 return GetRef<Stmt>(op);
480 } else {
481 return For(op->loop_var, op->min, extent, op->kind, body, op->thread_binding,
482 op->annotations);
483 }
484 }
485 // IfThenElse
486 Stmt VisitStmt_(const IfThenElseNode* op) final {
487 ICHECK(!op->condition.dtype().is_vector());
488 PrimExpr condition = this->VisitExpr(op->condition);
489 if (condition.dtype().is_vector()) {
490 return Scalarize(GetRef<Stmt>(op));
491 }
492 Stmt then_case = this->VisitStmt(op->then_case);
493 Optional<Stmt> else_case = NullOpt;
494 if (op->else_case) {
495 else_case = this->VisitStmt(op->else_case.value());
496 }
497 if (condition.same_as(op->condition) && then_case.same_as(op->then_case) &&
498 else_case.same_as(op->else_case)) {
499 return GetRef<Stmt>(op);
500 } else {
501 return IfThenElse(condition, then_case, else_case);
502 }
503 }

509 // LetStmt
510 Stmt VisitStmt_(const LetStmtNode* op) final {
511 PrimExpr value = this->VisitExpr(op->value);
512 ICHECK(!let_binding_.count(op->var)) << "SSA violation, a single var is binded twice";
513 let_binding_[op->var] = value;
514
515 if (value.dtype().lanes() != op->value.dtype().lanes()) {
516 Var new_var(op->var->name_hint, value.dtype());
517 let_binding_[op->var] = new_var;
518 return LetStmt(new_var, value, this->VisitStmt(op->body));
519 } else {
520 let_binding_[op->var] = op->var;
521 Stmt body = this->VisitStmt(op->body);
522 if (value.same_as(op->value) && body.same_as(op->body)) {
523 return GetRef<Stmt>(op);
524 } else {
525 return LetStmt(op->var, value, body);
526 }
527 }
528 }
529 // Allocate
530 Stmt VisitStmt_(const AllocateNode* op) final {
531 // Mutate the condition
532 PrimExpr condition = this->VisitExpr(op->condition);
533 if (condition.dtype().is_vector()) {
534 LOG(WARNING) << "Cannot handle vector extent in alloc of " << op->buffer_var->name_hint;
535 return Scalarize(GetRef<Stmt>(op));
536 }
537
538 // Mutate the extents
539 Array<PrimExpr> extents;
540 for (const auto& extent : op->extents) {
541 PrimExpr new_ext = this->VisitExpr(extent);
542 if (new_ext.dtype().is_vector()) {
543 LOG(WARNING) << "Cannot handle vector extent in alloc of " << op->buffer_var->name_hint;
544 return Scalarize(GetRef<Stmt>(op));
545 }
546 extents.push_back(new_ext);
547 }
548
549 // TODO(Lunderberg): Move this pass to be prior to
550 // StorageFlatten/FlattenBuffer. That will allow this pass to be
551 // implemented as adding a new buffer dimension, which is later
552 // flattened.
553
554 // Extend the least significant dimension by a factor of
555 // var_lanes_. Typically, this will be a 1-d index into a flat
556 // memory space.
557 extents.Set(extents.size() - 1, extents[extents.size() - 1] * var_lanes_);
558
559 // Rewrite access to the buffer in the body.
560 Stmt body = VecAllocAccess(op->buffer_var.get(), var_, var_lanes_)(op->body);
561 body = this->VisitStmt(body);
562 return Allocate(op->buffer_var, op->dtype, extents, condition, body);
563 }

565 // scalarize the statment
566 Stmt Scalarize(Stmt stmt) {
567 Var idx(var_->name_hint + ".s", var_->dtype);
568 Map<Var, PrimExpr> values{{var_, idx}};
569 stmt = Substitute(stmt, values);
570 return For(idx, IntImm(var_->dtype, 0), IntImm(var_->dtype, var_lanes_), ForKind::kSerial,
571 stmt);
572 }

576 return Stmt();

580 // analyzer
581 arith::Analyzer analyzer_;
582 // deep equal
583 ExprDeepEqual deep_equal_;
584 // variable to be replaced
585 Var var_;
586 // the lanes.
587 int var_lanes_;
588 // ramp representing the var.
589 PrimExpr ramp_;
590 // flag to mark requirment of scalarization.
591 bool need_scalarize_{false};
592 // Let binding
593 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectPtrEqual> let_binding_;
594 // vectorizable property
595 OpAttrMap<TVectorizable> op_vectorizable_ = Op::GetAttrMap<TVectorizable>("TVectorizable");

597 // mutate array, with given lane requirement
598 // when finished, p_lane updates the lane requirement.
599 Array<PrimExpr> MutateArray(Array<PrimExpr> arr, int* p_lanes) {
600 if (arr.size() == 0) return arr;
601 int& lanes = *p_lanes;
602 bool changed = false;
603 std::vector<PrimExpr> new_arr(arr.size());
604 for (size_t i = 0; i < arr.size(); i++) {
605 PrimExpr old_elem = arr[i];
606 PrimExpr new_elem = this->VisitExpr(old_elem);
607 if (!new_elem.same_as(old_elem)) changed = true;
608 new_arr[i] = new_elem;
609 lanes = std::max(lanes, new_elem.dtype().lanes());
610 }
611
612 for (size_t i = 0; i < arr.size(); ++i) {
613 if (new_arr[i].dtype().lanes() != lanes) {
614 new_arr[i] = BroadcastTo(new_arr[i], lanes);
615 changed = true;
616 }
617 }
618 if (!changed) return arr;
619 return Array<PrimExpr>(new_arr);
620 }
621 template <typename TOp, typename T>
622 PrimExpr BinaryVec(const T* op) {
623 static_assert(std::is_same<typename TOp::ContainerType, T>::value, "constraint");
624 PrimExpr a = this->VisitExpr(op->a);
625 PrimExpr b = this->VisitExpr(op->b);
626 if (a.same_as(op->a) && b.same_as(op->b)) {
627 return GetRef<PrimExpr>(op);
628 } else {
629 int lanes = std::max(a.dtype().lanes(), b.dtype().lanes());
630 return TOp(BroadcastTo(a, lanes), BroadcastTo(b, lanes));
631 }
632 }
633 template <typename T, typename FCompute>
634 PrimExpr AddSubVec(const T* op, FCompute fcompute) {
635 PrimExpr a = this->VisitExpr(op->a);
636 PrimExpr b = this->VisitExpr(op->b);
637 if (a.same_as(op->a) && b.same_as(op->b)) {
638 return GetRef<PrimExpr>(op);
639 } else {
640 int lanes = std::max(a.dtype().lanes(), b.dtype().lanes());
641 if (lanes != 1) {
642 const RampNode* b_ramp = b.as<RampNode>();
643 const RampNode* a_ramp = a.as<RampNode>();
644 if (a.dtype().lanes() == 1 && b_ramp) {
645 return Ramp(fcompute(a, b_ramp->base),
646 fcompute(make_zero(b_ramp->stride.dtype()), b_ramp->stride), b_ramp->lanes);
647 }
648 if (b.dtype().lanes() == 1 && a_ramp) {
649 return Ramp(fcompute(a_ramp->base, b), a_ramp->stride, a_ramp->lanes);
650 }
651 }
652 return fcompute(BroadcastTo(a, lanes), BroadcastTo(b, lanes));
653 }
654 }
655};
656
657class LoopVectorizer : public StmtMutator {
658 public:
659 Stmt VisitStmt_(const ForNode* op) final {
660 if (op->kind == ForKind::kVectorized) {
661 ICHECK(is_zero(op->min));
662 auto* extent_as_int = op->extent.as<IntImmNode>();
663 if (!extent_as_int || extent_as_int->value < 1) {
664 LOG(FATAL) << "Failed to vectorize loop with extent " << op->extent;
665 }
666 return Vectorizer(op->loop_var, static_cast<int>(extent_as_int->value))(op->body);
667 } else {
668 return StmtMutator::VisitStmt_(op);
669 }
670 }
671};
672
673Stmt VectorizeLoop(Stmt stmt) { return LoopVectorizer()(std::move(stmt)); }
674
675class VectorizeSkipper : public StmtMutator {
676 public:
677 Stmt VisitStmt_(const ForNode* op) final {
678 Stmt stmt = StmtMutator::VisitStmt_(op);
679 op = stmt.as<ForNode>();
680 if (op->kind == ForKind::kVectorized) {
681 return For(op->loop_var, op->min, op->extent, ForKind::kSerial, op->body);
682 } else {
683 return stmt;
684 }
685 }
686};
687
688Stmt SkipVectorize(Stmt stmt) { return VectorizeSkipper()(std::move(stmt)); }
