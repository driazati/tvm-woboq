20#include <tvm/runtime/registry.h>
21#include <tvm/tir/expr.h>
22#include <tvm/tir/stmt_functor.h>
23#include <tvm/tir/transform.h>
24
25namespace tvm {
26namespace tir {
27
28class AssertSkipper : public StmtMutator {
29 public:
30 Stmt VisitStmt_(const AssertStmtNode* op) final {
31 Stmt stmt = StmtMutator::VisitStmt_(op);
32 op = stmt.as<AssertStmtNode>();
33 return op->body;
34 }
35};
36
37Stmt SkipAssert(Stmt stmt) { return AssertSkipper()(std::move(stmt)); }
38
39namespace transform {
40
41Pass SkipAssert() {
42 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
43 auto* n = f.CopyOnWrite();
44 n->body = AssertSkipper()(std::move(n->body));
45 return f;
46 };
47 return CreatePrimFuncPass(pass_func, 0, "tir.SkipAssert", {});
48}
49
50TVM_REGISTER_GLOBAL("tir.transform.SkipAssert").set_body_typed(SkipAssert);
51
52} // namespace transform
53
54} // namespace tir
55} // namespace tvm
