63// Mark the statement of each stage.
64class NoOpRemover : public arith::IRMutatorWithAnalyzer {
65 public:
66 static Stmt Apply(Stmt stmt, arith::Analyzer* analyzer,
67 std::optional<ControlFlowGraph> touch_pattern, const StmtNode* context) {
68 NoOpRemover visitor(analyzer, touch_pattern, context);
69 return visitor(std::move(stmt));
70 }
71
72 private:
73 using Parent = IRMutatorWithAnalyzer;
74 using Parent::VisitStmt;
75 using Parent::VisitStmt_;
76
77 NoOpRemover(arith::Analyzer* analyzer, std::optional<ControlFlowGraph> touch_pattern,
78 const StmtNode* context)
79 : Parent(analyzer), touch_pattern_(touch_pattern), context_(context) {}
80
81 Stmt VisitStmt_(const LetStmtNode* op) final {
82 Stmt stmt = Parent::VisitStmt_(op);
83 op = stmt.as<LetStmtNode>();
84 if (is_no_op(op->body)) {
85 return MakeEvaluate(op->value);
86 }
87
88 bool body_uses_bound_variable =
89 !UsesVar(op->body, [&](const VarNode* var) { return var == op->var.get(); });
90 if (body_uses_bound_variable && HasSideEffect(op->value)) {
91 return SeqStmt({MakeEvaluate(op->value), op->body});
92 } else if (body_uses_bound_variable) {
93 return op->body;
94 } else {
95 return stmt;
96 }
97 }
98 Stmt VisitStmt_(const AttrStmtNode* op) final {
99 if (op->attr_key == "pragma_debug_skip_region") {
100 return MakeEvaluate(0);
101 } else if (op->attr_key == attr::async_wait_queue_scope) {
102 auto wait_attrs = GetAsyncWaitAttributes(op);
103 auto wait_cnt = wait_attrs.second;
104 arith::Analyzer ana;
105 if (ana.CanProve(wait_cnt < 0)) {
106 // A negative wait count can arise if it depends on a loop variable.
107 // For example, a wait count 1 - i can be negative after loop unrolling.
108 // We assume that such wait is a nop.
109 auto inner = op->body.as<AttrStmtNode>();
110 ICHECK(inner);
111 return Parent::VisitStmt(inner->body);
112 }
113 }
114
115 Stmt stmt = Parent::VisitStmt_(op);
116 op = stmt.as<AttrStmtNode>();
117 return is_no_op(op->body) ? MakeEvaluate(op->value) : stmt;
118 }
119 Stmt VisitStmt_(const IfThenElseNode* op) final {
120 Stmt stmt = Parent::VisitStmt_(op);
121 op = stmt.as<IfThenElseNode>();
122 if (op->else_case) {
123 bool no_op_else = is_no_op(op->else_case.value());
124 bool no_op_then = is_no_op(op->then_case);
125 if (no_op_else && no_op_then) {
126 return MakeEvaluate(op->condition);
127 } else if (no_op_else) {
128 return IfThenElse(op->condition, op->then_case);
129 } else if (no_op_then) {
130 return IfThenElse(!op->condition, op->else_case.value());
131 } else {
132 return stmt;
133 }
134 } else {
135 if (is_no_op(op->then_case)) {
136 return MakeEvaluate(op->condition);
137 } else {
138 return stmt;
139 }
140 }
141 }
142 Stmt VisitStmt_(const ForNode* op) final {
143 auto extent_range = arith::EvalSet(op->extent, var_range_map_);
144 if (!arith::is_neg_inf(extent_range.max()) && !arith::is_pos_inf(extent_range.max()) &&
145 analyzer_->CanProve(extent_range.max() <= 0)) {
146 return Evaluate(0);
147 }
148 var_range_map_[op->loop_var.get()] = arith::IntSet::FromMinExtent(op->min, op->extent);
149 Stmt stmt = Parent::VisitStmt_(op);
150 var_range_map_.erase(op->loop_var.get());
151 op = stmt.as<ForNode>();
152 if (is_zero(op->extent)) {
153 return Evaluate(0);
154 }
155 return is_no_op(op->body) ? MakeEvaluate({op->min, op->extent}) : stmt;
156 }
157 Stmt VisitStmt_(const AllocateNode* op) final {
158 Stmt stmt = StmtMutator::VisitStmt_(op);
159 op = stmt.as<AllocateNode>();
160 return is_no_op(op->body) ? MakeEvaluate(op->extents) : stmt;
161 }
162
163 Stmt VisitStmt_(const ProducerRealizeNode* op) final {
164 Stmt stmt = StmtMutator::VisitStmt_(op);
165 op = stmt.as<ProducerRealizeNode>();
166 return is_no_op(op->body) ? op->body : stmt;
167 }
168 Stmt VisitStmt_(const EvaluateNode* op) final {
169 if (HasSideEffect(op->value)) {
170 return GetRef<Stmt>(op);
171 } else {
172 return Evaluate(0);
173 }
174 }
175
176 Stmt VisitStmt_(const SeqStmtNode* op) final {
177 auto ret = Downcast<SeqStmt>(StmtMutator::VisitSeqStmt_(op, true));
178
179 bool need_compact = std::any_of(ret->seq.begin(), ret->seq.end(),
180 [](const auto& stmt) { return is_no_op(stmt); });
181
182 if (need_compact) {
183 Array<Stmt> filtered;
184 for (Stmt stmt : ret->seq) {
185 if (!is_no_op(stmt)) {
186 filtered.push_back(std::move(stmt));
187 }
188 }
189 ret = SeqStmt(filtered);
190 }
191
192 if (ret->size() == 0) {
193 return Evaluate(0);
194 } else if (ret->size() == 1) {
195 return ret->seq[0];
196 } else {
197 return std::move(ret);
198 }
199 }
200
201 Stmt VisitStmt_(const BufferStoreNode* op) final {
202 BufferStore store = GetRef<BufferStore>(op);
203
204 // Helper function that returns a statement containing only the
205 // side effects of evaluating this BufferStore, but not the store
206 // itself.
207 auto only_side_effects = [&]() {
208 Array<Stmt> statements;
209 statements.push_back(MakeEvaluate(store->value));
210 for (const auto& index : store->indices) {
211 statements.push_back(MakeEvaluate(index));
212 }
213 return this->VisitStmt(SeqStmt(statements));
214 };
215
216 if (touch_pattern_.has_value()) {
217 // A write that is later overwritten is a no-op.
218 Stmt context = context_ ? GetRef<Stmt>(context_) : store;
219 if (touch_pattern_->IsOverwrittenWithoutEffect(store, context)) {
220 touch_pattern_->RemoveStore(store);
221 return only_side_effects();
222 }
223 }
224
225 // A write whose destination is known to already contain the
226 // values to be written is a no-op.
227 // PrimExpr stores_existing_value = store->value == BufferLoad(store->buffer, store->indices);
228 PrimExpr stores_existing_value = store->value - BufferLoad(store->buffer, store->indices) == 0;
229 if (touch_pattern_.has_value()) {
230 Stmt context_arg = context_ ? GetRef<Stmt>(context_) : Stmt(store);
231 stores_existing_value =
232 touch_pattern_->SimplifyInContext(stores_existing_value, context_arg, analyzer_);
233 } else {
234 stores_existing_value = analyzer_->Simplify(stores_existing_value);
235 }
236 if (is_one(stores_existing_value)) {
237 return only_side_effects();
238 }
239
240 // If the stored value is a load from the same location, the
241 // statement is a no-op, regardless of contextual information.
242 if (const BufferLoadNode* load = store->value.as<BufferLoadNode>()) {
243 if (load->buffer->data.same_as(store->buffer->data) &&
244 analyzer_->CanProveEqual(load->buffer->elem_offset, store->buffer->elem_offset) &&
245 ArrayValueEqual(load->buffer->shape, store->buffer->shape) &&
246 ArrayValueEqual(load->buffer->strides, store->buffer->strides) &&
247 ArrayValueEqual(load->indices, store->indices)) {
248 return only_side_effects();
249 }
250 }
251
252 return std::move(store);
253 }
254
255 private:
256 bool ArrayValueEqual(const Array<PrimExpr>& a, const Array<PrimExpr>& b) {
257 if (a.size() != b.size()) {
258 return false;
259 }
260 for (size_t i = 0; i < a.size(); i++) {
261 if (!analyzer_->CanProveEqual(a[i], b[i])) {
262 return false;
263 }
264 }
265 return true;
266 }
267
268 bool HasSideEffect(const PrimExpr& value) {
269 return SideEffect(value) > CallEffectKind::kReadState;
270 }
271
272 Stmt MakeEvaluate(PrimExpr value) {
273 if (SideEffect(value) > CallEffectKind::kReadState) {
274 return Evaluate(value);
275 } else {
276 return Evaluate(0);
277 }
278 }
279 Stmt MakeEvaluate(const Array<PrimExpr>& values) {
280 Array<Stmt> stmts;
281 for (PrimExpr e : values) {
282 if (SideEffect(e) > CallEffectKind::kReadState) {
283 stmts.push_back(Evaluate(e));
284 }
285 }
286
287 if (stmts.size() == 0) {
288 return Evaluate(0);
289 } else if (stmts.size() == 1) {
290 return stmts[0];
291 } else {
292 return SeqStmt(stmts);
293 }
294 }
295
296 std::unordered_map<const VarNode*, arith::IntSet> var_range_map_;
297 std::optional<ControlFlowGraph> touch_pattern_;
298 const StmtNode* context_;
299};
300
301Stmt RemoveNoOp(Stmt stmt, arith::Analyzer* analyzer, std::optional<ControlFlowGraph> touch_pattern,
302 const StmtNode* context) {
303 return NoOpRemover::Apply(std::move(stmt), analyzer, std::move(touch_pattern), context);
304}
305
306namespace transform {
307
308Pass RemoveNoOp() {
309 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
310 std::optional<ControlFlowGraph> touch_pattern = std::nullopt;
311
312 RemoveNoOpConfig config = ctx->GetConfig<RemoveNoOpConfig>("tir.RemoveNoOp")
313 .value_or(AttrsWithDefaultValues<RemoveNoOpConfig>());
314 if (config->use_dataflow_analysis) {
315 touch_pattern.emplace(f->body);
316 }
317
318 arith::Analyzer analyzer;
319
320 auto* n = f.CopyOnWrite();
321 n->body = NoOpRemover::Apply(std::move(n->body), &analyzer, std::move(touch_pattern), nullptr);
322 return f;
323 };
324 return CreatePrimFuncPass(pass_func, 0, "tir.RemoveNoOp", {});
325}
