40// Mark the statement of each stage.
41class NoOpRemover : public StmtMutator {
42 public:
43 Stmt VisitStmt_(const LetStmtNode* op) final {
44 Stmt stmt = StmtMutator::VisitStmt_(op);
45 op = stmt.as<LetStmtNode>();
46 return is_no_op(op->body) ? MakeEvaluate(op->value) : stmt;
47 }
48 Stmt VisitStmt_(const AttrStmtNode* op) final {
49 if (op->attr_key == "pragma_debug_skip_region") {
50 return MakeEvaluate(0);
51 } else if (op->attr_key == attr::async_wait_queue_scope) {
52 auto wait_attrs = GetAsyncWaitAttributes(op);
53 auto wait_cnt = wait_attrs.second;
54 arith::Analyzer ana;
55 if (ana.CanProve(wait_cnt < 0)) {
56 // A negative wait count can arise if it depends on a loop variable.
57 // For example, a wait count 1 - i can be negative after loop unrolling.
58 // We assume that such wait is a nop.
59 auto inner = op->body.as<AttrStmtNode>();
60 ICHECK(inner);
61 return StmtMutator::VisitStmt(inner->body);
62 }
63 }
64
65 Stmt stmt = StmtMutator::VisitStmt_(op);
66 op = stmt.as<AttrStmtNode>();
67 return is_no_op(op->body) ? MakeEvaluate(op->value) : stmt;
68 }
69 Stmt VisitStmt_(const IfThenElseNode* op) final {
70 Stmt stmt = StmtMutator::VisitStmt_(op);
71 op = stmt.as<IfThenElseNode>();
72 if (op->else_case.defined()) {
73 if (is_no_op(op->else_case)) {
74 if (is_no_op(op->then_case)) {
75 return MakeEvaluate(op->condition);
76 } else {
77 return IfThenElse(op->condition, op->then_case);
78 }
79 } else {
80 return stmt;
81 }
82 } else {
83 if (is_no_op(op->then_case)) {
84 return MakeEvaluate(op->condition);
85 } else {
86 return stmt;
87 }
88 }
89 }
90 Stmt VisitStmt_(const ForNode* op) final {
91 var_range_map_[op->loop_var.get()] = arith::IntSet::FromMinExtent(op->min, op->extent);
92 auto extent_range = arith::EvalSet(op->extent, var_range_map_);
93 if (!arith::is_neg_inf(extent_range.max()) && !arith::is_pos_inf(extent_range.max()) &&
94 analyzer_.CanProve(extent_range.max() <= 0)) {
95 return Evaluate(0);
96 }
97 Stmt stmt = StmtMutator::VisitStmt_(op);
98 var_range_map_.erase(op->loop_var.get());
99 op = stmt.as<ForNode>();
100 if (is_zero(op->extent)) {
101 return Evaluate(0);
102 }
103 return is_no_op(op->body) ? MakeEvaluate({op->min, op->extent}) : stmt;
104 }
105 Stmt VisitStmt_(const AllocateNode* op) final {
106 Stmt stmt = StmtMutator::VisitStmt_(op);
107 op = stmt.as<AllocateNode>();
108 return is_no_op(op->body) ? MakeEvaluate(op->extents) : stmt;
109 }
110
111 Stmt VisitStmt_(const ProducerRealizeNode* op) final {
112 Stmt stmt = StmtMutator::VisitStmt_(op);
113 op = stmt.as<ProducerRealizeNode>();
114 return is_no_op(op->body) ? op->body : stmt;
115 }
116 Stmt VisitStmt_(const EvaluateNode* op) final {
117 if (SideEffect(op->value) > CallEffectKind::kReadState) return GetRef<Stmt>(op);
118 return Evaluate(0);
119 }
120
121 Stmt VisitStmt_(const SeqStmtNode* op) final {
122 Stmt ret = StmtMutator::VisitSeqStmt_(op, true);
123 op = ret.as<SeqStmtNode>();
124 ICHECK(op != nullptr);
125 bool need_compact = false;
126 for (size_t i = 0; i < op->size(); ++i) {
127 if (is_no_op(op->seq[i])) need_compact = true;
128 }
129 if (need_compact) {
130 auto n = CopyOnWrite(op);
131 size_t top = 0;
132 for (size_t i = 0; i < n->seq.size(); ++i) {
133 if (!is_no_op(n->seq[i])) {
134 n->seq.Set(top++, n->seq[i]);
135 }
136 }
137 if (top == 1) {
138 return n->seq[0];
139 } else {
140 n->seq.resize(top);
141 return Stmt(n);
142 }
143 } else {
144 if (op->size() == 1) {
145 return op->seq[0];
146 } else {
147 return ret;
148 }
149 }
150 }
151
152 private:
153 Stmt MakeEvaluate(PrimExpr value) {
154 if (SideEffect(value) > CallEffectKind::kReadState) {
155 return Evaluate(value);
156 } else {
157 return Evaluate(0);
158 }
159 }
160 Stmt MakeEvaluate(const Array<PrimExpr>& values) {
161 Stmt stmt;
162 for (PrimExpr e : values) {
163 if (SideEffect(e) > CallEffectKind::kReadState) {
164 if (stmt.defined()) {
165 stmt = SeqStmt({stmt, Evaluate(e)});
166 } else {
167 stmt = Evaluate(e);
168 }
169 }
170 }
171 return stmt.defined() ? stmt : Evaluate(0);
172 }
173
174 std::unordered_map<const VarNode*, arith::IntSet> var_range_map_;
175 arith::Analyzer analyzer_;
176};
177
178Stmt RemoveNoOp(Stmt stmt) { return NoOpRemover()(std::move(stmt)); }
179
180namespace transform {
181
182Pass RemoveNoOp() {
183 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
184 auto* n = f.CopyOnWrite();
185 n->body = NoOpRemover()(std::move(n->body));
186 return f;
187 };
188 return CreatePrimFuncPass(pass_func, 0, "tir.RemoveNoOp", {});
189}
190
191TVM_REGISTER_GLOBAL("tir.transform.RemoveNoOp").set_body_typed(RemoveNoOp);
