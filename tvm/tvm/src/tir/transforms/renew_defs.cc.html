20/*!
21 * \file renew_defs.cc
22 * \brief Renew the definition nodes for a TIR, including Var, Buffer and IterVar.
23 */
24
25#include <tvm/tir/stmt_functor.h>
26#include <tvm/tir/transform.h>
27
28#include "../ir/functor_common.h"

30namespace tvm {
31namespace tir {

33#define STMT_REGENERATE_VAR_DEF(NODE, FIELD) \
34 Stmt VisitStmt_(const NODE* op) final { \
35 Var new_var = this->ReDefineVar(op->FIELD); \
36 Stmt stmt = StmtExprMutator::VisitStmt_(op); \
37 op = stmt.as<NODE>(); \
38 ICHECK(op != nullptr); \
39 auto n = make_object<NODE>(*op); \
40 n->FIELD = std::move(new_var); \
41 return Stmt(n); \
42 }
43
44class RenewDefMutator : public StmtExprMutator {
45 public:
46 static PrimFunc Transform(const PrimFunc& func) {
47 RenewDefMutator generator;
48 // Redefine params
49 Array<Var> params;
50 for (const auto& param : func->params) {
51 params.push_back(generator.ReDefineVar(param));
52 }
53 // Redefine buffers in order
54 // TODO(Siyuan Feng): checking var is used after define
55 Map<tir::Var, Buffer> buffer_map;
56 for (const auto& param : func->params) {
57 if (param->dtype.is_handle()) {
58 const Buffer& buffer = func->buffer_map.at(param);
59 Var new_param = Downcast<Var>(generator.VisitExpr(param));
60 Buffer new_buffer = generator.VisitBuffer(buffer, true);
61 buffer_map.Set(new_param, new_buffer);
62 }
63 }
64 // Visit body
65 Stmt body = generator(func->body);
66 // Recreate function
67 auto n = make_object<PrimFuncNode>(*func.get());
68 n->params = std::move(params);
69 n->buffer_map = std::move(buffer_map);
70 n->body = std::move(body);
71 return PrimFunc(n);
72 }
73
74 private:
75 Stmt operator()(Stmt stmt) {
76 // override StmtMutator::operator() to disable copy_on_write
77 // Since this pass tries to explict create a new function rather than update the existing one
78 allow_copy_on_write_ = false;
79 return VisitStmt(stmt);
80 }
81
82 PrimExpr VisitExpr(const PrimExpr& expr) final {
83 auto it = remap_.find(expr);
84 if (it != remap_.end()) {
85 return Downcast<PrimExpr>((*it).second);
86 } else {
87 return ExprMutator::VisitExpr(expr);
88 }
89 }
90
91 private:
92 STMT_REGENERATE_VAR_DEF(LetStmtNode, var);
93 STMT_REGENERATE_VAR_DEF(AllocateNode, buffer_var);
94 STMT_REGENERATE_VAR_DEF(AllocateConstNode, buffer_var);
95 STMT_REGENERATE_VAR_DEF(ForNode, loop_var);
96
97 Stmt VisitStmt_(const BlockNode* op) final {
98 // Step 0. Re-define Itervars
99 Array<IterVar> iter_vars = MutateArray(
100 op->iter_vars, std::bind(&RenewDefMutator::VisitIterVar, this, std::placeholders::_1));
101
102 // Step 1. Re-define buffers allocate under the block
103 Array<Buffer> alloc_buffers = MutateArray(
104 op->alloc_buffers,
105 std::bind(&RenewDefMutator::VisitBuffer, this, std::placeholders::_1, /*define=*/true));
106
107 // Step 2. Re-define match_buffers
108 Array<MatchBufferRegion> match_buffers =
109 MutateArray(op->match_buffers,
110 std::bind(&RenewDefMutator::VisitMatchBuffer, this, std::placeholders::_1));
111
112 // Step 3. Visit body
113 Stmt stmt = StmtExprMutator::VisitStmt_(op);
114 op = stmt.as<BlockNode>();
115 ICHECK(op);
116
117 // Step 4. Revisit access region
118 Array<BufferRegion> reads = MutateArray(
119 op->reads, std::bind(&RenewDefMutator::VisitBufferRegion, this, std::placeholders::_1));
120 Array<BufferRegion> writes = MutateArray(
121 op->writes, std::bind(&RenewDefMutator::VisitBufferRegion, this, std::placeholders::_1));
122
123 // Step 5. Regenerate block. Since the defs are changed, we need to create a new block
124 auto n = make_object<BlockNode>(*op);
125 n->iter_vars = std::move(iter_vars);
126 n->alloc_buffers = std::move(alloc_buffers);
127 n->match_buffers = std::move(match_buffers);
128 n->reads = std::move(reads);
129 n->writes = std::move(writes);
130
131 return Stmt(n);
132 }
133
134 Stmt VisitStmt_(const BufferStoreNode* op) final {
135 Stmt stmt = StmtExprMutator::VisitStmt_(op);
136 op = stmt.as<BufferStoreNode>();
137 ICHECK(op != nullptr);
138 Buffer buffer = VisitDeclOrRemapBuffer(op->buffer);
139 if (buffer.same_as(op->buffer)) {
140 return stmt;
141 } else {
142 auto n = make_object<BufferStoreNode>(*op);
143 n->buffer = std::move(buffer);
144 return BufferStore(n);
145 }
146 }
147
148 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
149 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
150 op = expr.as<BufferLoadNode>();
151 ICHECK(op != nullptr);
152 Buffer buffer = VisitDeclOrRemapBuffer(op->buffer);
153 if (buffer.same_as(op->buffer)) {
154 return expr;
155 } else {
156 auto n = make_object<BufferLoadNode>(*op);
157 n->buffer = std::move(buffer);
158 return BufferLoad(n);
159 }
160 }
161
162 PrimExpr VisitExpr_(const LoadNode* op) final {
163 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
164 return PrimExpr();
165 }
166
167 Stmt VisitStmt_(const StoreNode* op) final {
168 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
169 return Stmt();
170 }
171
172 private:
173 Var ReDefineVar(const Var& var) {
174 Var new_var = Var(make_object<VarNode>(*var.get()));
175 this->AddDefRemap(var, new_var);
176 return new_var;
177 }
178
179 template <typename T>
180 void AddDefRemap(const T& source, const T& target) {
181 ICHECK(remap_.count(source) == 0);
182 remap_.Set(source, target);
183 }
184
185 Buffer VisitBuffer(const Buffer& buffer, bool define = false) {
186 auto it = remap_.find(buffer);
187 if (it != remap_.end()) {
188 return Downcast<Buffer>((*it).second);
189 }
190 ICHECK(define);
191
192 auto redefine_if_is_var = [this](const PrimExpr& expr) -> PrimExpr {
193 auto it = remap_.find(expr);
194 if (it != remap_.end()) {
195 return Downcast<PrimExpr>((*it).second);
196 } else if (const VarNode* var = expr.as<VarNode>()) {
197 return this->ReDefineVar(GetRef<Var>(var));
198 } else {
199 return ExprMutator::VisitExpr(expr);
200 }
201 };
202
203 // update data
204 Var data = Downcast<Var>(redefine_if_is_var(buffer->data));
205 // update shape
206 Array<PrimExpr> shape = MutateArray(buffer->shape, redefine_if_is_var);
207 // update strides
208 Array<PrimExpr> strides = MutateArray(buffer->strides, redefine_if_is_var);
209 // update elem_offset
210 PrimExpr elem_offset = redefine_if_is_var(buffer->elem_offset);
211
212 auto n = make_object<BufferNode>(*buffer.get());
213 n->data = std::move(data);
214 n->shape = std::move(shape);
215 n->strides = std::move(strides);
216 n->elem_offset = std::move(elem_offset);
217 Buffer new_buffer(n);
218 this->AddDefRemap(buffer, new_buffer);
219 return new_buffer;
220 }
221
222 IterVar VisitIterVar(const IterVar& iter_var) {
223 auto it = remap_.find(iter_var);
224 if (it != remap_.end()) {
225 return Downcast<IterVar>((*it).second);
226 }
227 PrimExpr min = VisitExpr(iter_var->dom->min);
228 PrimExpr extent = VisitExpr(iter_var->dom->extent);
229 IterVar new_iter_var(Range(min, extent), ReDefineVar(iter_var->var), iter_var->iter_type,
230 iter_var->thread_tag);
231 this->AddDefRemap(iter_var, new_iter_var);
232 return new_iter_var;
233 }
234
235 Buffer VisitDeclOrRemapBuffer(const Buffer& buffer) {
236 // If the buffer has been remapped, return the remapped buffer, otherwise,
237 // return the declared one.
238 // Due to a recent PR, we can allow undefined buffer appearing in BufferLoad/Store. We need
239 // to remap them but will not create new var
240 auto it = remap_.find(buffer);
241 if (it != remap_.end()) {
242 return Downcast<Buffer>((*it).second);
243 }
244 Var data = Downcast<Var>(VisitExpr(buffer->data));
245 Array<PrimExpr> shape = MutateArray(
246 buffer->shape, std::bind(&RenewDefMutator::VisitExpr, this, std::placeholders::_1));
247 Array<PrimExpr> strides = MutateArray(
248 buffer->strides, std::bind(&RenewDefMutator::VisitExpr, this, std::placeholders::_1));
249 PrimExpr elem_offset = VisitExpr(buffer->elem_offset);
250
251 auto n = make_object<BufferNode>(*buffer.get());
252 n->data = std::move(data);
253 n->shape = std::move(shape);
254 n->strides = std::move(strides);
255 n->elem_offset = std::move(elem_offset);
256 Buffer new_buffer(n);
257 this->AddDefRemap(buffer, new_buffer);
258 return new_buffer;
259 }
260
261 MatchBufferRegion VisitMatchBuffer(const MatchBufferRegion& match_buffer) {
262 Buffer buffer = VisitBuffer(match_buffer->buffer, /*define=*/true);
263 BufferRegion region = VisitBufferRegion(match_buffer->source);
264 return MatchBufferRegion(std::move(buffer), std::move(region));
265 }
266
267 Range VisitRange(const Range& range) {
268 PrimExpr min = VisitExpr(range->min);
269 PrimExpr extent = VisitExpr(range->extent);
270 if (min.same_as(range->min) && extent.same_as(range->extent)) {
271 return range;
272 } else {
273 return Range::FromMinExtent(std::move(min), std::move(extent));
274 }
275 }
276
277 BufferRegion VisitBufferRegion(const BufferRegion& buffer_region) {
278 Buffer buffer = VisitBuffer(buffer_region->buffer);
279 Array<Range> region =
280 MutateArray(buffer_region->region,
281 std::bind(&RenewDefMutator::VisitRange, this, std::placeholders::_1));
282 if (buffer.same_as(buffer_region->buffer) && region.same_as(buffer_region->region)) {
283 return buffer_region;
284 } else {
285 return BufferRegion(std::move(buffer), std::move(region));
286 }
287 }
288
289 Map<ObjectRef, ObjectRef> remap_;
290};
291
292PrimFunc RenewDefs(const PrimFunc& func) { return RenewDefMutator::Transform(func); }

294TVM_REGISTER_GLOBAL("tir.RenewDefs").set_body_typed(RenewDefs);
