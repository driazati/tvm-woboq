35/*!
36 * \brief Helper mutator to implement lowering of custom datatypes.
37 *
38 * Lowering datatypes works as follows: for every expression containing a custom
39 * datatype, we search for a global (registered by the implementer of the custom
40 * datatype) for lowering this type of expression, and uses it to lower the
41 * expression.
42 */
43class CustomDatatypesLowerer : public StmtExprMutator {
44 public:
45 explicit CustomDatatypesLowerer(const std::string& target) : target_(target) {}
46
47 PrimExpr VisitExpr_(const CastNode* op) final {
48 auto type_code = op->dtype.code();
49 auto src_type_code = op->value.dtype().code();
50 // If either datatype is a registered custom datatype, we must lower.
51 bool to_be_lowered = datatype::Registry::Global()->GetTypeRegistered(type_code) ||
52 datatype::Registry::Global()->GetTypeRegistered(src_type_code);
53 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
54 if (to_be_lowered) {
55 auto lower = datatype::GetCastLowerFunc(target_, type_code, src_type_code);
56 ICHECK(lower) << "Cast lowering function for target " << target_ << " destination type "
57 << static_cast<unsigned>(type_code) << " source type "
58 << static_cast<unsigned>(src_type_code) << " not found";
59 return (*lower)(expr);
60 }
61 return expr;
62 }
63
64 PrimExpr VisitExpr_(const FloatImmNode* imm) final {
65 auto type_code = imm->dtype.code();
66 auto e = GetRef<PrimExpr>(imm);
67 if (datatype::Registry::Global()->GetTypeRegistered(type_code)) {
68 auto lower = datatype::GetFloatImmLowerFunc(target_, type_code);
69 ICHECK(lower) << "FloatImm lowering function for target " << target_ << " type "
70 << static_cast<unsigned>(type_code) << " not found";
71 return (*lower)(e);
72 }
73 return e;
74 }
75
76 PrimExpr VisitExpr_(const VarNode* op) final {
77 Var var = GetRef<Var>(op);
78
79 auto itr = var_remap_.find(var);
80 if (itr != var_remap_.end()) {
81 return itr->second;
82 } else {
83 return std::move(var);
84 }
85 }
86
87 Stmt VisitStmt_(const AllocateNode* allocate) final {
88 bool to_be_lowered = datatype::Registry::Global()->GetTypeRegistered(allocate->dtype.code());
89
90 if (to_be_lowered) {
91 auto new_allocate_type = DataType::UInt(allocate->dtype.bits(), allocate->dtype.lanes());
92 auto new_buffer_var =
93 Var(allocate->buffer_var->name_hint, PointerType(PrimType(new_allocate_type)));
94 var_remap_[allocate->buffer_var] = new_buffer_var;
95
96 Stmt stmt = StmtExprMutator::VisitStmt_(allocate);
97 allocate = stmt.as<AllocateNode>();
98
99 return Allocate(new_buffer_var, new_allocate_type, allocate->extents, allocate->condition,
100 allocate->body);
101 } else {
102 return StmtExprMutator::VisitStmt_(allocate);
103 }
104 }
105
106 PrimExpr VisitExpr_(const LoadNode* op) final {
107 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
108 return PrimExpr();
109 }
110
111 Stmt VisitStmt_(const StoreNode* op) final {
112 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
113 return Stmt();
114 }
115
116 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
117 auto node = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
118 auto modified = VisitBufferAccess(node);
119
120 // Not needed for BufferStoreNode, so we can't just call
121 // LegalizeDtype() in VisitBufferAccess.
122 if (node.same_as(modified)) {
123 return std::move(node);
124 } else {
125 auto writer = modified.CopyOnWrite();
126 writer->LegalizeDType();
127 return std::move(modified);
128 }
129 }
130
131 Stmt VisitStmt_(const BufferStoreNode* op) final {
132 auto node = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
133 return VisitBufferAccess(std::move(node));
134 }
135
136 template <typename Node>
137 Node VisitBufferAccess(Node node) {
138 Buffer new_buf = GetRemappedBuffer(node->buffer);
139 if (!new_buf.same_as(node->buffer)) {
140 auto writer = node.CopyOnWrite();
141 writer->buffer = new_buf;
142 }
143
144 return node;
145 }
146
147 Buffer GetRemappedBuffer(Buffer buf) {
148 auto key = buf;
149 auto cache_it = buf_remap_.find(key);
150 if (cache_it != buf_remap_.end()) {
151 return cache_it->second;
152 }
153
154 bool to_be_lowered = datatype::Registry::Global()->GetTypeRegistered(buf->dtype.code());
155
156 if (to_be_lowered) {
157 auto new_load_type = DataType::UInt(buf->dtype.bits());
158 auto writer = buf.CopyOnWrite();
159 writer->dtype = new_load_type;
160
161 auto var_it = var_remap_.find(buf->data);
162 if (var_it != var_remap_.end()) {
163 writer->data = var_it->second;
164 }
165 }
166
167 buf_remap_[key] = buf;
168 return buf;
169 }
170
171 Stmt VisitStmt_(const AttrStmtNode* op) final {
172 Stmt ret = StmtExprMutator::VisitStmt_(op);
173 op = ret.as<AttrStmtNode>();
174 // Due to legacy reasons, some attr node can contain
175 // information(e.g. alignment) of buffer variables.
176 // remap these vars when needed
177 // TODO(tvm-team): remove the rewriting once the buffer var
178 // attrs are being refactored into the corresponding definition node
179 if (const auto* var_node = op->node.as<VarNode>()) {
180 auto it = var_remap_.find(GetRef<Var>(var_node));
181 if (it != var_remap_.end()) {
182 return AttrStmt(it->second, op->attr_key, op->value, op->body);
183 }
184 }
185 return ret;
186 }
187
188 PrimExpr VisitExpr_(const CallNode* call) final {
189 bool to_be_lowered = datatype::Registry::Global()->GetTypeRegistered(call->dtype.code());
190 PrimExpr expr = StmtExprMutator::VisitExpr_(call);
191 call = expr.as<CallNode>();
192 if (to_be_lowered) {
193 auto op = call->op.as<OpNode>();
194 ICHECK(op != nullptr) << "Lowering non-intrinsic Calls not implemented";
195 auto lower = datatype::GetIntrinLowerFunc(target_, op->name, call->dtype.code());
196 ICHECK(lower) << "Intrinsic lowering function for target " << target_ << ", intrinsic name "
197 << op->name << ", type " << static_cast<unsigned>(call->dtype.code())
198 << " not found";
199 return (*lower)(expr);
200 }
201 return expr;
202 }
203
204#define TVM_DEFINE_MUTATE_CUSTOM_DTYPE(OP, NodeName) \
205 PrimExpr VisitExpr_(const NodeName* op) final { \
206 auto type_code = op->dtype.code(); \
207 bool to_be_lowered = datatype::Registry::Global()->GetTypeRegistered(type_code); \
208 PrimExpr expr = StmtExprMutator::VisitExpr_(op); \
209 op = expr.as<NodeName>(); \
210 if (to_be_lowered) { \
211 auto lower = datatype::Get##OP##LowerFunc(target_, type_code); \
212 ICHECK(lower) << #OP " lowering function for target " << target_ << " type " \
213 << static_cast<unsigned>(type_code) << " not found"; \
214 return (*lower)(expr); \
215 } \
216 return expr; \
217 }
218
219 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Add, AddNode);
220 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Sub, SubNode);
221 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Mul, MulNode);
222 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Div, DivNode);
223 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Mod, ModNode);
224 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Min, MinNode);
225 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(Max, MaxNode);
226 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(EQ, EQNode);
227 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(NE, NENode);
228 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(LT, LTNode);
229 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(LE, LENode);
230 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(GT, GTNode);
231 TVM_DEFINE_MUTATE_CUSTOM_DTYPE(GE, GENode);
232 // Later changes may need to add more mutate functions as we support workloads with more ops.
233
234#undef TVM_DEFINE_MUTATE_CUSTOM_DTYPE
235
236 private:
237 std::string target_;
238 // remap buffer vars
239 std::unordered_map<Var, Var, ObjectPtrHash, ObjectPtrEqual> var_remap_;
240 std::unordered_map<Buffer, Buffer, ObjectPtrHash, ObjectPtrEqual> buf_remap_;
241};
242
243namespace transform {
244
245Pass LowerCustomDatatypes() {
246 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
247 auto* n = f.CopyOnWrite();
248 auto target = f->GetAttr<Target>(tvm::attr::kTarget);
249 ICHECK(target.defined()) << "LowerCustomDatatypes: Require the target attribute";
250
251 n->body = CustomDatatypesLowerer(target.value()->kind->name)(std::move(n->body));
252 return f;
253 };
254 return CreatePrimFuncPass(pass_func, 0, "tir.LowerCustomDatatypes", {});
255}
