37// Calculate the statistics of packed function.
38// These information are needed during codegen.
39class BuiltinLower : public StmtExprMutator {
40 public:
41 // NOTE: Right now, we make the following scoping requirement
42 // for memory allocated by the following primitives
43 // - tvm_stack_make_array
44 // - tvm_stack_make_shape
45 // - arg stack
46 //
47 // Scoping and liveness rules:
48 // - Every call_packed introduce a new scope.
49 // - The memory allocated by tvm_stack_make_array/make_shape will
50 // no longer become valid outside the scope (and may be reused by
51 // subsequent call_packed.
52 // - TODO(tvm-team): we might consider a root scope so stack_make_shape
53 // can be called out-side call_packed.
54 //
55 // Example:
56 // {
57 // call_packed(make_shape1(...),
58 // call_packed(make_shape2(...))
59 // call_packed(make_shape3(...))
60 // }
61 //
62 // In this case, make_shape1 and make_shape2 should not share memory,
63 // but they can share memory with make_shape3.
64 //
65 // Rationale: most of the packed calls needs their own internal
66 // argument stack, and those stack can be shared across calls.
67 // Scoping is a quick way to enable sharing without having
68 // to do full-scale liveness analysis and it does its job.
69 // Alternative approaches can also be used.
70 struct StackSizes {
71 // If a tvm_stack_make_shape call has no arguments, it is still
72 // valid and represents a scalar shape (). Therefore, -1 is used
73 // to represent "no shape arguments exist", while 0 represents
74 // "shape arguments exist, all of which are size 0".
75 int64_t shape_stack{-1};
76 uint64_t array_stack{0};
77 uint64_t arg_stack{0};
78 };
79
80 // Record stack frame for existing scope.
81 struct AllocaScope {
82 Buffer stack_shape;
83 Var stack_array = Var("stack_array", DataType::Handle());
84 Var stack_value = Var("stack_value", DataType::Handle());
85 Buffer stack_tcode;
86
87 StackSizes max_sizes;
88 StackSizes run_sizes;
89
90 void UpdateMax() {
91 max_sizes.shape_stack = std::max(max_sizes.shape_stack, run_sizes.shape_stack);
92 max_sizes.array_stack = std::max(max_sizes.array_stack, run_sizes.array_stack);
93 max_sizes.arg_stack = std::max(max_sizes.arg_stack, run_sizes.arg_stack);
94 }
95
96 void AssertMaxIsValid() const {
97 ICHECK((max_sizes.shape_stack >= run_sizes.shape_stack) ||
98 (max_sizes.array_stack >= run_sizes.array_stack) ||
99 (max_sizes.arg_stack >= run_sizes.arg_stack));
100 }
101 };
102
103 Stmt Build(Stmt stmt) { return this->VisitBodyAndRealizeAlloca(stmt); }
104
105 StackSizes GetMaxStack(Stmt stmt) {
106 BuiltinLower precheck;
107 precheck.is_precheck_ = true;
108 precheck.device_id_ = this->device_id_;
109 precheck.device_type_ = this->device_type_;
110
111 precheck.alloca_scope_.emplace_back();
112 {
113 // NOTE: this scope reference is invalid after any mutation is applied to alloca_scope_.
114 auto& scope = precheck.alloca_scope_.back();
115 scope.stack_shape =
116 decl_buffer({IntImm(DataType::Int(64), 0)}, DataType::Int(64), "stack_shape");
117 scope.stack_tcode =
118 decl_buffer({IntImm(DataType::UInt(64), 0)}, DataType::Int(32), "stack_tcode");
119 }
120
121 precheck.VisitStmt(stmt);
122
123 ICHECK_EQ(precheck.alloca_scope_.size(), 1);
124 return precheck.alloca_scope_[0].max_sizes;
125 }
126
127 // Allcoate stack frames, only at parallel-for or root.
128 Stmt VisitBodyAndRealizeAlloca(Stmt stmt) {
129 // Only perform the precheck up to the point where we would add a
130 // new scope.
131 if (is_precheck_) {
132 return stmt;
133 }
134
135 alloca_scope_.emplace_back();
136 {
137 // NOTE: this scope reference is invalid after any mutation is applied to alloca_scope_.
138 auto& scope = alloca_scope_.back();
139
140 // Initial check to identify maximum stack sizes. These are used
141 // to construct Buffer objects to hold the stack, which are then
142 // used when mutating.
143 scope.max_sizes = GetMaxStack(stmt);
144
145 if (scope.max_sizes.shape_stack != -1) {
146 scope.stack_shape = decl_buffer({IntImm(DataType::Int(64), scope.max_sizes.shape_stack)},
147 DataType::Int(64), "stack_shape");
148 stmt = LetStmt(scope.stack_shape->data, StackAlloca("shape", scope.max_sizes.shape_stack),
149 stmt);
150 }
151
152 if (scope.max_sizes.array_stack != 0) {
153 stmt = LetStmt(scope.stack_array, StackAlloca("array", scope.max_sizes.array_stack), stmt);
154 }
155
156 if (scope.max_sizes.arg_stack != 0) {
157 scope.stack_tcode = decl_buffer({IntImm(DataType::UInt(64), scope.max_sizes.arg_stack)},
158 DataType::Int(32), "stack_tcode");
159 stmt =
160 LetStmt(scope.stack_value, StackAlloca("arg_value", scope.max_sizes.arg_stack), stmt);
161
162 stmt = LetStmt(scope.stack_tcode->data, StackAlloca("arg_tcode", scope.max_sizes.arg_stack),
163 stmt);
164 }
165 }
166
167 stmt = this->VisitStmt(stmt);
168
169 ICHECK(!alloca_scope_.empty());
170 alloca_scope_.pop_back();
171
172 return stmt;
173 }
174
175 Stmt VisitStmt(const Stmt& s) final {
176 // allocate space to hold prepare stmts before s
177 prep_seq_stack_.emplace_back(std::vector<Stmt>());
178
179 auto scope_size = alloca_scope_.size();
180 auto stmt = StmtExprMutator::VisitStmt(s);
181 {
182 // NOTE: this scope reference is invalid after any mutation is applied to alloca_scope_.
183 auto& scope = alloca_scope_.back();
184 // This invariant asserts the assumption that
185 // make_stack_shape only happens within a call_packed.
186 // We could relax this in the future if we want to
187 // introduce root scope as a separate scope
188 ICHECK_EQ(alloca_scope_.size(), scope_size)
189 << "alloca_scope_ length is different before and after recursion";
190 ICHECK_EQ(scope.run_sizes.shape_stack, -1)
191 << "Expect no tvm_stack_make_shape outside of CallNodes";
192 ICHECK_EQ(scope.run_sizes.array_stack, 0)
193 << "Expect no tvm_stack_make_array outside of CallNodes";
194 }
195
196 auto prep_seq = std::move(prep_seq_stack_.back());
197 prep_seq_stack_.pop_back();
198
199 if (prep_seq.size() != 0) {
200 Stmt ret = SeqStmt::Flatten(prep_seq, stmt);
201 return ret;
202 } else {
203 return stmt;
204 }
205 }

207 Stmt VisitStmt_(const LetStmtNode* op) final {
208 if (const CallNode* call = op->value.as<CallNode>()) {
209 if (call->op.same_as(builtin::nd_mem_alloc_with_scope())) {
210 return StmtExprMutator::VisitStmt(MakeNdMemAllocWithScope(op, call));
211 }
212 }
213 return StmtExprMutator::VisitStmt_(op);
214 }
215
216 Stmt VisitStmt_(const AllocateNode* op) {
217 // Lower allocate to device allocate when needed.
218 Stmt stmt = StmtExprMutator::VisitStmt_(op);
219 op = stmt.as<AllocateNode>();
220 // Get constant allocation bound.
221 int64_t nbytes = GetVectorBytes(op->dtype);
222 // If the buffers are for CPU and have global scope,
223 // and less than runtime::kMaxStackAlloca heuristic
224 // they are not serviced with TVMBackendWorkspaceAlloc calls
225 // to be placed on stack.
226 if (op->annotations.count(transform::kDisableLowerTVMBuiltin)) {
227 if (Downcast<Bool>(op->annotations[transform::kDisableLowerTVMBuiltin])) {
228 return stmt;
229 }
230 }
231 if (device_type_.defined()) {
232 if (const auto* dev_type = device_type_.as<IntImmNode>()) {
233 auto storage_scope = Downcast<PointerType>(op->buffer_var->type_annotation)->storage_scope;
234 if (dev_type->value == kDLCPU && storage_scope == "global") {
235 size_t constant_size = op->ConstantAllocationSize();
236 if (constant_size > 0 && constant_size * nbytes < runtime::kMaxStackAlloca) {
237 return stmt;
238 }
239 }
240 }
241 }
242 PrimExpr total_bytes = make_const(op->extents[0].dtype(), nbytes);
243 for (size_t i = 0; i < op->extents.size(); ++i) {
244 total_bytes = total_bytes * op->extents[i];
245 }
246 ICHECK(device_type_.defined()) << "Unknown device type in current IR";
247 ICHECK(device_id_.defined()) << "Unknown device id in current IR";
248 Stmt throw_last_error = Evaluate(Call(DataType::Int(32), builtin::tvm_throw_last_error(), {}));
249
250 Stmt body = SeqStmt({IfThenElse(Call(DataType::Bool(1), builtin::isnullptr(), {op->buffer_var}),
251 throw_last_error),
252 op->body});
253 Stmt alloca = LetStmt(
254 op->buffer_var,
255 Call(op->buffer_var.dtype(), Op::Get("tir.TVMBackendAllocWorkspace"),
256 {cast(DataType::Int(32), device_type_), cast(DataType::Int(32), device_id_),
257 cast(DataType::UInt(64), total_bytes), IntImm(DataType::Int(32), op->dtype.code()),
258 IntImm(DataType::Int(32), op->dtype.bits())}),
259 body);
260
261 PrimExpr free_op = Call(DataType::Int(32), Op::Get("tir.TVMBackendFreeWorkspace"),
262 {cast(DataType::Int(32), device_type_),
263 cast(DataType::Int(32), device_id_), op->buffer_var});
264 Stmt free_stmt = IfThenElse(free_op != make_zero(DataType::Int(32)), throw_last_error);
265 body = SeqStmt({alloca, free_stmt});
266 body = AttrStmt(op->buffer_var, attr::storage_alignment,
267 make_const(DataType::Int(32), runtime::kTempAllocaAlignment), body);
268 return body;
269 }

271 Stmt VisitStmt_(const AttrStmtNode* op) final {
272 if (op->attr_key == attr::device_id) {
273 ICHECK(!device_id_.defined());
274 device_id_ = op->value;
275 return this->VisitStmt(op->body);
276 } else if (op->attr_key == attr::device_type) {
277 ICHECK(!device_type_.defined());
278 device_type_ = op->value;
279 return this->VisitStmt(op->body);
280 } else {
281 return StmtExprMutator::VisitStmt_(op);
282 }
283 }
284 Stmt VisitStmt_(const ForNode* op) final {
285 PrimExpr min = this->VisitExpr(op->min);
286 PrimExpr extent = this->VisitExpr(op->extent);
287 Stmt body;
288
289 if (op->kind == ForKind::kParallel) {
290 body = this->VisitBodyAndRealizeAlloca(op->body);
291 } else {
292 body = this->VisitStmt(op->body);
293 }
294
295 if (min.same_as(op->min) && extent.same_as(op->extent) && body.same_as(op->body)) {
296 return GetRef<Stmt>(op);
297 } else {
298 auto n = CopyOnWrite(op);
299 n->min = std::move(min);
300 n->extent = std::move(extent);
301 n->body = std::move(body);
302 return Stmt(n);
303 }
304 }
305 PrimExpr VisitExpr_(const CallNode* op) final {
306 if (op->op.same_as(builtin::tvm_call_packed())) {
307 return MakeCallPacked(op, /* use_string_lookup */ true);
308 } else if (op->op.same_as(builtin::tvm_call_cpacked())) {
309 return MakeCallPacked(op, /* use_string_lookup */ false);
310 } else if (op->op.same_as(builtin::tvm_call_trace_packed())) {
311 return MakeCallTracePacked(op);
312 } else if (op->op.same_as(builtin::tvm_stack_make_shape())) {
313 return MakeShape(op);
314 } else if (op->op.same_as(builtin::tvm_stack_make_array())) {
315 return MakeArray(op);
316 } else if (op->op.same_as(builtin::tvm_context_id())) {
317 return make_zero(op->dtype);
318 } else if (op->op.same_as(builtin::mem_copy())) {
319 return MakeMemCopy(op);
320 } else {
321 return StmtExprMutator::VisitExpr_(op);
322 }
323 }
324
325 PrimExpr MakeMemCopy(const CallNode* op) {
326 PrimExpr dst = op->args[0];
327 PrimExpr src = op->args[1];
328 PrimExpr size = op->args[2];
329
330 std::string fdevapi_prefix =
331 "device_api." + std::string(runtime::DeviceName(device_type_.as<IntImmNode>()->value));
332
333 Call call_packed = Call(DataType::Int(32), builtin::tvm_call_packed(),
334 {StringImm(fdevapi_prefix + ".mem_copy"), dst, src, size});
335 return VisitExpr(call_packed);
336 }

338 // call shape
339 PrimExpr MakeShape(const CallNode* op) {
340 // if args.size() == 0, it represents a scalar shape ()
341 ICHECK(!alloca_scope_.empty());
342 auto& scope = alloca_scope_.back();
343 auto& prep_seq = prep_seq_stack_.back();
344 if (scope.run_sizes.shape_stack == -1) {
345 scope.run_sizes.shape_stack = 0;
346 }
347 int64_t stack_begin = scope.run_sizes.shape_stack;
348 scope.run_sizes.shape_stack += op->args.size();
349 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
350 op = expr.as<CallNode>();
351 // no need to perform any store for a scalar shape
352 for (size_t i = 0; i < op->args.size(); ++i) {
353 prep_seq.emplace_back(BufferStore(scope.stack_shape, cast(DataType::Int(64), op->args[i]),
354 {ConstInt32(stack_begin + i)}));
355 }
356 return AddressOffset(scope.stack_shape->data, DataType::Int(64), stack_begin);
357 }
358 // make array
359 PrimExpr MakeArray(const CallNode* op) {
360 ICHECK(!alloca_scope_.empty());
361 auto& scope = alloca_scope_.back();
362 auto& prep_seq = prep_seq_stack_.back();
363
364 size_t idx = scope.run_sizes.array_stack;
365 scope.run_sizes.array_stack += 1;
366 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
367 op = expr.as<CallNode>();
368
369 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrData, op->args[0]));
370 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrShape, op->args[1]));
371 PrimExpr strides = op->args[2];
372 if (!strides.defined() || is_zero(strides)) {
373 strides = make_zero(DataType::Handle());
374 }
375 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrStrides, strides));
376 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrNDim, op->args[3]));
377 DataType dtype = op->args[4].dtype();
378 prep_seq.emplace_back(
379 TVMStructSet(scope.stack_array, idx, builtin::kArrTypeCode,
380 make_const(DataType::UInt(8), static_cast<int>(dtype.code()))));
381 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrTypeBits,
382 make_const(DataType::UInt(8), dtype.bits())));
383 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrTypeLanes,
384 make_const(DataType::UInt(16), dtype.lanes())));
385 // set byte offset
386 int data_bytes = GetVectorBytes(dtype);
387 PrimExpr elem_offset = op->args[5];
388 PrimExpr byte_offset;
389 if (!is_zero(elem_offset)) {
390 byte_offset = elem_offset * make_const(elem_offset.dtype(), data_bytes);
391 } else {
392 byte_offset = elem_offset;
393 }
394 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrByteOffset,
395 cast(DataType::UInt(64), byte_offset)));
396 ICHECK(device_type_.defined()) << "Unknown device type in current IR";
397 ICHECK(device_id_.defined()) << "Unknown device id in current IR";
398 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrDeviceId,
399 cast(DataType::Int(32), device_id_)));
400 prep_seq.emplace_back(TVMStructSet(scope.stack_array, idx, builtin::kArrDeviceType,
401 cast(DataType::Int(32), device_type_)));
402 return TVMStructGet(DataType::Handle(), scope.stack_array, idx, builtin::kArrAddr);
403 }
404 // call packed.
405 PrimExpr MakeCallPacked(const CallNode* op, bool use_string_lookup) {
406 auto& scope = alloca_scope_.back();
407 auto& prep_seq = prep_seq_stack_.back();
408
409 int64_t restore_shape_stack = scope.run_sizes.shape_stack;
410 size_t restore_array_stack = scope.run_sizes.array_stack;
411 size_t arg_stack_begin = scope.run_sizes.arg_stack;
412
413 size_t arg_count = op->args.size();
414
415 // cpacked expects a resource_handle parameter
416 if (!use_string_lookup) {
417 arg_count--;
418 }
419
420 scope.run_sizes.arg_stack += arg_count;
421 // Specially handle the buffer packed intrinsic
422 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
423 op = expr.as<CallNode>();
424 for (size_t i = 1; i < arg_count; ++i) {
425 PrimExpr stack_index = ConstInt32(arg_stack_begin + i - 1);
426 PrimExpr arg = op->args[i];
427 DataType t = arg.dtype();
428 DataType api_type = APIType(t);
429 if (t != api_type) {
430 arg = Cast(api_type, arg);
431 }
432 prep_seq.emplace_back(TVMStructSet(scope.stack_value,
433 static_cast<int>(arg_stack_begin + i - 1),
434 builtin::kTVMValueContent, arg));
435 int arg_tcode = api_type.code();
436 if (api_type.is_handle() && arg.as<StringImmNode>()) {
437 arg_tcode = kTVMStr;
438 }
439 if (IsArrayHandle(arg)) arg_tcode = kTVMDLTensorHandle;
440 prep_seq.emplace_back(BufferStore(scope.stack_tcode, ConstInt32(arg_tcode), {stack_index}));
441 }
442 // Verify stack size matches earlier value.
443 if (is_precheck_) {
444 scope.UpdateMax();
445 } else {
446 scope.AssertMaxIsValid();
447 }
448 scope.run_sizes.shape_stack = restore_shape_stack;
449 scope.run_sizes.array_stack = restore_array_stack;
450 scope.run_sizes.arg_stack = arg_stack_begin;
451 Array<PrimExpr> packed_args = {op->args[0], scope.stack_value, scope.stack_tcode->data,
452 ConstInt32(arg_stack_begin),
453 ConstInt32(arg_stack_begin + op->args.size() - 1)};
454
455 // cpacked call resource_handle
456 if (!use_string_lookup) {
457 PrimExpr last_arg = op->args[arg_count];
458 const VarNode* var_node = last_arg.as<VarNode>();
459 if (var_node != nullptr) {
460 tir::Var resource_handle = GetRef<Var>(var_node);
461 packed_args.push_back(StringImm(resource_handle->name_hint));
462 } else {
463 packed_args.push_back(last_arg);
464 }
465 }
466
467 auto builtin_call = use_string_lookup ? builtin::tvm_call_packed_lowered()
468 : builtin::tvm_call_cpacked_lowered();
469 return Call(op->dtype, builtin_call, packed_args);
470 }

472 PrimExpr MakeCallTracePacked(const CallNode* op) {
473 ICHECK(!alloca_scope_.empty());
474 auto& scope = alloca_scope_.back();
475 auto& prep_seq = prep_seq_stack_.back();
476
477 int64_t restore_shape_stack = scope.run_sizes.shape_stack;
478 size_t restore_array_stack = scope.run_sizes.array_stack;
479 size_t arg_stack_begin = scope.run_sizes.arg_stack;
480 scope.run_sizes.arg_stack += op->args.size();
481 size_t args_size = op->args.size();
482 ICHECK_GT(args_size, 0);
483 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
484 op = expr.as<CallNode>();
485 for (size_t i = 1; i < op->args.size(); ++i) {
486 PrimExpr stack_index = ConstInt32(arg_stack_begin + i - 1);
487 PrimExpr arg = op->args[i];
488 DataType t = arg.dtype();
489 DataType api_type = APIType(t);
490 if (t != api_type) {
491 arg = Cast(api_type, arg);
492 }
493 prep_seq.emplace_back(TVMStructSet(scope.stack_value,
494 static_cast<int>(arg_stack_begin + i - 1),
495 builtin::kTVMValueContent, arg));
496 int arg_tcode = api_type.code();
497 ICHECK(!IsArrayHandle(arg)) << "Trace does not support Buffers";
498 prep_seq.emplace_back(BufferStore(scope.stack_tcode, ConstInt32(arg_tcode), {stack_index}));
499 }
500 // Verify stack size matches earlier value.
501 if (is_precheck_) {
502 scope.UpdateMax();
503 } else {
504 scope.AssertMaxIsValid();
505 }
506 scope.run_sizes.shape_stack = restore_shape_stack;
507 scope.run_sizes.array_stack = restore_array_stack;
508 // Update the top of the stack, so we can use more than one
509 // packed function's arguments with the one stack.
510 scope.run_sizes.arg_stack = arg_stack_begin + args_size - 1;
511 Array<PrimExpr> packed_args = {op->args[0], scope.stack_value, scope.stack_tcode->data,
512 ConstInt32(arg_stack_begin),
513 ConstInt32(arg_stack_begin + op->args.size() - 1),
514 // Pass traced value.
515 op->args[args_size - 1]};
516 return Call(op->dtype, builtin::tvm_call_trace_packed_lowered(), packed_args);
517 }

519 Stmt MakeNdMemAllocWithScope(const LetStmtNode* let, const CallNode* call) {
520 ICHECK(device_type_.defined()) << "Unknown device type in current IR";
521 ICHECK(device_id_.defined()) << "Unknown device id in current IR";
522 Stmt throw_last_error = Evaluate(Call(DataType::Int(32), builtin::tvm_throw_last_error(), {}));
523
524 Stmt body = SeqStmt(
525 {IfThenElse(Call(DataType::Bool(1), builtin::isnullptr(), {let->var}), throw_last_error),
526 let->body});
527
528 DataType dtype =
529 let->var->type_annotation.as<PointerTypeNode>()->element_type.as<PrimTypeNode>()->dtype;
530
531 std::string fdevapi_prefix = "device_api.";
532 fdevapi_prefix += runtime::DeviceName(device_type_.as<IntImmNode>()->value);
533
534 Array<PrimExpr> args = {
535 StringImm(fdevapi_prefix + ".alloc_nd"),
536 device_type_,
537 device_id_,
538 IntImm(DataType::Int(32), dtype.code()),
539 IntImm(DataType::Int(32), dtype.bits()),
540 };
541
542 for (size_t i = 0; i < call->args.size(); ++i) {
543 args.push_back(call->args[i]);
544 }
545
546 Call call_packed = Call(let->var.dtype(), builtin::tvm_call_packed(), args);
547 Stmt alloca = LetStmt(let->var, call_packed, body);
548
549 PrimExpr storage_scope = call->args[0];
550 Call free_op = Call(DataType::Int(32), builtin::tvm_call_packed(),
551 {StringImm(fdevapi_prefix + ".free_nd"), device_type_, device_id_,
552 storage_scope, let->var});
553
554 Stmt free_stmt = IfThenElse(free_op != make_zero(DataType::Int(32)), throw_last_error);
555 body = SeqStmt({alloca, free_stmt});
556 return body;
557 }
558
559 private:
560 bool IsArrayHandle(const PrimExpr& arg) {
561 // specially set array handle.
562 if (const CallNode* buf = arg.as<CallNode>()) {
563 if (buf->op.same_as(builtin::tvm_struct_get()) &&
564 buf->args[2].as<IntImmNode>()->value == builtin::kArrAddr) {
565 return true;
566 }
567 }
568 return false;
569 }

571 // The prepration sequence to be emitted before the current statement.
572 std::vector<std::vector<Stmt>> prep_seq_stack_;
573 PrimExpr device_type_;
574 PrimExpr device_id_;
575
