37// If expression is touched by var.
38class ExprTouched final : public StmtExprVisitor {
39 public:
40 explicit ExprTouched(const std::unordered_set<const VarNode*>& touched, bool check_write)
41 : touched_var_(touched), check_write_(check_write) {}
42
43 void VisitExpr(const PrimExpr& n) final {
44 // early stopping
45 if (expr_touched_ && !check_write_) return;
46 StmtExprVisitor::VisitExpr(n);
47 }
48 void VisitStmt(const Stmt& n) final {
49 // early stopping
50 if (expr_touched_ && !check_write_) return;
51 StmtExprVisitor::VisitStmt(n);
52 }
53 void VisitExpr_(const LoadNode* op) final {
54 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
55 }
56 void VisitExpr_(const BufferLoadNode* op) final {
57 HandleUseVar(op->buffer->data.get());
58 StmtExprVisitor::VisitExpr_(op);
59 }
60 void VisitExpr_(const VarNode* op) final { HandleUseVar(op); }
61 void VisitExpr_(const CallNode* op) final {
62 if (op->op.same_as(builtin::tvm_access_ptr())) {
63 const auto* rw_mask = op->args[4].as<IntImmNode>();
64 const VarNode* buffer_var = op->args[1].as<VarNode>();
65 ICHECK(buffer_var);
66 ICHECK(rw_mask);
67 // read
68 if (rw_mask->value & 1) {
69 HandleUseVar(buffer_var);
70 }
71 if (rw_mask->value & 2) {
72 HandleWriteVar(buffer_var);
73 }
74 this->VisitExpr(op->args[2]);
75 } else {
76 StmtExprVisitor::VisitExpr_(op);
77 }
78 }
79 void HandleUseVar(const VarNode* var) {
80 auto it = touched_var_.find(var);
81 if (it != touched_var_.end()) {
82 expr_touched_ = true;
83 }
84 // rember the used vars
85 // in case the var get touched later in a loop.
86 if (!expr_touched_) {
87 used_vars_.push_back(var);
88 }
89 }
90 void HandleWriteVar(const VarNode* var) { write_vars_.push_back(var); }
91 // the fields.
92 bool expr_touched_{false};
93 std::vector<const VarNode*> used_vars_;
94 std::vector<const VarNode*> write_vars_;
95 const std::unordered_set<const VarNode*>& touched_var_;
96 bool check_write_;
97};

99// Analyze if the buffers are invariant to value of var
100class VarTouchedAnalysis : public StmtVisitor {
101 public:
102 void VisitStmt_(const LetStmtNode* op) final {
103 ExprTouched tc(touched_var_, false);
104 tc(op->value);
105 Record(op->var.get(), tc);
106 this->VisitStmt(op->body);
107 }
108
109 void VisitStmt_(const StoreNode* op) final {
110 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
111 }
112
113 void VisitStmt_(const BufferStoreNode* op) final {
114 ExprTouched tc(touched_var_, false);
115 tc(op->value);
116 for (const auto& index : op->indices) {
117 tc(index);
118 }
119 Record(op->buffer->data.get(), tc);
120 }
121 void VisitStmt_(const ForNode* op) final {
122 ExprTouched tc(touched_var_, false);
123 tc(op->min);
124 tc(op->extent);
125 Record(op->loop_var.get(), tc);
126 this->VisitStmt(op->body);
127 }
128 // external function call
129 void VisitStmt_(const EvaluateNode* op) final {
130 ExprTouched tc(touched_var_, true);
131 tc(op->value);
132 for (const VarNode* var : tc.write_vars_) {
133 Record(var, tc);
134 }
135 }
136 void VisitStmt_(const AllocateNode* op) final {
137 ExprTouched tc(touched_var_, false);
138 for (size_t i = 0; i < op->extents.size(); ++i) {
139 tc(op->extents[i]);
140 }
141 tc.VisitExpr(op->condition);
142 Record(op->buffer_var.get(), tc);
143 this->VisitStmt(op->body);
144 }
145 void Record(const VarNode* var, const ExprTouched& tc) {
146 if (touched_var_.count(var)) return;
147 if (tc.expr_touched_) {
148 touched_var_.insert(var);
149 } else {
150 for (const VarNode* r : tc.used_vars_) {
151 if (r != var) {
152 affect_[r].push_back(var);
153 }
154 }
155 }
156 }
157
158 std::unordered_set<const VarNode*> TouchedVar(const Stmt& stmt, const VarNode* var) {
159 touched_var_.insert(var);
160 this->VisitStmt(stmt);
161 // do a DFS to push affect around dependency.
162 std::vector<const VarNode*> pending(touched_var_.begin(), touched_var_.end());
163 while (!pending.empty()) {
164 const VarNode* v = pending.back();
165 pending.pop_back();
166 for (const VarNode* r : affect_[v]) {
167 if (!touched_var_.count(r)) {
168 touched_var_.insert(r);
169 pending.push_back(r);
170 }
171 }
172 }
173 return std::move(touched_var_);
174 }
175
176 private:
177 // Whether variable is touched by the thread variable.
178 std::unordered_set<const VarNode*> touched_var_;
179 // x -> all the buffers x read from
180 std::unordered_map<const VarNode*, std::vector<const VarNode*>> affect_;
181};

183// Inject virtual thread loop
184// rewrite the buffer access pattern when necessary.
185class VTInjector : public arith::IRMutatorWithAnalyzer {
186 public:
187 using IRMutatorWithAnalyzer::VisitExpr_;
188 using IRMutatorWithAnalyzer::VisitStmt_;
189
190 // constructor
191 VTInjector(arith::Analyzer* analyzer, Var var, int num_threads,
192 const std::unordered_set<const VarNode*>& touched_var, bool allow_share)
193 : IRMutatorWithAnalyzer(analyzer),
194 var_(var),
195 num_threads_(num_threads),
196 touched_var_(touched_var),
197 allow_share_(allow_share) {}
198 // Inject VTLoop when needed.
199 Stmt VisitStmt(const Stmt& s) final {
200 ICHECK(!visit_touched_var_);
201 auto stmt = StmtExprMutator::VisitStmt(s);
202 if (visit_touched_var_ || trigger_base_inject_) {
203 if (!vt_loop_injected_) {
204 return InjectVTLoop(stmt, false);
205 }
206 visit_touched_var_ = false;
207 trigger_base_inject_ = false;
208 }
209 return stmt;
210 }
211 // Variable
212 PrimExpr VisitExpr_(const VarNode* op) final {
213 ICHECK(!alloc_remap_.count(op)) << "Buffer address may get rewritten in virtual thread";
214 if (touched_var_.count(op)) {
215 visit_touched_var_ = true;
216 }
217 return GetRef<PrimExpr>(op);
218 }
219 PrimExpr RewriteIndex(PrimExpr index, PrimExpr alloc_extent) const {
220 return analyzer_->Simplify(index + var_ * alloc_extent);
221 }
222 // Expression.
223 PrimExpr VisitExpr_(const CallNode* op) final {
224 if (op->op.same_as(builtin::tvm_access_ptr())) {
225 ICHECK_EQ(op->args.size(), 5U);
226 DataType dtype = op->args[0].dtype();
227 const VarNode* buffer = op->args[1].as<VarNode>();
228 auto it = alloc_remap_.find(buffer);
229 if (it == alloc_remap_.end()) return StmtExprMutator::VisitExpr_(op);
230 visit_touched_var_ = true;
231 PrimExpr offset = this->VisitExpr(op->args[2]);
232 PrimExpr extent = this->VisitExpr(op->args[3]);
233 PrimExpr stride = it->second / make_const(offset.dtype(), dtype.lanes());
234 offset = RewriteIndex(offset, stride);
235
236 return Call(op->dtype, op->op, {op->args[0], op->args[1], offset, extent, op->args[4]});
237 } else if (op->op.same_as(builtin::tvm_context_id())) {
238 return allow_share_ ? GetRef<PrimExpr>(op) : var_;
239 } else {
240 return StmtExprMutator::VisitExpr_(op);
241 }
242 }
243 Stmt VisitStmt_(const EvaluateNode* op) final {
244 trigger_base_inject_ = !allow_share_;
245 return StmtExprMutator::VisitStmt_(op);
246 }
247 // Load
248 PrimExpr VisitExpr_(const LoadNode* op) final {
249 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
250 return PrimExpr();
251 }
252 // Store
253 Stmt VisitStmt_(const StoreNode* op) final {
254 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
255 return Stmt();
256 }
257 // BufferLoad
258 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
259 auto node = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
260 return VisitBufferAccess(std::move(node));
261 }
262 // BufferStore
263 Stmt VisitStmt_(const BufferStoreNode* op) final {
264 auto node = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
265 trigger_base_inject_ = !allow_share_;
266 return VisitBufferAccess(std::move(node));
267 }
268
269 template <typename Node>
270 Node VisitBufferAccess(Node node) {
271 if (touched_var_.count(node->buffer->data.get())) {
272 visit_touched_var_ = true;
273 }
274
275 auto it = alloc_remap_.find(node->buffer->data.get());
276 if (it != alloc_remap_.end()) {
277 ICHECK_EQ(node->indices.size(), 1)
278 << "InjectVirtualThread expects rewritten allocations to be flat memory.";
279 auto writer = node.CopyOnWrite();
280 writer->buffer = GetRemappedBuffer(node->buffer, it->second);
281 writer->indices = {RewriteIndex(node->indices[0], it->second)};
282 }
283
284 return node;
285 }
286
287 Buffer GetRemappedBuffer(Buffer buf, PrimExpr alloc_extent) {
288 auto key = buf.get();
289 auto it = buf_remap_.find(key);
290 if (it != buf_remap_.end()) {
291 return it->second;
292 }
293
294 ICHECK_EQ(buf->shape.size(), 1) << "Expected buffers being rewritten to already be flattened.";
295 auto writer = buf.CopyOnWrite();
296 writer->shape = {buf->shape[0] * alloc_extent};
297
298 buf_remap_[key] = buf;
299 return buf;
300 }
301
302 // Attribute
303 Stmt VisitStmt_(const AttrStmtNode* op) final {
304 PrimExpr value = this->VisitExpr(op->value);
305 if (visit_touched_var_ && !vt_loop_injected_) {
306 return InjectVTLoop(GetRef<Stmt>(op), true);
307 } else if (!allow_share_ && !vt_loop_injected_ &&
308 (op->attr_key == attr::coproc_uop_scope || op->attr_key == attr::coproc_scope)) {
309 return InjectVTLoop(GetRef<Stmt>(op), true);
310 } else {
311 Stmt body = this->VisitStmt(op->body);
312 if (value.same_as(op->value) && body.same_as(op->body)) {
313 return GetRef<Stmt>(op);
314 } else {
315 return AttrStmt(op->node, op->attr_key, value, body);
316 }
317 }
318 }
319 // LetStmt
320 Stmt VisitStmt_(const LetStmtNode* op) final {
321 PrimExpr value = this->VisitExpr(op->value);
322 if (visit_touched_var_ && !vt_loop_injected_) {
323 return InjectVTLoop(GetRef<Stmt>(op), true);
324 }
325 visit_touched_var_ = false;
326 Stmt body = this->VisitStmt(op->body);
327 if (value.same_as(op->value) && body.same_as(op->body)) {
328 return GetRef<Stmt>(op);
329 } else {
330 return LetStmt(op->var, value, body);
331 }
332 }
333 // For
334 Stmt VisitStmt_(const ForNode* op) final {
335 ICHECK(is_zero(op->min));
336 PrimExpr extent = this->VisitExpr(op->extent);
337 if (visit_touched_var_ && !vt_loop_injected_) {
338 Stmt stmt = InjectVTLoop(GetRef<Stmt>(op), true);
339 ++max_loop_depth_;
340 return stmt;
341 }
342 visit_touched_var_ = false;
343 Stmt body = this->VisitStmt(op->body);
344 ++max_loop_depth_;
345 if (extent.same_as(op->extent) && body.same_as(op->body)) {
346 return GetRef<Stmt>(op);
347 } else {
348 auto n = CopyOnWrite(op);
349 n->extent = std::move(extent);
350 n->body = std::move(body);
351 return Stmt(n);
352 }
353 }
354 // IfThenElse
355 Stmt VisitStmt_(const IfThenElseNode* op) final {
356 PrimExpr condition = this->VisitExpr(op->condition);
357 if (visit_touched_var_ && !vt_loop_injected_) {
358 return InjectVTLoop(GetRef<Stmt>(op), true);
359 }
360 visit_touched_var_ = false;
361 ICHECK_EQ(max_loop_depth_, 0);
362 Stmt then_case = this->VisitStmt(op->then_case);
363 Stmt else_case;
364 if (op->else_case.defined()) {
365 int temp = max_loop_depth_;
366 max_loop_depth_ = 0;
367 else_case = this->VisitStmt(op->else_case);
368 max_loop_depth_ = std::max(temp, max_loop_depth_);
369 }
370 if (condition.same_as(op->condition) && then_case.same_as(op->then_case) &&
371 else_case.same_as(op->else_case)) {
372 return GetRef<Stmt>(op);
373 } else {
374 return IfThenElse(condition, then_case, else_case);
375 }
376 }
377
378 // While
379 Stmt VisitStmt_(const WhileNode* op) final {
380 // TODO(masahi): What should we do for While nodes?
381 LOG(FATAL) << "WhileNode in InjectVirtualThread not supported yet";
382 return Stmt();
383 }
384
385 // Seq
386 Stmt VisitStmt_(const SeqStmtNode* op) final {
387 ICHECK_EQ(max_loop_depth_, 0);
388 auto fmutate = [this](const Stmt& s) {
389 int temp = max_loop_depth_;
390 max_loop_depth_ = 0;
391 Stmt ret = this->VisitStmt(s);
392 max_loop_depth_ = std::max(max_loop_depth_, temp);
393 return ret;
394 };
395 return StmtMutator::VisitSeqStmt_(op, false, fmutate);
396 }
397 // Allocate
398 Stmt VisitStmt_(const AllocateNode* op) final {
399 Allocate node = GetRef<Allocate>(op);
400
401 PrimExpr condition = this->VisitExpr(op->condition);
402
403 Array<PrimExpr> extents =
404 op->extents.Map([this](const PrimExpr& extent) { return this->VisitExpr(extent); });
405
406 if (visit_touched_var_ && !vt_loop_injected_) {
407 return InjectVTLoop(GetRef<Stmt>(op), true);
408 }
409
410 visit_touched_var_ = false;
411
412 // Rewrite the buffer if its shape or any value stored in it
413 // depends on the virtual thread var. If `allow_share_` is false,
414 // then the buffer is always rewritten, even if separate virtual
415 // threads only read from the buffer.
416 if (touched_var_.count(op->buffer_var.get()) || !allow_share_) {
417 // place v on highest dimension.
418
419 // TODO(Lunderberg): Move pass to apply before
420 // StorageFlatten/FlattenBuffer. Would rewrite the Buffer to
421 // add the injected virtual thread as the first index.
422 ICHECK_EQ(extents.size(), 1)
423 << "InjectVirtualThread expects rewritten allocations to be flat memory.";
424 PrimExpr stride = extents[0];
425 extents = {stride * num_threads_};
426
427 // Mark the buffer var as touched. BufferLoad/BufferStore should
428 // access locations at `current_index + stride*vthread_var`.
429 alloc_remap_[op->buffer_var.get()] = stride;
430 }
431
432 // Mutate the body. Depends on alloc_remap_.
433 auto body = this->VisitStmt(op->body);
434
435 if (extents.same_as(op->extents) && body.same_as(op->body) &&
436 condition.same_as(op->condition)) {
437 return GetRef<Stmt>(op);
438 } else {
439 return Allocate(op->buffer_var, op->dtype, extents, condition, body);
440 }
441 }
442
443 // inject vthread loop
444 Stmt InjectVTLoop(Stmt stmt, bool before_mutation) {
445 ICHECK(!vt_loop_injected_);
446 // reset the flags
447 visit_touched_var_ = false;
448 trigger_base_inject_ = false;
449 vt_loop_injected_ = true;
450 if (before_mutation) {
451 stmt = this->VisitStmt(stmt);
452 }
453 // reset the flags after processing.
454 vt_loop_injected_ = false;
455 visit_touched_var_ = false;
456 // only unroll if number of vthreads are small
457 if (max_loop_depth_ == 0 && num_threads_ < 16) {
458 // do unrolling if it is inside innermost content.
459 Array<Stmt> seq;
460 for (int i = 0; i < num_threads_; ++i) {
461 seq.push_back(Substitute(stmt, {{var_, make_const(var_.dtype(), i)}}));
462 }
463 return SeqStmt::Flatten(seq);
464 } else {
465 // insert a for loop
466 Var idx(var_->name_hint + ".s", var_->dtype);
467 Map<Var, PrimExpr> values{{var_, idx}};
468 stmt = Substitute(stmt, values);
469 return For(idx, make_zero(idx.dtype()), make_const(idx.dtype(), num_threads_),
470 ForKind::kSerial, stmt);
471 }
472 }
473
474 private:
475 // vthread variable
476 Var var_;
477 // the threads/lanes
478 int num_threads_;
479 // whethe the loop is already injected.
480 bool vt_loop_injected_{false};
481 // whether current expression get touched.
482 bool visit_touched_var_{false};
483 // Trigger base stmt
484 bool trigger_base_inject_{false};
485 // the counter of loops in after mutation.
486 int max_loop_depth_{0};
487 // The variables that get touched.
488 const std::unordered_set<const VarNode*>& touched_var_;
489 // Whether allow shareding.
490 bool allow_share_;
491 /* \brief The allocations that get touched -> extent
492 *
493 * Maps from the buffer_var of an allocate node to the original
494 * extent of the allocation. Used when rewriting the indices of
495 * BufferLoad/BufferStore.
496 */
497 std::unordered_map<const VarNode*, PrimExpr> alloc_remap_;
498 /*! \brief Map of buffers that are modified.
499 *
500 * Buffers allocated or written to within the virtual thread loop
501 * must have one copy per virtual thread. This is done by enlarging
502 * the allocated buffer size, then modifying the indices at which
503 * each virtual thread accesses the buffer.
504 */
505 std::unordered_map<const BufferNode*, Buffer> buf_remap_;
506};
507
508class VirtualThreadInjector : public arith::IRMutatorWithAnalyzer {
509 public:
510 using IRMutatorWithAnalyzer::IRMutatorWithAnalyzer;
511 using IRMutatorWithAnalyzer::VisitStmt_;
512
513 Stmt VisitStmt_(const AttrStmtNode* op) final {
514 Stmt stmt = StmtMutator::VisitStmt_(op);
515 op = stmt.as<AttrStmtNode>();
516 if (op->attr_key == attr::virtual_thread) {
517 IterVar iv = Downcast<IterVar>(op->node);
518 bool allow_share = std::string(iv->thread_tag).substr(0, 7) == "vthread";
519 int nthread = static_cast<int>(op->value.as<IntImmNode>()->value);
520 VarTouchedAnalysis vs;
521 auto touched = vs.TouchedVar(op->body, iv->var.get());
522 VTInjector injector(analyzer_, iv->var, nthread, touched, allow_share);
523 return injector(op->body);
524 } else {
525 return stmt;
526 }
527 }
528
529 Stmt VisitStmt_(const ProducerStoreNode* op) final {
530 LOG(FATAL) << "Need to call StorageFlatten first";
531 return GetRef<Stmt>(op);
532 }
533};

535namespace transform {
536
537Pass InjectVirtualThread() {
538 auto pass_func = [](PrimFunc f, IRModule m, PassContext ctx) {
539 auto* n = f.CopyOnWrite();
540
541 arith::Analyzer analyzer;
542
543 n->body = VirtualThreadInjector(&analyzer)(std::move(n->body));
544 n->body = ConvertSSA(std::move(n->body));
545 return f;
546 };
547 return CreatePrimFuncPass(pass_func, 0, "tir.InjectVirtualThread", {});
548}
