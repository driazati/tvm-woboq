43static constexpr const char* kDeviceContextVar = "device_api_context";
44
45class ReturnRewriter : public StmtMutator {
46 public:
47 explicit ReturnRewriter(Var ret_var, Var ret_tcode) : ret_var_(ret_var), ret_tcode_(ret_tcode) {}
48
49 Stmt VisitStmt_(const ForNode* node) override {
50 if (node->kind == ForKind::kParallel) in_parallel_ += 1;
51 Stmt ret = StmtMutator::VisitStmt_(node);
52 if (node->kind == ForKind::kParallel) in_parallel_ -= 1;
53 return ret;
54 }
55
56 Stmt VisitStmt_(const EvaluateNode* node) override {
57 Stmt ret = StmtMutator::VisitStmt_(node);
58 const EvaluateNode* eval = ret.as<EvaluateNode>();
59 ICHECK(eval);
60 if (const CallNode* call = eval->value.as<CallNode>()) {
61 if (call->op.same_as(builtin::ret())) {
62 ICHECK_EQ(in_parallel_, 0) << "tir.ret cannot be used in parallel scope.";
63 ICHECK_EQ(call->args.size(), 1) << "tir.ret expect a single argument.";
64 ret = WriteToOut(call->args[0]);
65 }
66 }
67 return ret;
68 }
69
70 private:
71 struct ConvertedInfo {
72 int tcode{-1};
73 PrimExpr expr;
74 Buffer dummy_val_buffer;
75 Buffer dummy_tcode_buffer;
76 };
77
78 ConvertedInfo ConvertForFFI(PrimExpr val) {
79 ConvertedInfo info;
80
81 // convert val's data type to FFI data type, return type code
82 DataType dtype = val.dtype();
83 if (dtype.is_int() || dtype.is_uint()) {
84 info.tcode = kTVMArgInt;
85 info.expr = Cast(DataType::Int(64), val);
86 } else if (dtype.is_float()) {
87 info.tcode = kTVMArgFloat;
88 info.expr = Cast(DataType::Float(64), val);
89 } else if (dtype.is_void()) {
90 info.tcode = kTVMNullptr;
91 info.expr = val;
92 } else {
93 LOG(FATAL) << "data type " << dtype << " not supported yet";
94 }
95
96 // If multiple return locations have the same data type, use the
97 // same dummy buffer declaration.
98 auto it = dummy_val_buffer_map_.find(info.tcode);
99 if (it != dummy_val_buffer_map_.end()) {
100 info.dummy_val_buffer = it->second;
101 } else {
102 info.dummy_val_buffer = Buffer(ret_var_, info.expr.dtype(), {1}, {1}, ConstInt32(0),
103 ret_var_->name_hint, 0, 0, kDefault);
104 dummy_val_buffer_map_[info.tcode] = info.dummy_val_buffer;
105 }
106
107 // The tcode is always a 32-bit int, so we don't need to have a separate map.
108 if (!dummy_tcode_buffer_.defined()) {
109 dummy_tcode_buffer_ = Buffer(ret_tcode_, DataType::Int(32), {1}, {1}, ConstInt32(0),
110 ret_tcode_->name_hint, 0, 0, kDefault);
111 }
112 info.dummy_tcode_buffer = dummy_tcode_buffer_;
113
114 return info;
115 }
116
117 Stmt WriteToOut(PrimExpr val) {
118 auto info = ConvertForFFI(val);
119 Stmt store_val = BufferStore(info.dummy_val_buffer, info.expr, {0});
120 Stmt store_tcode = BufferStore(info.dummy_tcode_buffer, info.tcode, {0});
121 Stmt ret_zero = Evaluate(tvm::ret(0));
122 return SeqStmt({store_val, store_tcode, ret_zero});
123 }
124
125 Var ret_var_;
126 Var ret_tcode_;
127 int in_parallel_{0};
128
129 std::unordered_map<int, Buffer> dummy_val_buffer_map_;
130 Buffer dummy_tcode_buffer_;
131};
132
133Stmt RewriteReturn(Stmt body, Var ret_var, Var ret_tcode) {
134 ReturnRewriter rewriter(ret_var, ret_tcode);
135 return rewriter(body);
136}
137
138inline Stmt MakeAssertEQ(PrimExpr lhs, PrimExpr rhs, std::string msg) {
139 return AssertStmt(lhs == rhs, tvm::tir::StringImm(msg), Evaluate(0));
140}

142PrimFunc MakePackedAPI(PrimFunc&& func, int num_unpacked_args) {
143 auto global_symbol = func->GetAttr<String>(tvm::attr::kGlobalSymbol);
144 ICHECK(global_symbol) << "MakePackedAPI: Expect PrimFunc to have the global_symbol attribute";
145
146 auto target = func->GetAttr<Target>(tvm::attr::kTarget);
147 ICHECK(target.defined()) << "MakePackedAPI: Require the target attribute";
148 int target_device_type = target.value()->kind->device_type;
149
150 std::string name_hint = global_symbol.value();
151
152 auto* func_ptr = func.CopyOnWrite();
153 const Stmt nop = Evaluate(0);
154 int num_args = static_cast<int>(func_ptr->params.size());
155 ICHECK_LE(num_unpacked_args, num_args);
156 bool pack_args = (num_unpacked_args == -1) || (num_args > num_unpacked_args);
157 if (num_unpacked_args == -1) {
158 // reset to zero
159 num_unpacked_args = 0;
160 }
161 ICHECK_GE(num_unpacked_args, 0);
162 int num_packed_args = num_args - num_unpacked_args;
163 // Data field definitions
164 // The packed fields
165 Var v_packed_args("args", DataType::Handle());
166 Buffer buf_packed_arg_type_ids = decl_buffer({IntImm(DataType::Int(32), func_ptr->params.size())},
167 DataType::Int(32), "arg_type_ids");
168 Var v_num_packed_args("num_args", DataType::Int(32));
169 Var v_out_ret_value("out_ret_value", PointerType(PrimType(DataType::Void())));
170 Var v_out_ret_tcode("out_ret_tcode", PointerType(PrimType(DataType::Int(32))));
171 Var v_resource_handle("resource_handle", DataType::Handle());
172 // The arguments of the function.
173 Array<Var> args;
174 // The device context
175 Var device_id("dev_id");
176 Integer device_type(target_device_type);
177 // seq_init gives sequence of initialization
178 // seq_check gives sequence of later checks after init
179 std::vector<Stmt> seq_init, seq_check;
180 std::unordered_map<const VarNode*, PrimExpr> vmap;
181 ArgBinder binder(&vmap);
182 // ---------------------------
183 // local function definitions
184 // load i-th argument as type t
185 auto f_arg_value = [&](DataType t, int i) {
186 Array<PrimExpr> call_args{v_packed_args, IntImm(DataType::Int(32), i),
187 IntImm(DataType::Int(32), builtin::kTVMValueContent)};
188 // load 64 bit version
189 DataType api_type = APIType(t);
190 PrimExpr res = Call(api_type, builtin::tvm_struct_get(), call_args);
191 // cast to the target version.
192 if (api_type != t) {
193 res = Cast(t, res);
194 }
195 return res;
196 };
197 // ---------------------------
198 // start of logics
199 // add signature for packed arguments.
200 if (pack_args) {
201 args.push_back(v_packed_args);
202 args.push_back(buf_packed_arg_type_ids->data);
203 args.push_back(v_num_packed_args);
204 }
205
206 // Need to re-declare vars, in case some arguments also appears in the buffer.
207 std::vector<std::pair<Var, Var>> var_def;
208 std::vector<std::pair<Var, Buffer>> buffer_def;
209
210 for (int i = 0; i < static_cast<int>(func_ptr->params.size()); ++i) {
211 Var param = func_ptr->params[i];
212 std::string param_name;
213 if (param->name_hint.defined() && (!param->name_hint.empty())) {
214 param_name = "arg." + param->name_hint;
215 } else {
216 param_name = "arg" + std::to_string(i);
217 }
218 Var v_arg = Var(param_name, param->dtype);
219
220 // Pluck the device API context out based on name
221 if (param->name_hint == kDeviceContextVar) {
222 num_packed_args--;
223 v_resource_handle = param;
224 continue;
225 }
226
227 if (func_ptr->preflattened_buffer_map.count(param)) {
228 buffer_def.emplace_back(v_arg, func_ptr->preflattened_buffer_map[param]);
229 } else if (func_ptr->buffer_map.count(param)) {
230 buffer_def.emplace_back(v_arg, func_ptr->buffer_map[param]);
231 } else {
232 var_def.emplace_back(v_arg, param);
233 }
234
235 if (i < num_packed_args) {
236 // Value loads
237 seq_init.emplace_back(LetStmt(v_arg, f_arg_value(v_arg.dtype(), i), nop));
238 // type code checks
239 Var tcode(v_arg->name_hint + ".code", DataType::Int(32));
240 seq_init.emplace_back(
241 LetStmt(tcode, BufferLoad(buf_packed_arg_type_ids, {IntImm(DataType::Int(32), i)}), nop));
242 DataType t = v_arg.dtype();
243 if (t.is_handle()) {
244 std::ostringstream msg;
245 msg << name_hint << ": Expect arg[" << i << "] to be pointer";
246 seq_check.emplace_back(AssertStmt(tcode == kTVMOpaqueHandle || tcode == kTVMNDArrayHandle ||
247 tcode == kTVMDLTensorHandle || tcode == kTVMNullptr,
248 tvm::tir::StringImm(msg.str()), nop));
249 } else if (t.is_int() || t.is_uint()) {
250 std::ostringstream msg;
251 msg << name_hint << ": Expect arg[" << i << "] to be int";
252 seq_check.emplace_back(AssertStmt(tcode == kDLInt, tvm::tir::StringImm(msg.str()), nop));
253 } else {
254 ICHECK(t.is_float());
255 std::ostringstream msg;
256 msg << name_hint << ": Expect arg[" << i << "] to be float";
257 seq_check.emplace_back(AssertStmt(tcode == kDLFloat, tvm::tir::StringImm(msg.str()), nop));
258 }
259 } else {
260 args.push_back(v_arg);
261 }
262 }
263
264 // allow return value if the function is packed.
265 if (pack_args) {
266 args.push_back(v_out_ret_value);
267 args.push_back(v_out_ret_tcode);
268 args.push_back(v_resource_handle);
269 }
270
271 size_t expected_nargs = num_unpacked_args + (pack_args ? 6 : 0);
272 ICHECK_EQ(args.size(), expected_nargs);
273
274 // Arg definitions are defined before buffer binding to avoid the use before
275 // def errors.
276 //
277 // For example, for auto broadcasting, checks are required to guarantee that
278 // either 0 or the original stride will be correctly used. Checks here have
279 // to use the args that may have no let binding yet. Therefore, hoisting let
280 // binding for args before buffer declaration is needed.
281 for (const auto& kv : var_def) {
282 binder.Bind(kv.second, kv.first, kv.first->name_hint, true);
283 }
284
285 for (const auto& kv : buffer_def) {
286 binder.BindDLTensor(kv.second, device_type, device_id, kv.first, kv.first->name_hint);
287 }
288
289 if (num_unpacked_args == 0) {
290 func = WithAttr(std::move(func), tvm::attr::kCallingConv, Integer(CallingConv::kCPackedFunc));
291 }
292
293 Stmt body = RewriteReturn(func_ptr->body, v_out_ret_value, v_out_ret_tcode);
294 body = AttrStmt(make_zero(DataType::Int(32)), attr::compute_scope,
295 StringImm(name_hint + "_compute_"), body);
296 // Set device context
297 if (vmap.count(device_id.get())) {
298 PrimExpr node = StringImm("default");
299 seq_check.push_back(AttrStmt(node, attr::device_id, device_id, nop));
300 seq_check.push_back(AttrStmt(node, attr::device_type, device_type, nop));
301
302 if (runtime::DeviceAPI::NeedSetDevice(target_device_type)) {
303 Stmt set_device =
304 Evaluate(Call(DataType::Int(32), builtin::tvm_call_packed(),
305 {StringImm(runtime::symbol::tvm_set_device), device_type, device_id}));
306 body = SeqStmt({set_device, body});
307 }
308 }
309
310 if (pack_args) {
311 std::ostringstream num_args_error;
312 num_args_error << name_hint << ": num_args should be " << num_packed_args;
313 std::vector<Stmt> arg_assert = {
314 MakeAssertEQ(v_num_packed_args, num_packed_args, num_args_error.str())};
315 func_ptr->body =
316 MergeNest({arg_assert, seq_init, binder.init_nest(), seq_check, binder.asserts()}, body);
317 } else {
318 func_ptr->body = MergeNest({seq_init, binder.init_nest(), seq_check, binder.asserts()}, body);
319 }
320 func_ptr->params = args;
321
322 Array<Var> undefined = UndefinedVars(func_ptr->body, func_ptr->params);
323 if (undefined.size() != 0) {
324 std::ostringstream os;
325 for (Var v : undefined) {
326 os << " \'" << v->name_hint << "\' ";
327 }
328 os << " is not bound to any variables";
329 LOG(FATAL) << "Not all Vars are passed in api_args: " << os.str();
330 }
331
332 func_ptr->buffer_map = Map<Var, Buffer>();
333 func_ptr->checked_type_ = func_ptr->func_type_annotation();
334 func_ptr->ret_type = PrimType(DataType::Int(32));
335
336 // return the function.
337 return std::move(func);
338}

340namespace transform {

342Pass MakePackedAPI(int num_unpacked_args) {
343 // packed arguments anyway while `num_unpacked_args` is -1
344 auto pass_func = [num_unpacked_args](IRModule m, PassContext ctx) {
345 IRModuleNode* mptr = m.CopyOnWrite();
346 std::vector<std::pair<GlobalVar, PrimFunc>> updates;
347
348 for (const auto& kv : mptr->functions) {
349 if (auto* n = kv.second.as<PrimFuncNode>()) {
350 PrimFunc func = GetRef<PrimFunc>(n);
351 if (func->GetAttr<Integer>(tvm::attr::kCallingConv, Integer(CallingConv::kDefault)) ==
352 CallingConv::kDefault) {
353 auto updated_func = MakePackedAPI(std::move(func), num_unpacked_args);
354 updates.push_back({kv.first, updated_func});
355 }
356 }
357 }
358
359 for (const auto& pair : updates) {
360 mptr->AddUnchecked(pair.first, pair.second);
361 }
362 return m;
363 };
364
365 return tvm::transform::CreateModulePass(pass_func, 0, "tir.MakePackedAPI", {});
366}
