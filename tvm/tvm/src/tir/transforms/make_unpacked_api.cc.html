43PrimFunc MakeUnpackedAPI(PrimFunc&& func) {
44 auto global_symbol = func->GetAttr<String>(tvm::attr::kGlobalSymbol);
45 ICHECK(global_symbol) << "MakeUnpackedAPI: Expect PrimFunc to have the global_symbol attribute";
46
47 auto target = func->GetAttr<Target>(tvm::attr::kTarget);
48 ICHECK(target.defined()) << "MakeUnpackedAPI: Require the target attribute";
49
50 auto* func_ptr = func.CopyOnWrite();
51
52 // Setup device context
53 int target_device_type = target.value()->kind->device_type;
54 Integer device_type(target_device_type);
55 Integer device_id(0);
56 PrimExpr node = StringImm("default");
57 const Stmt nop = Evaluate(0);
58 std::vector<Stmt> device_init;
59
60 // Collect variables and buffers to map between
61 Array<Var> args;
62 Map<Var, Buffer> new_buffer_map;
63 for (const Var& param : func->params) {
64 // Ideally all func params should have Buffers defined in the buffer_map
65 // We should look to insert buffer_maps for all PrimFuncs that are returned
66 // to the core compiler.
67 if (func->buffer_map.find(param) != func->buffer_map.end()) {
68 args.push_back(func->buffer_map[param]->data);
69 // Rewiring the buffer_var to map to Buffers for low-level passes
70 // retain information about the buffer.
71 new_buffer_map.Set(func->buffer_map[param]->data, func->buffer_map[param]);
72 } else {
73 args.push_back(param);
74 }
75 }
76
77 if (func->buffer_map.size()) {
78 device_init.push_back(AttrStmt(node, attr::device_id, device_id, nop));
79 device_init.push_back(AttrStmt(node, attr::device_type, device_type, nop));
80 }
81
82 func_ptr->body = MergeNest(device_init, func_ptr->body);
83 func_ptr->params = args;
84 func_ptr->ret_type = PrimType(DataType::Int(32));
85 func_ptr->buffer_map = new_buffer_map;
86
87 // return the function.
88 return std::move(func);
89}
90
91namespace transform {
92
93Pass MakeUnpackedAPI() {
94 auto pass_func = [](IRModule m, PassContext ctx) {
95 IRModuleNode* mptr = m.CopyOnWrite();
96 std::vector<std::pair<GlobalVar, PrimFunc>> updates;
97
98 for (const auto& kv : mptr->functions) {
99 if (auto* n = kv.second.as<PrimFuncNode>()) {
100 PrimFunc func = GetRef<PrimFunc>(n);
101 if (func->GetAttr<Integer>(tvm::attr::kCallingConv, Integer(CallingConv::kDefault)) ==
102 CallingConv::kDefault) {
103 auto updated_func = MakeUnpackedAPI(std::move(func));
104 updates.push_back({kv.first, updated_func});
105 }
106 }
107 }
108
109 for (const auto& pair : updates) {
110 mptr->AddUnchecked(pair.first, pair.second);
111 }
112 return m;
113 };
114
115 return tvm::transform::CreateModulePass(pass_func, 0, "tir.MakeUnpackedAPI", {});
116}
