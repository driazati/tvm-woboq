49// Find a linear pattern of storage access
50// Used for liveness analysis.
51// Composite scopes(loop/thread_launch/IfThen) is represented by two points:
52// before_scope -> scope_body -> after_scope
53//
54// The linear_seq_ stores before_scope and after_scope.
55// The access to the arrays are stored at the after_scope point.
56//
57// Define "scope" as the body of For/thread_launch/IfThenElse
58// This pass tries to detect last point that we need to keep memory
59// alive under the same scope as allocate.
60// The storage need to be kept alive between allocate and last access.
61// The free point is only inserted at the same scope of allocate.
62//
63class LinearAccessPatternFinder final : public StmtExprVisitor {
64 public:
65 /*! \brief record the touch hist of statment. */
66 struct StmtEntry {
67 // The statment
68 const Object* stmt;
69 // The index in the linear_seq_ to point to end of the nested scope.
70 // This is only set to non-zero if stmt is a nested scope.
71 // if offset > 0, means this is the begin, the end entry is current_index + offset
72 // if offset < 0, means this is the end, the begin entry is current_index + offset
73 int64_t scope_pair_offset{0};
74 // The buffer variables this statment touched.
75 std::vector<const VarNode*> touched;
76 };
77 // The scope of each allocation
78 struct AllocEntry {
79 // The physical dimension of the allocation.
80 size_t num_physical_dimensions{0};
81 // scope level
82 size_t level{0};
83 // allocation stmt
84 const AllocateNode* alloc{nullptr};
85 };
86
87 void VisitStmt_(const AllocateNode* op) final {
88 size_t level = scope_.size();
89 const VarNode* buf = op->buffer_var.get();
90
91 AllocEntry entry;
92 entry.alloc = op;
93 entry.level = level;
94 // Since StorageRewrite occurs after StorageFlatten/FlattenBuffer,
95 // all allocations specify the extent of physical dimensions, and
96 // is 1 for flat memory spaces.
97 entry.num_physical_dimensions = op->extents.size();
98 alloc_info_[buf] = entry;
99
100 StmtExprVisitor::VisitStmt_(op);
101 }

103 void VisitStmt_(const StoreNode* op) final {
104 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
105 }

107 void VisitStmt_(const BufferStoreNode* op) final {
108 scope_.push_back(StmtEntry());
109 // visit subexpr
110 StmtExprVisitor::VisitStmt_(op);
111 // Add write access.
112 const VarNode* buf = op->buffer->data.get();
113 auto it = alloc_info_.find(buf);
114 if (it != alloc_info_.end() && it->second.alloc) {
115 ICHECK_LT(it->second.level, scope_.size());
116 scope_[it->second.level].touched.push_back(buf);
117
118 ICHECK_EQ(op->buffer->axis_separators.size() + 1, it->second.num_physical_dimensions)
119 << "Buffer " << op->buffer->name << " is allocated with "
120 << it->second.num_physical_dimensions
121 << " physical dimensions, but is accessed as having "
122 << op->buffer->axis_separators.size() + 1 << " physical dimensions" << std::endl;
123 }
124 StmtEntry e = scope_.back();
125 scope_.pop_back();
126 if (e.touched.size() != 0) {
127 e.stmt = op;
128 linear_seq_.push_back(e);
129 }
130 }

132 void VisitExpr_(const LoadNode* op) final {

136 void VisitExpr_(const BufferLoadNode* op) final {
137 // Add write access.
138 StmtExprVisitor::VisitExpr_(op);
139 const VarNode* buf = op->buffer->data.get();
140 auto it = alloc_info_.find(buf);
141 if (it != alloc_info_.end() && it->second.alloc) {
142 ICHECK_LT(it->second.level, scope_.size()) << "Load memory in places other than store.";
143 scope_[it->second.level].touched.push_back(buf);
144
145 ICHECK_EQ(op->buffer->axis_separators.size() + 1, it->second.num_physical_dimensions)
146 << "Buffer " << op->buffer->name << " is allocated with "
147 << it->second.num_physical_dimensions
148 << " physical dimensions, but is accessed as having "
149 << op->buffer->axis_separators.size() + 1 << " physical dimensions" << std::endl;
150 }
151 }

165 void VisitExpr_(const CallNode* op) final {
166 if (op->op.same_as(builtin::address_of())) {
167 const BufferLoadNode* load = op->args[0].as<BufferLoadNode>();
168 for (const auto& index : load->indices) {

176 void VisitExpr_(const VarNode* buf) final {
177 // Directly reference to the variable count as a read.
178 auto it = alloc_info_.find(buf);
179 if (it != alloc_info_.end() && it->second.alloc) {
180 ICHECK_LT(it->second.level, scope_.size()) << " buf=" << buf->name_hint;
181 scope_[it->second.level].touched.push_back(buf);
182 }

185 template <typename T>
186 void VisitNewScope(const T* op) {
187 scope_.push_back(StmtEntry());
188 StmtEntry e;
189 e.stmt = op;
190 int64_t begin_index = static_cast<int64_t>(linear_seq_.size());
191 // before scope.
192 linear_seq_.push_back(e);
193 StmtExprVisitor::VisitStmt_(op);
194 // after scope.
195 e.touched = std::move(scope_.back().touched);
196 scope_.pop_back();
197 int64_t end_index = static_cast<int64_t>(linear_seq_.size());
198 ICHECK_GT(end_index, begin_index);
199 e.scope_pair_offset = begin_index - end_index;
200 linear_seq_.push_back(e);
201 // record the pointer to end index.
202 ICHECK_NE(end_index, 0U);
203 linear_seq_[begin_index].scope_pair_offset = end_index - begin_index;
204 }

206 void VisitStmt_(const AttrStmtNode* op) final {
207 // Only record the outer most thread extent.
208 if (op->attr_key == attr::thread_extent && !in_thread_env_) {
209 in_thread_env_ = true;
210 VisitNewScope(op);
211 in_thread_env_ = false;
212 } else if (op->attr_key == attr::extern_scope) {
213 VisitNewScope(op);
214 } else if (op->attr_key == attr::virtual_thread) {
215 VisitNewScope(op);
216 } else {
217 StmtExprVisitor::VisitStmt_(op);
218 }
219 }
220
221 void VisitStmt_(const IfThenElseNode* op) final { VisitNewScope(op); }
222
223 void VisitStmt_(const ForNode* op) final { VisitNewScope(op); }
224
225 void VisitStmt_(const WhileNode* op) final { VisitNewScope(op); }
226
227 void VisitStmt_(const AssertStmtNode* op) final { VisitNewScope(op); }
228
229 void VisitStmt_(const LetStmtNode* op) final { VisitNewScope(op); }

231 // linearized access sequence.
232 std::vector<StmtEntry> linear_seq_;
233 // The storage scope of each buffer
234 std::unordered_map<const VarNode*, AllocEntry> alloc_info_;
235
236 private:
237 // Whether already in thread env.
238 bool in_thread_env_{false};
239 // The scope stack.
240 std::vector<StmtEntry> scope_;

243// Verify if the statement can be run safely via inplace fashion
244//
245// Detect pattern: dst[index] = f(src[index])
246//
247// WARNING: the current detection algorithm cannot handle the case
248// when a location in an array is written multiple times
249//
250// For example, the following program will pass the check,
251// but we cannot make A and B to be the same array.
252//
253// A[0] = B[0] + 1
254// A[0] = B[0] + 1
255//
256// The high level code generator needs to ensure that the generated
257// code only write each location of the target array once.
258//
259// This is the case with IR generated by the current compute schedule.
260// We explicitly return false if we find there is an extern block
261// which can be arbitrary IR.
262//
263// Neve-the-less, inplace detector should be used with care in mind.
264// We may also consider introduce a condition checker that checks
265// if every index only visited once for an absolute sufficient condition.
266//
267// The code after inplace transformation is no longer idempotent.
268//
269class InplaceOpVerifier : public StmtExprVisitor {
270 public:
271 bool Check(const Object* stmt, const VarNode* dst, const VarNode* src) {
272 dst_ = dst;
273 src_ = src;
274 result_ = true;
275 if (stmt->IsInstance<AttrStmtNode>()) {
276 VisitStmt_(static_cast<const AttrStmtNode*>(stmt));
277 } else if (stmt->IsInstance<ForNode>()) {
278 VisitStmt_(static_cast<const ForNode*>(stmt));
279 } else if (stmt->IsInstance<IfThenElseNode>()) {
280 VisitStmt_(static_cast<const IfThenElseNode*>(stmt));
281 } else if (stmt->IsInstance<WhileNode>()) {
282 VisitStmt_(static_cast<const WhileNode*>(stmt));
283 } else if (stmt->IsInstance<StoreNode>()) {
284 VisitStmt_(static_cast<const StoreNode*>(stmt));
285 } else if (stmt->IsInstance<BufferStoreNode>()) {
286 VisitStmt_(static_cast<const BufferStoreNode*>(stmt));
287 } else {
288 return false;
289 }
290 return result_;
291 }

293 using StmtExprVisitor::VisitStmt_;
294
295 void VisitStmt(const Stmt& n) final {
296 if (!result_) return;
297 StmtExprVisitor::VisitStmt(n);
298 }
299 void VisitExpr(const PrimExpr& n) final {
300 if (!result_) return;
301 StmtExprVisitor::VisitExpr(n);
302 }
303
304 void VisitExpr_(const VarNode* op) final {
305 // assume all opaque access is unsafe
306 if (op == dst_ || op == src_) {
307 result_ = false;
308 return;
309 }
310 }

314 }

316 void VisitStmt_(const BufferStoreNode* op) final {
317 ++mem_nest_;
318 for (const auto& index : op->indices) {
319 this->VisitExpr(index);
320 }
321 --mem_nest_;
322 if (op->buffer->data.get() == dst_) {
323 store_ = op;
324 this->VisitExpr(op->value);
325 store_ = nullptr;
326 } else {
327 this->VisitExpr(op->value);
328 }
329 }
330
331 void VisitStmt_(const AttrStmtNode* op) final {
332 // always reject extern code
333 if (op->attr_key == attr::extern_scope || op->attr_key == attr::volatile_scope) {
334 result_ = false;
335 return;
336 }
337 StmtExprVisitor::VisitStmt_(op);
338 }

340 void VisitExpr_(const LoadNode* op) final {
341 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
342 }
343
344 void VisitExpr_(const BufferLoadNode* op) final {
345 const VarNode* buf = op->buffer->data.get();
346 // cannot read from dst_ (no reduction)
347 if (buf == dst_) {
348 result_ = false;
349 return;
350 }
351 // do not allow indirect memory load
352 if (mem_nest_ != 0) {
353 result_ = false;
354 return;
355 }
356 if (src_ == buf) {
357 if (store_ == nullptr || store_->value.dtype() != op->dtype) {
358 result_ = false;
359 return;
360 }
361 ICHECK_EQ(store_->indices.size(), op->indices.size())
362 << "Store/Load occur to the same buffer " << buf->name_hint
363 << " with differing number of indices";
364 for (size_t i = 0; i < store_->indices.size(); i++) {
365 if (!tir::ExprDeepEqual()(store_->indices[i], op->indices[i])) {
366 result_ = false;
367 return;
368 }
369 }
370 }
371 ++mem_nest_;
372 StmtExprVisitor::VisitExpr_(op);
373 --mem_nest_;
374 }
375
376 private:
377 // result of the check
378 bool result_{true};
379 // destination memory
380 const VarNode* dst_;
381 // source variable
382 const VarNode* src_;
383 // counter of load,
384 // it is not safe to inplace when there is nested load like A[B[i]]
385 int mem_nest_{0};
386 // The current store to be inspected
387 const BufferStoreNode* store_{nullptr};
388};

390/* \brief Rewrite and merge memory allocation.
391 *
392 * Using LinearAccessPatternFinder, determines which buffers could share an
393 * allocation. This includes both sequential usage of the same buffer and
394 * merging small allocations at the same scope into a single larger allocation.
395 * The merging of small allocations requires the codegen to cast the resulting
396 * value from the storage type to the output type after access.
397 */
398class StoragePlanRewriter : public StmtExprMutator {
399 public:
400 using StmtEntry = LinearAccessPatternFinder::StmtEntry;
401 using AllocEntry = LinearAccessPatternFinder::AllocEntry;
402
403 Stmt Rewrite(Stmt stmt, bool detect_inplace) {
404 detect_inplace_ = detect_inplace;
405 // plan the rewrite
406 LinearAccessPatternFinder finder;
407 finder(stmt);
408 this->LivenessAnalysis(finder.linear_seq_);
409 this->PlanMemory(finder.linear_seq_, finder.alloc_info_);
410 this->PrepareNewAlloc();
411 // start rewrite
412 stmt = operator()(std::move(stmt));
413 if (attach_map_.count(nullptr)) {
414 return MakeAttach(attach_map_.at(nullptr), stmt);
415 }
416 return stmt;
417 }

419 Stmt VisitStmt_(const StoreNode* op) final {
420 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
421 return Stmt();
422 }

424 PrimExpr VisitExpr_(const LoadNode* op) final {
425 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
426 return PrimExpr();
427 }

429 template <typename Node>
430 Node VisitBufferAccess(Node node) {
431 auto it = alloc_map_.find(node->buffer->data.get());
432 if (it != alloc_map_.end()) {
433 Buffer buf = RemapBuffer(node->buffer, it->second->alloc_var);
434
435 Array<PrimExpr> indices = node->indices;
436 indices.Set(indices.size() - 1,
437 RemapIndex(node->buffer->dtype, indices[indices.size() - 1], it->second));
438
439 auto writer = node.CopyOnWrite();
440 writer->buffer = buf;
441 writer->indices = indices;
442 }
443 return node;
444 }
445
446 Buffer RemapBuffer(Buffer buf, Var new_backing_array) {
447 auto key = buf.get();
448 auto it = buffer_remap_.find(key);
449 if (it != buffer_remap_.end()) {
450 ICHECK_EQ(it->second->data.get(), new_backing_array.get())
451 << "Cannot remap buffer " << buf->name << " to use backing array "
452 << new_backing_array->name_hint << ", previously used backing array "
453 << it->second->data->name_hint;
454 return it->second;
455 }
456
457 Buffer remapped = Buffer(new_backing_array, buf->dtype, buf->shape, buf->strides,
458 buf->elem_offset, new_backing_array->name_hint, buf->data_alignment,
459 buf->offset_factor, buf->buffer_type, buf->axis_separators, buf->span);
460 buffer_remap_[key] = remapped;
461 return remapped;
462 }

477 if (it->second->bits_offset != 0) {

485 PrimExpr VisitExpr_(const CallNode* op) final {
486 if (op->op.same_as(builtin::tvm_access_ptr())) {
487 ICHECK_EQ(op->args.size(), 5U);
488 DataType dtype = op->args[0].dtype();
489 const VarNode* buffer = op->args[1].as<VarNode>();
490 auto it = alloc_map_.find(buffer);
491 if (it == alloc_map_.end()) {
492 return StmtExprMutator::VisitExpr_(op);
493 }
494 const StorageEntry* se = it->second;
495 PrimExpr offset = this->VisitExpr(op->args[2]);
496 PrimExpr extent = this->VisitExpr(op->args[3]);
497 uint64_t elem_bits = dtype.bits() * dtype.lanes();
498 ICHECK_EQ(se->bits_offset % elem_bits, 0U);
499 if (se->bits_offset != 0) {
500 offset = make_const(offset.dtype(), se->bits_offset / elem_bits) + offset;
501 }
502 return Call(op->dtype, op->op, {op->args[0], se->alloc_var, offset, extent, op->args[4]});
503 } else {
504 return StmtExprMutator::VisitExpr_(op);
505 }
506 }

508 Stmt VisitStmt_(const AttrStmtNode* op) final {
509 if (op->attr_key == attr::thread_extent || op->attr_key == attr::virtual_thread ||
510 attr::IsPragmaKey(op->attr_key)) {
511 // remake all the allocation at the attach scope.
512 if (attach_map_.count(op)) {
513 auto& svec = attach_map_[op];
514 Stmt stmt = StmtExprMutator::VisitStmt_(op);
515 op = stmt.as<AttrStmtNode>();

521 Stmt stmt = StmtExprMutator::VisitStmt_(op);

524 if (it == alloc_map_.end()) return stmt;
525 return AttrStmt(it->second->alloc_var, op->attr_key, op->value, op->body);
526 } else {
527 return StmtExprMutator::VisitStmt_(op);
528 }

547 private:
548 struct StorageEntry {
549 // The scope that this alloc attaches after
550 // For shared/local memory it is beginning of the thread extent.
551 // for global memory it is nullptr, means beginning of everything.
552 const Object* attach_scope_{nullptr};
553 // The constant size of the buffer in bits, only used if it is constant
554 uint64_t const_nbits{0};
555 // The storage scope.
556 StorageScope scope;
557 // The physical dimensionality of the allocations. Since
558 // StorageRewrite is applied after StorageFlatten/FlattenBuffer,
559 // this is size of `AllocateNode::extents`. If moved
560 size_t ndim;
561 // Allocs that shares this entry.
562 std::vector<const AllocateNode*> allocs;
563 // The children of this entry, not including itself.
564 std::vector<StorageEntry*> merged_children;
565 // The replacement allocation, if any.
566 Stmt new_alloc;
567 // The var expr of new allocation.
568 Var alloc_var;
569 // The allocation element type.
570 DataType elem_type;
571 // This is non-zero if this allocate is folded into another one
572 // the address(in bits) becomes alloc_var + bits_offset;
573 // can be effectively converted to the element type.
574 // We need to convert bit_offset to offset of specific element type later.
575 //
576 // We use bits(instead of bytes) to support non-conventional indexing in hardware.
577 // When we are merging buffer together, the bits_offset are set to be aligned
578 // to certain value given by the max_simd_bits property of the special memory.
579 //
580 // This allows effective sharing among different types as long as their alignment
581 // requirement fits into the max_simd_bits.
582 uint64_t bits_offset{0};
583 };

588 return scope.tag.length() != 0 && scope.tag != ".dyn" && scope.tag != ".workspace" &&

601 Stmt MakeAttach(const std::vector<StorageEntry*>& svec, Stmt body) {
602 std::vector<Stmt> nest;
603 for (StorageEntry* e : svec) {
604 if (e->new_alloc.defined()) {
605 nest.push_back(e->new_alloc);
606 }
607 }
608 return MergeNest(nest, body);
609 }
610 // Remap the index
611 PrimExpr RemapIndex(DataType dtype, PrimExpr index, StorageEntry* e) {
612 if (e->bits_offset == 0) return index;
613 uint64_t elem_bits = dtype.bits();
614 ICHECK_EQ(e->bits_offset % elem_bits, 0U);
615 return make_const(index.dtype(), e->bits_offset / elem_bits) + index;
616 }
617 // Prepare the new allocations
618 void PrepareNewAlloc() {
619 for (size_t i = 0; i < alloc_vec_.size(); ++i) {
620 StorageEntry* e = alloc_vec_[i].get();
621 attach_map_[e->attach_scope_].push_back(e);
622 }
623 // find allocation via attach map.
624 for (auto& kv : attach_map_) {
625 // find the element with the most amount of bytes.
626 std::vector<StorageEntry*>& vec = kv.second;
627 // try to find merge, for tagged memory
628 for (size_t i = 0; i < vec.size(); ++i) {
629 StorageEntry* e = vec[i];
630 if (IsSpecialTaggedMemory(e->scope)) {
631 ICHECK_NE(e->const_nbits, 0U) << "Special tagged memory must be const size";
632 for (size_t j = 0; j < i; ++j) {
633 if (e->scope == vec[j]->scope) {
634 vec[j]->merged_children.push_back(e);
635 break;
636 }
637 }
638 }
639 }
640 // Start allocation
641 for (size_t i = 0; i < vec.size(); ++i) {
642 StorageEntry* e = vec[i];
643 // already merged
644 if (e->bits_offset != 0) continue;
645 if (e->merged_children.size() != 0) {
646 NewAllocTagMerged(e);
647 continue;
648 }
649 // Get the allocation size;
650 e->alloc_var = e->allocs[0]->buffer_var;
651 DataType alloc_type = e->allocs[0]->dtype;
652 for (const AllocateNode* op : e->allocs) {
653 if (op->dtype.lanes() > alloc_type.lanes()) {
654 alloc_type = op->dtype;
655 }
656 }
657
658 if (e->allocs.size() == 1) {
659 // simply use the original allocation.
660 e->new_alloc = Allocate(e->alloc_var, alloc_type, e->allocs[0]->extents,
661 e->allocs[0]->condition, Evaluate(0));
662 if (IsSpecialTaggedMemory(e->scope)) {
663 MemoryInfo info = GetMemoryInfo(e->scope.to_string());
664 if (info.defined()) {
665 uint64_t total_elem = e->const_nbits / e->elem_type.bits();
666 ICHECK_LE(total_elem * e->elem_type.bits(), info->max_num_bits)
667 << "Allocation exceed bound of memory tag " << e->scope.to_string();
668 }
669 }
670 } else {
671 // Build a merged allocation
672 PrimExpr combo_size;
673 for (const AllocateNode* op : e->allocs) {
674 ICHECK_EQ(op->extents.size(), 1)
675 << "Buffer var " << op->buffer_var->name_hint
676 << " was identified as a re-usable allocation, but has " << op->extents.size()
677 << " physical dimensions. "
678 << "Currently, only flat 1-d memory spaces should be identified as re-usable "
679 "allocations.";
680 PrimExpr sz = op->extents[0];
681 auto nbits = op->dtype.bits() * op->dtype.lanes();
682 if (const auto* imm = sz.as<IntImmNode>()) {
683 if (imm->value > std::numeric_limits<int>::max() / nbits) {
684 LOG(WARNING) << "The allocation requires : " << imm->value << " * " << nbits
685 << " bits, which is greater than the maximum of"
686 " int32. The size is cast to int64."
687 << "\n";
688 sz = make_const(DataType::Int(64), imm->value);
689 }
690 }
691 // transform to bits
692 auto sz_nbits = sz * nbits;
693 if (combo_size.defined()) {
694 combo_size = max(combo_size, sz_nbits);
695 } else {
696 combo_size = sz_nbits;
697 }
698 }
699 // transform to alloc bytes
700 auto type_bits = alloc_type.bits() * alloc_type.lanes();
701 bool divided = analyzer_.CanProve(indexmod(combo_size, type_bits) == 0);
702 combo_size = indexdiv(combo_size, type_bits);
703 // round up for can not divided
704 if (!divided) {
705 combo_size = combo_size + make_const(DataType::Int(32), 1);
706 }
707 combo_size = analyzer_.Simplify(combo_size);
708 e->new_alloc =
709 Allocate(e->alloc_var, alloc_type, {combo_size}, const_true(), Evaluate(0));
710 if (IsSpecialTaggedMemory(e->scope)) {
711 MemoryInfo info = GetMemoryInfo(e->scope.to_string());
712 if (info.defined()) {
713 uint64_t total_elem = e->const_nbits / e->elem_type.bits();
714 ICHECK_LE(total_elem * e->elem_type.bits(), info->max_num_bits)
715 << "Allocation exceed bound of memory tag " << e->scope.to_string();
716 }
717 }
718 }
719 }
720 }
721 }
722 // New allocation for merged data
723 void NewAllocTagMerged(StorageEntry* e) {
724 ICHECK_NE(e->scope.tag.length(), 0U);
725 // allocate with element type.
726 ICHECK_NE(e->const_nbits, 0U);
727 MemoryInfo info = GetMemoryInfo(e->scope.to_string());
728 uint64_t total_bits = e->const_nbits;
729 // By default, align to 32 bits.
730 size_t align = 32;
731 if (info.defined()) {
732 align = info->max_simd_bits;
733 }
734 // Always align to max_simd_bits
735 // so we can remap types by keeping this property
736 if (total_bits % align != 0) {
737 total_bits += align - (total_bits % align);
738 }
739 e->alloc_var = e->allocs[0]->buffer_var;
740 for (StorageEntry* child : e->merged_children) {
741 ICHECK_NE(child->const_nbits, 0U);
742 ICHECK_NE(total_bits, 0U);
743 child->bits_offset = total_bits;
744 child->alloc_var = e->alloc_var;
745 total_bits += child->const_nbits;
746 if (total_bits % align != 0) {
747 total_bits += align - (total_bits % align);
748 }
749 }
750 uint64_t type_bits = e->elem_type.bits() * e->elem_type.lanes();
751 PrimExpr alloc_size =
752 make_const(e->allocs[0]->extents[0].dtype(), (total_bits + type_bits - 1) / type_bits);
753 e->new_alloc = Allocate(e->alloc_var, e->elem_type, {alloc_size}, const_true(), Evaluate(0));
754 if (info.defined()) {
755 ICHECK_LE(total_bits, info->max_num_bits)
756 << "Allocation exceed bound of memory tag " << e->scope.to_string();
757 }
758 }
759 // Liveness analysis to find gen and kill point of each variable.
760 void LivenessAnalysis(const std::vector<StmtEntry>& seq) {
761 // find kill point, do a reverse linear scan.
762 std::unordered_set<const VarNode*> touched;
763 for (size_t i = seq.size(); i != 0; --i) {
764 const StmtEntry& s = seq[i - 1];
765 for (const VarNode* buffer : s.touched) {
766 if (!touched.count(buffer)) {
767 touched.insert(buffer);
768 event_map_[s.stmt].kill.push_back(buffer);
769 }
770 }
771 }
772 // find gen point, do forward scan
773 touched.clear();
774 for (size_t i = 0; i < seq.size(); ++i) {
775 int64_t offset = seq[i].scope_pair_offset;
776 if (offset < 0) continue;
777 const StmtEntry& s = seq[i + offset];
778 for (const VarNode* buffer : s.touched) {
779 if (!touched.count(buffer)) {
780 touched.insert(buffer);
781 event_map_[s.stmt].gen.push_back(buffer);
782 }
783 }
784 }

786 void PlanNewScope(const Object* op) {
787 if (thread_scope_ != nullptr) {
788 ICHECK(thread_scope_ == op);
789 // erase all memory atatched to this scope.
790 for (auto it = const_free_map_.begin(); it != const_free_map_.end();) {
791 if (it->second->attach_scope_ == op) {
792 it = const_free_map_.erase(it);
793 } else {
794 ++it;
795 }
796 }
797 for (auto it = sym_free_list_.begin(); it != sym_free_list_.end();) {
798 if ((*it)->attach_scope_ == op) {
799 it = sym_free_list_.erase(it);
800 } else {
801 ++it;
802 }
803 }
804 thread_scope_ = nullptr;
805 } else {
806 thread_scope_ = op;
807 }

810 // Memory plan algorithm
811 void PlanMemory(const std::vector<StmtEntry>& seq,
812 const std::unordered_map<const VarNode*, AllocEntry>& alloc_info) {
813 std::unordered_set<const VarNode*> inplace_flag;
814
815 for (size_t i = 0; i < seq.size(); ++i) {
816 const StmtEntry& s = seq[i];
817 auto it = event_map_.find(seq[i].stmt);
818
819 // scope_pair_offset >= 0 means it is either
820 // - leaf stmt(offset = 0)
821 // - beginning of scope(offset < 0)
822 // In both cases, we need to handle the gen event correctly
823 if (it != event_map_.end() && seq[i].scope_pair_offset >= 0) {
824 // Inplace operation detection
825 // specially handle this
826 bool detect_inplace = detect_inplace_ && (it->second.gen.size() <= 2);
827
828 for (const VarNode* var : it->second.gen) {
829 ICHECK(alloc_info.count(var));
830 const AllocEntry& entry = alloc_info.at(var);
831 const AllocateNode* alloc = entry.alloc;
832 auto storage_scope = StorageScope::Create(GetPtrStorageScope(GetRef<Var>(var)));
833 StorageEntry* dst_entry = nullptr;
834 // inplace detection
835 if (detect_inplace) {
836 // only one inplace var for s.stmt
837 bool inplace_found = false;
838 for (const VarNode* src : it->second.kill) {
839 if (!inplace_flag.count(src) && alloc_map_.count(src)) {
840 InplaceOpVerifier visitor;
841 StorageEntry* src_entry = alloc_map_.at(src);
842 if (src_entry->scope == storage_scope &&
843 src_entry->attach_scope_ == thread_scope_ &&
844 src_entry->elem_type == alloc->dtype.element_of() &&
845 visitor.Check(s.stmt, var, src)) {
846 uint64_t const_nbits = static_cast<uint64_t>(alloc->ConstantAllocationSize()) *
847 alloc->dtype.bits() * alloc->dtype.lanes();
848 if (src_entry->const_nbits == const_nbits && !inplace_found) {
849 // successfully inplace
850 dst_entry = src_entry;
851 inplace_flag.insert(src);
852 inplace_found = true;
853 }
854 }
855 }
856 }
857 }
858 if (dst_entry == nullptr) {
859 dst_entry =
860 FindAlloc(alloc, thread_scope_, storage_scope, entry.num_physical_dimensions);
861 }
862 dst_entry->allocs.emplace_back(alloc);
863 alloc_map_[var] = dst_entry;
864 }
865 }
866 // enter/exit new scope
867 if (s.stmt->IsInstance<AttrStmtNode>()) {
868 const auto* op = static_cast<const AttrStmtNode*>(s.stmt);
869 if (op->attr_key == attr::thread_extent || op->attr_key == attr::virtual_thread ||
870 attr::IsPragmaKey(op->attr_key)) {
871 PlanNewScope(op);
872 } else {
873 ICHECK(op->attr_key == attr::extern_scope);
874 }
875 } else if (s.stmt->IsInstance<ForNode>()) {
876 const auto* op = static_cast<const ForNode*>(s.stmt);
877 if (op->kind == ForKind::kParallel) {
878 if (thread_scope_ == nullptr || thread_scope_ == op) {
879 PlanNewScope(op);
880 }
881 }
882 }
883 // scope_pair_offset <= 0 means it is either
884 // - leaf stmt(offset = 0)
885 // - end of scope(offset < 0)
886 // In both cases, we need to handle the kill event correctly
887 if (it != event_map_.end() && seq[i].scope_pair_offset <= 0) {
888 for (const VarNode* var : it->second.kill) {
889 // skip space which are already replaced by inplace
890 if (!inplace_flag.count(var)) {
891 this->Free(var);
892 }
893 }
894 }
895 }
896 }
897 // Allocate new storage entry.
898 StorageEntry* NewAlloc(const AllocateNode* op, const Object* attach_scope,
899 const StorageScope& scope, size_t const_nbits) {
900 ICHECK(op != nullptr);
901 // Re-use not successful, allocate a new buffer.
902 auto entry = std::make_unique<StorageEntry>();
903 entry->attach_scope_ = attach_scope;
904 entry->scope = scope;
905 entry->elem_type = op->dtype.element_of();
906 entry->const_nbits = const_nbits;
907 StorageEntry* e = entry.get();
908 alloc_vec_.emplace_back(std::move(entry));
909 return e;
910 }
911
912 StorageEntry* FindAlloc(const AllocateNode* op, const Object* attach_scope,
913 const StorageScope& scope, size_t num_physical_dimensions) {
914 ICHECK(op != nullptr);
915 // skip plan for local variable,
916 // compiler can do a better job with register allocation.
917 const uint64_t match_range = 16;
918 uint64_t op_elem_bits = op->dtype.bits() * op->dtype.lanes();
919 uint64_t const_nbits = static_cast<uint64_t>(op->ConstantAllocationSize() * op_elem_bits);
920
921 // If the size of the array isn't known at compile-time, it must
922 // have its own allocation with size determined at runtime.
923 bool is_known_size = (const_nbits != 0);
924
925 // Currently, only flat memory spaces can be re-used. Packing
926 // into N-d space (e.g. 2-d texture memory on GPUs) will require
927 // more in-depth algorithms.
928 bool is_flat_memory_space = (num_physical_dimensions == 1);
929
930 // disable reuse of small arrays, they will be lowered to registers in LLVM
931 // This rules only apply if we are using non special memory
932 bool is_small_array =
933 (scope.tag.length() == 0) && (scope.rank >= StorageRank::kWarp || op->dtype.is_handle() ||
934 (is_known_size && const_nbits <= 32));
935
936 if (is_small_array || !is_flat_memory_space) {
937 return NewAlloc(op, attach_scope, scope, const_nbits);
938 }
939
940 if (is_known_size) {
941 // constant allocation.
942 auto begin = const_free_map_.lower_bound(const_nbits / match_range);
943 auto mid = const_free_map_.lower_bound(const_nbits);
944 auto end = const_free_map_.upper_bound(const_nbits * match_range);
945 // start looking at the buffer that is bigger than the required size first
946 for (auto it = mid; it != end; ++it) {
947 StorageEntry* e = it->second;
948 if (e->attach_scope_ != attach_scope) continue;
949 if (e->scope != scope) continue;
950 // when not divided, no reuse, eg, float4 vs float3
951 if (e->bits_offset % op_elem_bits != 0) continue;
952 e->const_nbits = std::max(const_nbits, e->const_nbits);
953 const_free_map_.erase(it);
954 return e;
955 }
956 // then start looking at smaller buffers.
957 for (auto it = mid; it != begin;) {
958 --it;
959 StorageEntry* e = it->second;
960 if (e->attach_scope_ != attach_scope) continue;
961 if (e->scope != scope) continue;
962 if (e->elem_type != op->dtype.element_of()) continue;
963 e->const_nbits = std::max(const_nbits, e->const_nbits);
964 const_free_map_.erase(it);
965 return e;
966 }
967 } else {
968 // Simple strategy: round roubin.
969 for (auto it = sym_free_list_.begin(); it != sym_free_list_.end(); ++it) {
970 StorageEntry* e = *it;
971 if (e->attach_scope_ != attach_scope) continue;
972 if (e->scope != scope) continue;
973 if (e->elem_type != op->dtype.element_of()) continue;
974 sym_free_list_.erase(it);
975 return e;
976 }
977 }
978 return NewAlloc(op, attach_scope, scope, const_nbits);
979 }
980 // simulated free.
981 void Free(const VarNode* var) {
982 auto it = alloc_map_.find(var);
983 ICHECK(it != alloc_map_.end());
984 StorageEntry* e = it->second;
985 ICHECK_NE(e->allocs.size(), 0U);
986
987 // disable reuse of small arrays, they will be lowered to registers in LLVM
988 // This rules only apply if we are using non special memory
989 if (e->scope.tag.length() == 0) {
990 // Disable sharing of local memory.
991 if (e->scope.rank >= StorageRank::kWarp || e->allocs[0]->dtype.is_handle()) return;
992 // disable reuse of small arrays
993 if (e->const_nbits > 0 && e->const_nbits <= 32) return;
994 }
995 // normal free.
996 if (e->const_nbits != 0) {
997 const_free_map_.insert({e->const_nbits, e});
998 } else {
999 sym_free_list_.push_back(e);
1000 }
1001 }
1002 // thread scope.
1003 const Object* thread_scope_{nullptr};
1004 // whether enable inplace detection.
1005 bool detect_inplace_{false};
1006 // Locations of free ops.
1007 std::unordered_map<const Object*, EventEntry> event_map_;
1008 // constant size free map.
1009 std::multimap<uint64_t, StorageEntry*> const_free_map_;
1010 // symbolic free list, for non constant items.
1011 std::list<StorageEntry*> sym_free_list_;
1012 // The allocation attach map
1013 std::unordered_map<const Object*, std::vector<StorageEntry*> > attach_map_;
1014 // The allocation assign map
1015 std::unordered_map<const VarNode*, StorageEntry*> alloc_map_;
1016 // The allocations
1017 std::vector<std::unique_ptr<StorageEntry> > alloc_vec_;
1018 // The buffer objects being remapped
1019 std::unordered_map<const BufferNode*, Buffer> buffer_remap_;
1020 // analyzer
1021 arith::Analyzer analyzer_;
1022};

1024/* Helper struct containing information on how a buffer is declared and used
1025 *
1026 */
1027struct BufferVarInfo {
1028 enum DeclarationLocation {
1029 kPrimFuncParam = (1 << 0),
1030 kPrimFuncBufferMap = (1 << 1),
1031 kAllocateNode = (1 << 2),
1032 kAllocateConstNode = (1 << 3),
1033 kLetNode = (1 << 4),
1034 };
1035
1036 // The tir::Var that represents this buffer.
1037 Var var;
1038
1039 // The data type of an element of the buffer.
1040 DataType element_dtype;
1041
1042 /* The extent of the buffer.
1043 *
1044 * If multidimensional, the extent of the last dimension of the buffer. If the
1045 * size is unknown (e.g. pointer arguments to PrimFunc with no corresponding
1046 * entry in buffer_map), then extent is zero.
1047 */
1048 PrimExpr extent;
1049
1050 // Where the buffer was declared
1051 DeclarationLocation declaration_location;
1052
1053 // When accessed, which element type is it accessed as. This may
1054 // differ both in base type (e.g. int32* cast to float32* after
1055 // packing in StorageRewrite) or in number of lanes (e.g. float16*
1056 // cast to float16x4*).
1057 std::unordered_set<DataType> access_dtype;
1058
1059 DataType get_preferred_dtype() const {
1060 std::unordered_set<DataType> base_access_dtype;
1061 for (auto dtype : access_dtype) {
1062 base_access_dtype.insert(dtype.element_of());
1063 }
1064 // If the array is accessed as multiple base types within a
1065 // function, no point in changing the declared type. CodeGenC can
1066 // handle this with a type-cast prior to indexing. Vulkan will
1067 // raise an error at code-gen time, if a later pass doesn't split
1068 // it out.
1069 if (base_access_dtype.size() != 1) {
1070 return element_dtype;
1071 }
1072
1073 DataType preferred_base_type = *base_access_dtype.begin();
1074
1075 // If there is only one vectorizable size used to access the
1076 // buffer, and if that access size is compatible with the array
1077 // size, then the buffer is vectorizable. In the future, this
1078 // could be improved to allow vectorized buffer access of size
1079 // GCD(*lanes_used), if necessary.
1080 int preferred_lanes = element_dtype.lanes();
1081 if ((element_dtype.lanes() == 1) && (access_dtype.size() == 1)) {
1082 arith::Analyzer analyzer_;
1083 arith::ModularSet me = analyzer_.modular_set(extent);
1084
1085 int lanes = access_dtype.begin()->lanes();
1086 if ((me->coeff % lanes == 0) && (me->base % lanes == 0)) {
1087 preferred_lanes = lanes;
1088 }
1089 }
1090
1091 return preferred_base_type.with_lanes(preferred_lanes);
1092 }

1231 *
1232 * @param value_dtype The dtype of the value being stored to or
1233 * loaded from the buffer.

1236 *

1238 *
1239 * @param predicate The predicate used for the store/load.
1240 */
1241 void OnArrayAccess(DataType value_dtype, const VarNode* buffer, const Array<PrimExpr>& indices) {
1242 auto it = info_map_.find(buffer);
1243 ICHECK(it != info_map_.end()) << "Load/Store of buffer " << buffer->name_hint << " (" << buffer
1244 << ") occurred before its declaration.";
1245 BufferVarInfo& var_info = it->second;
1246
1247 if (value_dtype.element_of() == DataType::Bool()) {
1248 value_dtype = DataType::Int(8).with_lanes(value_dtype.lanes());
1249 }
1250
1251 if (var_info.element_dtype.is_handle()) {
1252 ICHECK(allow_untyped_pointers_) << "Variable " << buffer->name_hint
1253 << " was missing a type annotation in its declaration";
1254 var_info.element_dtype = value_dtype.element_of();
1255 }
1256
1257 for (int i = 0; i < static_cast<int>(indices.size()) - 1; i++) {
1258 ICHECK(indices[i].dtype().is_scalar())
1259 << "Only the last index of a buffer access may be a vector type.";
1260 }
1261 int index_lanes = indices.size() ? indices.back().dtype().lanes() : 1;
1262
1263 DataType access_dtype = value_dtype;
1264
1265 int lanes_used = var_info.element_dtype.lanes();
1266
1267 // This can happen due to a previous pass that had rewrite_store_load =
1268 // false. This occurs from the StorageRewrite in tvm::lower, followed by the
1269 // PointerValueTypeRewrite in BuildSPIRV. The rewrite_store_load = false is
1270 // necessary because the C-based codegens do not yet support vectorized
1271 // pointer types (e.g. float16x4*). Once they do, this if statement should
1272 // instead be replaced by the below ICHECK_EQ.
1273 if (index_lanes * var_info.element_dtype.lanes() != value_dtype.lanes()) {
1274 ICHECK_EQ(index_lanes, value_dtype.lanes());
1275 lanes_used = 1;
1276 var_info.element_dtype = var_info.element_dtype.with_lanes(1);
1277 }
1278
1279 // TODO(Lunderberg): Uncomment this check once it can be applied.
1280 // See https://discuss.tvm.apache.org/t/pre-rfc-vectorized-tir-buffers/10615
1281 // for discussion.
1282
1283 // ICHECK_EQ(index_lanes * var_info.element_dtype.lanes(), value_dtype.lanes())
1284 // << "Attempting to retrieve " << value_dtype.lanes() << " lanes of data with "
1285 // << index_lanes << " indices into an array whose elements have "
1286 // << var_info.element_dtype.lanes() << " lanes. "
1287 // << "Expected output with " << index_lanes * var_info.element_dtype.lanes()
1288 // << " lanes.";
1289
1290 // If the index is a RampNode with stride of 1 and offset
1291 // divisible by the number of number of lanes, and the predicate
1292 // does not apply any masking, then this array access could be
1293 // vectorized.
1294 if (indices.size()) {
1295 const RampNode* ramp_index = indices[indices.size() - 1].as<RampNode>();
1296 if (ramp_index && is_one(ramp_index->stride)) {
1297 arith::ModularSet me = analyzer_.modular_set(ramp_index->base);
1298 if ((me->coeff % ramp_index->lanes == 0) && (me->base % ramp_index->lanes == 0)) {
1299 lanes_used = ramp_index->lanes;
1300 }
1301 }
1302 }
1303
1304 var_info.access_dtype.insert(access_dtype.with_lanes(lanes_used));
1305 }

1308 std::unordered_map<const VarNode*, BufferVarInfo> info_map_;
1309
1310 //
1311 bool allow_untyped_pointers_{false};

1313 // internal analyzer
1314 arith::Analyzer analyzer_;

1317/* \brief Rewrites buffer/pointer variables from scalar types to vectorized
1318 * types.
1319 *
1320 * Some runtimes do not allow casting between composite types and the underlying
1321 * base type (e.g. Vulkan, casting from 1-lane float16* to 4-lane float16x4*).
1322 * In these cases, in order to have vectorized load/store on an array, the
1323 * element type of that array must be vectorized. This is in contrast to C-style
1324 * runtimes, in which `float16x4* vec = *(float16x4*)(float_arr + offset)` is
1325 * valid.
1326 *
1327 * By default, VectorTypeRewriter will attempt to rewrite all buffer variables to
1328 * vectorized access, if the load/store occurring in the PrimFunc are all
1329 * vectorized. This includes adjusting the indices being used to access the
1330 * array. (e.g. If `float16* scalar_arr` is being converted to `float16x4*
1331 * vec_arr`, then `scalar_arr[Ramp(offset, 1, 4)]` will be converted to
1332 * `vec_arr[offset/4]`.)
1333 *
1334 * Currently, several of the C-style runtimes do not support buffers whose
1335 * elements are vectorized types, or rely on the presence of the Ramp nodes to
1336 * identify vectorized loads. The boolean parameters in the constructor are to
1337 * mimic the previous behavior of VectorTypeRewriter, to avoid breaking these
1338 * runtimes. Once all runtimes support vectorized buffer elements, these
1339 * parameters can be removed.
1340 */
1341class VectorTypeRewriter : public StmtExprMutator {
1342 public:
1343 /* Constructor
1344 *
1345 * @param checker The VectorTypeAccessChecker that has previously read out
1346 * information from the PrimFunc
1347 *
1348 * @param rewrite_params Whether pointer-type parameters passed into the
1349 * function should be rewritten from scalar types to vectorized types.
1350 *
1351 * @param rewrite_buffer_map Whether buffers present in the buffer_map should
1352 * have their data variable be rewritten from scalar types to vectorized types.
1353 *
1354 * @param rewrite_allocate_node Whether the buffer variable associated with
1355 * AllocateNodes should be rewritten from scalar types to vectorized types.
1356 *
1357 * @param rewrite_indices Whether the indices to the Load and Store nodes
1358 * should be rewritten to correspond to the new buffer_var type.
1359 *
1360 * @param rewrite_let_node Whether pointer declarations in let nodes
1361 * should be re-written.
1362 */
1363 VectorTypeRewriter(const std::unordered_map<const VarNode*, BufferVarInfo>& info_map,
1364 bool rewrite_params = true, bool rewrite_buffer_map = true,
1365 bool rewrite_allocate_node = true, bool rewrite_indices = true,
1366 bool rewrite_let_node = true, bool rewrite_allocate_const_node = true)
1367 : rewrite_indices_(rewrite_indices) {
1368 int rewrite_mask = 0;
1369 if (rewrite_params) {
1370 rewrite_mask |= BufferVarInfo::kPrimFuncParam;
1371 }
1372 if (rewrite_buffer_map) {
1373 rewrite_mask |= BufferVarInfo::kPrimFuncBufferMap;
1374 }
1375 if (rewrite_allocate_node) {
1376 rewrite_mask |= BufferVarInfo::kAllocateNode;
1377 }
1378 if (rewrite_let_node) {
1379 rewrite_mask |= BufferVarInfo::kLetNode;
1380 }
1381 if (rewrite_allocate_const_node) {
1382 rewrite_mask |= BufferVarInfo::kAllocateConstNode;
1383 }
1384
1385 // Rewrite any buffer variables whose preferred type isn't their current type.
1386 for (const auto& pair : info_map) {
1387 const auto& var_info = pair.second;
1388 DataType preferred = var_info.get_preferred_dtype();
1389 if (preferred != var_info.element_dtype && (rewrite_mask & var_info.declaration_location)) {
1390 Var old_buffer_var = var_info.var;
1391 Var new_buffer_var(old_buffer_var->name_hint,
1392 PointerType(PrimType(preferred), GetPtrStorageScope(old_buffer_var)),
1393 old_buffer_var->span);
1394
1395 rewrite_map_[var_info.var.get()] = {var_info.var, new_buffer_var, var_info.element_dtype,
1396 preferred};
1397 }
1398 }
1399 }

1411 template <typename Node>
1412 Node VisitBufferAccess(Node node) {
1413 if (!rewrite_indices_) {
1414 return node;
1415 }
1416
1417 auto it = rewrite_map_.find(node->buffer->data.get());
1418 if (it == rewrite_map_.end()) {
1419 return node;
1420 }
1421 const auto& info = it->second;
1422
1423 Array<PrimExpr> indices = node->indices;
1424
1425 const RampNode* ramp_index = indices[indices.size() - 1].as<RampNode>();
1426 if (ramp_index && is_one(ramp_index->stride)) {
1427 PrimExpr new_index =
1428 ramp_index->base / make_const(ramp_index->base.dtype(), ramp_index->lanes);
1429 if (ramp_index->lanes != info.factor()) {
1430 new_index = Ramp(new_index, ramp_index->stride, ramp_index->lanes / info.factor(),
1431 ramp_index->span);
1432 }
1433
1434 indices.Set(indices.size() - 1, new_index);
1435 }
1436
1437 auto writer = node.CopyOnWrite();
1438 writer->buffer = RemapBuffer(node->buffer);
1439 writer->indices = indices;
1440
1441 return node;
1442 }

1475 Buffer RemapBuffer(Buffer buf) {
1476 auto cache_key = buf.get();
1477
1478 auto cache_it = buffer_map_.find(cache_key);
1479 if (cache_it != buffer_map_.end()) {
1480 return cache_it->second;
1481 }
1482
1483 auto info_it = rewrite_map_.find(buf->data.get());
1484 if (info_it != rewrite_map_.end()) {
1485 auto& info = info_it->second;
1486
1487 Array<PrimExpr> shape = buf->shape;
1488 PrimExpr last_dim = shape[shape.size() - 1];
1489 shape.Set(shape.size() - 1, last_dim / make_const(last_dim.dtype(), info.factor()));
1490
1491 auto writer = buf.CopyOnWrite();
1492 writer->data = info.new_buffer_var;
1493 writer->dtype = info.new_element_dtype;
1494 writer->shape = shape;
1495 }
1496
1497 buffer_map_[cache_key] = buf;
1498 return buf;
1499 }

1501 PrimExpr VisitExpr_(const CallNode* op) final {
1502 if (op->op.same_as(builtin::tvm_access_ptr())) {
1503 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
1504 op = expr.as<CallNode>();
1505
1506 if (!rewrite_indices_) {
1507 return expr;
1508 }
1509

1512 if (it == rewrite_map_.end()) {
1513 return expr;

1518 PrimExpr extent = op->args[3];
1519 PrimExpr flag = op->args[4];
1520
1521 PrimExpr e_dtype = tir::TypeAnnotation(info.new_element_dtype);
1522 int factor = info.factor();
1523 extent = extent / make_const(extent.dtype(), factor);
1524 index = index / make_const(index.dtype(), factor);
1525 Array<PrimExpr> acc_args{e_dtype, info.new_buffer_var, index, extent, flag};
1526 return Call(info.new_element_dtype, builtin::tvm_access_ptr(), acc_args);
1527
1528 } else {
1529 return StmtExprMutator::VisitExpr_(op);
1530 }

1573 /* Update the parameters and all remaining variable references
1574 *
1575 * Should be called after calling operator() on the body of the
1576 * function.
1577 *
1578 * @param func A pointer to the PrimFunc being modified.
1579 */
1580 void Finalize(PrimFunc* func_ptr) {
1581 ICHECK(func_ptr) << "Finalize expects a non-null pointer";
1582 auto& func = *func_ptr;
1583 auto* n = func.CopyOnWrite();
1584
1585 // Remap any remaining references to the old buffer variables
1586 Map<Var, PrimExpr> var_remap;
1587 for (const auto& pair : rewrite_map_) {
1588 const auto& info = pair.second;
1589 var_remap.Set(info.old_buffer_var, info.new_buffer_var);
1590 }
1591 n->body = Substitute(n->body, var_remap);
1592
1593 // Remap the argument list to use the new buffer variables.
1594 Array<Var> new_params;
1595 for (const auto& old_param : n->params) {
1596 auto it = rewrite_map_.find(old_param.get());
1597 if (it == rewrite_map_.end()) {
1598 new_params.push_back(old_param);
1599 } else {
1600 const auto& info = it->second;
1601 new_params.push_back(info.new_buffer_var);
1602 }
1603 }
1604 n->params = new_params;
1605
1606 // Remap the Buffer objects in PrimFunc::buffer_map so that the
1607 // buffers use the new buffer variables
1608 Map<Var, Buffer> new_buffer_map;
1609 for (const auto& pair : n->buffer_map) {
1610 Var key = pair.first;
1611 Buffer old_buffer = pair.second;
1612 Var old_var = old_buffer->data;
1613 Buffer new_buffer = RemapBuffer(old_buffer);
1614 new_buffer_map.Set(key, new_buffer);
1615 }
1616 n->buffer_map = new_buffer_map;
1617 }

1621 Var old_buffer_var;
1622 Var new_buffer_var;
1623 DataType old_element_dtype;
1624 DataType new_element_dtype;

1626 int factor() const {
1627 int old_lanes = old_element_dtype.lanes();
1628 int new_lanes = new_element_dtype.lanes();
1629 ICHECK_EQ(new_lanes % old_lanes, 0);
1630 return new_lanes / old_lanes;
1631 }
1632 };

1636 std::unordered_map<const BufferNode*, Buffer> buffer_map_;

1639// Rewrite allocates, pointer parameters, and buffer map into vectorized versions
1640// if each access into a buffer is the same vector type.
1641PrimFunc PointerValueTypeRewrite(PrimFunc f, bool allow_untyped_pointers = false,
1642 bool rewrite_params = true, bool rewrite_buffer_map = true,
1643 bool rewrite_allocate_node = true, bool rewrite_indices = true,
1644 bool rewrite_let_node = true,
1645 bool rewrite_allocate_const_node = true) {
1646 VectorTypeAccessChecker checker(f->params, f->buffer_map, allow_untyped_pointers);
1647 checker(f->body);
1648
1649 VectorTypeRewriter rewriter(checker.info_map_, rewrite_params, rewrite_buffer_map,
1650 rewrite_allocate_node, rewrite_indices, rewrite_let_node,
1651 rewrite_allocate_const_node);
1652 PrimFuncNode* n = f.CopyOnWrite();
1653 n->body = rewriter(std::move(n->body));
1654 rewriter.Finalize(&f);
1655
1656 return f;
1657}
