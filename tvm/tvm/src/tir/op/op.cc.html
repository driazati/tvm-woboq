51runtime::DataType GetRuntimeDataType(const Type& type) {
52 if (auto* n = type.as<PrimTypeNode>()) {
53 return n->dtype;
54 } else if (type.as<PointerTypeNode>()) {
55 return DataType::Handle();
56 } else if (IsVoidType(type)) {
57 return DataType::Void();
58 } else {
59 LOG(FATAL) << "Type " << type << " does not have a corresponding runtime::DataType";
60 return DataType::Handle();
61 }
62}

64Type GetType(const PrimExpr& expr) {
65 // TODO(tqchen): add recursive type inference for Call here
66 // once we introduced the corresponding fields to the IR.
67 if (auto* ptr = expr.as<tir::VarNode>()) {
68 // If Var has a more refined type annotation,
69 // return the type anotation
70 if (ptr->type_annotation.defined()) {
71 return ptr->type_annotation;

73 }
74 // Default: return the type indicated by the dtype.
75 runtime::DataType dtype = expr.dtype();
76 return GetTypeFromRuntimeDataType(dtype);
77}

100// The public function with a quick checking path.
101void BinaryOpMatchTypes(PrimExpr& lhs, PrimExpr& rhs, Span span) { // NOLINT(*)
102 CHECK(lhs.defined()) << "ValueError: `lhs` is null in the binary operator";
103 CHECK(rhs.defined()) << "ValueError: `rhs` is null in the binary operator";
104 if (lhs.dtype() == rhs.dtype()) return;
105 DataType ltype = lhs.dtype();
106 DataType rtype = rhs.dtype();
107 if (ltype.lanes() == 1 && rtype.lanes() != 1) {
108 lhs = tir::Broadcast(lhs, rtype.lanes());
109 } else if (rtype.lanes() == 1 && ltype.lanes() != 1) {
110 rhs = tir::Broadcast(rhs, ltype.lanes());
111 } else {
112 ICHECK(ltype.lanes() == rtype.lanes()) << "Cannot match type " << ltype << " vs " << rtype;
113 }
114 if (lhs.dtype() == rhs.dtype()) return;
115
116 ltype = lhs.dtype();
117 rtype = rhs.dtype();
118 // We keep dtypes conversion to be relatively consistent to reduce the amount code generated by
119 // operators. This can be helpful for users to find potential type conversion problems. The
120 // following are exceptions:
121 if (ltype.is_float() && rtype.is_float()) {
122 // Given two dissimilar floats, cast the lower bit version to the higher bit version.
123 // E.g. fp16 + fp32 --> fp32 + fp32
124 if (ltype.bits() < rtype.bits()) {
125 lhs = cast(rtype, lhs);
126 } else {
127 rhs = cast(ltype, rhs);
128 }
129 } else if (!ltype.is_float() &&
130 (rtype.is_float() || datatype::Registry::Global()->GetTypeRegistered(rtype.code()))) {
131 // Cast int->float when the other operand is a float
132 lhs = cast(rtype, lhs);
133 } else if ((ltype.is_float() || datatype::Registry::Global()->GetTypeRegistered(ltype.code())) &&
134 !rtype.is_float()) {
135 // Cast int->float when the other operand is a float
136 rhs = cast(ltype, rhs);
137 } else if (!ltype.is_bfloat16() &&
138 (rtype.is_bfloat16() ||
139 datatype::Registry::Global()->GetTypeRegistered(rtype.code()))) {
140 // Cast int->bfloat16 when the other operand is a bfloat16
141 lhs = cast(rtype, lhs);
142 } else if ((ltype.is_bfloat16() ||
143 datatype::Registry::Global()->GetTypeRegistered(ltype.code())) &&
144 !rtype.is_bfloat16()) {
145 // Cast int->bfloat16 when the other operand is a bfloat16
146 rhs = cast(ltype, rhs);
147 } else if ((ltype.is_int() && rtype.is_int()) || (ltype.is_uint() && rtype.is_uint())) {
148 // Promote int to higher bits e.g. int8 + int16 --> int16 + int16
149 if (ltype.bits() < rtype.bits()) {
150 lhs = cast(rtype, lhs);
151 } else {
152 rhs = cast(ltype, rhs);
153 }
154 } else if ((ltype.is_int() && rtype.is_uint()) || (ltype.is_uint() && rtype.is_int())) {
155 // Handle mixing signed and unsigned integers
156 if (ltype.bits() < rtype.bits()) {
157 lhs = cast(rtype, lhs);
158 } else if (ltype.bits() > rtype.bits()) {
159 rhs = cast(ltype, rhs);
160 } else {
161 // The width of signed and unsigned integers is same.
162 if (ltype.is_uint()) {
163 rhs = cast(ltype, rhs);
164 } else {
165 lhs = cast(rtype, lhs);
166 }
167 }
168 } else {
169 LOG(FATAL) << "Cannot match type " << ltype << " vs " << rtype;
170 }
171}

177// maximum and min limits
178PrimExpr max_value(const DataType& dtype, Span span) {
179 using namespace tir;
180 ICHECK_EQ(dtype.lanes(), 1);
181 if (dtype.is_int()) {
182 if (dtype.bits() == 64) {
183 return IntImm(dtype, std::numeric_limits<int64_t>::max(), span);
184 } else if (dtype.bits() < 64) {
185 int64_t val = 1;
186 val = (val << (dtype.bits() - 1)) - 1;
187 return IntImm(dtype, val, span);
188 }
189 } else if (dtype.is_uint()) {
190 if (dtype.bits() == 64) {
191 return make_const(dtype, std::numeric_limits<uint64_t>::max(), span);
192 } else if (dtype.bits() < 64) {
193 uint64_t val = 1;
194 val = (val << static_cast<uint64_t>(dtype.bits())) - 1;
195 return IntImm(dtype, static_cast<int64_t>(val), span);
196 }
197 } else if (dtype.is_float()) {
198 if (dtype.bits() == 64) {
199 return FloatImm(dtype, std::numeric_limits<double>::max(), span);
200 } else if (dtype.bits() == 32) {
201 return FloatImm(dtype, std::numeric_limits<float>::max(), span);
202 } else if (dtype.bits() == 16) {
203 return FloatImm(dtype, 65504.0, span);
204 }
205 } else if (dtype.is_bfloat16()) {
206 return FloatImm(dtype, std::numeric_limits<float>::max(), span);
207 }
208 LOG(FATAL) << "Cannot decide max_value for type" << dtype;
209 return PrimExpr();
210}

212PrimExpr min_value(const DataType& dtype, Span span) {
213 using namespace tir;
214 ICHECK_EQ(dtype.lanes(), 1);
215 if (datatype::Registry::Global()->GetTypeRegistered(dtype.code())) {
216 // TODO(tkonolige): need to convert all registered min functions to use the span.
217 auto f = datatype::GetMinFunc(dtype.code());
218 ICHECK(f) << "No minimum function registered for custom dtype " << (unsigned int)dtype.code();
219 // TODO(@hypercubestart) Document this change (and others associated with the overflowing
220 // floatimm min bug)
221 return (*f)(dtype.bits());
222 } else if (dtype.is_int()) {
223 if (dtype.bits() == 64) {
224 return IntImm(dtype, std::numeric_limits<int64_t>::lowest(), span);
225 } else if (dtype.bits() < 64) {
226 int64_t val = 1;
227 val = -(val << (dtype.bits() - 1));
228 return IntImm(dtype, val, span);
229 }
230 } else if (dtype.is_uint()) {
231 return IntImm(dtype, 0, span);
232 } else if (dtype.is_float()) {
233 if (dtype.bits() == 64) {
234 return FloatImm(dtype, std::numeric_limits<double>::lowest(), span);
235 } else if (dtype.bits() == 32) {
236 return FloatImm(dtype, std::numeric_limits<float>::lowest(), span);
237 } else if (dtype.bits() == 16) {
238 return FloatImm(dtype, -65504.0, span);
239 }
240 } else if (dtype.is_bfloat16()) {
241 return FloatImm(dtype, std::numeric_limits<float>::lowest(), span);
242 }
243 LOG(FATAL) << "Cannot decide min_value for type" << dtype;
244 return PrimExpr();
245}

247// infinity
248PrimExpr infinity(const DataType& dtype, Span span) {
249 using namespace tir;
250 ICHECK_EQ(dtype.lanes(), 1);
251 if (dtype.is_float()) {
252 if (dtype.bits() == 64) {
253 return FloatImm(dtype, std::numeric_limits<double>::infinity(), span);
254 } else if (dtype.bits() == 32 || dtype.bits() == 16) {
255 return FloatImm(dtype, std::numeric_limits<float>::infinity(), span);
256 }
257 }
258 LOG(FATAL) << "Cannot decide infinity for type " << dtype;
259 return PrimExpr();
260}

263template <typename ValueType>
264inline bool ConstPowerHelper(ValueType val, int* shift) {
265 if (val <= 0) return false;
266 shift[0] = 0;
267 while (val != 0) {
268 if (val & 1) {
269 return (val == 1);
270 }
271 ++shift[0];
272 val = val >> 1;
273 }
274 return true;
275}

277bool is_const_power_of_two_integer(const PrimExpr& x, int* shift) {
278 if (const auto* op = x.as<tir::IntImmNode>()) {
279 return ConstPowerHelper(op->value, shift);
280 } else {
281 return false;
282 }
283}

286PrimExpr cast(const DataType& t, PrimExpr value, Span span) {
287 using tir::FloatImmNode;
288 if (value.dtype() == t) return value;
289 // const fold IntImm as they are used in index computations
290 if (t.lanes() == 1) {
291 if (const IntImmNode* op = value.as<IntImmNode>()) {
292 return make_const(t, op->value, op->span);
293 } else if (const FloatImmNode* op = value.as<FloatImmNode>()) {
294 return make_const(t, op->value, op->span);
295 }
296 ICHECK(!value.dtype().is_handle()) << "Can't cast a handle to other types.";
297 return tir::Cast(t, value, span);
298 } else {
299 if (value.dtype().lanes() == 1) {
300 // manually unroll cast
301 DataType vtype = t.element_of();
302 if (value.dtype() != vtype) {
303 if (const IntImmNode* op = value.as<IntImmNode>()) {
304 value = make_const(vtype, op->value, op->span);
305 } else if (const FloatImmNode* op = value.as<FloatImmNode>()) {
306 value = make_const(vtype, op->value, op->span);
307 } else {
308 value = tir::Cast(vtype, value, span);
309 }
310 }
311 return tir::Broadcast(value, t.lanes(), span);
312 } else {
313 ICHECK(value.dtype().lanes() == t.lanes());
314 return tir::Cast(t, value, span);
315 }
316 }
317}

441// if_then_else
442PrimExpr if_then_else(PrimExpr cond, PrimExpr true_value, PrimExpr false_value, Span span) {
443 ICHECK(cond.dtype() == DataType::Bool(1))
444 << "if_then_else only accept the condition to be boolean type.";
445 BinaryOpMatchTypes(true_value, false_value, span);
446 if (const IntImmNode* op = cond.as<IntImmNode>()) {
447 if (op->value != 0) {
448 return true_value;
449 } else {
450 return false_value;
451 }
452 }
453
454 return tir::Call(true_value.dtype(), tir::builtin::if_then_else(),
455 {cond, true_value, false_value}, span);
456}

556// shift right
557PrimExpr operator>>(PrimExpr a, PrimExpr b) { return right_shift(a, b); }
558
559PrimExpr right_shift(PrimExpr a, PrimExpr b, Span span) {
560 type_check_integer_args(a, b, ">> operator (right shift)");
561
562 BinaryOpMatchTypes(a, b, span);
563 TVM_INDEX_CONST_PROPAGATION({
564 const DataType& rtype = a.dtype();
565 if (pb)
566 ICHECK(pb->value >= 0 && pb->value < rtype.bits())
567 << "Shift amount must be non-negative and less than " << rtype.bits() << " for type "
568 << rtype;
569 if (pa && pb) {
570 return IntImm(rtype, (pa->value >> pb->value), span);
571 }
572 if (pb) {
573 if (pb->value == 0) return a;

658PrimExpr abs(PrimExpr x, Span span) {
659 if (x.dtype().is_int()) {
660 using tir::IntImmNode;
661 const IntImmNode* px = x.as<IntImmNode>();

674 } else if (x.dtype().is_uint()) {
675 return x;
676 } else {
677 LOG(FATAL) << "Data type " << x.dtype()
678 << " not supported for absolute op. Skipping absolute op...";
679 return x;
680 }
681}

687 DataType t = DataType::Bool(x.dtype().lanes());
688 if (x.dtype().is_int() || x.dtype().is_uint()) {

699 } else {

969 if (lhs_is_int) { \

971 } else if (rhs_is_int) { \
972 *ret = (Func(args[0].operator PrimExpr(), args[1].operator int(), args[2])); \
973 } else { \
