57/*! Returns the arguments of the given statement */
58Array<PrimExpr> GetStmtArgs(const Stmt& stmt) {
59 auto attr{stmt.as<AttrStmtNode>()};
60 Stmt eval_stmt{attr ? attr->body : stmt};
61 auto eval{eval_stmt.as<EvaluateNode>()};
62 ICHECK(eval) << "Expected statement to be an evaluate node, but was " << eval_stmt->GetTypeKey();
63 auto call{eval->value.as<CallNode>()};
64 ICHECK(call) << "Expected expression to be a call node, but was " << eval->value->GetTypeKey();
65 return call->args;
66}
67
68enum class StmtType { global_copy, local_copy, compute };
69
70/*! Returns the type of the given statement */
71StmtType GetStmtType(const Stmt& stmt) {
72 Array<PrimExpr> args{GetStmtArgs(stmt)};
73 if (args[0].as<StringImmNode>()->value == "ethosu_copy") {
74 if (args[3].as<BufferLoadNode>()->buffer.scope() == "global") {
75 return StmtType::global_copy;
76 } else {
77 return StmtType::local_copy;
78 }
79 }
80 return StmtType::compute;
81}
82/*! Returns the buffer read my the given copy statement */
83Buffer GetCopyReadBuffer(const Stmt& stmt) {
84 Array<PrimExpr> args{GetStmtArgs(stmt)};
85 return args[1].as<BufferLoadNode>()->buffer;
86}
87
88/*! Returns the buffer written my the given copy statement */
89Buffer GetCopyWriteBuffer(const Stmt& stmt) {
90 Array<PrimExpr> args{GetStmtArgs(stmt)};
91 return args[3].as<BufferLoadNode>()->buffer;
92}
93
94/*! Returns the length of the given copy statement */
95int64_t GetCopyLength(const Stmt& stmt) {
96 Array<PrimExpr> args{GetStmtArgs(stmt)};
97 return args[2].as<IntImmNode>()->value;
98}
99
100/*! Returns the cycles of the given statement */
101int64_t GetStmtCycles(const Stmt& stmt) {
102 auto attr{stmt.as<AttrStmtNode>()};
103 if (attr && attr->attr_key == "pragma_compute_cycles_hint") {
104 int64_t cycles{Downcast<Integer>(attr->value)->value};
105 return cycles;
106 }
107 return 0;
108}

111/*!
112 * \brief This mutator moves allocates to the top of the body of the main
113 * function.
114 *
115 * Note: This pass can currently only be run in conjunction with the
116 * LowerToTIR() pass as it expects a single primitive function called
117 * "main" that is being offloaded to the NPU.
118 *
119 * For example,
120 * Before:
121 * allocate {
122 * extern_call(...)
123 * allocate {
124 * extern_call(...)
125 * }
126 * }
127 *
128 * After:
129 * allocate {
130 * allocate {
131 * extern_call(...)
132 * extern_call(...)
133 * }
134 * }
135 */
136class HoistAllocatesMutator : public StmtExprMutator {
137 public:
138 HoistAllocatesMutator() {}
139
140 PrimFunc operator()(PrimFunc main_func) {
141 Stmt new_main_func_body = SeqStmt::Flatten(this->VisitStmt(main_func->body));
142
143 // Write all allocates that were removed in reverse order
144 for (auto it = allocates_.rbegin(); it != allocates_.rend(); it++) {
145 Allocate current_alloc = *it;
146 if (it != allocates_.rbegin()) {
147 new_main_func_body = SeqStmt({new_main_func_body});
148 }
149 new_main_func_body =
150 Allocate(current_alloc->buffer_var, current_alloc->dtype, current_alloc->extents,
151 current_alloc->condition, new_main_func_body, current_alloc->annotations,
152 current_alloc->span);
153 }
154
155 PrimFunc new_main_func =
156 PrimFunc(main_func->params, new_main_func_body, main_func->ret_type, main_func->buffer_map,
157 main_func->preflattened_buffer_map, main_func->attrs);
158 return new_main_func;
159 }
160
161 private:
162 Stmt VisitStmt_(const AllocateNode* op) override {
163 allocates_.push_back(GetRef<Allocate>(op));
164 return VisitStmt(op->body);
165 }
166
167 /*! A stack to store allocates as they are visited. */
168 std::vector<Allocate> allocates_;
169};

171/*!
172 * \brief A pass to hoist allocate nodes to the top of the body of the main function.
173 *
174 * \return tvm::transform::Pass
175 */
176tvm::transform::Pass HoistAllocates() {
177 auto pass_func = [=](PrimFunc f, IRModule mod, tvm::transform::PassContext ctx) {
178 ICHECK(mod->GetGlobalVars().size() == 1 && mod->ContainGlobalVar("main"))
179 << "Expected a single primitive function called 'main'. Please run the HoistAllocates pass "
180 "in conjunction with the LowerToTIR() pass.";
181 return HoistAllocatesMutator()(f);
182 };
183 return tvm::tir::transform::CreatePrimFuncPass(pass_func, 0, "tir.contrib.ethos-u.HoistAllocates",
184 {});
185}

189/*!
190 * \brief Reorders copy and compute nodes in such a way that independent DMA copies
191 * and computes happen in parallel.
192 * Copies to buffers with local scope are not reordered since they copy LUT
193 * into the SHRAM and that already happens in parallel with copying weights into
194 * the weights encoder.
195 */
196class CopyComputeReorderingMutator : public StmtExprMutator {
197 public:
198 explicit CopyComputeReorderingMutator(int max_copy_movements, bool reorder_by_cycles)
199 : _max_copy_movements{max_copy_movements}, _reorder_by_cycles{reorder_by_cycles} {}
200
201 PrimFunc operator()(PrimFunc main_func) {
202 if (_max_copy_movements > 0) {
203 auto prim_func_node{main_func.CopyOnWrite()};
204 prim_func_node->body = this->VisitStmt(main_func->body);
205 return GetRef<PrimFunc>(prim_func_node);
206 }
207 return main_func;
208 }
209
210 private:
211 // A structure to hold a compute op with the corresponding weights/bias copy and LUT copy
212 struct OpWithCopies {
213 Stmt compute_op{};
214 Stmt global_copy{};
215 Stmt local_copy{};
216 };
217
218 Stmt VisitStmt_(const SeqStmtNode* op) override {
219 if (op->size() <= 1) {
220 return StmtExprMutator::VisitStmt_(op);
221 }
222
223 auto seq_stmt{GetRef<SeqStmt>(op)};
224 std::vector<Stmt> new_seq(seq_stmt->size());
225 std::copy(seq_stmt->seq.begin(), seq_stmt->seq.end(), new_seq.begin());
226
227 // Reorder the copies and computes based on the cycle count
228 if (_reorder_by_cycles) {
229 // We can't hide the first copy, so ignore it for the purpose of hiding copies
230 Stmt first_copy{};
231 if (stmt_is_global_copy(new_seq[0]) ||
232 (stmt_is_local_copy(new_seq[0]) && stmt_is_global_copy(new_seq[1]))) {
233 auto copy_position = stmt_is_global_copy(new_seq[0]) ? 0 : 1;
234 first_copy = new_seq[copy_position];
235 new_seq.erase(new_seq.begin() + copy_position);
236 }
237
238 // Build up a list of cells with the compute op and the copy ops that directly preceed it
239 std::vector<OpWithCopies> ops{};
240 for (size_t idx = 0; idx < new_seq.size(); ++idx) {
241 if (stmt_is_compute_op(new_seq[idx])) {
242 OpWithCopies new_op;
243 new_op.compute_op = new_seq[idx];
244 if (idx > 0) {
245 auto prev_op = new_seq[idx - 1];
246 if (!stmt_is_compute_op(prev_op)) {
247 if (stmt_is_local_copy(prev_op)) {
248 new_op.local_copy = prev_op;
249 } else {
250 new_op.global_copy = prev_op;
251 }
252 if (idx > 1) {
253 auto prev_prev_op = new_seq[idx - 2];
254 if (!stmt_is_compute_op(prev_prev_op)) {
255 if (stmt_is_local_copy(prev_prev_op)) {
256 new_op.local_copy = prev_prev_op;
257 } else {
258 new_op.global_copy = prev_prev_op;
259 }
260 }
261 }
262 }
263 }
264 ops.push_back(new_op);
265 }
266 }
267
268 // Move the global copies up by one. If in general the computes take longer than the copies,
269 // that should be good enough
270 for (size_t idx = 1; idx < ops.size(); ++idx) {
271 if (ops[idx].global_copy.as<AttrStmtNode>()) {
272 ops[idx - 1].global_copy = ops[idx].global_copy;
273 ops[idx].global_copy = {};
274 }
275 }
276
277 // If there are long copies, try to hide them further
278 for (size_t idx = ops.size() - 1; idx > 0; --idx) {
279 if (ops[idx].global_copy.as<AttrStmtNode>()) {
280 // Check whether the copy is hidden
281 int64_t copy_cycles{GetStmtCycles(ops[idx].global_copy)};
282 int64_t compute_cycles{GetStmtCycles(ops[idx].compute_op)};
283 bool is_hidden = compute_cycles >= copy_cycles;
284
285 // If the previous compute op is not already hiding another copy, move the copy back, so
286 // that it would be hidden by multiple computes
287 while (!is_hidden && !ops[idx - 1].global_copy.as<AttrStmtNode>() && (idx > 0)) {
288 int64_t new_compute_cycles{GetStmtCycles(ops[idx - 1].compute_op)};
289 ops[idx - 1].global_copy = ops[idx].global_copy;
290 ops[idx].global_copy = {};
291 compute_cycles += new_compute_cycles;
292 is_hidden = compute_cycles >= copy_cycles;
293 --idx;
294 }
295 }
296 }
297
298 // Reconstruct the op sequence from the vector of OpWithCopies
299 new_seq.clear();
300 if (first_copy.as<AttrStmtNode>()) {
301 new_seq.push_back(first_copy);
302 }
303 for (auto& op : ops) {
304 if (op.global_copy.as<AttrStmtNode>()) {
305 new_seq.push_back(op.global_copy);
306 }
307 if (op.local_copy.as<EvaluateNode>()) {
308 new_seq.push_back(op.local_copy);
309 }
310 if (op.compute_op.as<AttrStmtNode>()) {
311 new_seq.push_back(op.compute_op);
312 }
313 }
314 } else {
315 // Each copy statement to a buffer with global scope is moved up
316 // at most `_max_copy_movements` times.
317 for (size_t index = 0; index < new_seq.size(); ++index) {
318 if (GetStmtType(new_seq[index]) == StmtType::global_copy) {
319 int lower = std::max(0, static_cast<int>(index) - _max_copy_movements);
320 for (int i = index; i > lower && (GetStmtType(new_seq[i - 1]) == StmtType::compute);
321 --i) {
322 std::swap(new_seq[i - 1], new_seq[i]);
323 }
324 }
325 }
326 }
327
328 auto seq_stmt_node{CopyOnWrite(op)};
329 seq_stmt_node->seq = std::move(new_seq);
330 return Stmt{seq_stmt_node};
331 }
332
333 bool stmt_is_global_copy(const Stmt& stmt) { return GetStmtType(stmt) == StmtType::global_copy; }
334
335 bool stmt_is_local_copy(const Stmt& stmt) { return GetStmtType(stmt) == StmtType::local_copy; }
336
337 bool stmt_is_compute_op(const Stmt& stmt) { return GetStmtType(stmt) == StmtType::compute; }
338
339 /*! The maximum number of movements allowed for a copy. */
340 int _max_copy_movements;
341 /*! Whether we use the cycle hint to determine the reordering. */
342 bool _reorder_by_cycles;
343};
344
345/*!
346 * \brief A pass to reorder copy and compute nodes in such a way that independent DMA copies
347 * and computes happen in parallel. If reorder_by_cycles is set, we will ignore the
348 * max_copy_movements value.
349 *
350 * \param max_copy_movements: The maximum number of movements allowed for a copy.
351 * If None, the pass context option tir.contrib.ethos-u.copy_compute_reordering_max_copy_movements
352 * is used if provided, otherwise the default value will be 1.
353 *
354 * \param reorder_by_cycles: Whether to reorder copies and computes by cycles.
355 * If None, the pass context option tir.contrib.ethos-u.copy_compute_reordering_reorder_by_cycles
356 * is used if provided, otherwise the default value will be False. If the value is True,
357 * max_copy_movements will be ignored.
358 * \return tvm::transform::Pass
359 */
360tvm::transform::Pass CopyComputeReordering(Optional<Integer> max_copy_movements,
361 Optional<Bool> reorder_by_cycles) {
362 auto pass_func = [=](PrimFunc f, IRModule mod, tvm::transform::PassContext ctx) {
363 ICHECK(mod->GetGlobalVars().size() == 1 && mod->ContainGlobalVar("main"))
364 << "Expected a single primitive function called 'main'. Please run the "
365 "CopyComputeReordering "
366 "pass in conjunction with the LowerToTIR() pass.";
367
368 auto copy_movements = max_copy_movements.value_or(
369 ctx->GetConfig(kCopyComputeReorderingMaxCopyMovements, Integer(1)).value());
370 auto reorder = reorder_by_cycles.value_or(
371 ctx->GetConfig(kCopyComputeReorderingReorderByCycles, Bool(false)).value());
372 return CopyComputeReorderingMutator(copy_movements.IntValue(), reorder)(f);
373 };
374 return tvm::tir::transform::CreatePrimFuncPass(pass_func, 0,
375 "tir.contrib.ethos-u.CopyComputeReordering", {});
376}

379 .set_body_typed(CopyComputeReordering);

396/*!
397 * \brief This extractor collects information used by the MergeConstantsMutator
398 */
399class MergeConstantsInfoExtractor : public StmtExprVisitor {
400 public:
401 class Info {
402 public:
403 /*! A stack to store allocates as they are visited. */
404 std::vector<Allocate> allocates{};
405
406 /*! A list that contains in the i-th position the write buffer of the i-th statement
407 * if that statement is a copy to a buffer with global scope */
408 std::vector<Optional<Buffer>> copy_write_buffers{};
409
410 /*! Maps a copy's write buffer to an index representing the
411 * new buffer and an offset in that buffer */
412 std::unordered_map<const BufferNode*, std::pair<int /* new buffer index */, int /* offset */>>
413 old_to_new_write_buffer{};
414
415 /*! Maps an index representing a new buffer to the length of that buffer */
416 std::unordered_map<int /* new buffer index */, int /* length */> new_buffers_length{};
417
418 /*! Maps an index representing a new buffer to the cycless needed to copy that buffer */
419 std::unordered_map<int /* new buffer index */, int64_t> cycless{};
420 };
421
422 Info operator()(PrimFunc main_func) {
423 this->VisitStmt(main_func->body);
424 return std::move(_info);
425 }
426
427 private:
428 /*! The information collected by this extractor */
429 Info _info{};
430
431 void VisitStmt_(const AllocateNode* op) override {
432 _info.allocates.push_back(GetRef<Allocate>(op));
433 VisitStmt(op->body);
434 }
435
436 void VisitStmt_(const SeqStmtNode* op) override {
437 if (op->size() <= 1) {
438 StmtExprVisitor::VisitStmt_(op);
439 return;
440 }
441
442 auto seq_stmt{GetRef<SeqStmt>(op)};
443 for (size_t i = 0; i < seq_stmt.size(); ++i) {
444 Stmt stmt{seq_stmt[i]};
445 switch (GetStmtType(stmt)) {
446 case StmtType::global_copy: {
447 Buffer write_buffer{GetCopyWriteBuffer(stmt)};
448 _info.copy_write_buffers.push_back(write_buffer);
449 _info.old_to_new_write_buffer[write_buffer.as<BufferNode>()] = std::make_pair(-1, -1);
450 break;
451 }
452 case StmtType::local_copy: {
453 _info.copy_write_buffers.push_back(Optional<Buffer>{});
454 break;
455 }
456 case StmtType::compute: {
457 _info.copy_write_buffers.push_back(Optional<Buffer>{});
458 std::vector<Buffer> buffers{GetCopiedBuffersUsedByStmt(stmt)};
459 if (buffers.empty()) {
460 continue;
461 }
462 _info.new_buffers_length[i] = 0;
463 for (Buffer buffer : buffers) {
464 for (size_t j{i - 1}; j >= 0; --j) {
465 if (_info.copy_write_buffers[j] == buffer) {
466 _info.old_to_new_write_buffer[buffer.as<BufferNode>()] =
467 std::make_pair(i, _info.new_buffers_length[i]);
468 _info.new_buffers_length[i] += GetCopyLength(seq_stmt[j]);
469 _info.cycless[i] += GetStmtCycles(seq_stmt[j]);
470 break;
471 }
472 }
473 }
474 break;
475 }
476 }
477 }
478 }
479
480 /*! Get all buffers written by copies and used by a given statement */
481 std::vector<Buffer> GetCopiedBuffersUsedByStmt(const Stmt& stmt) {
482 std::vector<Buffer> buffers{};
483 for (PrimExpr arg : GetStmtArgs(stmt)) {
484 if (auto buffer_load = arg.as<BufferLoadNode>()) {
485 Buffer buffer{buffer_load->buffer};
486 // Check if the buffer has already been added
487 if (std::find(buffers.begin(), buffers.end(), buffer) == buffers.end()) {
488 // Check if the buffer is copied
489 if (_info.old_to_new_write_buffer.count(buffer.as<BufferNode>())) {
490 buffers.push_back(buffer);
491 }
492 }
493 }
494 }
495 return buffers;
496 }
497};

499/*!
500 * \brief This mutator looks for the constants used by each compute operator
501 * and merges them into a single buffer.
502 * Constants written to a buffer with local scope are not merged.
503 */
504class MergeConstantsMutator : public StmtExprMutator {
505 public:
506 explicit MergeConstantsMutator(MergeConstantsInfoExtractor::Info info) : _info{std::move(info)} {}
507
508 PrimFunc operator()(PrimFunc main_func, const Map<IntImm, runtime::NDArray>& const_dict) {
509 // Rewrite
510 Stmt new_body = RewritePrimFuncBody(main_func->body);
511 std::unordered_set<const VarNode*> params_to_delete{};
512 Map<Var, Buffer> new_buffer_map{MakeNewBufferMap(main_func->buffer_map, &params_to_delete)};
513 Array<Var> new_params{MakeNewParams(main_func->params, params_to_delete)};
514
515 // Make the new const dict
516 Array<Array<IntImm>> args_to_merge{GetArgsToMerge(main_func->buffer_map, main_func->params)};
517 Array<Array<IntImm>> buffers_to_merge{
518 GetArgsToMergeWithoutArgsNotInConstDict(args_to_merge, const_dict)};
519 Map<IntImm, runtime::NDArray> new_const_dict{MakeNewConstDict(buffers_to_merge, const_dict)};
520
521 // Make the new prim func
522 auto prim_func_node{main_func.CopyOnWrite()};
523 prim_func_node->body = std::move(new_body);
524 prim_func_node->buffer_map = std::move(new_buffer_map);
525 prim_func_node->params = std::move(new_params);
526 prim_func_node->preflattened_buffer_map = {};
527 PrimFunc f{GetRef<PrimFunc>(prim_func_node)};
528
529 // Add the new const dict as an attribute
530 f = WithAttr(std::move(f), "ethos-u.const_dict", new_const_dict);
531
532 return f;
533 }
534
535 private:
536 /*! The information collected by the MergeConstantsInfoExtractor */
537 MergeConstantsInfoExtractor::Info _info;
538
539 /*! Maps an index representing a new buffer to the new buffer */
540 std::unordered_map<int /* new buffer index */, Buffer> new_buffers{};
541
542 /*! Maps a copy's read buffer to the new copy's read buffer */
543 std::unordered_map<const BufferNode*, Buffer> old_to_new_read_buffers{};
544
545 /*! Maps an index representing a new buffer to the list of buffers to be merged in the new buffer
546 */
547 std::unordered_map<int /* new buffer index */, std::vector<Buffer>> buffers_to_merge{};
548
549 /*! A set of buffers to delete */
550 std::unordered_set<const BufferNode*> buffers_to_delete{};
551
552 Stmt RewritePrimFuncBody(Stmt body) {
553 std::unordered_map<const VarNode*, Allocate> var_to_allocate{};
554
555 // Rewrite old allocates
556 std::unordered_set<const VarNode*> buffer_vars{GetVarsForWrittenCopyBuffers()};
557 for (auto it{_info.allocates.rbegin()}; it != _info.allocates.rend(); ++it) {
558 Allocate alloc{*it};
559 var_to_allocate[alloc->buffer_var.get()] = alloc;
560 if (buffer_vars.count(alloc->buffer_var.as<VarNode>()) == 0) {
561 body = Allocate(alloc->buffer_var, alloc->dtype, alloc->extents, alloc->condition, body,
562 alloc->annotations, alloc->span);
563 }
564 }
565
566 // Rewrite new allocates
567 for (auto it{_info.copy_write_buffers.rbegin()}; it != _info.copy_write_buffers.rend(); ++it) {
568 if (Optional<Buffer> buffer_opt = *it) {
569 Buffer old_write_buffer{buffer_opt.value()};
570 int new_buffer_index{
571 _info.old_to_new_write_buffer[old_write_buffer.as<BufferNode>()].first};
572
573 // Check if the allocate has already been created
574 if (new_buffers.count(new_buffer_index) == 0) {
575 BufferNode* new_buffer{old_write_buffer.CopyOnWrite()};
576 new_buffer->shape = {_info.new_buffers_length[new_buffer_index]};
577
578 new_buffers[new_buffer_index] = GetRef<Buffer>(new_buffer);
579
580 Allocate old_allocate{var_to_allocate[old_write_buffer->data.get()]};
581 body = Allocate(new_buffer->data, new_buffer->dtype, new_buffer->shape, tir::const_true(),
582 body, old_allocate->annotations, old_allocate->span);
583 }
584 }
585 }
586
587 // Rewrite operators
588 return this->VisitStmt(body);
589 }
590
591 Stmt VisitStmt_(const AllocateNode* op) override {
592 auto allocate{CopyOnWrite(op)};
593 allocate->body = this->VisitStmt(op->body);
594 return Stmt(allocate);
595 }
596
597 Stmt VisitStmt_(const SeqStmtNode* op) override {
598 if (op->size() <= 1) {
599 return StmtExprMutator::VisitStmt_(op);
600 }
601
602 Array<Stmt> new_seq{};
603 SeqStmt seq_stmt{GetRef<SeqStmt>(op)};
604 for (size_t i{0}; i < seq_stmt.size(); ++i) {
605 Stmt stmt{seq_stmt[i]};
606
607 switch (GetStmtType(stmt)) {
608 case StmtType::global_copy: {
609 Buffer old_write_buffer{_info.copy_write_buffers[i].value()};
610 std::pair<int, int> pair{
611 _info.old_to_new_write_buffer[old_write_buffer.as<BufferNode>()]};
612 int new_buffer_index{pair.first};
613 int new_buffer_offset{pair.second};
614 UpdateBuffersToMergeAndDelete(stmt, new_buffer_index, new_buffer_offset);
615
616 if (!IsCopyToBeDeleted(new_buffer_offset)) {
617 Optional<PrimExpr> cycless{GetMergedCycles(new_buffer_index)};
618 new_seq.push_back(MakeNewStmt(
619 stmt, MakeNewCopyArgs(stmt, old_write_buffer, new_buffer_index), cycless));
620 }
621 break;
622 }
623 case StmtType::local_copy: {
624 new_seq.push_back(stmt);
625 break;
626 }
627 case StmtType::compute: {
628 new_seq.push_back(MakeNewStmt(stmt, MakeNewComputeArgs(stmt)));
629 break;
630 }
631 }
632 }
633 return SeqStmt(new_seq, op->span);
634 }

636 /*! Returns the variables of the buffers written by copies */
637 std::unordered_set<const VarNode*> GetVarsForWrittenCopyBuffers() {
638 std::unordered_set<const VarNode*> buffer_vars{};
639 std::transform(_info.old_to_new_write_buffer.begin(), _info.old_to_new_write_buffer.end(),
640 std::inserter(buffer_vars, buffer_vars.begin()),
641 [](std::pair<const BufferNode*, std::pair<int, int>> pair) -> const VarNode* {
642 return pair.first->data.as<VarNode>();
643 });
644 return buffer_vars;
645 }

647 /*! Returns the cycles of the new buffer at the given index */
648 Optional<PrimExpr> GetMergedCycles(int new_buffer_index) {
649 auto it = _info.cycless.find(new_buffer_index);
650 if (it != _info.cycless.end()) {
651 return Integer(it->second);

653 return Optional<PrimExpr>{};

657 bool IsCopyToBeDeleted(int new_buffer_offset) { return new_buffer_offset > 0; }

659 Array<PrimExpr> MakeNewCopyArgs(const Stmt& stmt, const Buffer& old_write_buffer,
660 int new_buffer_index) {
661 Array<PrimExpr> args{GetStmtArgs(stmt)};
662 int new_length{_info.new_buffers_length[new_buffer_index]};
663
664 Array<PrimExpr> new_args{};
665 for (size_t i = 0; i < args.size(); ++i) {
666 switch (i) {
667 case 1: /* read_address */ {
668 auto buffer_load = args[1].as<BufferLoadNode>();
669 Buffer buffer{buffer_load->buffer};
670 Buffer new_buffer{buffer->data,
671 buffer->dtype,
672 {new_length},
673 buffer->strides,
674 buffer->elem_offset,
675 buffer->name,
676 buffer->data_alignment,
677 buffer->offset_factor,
678 buffer->buffer_type,
679 buffer->axis_separators,
680 buffer->span};
681 old_to_new_read_buffers[buffer.as<BufferNode>()] = new_buffer;
682 new_args.push_back(BufferLoad(new_buffer, buffer_load->indices, buffer_load->span));
683 break;
684 }
685 case 2: /* length */ {
686 new_args.push_back(new_length);
687 break;
688 }
689 case 3: /* write_address */ {
690 new_args.push_back(MakeNewBufferLoad(old_write_buffer, 0, true).value());
691 break;
692 }
693 default:
694 new_args.push_back(args[i]);
695 break;
696 }
697 }
698 return new_args;
699 }

701 Array<PrimExpr> MakeNewComputeArgs(const Stmt& stmt) {
702 Array<PrimExpr> args{GetStmtArgs(stmt)};
703 Array<PrimExpr> new_args{};
704 for (size_t i = 0; i < args.size(); ++i) {
705 if (auto buffer_load = args[i].as<BufferLoadNode>()) {
706 BufferLoad new_buffer_load{
707 MakeNewBufferLoad(buffer_load->buffer, buffer_load->indices[0], false)
708 .value_or(GetRef<BufferLoad>(buffer_load))};
709 new_args.push_back(new_buffer_load);
710 } else {
711 new_args.push_back(args[i]);
712 }
713 }
714 return new_args;
715 }

717 Stmt MakeNewStmt(const Stmt& stmt, const Array<PrimExpr>& new_args,
718 Optional<PrimExpr> cycless = Optional<PrimExpr>{}) {
719 auto attr{stmt.as<AttrStmtNode>()};
720 Stmt eval_stmt{attr ? attr->body : stmt};
721 auto eval{eval_stmt.as<EvaluateNode>()};
722 ICHECK(eval) << "Expected statement to be an evaluate node, but was "
723 << eval_stmt->GetTypeKey();
724 auto call{eval->value.as<CallNode>()};
725 ICHECK(call) << "Expected expression to be a call node, but was " << eval->value->GetTypeKey();
726
727 Call new_call{call->dtype, call->op, new_args, call->span};
728 Evaluate new_eval{new_call, eval->span};
729
730 if (attr) {
731 ICHECK(attr->attr_key == "pragma_compute_cycles_hint");
732 PrimExpr value = cycless.value_or(attr->value);
733 return AttrStmt{attr->node, attr->attr_key, value, new_eval, attr->span};
734 } else {
735 return std::move(new_eval);
736 }
737 }

739 Optional<BufferLoad> MakeNewBufferLoad(const Buffer& write_buffer, const PrimExpr& old_index,
740 bool only_old_index) {
741 auto it = _info.old_to_new_write_buffer.find(write_buffer.as<BufferNode>());
742 if (it != _info.old_to_new_write_buffer.end()) {
743 std::pair<int, int> pair{it->second};
744 int new_buffer_index{pair.first};
745 PrimExpr new_index{only_old_index ? old_index : (pair.second + old_index)};
746 return BufferLoad{new_buffers[new_buffer_index], {new_index}};
747 }
748 return Optional<BufferLoad>{};
749 }

751 Map<tir::Var, Buffer> MakeNewBufferMap(const Map<tir::Var, Buffer>& buffer_map,
752 std::unordered_set<const VarNode*>* params_to_delete) {
753 Map<tir::Var, Buffer> new_buffer_map{};
754 for (std::pair<Var, Buffer> pair : buffer_map) {
755 Var var{pair.first};
756 Buffer buffer{pair.second};
757
758 if (buffers_to_delete.count(buffer.as<BufferNode>()) == 1) {
759 params_to_delete->insert(var.as<VarNode>());
760 } else if (old_to_new_read_buffers.count(buffer.as<BufferNode>()) == 1) {
761 new_buffer_map.Set(var, old_to_new_read_buffers[buffer.as<BufferNode>()]);
762 } else {
763 new_buffer_map.Set(var, buffer);
764 }
765 }
766 return new_buffer_map;

780 void UpdateBuffersToMergeAndDelete(const Stmt& stmt, int new_buffer_index,
781 int new_buffer_offset) {
782 Array<PrimExpr> args{GetStmtArgs(stmt)};
783 Buffer read_buffer{GetCopyReadBuffer(stmt)};
784

786 buffers_to_merge[new_buffer_index] = std::vector<Buffer>{read_buffer};
787 } else {
788 buffers_to_merge[new_buffer_index].push_back(read_buffer);
789 }

792 buffers_to_delete.insert(read_buffer.as<BufferNode>());
793 }

796 /*! Returns an array whose elements are the indices of the function arguments to be merged.
797 * Example: if a function has three arguments and the second and the third ones must
798 * be merged then the array is: [[0], [1, 2], [3]] */
799 Array<Array<IntImm>> GetArgsToMerge(const Map<Var, Buffer>& buffer_map,
800 const Array<Var>& params) {
801 std::unordered_map<const BufferNode*, Var> buffer_to_var{};
802 for (std::pair<Var, Buffer> var_buffer : buffer_map) {
803 buffer_to_var[var_buffer.second.as<BufferNode>()] = var_buffer.first;
804 }
805
806 std::unordered_map<const VarNode*, int> var_to_index{};
807 for (int i = 0; i < static_cast<int>(params.size()); ++i) {
808 var_to_index[params[i].as<VarNode>()] = i;
809 }
810
811 std::vector<Array<IntImm>> vector{};
812 for (std::pair<int, std::vector<Buffer>> index_vector : buffers_to_merge) {
813 std::vector<IntImm> indices{};
814 for (Buffer buffer : index_vector.second) {
815 const VarNode* var{buffer_to_var[buffer.as<BufferNode>()].as<VarNode>()};
816 IntImm index{DataType::Int(64), var_to_index[var]};
817 var_to_index.erase(var);
818 auto it = std::find_if(indices.begin(), indices.end(),
819 [&](IntImm value) { return value->value == index->value; });
820 if (it == indices.end()) {
821 indices.push_back(index);
822 }
823 }
824 vector.push_back(Array<IntImm>{indices});
825 }
826
827 for (std::pair<const VarNode*, int> var_index : var_to_index) {
828 vector.push_back(Array<IntImm>{IntImm(DataType::Int(64), var_index.second)});
829 }
830 std::sort(vector.begin(), vector.end(),
831 [](Array<IntImm> a, Array<IntImm> b) { return a[0]->value < b[0]->value; });
832 return vector;
833 }

835 Array<Array<IntImm>> GetArgsToMergeWithoutArgsNotInConstDict(
836 const Array<Array<IntImm>>& args_to_merge, const Map<IntImm, runtime::NDArray>& const_dict) {
837 Array<Array<IntImm>> new_args_to_merge{};
838 for (Array<IntImm> args : args_to_merge) {
839 IntImm key{args[0]};

841 [&](std::pair<tvm::IntImm, runtime::NDArray> pair) {
842 return pair.first->value == key->value;
843 });
844 if (it != const_dict.end()) {
845 new_args_to_merge.push_back(args);
846 }
847 }
848 return new_args_to_merge;

851 Map<IntImm, runtime::NDArray> MakeNewConstDict(const Array<Array<IntImm>>& args_to_merge,
852 Map<IntImm, runtime::NDArray> const_dict) {
853 Map<IntImm, runtime::NDArray> new_const_dict{};
854 if (args_to_merge.size() == 0) {
855 return new_const_dict;
856 }
857
858 int64_t key = args_to_merge[0][0]->value;
859 for (Array<IntImm> args : args_to_merge) {
860 int64_t size = 0;
861 for (IntImm arg : args) {
862 auto it = std::find_if(const_dict.begin(), const_dict.end(),
863 [&](auto pair) { return pair.first->value == arg->value; });
864 runtime::NDArray arg_constant{(*it).second};
865 size += runtime::GetDataSize(*arg_constant.operator->());
866 }
867
868 runtime::NDArray constant = runtime::NDArray::Empty({size}, DataType::UInt(8), {kDLCPU, 0});
869
870 size_t offset = 0;
871 for (IntImm arg : args) {
872 auto it = std::find_if(const_dict.begin(), const_dict.end(),
873 [&](auto pair) { return pair.first->value == arg->value; });
874 runtime::NDArray arg_constant{(*it).second};
875 size_t nbytes = runtime::GetDataSize(*arg_constant.operator->());
876 arg_constant.CopyToBytes(static_cast<uint8_t*>(constant->data) + offset, nbytes);
877 offset += nbytes;
878 }
879 new_const_dict.Set(IntImm(DataType::Int(64), key), constant);
880 key += 1;
881 }
882 return new_const_dict;
883 }

886/*!
887 * \brief This pass looks for the constants used by each compute operator
888 * and merges them into a single buffer.
889 * Constants written to a buffer with local scope are not merged.
890 * \return tvm::transform::Pass
891 */
892tvm::transform::Pass MergeConstants() {
893 auto pass_func = [=](PrimFunc f, IRModule mod, tvm::transform::PassContext ctx) {
894 ICHECK(mod->GetGlobalVars().size() == 1 && mod->ContainGlobalVar("main"))
895 << "Expected a single primitive function called 'main'. Please run the "
896 "MergeConstants pass in conjunction with the LowerToTIR() pass.";
897 Optional<Map<IntImm, runtime::NDArray>> const_dict{
898 f->attrs.GetAttr("ethos-u.const_dict", Optional<Map<IntImm, runtime::NDArray>>{})};
899 ICHECK(const_dict) << "Expected a ethos-u.const_dict attribute";
900
901 MergeConstantsInfoExtractor::Info info{MergeConstantsInfoExtractor()(f)};
902 f = RemoveAllocatesMutator()(f);
903 return MergeConstantsMutator(info)(f, const_dict.value());
904 };
905 return tvm::tir::transform::CreatePrimFuncPass(pass_func, 0, "tir.contrib.ethos-u.MergeConstants",
906 {});
907}
