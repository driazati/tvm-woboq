41/*! \brief Helper function to check whether the given var is in function parameter list. */
42inline bool IsParam(const PrimFunc& func, const Var& param) {
43 return std::any_of(func->params.begin(), func->params.end(),
44 [&](const Var& var) { return var.same_as(param); });
45}

49// Try fold constants if op's child get specialized to constant.
50#define DEFINE_SPECIALIZER_BINARY_OP_MUTATE(BinaryNode, BinaryFunc) \
51 PrimExpr VisitExpr_(const BinaryNode* op) final { \
52 PrimExpr a = VisitExpr(op->a); \
53 PrimExpr b = VisitExpr(op->b); \
54 if (a.same_as(op->a) && b.same_as(op->b)) { \
55 return GetRef<PrimExpr>(op); \
56 } else { \
57 return BinaryFunc(a, b); \
58 } \
59 }
60#define DEFINE_SPECIALIZER_UNARY_OP_MUTATE(UnaryNode, UnaryFunc) \
61 PrimExpr VisitExpr_(const UnaryNode* op) final { \
62 PrimExpr a = VisitExpr(op->a); \
63 if (a.same_as(op->a)) { \
64 return GetRef<PrimExpr>(op); \
65 } else { \
66 return UnaryFunc(a); \
67 } \
68 }

70/*! \brief Mutator to specialize function and remove const parameters */
71class PrimFuncSpecializer : public StmtExprMutator {
72 public:
73 explicit PrimFuncSpecializer(const VarMap& var_map) : var_map_(var_map) {}
74
75 static PrimFunc Specialize(PrimFunc f, const VarMap& var_map) {
76 PrimFuncSpecializer specializer(var_map);
77 // Updating Buffer map
78 Map<Var, Buffer> buffer_map;
79 bool buffer_map_updated = false;
80 for (const auto& it : f->buffer_map) {
81 const Var& var = it.first;
82 const Buffer& buffer = it.second;
83 Buffer new_buffer = specializer.MutateBuffer(buffer);
84 buffer_map.Set(var, new_buffer);
85 if (!new_buffer.same_as(buffer)) {
86 buffer_map_updated = true;
87 specializer.buffer_map_[buffer] = new_buffer;
88 }
89 }
90
91 // Updating parmeters
92 Array<Var> params;
93 bool param_updated = false;
94 for (const auto& var : f->params) {
95 // Remove parmeters which has been specialized.
96 if (var_map.find(var) == var_map.end()) {
97 params.push_back(var);
98 } else {
99 param_updated = true;
100 }
101 }
102
103 // Updating function body
104 Stmt body = specializer(f->body);
105
106 if (param_updated || buffer_map_updated || !f->body.same_as(body)) {
107 PrimFuncNode* f_ptr = f.CopyOnWrite();
108 f_ptr->params = std::move(params);
109 f_ptr->buffer_map = std::move(buffer_map);
110 f_ptr->body = std::move(body);
111 }
112 return f;
113 }
114
115 private:
116 Stmt VisitStmt_(const BlockNode* op) final {
117 // Step.0. Define buffer mappings which is allocated inside the block
118 Array<Buffer> alloc_buffers = op->alloc_buffers.Map(
119 std::bind(&PrimFuncSpecializer::MutateAllocBuffer, this, std::placeholders::_1));
120
121 // Step.1. Recursively visit block body
122 Stmt stmt = StmtExprMutator::VisitStmt_(op);
123 op = stmt.as<BlockNode>();
124 ICHECK(op != nullptr);
125
126 Array<BufferRegion> reads = op->reads.Map(
127 std::bind(&PrimFuncSpecializer::MutateBufferRegion, this, std::placeholders::_1));
128 Array<BufferRegion> writes = op->writes.Map(
129 std::bind(&PrimFuncSpecializer::MutateBufferRegion, this, std::placeholders::_1));
130
131 if (alloc_buffers.same_as(op->alloc_buffers) && reads.same_as(op->reads)) {
132 return GetRef<Block>(op);
133 } else {
134 ObjectPtr<BlockNode> n = CopyOnWrite(op);
135 n->alloc_buffers = std::move(alloc_buffers);
136 n->reads = std::move(reads);
137 n->writes = std::move(writes);
138 return Stmt(n);
139 }
140 }
141
142 Stmt VisitStmt_(const BufferStoreNode* op) final {
143 Stmt stmt = StmtExprMutator::VisitStmt_(op);
144 op = stmt.as<BufferStoreNode>();
145 ICHECK(op != nullptr);
146 auto it = buffer_map_.find(op->buffer);
147 if (it == buffer_map_.end()) {
148 return GetRef<BufferStore>(op);
149 } else {
150 auto n = CopyOnWrite(op);
151 n->buffer = it->second;
152 return Stmt(n);
153 }
154 }
155
156 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
157 PrimExpr expr = StmtExprMutator::VisitExpr_(op);
158 op = expr.as<BufferLoadNode>();
159 ICHECK(op != nullptr);
160 auto it = buffer_map_.find(op->buffer);
161 if (it == buffer_map_.end()) {
162 return GetRef<BufferLoad>(op);
163 } else {
164 auto n = make_object<BufferLoadNode>(*op);
165 n->buffer = it->second;
166 return PrimExpr(n);
167 }
168 }
169
170 PrimExpr VisitExpr_(const VarNode* op) final {
171 auto it = var_map_.find(GetRef<Var>(op));
172 if (it == var_map_.end()) {
173 return GetRef<PrimExpr>(op);
174 } else {
175 return it->second;
176 }
177 }
178
179 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(AddNode, add);
180 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(SubNode, sub);
181 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(MulNode, mul);
182 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(DivNode, div);
183 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(ModNode, truncmod);
184 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(FloorDivNode, floordiv);
185 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(FloorModNode, floormod);
186 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(MaxNode, max);
187 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(MinNode, min);
188 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(EQNode, equal);
189 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(NENode, not_equal);
190 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(LTNode, less);
191 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(LENode, less_equal);
192 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(GTNode, greater);
193 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(GENode, greater_equal);
194 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(AndNode, logical_and);
195 DEFINE_SPECIALIZER_BINARY_OP_MUTATE(OrNode, logical_or);
196 DEFINE_SPECIALIZER_UNARY_OP_MUTATE(NotNode, logical_not);
197
198 private:
199 Buffer MutateBuffer(const Buffer& buffer) {
200 Array<PrimExpr> shape = buffer->shape.Map([this](const PrimExpr& e) { return VisitExpr(e); });
201 Array<PrimExpr> strides =
202 buffer->strides.Map([this](const PrimExpr& e) { return VisitExpr(e); });
203
204 PrimExpr elem_offset = VisitExpr(buffer->elem_offset);
205
206 if (buffer->elem_offset.same_as(elem_offset) && buffer->shape.same_as(shape) &&
207 buffer->strides.same_as(strides)) {
208 return buffer;
209 } else {
210 auto n = make_object<BufferNode>(*buffer.get());
211 n->elem_offset = std::move(elem_offset);
212 n->shape = std::move(shape);
213 n->strides = std::move(strides);
214 return Buffer(n);
215 }
216 }
217
218 Range MutateRange(const Range& range) {
219 PrimExpr min = this->VisitExpr(range->min);
220 PrimExpr extent = this->VisitExpr(range->extent);
221 if (min.same_as(range->min) && extent.same_as(range->extent)) {
222 return range;
223 } else {
224 return Range::FromMinExtent(std::move(min), std::move(extent));
225 }
226 }
227
228 Buffer MutateAllocBuffer(const Buffer& alloc_buf) {
229 Buffer buf = MutateBuffer(alloc_buf);
230 if (buf.same_as(alloc_buf)) {
231 return alloc_buf;
232 } else {
233 ICHECK(buffer_map_.find(alloc_buf) == buffer_map_.end());
234 buffer_map_[alloc_buf] = buf;
235 return buf;
236 }
237 }
238
239 BufferRegion MutateBufferRegion(const BufferRegion& buffer_region) {
240 auto it = buffer_map_.find(buffer_region->buffer);
241 Array<Range> region = buffer_region->region.Map(
242 std::bind(&PrimFuncSpecializer::MutateRange, this, std::placeholders::_1));
243 if (it == buffer_map_.end() && region.same_as(buffer_region->region)) {
244 return buffer_region;
245 } else {
246 return BufferRegion(it->second, std::move(region));
247 }
248 }
249
250 private:
251 /*! \brief The vars to be substitute and their values */
252 const VarMap& var_map_;
253 /*! \brief map from old buffer to mutated buffer */
254 std::unordered_map<Buffer, Buffer, ObjectPtrHash, ObjectPtrEqual> buffer_map_;
255};

257/*!
258 * \brief Update Specialize var map with buffer matching.
259 * \param func The function to be specialized.
260 * \param param The given function parameter
261 * \param specific_buf The matching buffer.
262 * \param var_map The var mapping to be updated.
263 * \note This function will match target buffer's shape, strides and element_offset
264 * For example, we define a buffer in PrimFunc:
265 * A = T.match_buffer(a, [m, n])
266 *
267 * Then we match it with a buffer B = tir.decl_buffer((8, 16))
268 *
269 * It means we have two var mappings here: m = 8 and n = 16
270 *
271 * If the buffer signature is not a Var, the mapping will fail.
272 * e.g. A = T.match_buffer(a, [m * 2, n + 1])
273 */
274void UpdateSpecializeVarMap(const PrimFunc& func, const Var& param, const Buffer& specific_buf,
275 VarMap* var_map) {
276 // preliminaries
277 tir::ExprDeepEqual equal;
278
279 auto it = func->buffer_map.find(param);
280 CHECK(it != func->buffer_map.end())
281 << "ValueError: specialize expects param to be in PrimFunc's buffer_map";
282 const Buffer& buf_to_specialize = (*it).second;
283
284 // build var mapping using specific_buf's parameters
285 auto build_var_mapping = [&](const PrimExpr& new_expr, const PrimExpr& old_expr) {
286 if (!equal(new_expr, old_expr)) {
287 CHECK(old_expr->IsInstance<VarNode>())
288 << "TypeError: The signature of target buffer exprected an independent Var, but got "
289 << old_expr << ".";
290 const Var& var = Downcast<Var>(old_expr);
291 auto it = var_map->find(var);
292 if (it != var_map->end()) {
293 CHECK(equal(it->second, new_expr))
294 << "ValueError: The assigned value of var " << var << " mismatched. " << it->second
295 << " vs. " << new_expr << ".";
296 } else {
297 (*var_map)[var] = new_expr;
298 }
299 }
300 };
301
302 // Check buffer dimensions
303 CHECK(specific_buf->shape.size() == buf_to_specialize->shape.size())
304 << "ValueError: The buffer dimensions mismatched" << buf_to_specialize->shape.size()
305 << " vs. " << specific_buf->shape.size() << ".";
306
307 CHECK(specific_buf->strides.size() == buf_to_specialize->strides.size())
308 << "ValueError: The buffer strides dimensions mismatched" << buf_to_specialize->strides.size()
309 << " vs. " << specific_buf->strides.size() << ".";
310
311 // Updating var mapping using specific_expr
312 for (size_t i = 0; i < specific_buf->shape.size(); ++i) {
313 build_var_mapping(specific_buf->shape[i], buf_to_specialize->shape[i]);
314 }
315 for (size_t i = 0; i < specific_buf->strides.size(); ++i) {
316 build_var_mapping(specific_buf->strides[i], buf_to_specialize->strides[i]);
317 }
318 build_var_mapping(specific_buf->elem_offset, buf_to_specialize->elem_offset);
319
320 // Check data_alignment and offset_factor.
321 // These two signatures are int, so we do not need map them.
322 CHECK_EQ(specific_buf->data_alignment, buf_to_specialize->data_alignment)
323 << "ValueError: The buffer data_alignment mismatched" << buf_to_specialize->data_alignment
324 << " vs. " << specific_buf->data_alignment << ".";
325
326 CHECK_EQ(specific_buf->offset_factor, buf_to_specialize->offset_factor)
327 << "ValueError: The buffer offset_factor mismatched" << buf_to_specialize->offset_factor
328 << " vs. " << specific_buf->offset_factor << ".";
329}

331/*!
332 * \brief Update Specialize var map with parameter value.
333 * \param func The function to be specialized.
334 * \param param The given function parameter
335 * \param specific_expr The parameter value.
336 * \param var_map The var mapping to be updated.
337 */
338void UpdateSpecializeVarMap(const PrimFunc& func, const Var& param, const PrimExpr& specific_expr,
339 VarMap* var_map) {
340 // check param is in PrimFunc's parameters
341 CHECK(IsParam(func, param)) << "ValueError: Specialize expects param to be in PrimFunc's params";
342 // specialize a param not in buffer_map
343 CHECK_EQ(func->buffer_map.count(param), 0)
344 << "ValueError: Specialize expects param to not be in PrimFunc's buffer_map";
345 // build var mapping using specific_expr
346 (*var_map)[param] = specific_expr;
347}
348
349/**************** Implementation ****************/
350
351PrimFunc Specialize(PrimFunc func, const Map<Var, ObjectRef>& param_map) {
352 VarMap var_map;
353 for (const auto& kv : param_map) {
354 const Var& param = kv.first;
355 const ObjectRef& instance = kv.second;
356 if (instance->IsInstance<BufferNode>()) {
357 UpdateSpecializeVarMap(func, param, Downcast<Buffer>(instance), &var_map);
358 } else if (instance->IsInstance<PrimExprNode>()) {
359 UpdateSpecializeVarMap(func, param, Downcast<PrimExpr>(instance), &var_map);
360 } else {
361 CHECK(instance.defined()) << "Specialize instance is not defined for param " << param;
362 LOG(FATAL) << "TypeError: specialize expected instance to be Buffer or PrimExpr, but got "
363 << instance->GetTypeKey();
364 }
365 }
366 return PrimFuncSpecializer::Specialize(func, std::move(var_map));
367}
