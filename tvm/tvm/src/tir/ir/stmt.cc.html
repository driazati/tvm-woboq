34// LetStmt
35LetStmt::LetStmt(Var var, PrimExpr value, Stmt body, Span span) {
36 ICHECK(value.defined());
37 ICHECK(body.defined());
38 auto vdtype = value.dtype();
39 // It is still valid to bind a pointer type
40 // var to a value that is of type handle.
41 if (var->type_annotation.as<PointerTypeNode>()) {
42 ICHECK(vdtype.is_handle());
43 } else {
44 ICHECK_EQ(value.dtype(), var.dtype());
45 }
46
47 ObjectPtr<LetStmtNode> node = make_object<LetStmtNode>();
48 node->var = std::move(var);
49 node->value = std::move(value);
50 node->body = std::move(body);
51 node->span = std::move(span);
52 data_ = std::move(node);
53}

140// For
141For::For(Var loop_var, PrimExpr min, PrimExpr extent, ForKind kind, Stmt body,
142 Optional<IterVar> thread_binding, Map<String, ObjectRef> annotations, Span span) {
143 ICHECK(min.defined());
144 ICHECK(extent.defined());
145 ICHECK(min.dtype().is_scalar());
146 ICHECK(extent.dtype().is_scalar());
147 ICHECK(loop_var.dtype().is_scalar());
148 ICHECK(body.defined());
149
150 // When extent or min is an IntImm but has narrower dtype than loop_var, we directly promote them
151 // without raising errors.
152 auto try_promote_imm_dtype = [&](const PrimExpr& e) {
153 ICHECK(e.dtype().bits() <= loop_var.dtype().bits())
154 << " Loop variable's dtype (" << loop_var.dtype()
155 << ") is narrower than that of `min` or `extent` (" << e.dtype() << ")";
156 const IntImmNode* a = e.as<IntImmNode>();
157 if (a && e.dtype().bits() < loop_var.dtype().bits()) {
158 return make_const(loop_var.dtype(), a->value);
159 } else {
160 return e;
161 }
162 };
163
164 min = try_promote_imm_dtype(min);
165 extent = try_promote_imm_dtype(extent);
166
167 ICHECK(loop_var.dtype() == min.dtype()) << loop_var.dtype() << " vs " << min.dtype();
168 ICHECK(loop_var.dtype() == extent.dtype()) << loop_var.dtype() << " vs " << extent.dtype();
169
170 ObjectPtr<ForNode> node = make_object<ForNode>();
171 node->loop_var = std::move(loop_var);
172 node->min = std::move(min);
173 node->extent = std::move(extent);
174 node->kind = kind;
175 node->body = std::move(body);
176 node->thread_binding = std::move(thread_binding);
177 node->annotations = std::move(annotations);
178 node->span = std::move(span);
179 data_ = std::move(node);
180}
181
182TVM_REGISTER_GLOBAL("tir.For").set_body_typed(
183 [](Var loop_var, PrimExpr min, PrimExpr extent, int kind, Stmt body,
184 Optional<IterVar> thread_binding, Optional<Map<String, ObjectRef>> annotations, Span span) {
185 return For(loop_var, min, extent, static_cast<ForKind>(kind), body, thread_binding,
186 annotations.value_or(Map<String, ObjectRef>()), span);
187 });

189TVM_REGISTER_NODE_TYPE(ForNode);

191std::ostream& operator<<(std::ostream& out, ForKind type) { // NOLINT(*)
192 switch (type) {
193 case ForKind::kSerial:
194 out << "for";
195 break;
196 case ForKind::kParallel:
197 out << "parallel";
198 break;
199 case ForKind::kUnrolled:
200 out << "unrolled";
201 break;
202 case ForKind::kVectorized:
203 out << "vectorized";
204 break;
205 case ForKind::kThreadBinding:
206 out << "launch_thread";
207 break;
208 }
209 return out;
210}

262// Store
263Store::Store(Var buffer_var, PrimExpr value, PrimExpr index, PrimExpr predicate, Span span) {
264 LOG(FATAL) << "Unexpected use of deprecated Store node for buffer " << buffer_var->name_hint
265 << ". Use BufferStore instead.";
266 ICHECK(value.defined());
267 ICHECK(index.defined());
268 ICHECK(predicate.defined());
269
270 // Assume that the array elements have 1 lane, unless a type
271 // annotation tells us otherwise.
272 int element_lanes = 1;
273 auto pointer_type = tir::GetPointerType(buffer_var->type_annotation);
274 if (pointer_type.first) {
275 // Currently cannot check element type of array, see Load::Load
276 // for details.
277
278 // TODO(Lunderberg): Uncomment this check once it can be applied.
279 // See https://discuss.tvm.apache.org/t/pre-rfc-vectorized-tir-buffers/10615
280 // for discussion.
281
282 // ICHECK_EQ(value.dtype().element_of(), pointer_type.second.element_of())
283 // << "Type mismatch, cannot store type " << value.dtype() << " into buffer "
284 // << buffer_var->name_hint << " of type " << pointer_type.second;
285 element_lanes = pointer_type.second.lanes();
286 }
287
288 ICHECK((value.dtype().lanes() == element_lanes * index.dtype().lanes()) ||
289 (value.dtype().lanes() == index.dtype().lanes()));
290 ICHECK((value.dtype().lanes() == element_lanes * predicate.dtype().lanes()) ||
291 (value.dtype().lanes() == index.dtype().lanes()));
292
293 ObjectPtr<StoreNode> node = make_object<StoreNode>();
294 node->buffer_var = std::move(buffer_var);
295 node->value = std::move(value);
296 node->index = std::move(index);
297 node->predicate = std::move(predicate);
298 node->span = std::move(span);
299 data_ = std::move(node);
300}
301
302TVM_REGISTER_GLOBAL("tir.Store").set_body([](TVMArgs args, TVMRetValue* ret) {
303 PrimExpr value = args[1];
304 if (args.size() == 3) {
305 *ret = Store(args[0], value, args[2], const_true(value.dtype().lanes()), Span());
306 } else if (args.size() == 4) {
307 *ret = Store(args[0], value, args[2], args[3], Span());
308 } else {
309 *ret = Store(args[0], value, args[2], args[3], args[4]);
310 }
311});

376 }
377 ICHECK(body.defined());

392int64_t AllocateNode::ConstantAllocationSize(const Array<PrimExpr>& extents) {
393 int64_t result = 1;
394 for (size_t i = 0; i < extents.size(); ++i) {
395 if (const IntImmNode* int_size = extents[i].as<IntImmNode>()) {
396 result *= int_size->value;
397 if (result > std::numeric_limits<int64_t>::max()) {
398 return 0;
399 }
400 } else {
401 return 0;
402 }
403 }
404 return static_cast<int64_t>(result);
405}

435// Const
436// The constructor to create a IRNode with constant data
437// depending on the type of ObjectRef, it will either
438// create AllocateConstNode with irmod_storage_idx or data
439AllocateConst::AllocateConst(Var buffer_var, DataType dtype, Array<PrimExpr> extents,
440 ObjectRef data_or_idx, Stmt body, Map<String, ObjectRef> annotations,
441 Span span) {
442 ICHECK(IsPointerType(buffer_var->type_annotation, dtype))
443 << "The allocated data type (" << dtype
444 << ") does not match the type annotation of the buffer " << buffer_var << " ("
445 << buffer_var->type_annotation
446 << "). The data type should be an element of the pointer type.";
447
448 for (size_t i = 0; i < extents.size(); ++i) {
449 ICHECK(extents[i].defined());
450 ICHECK(extents[i].dtype().is_scalar());
451 }
452 ICHECK(body.defined());
453 ICHECK(data_or_idx.defined());
454
455 ObjectPtr<AllocateConstNode> node = make_object<AllocateConstNode>();
456 node->buffer_var = std::move(buffer_var);
457 node->dtype = dtype;
458 node->extents = std::move(extents);
459 node->body = std::move(body);
460 node->annotations = annotations;
461 node->span = std::move(span);
462 if (data_or_idx->IsInstance<runtime::NDArray::ContainerType>()) {
463 node->data = Optional<tvm::runtime::NDArray>(Downcast<runtime::NDArray>(data_or_idx));
464 node->irmod_storage_idx = Optional<Integer>();
465 } else if (data_or_idx->IsInstance<IntImmNode>()) {
466 node->data = Optional<tvm::runtime::NDArray>();
467 node->irmod_storage_idx = Optional<Integer>(Downcast<Integer>(data_or_idx));
468 } else {
469 LOG(FATAL) << "Data type not supported: " << data_or_idx->GetTypeKey();
470 }
471 data_ = std::move(node);
472}

719BufferStore::BufferStore(Buffer buffer, PrimExpr value, Array<PrimExpr> indices, Span span) {
720 ICHECK_EQ(buffer->shape.size(), indices.size())
721 << "Buffer " << buffer->name << " is " << buffer->shape.size()
722 << "-dimensional, cannot be indexed with the " << indices.size()
723 << "-dimensional indices provided.";
724
725 for (int i = 0; i < static_cast<int>(indices.size()) - 1; i++) {

734 << "Cannot store value with " << value.dtype().lanes() << ", expected value with "
735 << index_lanes * buffer_lanes << " (" << index_lanes << " index lanes * " << buffer_lanes
736 << " buffer element lanes)";
737
738 ObjectPtr<BufferStoreNode> node = make_object<BufferStoreNode>();
739 node->buffer = std::move(buffer);
740 node->value = std::move(value);
741 node->indices = std::move(indices);
742 node->span = std::move(span);

863// MatchBufferRegion
864MatchBufferRegion::MatchBufferRegion(Buffer buffer, BufferRegion source) {
865 const Buffer& source_buffer = source->buffer;
866 arith::Analyzer analyzer;
867 // Check scope and dtype
868 CHECK_EQ(buffer.scope(), source_buffer.scope())
869 << "MatchBuffer " << buffer << " scope mismatch:" << buffer.scope() << " vs. "
870 << source_buffer.scope();
871 CHECK_EQ(buffer->dtype, source_buffer->dtype)
872 << "MatchBuffer " << buffer << " data type mismatch:" << buffer->dtype << " vs. "
873 << source_buffer->dtype;
874
875 // Check data_alignment
876 CHECK(source_buffer->data_alignment % buffer->data_alignment == 0)
877 << "Trying to match buffer to another one with lower alignment requirement "
878 << " required_alignment=" << buffer->data_alignment
879 << ", provided_alignment=" << source_buffer->data_alignment;
880
881 // Check BufferType. AutoBroadcast is not allowed for now.
882 CHECK(buffer->buffer_type == BufferType::kDefault &&
883 source_buffer->buffer_type == BufferType::kDefault)
884 << "AutoBroadcast is not allowed in MatchBuffer";
885
886 // Validate shape
887 CHECK(source->region.size() >= buffer->shape.size())
888 << "Dimension of source Region expected to be larger or equal than target buffer shape, but "
889 "got "
890 << source->region.size() << " vs. " << buffer->shape.size();
891 size_t offset = source->region.size() - buffer->shape.size();
892 for (size_t i = 0; i < offset; ++i) {
893 CHECK(analyzer.CanProve(source->region[i]->extent == 1))
894 << "The higher dimension should be 1, but got " << source->region[i]->extent << ".";
895 }
896 for (size_t i = 0; i < buffer->shape.size(); ++i) {
897 const Range& source_range = source->region[i + offset];
898 const PrimExpr& buffer_shape = buffer->shape[i];
899 if (!buffer_shape->IsInstance<VarNode>()) {
900 CHECK(analyzer.CanProve(source_range->extent == buffer_shape))
901 << "The dimension mismatched between source region and target buffer shape, got "
902 << source_range->extent << " vs. " << buffer_shape << ".";
903 }
904 }
905 // Note that we do not check elem_offset and strides in this function
906
907 // Construction
908 ObjectPtr<MatchBufferRegionNode> node = make_object<MatchBufferRegionNode>();
909 node->buffer = std::move(buffer);
910 node->source = std::move(source);
911 data_ = std::move(node);
912}
