30// Get the function type of a PrimFunc
31PrimFunc::PrimFunc(Array<tir::Var> params, Stmt body, Type ret_type,
32 Map<tir::Var, Buffer> buffer_map,
33 Optional<Map<tir::Var, Buffer>> preflattened_buffer_map, DictAttrs attrs,
34 Span span) {
35 // Assume void-return type for now
36 // TODO(tvm-team) consider type deduction from body.
37 if (!ret_type.defined()) {
38 ret_type = VoidType();
39 }
40 auto n = make_object<PrimFuncNode>();
41 n->params = std::move(params);
42 n->body = std::move(body);
43 n->ret_type = std::move(ret_type);
44 n->buffer_map = std::move(buffer_map);
45 n->preflattened_buffer_map = preflattened_buffer_map.value_or(Map<tir::Var, Buffer>());
46 n->attrs = std::move(attrs);
47 n->checked_type_ = n->func_type_annotation();
48 n->span = std::move(span);
49 data_ = std::move(n);
50}
51
52FuncType PrimFuncNode::func_type_annotation() const {
53 Array<Type> param_types;
54 for (auto param : this->params) {
55 param_types.push_back(GetType(param));
56 }
57 return FuncType(param_types, ret_type, {}, {});
58}

62class TensorIntrinManager {
63 public:
64 Map<String, tir::TensorIntrin> reg;
65
66 static TensorIntrinManager* Global() {
67 static TensorIntrinManager* inst = new TensorIntrinManager();
68 return inst;
69 }
70};
71
72TensorIntrin::TensorIntrin(PrimFunc desc, PrimFunc impl) {
73 // Check the number of func var is equal
74 CHECK_EQ(desc->params.size(), impl->params.size())
75 << "ValueError: The number of parameters of the description and the implementation of the "
76 "tensor intrinsic doesn't match.";
77 for (size_t i = 0; i < desc->params.size(); i++) {
78 CHECK(desc->params[i]->dtype.is_handle()) << "ValueError: Parameters of the description of the "
79 "tensor intrinsic should be handle only.";
80 CHECK(impl->params[i]->dtype.is_handle()) << "ValueError: Parameters of the implementation of "
81 "the tensor intrinsic should be handle only.";
82 }
83 ICHECK_EQ(desc->buffer_map.size(), impl->buffer_map.size());
84
85 ObjectPtr<TensorIntrinNode> n = make_object<TensorIntrinNode>();
86 n->desc = std::move(desc);
87 n->impl = std::move(impl);
88 data_ = std::move(n);
89}
90
91void TensorIntrin::Register(String name, TensorIntrin intrin, bool override) {
92 TensorIntrinManager* manager = TensorIntrinManager::Global();
93 if (!override) {
94 CHECK_EQ(manager->reg.count(name), 0)
95 << "ValueError: TensorIntrin '" << name << "' has already been registered";
96 }
97 manager->reg.Set(name, intrin);
98}
99
100Optional<TensorIntrin> TensorIntrin::Get(String name, bool allow_missing) {
101 const TensorIntrinManager* manager = TensorIntrinManager::Global();
102 auto it = manager->reg.find(name);
103 if (it == manager->reg.end()) {
104 if (allow_missing) {
105 return NullOpt;
106 } else {
107 LOG(FATAL) << "ValueError: TensorIntrin '" << name << "' is not registered";
108 }
109 }
110 return (*it).second;
111}

115TVM_STATIC_IR_FUNCTOR(ReprPrinter, vtable)
116 .set_dispatch<PrimFuncNode>([](const ObjectRef& ref, ReprPrinter* p) {
117 // TODO(tvm-team) redirect to Text printer once we have a good text format.
118 auto* node = static_cast<const PrimFuncNode*>(ref.get());
119 p->stream << "PrimFunc(" << node->params << ") ";
120 if (node->attrs.defined()) {
121 p->stream << "attrs=" << node->attrs;
122 }
123 p->stream << " {\n";
124 p->indent += 2;
125 p->Print(node->body);
126 p->indent -= 2;
127 p->stream << "}\n";
128 });

133 Map<tir::Var, Buffer> preflattened_buffer_map, DictAttrs attrs, Span span) {
134 return PrimFunc(params, body, ret_type, buffer_map, preflattened_buffer_map, attrs, span);
135 });

137TVM_REGISTER_GLOBAL("tir.TensorIntrin")
138 .set_body_typed([](PrimFunc desc_func, PrimFunc intrin_func) {
139 return TensorIntrin(desc_func, intrin_func);
140 });
