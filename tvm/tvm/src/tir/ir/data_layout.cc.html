40const LayoutAxis LayoutAxis::UPPER_CASE[] = {
41 LayoutAxis('A'), LayoutAxis('B'), LayoutAxis('C'), LayoutAxis('D'), LayoutAxis('E'),
42 LayoutAxis('F'), LayoutAxis('G'), LayoutAxis('H'), LayoutAxis('I'), LayoutAxis('J'),
43 LayoutAxis('K'), LayoutAxis('L'), LayoutAxis('M'), LayoutAxis('N'), LayoutAxis('O'),
44 LayoutAxis('P'), LayoutAxis('Q'), LayoutAxis('R'), LayoutAxis('S'), LayoutAxis('T'),

56const LayoutAxis& LayoutAxis::Get(const char name) {
57 ICHECK((name >= 'A' && name <= 'Z') || (name >= 'a' && name <= 'z'))
58 << "Invalid layout axis name: " << name << ". Has to be A-Z or a-z.";
59 return (name >= 'A' && name <= 'Z') ? LayoutAxis::UPPER_CASE[name - 'A']
60 : LayoutAxis::LOWER_CASE[name - 'a'];
61}

63const LayoutAxis& LayoutAxis::Get(const IterVar& itvar) {
64 const std::string axis = itvar->var.get()->name_hint;
65 ICHECK_EQ(axis.size(), 1) << "Invalid layout axis " << axis;
66 return LayoutAxis::Get(axis[0]);
67}

70 ICHECK_EQ(name.length(), 1) << "Invalid axis " << name;
71 return LayoutAxis::Get(name[0]);

74Layout::Layout(const Array<IterVar>& axes) {
75 auto node = make_object<LayoutNode>();
76 node->axes = axes;
77 std::ostringstream repr;
78 for (const IterVar& axis : axes) {
79 if (const auto* factor = axis->dom->extent.as<IntImmNode>()) {
80 ICHECK_GT(factor->value, 0);
81 repr << factor->value;
82 }
83 ICHECK_EQ(axis->var.get()->name_hint.size(), 1)
84 << "Invalid layout axis " << axis->var.get()->name_hint;
85 char c = axis->var.get()->name_hint.operator std::string()[0];
86 ICHECK((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z')) << "Invalid layout axis " << c;
87 repr << axis->var.get()->name_hint;
88 }
89 node->name = repr.str();
90 data_ = std::move(node);
91}

93Layout::Layout(const std::string& name) { // NOLINT(*)
94 if (name == "__undef__") return;
95
96 auto node = make_object<LayoutNode>();
97 node->name = name;
98
99 if (name.empty()) return; // scalar
100
101 // parse layout string
102 int32_t factor = 0;
103 for (char c : name) {
104 if (c >= 'A' && c <= 'Z') {
105 ICHECK_EQ(factor, 0) << "Invalid layout " << name << ": invalid factor size " << factor
106 << " before dimension " << c;
107 std::string shape_name("_shape");
108 shape_name.insert(0, 1, c);
109 IterVar axis =
110 IterVar(Range(PrimExpr(0), Var(shape_name)), Var(std::string(1, c)), tir::kDataPar);
111 node->axes.push_back(axis);
112 } else if (c >= 'a' && c <= 'z') {
113 ICHECK_GT(factor, 0) << "Invalid layout " << name << ": invalid factor size " << factor
114 << " for dimension " << c;
115 IterVar axis =
116 IterVar(Range(PrimExpr(0), PrimExpr(factor)), Var(std::string(1, c)), tir::kDataPar);
117 node->axes.push_back(axis);
118 factor = 0;
119 } else if (c >= '0' && c <= '9') {
120 ICHECK(factor >= 0) << "Invalid layout " << name << ": _ is adjacent to a number.";
121 factor = factor * 10 + c - '0';
122 } else {
123 LOG(FATAL) << "Invalid layout " << name;
124 }
125 }
126
127 // validate layout
128 std::vector<bool> exist_axis(256, false);
129 for (const IterVar& v : node->axes) {
130 auto axis_str = v->var.get()->name_hint.operator std::string();
131 ICHECK_EQ(axis_str.size(), 1);
132 char axis = axis_str[0];
133 ICHECK((axis >= 'a' && axis <= 'z') || (axis >= 'A' && axis <= 'Z'));
134 exist_axis[axis] = true;
135 }
136 for (const IterVar& v : node->axes) {
137 char axis = v->var.get()->name_hint.operator std::string()[0];
138 if (axis >= 'a' && axis <= 'z') {
139 ICHECK(exist_axis[axis - 'a' + 'A'])
140 << "Invalid layout " << name << ": missing axis " << std::toupper(axis);
141 }
142 }
143 data_ = std::move(node);
144}

146Layout Layout::SubLayout(size_t pos, size_t len) const {
147 if (!defined() || pos > ndim()) return Layout::Undef();
148 if (len == 0) return Layout(Array<IterVar>());
149 if (pos + len > ndim()) len = ndim() - pos;
150 Array<IterVar> new_layout;
151 const auto axes = operator->()->axes;
152 for (size_t i = pos; i < pos + len; ++i) {
153 new_layout.push_back(axes[i]);
154 }
155 return Layout(new_layout);
156}

158Layout Layout::Split(const LayoutAxis& axis, size_t target_pos, int32_t factor) const {
159 if (!defined()) return Layout::Undef();
160 const std::string& name = operator->()->name;
161 const auto axes = operator->()->axes;
162 ICHECK(target_pos <= this->ndim())
163 << "Invalid split position " << target_pos << " for layout " << name;
164 ICHECK(axis.IsPrimal()) << "Cannot split a subordinate axis " << axis;
165 ICHECK(this->Contains(axis)) << "Axis " << axis << " does not exist in " << name;
166 ICHECK(!this->Contains(axis.ToSubordinate()))
167 << "Axis " << axis << " has already been split in " << name;
168 ICHECK(factor > 0) << "Invalid split size " << factor;
169 Array<IterVar> new_layout;
170 for (size_t i = 0; i <= this->ndim(); ++i) {
171 if (i == target_pos) {
172 new_layout.push_back(IterVar(Range(PrimExpr(0), PrimExpr(factor)),
173 Var(axis.ToSubordinate().name()), tir::kDataPar));
174 }
175 if (i == this->ndim()) break;
176 new_layout.push_back(axes[i]);
177 }
178 return Layout(new_layout);
179}

181int32_t Layout::FactorOf(const LayoutAxis& axis) const {
182 if (!defined()) return -1;
183 const LayoutAxis& sub = axis.ToSubordinate();
184
185 int32_t factor = 1;
186 bool has_sub = false;
187 for (const IterVar& itvar : operator->()->axes) {
188 if (sub == LayoutAxis::Get(itvar)) {
189 has_sub = true;
190 int32_t val = itvar->dom->extent.as<IntImmNode>()->value;
191 ICHECK(val);
192 factor *= val;
193 }
194 }
195 factor = has_sub ? factor : -1;
196
197 return factor;
198}

206inline bool GetStoreRule(Array<PrimExpr>* index_rule, Array<PrimExpr>* shape_rule,
207 const Layout& src_layout, const Layout& dst_layout) {
208 if (!src_layout.defined() || src_layout.name().empty()) {
209 LOG(WARNING) << "src layout '" << src_layout.name() << "' is invalid.";
210 return false;
211 }
212 if (!dst_layout.defined() || dst_layout.name().empty()) {
213 LOG(WARNING) << "dst layout '" << dst_layout.name() << "' is invalid.";
214 return false;
215 }
216
217 for (size_t i = 0; i < dst_layout.ndim(); ++i) {
218 const auto& store_axis = dst_layout[i];
219 const IterVar& store_axis_impl = dst_layout->axes[i];
220 PrimExpr index_store(0);
221
222 for (size_t j = 0; j < src_layout.ndim(); ++j) {
223 const auto& orig_axis = src_layout[j];
224 const IterVar& orig_axis_impl = src_layout->axes[j];
225 if (store_axis.ToPrimal() == orig_axis.ToPrimal()) {
226 if (orig_axis.IsPrimal()) {
227 PrimExpr orig_var = orig_axis_impl->var;
228 const int32_t factor = src_layout.FactorOf(orig_axis);
229 if (factor > 0) {
230 orig_var = orig_var * factor;
231 }
232 index_store = index_store + orig_var;
233 } else {
234 PrimExpr factor(1);
235 for (size_t k = j + 1; k < src_layout.ndim(); ++k) {
236 if (LayoutAxis::Get(orig_axis_impl) == LayoutAxis::Get(src_layout->axes[k])) {
237 factor = factor * src_layout->axes[k]->dom->extent;
238 }
239 }
240 index_store = index_store + orig_axis_impl->var * factor;
241 }
242 }
243 }
244 if (tir::is_zero(index_store)) {
245 LOG(WARNING) << "layout '" << src_layout.name() << "'-->'" << dst_layout.name()
246 << "' is not convertible.";
247 return false;
248 }
249
250 PrimExpr shape_store = index_store;
251 if (store_axis.IsPrimal()) {
252 const int32_t factor = dst_layout.FactorOf(store_axis);
253 if (factor > 0) {
254 shape_store = shapediv(index_store, PrimExpr(factor));
255 index_store = indexdiv(index_store, PrimExpr(factor));
256 }
257 } else {
258 PrimExpr stride(1);
259 PrimExpr factor(1);
260 for (size_t j = i; j < dst_layout.ndim(); ++j) {
261 if (LayoutAxis::Get(store_axis_impl) == LayoutAxis::Get(dst_layout->axes[j])) {
262 stride = stride * dst_layout->axes[j]->dom->extent;
263 if (j > i) {
264 factor = factor * dst_layout->axes[j]->dom->extent;
265 }
266 }
267 }
268 shape_store = indexdiv(indexmod(index_store, stride), factor);
269 index_store = indexdiv(indexmod(index_store, stride), factor);
270 }
271
272 index_rule->push_back(index_store);
273 shape_rule->push_back(shape_store);
274 }
275
276 std::stringstream ss;
277 ss << "index rule for " << src_layout.name() << "-->" << dst_layout.name() << ": [ ";
278 for (const auto& r : *index_rule) {
279 ss << r << ", ";
280 }
281 ss << "]" << std::endl;
282
283 ss << "shape rule for " << src_layout.name() << "-->" << dst_layout.name() << ": [ ";
284 for (const auto& r : *shape_rule) {
285 ss << r << ", ";
286 }
287 ss << "]" << std::endl;
288 VLOG(1) << std::endl << ss.str();
289
290 return true;
291}

293inline Array<PrimExpr> TransformIndex(const Array<PrimExpr>& src_index,
294 const Array<IterVar>& src_axis,
295 const Array<PrimExpr>& transform_rule) {
296 arith::Analyzer ana;
297 Array<PrimExpr> result;
298 std::unordered_map<const tir::VarNode*, PrimExpr> bind_map;
299 for (size_t i = 0; i < src_index.size(); ++i) {
300 bind_map[src_axis[i]->var.get()] = src_index[i];
301 }
302 for (PrimExpr rule : transform_rule) {
303 result.push_back(ana.Simplify(tir::Substitute(rule, bind_map)));
304 }
305 return result;
306}

308Array<PrimExpr> BijectiveLayout::ForwardIndex(const Array<PrimExpr>& src_index) const {
309 ICHECK(defined()) << "Cannot operate on an undefined bijective layout.";
310 const BijectiveLayoutNode* self = operator->();
311 ICHECK_EQ(src_index.size(), self->src_layout->axes.size())
312 << "Input mismatch with layout " << self->src_layout;
313 return TransformIndex(src_index, self->src_layout->axes, self->index_forward_rule);
314}

316Array<PrimExpr> BijectiveLayout::BackwardIndex(const Array<PrimExpr>& dst_index) const {
317 ICHECK(defined()) << "Cannot operate on an undefined bijective layout.";

319 ICHECK_EQ(dst_index.size(), self->dst_layout->axes.size())
320 << "Output mismatch with layout " << self->dst_layout;
321 return TransformIndex(dst_index, self->dst_layout->axes, self->index_backward_rule);

324inline Array<PrimExpr> TransformShape(const Array<PrimExpr>& src_shape,
325 const Array<IterVar>& src_axis,
326 const Array<IterVar>& target_axis,
327 const Array<PrimExpr>& transform_rule) {
328 arith::Analyzer ana;
329 ICHECK_EQ(src_shape.size(), src_axis.size())
330 << "Input shape size " << src_shape.size() << " mismatch with the exepected shape size "
331 << src_axis.size();
332 // bind variables for original axes
333 // for major-axis, bind the corresponding size
334 // for minor-axis, simply bind it as 0, so that we can reuse forward/backward_rule,
335 // e.g., (C * 16 + c) / 32
336 std::unordered_map<const tir::VarNode*, PrimExpr> bind_map;
337 std::unordered_set<size_t> symbolic_var_set;
338 for (size_t i = 0; i < src_shape.size(); ++i) {
339 PrimExpr orig_shape = src_shape[i];
340 IterVar orig_axis = src_axis[i];
341 if (orig_shape.as<tir::AnyNode>()) {
342 symbolic_var_set.insert(i);
343 }
344 if (!LayoutAxis::Get(orig_axis).IsPrimal()) {
345 if (orig_shape.defined()) {
346 const auto* orig_shape_const = orig_shape.as<IntImmNode>();
347 const auto* orig_axis_extent = orig_axis->dom->extent.as<IntImmNode>();
348 if (orig_shape_const) {
349 ICHECK_EQ(orig_shape_const->value, orig_axis_extent->value)
350 << "Input shape mismatch at index " << i << ". Expected " << orig_axis->dom->extent
351 << ", get " << orig_shape;
352 }
353 }
354 bind_map[orig_axis->var.get()] = IntImm(orig_axis->var->dtype, 0);
355 } else {
356 bind_map[orig_axis->var.get()] = orig_axis->var->dtype == orig_shape->dtype
357 ? orig_shape
358 : cast(orig_axis->var->dtype, orig_shape);
359 }
360 }
361 // infer the target shape,
362 // for major-axis, use the forward/backward_rule directly,
363 // for minor-axis, simply use the extent.
364 Array<PrimExpr> result;
365 ICHECK_EQ(transform_rule.size(), target_axis.size());
366 for (size_t i = 0; i < transform_rule.size(); ++i) {
367 PrimExpr rule = transform_rule[i];
368 IterVar axis = target_axis[i];
369 if (!LayoutAxis::Get(axis).IsPrimal()) {
370 result.push_back(axis->dom->extent);
371 } else {
372 if (symbolic_var_set.count(i)) {
373 result.push_back(tir::Any());
374 } else {
375 result.push_back(ana.Simplify(tir::Substitute(rule, bind_map)));
376 }
377 }
378 }
379
380 std::stringstream ss;
381 ss << "shape rule for " << Layout(src_axis).name() << "-->" << Layout(target_axis).name()
382 << ": [ ";
383 for (const auto& r : transform_rule) {
384 ss << r << ", ";
385 }
386 ss << "]" << std::endl;
387
388 ss << "shape transform: [ ";
389 for (const auto& s : src_shape) {
390 ss << s << ", ";
391 }
392 ss << "] --> [ ";
393 for (const auto& r : result) {
394 ss << r << ", ";
395 }
396 ss << "]" << std::endl;
397 VLOG(1) << std::endl << ss.str();
398
399 return result;
400}

402Array<PrimExpr> BijectiveLayout::ForwardShape(const Array<PrimExpr>& shape) const {
403 ICHECK(defined()) << "Cannot operate on an undefined bijective layout.";
404 const BijectiveLayoutNode* self = operator->();
405 return TransformShape(shape, self->src_layout->axes, self->dst_layout->axes,
406 self->shape_forward_rule);
407}

409Array<PrimExpr> BijectiveLayout::BackwardShape(const Array<PrimExpr>& shape) const {
410 ICHECK(defined()) << "Cannot operate on an undefined bijective layout.";
411 const BijectiveLayoutNode* self = operator->();
412 return TransformShape(shape, self->dst_layout->axes, self->src_layout->axes,
413 self->shape_backward_rule);
414}

416BijectiveLayout::BijectiveLayout(Layout src_layout, Layout dst_layout) {
417 auto n = make_object<BijectiveLayoutNode>();
418
419 n->src_layout = std::move(src_layout);
420 n->dst_layout = std::move(dst_layout);
421 // To be consistent with previous behavior, a nullptr layout is created
422 // when argument is invalid.
423 if (GetStoreRule(&n->index_forward_rule, &n->shape_forward_rule, n->src_layout, n->dst_layout)) {
424 ICHECK(GetStoreRule(&n->index_backward_rule, &n->shape_backward_rule, n->dst_layout,
425 n->src_layout));
426 data_ = std::move(n);
427 }
428}
