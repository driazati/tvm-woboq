34#define TVM_DEFINE_BINOP_CONSTRUCTOR(Name) \
35 Name::Name(PrimExpr a, PrimExpr b, Span span) { \
36 using T = Name::ContainerType; \
37 ICHECK(a.defined()) << "ValueError: a is undefined\n"; \
38 ICHECK(b.defined()) << "ValueError: b is undefined\n"; \
39 CHECK(a.dtype() == b.dtype()) << "TypeError: mismatched types. " << a.dtype() << " vs. " \
40 << b.dtype() << "\n"; \
41 ObjectPtr<T> node = make_object<T>(); \
42 node->dtype = a.dtype(); \
43 node->a = std::move(a); \
44 node->b = std::move(b); \
45 node->span = std::move(span); \
46 data_ = std::move(node); \
47 }

60 node->span = std::move(span); \
61 data_ = std::move(node); \
62 }

83Var Var::copy_with_suffix(const String& suffix) const {
84 const VarNode* node = get();
85 ObjectPtr<VarNode> new_ptr;
86 if (auto* ptr = this->as<SizeVarNode>()) {
87 new_ptr = make_object<SizeVarNode>(*ptr);
88 } else {
89 new_ptr = make_object<VarNode>(*node);
90 }
91 new_ptr->name_hint = new_ptr->name_hint + suffix;
92 return Var(new_ptr);
93}
94
95Var Var::copy_with_dtype(DataType dtype) const {
96 const VarNode* node = get();
97 ObjectPtr<VarNode> new_ptr;
98 if (auto* ptr = this->as<SizeVarNode>()) {
99 new_ptr = make_object<SizeVarNode>(*ptr);
100 } else {
101 new_ptr = make_object<VarNode>(*node);
102 }
103 new_ptr->type_annotation = GetTypeFromRuntimeDataType(dtype);
104 new_ptr->dtype = std::move(dtype);
105 return Var(new_ptr);
106}

149// IterVar
150IterVar::IterVar(Range dom, Var var, IterVarType t, String thread_tag, Span span) {
151 ObjectPtr<IterVarNode> n = make_object<IterVarNode>();
152 if (dom.defined() && dom->extent.defined()) {
153 CHECK(dom->extent.dtype().is_int())
154 << "The dtype of the domain of an IterVar must be an integer type. However, the domain's "
155 "dtype is "
156 << dom->extent.dtype();
157 CHECK_EQ(dom->extent.dtype(), var.dtype())
158 << "The dtype of the extent of an IterVar (" << dom->extent.dtype()
159 << ") must match its associated Var's dtype (" << var.dtype() << ")";
160 }
161 n->dom = dom;
162 n->var = var;
163 n->iter_type = t;
164 n->thread_tag = thread_tag;
165 n->span = std::move(span);
166 data_ = std::move(n);
167}

169TVM_REGISTER_GLOBAL("tir.IterVar")
170 .set_body_typed([](Range dom, Var var, int iter_type, String thread_tag, Span span) {
171 return IterVar(dom, var, static_cast<IterVarType>(iter_type), thread_tag, span);
172 });

177 p->stream << "iter_var(";

179 p->stream << op->var->name_hint << ", ";

181 if (op->dom.defined()) {
182 p->stream << op->dom;

602// Select
603Select::Select(PrimExpr condition, PrimExpr true_value, PrimExpr false_value, Span span) {
604 ICHECK(condition.defined()) << "ValueError: condition is undefined";
605 ICHECK(true_value.defined()) << "ValueError: true_value is undefined";
606 ICHECK(false_value.defined()) << "ValueError: true_value is undefined";
607 ICHECK(condition.dtype().is_bool());
608 ICHECK(condition.dtype().lanes() == true_value.dtype().lanes() || condition.dtype().lanes() == 1);
609 ICHECK(false_value.dtype() == true_value.dtype()) << "TypeError: mismatched types";
610
611 ObjectPtr<SelectNode> node = make_object<SelectNode>();
612 node->dtype = true_value.dtype();
613 node->condition = std::move(condition);
614 node->true_value = std::move(true_value);
615 node->false_value = std::move(false_value);
616 node->span = std::move(span);
617 data_ = std::move(node);
618}

623 });

639// Load
640Load::Load(DataType dtype, Var buffer_var, PrimExpr index, PrimExpr predicate, Span span) {
641 LOG(FATAL) << "Unexpected use of deprecated Store node for buffer " << buffer_var->name_hint
642 << ". Use BufferStore instead.";
643 ICHECK(buffer_var.defined());
644 ICHECK(predicate.defined());
645 ICHECK(index.defined());
646
647 // Assume that the array elements have 1 lane, unless a type
648 // annotation tells us otherwise.
649 int element_lanes = 1;
650 auto pointer_type = tir::GetPointerType(buffer_var->type_annotation);
651 if (pointer_type.has_value()) {
652 // Cannot check element type of array, as it may be different than
653 // the loaded type in some cases.
654 //
655 // 1. Booleans use DataType::Int(8) while stored, and the codegens
656 // handle cast to boolean.
657 //
658 // 2. The StorageRewrite pass can merge multiple allocations at
659 // the same scope, regardless of element type. The codegen is
660 // then responsible for casting to the output type.
661
662 // TODO(Lunderberg): Uncomment this check once it can be applied.
663 // See https://discuss.tvm.apache.org/t/pre-rfc-vectorized-tir-buffers/10615
664 // for discussion.
665
666 // ICHECK(dtype.element_of() == pointer_type->element_of())
667 // << "Type mismatch, cannot load type " << dtype << " from buffer " <<
668 // buffer_var->name_hint
669 // << " of type " << pointer_type.value();
670 element_lanes = pointer_type->lanes();
671 }
672
673 // The C-based codegens assume that all loads occur on a array with
674 // non-vectorized elements, and cast between
675 // vectorized/non-vectorized arrays as needed. Ideally, these
676 // should be changed to explicit casts in the TIR graph, rather than
677 // being handled at the code-gen level.
678 ICHECK((dtype.lanes() == element_lanes * index.dtype().lanes()) ||
679 (dtype.lanes() == index.dtype().lanes()));
680 ICHECK((dtype.lanes() == element_lanes * predicate.dtype().lanes()) ||
681 (dtype.lanes() == index.dtype().lanes()));
682
683 ObjectPtr<LoadNode> node = make_object<LoadNode>();
684 node->dtype = dtype;
685 node->buffer_var = std::move(buffer_var);
686 node->index = std::move(index);
687 node->predicate = std::move(predicate);
688 node->span = std::move(span);
689
690 data_ = std::move(node);
691}

947// CommReducer
948CommReducer::CommReducer(Array<Var> lhs, Array<Var> rhs, Array<PrimExpr> result,
949 Array<PrimExpr> identity_element, Span span) {
950 size_t n_group = result.size();
951 CHECK_EQ(lhs.size(), n_group) << "ValueError: The number of vars in `lhs` must equal to the "
952 "number of elements in `results`";
953 CHECK_EQ(rhs.size(), n_group) << "ValueError: The number of vars in `rhs` must equal to the "
954 "number of elements in `results`";
955 CHECK_EQ(identity_element.size(), n_group)
956 << "ValueError: The number of identities must equal to the number of elements in `results`";
957
958 // Change the dtype of input vars to adapt to the dtype of identities
959 ArrayNode* p_lhs = lhs.CopyOnWrite();
960 ArrayNode* p_rhs = rhs.CopyOnWrite();
961 std::unordered_map<const VarNode*, PrimExpr> var_map;
962 var_map.reserve(n_group * 2);
963 for (int i = 0; i < static_cast<int>(n_group); ++i) {
964 DataType dtype = identity_element[i].dtype();
965 Var l = lhs[i].copy_with_dtype(dtype);
966 Var r = rhs[i].copy_with_dtype(dtype);
967 var_map[lhs[i].get()] = l;
968 var_map[rhs[i].get()] = r;
969
970 p_lhs->SetItem(i, l);
971 p_rhs->SetItem(i, r);
972 }
973
974 ArrayNode* p_result = result.CopyOnWrite();
975 for (int i = 0; i < static_cast<int>(n_group); ++i) {
976 p_result->SetItem(i, Substitute(result[i], var_map));
977 }
978
979 auto node = make_object<CommReducerNode>();
980 node->lhs = lhs;
981 node->rhs = rhs;
982 node->result = result;
983 node->identity_element = identity_element;
984 node->span = std::move(span);
985 data_ = std::move(node);
986}

988Array<PrimExpr> CommReducerNode::operator()(Array<PrimExpr> a, Array<PrimExpr> b) const {
989 ICHECK_EQ(a.size(), b.size());
990 ICHECK_EQ(lhs.size(), a.size());
991 ICHECK_EQ(rhs.size(), b.size());
992 Map<Var, PrimExpr> value_map;
993 for (size_t i = 0; i < a.size(); ++i) {
994 value_map.Set(lhs[i], a[i]);
995 value_map.Set(rhs[i], b[i]);
996 }
997 auto ret = this->result;
998 ret.MutateByApply([&value_map](const PrimExpr& e) { return Substitute(e, value_map); });
999 return ret;
1000}
