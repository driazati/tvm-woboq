37IndexMap::IndexMap(Array<Var> initial_indices, Array<PrimExpr> final_indices) {
38 auto n = make_object<IndexMapNode>();
39 n->initial_indices = std::move(initial_indices);
40 n->final_indices = std::move(final_indices);
41 data_ = std::move(n);
42}

44IndexMap IndexMap::FromFunc(int ndim, runtime::TypedPackedFunc<Array<PrimExpr>(Array<Var>)> func) {
45 Array<Var> initial_indices;
46 initial_indices.reserve(ndim);
47 for (int i = 0; i < ndim; ++i) {
48 initial_indices.push_back(Var("i" + std::to_string(i), DataType::Int(32)));
49 }
50 return IndexMap(initial_indices, func(initial_indices));
51}

53std::pair<IndexMap, PrimExpr> IndexMap::NonSurjectiveInverse(Array<Range> initial_ranges) const {
54 // Dummy variables to represent the inverse's inputs.
55 Array<Var> output_vars;
56 for (size_t i = 0; i < (*this)->final_indices.size(); i++) {
57 PrimExpr index = (*this)->final_indices[i];
58 // TODO(Lunderberg): Better names for these variables. A variable
59 // that is passed through unmodified (`index` is an element of
60 // `initial_indices`) should use that input index's name. A pair
61 // of output indices variables split from a single input index
62 // should be named (X.outer,X.inner).
63 std::stringstream ss;
64 ss << "axis" << i;
65 Var var_index(ss.str(), index.dtype());
66 output_vars.push_back(var_index);
67 }
68
69 // Dummy ranges for the extent of each input.
70 Map<Var, Range> input_iters;
71 ICHECK_EQ((*this)->initial_indices.size(), initial_ranges.size());
72 for (size_t i = 0; i < initial_ranges.size(); i++) {
73 input_iters.Set((*this)->initial_indices[i], initial_ranges[i]);
74 }
75
76 // Unpack the output indices into linear combinations of the initial
77 // indices.
78 arith::Analyzer analyzer;
79 auto padded_iter_map = DetectIterMap((*this)->final_indices, input_iters, /* predicate = */ 1,
80 /*check_level=*/arith::IterMapLevel::NoCheck, &analyzer,
81 /*simplify_trivial_iterators=*/false);
82 CHECK(padded_iter_map->errors.empty()) << "Could not parse mapping as sum of iterators. "
83 << "Error: " << padded_iter_map->errors[0];
84
85 // Determine expressions for the input variables, in terms of the
86 // output variables.
87 Map<Var, PrimExpr> inverse_exprs_map = InverseAffineIterMap(
88 padded_iter_map->indices, Array<PrimExpr>(output_vars.begin(), output_vars.end()));
89
90 // Unpack the map to an array, maintaining the same parameter order.
91 Array<PrimExpr> inverse_exprs;
92 for (const auto& index : (*this)->initial_indices) {
93 inverse_exprs.push_back(inverse_exprs_map.at(index));
94 }
95
96 PrimExpr padding_predicate = padded_iter_map->padding_predicate;
97 padding_predicate = arith::NormalizeIterMapToExpr(padding_predicate);
98 padding_predicate = Substitute(padding_predicate, inverse_exprs_map);
99
100 {
101 auto output_ranges = (*this)->MapRanges(initial_ranges);
102 ICHECK_EQ(output_ranges.size(), output_vars.size());
103
104 arith::Analyzer analyzer;
105 for (size_t i = 0; i < output_vars.size(); ++i) {
106 analyzer.Bind(output_vars[i], output_ranges[i]);
107 }
108
109 // Additional simplification steps required to unwrap nested floordiv/floormod
110 padding_predicate = analyzer.Simplify(padding_predicate, 10);
111 }
112
113 return {IndexMap(output_vars, inverse_exprs), padding_predicate};
114}

116IndexMap IndexMap::Inverse(Array<Range> initial_ranges) const {
117 // Dummy variables to represent the inverse's inputs.
118 Array<Var> output_vars;
119 for (size_t i = 0; i < (*this)->final_indices.size(); i++) {
120 PrimExpr index = (*this)->final_indices[i];
121 // TODO(Lunderberg): Better names for these variables. A variable
122 // that is passed through unmodified (`index` is an element of
123 // `initial_indices`) should use that input index's name. A pair
124 // of output indices variables split from a single input index
125 // should be named (X.outer,X.inner).
126 std::stringstream ss;
127 ss << "axis" << i;
128 Var var_index(ss.str(), index.dtype());
129 output_vars.push_back(var_index);
130 }
131
132 // Dummy ranges for the extent of each input.
133 Map<Var, Range> input_iters;
134 ICHECK_EQ((*this)->initial_indices.size(), initial_ranges.size());
135 for (size_t i = 0; i < initial_ranges.size(); i++) {
136 input_iters.Set((*this)->initial_indices[i], initial_ranges[i]);
137 }
138
139 // Unpack the output indices into linear combinations of the initial
140 // indices.
141 arith::Analyzer analyzer;
142 auto iter_map = DetectIterMap((*this)->final_indices, input_iters, /* predicate = */ 1,
143 /* check_level = */ arith::IterMapLevel::Bijective, &analyzer,
144 /* simplify_trivial_iterators = */ false);
145 CHECK(iter_map->indices.size()) << "Index transformation was not bijective.";
146
147 // Determine expressions for the input variables, in terms of the
148 // output variables.
149 Map<Var, PrimExpr> inverse_exprs_map = InverseAffineIterMap(
150 iter_map->indices, Array<PrimExpr>(output_vars.begin(), output_vars.end()));
151
152 // Unpack the map to an array, maintaining the same parameter order.
153 Array<PrimExpr> inverse_exprs;
154 for (int i = 0, n = (*this)->initial_indices.size(); i < n; ++i) {
155 Var index = (*this)->initial_indices[i];
156 if (is_one(initial_ranges[i]->extent) && !inverse_exprs_map.count(index)) {
157 inverse_exprs.push_back(initial_ranges[i]->min);
158 } else {
159 inverse_exprs.push_back(inverse_exprs_map.at(index));
160 }
161 }
162
163 return IndexMap(output_vars, inverse_exprs);
164}

166Array<PrimExpr> IndexMapNode::MapIndices(const Array<PrimExpr>& indices,
167 arith::Analyzer* analyzer) const {
168 ICHECK_EQ(indices.size(), initial_indices.size());
169
170 Map<Var, PrimExpr> vmap;
171
172 for (size_t i = 0; i < initial_indices.size(); i++) {
173 vmap.Set(initial_indices[i], indices[i]);
174 }
175
176 arith::Analyzer local_analyzer;
177 if (!analyzer) {
178 analyzer = &local_analyzer;
179 }
180
181 Array<PrimExpr> output = final_indices;
182 output.MutateByApply(
183 [&](const PrimExpr& index) { return analyzer->Simplify(Substitute(index, vmap)); });
184
185 return output;
186}

188Array<Range> IndexMapNode::MapRanges(const Array<Range>& ranges, arith::Analyzer* analyzer) const {
189 ICHECK_EQ(ranges.size(), initial_indices.size());
190
191 Map<Var, Range> input_iters;
192 for (size_t i = 0; i < initial_indices.size(); i++) {
193 input_iters.Set(initial_indices[i], ranges[i]);
194 }
195
196 std::unordered_map<const VarNode*, arith::IntSet> dom_map;
197 for (size_t i = 0; i < initial_indices.size(); i++) {
198 dom_map[initial_indices[i].get()] = arith::IntSet::FromRange(ranges[i]);
199 }
200
201 arith::Analyzer local_analyzer;
202 if (!analyzer) {
203 analyzer = &local_analyzer;
204 }
205
206 Array<Range> output;
207 for (const auto& final_index : final_indices) {
208 auto int_set = arith::EvalSet(final_index, dom_map);
209 output.push_back(Range::FromMinExtent(analyzer->Simplify(int_set.min()),
210 analyzer->Simplify(int_set.max() - int_set.min() + 1)));
211 }
212
213 return output;
214}

216Array<PrimExpr> IndexMapNode::MapShape(const Array<PrimExpr>& shape,
217 arith::Analyzer* analyzer) const {
218 ICHECK_EQ(shape.size(), initial_indices.size());
219
220 Array<Range> ranges;
221 for (auto& dim : shape) {
222 ranges.push_back(Range(0, dim));
223 }
224 Array<Range> mapped = MapRanges(std::move(ranges), analyzer);
225
226 Array<PrimExpr> output;
227 for (auto& range : mapped) {
228 ICHECK(is_zero(range->min));
229 output.push_back(range->extent);
230 }
231
232 return output;
233}

235String IndexMapNode::ToPythonString() const {
236 std::unordered_set<std::string> used_names;
237 Map<Var, PrimExpr> var_remap;
238 for (const Var& initial_index : initial_indices) {
239 if (used_names.count(initial_index->name_hint)) {
240 std::string new_name = initial_index->name_hint + std::to_string(used_names.size());
241 used_names.insert(new_name);
242 var_remap.Set(initial_index, Var(new_name));
243 } else {
244 used_names.insert(initial_index->name_hint);
245 }
246 }
247 std::ostringstream oss;
248 oss << "lambda ";
249 for (size_t i = 0; i < initial_indices.size(); ++i) {
250 if (i != 0) {
251 oss << ", ";
252 }
253 auto it = var_remap.find(initial_indices[i]);
254 if (it != var_remap.end()) {
255 oss << (*it).second;
256 } else {
257 oss << initial_indices[i];
258 }
259 }
260 oss << ": (";
261 for (size_t i = 0; i < final_indices.size(); ++i) {
262 oss << Substitute(final_indices[i], var_remap);
263 oss << ", ";
264 }
265 oss << ")";
266 return String(oss.str());
267}
