37IndexMap::IndexMap(Array<Var> initial_indices, Array<PrimExpr> final_indices,
38 Optional<IndexMap> inverse_index_map) {
39 auto n = make_object<IndexMapNode>();
40 n->initial_indices = std::move(initial_indices);
41 n->final_indices = std::move(final_indices);
42 n->inverse_index_map = std::move(inverse_index_map);
43 data_ = std::move(n);
44}
45
46IndexMap IndexMap::FromFunc(int ndim, runtime::TypedPackedFunc<Array<PrimExpr>(Array<Var>)> func,
47 Optional<IndexMap> inverse_index_map) {
48 Array<Var> initial_indices;
49 initial_indices.reserve(ndim);
50 for (int i = 0; i < ndim; ++i) {
51 initial_indices.push_back(Var("i" + std::to_string(i), DataType::Int(32)));
52 }
53 return IndexMap(initial_indices, func(initial_indices), std::move(inverse_index_map));
54}

56std::pair<IndexMap, PrimExpr> IndexMap::NonSurjectiveInverse(Array<Range> initial_ranges) const {
57 if ((*this)->inverse_index_map.defined()) {
58 // return the pre-defined inverse index map if exists. In this
59 // case, the user-defined inverse is assumed to be correct and
60 // bijective.
61 PrimExpr padding_predicate = Bool(false);
62 return {Downcast<IndexMap>((*this)->inverse_index_map.value()), padding_predicate};
63 }
64
65 // Dummy variables to represent the inverse's inputs.
66 Array<Var> output_vars;
67 for (size_t i = 0; i < (*this)->final_indices.size(); i++) {
68 PrimExpr index = (*this)->final_indices[i];
69 // TODO(Lunderberg): Better names for these variables. A variable
70 // that is passed through unmodified (`index` is an element of
71 // `initial_indices`) should use that input index's name. A pair
72 // of output indices variables split from a single input index
73 // should be named (X.outer,X.inner).
74 std::stringstream ss;
75 ss << "axis" << i;
76 Var var_index(ss.str(), index.dtype());
77 output_vars.push_back(var_index);
78 }
79
80 // Dummy ranges for the extent of each input.
81 Map<Var, Range> input_iters;
82 ICHECK_EQ((*this)->initial_indices.size(), initial_ranges.size());
83 for (size_t i = 0; i < initial_ranges.size(); i++) {
84 input_iters.Set((*this)->initial_indices[i], initial_ranges[i]);
85 }
86
87 // Unpack the output indices into linear combinations of the initial
88 // indices.
89 arith::Analyzer analyzer;
90 auto padded_iter_map = DetectIterMap((*this)->final_indices, input_iters, /* predicate = */ 1,
91 /*check_level=*/arith::IterMapLevel::NoCheck, &analyzer,
92 /*simplify_trivial_iterators=*/false);
93 CHECK(padded_iter_map->errors.empty()) << "Could not parse mapping as sum of iterators. "
94 << "Error: " << padded_iter_map->errors[0];
95
96 // Determine expressions for the input variables, in terms of the
97 // output variables.
98 Map<Var, PrimExpr> inverse_exprs_map = InverseAffineIterMap(
99 padded_iter_map->indices, Array<PrimExpr>(output_vars.begin(), output_vars.end()));
100
101 // Unpack the map to an array, maintaining the same parameter order.
102 Array<PrimExpr> inverse_exprs;
103 for (int i = 0, n = (*this)->initial_indices.size(); i < n; ++i) {
104 Var index = (*this)->initial_indices[i];
105 PrimExpr expr;
106 if (is_one(initial_ranges[i]->extent) && !inverse_exprs_map.count(index)) {
107 expr = initial_ranges[i]->min;
108 } else {
109 expr = inverse_exprs_map.at(index);
110 }
111 inverse_exprs.push_back(analyzer.Simplify(expr));
112 }
113
114 PrimExpr padding_predicate = padded_iter_map->padding_predicate;
115 padding_predicate = arith::NormalizeIterMapToExpr(padding_predicate);
116 padding_predicate = Substitute(padding_predicate, inverse_exprs_map);
117
118 {
119 auto output_ranges = (*this)->MapRanges(initial_ranges);
120 ICHECK_EQ(output_ranges.size(), output_vars.size());
121
122 arith::Analyzer analyzer;
123 for (size_t i = 0; i < output_vars.size(); ++i) {
124 analyzer.Bind(output_vars[i], output_ranges[i]);
125 }
126
127 // Additional simplification steps required to unwrap nested floordiv/floormod
128 padding_predicate = analyzer.Simplify(padding_predicate, 10);
129 }
130
131 return {IndexMap(output_vars, inverse_exprs), padding_predicate};
132}

134IndexMap IndexMap::Inverse(Array<Range> initial_ranges) const {
135 auto [inverse, padding_predicate] = NonSurjectiveInverse(std::move(initial_ranges));
136 arith::Analyzer analyzer;
137 CHECK(analyzer.CanProve(!padding_predicate))
138 << "Bijective inverse should not contain padding, but inverse of " << *this << " over range "
139 << initial_ranges << " resulted in a padding predicate of " << padding_predicate;
140 return inverse;
141}

143Array<PrimExpr> IndexMapNode::MapIndices(const Array<PrimExpr>& indices,
144 arith::Analyzer* analyzer) const {
145 ICHECK_EQ(indices.size(), initial_indices.size());
146
147 Map<Var, PrimExpr> vmap;
148
149 for (size_t i = 0; i < initial_indices.size(); i++) {
150 vmap.Set(initial_indices[i], indices[i]);
151 }
152
153 arith::Analyzer local_analyzer;
154 if (!analyzer) {
155 analyzer = &local_analyzer;
156 }
157
158 Array<PrimExpr> output = final_indices.Map(
159 [&](PrimExpr index) { return analyzer->Simplify(Substitute(std::move(index), vmap)); });
160
161 return output;
162}

164Array<Range> IndexMapNode::MapRanges(const Array<Range>& ranges, arith::Analyzer* analyzer) const {
165 ICHECK_EQ(ranges.size(), initial_indices.size());
166
167 Map<Var, Range> input_iters;
168 for (size_t i = 0; i < initial_indices.size(); i++) {
169 input_iters.Set(initial_indices[i], ranges[i]);
170 }
171
172 std::unordered_map<const VarNode*, arith::IntSet> dom_map;
173 for (size_t i = 0; i < initial_indices.size(); i++) {
174 dom_map[initial_indices[i].get()] = arith::IntSet::FromRange(ranges[i]);
175 }
176
177 arith::Analyzer local_analyzer;
178 if (!analyzer) {
179 analyzer = &local_analyzer;
180 }
181
182 Array<Range> output;
183 for (const auto& final_index : final_indices) {
184 auto int_set = arith::EvalSet(final_index, dom_map);
185 output.push_back(Range::FromMinExtent(analyzer->Simplify(int_set.min()),
186 analyzer->Simplify(int_set.max() - int_set.min() + 1)));
187 }
188
189 return output;
190}

192Array<PrimExpr> IndexMapNode::MapShape(const Array<PrimExpr>& shape,
193 arith::Analyzer* analyzer) const {
194 ICHECK_EQ(shape.size(), initial_indices.size());
195
196 Array<Range> ranges;
197 for (auto& dim : shape) {
198 ranges.push_back(Range(0, dim));
199 }
200 Array<Range> mapped = MapRanges(std::move(ranges), analyzer);
201
202 Array<PrimExpr> output;
203 for (auto& range : mapped) {
204 ICHECK(is_zero(range->min));
205 output.push_back(range->extent);
206 }
207
208 return output;
209}

211/*!
212 * \brief Auxilarry function to comvert an index map to lambda expression in Python.
213 * \param initial_indices The initial indices in the index map.
214 * \param final_indices The final indices in the index map.
215 * \return The lambda expression string.
216 */
217std::string IndexMap2PythonLambdaExpr(const Array<Var>& initial_indices,
218 const Array<PrimExpr>& final_indices) {
219 std::unordered_set<std::string> used_names;
220 Map<Var, PrimExpr> var_remap;
221 for (const Var& initial_index : initial_indices) {
222 if (used_names.count(initial_index->name_hint)) {
223 std::string new_name = initial_index->name_hint + std::to_string(used_names.size());
224 used_names.insert(new_name);
225 var_remap.Set(initial_index, Var(new_name));
226 } else {
227 used_names.insert(initial_index->name_hint);
228 }
229 }
230 std::ostringstream oss;
231 oss << "lambda ";
232 for (size_t i = 0; i < initial_indices.size(); ++i) {
233 if (i != 0) {
234 oss << ", ";
235 }
236 auto it = var_remap.find(initial_indices[i]);
237 if (it != var_remap.end()) {
238 oss << (*it).second;
239 } else {
240 oss << initial_indices[i];
241 }
242 }
243 oss << ": (";
244 for (size_t i = 0; i < final_indices.size(); ++i) {
245 if (i != 0) {
246 oss << " ";
247 }
248 oss << Substitute(final_indices[i], var_remap);
249 oss << ",";
250 }
251 oss << ")";
252 return oss.str();
253}
254
255String IndexMapNode::ToPythonString() const {
256 std::string lambda_expr = IndexMap2PythonLambdaExpr(initial_indices, final_indices);
257 if (!inverse_index_map.defined()) {
258 return String(lambda_expr);
259 }
260 // Also convert the inverse index map.
261 IndexMap inverse = Downcast<IndexMap>(inverse_index_map.value());
262 std::string inverse_lambda_expr =
263 IndexMap2PythonLambdaExpr(inverse->initial_indices, inverse->final_indices);
264 std::ostringstream oss;
265 oss << "tvm.tir.IndexMap.from_func(" << lambda_expr
266 << ", inverse_index_map=" << inverse_lambda_expr << ")";
267 return String(oss.str());
268}
