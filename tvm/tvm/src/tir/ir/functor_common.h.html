25#ifndef TVM_TIR_IR_FUNCTOR_COMMON_H_
26#define TVM_TIR_IR_FUNCTOR_COMMON_H_
27
28namespace tvm {
29namespace tir {
30
31// Implementation of Visitors
32template <typename T, typename F>
33inline void VisitArray(const Array<T>& arr, F fvisit) {
34 for (size_t i = 0; i < arr.size(); i++) {
35 fvisit(arr[i]);
36 }
37}
38
39template <typename T, typename F>
40inline Array<T> MutateArray(Array<T> arr, F fmutate) {
41 return arr.Map(fmutate);
42}
43
44} // namespace tir
