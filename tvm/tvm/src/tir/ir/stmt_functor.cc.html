127void StmtVisitor::VisitStmt_(const BlockNode* op) {
128 auto fvisit_buffer_region = [this](const BufferRegion& s) {
129 for (const auto& range : s->region) {
130 this->VisitExpr(range->min);
131 this->VisitExpr(range->extent);
132 }
133 };
134 VisitArray(op->iter_vars, [this](const IterVar& iter_var) {
135 this->VisitExpr(iter_var->dom->min);
136 this->VisitExpr(iter_var->dom->extent);
137 });
138 VisitArray(op->reads, fvisit_buffer_region);
139 VisitArray(op->writes, fvisit_buffer_region);
140 VisitArray(op->match_buffers,
141 [fvisit_buffer_region](const MatchBufferRegion& match_buffer_region) {
142 fvisit_buffer_region(match_buffer_region->source);
143 });
144 if (op->init.defined()) {
145 this->VisitStmt(op->init.value());
146 }
147 this->VisitStmt(op->body);
148}

152 this->VisitExpr(op->predicate);
153 this->VisitStmt(op->block);

156class StmtMutator::Internal {
157 public:
158 /*!
159 * \brief Mutate array's element by fmutate function.
160 *
161 * \note Use extra care for copy on write setting.
162 *
163 * In particular, consider the following case of two reference chains:
164 * - strongref0 -> loop0 -> loop1 -> loop2
165 * - strongref1 -> loop3 -> loop1 -> loop2
166 *
167 * Think of the case of calling MutateArray on loop1->loop2(as const reference).
168 * When both strongref0 and strongref1 exists, the context does not allow copy
169 * on write, even though loop1 uniquely refers to loop2.
170 *
171 * \param self The pointer to the mutator.
172 * \param arr Array to be mutated, const reference is used to allow copy on write
173 * mutation in a recursive visitor.
174 * \param fmutate The mutator function.
175 * \return The mutated array, a new copy can be created.
176 */
177 template <typename T, typename F>
178 static Array<T> MutateArray(StmtMutator* self, const Array<T>& arr, F fmutate) {
179 if (self->allow_copy_on_write_ && arr.unique()) {
180 // if we allow copy on write, we can directly
181 // call the inplace mutate function.
182 const_cast<Array<T>&>(arr).MutateByApply(fmutate);
183 return arr;
184 } else {
185 bool allow_cow = false;
186 Array<T> copy = arr;
187 std::swap(allow_cow, self->allow_copy_on_write_);
188 copy.MutateByApply(fmutate);
189 std::swap(allow_cow, self->allow_copy_on_write_);
190 return copy;
191 }
192 }
193
194 static Array<IterVar> Mutate(StmtMutator* self, const Array<IterVar>& arr) {
195 auto fmutate = [self](const IterVar& iter_var) {
196 PrimExpr min = self->VisitExpr(iter_var->dom->min);
197 PrimExpr extent = self->VisitExpr(iter_var->dom->extent);
198 if (min.same_as(iter_var->dom->min) && extent.same_as(iter_var->dom->extent)) {
199 return iter_var;
200 } else {
201 return IterVar(Range(min, extent), iter_var->var, iter_var->iter_type,
202 iter_var->thread_tag);
203 }
204 };
205 return MutateArray(self, arr, fmutate);
206 }
207
208 static Array<PrimExpr> Mutate(StmtMutator* self, const Array<PrimExpr>& arr) {
209 auto fmutate = [self](const PrimExpr& e) { return self->VisitExpr(e); };
210 return MutateArray(self, arr, fmutate);
211 }
212
213 static Array<Stmt> Mutate(StmtMutator* self, const Array<Stmt>& arr) {
214 auto fmutate = [self](const Stmt& s) { return self->VisitStmt(s); };
215 return MutateArray(self, arr, fmutate);
216 }
217
218 static Array<Range> Mutate(StmtMutator* self, const Array<Range>& arr) {
219 auto fmutate = [self](const Range& r) {
220 PrimExpr min = self->VisitExpr(r->min);
221 PrimExpr extent = self->VisitExpr(r->extent);
222 if (min.same_as(r->min) && extent.same_as(r->extent)) {
223 return r;
224 } else {
225 return Range::FromMinExtent(min, extent);
226 }
227 };
228 return MutateArray(self, arr, fmutate);
229 }
230
231 static Array<BufferRegion> Mutate(StmtMutator* self, const Array<BufferRegion>& arr) {
232 auto fmutate = [self](const BufferRegion& buffer_region) {
233 Array<Range> region = Mutate(self, buffer_region->region);
234 if (region.same_as(buffer_region->region)) {
235 return buffer_region;
236 } else {
237 return BufferRegion(buffer_region->buffer, region);
238 }
239 };
240 return MutateArray(self, arr, fmutate);
241 }
242
243 static Array<MatchBufferRegion> Mutate(StmtMutator* self, const Array<MatchBufferRegion>& arr) {
244 auto fmutate = [self](const MatchBufferRegion& match_buffer_region) {
245 Array<Range> region = Mutate(self, match_buffer_region->source->region);
246 if (region.same_as(match_buffer_region->source->region)) {
247 return match_buffer_region;
248 } else {
249 return MatchBufferRegion(match_buffer_region->buffer,
250 BufferRegion(match_buffer_region->source->buffer, region));
251 }
252 };
253 return MutateArray(self, arr, fmutate);
254 }
255};

257Stmt StmtMutator::VisitStmt_(const AttrStmtNode* op) {
258 PrimExpr value = this->VisitExpr(op->value);
259 Stmt body = this->VisitStmt(op->body);
260 if (value.same_as(op->value) && body.same_as(op->body)) {
261 return GetRef<Stmt>(op);
262 } else {
263 auto n = CopyOnWrite(op);
264 n->value = std::move(value);
265 n->body = std::move(body);
266 return Stmt(n);
267 }

457// advanced visit function for seqstmt.
458Stmt StmtMutator::VisitSeqStmt_(const SeqStmtNode* op, bool flatten_before_visit,
459 std::function<Stmt(const Stmt&)> fmutate) {
460 if (flatten_before_visit) {
461 // Pass 1, check if we need to flatten.
462 bool need_flatten = false;
463 for (size_t i = 0; i < op->seq.size(); ++i) {
464 Stmt tmp = (*op)[i];
465 if (tmp.as<SeqStmtNode>()) need_flatten = true;
466 }
467 flatten_before_visit = need_flatten;
468 }
469 // function to run the visit.
470 auto frunvisit = [&](const SeqStmtNode* op) {
471 Array<Stmt> seq = fmutate != nullptr ? Internal::MutateArray(this, op->seq, fmutate)
472 : Internal::Mutate(this, op->seq);
473 if (seq.same_as(op->seq)) {
474 return GetRef<Stmt>(op);
475 } else {
476 auto n = CopyOnWrite(op);
477 n->seq = std::move(seq);
478 return Stmt(n);
479 }
480 };
481 if (flatten_before_visit) {
482 Array<Stmt> seq;
483 SeqStmt::Flattener flattener(&seq);
484 flattener(0, op->seq);
485 // NOTE: If copy on write is allowed
486 // the assignment to seq below will
487 // destruct the original seq.
488 //
489 // Such destruction removes duplicated reference
490 // count to children and still enables COW for
491 // child Stmt.
492 ObjectPtr<SeqStmtNode> n = CopyOnWrite(op);
493 n->seq = std::move(seq);
494 return frunvisit(n.operator->());
495 } else {
496 return frunvisit(op);
497 }
498}

527Stmt StmtMutator::VisitStmt_(const BlockNode* op) {
528 Array<IterVar> iter_vars = Internal::Mutate(this, op->iter_vars);
529 Array<BufferRegion> reads = Internal::Mutate(this, op->reads);
530 Array<BufferRegion> writes = Internal::Mutate(this, op->writes);
531 Array<MatchBufferRegion> match_buffers = Internal::Mutate(this, op->match_buffers);
532 Optional<Stmt> init = NullOpt;
533 if (op->init.defined()) {
534 init = VisitStmt(op->init.value());
535 }
536 Stmt body = VisitStmt(op->body);
537 if (iter_vars.same_as(op->iter_vars) && reads.same_as(op->reads) && writes.same_as(op->writes) &&
538 body.same_as(op->body) && init.same_as(op->init) &&
539 match_buffers.same_as(op->match_buffers)) {
540 return GetRef<Block>(op);
541 } else {
542 auto n = CopyOnWrite(op);
543 n->iter_vars = std::move(iter_vars);
544 n->reads = std::move(reads);
545 n->writes = std::move(writes);
546 n->body = std::move(body);
547 n->init = std::move(init);
548 n->match_buffers = std::move(match_buffers);
549 return Stmt(n);
550 }
551}

553Stmt StmtMutator::VisitStmt_(const BlockRealizeNode* op) {
554 Array<PrimExpr> v = Internal::Mutate(this, op->iter_values);
555 PrimExpr pred = this->VisitExpr(op->predicate);
556 Stmt block = this->VisitStmt(op->block);
557 if (v.same_as(op->iter_values) && pred.same_as(op->predicate) && block.same_as(op->block)) {
558 return GetRef<Stmt>(op);
559 } else {
560 auto n = CopyOnWrite(op);
561 n->iter_values = std::move(v);
562 n->predicate = std::move(pred);
563 n->block = Downcast<Block>(block);
564 return Stmt(n);
565 }

569class IRApplyVisit : public StmtExprVisitor {
570 public:
571 explicit IRApplyVisit(std::function<void(const ObjectRef&)> f) : f_(f) {}
572
573 void VisitExpr(const PrimExpr& node) final {
574 if (visited_.count(node.get()) != 0) return;
575 visited_.insert(node.get());
576 ExprVisitor::VisitExpr(node);
577 f_(node);
578 }
579
580 void VisitStmt(const Stmt& node) final {
581 if (visited_.count(node.get()) != 0) return;
582 visited_.insert(node.get());
583 StmtVisitor::VisitStmt(node);
584 f_(node);
585 }
586
587 private:
588 std::function<void(const ObjectRef&)> f_;
589 std::unordered_set<const Object*> visited_;
590};

592void PostOrderVisit(const ObjectRef& node, std::function<void(const ObjectRef&)> fvisit) {
593 if (node.as<StmtNode>()) {
594 IRApplyVisit visitor(fvisit);
595 visitor(Downcast<Stmt>(node));
596 } else {
597 IRApplyVisit visitor(fvisit);
598 visitor(Downcast<PrimExpr>(node));
599 }

602class IRTransformer final : public StmtExprMutator {
603 public:
604 IRTransformer(const runtime::PackedFunc& f_preorder, const runtime::PackedFunc& f_postorder,
605 const std::unordered_set<uint32_t>& only_enable)
606 : f_preorder_(f_preorder), f_postorder_(f_postorder), only_enable_(only_enable) {}
607
608 Stmt VisitStmt(const Stmt& stmt) final {
609 return MutateInternal<Stmt>(stmt, [this](const Stmt& s) { return this->BaseVisitStmt(s); });
610 }
611 PrimExpr VisitExpr(const PrimExpr& expr) final {
612 return MutateInternal<PrimExpr>(expr,
613 [this](const PrimExpr& e) { return this->BaseVisitExpr(e); });
614 }
615
616 private:
617 // NOTE: redirect to parent's call
618 // This is used to get around limitation of gcc-4.8
619 Stmt BaseVisitStmt(const Stmt& s) { return StmtMutator::VisitStmt(s); }
620 PrimExpr BaseVisitExpr(const PrimExpr& e) { return ExprMutator::VisitExpr(e); }
621
622 template <typename T, typename F>
623 T MutateInternal(const T& node, F fmutate) {
624 if (only_enable_.size() && !only_enable_.count(node->type_index())) {
625 return fmutate(node);
626 }
627 if (f_preorder_ != nullptr) {
628 T pre = f_preorder_(node);
629 if (pre.defined()) return pre;
630 }
631 T new_node = fmutate(node);
632 if (f_postorder_ != nullptr) {
633 T post = f_postorder_(new_node);
634 if (post.defined()) return post;
635 }
636 return new_node;
637 }
638 // The functions
639 const runtime::PackedFunc& f_preorder_;
640 const runtime::PackedFunc& f_postorder_;
641 // type indices enabled.
642 const std::unordered_set<uint32_t>& only_enable_;
643};
644
645Stmt IRTransform(Stmt ir_node, const runtime::PackedFunc& f_preorder,
646 const runtime::PackedFunc& f_postorder, Optional<Array<String>> only_enable) {
647 std::unordered_set<uint32_t> only_type_index;
648 if (only_enable.defined()) {
649 for (auto s : only_enable.value()) {
650 only_type_index.insert(Object::TypeKey2Index(s.c_str()));
651 }
652 }
653 IRTransformer transform(f_preorder, f_postorder, only_type_index);
654 return transform(std::move(ir_node));
655}

657class IRSubstitute : public StmtExprMutator {
658 public:
659 explicit IRSubstitute(std::function<Optional<PrimExpr>(const Var&)> vmap) : vmap_(vmap) {}
660
661 PrimExpr VisitExpr_(const VarNode* op) final {
662 Var var = GetRef<Var>(op);
663 auto ret = vmap_(var);
664 if (ret.defined()) {
665 // Allow substitution of void variables with any expression. The TVM script parser
666 // uses void variables for lambda parameters (since exact types are not known yet).
667 if (!var.dtype().is_void()) {
668 PrimExpr ret_ex = Downcast<PrimExpr>(ret.value());
669 ICHECK(ret_ex.dtype() == var.dtype()) << "substituting " << var << ":" << var.dtype()
670 << " -> " << ret_ex << ":" << ret_ex.dtype();
671 }
672 return ret.value();
673 }
674 return std::move(var);
675 }
676
677 PrimExpr VisitExpr_(const LoadNode* op) final {
678 LOG(FATAL) << "Unexpected use of deprecated LoadNode. Please use BufferLoadNode instead.";
679 return PrimExpr();
680 }
681
682 Stmt VisitStmt_(const StoreNode* op) final {
683 LOG(FATAL) << "Unexpected use of deprecated StoreNode. Please use BufferStoreNode instead.";
684 return Stmt();
685 }
686
687 PrimExpr VisitExpr_(const BufferLoadNode* op) final {
688 auto node = Downcast<BufferLoad>(StmtExprMutator::VisitExpr_(op));
689 return VisitBufferAccess(std::move(node));
690 }
691
692 Stmt VisitStmt_(const BufferStoreNode* op) final {
693 auto node = Downcast<BufferStore>(StmtExprMutator::VisitStmt_(op));
694 return VisitBufferAccess(std::move(node));
695 }
696
697 template <typename Node>
698 Node VisitBufferAccess(Node node) {
699 Buffer new_buf = GetRemappedBuffer(node->buffer);
700
701 if (!new_buf.same_as(node->buffer)) {
702 auto writer = node.CopyOnWrite();
703 writer->buffer = new_buf;
704 }
705
706 return node;
707 }
708
709 Buffer GetRemappedBuffer(Buffer buf) {
710 auto key = buf.get();
711 auto it = buf_remap_.find(key);
712 if (it != buf_remap_.end()) {
713 return it->second;
714 }
715
716 auto new_buffer_var = vmap_(buf->data);
717 if (new_buffer_var.defined() && !new_buffer_var.value().same_as(buf->data)) {
718 auto writer = buf.CopyOnWrite();
719 writer->data = Downcast<Var>(new_buffer_var);
720 }
721
722 buf_remap_[key] = buf;
723 return buf;
724 }
725
726 Stmt VisitStmt_(const AttrStmtNode* op) final {
727 Stmt ret = StmtExprMutator::VisitStmt_(op);
728 op = ret.as<AttrStmtNode>();
729 // remap var node in attr
730 if (const auto* var_node = op->node.as<VarNode>()) {
731 if (auto mapped_var = vmap_(GetRef<Var>(var_node))) {
732 return AttrStmt(mapped_var, op->attr_key, op->value, op->body);
733 }
734 }
735 return ret;
736 }
737
738 private:
739 // Caller provided function that defines the variables to be remapped.
740 std::function<Optional<PrimExpr>(const Var&)> vmap_;
741
742 /* \brief Generated map to track buffers being remapped.
743 *
744 * If a `Var BufferNode::data` is remapped, then all buffers
745 * containing that data pointer should also be remapped. This map
746 * is used to track buffer modifications, and ensure all instances
747 * of a buffer are replaced by the same modified buffer object.
748 */
749 std::unordered_map<const BufferNode*, Buffer> buf_remap_;
750};
751
752Stmt Substitute(Stmt stmt, std::function<Optional<PrimExpr>(const Var&)> vmap) {
753 return IRSubstitute(vmap)(std::move(stmt));
754}
755
756PrimExpr Substitute(PrimExpr expr, std::function<Optional<PrimExpr>(const Var&)> vmap) {
757 return IRSubstitute(vmap)(std::move(expr));
758}
759
760Array<Range> Substitute(const Array<Range>& region, const Map<Var, PrimExpr>& vmap) {
761 Array<Range> result;
762 result.reserve(region.size());
763 for (const Range& range : region) {
764 PrimExpr min = Substitute(range->min, vmap);
765 PrimExpr extent = Substitute(range->extent, vmap);
766 result.push_back(Range::FromMinExtent(std::move(min), std::move(extent)));
767 }
768 return result;
769}

771void PreOrderVisit(const ObjectRef& stmt_or_expr,
772 const std::function<bool(const ObjectRef&)>& fvisit) {
773 class PreOrderVisitor : public StmtExprVisitor {
774 public:
775 explicit PreOrderVisitor(const std::function<bool(const ObjectRef&)>& f) : f_(f) {}
776
777 private:
778 void VisitExpr(const PrimExpr& expr) final {
779 const PrimExprNode* p_expr = expr.get();
780 if (visited_.count(p_expr) == 0) {
781 visited_.insert(p_expr);
782 if (f_(expr)) {
783 ExprVisitor::VisitExpr(expr);
784 }
785 }
786 }
787
788 void VisitStmt(const Stmt& stmt) final {
789 const StmtNode* p_stmt = stmt.get();
790 if (visited_.count(p_stmt) == 0) {
791 visited_.insert(p_stmt);
792 if (f_(stmt)) {
793 StmtVisitor::VisitStmt(stmt);
794 }
795 }
796 }
797
798 const std::function<bool(const ObjectRef&)>& f_;
799 std::unordered_set<const Object*> visited_;
800 };
801
802 PreOrderVisitor visitor(fvisit);
803 if (const auto* stmt = stmt_or_expr.as<StmtNode>()) {
804 visitor(GetRef<Stmt>(stmt));
805 } else if (const auto* expr = stmt_or_expr.as<PrimExprNode>()) {
806 visitor(GetRef<PrimExpr>(expr));
807 } else {
808 LOG(FATAL) << "InternalError: PreOrderVisit does not accept object with type: "
809 << stmt_or_expr->GetTypeKey();
810 }
811}

827 } else {
828 return Substitute(Downcast<PrimExpr>(node), vmap);
829 }
