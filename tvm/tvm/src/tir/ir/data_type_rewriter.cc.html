34Stmt DataTypeLegalizer::VisitStmt_(const ForNode* op) {
35 Stmt s = StmtExprMutator::VisitStmt_(op);
36 op = s.as<ForNode>();
37 ICHECK(op != nullptr) << "Expected type to be ForNode, but get " << s->GetTypeKey();
38 PrimExpr e = VisitExpr(op->loop_var);
39 Var var = Downcast<Var>(e);
40 return For(var, cast(var.dtype(), op->min), cast(var.dtype(), op->extent), op->kind, op->body,
41 op->thread_binding, op->annotations);
42}

44Stmt DataTypeLegalizer::VisitStmt_(const BlockRealizeNode* op) {
45 BlockRealize realize = Downcast<BlockRealize>(StmtExprMutator::VisitStmt_(op));
46 Array<PrimExpr> new_iter_values;
47 bool changed = false;
48 for (int i = 0; i < static_cast<int>(op->iter_values.size()); ++i) {
49 auto dtype = realize->block->iter_vars[i]->var->dtype;
50 if (op->iter_values[i]->dtype != dtype) {
51 new_iter_values.push_back(cast(dtype, realize->iter_values[i]));
52 changed = true;
53 } else {
54 new_iter_values.push_back(realize->iter_values[i]);
55 }
56 }
57 if (changed) {
58 realize.CopyOnWrite()->iter_values = std::move(new_iter_values);
59 }
60 return std::move(realize);
61}

63Stmt DataTypeLegalizer::VisitStmt_(const BlockNode* op) {
64 Block new_block = Downcast<Block>(StmtExprMutator::VisitStmt_(op));
65 Array<IterVar> new_iter_vars = MutateArray(new_block->iter_vars, [/*this*/](const IterVar& iter) {
66 auto dtype = iter->var.dtype();
67 if (iter->dom->min->dtype != dtype || iter->dom->extent->dtype != dtype) {
68 IterVar new_iter = iter;
69 new_iter.CopyOnWrite()->dom =
70 Range(cast(dtype, iter->dom->min), cast(dtype, iter->dom->extent));
71 return new_iter;
72 } else {
73 return iter;
74 }
75 });
76 if (!op->iter_vars.same_as(new_iter_vars)) {
77 new_block.CopyOnWrite()->iter_vars = std::move(new_iter_vars);
78 }
79 return std::move(new_block);
80}

82Stmt DataTypeLegalizer::VisitStmt_(const AttrStmtNode* op) {
83 if (op->attr_key == attr::thread_extent || op->attr_key == attr::virtual_thread) {
84 Stmt s = StmtExprMutator::VisitStmt_(op);
85 op = s.as<AttrStmtNode>();
86 ICHECK(op != nullptr) << "Expected type to be AttrStmtNode"
87 << ", but get " << s->GetTypeKey();
88 const IterVarNode* iv = op->node.as<IterVarNode>();
89 ICHECK(iv != nullptr) << "Expected type to be IterVarNode"
90 << ", but get " << op->node->GetTypeKey();
91 PrimExpr e = VisitExpr(iv->var);
92 Var var = Downcast<Var>(e);
93 if (ivmap_.find(iv) == ivmap_.end()) {
94 Range dom = iv->dom;
95 if (dom.defined()) {
96 PrimExpr extend = dom->extent;
97 ICHECK(extend.dtype().is_int() && var.dtype().is_int());
98 if (var.dtype().bits() != extend.dtype().bits()) {
99 DataType dtype = var.dtype();
100 dom = Range(cast(dtype, dom->min), cast(dtype, extend), dom->span);
101 }
102 }
103 ivmap_[iv] = IterVar(dom, var, iv->iter_type, iv->thread_tag);
104 }
105 return AttrStmt(ivmap_[iv], op->attr_key, cast(var.dtype(), op->value), op->body);
106 }
107 return StmtExprMutator::VisitStmt_(op);
108}

110PrimExpr DataTypeLegalizer::VisitExpr_(const SelectNode* op) {
111 PrimExpr condition = this->VisitExpr(op->condition);
112 PrimExpr true_value = this->VisitExpr(op->true_value);
113 PrimExpr false_value = this->VisitExpr(op->false_value);
114 if (condition.same_as(op->condition) && true_value.same_as(op->true_value) &&
115 false_value.same_as(op->false_value) && true_value.dtype() == false_value.dtype()) {
116 return GetRef<PrimExpr>(op);
117 } else {
118 int bits = std::max(true_value.dtype().bits(), false_value.dtype().bits());
119 DataType dtype = true_value.dtype().with_bits(bits);
120 if (true_value.dtype() != dtype) true_value = cast(dtype, true_value);
121 if (false_value.dtype() != dtype) false_value = cast(dtype, false_value);
122 return Select(condition, true_value, false_value);
123 }
124}

126PrimExpr DataTypeLegalizer::VisitExpr_(const RampNode* op) {
127 PrimExpr base = VisitExpr(op->base);
128 PrimExpr stride = VisitExpr(op->stride);
129 if (base.same_as(op->base) && stride.same_as(op->stride) && base.dtype() == stride.dtype()) {
130 return GetRef<PrimExpr>(op);
131 } else {
132 ICHECK(base.dtype().is_int() && stride.dtype().is_int());
133 int bits = std::max(base.dtype().bits(), stride.dtype().bits());
134 DataType dtype = base.dtype().with_bits(bits);
135 if (base.dtype() != dtype) base = cast(dtype, base);
136 if (stride.dtype() != dtype) stride = cast(dtype, stride);
137 return Ramp(base, stride, op->lanes);
138 }
139}

146 PrimExpr DataTypeLegalizer::VisitExpr_(const OP* op) { \
147 PrimExpr a = this->VisitExpr(op->a); \
148 PrimExpr b = this->VisitExpr(op->b); \
149 if (op->a.same_as(a) && op->b.same_as(b) && a.dtype() == b.dtype()) { \
150 return GetRef<PrimExpr>(op); \

174PrimExpr DataTypeLegalizer::VisitExpr_(const CallNode* op) {
175 PrimExpr e = StmtExprMutator::VisitExpr_(op);
176 op = e.as<CallNode>();
177 static const Op& builtin_pow_ = Op::Get("tir.pow");
178 ICHECK(op != nullptr) << "Expected type to be CallNode"
179 << ", but get " << e->GetTypeKey();
180 if (op->op.same_as(builtin::shift_right())) {
181 return op->args[0] >> op->args[1];
182 } else if (op->op.same_as(builtin::shift_left())) {
183 return op->args[0] << op->args[1];
184 } else if (op->op.same_as(builtin::bitwise_and())) {
185 return op->args[0] & op->args[1];
186 } else if (op->op.same_as(builtin::bitwise_or())) {
187 return op->args[0] | op->args[1];
188 } else if (op->op.same_as(builtin::bitwise_xor())) {
189 return op->args[0] ^ op->args[1];
190 } else if (op->op.same_as(builtin_pow_)) {
191 return pow(op->args[0], op->args[1]);
192 } else if (op->op.same_as(builtin::if_then_else())) {
193 return if_then_else(op->args[0], op->args[1], op->args[2]);
194 }
195 return e;
196}

198Stmt IndexDataTypeRewriter::VisitStmt_(const AllocateNode* op) {
199 bool is_enabled = is_enabled_;
200 is_enabled_ = true;
201 auto new_extents = op->extents.Map([this](const PrimExpr& e) { return this->VisitExpr(e); });
202 auto new_cond = VisitExpr(op->condition);
203 is_enabled_ = is_enabled;
204 auto new_body = this->VisitStmt(op->body);
205 if (!new_extents.same_as(op->extents) || !new_cond.same_as(op->condition) ||
206 !new_body.same_as(op->body)) {
207 Allocate new_allocate = GetRef<Allocate>(op);
208 auto* n = new_allocate.CopyOnWrite();
209 n->extents = std::move(new_extents);
210 n->condition = std::move(new_cond);
211 n->body = std::move(new_body);
212 return std::move(new_allocate);
213 } else {
214 return GetRef<Stmt>(op);
215 }
216}

227Stmt IndexDataTypeRewriter::VisitStmt_(const BlockRealizeNode* op) {
228 bool is_condition = is_condition_;
229 is_condition_ = true;
230 auto new_predicate = VisitExpr(op->predicate);
231 is_condition_ = is_condition;
232
233 bool is_enabled = is_enabled_;
234 is_enabled_ = true;
235 auto new_iter_values =
236 op->iter_values.Map([this](const PrimExpr& e) { return this->VisitExpr(e); });
237 is_enabled_ = is_enabled;
238 Block new_body = Downcast<Block>(this->VisitStmt(op->block));
239 if (!new_predicate.same_as(op->predicate) || !new_iter_values.same_as(op->iter_values) ||
240 !new_body.same_as(op->block)) {
241 BlockRealize new_block_realize = GetRef<BlockRealize>(op);
242 auto* n = new_block_realize.CopyOnWrite();
243 n->predicate = std::move(new_predicate);
244 n->iter_values = std::move(new_iter_values);
245 n->block = std::move(new_body);
246 return std::move(new_block_realize);
247 } else {
248 return GetRef<Stmt>(op);
249 }
250}

252Stmt IndexDataTypeRewriter::VisitStmt_(const BlockNode* op) {
253 Array<Buffer> new_alloc_buffers =
254 op->alloc_buffers.Map([this](const Buffer& buffer) { return this->VisitBuffer(buffer); });
255 Array<MatchBufferRegion> new_match_buffers =
256 op->match_buffers.Map([this](const MatchBufferRegion& match_buffer_region) {
257 Buffer new_buffer = this->VisitBuffer(match_buffer_region->buffer);
258 BufferRegion new_buffer_region = this->VisitBufferRegion(match_buffer_region->source);
259 if (!new_buffer.same_as(match_buffer_region->buffer) ||
260 !new_buffer_region.same_as(match_buffer_region->source)) {
261 return MatchBufferRegion(new_buffer, new_buffer_region);
262 } else {
263 return match_buffer_region;
264 }
265 });
266 Array<BufferRegion> new_reads = op->reads.Map(
267 [this](const BufferRegion& buffer_region) { return this->VisitBufferRegion(buffer_region); });
268 Array<BufferRegion> new_writes = op->writes.Map(
269 [this](const BufferRegion& buffer_region) { return this->VisitBufferRegion(buffer_region); });
270 Array<IterVar> new_iter_vars =
271 op->iter_vars.Map([this](const IterVar& iter_var) { return this->VisitIterVar(iter_var); });
272 Optional<Stmt> new_init = NullOpt;
273 if (op->init.defined()) {
274 new_init = this->VisitStmt(op->init.value());
275 }
276 Map<String, ObjectRef> new_annotations = VisitBlockAnnotations(op->annotations);
277 Stmt new_body = this->VisitStmt(op->body);
278
279 if (!new_init.same_as(op->init) || !new_body.same_as(op->body) ||
280 !new_alloc_buffers.same_as(op->alloc_buffers) ||
281 !new_match_buffers.same_as(op->match_buffers) || !new_reads.same_as(op->reads) ||
282 !new_writes.same_as(op->writes) || new_iter_vars.same_as(op->iter_vars) ||
283 !new_annotations.same_as(op->annotations)) {
284 Block new_block = GetRef<Block>(op);
285 BlockNode* n = new_block.CopyOnWrite();
286 n->alloc_buffers = std::move(new_alloc_buffers);
287 n->match_buffers = std::move(new_match_buffers);
288 n->reads = std::move(new_reads);
289 n->writes = std::move(new_writes);
290 n->iter_vars = std::move(new_iter_vars);
291 n->init = std::move(new_init);
292 n->annotations = std::move(new_annotations);
293 n->body = std::move(new_body);
294 return std::move(new_block);
295 }
296 return GetRef<Stmt>(op);
297}

299Map<String, ObjectRef> IndexDataTypeRewriter::VisitBlockAnnotations(
300 const Map<String, ObjectRef>& annotations) {
301 auto new_annotations = annotations;
302
303 std::function<ObjectRef(const ObjectRef&)> f_mutate_obj =
304 [this, &f_mutate_obj](const ObjectRef& obj) -> ObjectRef {
305 if (!obj.defined()) {
306 return obj;
307 }
308 if (obj->IsInstance<BufferNode>()) {
309 Buffer buffer = Downcast<Buffer>(obj);
310 if (Buffer new_buffer = GetRemappedBuffer(buffer); !new_buffer.same_as(buffer)) {
311 return new_buffer;
312 }
313 } else if (obj->IsInstance<ArrayNode>()) {
314 return Downcast<Array<ObjectRef>>(obj).Map(f_mutate_obj);
315 }
316 return obj;
317 };
318 for (const auto& [key, value] : annotations) {
319 auto new_value = f_mutate_obj(value);
320 if (!new_value.same_as(value)) {
321 new_annotations.Set(key, new_value);
322 }
323 }
324 return new_annotations;
325}

327Buffer IndexDataTypeRewriter::GetRemappedBuffer(const Buffer& buffer) {
328 if (auto it = buffer_remap_.find(buffer); it != buffer_remap_.end()) {
329 return (*it).second;
330 }
331 return buffer;
332}

334IterVar IndexDataTypeRewriter::VisitIterVar(const IterVar& iter_var) {
335 bool is_enabled = is_enabled_;
336 is_enabled_ = true;
337 Var new_var = Downcast<Var>(VisitExpr(iter_var->var));
338 PrimExpr min = VisitExpr(iter_var->dom->min);
339 PrimExpr extent = VisitExpr(iter_var->dom->extent);
340 is_enabled_ = is_enabled;
341 if (!new_var.same_as(iter_var->var) || !min.same_as(iter_var->dom->min) ||
342 !extent.same_as(iter_var->dom->extent)) {
343 IterVar new_iter_var = iter_var;
344 IterVarNode* n = new_iter_var.CopyOnWrite();
345 n->var = std::move(new_var);
346 n->dom = Range(min, extent);
347 return new_iter_var;
348 }
349 return iter_var;
350}

352Buffer IndexDataTypeRewriter::VisitBuffer(const Buffer& buffer) {
353 bool is_enabled = is_enabled_;
354
355 is_enabled_ = true;
356 Array<PrimExpr> new_shape =
357 buffer->shape.Map([&](const PrimExpr& e) { return this->VisitExpr(e); });
358 Array<PrimExpr> new_strides =
359 buffer->strides.Map([&](const PrimExpr& e) { return this->VisitExpr(e); });
360 auto new_elem_offset = VisitExpr(buffer->elem_offset);
361 is_enabled_ = is_enabled;
362
363 if (!buffer->shape.same_as(new_shape) || !buffer->strides.same_as(new_strides) ||
364 !buffer->elem_offset.same_as(new_elem_offset)) {
365 Buffer new_buffer = buffer;
366 BufferNode* new_buffer_node = new_buffer.CopyOnWrite();
367 new_buffer_node->shape = std::move(new_shape);
368 new_buffer_node->strides = std::move(new_strides);
369 new_buffer_node->elem_offset = std::move(new_elem_offset);
370 buffer_remap_.Set(buffer, new_buffer);
371 return new_buffer;
372 } else {
373 return buffer;
374 }
375}

377BufferRegion IndexDataTypeRewriter::VisitBufferRegion(const BufferRegion& buffer_region) {
378 Buffer remapped_buffer = GetRemappedBuffer(buffer_region->buffer);
379
380 bool is_enabled = is_enabled_;
381 is_enabled_ = true;
382 auto new_region = buffer_region->region.Map([&](const Range& range) {
383 return Range::FromMinExtent(this->VisitExpr(range->min), this->VisitExpr(range->extent));
384 });
385 is_enabled_ = is_enabled;
386
387 if (!remapped_buffer.same_as(buffer_region->buffer) ||
388 !new_region.same_as(buffer_region->region)) {
389 return BufferRegion(remapped_buffer, new_region);
390 } else {
391 return buffer_region;
392 }
393}

395Stmt IndexDataTypeRewriter::VisitStmt_(const BufferStoreNode* op) {
396 BufferStore store = GetRef<BufferStore>(op);
397
398 Buffer new_buffer = GetRemappedBuffer(op->buffer);
399 auto value = this->VisitExpr(op->value);
400 auto indices = VisitIndices(op->indices);
401
402 if (!new_buffer.same_as(op->buffer) || !value.same_as(op->value) ||
403 !indices.same_as(op->indices)) {
404 auto writer = store.CopyOnWrite();
405 writer->buffer = new_buffer;
406 writer->value = value;
407 writer->indices = indices;
408 }
409
410 return std::move(store);
411}

428Array<PrimExpr> IndexDataTypeRewriter::VisitIndices(Array<PrimExpr> indices) {
429 bool is_enabled = is_enabled_;
430 is_enabled_ = true;
431
432 auto fmutate = [this](const PrimExpr& index) { return this->VisitExpr(index); };
433 indices.MutateByApply(fmutate);
434
435 is_enabled_ = is_enabled;
436
437 return indices;
438}

440Stmt IndexDataTypeRewriter::VisitStmt_(const IfThenElseNode* op) {
441 bool is_condition = is_condition_;
442 is_condition_ = true;
443 PrimExpr cond = VisitExpr(op->condition);
444 is_condition_ = is_condition;
445
446 Stmt then_case = VisitStmt(op->then_case);
447 Optional<Stmt> else_case =
448 op->else_case.defined() ? Optional<Stmt>{VisitStmt(op->else_case.value())} : NullOpt;
449 if (!cond.same_as(op->condition) || !then_case.same_as(op->then_case) ||
450 !else_case.same_as(op->else_case)) {
451 IfThenElse new_stmt = GetRef<IfThenElse>(op);
452 auto* n = new_stmt.CopyOnWrite();
453 n->condition = std::move(cond);
454 n->then_case = std::move(then_case);
455 n->else_case = std::move(else_case);
456 return std::move(new_stmt);
457 }
458 return GetRef<Stmt>(op);
459}

461Stmt IndexDataTypeRewriter::VisitStmt_(const ForNode* op) {
462 bool is_enabled = is_enabled_;
463 is_enabled_ = true;
464 Var new_loop_var = Downcast<Var>(VisitExpr(op->loop_var));
465 PrimExpr min = VisitExpr(op->min);
466 PrimExpr extent = VisitExpr(op->extent);
467 is_enabled_ = is_enabled;
468
469 Stmt new_body = VisitStmt(op->body);
470
471 if (!new_loop_var.same_as(op->loop_var) || !min.same_as(op->min) || !extent.same_as(op->extent) ||
472 !new_body.same_as(op->body)) {
473 For new_for = GetRef<For>(op);
474 auto* n = new_for.CopyOnWrite();
475 n->loop_var = new_loop_var;
476 n->min = cast(new_loop_var.dtype(), min);
477 n->extent = cast(new_loop_var.dtype(), extent);
478 n->body = new_body;
479 return std::move(new_for);
480 } else {
481 return GetRef<Stmt>(op);
482 }
483}

501PrimExpr IndexDataTypeRewriter::VisitExpr_(const CallNode* op) {
502 // handle if_then_else condition
503 if (op->op.same_as(builtin::if_then_else())) {
504 bool is_condition = is_condition_;
505 is_condition_ = true;
506 PrimExpr cond = VisitExpr(op->args[0]);
507 is_condition_ = is_condition;
508 return if_then_else(cond, VisitExpr(op->args[1]), VisitExpr(op->args[2]));
509 }
510 return Parent::VisitExpr_(op);
511}

515IndexDataTypeNormalizer::IndexDataTypeNormalizer(DataType target_data_type)
516 : target_data_type_(std::move(target_data_type)) {}
517
518PrimFunc IndexDataTypeNormalizer::Rewrite(PrimFunc func) {
519 Map<Var, Buffer> new_buffer_map = func->buffer_map;
520 for (const auto& [var, buffer] : func->buffer_map) {
521 new_buffer_map.Set(var, VisitBuffer(buffer));
522 }
523 PrimFuncNode* new_func = func.CopyOnWrite();
524 new_func->buffer_map = std::move(new_buffer_map);
525 new_func->body = VisitStmt(std::move(new_func->body));
526 return func;
527}

529PrimExpr IndexDataTypeNormalizer::VisitExpr_(const IntImmNode* op) {
530 if (is_enabled_) {
531 ICHECK_LE(op->value, Downcast<Integer>(max_value(target_data_type_))->value);
532 return cast(target_data_type_, GetRef<IntImm>(op));
533 }
534 return GetRef<IntImm>(op);
535}

537PrimExpr IndexDataTypeNormalizer::VisitExpr_(const VarNode* op) {
538 if (auto it = var_remap_.find(GetRef<Var>(op)); it != var_remap_.end()) {
539 return (*it).second;
540 }
541 if (is_enabled_ && op->dtype != target_data_type_) {
542 Var new_var = GetRef<Var>(op).copy_with_dtype(target_data_type_);
543 var_remap_.Set(GetRef<Var>(op), new_var);
544 return std::move(new_var);
545 }
546 return GetRef<PrimExpr>(op);
547}

549PrimExpr IndexDataTypeNormalizer::VisitExpr_(const CastNode* op) {
550 if (is_enabled_) {
551 PrimExpr value = IndexDataTypeNormalizer::VisitExpr(op->value);
552 return value->dtype == target_data_type_ ? value : Cast(target_data_type_, value);
553 }
554 return IndexDataTypeRewriter::VisitExpr_(op);
555}
