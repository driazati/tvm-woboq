86void ExprVisitor::VisitExpr_(const ReduceNode* op) {
87 VisitArray(op->axis, [this](const IterVar& r) {
88 this->VisitExpr(r->dom->min);
89 this->VisitExpr(r->dom->extent);
90 });
91 VisitArray(op->source, [this](const PrimExpr& e) { this->VisitExpr(e); });
92 if (!op->init.empty()) {
93 VisitArray(op->init, [this](const PrimExpr& e) { this->VisitExpr(e); });
94 }
95 this->VisitExpr(op->condition);
96}

133PrimExpr ExprMutator::VisitExpr_(const BufferLoadNode* op) {
134 auto fmutate = [this](const PrimExpr& e) { return this->VisitExpr(e); };
135 Array<PrimExpr> indices = MutateArray(op->indices, fmutate);
136 if (indices.same_as(op->indices)) {
137 return GetRef<PrimExpr>(op);
138 } else {
139 return BufferLoad(op->buffer, indices);
140 }
141}

143PrimExpr ExprMutator::VisitExpr_(const ProducerLoadNode* op) {
144 auto fmutate = [this](const PrimExpr& e) { return this->VisitExpr(e); };
145 Array<PrimExpr> indices = MutateArray(op->indices, fmutate);
146 if (indices.same_as(op->indices)) {
147 return GetRef<PrimExpr>(op);
148 } else {
149 return ProducerLoad(op->producer, indices);
150 }
151}

153PrimExpr ExprMutator::VisitExpr_(const LetNode* op) {
154 PrimExpr value = this->VisitExpr(op->value);
155 PrimExpr body = this->VisitExpr(op->body);
156 if (value.same_as(op->value) && body.same_as(op->body)) {
157 return GetRef<PrimExpr>(op);
158 } else {
159 return Let(op->var, value, body);
160 }
161}
162
163PrimExpr ExprMutator::VisitExpr_(const CallNode* op) {
164 auto fmutate = [this](const PrimExpr& e) { return this->VisitExpr(e); };
165 Array<PrimExpr> args = MutateArray(op->args, fmutate);
166
167 if (args.same_as(op->args)) {
168 return GetRef<PrimExpr>(op);
169 } else {
170 return Call(op->dtype, op->op, args);
171 }
172}

210PrimExpr ExprMutator::VisitExpr_(const ReduceNode* op) {
211 auto fitervar = [this](const IterVar& v) {
212 Range r = v->dom;
213 PrimExpr min = this->VisitExpr(r->min);
214 PrimExpr extent = this->VisitExpr(r->extent);
215 if (min.same_as(r->min) && extent.same_as(r->extent)) {
216 return v;
217 } else {
218 return IterVar(Range::FromMinExtent(min, extent), v->var, v->iter_type, v->thread_tag);
219 }
220 };
221 Array<IterVar> axis = MutateArray(op->axis, fitervar);
222
223 auto fexpr = [this](const PrimExpr& e) { return this->VisitExpr(e); };
224 Array<PrimExpr> source = MutateArray(op->source, fexpr);
225 Array<PrimExpr> init = MutateArray(op->init, fexpr);
226
227 PrimExpr condition = this->VisitExpr(op->condition);
228
229 if (axis.same_as(op->axis) && source.same_as(op->source) && condition.same_as(op->condition) &&
230 init.same_as(op->init)) {
231 return GetRef<PrimExpr>(op);
232 } else {
233 return Reduce(op->combiner, source, axis, condition, op->value_index, init);
234 }
235}

237PrimExpr ExprMutator::VisitExpr_(const CastNode* op) {
238 PrimExpr value = this->VisitExpr(op->value);
239 if (value.same_as(op->value)) {
240 return GetRef<PrimExpr>(op);
241 } else {
242 return Cast(op->dtype, value);
243 }
244}

246PrimExpr ExprMutator::VisitExpr_(const NotNode* op) {
247 PrimExpr a = this->VisitExpr(op->a);
248 if (a.same_as(op->a)) {
249 return GetRef<PrimExpr>(op);
250 } else {
251 return Not(a);
252 }
253}

255PrimExpr ExprMutator::VisitExpr_(const SelectNode* op) {
256 PrimExpr condition = this->VisitExpr(op->condition);
257 PrimExpr true_value = this->VisitExpr(op->true_value);
258 PrimExpr false_value = this->VisitExpr(op->false_value);
259 if (condition.same_as(op->condition) && true_value.same_as(op->true_value) &&
260 false_value.same_as(op->false_value)) {
261 return GetRef<PrimExpr>(op);
262 } else {
263 return Select(condition, true_value, false_value);
264 }
265}
266
267PrimExpr ExprMutator::VisitExpr_(const RampNode* op) {
268 PrimExpr base = this->VisitExpr(op->base);
269 PrimExpr stride = this->VisitExpr(op->stride);
270 if (base.same_as(op->base) && stride.same_as(op->stride)) {
271 return GetRef<PrimExpr>(op);
272 } else {
273 return Ramp(base, stride, op->lanes);
274 }
275}

277PrimExpr ExprMutator::VisitExpr_(const BroadcastNode* op) {
278 PrimExpr value = this->VisitExpr(op->value);
279 if (value.same_as(op->value)) {
280 return GetRef<PrimExpr>(op);
281 } else {
282 return Broadcast(value, op->lanes);
283 }
284}

286PrimExpr ExprMutator::VisitExpr_(const ShuffleNode* op) {
287 auto fexpr = [this](const PrimExpr& e) { return this->VisitExpr(e); };
288 auto vectors = MutateArray(op->vectors, fexpr);
289 if (vectors.same_as(op->vectors)) {
290 return GetRef<PrimExpr>(op);
291 } else {
292 return Shuffle(vectors, op->indices);
293 }
