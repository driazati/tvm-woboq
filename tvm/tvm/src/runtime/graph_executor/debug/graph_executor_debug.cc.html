40/*!
41 * \brief Graph executor with debug .
42 *
43 * This is the extension of GraphExecutor class used for debugging
44 * TVM runtime PackedFunc API.
45 */
46class GraphExecutorDebug : public GraphExecutor {
47 public:
48 /*!
49 * \brief Run each operation in the graph and get the time per op for all ops.
50 * \param number The number of times to run this function for taking average.
51 * \param repeat The number of times to repeat the measurement.
52 * In total, the function will be invoked (1 + number x repeat) times,
53 * where the first one is warmed up and will be discarded in case
54 * there is lazy initialization.
55 * \param min_repeat_ms The minimum duration of one `repeat` in milliseconds.
56 * By default, one `repeat` contains `number` runs. If this parameter is set,
57 * the parameters `number` will be dynamically adjusted to meet the
58 * minimum duration requirement of one `repeat`.
59 * \param limit_zero_time_iterations The maximum number of repeats when
60 * measured time is equal to 0. It helps to avoid hanging during
61 * measurements.
62 * \param cooldown_interval_ms The cooldown interval in milliseconds between the number of repeats
63 * defined by `repeats_to_cooldown`.
64 * \param repeats_to_cooldown The number of repeats before the
65 * cooldown is activated.
66 * \return Returns a string with an encoded byte array. Where the first 8 bytes are int64_t
67 * representing the number of layers. Next the encoded real numbers are float32_t in the number of
68 * repeat multiplied by the number of layers.
69 */
70 std::string RunIndividual(int number, int repeat, int min_repeat_ms,
71 int limit_zero_time_iterations, int cooldown_interval_ms,
72 int repeats_to_cooldown) {
73 // warmup run
74 GraphExecutor::Run();
75 std::string tkey = module_->type_key();
76 std::vector<std::vector<double>> time_sec_per_op(op_execs_.size());
77 if (tkey == "rpc") {
78 // RPC modules rely on remote timing which implements the logic from the else branch.
79 for (size_t index = 0; index < op_execs_.size(); ++index) {
80 time_sec_per_op[index] =
81 RunOpRPC(index, number, repeat, min_repeat_ms, limit_zero_time_iterations,
82 cooldown_interval_ms, repeats_to_cooldown);
83 }
84 } else {
85 int op = 0;
86 for (size_t index = 0; index < op_execs_.size(); ++index) {
87 std::string result_str =
88 RunIndividualNode(index, number, repeat, min_repeat_ms, limit_zero_time_iterations,
89 cooldown_interval_ms, repeats_to_cooldown);
90 const double* blob_ptr = reinterpret_cast<const double*>(result_str.data());
91 for (int i = 0; i < repeat; ++i, ++blob_ptr) {
92 time_sec_per_op[index].push_back(*blob_ptr);
93 }
94 if (op_execs_[index]) {
95 LOG(INFO) << "Op #" << op << " " << GetNodeName(index) << ":";
96 for (size_t cur_repeat = 0; cur_repeat < time_sec_per_op[index].size(); cur_repeat++) {
97 const auto& data = time_sec_per_op[index][cur_repeat];
98 LOG(INFO) << "Iteration: " << cur_repeat << ": " << (data * 1e6) << " us/iter";
99 }
100 ++op;
101 }
102 }
103 }
104
105 std::ostringstream os;
106 int64_t size = time_sec_per_op.size();
107 os.write(reinterpret_cast<char*>(&size), sizeof(int64_t));
108 for (size_t index = 0; index < time_sec_per_op.size(); ++index) {
109 for (auto& repeat_data : time_sec_per_op[index]) {
110 // To have good behavior when calculating total time, etc.
111 double data = std::isnan(repeat_data) ? 0 : repeat_data;
112 os.write(reinterpret_cast<char*>(&data), sizeof(double));
113 }
114 }
115 return os.str();
116 }
117
118 std::string RunIndividualNode(int node_index, int number, int repeat, int min_repeat_ms,
119 int limit_zero_time_iterations, int cooldown_interval_ms,
120 int repeats_to_cooldown) {
121 std::string tkey = module_->type_key();
122
123 if (tkey == "rpc") {
124 LOG(FATAL) << "RPC measurements should not use RunIndividualNode!";
125 }
126
127 if (!op_execs_[node_index]) {
128 // don't return anything...
129 std::ostringstream os;
130 double zero = 0;
131 for (int i = 0; i < repeat; ++i) {
132 os.write(reinterpret_cast<char*>(&zero), sizeof(double));
133 }
134 return os.str();
135 }
136
137 // assume host runs things which is first device
138 Device& d = devices_[0];
139 PackedFunc time_evaluator = profiling::WrapTimeEvaluator(
140 TypedPackedFunc<void()>([this, node_index]() { this->RunOpHost(node_index); }), d, number,
141 repeat, min_repeat_ms, limit_zero_time_iterations, cooldown_interval_ms,
142 repeats_to_cooldown);
143 return time_evaluator();
144 }
145
146 std::vector<double> RunOpRPC(int index, int number, int repeat, int min_repeat_ms,
147 int limit_zero_time_iterations, int cooldown_interval_ms,
148 int repeats_to_cooldown) {
149 std::vector<double> results(repeat, 0);
150 // Right now we expect either "tvm_op" for nodes which run PackedFunc or "null" for nodes
151 // which represent inputs/parameters to the graph. Other types may be supported in the
152 // future, but consideration would be needed as to how to do that over RPC before we support
153 // it here.
154 if (nodes_[index].op_type != "tvm_op") {
155 CHECK_EQ(nodes_[index].op_type, "null")
156 << "Don't know how to run op type " << nodes_[index].op_type
157 << " remotely over RPC right now";
158
159 // NOTE: GraphExecutorDebug expects graph nodes to have an "op" attribute of "tvm_op" or
160 // "null" and "null" is a placeholder node for a parameter or input.
161 return results;
162 }
163
164 const Device& dev = data_entry_[entry_id(index, 0)]->device;
165 TVMOpParam param = nodes_[index].param;
166 std::string name = param.func_name;
167 uint32_t num_inputs = param.num_inputs;
168 uint32_t num_outputs = param.num_outputs;
169
170 PackedFunc time_eval =
171 runtime::Registry::Get("runtime.RPCTimeEvaluator")
172 ->
173 operator()(module_, name, static_cast<int>(dev.device_type), dev.device_id, number,
174 repeat, min_repeat_ms, limit_zero_time_iterations, cooldown_interval_ms,
175 repeats_to_cooldown, "");
176
177 int num_flat_args = num_inputs + num_outputs;
178 auto values = std::make_unique<TVMValue[]>(num_flat_args);
179 auto type_codes = std::make_unique<int[]>(num_flat_args);
180 TVMArgsSetter setter(values.get(), type_codes.get());
181 int offs = 0;
182 const auto& inode = nodes_[index];
183 for (const auto& e : inode.inputs) {
184 uint32_t eid = this->entry_id(e);
185 DLTensor* arg = const_cast<DLTensor*>(data_entry_[eid].operator->());
186 setter(offs, arg);
187 offs++;
188 }
189 for (uint32_t i = 0; i < num_outputs; ++i) {
190 uint32_t eid = this->entry_id(index, i);
191 DLTensor* arg = const_cast<DLTensor*>(data_entry_[eid].operator->());
192 setter(offs, arg);
193 offs++;
194 }
195 TVMRetValue rv;
196 time_eval.CallPacked(TVMArgs(values.get(), type_codes.get(), num_flat_args), &rv);
197 std::string results_str = rv.operator std::string();
198 const double* blob_ptr = reinterpret_cast<const double*>(results_str.data());
199 for (int i = 0; i < repeat; ++i, ++blob_ptr) {
200 results[i] = *blob_ptr;
201 }
202
203 std::ostringstream os;
204 for (auto& repeat_data : results) {
205 os << std::to_string(repeat_data) << ", ";
206 }
207 LOG(INFO) << "Got op timing: " << os.str();
208 return results;
209 }

211 Timer RunOpHost(int index) {
212 const Device& dev = data_entry_[entry_id(index, 0)]->device;
213 Timer t = Timer::Start(dev);
214 op_execs_[index]();
215 t->Stop();
216 return t;
217 }

226 /*!
227 * \brief Get the node index given the name of node.
228 * \param name The name of the node.
229 * \return The index of node.
230 */
231 int GetNodeIndex(const std::string& name) const {
232 for (size_t nid = 0; nid < GetNumOfNodes(); ++nid) {
233 if (GetNodeName(nid) == name) {
234 return static_cast<int>(nid);
235 }
236 }
237 LOG(FATAL) << "cannot find " << name << " among nodex";
238 return -1;
239 }
240
241 /*!
242 * \brief Execute index-th node in the network.
243 *
244 * This method will do a partial run of the graph
245 * up to index-th node.
246 *
247 * \param node: The index of the node.
248 */
249 void ExecuteNode(int node) {
250 ICHECK_LT(static_cast<size_t>(node), op_execs_.size());
251
252 int start_ind;
253 int end_ind;
254 if (node < last_executed_node_) {
255 start_ind = 0;
256 end_ind = node;
257 } else if (node > last_executed_node_) {
258 start_ind = last_executed_node_ + 1;
259 end_ind = node;
260 } else {
261 return;
262 }
263
264 for (int i = start_ind; i <= end_ind; i++) {
265 if (op_execs_[i]) op_execs_[i]();
266 }
267 last_executed_node_ = end_ind;
268 }
269
270 /*!
271 * \brief Returns index-th output of node.
272 *
273 * This method will return index-th out_ind output
274 * of index-th node in the network.
275 *
276 * \param node: The index of the node.
277 * \param out_ind: The index of the output.
278 * \return Output array.
279 */
280 NDArray GetNodeOutput(int node, int out_ind) {
281 ICHECK_EQ(node, last_executed_node_);
282 ICHECK_LT(entry_id(node, out_ind), data_entry_.size());
283 return data_entry_[entry_id(node, out_ind)].CopyTo({kDLCPU, 0});
284 }
285
286 /*!
287 * \brief Copy index-th node to data_out.
288 *
289 * This method will do a partial run of the graph
290 * from begining upto the index-th node and return output of index-th node.
291 * This is costly operation and suggest to use only for debug porpose.
292 *
293 * \param index: The index of the node.
294 * \param data_out the node data.
295 */
296 void DebugGetNodeOutput(int index, DLTensor* data_out) {
297 ICHECK_LT(static_cast<size_t>(index), op_execs_.size());
298 uint32_t eid = index;
299
300 for (size_t i = 0; i < op_execs_.size(); ++i) {
301 if (op_execs_[i]) op_execs_[i]();
302 if (static_cast<int>(i) == index) break;
303 }
304
305 data_entry_[eid].CopyTo(data_out);
306 }

308 /*!
309 * \brief Profile execution time of the module.
310 *
311 * We run the entire module while recording overall and per-op timing
312 * information. The module may be run multiple times to ensure everything is
313 * warmed up. This function is a more correct reflection of actual runtime of
314 * the module compared to GraphRuntimeDebug::RunIndividual as it runs the
315 * entire graph in order.
316 *
317 * \param collectors Optional user defined `MetricCollector`s to use with this profiling run.
318 *
319 * \returns A table of per-op runtimes and total times.
320 */
321 profiling::Report Profile(Array<profiling::MetricCollector> collectors) {
322 std::vector<profiling::MetricCollector> cs(collectors.begin(), collectors.end());
323 profiling::Profiler prof(devices_, cs, {{String("Executor"), String("Graph")}});
324
325 // warm up. 1 iteration does not seem enough.
326 for (int i = 0; i < 3; i++) {
327 GraphExecutor::Run();
328 }
329
330 prof.Start();
331 for (size_t i = 0; i < op_execs_.size(); ++i) {
332 if (op_execs_[i]) {
333 // get argument shapes
334 std::vector<NDArray> shapes;
335 for (const auto& e : nodes_[i].inputs) {
336 uint32_t eid = entry_id(e);
337 shapes.push_back(data_entry_[eid]);
338 }
339 for (uint32_t j = 0; j < nodes_[i].param.num_outputs; ++j) {
340 uint32_t eid = entry_id(i, j);
341 shapes.push_back(data_entry_[eid]);
342 }
343
344 uint32_t eid = entry_id(i, 0);
345 const Device& dev = data_entry_[eid]->device;
346
347 std::unordered_map<std::string, ObjectRef> metrics;
348 for (auto p : nodes_[i].param.attrs) {
349 if (std::string(p.first).find("layout") != std::string::npos) {
350 metrics[p.first] = p.second;
351 }
352 }
353 if (nodes_[i].param.attrs.find("hash") != nodes_[i].param.attrs.end()) {
354 metrics["Hash"] = Downcast<String>(nodes_[i].param.attrs.at("hash"));
355 }
356 metrics["Argument Shapes"] = profiling::ShapeString(shapes);
357 prof.StartCall(nodes_[i].param.func_name, dev, metrics);
358 op_execs_[i]();
359 prof.StopCall();
360 }
361 }
362 prof.Stop();
363 return prof.Report();
364 }
365
366 private:
367 int last_executed_node_ = -1;
368};

370/*!
371 * \brief GetFunction Get the function based on input.
372 * \param name The function which needs to be invoked.
373 * \param sptr_to_self Packed function pointer.
374 */
375PackedFunc GraphExecutorDebug::GetFunction(const std::string& name,
376 const ObjectPtr<Object>& sptr_to_self) {
377 // return member functions during query.
378 if (name == "debug_get_output") {
379 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
380 if (String::CanConvertFrom(args[0])) {
381 this->DebugGetNodeOutput(this->GetNodeIndex(args[0]), args[1]);
382 } else {
383 this->DebugGetNodeOutput(args[0], args[1]);
384 }
385 });
386 } else if (name == "execute_node") {
387 return PackedFunc(
388 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { this->ExecuteNode(args[0]); });
389 } else if (name == "get_node_output") {
390 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
391 *rv = this->GetNodeOutput(args[0], args[1]);
392 });
393 } else if (name == "run_individual") {
394 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
395 int number = args[0];
396 int repeat = args[1];
397 int min_repeat_ms = args[2];
398 int limit_zero_time_iterations = args[3];
399 int cooldown_interval_ms = args[4];
400 int repeats_to_cooldown = args[5];
401 ICHECK_GT(number, 0);
402 ICHECK_GT(repeat, 0);
403 ICHECK_GE(min_repeat_ms, 0);
404 ICHECK_GE(limit_zero_time_iterations, 0);
405 ICHECK_GE(cooldown_interval_ms, 0);
406 ICHECK_GT(repeats_to_cooldown, 0);
407 std::string blob =
408 this->RunIndividual(number, repeat, min_repeat_ms, limit_zero_time_iterations,
409 cooldown_interval_ms, repeats_to_cooldown);
410 TVMByteArray arr;
411 arr.size = blob.length();
412 arr.data = blob.data();
413 *rv = arr;
414 });
415 } else if (name == "run_individual_node") {
416 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
417 int node_index = args[0];
418 int number = args[1];
419 int repeat = args[2];
420 int min_repeat_ms = args[3];
421 int limit_zero_time_iterations = args[4];
422 int cooldown_interval_ms = args[5];
423 int repeats_to_cooldown = args[6];
424 ICHECK_GE(node_index, 0);
425 ICHECK_LT(node_index, nodes_.size());
426 ICHECK_GT(number, 0);
427 ICHECK_GT(repeat, 0);
428 ICHECK_GE(min_repeat_ms, 0);
429 ICHECK_GE(limit_zero_time_iterations, 0);
430 ICHECK_GE(cooldown_interval_ms, 0);
431 ICHECK_GT(repeats_to_cooldown, 0);
432 std::string blob = this->RunIndividualNode(node_index, number, repeat, min_repeat_ms,
433 limit_zero_time_iterations, cooldown_interval_ms,
434 repeats_to_cooldown);
435 TVMByteArray arr;
436 arr.size = blob.length();
437 arr.data = blob.data();
438 *rv = arr;
439 });
440 } else if (name == "profile") {
441 return TypedPackedFunc<profiling::Report(Array<profiling::MetricCollector>)>(
442 [sptr_to_self, this](Array<profiling::MetricCollector> collectors) {
443 // We cannot send Arrays over rpc, so in order to support profiling
444 // on remotes, we accept a nullptr for collectors.
445 if (collectors.defined()) {
446 return this->Profile(collectors);
447 } else {
448 return this->Profile({});
449 }
450 });
451 } else if (name == "profile_rpc") {
452 // We cannot return a Report over RPC because TMV RPC mechanism only
453 // supports a subset of Object classes. Instead we serialize it on the
454 // remote (here) and deserialize it on the other end.
455 return TypedPackedFunc<std::string()>([sptr_to_self, this]() {
456 PackedFunc profile = GetFunction("profile", sptr_to_self);
457 profiling::Report report = profile(Array<profiling::MetricCollector>());
458 return report->AsJSON();
459 });
460 } else {
461 return GraphExecutor::GetFunction(name, sptr_to_self);
462 }
463}

465/*!
466 * \brief GraphExecutorDebugCreate Get the function based on input.
467 * \param sym_json The graph symbol in json format.
468 * \param m Compiled module which will be loaded.
469 * \param devs All devices.
470 */
471Module GraphExecutorDebugCreate(const std::string& sym_json, const tvm::runtime::Module& m,
472 const std::vector<Device>& devs,
473 PackedFunc lookup_linked_param_func) {
474 auto exec = make_object<GraphExecutorDebug>();
475 exec->Init(sym_json, m, devs, lookup_linked_param_func);
476 return Module(exec);
477}

479TVM_REGISTER_GLOBAL("tvm.graph_executor_debug.create").set_body([](TVMArgs args, TVMRetValue* rv) {
480 ICHECK_GE(args.num_args, 4) << "The expected number of arguments for graph_executor.create is "
481 "at least 4, but it has "
482 << args.num_args;
483 PackedFunc lookup_linked_param_func;
484 int dev_start_arg = 2;
485 if (args[2].type_code() == kTVMPackedFuncHandle) {
486 lookup_linked_param_func = args[2];
487 dev_start_arg++;
488 }
489
490 *rv = GraphExecutorDebugCreate(args[0], args[1], GetAllDevice(args, dev_start_arg),
491 lookup_linked_param_func);
492});
