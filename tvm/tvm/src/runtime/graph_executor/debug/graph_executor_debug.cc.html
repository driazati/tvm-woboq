40/*!
41 * \brief Graph executor with debug .
42 *
43 * This is the extension of GraphExecutor class used for debugging
44 * TVM runtime PackedFunc API.
45 */
46class GraphExecutorDebug : public GraphExecutor {
47 public:
48 /*!
49 * \brief Run each operation in the graph and get the time per op for all ops.
50 * \param number The number of times to run this function for taking average.
51 * \param repeat The number of times to repeat the measurement.
52 * In total, the function will be invoked (1 + number x repeat) times,
53 * where the first one is warmed up and will be discarded in case
54 * there is lazy initialization.
55 * \param min_repeat_ms The minimum duration of one `repeat` in milliseconds.
56 * By default, one `repeat` contains `number` runs. If this parameter is set,
57 * the parameters `number` will be dynamically adjusted to meet the
58 * minimum duration requirement of one `repeat`.
59 * \param cooldown_interval_ms The cooldown interval in milliseconds between the number of repeats
60 * defined by `repeats_to_cooldown`.
61 * \param repeats_to_cooldown The number of repeats before the
62 * cooldown is activated.
63 * \return Returns a string with an encoded byte array. Where the first 8 bytes are int64_t
64 * representing the number of layers. Next the encoded real numbers are float32_t in the number of
65 * repeat multiplied by the number of layers.
66 */
67 std::string RunIndividual(int number, int repeat, int min_repeat_ms, int cooldown_interval_ms,
68 int repeats_to_cooldown) {
69 // warmup run
70 GraphExecutor::Run();
71 std::string tkey = module_->type_key();
72 std::vector<std::vector<double>> time_sec_per_op(op_execs_.size());
73 if (tkey == "rpc") {
74 // RPC modules rely on remote timing which implements the logic from the else branch.
75 for (size_t index = 0; index < op_execs_.size(); ++index) {
76 time_sec_per_op[index] = RunOpRPC(index, number, repeat, min_repeat_ms,
77 cooldown_interval_ms, repeats_to_cooldown);
78 }
79 } else {
80 int op = 0;
81 for (size_t index = 0; index < op_execs_.size(); ++index) {
82 std::string result_str = RunIndividualNode(index, number, repeat, min_repeat_ms,
83 cooldown_interval_ms, repeats_to_cooldown);
84 const double* blob_ptr = reinterpret_cast<const double*>(result_str.data());
85 for (int i = 0; i < repeat; ++i, ++blob_ptr) {
86 time_sec_per_op[index].push_back(*blob_ptr);
87 }
88 if (op_execs_[index]) {
89 LOG(INFO) << "Op #" << op << " " << GetNodeName(index) << ":";
90 for (size_t cur_repeat = 0; cur_repeat < time_sec_per_op[index].size(); cur_repeat++) {
91 const auto& data = time_sec_per_op[index][cur_repeat];
92 LOG(INFO) << "Iteration: " << cur_repeat << ": " << (data * 1e6) << " us/iter";
93 }
94 ++op;
95 }
96 }
97 }
98
99 std::ostringstream os;
100 int64_t size = time_sec_per_op.size();
101 os.write(reinterpret_cast<char*>(&size), sizeof(int64_t));
102 for (size_t index = 0; index < time_sec_per_op.size(); ++index) {
103 for (auto& repeat_data : time_sec_per_op[index]) {
104 // To have good behavior when calculating total time, etc.
105 double data = std::isnan(repeat_data) ? 0 : repeat_data;
106 os.write(reinterpret_cast<char*>(&data), sizeof(double));
107 }
108 }
109 return os.str();
110 }
111
112 std::string RunIndividualNode(int node_index, int number, int repeat, int min_repeat_ms,
113 int cooldown_interval_ms, int repeats_to_cooldown) {
114 std::string tkey = module_->type_key();
115
116 if (tkey == "rpc") {
117 LOG(FATAL) << "RPC measurements should not use RunIndividualNode!";
118 }
119
120 if (!op_execs_[node_index] || nodes_[node_index].param.func_name == "__nop") {
121 // don't return anything...
122 std::ostringstream os;
123 double zero = 0;
124 for (int i = 0; i < repeat; ++i) {
125 os.write(reinterpret_cast<char*>(&zero), sizeof(double));
126 }
127 return os.str();
128 }
129
130 // assume host runs things which is first device
131 Device& d = devices_[0];
132 PackedFunc time_evaluator = profiling::WrapTimeEvaluator(
133 TypedPackedFunc<void()>([this, node_index]() { this->RunOpHost(node_index); }), d, number,
134 repeat, min_repeat_ms, cooldown_interval_ms, repeats_to_cooldown);
135 return time_evaluator();
136 }
137
138 std::vector<double> RunOpRPC(int index, int number, int repeat, int min_repeat_ms,
139 int cooldown_interval_ms, int repeats_to_cooldown) {
140 std::vector<double> results(repeat, 0);
141 // Right now we expect either "tvm_op" for nodes which run PackedFunc or "null" for nodes
142 // which represent inputs/parameters to the graph. Other types may be supported in the
143 // future, but consideration would be needed as to how to do that over RPC before we support
144 // it here.
145 if (nodes_[index].op_type != "tvm_op") {
146 CHECK_EQ(nodes_[index].op_type, "null")
147 << "Don't know how to run op type " << nodes_[index].op_type
148 << " remotely over RPC right now";
149
150 // NOTE: GraphExecutorDebug expects graph nodes to have an "op" attribute of "tvm_op" or
151 // "null" and "null" is a placeholder node for a parameter or input.
152 return results;
153 }
154
155 if (nodes_[index].param.func_name == "__nop") {
156 LOG_INFO << "Skipping __nop function";
157 return results;
158 }
159
160 const Device& dev = data_entry_[entry_id(index, 0)]->device;
161 TVMOpParam param = nodes_[index].param;
162 std::string name = param.func_name;
163 uint32_t num_inputs = param.num_inputs;
164 uint32_t num_outputs = param.num_outputs;
165
166 PackedFunc time_eval =
167 runtime::Registry::Get("runtime.RPCTimeEvaluator")
168 ->
169 operator()(module_, name, static_cast<int>(dev.device_type), dev.device_id, number,
170 repeat, min_repeat_ms, cooldown_interval_ms, repeats_to_cooldown, "");
171
172 int num_flat_args = num_inputs + num_outputs;
173 std::unique_ptr<TVMValue> values(new TVMValue[num_flat_args]);
174 std::unique_ptr<int> type_codes(new int[num_flat_args]);
175 TVMArgsSetter setter(values.get(), type_codes.get());
176 int offs = 0;
177 const auto& inode = nodes_[index];
178 for (const auto& e : inode.inputs) {
179 uint32_t eid = this->entry_id(e);
180 DLTensor* arg = const_cast<DLTensor*>(data_entry_[eid].operator->());
181 setter(offs, arg);
182 offs++;
183 }
184 for (uint32_t i = 0; i < num_outputs; ++i) {
185 uint32_t eid = this->entry_id(index, i);
186 DLTensor* arg = const_cast<DLTensor*>(data_entry_[eid].operator->());
187 setter(offs, arg);
188 offs++;
189 }
190 TVMRetValue rv;
191 time_eval.CallPacked(TVMArgs(values.get(), type_codes.get(), num_flat_args), &rv);
192 std::string results_str = rv.operator std::string();
193 const double* blob_ptr = reinterpret_cast<const double*>(results_str.data());
194 for (int i = 0; i < repeat; ++i, ++blob_ptr) {
195 results[i] = *blob_ptr;
196 }
197
198 std::ostringstream os;
199 for (auto& repeat_data : results) {
200 os << std::to_string(repeat_data) << ", ";
201 }
202 LOG(INFO) << "Got op timing: " << os.str();
203 return results;
204 }
205
206 Timer RunOpHost(int index) {
207 const Device& dev = data_entry_[entry_id(index, 0)]->device;
208 Timer t = Timer::Start(dev);
209 op_execs_[index]();
210 t->Stop();
211 return t;
212 }

221 /*!
222 * \brief Get the node index given the name of node.
223 * \param name The name of the node.
224 * \return The index of node.
225 */
226 int GetNodeIndex(const std::string& name) const {
227 for (size_t nid = 0; nid < GetNumOfNodes(); ++nid) {
228 if (GetNodeName(nid) == name) {
229 return static_cast<int>(nid);
230 }
231 }
232 LOG(FATAL) << "cannot find " << name << " among nodex";
233 return -1;
234 }
235
236 /*!
237 * \brief Execute index-th node in the network.
238 *
239 * This method will do a partial run of the graph
240 * up to index-th node.
241 *
242 * \param node: The index of the node.
243 */
244 void ExecuteNode(int node) {
245 ICHECK_LT(static_cast<size_t>(node), op_execs_.size());
246
247 int start_ind;
248 int end_ind;
249 if (node < last_executed_node_) {
250 start_ind = 0;
251 end_ind = node;
252 } else if (node > last_executed_node_) {
253 start_ind = last_executed_node_ + 1;
254 end_ind = node;
255 } else {
256 return;
257 }
258
259 for (int i = start_ind; i <= end_ind; i++) {
260 if (op_execs_[i]) op_execs_[i]();
261 }
262 last_executed_node_ = end_ind;
263 }
264
265 /*!
266 * \brief Returns index-th output of node.
267 *
268 * This method will return index-th out_ind output
269 * of index-th node in the network.
270 *
271 * \param node: The index of the node.
272 * \param out_ind: The index of the output.
273 * \return Output array.
274 */
275 NDArray GetNodeOutput(int node, int out_ind) {
276 ICHECK_EQ(node, last_executed_node_);
277 ICHECK_LT(entry_id(node, out_ind), data_entry_.size());
278 return data_entry_[entry_id(node, out_ind)].CopyTo({kDLCPU, 0});
279 }
280
281 /*!
282 * \brief Copy index-th node to data_out.
283 *
284 * This method will do a partial run of the graph
285 * from begining upto the index-th node and return output of index-th node.
286 * This is costly operation and suggest to use only for debug porpose.
287 *
288 * \param index: The index of the node.
289 * \param data_out the node data.
290 */
291 void DebugGetNodeOutput(int index, DLTensor* data_out) {
292 ICHECK_LT(static_cast<size_t>(index), op_execs_.size());
293 uint32_t eid = index;
294
295 for (size_t i = 0; i < op_execs_.size(); ++i) {
296 if (op_execs_[i]) op_execs_[i]();
297 if (static_cast<int>(i) == index) break;
298 }
299
300 data_entry_[eid].CopyTo(data_out);
301 }
302
303 /*!
304 * \brief Profile execution time of the module.
305 *
306 * We run the entire module while recording overall and per-op timing
307 * information. The module may be run multiple times to ensure everything is
308 * warmed up. This function is a more correct reflection of actual runtime of
309 * the module compared to GraphRuntimeDebug::RunIndividual as it runs the
310 * entire graph in order.
311 *
312 * \param collectors Optional user defined `MetricCollector`s to use with this profiling run.
313 *
314 * \returns A table of per-op runtimes and total times.
315 */
316 profiling::Report Profile(Array<profiling::MetricCollector> collectors) {
317 std::vector<profiling::MetricCollector> cs(collectors.begin(), collectors.end());
318 profiling::Profiler prof(devices_, cs, {{String("Executor"), String("Graph")}});
319
320 // warm up. 1 iteration does not seem enough.
321 for (int i = 0; i < 3; i++) {
322 GraphExecutor::Run();
323 }
324
325 prof.Start();
326 for (size_t i = 0; i < op_execs_.size(); ++i) {
327 if (op_execs_[i]) {
328 // get argument shapes
329 std::vector<NDArray> shapes;
330 for (const auto& e : nodes_[i].inputs) {
331 uint32_t eid = entry_id(e);
332 shapes.push_back(data_entry_[eid]);
333 }
334 for (uint32_t j = 0; j < nodes_[i].param.num_outputs; ++j) {
335 uint32_t eid = entry_id(i, j);
336 shapes.push_back(data_entry_[eid]);
337 }
338
339 uint32_t eid = entry_id(i, 0);
340 const Device& dev = data_entry_[eid]->device;
341
342 std::unordered_map<std::string, ObjectRef> metrics;
343 for (auto p : nodes_[i].param.attrs) {
344 if (std::string(p.first).find("layout") != std::string::npos) {
345 metrics[p.first] = p.second;
346 }
347 }
348 if (nodes_[i].param.attrs.find("hash") != nodes_[i].param.attrs.end()) {
349 metrics["Hash"] = Downcast<String>(nodes_[i].param.attrs.at("hash"));
350 }
351 metrics["Argument Shapes"] = profiling::ShapeString(shapes);
352 prof.StartCall(nodes_[i].param.func_name, dev, metrics);
353 op_execs_[i]();
354 prof.StopCall();
355 }
356 }
357 prof.Stop();
358 return prof.Report();
359 }
360
361 private:
362 int last_executed_node_ = -1;
363};

365/*!
366 * \brief GetFunction Get the function based on input.
367 * \param name The function which needs to be invoked.
368 * \param sptr_to_self Packed function pointer.
369 */
370PackedFunc GraphExecutorDebug::GetFunction(const std::string& name,
371 const ObjectPtr<Object>& sptr_to_self) {
372 // return member functions during query.
373 if (name == "debug_get_output") {
374 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
375 if (String::CanConvertFrom(args[0])) {
376 this->DebugGetNodeOutput(this->GetNodeIndex(args[0]), args[1]);
377 } else {
378 this->DebugGetNodeOutput(args[0], args[1]);
379 }
380 });
381 } else if (name == "execute_node") {
382 return PackedFunc(
383 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { this->ExecuteNode(args[0]); });
384 } else if (name == "get_node_output") {
385 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
386 *rv = this->GetNodeOutput(args[0], args[1]);
387 });
388 } else if (name == "run_individual") {
389 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
390 int number = args[0];
391 int repeat = args[1];
392 int min_repeat_ms = args[2];
393 int cooldown_interval_ms = args[3];
394 int repeats_to_cooldown = args[4];
395 ICHECK_GT(number, 0);
396 ICHECK_GT(repeat, 0);
397 ICHECK_GE(min_repeat_ms, 0);
398 ICHECK_GE(cooldown_interval_ms, 0);
399 ICHECK_GT(repeats_to_cooldown, 0);
400 std::string blob = this->RunIndividual(number, repeat, min_repeat_ms, cooldown_interval_ms,
401 repeats_to_cooldown);
402 TVMByteArray arr;
403 arr.size = blob.length();
404 arr.data = blob.data();
405 *rv = arr;
406 });
407 } else if (name == "run_individual_node") {
408 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
409 int node_index = args[0];
410 int number = args[1];
411 int repeat = args[2];
412 int min_repeat_ms = args[3];
413 int cooldown_interval_ms = args[4];
414 int repeats_to_cooldown = args[5];
415 ICHECK_GE(node_index, 0);
416 ICHECK_LT(node_index, nodes_.size());
417 ICHECK_GT(number, 0);
418 ICHECK_GT(repeat, 0);
419 ICHECK_GE(min_repeat_ms, 0);
420 ICHECK_GE(cooldown_interval_ms, 0);
421 ICHECK_GT(repeats_to_cooldown, 0);
422 std::string blob = this->RunIndividualNode(node_index, number, repeat, min_repeat_ms,
423 cooldown_interval_ms, repeats_to_cooldown);
424 TVMByteArray arr;
425 arr.size = blob.length();
426 arr.data = blob.data();
427 *rv = arr;
428 });
429 } else if (name == "profile") {
430 return TypedPackedFunc<profiling::Report(Array<profiling::MetricCollector>)>(
431 [sptr_to_self, this](Array<profiling::MetricCollector> collectors) {
432 // We cannot send Arrays over rpc, so in order to support profiling
433 // on remotes, we accept a nullptr for collectors.
434 if (collectors.defined()) {
435 return this->Profile(collectors);
436 } else {
437 return this->Profile({});
438 }
439 });
440 } else if (name == "profile_rpc") {
441 // We cannot return a Report over RPC because TMV RPC mechanism only
442 // supports a subset of Object classes. Instead we serialize it on the
443 // remote (here) and deserialize it on the other end.
444 return TypedPackedFunc<std::string()>([sptr_to_self, this]() {
445 PackedFunc profile = GetFunction("profile", sptr_to_self);
446 profiling::Report report = profile(Array<profiling::MetricCollector>());
447 return report->AsJSON();
448 });
449 } else {
450 return GraphExecutor::GetFunction(name, sptr_to_self);
451 }
452}

454/*!
455 * \brief GraphExecutorDebugCreate Get the function based on input.
456 * \param sym_json The graph symbol in json format.
457 * \param m Compiled module which will be loaded.
458 * \param devs All devices.
459 */
460Module GraphExecutorDebugCreate(const std::string& sym_json, const tvm::runtime::Module& m,
461 const std::vector<Device>& devs,
462 PackedFunc lookup_linked_param_func) {
463 auto exec = make_object<GraphExecutorDebug>();
464 exec->Init(sym_json, m, devs, lookup_linked_param_func);
465 return Module(exec);
466}

468TVM_REGISTER_GLOBAL("tvm.graph_executor_debug.create").set_body([](TVMArgs args, TVMRetValue* rv) {
469 ICHECK_GE(args.num_args, 4) << "The expected number of arguments for graph_executor.create is "
470 "at least 4, but it has "
471 << args.num_args;
472 PackedFunc lookup_linked_param_func;
473 int dev_start_arg = 2;
474 if (args[2].type_code() == kTVMPackedFuncHandle) {
475 lookup_linked_param_func = args[2];
476 dev_start_arg++;
477 }
478
479 *rv = GraphExecutorDebugCreate(args[0], args[1], GetAllDevice(args, dev_start_arg),
480 lookup_linked_param_func);
481});
