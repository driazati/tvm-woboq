50inline size_t GetDataAlignment(const DLTensor& arr) {
51 size_t align = (arr.dtype.bits / 8) * arr.dtype.lanes;
52 if (align < kAllocAlignment) return kAllocAlignment;
53 return align;
54}

58/*!
59 * \brief Run all the operations one by one.
60 */
61void GraphExecutor::Run() {
62 // setup the array and requirements.
63 for (size_t i = 0; i < op_execs_.size(); ++i) {
64 if (op_execs_[i]) op_execs_[i]();
65 }
66}

68/*!
69 * \brief Initialize the graph executor with graph and device.
70 * \param graph_json The execution graph.
71 * \param module The module containing the compiled functions for the host
72 * processor.
73 * \param devs The devices of the host and devices where graph nodes will be
74 * executed on.
75 * \param lookup_linked_param_func Linked parameter lookup function. Default is nullptr.
76 */
77void GraphExecutor::Init(const std::string& graph_json, tvm::runtime::Module module,
78 const std::vector<Device>& devs,
79 const PackedFunc lookup_linked_param_func) {
80 std::istringstream is(graph_json);
81 dmlc::JSONReader reader(&is);
82 this->Load(&reader);
83 module_ = module;
84 devices_ = devs;
85 lookup_linked_param_ = lookup_linked_param_func;
86 if (lookup_linked_param_ == nullptr) {
87 lookup_linked_param_ = PackedFunc(
88 [this](TVMArgs args, TVMRetValue* rv) { this->DefaultLookupLinkedParam(args, rv); });
89 }
90 this->SetupStorage();
91 this->SetupOpExecs();
92 for (size_t i = 0; i < input_nodes_.size(); i++) {
93 const uint32_t nid = input_nodes_[i];
94 std::string& name = nodes_[nid].name;
95 input_map_[name] = i;
96 }
97 for (size_t i = 0; i < outputs_.size(); i++) {
98 const uint32_t nid = outputs_[i].node_id;
99 std::string& name = nodes_[nid].name;
100 std::stringstream ss;
101 ss << name << ":" << i;
102 output_map_[ss.str()] = i;
103 }
104}

119/*!
120 * \brief Get the input info of Graph by parsing the input nodes.
121 * \return The shape and dtype tuple.
122 */
123std::tuple<GraphExecutor::ShapeInfo, GraphExecutor::DtypeInfo> GraphExecutor::GetInputInfo() const {
124 GraphExecutor::ShapeInfo shape_dict;
125 GraphExecutor::DtypeInfo dtype_dict;
126 for (uint32_t nid : input_nodes_) {
127 CHECK_LE(nid, nodes_.size());
128 std::string name = nodes_[nid].name;
129 if (param_names_.find(name) == param_names_.end()) {
130 CHECK_LE(nid, attrs_.shape.size());
131 auto shape = attrs_.shape[nid];
132 shape_dict.Set(name, ShapeTuple(shape));
133 CHECK_LE(nid, attrs_.dltype.size());
134 auto dtype = attrs_.dltype[nid];
135 dtype_dict.Set(name, String(dtype));
136 }
137 }
138 return std::make_tuple(shape_dict, dtype_dict);
139}

153/*!
154 * \brief set index-th input to the graph.
155 * \param index The input index.
156 * \param data_in The input data.
157 */
158void GraphExecutor::SetInput(int index, DLTensor* data_in) {
159 ICHECK_LT(static_cast<size_t>(index), input_nodes_.size());

163/*!
164 * \brief Check the legality of external DLTensor*.
165 * \param external The external DLTensor*.
166 * \param eid The data_enrty_ index.
167 */
168void GraphExecutor::CheckExternalDLTensor(const DLTensor* external, uint32_t eid) const {
169 const DLTensor* internal = data_entry_[eid].operator->();
170
171 ICHECK_EQ(data_alignment_[eid], details::GetDataAlignment(*external));
172 ICHECK_EQ(reinterpret_cast<size_t>(static_cast<char*>(external->data) + external->byte_offset) %
173 kAllocAlignment,
174 0);
175 ICHECK_EQ(internal->ndim, static_cast<size_t>(external->ndim));
176 ICHECK_EQ(internal->device.device_type, external->device.device_type);
177 ICHECK_EQ(internal->device.device_id, external->device.device_id);
178 for (auto i = 0; i < external->ndim; ++i) {
179 ICHECK_EQ(internal->shape[i], external->shape[i]);
180 }
181}
182/*!
183 * \brief set index-th input to the graph without copying the data.
184 * \param index The input index.
185 * \param data_ref The input data that is referred.
186 */
187void GraphExecutor::SetInputZeroCopy(int index, DLTensor* data_ref) {
188 ICHECK_LT(static_cast<size_t>(index), input_nodes_.size());
189 uint32_t eid = this->entry_id(input_nodes_[index], 0);
190 // check the consistency of input
191 CheckExternalDLTensor(data_ref, eid);
192 // Update the data pointer for each argument of each op
193 for (DLTensor* t : input_dltensors_[eid]) {
194 t->data = static_cast<char*>(data_ref->data) + data_ref->byte_offset;
195 }
196}
197/*!
198 * \brief set index-th output to the graph without copying the data.
199 * \param index The output index.
200 * \param data_ref The output data that is referred.
201 */
202void GraphExecutor::SetOutputZeroCopy(int index, DLTensor* data_ref) {
203 ICHECK_LT(static_cast<size_t>(index), outputs_.size());
204 ICHECK_LT(static_cast<size_t>(index), output_dltensors_.size());
205 const NodeEntry& output_node = outputs_[index];
206 uint32_t output_node_eid = this->entry_id(output_node);
207
208 // check the consistency of output
209 CheckExternalDLTensor(data_ref, output_node_eid);
210
211 // Update the data pointer for output op
212 for (DLTensor* t : output_dltensors_[output_node_eid]) {
213 t->data = static_cast<char*>(data_ref->data) + data_ref->byte_offset;
214 }
215
216 // Update the input of the op connected to the output
217 for (DLTensor* t : both_output_opinput_dltensors_[output_node_eid]) {
218 t->data = static_cast<char*>(data_ref->data) + data_ref->byte_offset;
219 }
220}

255/*!
256 * \brief Copy index-th output to data_out.
257 * \param index The output index.
258 * \param data_out the output data.
259 */
260void GraphExecutor::CopyOutputTo(int index, DLTensor* data_out) {
261 ICHECK_LT(static_cast<size_t>(index), outputs_.size());
262 uint32_t eid = this->entry_id(outputs_[index]);
263
264 // Check the shapes to avoid receiving in different dimension but same size.
265 const NDArray& data = data_entry_[eid];
266 ICHECK_EQ(data->ndim, data_out->ndim);
267 for (int32_t j = 0; j < data->ndim; ++j) {
268 ICHECK_EQ(data->shape[j], data_out->shape[j]);
269 }
270
271 data_entry_[eid].CopyTo(data_out);
272}

283void GraphExecutor::LoadParams(dmlc::Stream* strm) {
284 Map<String, NDArray> params = ::tvm::runtime::LoadParams(strm);
285 for (auto& p : params) {
286 param_names_.insert(p.first);
287 int in_idx = GetInputIndex(p.first);
288 if (in_idx < 0) continue;
289 uint32_t eid = this->entry_id(input_nodes_[in_idx], 0);
290 data_entry_[eid].CopyFrom(p.second);
291 }

294void GraphExecutor::ShareParams(const GraphExecutor& other, dmlc::Stream* strm) {
295 uint64_t header, reserved;
296 ICHECK(strm->Read(&header)) << "Invalid parameters file format";
297 ICHECK(header == kTVMNDArrayListMagic) << "Invalid parameters file format";
298 ICHECK(strm->Read(&reserved)) << "Invalid parameters file format";
299 std::vector<std::string> names;
300 ICHECK(strm->Read(&names)) << "Invalid parameters file format";
301 uint64_t sz;
302 strm->Read(&sz);
303 size_t size = static_cast<size_t>(sz);
304 ICHECK(size == names.size()) << "Invalid parameters file format";
305 for (size_t i = 0; i < size; ++i) {
306 int in_idx = GetInputIndex(names[i]);
307 if (in_idx < 0) continue;
308 uint32_t eid = this->entry_id(input_nodes_[in_idx], 0);
309 ICHECK_LT(eid, data_entry_.size());
310 ICHECK_EQ(data_entry_[eid].use_count(), 1);
311 data_entry_[eid] = other.GetInput(GetInputIndex(names[i]));
312 ICHECK_GT(data_entry_[eid].use_count(), 1);
313 const DLTensor* tmp = data_entry_[eid].operator->();
314 data_alignment_[eid] = details::GetDataAlignment(*tmp);
315 }
316 this->SetupOpExecs();
317}

320 // container is the NDArray::Container which needs to get deleted.

322 delete static_cast<NDArray::Container*>(container);

325void GraphExecutor::DefaultLookupLinkedParam(TVMArgs args, TVMRetValue* rv) {
326 Module mod = args[0];
327 int64_t storage_id = args[1];
328 DLTensor* template_tensor = args[2];
329 Device dev = args[3];
330 // Get pre-linked parameter lookup function, if it was generated. When pf == nullptr, no linked
331 // params are present.
332 if (!module_lookup_linked_param_valid_) {
333 module_lookup_linked_param_ =
334 mod.GetFunction(::tvm::runtime::symbol::tvm_lookup_linked_param, true);
335 }
336 if (module_lookup_linked_param_ == nullptr) {
337 *rv = nullptr;
338 return;
339 }
340
341 TVMRetValue opaque_handle = module_lookup_linked_param_(storage_id);
342 if (opaque_handle.type_code() == kTVMNullptr) {
343 *rv = nullptr;
344 return;
345 }
346
347 std::vector<int64_t> shape_vec{template_tensor->shape,
348 template_tensor->shape + template_tensor->ndim};
349
350 auto* container = new NDArray::Container(static_cast<void*>(opaque_handle), shape_vec,
351 template_tensor->dtype, dev);
352 container->SetDeleter(GraphExecutor::LinkedNDArrayDeleter);
353 *rv = NDArray(GetObjectPtr<Object>(container));
354}

356void GraphExecutor::SetupStorage() {
357 // Grab saved optimization plan from graph.
358 std::vector<DLDataType> vtype;
359 for (const std::string& s_type : attrs_.dltype) {
360 vtype.push_back(tvm::runtime::String2DLDataType(s_type));
361 }
362
363 // Size and device type of each storage pool entry.
364 std::vector<PoolEntry> pool_entry;
365 // Find the maximum space size.
366 for (size_t i = 0; i < attrs_.shape.size(); ++i) {
367 int storage_id = attrs_.storage_id[i];
368 std::string storage_scope = attrs_.storage_scope.empty() ? "" : attrs_.storage_scope[i];
369 // Use the fallback device if no device index is available.
370 int device_type = static_cast<int>(devices_[0].device_type);
371 if (!attrs_.device_index.empty()) {
372 device_type = attrs_.device_index[i];
373 }
374
375 uint32_t sid = static_cast<uint32_t>(storage_id);
376 if (sid >= pool_entry.size()) {
377 pool_entry.resize(sid + 1, {-1, {0}, {}});
378 } else {
379 ICHECK(pool_entry[sid].device_type == -1 || pool_entry[sid].device_type == device_type)
380 << "The same pool entry cannot be assigned to multiple devices";
381 }
382 TVMRetValue lookup_rv;
383 {
384 std::vector<int64_t> shape_vec{attrs_.shape[i].begin(), attrs_.shape[i].end()};
385 DLTensor template_tensor{nullptr, Device{kDLCPU, 0}, static_cast<int>(shape_vec.size()),
386 vtype[i], shape_vec.data(), nullptr,
387 0};
388 lookup_rv = lookup_linked_param_(module_, sid, &template_tensor, devices_[0]);
389 }
390 if (lookup_rv.type_code() != kTVMNullptr) {
391 pool_entry[sid].linked_param = lookup_rv;
392 }
393 pool_entry[sid].param_data_entry = i;
394 pool_entry[sid].device_type = device_type;
395 pool_entry[sid].scope = storage_scope;
396
397 DLDataType t = vtype[i];
398 if (!details::Is2DStorage(storage_scope)) {
399 size_t size = 1;
400 for (int64_t sz : attrs_.shape[i]) {
401 size *= static_cast<size_t>(sz);
402 }
403 size_t bits = t.bits * t.lanes;
404 ICHECK(bits % 8U == 0U || bits == 1U || bits == 4U);
405 int64_t bytes = ((bits + 7U) / 8U) * size;
406 pool_entry[sid].shape[0] = std::max(pool_entry[sid].shape[0], bytes);
407 pool_entry[sid].dtype = DLDataType{kDLFloat, 32, 1};
408 } else {
409 if (pool_entry[sid].shape.size() == 1) {
410 pool_entry[sid].shape.resize(3, 0);
411 }
412 size_t axis = runtime::DefaultTextureLayoutSeparator(attrs_.shape[i].size(), storage_scope);
413 auto shape = ApplyTexture2DFlattening<int64_t>(attrs_.shape[i], attrs_.shape[i].size(), axis);
414 pool_entry[sid].shape[0] = std::max(pool_entry[sid].shape[0], shape.height);
415 pool_entry[sid].shape[1] = std::max(pool_entry[sid].shape[1], shape.width);
416 CHECK(pool_entry[sid].shape[2] == 0 || pool_entry[sid].shape[2] == shape.channel)
417 << pool_entry[sid].shape[2] << " != " << shape.channel
418 << ", texture channel length must be consistent within a storage pool";
419 pool_entry[sid].shape[2] = shape.channel;
420 CHECK(pool_entry[sid].dtype.bits == 0 || TypeEqual(pool_entry[sid].dtype, t))
421 << DLDataType2String(pool_entry[sid].dtype) << " != " << DLDataType2String(t)
422 << ", pool entry for 2d texure allocations must be of the same type;"
423 << " downstream error from memory planner likely";
424 pool_entry[sid].dtype = t;
425 }
426 }
427
428 // Allocate the space.
429 for (const auto& pit : pool_entry) {
430 // This for loop is very fast since there are usually only a couple of
431 // devices available on the same hardware.
432 const auto& cit = std::find_if(devices_.begin(), devices_.end(), [&pit](const Device& d) {
433 return pit.device_type == static_cast<int>(d.device_type);
434 });
435 Device dev = cit == devices_.end() ? devices_[0] : *cit;
436 if (pit.linked_param.defined()) {
437 storage_pool_.push_back(pit.linked_param);
438 } else {
439 std::vector<int64_t> shape = pit.shape;
440 if (shape.size() == 1) {
441 shape[0] = (shape[0] + 3) / 4;
442 }
443 Optional<String> mem_scope;
444 if (!pit.scope.empty()) {
445 mem_scope = String(pit.scope);
446 }
447 storage_pool_.push_back(NDArray::Empty(shape, pit.dtype, dev, mem_scope));
448 }
449 }
450
451 // Assign the pooled entries. A unified memory pool is used to simplifiy
452 // memory assignment for each node entry. The allocated memory on each device
453 // is mapped to this pool.
454 data_entry_.resize(num_node_entries());
455 data_alignment_.resize(num_node_entries());
456 for (size_t i = 0; i < data_entry_.size(); ++i) {
457 int storage_id = attrs_.storage_id[i];
458 ICHECK_LT(static_cast<size_t>(storage_id), storage_pool_.size());
459 data_entry_[i] = storage_pool_[storage_id].CreateView(attrs_.shape[i], vtype[i]);
460
461 const DLTensor* tmp = data_entry_[i].operator->();
462 data_alignment_[i] = details::GetDataAlignment(*tmp);
463 }
464}

466void GraphExecutor::SetupOpExecs() {
467 op_execs_.resize(this->GetNumOfNodes());
468 input_dltensors_.resize(num_node_entries());
469 output_dltensors_.resize(num_node_entries());
470 both_output_opinput_dltensors_.resize(num_node_entries());
471 std::unordered_set<uint32_t> input_node_eids;
472 for (size_t i = 0; i < input_nodes_.size(); i++) {
473 uint32_t nid = input_nodes_[i];
474 input_node_eids.insert(entry_id(nid, 0));
475 }
476 std::unordered_set<uint32_t> output_node_eids;
477 for (size_t i = 0; i < outputs_.size(); i++) {
478 output_node_eids.insert(entry_id(outputs_[i]));
479 }
480
481 // setup the array and requirements.
482 for (uint32_t nid = 0; nid < this->GetNumOfNodes(); ++nid) {
483 const auto& inode = nodes_[nid];
484 if (inode.op_type == "null") continue;
485 std::vector<DLTensor> args;
486 for (const auto& e : inode.inputs) {
487 uint32_t eid = this->entry_id(e);
488 args.push_back(*(data_entry_[eid].operator->()));
489 }
490 for (uint32_t index = 0; index < inode.param.num_outputs; ++index) {
491 uint32_t eid = this->entry_id(nid, index);
492 args.push_back(*(data_entry_[eid].operator->()));
493 }
494 ICHECK(inode.op_type == "tvm_op") << "Can only take tvm_op as op";
495
496 std::shared_ptr<OpArgs> op_args = nullptr;
497 std::tie(op_execs_[nid], op_args) = CreateTVMOp(inode.param, args);
498
499 for (size_t i = 0; i < inode.inputs.size(); i++) {
500 uint32_t input_eid = this->entry_id(inode.inputs[i]);
501 // check if op input is model input
502 if (input_node_eids.count(input_eid) > 0) {
503 input_dltensors_[input_eid].push_back(
504 static_cast<DLTensor*>(op_args->arg_values[i].v_handle));
505 }
506 // check if any model output is the input of the op
507 if (output_node_eids.count(input_eid) > 0) {
508 both_output_opinput_dltensors_[input_eid].push_back(
509 static_cast<DLTensor*>(op_args->arg_values[i].v_handle));
510 }
511 }
512
513 for (uint32_t i = inode.inputs.size(); i < inode.inputs.size() + inode.param.num_outputs; ++i) {
514 uint32_t output_eid = this->entry_id(nid, i - inode.inputs.size());
515 // check if op output is model output
516 if (output_node_eids.count(output_eid) > 0) {
517 output_dltensors_[output_eid].push_back(
518 static_cast<DLTensor*>(op_args->arg_values[i].v_handle));
519 }
520 }
521 }
522}

524std::pair<std::function<void()>, std::shared_ptr<GraphExecutor::OpArgs>> GraphExecutor::CreateTVMOp(
525 const TVMOpParam& param, const std::vector<DLTensor>& args) {
526 std::shared_ptr<GraphExecutor::OpArgs> arg_ptr = std::make_shared<GraphExecutor::OpArgs>();
527 // setup address.
528 arg_ptr->args = args;
529 if (param.flatten_data) {
530 arg_ptr->shape_data.resize(arg_ptr->args.size());
531 }
532 for (size_t i = 0; i < arg_ptr->args.size(); ++i) {
533 TVMValue v;
534 DLTensor* t = &arg_ptr->args[i];
535 v.v_handle = t;
536 arg_ptr->arg_values.push_back(v);
537 arg_ptr->arg_tcodes.push_back(kTVMDLTensorHandle);
538 if (param.flatten_data) {
539 arg_ptr->shape_data[i] =
540 std::accumulate(t->shape, t->shape + t->ndim, 1, std::multiplies<int64_t>());
541 t->ndim = 1;
542 t->shape = &(arg_ptr->shape_data[i]);
543 }
544 }
545
546 if (param.func_name == "__nop") {
547 return {[]() {}, arg_ptr};
548 } else if (param.func_name == "__copy") {
549 // Perform cross device data copy.
550 // Directly copy data from the input to the output.
551 // TODO(mbs): device_copy cleanup.
552 auto fexec = [arg_ptr]() {
553 DLTensor* from = static_cast<DLTensor*>(arg_ptr->arg_values[0].v_handle);
554 DLTensor* to = static_cast<DLTensor*>(arg_ptr->arg_values[1].v_handle);
555 TVM_CCALL(TVMArrayCopyFromTo(from, to, nullptr));
556 };
557 return {fexec, arg_ptr};
558 }
559
560 // Get compiled function from the module that contains both host and device
561 // code.
562 tvm::runtime::PackedFunc pf = module_.GetFunction(param.func_name, true);
563 ICHECK(pf != nullptr) << "no such function in module: " << param.func_name;
564
565 auto fexec = [arg_ptr, pf]() {
566 TVMRetValue rv;
567 TVMArgs targs(arg_ptr->arg_values.data(), arg_ptr->arg_tcodes.data(),
568 static_cast<int>(arg_ptr->arg_values.size()));
569 pf.CallPacked(targs, &rv);
570 };
571 return {fexec, arg_ptr};
572}

574PackedFunc GraphExecutor::GetFunction(const std::string& name,
575 const ObjectPtr<Object>& sptr_to_self) {
576 // Return member functions during query.
577 if (name == "set_input") {
578 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
579 if (String::CanConvertFrom(args[0])) {
580 int in_idx = this->GetInputIndex(args[0].operator String());
581 if (in_idx >= 0) this->SetInput(in_idx, args[1]);
582 } else {
583 this->SetInput(args[0], args[1]);
584 }
585 });
586 } else if (name == "set_input_zero_copy") {
587 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
588 if (String::CanConvertFrom(args[0])) {
589 int in_idx = this->GetInputIndex(args[0].operator String());
590 if (in_idx >= 0) this->SetInputZeroCopy(in_idx, args[1]);
591 } else {
592 this->SetInputZeroCopy(args[0], args[1]);
593 }
594 });
595 } else if (name == "set_output_zero_copy") {
596 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
597 if (String::CanConvertFrom(args[0])) {
598 int out_idx = this->GetOutputIndex(args[0].operator String());
599 if (out_idx >= 0) this->SetOutputZeroCopy(out_idx, args[1]);
600 } else {
601 this->SetOutputZeroCopy(args[0], args[1]);
602 }
603 });
604 } else if (name == "get_output") {
605 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
606 if (args.num_args == 2) {
607 this->CopyOutputTo(args[0], args[1]);
608 } else {
609 *rv = this->GetOutput(args[0]);
610 }
611 });
612 } else if (name == "get_input") {
613 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
614 int in_idx = 0;
615 if (String::CanConvertFrom(args[0])) {
616 in_idx = this->GetInputIndex(args[0].operator String());
617 } else {
618 in_idx = args[0];
619 }
620 if (in_idx >= 0) {
621 *rv = this->GetInput(in_idx);
622 }
623 });
624 } else if (name == "get_num_outputs") {
625 return PackedFunc(
626 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->NumOutputs(); });
627 } else if (name == "get_num_inputs") {
628 return PackedFunc(
629 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->NumInputs(); });
630 } else if (name == "run") {
631 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { this->Run(); });
632 } else if (name == "run_from_inputs") {
633 return PackedFunc(
634 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
635 CHECK(args.size() % 2 == 0)
636 << "Number of arguments to run_from_inputs must be an even number of key-value pairs";
637 Device host{static_cast<DLDeviceType>(args[0].operator int()), args[1].operator int()};
638 for (int i = 2; i < args.size(); i += 2) {
639 if (String::CanConvertFrom(args[i])) {
640 int in_idx = this->GetInputIndex(args[i].operator String());
641 if (in_idx >= 0) {
642 this->SetInput(in_idx, args[i + 1]);
643 } else {
644 LOG(FATAL) << args[i].operator String() << " is not a valid input name";
645 }
646 } else {
647 this->SetInput(args[i], args[i + 1]);
648 }
649 }
650 this->Run();
651 Array<NDArray> outputs;
652 for (int i = 0; i < this->NumOutputs(); i++) {
653 NDArray out = this->GetOutput(i);
654 NDArray a = NDArray::Empty(out.Shape(), out.DataType(), host);
655 a.CopyFrom(out);
656 outputs.push_back(a);
657 }
658 *rv = outputs;
659 });
660 } else if (name == "load_params") {
661 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
662 this->LoadParams(args[0].operator std::string());
663 });
664 } else if (name == "share_params") {
665 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
666 const auto& module = args[0].operator Module();
667 ICHECK_EQ(module.operator->()->type_key(), std::string("GraphExecutor"));
668 const auto& param_blob = args[1].operator std::string();
669 dmlc::MemoryStringStream strm(const_cast<std::string*>(&param_blob));
670 this->ShareParams(dynamic_cast<const GraphExecutor&>(*module.operator->()), &strm);
671 });
672 } else if (name == "get_input_index") {
673 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
674 CHECK(String::CanConvertFrom(args[0])) << "Input key is not a string";
675 *rv = this->GetInputIndex(args[0].operator String());
676 });
677 } else if (name == "get_input_info") {
678 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
679 auto [shape_info, dtype_info] = this->GetInputInfo();
680 Map<String, ObjectRef> input_info;
681 input_info.Set("shape", shape_info);
682 input_info.Set("dtype", dtype_info);
683 *rv = input_info;
684 });
685 } else {
686 return PackedFunc();
687 }
688}

698// Get all devices for the host and other runtime devices.
699std::vector<Device> GetAllDevice(const TVMArgs& args, int dev_start_arg) {
700 // Reserve the first item as the fallback device.
701 std::vector<Device> ret;
702 Device dev;
703 for (int i = dev_start_arg; i < args.num_args; i += 2) {
704 int dev_type = args[i];
705 dev.device_type = static_cast<DLDeviceType>(dev_type);
706 dev.device_id = args[i + 1];
707 ret.push_back(dev);
708 }
709 return ret;
710}

712// 4-argument version is currently reserved to keep support of calling
713// from tvm4j and javascript, since they don't have heterogeneous
714// execution support yet. For heterogenenous execution, at least 5 arguments will
715// be passed in. The third one is the number of devices.
716// Eventually, we will only probably pass Device for all the languages.
717TVM_REGISTER_GLOBAL("tvm.graph_executor.create").set_body([](TVMArgs args, TVMRetValue* rv) {
718 ICHECK_GE(args.num_args, 4) << "The expected number of arguments for graph_executor.create is "
719 "at least 4, but it has "
720 << args.num_args;
721 PackedFunc lookup_linked_param_func;
722 int dev_start_arg = 2;
723 if (args[2].type_code() == kTVMPackedFuncHandle) {
724 lookup_linked_param_func = args[2];
725 dev_start_arg++;
726 }
727 const auto& devices = GetAllDevice(args, dev_start_arg);
728 *rv = GraphExecutorCreate(args[0], args[1], devices, lookup_linked_param_func);
729});
