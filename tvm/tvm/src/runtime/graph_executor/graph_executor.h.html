216 // Node entry
217 struct NodeEntry {
218 uint32_t node_id;
219 uint32_t index;
220 uint32_t version;
221 inline bool operator==(const NodeEntry& other) const {
222 return node_id == other.node_id && index == other.index && version == other.version;
223 }
224 // JSON Loader
225 void Load(dmlc::JSONReader* reader) {
226 reader->BeginArray();
227 ICHECK(reader->NextArrayItem()) << "invalid json format";
228 reader->Read(&node_id);
229 ICHECK(reader->NextArrayItem()) << "invalid json format";
230 reader->Read(&index);
231 if (reader->NextArrayItem()) {
232 reader->Read(&version);
233 ICHECK(!reader->NextArrayItem()) << "invalid json format";
234 } else {
235 version = 0;
236 }
237 }
238 };
239 // Node
240 struct Node {
241 // operator type in string
242 std::string op_type;
243 // name of the op
244 std::string name;
245 // parameters
246 TVMOpParam param;
247 // inputs
248 std::vector<NodeEntry> inputs;
249 // control deps
250 std::vector<uint32_t> control_deps;
251
252 // JSON Loader
253 void LoadAttrs(dmlc::JSONReader* reader, TVMOpParam* param) {
254 int bitmask = 0;
255 std::string key, value;
256 reader->BeginObject();
257 while (reader->NextObjectItem(&key)) {
258 reader->Read(&value);
259 if (key == "func_name") {
260 param->func_name = value;
261 bitmask |= 1;
262 } else if (key == "num_inputs") {
263 param->num_inputs = strtoul(value.c_str(), nullptr, 10);
264 bitmask |= 2;
265 } else if (key == "num_outputs") {
266 param->num_outputs = strtoul(value.c_str(), nullptr, 10);
267 bitmask |= 4;
268 } else if (key == "flatten_data") {
269 param->flatten_data = strtoul(value.c_str(), nullptr, 10);
270 bitmask |= 8;
271 } else {
272 param->attrs[key] = String(value);
273 }
274 }
275 ICHECK_EQ(bitmask, 1 | 2 | 4 | 8) << "invalid format";
276 }
277 // JSON Loader
278 void Load(dmlc::JSONReader* reader) {
279 reader->BeginObject();
280 int bitmask = 0;
281 std::string key;
282 while (reader->NextObjectItem(&key)) {
283 if (key == "op") {
284 reader->Read(&op_type);
285 bitmask |= 1;
286 } else if (key == "name") {
287 reader->Read(&name);
288 bitmask |= 2;
289 } else if (key == "inputs") {
290 reader->Read(&inputs);
291 bitmask |= 4;
292 } else if (key == "attr" || key == "attrs") {
293 this->LoadAttrs(reader, &param);
294 } else if (key == "control_deps") {
295 reader->Read(&control_deps);
296 } else {
297 LOG(FATAL) << "do not support key " << key;
298 }
299 }
300 ICHECK_EQ(bitmask, 1 | 2 | 4) << "invalid format";
301 }
302 };
303 struct GraphAttr {
304 size_t storage_num_not_alloctaed{0};
305 std::vector<int> storage_id;
306 std::vector<int> device_index;
307 std::vector<std::string> dltype;
308 std::vector<std::string> storage_scope;
309 std::vector<std::vector<int64_t>> shape;
310 // The graph attribute fields.
311 void Load(dmlc::JSONReader* reader) {
312 reader->BeginObject();
313 int bitmask = 0;
314 std::string key, type;
315 while (reader->NextObjectItem(&key)) {
316 if (key == "dltype") {
317 reader->BeginArray();
318 ICHECK(reader->NextArrayItem());
319 reader->Read(&type);
320 ICHECK_EQ(type, "list_str");
321 ICHECK(reader->NextArrayItem());
322 reader->Read(&dltype);
323 ICHECK(!reader->NextArrayItem());
324 bitmask |= 1;
325 } else if (key == "storage_id") {
326 reader->BeginArray();
327 ICHECK(reader->NextArrayItem());
328 reader->Read(&type);
329 ICHECK_EQ(type, "list_int");
330 ICHECK(reader->NextArrayItem());
331 reader->Read(&storage_id);
332 ICHECK(!reader->NextArrayItem());
333 bitmask |= 2;
334 } else if (key == "storage_scope") {
335 reader->BeginArray();
336 ICHECK(reader->NextArrayItem());
337 reader->Read(&type);
338 ICHECK_EQ(type, "list_str");
339 ICHECK(reader->NextArrayItem());
340 reader->Read(&storage_scope);
341 ICHECK(!reader->NextArrayItem());
342 bitmask |= 1;
343 } else if (key == "shape") {
344 reader->BeginArray();
345 ICHECK(reader->NextArrayItem());
346 reader->Read(&type);
347 ICHECK_EQ(type, "list_shape");
348 ICHECK(reader->NextArrayItem());
349 reader->Read(&shape);
350 ICHECK(!reader->NextArrayItem());
351 bitmask |= 4;
352 } else if (key == "device_index") {
353 reader->BeginArray();
354 ICHECK(reader->NextArrayItem());
355 reader->Read(&type);
356 ICHECK_EQ(type, "list_int");
357 ICHECK(reader->NextArrayItem());
358 reader->Read(&device_index);
359 ICHECK(!reader->NextArrayItem());
360 } else {
361 reader->BeginArray();
362 ICHECK(reader->NextArrayItem());
363 reader->Read(&type);
364 if (type == "list_int") {
365 ICHECK(reader->NextArrayItem());
366 std::vector<int> temp;
367 reader->Read(&temp);
368 } else if (type == "size_t") {
369 ICHECK(reader->NextArrayItem());
370 size_t temp;
371 reader->Read(&temp);
372 } else {
373 LOG(FATAL) << "cannot skip graph attr " << key;
374 }
375 ICHECK(!reader->NextArrayItem());
376 }
377 }
378 ICHECK_EQ(bitmask, 1 | 2 | 4) << "invalid format";
379 }
380 };
381 // The graph attribute fields.
382 void Load(dmlc::JSONReader* reader) {
383 reader->BeginObject();
384 int bitmask = 0;
385 std::string key;
386 while (reader->NextObjectItem(&key)) {
387 if (key == "nodes") {
388 reader->Read(&nodes_);
389 bitmask |= 1;
390 } else if (key == "arg_nodes") {
391 reader->Read(&input_nodes_);
392 bitmask |= 2;
393 } else if (key == "node_row_ptr") {
394 reader->Read(&node_row_ptr_);
395 bitmask |= 4;
396 } else if (key == "heads") {
397 reader->Read(&outputs_);
398 bitmask |= 8;
399 } else if (key == "attrs") {
400 reader->Read(&attrs_);
401 bitmask |= 16;
402 } else if (key == "metadata") {
403 break;
404 } else {
405 LOG(FATAL) << "key " << key << " is not supported";
406 }
407 }
408 ICHECK_EQ(bitmask, 1 | 2 | 4 | 8 | 16) << "invalid format";
409 }
