43// deleter of RPC remote array
44static void RemoteNDArrayDeleter(Object* obj) {
45 auto* ptr = static_cast<NDArray::Container*>(obj);
46 RemoteSpace* space = static_cast<RemoteSpace*>(ptr->dl_tensor.data);
47 if (ptr->manager_ctx != nullptr) {
48 space->sess->FreeHandle(ptr->manager_ctx, kTVMNDArrayHandle);
49 }
50 delete space;
51 delete ptr;
52}

54/*!
55 * \brief Build a local NDArray with remote backing storage.
56 * \param sess the RPCSession which owns the given handle.
57 * \param handle A pointer valid on the remote end which should form the `data` field of the
58 * underlying DLTensor.
59 * \param template_tensor An empty DLTensor whose shape and dtype fields are used to fill the newly
60 * created array. Needed because it's difficult to pass a shape vector as a PackedFunc arg.
61 * \param dev Remote device used with this tensor. Must have non-zero RPCSessMask.
62 * \param remote_ndarray_handle The handle returned by RPC server to identify the NDArray.
63 */
64NDArray NDArrayFromRemoteOpaqueHandle(std::shared_ptr<RPCSession> sess, void* handle,
65 DLTensor* template_tensor, Device dev,
66 void* remote_ndarray_handle) {
67 ICHECK_EQ(sess->table_index(), GetRPCSessionIndex(dev))
68 << "The Device given does not belong to the given session";
69 RemoteSpace* space = new RemoteSpace();
70 space->sess = sess;
71 space->data = handle;
72 std::vector<int64_t> shape_vec{template_tensor->shape,
73 template_tensor->shape + template_tensor->ndim};
74 NDArray::Container* data = new NDArray::Container(static_cast<void*>(space), std::move(shape_vec),
75 template_tensor->dtype, dev);
76 data->manager_ctx = remote_ndarray_handle;
77 data->SetDeleter(RemoteNDArrayDeleter);
78 return NDArray(GetObjectPtr<Object>(data));
79}

81/*!
82 * \brief A wrapped remote function as a PackedFunc.
83 */
84class RPCWrappedFunc : public Object {
85 public:
86 RPCWrappedFunc(void* handle, std::shared_ptr<RPCSession> sess) : handle_(handle), sess_(sess) {}
87
88 void operator()(TVMArgs args, TVMRetValue* rv) const {
89 std::vector<TVMValue> values(args.values, args.values + args.size());
90 std::vector<int> type_codes(args.type_codes, args.type_codes + args.size());
91 std::vector<std::unique_ptr<DLTensor>> temp_dltensors;
92
93 // scan and check whether we need rewrite these arguments
94 // to their remote variant.
95 for (int i = 0; i < args.size(); ++i) {
96 if (args[i].IsObjectRef<String>()) {
97 String str = args[i];
98 type_codes[i] = kTVMStr;
99 values[i].v_str = str.c_str();
100 continue;
101 }
102 int tcode = type_codes[i];
103 switch (tcode) {
104 case kTVMDLTensorHandle:
105 case kTVMNDArrayHandle: {
106 // Pass NDArray as DLTensor, NDArray and DLTensor
107 // are compatible to each other, just need to change the index.
108 type_codes[i] = kTVMDLTensorHandle;
109 // translate to a remote view of DLTensor
110 auto dptr = std::make_unique<DLTensor>(*static_cast<DLTensor*>(values[i].v_handle));
111 dptr->device = RemoveSessMask(dptr->device);
112 dptr->data = static_cast<RemoteSpace*>(dptr->data)->data;
113 values[i].v_handle = dptr.get();
114 temp_dltensors.emplace_back(std::move(dptr));
115 break;
116 }
117 case kDLDevice: {
118 values[i].v_device = RemoveSessMask(values[i].v_device);
119 break;
120 }
121 case kTVMPackedFuncHandle:
122 case kTVMModuleHandle: {
123 values[i].v_handle = UnwrapRemoteValueToHandle(TVMArgValue(values[i], tcode));
124 break;
125 }
126 }
127 }
128 auto set_return = [this, rv](TVMArgs args) { this->WrapRemoteReturnToValue(args, rv); };
129 sess_->CallFunc(handle_, values.data(), type_codes.data(), args.size(), set_return);
130 }
131
132 ~RPCWrappedFunc() {
133 try {
134 sess_->FreeHandle(handle_, kTVMPackedFuncHandle);
135 } catch (const Error& e) {
136 // fault tolerance to remote close
137 }
138 }
139
140 private:
141 // remote function handle
142 void* handle_{nullptr};
143 // pointer to the session.
144 std::shared_ptr<RPCSession> sess_;
145
146 // unwrap a remote value to the underlying handle.
147 void* UnwrapRemoteValueToHandle(const TVMArgValue& arg) const;
148 // wrap a remote return via Set
149 void WrapRemoteReturnToValue(TVMArgs args, TVMRetValue* rv) const;
150
151 // remove a remote session mask
152 Device RemoveSessMask(Device dev) const {
153 ICHECK(IsRPCSessionDevice(dev)) << "Can not pass in local device";
154 ICHECK_EQ(GetRPCSessionIndex(dev), sess_->table_index())
155 << "Can not pass in device with a different remote session";
156 return RemoveRPCSessionMask(dev);
157 }
158};

160// RPC that represents a remote module session.
161class RPCModuleNode final : public ModuleNode {
162 public:
163 RPCModuleNode(void* module_handle, std::shared_ptr<RPCSession> sess)
164 : module_handle_(module_handle), sess_(sess) {}
165
166 ~RPCModuleNode() {
167 if (module_handle_ != nullptr) {
168 try {
169 sess_->FreeHandle(module_handle_, kTVMModuleHandle);
170 } catch (const Error& e) {
171 // fault tolerance to remote close
172 }
173 module_handle_ = nullptr;
174 }
175 }
176
177 const char* type_key() const final { return "rpc"; }
178
179 PackedFunc GetFunction(const std::string& name, const ObjectPtr<Object>& sptr_to_self) final {
180 if (name == "CloseRPCConnection") {
181 return PackedFunc([this](TVMArgs, TVMRetValue*) { sess_->Shutdown(); });
182 }
183
184 if (module_handle_ == nullptr) {
185 return WrapRemoteFunc(sess_->GetFunction(name));
186 } else {
187 InitRemoteFunc(&remote_mod_get_function_, "tvm.rpc.server.ModuleGetFunction");
188 return remote_mod_get_function_(GetRef<Module>(this), name, true);
189 }
190 }
191
192 std::string GetSource(const std::string& format) final {
193 LOG(FATAL) << "GetSource for rpc Module is not supported";
194 return "";
195 }
196
197 PackedFunc GetTimeEvaluator(const std::string& name, Device dev, int number, int repeat,
198 int min_repeat_ms, int limit_zero_time_iterations,
199 int cooldown_interval_ms, int repeats_to_cooldown,
200 const std::string& f_preproc_name) {
201 InitRemoteFunc(&remote_get_time_evaluator_, "runtime.RPCTimeEvaluator");
202 // Remove session mask because we pass dev by parts.
203 ICHECK_EQ(GetRPCSessionIndex(dev), sess_->table_index())
204 << "ValueError: Need to pass the matched remote device to RPCModule.GetTimeEvaluator";
205 dev = RemoveRPCSessionMask(dev);
206
207 if (module_handle_ != nullptr) {
208 return remote_get_time_evaluator_(GetRef<Module>(this), name,
209 static_cast<int>(dev.device_type), dev.device_id, number,
210 repeat, min_repeat_ms, limit_zero_time_iterations,
211 cooldown_interval_ms, repeats_to_cooldown, f_preproc_name);
212 } else {
213 return remote_get_time_evaluator_(Optional<Module>(nullptr), name,
214 static_cast<int>(dev.device_type), dev.device_id, number,
215 repeat, min_repeat_ms, limit_zero_time_iterations,
216 cooldown_interval_ms, repeats_to_cooldown, f_preproc_name);
217 }
218 }

230 const std::shared_ptr<RPCSession>& sess() { return sess_; }

232 void* module_handle() const { return module_handle_; }

234 private:
235 template <typename FType>
236 void InitRemoteFunc(FType* func, const std::string& name) {
237 if (*func != nullptr) return;
238 RPCSession::PackedFuncHandle handle = sess_->GetFunction(name);
239 ICHECK(handle != nullptr) << "Cannot found remote function " << name;
240 *func = WrapRemoteFunc(handle);
241 }
242
243 PackedFunc WrapRemoteFunc(RPCSession::PackedFuncHandle handle) {
244 if (handle == nullptr) return PackedFunc();
245 auto wf = std::make_shared<RPCWrappedFunc>(handle, sess_);
246 return PackedFunc([wf](TVMArgs args, TVMRetValue* rv) { return wf->operator()(args, rv); });
247 }
248
249 // The module handle
250 void* module_handle_{nullptr};
251 // The local channel
252 std::shared_ptr<RPCSession> sess_;
253 // remote function to get time evaluator
254 TypedPackedFunc<PackedFunc(Optional<Module>, std::string, int, int, int, int, int, int, int, int,
255 std::string)>
256 remote_get_time_evaluator_;
257 // remote function getter for modules.
258 TypedPackedFunc<PackedFunc(Module, std::string, bool)> remote_mod_get_function_;
259 // remote function getter for load module
260 TypedPackedFunc<Module(std::string)> remote_load_module_;
261 // remote function getter for load module
262 TypedPackedFunc<void(Module, Module)> remote_import_module_;
263};

265void* RPCWrappedFunc::UnwrapRemoteValueToHandle(const TVMArgValue& arg) const {
266 if (arg.type_code() == kTVMModuleHandle) {
267 Module mod = arg;
268 std::string tkey = mod->type_key();
269 ICHECK_EQ(tkey, "rpc") << "ValueError: Cannot pass a non-RPC module to remote";
270 auto* rmod = static_cast<RPCModuleNode*>(mod.operator->());
271 ICHECK(rmod->sess() == sess_)
272 << "ValueError: Cannot pass in module into a different remote session";
273 return rmod->module_handle();
274 } else {
275 LOG(FATAL) << "ValueError: Cannot pass type " << runtime::ArgTypeCode2Str(arg.type_code())
276 << " as an argument to the remote";
277 return nullptr;
278 }
279}

281void RPCWrappedFunc::WrapRemoteReturnToValue(TVMArgs args, TVMRetValue* rv) const {
282 int tcode = args[0];
283
284 if (tcode == kTVMNullptr) return;
285 if (tcode == kTVMPackedFuncHandle) {
286 ICHECK_EQ(args.size(), 2);
287 void* handle = args[1];
288 auto wf = std::make_shared<RPCWrappedFunc>(handle, sess_);
289 *rv = PackedFunc([wf](TVMArgs args, TVMRetValue* rv) { return wf->operator()(args, rv); });
290 } else if (tcode == kTVMModuleHandle) {
291 ICHECK_EQ(args.size(), 2);
292 void* handle = args[1];
293 auto n = make_object<RPCModuleNode>(handle, sess_);
294 *rv = Module(n);
295 } else if (tcode == kTVMDLTensorHandle || tcode == kTVMNDArrayHandle) {
296 ICHECK_EQ(args.size(), 3);
297 DLTensor* tensor = args[1];
298 void* nd_handle = args[2];
299 *rv = NDArrayFromRemoteOpaqueHandle(sess_, tensor->data, tensor,
300 AddRPCSessionMask(tensor->device, sess_->table_index()),
301 nd_handle);
302 } else {
303 ICHECK_EQ(args.size(), 2);
304 *rv = args[1];
305 }
306}

321/*!
322 * \brief Flush the cache.
323 * \param addr The address of data we want to flush
324 * \param len The length of data
325 */
326/*
327 * When we are in the tuning of TVM, we will make TVM occupy
328 * the cache fully and doesn't flush it during iteration.
329 * This has problems then in e2e testing, since arrays that
330 * we assume exist in cache (ie. weights) are evicted during e2e runs,
331 * which leads to lower performance.
332 */
333inline void CPUCacheFlushImpl(const char* addr, unsigned int len) {
334#if (defined(_M_X64) || defined(__x86_64__) || defined(__aarch64__))
335
336#if defined(__aarch64__)
337 size_t ctr_el0 = 0;
338 asm volatile("mrs %0, ctr_el0" : "=r"(ctr_el0));
339 const size_t cache_line = 4 << ((ctr_el0 >> 16) & 15);
340#else
341 const size_t cache_line = 64;
342#endif
343
344 if (addr == nullptr || len <= 0) {
345 return;
346 }
347
348 for (uintptr_t uptr = (uintptr_t)addr & ~(cache_line - 1); uptr < (uintptr_t)addr + len;
349 uptr += cache_line) {
350#if defined(__aarch64__)
351 asm volatile("dc civac, %0\n\t" : : "r"(reinterpret_cast<const void*>(uptr)) : "memory");
352#else
353 _mm_clflush(reinterpret_cast<const void*>(uptr));
354#endif
355 }
356
357#if defined(__aarch64__)
358 asm volatile("dmb ishst" : : : "memory");
359#endif
360
361#endif
362}
363
364inline void CPUCacheFlush(int begin_index, const TVMArgs& args) {
365 for (int i = begin_index; i < args.size(); i++) {
366 CPUCacheFlushImpl(static_cast<char*>((args[i].operator DLTensor*()->data)),
367 GetDataSize(*(args[i].operator DLTensor*())));
368 }
369}

371TVM_REGISTER_GLOBAL("runtime.RPCTimeEvaluator")
372 .set_body_typed([](Optional<Module> opt_mod, std::string name, int device_type, int device_id,
373 int number, int repeat, int min_repeat_ms, int limit_zero_time_iterations,
374 int cooldown_interval_ms, int repeats_to_cooldown,
375 std::string f_preproc_name) {
376 Device dev;
377 dev.device_type = static_cast<DLDeviceType>(device_type);
378 dev.device_id = device_id;
379 if (opt_mod.defined()) {
380 Module m = opt_mod.value();
381 std::string tkey = m->type_key();
382 if (tkey == "rpc") {
383 return static_cast<RPCModuleNode*>(m.operator->())
384 ->GetTimeEvaluator(name, dev, number, repeat, min_repeat_ms,
385 limit_zero_time_iterations, cooldown_interval_ms,
386 repeats_to_cooldown, f_preproc_name);
387 } else {
388 PackedFunc f_preproc;
389 if (!f_preproc_name.empty()) {
390 auto* pf_preproc = runtime::Registry::Get(f_preproc_name);
391 ICHECK(pf_preproc != nullptr)
392 << "Cannot find " << f_preproc_name << " in the global function";
393 f_preproc = *pf_preproc;
394 }
395 PackedFunc pf = m.GetFunction(name, true);
396 CHECK(pf != nullptr) << "Cannot find " << name << " in the global registry";
397 return profiling::WrapTimeEvaluator(pf, dev, number, repeat, min_repeat_ms,
398 limit_zero_time_iterations, cooldown_interval_ms,
399 repeats_to_cooldown, f_preproc);
400 }
401 } else {
402 auto* pf = runtime::Registry::Get(name);
403 ICHECK(pf != nullptr) << "Cannot find " << name << " in the global function";
404 PackedFunc f_preproc;
405 if (!f_preproc_name.empty()) {
406 auto* pf_preproc = runtime::Registry::Get(f_preproc_name);
407 ICHECK(pf_preproc != nullptr)
408 << "Cannot find " << f_preproc_name << " in the global function";
409 f_preproc = *pf_preproc;
410 }
411 return profiling::WrapTimeEvaluator(*pf, dev, number, repeat, min_repeat_ms,
412 limit_zero_time_iterations, cooldown_interval_ms,
413 repeats_to_cooldown, f_preproc);
414 }
415 });
