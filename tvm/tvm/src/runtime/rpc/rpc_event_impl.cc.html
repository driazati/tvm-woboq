20/*!
21 * \file rpc_event_impl.cc
22 * \brief Event driven RPC server implementation.
23 */
24#include <tvm/runtime/registry.h>
25
26#include <memory>
27
28#include "rpc_endpoint.h"
29#include "rpc_local_session.h"
30
31namespace tvm {
32namespace runtime {
33
34PackedFunc CreateEventDrivenServer(PackedFunc fsend, std::string name, std::string remote_key) {
35 static PackedFunc frecv(
36 [](TVMArgs args, TVMRetValue* rv) { LOG(FATAL) << "Do not allow explicit receive"; });
37
38 auto ch = std::make_unique<CallbackChannel>(fsend, frecv);
39 std::shared_ptr<RPCEndpoint> sess = RPCEndpoint::Create(std::move(ch), name, remote_key);
40 return PackedFunc([sess](TVMArgs args, TVMRetValue* rv) {
41 int ret = sess->ServerAsyncIOEventHandler(args[0], args[1]);
42 *rv = ret;
43 });
44}
45
46TVM_REGISTER_GLOBAL("rpc.CreateEventDrivenServer").set_body_typed(CreateEventDrivenServer);
47} // namespace runtime
48} // namespace tvm
