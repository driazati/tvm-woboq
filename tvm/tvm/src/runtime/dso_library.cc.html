40/*!
41 * \brief Dynamic shared library object used to load
42 * and retrieve symbols by name. This is the default
43 * module TVM uses for host-side AOT compilation.
44 */
45class DSOLibrary final : public Library {
46 public:
47 ~DSOLibrary();
48 /*!
49 * \brief Initialize by loading and storing
50 * a handle to the underlying shared library.
51 * \param name The string name/path to the
52 * shared library over which to initialize.
53 */
54 void Init(const std::string& name);
55 /*!
56 * \brief Returns the symbol address within
57 * the shared library for a given symbol name.
58 * \param name The name of the symbol.
59 * \return The symbol.
60 */
61 void* GetSymbol(const char* name) final;
62
63 private:
64 /*! \brief Private implementation of symbol lookup.
65 * Implementation is operating system dependent.
66 * \param The name of the symbol.
67 * \return The symbol.
68 */
69 void* GetSymbol_(const char* name);
70 /*! \brief Implementation of shared library load.
71 * Implementation is operating system dependent.
72 * \param The name/path of the shared library.
73 */
74 void Load(const std::string& name);
75 /*! \brief Implementation of shared library unload.
76 * Implementation is operating system dependent.
77 */
78 void Unload();
79
80#if defined(_WIN32)
81 //! \brief Windows library handle
82 HMODULE lib_handle_{nullptr};
83#else
84 // \brief Linux library handle
85 void* lib_handle_{nullptr};
86#endif
87};
88
89DSOLibrary::~DSOLibrary() {
90 if (lib_handle_) Unload();
91}
92
93void DSOLibrary::Init(const std::string& name) { Load(name); }
94
95void* DSOLibrary::GetSymbol(const char* name) { return GetSymbol_(name); }
96
97#if defined(_WIN32)
98
99void* DSOLibrary::GetSymbol_(const char* name) {
100 return reinterpret_cast<void*>(GetProcAddress(lib_handle_, (LPCSTR)name)); // NOLINT(*)
101}
102
103void DSOLibrary::Load(const std::string& name) {
104 // use wstring version that is needed by LLVM.
105 std::wstring wname(name.begin(), name.end());
106 lib_handle_ = LoadLibraryW(wname.c_str());
107 ICHECK(lib_handle_ != nullptr) << "Failed to load dynamic shared library " << name;
108}
109
110void DSOLibrary::Unload() {
111 FreeLibrary(lib_handle_);
112 lib_handle_ = nullptr;
113}
114
115#else
116
117void DSOLibrary::Load(const std::string& name) {
118 lib_handle_ = dlopen(name.c_str(), RTLD_LAZY | RTLD_LOCAL);
119 ICHECK(lib_handle_ != nullptr) << "Failed to load dynamic shared library " << name << " "
120 << dlerror();
121}
122
123void* DSOLibrary::GetSymbol_(const char* name) { return dlsym(lib_handle_, name); }
124
125void DSOLibrary::Unload() {
126 dlclose(lib_handle_);
127 lib_handle_ = nullptr;
128}
129
130#endif
131
132ObjectPtr<Library> CreateDSOLibraryObject(std::string library_path) {
133 auto n = make_object<DSOLibrary>();
134 n->Init(library_path);
135 return n;
136}
