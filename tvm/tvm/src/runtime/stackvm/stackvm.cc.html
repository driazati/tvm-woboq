38#define STACK_VM_BINOP(OP, FIELD) \
39 { \
40 stack[sp - 1].FIELD = stack[sp - 1].FIELD OP stack[sp].FIELD; \
41 sp -= 1; \
42 pc += 1; \
43 }
44
45#define STACK_VM_CMPOP(OP, FIELD) \
46 { \
47 stack[sp - 1].v_int64 = stack[sp - 1].FIELD OP stack[sp].FIELD; \
48 sp -= 1; \
49 pc += 1; \
50 }
51
52#define STACK_VM_LOAD(FIELD, DST_TYPE, SRC_TYPE) \
53 { \
54 int index = code[pc + 1].v_int; \
55 stack[sp] FIELD = static_cast<DST_TYPE>(static_cast<SRC_TYPE*>(stack[sp].v_handle)[index]); \
56 pc += 2; \
57 }
58
59#define STACK_VM_STORE(FIELD, DST_TYPE) \
60 { \
61 int index = code[pc + 1].v_int; \
62 static_cast<DST_TYPE*>(stack[sp - 1].v_handle)[index] = \
63 static_cast<DST_TYPE>(stack[sp] FIELD); \
64 sp -= 2; \
65 pc += 2; \
66 }
67
68#define STACK_VM_PRINT_CODE0(CODE) \
69 case CODE: { \
70 os << "[" << pc << "]\t" << #CODE << std::endl; \
71 return pc + 1; \
72 }
73
74#define STACK_VM_PRINT_CODE1(CODE) \
75 case CODE: { \
76 os << "[" << pc << "]\t" << #CODE << " " << code[pc + 1].v_int << "\n" \
77 << "[" << pc + 1 << "]" << std::endl; \
78 return pc + 2; \
79 }
80
81#define STACK_VM_PRINT_CODE2(CODE) \
82 case CODE: { \
83 os << "[" << pc << "]\t" << #CODE << " " << code[pc + 1].v_int << " " << code[pc + 2].v_int \
84 << "\n" \
85 << "[" << pc + 1 << "]" << std::endl \
86 << "[" << pc + 2 << "]" << std::endl; \
87 return pc + 3; \
88 }
89
90#define STACK_VM_PRINT_HEAP_ACCESS(CODE) \
91 case CODE: { \
92 os << "[" << pc << "]\t" << #CODE << " " << code[pc + 1].v_int << " " \
93 << heap_id_name[code[pc + 1].v_int] << "\n" \
94 << "[" << pc + 1 << "]" << std::endl; \
95 return pc + 2; \
96 }
97
98#define STACK_VM_PRINT_JUMP(CODE) \
99 case CODE: { \
100 os << "[" << pc << "]\t" << #CODE << " rel=" << code[pc + 1].v_int << " to " \
101 << pc + code[pc + 1].v_int << '\n' \
102 << "[" << pc + 1 << "]" << std::endl; \
103 return pc + 2; \
104 }

106int64_t StackVM::PrintCode(std::ostream& os, int64_t pc) const {
107 switch (code[pc].op_code) {
108 // int
109 STACK_VM_PRINT_CODE0(ADD_I64);
110 STACK_VM_PRINT_CODE0(SUB_I64);
111 STACK_VM_PRINT_CODE0(MUL_I64);
112 STACK_VM_PRINT_CODE0(MOD_I64);
113 STACK_VM_PRINT_CODE0(DIV_I64);
114 STACK_VM_PRINT_CODE0(EQ_I64);
115 STACK_VM_PRINT_CODE0(LT_I64);
116 STACK_VM_PRINT_CODE0(LE_I64);
117 // floats
118 STACK_VM_PRINT_CODE0(ADD_F64);
119 STACK_VM_PRINT_CODE0(SUB_F64);
120 STACK_VM_PRINT_CODE0(MUL_F64);
121 STACK_VM_PRINT_CODE0(DIV_F64);
122 STACK_VM_PRINT_CODE0(EQ_F64);
123 STACK_VM_PRINT_CODE0(LT_F64);
124 STACK_VM_PRINT_CODE0(LE_F64);
125 // handle.
126 STACK_VM_PRINT_CODE0(EQ_HANDLE);
127 // addressing load
128 STACK_VM_PRINT_CODE1(ARRAY_LOAD_UINT32);
129 STACK_VM_PRINT_CODE1(ARRAY_LOAD_INT32);
130 STACK_VM_PRINT_CODE1(ARRAY_LOAD_INT64);
131 STACK_VM_PRINT_CODE1(ARRAY_LOAD_FP64);
132 STACK_VM_PRINT_CODE1(ARRAY_LOAD_HANDLE);
133 STACK_VM_PRINT_CODE1(ARRAY_LOAD_TVMVALUE);
134 STACK_VM_PRINT_CODE1(ARRAY_STORE_UINT32);
135 STACK_VM_PRINT_CODE1(ARRAY_STORE_INT32);
136 STACK_VM_PRINT_CODE1(ARRAY_STORE_INT64);
137 STACK_VM_PRINT_CODE1(ARRAY_STORE_FP64);
138 STACK_VM_PRINT_CODE1(ARRAY_STORE_HANDLE);
139 STACK_VM_PRINT_CODE1(ARRAY_STORE_TVMVALUE);
140 STACK_VM_PRINT_CODE0(NOT);
141 STACK_VM_PRINT_CODE0(ADDR_ADD);
142 // stack ops
143 STACK_VM_PRINT_CODE1(PUSH_I64);
144 STACK_VM_PRINT_CODE1(PUSH_VALUE);
145 STACK_VM_PRINT_CODE0(POP);
146 STACK_VM_PRINT_CODE0(SELECT);
147 STACK_VM_PRINT_HEAP_ACCESS(STORE_HEAP);
148 STACK_VM_PRINT_HEAP_ACCESS(LOAD_HEAP);
149 STACK_VM_PRINT_CODE1(ASSERT);
150 STACK_VM_PRINT_JUMP(RJUMP_IF_TRUE);
151 STACK_VM_PRINT_JUMP(RJUMP_IF_FALSE);
152 STACK_VM_PRINT_JUMP(RJUMP);
153 STACK_VM_PRINT_CODE1(ASSERT_SP);
154 // Intrinsics
155 STACK_VM_PRINT_CODE2(TVM_STRUCT_GET);
156 STACK_VM_PRINT_CODE2(TVM_STRUCT_SET);
157 // Allocate data by 8 bytes.
158 STACK_VM_PRINT_CODE1(TVM_STACK_ALLOCA_BY_8BYTE);
159 STACK_VM_PRINT_CODE0(TVM_DEVICE_ALLOCA);
160 STACK_VM_PRINT_CODE0(TVM_DEVICE_FREE);
161 STACK_VM_PRINT_CODE0(TVM_THROW_LAST_ERROR);
162 // packed function.
163 case CALL_PACKED_LOWERED: {
164 int call_fid = code[pc + 1].v_int;
165 int begin = code[pc + 2].v_int;
166 int end = code[pc + 3].v_int;
167 os << "[" << pc << "]\tCALL_PACKED_FUNC "
168 << " fid=" << call_fid << " begin=" << begin << " end=" << end;
169 os << '\n';
170 for (int i = 0; i < 3; ++i) {
171 os << "[" << pc + 1 + i << "]" << std::endl;
172 }
173 return pc + 4;
174 }
175 }
176 LOG(FATAL) << "unknown op code " << code[pc].op_code;
177 return 0;
178}

180std::ostream& operator<<(std::ostream& os, const StackVM& vm) { // NOLINT(*)
181 int64_t pc = 0;
182 const int64_t code_size = static_cast<int64_t>(vm.code.size());
183 os << "Program dump: code-size=" << code_size << '\n' << "----------begin-----------------\n";
184 while (pc < code_size) {
185 pc = vm.PrintCode(os, pc);
186 }
187 os << "----------end--------------------\n";
188 return os;
189}

191void StackVM::Run(const runtime::TVMArgs& args, runtime::ModuleNode* mod_ctx) const {
192 StackVM::State* s = StackVM::ThreadLocalState();
193 if (s->heap.size() < heap_size) {
194 s->heap.resize(heap_size);
195 }
196 s->sp = 0;
197 s->pc = 0;
198 s->mod_ctx = mod_ctx;
199 s->heap[0].v_handle = (void*)args.values; // NOLINT(*)
200 s->heap[1].v_handle = (void*)args.type_codes; // NOLINT(*)
201 s->heap[2].v_int64 = args.num_args;
202 this->Run(s);
203}

205void StackVM::InitCache() {
206 extern_func_cache_.clear();
207 extern_func_cache_.resize(extern_func_name.size(), PackedFunc(nullptr));
208}

210void StackVM::Save(dmlc::Stream* strm) const {
211 // to be endian invariant.
212 std::vector<int32_t> code_copy(code.size());
213 std::transform(code.begin(), code.end(), code_copy.begin(), [](Code c) { return c.v_int; });
214 strm->Write(code_copy);
215 strm->Write(str_data);
216 strm->Write(extern_func_name);
217 strm->Write(heap_id_name);
218 strm->Write(heap_size);
219 strm->Write(stack_size);
220}
221
222bool StackVM::Load(dmlc::Stream* strm) {
223 // to be endian invariant.
224 std::vector<int32_t> code_copy;
225 if (!strm->Read(&code_copy)) return false;
226 code.resize(code_copy.size());
227 std::transform(code_copy.begin(), code_copy.end(), code.begin(), [](int v) {
228 Code code;
229 code.v_int = v;
230 return code;
231 });
232 if (!strm->Read(&str_data)) return false;
233 if (!strm->Read(&extern_func_name)) return false;
234 if (!strm->Read(&heap_id_name)) return false;
235 if (!strm->Read(&heap_size)) return false;
236 if (!strm->Read(&stack_size)) return false;
237 this->InitCache();
238 return true;
239}

241void StackVM::Run(State* s) const {
242 int64_t sp = s->sp;
243 int64_t pc = s->pc;
244 int64_t alloca_sp = s->sp;
245 std::vector<TVMValue>& stack = s->stack;
246 std::vector<TVMValue>& heap = s->heap;
247 if (stack.size() < stack_size) {
248 stack.resize(stack_size);
249 }
250 int64_t stack_cap = static_cast<int64_t>(stack_size - 4);
251 if (heap.size() < heap_size) {
252 heap.resize(heap_size);
253 }
254 const int64_t code_size = static_cast<int64_t>(code.size());
255 while (pc < code_size) {
256 switch (code[pc].op_code) {
257 case ADD_I64:
258 STACK_VM_BINOP(+, v_int64);
259 break;
260 case SUB_I64:
261 STACK_VM_BINOP(-, v_int64);
262 break;
263 case MUL_I64:
264 STACK_VM_BINOP(*, v_int64);
265 break;
266 case DIV_I64:
267 STACK_VM_BINOP(/, v_int64);
268 break;
269 case MOD_I64:
270 STACK_VM_BINOP(%, v_int64);
271 break;
272 case EQ_I64:
273 STACK_VM_CMPOP(==, v_int64);
274 break;
275 case LT_I64:
276 STACK_VM_CMPOP(<, v_int64);
277 break;
278 case LE_I64:
279 STACK_VM_CMPOP(<=, v_int64);
280 break;
281 case ADD_F64:
282 STACK_VM_BINOP(+, v_float64);
283 break;
284 case SUB_F64:
285 STACK_VM_BINOP(-, v_float64);
286 break;
287 case MUL_F64:
288 STACK_VM_BINOP(*, v_float64);
289 break;
290 case DIV_F64:
291 STACK_VM_BINOP(/, v_float64);
292 break;
293 case EQ_F64:
294 STACK_VM_CMPOP(==, v_float64);
295 break;
296 case LT_F64:
297 STACK_VM_CMPOP(<, v_float64);
298 break;
299 case LE_F64:
300 STACK_VM_CMPOP(<=, v_float64);
301 break;
302 case EQ_HANDLE:
303 STACK_VM_CMPOP(==, v_handle);
304 break;
305 // addressing
306 case ARRAY_LOAD_UINT32:
307 STACK_VM_LOAD(.v_int64, int64_t, uint32_t);
308 break;
309 case ARRAY_LOAD_INT32:
310 STACK_VM_LOAD(.v_int64, int64_t, int32_t);
311 break;
312 case ARRAY_LOAD_INT64:
313 STACK_VM_LOAD(.v_int64, int64_t, int64_t);
314 break;
315 case ARRAY_LOAD_FP64:
316 STACK_VM_LOAD(.v_float64, double, double);
317 break;
318 case ARRAY_LOAD_HANDLE:
319 STACK_VM_LOAD(.v_handle, void*, void*);
320 break;
321 case ARRAY_LOAD_TVMVALUE:
322 STACK_VM_LOAD(, TVMValue, TVMValue);
323 break;
324 // store
325 case ARRAY_STORE_UINT32:
326 STACK_VM_STORE(.v_int64, uint32_t);
327 break;
328 case ARRAY_STORE_INT32:
329 STACK_VM_STORE(.v_int64, int32_t);
330 break;
331 case ARRAY_STORE_INT64:
332 STACK_VM_STORE(.v_int64, int64_t);
333 break;
334 case ARRAY_STORE_FP64:
335 STACK_VM_STORE(.v_float64, double);
336 break;
337 case ARRAY_STORE_HANDLE:
338 STACK_VM_STORE(.v_handle, void*);
339 break;
340 case ARRAY_STORE_TVMVALUE:
341 STACK_VM_STORE(, TVMValue);
342 break;
343 // add
344 case ADDR_ADD: {
345 stack[sp - 1].v_handle = (char*)(stack[sp - 1].v_handle) + stack[sp].v_int64; // NOLINT(*)
346 sp = sp - 1;
347 pc = pc + 1;
348 break;
349 }
350 case NOT: {
351 stack[sp].v_int64 = !stack[sp].v_int64;
352 pc += 1;
353 break;
354 }
355 case PUSH_I64: {
356 stack[sp + 1].v_int64 = code[pc + 1].v_int;
357 sp += 1;
358 pc += 2;
359 break;
360 }
361 case PUSH_VALUE: {
362 int relpos = code[pc + 1].v_int;
363 ICHECK_LE(relpos, 0);
364 stack[sp + 1] = stack[sp + relpos];
365 sp += 1;
366 pc += 2;
367 break;
368 }
369 case POP: {
370 sp -= 1;
371 pc += 1;
372 break;
373 }
374 case SELECT: {
375 stack[sp - 2] = (stack[sp].v_int64 ? stack[sp - 2] : stack[sp - 1]);
376 sp -= 2;
377 pc += 1;
378 break;
379 }
380 case LOAD_HEAP: {
381 stack[sp + 1] = heap[code[pc + 1].v_int];
382 sp += 1;
383 pc += 2;
384 break;
385 }
386 case STORE_HEAP: {
387 heap[code[pc + 1].v_int] = stack[sp];
388 sp -= 1;
389 pc += 2;
390 break;
391 }
392 case ASSERT: {
393 ICHECK(stack[sp].v_int64) << str_data[code[pc + 1].v_int];
394 sp -= 1;
395 pc += 2;
396 break;
397 }
398 case RJUMP_IF_TRUE: {
399 if (stack[sp].v_int64) {
400 pc += code[pc + 1].v_int;
401 } else {
402 pc += 2;
403 }
404 break;
405 }
406 case RJUMP_IF_FALSE: {
407 if (!stack[sp].v_int64) {
408 pc += code[pc + 1].v_int;
409 } else {
410 pc += 2;
411 }
412 break;
413 }
414 case RJUMP: {
415 pc += code[pc + 1].v_int;
416 break;
417 }
418 case ASSERT_SP: {
419 int64_t expected = code[pc + 1].v_int;
420 ICHECK_EQ(sp, expected) << "sp assertion failed, expected=" << expected << " now=" << sp
421 << ", pc=" << pc;
422 pc += 2;
423 break;
424 }
425 case CALL_PACKED_LOWERED: {
426 // call packed function.
427 TVMValue* value_stack = static_cast<TVMValue*>(stack[sp - 1].v_handle);
428 int* type_stack = static_cast<int*>(stack[sp].v_handle);
429 int call_fid = code[pc + 1].v_int;
430 int begin = code[pc + 2].v_int;
431 int end = code[pc + 3].v_int;
432 int num_args = end - begin;
433 static_assert(sizeof(Code) == sizeof(int) && alignof(Code) == alignof(int), "asusmption");
434 runtime::TVMRetValue rv;
435 GetExtern(s, call_fid)
436 .CallPacked(runtime::TVMArgs(value_stack + begin, type_stack + begin, num_args), &rv);
437 sp = sp - 1;
438 stack[sp] = rv.value();
439 pc += 4;
440 break;
441 }
442 // intrinsics
443 case TVM_STRUCT_GET: {
444 int index = code[pc + 1].v_int;
445 int kind = code[pc + 2].v_int;
446 DLTensor* arr = static_cast<DLTensor*>(stack[sp].v_handle);
447 switch (kind) {
448 case StackVM::kArrData: {
449 stack[sp].v_handle = arr[index].data;
450 break;
451 }
452 case StackVM::kArrShape: {
453 stack[sp].v_handle = arr[index].shape;
454 break;
455 }
456 case StackVM::kArrStrides: {
457 stack[sp].v_handle = arr[index].strides;
458 break;
459 }
460 case StackVM::kArrNDim: {
461 stack[sp].v_int64 = arr[index].ndim;
462 break;
463 }
464 case StackVM::kArrTypeCode: {
465 stack[sp].v_int64 = static_cast<int64_t>(arr[index].dtype.code);
466 break;
467 }
468 case StackVM::kArrTypeBits: {
469 stack[sp].v_int64 = static_cast<int64_t>(arr[index].dtype.bits);
470 break;
471 }
472 case StackVM::kArrTypeLanes: {
473 stack[sp].v_int64 = static_cast<int64_t>(arr[index].dtype.lanes);
474 break;
475 }
476 case StackVM::kArrByteOffset: {
477 stack[sp].v_int64 = static_cast<int64_t>(arr[index].byte_offset);
478 break;
479 }
480 case StackVM::kArrDeviceId: {
481 stack[sp].v_int64 = arr[index].device.device_id;
482 break;
483 }
484 case StackVM::kArrDeviceType: {
485 stack[sp].v_int64 = static_cast<int64_t>(arr[index].device.device_type);
486 break;
487 }
488 case StackVM::kArrAddr: {
489 stack[sp].v_handle = arr + index;
490 break;
491 }
492 case StackVM::kTVMValueContent: {
493 stack[sp] = static_cast<TVMValue*>(stack[sp].v_handle)[index];
494 break;
495 }
496 default:
497 LOG(FATAL) << "unhandled get " << kind;
498 }
499 pc = pc + 3;
500 break;
501 }
502 case TVM_STRUCT_SET: {
503 int index = code[pc + 1].v_int;
504 int kind = code[pc + 2].v_int;
505 DLTensor* arr = static_cast<DLTensor*>(stack[sp - 1].v_handle);
506 switch (kind) {
507 case StackVM::kArrData: {
508 arr[index].data = stack[sp].v_handle;
509 break;
510 }
511 case StackVM::kArrShape: {
512 arr[index].shape = static_cast<int64_t*>(stack[sp].v_handle);
513 break;
514 }
515 case StackVM::kArrStrides: {
516 arr[index].strides = static_cast<int64_t*>(stack[sp].v_handle);
517 break;
518 }
519 case StackVM::kArrNDim: {
520 arr[index].ndim = static_cast<int>(stack[sp].v_int64);
521 break;
522 }
523 case StackVM::kArrTypeCode: {
524 arr[index].dtype.code = static_cast<uint8_t>(stack[sp].v_int64);
525 break;
526 }
527 case StackVM::kArrTypeBits: {
528 arr[index].dtype.bits = static_cast<uint8_t>(stack[sp].v_int64);
529 break;
530 }
531 case StackVM::kArrTypeLanes: {
532 arr[index].dtype.lanes = static_cast<uint16_t>(stack[sp].v_int64);
533 break;
534 }
535 case StackVM::kArrByteOffset: {
536 arr[index].byte_offset = static_cast<uint64_t>(stack[sp].v_int64);
537 break;
538 }
539 case StackVM::kArrDeviceId: {
540 arr[index].device.device_id = static_cast<int>(stack[sp].v_int64);
541 break;
542 }
543 case StackVM::kArrDeviceType: {
544 arr[index].device.device_type = static_cast<DLDeviceType>(stack[sp].v_int64);
545 break;
546 }
547 case StackVM::kTVMValueContent: {
548 static_cast<TVMValue*>(stack[sp - 1].v_handle)[index] = stack[sp];
549 break;
550 }
551 default:
552 LOG(FATAL) << "unhandled tvm_struct_set " << kind;
553 }
554 sp -= 2;
555 pc += 3;
556 break;
557 }
558 // alloca
559 case TVM_STACK_ALLOCA_BY_8BYTE: {
560 static_assert(sizeof(TVMValue) == 8, "invariance");
561 int num = code[pc + 1].v_int;
562 void* addr = &stack[sp] + 1;
563 sp = sp + num + 1;
564 alloca_sp = sp - 1;
565 stack[sp].v_handle = addr;
566 pc = pc + 2;
567 break;
568 }
569 case TVM_DEVICE_ALLOCA: {
570 int device_type = static_cast<int>(stack[sp - 4].v_int64);
571 int device_id = static_cast<int>(stack[sp - 3].v_int64);
572 size_t nbytes = static_cast<size_t>(stack[sp - 2].v_int64);
573 int dtype_code_hint = static_cast<int>(stack[sp - 1].v_int64);
574 int dtype_bits_hint = static_cast<int>(stack[sp].v_int64);
575 void* ptr = TVMBackendAllocWorkspace(device_type, device_id, nbytes, dtype_code_hint,
576 dtype_bits_hint);
577 stack[sp - 4].v_handle = ptr;
578 sp = sp - 4;
579 pc = pc + 1;
580 break;
581 }
582 case TVM_DEVICE_FREE: {
583 int device_type = static_cast<int>(stack[sp - 2].v_int64);
584 int device_id = static_cast<int>(stack[sp - 1].v_int64);
585 void* ptr = stack[sp].v_handle;
586 int ret = TVMBackendFreeWorkspace(device_type, device_id, ptr);
587 stack[sp - 2].v_int64 = ret;
588 sp = sp - 2;
589 pc = pc + 1;
590 break;
591 }
592 case TVM_THROW_LAST_ERROR: {
593 LOG(FATAL) << TVMGetLastError();
594 break;
595 }
596 }
597 ICHECK_GE(sp, alloca_sp) << "touch allocated space";
598 ICHECK_LT(sp, stack_cap) << "Stack overflow";
599 }
600}

602const PackedFunc& StackVM::GetExtern(State* s, int fid) const {
603 ICHECK_LT(static_cast<size_t>(fid), extern_func_cache_.size());
604 // allow race write in this, since write is idempotent
605 PackedFunc& f = extern_func_cache_[fid];
606 if (f == nullptr) {
607 ICHECK(s->mod_ctx != nullptr) << "No local context is set in stackvm";
608 const PackedFunc* pf = s->mod_ctx->GetFuncFromEnv(extern_func_name[fid]);
609 ICHECK(pf != nullptr);
610 f = *pf;
611 }
612 return f;
613}
