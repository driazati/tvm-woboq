37template <typename DType>
38bool CompareAscend(const std::pair<int64_t, DType>& lhs, const std::pair<int64_t, DType>& rhs) {
39 return lhs.second < rhs.second;
40}
41
42template <typename DType>
43bool CompareDescend(const std::pair<int64_t, DType>& lhs, const std::pair<int64_t, DType>& rhs) {
44 return lhs.second > rhs.second;
45}
46
47struct float16 {
48 uint16_t bits;
49 float to_float() const {
50 return __extendXfYf2__<uint16_t, uint16_t, 10, float, uint32_t, 23>(bits);
51 }
52};
53
54template <>
55bool CompareAscend(const std::pair<int64_t, float16>& lhs, const std::pair<int64_t, float16>& rhs) {
56 return lhs.second.to_float() < rhs.second.to_float();
57}
58
59template <>
60bool CompareDescend(const std::pair<int64_t, float16>& lhs,
61 const std::pair<int64_t, float16>& rhs) {
62 return lhs.second.to_float() > rhs.second.to_float();
63}

65// Argsort implemented C library sort for nms.
66// Return indices of sorted tensor.
67// By default, the last axis will be used to sort.
68// sort_num specify the number of elements to be sorted.
69// If input tensor has dimension (d0, d1, ..., d(k-1), dk, d(k+1), ..., d(n-1))
70// and sort axis is dk. sort_num should have dimension of
71// (d1, d2, ..., d(k-1), d(k+1), ..., dn).
72TVM_REGISTER_GLOBAL("tvm.contrib.sort.argsort_nms").set_body([](TVMArgs args, TVMRetValue* ret) {
73 DLTensor* input = args[0];
74 DLTensor* sort_num = args[1];
75 DLTensor* output = args[2];
76 int32_t axis = args[3];
77 bool is_ascend = args[4];
78
79 auto dtype = input->dtype;
80 auto data_ptr = static_cast<float*>(input->data);
81 auto sort_num_ptr = static_cast<int32_t*>(sort_num->data);
82 std::vector<std::pair<int32_t, float>> sorter;
83 int64_t axis_mul_before = 1;
84 int64_t axis_mul_after = 1;
85
86 if (axis < 0) {
87 axis = input->ndim + axis;
88 }
89
90 // Currently only supports input dtype to be float32.
91 ICHECK_EQ(dtype.code, 2) << "Currently only supports input dtype "
92 "to be float.";
93#if (__ARM_FEATURE_FP16_SCALAR_ARITHMETIC != 1)
94 ICHECK_EQ(dtype.bits, 32) << "Currently only supports input dtype "
95 "to be float32.";
96#endif
97 ICHECK_LT(axis, input->ndim) << "Axis out of boundary for "
98 "input ndim "
99 << input->ndim;
100
101 for (int i = 0; i < input->ndim; ++i) {
102 if (i < axis) {
103 axis_mul_before *= input->shape[i];
104 } else if (i > axis) {
105 axis_mul_after *= input->shape[i];
106 }
107 }
108
109 for (int64_t i = 0; i < axis_mul_before; ++i) {
110 for (int64_t j = 0; j < axis_mul_after; ++j) {
111 sorter.clear();
112 int32_t current_sort_num = *(sort_num_ptr + i * axis_mul_after + j);
113 int64_t base_idx = i * input->shape[axis] * axis_mul_after + j;
114 for (int64_t k = 0; k < current_sort_num; ++k) {
115 int64_t full_idx = base_idx + k * axis_mul_after;
116 sorter.emplace_back(std::make_pair(k, *(data_ptr + full_idx)));
117 }
118 if (is_ascend) {
119#if (__ARM_FEATURE_FP16_SCALAR_ARITHMETIC == 1)
120 if (dtype.bits == 16) {
121 std::stable_sort(sorter.begin(), sorter.end(), CompareAscend<__fp16>);
122 } else {
123#endif
124 std::stable_sort(sorter.begin(), sorter.end(), CompareAscend<float>);
125#if (__ARM_FEATURE_FP16_SCALAR_ARITHMETIC == 1)
126 }
127#endif
128 } else {
129#if (__ARM_FEATURE_FP16_SCALAR_ARITHMETIC == 1)
130 if (dtype.bits == 16) {
131 std::stable_sort(sorter.begin(), sorter.end(), CompareDescend<__fp16>);
132 } else {
133#endif
134 std::stable_sort(sorter.begin(), sorter.end(), CompareDescend<float>);
135#if (__ARM_FEATURE_FP16_SCALAR_ARITHMETIC == 1)
136 }
137#endif
138 }
139 for (int32_t k = 0; k < input->shape[axis]; ++k) {
140 *(static_cast<int32_t*>(output->data) + base_idx + k * axis_mul_after) =
141 k < static_cast<int32_t>(sorter.size()) ? sorter[k].first : k;
142 }
143 }
144 }
145});

147template <typename DataType, typename OutType>
148void sort_impl(
149 DLTensor* input, DLTensor* output, int32_t axis, bool is_ascend,
150 std::function<void(OutType*, size_t, const std::pair<int64_t, DataType>&)> epilogue) {
151 auto data_ptr = static_cast<DataType*>(input->data);
152 auto out_ptr = static_cast<OutType*>(output->data);
153 std::vector<std::pair<int64_t, DataType>> sorter;
154
155 int axis_mul_before = 1;
156 int axis_mul_after = 1;
157 for (int i = 0; i < input->ndim; ++i) {
158 if (i < axis) {
159 axis_mul_before *= input->shape[i];
160 } else if (i > axis) {
161 axis_mul_after *= input->shape[i];
162 }
163 }
164
165 for (int i = 0; i < axis_mul_before; ++i) {
166 for (int j = 0; j < axis_mul_after; ++j) {
167 sorter.clear();
168 int64_t base_idx = i * input->shape[axis] * axis_mul_after + j;
169 for (int64_t k = 0; k < input->shape[axis]; ++k) {
170 int64_t full_idx = base_idx + k * axis_mul_after;
171 sorter.emplace_back(std::make_pair(k, data_ptr[full_idx]));
172 }
173 if (is_ascend) {
174 std::stable_sort(sorter.begin(), sorter.end(), CompareAscend<DataType>);
175 } else {
176 std::stable_sort(sorter.begin(), sorter.end(), CompareDescend<DataType>);
177 }
178 for (int64_t k = 0; k < input->shape[axis]; ++k) {
179 epilogue(out_ptr, base_idx + k * axis_mul_after, sorter[k]);
180 }
181 }
182 }
183}

185template <typename DataType, typename OutType>
186void argsort(DLTensor* input, DLTensor* output, int32_t axis, bool is_ascend) {
187 return sort_impl<DataType, OutType>(
188 input, output, axis, is_ascend,
189 [](OutType* out_ptr, size_t index, const std::pair<int64_t, DataType>& sort_pair) {
190 out_ptr[index] = static_cast<OutType>(sort_pair.first);
191 });
192}

194template <typename DataType>
195void sort(DLTensor* input, DLTensor* output, int32_t axis, bool is_ascend) {
196 return sort_impl<DataType, DataType>(
197 input, output, axis, is_ascend,
198 [](DataType* out_ptr, size_t index, const std::pair<int64_t, DataType>& sort_pair) {
199 out_ptr[index] = sort_pair.second;
200 });

258 if (out_dtype == "int32") {

261 argsort<int32_t, int64_t>(input, output, axis, is_ascend);
262 } else if (out_dtype == "float32") {
263 argsort<int32_t, float>(input, output, axis, is_ascend);
264 } else if (out_dtype == "float64") {
265 argsort<int32_t, double>(input, output, axis, is_ascend);
266 } else {
267 LOG(FATAL) << "Unsupported output dtype: " << out_dtype;

269 } else if (data_dtype == "int64") {
270 if (out_dtype == "int32") {

272 } else if (out_dtype == "int64") {
273 argsort<int64_t, int64_t>(input, output, axis, is_ascend);
274 } else if (out_dtype == "float32") {
275 argsort<int64_t, float>(input, output, axis, is_ascend);

277 argsort<int64_t, double>(input, output, axis, is_ascend);

334 } else if (data_dtype == "float16") {
335 sort<float16>(input, output, axis, is_ascend);
336 } else {
337 LOG(FATAL) << "Unsupported input dtype: " << data_dtype;
338 }

341template <typename DataType, typename IndicesType>
342void topk(DLTensor* input, DLTensor* out_values, DLTensor* out_indices, int k, int axis,
343 bool is_ascend) {
344 DataType* data_ptr = static_cast<DataType*>(input->data);
345 DataType* values_ptr =
346 (out_values == nullptr) ? nullptr : static_cast<DataType*>(out_values->data);
347 IndicesType* indices_ptr =
348 (out_indices == nullptr) ? nullptr : static_cast<IndicesType*>(out_indices->data);
349 std::vector<std::pair<int64_t, DataType>> sorter;
350
351 int axis_mul_before = 1;
352 int axis_mul_after = 1;
353 for (int i = 0; i < input->ndim; ++i) {
354 if (i < axis) {
355 axis_mul_before *= input->shape[i];
356 } else if (i > axis) {
357 axis_mul_after *= input->shape[i];
358 }
359 }
360 if (k < 1) {
361 k = input->shape[axis];
362 }
363
364 for (int i = 0; i < axis_mul_before; ++i) {
365 for (int j = 0; j < axis_mul_after; ++j) {
366 sorter.clear();
367 int64_t src_base_idx = i * input->shape[axis] * axis_mul_after + j;
368 int64_t dst_base_idx = i * k * axis_mul_after + j;
369 for (int64_t kk = 0; kk < input->shape[axis]; ++kk) {
370 int64_t full_idx = src_base_idx + kk * axis_mul_after;
371 sorter.emplace_back(std::make_pair(kk, data_ptr[full_idx]));
372 }
373 if (is_ascend) {
374 std::stable_sort(sorter.begin(), sorter.end(), CompareAscend<DataType>);
375 } else {
376 std::stable_sort(sorter.begin(), sorter.end(), CompareDescend<DataType>);
377 }
378 int64_t cnt = k > 0 ? k : input->shape[axis];
379 for (int64_t kk = 0; kk < cnt; ++kk) {
380 if (indices_ptr != nullptr) {
381 indices_ptr[dst_base_idx + kk * axis_mul_after] =
382 static_cast<IndicesType>(sorter[kk].first);
383 }
384 if (values_ptr != nullptr) {
385 values_ptr[dst_base_idx + kk * axis_mul_after] = static_cast<DataType>(sorter[kk].second);
386 }
387 }
388 }
389 }
390}
