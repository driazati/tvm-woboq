44class DefaultTimerNode : public TimerNode {
45 public:
46 virtual void Start() {
47 TVMSynchronize(device_.device_type, device_.device_id, nullptr);
48 start_ = std::chrono::high_resolution_clock::now();
49 }
50 virtual void Stop() {
51 TVMSynchronize(device_.device_type, device_.device_id, nullptr);
52 duration_ = std::chrono::high_resolution_clock::now() - start_;
53 }
54 virtual int64_t SyncAndGetElapsedNanos() { return duration_.count(); }
55 virtual ~DefaultTimerNode() {}
56
57 explicit DefaultTimerNode(Device dev) : device_(dev) {}
58 static constexpr const char* _type_key = "DefaultTimerNode";
59 TVM_DECLARE_FINAL_OBJECT_INFO(DefaultTimerNode, TimerNode);
60
61 private:
62 std::chrono::high_resolution_clock::time_point start_;
63 std::chrono::duration<int64_t, std::nano> duration_;
64 Device device_;
65};

67TVM_REGISTER_OBJECT_TYPE(DefaultTimerNode);
68TVM_REGISTER_OBJECT_TYPE(TimerNode);

92// keep track of which timers are not defined but we have already warned about
93std::set<DLDeviceType> seen_devices;
94std::mutex seen_devices_lock;
95
96Timer Timer::Start(Device dev) {
97 auto f = Registry::Get(std::string("profiling.timer.") + DeviceName(dev.device_type));
98 if (f == nullptr) {
99 {
100 std::lock_guard<std::mutex> lock(seen_devices_lock);
101 if (seen_devices.find(dev.device_type) == seen_devices.end()) {
102 LOG(WARNING)
103 << "No timer implementation for " << DeviceName(dev.device_type)
104 << ", using default timer instead. It may be inaccurate or have extra overhead.";
105 seen_devices.insert(dev.device_type);
106 }
107 }
108 Timer t = DefaultTimer(dev);
109 t->Start();
110 return t;
111 } else {
112 Timer t = f->operator()(dev);
113 t->Start();
114 return t;
115 }
116}

122Profiler::Profiler(std::vector<Device> devs, std::vector<MetricCollector> metric_collectors,
123 std::unordered_map<String, ObjectRef> configuration)
124 : devs_(devs), collectors_(metric_collectors), configuration_(configuration) {
125 is_running_ = false;
126 std::vector<DeviceWrapper> wrapped_devs;
127 for (auto dev : devs) {
128 wrapped_devs.push_back(DeviceWrapper(make_object<DeviceWrapperNode>(dev)));
129 }
130 for (auto& x : collectors_) {
131 x->Init(wrapped_devs);
132 }
133 // reset the thread pool so that PAPI eventset hooks are set in all threads.
134 threading::ResetThreadPool();
135
136 configuration_[String("Number of threads")] =
137 ObjectRef(make_object<CountNode>(threading::NumThreads()));
138}
139
140void Profiler::Start() {
141 is_running_ = true;
142 for (auto dev : devs_) {
143 StartCall("Total", dev, {});
144 }
145}

147void Profiler::StartCall(String name, Device dev,
148 std::unordered_map<std::string, ObjectRef> extra_metrics) {
149 std::vector<std::pair<MetricCollector, ObjectRef>> objs;
150 for (auto& collector : collectors_) {
151 ObjectRef obj = collector->Start(dev);
152 if (obj.defined()) {
153 objs.emplace_back(collector, obj);
154 }
155 }
156 in_flight_.push(CallFrame{dev, name, Timer::Start(dev), extra_metrics, objs});
157}
158
159void Profiler::StopCall(std::unordered_map<std::string, ObjectRef> extra_metrics) {
160 CallFrame cf = in_flight_.top();
161 cf.timer->Stop();
162 for (auto& p : extra_metrics) {
163 cf.extra_metrics[p.first] = p.second;
164 }
165 // collect the extra metrics from user defined collectors
166 for (const auto& obj : cf.extra_collectors) {
167 auto collector_metrics = obj.first->Stop(obj.second);
168 for (auto& p : collector_metrics) {
169 cf.extra_metrics[p.first] = p.second;
170 }
171 }
172 in_flight_.pop();
173 calls_.push_back(cf);
174}
175
176void Profiler::Stop() {
177 is_running_ = false;
178 for (size_t i = 0; i < devs_.size(); i++) {
179 StopCall();
180 }
181}

183std::vector<int64_t> ToShape(NDArray shape_tensor) {
184 std::vector<int64_t> shape;
185 auto rank = shape_tensor.Shape().size();
186 auto dtype = shape_tensor.DataType();
187
188 // For 0-rank shapes we need to allocate a single scalar.
189 if (rank == 0) {
190 return shape;
191 }
192
193 // Otherwise we should be rank-1, and we will extract the number of dimensions
194 // for the output vector.
195 ICHECK_EQ(rank, 1U) << "shape tensor should be a k-length vector, found " << rank;
196 int64_t ndim = shape_tensor.Shape().at(0);
197 shape.resize(ndim);
198
199 const DLTensor* dl_tensor = shape_tensor.operator->();
200 if (dtype.is_int() && dtype.bits() == 32 && dtype.lanes() == 1) {
201 int32_t* dims = reinterpret_cast<int32_t*>(dl_tensor->data);
202 shape.assign(dims, dims + ndim);
203 } else if (dtype.is_int() && dtype.bits() == 64 && dtype.lanes() == 1) {
204 int64_t* dims = reinterpret_cast<int64_t*>(dl_tensor->data);
205 shape.assign(dims, dims + ndim);
206 } else {
207 LOG(FATAL) << "invalid shape tensor datatype: " << dtype;
208 }
209
210 return shape;
211}

215String ShapeString(const std::vector<int64_t>& shape, DLDataType dtype) {
216 std::stringstream sizes;
217 sizes << dtype << "[";
218 for (size_t i = 0; i < shape.size(); i++) {
219 if (i != 0) {
220 sizes << ", ";
221 }
222 sizes << shape[i];
223 }
224 sizes << "]";
225 return String(sizes.str());
226}
227
228String ShapeString(const std::vector<NDArray>& shapes) {
229 std::stringstream sizes;
230 for (const NDArray& ary : shapes) {
231 if (sizes.tellp() > 0) {
232 sizes << ", ";
233 }
234 auto shape = ary.Shape();
235 sizes << ary.DataType() << "[";
236 for (size_t i = 0; i < shape.size(); i++) {
237 if (i != 0) {
238 sizes << ", ";
239 }
240 sizes << shape[i];
241 }
242 sizes << "]";
243 }
244 return String(sizes.str());
245}

247String ReportNode::AsCSV() const {
248 // get unique headers
249 std::set<std::string> unique_headers;
250
251 for (auto row : calls) {
252 for (auto p : row) {
253 unique_headers.insert(p.first);
254 }
255 }
256
257 std::vector<std::string> headers;
258 for (auto x : unique_headers) {
259 headers.push_back(x);
260 }
261
262 std::stringstream s;
263
264 for (size_t i = 0; i < headers.size(); i++) {
265 std::string header = headers[i];
266 s << header;
267 if (i < headers.size() - 1) {
268 s << ",";
269 }
270 }
271 s << std::endl;
272 for (auto row : calls) {
273 for (size_t i = 0; i < headers.size(); i++) {
274 std::string header = headers[i];
275 auto it = row.find(header);
276 if (it != row.end()) {
277 std::string val;
278 if ((*it).second.as<CountNode>()) {
279 s << (*it).second.as<CountNode>()->value;
280 } else if ((*it).second.as<DurationNode>()) {
281 s << (*it).second.as<DurationNode>()->microseconds;
282 } else if ((*it).second.as<PercentNode>()) {
283 s << (*it).second.as<PercentNode>()->percent;
284 } else if ((*it).second.as<RatioNode>()) {
285 s << (*it).second.as<RatioNode>()->ratio;
286 } else if ((*it).second.as<StringObj>()) {
287 s << "\"" << Downcast<String>((*it).second) << "\"";
288 }
289 }
290 if (i < headers.size() - 1) {
291 s << ",";
292 }
293 }
294 s << std::endl;
295 }
296 return s.str();
297}

300void metric_as_json(std::ostream& os, ObjectRef o) {
301 if (o.as<StringObj>()) {
302 os << "{\"string\":"
303 << "\"" << Downcast<String>(o) << "\""
304 << "}";
305 } else if (const CountNode* n = o.as<CountNode>()) {
306 os << "{\"count\":" << n->value << "}";
307 } else if (const DurationNode* n = o.as<DurationNode>()) {
308 os << "{\"microseconds\":" << std::setprecision(std::numeric_limits<double>::max_digits10)
309 << std::fixed << n->microseconds << "}";
310 } else if (const PercentNode* n = o.as<PercentNode>()) {
311 os << "{\"percent\":" << std::setprecision(std::numeric_limits<double>::max_digits10)
312 << std::fixed << n->percent << "}";
313 } else if (const RatioNode* n = o.as<RatioNode>()) {
314 os << "{\"ratio\":" << std::setprecision(std::numeric_limits<double>::max_digits10)
315 << std::fixed << n->ratio << "}";
316 } else {
317 LOG(FATAL) << "Unprintable type " << o->GetTypeKey();
318 }
319}

322String ReportNode::AsJSON() const {
323 std::ostringstream s;
324 // DMLC's JSONWriter does not allow us to write a key value pair without
325 // implementing Write for the value. We want a specific write for the value,
326 // so we would have to implement a custom data structure for each type of
327 // value we want to print. Instead we construct the json by hand because it
328 // is easier.
329 s << "{";
330
331 s << "\"calls\":[";
332 for (size_t i = 0; i < calls.size(); i++) {
333 size_t j = 0;
334 s << "{";
335 for (const auto& kv : calls[i]) {
336 s << "\"" << kv.first << "\":";
337 metric_as_json(s, kv.second);
338 if (j < calls[i].size() - 1) {
339 s << ",";
340 }
341 j++;
342 }
343 s << "}";
344 if (i < calls.size() - 1) {
345 s << ",";
346 }
347 }
348 s << "],"; // end calls
349
350 s << "\"device_metrics\":{";
351 size_t i = 0;
352 for (const auto& dev_kv : device_metrics) {
353 size_t j = 0;
354 s << "\"" << dev_kv.first << "\":{";
355 for (const auto& metric_kv : dev_kv.second) {
356 s << "\"" << metric_kv.first << "\":";
357 metric_as_json(s, metric_kv.second);
358 if (j < dev_kv.second.size() - 1) {
359 s << ",";
360 }
361 j++;
362 }
363 s << "}";
364 if (i < device_metrics.size() - 1) {
365 s << ",";
366 }
367 i++;
368 }
369 s << "},"; // end device metrics
370
371 s << "\"configuration\":{";
372 size_t k = 0;
373 for (const auto& kv : configuration) {
374 s << "\"" << kv.first << "\":";
375 metric_as_json(s, kv.second);
376 if (k < configuration.size() - 1) {
377 s << ",";
378 }
379 k++;
380 }
381 s << "}"; // end configuration
382 s << "}";
383 return s.str();
384}

386// Aggregate a set of values for a metric. Computes sum for Duration, Count,
387// and Percent; average for Ratio; and assumes all Strings are the same. All
388// ObjectRefs in metrics must have the same type.
389ObjectRef AggregateMetric(const std::vector<ObjectRef>& metrics) {
390 ICHECK_GT(metrics.size(), 0) << "Must pass a non-zero number of metrics";
391 if (metrics[0].as<DurationNode>()) {
392 double sum = 0;
393 for (auto& metric : metrics) {
394 sum += metric.as<DurationNode>()->microseconds;
395 }
396 return ObjectRef(make_object<DurationNode>(sum));
397 } else if (metrics[0].as<CountNode>()) {
398 int64_t sum = 0;
399 for (auto& metric : metrics) {
400 sum += metric.as<CountNode>()->value;
401 }
402 return ObjectRef(make_object<CountNode>(sum));
403 } else if (metrics[0].as<PercentNode>()) {
404 double sum = 0;
405 for (auto& metric : metrics) {
406 sum += metric.as<PercentNode>()->percent;
407 }
408 return ObjectRef(make_object<PercentNode>(sum));
409 } else if (metrics[0].as<RatioNode>()) {
410 double sum = 0;
411 for (auto& metric : metrics) {
412 sum += metric.as<RatioNode>()->ratio;
413 }
414 return ObjectRef(make_object<RatioNode>(sum / metrics.size()));
415 } else if (metrics[0].as<StringObj>()) {
416 for (auto& m : metrics) {
417 if (Downcast<String>(metrics[0]) != Downcast<String>(m)) {
418 return ObjectRef(String(""));
419 }
420 }
421 // Assume all strings in metrics are the same.
422 return metrics[0];
423 } else {
424 LOG(FATAL) << "Can only aggregate metrics with types DurationNode, CountNode, "
425 "PercentNode, RatioNode, and StringObj, but got "
426 << metrics[0]->GetTypeKey();
427 return ObjectRef(); // To silence warnings
428 }
429}

431// Try and set the locale of the provided stringstream so that it will print
432// numbers with thousands separators. Sometimes users will have a misconfigured
433// system where an invalid locale is set, so we catch and ignore any locale
434// errors.
435static void set_locale_for_separators(std::stringstream& s) {
436 try {
437 // empty string indicates locale should be the user's default, see man 3 setlocale
438 s.imbue(std::locale(""));
439 } catch (std::runtime_error& e) {
440 }
441}
442
443static String print_metric(ObjectRef metric) {
444 std::string val;
445 if (metric.as<CountNode>()) {
446 std::stringstream s;
447 set_locale_for_separators(s);
448 s << std::fixed << metric.as<CountNode>()->value;
449 val = s.str();
450 } else if (metric.as<DurationNode>()) {
451 std::stringstream s;
452 set_locale_for_separators(s);
453 s << std::fixed << std::setprecision(2) << metric.as<DurationNode>()->microseconds;
454 val = s.str();
455 } else if (metric.as<PercentNode>()) {
456 std::stringstream s;
457 s << std::fixed << std::setprecision(2) << metric.as<PercentNode>()->percent;
458 val = s.str();
459 } else if (metric.as<RatioNode>()) {
460 std::stringstream s;
461 set_locale_for_separators(s);
462 s << std::setprecision(2) << metric.as<RatioNode>()->ratio;
463 val = s.str();
464 } else if (metric.as<StringObj>()) {
465 val = Downcast<String>(metric);
466 } else {
467 LOG(FATAL) << "Cannot print metric of type " << metric->GetTypeKey();
468 }
469 return val;
470}

472String ReportNode::AsTable(bool sort, bool aggregate, bool compute_col_sums) const {
473 // aggregate calls by op hash (or op name if hash is not set) + argument shapes
474 std::vector<Map<String, ObjectRef>> aggregated_calls;
475 if (aggregate) {
476 std::unordered_map<std::string, std::vector<size_t>> aggregates;
477 for (size_t i = 0; i < calls.size(); i++) {
478 auto& frame = calls[i];
479 auto it = frame.find("Hash");
480 std::string name = Downcast<String>(frame["Name"]);
481 if (it != frame.end()) {
482 name = Downcast<String>((*it).second);
483 }
484 if (frame.find("Argument Shapes") != frame.end()) {
485 name += Downcast<String>(frame["Argument Shapes"]);
486 }
487 if (frame.find("Device") != frame.end()) {
488 name += Downcast<String>(frame["Device"]);
489 }
490
491 if (aggregates.find(name) == aggregates.end()) {
492 aggregates[name] = {i};
493 } else {
494 aggregates[name].push_back(i);
495 }
496 }
497 for (const auto& p : aggregates) {
498 std::unordered_map<String, ObjectRef> aggregated;
499 std::unordered_set<std::string> metrics;
500 for (auto& call : calls) {
501 for (auto& metric : call) {
502 metrics.insert(metric.first);
503 }
504 }
505 for (const std::string& metric : metrics) {
506 std::vector<ObjectRef> per_call;
507 for (auto i : p.second) {
508 auto& call = calls[i];
509 auto it = std::find_if(call.begin(), call.end(),
510 [&metric](const std::pair<String, ObjectRef>& call_metric) {
511 return std::string(call_metric.first) == metric;
512 });
513 if (it != call.end()) {
514 per_call.push_back((*it).second);
515 }
516 }
517 if (per_call.size() > 0) {
518 aggregated[metric] = AggregateMetric(per_call);
519 }
520 }
521 aggregated_calls.push_back(aggregated);
522 }
523 } else {
524 for (auto call : calls) {
525 aggregated_calls.push_back(call);
526 }
527 }
528
529 // sort rows by duration
530 if (sort) {
531 std::sort(aggregated_calls.begin(), aggregated_calls.end(),
532 [&](const Map<String, ObjectRef>& a, const Map<String, ObjectRef>& b) {
533 return a.at("Duration (us)").as<DurationNode>()->microseconds >
534 b.at("Duration (us)").as<DurationNode>()->microseconds;
535 });
536 }
537
538 // compute columnwise sums
539 if (compute_col_sums) {
540 std::unordered_map<String, ObjectRef> col_sums;
541 for (auto call : aggregated_calls) {
542 for (auto p : call) {
543 if (p.second.as<CountNode>()) {
544 int64_t val = p.second.as<CountNode>()->value;
545 auto it = col_sums.find(p.first);
546 if (it != col_sums.end()) {
547 val += it->second.as<CountNode>()->value;
548 }
549 col_sums[p.first] = ObjectRef(make_object<CountNode>(val));
550 } else if (p.second.as<DurationNode>()) {
551 double val = p.second.as<DurationNode>()->microseconds;
552 auto it = col_sums.find(p.first);
553 if (it != col_sums.end()) {
554 val += it->second.as<DurationNode>()->microseconds;
555 }
556 col_sums[p.first] = ObjectRef(make_object<DurationNode>(val));
557 } else if (p.second.as<PercentNode>()) {
558 double val = p.second.as<PercentNode>()->percent;
559 auto it = col_sums.find(p.first);
560 if (it != col_sums.end()) {
561 val += it->second.as<PercentNode>()->percent;
562 }
563 col_sums[p.first] = ObjectRef(make_object<PercentNode>(val));
564 } else if (p.second.as<RatioNode>()) {
565 // It does not make sense to sum ratios
566 }
567 }
568 }
569 col_sums["Name"] = String("Sum");
570 aggregated_calls.push_back({{String("Name"), String("----------")}}); // separator
571 aggregated_calls.push_back(col_sums);
572 }
573
574 // per-device metrics
575 for (auto p : device_metrics) {
576 Map<String, ObjectRef> metrics = p.second;
577 metrics.Set("Name", String("Total"));
578 aggregated_calls.push_back(metrics);
579 }
580
581 // Table formatting
582 std::set<std::string> unique_headers;
583 for (auto row : aggregated_calls) {
584 for (auto p : row) {
585 unique_headers.insert(p.first);
586 }
587 }
588
589 // always include these headers in this order
590 std::vector<std::string> headers = {"Name", "Duration (us)", "Percent",
591 "Device", "Count", "Argument Shapes"};
592 for (auto header : unique_headers) {
593 if (std::find(headers.begin(), headers.end(), header) == headers.end()) {
594 headers.push_back(header);
595 }
596 }
597
598 // Switch layout from row major to column major so we can easily compute column widths.
599 std::vector<std::vector<std::string>> cols;
600 for (auto header : headers) {
601 cols.push_back({header});
602 }
603 for (auto row : aggregated_calls) {
604 for (size_t i = 0; i < headers.size(); i++) {
605 auto it = row.find(headers[i]);
606 if (it == row.end()) {
607 // fill empty data with empty strings
608 cols[i].push_back("");
609 } else {
610 cols[i].push_back(print_metric((*it).second));
611 }
612 }
613 }
614
615 std::vector<size_t> widths;
616 for (auto v : cols) {
617 size_t width = 0;
618 for (auto x : v) {
619 width = std::max(width, x.size());
620 }
621 widths.push_back(width);
622 }
623 size_t length = 0;
624 for (auto v : cols) {
625 length = std::max(length, v.size());
626 }
627
628 std::stringstream s;
629 for (size_t row = 0; row < length; row++) {
630 for (size_t col = 0; col < cols.size(); col++) {
631 // left align first column
632 if (col == 0) {
633 s << std::left;
634 } else {
635 s << std::right;
636 }
637 if (row < cols[col].size()) {
638 s << std::setw(widths[col]) << cols[col][row] << " ";
639 } else {
640 s << std::setw(widths[col]) << ""
641 << " ";
642 }
643 }
644 s << std::endl;
645 }
646
647 // Add configuration information. It will not be aligned with the columns.
648 s << std::endl << "Configuration" << std::endl << "-------------" << std::endl;
649 for (auto kv : configuration) {
650 s << kv.first << ": " << print_metric(kv.second) << std::endl;
651 }
652 return s.str();
653}

659Report Profiler::Report() {
660 // sync all timers and normalize rows
661 std::vector<std::unordered_map<String, ObjectRef>> rows;
662 for (auto& cf : calls_) {
663 std::unordered_map<String, ObjectRef> row;
664 double us = cf.timer->SyncAndGetElapsedNanos() / 1e3;
665 row["Duration (us)"] = ObjectRef(make_object<DurationNode>(us));
666 row["Count"] = ObjectRef(make_object<CountNode>(1));
667 row["Name"] = cf.name;
668 row["Device"] = String(DeviceString(cf.dev));
669 for (auto p : cf.extra_metrics) {
670 row[p.first] = p.second;
671 }
672 rows.push_back(row);
673 }
674
675 // the last couple of call frames are the overall times
676 double overall_time_us = 0;
677 std::unordered_map<String, Map<String, ObjectRef>> device_metrics;
678 for (size_t i = 0; i < devs_.size(); i++) {
679 auto row = rows[rows.size() - 1];
680 rows.pop_back();
681 device_metrics[Downcast<String>(row["Device"])] = row;
682 overall_time_us =
683 std::max(overall_time_us, row["Duration (us)"].as<DurationNode>()->microseconds);
684 }
685
686 // Calculate percentages
687 for (auto& row : rows) {
688 row["Percent"] = ObjectRef(make_object<PercentNode>(
689 row["Duration (us)"].as<DurationNode>()->microseconds / overall_time_us * 100));
690 }
691
692 // convert to map
693 std::vector<Map<String, ObjectRef>> converted_rows;
694 for (const auto& row : rows) {
695 converted_rows.push_back(row);
696 }
697
698 return profiling::Report(converted_rows, device_metrics, configuration_);
699}

711Map<String, ObjectRef> parse_metrics(dmlc::JSONReader* reader) {
712 reader->BeginObject();
713 std::string metric_name, metric_value_name;
714 Map<String, ObjectRef> metrics;
715 while (reader->NextObjectItem(&metric_name)) {
716 ObjectRef o;
717 reader->BeginObject();
718 reader->NextObjectItem(&metric_value_name);
719 if (metric_value_name == "microseconds") {
720 double microseconds;
721 reader->Read(&microseconds);
722 o = ObjectRef(make_object<DurationNode>(microseconds));
723 } else if (metric_value_name == "percent") {
724 double percent;
725 reader->Read(&percent);
726 o = ObjectRef(make_object<PercentNode>(percent));
727 } else if (metric_value_name == "count") {
728 int64_t count;
729 reader->Read(&count);
730 o = ObjectRef(make_object<CountNode>(count));
731 } else if (metric_value_name == "ratio") {
732 double ratio;
733 reader->Read(&ratio);
734 o = ObjectRef(make_object<RatioNode>(ratio));
735 } else if (metric_value_name == "string") {
736 std::string s;
737 reader->Read(&s);
738 o = String(s);
739 } else {
740 LOG(FATAL) << "Cannot parse metric of type " << metric_value_name
741 << " valid types are microseconds, percent, count.";
742 }
743 metrics.Set(metric_name, o);
744 // Necessary to make sure that the parser hits the end of the object.
745 ICHECK(!reader->NextObjectItem(&metric_value_name));
746 // EndObject does not exist, leaving this here for clarity
747 // reader.EndObject();
748 }
749 // reader.EndObject();
750 return metrics;
751}

753Report Report::FromJSON(String json) {
754 std::stringstream input(json.operator std::string());
755 dmlc::JSONReader reader(&input);
756 std::string key;
757 Array<Map<String, ObjectRef>> calls;
758 Map<String, Map<String, ObjectRef>> device_metrics;
759 Map<String, ObjectRef> configuration;
760
761 reader.BeginObject();
762 while (reader.NextObjectItem(&key)) {
763 if (key == "calls") {
764 reader.BeginArray();
765 while (reader.NextArrayItem()) {
766 calls.push_back(parse_metrics(&reader));
767 }
768 // reader.EndArray();
769 } else if (key == "device_metrics") {
770 reader.BeginObject();
771 std::string device_name;
772 while (reader.NextObjectItem(&device_name)) {
773 device_metrics.Set(device_name, parse_metrics(&reader));
774 }
775 // reader.EndObject();
776 } else if (key == "configuration") {
777 configuration = parse_metrics(&reader);
778 }
779 }
780
781 return Report(calls, device_metrics, configuration);
782}

802PackedFunc ProfileFunction(Module mod, std::string func_name, int device_type, int device_id,
803 int warmup_iters, Array<MetricCollector> collectors) {
804 // Module::GetFunction is not const, so this lambda has to be mutable
805 return PackedFunc([=](TVMArgs args, TVMRetValue* ret) mutable {
806 PackedFunc f = mod.GetFunction(func_name);
807 CHECK(f.defined()) << "There is no function called \"" << func_name << "\" in the module";
808 Device dev{static_cast<DLDeviceType>(device_type), device_id};
809
810 // warmup
811 for (int i = 0; i < warmup_iters; i++) {
812 f.CallPacked(args, ret);
813 }
814
815 for (auto& collector : collectors) {
816 collector->Init({DeviceWrapper(dev)});
817 }
818 std::vector<Map<String, ObjectRef>> results;
819 results.reserve(collectors.size());
820 std::vector<std::pair<MetricCollector, ObjectRef>> collector_data;
821 collector_data.reserve(collectors.size());
822 for (auto& collector : collectors) {
823 ObjectRef o = collector->Start(dev);
824 // If not defined, then the collector cannot time this device.
825 if (o.defined()) {
826 collector_data.push_back({collector, o});
827 }
828 }
829
830 // TODO(tkonolige): repeated calls if the runtime is small?
831 f.CallPacked(args, ret);
832
833 for (auto& kv : collector_data) {
834 results.push_back(kv.first->Stop(kv.second));
835 }
836 Map<String, ObjectRef> combined_results;
837 for (auto m : results) {
838 for (auto p : m) {
839 // assume that there is no shared metric name between collectors
840 combined_results.Set(p.first, p.second);
841 }
842 }
843 *ret = combined_results;
844 });
845}

847TVM_REGISTER_GLOBAL("runtime.profiling.ProfileFunction")
848 .set_body_typed<PackedFunc(Module, String, int, int, int,
849 Array<MetricCollector>)>([](Module mod, String func_name,
850 int device_type, int device_id,
851 int warmup_iters,
852 Array<MetricCollector> collectors) {
853 if (mod->type_key() == std::string("rpc")) {
854 LOG(FATAL)
855 << "Profiling a module over RPC is not yet supported"; // because we can't send
856 // MetricCollectors over rpc.
857 throw;
858 } else {
859 return ProfileFunction(mod, func_name, device_type, device_id, warmup_iters, collectors);
860 }
861 });

863PackedFunc WrapTimeEvaluator(PackedFunc pf, Device dev, int number, int repeat, int min_repeat_ms,
864 int limit_zero_time_iterations, int cooldown_interval_ms,
865 int repeats_to_cooldown, PackedFunc f_preproc) {
866 ICHECK(pf != nullptr);
867
868 if (static_cast<int>(dev.device_type) == static_cast<int>(kDLMicroDev)) {
869 auto get_micro_time_evaluator = runtime::Registry::Get("micro._GetMicroTimeEvaluator");
870 ICHECK(get_micro_time_evaluator != nullptr) << "micro backend not enabled";
871 return (*get_micro_time_evaluator)(pf, dev, number, repeat);
872 }
873
874 auto ftimer = [pf, dev, number, repeat, min_repeat_ms, limit_zero_time_iterations,
875 cooldown_interval_ms, repeats_to_cooldown,
876 f_preproc](TVMArgs args, TVMRetValue* rv) mutable {
877 TVMRetValue temp;
878 std::ostringstream os;
879 // skip first time call, to activate lazy compilation components.
880 pf.CallPacked(args, &temp);
881
882 DeviceAPI::Get(dev)->StreamSync(dev, nullptr);
883
884 for (int i = 0; i < repeat; ++i) {
885 if (f_preproc != nullptr) {
886 f_preproc.CallPacked(args, &temp);
887 }
888 double duration_ms = 0.0;
889 int absolute_zero_times = 0;
890 do {
891 if (duration_ms > 0.0) {
892 const double golden_ratio = 1.618;
893 number = static_cast<int>(
894 std::max((min_repeat_ms / (duration_ms / number) + 1), number * golden_ratio));
895 }
896
897 // start timing
898 Timer t = Timer::Start(dev);
899 for (int j = 0; j < number; ++j) {
900 pf.CallPacked(args, &temp);
901 }
902 t->Stop();
903 int64_t t_nanos = t->SyncAndGetElapsedNanos();
904 if (t_nanos == 0) absolute_zero_times++;
905 duration_ms = t_nanos / 1e6;
906 } while (duration_ms < min_repeat_ms && absolute_zero_times < limit_zero_time_iterations);
907
908 double speed = duration_ms / 1e3 / number;
909 os.write(reinterpret_cast<char*>(&speed), sizeof(speed));
910
911 if (cooldown_interval_ms > 0 && (i % repeats_to_cooldown) == 0) {
912 std::this_thread::sleep_for(std::chrono::milliseconds(cooldown_interval_ms));
913 }
914 }
915
916 std::string blob = os.str();
917 TVMByteArray arr;
918 arr.size = blob.length();
919 arr.data = blob.data();
920 // return the time.
921 *rv = arr;
922 };
923 return PackedFunc(ftimer);
924}
