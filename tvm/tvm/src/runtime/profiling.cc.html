44class DefaultTimerNode : public TimerNode {
45 public:
46 virtual void Start() {
47 TVMSynchronize(device_.device_type, device_.device_id, nullptr);
48 start_ = std::chrono::high_resolution_clock::now();
49 }
50 virtual void Stop() {
51 TVMSynchronize(device_.device_type, device_.device_id, nullptr);
52 duration_ = std::chrono::high_resolution_clock::now() - start_;
53 }
54 virtual int64_t SyncAndGetElapsedNanos() { return duration_.count(); }
55 virtual ~DefaultTimerNode() {}
56
57 explicit DefaultTimerNode(Device dev) : device_(dev) {}
58 static constexpr const char* _type_key = "DefaultTimerNode";
59 TVM_DECLARE_FINAL_OBJECT_INFO(DefaultTimerNode, TimerNode);
60
61 private:
62 std::chrono::high_resolution_clock::time_point start_;
63 std::chrono::duration<int64_t, std::nano> duration_;
64 Device device_;
65};
66
67TVM_REGISTER_OBJECT_TYPE(DefaultTimerNode);
68TVM_REGISTER_OBJECT_TYPE(TimerNode);

75 virtual void Stop() { duration_ = std::chrono::high_resolution_clock::now() - start_; }
76 virtual int64_t SyncAndGetElapsedNanos() { return duration_.count(); }
77 virtual ~CPUTimerNode() {}

92Timer Timer::Start(Device dev) {
93 auto f = Registry::Get(std::string("profiling.timer.") + DeviceName(dev.device_type));
94 if (f == nullptr) {
95 Timer t = DefaultTimer(dev);
96 t->Start();
97 return t;
98 } else {
99 Timer t = f->operator()(dev);
100 t->Start();
101 return t;
102 }
103}

109Profiler::Profiler(std::vector<Device> devs, std::vector<MetricCollector> metric_collectors,
110 std::unordered_map<String, ObjectRef> configuration)
111 : devs_(devs), collectors_(metric_collectors), configuration_(configuration) {
112 is_running_ = false;
113 std::vector<DeviceWrapper> wrapped_devs;
114 for (auto dev : devs) {
115 wrapped_devs.push_back(DeviceWrapper(make_object<DeviceWrapperNode>(dev)));
116 }
117 for (auto& x : collectors_) {
118 x->Init(wrapped_devs);
119 }
120 // reset the thread pool so that PAPI eventset hooks are set in all threads.
121 threading::ResetThreadPool();
122
123 configuration_[String("Number of threads")] =
124 ObjectRef(make_object<CountNode>(threading::NumThreads()));
125}
126
127void Profiler::Start() {
128 is_running_ = true;
129 for (auto dev : devs_) {
130 StartCall("Total", dev, {});
131 }
132}
133
134void Profiler::StartCall(String name, Device dev,
135 std::unordered_map<std::string, ObjectRef> extra_metrics) {
136 std::vector<std::pair<MetricCollector, ObjectRef>> objs;
137 for (auto& collector : collectors_) {
138 ObjectRef obj = collector->Start(dev);
139 if (obj.defined()) {
140 objs.emplace_back(collector, obj);
141 }
142 }
143 in_flight_.push(CallFrame{dev, name, Timer::Start(dev), extra_metrics, objs});
144}
145
146void Profiler::StopCall(std::unordered_map<std::string, ObjectRef> extra_metrics) {
147 CallFrame cf = in_flight_.top();
148 cf.timer->Stop();
149 for (auto& p : extra_metrics) {
150 cf.extra_metrics[p.first] = p.second;
151 }
152 // collect the extra metrics from user defined collectors
153 for (const auto& obj : cf.extra_collectors) {
154 auto collector_metrics = obj.first->Stop(obj.second);
155 for (auto& p : collector_metrics) {
156 cf.extra_metrics[p.first] = p.second;
157 }
158 }
159 in_flight_.pop();
160 calls_.push_back(cf);
161}
162
163void Profiler::Stop() {
164 is_running_ = false;
165 for (size_t i = 0; i < devs_.size(); i++) {
166 StopCall();
167 }

170std::vector<int64_t> ToShape(NDArray shape_tensor) {
171 std::vector<int64_t> shape;
172 auto rank = shape_tensor.Shape().size();
173 auto dtype = shape_tensor.DataType();
174
175 // For 0-rank shapes we need to allocate a single scalar.
176 if (rank == 0) {
177 return shape;
178 }
179
180 // Otherwise we should be rank-1, and we will extract the number of dimensions
181 // for the output vector.
182 ICHECK_EQ(rank, 1U) << "shape tensor should be a k-length vector, found " << rank;
183 int64_t ndim = shape_tensor.Shape().at(0);
184 shape.resize(ndim);
185
186 const DLTensor* dl_tensor = shape_tensor.operator->();
187 if (dtype.is_int() && dtype.bits() == 32 && dtype.lanes() == 1) {
188 int32_t* dims = reinterpret_cast<int32_t*>(dl_tensor->data);
189 shape.assign(dims, dims + ndim);
190 } else if (dtype.is_int() && dtype.bits() == 64 && dtype.lanes() == 1) {
191 int64_t* dims = reinterpret_cast<int64_t*>(dl_tensor->data);
192 shape.assign(dims, dims + ndim);
193 } else {
194 LOG(FATAL) << "invalid shape tensor datatype: " << dtype;
195 }
196
197 return shape;
198}

202String ShapeString(const std::vector<int64_t>& shape, DLDataType dtype) {
203 std::stringstream sizes;
204 sizes << dtype << "[";
205 for (size_t i = 0; i < shape.size(); i++) {
206 if (i != 0) {
207 sizes << ", ";
208 }
209 sizes << shape[i];
210 }
211 sizes << "]";
212 return String(sizes.str());
213}

215String ShapeString(const std::vector<NDArray>& shapes) {
216 std::stringstream sizes;
217 for (const NDArray& ary : shapes) {
218 if (sizes.tellp() > 0) {
219 sizes << ", ";
220 }
221 auto shape = ary.Shape();
222 sizes << ary.DataType() << "[";
223 for (size_t i = 0; i < shape.size(); i++) {
224 if (i != 0) {
225 sizes << ", ";
226 }
227 sizes << shape[i];
228 }
229 sizes << "]";
230 }
231 return String(sizes.str());
232}

234String ReportNode::AsCSV() const {
235 // get unique headers
236 std::set<std::string> unique_headers;
237
238 for (auto row : calls) {
239 for (auto p : row) {
240 unique_headers.insert(p.first);
241 }
242 }
243
244 std::vector<std::string> headers;
245 for (auto x : unique_headers) {
246 headers.push_back(x);
247 }
248
249 std::stringstream s;
250
251 for (size_t i = 0; i < headers.size(); i++) {
252 std::string header = headers[i];
253 s << header;
254 if (i < headers.size() - 1) {
255 s << ",";
256 }
257 }
258 s << std::endl;
259 for (auto row : calls) {
260 for (size_t i = 0; i < headers.size(); i++) {
261 std::string header = headers[i];
262 auto it = row.find(header);
263 if (it != row.end()) {
264 std::string val;
265 if ((*it).second.as<CountNode>()) {
266 s << (*it).second.as<CountNode>()->value;
267 } else if ((*it).second.as<DurationNode>()) {
268 s << (*it).second.as<DurationNode>()->microseconds;
269 } else if ((*it).second.as<PercentNode>()) {
270 s << (*it).second.as<PercentNode>()->percent;
271 } else if ((*it).second.as<RatioNode>()) {
272 s << (*it).second.as<RatioNode>()->ratio;
273 } else if ((*it).second.as<StringObj>()) {
274 s << "\"" << Downcast<String>((*it).second) << "\"";
275 }
276 }
277 if (i < headers.size() - 1) {
278 s << ",";
279 }
280 }
281 s << std::endl;
282 }
283 return s.str();
284}

287void metric_as_json(std::ostream& os, ObjectRef o) {
288 if (o.as<StringObj>()) {
289 os << "{\"string\":"
290 << "\"" << Downcast<String>(o) << "\""
291 << "}";
292 } else if (const CountNode* n = o.as<CountNode>()) {
293 os << "{\"count\":" << n->value << "}";
294 } else if (const DurationNode* n = o.as<DurationNode>()) {
295 os << "{\"microseconds\":" << std::setprecision(std::numeric_limits<double>::max_digits10)
296 << std::fixed << n->microseconds << "}";
297 } else if (const PercentNode* n = o.as<PercentNode>()) {
298 os << "{\"percent\":" << std::setprecision(std::numeric_limits<double>::max_digits10)
299 << std::fixed << n->percent << "}";
300 } else if (const RatioNode* n = o.as<RatioNode>()) {
301 os << "{\"ratio\":" << std::setprecision(std::numeric_limits<double>::max_digits10)
302 << std::fixed << n->ratio << "}";
303 } else {
304 LOG(FATAL) << "Unprintable type " << o->GetTypeKey();
305 }
306}

309String ReportNode::AsJSON() const {
310 std::ostringstream s;
311 // DMLC's JSONWriter does not allow us to write a key value pair without
312 // implementing Write for the value. We want a specific write for the value,
313 // so we would have to implement a custom data structure for each type of
314 // value we want to print. Instead we construct the json by hand because it
315 // is easier.
316 s << "{";
317
318 s << "\"calls\":[";
319 for (size_t i = 0; i < calls.size(); i++) {
320 size_t j = 0;
321 s << "{";
322 for (const auto& kv : calls[i]) {
323 s << "\"" << kv.first << "\":";
324 metric_as_json(s, kv.second);
325 if (j < calls[i].size() - 1) {
326 s << ",";
327 }
328 j++;
329 }
330 s << "}";
331 if (i < calls.size() - 1) {
332 s << ",";
333 }
334 }
335 s << "],"; // end calls
336
337 s << "\"device_metrics\":{";
338 size_t i = 0;
339 for (const auto& dev_kv : device_metrics) {
340 size_t j = 0;
341 s << "\"" << dev_kv.first << "\":{";
342 for (const auto& metric_kv : dev_kv.second) {
343 s << "\"" << metric_kv.first << "\":";
344 metric_as_json(s, metric_kv.second);
345 if (j < dev_kv.second.size() - 1) {
346 s << ",";
347 }
348 j++;
349 }
350 s << "}";
351 if (i < device_metrics.size() - 1) {
352 s << ",";
353 }
354 i++;
355 }
356 s << "},"; // end device metrics
357
358 s << "\"configuration\":{";
359 size_t k = 0;
360 for (const auto& kv : configuration) {
361 s << "\"" << kv.first << "\":";
362 metric_as_json(s, kv.second);
363 if (k < configuration.size() - 1) {
364 s << ",";
365 }
366 k++;
367 }
368 s << "}"; // end configuration
369 s << "}";
370 return s.str();
371}

373// Aggregate a set of values for a metric. Computes sum for Duration, Count,
374// and Percent; average for Ratio; and assumes all Strings are the same. All
375// ObjectRefs in metrics must have the same type.
376ObjectRef AggregateMetric(const std::vector<ObjectRef>& metrics) {
377 ICHECK_GT(metrics.size(), 0) << "Must pass a non-zero number of metrics";
378 if (metrics[0].as<DurationNode>()) {
379 double sum = 0;
380 for (auto& metric : metrics) {
381 sum += metric.as<DurationNode>()->microseconds;
382 }
383 return ObjectRef(make_object<DurationNode>(sum));
384 } else if (metrics[0].as<CountNode>()) {
385 int64_t sum = 0;
386 for (auto& metric : metrics) {
387 sum += metric.as<CountNode>()->value;
388 }
389 return ObjectRef(make_object<CountNode>(sum));
390 } else if (metrics[0].as<PercentNode>()) {
391 double sum = 0;
392 for (auto& metric : metrics) {
393 sum += metric.as<PercentNode>()->percent;
394 }
395 return ObjectRef(make_object<PercentNode>(sum));
396 } else if (metrics[0].as<RatioNode>()) {
397 double sum = 0;
398 for (auto& metric : metrics) {
399 sum += metric.as<RatioNode>()->ratio;
400 }
401 return ObjectRef(make_object<RatioNode>(sum / metrics.size()));
402 } else if (metrics[0].as<StringObj>()) {
403 for (auto& m : metrics) {
404 if (Downcast<String>(metrics[0]) != Downcast<String>(m)) {
405 return ObjectRef(String(""));
406 }
407 }
408 // Assume all strings in metrics are the same.
409 return metrics[0];
410 } else {
411 LOG(FATAL) << "Can only aggregate metrics with types DurationNode, CountNode, "
412 "PercentNode, RatioNode, and StringObj, but got "
413 << metrics[0]->GetTypeKey();
414 return ObjectRef(); // To silence warnings
415 }
416}

418// Try and set the locale of the provided stringstream so that it will print
419// numbers with thousands separators. Sometimes users will have a misconfigured
420// system where an invalid locale is set, so we catch and ignore any locale
421// errors.
422static void set_locale_for_separators(std::stringstream& s) {
423 try {
424 // empty string indicates locale should be the user's default, see man 3 setlocale
425 s.imbue(std::locale(""));
426 } catch (std::runtime_error& e) {
427 }
428}
429
430static String print_metric(ObjectRef metric) {
431 std::string val;
432 if (metric.as<CountNode>()) {
433 std::stringstream s;
434 set_locale_for_separators(s);
435 s << std::fixed << metric.as<CountNode>()->value;
436 val = s.str();
437 } else if (metric.as<DurationNode>()) {
438 std::stringstream s;
439 set_locale_for_separators(s);
440 s << std::fixed << std::setprecision(2) << metric.as<DurationNode>()->microseconds;
441 val = s.str();
442 } else if (metric.as<PercentNode>()) {
443 std::stringstream s;
444 s << std::fixed << std::setprecision(2) << metric.as<PercentNode>()->percent;
445 val = s.str();
446 } else if (metric.as<RatioNode>()) {
447 std::stringstream s;
448 set_locale_for_separators(s);
449 s << std::setprecision(2) << metric.as<RatioNode>()->ratio;
450 val = s.str();
451 } else if (metric.as<StringObj>()) {
452 val = Downcast<String>(metric);
453 } else {
454 LOG(FATAL) << "Cannot print metric of type " << metric->GetTypeKey();
455 }
456 return val;
457}

459String ReportNode::AsTable(bool sort, bool aggregate, bool compute_col_sums) const {
460 // aggregate calls by op hash (or op name if hash is not set) + argument shapes
461 std::vector<Map<String, ObjectRef>> aggregated_calls;
462 if (aggregate) {
463 std::unordered_map<std::string, std::vector<size_t>> aggregates;
464 for (size_t i = 0; i < calls.size(); i++) {
465 auto& frame = calls[i];
466 auto it = frame.find("Hash");
467 std::string name = Downcast<String>(frame["Name"]);
468 if (it != frame.end()) {
469 name = Downcast<String>((*it).second);
470 }
471 if (frame.find("Argument Shapes") != frame.end()) {
472 name += Downcast<String>(frame["Argument Shapes"]);
473 }
474 if (frame.find("Device") != frame.end()) {
475 name += Downcast<String>(frame["Device"]);
476 }
477
478 if (aggregates.find(name) == aggregates.end()) {
479 aggregates[name] = {i};
480 } else {
481 aggregates[name].push_back(i);
482 }
483 }
484 for (const auto& p : aggregates) {
485 std::unordered_map<String, ObjectRef> aggregated;
486 std::unordered_set<std::string> metrics;
487 for (auto& call : calls) {
488 for (auto& metric : call) {
489 metrics.insert(metric.first);
490 }
491 }
492 for (const std::string& metric : metrics) {
493 std::vector<ObjectRef> per_call;
494 for (auto i : p.second) {
495 auto& call = calls[i];
496 auto it = std::find_if(call.begin(), call.end(),
497 [&metric](const std::pair<String, ObjectRef>& call_metric) {
498 return std::string(call_metric.first) == metric;
499 });
500 if (it != call.end()) {
501 per_call.push_back((*it).second);
502 }
503 }
504 if (per_call.size() > 0) {
505 aggregated[metric] = AggregateMetric(per_call);
506 }
507 }
508 aggregated_calls.push_back(aggregated);
509 }
510 } else {
511 for (auto call : calls) {
512 aggregated_calls.push_back(call);
513 }
514 }
515
516 // sort rows by duration
517 if (sort) {
518 std::sort(aggregated_calls.begin(), aggregated_calls.end(),
519 [&](const Map<String, ObjectRef>& a, const Map<String, ObjectRef>& b) {
520 return a.at("Duration (us)").as<DurationNode>()->microseconds >
521 b.at("Duration (us)").as<DurationNode>()->microseconds;
522 });
523 }
524
525 // compute columnwise sums
526 if (compute_col_sums) {
527 std::unordered_map<String, ObjectRef> col_sums;
528 for (auto call : aggregated_calls) {
529 for (auto p : call) {
530 if (p.second.as<CountNode>()) {
531 int64_t val = p.second.as<CountNode>()->value;
532 auto it = col_sums.find(p.first);
533 if (it != col_sums.end()) {
534 val += it->second.as<CountNode>()->value;
535 }
536 col_sums[p.first] = ObjectRef(make_object<CountNode>(val));
537 } else if (p.second.as<DurationNode>()) {
538 double val = p.second.as<DurationNode>()->microseconds;
539 auto it = col_sums.find(p.first);
540 if (it != col_sums.end()) {
541 val += it->second.as<DurationNode>()->microseconds;
542 }
543 col_sums[p.first] = ObjectRef(make_object<DurationNode>(val));
544 } else if (p.second.as<PercentNode>()) {
545 double val = p.second.as<PercentNode>()->percent;
546 auto it = col_sums.find(p.first);
547 if (it != col_sums.end()) {
548 val += it->second.as<PercentNode>()->percent;
549 }
550 col_sums[p.first] = ObjectRef(make_object<PercentNode>(val));
551 } else if (p.second.as<RatioNode>()) {
552 // It does not make sense to sum ratios
553 }
554 }
555 }
556 col_sums["Name"] = String("Sum");
557 aggregated_calls.push_back({{String("Name"), String("----------")}}); // separator
558 aggregated_calls.push_back(col_sums);
559 }
560
561 // per-device metrics
562 for (auto p : device_metrics) {
563 Map<String, ObjectRef> metrics = p.second;
564 metrics.Set("Name", String("Total"));
565 aggregated_calls.push_back(metrics);
566 }
567
568 // Table formatting
569 std::set<std::string> unique_headers;
570 for (auto row : aggregated_calls) {
571 for (auto p : row) {
572 unique_headers.insert(p.first);
573 }
574 }
575
576 // always include these headers in this order
577 std::vector<std::string> headers = {"Name", "Duration (us)", "Percent",
578 "Device", "Count", "Argument Shapes"};
579 for (auto header : unique_headers) {
580 if (std::find(headers.begin(), headers.end(), header) == headers.end()) {
581 headers.push_back(header);
582 }
583 }
584
585 // Switch layout from row major to column major so we can easily compute column widths.
586 std::vector<std::vector<std::string>> cols;
587 for (auto header : headers) {
588 cols.push_back({header});
589 }
590 for (auto row : aggregated_calls) {
591 for (size_t i = 0; i < headers.size(); i++) {
592 auto it = row.find(headers[i]);
593 if (it == row.end()) {
594 // fill empty data with empty strings
595 cols[i].push_back("");
596 } else {
597 cols[i].push_back(print_metric((*it).second));
598 }
599 }
600 }
601
602 std::vector<size_t> widths;
603 for (auto v : cols) {
604 size_t width = 0;
605 for (auto x : v) {
606 width = std::max(width, x.size());
607 }
608 widths.push_back(width);
609 }
610 size_t length = 0;
611 for (auto v : cols) {
612 length = std::max(length, v.size());
613 }
614
615 std::stringstream s;
616 for (size_t row = 0; row < length; row++) {
617 for (size_t col = 0; col < cols.size(); col++) {
618 // left align first column
619 if (col == 0) {
620 s << std::left;
621 } else {
622 s << std::right;
623 }
624 if (row < cols[col].size()) {
625 s << std::setw(widths[col]) << cols[col][row] << " ";
626 } else {
627 s << std::setw(widths[col]) << ""
628 << " ";
629 }
630 }
631 s << std::endl;
632 }
633
634 // Add configuration information. It will not be aligned with the columns.
635 s << std::endl << "Configuration" << std::endl << "-------------" << std::endl;
636 for (auto kv : configuration) {
637 s << kv.first << ": " << print_metric(kv.second) << std::endl;
638 }
639 return s.str();
640}

646Report Profiler::Report() {
647 // sync all timers and normalize rows
648 std::vector<std::unordered_map<String, ObjectRef>> rows;
649 for (auto& cf : calls_) {
650 std::unordered_map<String, ObjectRef> row;
651 double us = cf.timer->SyncAndGetElapsedNanos() / 1e3;
652 row["Duration (us)"] = ObjectRef(make_object<DurationNode>(us));
653 row["Count"] = ObjectRef(make_object<CountNode>(1));
654 row["Name"] = cf.name;
655 row["Device"] = String(DeviceString(cf.dev));
656 for (auto p : cf.extra_metrics) {
657 row[p.first] = p.second;
658 }
659 rows.push_back(row);
660 }
661
662 // the last couple of call frames are the overall times
663 double overall_time_us = 0;
664 std::unordered_map<String, Map<String, ObjectRef>> device_metrics;
665 for (size_t i = 0; i < devs_.size(); i++) {
666 auto row = rows[rows.size() - 1];
667 rows.pop_back();
668 device_metrics[Downcast<String>(row["Device"])] = row;
669 overall_time_us =
670 std::max(overall_time_us, row["Duration (us)"].as<DurationNode>()->microseconds);
671 }
672
673 // Calculate percentages
674 for (auto& row : rows) {
675 row["Percent"] = ObjectRef(make_object<PercentNode>(
676 row["Duration (us)"].as<DurationNode>()->microseconds / overall_time_us * 100));
677 }
678
679 // convert to map
680 std::vector<Map<String, ObjectRef>> converted_rows;
681 for (const auto& row : rows) {
682 converted_rows.push_back(row);
683 }
684
685 return profiling::Report(converted_rows, device_metrics, configuration_);
686}

688Report::Report(Array<Map<String, ObjectRef>> calls,
689 Map<String, Map<String, ObjectRef>> device_metrics,
690 Map<String, ObjectRef> configuration) {

693 node->device_metrics = std::move(device_metrics);
694 node->configuration = std::move(configuration);

698Map<String, ObjectRef> parse_metrics(dmlc::JSONReader* reader) {
699 reader->BeginObject();
700 std::string metric_name, metric_value_name;
701 Map<String, ObjectRef> metrics;
702 while (reader->NextObjectItem(&metric_name)) {
703 ObjectRef o;
704 reader->BeginObject();
705 reader->NextObjectItem(&metric_value_name);
706 if (metric_value_name == "microseconds") {
707 double microseconds;
708 reader->Read(&microseconds);
709 o = ObjectRef(make_object<DurationNode>(microseconds));
710 } else if (metric_value_name == "percent") {
711 double percent;
712 reader->Read(&percent);
713 o = ObjectRef(make_object<PercentNode>(percent));
714 } else if (metric_value_name == "count") {
715 int64_t count;
716 reader->Read(&count);
717 o = ObjectRef(make_object<CountNode>(count));
718 } else if (metric_value_name == "ratio") {
719 double ratio;
720 reader->Read(&ratio);
721 o = ObjectRef(make_object<RatioNode>(ratio));
722 } else if (metric_value_name == "string") {
723 std::string s;
724 reader->Read(&s);
725 o = String(s);
726 } else {
727 LOG(FATAL) << "Cannot parse metric of type " << metric_value_name
728 << " valid types are microseconds, percent, count.";
729 }
730 metrics.Set(metric_name, o);
731 // Necessary to make sure that the parser hits the end of the object.
732 ICHECK(!reader->NextObjectItem(&metric_value_name));
733 // EndObject does not exist, leaving this here for clarity
734 // reader.EndObject();
735 }
736 // reader.EndObject();
737 return metrics;
738}

740Report Report::FromJSON(String json) {
741 std::stringstream input(json.operator std::string());
742 dmlc::JSONReader reader(&input);
743 std::string key;
744 Array<Map<String, ObjectRef>> calls;
745 Map<String, Map<String, ObjectRef>> device_metrics;
746 Map<String, ObjectRef> configuration;
747
748 reader.BeginObject();
749 while (reader.NextObjectItem(&key)) {
750 if (key == "calls") {
751 reader.BeginArray();
752 while (reader.NextArrayItem()) {
753 calls.push_back(parse_metrics(&reader));
754 }
755 // reader.EndArray();
756 } else if (key == "device_metrics") {
757 reader.BeginObject();
758 std::string device_name;
759 while (reader.NextObjectItem(&device_name)) {
760 device_metrics.Set(device_name, parse_metrics(&reader));
761 }
762 // reader.EndObject();
763 } else if (key == "configuration") {
764 configuration = parse_metrics(&reader);
765 }
766 }
767
768 return Report(calls, device_metrics, configuration);
769}

789PackedFunc ProfileFunction(Module mod, std::string func_name, int device_type, int device_id,
790 int warmup_iters, Array<MetricCollector> collectors) {
791 // Module::GetFunction is not const, so this lambda has to be mutable
792 return PackedFunc([=](TVMArgs args, TVMRetValue* ret) mutable {
793 PackedFunc f = mod.GetFunction(func_name);
794 CHECK(f.defined()) << "There is no function called \"" << func_name << "\" in the module";
795 Device dev{static_cast<DLDeviceType>(device_type), device_id};
796
797 // warmup
798 for (int i = 0; i < warmup_iters; i++) {
799 f.CallPacked(args, ret);
800 }
801
802 for (auto& collector : collectors) {
803 collector->Init({DeviceWrapper(dev)});
804 }
805 std::vector<Map<String, ObjectRef>> results;
806 results.reserve(collectors.size());
807 std::vector<std::pair<MetricCollector, ObjectRef>> collector_data;
808 collector_data.reserve(collectors.size());
809 for (auto& collector : collectors) {
810 ObjectRef o = collector->Start(dev);
811 // If not defined, then the collector cannot time this device.
812 if (o.defined()) {
813 collector_data.push_back({collector, o});
814 }
815 }
816
817 // TODO(tkonolige): repeated calls if the runtime is small?
818 f.CallPacked(args, ret);
819
820 for (auto& kv : collector_data) {
821 results.push_back(kv.first->Stop(kv.second));
822 }
823 Map<String, ObjectRef> combined_results;
824 for (auto m : results) {
825 for (auto p : m) {
826 // assume that there is no shared metric name between collectors
827 combined_results.Set(p.first, p.second);
828 }
829 }
830 *ret = combined_results;
831 });
832}

847 }

850PackedFunc WrapTimeEvaluator(PackedFunc pf, Device dev, int number, int repeat, int min_repeat_ms,
851 int limit_zero_time_iterations, int cooldown_interval_ms,
852 int repeats_to_cooldown, PackedFunc f_preproc) {
853 ICHECK(pf != nullptr);
854
855 if (static_cast<int>(dev.device_type) == static_cast<int>(kDLMicroDev)) {
856 auto get_micro_time_evaluator = runtime::Registry::Get("micro._GetMicroTimeEvaluator");
857 ICHECK(get_micro_time_evaluator != nullptr) << "micro backend not enabled";
858 return (*get_micro_time_evaluator)(pf, dev, number, repeat);
859 }
860
861 auto ftimer = [pf, dev, number, repeat, min_repeat_ms, limit_zero_time_iterations,
862 cooldown_interval_ms, repeats_to_cooldown,
863 f_preproc](TVMArgs args, TVMRetValue* rv) mutable {
864 TVMRetValue temp;
865 std::ostringstream os;
866 // skip first time call, to activate lazy compilation components.
867 pf.CallPacked(args, &temp);
868
869 DeviceAPI::Get(dev)->StreamSync(dev, nullptr);
870
871 for (int i = 0; i < repeat; ++i) {
872 if (f_preproc != nullptr) {
873 f_preproc.CallPacked(args, &temp);
874 }
875 double duration_ms = 0.0;
876 int absolute_zero_times = 0;
877 do {
878 if (duration_ms > 0.0) {
879 const double golden_ratio = 1.618;
880 number = static_cast<int>(
881 std::max((min_repeat_ms / (duration_ms / number) + 1), number * golden_ratio));
882 }
883
884 // start timing
885 Timer t = Timer::Start(dev);
886 for (int j = 0; j < number; ++j) {
887 pf.CallPacked(args, &temp);
888 }
889 t->Stop();
890 int64_t t_nanos = t->SyncAndGetElapsedNanos();
891 if (t_nanos == 0) absolute_zero_times++;
892 duration_ms = t_nanos / 1e6;
893 } while (duration_ms < min_repeat_ms && absolute_zero_times < limit_zero_time_iterations);
894
895 double speed = duration_ms / 1e3 / number;
896 os.write(reinterpret_cast<char*>(&speed), sizeof(speed));
897
898 if (cooldown_interval_ms > 0 && (i % repeats_to_cooldown) == 0) {
899 std::this_thread::sleep_for(std::chrono::milliseconds(cooldown_interval_ms));
900 }
901 }
902
903 std::string blob = os.str();
904 TVMByteArray arr;
905 arr.size = blob.length();
906 arr.data = blob.data();
907 // return the time.
908 *rv = arr;
909 };
910 return PackedFunc(ftimer);
911}
