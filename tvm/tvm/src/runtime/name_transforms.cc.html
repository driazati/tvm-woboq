24#include <algorithm>
25#include <cctype>
26#include <string>
27
28namespace tvm {
29namespace runtime {
30
31std::string SanitizeName(const std::string& name) {
32 ICHECK(!name.empty()) << "Name is empty";
33
34 auto isNotAlnum = [](char c) { return !std::isalnum(c); };
35 std::string sanitized_input = name;
36 std::replace_if(sanitized_input.begin(), sanitized_input.end(), isNotAlnum, '_');
37
38 return sanitized_input;
39}
40
41TVM_REGISTER_GLOBAL("runtime.SanitizeName").set_body_typed(SanitizeName);
42
