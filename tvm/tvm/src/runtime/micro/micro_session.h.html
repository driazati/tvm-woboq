20/*!
21 * \file micro_session.h
22 * \brief session to manage multiple micro modules
23 *
24 * Each session consists of an interaction with a *single* logical device.
25 * Within that interaction, multiple TVM modules can be loaded on the logical
26 * device.
27 *
28 * Multiple sessions can exist simultaneously, but there is only ever one
29 * *active* session. The idea of an active session mainly has implications for
30 * the frontend, in that one must make a session active in order to allocate
31 * new TVM objects on it. Aside from that, previously allocated objects can be
32 * used even if the session which they belong to is not currently active.
33 */
34#ifndef TVM_RUNTIME_MICRO_MICRO_SESSION_H_
35#define TVM_RUNTIME_MICRO_MICRO_SESSION_H_
36
37#endif // TVM_RUNTIME_MICRO_MICRO_SESSION_H_
