50class CallbackWriteStream : public WriteStream {
51 public:
52 explicit CallbackWriteStream(PackedFunc fsend, ::std::chrono::microseconds write_timeout)
53 : fsend_{fsend}, write_timeout_{write_timeout} {}
54
55 ssize_t Write(const uint8_t* data, size_t data_size_bytes) override {
56 TVMByteArray bytes;
57 bytes.data = (const char*)data;
58 bytes.size = data_size_bytes;
59 if (write_timeout_ == ::std::chrono::microseconds::zero()) {
60 fsend_(bytes, nullptr);
61 } else {
62 fsend_(bytes, write_timeout_.count());
63 }
64
65 return static_cast<ssize_t>(data_size_bytes);
66 }
67
68 void PacketDone(bool is_valid) override {}
69
70 void SetWriteTimeout(::std::chrono::microseconds timeout) { write_timeout_ = timeout; }
71
72 private:
73 PackedFunc fsend_;
74 ::std::chrono::microseconds write_timeout_;
75};

77class MicroTransportChannel : public RPCChannel {
78 public:
79 enum class State : uint8_t {
80 kReset = 0, // state entered before the transport has been read or written to.
81 kSessionTerminated = 1, // session is terminated, but transport is alive.
82 kSessionEstablished = 2, // session is alive.
83 };
84
85 /*!
86 * \brief Construct a new MicroTransportChannel.
87 * \param fsend A PackedFunc accepting (data_bytes, timeout_usec) and returning the number of
88 * bytes sent. If a timeout_usec elapses before all data is sent, it should return 0.
89 * \param frecv A PackedFunc accepting (num_bytes, timeout_usec) and returning a string containing
90 * the received data. Must not return an empty string, except to indicate a timeout.
91 * \param session_start_retry_timeout During session initialization, the session start message is
92 * re-sent after this many microseconds elapse without a reply. If 0, the session start message
93 * is sent only once.
94 * \param session_start_timeout Session initialization is considered "timed out" if no reply is
95 * received this many microseconds after the session start is sent. If 0, a session start never
96 * times out.
97 * \param session_established_timeout Timeout used for the Recv() function. This is used for
98 * messages sent after a session is already established. If 0, Recv() never times out.
99 */
100 MicroTransportChannel(PackedFunc fsend, PackedFunc frecv,
101 ::std::chrono::microseconds session_start_retry_timeout,
102 ::std::chrono::microseconds session_start_timeout,
103 ::std::chrono::microseconds session_established_timeout)
104 : state_{State::kReset},
105 session_start_retry_timeout_{session_start_retry_timeout},
106 session_start_timeout_{session_start_timeout},
107 session_established_timeout_{session_established_timeout},
108 write_stream_{fsend, session_start_timeout},
109 framer_{&write_stream_},
110 receive_buffer_{new uint8_t[TVM_CRT_MAX_PACKET_SIZE_BYTES], TVM_CRT_MAX_PACKET_SIZE_BYTES},
111 session_{&framer_, &receive_buffer_, &HandleMessageReceivedCb, this},
112 unframer_{session_.Receiver()},
113 did_receive_message_{false},
114 frecv_{frecv},
115 message_buffer_{nullptr} {}
116
117 private:
118 static constexpr const size_t kReceiveBufferSizeBytes = 128;
119
120 /*
121 * \brief Receive data until either pf() returns true or a timeout occurs.
122 *
123 * The condition function is called first, so this function may return without performing a read.
124 * Following this call, received data is consumed and frecv_ is invoked until the timeout occurs
125 * or the condition function passes.
126 *
127 * \param pf A condition function that returns true when enough data has been received for the
128 * caller to proceed.
129 * \param timeout Pointer to number of microseconds to wait before timing out. If nullptr, no
130 * timeout ever occurs in this function, so it may block forever. If 0, a single non-blocking
131 * read is performed, and any data returned is processed.
132 * \return true if the condition passed, false if the timeout expired.
133 */
134 bool ReceiveUntil(TypedPackedFunc<bool(void)> pf, ::std::chrono::microseconds* timeout) {
135 if (pf()) {
136 return true;
137 }
138
139 auto end_time = ::std::chrono::steady_clock::now();
140 if (timeout != nullptr) {
141 end_time += *timeout;
142 }
143 for (;;) {
144 if (ConsumeReceivedPayload(pf)) {
145 return true;
146 }
147
148 ::std::string chunk;
149 size_t bytes_needed = unframer_.BytesNeeded();
150 CHECK_GT(bytes_needed, 0) << "unframer unexpectedly needs no data";
151 if (timeout != nullptr) {
152 ::std::chrono::microseconds iter_timeout{
153 ::std::max(::std::chrono::microseconds{0},
154 ::std::chrono::duration_cast<::std::chrono::microseconds>(
155 end_time - ::std::chrono::steady_clock::now()))};
156 chunk = frecv_(bytes_needed, iter_timeout.count()).operator std::string();
157 } else {
158 chunk = frecv_(bytes_needed, nullptr).operator std::string();
159 }
160 pending_chunk_ = chunk;
161 if (pending_chunk_.size() == 0) {
162 // Timeout occurred
163 return false;
164 }
165 }
166 }
167
168 static constexpr const int kNumRandRetries = 10;
169 static std::atomic<unsigned int> random_seed;
170
171 inline uint8_t GenerateRandomNonce() {
172 // NOTE: this is bad concurrent programming but in practice we don't really expect race
173 // conditions here, and even if they occur we don't particularly care whether a competing
174 // process computes a different random seed. This value is just chosen pseudo-randomly to
175 // form an initial distinct session id. Here we just want to protect against bad loads causing
176 // confusion.
177 unsigned int seed = random_seed.load();
178 if (seed == 0) {
179 seed = (unsigned int)time(nullptr);
180 }
181 uint8_t initial_nonce = 0;
182 for (int i = 0; i < kNumRandRetries && initial_nonce == 0; ++i) {
183 initial_nonce = rand_r(&seed);
184 }
185 random_seed.store(seed);
186 ICHECK_NE(initial_nonce, 0) << "rand() does not seem to be producing random values";
187 return initial_nonce;
188 }
189
190 bool StartSessionInternal() {
191 using ::std::chrono::duration_cast;
192 using ::std::chrono::microseconds;
193 using ::std::chrono::steady_clock;
194
195 steady_clock::time_point start_time = steady_clock::now();
196 ICHECK_EQ(kTvmErrorNoError, session_.Initialize(GenerateRandomNonce()));
197 ICHECK_EQ(kTvmErrorNoError, session_.StartSession());
198
199 if (session_start_timeout_ == microseconds::zero() &&
200 session_start_retry_timeout_ == microseconds::zero()) {
201 ICHECK(ReceiveUntil([this]() -> bool { return session_.IsEstablished(); }, nullptr))
202 << "ReceiveUntil indicated timeout expired, but no timeout set!";
203 ICHECK(session_.IsEstablished()) << "Session not established, but should be";
204 return true;
205 }
206
207 auto session_start_end_time = start_time + session_start_timeout_;
208 steady_clock::time_point end_time;
209 if (session_start_retry_timeout_ != ::std::chrono::microseconds::zero()) {
210 end_time = start_time + session_start_retry_timeout_;
211 } else {
212 end_time = session_start_end_time;
213 }
214
215 while (!session_.IsEstablished()) {
216 microseconds time_remaining =
217 ::std::max(microseconds{0}, duration_cast<microseconds>(end_time - steady_clock::now()));
218 if (ReceiveUntil([this]() -> bool { return session_.IsEstablished(); }, &time_remaining)) {
219 break;
220 }
221
222 if (session_start_timeout_ != microseconds::zero() && end_time >= session_start_end_time) {
223 return false;
224 }
225 end_time += session_start_retry_timeout_;
226
227 ICHECK_EQ(kTvmErrorNoError, session_.Initialize(GenerateRandomNonce()));
228 ICHECK_EQ(kTvmErrorNoError, session_.StartSession());
229 }
230
231 return true;
232 }
233
234 public:
235 bool StartSession() {
236 ICHECK(state_ == State::kReset)
237 << "MicroSession: state_: expected kReset, got " << uint8_t(state_);
238
239 bool to_return = StartSessionInternal();
240 if (to_return) {
241 write_stream_.SetWriteTimeout(session_established_timeout_);
242 }
243
244 return to_return;
245 }
246
247 size_t Send(const void* data, size_t size) override {
248 const uint8_t* data_bytes = static_cast<const uint8_t*>(data);
249 tvm_crt_error_t err = session_.SendMessage(MessageType::kNormal, data_bytes, size);
250 ICHECK(err == kTvmErrorNoError) << "SendMessage returned " << err;
251
252 return size;
253 }
254
255 size_t Recv(void* data, size_t size) override {
256 size_t num_bytes_recv = 0;
257 while (num_bytes_recv < size) {
258 if (message_buffer_ != nullptr) {
259 num_bytes_recv += message_buffer_->Read(static_cast<uint8_t*>(data), size);
260 if (message_buffer_->ReadAvailable() == 0) {
261 message_buffer_ = nullptr;
262 session_.ClearReceiveBuffer();
263 }
264 if (num_bytes_recv == size) {
265 ICHECK(message_buffer_ == nullptr || message_buffer_->ReadAvailable() > 0);
266 return num_bytes_recv;
267 }
268 }
269
270 did_receive_message_ = false;
271 if (session_established_timeout_ == ::std::chrono::microseconds::zero()) {
272 ICHECK(ReceiveUntil([this]() -> bool { return did_receive_message_; }, nullptr))
273 << "ReceiveUntil timeout expired, but no timeout configured!";
274 } else {
275 if (!ReceiveUntil([this]() -> bool { return did_receive_message_; },
276 &session_established_timeout_)) {
277 std::stringstream ss;
278 ss << "MicroSessionTimeoutError: failed to read reply message after timeout "
279 << session_established_timeout_.count() / 1e6 << "s";
280
281 throw std::runtime_error(ss.str());
282 }
283 }
284 }
285
286 return num_bytes_recv;
287 }
288
289 FrameBuffer* GetReceivedMessage() {
290 if (did_receive_message_) {
291 did_receive_message_ = false;
292 return message_buffer_;
293 }
294
295 return nullptr;
296 }
297
298 private:
299 /*!
300 * \brief Consume the entire received payload, unless the pf condition is met halfway through.
301 *
302 * This function expects pending_chunk_ to contain a chunk of unprocessed packet data. It
303 * repeatedly writes the chunk to the Unframer until either a) pf() returns True or b) no more
304 * data remains to be written.
305 *
306 * \param pf A PackedFunc which returns true when ReceiveUntil should return.
307 * \returns true if pf() returned true during processing; false otherwise.
308 */
309 bool ConsumeReceivedPayload(TypedPackedFunc<bool(void)> pf) {
310 while (pending_chunk_.size() > 0) {
311 size_t bytes_consumed = 0;
312 int unframer_error = unframer_.Write((const uint8_t*)pending_chunk_.data(),
313 pending_chunk_.size(), &bytes_consumed);
314
315 ICHECK(bytes_consumed <= pending_chunk_.size())
316 << "consumed " << bytes_consumed << " want <= " << pending_chunk_.size();
317 pending_chunk_ = pending_chunk_.substr(bytes_consumed);
318 if (unframer_error < 0) {
319 LOG(ERROR) << "unframer got error code: " << unframer_error;
320 } else {
321 if (pf()) {
322 return true;
323 }
324 }
325 }
326
327 return false;
328 }

330 static void HandleMessageReceivedCb(void* context, MessageType message_type, FrameBuffer* buf) {
331 static_cast<MicroTransportChannel*>(context)->HandleMessageReceived(message_type, buf);
332 }
333
334 void HandleMessageReceived(MessageType message_type, FrameBuffer* buf) {
335 size_t message_size_bytes;
336 switch (message_type) {
337 case MessageType::kStartSessionInit:
338 break;
339
340 case MessageType::kStartSessionReply:
341 state_ = State::kSessionEstablished;
342 break;
343
344 case MessageType::kTerminateSession:
345 if (state_ == State::kReset) {
346 state_ = State::kSessionTerminated;
347 } else if (state_ == State::kSessionTerminated) {
348 LOG(FATAL) << "SessionTerminatedError: multiple session-terminated messages received; "
349 "device in reboot loop?";
350 } else if (state_ == State::kSessionEstablished) {
351 LOG(FATAL) << "SessionTerminatedError: remote device terminated connection";
352 }
353 break;
354
355 case MessageType::kLog:
356 uint8_t message[1024];
357 message_size_bytes = buf->ReadAvailable();
358 if (message_size_bytes == 0) {
359 return;
360 } else if (message_size_bytes > sizeof(message) - 1) {
361 LOG(ERROR) << "Remote log message is too long to display: " << message_size_bytes
362 << " bytes";
363 return;
364 }
365
366 ICHECK_EQ(buf->Read(message, sizeof(message) - 1), message_size_bytes);
367 message[message_size_bytes] = 0;
368 LOG(INFO) << "remote: " << message;
369 session_.ClearReceiveBuffer();
370 return;
371
372 case MessageType::kNormal:
373 did_receive_message_ = true;
374 message_buffer_ = buf;
375 break;
376 }
377 }
378
379 State state_;
380 ::std::chrono::microseconds session_start_retry_timeout_;
381 ::std::chrono::microseconds session_start_timeout_;
382 ::std::chrono::microseconds session_established_timeout_;
383 CallbackWriteStream write_stream_;
384 Framer framer_;
385 FrameBuffer receive_buffer_;
386 Session session_;
387 Unframer unframer_;
388 bool did_receive_message_;
389 PackedFunc frecv_;
390 FrameBuffer* message_buffer_;
391 std::string pending_chunk_;
392};

396TVM_REGISTER_GLOBAL("micro._rpc_connect").set_body([](TVMArgs args, TVMRetValue* rv) {
397 MicroTransportChannel* micro_channel =
398 new MicroTransportChannel(args[1], args[2], ::std::chrono::microseconds(uint64_t(args[3])),
399 ::std::chrono::microseconds(uint64_t(args[4])),
400 ::std::chrono::microseconds(uint64_t(args[5])));
401 if (!micro_channel->StartSession()) {
402 std::stringstream ss;
403 ss << "MicroSessionTimeoutError: session start handshake failed after " << double(args[4]) / 1e6
404 << "s";
405 throw std::runtime_error(ss.str());
406 }
407 std::unique_ptr<RPCChannel> channel(micro_channel);
408 bool enable_logging = false;
409 if (args.num_args > 7) {
410 enable_logging = args[7];
411 }
412 if (enable_logging) {
413 channel.reset(new RPCChannelLogging(std::move(channel)));
414 }
415 auto ep = RPCEndpoint::Create(std::move(channel), args[0], "", args[6]);
416 auto sess = CreateClientSession(ep);
417 *rv = CreateRPCSessionModule(sess);
418});
