20/*!
21 * \file builtin_fp16.cc
22 * \brief Functions for conversion between fp32 and fp16
23 */
24#include <builtin_fp16.h>
25#include <tvm/runtime/c_runtime_api.h>
26
27extern "C" {
28
29// disable under msvc
30#ifndef _MSC_VER
31
32TVM_DLL TVM_WEAK uint16_t __gnu_f2h_ieee(float a) {
33 return __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 10>(a);
34}
35
36TVM_DLL TVM_WEAK float __gnu_h2f_ieee(uint16_t a) {
37 return __extendXfYf2__<uint16_t, uint16_t, 10, float, uint32_t, 23>(a);
38}
39
40#else
41
42TVM_DLL uint16_t __gnu_f2h_ieee(float a) {
43 return __truncXfYf2__<float, uint32_t, 23, uint16_t, uint16_t, 10>(a);
44}
45
46TVM_DLL float __gnu_h2f_ieee(uint16_t a) {
47 return __extendXfYf2__<uint16_t, uint16_t, 10, float, uint32_t, 23>(a);
48}
49
50#endif
51}
