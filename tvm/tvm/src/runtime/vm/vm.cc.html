69inline ObjectRef CopyTo(ObjectRef src, const DLDevice& dev) {
70 if (src->IsInstance<NDArray::ContainerType>()) {
71 auto nd_array = Downcast<NDArray>(src);
72 // TODO(mbs): Should respect device id also.
73 // TODO(vvchernov): it still does not work for different device id
74 // due to simple implementation of Get() and AllocDataSpace() methods
75 // see tvm/src/runtime/c_runtime_api.cc: L139
76 // tvm/src/runtime/cpu_device_api.cc: L47
77 if (nd_array->device.device_type != dev.device_type ||
78 nd_array->device.device_id != dev.device_id) {
79 VLOG(2) << "copying from " << nd_array->device.device_type << "["
80 << nd_array->device.device_id << "] to " << dev.device_type << "[" << dev.device_id
81 << "]";
82 return nd_array.CopyTo(dev);
83 }
84 return src;
85 } else {
86 ICHECK(src->IsInstance<ADTObj>())
87 << "VM data must be NDArray or a list of NDArray, but received: " << src->_type_key;
88 std::vector<ObjectRef> ret;
89 ADT adt = Downcast<ADT>(src);
90 for (size_t i = 0; i < adt.size(); i++) {
91 ret.push_back(CopyTo(adt[i], dev));
92 }
93 return ADT(adt->tag, ret.begin(), ret.end());
94 }

97std::vector<int64_t> ToShape(NDArray shape_tensor) {
98 std::vector<int64_t> shape;
99 auto rank = shape_tensor.Shape().size();
100 auto dtype = shape_tensor.DataType();
101
102 // For 0-rank shapes we need to allocate a single scalar.
103 if (rank == 0) {
104 return shape;
105 }
106
107 // Otherwise we should be rank-1, and we will extract the number of dimensions
108 // for the output vector.
109 ICHECK_EQ(rank, 1U) << "shape tensor should be a k-length vector, found " << rank;
110 int64_t ndim = shape_tensor.Shape().at(0);
111 shape.resize(ndim);
112
113 const DLTensor* dl_tensor = shape_tensor.operator->();
114 if (dtype.is_int() && dtype.bits() == 32 && dtype.lanes() == 1) {
115 int32_t* dims = reinterpret_cast<int32_t*>(dl_tensor->data);
116 shape.assign(dims, dims + ndim);
117 } else if (dtype.is_int() && dtype.bits() == 64 && dtype.lanes() == 1) {
118 int64_t* dims = reinterpret_cast<int64_t*>(dl_tensor->data);
119 shape.assign(dims, dims + ndim);
120 } else {
121 LOG(FATAL) << "invalid shape tensor datatype: " << dtype;
122 }
123
124 return shape;
125}

130PackedFunc VirtualMachine::GetFunction(const std::string& name,
131 const ObjectPtr<Object>& sptr_to_self) {
132 if (name == "invoke") {
133 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
134 ICHECK(exec_) << "The executable is not created yet.";
135
136 std::string func_name = args[0];
137 auto git = exec_->global_map.find(func_name);
138 ICHECK(git != exec_->global_map.end())
139 << "Cannot find function " << func_name << " in the executable";
140 auto func = exec_->functions[git->second];
141 if (func.params.empty()) {
142 *rv = Invoke(func, {});
143 } else {
144 auto it = inputs_.find(func_name);
145 ICHECK(it != inputs_.end()) << "Input has not been set for function " << func_name;
146 const std::vector<ObjectRef>& func_args = it->second;
147 *rv = Invoke(func, func_args);
148 }
149 });
150 } else if (name == "invoke_stateful") {
151 // TODO(tkonolige, jroesch, tqchen): invoke_stateful and get_output are
152 // stop-gap measure to allow using vm over a remote connection.
153 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
154 PackedFunc invoke = GetFunction("invoke", sptr_to_self);
155 TVMRetValue rv_;
156 invoke.CallPacked(args, &rv_);
157 });
158 } else if (name == "invoke_return_to_device") {
159 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
160 Device host{static_cast<DLDeviceType>(args[1].operator int()), args[2].operator int()};
161
162 SetInput(args[0].operator std::string(), args, 3);
163 PackedFunc invoke = GetFunction("invoke", sptr_to_self);
164 TVMRetValue rv_;
165 invoke.CallPacked(args, &rv_); // Invoke only uses the first arg, so the rest of the args
166 // should not cause an issue
167 if (rv_.type_code() == kTVMObjectHandle) {
168 ADT adt = Downcast<ADT>(rv_.operator ObjectRef());
169 std::vector<ObjectRef> transfered;
170 for (size_t i = 0; i < adt.size(); i++) {
171 transfered.push_back(CopyTo(adt[i], host));
172 }
173 *rv = ADT(adt.tag(), transfered);
174 } else {
175 *rv = CopyTo(rv_, host);
176 }
177 });
178 } else if (name == "get_output") {
179 return TypedPackedFunc<NDArray(int64_t)>([this](int64_t index) {
180 if (this->return_register_.as<ADTObj>()) {
181 return Downcast<NDArray>(Downcast<ADT>(this->return_register_)[index]);
182 } else {
183 CHECK_EQ(index, 0) << "VM output contains only one item, but you are trying to get the "
184 << index << "th.";
185 return Downcast<NDArray>(this->return_register_);
186 }
187 });
188 } else if (name == "get_num_outputs") {
189 return TypedPackedFunc<int64_t(void)>([this]() -> int64_t {
190 // single output is an NDArray not an ADT
191 if (this->return_register_.as<ADTObj>()) {
192 return Downcast<ADT>(this->return_register_).size();
193 } else {
194 return 1;
195 }
196 });
197 } else if (name == "get_input_index") {
198 return TypedPackedFunc<int64_t(std::string, std::string)>(
199 [this](std::string input_name, std::string func_name) {
200 return GetInputIndexFromVMFunction(func_name, input_name);
201 });
202 } else if (name == "init") {
203 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
204 ICHECK_EQ(args.size() % 3, 0);
205 std::vector<Device> devices;
206 std::vector<AllocatorType> alloc_types;
207 for (int i = 0; i < args.size() / 3; ++i) {
208 Device dev;
209 int device_type = args[i * 3];
210 dev.device_type = DLDeviceType(device_type);
211 dev.device_id = args[i * 3 + 1];
212 int type = args[i * 3 + 2];
213 devices.push_back(dev);
214 alloc_types.push_back(AllocatorType(type));
215 }
216 this->Init(devices, alloc_types);
217 });
218 } else if (name == "set_input") {
219 return PackedFunc(
220 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { SetInput(args[0], args, 1); });
221 } else if (name == "set_one_input") {
222 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
223 ICHECK_EQ(args.size(), 3) << "The expected number of arguments is 3 "
224 << "(func_name, index or name, tensor)";
225 SetOneInput(args[0], args[1], args[2]);
226 });
227 } else if (name == "load_late_bound_consts") {
228 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
229 CHECK_EQ(args.size(), 1);
230 std::string path = args[0];
231 exec_->LoadLateBoundConstantsFromFile(path);
232 });
233 } else {
234 LOG(FATAL) << "Unknown packed function: " << name;
235 return PackedFunc([sptr_to_self, name](TVMArgs args, TVMRetValue* rv) {});
236 }
237}

239void VirtualMachine::SetInput(std::string func_name, TVMArgs args, int offset) {
240 const auto& vm_func = CheckAndGetVMFunction(func_name);
241 size_t params_num = vm_func.params.size();
242 ICHECK_EQ(args.size() - offset, params_num)
243 << "The number of provided parameters doesn't match the number of arguments";
244 std::vector<ObjectRef> func_args(params_num);
245 for (int i = offset; i < args.size(); ++i) {
246 int index = i - offset;
247 Device dev = GetDevice(vm_func.param_device_indexes[index]);
248 SetInputTensorWithIndex(func_args, args[i], index, dev);
249 }
250 inputs_.erase(func_name);
251 inputs_.emplace(func_name, func_args);
252}
253
254void VirtualMachine::SetOneInput(std::string func_name, const TVMArgValue& tag,
255 const TVMArgValue& tensor) {
256 const auto& vm_func = CheckAndGetVMFunction(func_name);
257 size_t params_num = vm_func.params.size();
258
259 int inp_index = 0;
260 if (tag.type_code() == kTVMArgInt) {
261 inp_index = tag;
262 } else if (tag.type_code() == kTVMStr) {
263 inp_index = static_cast<int>(GetInputIndexFromName(vm_func.params, tag));
264 } else {
265 LOG(FATAL) << "The type of input tensor tag (" << tag.type_code()
266 << ") doesn't match integer or string";
267 }
268 ICHECK_LT(inp_index, params_num);
269
270 CreateInputsOrCheckSize(func_name, params_num);
271 Device dev = GetDevice(vm_func.param_device_indexes[inp_index]);
272 SetInputTensorWithIndex(inputs_[func_name], tensor, inp_index, dev);
273}

281int64_t VirtualMachine::GetInputIndexFromName(const std::vector<std::string>& params,
282 const std::string& input_name) const {
283 // TODO(vvchernov): excess integer type?
284 for (uint64_t i = 0; i < params.size(); i++) {
285 if (input_name == params[i]) {
286 return static_cast<int64_t>(i);
287 }
288 }
289 return static_cast<int64_t>(-1);

297void VirtualMachine::CreateInputsOrCheckSize(const std::string& func_name, size_t size) {
298 if (inputs_.count(func_name)) {
299 ICHECK_EQ(inputs_[func_name].size(), size)
300 << "The size of function" << func_name
301 << " doesn't match the number of provided parameters";
302 } else {
303 std::vector<ObjectRef> func_args(size);
304 inputs_.emplace(func_name, func_args);
305 }

308void VirtualMachine::SetInputTensorWithIndex(std::vector<ObjectRef>& tensors,
309 const TVMArgValue& inp_tensor, int index, Device dev) {
310 if (inp_tensor.type_code() == kTVMDLTensorHandle) {
311 if (NDArray::AbilityOfZeroCopyForDLTensor(inp_tensor, dev)) {
312 tensors[index] = NDArray::FromExternalDLTensor(*inp_tensor);
313 } else {
314 tensors[index] = NDArray::NewFromDLTensor(inp_tensor, dev);
315 }
316 } else {
317 tensors[index] = CopyTo(inp_tensor, dev);
318 }
319}

347void VirtualMachine::InvokeGlobal(const VMFunction& func, const std::vector<ObjectRef>& args) {
348 VLOG(2) << "Invoking global " << func.name << " with " << args.size() << " args";
349
350 PushFrame(func.params.size(), this->pc_ + 1, func);
351 for (size_t i = 0; i < args.size(); ++i) {
352 WriteRegister(i, args[i]);
353 VLOG(2) << "arg " << i << " = "
354 << RuntimeObject2String(args[i], GetDevice(exec_->host_device_index));
355 }
356
357 code_ = func.instructions.data();
358 pc_ = 0;
359}

361ObjectRef VirtualMachine::Invoke(const VMFunction& func, const std::vector<ObjectRef>& args) {
362 VLOG(2) << "Executing Function: " << std::endl << func;
363 for (int i = 0; i < static_cast<int>(devices_.size()); ++i) {
364 VLOG(2) << "Device " << i << " has device type " << devices_[i].device_type << " and device id "
365 << devices_[i].device_id
366 << (i == exec_->host_device_index ? " (using as host device)" : "");
367 }
368
369 InvokeGlobal(func, args);
370 RunLoop();
371 return return_register_;
372}

383void VirtualMachine::InvokePacked(Index packed_index, const PackedFunc& func, Index arg_count,
384 Index output_size, const std::vector<ObjectRef>& args) {
385 size_t arity = 0;
386 for (Index i = 0; i < arg_count; i++) {
387 if (const auto* obj = args[i].as<ADTObj>()) {
388 arity += obj->size;
389 } else {
390 ++arity;
391 }
392 }
393
394 std::vector<TVMValue> values(arity);
395 std::vector<int> codes(arity);
396 runtime::TVMArgsSetter setter(values.data(), codes.data());
397 int idx = 0;
398 bool is_empty_output = false;
399 for (Index i = 0; i < arg_count; i++) {
400 if (const auto* dt_cell = args[i].as<ADTObj>()) {
401 for (size_t fi = 0; fi < dt_cell->size; ++fi) {
402 auto obj = (*dt_cell)[fi];
403 auto nd_array = Downcast<NDArray>(obj);
404 setter(idx++, nd_array);
405 }
406 } else {
407 auto nd_array = Downcast<NDArray>(args[i]);
408 // We can safely skip CallPacked if there is only one
409 // output and it is empty.
410 if (i == arg_count - 1 && output_size == 1) {
411 for (const auto& dim : nd_array.Shape()) {
412 if (!dim) {
413 is_empty_output = true;
414 break;
415 }
416 }
417 }
418 setter(idx++, nd_array);
419 }
420 }
421
422 if (!is_empty_output) {
423 TVMRetValue rv;
424 func.CallPacked(TVMArgs(values.data(), codes.data(), arity), &rv);
425 }
426}

428void VirtualMachine::LoadExecutable(const ObjectPtr<Executable>& exec) {
429 ICHECK(exec) << "The executable is not created yet.";
430 ICHECK(exec->late_bound_constant_names.empty())
431 << "Need to load late-bound-constants before creating VM";
432 exec_ = exec;
433
434 runtime::Module lib = exec_->GetLib();
435
436 ICHECK(exec_->primitive_map.empty() || lib.operator->())
437 << "If the executable has declared primitive functions, the "
438 << "generated kernel library must non-be null.";
439
440 for (const auto& it : exec_->primitive_map) {
441 const auto& packed_name = it.first;
442 auto packed_index = static_cast<size_t>(it.second);
443 if (packed_funcs_.size() <= packed_index) {
444 packed_funcs_.resize(packed_index + 1);
445 }
446 tvm::runtime::PackedFunc pf = lib.GetFunction(packed_name, /*query_imports=*/true);
447 ICHECK(pf != nullptr) << "Cannot find function in module: " << packed_name;
448 packed_funcs_[packed_index] = pf;
449 }
450 for (size_t i = 0; i < packed_funcs_.size(); ++i) {
451 ICHECK(packed_funcs_[i] != nullptr) << "Packed function " << i << " is not initialized";
452 }
453}

455void VirtualMachine::Init(const std::vector<Device>& physical_devices,
456 const std::vector<AllocatorType>& alloc_types) {
457 ICHECK_EQ(physical_devices.size(), alloc_types.size());
458
459 // Find a physical device to represent each virtual device the VM code requires.
460 // (Recall the VM instructions refer to devices by "device index" into this vector of
461 // virtual devices.)
462 const size_t num_virtual_devices = exec_->virtual_devices.size();
463 devices_.reserve(num_virtual_devices);
464 allocators_.reserve(num_virtual_devices);
465
466 for (size_t device_index = 0; device_index < num_virtual_devices; ++device_index) {
467 // We'll retain the legacy behaviour and just match by device type.
468 // TODO(mbs): Generalize.
469 DLDeviceType virtual_device_type = exec_->virtual_devices[device_index].device_type;
470 auto itr = std::find_if(physical_devices.begin(), physical_devices.end(),
471 [virtual_device_type](const Device& physical_device) {
472 return physical_device.device_type == virtual_device_type;
473 });
474 CHECK(itr != physical_devices.end())
475 << "Unable to find a physical device (from among the " << physical_devices.size()
476 << " given) to match the virtual device with device type " << virtual_device_type;
477 const size_t i = std::distance(physical_devices.begin(), itr);
478 devices_.push_back(*itr);
479 allocators_.push_back(MemoryManager::GetOrCreateAllocator(*itr, alloc_types[i]));
480 }
481}

489int64_t VirtualMachine::LoadScalarInt(Index r) const {
490 int64_t result = 0;
491 const auto& obj = ReadRegister(r);
492 NDArray array = Downcast<NDArray>(CopyTo(obj, GetDevice(exec_->host_device_index)));
493
494 switch (array->dtype.bits) {
495 case 1: {
496 result = reinterpret_cast<bool*>(array->data)[0];
497 break;
498 }
499 case 8: {
500 result = reinterpret_cast<int8_t*>(array->data)[0];
501 break;
502 }
503 case 16: {
504 result = reinterpret_cast<int16_t*>(array->data)[0];
505 break;
506 }
507 case 32: {
508 result = reinterpret_cast<int32_t*>(array->data)[0];
509 break;
510 }
511 case 64: {
512 result = reinterpret_cast<int64_t*>(array->data)[0];
513 break;
514 }
515 default:
516 LOG(FATAL) << "Unknown scalar int type: " << DLDataType2String(array->dtype);
517 }
518 return result;
519}

521void VirtualMachine::RunLoop() {
522 ICHECK(this->exec_);
523 ICHECK(this->code_);
524 pc_ = 0;
525 Index frame_start = frames_.size();
526 while (true) {
527 main_loop:
528 auto const& instr = code_[this->pc_];
529 VLOG(2) << "Executing(" << pc_ << "): " << instr;
530
531 switch (instr.op) {
532 case Opcode::Move: {
533 ObjectRef from_obj;
534 from_obj = ReadRegister(instr.from);
535 WriteRegister(instr.dst, from_obj);
536 pc_++;
537 goto main_loop;
538 }
539 case Opcode::Fatal: {
540 throw std::runtime_error("VM encountered fatal error");
541 }
542 case Opcode::LoadConst: {
543 bool is_not_cached = const_pool_.size() <= static_cast<size_t>(instr.const_index) ||
544 !const_pool_[instr.const_index].defined();
545 if (is_not_cached) {
546 OpStartHook(instr);
547 }
548 auto constant_obj = exec_->constants[instr.const_index];
549 // We cache the allocated object in the constant pool. To measure, the
550 // first iteration will set the pool up. The other iterations will
551 // directly reuse the allocated objects.
552 if (const_pool_.size() <= static_cast<size_t>(instr.const_index)) {
553 const_pool_.resize(instr.const_index + 1);
554 }
555
556 if (!const_pool_[instr.const_index].defined()) {
557 Device dev = GetDevice(exec_->const_device_indexes[instr.const_index]);
558 const_pool_[instr.const_index] = CopyTo(constant_obj, dev);
559 }
560 WriteRegister(instr.dst, const_pool_[instr.const_index]);
561 if (is_not_cached) {
562 OpStopHook();
563 }
564 pc_++;
565 goto main_loop;
566 }
567 case Opcode::LoadConsti: {
568 auto tensor = NDArray::Empty({1}, {kDLInt, 64, 1}, GetDevice(exec_->host_device_index));
569 reinterpret_cast<int64_t*>(tensor->data)[0] = instr.load_consti.val;
570 WriteRegister(instr.dst, tensor);
571 pc_++;
572 goto main_loop;
573 }
574 case Opcode::Invoke: {
575 std::vector<ObjectRef> args;
576 for (Index i = 0; i < instr.num_args; ++i) {
577 args.push_back(ReadRegister(instr.invoke_args_registers[i]));
578 }
579 InvokeGlobal(exec_->functions[instr.func_index], args);
580 frames_.back().caller_return_register = instr.dst;
581 goto main_loop;
582 }
583 case Opcode::InvokePacked: {
584 ICHECK_LE(instr.packed_index, packed_funcs_.size());
585 const auto& func = packed_funcs_[instr.packed_index];
586 const auto& arity = instr.arity;
587 std::vector<ObjectRef> args;
588 for (Index i = 0; i < arity; ++i) {
589 auto arg = ReadRegister(instr.packed_args[i]);
590 args.push_back(arg);
591#if TVM_LOG_DEBUG
592 if (i < arity) {
593 const bool is_input = i < arity - instr.output_size;
594 VLOG(2) << (is_input ? "input" : "placeholder") << " arg " << i << " = "
595 << RuntimeObject2String(arg, GetDevice(exec_->host_device_index),
596 /*show_contents=*/is_input);
597 }
598#endif
599 }
600
601 // We no longer need to write the registers back, we write directly
602 // through the registers mutably.
603 InvokePacked(instr.packed_index, func, arity, instr.output_size, args);
604
605#if TVM_LOG_DEBUG
606 for (Index i = arity - instr.output_size; i < arity; ++i) {
607 auto arg = ReadRegister(instr.packed_args[i]);
608 VLOG(2) << "output arg " << i << " = "
609 << RuntimeObject2String(arg, GetDevice(exec_->host_device_index));
610 }
611#endif
612
613 pc_++;
614 goto main_loop;
615 }
616 case Opcode::InvokeClosure: {
617 auto object = ReadRegister(instr.closure);
618 const auto* closure = object.as<VMClosureObj>();
619 ICHECK(closure);
620 std::vector<ObjectRef> args;
621 for (auto free_var : closure->free_vars) {
622 args.push_back(free_var);
623 }
624 for (Index i = 0; i < instr.num_closure_args; ++i) {
625 args.push_back(ReadRegister(instr.closure_args[i]));
626 }
627 InvokeGlobal(exec_->functions[closure->func_index], args);
628 frames_.back().caller_return_register = instr.dst;
629 goto main_loop;
630 }
631 case Opcode::GetField: {
632 auto object = ReadRegister(instr.object);
633 const auto& tuple = Downcast<ADT>(object);
634 auto field = tuple[instr.field_index];
635 WriteRegister(instr.dst, field);
636 pc_++;
637 goto main_loop;
638 }
639 case Opcode::GetTag: {
640 auto object = ReadRegister(instr.get_tag.object);
641 const auto& adt = Downcast<ADT>(object);
642 auto tag = adt.tag();
643 auto tag_tensor = NDArray::Empty({1}, {kDLInt, 32, 1}, GetDevice(exec_->host_device_index));
644 reinterpret_cast<int32_t*>(tag_tensor->data)[0] = tag;
645 WriteRegister(instr.dst, tag_tensor);
646 pc_++;
647 goto main_loop;
648 }
649 case Opcode::Goto: {
650 pc_ += instr.pc_offset;
651 goto main_loop;
652 }
653 case Opcode::If: {
654 int32_t test_val = LoadScalarInt(instr.if_op.test);
655 int32_t target_val = LoadScalarInt(instr.if_op.target);
656
657 if (test_val == target_val) {
658 ICHECK_NE(instr.if_op.true_offset, 0);
659 pc_ += instr.if_op.true_offset;
660 } else {
661 ICHECK_NE(instr.if_op.false_offset, 0);
662 pc_ += instr.if_op.false_offset;
663 }
664
665 goto main_loop;
666 }
667 case Opcode::AllocTensor: {
668 OpStartHook(instr);
669 auto shape = std::vector<int64_t>(instr.alloc_tensor.ndim);
670
671 for (uint32_t i = 0; i < instr.alloc_tensor.ndim; ++i) {
672 shape[i] = instr.alloc_tensor.shape[i];
673 }
674
675 auto storage_obj = ReadRegister(instr.alloc_tensor.storage);
676 auto offset = LoadScalarInt(instr.alloc_tensor.offset);
677 auto storage = Downcast<Storage>(storage_obj);
678 auto obj = storage->AllocNDArray(offset, shape, instr.alloc_tensor.dtype);
679 VLOG(2) << "allocated "
680 << RuntimeObject2String(obj, GetDevice(exec_->host_device_index),
681 /*show_contents=*/false);
682
683 WriteRegister(instr.dst, obj);
684 OpStopHook();
685 pc_++;
686 goto main_loop;
687 }
688 case Opcode::AllocTensorReg: {
689 OpStartHook(instr);
690 Device cpu_dev = GetDevice(exec_->host_device_index);
691 auto shape_obj = ReadRegister(instr.alloc_tensor_reg.shape_register);
692 NDArray shape_tensor = Downcast<NDArray>(CopyTo(shape_obj, cpu_dev));
693 auto shape = ToShape(shape_tensor);
694 auto storage_obj = ReadRegister(instr.alloc_tensor_reg.storage);
695 auto storage = Downcast<Storage>(storage_obj);
696 auto offset = LoadScalarInt(instr.alloc_tensor.offset);
697 auto obj = storage->AllocNDArray(offset, shape, instr.alloc_tensor_reg.dtype);
698 VLOG(2) << "allocated "
699 << RuntimeObject2String(obj, GetDevice(exec_->host_device_index),
700 /*show_contents=*/false);
701
702 WriteRegister(instr.dst, obj);
703 OpStopHook();
704 pc_++;
705 goto main_loop;
706 }
707 case Opcode::AllocADT: {
708 std::vector<ObjectRef> fields;
709 for (Index i = 0; i < instr.num_fields; ++i) {
710 fields.push_back(ReadRegister(instr.datatype_fields[i]));
711 }
712 ObjectRef obj = ADT(instr.constructor_tag, fields);
713 WriteRegister(instr.dst, obj);
714 pc_++;
715 goto main_loop;
716 }
717 case Opcode::AllocClosure: {
718 std::vector<ObjectRef> free_vars;
719 for (Index i = 0; i < instr.num_freevar; i++) {
720 free_vars.push_back(ReadRegister(instr.free_vars[i]));
721 }
722 WriteRegister(instr.dst, VMClosure(instr.func_index, free_vars));
723 pc_++;
724 goto main_loop;
725 }
726 case Opcode::AllocStorage: {
727 OpStartHook(instr);
728 auto size = LoadScalarInt(instr.alloc_storage.allocation_size);
729 auto alignment = instr.alloc_storage.alignment;
730
731 auto storage_obj = SimpleObjAllocator().make_object<StorageObj>();
732 Allocator* allocator = GetAllocator(instr.alloc_storage.device_index);
733 ICHECK(allocator) << "Did you forget to init the VirtualMachine with devices?";
734 VLOG(2) << "allocating with allocation_size=" << size << ", alignment=" << alignment
735 << ", dtype_hint=" << DLDataType2String(instr.alloc_storage.dtype_hint)
736 << ", device_index=" << instr.alloc_storage.device_index;
737
738 storage_obj->buffer = allocator->Alloc(size, alignment, instr.alloc_storage.dtype_hint);
739 Storage storage(storage_obj);
740 WriteRegister(instr.dst, storage);
741 OpStopHook();
742 pc_++;
743 goto main_loop;
744 }
745 case Opcode::ShapeOf: {
746 auto input = ReadRegister(instr.shape_of.tensor);
747 NDArray input_array = Downcast<NDArray>(input);
748 int ndim = input_array->ndim;
749 auto out_tensor =
750 NDArray::Empty({ndim}, {kDLInt, 64, 1}, GetDevice(exec_->host_device_index));
751 for (int i = 0; i < ndim; ++i) {
752 reinterpret_cast<int64_t*>(out_tensor->data)[i] = input_array->shape[i];
753 }
754 VLOG(2) << "shape = "
755 << RuntimeObject2String(out_tensor, GetDevice(exec_->host_device_index));
756 WriteRegister(instr.dst, out_tensor);
757 pc_++;
758 goto main_loop;
759 }
760 case Opcode::Ret: {
761 // If we have hit the point from which we started
762 // running, we should return to the caller breaking
763 // the dispatch loop.
764 return_register_ = ReadRegister(instr.result);
765 auto caller_return_register = frames_.back().caller_return_register;
766
767 if (PopFrame() == frame_start) {
768 return;
769 // Otherwise we are just returning from a local call.
770 } else {
771 WriteRegister(caller_return_register, return_register_);
772 goto main_loop;
773 }
774 }
775 case Opcode::ReshapeTensor: {
776 OpStartHook(instr);
777 Device cpu_dev = GetDevice(exec_->host_device_index);
778 auto tensor_obj = ReadRegister(instr.reshape_tensor.tensor);
779 NDArray tensor_arr = Downcast<NDArray>(tensor_obj);
780 // Read the shape from shape tensor
781 auto shape_obj = ReadRegister(instr.reshape_tensor.newshape);
782 NDArray shape_tensor = Downcast<NDArray>(CopyTo(shape_obj, cpu_dev));
783 const DLTensor* dl_tensor = shape_tensor.operator->();
784 ICHECK_EQ(dl_tensor->dtype.code, 0u);
785 ICHECK_EQ(dl_tensor->dtype.bits, 64u);
786 int64_t* dims = reinterpret_cast<int64_t*>(dl_tensor->data);
787 int64_t ndim = shape_tensor->shape[0];
788 std::vector<int64_t> shape(dims, dims + ndim);
789 // Reshape the input tensor
790 auto out_tensor = tensor_arr.CreateView(shape, tensor_arr->dtype);
791 VLOG(2) << "reshaped "
792 << RuntimeObject2String(tensor_obj, GetDevice(exec_->host_device_index)) << " to "
793 << RuntimeObject2String(out_tensor, GetDevice(exec_->host_device_index));
794 WriteRegister(instr.dst, out_tensor);
795 OpStopHook();
796 pc_++;
797 goto main_loop;
798 }
799 case Opcode::DeviceCopy: {
800 OpStartHook(instr);
801 auto tensor_src = ReadRegister(instr.device_copy.src);
802 NDArray src_data = Downcast<NDArray>(tensor_src);
803 Device actual_src_dev = src_data->device;
804 Device inst_src_dev = GetDevice(instr.device_copy.src_device_index);
805 ICHECK_EQ(actual_src_dev.device_type, inst_src_dev.device_type);
806 ICHECK_EQ(actual_src_dev.device_id, inst_src_dev.device_id);
807 Device dst_dev = GetDevice(instr.device_copy.dst_device_index);
808
809 NDArray dst_data = src_data.CopyTo(dst_dev);
810 WriteRegister(instr.dst, dst_data);
811 OpStopHook();
812 pc_++;
813 goto main_loop;
814 }
815 case Opcode::KillRegister: {
816 OpStartHook(instr);
817 WriteRegister(instr.dst, ObjectRef());
818 OpStopHook();
819 pc_++;
820 goto main_loop;
821 }
822 default:
823 LOG(FATAL) << "Unknown instruction opcode: " << int(instr.op);
824 }
825 }
826}

828runtime::Module CreateVirtualMachine(Executable* exec) {
829 auto vm = make_object<VirtualMachine>();
830 vm->LoadExecutable(GetObjectPtr<Executable>(exec));
831 return runtime::Module(vm);
832}
