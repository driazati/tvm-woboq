69inline ObjectRef CopyTo(ObjectRef src, const DLDevice& dev) {
70 if (src->IsInstance<NDArray::ContainerType>()) {
71 auto nd_array = Downcast<NDArray>(src);
72 // TODO(mbs): Should respect device id also.
73 // TODO(vvchernov): it still does not work for different device id
74 // due to simple implementation of Get() and AllocDataSpace() methods
75 // see tvm/src/runtime/c_runtime_api.cc: L139
76 // tvm/src/runtime/cpu_device_api.cc: L47
77 if (nd_array->device.device_type != dev.device_type ||
78 nd_array->device.device_id != dev.device_id) {
79 VLOG(2) << "copying from " << nd_array->device.device_type << "["
80 << nd_array->device.device_id << "] to " << dev.device_type << "[" << dev.device_id
81 << "]";
82 return nd_array.CopyTo(dev);
83 }
84 return src;
85 } else {
86 ICHECK(src->IsInstance<ADTObj>())
87 << "VM data must be NDArray or a list of NDArray, but received: " << src->_type_key;
88 std::vector<ObjectRef> ret;
89 ADT adt = Downcast<ADT>(src);
90 for (size_t i = 0; i < adt.size(); i++) {
91 ret.push_back(CopyTo(adt[i], dev));
92 }
93 return ADT(adt->tag, ret.begin(), ret.end());
94 }

97std::vector<int64_t> ToShape(NDArray shape_tensor) {
98 std::vector<int64_t> shape;
99 auto rank = shape_tensor.Shape().size();
100 auto dtype = shape_tensor.DataType();
101
102 // For 0-rank shapes we need to allocate a single scalar.
103 if (rank == 0) {
104 return shape;
105 }
106
107 // Otherwise we should be rank-1, and we will extract the number of dimensions
108 // for the output vector.
109 ICHECK_EQ(rank, 1U) << "shape tensor should be a k-length vector, found " << rank;
110 int64_t ndim = shape_tensor.Shape().at(0);
111 shape.resize(ndim);
112
113 const DLTensor* dl_tensor = shape_tensor.operator->();
114 if (dtype.is_int() && dtype.bits() == 32 && dtype.lanes() == 1) {
115 int32_t* dims = reinterpret_cast<int32_t*>(dl_tensor->data);
116 shape.assign(dims, dims + ndim);
117 } else if (dtype.is_int() && dtype.bits() == 64 && dtype.lanes() == 1) {
118 int64_t* dims = reinterpret_cast<int64_t*>(dl_tensor->data);
119 shape.assign(dims, dims + ndim);
120 } else {
121 LOG(FATAL) << "invalid shape tensor datatype: " << dtype;
122 }
123
124 return shape;
125}

130PackedFunc VirtualMachine::GetFunction(const std::string& name,
131 const ObjectPtr<Object>& sptr_to_self) {
132 if (name == "invoke") {
133 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
134 ICHECK(exec_) << "The executable is not created yet.";
135
136 std::string func_name = args[0];
137 auto git = exec_->global_map.find(func_name);
138 ICHECK(git != exec_->global_map.end())
139 << "Cannot find function " << func_name << " in the executable";
140 auto func = exec_->functions[git->second];
141 if (func.params.empty()) {
142 *rv = Invoke(func, {});
143 } else {
144 auto it = inputs_.find(func_name);
145 ICHECK(it != inputs_.end()) << "Input has not been set for function " << func_name;
146 const std::vector<ObjectRef>& input_args = it->second;
147 if (set_outputs_enabled_.count(func_name) && set_outputs_enabled_[func_name]) {
148 ICHECK(outputs_.count(func_name))
149 << "Outputs have not been set for function " << func_name;
150 *rv = Invoke(func, input_args, outputs_[func_name]);
151 outputs_[func_name].clear();
152 set_outputs_enabled_[func_name] = false;
153 } else {
154 *rv = Invoke(func, input_args);
155 }
156 }
157 });
158 } else if (name == "invoke_stateful") {
159 // TODO(tkonolige, jroesch, tqchen): invoke_stateful and get_output are
160 // stop-gap measure to allow using vm over a remote connection.
161 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
162 PackedFunc invoke = GetFunction("invoke", sptr_to_self);
163 TVMRetValue rv_;
164 invoke.CallPacked(args, &rv_);
165 });
166 } else if (name == "invoke_return_to_device") {
167 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
168 Device host{static_cast<DLDeviceType>(args[1].operator int()), args[2].operator int()};
169
170 SetInput(args[0].operator std::string(), args, 3);
171 PackedFunc invoke = GetFunction("invoke", sptr_to_self);
172 TVMRetValue rv_;
173 invoke.CallPacked(args, &rv_); // Invoke only uses the first arg, so the rest of the args
174 // should not cause an issue
175 if (rv_.type_code() == kTVMObjectHandle) {
176 ADT adt = Downcast<ADT>(rv_.operator ObjectRef());
177 std::vector<ObjectRef> transfered;
178 for (size_t i = 0; i < adt.size(); i++) {
179 transfered.push_back(CopyTo(adt[i], host));
180 }
181 *rv = ADT(adt.tag(), transfered);
182 } else {
183 *rv = CopyTo(rv_, host);
184 }
185 });
186 } else if (name == "get_output") {
187 return TypedPackedFunc<NDArray(int64_t)>([this](int64_t index) {
188 if (this->return_register_.as<ADTObj>()) {
189 return Downcast<NDArray>(Downcast<ADT>(this->return_register_)[index]);
190 } else {
191 CHECK_EQ(index, 0) << "VM output contains only one item, but you are trying to get the "
192 << index << "th.";
193 return Downcast<NDArray>(this->return_register_);
194 }
195 });
196 } else if (name == "get_num_outputs") {
197 return TypedPackedFunc<int64_t(void)>([this]() -> int64_t {
198 // single output is an NDArray not an ADT
199 if (this->return_register_.as<ADTObj>()) {
200 return Downcast<ADT>(this->return_register_).size();
201 } else {
202 return 1;
203 }
204 });
205 } else if (name == "get_input_index") {
206 return TypedPackedFunc<int64_t(std::string, std::string)>(
207 [this](std::string input_name, std::string func_name) {
208 return GetInputIndexFromVMFunction(func_name, input_name);
209 });
210 } else if (name == "init") {
211 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
212 ICHECK_EQ(args.size() % 3, 0);
213 std::vector<Device> devices;
214 std::vector<AllocatorType> alloc_types;
215 for (int i = 0; i < args.size() / 3; ++i) {
216 Device dev;
217 int device_type = args[i * 3];
218 dev.device_type = DLDeviceType(device_type);
219 dev.device_id = args[i * 3 + 1];
220 int type = args[i * 3 + 2];
221 devices.push_back(dev);
222 alloc_types.push_back(AllocatorType(type));
223 }
224 this->Init(devices, alloc_types);
225 });
226 } else if (name == "set_input") {
227 return PackedFunc(
228 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { SetInput(args[0], args, 1); });
229 } else if (name == "set_one_input") {
230 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
231 ICHECK_EQ(args.size(), 3) << "The expected number of arguments is 3 "
232 << "(func_name, index or name, tensor)";
233 SetOneInput(args[0], args[1], args[2]);
234 });
235 } else if (name == "set_outputs") {
236 return PackedFunc(
237 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { SetOutputs(args[0], args); });
238 } else if (name == "load_late_bound_consts") {
239 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
240 CHECK_EQ(args.size(), 1);
241 std::string path = args[0];
242 exec_->LoadLateBoundConstantsFromFile(path);
243 });
244 } else {
245 LOG(FATAL) << "Unknown packed function: " << name;
246 return PackedFunc([sptr_to_self, name](TVMArgs args, TVMRetValue* rv) {});
247 }
248}

250void VirtualMachine::SetInput(std::string func_name, TVMArgs args, int offset) {
251 const auto& vm_func = CheckAndGetVMFunction(func_name);
252 size_t params_num = vm_func.params.size();
253 ICHECK_EQ(args.size() - offset, params_num)
254 << "The number of provided parameters doesn't match the number of arguments";
255 std::vector<ObjectRef> func_args(params_num);
256 for (int i = offset; i < args.size(); ++i) {
257 int index = i - offset;
258 Device dev = GetDevice(vm_func.param_device_indexes[index]);
259 SetInputTensorWithIndex(func_args, args[i], index, dev);
260 }
261 inputs_.erase(func_name);
262 inputs_.emplace(func_name, func_args);
263}

265void VirtualMachine::SetOneInput(std::string func_name, const TVMArgValue& tag,
266 const TVMArgValue& tensor) {
267 const auto& vm_func = CheckAndGetVMFunction(func_name);
268 size_t params_num = vm_func.params.size();
269
270 int inp_index = 0;
271 if (tag.type_code() == kTVMArgInt) {
272 inp_index = tag;
273 } else if (tag.type_code() == kTVMStr) {
274 inp_index = static_cast<int>(GetInputIndexFromName(vm_func.params, tag));
275 } else {
276 LOG(FATAL) << "The type of input tensor tag (" << tag.type_code()
277 << ") doesn't match integer or string";
278 }
279 ICHECK_LT(inp_index, params_num);
280
281 CreateInputsOrCheckSize(func_name, params_num);
282 Device dev = GetDevice(vm_func.param_device_indexes[inp_index]);
283 SetInputTensorWithIndex(inputs_[func_name], tensor, inp_index, dev);
284}

286void VirtualMachine::SetOutputs(std::string func_name, TVMArgs args) {
287 set_outputs_enabled_[func_name] = true;
288 size_t outputs_size = args.size();
289 // First args is func_name
290 ICHECK_GT(outputs_size, 1) << "There is no output arguments set";
291
292 std::vector<ObjectRef> func_args(outputs_size - 1);
293 for (size_t i = 1; i < outputs_size; ++i) {
294 // TODO(vvchernov): device?
295 func_args[i - 1] = TensorFromTVMArgValueToObjectRef(args[i]);
296 }
297 outputs_.erase(func_name);
298 outputs_.emplace(func_name, func_args);
299}

313void VirtualMachine::SetOutputTensorsToRegister(const std::string& func_name,
314 const std::vector<ObjectRef>& outputs) {
315 size_t size = outputs.size();
316
317 if (output_tensor_reg_indices_[func_name].empty()) {
318 output_tensor_reg_indices_[func_name] = GetOutputTensorRegIndices();
319 }
320 auto& reg_indices = output_tensor_reg_indices_[func_name];
321 ICHECK_EQ(reg_indices.size(), size)
322 << "Number of outside output tensors should be equal to model outputs number";
323 size_t i = 0;
324 for (auto it = reg_indices.begin(); it != reg_indices.end(); ++it, ++i) {
325 WriteRegister(*it, outputs[i]);
326 }
327}

329ObjectRef VirtualMachine::TensorFromTVMArgValueToObjectRef(const TVMArgValue& output_tensor) const {
330 if (output_tensor.type_code() == kTVMDLTensorHandle) {
331 DLTensor* dl_tensor = output_tensor;
332 return NDArray::FromExternalDLTensor(*dl_tensor);
333 } else if (output_tensor.type_code() == kTVMNDArrayHandle) {
334 return output_tensor.AsObjectRef<tvm::runtime::NDArray>();
335 } else {
336 LOG(FATAL) << "It supports tensor of DLTensor or NDArray type only! Given type is "
337 << output_tensor.type_code();
338 }
339 return ObjectRef();
340}

375void VirtualMachine::SetInputTensorWithIndex(std::vector<ObjectRef>& tensors,
376 const TVMArgValue& inp_tensor, int index, Device dev) {
377 if (inp_tensor.type_code() == kTVMDLTensorHandle) {
378 if (NDArray::AbilityOfZeroCopyForDLTensor(inp_tensor, dev)) {
379 tensors[index] = NDArray::FromExternalDLTensor(*inp_tensor);
380 } else {
381 tensors[index] = NDArray::NewFromDLTensor(inp_tensor, dev);
382 }
383 } else {
384 tensors[index] = CopyTo(inp_tensor, dev);
385 }

414void VirtualMachine::InvokeGlobal(const VMFunction& func, const std::vector<ObjectRef>& args) {
415 VLOG(2) << "Invoking global " << func.name << " with " << args.size() << " args";
416
417 PushFrame(func.params.size(), this->pc_ + 1, func);
418 for (size_t i = 0; i < args.size(); ++i) {
419 WriteRegister(i, args[i]);
420 VLOG(2) << "arg " << i << " = "
421 << RuntimeObject2String(args[i], GetDevice(exec_->host_device_index));
422 }
423
424 code_ = func.instructions.data();
425 pc_ = 0;
426}

451void VirtualMachine::InvokePacked(Index packed_index, const PackedFunc& func, Index arg_count,
452 Index output_size, const std::vector<ObjectRef>& args) {
453 size_t arity = 0;
454 for (Index i = 0; i < arg_count; i++) {
455 if (const auto* obj = args[i].as<ADTObj>()) {
456 arity += obj->size;
457 } else {
458 ++arity;
459 }
460 }
461
462 std::vector<TVMValue> values(arity);
463 std::vector<int> codes(arity);
464 runtime::TVMArgsSetter setter(values.data(), codes.data());
465 int idx = 0;
466 bool is_empty_output = false;
467 for (Index i = 0; i < arg_count; i++) {
468 if (const auto* dt_cell = args[i].as<ADTObj>()) {
469 for (size_t fi = 0; fi < dt_cell->size; ++fi) {
470 auto obj = (*dt_cell)[fi];
471 auto nd_array = Downcast<NDArray>(obj);
472 setter(idx++, nd_array);
473 }
474 } else {
475 auto nd_array = Downcast<NDArray>(args[i]);
476 // We can safely skip CallPacked if there is only one
477 // output and it is empty.
478 if (i == arg_count - 1 && output_size == 1) {
479 for (const auto& dim : nd_array.Shape()) {
480 if (!dim) {
481 is_empty_output = true;
482 break;
483 }
484 }
485 }
486 setter(idx++, nd_array);
487 }
488 }
489
490 if (!is_empty_output) {
491 TVMRetValue rv;
492 func.CallPacked(TVMArgs(values.data(), codes.data(), arity), &rv);
493 }
494}

496void VirtualMachine::LoadExecutable(const ObjectPtr<Executable>& exec) {
497 ICHECK(exec) << "The executable is not created yet.";
498 ICHECK(exec->late_bound_constant_names.empty())
499 << "Need to load late-bound-constants before creating VM";
500 exec_ = exec;
501
502 runtime::Module lib = exec_->GetLib();
503
504 ICHECK(exec_->primitive_map.empty() || lib.operator->())
505 << "If the executable has declared primitive functions, the "
506 << "generated kernel library must non-be null.";
507
508 for (const auto& it : exec_->primitive_map) {
509 const auto& packed_name = it.first;
510 auto packed_index = static_cast<size_t>(it.second);
511 if (packed_funcs_.size() <= packed_index) {
512 packed_funcs_.resize(packed_index + 1);
513 }
514 tvm::runtime::PackedFunc pf = lib.GetFunction(packed_name, /*query_imports=*/true);
515 ICHECK(pf != nullptr) << "Cannot find function in module: " << packed_name;
516 packed_funcs_[packed_index] = pf;
517 }
518 for (size_t i = 0; i < packed_funcs_.size(); ++i) {
519 ICHECK(packed_funcs_[i] != nullptr) << "Packed function " << i << " is not initialized";
520 }
521}

523void VirtualMachine::Init(const std::vector<Device>& physical_devices,
524 const std::vector<AllocatorType>& alloc_types) {
525 ICHECK_EQ(physical_devices.size(), alloc_types.size());
526
527 // Find a physical device to represent each virtual device the VM code requires.
528 // (Recall the VM instructions refer to devices by "device index" into this vector of
529 // virtual devices.)
530 const size_t num_virtual_devices = exec_->virtual_devices.size();
531 devices_.reserve(num_virtual_devices);
532 allocators_.reserve(num_virtual_devices);
533
534 for (size_t device_index = 0; device_index < num_virtual_devices; ++device_index) {
535 // We'll retain the legacy behaviour and just match by device type.
536 // TODO(mbs): Generalize.
537 DLDeviceType virtual_device_type = exec_->virtual_devices[device_index].device_type;
538 auto itr = std::find_if(physical_devices.begin(), physical_devices.end(),
539 [virtual_device_type](const Device& physical_device) {
540 return physical_device.device_type == virtual_device_type;
541 });
542 CHECK(itr != physical_devices.end())
543 << "Unable to find a physical device (from among the " << physical_devices.size()
544 << " given) to match the virtual device with device type " << virtual_device_type;
545 const size_t i = std::distance(physical_devices.begin(), itr);
546 devices_.push_back(*itr);
547 allocators_.push_back(MemoryManager::GetOrCreateAllocator(*itr, alloc_types[i]));
548 }
549}

557int64_t VirtualMachine::LoadScalarInt(Index r) const {
558 int64_t result = 0;
559 const auto& obj = ReadRegister(r);
560 NDArray array = Downcast<NDArray>(CopyTo(obj, GetDevice(exec_->host_device_index)));
561
562 switch (array->dtype.bits) {
563 case 1: {
564 result = reinterpret_cast<bool*>(array->data)[0];
565 break;
566 }
567 case 8: {
568 result = reinterpret_cast<int8_t*>(array->data)[0];
569 break;
570 }
571 case 16: {
572 result = reinterpret_cast<int16_t*>(array->data)[0];
573 break;
574 }
575 case 32: {
576 result = reinterpret_cast<int32_t*>(array->data)[0];
577 break;
578 }
579 case 64: {
580 result = reinterpret_cast<int64_t*>(array->data)[0];
581 break;
582 }
583 default:
584 LOG(FATAL) << "Unknown scalar int type: " << DLDataType2String(array->dtype);
585 }
586 return result;
587}

598void VirtualMachine::CalculatePreResultOpIndex(Index res_index) {
599 if (preresult_op_index_ == -1) {
600 preresult_op_index_ = 0;
601 while (code_[preresult_op_index_].dst != res_index) {
602 ++preresult_op_index_;
603 }
604 }
605}

607std::vector<Index> VirtualMachine::GetOutputTensorRegIndices() {
608 std::vector<Index> reg_indices;
609 Index res_index = GetResultRegisterIndex();
610 CalculatePreResultOpIndex(res_index);
611 auto& preres_instr = code_[preresult_op_index_];
612 auto op_code = preres_instr.op;
613 if (op_code == Opcode::AllocTensor) {
614 reg_indices.emplace_back(res_index);
615 } else if (op_code == Opcode::AllocADT) {
616 for (Index i = 0; i < preres_instr.num_fields; ++i) {
617 reg_indices.push_back(preres_instr.datatype_fields[i]);
618 }
619 } else if (op_code == Opcode::ReshapeTensor) {
620 reg_indices.push_back(preres_instr.reshape_tensor.tensor);
621 } else {
622 LOG(FATAL) << "Operation " << size_t(op_code) << " is not supported for set_outputs method";
623 }
624 return reg_indices;
625}

627void VirtualMachine::RunLoop(const std::vector<Index>& output_tensor_reg_indices) {
628 ICHECK(this->exec_);
629 ICHECK(this->code_);
630 pc_ = 0;
631 Index frame_start = frames_.size();
632 while (true) {
633 main_loop:
634 auto const& instr = code_[this->pc_];
635 VLOG(2) << "Executing(" << pc_ << "): " << instr;
636
637 switch (instr.op) {
638 case Opcode::Move: {
639 ObjectRef from_obj;
640 from_obj = ReadRegister(instr.from);
641 WriteRegister(instr.dst, from_obj);
642 pc_++;
643 goto main_loop;
644 }
645 case Opcode::Fatal: {
646 throw std::runtime_error("VM encountered fatal error");
647 }
648 case Opcode::LoadConst: {
649 bool is_not_cached = const_pool_.size() <= static_cast<size_t>(instr.const_index) ||
650 !const_pool_[instr.const_index].defined();
651 if (is_not_cached) {
652 OpStartHook(instr);
653 }
654 auto constant_obj = exec_->constants[instr.const_index];
655 // We cache the allocated object in the constant pool. To measure, the
656 // first iteration will set the pool up. The other iterations will
657 // directly reuse the allocated objects.
658 if (const_pool_.size() <= static_cast<size_t>(instr.const_index)) {
659 const_pool_.resize(instr.const_index + 1);
660 }
661
662 if (!const_pool_[instr.const_index].defined()) {
663 Device dev = GetDevice(exec_->const_device_indexes[instr.const_index]);
664 const_pool_[instr.const_index] = CopyTo(constant_obj, dev);
665 }
666 WriteRegister(instr.dst, const_pool_[instr.const_index]);
667 if (is_not_cached) {
668 OpStopHook();
669 }
670 pc_++;
671 goto main_loop;
672 }
673 case Opcode::LoadConsti: {
674 auto tensor = NDArray::Empty({1}, {kDLInt, 64, 1}, GetDevice(exec_->host_device_index));
675 reinterpret_cast<int64_t*>(tensor->data)[0] = instr.load_consti.val;
676 WriteRegister(instr.dst, tensor);
677 pc_++;
678 goto main_loop;
679 }
680 case Opcode::Invoke: {
681 std::vector<ObjectRef> args;
682 for (Index i = 0; i < instr.num_args; ++i) {
683 args.push_back(ReadRegister(instr.invoke_args_registers[i]));
684 }
685 InvokeGlobal(exec_->functions[instr.func_index], args);
686 frames_.back().caller_return_register = instr.dst;
687 goto main_loop;
688 }
689 case Opcode::InvokePacked: {
690 ICHECK_LE(instr.packed_index, packed_funcs_.size());
691 const auto& func = packed_funcs_[instr.packed_index];
692 const auto& arity = instr.arity;
693 std::vector<ObjectRef> args;
694 for (Index i = 0; i < arity; ++i) {
695 auto arg = ReadRegister(instr.packed_args[i]);
696 args.push_back(arg);
697#if TVM_LOG_DEBUG
698 if (i < arity) {
699 const bool is_input = i < arity - instr.output_size;
700 VLOG(2) << (is_input ? "input" : "placeholder") << " arg " << i << " = "
701 << RuntimeObject2String(arg, GetDevice(exec_->host_device_index),
702 /*show_contents=*/is_input);
703 }
704#endif
705 }
706
707 // We no longer need to write the registers back, we write directly
708 // through the registers mutably.
709 InvokePacked(instr.packed_index, func, arity, instr.output_size, args);
710
711#if TVM_LOG_DEBUG
712 for (Index i = arity - instr.output_size; i < arity; ++i) {
713 auto arg = ReadRegister(instr.packed_args[i]);
714 VLOG(2) << "output arg " << i << " = "
715 << RuntimeObject2String(arg, GetDevice(exec_->host_device_index));
716 }
717#endif
718
719 pc_++;
720 goto main_loop;
721 }
722 case Opcode::InvokeClosure: {
723 auto object = ReadRegister(instr.closure);
724 const auto* closure = object.as<VMClosureObj>();
725 ICHECK(closure);
726 std::vector<ObjectRef> args;
727 for (auto free_var : closure->free_vars) {
728 args.push_back(free_var);
729 }
730 for (Index i = 0; i < instr.num_closure_args; ++i) {
731 args.push_back(ReadRegister(instr.closure_args[i]));
732 }
733 InvokeGlobal(exec_->functions[closure->func_index], args);
734 frames_.back().caller_return_register = instr.dst;
735 goto main_loop;
736 }
737 case Opcode::GetField: {
738 auto object = ReadRegister(instr.object);
739 const auto& tuple = Downcast<ADT>(object);
740 auto field = tuple[instr.field_index];
741 WriteRegister(instr.dst, field);
742 pc_++;
743 goto main_loop;
744 }
745 case Opcode::GetTag: {
746 auto object = ReadRegister(instr.get_tag.object);
747 const auto& adt = Downcast<ADT>(object);
748 auto tag = adt.tag();
749 auto tag_tensor = NDArray::Empty({1}, {kDLInt, 32, 1}, GetDevice(exec_->host_device_index));
750 reinterpret_cast<int32_t*>(tag_tensor->data)[0] = tag;
751 WriteRegister(instr.dst, tag_tensor);
752 pc_++;
753 goto main_loop;
754 }
755 case Opcode::Goto: {
756 pc_ += instr.pc_offset;
757 goto main_loop;
758 }
759 case Opcode::If: {
760 int32_t test_val = LoadScalarInt(instr.if_op.test);
761 int32_t target_val = LoadScalarInt(instr.if_op.target);
762
763 if (test_val == target_val) {
764 ICHECK_NE(instr.if_op.true_offset, 0);
765 pc_ += instr.if_op.true_offset;
766 } else {
767 ICHECK_NE(instr.if_op.false_offset, 0);
768 pc_ += instr.if_op.false_offset;
769 }
770
771 goto main_loop;
772 }
773 case Opcode::AllocTensor: {
774 OpStartHook(instr);
775 if (!output_tensor_reg_indices.empty() && FindIndex(output_tensor_reg_indices, instr.dst)) {
776 WriteAllocatedTensorFromOutside(instr);
777 } else {
778 WriteAllocatedTensor(instr);
779 }
780 OpStopHook();
781 pc_++;
782 goto main_loop;
783 }
784 case Opcode::AllocTensorReg: {
785 OpStartHook(instr);
786 Device cpu_dev = GetDevice(exec_->host_device_index);
787 auto shape_obj = ReadRegister(instr.alloc_tensor_reg.shape_register);
788 NDArray shape_tensor = Downcast<NDArray>(CopyTo(shape_obj, cpu_dev));
789 auto shape = ToShape(shape_tensor);
790 auto storage_obj = ReadRegister(instr.alloc_tensor_reg.storage);
791 auto storage = Downcast<Storage>(storage_obj);
792 auto offset = LoadScalarInt(instr.alloc_tensor.offset);
793 auto obj = storage->AllocNDArray(offset, shape, instr.alloc_tensor_reg.dtype);
794 VLOG(2) << "allocated "
795 << RuntimeObject2String(obj, GetDevice(exec_->host_device_index),
796 /*show_contents=*/false);
797
798 WriteRegister(instr.dst, obj);
799 OpStopHook();
800 pc_++;
801 goto main_loop;
802 }
803 case Opcode::AllocADT: {
804 std::vector<ObjectRef> fields;
805 for (Index i = 0; i < instr.num_fields; ++i) {
806 fields.push_back(ReadRegister(instr.datatype_fields[i]));
807 }
808 ObjectRef obj = ADT(instr.constructor_tag, fields);
809 WriteRegister(instr.dst, obj);
810 pc_++;
811 goto main_loop;
812 }
813 case Opcode::AllocClosure: {
814 std::vector<ObjectRef> free_vars;
815 for (Index i = 0; i < instr.num_freevar; i++) {
816 free_vars.push_back(ReadRegister(instr.free_vars[i]));
817 }
818 WriteRegister(instr.dst, VMClosure(instr.func_index, free_vars));
819 pc_++;
820 goto main_loop;
821 }
822 case Opcode::AllocStorage: {
823 OpStartHook(instr);
824 auto size = LoadScalarInt(instr.alloc_storage.allocation_size);
825 auto alignment = instr.alloc_storage.alignment;
826
827 auto storage_obj = SimpleObjAllocator().make_object<StorageObj>();
828 Allocator* allocator = GetAllocator(instr.alloc_storage.device_index);
829 ICHECK(allocator) << "Did you forget to init the VirtualMachine with devices?";
830 VLOG(2) << "allocating with allocation_size=" << size << ", alignment=" << alignment
831 << ", dtype_hint=" << DLDataType2String(instr.alloc_storage.dtype_hint)
832 << ", device_index=" << instr.alloc_storage.device_index;
833
834 storage_obj->buffer = allocator->Alloc(size, alignment, instr.alloc_storage.dtype_hint);
835 Storage storage(storage_obj);
836 WriteRegister(instr.dst, storage);
837 OpStopHook();
838 pc_++;
839 goto main_loop;
840 }
841 case Opcode::ShapeOf: {
842 auto input = ReadRegister(instr.shape_of.tensor);
843 NDArray input_array = Downcast<NDArray>(input);
844 int ndim = input_array->ndim;
845 auto out_tensor =
846 NDArray::Empty({ndim}, {kDLInt, 64, 1}, GetDevice(exec_->host_device_index));
847 for (int i = 0; i < ndim; ++i) {
848 reinterpret_cast<int64_t*>(out_tensor->data)[i] = input_array->shape[i];
849 }
850 VLOG(2) << "shape = "
851 << RuntimeObject2String(out_tensor, GetDevice(exec_->host_device_index));
852 WriteRegister(instr.dst, out_tensor);
853 pc_++;
854 goto main_loop;
855 }
856 case Opcode::Ret: {
857 // If we have hit the point from which we started
858 // running, we should return to the caller breaking
859 // the dispatch loop.
860 return_register_ = ReadRegister(instr.result);
861 auto caller_return_register = frames_.back().caller_return_register;
862
863 if (PopFrame() == frame_start) {
864 return;
865 // Otherwise we are just returning from a local call.
866 } else {
867 WriteRegister(caller_return_register, return_register_);
868 goto main_loop;
869 }
870 }
871 case Opcode::ReshapeTensor: {
872 OpStartHook(instr);
873 Device cpu_dev = GetDevice(exec_->host_device_index);
874 auto tensor_obj = ReadRegister(instr.reshape_tensor.tensor);
875 NDArray tensor_arr = Downcast<NDArray>(tensor_obj);
876 // Read the shape from shape tensor
877 auto shape_obj = ReadRegister(instr.reshape_tensor.newshape);
878 NDArray shape_tensor = Downcast<NDArray>(CopyTo(shape_obj, cpu_dev));
879 const DLTensor* dl_tensor = shape_tensor.operator->();
880 ICHECK_EQ(dl_tensor->dtype.code, 0u);
881 ICHECK_EQ(dl_tensor->dtype.bits, 64u);
882 int64_t* dims = reinterpret_cast<int64_t*>(dl_tensor->data);
883 int64_t ndim = shape_tensor->shape[0];
884 std::vector<int64_t> shape(dims, dims + ndim);
885 // Reshape the input tensor
886 auto out_tensor = tensor_arr.CreateView(shape, tensor_arr->dtype);
887 VLOG(2) << "reshaped "
888 << RuntimeObject2String(tensor_obj, GetDevice(exec_->host_device_index)) << " to "
889 << RuntimeObject2String(out_tensor, GetDevice(exec_->host_device_index));
890 WriteRegister(instr.dst, out_tensor);
891 OpStopHook();
892 pc_++;
893 goto main_loop;
894 }
895 case Opcode::DeviceCopy: {
896 OpStartHook(instr);
897 auto tensor_src = ReadRegister(instr.device_copy.src);
898 NDArray src_data = Downcast<NDArray>(tensor_src);
899 Device actual_src_dev = src_data->device;
900 Device inst_src_dev = GetDevice(instr.device_copy.src_device_index);
901 ICHECK_EQ(actual_src_dev.device_type, inst_src_dev.device_type);
902 ICHECK_EQ(actual_src_dev.device_id, inst_src_dev.device_id);
903 Device dst_dev = GetDevice(instr.device_copy.dst_device_index);
904
905 NDArray dst_data = src_data.CopyTo(dst_dev);
906 WriteRegister(instr.dst, dst_data);
907 OpStopHook();
908 pc_++;
909 goto main_loop;
910 }
911 case Opcode::KillRegister: {
912 OpStartHook(instr);
913 WriteRegister(instr.dst, ObjectRef());
914 OpStopHook();
915 pc_++;
916 goto main_loop;
917 }
918 default:
919 LOG(FATAL) << "Unknown instruction opcode: " << int(instr.op);
920 }
921 }
922}

924void VirtualMachine::WriteAllocatedTensor(const Instruction& instr) {
925 auto shape = std::vector<int64_t>(instr.alloc_tensor.ndim);
926
927 for (uint32_t i = 0; i < instr.alloc_tensor.ndim; ++i) {
928 shape[i] = instr.alloc_tensor.shape[i];
929 }
930
931 auto storage_obj = ReadRegister(instr.alloc_tensor.storage);
932 auto offset = LoadScalarInt(instr.alloc_tensor.offset);
933 auto storage = Downcast<Storage>(storage_obj);
934 auto obj = storage->AllocNDArray(offset, shape, instr.alloc_tensor.dtype);
935 VLOG(2) << "allocated "
936 << RuntimeObject2String(obj, GetDevice(exec_->host_device_index),
937 /*show_contents=*/false);
938
939 WriteRegister(instr.dst, obj);
940}

942void VirtualMachine::WriteAllocatedTensorFromOutside(const Instruction& instr) {
943 // External tensor(s) has been already written to the register (instr.dst)
944 auto ex_arr = Downcast<NDArray>(ReadRegister(instr.dst));
945 auto ex_shape = ex_arr.Shape();
946 auto ex_size = ex_shape.size();
947 auto ex_dtype = ex_arr->dtype;
948
949 auto in_size = instr.alloc_tensor.ndim;
950 auto in_dtype = instr.alloc_tensor.dtype;
951 ICHECK_EQ(TypeEqual(in_dtype, ex_dtype), true)
952 << "Data types mismatching for internal and external output tensors";
953
954 bool size_check = false;
955 if (ex_size != in_size) {
956 size_check = true;
957 } else {
958 for (size_t i = 0; i < in_size; ++i) {
959 if (ex_shape[i] != instr.alloc_tensor.shape[i]) {
960 size_check = true;
961 break;
962 }
963 }
964 }
965
966 if (size_check) {
967 // Match element number
968 size_t in_el_num = 1, ex_el_num = 1;
969 for (size_t i = 0; i < ex_size; ++i) {
970 ex_el_num *= ex_shape[i];
971 }
972 for (size_t i = 0; i < in_size; ++i) {
973 in_el_num *= instr.alloc_tensor.shape[i];
974 }
975 ICHECK_EQ(in_el_num, ex_el_num)
976 << "Element number mismatching of internal and external output tensors";
977 if (code_[preresult_op_index_].op == Opcode::ReshapeTensor) {
978 int64_t* dims = instr.alloc_tensor.shape;
979 std::vector<int64_t> ref_shape(dims, dims + int64_t(in_size));
980 auto reshaped_tensor = ex_arr.CreateView(ref_shape, ex_dtype);
981 WriteRegister(instr.dst, reshaped_tensor);
982 } else {
983 LOG(FATAL) << "Internal and external output tensor shapes are mismatched";
984 }
985 }
986}

990 return it != indices.end();
