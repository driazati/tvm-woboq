58PackedFunc Executable::GetFunction(const std::string& name, const ObjectPtr<Object>& sptr_to_self) {
59 if (name == "get_lib") {
60 return PackedFunc(
61 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetLib(); });
62 } else if (name == "get_bytecode") {
63 return PackedFunc(
64 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetBytecode(); });
65 } else if (name == "get_constants") {
66 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetConstants(); });
67 } else if (name == "get_virtual_devices") {
68 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetVirtualDevices(); });
69 } else if (name == "get_primitives") {
70 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetPrimitives(); });
71 } else if (name == "get_stats") {
72 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->Stats(); });
73 } else if (name == "save") {
74 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->Save(); });
75 } else if (name == "get_function_arity") {
76 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
77 std::string func_name = args[0];
78 *rv = this->GetFunctionArity(func_name);
79 });
80 } else if (name == "get_function_param_name") {
81 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
82 std::string func_name = args[0];
83 int index = args[1];
84 *rv = this->GetFunctionParameterName(func_name, index);
85 });
86 } else if (name == "vm_load_executable") {
87 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
88 auto vm = make_object<VirtualMachine>();
89 ICHECK(sptr_to_self.get() == this);
90 vm->LoadExecutable(GetObjectPtr<Executable>(this));
91 *rv = Module(vm);
92 });
93 } else if (name == "move_late_bound_consts") {
94 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
95 CHECK_EQ(args.size(), 2);
96 std::string path = args[0];
97 uint64_t byte_limit = args[1];
98 MoveLateBoundConstantsToFile(path, static_cast<size_t>(byte_limit));
99 });
100 } else if (name == "get_late_bound_consts") {
101 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
102 CHECK_EQ(args.size(), 1);
103 uint64_t byte_limit = args[0];
104 Map<String, NDArray> consts = GetLateBoundConstants(static_cast<size_t>(byte_limit));
105 *rv = consts;
106 });
107 } else if (name == "load_late_bound_consts") {
108 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
109 CHECK_EQ(args.size(), 1);
110 std::string path = args[0];
111 LoadLateBoundConstantsFromFile(path);
112 });
113 } else if (name == "load_late_bound_consts_from_map") {
114 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
115 CHECK_EQ(args.size(), 1);
116 Map<String, NDArray> map = args[0];
117 LoadLateBoundConstantsFromMap(map);
118 });
119 } else {
120 LOG(FATAL) << "Unknown packed function: " << name;
121 return PackedFunc();
122 }
123}

125const VMFunction& Executable::GetVMFunctionWithName(const std::string& func_name) const {
126 auto it = global_map.find(func_name);
127 ICHECK(it != global_map.end()) << "Cannot find function " << func_name << " in executable";
128 return functions[it->second];
129}

131int Executable::GetFunctionArity(std::string func_name) const {
132 const auto& func = GetVMFunctionWithName(func_name);
133 return func.params.size();
134}

136std::string Executable::GetFunctionParameterName(std::string func_name, uint32_t index) const {
137 const auto& func = GetVMFunctionWithName(func_name);
138 ICHECK_LT(index, func.params.size()) << "Invalid parameter index";
139 return func.params[index];

142std::string Executable::GetBytecode() const {
143 std::ostringstream oss;
144
145 for (size_t i = 0; i < functions.size(); ++i) {
146 const auto& func = functions[i];
147 // Print the header of the function format.
148 oss << "VM Function[" << i << "]: " << func.name << "(";
149 bool first = true;
150 for (const auto& param : func.params) {
151 if (!first) {
152 oss << ", ";
153 }
154 oss << param;
155 first = false;
156 }
157 oss << ")" << std::endl;
158 oss << "# reg file size = " << func.register_file_size << std::endl;
159 oss << "# instruction count = " << func.instructions.size() << std::endl;
160
161 // Print the instructions of a `VMFunction`.
162 // The part after ";" is the instruction in text format.
163 oss << "opcode, fields # inst(text):" << std::endl;
164 for (size_t idx = 0; idx < func.instructions.size(); ++idx) {
165 const auto& instr = func.instructions[idx];
166 const auto& serialized_instr = SerializeInstruction(instr);
167 std::ostringstream line;
168 line << std::setw(2) << idx << ": " << serialized_instr.opcode << " ";
169 for (auto it : serialized_instr.fields) {
170 line << it << " ";
171 }
172 oss << std::setw(40) << std::setfill(' ') << std::left << line.str();
173 oss << " # " << instr;
174 if (oss.str().back() != '\n') oss << std::endl;
175 }
176 oss << std::endl;
177 }
178
179 return oss.str();
180}

182std::string Executable::GetConstants() const {
183 std::ostringstream oss;
184 for (size_t i = 0; i < constants.size(); ++i) {
185 const auto& constant = constants[i];
186 auto ndarray = Downcast<NDArray>(constant);
187 oss << "VM Const[" << i
188 << "]: " << RuntimeObject2String(ndarray, virtual_devices[host_device_index])
189 << " on device index " << const_device_indexes[i] << std::endl;
190 }
191 return oss.str();
192}

198 oss << "VM VirtualDevice[" << i << "]: device type " << device.device_type << " and id "

204std::string Executable::GetPrimitives() const {
205 std::ostringstream os;
206 std::vector<std::pair<int, std::string>> entries;
207 entries.reserve(primitive_map.size());
208 for (const auto& kv : primitive_map) {
209 entries.emplace_back(kv.second, kv.first);
210 }
211 std::sort(entries.begin(), entries.end(),
212 [](const std::pair<int, std::string>& left, const std::pair<int, std::string>& right) {
213 return left.first < right.first;
214 });
215 for (const auto& entry : entries) {
216 os << "VM PackedFunc[" << entry.first << "]: " << entry.second << std::endl;
217 }
218 return os.str();
219}

221std::string Executable::Stats() const {
222 std::ostringstream oss;
223 oss << "Relay VM executable statistics:" << std::endl;
224
225 // Get the number of constants and the shape of each of them.
226 oss << " Constant shapes (# " << constants.size() << "): [";
227 for (const auto& it : constants) {
228 const auto constant = Downcast<NDArray>(it);
229 const auto& shape = constant.Shape();
230
231 // Scalar
232 if (shape.empty()) {
233 oss << "scalar, ";
234 continue;
235 }
236
237 oss << "[";
238 for (auto s : shape) {
239 oss << s << ", ";
240 }
241 oss.seekp(-2, oss.cur);
242 oss << "], " << std::endl;
243 }
244 if (!constants.empty()) oss.seekp(-2, oss.cur);
245 oss << "]" << std::endl;
246
247 // Get the number of globals and the name of each of them.
248 oss << " Globals (#" << global_map.size() << "): [";
249 for (const auto& it : global_map) {
250 oss << "(\"" << it.first << "\", " << it.second << ")"
251 << ", ";
252 }
253 if (!global_map.empty()) oss.seekp(-2, oss.cur);
254 oss << "]" << std::endl;
255
256 // Get the number of primitive ops and the name of each of them.
257 oss << " Primitive ops (#" << primitive_map.size() << "): [";
258 std::vector<std::string> prim_ops;
259 for (const auto& it : primitive_map) {
260 auto packed_index = static_cast<size_t>(it.second);
261 if (prim_ops.size() <= packed_index) {
262 prim_ops.resize(packed_index + 1);
263 }
264 prim_ops[packed_index] = it.first;
265 }
266 for (const auto& it : prim_ops) {
267 oss << it << ", ";
268 }
269 if (!prim_ops.empty()) oss.seekp(-2, oss.cur);
270 oss << "]" << std::endl;
271
272 return oss.str();
273}

276 uint64_t header = kTVMVMBytecodeMagic;

279 strm->Write(version);

282TVMByteArray Executable::Save() {
283 // Initialize the stream object.
284 code_.clear();
285 dmlc::MemoryStringStream strm(&code_);
286
287 // Save header
288 SaveHeader(&strm);
289
290 // Save virtual devices section.
291 SaveVirtualDevicesSection(&strm);
292
293 // Global section.
294 SaveGlobalSection(&strm);
295
296 // Constant section.
297 SaveConstantSection(&strm);
298
299 // Primitive names.
300 SavePrimitiveOpNames(&strm);
301
302 // Code section.
303 SaveCodeSection(&strm);
304
305 TVMByteArray arr;
306 arr.data = code_.c_str();
307 arr.size = code_.length();
308 return arr;
309}

316Map<String, NDArray> Executable::GetLateBoundConstants(size_t byte_limit) {
317 ICHECK(late_bound_constant_names.empty());
318 late_bound_constant_names.reserve(constants.size());
319 Map<String, NDArray> map;
320 size_t total_late_bound_bytes = 0;
321 for (size_t const_index = 0; const_index < constants.size(); ++const_index) {
322 const auto ndarray = Downcast<NDArray>(constants[const_index]);
323 ICHECK(ndarray.defined()) << "Undefined constant at index " << const_index;
324 size_t num_bytes = runtime::GetDataSize(*ndarray.operator->());
325 if (num_bytes < byte_limit) {
326 // Leave as immediate.
327 late_bound_constant_names.emplace_back(nullptr);
328 continue;
329 }
330 total_late_bound_bytes += num_bytes;
331 std::ostringstream os;
332 os << "const_" << const_index;
333 String name = os.str();
334 map.Set(name, Downcast<NDArray>(std::move(constants[const_index])));
335 late_bound_constant_names.emplace_back(std::move(name));
336 }
337 VLOG(1) << "moved " << map.size() << " constants of " << total_late_bound_bytes
338 << " bytes (out of " << constants.size() << " overall) to be late-bound";
339 return map;
340}

342void Executable::MoveLateBoundConstantsToStream(dmlc::Stream* stream, size_t byte_limit) {
343 Map<String, NDArray> map = GetLateBoundConstants(byte_limit);
344 runtime::SaveParams(stream, map);
345}

347void Executable::MoveLateBoundConstantsToFile(const std::string& path, size_t byte_limit) {

354void Executable::LoadLateBoundConstantsFromStream(dmlc::Stream* stream) {
355 if (late_bound_constant_names.empty()) {
356 VLOG(1) << "Found no late-bound constants to load";
357 return;
358 }
359 ICHECK_EQ(late_bound_constant_names.size(), constants.size());
360 Map<String, NDArray> map = runtime::LoadParams(stream);
361 VLOG(1) << "loaded " << map.size() << " late-bound constants";
362 LoadLateBoundConstantsFromMap(map);
363}
364
365void Executable::LoadLateBoundConstantsFromMap(Map<String, NDArray> map) {
366 for (size_t const_index = 0; const_index < constants.size(); ++const_index) {
367 if (!late_bound_constant_names[const_index].defined()) {
368 ICHECK(constants[const_index].defined())
369 << "Undefined immediate constant at index " << const_index;
370 continue;
371 }
372 const String& name = late_bound_constant_names[const_index];
373 ICHECK(!constants[const_index].defined()) << "Unexpected constant at index " << const_index;
374 auto itr = map.find(name);
375 ICHECK(itr != map.end()) << "No binding for late-bound constant at index " << const_index
376 << " with name '" << name << "'";
377 constants[const_index] = (*itr).second;
378 map.erase(name);
379 }
380 late_bound_constant_names.clear();
381 ICHECK(map.empty()) << "Have " << map.size() << " unused late-bound constants";
382}

384void Executable::LoadLateBoundConstantsFromFile(const std::string& path) {
385 std::string bytes;
386 LoadBinaryFromFile(path, &bytes);
387 dmlc::MemoryStringStream stream(&bytes);
388 LoadLateBoundConstantsFromStream(&stream);
389}

391void Executable::SaveGlobalSection(dmlc::Stream* strm) {
392 std::vector<std::pair<std::string, Index>> globals(this->global_map.begin(),
393 this->global_map.end());
394 auto comp = [](const std::pair<std::string, Index>& a, const std::pair<std::string, Index>& b) {
395 return a.second < b.second;
396 };
397 std::sort(globals.begin(), globals.end(), comp);
398
399 std::vector<std::string> glbs;
400 for (const auto& it : globals) {
401 glbs.push_back(it.first);
402 }
403 strm->Write(glbs);
404}

406namespace {
407// Tags to distinguish immediate vs late-bound constants in constants table bytestream.
408constexpr uint32_t kImmediateConstTag = 0;
409constexpr uint32_t kLateBoundConstTag = 1;
410} // namespace
411
412void Executable::SaveConstantSection(dmlc::Stream* stream) {
413 // Save the overall number of constants.
414 stream->Write(static_cast<uint64_t>(constants.size()));
415
416 for (size_t const_index = 0; const_index < constants.size(); ++const_index) {
417 if (late_bound_constant_names.empty() || !late_bound_constant_names[const_index].defined()) {
418 // Tag immediate constants by 0.
419 stream->Write(kImmediateConstTag);
420 // Write as DLTensor.
421 const auto ndarray = Downcast<runtime::NDArray>(constants[const_index]);
422 ICHECK(ndarray.defined());
423 runtime::SaveDLTensor(stream, ndarray.operator->());
424 VLOG(1) << "save " << const_index << " as immediate";
425 } else {
426 // Tag late-bound constants by 1.
427 const String& name = late_bound_constant_names[const_index];
428 ICHECK(!constants[const_index].defined());
429 stream->Write(kLateBoundConstTag);
430 // Write a string.
431 stream->Write(std::string(name));
432 VLOG(1) << "save " << const_index << " as late-bound";
433 }
434 }
435
436 VLOG(1) << "saved " << constants.size() << " constants";
437
438 // Save the const to device index mapping.
439 stream->Write(const_device_indexes);
440}
441
442void Executable::LoadConstantSection(dmlc::Stream* stream) {
443 uint64_t sz;
444 // Load the overall number of constants.
445 STREAM_CHECK(stream->Read(&sz, sizeof(sz)), "constants table size");
446 size_t size = static_cast<size_t>(sz);
447
448 VLOG(1) << "loading " << size << " constants";
449
450 constants.resize(size);
451 late_bound_constant_names.resize(size);
452 bool any_late_bound = false;
453
454 // Load each of the constants.
455 for (size_t const_index = 0; const_index < size; const_index++) {
456 uint32_t tag;
457 STREAM_CHECK(stream->Read(&tag, sizeof(tag)), "constant tag");
458 if (tag == kImmediateConstTag) {
459 // Immediate constants tagged by 0.
460 VLOG(1) << "load " << const_index << " as immediate";
461 runtime::NDArray ndarray;
462 STREAM_CHECK(ndarray.Load(stream), "constant tensor");
463 constants[const_index] = std::move(ndarray);
464 late_bound_constant_names[const_index] = String(ObjectPtr<StringObj>(nullptr));
465 } else if (tag == kLateBoundConstTag) {
466 // Late-bound constants tagged by 1.
467 VLOG(1) << "load " << const_index << " as late-bound";
468 std::string name;
469 STREAM_CHECK(stream->Read(&name), "late-bound constant name");
470 constants[const_index] = NDArray(nullptr);
471 late_bound_constant_names[const_index] = std::move(name);
472 any_late_bound = true;
473 } else {
474 STREAM_CHECK(false, "constant tag");
475 }
476 }
477
478 if (!any_late_bound) {
479 late_bound_constant_names.clear();
480 }
481
482 // Load the const to device index mapping.
483 std::vector<Index> indexes;
484 indexes.reserve(size);
485 STREAM_CHECK(stream->Read(&indexes), "constant devices");
486 ICHECK_EQ(size, indexes.size());
487 const_device_indexes = std::move(indexes);
488}

490void Executable::SavePrimitiveOpNames(dmlc::Stream* strm) {
491 std::vector<std::string> primitive_names;
492 for (const auto& it : this->primitive_map) {
493 auto packed_index = static_cast<size_t>(it.second);
494 if (primitive_names.size() <= packed_index) {
495 primitive_names.resize(packed_index + 1);
496 }
497 primitive_names[packed_index] = it.first;
498 }
499 strm->Write(primitive_names);
500 std::map<uint64_t, std::map<std::string, std::string>> primitive_attrs;
501 for (const auto& it : this->op_attrs) {
502 auto packed_index = static_cast<size_t>(it.first);
503 std::map<std::string, std::string> attrs;
504 for (const auto& elem : it.second) {
505 // TODO(tkonolige): cannot serialize ObjectRefs with dmlc's serializer, so we just serialize
506 // strings for now
507 if (elem.second.as<StringObj>()) {
508 attrs[elem.first] = Downcast<String>(elem.second);
509 }
510 }
511 primitive_attrs[packed_index] = attrs;
512 }
513 strm->Write(primitive_attrs);
514}

516// Serialize a virtual machine instruction. It creates a list that contains the
517// hash, opcode, and all fields of an instruction.
518//
519// For example, the function signature used to create an `AllocTensor`
520// instruction is:
521// Instruction AllocTensor(std::vector<Index> shape, DLDataType dtype, RegName dst)
522//
523// The serialized form will be:
524// `hash 5 dtype.code dtype.bits dtype.lanes ndim dst_register val1 val2 ... valn`
525//
526// where hash is the hash of serialized instruction that is computed internally
527// by the `VMInstructionExecutable`. It is used for sanity check before decoding.
528// 5 shows opcode of `AllocTensor`, `(dtype.code dtype.bits dtype.lanes)`
529// represents a `DLDataType`, `ndim` is the number of dimensions, `dst_register`
530// is the destination register, and the rest of it together indicates the shape
531// of the tensor to be allocated.
532VMInstructionSerializer SerializeInstruction(const Instruction& instr) {
533 std::vector<Index> fields;
534 // Save the opcode.
535 VLOG(2) << "Serializing: " << instr << std::endl;
536 switch (instr.op) {
537 case Opcode::Move: {
538 // Number of fields = 2
539 fields.assign({instr.from, instr.dst});
540 break;
541 }
542 case Opcode::Ret: {
543 // Number of fields = 1
544 fields.push_back(instr.result);
545 break;
546 }
547 case Opcode::Fatal: {
548 // Number of fields = 0
549 break;
550 }
551 case Opcode::InvokePacked: {
552 // Number of fields = 3 + instr.arity
553 // Note that arity includes both input arguments and outputs. We will
554 // put all the `arity` number of fields in the end for serialization.
555 fields.assign({instr.packed_index, instr.arity, instr.output_size});
556 // Save the args.
557 fields.insert(fields.end(), instr.packed_args, instr.packed_args + instr.arity);
558 break;
559 }
560 case Opcode::AllocTensor: {
561 // Number of fields = 7 + instr.alloc_tensor.ndim
562 fields.push_back(instr.alloc_tensor.storage);
563 fields.push_back(instr.alloc_tensor.offset);
564 // Save `DLDataType` and the dst register.
565 const auto& dtype = instr.alloc_tensor.dtype;
566 fields.push_back(dtype.code);
567 fields.push_back(dtype.bits);
568 fields.push_back(dtype.lanes);
569
570 // The number of dimensions is not needed for constructing an
571 // `AllocTensor` instruction as it equals to the length of the `shape`
572 // vector. However, we save it to conveniently deserialize the instruction
573 // because we will know how many fields are needed by the `shape` argument.
574 fields.push_back(instr.alloc_tensor.ndim);
575 fields.push_back(instr.dst);
576
577 // Save the shape of the tensor.
578 // Note that this field is rotated to the end of the list.
579 fields.insert(fields.end(), instr.alloc_tensor.shape,
580 instr.alloc_tensor.shape + instr.alloc_tensor.ndim);
581 break;
582 }
583 case Opcode::AllocTensorReg: {
584 // Number of fields = 7
585 fields.push_back(instr.alloc_tensor_reg.storage);
586 fields.push_back(instr.alloc_tensor_reg.offset);
587 fields.push_back(instr.alloc_tensor_reg.shape_register);
588 // Save `DLDataType` and the dst register.
589 const auto& dtype = instr.alloc_tensor_reg.dtype;
590 fields.push_back(dtype.code);
591 fields.push_back(dtype.bits);
592 fields.push_back(dtype.lanes);
593 fields.push_back(instr.dst);
594 break;
595 }
596 case Opcode::AllocStorage: {
597 fields.push_back(instr.alloc_storage.allocation_size);
598 fields.push_back(instr.alloc_storage.alignment);
599 // Save `DLDataType` and the dst register.
600 const auto& dtype = instr.alloc_storage.dtype_hint;
601 fields.push_back(dtype.code);
602 fields.push_back(dtype.bits);
603 fields.push_back(dtype.lanes);
604 fields.push_back(instr.alloc_storage.device_index);
605 fields.push_back(instr.dst);
606 break;
607 }
608 case Opcode::AllocADT: {
609 // Number of fields = 3 + instr.num_fields
610 fields.assign({instr.constructor_tag, instr.num_fields, instr.dst});
611
612 // Save the fields.
613 fields.insert(fields.end(), instr.datatype_fields, instr.datatype_fields + instr.num_fields);
614 break;
615 }
616 case Opcode::AllocClosure: {
617 // Number of fields = 3 + instr.num_freevar
618 fields.assign({instr.clo_index, instr.num_freevar, instr.dst});
619
620 // Save the free vars.
621 fields.insert(fields.end(), instr.free_vars, instr.free_vars + instr.num_freevar);
622 break;
623 }
624 case Opcode::If: {
625 // Number of fields = 4
626 fields.assign({instr.if_op.test, instr.if_op.target, instr.if_op.true_offset,
627 instr.if_op.false_offset});
628 break;
629 }
630 case Opcode::Invoke: {
631 // Number of fields = 3 + instr.num_args
632 fields.assign({instr.func_index, instr.num_args, instr.dst});
633
634 // Save the args.
635 fields.insert(fields.end(), instr.invoke_args_registers,
636 instr.invoke_args_registers + instr.num_args);
637 break;
638 }
639 case Opcode::InvokeClosure: {
640 // Number of fields = 3 + instr.num_closure_args
641 fields.assign({instr.closure, instr.num_closure_args, instr.dst});
642
643 // Save the args.
644 fields.insert(fields.end(), instr.closure_args, instr.closure_args + instr.num_closure_args);
645 break;
646 }
647 case Opcode::LoadConst: {
648 // Number of fields = 2
649 fields.assign({instr.const_index, instr.dst});
650 break;
651 }
652 case Opcode::LoadConsti: {
653 // Number of fields = 2
654 fields.assign({instr.load_consti.val, instr.dst});
655 break;
656 }
657 case Opcode::GetField: {
658 // Number of fields = 3
659 fields.assign({instr.object, instr.field_index, instr.dst});
660 break;
661 }
662 case Opcode::GetTag: {
663 // Number of fields = 2
664 fields.assign({instr.get_tag.object, instr.dst});
665 break;
666 }
667 case Opcode::Goto: {
668 // Number of fields = 1
669 fields.push_back(instr.pc_offset);
670 break;
671 }
672 case Opcode::ShapeOf: {
673 // Number of fields = 2
674 fields.assign({instr.shape_of.tensor, instr.dst});
675 break;
676 }
677 case Opcode::ReshapeTensor: {
678 // Number of fields = 3
679 fields.assign({instr.reshape_tensor.tensor, instr.reshape_tensor.newshape, instr.dst});
680 break;
681 }
682 case Opcode::DeviceCopy: {
683 // Number of fields = 4
684 fields.assign({instr.device_copy.src, instr.device_copy.src_device_index,
685 instr.device_copy.dst_device_index, instr.dst});
686 break;
687 }
688 case Opcode::KillRegister: {
689 fields.assign({instr.dst});
690 break;
691 }
692 default:
693 LOG(FATAL) << "Invalid opcode" << static_cast<int>(instr.op);
694 break;
695 }
696
697 return VMInstructionSerializer(static_cast<Index>(instr.op), fields);
698}

751runtime::Module Executable::Load(const std::string& code, const runtime::Module lib) {
752 auto exec = make_object<Executable>();
753
754 // Support null-initialization of lib, to enable initialization during
755 // deserialization before we have deserialized the imports.
756 if (lib.defined()) {
757 exec->SetLib(lib);
758 }
759
760 exec->code_ = code;
761 dmlc::MemoryStringStream strm(&exec->code_);
762
763 // Load header.
764 LoadHeader(&strm);
765
766 // Virtual devices section
767 exec->LoadVirtualDevicesSection(&strm);
768
769 // Global section.
770 exec->LoadGlobalSection(&strm);
771
772 // Constant section.
773 exec->LoadConstantSection(&strm);
774
775 // Primitive names that will be invoked by `InvokePacked` instructions.
776 exec->LoadPrimitiveOpNames(&strm);
777
778 // Code section.
779 exec->LoadCodeSection(&strm);
780
781 return runtime::Module(exec);
782}

816// Extract the `cnt` number of fields started at `start` from the list
817// `instr_fields`.
818inline std::vector<Index> ExtractFields(const std::vector<Index>& instr_fields, Index start,
819 Index cnt) {
820 ICHECK_LE(static_cast<size_t>(start + cnt), instr_fields.size());
821 std::vector<Index> ret;
822 for (auto i = start; i < start + cnt; i++) {
823 ret.push_back(instr_fields[i]);
824 }
825 return ret;
826}

828Instruction DeserializeInstruction(const VMInstructionSerializer& instr) {
829 Opcode opcode = static_cast<Opcode>(instr.opcode);
830 switch (opcode) {
831 case Opcode::Move: {
832 // Number of fields = 2
833 DCHECK_EQ(instr.fields.size(), 2U);
834 return Instruction::Move(instr.fields[0], instr.fields[1]);
835 }
836 case Opcode::Ret: {
837 // Number of fields = 1
838 DCHECK_EQ(instr.fields.size(), 1U);
839 return Instruction::Ret(instr.fields[0]);
840 }
841 case Opcode::Fatal: {
842 // Number of fields = 0
843 DCHECK(instr.fields.empty());
844 return Instruction::Fatal();
845 }
846 case Opcode::InvokePacked: {
847 // Number of fields = 3 + instr.arity
848 DCHECK_GE(instr.fields.size(), 3U);
849 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
850
851 Index packed_index = instr.fields[0];
852 Index arity = instr.fields[1];
853 Index output_size = instr.fields[2];
854 std::vector<RegName> args = ExtractFields(instr.fields, 3, arity);
855 return Instruction::InvokePacked(packed_index, arity, output_size, args);
856 }
857 case Opcode::AllocTensor: {
858 // Number of fields = 7 + instr.alloc_tensor.ndim
859 DCHECK_GE(instr.fields.size(), 7U);
860 DCHECK_EQ(instr.fields.size(), 7U + static_cast<size_t>(instr.fields[5]));
861
862 RegName storage_reg = instr.fields[0];
863 RegName offset = instr.fields[1];
864
865 DLDataType dtype;
866 dtype.code = instr.fields[2];
867 dtype.bits = instr.fields[3];
868 dtype.lanes = instr.fields[4];
869
870 Index ndim = instr.fields[5];
871 RegName dst = instr.fields[6];
872
873 std::vector<Index> shape = ExtractFields(instr.fields, 7, ndim);
874
875 return Instruction::AllocTensor(storage_reg, offset, shape, dtype, dst);
876 }
877 case Opcode::AllocTensorReg: {
878 // Number of fields = 7
879 DCHECK_EQ(instr.fields.size(), 7U);
880
881 RegName storage_reg = instr.fields[0];
882 RegName offset = instr.fields[1];
883 Index shape_register = instr.fields[2];
884
885 DLDataType dtype;
886 dtype.code = instr.fields[3];
887 dtype.bits = instr.fields[4];
888 dtype.lanes = instr.fields[5];
889
890 RegName dst = instr.fields[6];
891
892 return Instruction::AllocTensorReg(storage_reg, offset, shape_register, dtype, dst);
893 }
894 case Opcode::AllocADT: {
895 // Number of fields = 3 + instr.num_fields
896 DCHECK_GE(instr.fields.size(), 3U);
897 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
898
899 Index constructor_tag = instr.fields[0];
900 Index num_fields = instr.fields[1];
901 RegName dst = instr.fields[2];
902 std::vector<Index> fields = ExtractFields(instr.fields, 3, num_fields);
903
904 return Instruction::AllocADT(constructor_tag, num_fields, fields, dst);
905 }
906 case Opcode::AllocClosure: {
907 // Number of fields = 3 + instr.num_freevar
908 DCHECK_GE(instr.fields.size(), 3U);
909 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
910
911 Index clo_index = instr.fields[0];
912 Index num_freevar = instr.fields[1];
913 RegName dst = instr.fields[2];
914 std::vector<Index> free_vars = ExtractFields(instr.fields, 3, num_freevar);
915
916 return Instruction::AllocClosure(clo_index, num_freevar, free_vars, dst);
917 }
918 case Opcode::AllocStorage: {
919 // Number of fields = 7
920 DCHECK_GE(instr.fields.size(), 7U);
921 Index allocation_size = instr.fields[0];
922 Index alignment = instr.fields[1];
923
924 DLDataType dtype;
925 dtype.code = instr.fields[2];
926 dtype.bits = instr.fields[3];
927 dtype.lanes = instr.fields[4];
928
929 Index device_type = instr.fields[5];
930 RegName dst = instr.fields[6];
931
932 return Instruction::AllocStorage(allocation_size, alignment, dtype, device_type, dst);
933 }
934 case Opcode::If: {
935 // Number of fields = 4
936 DCHECK_EQ(instr.fields.size(), 4U);
937 Index test = instr.fields[0];
938 Index target = instr.fields[1];
939 Index true_offset = instr.fields[2];
940 Index false_offset = instr.fields[3];
941
942 return Instruction::If(test, target, true_offset, false_offset);
943 }
944 case Opcode::Invoke: {
945 // Number of fields = 3 + instr.num_args
946 DCHECK_GE(instr.fields.size(), 3U);
947 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
948
949 Index func_index = instr.fields[0];
950 Index num_args = instr.fields[1];
951 RegName dst = instr.fields[2];
952 std::vector<Index> args = ExtractFields(instr.fields, 3, num_args);
953
954 return Instruction::Invoke(func_index, args, dst);
955 }
956 case Opcode::InvokeClosure: {
957 // Number of fields = 3 + instr.num_closure_args
958 DCHECK_GE(instr.fields.size(), 3U);
959 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
960
961 Index closure = instr.fields[0];
962 Index num_closure_args = instr.fields[1];
963 RegName dst = instr.fields[2];
964 std::vector<Index> args = ExtractFields(instr.fields, 3, num_closure_args);
965
966 return Instruction::InvokeClosure(closure, args, dst);
967 }
968 case Opcode::LoadConst: {
969 // Number of fields = 2
970 DCHECK_EQ(instr.fields.size(), 2U);
971 return Instruction::LoadConst(instr.fields[0], instr.fields[1]);
972 }
973 case Opcode::LoadConsti: {
974 // Number of fields = 2
975 DCHECK_EQ(instr.fields.size(), 2U);
976 return Instruction::LoadConsti(instr.fields[0], instr.fields[1]);
977 }
978 case Opcode::GetField: {
979 // Number of fields = 3
980 DCHECK_EQ(instr.fields.size(), 3U);
981 return Instruction::GetField(instr.fields[0], instr.fields[1], instr.fields[2]);
982 }
983 case Opcode::GetTag: {
984 // Number of fields = 2
985 DCHECK_EQ(instr.fields.size(), 2U);
986 return Instruction::GetTag(instr.fields[0], instr.fields[1]);
987 }
988 case Opcode::Goto: {
989 // Number of fields = 1
990 DCHECK_EQ(instr.fields.size(), 1U);
991 return Instruction::Goto(instr.fields[0]);
992 }
993 case Opcode::ShapeOf: {
994 // Number of fields = 2
995 DCHECK_EQ(instr.fields.size(), 2U);
996 return Instruction::ShapeOf(instr.fields[0], instr.fields[1]);
997 }
998 case Opcode::ReshapeTensor: {
999 // Number of fields = 3
1000 DCHECK_EQ(instr.fields.size(), 3U);
1001 return Instruction::ReshapeTensor(instr.fields[0], instr.fields[1], instr.fields[2]);
1002 }
1003 case Opcode::DeviceCopy: {
1004 // Number of fields = 4
1005 DCHECK_EQ(instr.fields.size(), 4U);
1006 return Instruction::DeviceCopy(instr.fields[0], instr.fields[1], instr.fields[2],
1007 instr.fields[3]);
1008 }
1009 case Opcode::KillRegister: {
1010 DCHECK_EQ(instr.fields.size(), 1U);
1011 return Instruction::KillRegister(instr.fields[0]);
1012 }
1013 default:
1014 LOG(FATAL) << "Invalid opcode" << instr.opcode;
1015 return Instruction();
1016 }
1017}

1019void Executable::LoadCodeSection(dmlc::Stream* strm) {
1020 // Load the number of functions.
1021 uint64_t sz;
1022 STREAM_CHECK(strm->Read(&sz, sizeof(sz)), "code");
1023
1024 size_t num_funcs = static_cast<size_t>(sz);
1025 this->functions.resize(num_funcs);
1026 for (size_t i = 0; i < num_funcs; i++) {
1027 // Load the function info.
1028 VMFunctionSerializer loaded_func;
1029 STREAM_CHECK(loaded_func.Load(strm), "code/function");
1030
1031 // Load the instructions.
1032 std::vector<Instruction> instructions;
1033 for (size_t j = 0; j < loaded_func.num_instructions; j++) {
1034 VMInstructionSerializer instr;
1035 std::vector<Index> instr_fields;
1036 STREAM_CHECK(instr.Load(strm), "code/instruction");
1037 instructions.push_back(DeserializeInstruction(instr));
1038 }
1039
1040 // Create the VM function.
1041 VMFunction vm_func =
1042 VMFunction(loaded_func.name, loaded_func.params, instructions,
1043 loaded_func.register_file_size, loaded_func.param_device_indexes);
1044 auto it = this->global_map.find(loaded_func.name);
1045 ICHECK(it != this->global_map.end());
1046 ICHECK_LE(it->second, this->global_map.size());
1047 this->functions[it->second] = vm_func;
1048 }
1049}
