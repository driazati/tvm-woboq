58PackedFunc Executable::GetFunction(const std::string& name, const ObjectPtr<Object>& sptr_to_self) {
59 if (name == "get_lib") {
60 return PackedFunc(
61 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetLib(); });
62 } else if (name == "get_bytecode") {
63 return PackedFunc(
64 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetBytecode(); });
65 } else if (name == "get_constants") {
66 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetConstants(); });
67 } else if (name == "get_virtual_devices") {
68 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetVirtualDevices(); });
69 } else if (name == "get_primitives") {
70 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) { *rv = this->GetPrimitives(); });
71 } else if (name == "get_stats") {
72 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->Stats(); });
73 } else if (name == "save") {
74 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->Save(); });
75 } else if (name == "get_function_arity") {
76 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
77 std::string func_name = args[0];
78 *rv = this->GetFunctionArity(func_name);
79 });
80 } else if (name == "get_function_param_name") {
81 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
82 std::string func_name = args[0];
83 int index = args[1];
84 *rv = this->GetFunctionParameterName(func_name, index);
85 });
86 } else if (name == "vm_load_executable") {
87 return PackedFunc([sptr_to_self, this](TVMArgs args, TVMRetValue* rv) {
88 auto vm = make_object<VirtualMachine>();
89 ICHECK(sptr_to_self.get() == this);
90 vm->LoadExecutable(GetObjectPtr<Executable>(this));
91 *rv = Module(vm);
92 });
93 } else if (name == "move_late_bound_consts") {
94 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
95 CHECK_EQ(args.size(), 2);
96 std::string path = args[0];
97 uint64_t byte_limit = args[1];
98 MoveLateBoundConstantsToFile(path, static_cast<size_t>(byte_limit));
99 });
100 } else if (name == "load_late_bound_consts") {
101 return PackedFunc([this](TVMArgs args, TVMRetValue* rv) {
102 CHECK_EQ(args.size(), 1);
103 std::string path = args[0];
104 LoadLateBoundConstantsFromFile(path);
105 });
106 } else {
107 LOG(FATAL) << "Unknown packed function: " << name;
108 return PackedFunc();
109 }
110}

112const VMFunction& Executable::GetVMFunctionWithName(const std::string& func_name) const {
113 auto it = global_map.find(func_name);
114 ICHECK(it != global_map.end()) << "Cannot find function " << func_name << " in executable";
115 return functions[it->second];
116}
117
118int Executable::GetFunctionArity(std::string func_name) const {
119 const auto& func = GetVMFunctionWithName(func_name);
120 return func.params.size();
121}

123std::string Executable::GetFunctionParameterName(std::string func_name, uint32_t index) const {
124 const auto& func = GetVMFunctionWithName(func_name);
125 ICHECK_LT(index, func.params.size()) << "Invalid parameter index";
126 return func.params[index];
127}

129std::string Executable::GetBytecode() const {
130 std::ostringstream oss;
131
132 for (size_t i = 0; i < functions.size(); ++i) {
133 const auto& func = functions[i];
134 // Print the header of the function format.
135 oss << "VM Function[" << i << "]: " << func.name << "(";
136 bool first = true;
137 for (const auto& param : func.params) {
138 if (!first) {
139 oss << ", ";
140 }
141 oss << param;
142 first = false;
143 }
144 oss << ")" << std::endl;
145 oss << "# reg file size = " << func.register_file_size << std::endl;
146 oss << "# instruction count = " << func.instructions.size() << std::endl;
147
148 // Print the instructions of a `VMFunction`.
149 // The part after ";" is the instruction in text format.
150 oss << "opcode, fields # inst(text):" << std::endl;
151 for (size_t idx = 0; idx < func.instructions.size(); ++idx) {
152 const auto& instr = func.instructions[idx];
153 const auto& serialized_instr = SerializeInstruction(instr);
154 std::ostringstream line;
155 line << std::setw(2) << idx << ": " << serialized_instr.opcode << " ";
156 for (auto it : serialized_instr.fields) {
157 line << it << " ";
158 }
159 oss << std::setw(40) << std::setfill(' ') << std::left << line.str();
160 oss << " # " << instr;
161 if (oss.str().back() != '\n') oss << std::endl;
162 }
163 oss << std::endl;
164 }
165
166 return oss.str();
167}

169std::string Executable::GetConstants() const {
170 std::ostringstream oss;
171 for (size_t i = 0; i < constants.size(); ++i) {
172 const auto& constant = constants[i];
173 auto ndarray = Downcast<NDArray>(constant);
174 oss << "VM Const[" << i
175 << "]: " << RuntimeObject2String(ndarray, virtual_devices[host_device_index])
176 << " on device index " << const_device_indexes[i] << std::endl;
177 }
178 return oss.str();
179}

181std::string Executable::GetVirtualDevices() const {
182 std::ostringstream oss;
183 for (size_t i = 0; i < virtual_devices.size(); ++i) {
184 const auto& device = virtual_devices[i];
185 oss << "VM VirtualDevice[" << i << "]: device type " << device.device_type << " and id "

191std::string Executable::GetPrimitives() const {
192 std::ostringstream os;
193 std::vector<std::pair<int, std::string>> entries;
194 entries.reserve(primitive_map.size());
195 for (const auto& kv : primitive_map) {
196 entries.emplace_back(kv.second, kv.first);
197 }
198 std::sort(entries.begin(), entries.end(),
199 [](const std::pair<int, std::string>& left, const std::pair<int, std::string>& right) {
200 return left.first < right.first;
201 });
202 for (const auto& entry : entries) {
203 os << "VM PackedFunc[" << entry.first << "]: " << entry.second << std::endl;
204 }
205 return os.str();
206}

208std::string Executable::Stats() const {
209 std::ostringstream oss;
210 oss << "Relay VM executable statistics:" << std::endl;
211
212 // Get the number of constants and the shape of each of them.
213 oss << " Constant shapes (# " << constants.size() << "): [";
214 for (const auto& it : constants) {
215 const auto constant = Downcast<NDArray>(it);
216 const auto& shape = constant.Shape();
217
218 // Scalar
219 if (shape.empty()) {
220 oss << "scalar, ";
221 continue;
222 }
223
224 oss << "[";
225 for (auto s : shape) {
226 oss << s << ", ";
227 }
228 oss.seekp(-2, oss.cur);
229 oss << "], " << std::endl;
230 }
231 if (!constants.empty()) oss.seekp(-2, oss.cur);
232 oss << "]" << std::endl;
233
234 // Get the number of globals and the name of each of them.
235 oss << " Globals (#" << global_map.size() << "): [";
236 for (const auto& it : global_map) {
237 oss << "(\"" << it.first << "\", " << it.second << ")"
238 << ", ";
239 }
240 if (!global_map.empty()) oss.seekp(-2, oss.cur);
241 oss << "]" << std::endl;
242
243 // Get the number of primitive ops and the name of each of them.
244 oss << " Primitive ops (#" << primitive_map.size() << "): [";
245 std::vector<std::string> prim_ops;
246 for (const auto& it : primitive_map) {
247 auto packed_index = static_cast<size_t>(it.second);
248 if (prim_ops.size() <= packed_index) {
249 prim_ops.resize(packed_index + 1);
250 }
251 prim_ops[packed_index] = it.first;
252 }
253 for (const auto& it : prim_ops) {
254 oss << it << ", ";
255 }
256 if (!prim_ops.empty()) oss.seekp(-2, oss.cur);
257 oss << "]" << std::endl;
258
259 return oss.str();
260}

262void SaveHeader(dmlc::Stream* strm) {
263 uint64_t header = kTVMVMBytecodeMagic;
264 strm->Write(header);

266 strm->Write(version);

269TVMByteArray Executable::Save() {
270 // Initialize the stream object.
271 code_.clear();
272 dmlc::MemoryStringStream strm(&code_);
273
274 // Save header
275 SaveHeader(&strm);
276
277 // Save virtual devices section.
278 SaveVirtualDevicesSection(&strm);
279
280 // Global section.
281 SaveGlobalSection(&strm);
282
283 // Constant section.
284 SaveConstantSection(&strm);
285
286 // Primitive names.
287 SavePrimitiveOpNames(&strm);
288
289 // Code section.
290 SaveCodeSection(&strm);
291
292 TVMByteArray arr;
293 arr.data = code_.c_str();
294 arr.size = code_.length();
295 return arr;
296}

303void Executable::MoveLateBoundConstantsToStream(dmlc::Stream* stream, size_t byte_limit) {
304 ICHECK(late_bound_constant_names.empty());
305 late_bound_constant_names.reserve(constants.size());
306 Map<String, NDArray> map;
307 size_t total_late_bound_bytes = 0;
308 for (size_t const_index = 0; const_index < constants.size(); ++const_index) {
309 const auto ndarray = Downcast<NDArray>(constants[const_index]);
310 ICHECK(ndarray.defined()) << "Undefined constant at index " << const_index;
311 size_t num_bytes = runtime::GetDataSize(*ndarray.operator->());
312 if (num_bytes < byte_limit) {
313 // Leave as immediate.
314 late_bound_constant_names.emplace_back(nullptr);
315 continue;
316 }
317 total_late_bound_bytes += num_bytes;
318 std::ostringstream os;
319 os << "const_" << const_index;
320 String name = os.str();
321 map.Set(name, Downcast<NDArray>(std::move(constants[const_index])));
322 late_bound_constant_names.emplace_back(std::move(name));
323 }
324 VLOG(1) << "moved " << map.size() << " constants of " << total_late_bound_bytes
325 << " bytes (out of " << constants.size() << " overall) to be late-bound";
326 runtime::SaveParams(stream, map);
327}

329void Executable::MoveLateBoundConstantsToFile(const std::string& path, size_t byte_limit) {
330 std::string bytes;
331 dmlc::MemoryStringStream stream(&bytes);
332 MoveLateBoundConstantsToStream(&stream, byte_limit);
333 SaveBinaryToFile(path, bytes);
334}

336void Executable::LoadLateBoundConstantsFromStream(dmlc::Stream* stream) {
337 if (late_bound_constant_names.empty()) {
338 VLOG(1) << "Found no late-bound constants to load";
339 return;
340 }
341 ICHECK_EQ(late_bound_constant_names.size(), constants.size());
342 Map<String, NDArray> map = runtime::LoadParams(stream);
343 VLOG(1) << "loaded " << map.size() << " late-bound constants";
344 for (size_t const_index = 0; const_index < constants.size(); ++const_index) {
345 if (!late_bound_constant_names[const_index].defined()) {
346 ICHECK(constants[const_index].defined())
347 << "Undefined immediate constant at index " << const_index;
348 continue;
349 }
350 const String& name = late_bound_constant_names[const_index];
351 ICHECK(!constants[const_index].defined()) << "Unexpected constant at index " << const_index;
352 auto itr = map.find(name);
353 ICHECK(itr != map.end()) << "No binding for late-bound constant at index " << const_index
354 << " with name '" << name << "'";
355 constants[const_index] = (*itr).second;
356 map.erase(name);
357 }
358 late_bound_constant_names.clear();
359 ICHECK(map.empty()) << "Have " << map.size() << " unused late-bound constants";
360}
361
362void Executable::LoadLateBoundConstantsFromFile(const std::string& path) {
363 std::string bytes;
364 LoadBinaryFromFile(path, &bytes);
365 dmlc::MemoryStringStream stream(&bytes);
366 LoadLateBoundConstantsFromStream(&stream);
367}

369void Executable::SaveGlobalSection(dmlc::Stream* strm) {
370 std::vector<std::pair<std::string, Index>> globals(this->global_map.begin(),
371 this->global_map.end());
372 auto comp = [](const std::pair<std::string, Index>& a, const std::pair<std::string, Index>& b) {
373 return a.second < b.second;
374 };
375 std::sort(globals.begin(), globals.end(), comp);
376
377 std::vector<std::string> glbs;
378 for (const auto& it : globals) {
379 glbs.push_back(it.first);
380 }
381 strm->Write(glbs);
382}

384namespace {
385// Tags to distinguish immediate vs late-bound constants in constants table bytestream.
386constexpr uint32_t kImmediateConstTag = 0;
387constexpr uint32_t kLateBoundConstTag = 1;
388} // namespace
389
390void Executable::SaveConstantSection(dmlc::Stream* stream) {
391 // Save the overall number of constants.
392 stream->Write(static_cast<uint64_t>(constants.size()));
393
394 for (size_t const_index = 0; const_index < constants.size(); ++const_index) {
395 if (late_bound_constant_names.empty() || !late_bound_constant_names[const_index].defined()) {
396 // Tag immediate constants by 0.
397 stream->Write(kImmediateConstTag);
398 // Write as DLTensor.
399 const auto ndarray = Downcast<runtime::NDArray>(constants[const_index]);
400 ICHECK(ndarray.defined());
401 runtime::SaveDLTensor(stream, ndarray.operator->());
402 VLOG(1) << "save " << const_index << " as immediate";
403 } else {
404 // Tag late-bound constants by 1.
405 const String& name = late_bound_constant_names[const_index];
406 ICHECK(!constants[const_index].defined());
407 stream->Write(kLateBoundConstTag);
408 // Write a string.
409 stream->Write(std::string(name));
410 VLOG(1) << "save " << const_index << " as late-bound";
411 }
412 }
413
414 VLOG(1) << "saved " << constants.size() << " constants";
415
416 // Save the const to device index mapping.
417 stream->Write(const_device_indexes);
418}
419
420void Executable::LoadConstantSection(dmlc::Stream* stream) {
421 uint64_t sz;
422 // Load the overall number of constants.
423 STREAM_CHECK(stream->Read(&sz, sizeof(sz)), "constants table size");
424 size_t size = static_cast<size_t>(sz);
425
426 VLOG(1) << "loading " << size << " constants";
427
428 constants.resize(size);
429 late_bound_constant_names.resize(size);
430 bool any_late_bound = false;
431
432 // Load each of the constants.
433 for (size_t const_index = 0; const_index < size; const_index++) {
434 uint32_t tag;
435 STREAM_CHECK(stream->Read(&tag, sizeof(tag)), "constant tag");
436 if (tag == kImmediateConstTag) {
437 // Immediate constants tagged by 0.
438 VLOG(1) << "load " << const_index << " as immediate";
439 runtime::NDArray ndarray;
440 STREAM_CHECK(ndarray.Load(stream), "constant tensor");
441 constants[const_index] = std::move(ndarray);
442 late_bound_constant_names[const_index] = String(ObjectPtr<StringObj>(nullptr));
443 } else if (tag == kLateBoundConstTag) {
444 // Late-bound constants tagged by 1.
445 VLOG(1) << "load " << const_index << " as late-bound";
446 std::string name;
447 STREAM_CHECK(stream->Read(&name), "late-bound constant name");
448 constants[const_index] = NDArray(nullptr);
449 late_bound_constant_names[const_index] = std::move(name);
450 any_late_bound = true;
451 } else {
452 STREAM_CHECK(false, "constant tag");
453 }
454 }
455
456 if (!any_late_bound) {
457 late_bound_constant_names.clear();
458 }
459
460 // Load the const to device index mapping.
461 std::vector<Index> indexes;
462 indexes.reserve(size);
463 STREAM_CHECK(stream->Read(&indexes), "constant devices");
464 ICHECK_EQ(size, indexes.size());
465 const_device_indexes = std::move(indexes);
466}

468void Executable::SavePrimitiveOpNames(dmlc::Stream* strm) {
469 std::vector<std::string> primitive_names;
470 for (const auto& it : this->primitive_map) {
471 auto packed_index = static_cast<size_t>(it.second);
472 if (primitive_names.size() <= packed_index) {
473 primitive_names.resize(packed_index + 1);
474 }
475 primitive_names[packed_index] = it.first;
476 }
477 strm->Write(primitive_names);
478 std::map<uint64_t, std::map<std::string, std::string>> primitive_attrs;
479 for (const auto& it : this->op_attrs) {
480 auto packed_index = static_cast<size_t>(it.first);
481 std::map<std::string, std::string> attrs;
482 for (const auto& elem : it.second) {
483 // TODO(tkonolige): cannot serialize ObjectRefs with dmlc's serializer, so we just serialize
484 // strings for now
485 if (elem.second.as<StringObj>()) {
486 attrs[elem.first] = Downcast<String>(elem.second);
487 }
488 }
489 primitive_attrs[packed_index] = attrs;
490 }
491 strm->Write(primitive_attrs);
492}

494// Serialize a virtual machine instruction. It creates a list that contains the
495// hash, opcode, and all fields of an instruction.
496//
497// For example, the function signature used to create an `AllocTensor`
498// instruction is:
499// Instruction AllocTensor(std::vector<Index> shape, DLDataType dtype, RegName dst)
500//
501// The serialized form will be:
502// `hash 5 dtype.code dtype.bits dtype.lanes ndim dst_register val1 val2 ... valn`
503//
504// where hash is the hash of serialized instruction that is computed internally
505// by the `VMInstructionExecutable`. It is used for sanity check before decoding.
506// 5 shows opcode of `AllocTensor`, `(dtype.code dtype.bits dtype.lanes)`
507// represents a `DLDataType`, `ndim` is the number of dimensions, `dst_register`
508// is the destination register, and the rest of it together indicates the shape
509// of the tensor to be allocated.
510VMInstructionSerializer SerializeInstruction(const Instruction& instr) {
511 std::vector<Index> fields;
512 // Save the opcode.
513 VLOG(2) << "Serializing: " << instr << std::endl;
514 switch (instr.op) {
515 case Opcode::Move: {
516 // Number of fields = 2
517 fields.assign({instr.from, instr.dst});
518 break;
519 }
520 case Opcode::Ret: {
521 // Number of fields = 1
522 fields.push_back(instr.result);
523 break;
524 }
525 case Opcode::Fatal: {
526 // Number of fields = 0
527 break;
528 }
529 case Opcode::InvokePacked: {
530 // Number of fields = 3 + instr.arity
531 // Note that arity includes both input arguments and outputs. We will
532 // put all the `arity` number of fields in the end for serialization.
533 fields.assign({instr.packed_index, instr.arity, instr.output_size});
534 // Save the args.
535 fields.insert(fields.end(), instr.packed_args, instr.packed_args + instr.arity);
536 break;
537 }
538 case Opcode::AllocTensor: {
539 // Number of fields = 7 + instr.alloc_tensor.ndim
540 fields.push_back(instr.alloc_tensor.storage);
541 fields.push_back(instr.alloc_tensor.offset);
542 // Save `DLDataType` and the dst register.
543 const auto& dtype = instr.alloc_tensor.dtype;
544 fields.push_back(dtype.code);
545 fields.push_back(dtype.bits);
546 fields.push_back(dtype.lanes);
547
548 // The number of dimensions is not needed for constructing an
549 // `AllocTensor` instruction as it equals to the length of the `shape`
550 // vector. However, we save it to conveniently deserialize the instruction
551 // because we will know how many fields are needed by the `shape` argument.
552 fields.push_back(instr.alloc_tensor.ndim);
553 fields.push_back(instr.dst);
554
555 // Save the shape of the tensor.
556 // Note that this field is rotated to the end of the list.
557 fields.insert(fields.end(), instr.alloc_tensor.shape,
558 instr.alloc_tensor.shape + instr.alloc_tensor.ndim);
559 break;
560 }
561 case Opcode::AllocTensorReg: {
562 // Number of fields = 7
563 fields.push_back(instr.alloc_tensor_reg.storage);
564 fields.push_back(instr.alloc_tensor_reg.offset);
565 fields.push_back(instr.alloc_tensor_reg.shape_register);
566 // Save `DLDataType` and the dst register.
567 const auto& dtype = instr.alloc_tensor_reg.dtype;
568 fields.push_back(dtype.code);
569 fields.push_back(dtype.bits);
570 fields.push_back(dtype.lanes);
571 fields.push_back(instr.dst);
572 break;
573 }
574 case Opcode::AllocStorage: {
575 fields.push_back(instr.alloc_storage.allocation_size);
576 fields.push_back(instr.alloc_storage.alignment);
577 // Save `DLDataType` and the dst register.
578 const auto& dtype = instr.alloc_storage.dtype_hint;
579 fields.push_back(dtype.code);
580 fields.push_back(dtype.bits);
581 fields.push_back(dtype.lanes);
582 fields.push_back(instr.alloc_storage.device_index);
583 fields.push_back(instr.dst);
584 break;
585 }
586 case Opcode::AllocADT: {
587 // Number of fields = 3 + instr.num_fields
588 fields.assign({instr.constructor_tag, instr.num_fields, instr.dst});
589
590 // Save the fields.
591 fields.insert(fields.end(), instr.datatype_fields, instr.datatype_fields + instr.num_fields);
592 break;
593 }
594 case Opcode::AllocClosure: {
595 // Number of fields = 3 + instr.num_freevar
596 fields.assign({instr.clo_index, instr.num_freevar, instr.dst});
597
598 // Save the free vars.
599 fields.insert(fields.end(), instr.free_vars, instr.free_vars + instr.num_freevar);
600 break;
601 }
602 case Opcode::If: {
603 // Number of fields = 4
604 fields.assign({instr.if_op.test, instr.if_op.target, instr.if_op.true_offset,
605 instr.if_op.false_offset});
606 break;
607 }
608 case Opcode::Invoke: {
609 // Number of fields = 3 + instr.num_args
610 fields.assign({instr.func_index, instr.num_args, instr.dst});
611
612 // Save the args.
613 fields.insert(fields.end(), instr.invoke_args_registers,
614 instr.invoke_args_registers + instr.num_args);
615 break;
616 }
617 case Opcode::InvokeClosure: {
618 // Number of fields = 3 + instr.num_closure_args
619 fields.assign({instr.closure, instr.num_closure_args, instr.dst});
620
621 // Save the args.
622 fields.insert(fields.end(), instr.closure_args, instr.closure_args + instr.num_closure_args);
623 break;
624 }
625 case Opcode::LoadConst: {
626 // Number of fields = 2
627 fields.assign({instr.const_index, instr.dst});
628 break;
629 }
630 case Opcode::LoadConsti: {
631 // Number of fields = 2
632 fields.assign({instr.load_consti.val, instr.dst});
633 break;
634 }
635 case Opcode::GetField: {
636 // Number of fields = 3
637 fields.assign({instr.object, instr.field_index, instr.dst});
638 break;
639 }
640 case Opcode::GetTag: {
641 // Number of fields = 2
642 fields.assign({instr.get_tag.object, instr.dst});
643 break;
644 }
645 case Opcode::Goto: {
646 // Number of fields = 1
647 fields.push_back(instr.pc_offset);
648 break;
649 }
650 case Opcode::ShapeOf: {
651 // Number of fields = 2
652 fields.assign({instr.shape_of.tensor, instr.dst});
653 break;
654 }
655 case Opcode::ReshapeTensor: {
656 // Number of fields = 3
657 fields.assign({instr.reshape_tensor.tensor, instr.reshape_tensor.newshape, instr.dst});
658 break;
659 }
660 case Opcode::DeviceCopy: {
661 // Number of fields = 4
662 fields.assign({instr.device_copy.src, instr.device_copy.src_device_index,
663 instr.device_copy.dst_device_index, instr.dst});
664 break;
665 }
666 case Opcode::KillRegister: {
667 fields.assign({instr.dst});
668 break;
669 }
670 default:
671 LOG(FATAL) << "Invalid opcode" << static_cast<int>(instr.op);
672 break;
673 }
674
675 return VMInstructionSerializer(static_cast<Index>(instr.op), fields);
676}

679 // Save the number of functions.
680 strm->Write(static_cast<uint64_t>(this->functions.size()));

683 VMFunctionSerializer func_format(func.name, func.register_file_size, func.instructions.size(),

729runtime::Module Executable::Load(const std::string& code, const runtime::Module lib) {
730 auto exec = make_object<Executable>();
731
732 // Support null-initialization of lib, to enable initialization during
733 // deserialization before we have deserialized the imports.
734 if (lib.defined()) {
735 exec->SetLib(lib);
736 }
737
738 exec->code_ = code;
739 dmlc::MemoryStringStream strm(&exec->code_);
740
741 // Load header.
742 LoadHeader(&strm);
743
744 // Virtual devices section
745 exec->LoadVirtualDevicesSection(&strm);
746
747 // Global section.
748 exec->LoadGlobalSection(&strm);
749
750 // Constant section.
751 exec->LoadConstantSection(&strm);
752
753 // Primitive names that will be invoked by `InvokePacked` instructions.
754 exec->LoadPrimitiveOpNames(&strm);
755
756 // Code section.
757 exec->LoadCodeSection(&strm);
758
759 return runtime::Module(exec);
760}

794// Extract the `cnt` number of fields started at `start` from the list
795// `instr_fields`.
796inline std::vector<Index> ExtractFields(const std::vector<Index>& instr_fields, Index start,
797 Index cnt) {
798 ICHECK_LE(static_cast<size_t>(start + cnt), instr_fields.size());
799 std::vector<Index> ret;
800 for (auto i = start; i < start + cnt; i++) {
801 ret.push_back(instr_fields[i]);
802 }
803 return ret;
804}

806Instruction DeserializeInstruction(const VMInstructionSerializer& instr) {
807 Opcode opcode = static_cast<Opcode>(instr.opcode);
808 switch (opcode) {
809 case Opcode::Move: {
810 // Number of fields = 2
811 DCHECK_EQ(instr.fields.size(), 2U);
812 return Instruction::Move(instr.fields[0], instr.fields[1]);
813 }
814 case Opcode::Ret: {
815 // Number of fields = 1
816 DCHECK_EQ(instr.fields.size(), 1U);
817 return Instruction::Ret(instr.fields[0]);
818 }
819 case Opcode::Fatal: {
820 // Number of fields = 0
821 DCHECK(instr.fields.empty());
822 return Instruction::Fatal();
823 }
824 case Opcode::InvokePacked: {
825 // Number of fields = 3 + instr.arity
826 DCHECK_GE(instr.fields.size(), 3U);
827 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
828
829 Index packed_index = instr.fields[0];
830 Index arity = instr.fields[1];
831 Index output_size = instr.fields[2];
832 std::vector<RegName> args = ExtractFields(instr.fields, 3, arity);
833 return Instruction::InvokePacked(packed_index, arity, output_size, args);
834 }
835 case Opcode::AllocTensor: {
836 // Number of fields = 7 + instr.alloc_tensor.ndim
837 DCHECK_GE(instr.fields.size(), 7U);
838 DCHECK_EQ(instr.fields.size(), 7U + static_cast<size_t>(instr.fields[5]));
839
840 RegName storage_reg = instr.fields[0];
841 RegName offset = instr.fields[1];
842
843 DLDataType dtype;
844 dtype.code = instr.fields[2];
845 dtype.bits = instr.fields[3];
846 dtype.lanes = instr.fields[4];
847
848 Index ndim = instr.fields[5];
849 RegName dst = instr.fields[6];
850
851 std::vector<Index> shape = ExtractFields(instr.fields, 7, ndim);
852
853 return Instruction::AllocTensor(storage_reg, offset, shape, dtype, dst);
854 }
855 case Opcode::AllocTensorReg: {
856 // Number of fields = 7
857 DCHECK_EQ(instr.fields.size(), 7U);
858
859 RegName storage_reg = instr.fields[0];
860 RegName offset = instr.fields[1];
861 Index shape_register = instr.fields[2];
862
863 DLDataType dtype;
864 dtype.code = instr.fields[3];
865 dtype.bits = instr.fields[4];
866 dtype.lanes = instr.fields[5];
867
868 RegName dst = instr.fields[6];
869
870 return Instruction::AllocTensorReg(storage_reg, offset, shape_register, dtype, dst);
871 }
872 case Opcode::AllocADT: {
873 // Number of fields = 3 + instr.num_fields
874 DCHECK_GE(instr.fields.size(), 3U);
875 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
876
877 Index constructor_tag = instr.fields[0];
878 Index num_fields = instr.fields[1];
879 RegName dst = instr.fields[2];
880 std::vector<Index> fields = ExtractFields(instr.fields, 3, num_fields);
881
882 return Instruction::AllocADT(constructor_tag, num_fields, fields, dst);
883 }
884 case Opcode::AllocClosure: {
885 // Number of fields = 3 + instr.num_freevar
886 DCHECK_GE(instr.fields.size(), 3U);
887 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
888
889 Index clo_index = instr.fields[0];
890 Index num_freevar = instr.fields[1];
891 RegName dst = instr.fields[2];
892 std::vector<Index> free_vars = ExtractFields(instr.fields, 3, num_freevar);
893
894 return Instruction::AllocClosure(clo_index, num_freevar, free_vars, dst);
895 }
896 case Opcode::AllocStorage: {
897 // Number of fields = 7
898 DCHECK_GE(instr.fields.size(), 7U);
899 Index allocation_size = instr.fields[0];
900 Index alignment = instr.fields[1];
901
902 DLDataType dtype;
903 dtype.code = instr.fields[2];
904 dtype.bits = instr.fields[3];
905 dtype.lanes = instr.fields[4];
906
907 Index device_type = instr.fields[5];
908 RegName dst = instr.fields[6];
909
910 return Instruction::AllocStorage(allocation_size, alignment, dtype, device_type, dst);
911 }
912 case Opcode::If: {
913 // Number of fields = 4
914 DCHECK_EQ(instr.fields.size(), 4U);
915 Index test = instr.fields[0];
916 Index target = instr.fields[1];
917 Index true_offset = instr.fields[2];
918 Index false_offset = instr.fields[3];
919
920 return Instruction::If(test, target, true_offset, false_offset);
921 }
922 case Opcode::Invoke: {
923 // Number of fields = 3 + instr.num_args
924 DCHECK_GE(instr.fields.size(), 3U);
925 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
926
927 Index func_index = instr.fields[0];
928 Index num_args = instr.fields[1];
929 RegName dst = instr.fields[2];
930 std::vector<Index> args = ExtractFields(instr.fields, 3, num_args);
931
932 return Instruction::Invoke(func_index, args, dst);
933 }
934 case Opcode::InvokeClosure: {
935 // Number of fields = 3 + instr.num_closure_args
936 DCHECK_GE(instr.fields.size(), 3U);
937 DCHECK_EQ(instr.fields.size(), 3U + static_cast<size_t>(instr.fields[1]));
938
939 Index closure = instr.fields[0];
940 Index num_closure_args = instr.fields[1];
941 RegName dst = instr.fields[2];
942 std::vector<Index> args = ExtractFields(instr.fields, 3, num_closure_args);
943
944 return Instruction::InvokeClosure(closure, args, dst);
945 }
946 case Opcode::LoadConst: {
947 // Number of fields = 2
948 DCHECK_EQ(instr.fields.size(), 2U);
949 return Instruction::LoadConst(instr.fields[0], instr.fields[1]);
950 }
951 case Opcode::LoadConsti: {
952 // Number of fields = 2
953 DCHECK_EQ(instr.fields.size(), 2U);
954 return Instruction::LoadConsti(instr.fields[0], instr.fields[1]);
955 }
956 case Opcode::GetField: {
957 // Number of fields = 3
958 DCHECK_EQ(instr.fields.size(), 3U);
959 return Instruction::GetField(instr.fields[0], instr.fields[1], instr.fields[2]);
960 }
961 case Opcode::GetTag: {
962 // Number of fields = 2
963 DCHECK_EQ(instr.fields.size(), 2U);
964 return Instruction::GetTag(instr.fields[0], instr.fields[1]);
965 }
966 case Opcode::Goto: {
967 // Number of fields = 1
968 DCHECK_EQ(instr.fields.size(), 1U);
969 return Instruction::Goto(instr.fields[0]);
970 }
971 case Opcode::ShapeOf: {
972 // Number of fields = 2
973 DCHECK_EQ(instr.fields.size(), 2U);
974 return Instruction::ShapeOf(instr.fields[0], instr.fields[1]);
975 }
976 case Opcode::ReshapeTensor: {
977 // Number of fields = 3
978 DCHECK_EQ(instr.fields.size(), 3U);
979 return Instruction::ReshapeTensor(instr.fields[0], instr.fields[1], instr.fields[2]);
980 }
981 case Opcode::DeviceCopy: {
982 // Number of fields = 4
983 DCHECK_EQ(instr.fields.size(), 4U);
984 return Instruction::DeviceCopy(instr.fields[0], instr.fields[1], instr.fields[2],
985 instr.fields[3]);
986 }
987 case Opcode::KillRegister: {
988 DCHECK_EQ(instr.fields.size(), 1U);
989 return Instruction::KillRegister(instr.fields[0]);
990 }
991 default:
992 LOG(FATAL) << "Invalid opcode" << instr.opcode;
993 return Instruction();
994 }
995}

997void Executable::LoadCodeSection(dmlc::Stream* strm) {
998 // Load the number of functions.
999 uint64_t sz;
1000 STREAM_CHECK(strm->Read(&sz, sizeof(sz)), "code");
1001
1002 size_t num_funcs = static_cast<size_t>(sz);
1003 this->functions.resize(num_funcs);
1004 for (size_t i = 0; i < num_funcs; i++) {
1005 // Load the function info.
1006 VMFunctionSerializer loaded_func;
1007 STREAM_CHECK(loaded_func.Load(strm), "code/function");
1008
1009 // Load the instructions.
1010 std::vector<Instruction> instructions;
1011 for (size_t j = 0; j < loaded_func.num_instructions; j++) {
1012 VMInstructionSerializer instr;
1013 std::vector<Index> instr_fields;
1014 STREAM_CHECK(instr.Load(strm), "code/instruction");
1015 instructions.push_back(DeserializeInstruction(instr));
1016 }
1017
1018 // Create the VM function.
1019 VMFunction vm_func =
1020 VMFunction(loaded_func.name, loaded_func.params, instructions,
1021 loaded_func.register_file_size, loaded_func.param_device_indexes);
1022 auto it = this->global_map.find(loaded_func.name);
1023 ICHECK(it != this->global_map.end());
1024 ICHECK_LE(it->second, this->global_map.size());
1025 this->functions[it->second] = vm_func;
1026 }
1027}
