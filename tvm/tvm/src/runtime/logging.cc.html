42 size_t max_size;

56std::string DemangleName(std::string name) {
57 int status = 0;
58 size_t length = name.size();
59 std::unique_ptr<char, void (*)(void* __ptr)> demangled_name = {
60 abi::__cxa_demangle(name.c_str(), nullptr, &length, &status), &std::free};
61 if (demangled_name && status == 0 && length > 0) {
62 return demangled_name.get();
63 } else {
64 return name;
65 }
66}

68void BacktraceErrorCallback(void* data, const char* msg, int errnum) {
69 // do nothing
70}

72void BacktraceSyminfoCallback(void* data, uintptr_t pc, const char* symname, uintptr_t symval,
73 uintptr_t symsize) {
74 auto str = reinterpret_cast<std::string*>(data);
75
76 if (symname != nullptr) {
77 std::string tmp(symname, symsize);
78 *str = DemangleName(tmp.c_str());
79 } else {
80 std::ostringstream s;
81 s << "0x" << std::setfill('0') << std::setw(sizeof(uintptr_t) * 2) << std::hex << pc;
82 *str = s.str();
83 }
84}
85
86int BacktraceFullCallback(void* data, uintptr_t pc, const char* filename, int lineno,
87 const char* symbol) {
88 auto stack_trace = reinterpret_cast<BacktraceInfo*>(data);
89 std::stringstream s;
90
91 std::unique_ptr<std::string> symbol_str = std::make_unique<std::string>("<unknown>");
92 if (symbol != nullptr) {
93 *symbol_str = DemangleName(symbol);
94 } else {
95 // see if syminfo gives anything
96 backtrace_syminfo(_bt_state, pc, BacktraceSyminfoCallback, BacktraceErrorCallback,
97 symbol_str.get());
98 }
99 s << *symbol_str;
100
101 if (filename != nullptr) {
102 s << std::endl << " at " << filename;
103 if (lineno != 0) {
104 s << ":" << lineno;
105 }
106 }
107 // Skip tvm::backtrace and tvm::LogFatal::~LogFatal at the beginning of the trace as they don't
108 // add anything useful to the backtrace.
109 if (!(stack_trace->lines.size() == 0 &&
110 (symbol_str->find("tvm::runtime::Backtrace", 0) == 0 ||
111 symbol_str->find("tvm::runtime::detail::LogFatal", 0) == 0))) {
112 stack_trace->lines.push_back(s.str());
113 }
114 // TVMFuncCall denotes the API boundary so we stop there. Exceptions should be caught there.
115 if (*symbol_str == "TVMFuncCall" || stack_trace->lines.size() >= stack_trace->max_size) {
116 return 1;
117 }
118 return 0;
119}

122std::string Backtrace() {
123 BacktraceInfo bt;
124
125 // Limit backtrace length based on TVM_BACKTRACE_LIMIT env variable
126 auto user_limit_s = getenv("TVM_BACKTRACE_LIMIT");
127 const auto default_limit = 500;
128
129 if (user_limit_s == nullptr) {
130 bt.max_size = default_limit;
131 } else {
132 // Parse out the user-set backtrace limit
133 try {
134 bt.max_size = std::stoi(user_limit_s);
135 } catch (const std::invalid_argument& e) {
136 bt.max_size = default_limit;
137 }
138 }
139
140 if (_bt_state == nullptr) {
141 return "";
142 }
143 // libbacktrace eats memory if run on multiple threads at the same time, so we guard against it
144 static std::mutex m;
145 std::lock_guard<std::mutex> lock(m);
146 backtrace_full(_bt_state, 0, BacktraceFullCallback, BacktraceErrorCallback, &bt);
147
148 std::ostringstream s;
149 s << "Stack trace:\n";
150 for (size_t i = 0; i < bt.lines.size(); i++) {
151 s << " " << i << ": " << bt.lines[i] << "\n";
152 }
153
154 return s.str();
155}

187constexpr const char* kSrcPrefix = "/src/";
188// Note: Better would be std::char_traits<const char>::length(kSrcPrefix) but it is not
189// a constexpr on all compilation targets.
190constexpr const size_t kSrcPrefixLength = 5;
191constexpr const char* kDefaultKeyword = "DEFAULT";
192} // namespace
193
194namespace {
195/*! \brief Convert __FILE__ to a vlog_level_map_ key, which strips any prefix ending iwth src/ */
196std::string FileToVLogMapKey(const std::string& filename) {
197 // Canonicalize the filename.
198 // TODO(mbs): Not Windows friendly.
199 size_t last_src = filename.rfind(kSrcPrefix, std::string::npos, kSrcPrefixLength);
200 if (last_src == std::string::npos) {
201 std::string no_slash_src{kSrcPrefix + 1};
202 if (filename.substr(0, no_slash_src.size()) == no_slash_src) {
203 return filename.substr(no_slash_src.size());
204 }
205 }
206 // Strip anything before the /src/ prefix, on the assumption that will yield the
207 // TVM project relative filename. If no such prefix fallback to filename without
208 // canonicalization.
209 return (last_src == std::string::npos) ? filename : filename.substr(last_src + kSrcPrefixLength);
210}
211} // namespace

213/* static */
214TvmLogDebugSettings TvmLogDebugSettings::ParseSpec(const char* opt_spec) {
215 TvmLogDebugSettings settings;
216 if (opt_spec == nullptr) {
217 // DLOG and VLOG disabled.
218 return settings;
219 }
220 std::string spec(opt_spec);
221 if (spec.empty() || spec == "0") {
222 // DLOG and VLOG disabled.
223 return settings;
224 }
225 settings.dlog_enabled_ = true;
226 if (spec == "1") {
227 // Legacy specification for enabling just DLOG.
228 return settings;
229 }
230 std::istringstream spec_stream(spec);
231 auto tell_pos = [&](const std::string& last_read) {
232 int pos = spec_stream.tellg();
233 if (pos == -1) {
234 LOG(INFO) << "override pos: " << last_read;
235 // when pos == -1, failbit was set due to std::getline reaching EOF without seeing delimiter.
236 pos = spec.size() - last_read.size();
237 }
238 return pos;
239 };
240 while (spec_stream) {
241 std::string name;
242 if (!std::getline(spec_stream, name, '=')) {
243 // Reached end.
244 break;
245 }
246 if (name.empty()) {
247 LOG(FATAL) << "TVM_LOG_DEBUG ill-formed at position " << tell_pos(name) << ": empty filename";
248 return settings;
249 }
250
251 name = FileToVLogMapKey(name);
252
253 std::string level;
254 if (!std::getline(spec_stream, level, ',')) {
255 LOG(FATAL) << "TVM_LOG_DEBUG ill-formed at position " << tell_pos(level)
256 << ": expecting \"=<level>\" after \"" << name << "\"";
257 return settings;
258 }
259 if (level.empty()) {
260 LOG(FATAL) << "TVM_LOG_DEBUG ill-formed at position " << tell_pos(level)
261 << ": empty level after \"" << name << "\"";
262 return settings;
263 }
264 // Parse level, default to 0 if ill-formed which we don't detect.
265 char* end_of_level = nullptr;
266 int level_val = static_cast<int>(strtol(level.c_str(), &end_of_level, 10));
267 if (end_of_level != level.c_str() + level.size()) {
268 LOG(FATAL) << "TVM_LOG_DEBUG ill-formed at position " << tell_pos(level)
269 << ": invalid level: \"" << level << "\"";
270 return settings;
271 }
272 LOG(INFO) << "TVM_LOG_DEBUG enables VLOG statements in '" << name << "' up to level " << level;
273 settings.vlog_level_map_.emplace(name, level_val);
274 }
275 return settings;
276}

278bool TvmLogDebugSettings::VerboseEnabledImpl(const std::string& filename, int level) const {
279 // Check for exact match.
280 auto itr = vlog_level_map_.find(FileToVLogMapKey(filename));
281 if (itr != vlog_level_map_.end()) {
282 return level <= itr->second;
283 }
284 // Check for default.
285 itr = vlog_level_map_.find(kDefaultKeyword);
286 if (itr != vlog_level_map_.end()) {
287 return level <= itr->second;
288 }
289 return false;
290}
