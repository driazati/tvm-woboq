20/*!
21 * \file src/runtime/const_loader_module.cc
22 * \brief A wrapper for initializing imported modules using constant NDArray. This
23 * module is intended to be used by various runtime in the TVM stack, i.e.
24 * graph executor, relay VM, AOT runtime, and various user defined runtimes. It
25 * paves the way to separate the code and metedata, which makes compilation
26 * and/or interpretation more convenient. In addition, the clear separation of
27 * code and constants significantly reduces the efforts for handling external
28 * codegen and runtimes.
29 */

44/*!
45 * \brief The const-loader module is designed to manage initialization of the
46 * imported submodules for the C++ runtime.
47 */
48class ConstLoaderModuleNode : public ModuleNode {
49 public:
50 ConstLoaderModuleNode(
51 const std::unordered_map<std::string, NDArray>& const_var_ndarray,
52 const std::unordered_map<std::string, std::vector<std::string>>& const_vars_by_symbol)
53 : const_var_ndarray_(const_var_ndarray), const_vars_by_symbol_(const_vars_by_symbol) {
54 VLOG(1) << "Creating ConstLoaderModule";
55 // Only the related submodules are cached to reduce the number of runtime
56 // symbol lookup for initialization. Otherwise, symbols/primitives in the
57 // DSO module will also be cached but they never need to be initialized.
58 for (const auto& kv : const_vars_by_symbol_) {
59 for (const auto& var : kv.second) {
60 VLOG(1) << "ConstLoaderModuleNode has constant '" << var << "' for function '" << kv.first
61 << "'";
62 ICHECK_GT(const_var_ndarray_.count(var), 0)
63 << "ConstLoaderModuleNode is missing entry for constant '" << var << "' for function '"
64 << kv.first << "'";
65 }
66 initialized_[kv.first] = false;
67 }
68 }
69
70 PackedFunc GetFunction(const std::string& name, const ObjectPtr<Object>& sptr_to_self) final {
71 VLOG(1) << "ConstLoaderModuleNode::GetFunction(" << name << ")";
72 // Initialize and memoize the module.
73 // Usually, we have some warmup runs. The module initialization should be
74 // done at this stage. Therefore, runtime overhead is not a concern.
75 if (initialized_.count(name) && !initialized_.at(name)) {
76 this->InitSubModule(name);
77 initialized_[name] = true;
78 }
79
80 // Run the module.
81 // Normally we would only have a limited number of submodules. The runtime
82 // symobl lookup overhead should be minimal.
83 ICHECK(!this->imports().empty());
84 for (Module it : this->imports()) {
85 PackedFunc pf = it.GetFunction(name);
86 if (pf != nullptr) return pf;
87 }
88 return PackedFunc(nullptr);
89 }
90
91 const char* type_key() const final { return "const_loader"; }
92
93 /*!
94 * \brief Get the list of constants that is required by the given module.
95 * \param symbol The symbol that is being queried.
96 * \return The list of needed NDArray.
97 */
98 Array<NDArray> GetRequiredConstants(const std::string& symbol) {
99 Array<NDArray> ret;
100 ICHECK_GT(const_vars_by_symbol_.count(symbol), 0U)
101 << "No constants known for function '" << symbol << "'";
102 std::vector<std::string> vars = const_vars_by_symbol_[symbol];
103 for (const auto& var : vars) {
104 ICHECK_GT(const_var_ndarray_.count(var), 0U)
105 << "No such constant variable '" << var << "' for function '" << symbol << "'";
106 ret.push_back(const_var_ndarray_[var]);
107 }
108 return ret;
109 }
110
111 /*!
112 * \brief Initialize each imported module.
113 * \param symobl The symbol used for initializing a module. It is also used
114 * for runtime lookup.
115 *
116 * \note A module could be like the following:
117 * ConstLoaderModuleNode (contains all the constants)
118 * - CSourceModule
119 * - JSON runtime module
120 *
121 * The initializer iterates through the imported modules and intilizes the
122 * found module accordingly by passing the needed constants into it.
123 */
124 void InitSubModule(const std::string& symbol) {
125 PackedFunc init(nullptr);
126 for (Module it : this->imports()) {
127 // Get the initialization function from the imported modules.
128 std::string init_name = "__init_" + symbol;
129 init = it.GetFunction(init_name, false);
130 if (init != nullptr) {
131 auto md = GetRequiredConstants(symbol);
132 // Initialize the module with constants.
133 int ret = init(md);
134 // Report the error if initialization is failed.
135 ICHECK_EQ(ret, 0) << TVMGetLastError();
136 break;
137 }
138 }
139 }
140
141 void SaveToBinary(dmlc::Stream* stream) final {
142 std::vector<std::string> variables;
143 std::vector<NDArray> const_var_ndarray;
144 for (const auto& it : const_var_ndarray_) {
145 String var_name = it.first;
146 variables.push_back(var_name);
147 const_var_ndarray.push_back(it.second);
148 }
149
150 // Save all variables in the function.
151 stream->Write(variables);
152 // Save all constant data.
153 uint64_t sz = static_cast<uint64_t>(const_var_ndarray.size());
154 stream->Write(sz);
155 for (uint64_t i = 0; i < sz; i++) {
156 const_var_ndarray[i].Save(stream);
157 }
158
159 // Save the symbol to list of required constant variables mapping
160 std::vector<std::string> symbols;
161 std::vector<std::vector<std::string>> const_vars;
162 for (const auto& it : const_vars_by_symbol_) {
163 symbols.push_back(it.first);
164 const_vars.push_back(it.second);
165 }
166
167 stream->Write(symbols);
168 sz = static_cast<uint64_t>(const_vars_by_symbol_.size());
169 stream->Write(sz);
170 for (uint64_t i = 0; i < sz; i++) {
171 stream->Write(const_vars[i]);
172 }
173 }
174
175 static Module LoadFromBinary(void* strm) {
176 dmlc::Stream* stream = static_cast<dmlc::Stream*>(strm);
177
178 // Load the variables.
179 std::vector<std::string> variables;
180 ICHECK(stream->Read(&variables)) << "Loading variable names failed";
181 uint64_t sz;
182 ICHECK(stream->Read(&sz, sizeof(sz))) << "Loading number of vars failed";
183 ICHECK_EQ(static_cast<size_t>(sz), variables.size())
184 << "The number of variables and ndarray counts must match";
185 // Load the list of ndarray.
186 std::vector<NDArray> arrays;
187 for (uint64_t i = 0; i < sz; i++) {
188 NDArray temp;
189 temp.Load(stream);
190 arrays.push_back(temp);
191 }
192
193 std::unordered_map<std::string, NDArray> const_var_ndarray;
194 for (uint64_t i = 0; i < sz; i++) {
195 ICHECK_EQ(const_var_ndarray.count(variables[i]), 0U);
196 const_var_ndarray[variables[i]] = arrays[i];
197 }
198
199 // Load the symbol to list of required constant variables mapping
200 std::vector<std::string> symbols;
201 ICHECK(stream->Read(&symbols)) << "Loading symbols failed";
202 ICHECK(stream->Read(&sz, sizeof(sz))) << "Loading number of symbols failed";
203 ICHECK_EQ(static_cast<size_t>(sz), symbols.size());
204 std::vector<std::vector<std::string>> const_vars;
205 for (uint64_t i = 0; i < sz; i++) {
206 std::vector<std::string> vars;
207 ICHECK(stream->Read(&vars)) << "Loading const variables failed";
208 const_vars.push_back(vars);
209 }
210
211 std::unordered_map<std::string, std::vector<std::string>> const_vars_by_symbol;
212 for (uint64_t i = 0; i < sz; i++) {
213 const_vars_by_symbol[symbols[i]] = const_vars[i];
214 }
215
216 auto n = make_object<ConstLoaderModuleNode>(const_var_ndarray, const_vars_by_symbol);
217 return Module(n);
218 }
219
220 private:
221 /*!
222 * \brief Record if a module is initialized. It is needed by imported
223 * modules using execution engine.
224 */
225 std::unordered_map<std::string, bool> initialized_;
226 /*! \brief Variable name to NDArray mapping. */
227 std::unordered_map<std::string, NDArray> const_var_ndarray_;
228 /*! \brief Symbol name to required constant variables mapping. */
229 std::unordered_map<std::string, std::vector<std::string>> const_vars_by_symbol_;
230};
231
232Module ConstLoaderModuleCreate(
233 const std::unordered_map<std::string, NDArray>& const_var_ndarray,
234 const std::unordered_map<std::string, std::vector<std::string>>& const_vars_by_symbol) {
235 auto n = make_object<ConstLoaderModuleNode>(const_var_ndarray, const_vars_by_symbol);
236 return Module(n);
237}

240 .set_body_typed(ConstLoaderModuleNode::LoadFromBinary);
