38/*!
39 * \brief Memory hierachy rank in the storage system
40 * \note The global rank and shared rank have one to one
41 * correspondence to the thread rank.
42 */
43enum class StorageRank {
44 /*! \brief global memory */
45 kGlobal = 0,
46 /*! \brief shared memory among thread group */
47 kShared = 1,
48 /*!
49 * \brief reserved for warp memory.
50 * This is only used by programming model.
51 * There is no such memory usually in GPU.
52 * Instead, we can simulate it by registers and shuffle.
53 */
54 kWarp = 2,
55 /*! \brief thread local memory */
56 kLocal = 3,
57 /*! \brief wmma scope memory of matrix_a */
58 kWMMAMatrixA = 4,
59 /*! \brief wmma scope memory of matrix_b */
60 kWMMAMatrixB = 5,
61 /*! \brief wmma scope memory of accumulator */
62 kWMMAAccumulator = 6,
63 /*! \brief global scope texture memory */
64 kTexture = 7,
65};
66
67/*!
68 * \param thread_scope_rank The thread scope rank
69 * \return default storage rank given the thread scope
70 */
71inline StorageRank DefaultStorageRank(int thread_scope_rank) {
72 switch (thread_scope_rank) {
73 case -1:
74 return StorageRank::kGlobal;
75 case 0:
76 return StorageRank::kShared;
77 case 1:
78 return StorageRank::kLocal;
79 default: {
80 LOG(FATAL) << "unknown rank";
81 return StorageRank::kGlobal;
82 }
83 }
84}

86/*! \brief class to represent storage scope */
87struct StorageScope {
88 /*! \brief The rank of the storage */
89 StorageRank rank{StorageRank::kGlobal};
90 /*! \brief tag for special purpose memory. */
91 std::string tag;
92 // comparator
93 inline bool operator==(const StorageScope& other) const {
94 return rank == other.rank && tag == other.tag;
95 }
96 inline bool operator!=(const StorageScope& other) const { return !(*this == other); }
97 inline std::string to_string() const {
98 std::string ret;
99 switch (rank) {
100 case StorageRank::kGlobal:
101 return "global" + tag;
102 case StorageRank::kShared:
103 return "shared" + tag;
104 case StorageRank::kWarp:
105 return "warp" + tag;
106 case StorageRank::kLocal:
107 return "local" + tag;
108 case StorageRank::kWMMAMatrixA:
109 return "wmma.matrix_a" + tag;
110 case StorageRank::kWMMAMatrixB:
111 return "wmma.matrix_b" + tag;
112 case StorageRank::kWMMAAccumulator:
113 return "wmma.accumulator" + tag;
114 case StorageRank::kTexture:
115 return "texture" + tag;
116 default:
117 LOG(FATAL) << "unknown storage scope";
118 return "";
119 }
120 }
121 /*!
122 * \brief Create storage scope from string
123 * \param s The string to be parsed.
124 * \return The storage scope.
125 */
126 static StorageScope Create(const std::string& s) {
127 StorageScope r;
128 if (s.empty()) {
129 r.rank = StorageRank::kGlobal;
130 } else if (s.compare(0, 6, "global") == 0) {
131 r.rank = StorageRank::kGlobal;
132 r.tag = s.substr(6, std::string::npos);
133 } else if (s.compare(0, 6, "shared") == 0) {
134 r.rank = StorageRank::kShared;
135 r.tag = s.substr(6, std::string::npos);
136 } else if (s.compare(0, 4, "warp") == 0) {
137 r.rank = StorageRank::kWarp;
138 r.tag = s.substr(4, std::string::npos);
139 } else if (s.compare(0, 5, "local") == 0) {
140 r.rank = StorageRank::kLocal;
141 r.tag = s.substr(5, std::string::npos);
142 } else if (s.compare(0, 13, "wmma.matrix_a") == 0) {
143 r.rank = StorageRank::kWMMAMatrixA;
144 r.tag = s.substr(13, std::string::npos);
145 } else if (s.compare(0, 13, "wmma.matrix_b") == 0) {
146 r.rank = StorageRank::kWMMAMatrixB;
147 r.tag = s.substr(13, std::string::npos);
148 } else if (s.compare(0, 16, "wmma.accumulator") == 0) {
149 r.rank = StorageRank::kWMMAAccumulator;
150 r.tag = s.substr(16, std::string::npos);
151 } else if (s.compare(0, 7, "texture") == 0) {
152 r.rank = StorageRank::kTexture;
153 r.tag = s.substr(7, std::string::npos);
154 } else {
155 LOG(FATAL) << "unknown storage scope " << s;
156 }
157 return r;
158 }
159};

161/*! \brief class to represent thread scope */
162struct ThreadScope {
163 /*! \brief The rank of thread scope */
164 int rank{0};
165 /*! \brief the dimension index under the rank */
166 int dim_index{0};
167 /*!
168 * \brief Create storage scope from string
169 * \param s The string to be parsed.
170 * \return The storage scope.
171 */
172 static ThreadScope Create(const std::string& s) {
173 ThreadScope r;
174 if (s.compare(0, 7, "vthread") == 0 || s == "cthread") {
175 // virtual thread at the same level as local
176 r.rank = 1;
177 r.dim_index = -1;
178 } else if (s.compare(0, 9, "blockIdx.") == 0) {
179 r.rank = 0;
180 r.dim_index = static_cast<int>(s[9] - 'x');
181 } else if (s.compare(0, 10, "threadIdx.") == 0) {
182 r.rank = 1;
183 r.dim_index = static_cast<int>(s[10] - 'x');
184 } else {
185 LOG(FATAL) << "Unknown threadscope " << s;
186 }
187 return r;
188 }
189};

191/*! \brief workload specification */
192struct ThreadWorkLoad {
193 // array, first three are thread configuration.
194 size_t work_size[6];
195 // Dynamic shared memory allocation size in bytes.
196 size_t dyn_shmem_size{0};
197 /*!
198 * \param i The block dimension.
199 * \return i-th block dim
200 */
201 inline size_t block_dim(size_t i) const { return work_size[i + 3]; }
202 /*!
203 * \param i The grid dimension.
204 * \return i-th grid dim
205 */
206 inline size_t grid_dim(size_t i) const { return work_size[i]; }
207};
208/*! \brief Launch parameters configuration */
209class LaunchParamConfig {
210 public:
211 void Init(size_t base, const std::vector<std::string>& launch_param_tags) {
212 base_ = base;
213 std::vector<bool> filled(6, false);
214 for (size_t i = 0; i < launch_param_tags.size(); ++i) {
215 const std::string& tag = launch_param_tags[i];
216 if (tag == launch_param::kUseDynamicSharedMemoryTag) {
217 ICHECK_EQ(i, launch_param_tags.size() - 1)
218 << "kUseDynamicSharedMemoryTag should be the last tag in launch_param_tags.";
219 use_dyn_shared_memory_ = true;
220 } else {
221 ThreadScope ts = ThreadScope::Create(tag);
222 arg_index_map_.push_back(ts.rank * 3 + ts.dim_index);
223 filled[ts.rank * 3 + ts.dim_index] = true;
224 }
225 }
226 work_dim_ = 1;
227 for (int i = 0; i < 3; ++i) {
228 if (filled[i] || filled[i + 3]) {
229 work_dim_ = i + 1;
230 }
231 }
232 }
233 // extract workload from arguments.
234 ThreadWorkLoad Extract(TVMArgs x) const {
235 ThreadWorkLoad w;
236 std::fill(w.work_size, w.work_size + 6, 1);
237 for (size_t i = 0; i < arg_index_map_.size(); ++i) {
238 // Dynamic shapes can result in 0 dim size. Guard to ensure that the dim size is at least 1.
239 size_t size = static_cast<size_t>(x.values[base_ + i].v_int64);
240 if (size > 0) {
241 w.work_size[arg_index_map_[i]] = size;
242 }
243 }
244 if (use_dyn_shared_memory_) {
245 w.dyn_shmem_size = static_cast<size_t>(x.values[base_ + arg_index_map_.size()].v_int64);
246 }
247 return w;
248 }
249 // return the work dim
250 size_t work_dim() const { return work_dim_; }
251
252 private:
253 /*! \brief base axis */
254 size_t base_;
255 /*! \brief The worker dimension */
256 size_t work_dim_;
257 /*! \brief The index mapping. */
258 std::vector<uint32_t> arg_index_map_;
259 /*! \brief Whether or not use dynamic shared memory. */
260 bool use_dyn_shared_memory_{false};
261};
