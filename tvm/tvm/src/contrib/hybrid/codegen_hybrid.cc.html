39std::string dot_to_underscore(std::string s) {
40 for (auto& ch : s)
41 if (ch == '.') ch = '_';
42 return s;
43}
44
45std::string CodeGenHybrid::GetUniqueName(std::string prefix) {
46 prefix = dot_to_underscore(prefix);
47 auto it = ids_allocated_.find(prefix);
48 if (it != ids_allocated_.end()) {
49 while (true) {
50 std::ostringstream os;
51 os << prefix << (++it->second);
52 std::string name = os.str();
53 if (ids_allocated_.count(name) == 0) {
54 prefix = name;
55 break;
56 }
57 }
58 }
59 ids_allocated_[prefix] = 0;
60 return prefix;
61}

65void CodeGenHybrid::PrintType(DataType t, std::ostream& os) {
66 if (t.is_float()) {
67 os << "float";
68 ICHECK(t.bits() == 16 || t.bits() == 32 || t.bits() == 64);
69 } else if (t.is_int()) {
70 os << "int";
71 ICHECK(t.bits() == 8 || t.bits() == 16 || t.bits() == 32 || t.bits() == 64);
72 } else if (t.is_bfloat16()) {
73 os << "bfloat";
74 ICHECK(t.bits() == 16);
75 } else {
76 ICHECK(t.is_uint()) << "Unsupported type " << t;
77 os << "uint";
78 ICHECK(t.bits() == 8 || t.bits() == 16 || t.bits() == 32 || t.bits() == 64);
79 }
80 os << t.bits();
81}

95template <typename T>
96inline void PrintBinaryExpr(const T* op, const char* opstr,
97 std::ostream& os, // NOLINT(*)
98 CodeGenHybrid* p) {
99 ICHECK(op->dtype.lanes() == 1) << "vec bin op not implemented";
100 if (isalpha(opstr[0])) {
101 os << opstr << '(';
102 p->PrintExpr(op->a, os);
103 os << ", ";
104 p->PrintExpr(op->b, os);
105 os << ')';
106 } else {
107 os << '(';
108 p->PrintExpr(op->a, os);
109 if (!strcmp(opstr, "&&")) opstr = "and";
110 if (!strcmp(opstr, "||")) opstr = "or";
111 os << ' ' << opstr << ' ';
112 p->PrintExpr(op->b, os);
113 os << ')';
114 }
115}

117inline void PrintBinaryIntrinsitc(const CallNode* op, const char* opstr,
118 std::ostream& os, // NOLINT(*)
119 CodeGenHybrid* p) {
120 ICHECK(op->dtype.lanes() == 1) << "vec bin intrin not implemented";
121 ICHECK_EQ(op->args.size(), 2U);
122 os << '(';
123 p->PrintExpr(op->args[0], os);
124 os << opstr;
125 p->PrintExpr(op->args[1], os);
126 os << ')';
127}

209void CodeGenHybrid::VisitExpr_(const ProducerLoadNode* op, std::ostream& os) { // NOLINT(*)
210 auto tensor = Downcast<Tensor>(op->producer);
211
212 os << GetTensorID(tensor);
213 os << "[";
214 for (size_t i = 0; i < op->indices.size(); ++i) {
215 if (i) os << ", ";
216 std::stringstream idx;
217 PrintExpr(op->indices[i], idx);
218 os << idx.str();
219 }
220 os << "]";
221}
222void CodeGenHybrid::VisitExpr_(const CallNode* op, std::ostream& os) { // NOLINT(*)
223 if (op->op.same_as(builtin::bitwise_and())) {
224 PrintBinaryIntrinsitc(op, "&", os, this);
225 } else if (op->op.same_as(builtin::bitwise_xor())) {
226 PrintBinaryIntrinsitc(op, "^", os, this);
227 } else if (op->op.same_as(builtin::bitwise_or())) {
228 PrintBinaryIntrinsitc(op, "|", os, this);
229 } else if (op->op.same_as(builtin::shift_left())) {
230 PrintBinaryIntrinsitc(op, "<<", os, this);
231 } else if (op->op.same_as(builtin::shift_right())) {
232 PrintBinaryIntrinsitc(op, ">>", os, this);
233 } else if (op->op.same_as(builtin::bitwise_not())) {
234 ICHECK_EQ(op->args.size(), 1U);
235 os << "(~";
236 PrintExpr(op->args[0], os);
237 os << ')';
238 } else if (op->op.same_as(builtin::if_then_else())) {
239 PrintExpr(op->args[1], os);
240 os << " if ";
241 PrintExpr(op->args[0], os);
242 os << " else ";
243 PrintExpr(op->args[2], os);
244 } else if (op->op.same_as(builtin::call_pure_extern()) ||
245 op->op.same_as(builtin::call_extern())) {
246 StringImm fname = Downcast<StringImm>(op->args[0]);
247 os << fname << "(";
248 for (size_t i = 1; i < op->args.size(); i++) {
249 PrintExpr(op->args[i], os);
250 if (i < op->args.size() - 1) {
251 os << ", ";
252 }
253 }
254 os << ")";
255 } else {
256 auto* ptr_op = op->op.as<OpNode>();
257 ICHECK(ptr_op != nullptr);
258 std::string name = ptr_op->name;
259 ICHECK_EQ(name.compare(0, 4, "tir."), 0);
260 os << name.substr(4) << "(";
261 for (size_t i = 0; i < op->args.size(); i++) {
262 PrintExpr(op->args[i], os);
263 if (i < op->args.size() - 1) {
264 os << ", ";
265 }
266 }
267 os << ")";
268 }
269}

316void CodeGenHybrid::VisitStmt_(const AttrStmtNode* op) {
317 if (op->attr_key == tir::attr::thread_extent) {
318 auto iter_var = op->node.as<IterVarNode>();
319 ICHECK(iter_var);
320 binds_[iter_var->var.get()] = dot_to_underscore(iter_var->var->name_hint);
321 PrintIndent();
322 stream << "for " << binds_[iter_var->var.get()] << " in bind('" << iter_var->var->name_hint
323 << "', ";
324 PrintExpr(op->value, stream);
325 stream << "):\n";
326 indent_ += tab_;
327 PrintStmt(op->body);
328 indent_ -= tab_;
329 } else {
330 // For now we ignore the unsupported AttrStmt
331 PrintStmt(op->body);
332 }
333}

335void CodeGenHybrid::VisitStmt_(const ProducerRealizeNode* op) {
336 auto tensor = Downcast<Tensor>(op->producer);
337 if (!op->storage_scope.empty()) {
338 PrintIndent();
339 stream << GetTensorID(tensor) << " = allocate((";
340 for (size_t i = 0; i < op->bounds.size(); ++i) {
341 if (i) stream << ", ";
342 stream << PrintExpr(op->bounds[i]->extent);
343 }
344 if (op->bounds.size() == 1) stream << ", ";
345 stream << "), '";
346 PrintType(tensor->dtype, stream);
347 stream << "', '";
348 stream << op->storage_scope << "')\n";
349 }
350 PrintStmt(op->body);
351}

388bool is_noop(const Stmt& stmt) {
389 if (!stmt.defined()) return true;
390 if (auto eval = stmt.as<EvaluateNode>()) return is_const_int(eval->value);
391 return false;
392}

394void CodeGenHybrid::VisitStmt_(const IfThenElseNode* op) {
395 std::string cond = PrintExpr(op->condition);
396 PrintIndent();
397 stream << "if " << cond << ":\n";
398 indent_ += tab_;
399 PrintStmt(op->then_case);
400 indent_ -= tab_;
401
402 if (!is_noop(op->else_case)) {
403 PrintIndent();
404 stream << "else:\n";
405 indent_ += tab_;
406 PrintStmt(op->else_case);
407 indent_ -= tab_;
408 }
409}

425std::string CodeGenHybrid::GetVarID(const VarNode* v) {
426 if (binds_.count(v)) return binds_[v];
427 auto key = std::make_pair(static_cast<const Object*>(v), 0);
428 if (id_map_.count(key)) {
429 return id_map_[key];
430 }
431 return id_map_[key] = GetUniqueName(v->name_hint);
432}
433
434std::string CodeGenHybrid::GetTensorID(const Tensor& tensor) {
435 auto key = std::make_pair(tensor->op.get(), tensor->value_index);
436 if (id_map_.count(key)) {
437 return id_map_[key];
438 }
439 std::string name_hint = tensor->op->name;
440 if (tensor->op->num_outputs() > 1) {
441 name_hint += "_v" + std::to_string(tensor->value_index);
442 }
443 return id_map_[key] = GetUniqueName(name_hint);
444}

490void CodeGenHybrid::DumpStmt(const Stmt& stmt, const Array<ObjectRef>& inputs,
491 const Array<Tensor>& outputs, const std::string& name) {
492 ReserveKeywords();
493 GetUniqueName(name);
494
495 stream << "def " << name << "(";
496 for (size_t i = 0; i < inputs.size(); ++i) {
497 if (i) stream << ", ";
498 if (auto tensor = inputs[i].as<TensorNode>()) {
499 stream << GetTensorID(GetRef<Tensor>(tensor));
500 } else {
501 auto var = inputs[i].as<VarNode>();
502 ICHECK(var) << "Input should either be a tensor or a variable!";
503 stream << GetVarID(var);
504 }
505 }
506 stream << "):\n";
507 indent_ += tab_;
508 for (size_t i = 0; i < outputs.size(); ++i) {
509 PrintIndent();
510 stream << GetTensorID(outputs[i]) << " = output_tensor((";
511 for (size_t j = 0; j < outputs[i]->shape.size(); ++j) {
512 if (j) stream << ", ";
513 PrintExpr(outputs[i]->shape[j], stream);
514 }
515 if (outputs[i]->shape.size() == 1) stream << ", ";
516 stream << "), '" << outputs[i]->dtype << "')\n";
517 }
518 PrintStmt(stmt);
519 PrintIndent();
520 stream << "return ";
521 for (size_t i = 0; i < outputs.size(); ++i) {
522 if (i) stream << ", ";
523 stream << GetTensorID(outputs[i]);
524 }
525 stream << "\n";
526}

528TVM_REGISTER_GLOBAL("hybrid._Dump").set_body([](TVMArgs args, TVMRetValue* rv) {
529 CodeGenHybrid codegen;
530 if (args.size() == 4)
531 codegen.DumpStmt(args[0], args[1], args[2], args[3]);
532 else
533 codegen.DumpStmt(args[0], args[1], args[2]);
534 *rv = codegen.Finish();
