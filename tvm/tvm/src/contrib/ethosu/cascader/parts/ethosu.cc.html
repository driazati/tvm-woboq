38const std::vector<int64_t> EthosuPartNode::GetBytesRead(const std::vector<int>& block_shape,
39 const std::vector<int>& full_shape) {
40 std::vector<int64_t> bytes_per_input(propagators_.size(), 0);
41
42 std::vector<int> order;
43 std::vector<int> stripes;
44 std::vector<int> offset;
45 std::vector<float> strides;
46 for (size_t i = 0; i < block_shape.size(); i++) {
47 order.push_back(1);
48 stripes.push_back(round_up_divide(full_shape[i], block_shape[i]));
49 offset.push_back(0);
50 strides.push_back(static_cast<float>(block_shape[i]));
51 }
52
53 StripeConfig output_block_config(block_shape, full_shape, strides, order, stripes, offset);
54 auto input_block_configs = CalculateInputStripeConfigs(output_block_config);
55
56 int i = 0;
57 for (const auto& input_block_config : input_block_configs) {
58 std::map<std::vector<int>, int> input_blocks = CountStripes(input_block_config, false);
59 for (const auto& block : input_blocks) {
60 bytes_per_input[i] +=
61 mul_reduce(block.first) * block.second * input_tensors_[i]->GetDataType().bytes();
62 }
63 i++;
64 }
65
66 if (weight_tensor_idx_ != -1) {
67 bytes_per_input[weight_tensor_idx_] *= (stripes[height_idx_] * stripes[width_idx_]);
68 }
69
70 return bytes_per_input;
71}

73float EthosuPartNode::CalculateCost(const BlockConfig& block_config,
74 const StripeConfig& output_stripe_config) {
75 std::vector<int> output_block = block_config->GetOutputBlockShape();
76 std::vector<int> output_stripe_shape = output_stripe_config->GetShape();
77 auto input_stripe_configs = CalculateInputStripeConfigs(output_stripe_config);
78 std::vector<int> input_stripe_shape = input_stripe_configs[0]->GetShape();
79
80 std::vector<int64_t> bytes_per_input = GetBytesRead(output_block, output_stripe_shape);
81 bytes_per_input[0] *= subkernels_;
82
83 // Calculate bytes read per output element
84 float cost =
85 static_cast<float>(bytes_per_input[0] + bytes_per_input[1]) / mul_reduce(output_stripe_shape);
86
87 // Single buffering hardware optimization
88 if (mul_reduce(input_stripe_shape) <= 2 * mul_reduce(block_config->GetInputBlockShape())) {
89 cost /= 2;
90 }
91 return cost;
92}

94const BlockConfig EthosuPartNode::GetBlockConfig(const StripeConfig& output_stripe_config) {
95 BlockConfig best_block_config = valid_block_configs_[0];
96 float best_cost = CalculateCost(best_block_config, output_stripe_config);
97 std::vector<int> output_stripe_shape = output_stripe_config->GetShape();
98 auto input_stripe_configs = CalculateInputStripeConfigs(output_stripe_config);
99 std::vector<int> input_stripe_shape = input_stripe_configs[0]->GetShape();
100
101 for (const auto& block_config : valid_block_configs_) {
102 float relative_cost = CalculateCost(block_config, output_stripe_config);
103 if (relative_cost < best_cost) {
104 best_block_config = block_config;
105 best_cost = relative_cost;
106 }
107 }
108 return best_block_config;
109}

111const PerformanceInfo EthosuPartNode::GetPerformanceInfo(const StripeConfig& output_stripe_config,
112 BufferMode buffer_mode) {
113 BlockConfig block_config = GetBlockConfig(output_stripe_config);
114 std::vector<int> block_shape = block_config->GetOutputBlockShape();
115
116 std::vector<int64_t> bytes_per_input =
117 GetBytesRead(block_shape, output_stripe_config->GetShape());
118
119 float num_blocks = 1.0f;
120 for (size_t i = 0; i < block_shape.size(); i++) {
121 if (buffer_mode == BufferMode::RECOMPUTE) {
122 num_blocks *= std::max(static_cast<float>(output_stripe_config->GetShape()[i]) /
123 block_shape[i] * output_stripe_config->GetStripes()[i],
124 1.0f);
125 } else {
126 num_blocks *=
127 std::max(static_cast<float>(output_tensor_->GetShape()[i]) / block_shape[i], 1.0f);
128 }
129 }
130
131 float num_stripes = mul_reduce(output_stripe_config->GetStripes());
132 std::vector<int64_t> read_bytes;
133 for (int64_t stripe_bytes : bytes_per_input) {
134 read_bytes.push_back(num_stripes * stripe_bytes);
135 }
136 int64_t write_bytes =
137 num_blocks * mul_reduce(block_shape) * output_tensor_->GetDataType().bytes();
138
139 int block_output_cycles = block_config->GetOutputCycles();
140 int block_compute_cycles = block_config->GetComputeCycles();
141
142 int64_t total_cycles = 0;
143 if (block_output_cycles > block_compute_cycles) {
144 total_cycles = (block_output_cycles * num_blocks) + block_compute_cycles;
145 } else {
146 total_cycles = (block_compute_cycles * num_blocks) + block_output_cycles;
147 }
148
149 PerformanceInfo info(total_cycles, read_bytes, write_bytes, block_config);
150 return info;
151}

153EthosuPart::EthosuPart(const TESubgraph& subgraph, const std::vector<Propagator> propagators,
154 const std::vector<int>& output_quantum, int subkernels,
155 const std::vector<BlockConfig>& valid_block_configs, int weight_tensor_idx) {
156 auto n = make_object<EthosuPartNode>();
157 ICHECK_GT(propagators.size(), 0) << "The Part must include at least one Propagator.";
158 n->subgraph_ = subgraph;
159 n->propagators_ = std::move(propagators);
160 n->in_line_ = false;
161 n->input_tensors_.resize(propagators.size());
162 n->output_quantum_ = output_quantum;
163 n->valid_block_configs_ = valid_block_configs;
164 n->subkernels_ = subkernels;
165 n->weight_tensor_idx_ = weight_tensor_idx;
166 if (output_quantum.size() == 5) {
167 // NHCWB16 Format
168 n->height_idx_ = 1;
169 n->width_idx_ = 3;
170 } else {
171 // NHWC Format
172 n->height_idx_ = 1;
173 n->width_idx_ = 2;
174 }
175 data_ = std::move(n);
176}

178TVM_REGISTER_GLOBAL("contrib.ethosu.cascader.EthosuPart")
179 .set_body_typed([](Array<te::Tensor> subgraph_inputs, te::Tensor subgraph_output,
180 Array<Propagator> propagators, Array<Integer> output_quantum, int subkernels,
181 Array<BlockConfig> valid_block_configs, int weight_tensor_idx) {
182 std::vector<te::Tensor> vsubgraph_inputs(subgraph_inputs.begin(), subgraph_inputs.end());
183 std::vector<Propagator> vpropagators(propagators.begin(), propagators.end());
184 std::vector<int> voutput_quantum;
185 std::transform(output_quantum.begin(), output_quantum.end(),
186 std::back_inserter(voutput_quantum),
187 [](auto&& val) { return val.IntValue(); });
188 TESubgraph subgraph;
189 subgraph.input_tensors = vsubgraph_inputs;
190 subgraph.output_tensor = subgraph_output;
191 std::vector<BlockConfig> vvalid_block_configs(valid_block_configs.begin(),
192 valid_block_configs.end());
193 return EthosuPart(subgraph, vpropagators, voutput_quantum, subkernels, vvalid_block_configs,
194 weight_tensor_idx);
195 });
