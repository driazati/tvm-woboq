34const PerformanceInfo InlinePartNode::GetPerformanceInfo(const StripeConfig& output_stripe_config,
35 BufferMode buffer_mode) {
36 std::vector<int64_t> read_bytes(input_tensors_.size());
37 BlockConfig block_config = BlockConfig(std::vector<int>(1, 1), std::vector<int>(1, 1), 0, 0);
38 PerformanceInfo info(0, read_bytes, 0, block_config);
39 return info;
40}
41
42InlinePart::InlinePart(const TESubgraph& subgraph, const std::vector<Propagator> propagators) {
43 auto n = make_object<InlinePartNode>();
44 ICHECK_GT(propagators.size(), 0) << "The Part must include at least one Propagator.";
45 n->subgraph_ = subgraph;
46 n->propagators_ = std::move(propagators);
47 n->in_line_ = true;
48 n->input_tensors_.resize(propagators.size());
49 data_ = std::move(n);
50}
51
52TVM_REGISTER_GLOBAL("contrib.ethosu.cascader.InlinePart")
53 .set_body_typed([](Array<te::Tensor> subgraph_inputs, te::Tensor subgraph_output,
54 Array<Propagator> propagators) {
55 std::vector<te::Tensor> vsubgraph_inputs(subgraph_inputs.begin(), subgraph_inputs.end());
56 std::vector<Propagator> vpropagators(propagators.begin(), propagators.end());
57 TESubgraph subgraph;
58 subgraph.input_tensors = vsubgraph_inputs;
59 subgraph.output_tensor = subgraph_output;
60 return InlinePart(subgraph, vpropagators);
61 });
62
63TVM_REGISTER_NODE_TYPE(InlinePartNode);
