38/*! \brief Node to represent an inlined Part */
39class InlinePartNode : public PartNode {
40 public:
41 /*!
42 * \brief Get the performance information for a given output stripe config.
43 * \param output_stripe_config The output stripe config to compute the performance for.
44 * \param is_rolling Whether the output config should be computed as a rolling buffer.
45 * \return The performance information containing the compute cycles and read/write bytes.
46 */
47 const PerformanceInfo GetPerformanceInfo(const StripeConfig& output_stripe_config,
48 BufferMode buffer_mode) final;
49
50 static constexpr const char* _type_key = "contrib.ethosu.cascader.InlinePart";
51 TVM_DECLARE_FINAL_OBJECT_INFO(InlinePartNode, PartNode);
52
53 protected:
54 friend class InlinePart;
55};
56
57/*!
58 * \brief A class to describe a inlined Part in a Cascader graph.
59 * \note Inlined Parts have a few special properties. First by IsInline being true,
60 * the Cascader will not allocate any space for the outputs of the Part. This is because
61 * they will be directly consumed as they are produced by the following Part. Second, they
62 * are assumed to be 'free' and require no cycles to execute. Lastly, as they are 'free'
63 * the compute quantum is arbitrary, but by convention it is a single tensor element.
64 *
65 * Examples of inline Parts include strided_slice, reshape and concatenate - all of which
66 * get absorbed into the DMA functionality of Ethos-U compute primitives.
67 */
68class InlinePart : public Part {
69 public:
70 InlinePart(const TESubgraph& subgraph, const std::vector<Propagator> propagators);
71
72 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(InlinePart, Part, InlinePartNode);
73};
