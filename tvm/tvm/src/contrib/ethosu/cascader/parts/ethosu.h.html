36namespace ethosu {
37namespace cascader {
38
39/*! \brief Node to represent an EthosuPart */
40class EthosuPartNode : public PartNode {
41 public:
42 /*!
43 * \brief Get the optimal BlockConfig to use given a StripeConfig
44 * \param output_stripe_config The output StripeConfig.
45 */
46 const BlockConfig GetBlockConfig(const StripeConfig& output_stripe_config);
47 /*!
48 * \brief Get the preferred alignment in each axis for a stripe of the Part.
49 * \note This is used to bias the selection of StripeConfigs towards those that are integer
50 * multiples of a tensor intrinsic used to compute the Part.
51 */
52 const std::vector<int> GetStripeAlignHint() const final { return output_quantum_; }
53 /*!
54 * \brief Get the performance information for a given output stripe config.
55 * \param output_stripe_config The output stripe config to compute the performance for.
56 * \param buffer_mode The mode of buffering, rolling or recompute.
57 * \return The performance information containing the compute cycles and read/write bytes.
58 */
59 const PerformanceInfo GetPerformanceInfo(const StripeConfig& output_stripe_config,
60 BufferMode buffer_mode) final;
61
62 static constexpr const char* _type_key = "contrib.ethosu.cascader.EthosuPart";
63 TVM_DECLARE_FINAL_OBJECT_INFO(EthosuPartNode, PartNode);
64
65 protected:
66 friend class EthosuPart;
67
68 /*!
69 * \brief Get the size of input required (per input tensor) to compute a stripe given a
70 * block_shape
71 * \param block_shape The shape of the block(s) the stripe is split into
72 * \param stripe_shape The shape of the full stripe to compute.
73 * \return The bytes required per input tensor.
74 */
75 const std::vector<int64_t> GetBytesRead(const std::vector<int>& block_shape,
76 const std::vector<int>& full_shape);
77
78 /*!
79 * \brief Get cost heuristic of using a given block config with the associated stripe config
80 * \param block_config The block config that is being checked for the cost
81 * \param output_stripe_config The striping configuration associated with the operator
82 * \return A cost heuristic representative of the choice
83 */
84 float CalculateCost(const BlockConfig& block_config, const StripeConfig& output_stripe_config);
85
86 /*! \brief List of block configs that are valid for this part */
87 std::vector<BlockConfig> valid_block_configs_;
88 /*! \brief The output volume that is atomically computed */
89 std::vector<int> output_quantum_;
90 /*! \brief Index for output height dimension */
91 int height_idx_;
92 /*! \brief Index for output width dimension */
93 int width_idx_;
94 /*! \brief Index of weight tensor, -1 if the Part has no weights */
95 int weight_tensor_idx_;
96 /*! \brief Number of sub-kernels the kernel has been split into */
97 int subkernels_;
98};
99
100/*!
101 * \brief A class to describe a Part to be executed on an Arm(R) Ethos(TM)-U NPU.
102 * \note EthosuParts must be provided with an output quantum and the cycles taken to
103 * compute an output quantum which depend on the operator the NPU is computing.
104 */
105class EthosuPart : public Part {
106 public:
107 EthosuPart(const TESubgraph& subgraph, const std::vector<Propagator> propagators,
108 const std::vector<int>& output_quantum, int subkernels,
109 const std::vector<BlockConfig>& valid_block_configs, int weight_tensor_idx);
110
111 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(EthosuPart, Part, EthosuPartNode);
112};
