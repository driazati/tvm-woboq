54 if (val == ConstIntBound::kPosInf) {

73// internal entry for const int bound
74struct ConstIntBoundAnalyzer::Entry {
75 int64_t min_value;
76 int64_t max_value;
77
78 bool is_const(int64_t value) const { return min_value == max_value && min_value == value; }
79
80 bool operator==(const Entry& other) const {
81 return min_value == other.min_value && max_value == other.max_value;
82 }
83};

86 : public ExprFunctor<ConstIntBoundAnalyzer::Entry(const PrimExpr&)> {

88 /*! \brief additional bound info about expr in bound */
89 struct BoundInfo {
90 /*! \brief The expr */
91 PrimExpr expr;
92 /*! \brief The additional bound */
93 Entry bound;

95 BoundInfo() {}

99 void Bind(const Var& var, const Range& range, bool allow_override) {
100 Entry a = VisitExpr(range->min);
101 Entry b = VisitExpr(range->extent);
102 Entry ret;
103 ret.min_value = a.min_value;
104 ret.max_value = InfAwareAdd(a.max_value, InfAwareAdd(b.max_value, -1));
105 Update(var, ret, allow_override);
106 }
107
108 void Update(const Var& var, const Entry& info, bool allow_override) {
109 if (!allow_override) {
110 auto it = var_map_.find(var);
111 if (it != var_map_.end()) {
112 ICHECK(it->second == info)
113 << "Trying to update var \'" << var << "\'"
114 << " with a different const bound: "
115 << "original=" << ConstIntBound(it->second.min_value, it->second.max_value)
116 << ", new=" << ConstIntBound(info.min_value, info.max_value);
117 }
118 }
119 var_map_[var] = info;
120 }
121
122 Entry VisitExpr_(const LetNode* op) final {
123 auto it = var_map_.find(op->var);
124 // if the var has not been binded, update the info.
125 if (it == var_map_.end()) {
126 var_map_[op->var] = this->VisitExpr(op->value);
127 Entry ret = VisitExpr(op->body);
128 var_map_.erase(op->var);
129 return ret;
130 } else {
131 return VisitExpr(op->body);
132 }
133 }
134
135 void Update(const Var& var, const ConstIntBound& info, bool allow_override) {
136 Update(var, MakeBound(info->min_value, info->max_value), allow_override);
137 }

139 // Override visitor behaviors
140 Entry VisitExprDefault_(const Object* op) final {
141 return Everything(static_cast<const PrimExprNode*>(op)->dtype);
142 }
143
144 Entry VisitExpr(const PrimExpr& expr) final {
145 Entry res = ExprFunctor::VisitExpr(expr);
146 tir::ExprDeepEqual equal;
147 // a linear search over additional info
148 // assume we won't have a lot of conditions
149 for (const BoundInfo& info : additional_info_) {
150 if (equal(expr, info.expr)) {
151 res = Intersect(res, info.bound);
152 }
153 }
154 if (bound_) {
155 auto val = bound_->find(expr);
156 if (val != bound_->end()) {
157 auto everything = Everything(expr->dtype);
158 ICHECK(
159 (val->second->min_value == res.min_value && val->second->max_value == res.max_value) ||
160 (val->second->min_value == everything.min_value &&
161 val->second->max_value == everything.max_value))
162 << "Detected bound for " << expr << "conflicts with memorization";
163 }
164 (*bound_)[expr] = ConstIntBound(res.min_value, res.max_value);
165 }
166 return res;
167 }

169 Entry VisitExpr_(const RampNode* op) final {
170 // op = {base + i * stride | 0 <= i < lanes}
171 // Entry(op) = Union(Entry(base + i * stride) | 0 <= i < lanes)
172 // Note that `base + i * stride` is linear w.r.t. `i`
173 // Entry(op) = Union(Entry(base + i * stride) | i = 0, i = lanes-1)
174 Entry a = VisitExpr(op->base);
175 Entry b = VisitExpr(op->base + (op->lanes - 1) * op->stride);
176 return Union(a, b);
177 }
178
179 Entry VisitExpr_(const CastNode* op) final {
180 Entry a;
181
182 // int(ceil(log2(cast(n,"float64")))) is used as the
183 // implementation of topi.math.ceil_log2, and appears in iteration
184 // bounds.
185 if (auto opt = FindCeilLog2Arg(op)) {
186 a = CeilLog2Bounds(opt.value());
187 } else {
188 a = VisitExpr(op->value);
189 }
190
191 Entry b = Everything(op->dtype);
192 return Intersect(a, b);
193 }

195 Entry VisitExpr_(const IntImmNode* op) final { return MakeBound(op->value, op->value); }
196
197 Entry VisitExpr_(const AddNode* op) final {
198 Entry a = VisitExpr(op->a);
199 Entry b = VisitExpr(op->b);
200 Entry ret;
201 ret.min_value = InfAwareAdd(a.min_value, b.min_value);
202 ret.max_value = InfAwareAdd(a.max_value, b.max_value);
203 return ret;
204 }

206 Entry VisitExpr_(const SubNode* op) final {
207 Entry a = VisitExpr(op->a);
208 Entry b = VisitExpr(op->b);
209 Entry ret;
210 ret.min_value = InfAwareAdd(a.min_value, -b.max_value);
211 ret.max_value = InfAwareAdd(a.max_value, -b.min_value);
212 return ret;
213 }

215 Entry VisitExpr_(const MulNode* op) final {
216 Entry a = VisitExpr(op->a);
217 Entry b = VisitExpr(op->b);
218 return BinaryOpBoundary(a, b, InfAwareMul);
219 }
220
221 Entry VisitExpr_(const DivNode* op) final {
222 Entry a = VisitExpr(op->a);
223 Entry b = VisitExpr(op->b);
224 ICHECK(!b.is_const(0)) << "divide by zero";
225 return HandleDivision(a, b, op->dtype, InfAwareDiv);
226 }
227
228 Entry VisitExpr_(const ModNode* op) final {
229 Entry a = VisitExpr(op->a);
230 Entry b = VisitExpr(op->b);
231 if (b.min_value > 0) {
232 int64_t b_max_cap = InfAwareAdd(b.max_value, -1);
233 if (a.min_value >= 0) {
234 // 0 <= [a_min, a_max] < b_min
235 if (a.max_value < b.min_value) return a;
236 // other case, we can get close to 0
237 return MakeBound(0, std::min(a.max_value, b_max_cap));
238 } else {
239 return MakeBound(std::max(a.min_value, -b_max_cap),
240 std::min(std::max(a.max_value, (int64_t)0), b_max_cap));
241 }
242 } else {
243 ICHECK(!b.is_const(0)) << "mod by zero";
244 // mod by negative value is rare,
245 // and we just use the simpliest rule.
246 return Everything(op->dtype);
247 }
248 }
249
250 Entry VisitExpr_(const FloorDivNode* op) final {
251 Entry a = VisitExpr(op->a);
252 Entry b = VisitExpr(op->b);
253 ICHECK(!b.is_const(0)) << "floordiv by zero";
254 return HandleDivision(a, b, op->dtype, InfAwareFloorDiv);
255 }
256
257 Entry VisitExpr_(const FloorModNode* op) final {
258 /* let a / b = x + y, where x is integer, y \in [0, 1)
259 * floormod(a, b) = a - floordiv(a, b) * b
260 * floordiv(a, b) = x
261 * floormod(a, b) = a - floordiv(a, b) * b
262 * = a - x * b
263 * = a - (a / b - y) * b
264 * = a - a + y * b
265 * = y * b
266 * note that 0 <= y < 1
267 * when b > 0, 0 <= b * y < b
268 * 0 <= b * y <= b - 1
269 * when b < 0, b < b * y <= 0
270 * b + 1 <= b * y <= 0
271 * In all cases, min(0, b + 1) <= b * y <= max(0, b - 1)
272 * min(0, b_min + 1) <= b * y <= max(0, b_max - 1)
273 * That is, min(0, b_min + 1) <= floormod(a, b) <= max(0, b_max - 1)
274 */
275 Entry a = VisitExpr(op->a);
276 Entry b = VisitExpr(op->b);
277 if (b.min_value > 0) {
278 int64_t b_max_cap = InfAwareAdd(b.max_value, -1);
279 if (a.min_value >= 0) {
280 // 0 <= [a_min, a_max] < b_min
281 if (a.max_value < b.min_value) return a;
282 // other case, we can get close to 0
283 return MakeBound(0, std::min(a.max_value, b_max_cap));
284 } else {
285 return MakeBound(0, b_max_cap);
286 }
287 } else {
288 ICHECK(!b.is_const(0)) << "floormod by zero";
289 int64_t b_min_cap = InfAwareAdd(b.min_value, 1);
290 int64_t b_max_cap = InfAwareAdd(b.max_value, -1);
291 return Intersect(MakeBound(std::min(static_cast<int64_t>(0), b_min_cap),
292 std::max(static_cast<int64_t>(0), b_max_cap)),
293 Everything(op->dtype));
294 }
295 }

298 Entry a = VisitExpr(op->a);
299 Entry b = VisitExpr(op->b);
300 Entry ret;
301 ret.min_value = std::min(a.min_value, b.min_value);
302 ret.max_value = std::min(a.max_value, b.max_value);
303 return ret;

307 Entry a = VisitExpr(op->a);
308 Entry b = VisitExpr(op->b);
309 Entry ret;
310 ret.min_value = std::max(a.min_value, b.min_value);
311 ret.max_value = std::max(a.max_value, b.max_value);
312 return ret;

315 Entry VisitExpr_(const SelectNode* op) final {
316 Entry a = VisitExpr(op->true_value);
317 Entry b = VisitExpr(op->false_value);

321 Entry VisitExpr_(const CallNode* op) final {
322 // only special handle >> and & which can be
323 // used for index calculation.
324
325 if (op->op.same_as(tir::builtin::shift_right())) {
326 return VisitRightShift(op);
327 } else if (op->op.same_as(tir::builtin::shift_left())) {
328 return VisitLeftShift(op);
329 } else if (op->op.same_as(tir::builtin::bitwise_and())) {
330 return VisitBitwiseAnd(op);
331 } else {
332 return Everything(op->dtype);
333 }
334 }

336 Entry VisitExpr_(const VarNode* op) final {
337 Var v = GetRef<Var>(op);
338 auto it = var_map_.find(v);
339 if (it != var_map_.end()) {
340 return it->second;
341 } else {
342 return Everything(op->dtype);
343 }
344 }

346 Entry VisitExpr_(const SizeVarNode* op) final {
347 SizeVar v = GetRef<SizeVar>(op);
348 auto it = var_map_.find(v);
349 if (it != var_map_.end()) {
350 return it->second;
351 } else {
352 return MakeBound(0, kPosInf);
353 }
354 }
355
356 Entry VisitLeftShift(const CallNode* op) {
357 Entry a = VisitExpr(op->args[0]);
358 Entry b = VisitExpr(op->args[1]);
359
360 if (a.min_value < 0 || b.min_value < 0) {
361 // If either operand can negative, we may run into undefined
362 // behavior for some targets. In these cases, avoid making any
363 // assumptions about the result.
364 return Everything(op->dtype);
365 }
366
367 return BinaryOpBoundary(a, b, InfAwareLeftShift);
368 }
369
370 Entry VisitRightShift(const CallNode* op) {
371 Entry a = VisitExpr(op->args[0]);
372 Entry b = VisitExpr(op->args[1]);
373 return BinaryOpBoundary(a, b, InfAwareRightShift);
374 }
375
376 Entry VisitBitwiseAnd(const CallNode* op) {
377 Entry a = VisitExpr(op->args[0]);
378 Entry b = VisitExpr(op->args[1]);
379 // handle positive index case.
380 if (a.min_value >= 0 && b.min_value >= 0) {
381 return MakeBound(0, std::min(a.max_value, b.max_value));
382 } else {
383 if (b.min_value >= 0) {
384 return MakeBound(0, b.max_value);
385 }
386 if (a.min_value >= 0) {
387 return MakeBound(0, a.max_value);
388 }
389 return Everything(op->dtype);
390 }
391 }

393 std::function<void()> EnterConstraint(const PrimExpr& constraint) {
394 std::vector<BoundInfo> info = DetectBoundInfo(constraint);
395 if (info.size() == 0) return nullptr;
396 size_t old_size = additional_info_.size();
397 additional_info_.insert(additional_info_.end(), info.begin(), info.end());
398 size_t new_size = old_size + info.size();
399 auto frecover = [old_size, new_size, this]() {
400 ICHECK_EQ(additional_info_.size(), new_size);
401 additional_info_.resize(old_size);
402 };
403 return frecover;
404 }

408 // internal variable map
409 std::unordered_map<Var, Entry, ObjectPtrHash, ObjectPtrEqual> var_map_;

411 std::vector<BoundInfo> additional_info_;
412 // look up table for memorization
413 BoundMapType* bound_{nullptr};
414 // constants: the limit value means umlimited
415 // NOTE: kNegInf/kPosInf are used to represent infinity.
416 static const constexpr int64_t kNegInf = ConstIntBound::kNegInf;
417 static const constexpr int64_t kPosInf = ConstIntBound::kPosInf;
418 static_assert(-kNegInf == kPosInf, "invariant of inf");
419 // internal helper functions
420 /*!
421 * \brief Get boundary of binary op who are monotonic wrt to one argument.
422 * \param a The entry of the left operand.
423 * \param b The entry of the right operand.
424 * \param op The operator.
425 * \tparam F the operator function type.
426 * \return The result.
427 */
428 template <typename F>
429 static Entry BinaryOpBoundary(Entry a, Entry b, const F& op) {
430 Entry ret;
431 // The boundary point must be shihft of the original boundary.
432 int64_t v1 = op(a.min_value, b.min_value);
433 int64_t v2 = op(a.max_value, b.max_value);
434 int64_t v3 = op(a.min_value, b.max_value);
435 int64_t v4 = op(a.max_value, b.min_value);
436 ret.min_value = std::min(std::min(std::min(v1, v2), v3), v4);
437 ret.max_value = std::max(std::max(std::max(v1, v2), v3), v4);
438 return ret;
439 }
440 /*!
441 * \brief Get value boundaries of division (e.g. Div or FloorDiv).
442 * \param a The entry of the left operand.
443 * \param b The entry of the right operand.
444 * \param dt The data type of the division operator.
445 * \param op The division operator.
446 * \tparam F the operator function type.
447 * \return The result.
448 */
449 template <typename F>
450 static Entry HandleDivision(Entry a, Entry b, DataType dt, const F& op) {
451 // Here we have a / b.
452 // The largest value of the division will be for the smallest (with
453 // respect to the absolute value) value of b. If the range of b starts
454 // at a negative value and ends at a positive one, narrow it down to
455 // be closer to 0, because BinaryOpBoundary only checks end-points of
456 // the domain ranges.
457
458 // If the range of b contains 0, then some infinity will be involved
459 if (b.min_value <= 0 && 0 <= b.max_value && dt.is_int()) {
460 Entry b_neg = b.min_value < 0 ? MakeBound(b.min_value, -1) : Everything(dt);
461 Entry b_pos = b.max_value > 0 ? MakeBound(1, b.max_value) : Everything(dt);
462
463 Entry e_neg = BinaryOpBoundary(a, b_neg, op);
464 Entry e_pos = BinaryOpBoundary(a, b_pos, op);
465
466 return MakeBound(std::min(e_neg.min_value, e_pos.min_value),
467 std::max(e_neg.max_value, e_pos.max_value));
468 } else if (b.min_value == 0 && dt.is_uint()) {
469 // uints only have one sided bounds
470 Entry assumed_b = MakeBound(1, b.max_value);
471 return BinaryOpBoundary(a, assumed_b, op);
472 }
473 // If the range of b does not have 0, use BinaryOpBoundary.
474 return BinaryOpBoundary(a, b, op);
475 }
476 /*!
477 * \brief Compute x + y, aware of inf.
478 * \param x The left operand.
479 * \param y The right operand.
480 * \return the result.
481 */
482 static int64_t InfAwareAdd(int64_t x, int64_t y) {
483 if (x == kPosInf) {
484 ICHECK(y != kNegInf);
485 return kPosInf;
486 }
487 if (x == kNegInf) {
488 ICHECK(y != kPosInf);
489 return kNegInf;
490 }
491 if (y == kPosInf || y == kNegInf) return y;
492 if (WillOverflow<AddNode>(x, y, kNegInf, kPosInf)) {
493 if (x > 0) return kPosInf;
494 return kNegInf;
495 }
496 return x + y;
497 }
498 /*!
499 * \brief Compute x * y, aware of inf.
500 * \param x The left operand.
501 * \param y The right operand.
502 * \return the result.
503 */
504 static int64_t InfAwareMul(int64_t x, int64_t y) {
505 if (!WillOverflow<MulNode>(x, y, kNegInf, kPosInf)) return x * y;
506 if ((x > 0 && y > 0) || (x < 0 && y < 0)) return kPosInf;
507 return kNegInf;
508 }
509 /*!
510 * \brief Compute x / y, aware of inf.
511 * \param x The left operand.
512 * \param y The right operand.
513 * \return the result.
514 */
515 static int64_t InfAwareDiv(int64_t x, int64_t y) {
516 ICHECK_NE(y, 0);
517 if (x == kPosInf || x == kNegInf) {
518 if (y > 0) return x;
519 return -x;
520 }
521 return x / y;
522 }
523 /*!
524 * \brief Compute floodiv(x, y), aware of inf.
525 * \param x The left operand.
526 * \param y The right operand.
527 * \return the result.
528 */
529 static int64_t InfAwareFloorDiv(int64_t x, int64_t y) {
530 ICHECK_NE(y, 0);
531 if (x == kPosInf || x == kNegInf) {
532 if (y > 0) return x;
533 return -x;
534 }
535 return floordiv(x, y);
536 }
537 /*!
538 * \brief Compute x << y, aware of inf.
539 * \param x The left operand.
540 * \param y The right operand.
541 * \return the result.
542 */
543 static int64_t InfAwareLeftShift(int64_t x, int64_t y) {
544 if (x == kPosInf || x == kNegInf) return x;
545
546 // Can be replaced with std::bit_width in C++20
547 auto bit_width = [](int64_t as_signed) {
548 uint64_t val = std::abs(as_signed);
549 int num_bits = 0;
550 while (val) {
551 ++num_bits;
552 val >>= 1;
553 }
554 return num_bits;
555 };
556 int x_bits = bit_width(x);
557 if (x_bits + y < 64) {
558 return x << y;
559 } else {
560 return kPosInf;
561 }
562 }
563 /*!
564 * \brief Compute x >> y, aware of inf.
565 * \param x The left operand.
566 * \param y The right operand.
567 * \return the result.
568 */
569 static int64_t InfAwareRightShift(int64_t x, int64_t y) {
570 if (x == kPosInf || x == kNegInf) return x;
571 return x >> y;
572 }
573 /*!
574 * \brief Make a new bound entry.
575 */
576 static Entry MakeBound(int64_t min_value, int64_t max_value) {
577 Entry e;
578 e.min_value = (min_value == kPosInf) ? min_value - 1 : min_value;
579 e.max_value = (max_value == kNegInf) ? max_value + 1 : max_value;
580 return e;
581 }
582 /*!
583 * \brief Create union of two sets.
584 * \param a The left operand.
585 * \param b the right operand.
586 */
587 static Entry Union(Entry a, Entry b) {
588 Entry ret;
589 ret.min_value = std::min(a.min_value, b.min_value);
590 ret.max_value = std::max(a.max_value, b.max_value);
591 return ret;
592 }
593 /*!
594 * \brief Create intersect of two sets.
595 * \param a The left operand.
596 * \param b the right operand.
597 */
598 static Entry Intersect(Entry a, Entry b) {
599 Entry ret;
600 ret.min_value = std::max(a.min_value, b.min_value);
601 ret.max_value = std::min(a.max_value, b.max_value);
602 return ret;
603 }
604 /*!
605 * \brief return everything dtype can represent.
606 * \param dtype The data type.
607 * \return Bound that represent everything dtype can represent.
608 */
609 static Entry Everything(DataType dtype) {
610 if (!dtype.is_int() && !dtype.is_uint()) {
611 return MakeBound(kNegInf, kPosInf);
612 }
613 Entry ret;
614 int64_t vbits = dtype.bits() - static_cast<int>(dtype.is_int());
615 if (dtype.is_uint()) {
616 ret.min_value = 0;
617 } else {
618 if (vbits >= 63) {
619 ret.min_value = kNegInf;
620 } else {
621 ret.min_value = -(static_cast<int64_t>(1) << vbits);
622 }
623 }
624 if (vbits >= 63) {
625 ret.max_value = kPosInf;
626 } else {
627 ret.max_value = (static_cast<int64_t>(1) << vbits) - 1;
628 }
629 return ret;
630 }

632 /*!
633 * \brief Detect additional constant bound from cond, if any
634 * \param cond The constraint condition.
635 * \return List of detected bounds.
636 */
637 static std::vector<BoundInfo> DetectBoundInfo(const PrimExpr& cond) {
638 PVar<PrimExpr> x, y;
639 PVar<IntImm> c;
640 if ((x && y).Match(cond)) {
641 auto ret1 = DetectBoundInfo(x.Eval());
642 auto ret2 = DetectBoundInfo(y.Eval());
643 ret1.insert(ret1.end(), ret2.begin(), ret2.end());
644 return ret1;
645 }
646
647 // NOTE: canonical form always use <= or <
648 Entry bound;
649 if ((c <= x).Match(cond)) {
650 bound = MakeBound(c.Eval()->value, kPosInf);
651 } else if ((c < x).Match(cond)) {
652 bound = MakeBound(c.Eval()->value + 1, kPosInf);
653 } else if ((x <= c).Match(cond)) {
654 bound = MakeBound(kNegInf, c.Eval()->value);
655 } else if ((x < c).Match(cond)) {
656 bound = MakeBound(kNegInf, c.Eval()->value - 1);
657 } else if ((x == c).Match(cond) || (c == x).Match(cond)) {
658 bound = MakeBound(c.Eval()->value, c.Eval()->value);
659 } else {
660 return {};
661 }
662
663 // If the conditional is comparing two integers, do not assign a
664 // value to them.
665 if (x.Eval().as<IntImmNode>()) {
666 return {};
667 }
668
669 return {BoundInfo(x.Eval(), bound)};
670 }
671
672 /*!
673 * \brief Extract the argument from int(ceil(log2(arg)))
674 *
675 * This expression is used as the implementation of
676 * topi.math.ceil_log2, and can appear in iteration bounds.
677 */
678 static Optional<PrimExpr> FindCeilLog2Arg(const CastNode* op) {
679 if (op->dtype.is_int()) {
680 if (auto as_call = op->value.as<CallNode>()) {
681 if (as_call->op.same_as(Op::Get("tir.ceil"))) {
682 PrimExpr ceil_arg = as_call->args[0];
683 if (auto arg_call = ceil_arg.as<CallNode>()) {
684 if (arg_call->op.same_as(Op::Get("tir.log2"))) {
685 PrimExpr log_arg = arg_call->args[0];
686 return log_arg;
687 }
688 }
689 }
690 }
691 }
692 return NullOpt;
693 }
694
695 /*! \brief Propagate constraints through ceil(log2(arg))
696 *
697 * Helper function for CastNode visitor
698 */
699 Entry CeilLog2Bounds(PrimExpr arg) {
700 if (auto as_float = arg.as<FloatImmNode>()) {
701 // A cast from int to float may have already been simplified
702 // out. Normally we don't inspect floating-point arguments, but here we can
703 int64_t val = std::ceil(std::log2(as_float->value));
704 return MakeBound(val, val);
705 } else {
706 Entry arg_bounds = VisitExpr(arg);
707 return MakeBound(std::ceil(std::log2(arg_bounds.min_value)),
708 std::ceil(std::log2(arg_bounds.max_value)));
709 }
710 }
711};
