33Stmt IRMutatorWithAnalyzer::VisitStmt_(const ForNode* op) {
34 analyzer_->Bind(op->loop_var, Range::FromMinExtent(op->min, op->extent));
35 return StmtExprMutator::VisitStmt_(op);
36}

38Stmt IRMutatorWithAnalyzer::VisitStmt_(const BlockNode* op) {
39 for (const auto& iter_var : op->iter_vars) {
40 analyzer_->Bind(iter_var->var, iter_var->dom);
41 }
42 return StmtExprMutator::VisitStmt_(op);
43}

45Stmt IRMutatorWithAnalyzer::VisitStmt_(const LetStmtNode* op) {
46 PrimExpr value = this->VisitExpr(op->value);
47 if (SideEffect(value) <= CallEffectKind::kPure) {
48 analyzer_->Bind(op->var, value);
49 }
50 // We keep the let-binding here
51 // as sub-class may or maynot choose to replace it.
52 Stmt body = this->VisitStmt(op->body);
53 if (value.same_as(op->value) && body.same_as(op->body)) {
54 return GetRef<Stmt>(op);
55 } else {
56 auto n = this->CopyOnWrite(op);
57 n->value = std::move(value);
58 n->body = std::move(body);
59 return Stmt(n);
60 }
61}

63Stmt IRMutatorWithAnalyzer::VisitStmt_(const IfThenElseNode* op) {
64 PrimExpr condition = this->VisitExpr(op->condition);
65 PrimExpr real_condition = condition;
66 static auto op_likely = Op::Get("tir.likely");
67
68 if (auto call = condition.as<CallNode>()) {
69 if (call->op.same_as(op_likely)) {
70 real_condition = call->args[0];
71 }
72 }
73
74 Stmt then_case, else_case;
75 {
76 With<ConstraintContext> ctx(analyzer_, real_condition);
77 then_case = this->VisitStmt(op->then_case);
78 }
79 if (op->else_case.defined()) {
80 With<ConstraintContext> ctx(analyzer_, analyzer_->rewrite_simplify(Not(real_condition)));
81 else_case = this->VisitStmt(op->else_case);
82 }
83 if (is_one(real_condition)) return then_case;
84 if (is_zero(real_condition)) {
85 if (else_case.defined()) {
86 return else_case;
87 }
88 return Evaluate(0);
89 }
90
91 if (condition.same_as(op->condition) && then_case.same_as(op->then_case) &&
92 else_case.same_as(op->else_case)) {
93 return GetRef<Stmt>(op);
94 } else {
95 auto n = this->CopyOnWrite(op);
96 n->condition = std::move(condition);
97 n->then_case = std::move(then_case);
98 n->else_case = std::move(else_case);
99 return Stmt(n);
100 }
101}

103Stmt IRMutatorWithAnalyzer::VisitStmt_(const AttrStmtNode* op) {
104 if (op->attr_key == tir::attr::thread_extent || op->attr_key == tir::attr::virtual_thread) {
105 IterVar iv = Downcast<IterVar>(op->node);
106 ICHECK_NE(iv->thread_tag.length(), 0U);
107 analyzer_->Bind(iv->var, Range::FromMinExtent(0, op->value));
108 Stmt stmt = StmtExprMutator::VisitStmt_(op);
109 return stmt;
110 } else {
111 return StmtExprMutator::VisitStmt_(op);
112 }
113}

115Stmt IRMutatorWithAnalyzer::VisitStmt_(const AssertStmtNode* op) {
116 PrimExpr condition = this->VisitExpr(op->condition);
117 PrimExpr message = this->VisitExpr(op->message);
118 With<ConstraintContext> ctx(analyzer_, condition);
119 Stmt body = this->VisitStmt(op->body);
120
121 if (condition.same_as(op->condition) && message.same_as(op->message) && body.same_as(op->body)) {
122 return GetRef<Stmt>(op);
123 } else {
124 auto n = this->CopyOnWrite(op);
125 n->condition = std::move(condition);
126 n->message = std::move(message);
127 n->body = std::move(body);
128 return Stmt(n);
129 }
130}

132PrimExpr IRMutatorWithAnalyzer::VisitExpr_(const CallNode* op) {
133 // add condition context to if_then_else
134 static auto op_if_then_else = Op::Get("tir.if_then_else");
135 if (op->op.same_as(op_if_then_else)) {
136 PrimExpr cond = this->VisitExpr(op->args[0]);
137 PrimExpr true_value, false_value;
138 {
139 With<ConstraintContext> constraint(analyzer_, cond);
140 true_value = this->VisitExpr(op->args[1]);
141 }
142 {
143 With<ConstraintContext> constraint(analyzer_, analyzer_->rewrite_simplify(Not(cond)));
144 false_value = this->VisitExpr(op->args[2]);
145 }
146 if (is_zero(cond)) {
147 return false_value;
148 }
149 if (is_one(cond)) {
150 return true_value;
151 }
152 if (cond.same_as(op->args[0]) && true_value.same_as(op->args[1]) &&
153 false_value.same_as(op->args[2])) {
154 return GetRef<PrimExpr>(op);
155 } else {
156 return Call(op->dtype, op->op, {cond, true_value, false_value});
157 }
158 }
159 return StmtExprMutator::VisitExpr_(op);
160}

162PrimExpr IRMutatorWithAnalyzer::VisitExpr_(const LetNode* op) {
163 PrimExpr value = this->VisitExpr(op->value);
164 if (SideEffect(value) <= CallEffectKind::kPure) {
165 analyzer_->Bind(op->var, value);
166 }
167 // We keep the let-binding here
168 // as sub-class may or maynot choose to replace it.
169 PrimExpr body = this->VisitExpr(op->body);
170 if (value.same_as(op->value) && body.same_as(op->body)) {
171 return GetRef<PrimExpr>(op);
172 } else {
173 return Let(op->var, value, body);
174 }
175}

177PrimExpr IRMutatorWithAnalyzer::VisitExpr_(const SelectNode* op) {
178 PrimExpr cond = this->VisitExpr(op->condition);
179 PrimExpr true_value, false_value;
180 {
181 With<ConstraintContext> constraint(analyzer_, cond);
182 true_value = VisitExpr(op->true_value);
183 }
184 {
185 With<ConstraintContext> constraint(analyzer_, analyzer_->rewrite_simplify(Not(cond)));
186 false_value = VisitExpr(op->false_value);
187 }
188 if (is_zero(cond)) {
189 return false_value;
190 }
191 if (is_one(cond)) {
192 return true_value;
193 }
194 // normal path
195 if (cond.same_as(op->condition) && true_value.same_as(op->true_value) &&
196 false_value.same_as(op->false_value)) {
197 return GetRef<PrimExpr>(op);
198 } else {
199 return Select(cond, true_value, false_value);
200 }
201}

203PrimExpr IRMutatorWithAnalyzer::VisitExpr_(const ReduceNode* op) {
204 // Setup the domain information before simplification.
205 for (const IterVar& iv : op->axis) {
206 analyzer_->Bind(iv->var, iv->dom);
207 }
208 // Recursively call simplification when necessary.
209 return StmtExprMutator::VisitExpr_(op);
210}
