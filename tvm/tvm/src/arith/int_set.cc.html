62IntervalSet Intersect(Analyzer* analyzer, IntervalSet a, IntervalSet b) {
63 PrimExpr max_value = min(a->max_value, b->max_value);
64 PrimExpr min_value = max(a->min_value, b->min_value);
65 if ((max_value.dtype().is_int() || max_value.dtype().is_uint()) &&
66 (min_value.dtype().is_int() || min_value.dtype().is_uint()) &&
67 analyzer->CanProve(max_value < min_value)) {
68 return IntervalSet::Empty();
69 } else {
70 return IntervalSet(min_value, max_value);
71 }
72}
73
74IntervalSet Union(Analyzer* analyzer, IntervalSet a, IntervalSet b) {
75 if (a->IsEmpty()) return b;
76 if (b->IsEmpty()) return a;
77 PrimExpr max_value = max(a->max_value, b->max_value);
78 PrimExpr min_value = min(a->min_value, b->min_value);
79 return IntervalSet(min_value, max_value);
80}

104/*!
105 * \brief Combine two interval set under arithmetic operations.
106 * \note this can possibly relax the set.
107 */
108template <typename Op>
109inline IntervalSet Combine(Analyzer* analyzer, IntervalSet a, IntervalSet b, DataType dtype) {
110 if (a->IsSinglePoint() && b->IsSinglePoint()) {
111 PrimExpr expr;
112 if (auto res = TryConstFold<Op>(a->min_value, b->min_value)) {
113 expr = res.value();
114 } else {
115 expr = Op(a->min_value, b->min_value);
116 }
117 return IntervalSet::SinglePoint(expr);
118 }
119 if (is_logical_op<Op>::value) {
120 return IntervalSet(make_const(dtype, 0), make_const(dtype, 1));
121 }
122 if (a->IsEmpty()) return a;
123 if (b->IsEmpty()) return b;
124 if (a->IsEverything()) return a;
125 if (b->IsEverything()) return b;
126 return IntervalSet::Everything();
127}

129template <>
130inline IntervalSet Combine<tir::Add>(Analyzer* analyer, IntervalSet a, IntervalSet b,
131 DataType /* dtype */) {
132 if (a->IsSinglePoint() && b->IsSinglePoint()) {
133 return IntervalSet::SinglePoint(a->min_value + b->min_value);
134 }
135 if (a->IsEmpty()) return a;
136 if (b->IsEmpty()) return b;
137 PrimExpr min_value =
138 a->HasLowerBound() && b->HasLowerBound() ? a->min_value + b->min_value : neg_inf();
139 PrimExpr max_value =
140 a->HasUpperBound() && b->HasUpperBound() ? a->max_value + b->max_value : pos_inf();
141 return IntervalSet(min_value, max_value);
142}

144template <>
145inline IntervalSet Combine<tir::Sub>(Analyzer* analyer, IntervalSet a, IntervalSet b,
146 DataType /* dtype */) {
147 if (a->IsSinglePoint() && b->IsSinglePoint()) {
148 return IntervalSet::SinglePoint(a->min_value - b->min_value);
149 }
150 if (a->IsEmpty()) return a;
151 if (b->IsEmpty()) return b;
152 PrimExpr min_value =
153 a->HasLowerBound() && b->HasUpperBound() ? a->min_value - b->max_value : neg_inf();
154 PrimExpr max_value =
155 a->HasUpperBound() && b->HasLowerBound() ? a->max_value - b->min_value : pos_inf();
156 return IntervalSet(min_value, max_value);
157}

159template <>
160inline IntervalSet Combine<tir::Mul>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
161 DataType /* dtype */) {
162 if (a->IsSinglePoint() && b->IsSinglePoint()) {
163 return IntervalSet::SinglePoint(a->min_value * b->min_value);
164 }
165 if (a->IsEmpty()) return a;
166 if (b->IsEmpty()) return b;
167 if (a->IsSinglePoint()) {
168 std::swap(a, b);
169 }
170 if (b->IsSinglePoint()) {
171 if (is_zero(b->min_value)) return b;
172 if (is_one(b->min_value)) return a;
173 if (analyzer->CanProveGreaterEqual(b->min_value, 0)) {
174 PrimExpr min_value = a->HasLowerBound() ? a->min_value * b->min_value : neg_inf();
175 PrimExpr max_value = a->HasUpperBound() ? a->max_value * b->min_value : pos_inf();
176 return IntervalSet(min_value, max_value);
177 } else if (analyzer->CanProveGreaterEqual(-b->min_value, 1)) {
178 PrimExpr min_value = a->HasUpperBound() ? a->max_value * b->min_value : neg_inf();
179 PrimExpr max_value = a->HasLowerBound() ? a->min_value * b->min_value : pos_inf();
180 return IntervalSet(min_value, max_value);
181 } else if (a->HasUpperBound() && a->HasLowerBound()) {
182 using tir::Select;
183 PrimExpr sign = b->min_value >= make_zero(b->min_value.dtype().element_of());
184 PrimExpr e1 = a->min_value * b->min_value;
185 PrimExpr e2 = a->max_value * b->min_value;
186 return IntervalSet(Select(sign, e1, e2), Select(sign, e2, e1));
187 }
188 }
189 DLOG(WARNING) << "Return Everything in CombineInterval Mul";
190 return IntervalSet::Everything();
191}

193template <>
194inline IntervalSet Combine<tir::Div>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
195 DataType /* dtype */) {
196 if (a->IsSinglePoint() && b->IsSinglePoint()) {
197 return IntervalSet::SinglePoint(a->min_value / b->min_value);
198 }
199 if (a->IsEmpty()) return a;
200 if (b->IsEmpty()) return b;
201 if (b->IsSinglePoint()) {
202 if (is_zero(b->min_value)) {
203 LOG(FATAL) << "Divide by zero in CombineInterval Div";
204 }
205 if (is_one(b->min_value)) return a;
206 // no relaxation is needed in here due to set is inclusive
207 if (analyzer->CanProveGreaterEqual(b->min_value, 0)) {
208 PrimExpr min_value = a->HasLowerBound() ? a->min_value / b->min_value : neg_inf();
209 PrimExpr max_value = a->HasUpperBound() ? a->max_value / b->min_value : pos_inf();
210 return IntervalSet(min_value, max_value);
211 } else if (analyzer->CanProveGreaterEqual(-b->min_value, 1)) {
212 PrimExpr min_value = a->HasUpperBound() ? a->max_value / b->min_value : neg_inf();
213 PrimExpr max_value = a->HasLowerBound() ? a->min_value / b->min_value : pos_inf();
214 return IntervalSet(min_value, max_value);
215 } else if (a->HasUpperBound() && a->HasLowerBound()) {
216 using tir::Select;
217 PrimExpr sign = b->min_value >= make_zero(b->min_value.dtype().element_of());
218 PrimExpr e1 = a->min_value / b->min_value;
219 PrimExpr e2 = a->max_value / b->min_value;
220 return IntervalSet(Select(sign, e1, e2), Select(sign, e2, e1));
221 }
222 }
223 DLOG(WARNING) << "Return Everything in CombineInterval Div";
224 return IntervalSet::Everything();
225}

227template <>
228inline IntervalSet Combine<tir::Mod>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
229 DataType /* dtype */) {
230 if (a->IsSinglePoint() && b->IsSinglePoint()) {
231 return IntervalSet::SinglePoint(truncmod(a->min_value, b->min_value));
232 }
233 if (a->IsEmpty()) return a;
234 if (b->IsEmpty()) return b;
235
236 if (b->IsSinglePoint()) {
237 const PrimExpr& divisor = b->min_value;
238 if (is_zero(divisor)) {
239 LOG(FATAL) << "Modular by zero in CombineInterval Mod";
240 }
241 // We need to add more bound constraints throughout the code.
242 // The logic below assumes a is non-negative, which usually
243 // is the case of our application.
244 // TODO(tqchen): add bound constraints for a.
245 if (analyzer->CanProveGreaterEqual(divisor, 0)) {
246 return IntervalSet(make_zero(divisor.dtype()), divisor - 1);
247 } else {
248 PrimExpr bound = abs(divisor) - 1;
249 return IntervalSet(-bound, bound);
250 }
251 }
252 DLOG(WARNING) << "Return Everything in CombineInterval Mod";
253 return IntervalSet::Everything();
254}

256template <>
257inline IntervalSet Combine<tir::FloorDiv>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
258 DataType /* dtype */) {
259 if (a->IsSinglePoint() && b->IsSinglePoint()) {
260 return IntervalSet::SinglePoint(floordiv(a->min_value, b->min_value));
261 }
262 if (a->IsEmpty()) return a;
263 if (b->IsEmpty()) return b;
264 if (b->IsSinglePoint()) {
265 if (is_zero(b->min_value)) {
266 LOG(FATAL) << "Divide by zero in CombineInterval Div";
267 }
268 if (is_one(b->min_value)) return a;
269 // no relaxation is needed in here due to set is inclusive
270 if (analyzer->CanProveGreaterEqual(b->min_value, 0)) {
271 PrimExpr min_value = a->HasLowerBound() ? floordiv(a->min_value, b->min_value) : neg_inf();
272 PrimExpr max_value = a->HasUpperBound() ? floordiv(a->max_value, b->min_value) : pos_inf();
273 return IntervalSet(min_value, max_value);
274 } else if (analyzer->CanProveGreaterEqual(-b->min_value, 1)) {
275 PrimExpr min_value = a->HasUpperBound() ? floordiv(a->max_value, b->min_value) : neg_inf();
276 PrimExpr max_value = a->HasLowerBound() ? floordiv(a->min_value, b->min_value) : pos_inf();
277 return IntervalSet(min_value, max_value);
278 } else if (a->HasUpperBound() && a->HasLowerBound()) {
279 using tir::Select;
280 PrimExpr sign = b->min_value >= make_zero(b->min_value.dtype().element_of());
281 PrimExpr e1 = floordiv(a->min_value, b->min_value);
282 PrimExpr e2 = floordiv(a->max_value, b->min_value);
283 return IntervalSet(Select(sign, e1, e2), Select(sign, e2, e1));
284 }
285 }
286 DLOG(WARNING) << "Return Everything in CombineInterval Div";
287 return IntervalSet::Everything();

290template <>
291inline IntervalSet Combine<tir::FloorMod>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
292 DataType /* dtype */) {
293 if (a->IsSinglePoint() && b->IsSinglePoint()) {
294 return IntervalSet::SinglePoint(floormod(a->min_value, b->min_value));
295 }
296 if (a->IsEmpty()) return a;
297 if (b->IsEmpty()) return b;
298
299 if (b->IsSinglePoint()) {
300 const PrimExpr& divisor = b->min_value;
301 if (is_zero(divisor)) {
302 LOG(FATAL) << "Modular by zero in CombineInterval Mod";
303 }
304 if (analyzer->CanProveGreaterEqual(divisor, 0)) {
305 if (divisor.as<tir::IntImmNode>()) {
306 // a mod b = a - (a / b) * b if a_max / b == a_min / b
307 auto qmax = a->HasUpperBound() ? floordiv(a->max_value, divisor) : pos_inf();
308 auto qmin = a->HasLowerBound() ? floordiv(a->min_value, divisor) : neg_inf();
309 // We can compare +/- inf against each other, but cannot use
310 // operator== between the symbolic limits and an integer.
311 bool compatible_dtypes = !(qmin.dtype().is_handle() ^ qmax.dtype().is_handle());
312 if (compatible_dtypes && analyzer->CanProve(qmax == qmin)) {
313 auto tmax = a->max_value - divisor * qmin;
314 auto tmin = a->min_value - divisor * qmin;
315 return IntervalSet(tmin, tmax);
316 }
317 }
318 return IntervalSet(make_zero(divisor.dtype()), divisor - 1);
319 } else {
320 PrimExpr bound = abs(divisor) - 1;
321 return IntervalSet(-bound, bound);
322 }
323 }
324 DLOG(WARNING) << "Return Everything in CombineInterval Mod";
325 return IntervalSet::Everything();
326}

361// Simplified version of int set evaluator that operates on IntervalSet
362// We might use better set analysis in the future to replace the intervalset.
363class IntervalSetEvaluator : public ExprFunctor<IntervalSet(const PrimExpr&)> {
364 public:
365 IntervalSetEvaluator(Analyzer* analyzer, const Map<Var, IntSet>& dom_map, bool eval_vec = false)
366 : analyzer_(analyzer), dom_map_(dom_map), eval_vec_(eval_vec) {}
367
368 IntervalSet Eval(const PrimExpr& val) { return this->VisitExpr(val); }
369 // evaluate and relax the set
370 IntervalSet Eval(IntervalSet val) {
371 // avoid recursive indefinite recursive expansion.
372 if (static_cast<size_t>(recur_depth_) >= dom_map_.size()) return val;
373 ++recur_depth_;
374 IntervalSet min_set = this->Eval(val->min_value);
375 IntervalSet max_set = this->Eval(val->max_value);
376 --recur_depth_;
377 return IntervalSet(min_set->min_value, max_set->max_value);
378 }
379
380 IntervalSet VisitExpr_(const IntImmNode* op) final {
381 return IntervalSet::SinglePoint(GetRef<PrimExpr>(op));
382 }
383
384 IntervalSet VisitExpr_(const VarNode* op) final {
385 Var var = GetRef<Var>(op);
386 auto it = dom_map_.find(var);
387 if (it != dom_map_.end()) {
388 IntervalSet res = ToIntervalSet((*it).second);
389 if (res->min_value.same_as(var) && res->max_value.same_as(var)) {
390 return res;
391 }
392 // recursively evaluate mapped result
393 // in case the domain contains variables to be relaxed.
394 return Eval(res);
395 } else {
396 return IntervalSet::SinglePoint(var);
397 }
398 }
399
400 IntervalSet VisitExpr_(const AddNode* op) final { return VisitBinaryExpr_<Add>(op); }
401
402 IntervalSet VisitExpr_(const SubNode* op) final { return VisitBinaryExpr_<Sub>(op); }
403
404 IntervalSet VisitExpr_(const MulNode* op) final { return VisitBinaryExpr_<Mul>(op); }
405
406 IntervalSet VisitExpr_(const DivNode* op) final { return VisitBinaryExpr_<Div>(op); }
407
408 IntervalSet VisitExpr_(const ModNode* op) final { return VisitBinaryExpr_<Mod>(op); }
409
410 IntervalSet VisitExpr_(const FloorDivNode* op) final { return VisitBinaryExpr_<FloorDiv>(op); }
411
412 IntervalSet VisitExpr_(const FloorModNode* op) final { return VisitBinaryExpr_<FloorMod>(op); }
413
414 IntervalSet VisitExpr_(const MinNode* op) final { return VisitBinaryExpr_<Min>(op); }
415
416 IntervalSet VisitExpr_(const MaxNode* op) final { return VisitBinaryExpr_<Max>(op); }
417
418 IntervalSet VisitExpr_(const EQNode* op) final { return VisitBinaryExpr_<EQ>(op); }
419
420 IntervalSet VisitExpr_(const NENode* op) final { return VisitBinaryExpr_<NE>(op); }
421
422 IntervalSet VisitExpr_(const LTNode* op) final { return VisitBinaryExpr_<LT>(op); }
423
424 IntervalSet VisitExpr_(const LENode* op) final { return VisitBinaryExpr_<LE>(op); }
425
426 IntervalSet VisitExpr_(const GTNode* op) final { return VisitBinaryExpr_<GT>(op); }
427
428 IntervalSet VisitExpr_(const GENode* op) final { return VisitBinaryExpr_<GE>(op); }
429
430 IntervalSet VisitExpr_(const AndNode* op) final { return VisitBinaryExpr_<And>(op); }
431
432 IntervalSet VisitExpr_(const OrNode* op) final { return VisitBinaryExpr_<Or>(op); }
433
434 IntervalSet VisitExpr_(const RampNode* op) final {
435 ICHECK(eval_vec_);
436 IntervalSet base = Eval(op->base);
437 PVar<IntImm> stride;
438 if (stride.Match(op->stride)) {
439 DataType t = op->base.dtype();
440 int64_t vstride = stride.Eval()->value;
441 if (vstride > 0) {
442 return Combine<Add>(analyzer_, base,
443 IntervalSet(make_zero(t), make_const(t, vstride * (op->lanes - 1))),
444 op->dtype);
445 } else {
446 return Combine<Add>(analyzer_, base,
447 IntervalSet(make_const(t, vstride * (op->lanes - 1)), make_zero(t)),
448 op->dtype);
449 }
450 }
451 DLOG(WARNING) << "cannot evaluate set on expression " << GetRef<PrimExpr>(op);
452 return IntervalSet::Everything();
453 }
454
455 IntervalSet VisitExpr_(const BroadcastNode* op) final {
456 ICHECK(eval_vec_);
457 return VisitExpr(op->value);
458 }
459
460 IntervalSet VisitExpr_(const SelectNode* op) final {
461 IntervalSet true_set = this->Eval(op->true_value);
462 IntervalSet false_set = this->Eval(op->false_value);
463 return Union(analyzer_, false_set, true_set);
464 }
465
466 IntervalSet VisitExpr_(const CastNode* op) final {
467 IntervalSet value_set = this->Eval(op->value);
468 PrimExpr min_value =
469 value_set->HasLowerBound() ? cast(op->dtype, value_set->min_value) : neg_inf();
470 PrimExpr max_value =
471 value_set->HasUpperBound() ? cast(op->dtype, value_set->max_value) : pos_inf();
472 return IntervalSet(min_value, max_value);
473 }
474
475 IntervalSet VisitExpr_(const BufferLoadNode* op) final {
476 if (!(op->dtype.is_int() || op->dtype.is_uint())) {
477 DLOG(WARNING) << "cannot evaluate set BufferLoad which loads from a " << op->dtype
478 << " buffer";
479 return IntervalSet::Everything();
480 }
481 // If the indices do not contain any variables to be relaxed, return the BufferLoad itself.
482 // Otherwise return `IntervalSet::everything()` since we have no knowledge on the buffer data.
483 for (const PrimExpr& index : op->indices) {
484 if (UsesVar(index, [dom_map = &this->dom_map_](const VarNode* var) {
485 return dom_map->find(GetRef<Var>(var)) != dom_map->end();
486 })) {
487 return IntervalSet::Everything();
488 }
489 }
490 return IntervalSet::SinglePoint(GetRef<PrimExpr>(op));
491 }
492
493 IntervalSet VisitExprDefault_(const Object* op) final {
494 DLOG(WARNING) << "cannot evaluate set type " << op->GetTypeKey();
495 return IntervalSet::Everything();
496 }
497
498 private:
499 // whether set is exactly single point that equals value.
500 bool MatchPoint(const IntervalSet& set, const PrimExpr& value) const {
501 return set->min_value.same_as(value) && set->max_value.same_as(value);
502 }
503
504 template <typename TOp, typename T>
505 inline IntervalSet VisitBinaryExpr_(const T* op) {
506 static_assert(std::is_same<typename TOp::ContainerType, T>::value, "constraint");
507 IntervalSet a = this->Eval(op->a);
508 IntervalSet b = this->Eval(op->b);
509 if (MatchPoint(a, op->a) && MatchPoint(b, op->b)) {
510 return IntervalSet::SinglePoint(GetRef<PrimExpr>(op));
511 }
512 return Combine<TOp>(analyzer_, a, b, op->dtype);
513 }
514
515 // recursive depth
516 int recur_depth_{0};
517 // analyzer
518 Analyzer* analyzer_;
519 const Map<Var, IntSet>& dom_map_;
520 bool eval_vec_{false};
521};

582void IntSetAnalyzer::Impl::Update(const Var& var, const IntSet& info, bool can_override) {
583 if (!can_override) {
584 auto it = dom_map_.find(var);
585 if (it != dom_map_.end()) {
586 const IntSet& old_info = (*it).second;
587
588 ICHECK(ExprDeepEqual()(old_info.min(), info.min()))
589 << "Trying to update var \'" << var << "\'"
590 << " with a different minimum value: "
591 << "original=" << old_info.min() << ", new=" << info.min();
592
593 ICHECK(ExprDeepEqual()(old_info.max(), info.max()))
594 << "Trying to update var \'" << var << "\'"
595 << " with a different maximum value: "
596 << "original=" << old_info.max() << ", new=" << info.max();
597 }
598 }
599 dom_map_.Set(var, info);

606Map<Var, IntSet> IntSetAnalyzer::Impl::GetCurrentBounds() const {
607 // If either constraints_ or dom_map_ is empty, return the other to
608 // avoid constructing a new map.
609 if (constraints_.empty()) {
610 return dom_map_;
611 } else if (dom_map_.empty()) {
612 return constraints_;
613 }
614
615 // If neither is empty, construct a merged domain map with
616 // information from both sources.
617 Map<Var, IntSet> merged = dom_map_;
618 for (const auto& pair : constraints_) {
619 auto it = merged.find(pair.first);
620 if (it == merged.end()) {
621 merged.Set(pair.first, pair.second);
622 } else {
623 merged.Set(pair.first, Intersect({pair.second, (*it).second}));
624 }
625 }
626 return merged;
627}

629std::vector<std::pair<Var, IntSet>> IntSetAnalyzer::Impl::DetectBoundInfo(
630 const PrimExpr& constraint) {
631 PVar<Var> x;
632 PVar<PrimExpr> limit;
633
634 std::vector<std::pair<Var, IntSet>> bounds;
635 for (const PrimExpr& subconstraint : ExtractConstraints(constraint)) {
636 if ((x <= limit).Match(subconstraint)) {
637 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval())});
638 } else if ((x < limit).Match(subconstraint)) {
639 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval() - 1)});
640 } else if ((x >= limit).Match(subconstraint)) {
641 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval(), SymbolicLimits::pos_inf_)});
642 } else if ((x > limit).Match(subconstraint)) {
643 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval() + 1, SymbolicLimits::pos_inf_)});
644 } else if ((x == limit).Match(subconstraint)) {
645 bounds.push_back({x.Eval(), IntSet::SinglePoint(limit.Eval())});
646 }
647
648 if ((limit >= x).Match(subconstraint)) {
649 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval())});
650 } else if ((limit > x).Match(subconstraint)) {
651 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval() - 1)});
652 } else if ((limit <= x).Match(subconstraint)) {
653 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval(), SymbolicLimits::pos_inf_)});
654 } else if ((limit < x).Match(subconstraint)) {
655 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval() + 1, SymbolicLimits::pos_inf_)});
656 } else if ((limit == x).Match(subconstraint)) {
657 bounds.push_back({x.Eval(), IntSet::SinglePoint(limit.Eval())});
658 }
659 }
660 return bounds;
661}

664 return impl_->EnterConstraint(constraint);

666
667std::function<void()> IntSetAnalyzer::Impl::EnterConstraint(const PrimExpr& constraint) {
668 Map<Var, IntSet> cached_values;
669
670 auto bounds = DetectBoundInfo(constraint);
671
672 if (bounds.size() == 0) return nullptr;
673
674 // Collect the current values of each var that is changes by this
675 // constraint.
676 for (const auto& pair : bounds) {
677 auto it = constraints_.find(pair.first);
678 if (it == constraints_.end()) {
679 cached_values.Set(pair.first, IntSet());
680 } else {
681 cached_values.Set(pair.first, (*it).second);
682 }
683 }
684
685 // Update all constraints
686 for (const auto& pair : bounds) {
687 auto it = constraints_.find(pair.first);
688 if (it == constraints_.end()) {
689 constraints_.Set(pair.first, pair.second);
690 } else {
691 constraints_.Set(pair.first, Intersect({pair.second, (*it).second}));
692 }
693 }
694
695 auto frecover = [cached_values, this]() {
696 for (const auto& it : cached_values) {
697 if (it.second.defined()) {
698 constraints_.Set(it.first, it.second);
699 } else {
700 constraints_.erase(it.first);
701 }
702 }
703 };
704 return frecover;
705}

882IntSet UnionLowerBound(const Array<IntSet>& sets) {
883 if (sets.size() == 0) return IntSet::Nothing();
884 if (sets.size() == 1) return sets[0];
885 Analyzer analyzer;
886 bool is_first_interval = true;
887 PrimExpr min_inclusive{nullptr};
888 PrimExpr max_inclusive(nullptr);
889 for (const IntSet& int_set : sets) {
890 if (const auto* interval_set = int_set.as<IntervalSetNode>()) {
891 PrimExpr new_min_inclusive = interval_set->min_value;
892 PrimExpr new_max_inclusive = interval_set->max_value;
893 if (is_first_interval) {
894 is_first_interval = false;
895 min_inclusive = std::move(new_min_inclusive);
896 max_inclusive = std::move(new_max_inclusive);
897 continue;
898 }
899 bool bound_1 = is_neg_inf(new_min_inclusive) || is_pos_inf(max_inclusive) ||
900 analyzer.CanProve(new_min_inclusive <= max_inclusive + 1);
901 bool bound_2 = is_neg_inf(min_inclusive) || is_pos_inf(new_max_inclusive) ||
902 analyzer.CanProve(min_inclusive <= new_max_inclusive + 1);
903 if (bound_1 && bound_2) {
904 min_inclusive = min(min_inclusive, new_min_inclusive);
905 max_inclusive = max(max_inclusive, new_max_inclusive);
906 }
907 }
908 }
909 if (is_first_interval) {
910 return IntSet::Nothing();
911 }
912 return IntSet::Interval(min_inclusive, max_inclusive);
913}

1055/*! \brief Helper function to convert IterSumExpr to the actual touched range. */
1056static Optional<IntSet> EvalIterSum(const IterSumExpr& iter_min, const PrimExpr& extent,
1057 Analyzer* analyzer) {
1058 if (iter_min->args.empty()) {
1059 return IntSet::FromMinExtent(iter_min->base, extent);
1060 }
1061 ICHECK_EQ(iter_min->args.size(), 1) << "The `EvalIterSum` expects fused iter sum expr";
1062 const IterSplitExpr& split = iter_min->args[0];
1063 if (!analyzer->CanProve(extent >= split->scale)) {
1064 return NullOpt;
1065 }
1066
1067 const PrimExpr& base = iter_min->base;
1068 // IterSplitExpr: (source // lower_factor) % extent * scale
1069 // where `(source // lower_factor) % extent` is within [0, extent - 1]
1070 if (analyzer->CanProve(split->scale < 0)) {
1071 // If scale is negative, the var dom is [(extent - 1) * scale, 0]
1072 // The total base is `base + (extent - 1) * scale`,
1073 // while total extent is `dom_extent + (extent - 1) * (-scale)`
1074 const PrimExpr& var_extent = (split->extent - 1) * split->scale;
1075 return IntSet::FromMinExtent(base + var_extent, extent - var_extent);
1076 } else {
1077 // If scale is positive, the var dom is [0, (extent - 1) * scale]
1078 // The total dom is [base, dom_extent + (extent - 1) * scale]
1079 return IntSet::FromMinExtent(base, extent + (split->extent - 1) * split->scale);
1080 }
1081}

1083Optional<Array<IntSet>> EstimateRegionStrictBound(const Array<Range>& region,
1084 const Map<Var, Range>& var_dom,
1085 const PrimExpr& predicate, Analyzer* analyzer) {
1086 int ndim = region.size();
1087 Array<IterSumExpr> iter_sum_exprs{nullptr};
1088 {
1089 Array<PrimExpr> affine_indices;
1090 affine_indices.reserve(ndim);
1091 for (const Range& range : region) {
1092 if (!is_const_number(range->extent)) {
1093 // dynamic extent is not supported yet.
1094 return NullOpt;
1095 }
1096 affine_indices.push_back(range->min);
1097 }
1098 auto res = DetectIterMap(
1099 /*indices=*/affine_indices, /*input_iters=*/var_dom,
1100 /*predicate=*/predicate, /*check_level=*/IterMapLevel::Surjective, analyzer);
1101 iter_sum_exprs = res->indices;
1102 }
1103 if (iter_sum_exprs.empty()) {
1104 return NullOpt;
1105 }
1106 ICHECK_EQ(iter_sum_exprs.size(), ndim);
1107 Array<IntSet> result;
1108 result.reserve(ndim);
1109 for (int i = 0; i < ndim; ++i) {
1110 const IterSumExpr& sum_expr = iter_sum_exprs[i];
1111 const Range& range = region[i];
1112 Optional<IntSet> int_set = EvalIterSum(sum_expr, range->extent, analyzer);
1113 if (int_set.defined()) {
1114 result.push_back(int_set.value());
1115 } else {
1116 return NullOpt;
1117 }
1118 }
1119 return result;
1120}

1129Array<IntSet> EstimateRegionUpperBound(const Array<Range>& region, const Map<Var, Range>& var_dom,
1130 const PrimExpr& predicate, Analyzer* analyzer) {
1131 if (Optional<Array<arith::IntSet>> result = EstimateRegionStrictBound(
1132 /*region=*/region,
1133 /*var_dom=*/var_dom,
1134 /*predicate=*/predicate, /*analyzer=*/analyzer)) {
1135 return result.value();
1136 }
1137 Array<IntSet> result;
1138 result.reserve(region.size());
1139 // try estimate each dimension independently
1140 for (const Range& range : region) {
1141 auto res = DetectIterMap(
1142 /*indices=*/{range->min}, /*input_iters=*/var_dom,
1143 /*predicate=*/predicate, /*check_level=*/IterMapLevel::Surjective, analyzer);
1144 if (!res->indices.empty()) {
1145 ICHECK_EQ(res->indices.size(), 1U);
1146 IterSumExpr sum_expr = res->indices[0];
1147
1148 // dynamic extent is not supported yet.
1149 PrimExpr extent = range->extent;
1150 if (!is_const_number(extent)) {
1151 IntSet relaxed = EvalSet(extent, AsIntSet(var_dom));
1152 ICHECK(relaxed.HasUpperBound());
1153 extent = relaxed.max();
1154 }
1155
1156 if (Optional<IntSet> int_set = EvalIterSum(sum_expr, range->extent, analyzer)) {
1157 result.push_back(int_set.value());
1158 continue;
1159 }
1160 }
1161 // fallback to coarse grained evalset
1162 result.push_back(EvalSet(range, AsIntSet(var_dom)));
1163 }
1164 return result;
