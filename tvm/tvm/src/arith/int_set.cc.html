62IntervalSet Intersect(Analyzer* analyzer, IntervalSet a, IntervalSet b) {
63 PrimExpr max_value = min(a->max_value, b->max_value);
64 PrimExpr min_value = max(a->min_value, b->min_value);
65 if ((max_value.dtype().is_int() || max_value.dtype().is_uint()) &&
66 (min_value.dtype().is_int() || min_value.dtype().is_uint()) &&
67 analyzer->CanProve(max_value < min_value)) {
68 return IntervalSet::Empty();
69 } else {
70 return IntervalSet(min_value, max_value);
71 }
72}
73
74IntervalSet Union(Analyzer* analyzer, IntervalSet a, IntervalSet b) {
75 if (a->IsEmpty()) return b;
76 if (b->IsEmpty()) return a;
77 PrimExpr max_value = max(a->max_value, b->max_value);
78 PrimExpr min_value = min(a->min_value, b->min_value);
79 return IntervalSet(min_value, max_value);
80}

104/*!
105 * \brief Combine two interval set under arithmetic operations.
106 * \note this can possibly relax the set.
107 */
108template <typename Op>
109inline IntervalSet Combine(Analyzer* analyzer, IntervalSet a, IntervalSet b, DataType dtype) {
110 if (a->IsSinglePoint() && b->IsSinglePoint()) {
111 PrimExpr res = TryConstFold<Op>(a->min_value, b->min_value);
112 if (!res.defined()) res = Op(a->min_value, b->min_value);
113 return IntervalSet::SinglePoint(res);
114 }
115 if (is_logical_op<Op>::value) {
116 return IntervalSet(make_const(dtype, 0), make_const(dtype, 1));
117 }
118 if (a->IsEmpty()) return a;
119 if (b->IsEmpty()) return b;
120 if (a->IsEverything()) return a;
121 if (b->IsEverything()) return b;
122 return IntervalSet::Everything();
123}

125template <>
126inline IntervalSet Combine<tir::Add>(Analyzer* analyer, IntervalSet a, IntervalSet b,
127 DataType /* dtype */) {
128 if (a->IsSinglePoint() && b->IsSinglePoint()) {
129 return IntervalSet::SinglePoint(a->min_value + b->min_value);
130 }
131 if (a->IsEmpty()) return a;
132 if (b->IsEmpty()) return b;
133 PrimExpr min_value =
134 a->HasLowerBound() && b->HasLowerBound() ? a->min_value + b->min_value : neg_inf();
135 PrimExpr max_value =
136 a->HasUpperBound() && b->HasUpperBound() ? a->max_value + b->max_value : pos_inf();
137 return IntervalSet(min_value, max_value);
138}

140template <>
141inline IntervalSet Combine<tir::Sub>(Analyzer* analyer, IntervalSet a, IntervalSet b,
142 DataType /* dtype */) {
143 if (a->IsSinglePoint() && b->IsSinglePoint()) {
144 return IntervalSet::SinglePoint(a->min_value - b->min_value);
145 }
146 if (a->IsEmpty()) return a;
147 if (b->IsEmpty()) return b;
148 PrimExpr min_value =
149 a->HasLowerBound() && b->HasUpperBound() ? a->min_value - b->max_value : neg_inf();
150 PrimExpr max_value =
151 a->HasUpperBound() && b->HasLowerBound() ? a->max_value - b->min_value : pos_inf();
152 return IntervalSet(min_value, max_value);
153}

155template <>
156inline IntervalSet Combine<tir::Mul>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
157 DataType /* dtype */) {
158 if (a->IsSinglePoint() && b->IsSinglePoint()) {
159 return IntervalSet::SinglePoint(a->min_value * b->min_value);
160 }
161 if (a->IsEmpty()) return a;
162 if (b->IsEmpty()) return b;
163 if (a->IsSinglePoint()) {
164 std::swap(a, b);
165 }
166 if (b->IsSinglePoint()) {
167 if (is_zero(b->min_value)) return b;
168 if (is_one(b->min_value)) return a;
169 if (analyzer->CanProveGreaterEqual(b->min_value, 0)) {
170 PrimExpr min_value = a->HasLowerBound() ? a->min_value * b->min_value : neg_inf();
171 PrimExpr max_value = a->HasUpperBound() ? a->max_value * b->min_value : pos_inf();
172 return IntervalSet(min_value, max_value);
173 } else if (analyzer->CanProveGreaterEqual(-b->min_value, 1)) {
174 PrimExpr min_value = a->HasUpperBound() ? a->max_value * b->min_value : neg_inf();
175 PrimExpr max_value = a->HasLowerBound() ? a->min_value * b->min_value : pos_inf();
176 return IntervalSet(min_value, max_value);
177 } else if (a->HasUpperBound() && a->HasLowerBound()) {
178 using tir::Select;
179 PrimExpr sign = b->min_value >= make_zero(b->min_value.dtype().element_of());
180 PrimExpr e1 = a->min_value * b->min_value;
181 PrimExpr e2 = a->max_value * b->min_value;
182 return IntervalSet(Select(sign, e1, e2), Select(sign, e2, e1));
183 }
184 }
185 DLOG(WARNING) << "Return Everything in CombineInterval Mul";
186 return IntervalSet::Everything();
187}

189template <>
190inline IntervalSet Combine<tir::Div>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
191 DataType /* dtype */) {
192 if (a->IsSinglePoint() && b->IsSinglePoint()) {
193 return IntervalSet::SinglePoint(a->min_value / b->min_value);
194 }
195 if (a->IsEmpty()) return a;
196 if (b->IsEmpty()) return b;
197 if (b->IsSinglePoint()) {
198 if (is_zero(b->min_value)) {
199 LOG(FATAL) << "Divide by zero in CombineInterval Div";
200 }
201 if (is_one(b->min_value)) return a;
202 // no relaxation is needed in here due to set is inclusive
203 if (analyzer->CanProveGreaterEqual(b->min_value, 0)) {
204 PrimExpr min_value = a->HasLowerBound() ? a->min_value / b->min_value : neg_inf();
205 PrimExpr max_value = a->HasUpperBound() ? a->max_value / b->min_value : pos_inf();
206 return IntervalSet(min_value, max_value);
207 } else if (analyzer->CanProveGreaterEqual(-b->min_value, 1)) {
208 PrimExpr min_value = a->HasUpperBound() ? a->max_value / b->min_value : neg_inf();
209 PrimExpr max_value = a->HasLowerBound() ? a->min_value / b->min_value : pos_inf();
210 return IntervalSet(min_value, max_value);
211 } else if (a->HasUpperBound() && a->HasLowerBound()) {
212 using tir::Select;
213 PrimExpr sign = b->min_value >= make_zero(b->min_value.dtype().element_of());
214 PrimExpr e1 = a->min_value / b->min_value;
215 PrimExpr e2 = a->max_value / b->min_value;
216 return IntervalSet(Select(sign, e1, e2), Select(sign, e2, e1));
217 }
218 }
219 DLOG(WARNING) << "Return Everything in CombineInterval Div";
220 return IntervalSet::Everything();
221}

223template <>
224inline IntervalSet Combine<tir::Mod>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
225 DataType /* dtype */) {
226 if (a->IsSinglePoint() && b->IsSinglePoint()) {
227 return IntervalSet::SinglePoint(truncmod(a->min_value, b->min_value));
228 }
229 if (a->IsEmpty()) return a;
230 if (b->IsEmpty()) return b;
231
232 if (b->IsSinglePoint()) {
233 const PrimExpr& divisor = b->min_value;
234 if (is_zero(divisor)) {
235 LOG(FATAL) << "Modular by zero in CombineInterval Mod";
236 }
237 // We need to add more bound constraints throughout the code.
238 // The logic below assumes a is non-negative, which usually
239 // is the case of our application.
240 // TODO(tqchen): add bound constraints for a.
241 if (analyzer->CanProveGreaterEqual(divisor, 0)) {
242 return IntervalSet(make_zero(divisor.dtype()), divisor - 1);
243 } else {
244 PrimExpr bound = abs(divisor) - 1;
245 return IntervalSet(-bound, bound);
246 }
247 }
248 DLOG(WARNING) << "Return Everything in CombineInterval Mod";
249 return IntervalSet::Everything();
250}

252template <>
253inline IntervalSet Combine<tir::FloorDiv>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
254 DataType /* dtype */) {
255 if (a->IsSinglePoint() && b->IsSinglePoint()) {
256 return IntervalSet::SinglePoint(floordiv(a->min_value, b->min_value));
257 }
258 if (a->IsEmpty()) return a;
259 if (b->IsEmpty()) return b;
260 if (b->IsSinglePoint()) {
261 if (is_zero(b->min_value)) {
262 LOG(FATAL) << "Divide by zero in CombineInterval Div";
263 }
264 if (is_one(b->min_value)) return a;
265 // no relaxation is needed in here due to set is inclusive
266 if (analyzer->CanProveGreaterEqual(b->min_value, 0)) {
267 PrimExpr min_value = a->HasLowerBound() ? floordiv(a->min_value, b->min_value) : neg_inf();
268 PrimExpr max_value = a->HasUpperBound() ? floordiv(a->max_value, b->min_value) : pos_inf();
269 return IntervalSet(min_value, max_value);
270 } else if (analyzer->CanProveGreaterEqual(-b->min_value, 1)) {
271 PrimExpr min_value = a->HasUpperBound() ? floordiv(a->max_value, b->min_value) : neg_inf();
272 PrimExpr max_value = a->HasLowerBound() ? floordiv(a->min_value, b->min_value) : pos_inf();
273 return IntervalSet(min_value, max_value);
274 } else if (a->HasUpperBound() && a->HasLowerBound()) {
275 using tir::Select;
276 PrimExpr sign = b->min_value >= make_zero(b->min_value.dtype().element_of());
277 PrimExpr e1 = floordiv(a->min_value, b->min_value);
278 PrimExpr e2 = floordiv(a->max_value, b->min_value);
279 return IntervalSet(Select(sign, e1, e2), Select(sign, e2, e1));
280 }
281 }
282 DLOG(WARNING) << "Return Everything in CombineInterval Div";
283 return IntervalSet::Everything();
284}

286template <>
287inline IntervalSet Combine<tir::FloorMod>(Analyzer* analyzer, IntervalSet a, IntervalSet b,
288 DataType /* dtype */) {
289 if (a->IsSinglePoint() && b->IsSinglePoint()) {
290 return IntervalSet::SinglePoint(floormod(a->min_value, b->min_value));
291 }
292 if (a->IsEmpty()) return a;
293 if (b->IsEmpty()) return b;
294
295 if (b->IsSinglePoint()) {
296 const PrimExpr& divisor = b->min_value;
297 if (is_zero(divisor)) {
298 LOG(FATAL) << "Modular by zero in CombineInterval Mod";
299 }
300 if (analyzer->CanProveGreaterEqual(divisor, 0)) {
301 if (divisor.as<tir::IntImmNode>()) {
302 // a mod b = a - (a / b) * b if a_max / b == a_min / b
303 auto qmax = a->HasUpperBound() ? floordiv(a->max_value, divisor) : pos_inf();
304 auto qmin = a->HasLowerBound() ? floordiv(a->min_value, divisor) : neg_inf();
305 // We can compare +/- inf against each other, but cannot use
306 // operator== between the symbolic limits and an integer.
307 bool compatible_dtypes = !(qmin.dtype().is_handle() ^ qmax.dtype().is_handle());
308 if (compatible_dtypes && analyzer->CanProve(qmax == qmin)) {
309 auto tmax = a->max_value - divisor * qmin;
310 auto tmin = a->min_value - divisor * qmin;
311 return IntervalSet(tmin, tmax);
312 }
313 }
314 return IntervalSet(make_zero(divisor.dtype()), divisor - 1);
315 } else {
316 PrimExpr bound = abs(divisor) - 1;
317 return IntervalSet(-bound, bound);
318 }
319 }
320 DLOG(WARNING) << "Return Everything in CombineInterval Mod";
321 return IntervalSet::Everything();
322}

357// Simplified version of int set evaluator that operates on IntervalSet
358// We might use better set analysis in the future to replace the intervalset.
359class IntervalSetEvaluator : public ExprFunctor<IntervalSet(const PrimExpr&)> {
360 public:
361 IntervalSetEvaluator(Analyzer* analyzer, const Map<Var, IntSet>& dom_map, bool eval_vec = false)
362 : analyzer_(analyzer), dom_map_(dom_map), eval_vec_(eval_vec) {}
363
364 IntervalSet Eval(const PrimExpr& val) { return this->VisitExpr(val); }
365 // evaluate and relax the set
366 IntervalSet Eval(IntervalSet val) {
367 // avoid recursive indefinite recursive expansion.
368 if (static_cast<size_t>(recur_depth_) >= dom_map_.size()) return val;
369 ++recur_depth_;
370 IntervalSet min_set = this->Eval(val->min_value);
371 IntervalSet max_set = this->Eval(val->max_value);
372 --recur_depth_;
373 return IntervalSet(min_set->min_value, max_set->max_value);
374 }
375
376 IntervalSet VisitExpr_(const IntImmNode* op) final {
377 return IntervalSet::SinglePoint(GetRef<PrimExpr>(op));
378 }
379
380 IntervalSet VisitExpr_(const VarNode* op) final {
381 Var var = GetRef<Var>(op);
382 auto it = dom_map_.find(var);
383 if (it != dom_map_.end()) {
384 IntervalSet res = ToIntervalSet((*it).second);
385 if (res->min_value.same_as(var) && res->max_value.same_as(var)) {
386 return res;
387 }
388 // recursively evaluate mapped result
389 // in case the domain contains variables to be relaxed.
390 return Eval(res);
391 } else {
392 return IntervalSet::SinglePoint(var);
393 }
394 }
395
396 IntervalSet VisitExpr_(const AddNode* op) final { return VisitBinaryExpr_<Add>(op); }
397
398 IntervalSet VisitExpr_(const SubNode* op) final { return VisitBinaryExpr_<Sub>(op); }
399
400 IntervalSet VisitExpr_(const MulNode* op) final { return VisitBinaryExpr_<Mul>(op); }
401
402 IntervalSet VisitExpr_(const DivNode* op) final { return VisitBinaryExpr_<Div>(op); }
403
404 IntervalSet VisitExpr_(const ModNode* op) final { return VisitBinaryExpr_<Mod>(op); }
405
406 IntervalSet VisitExpr_(const FloorDivNode* op) final { return VisitBinaryExpr_<FloorDiv>(op); }
407
408 IntervalSet VisitExpr_(const FloorModNode* op) final { return VisitBinaryExpr_<FloorMod>(op); }
409
410 IntervalSet VisitExpr_(const MinNode* op) final { return VisitBinaryExpr_<Min>(op); }
411
412 IntervalSet VisitExpr_(const MaxNode* op) final { return VisitBinaryExpr_<Max>(op); }
413
414 IntervalSet VisitExpr_(const EQNode* op) final { return VisitBinaryExpr_<EQ>(op); }
415
416 IntervalSet VisitExpr_(const NENode* op) final { return VisitBinaryExpr_<NE>(op); }
417
418 IntervalSet VisitExpr_(const LTNode* op) final { return VisitBinaryExpr_<LT>(op); }
419
420 IntervalSet VisitExpr_(const LENode* op) final { return VisitBinaryExpr_<LE>(op); }
421
422 IntervalSet VisitExpr_(const GTNode* op) final { return VisitBinaryExpr_<GT>(op); }
423
424 IntervalSet VisitExpr_(const GENode* op) final { return VisitBinaryExpr_<GE>(op); }
425
426 IntervalSet VisitExpr_(const AndNode* op) final { return VisitBinaryExpr_<And>(op); }
427
428 IntervalSet VisitExpr_(const OrNode* op) final { return VisitBinaryExpr_<Or>(op); }
429
430 IntervalSet VisitExpr_(const RampNode* op) final {
431 ICHECK(eval_vec_);
432 IntervalSet base = Eval(op->base);
433 PVar<IntImm> stride;
434 if (stride.Match(op->stride)) {
435 DataType t = op->base.dtype();
436 int64_t vstride = stride.Eval()->value;
437 if (vstride > 0) {
438 return Combine<Add>(analyzer_, base,
439 IntervalSet(make_zero(t), make_const(t, vstride * (op->lanes - 1))),
440 op->dtype);
441 } else {
442 return Combine<Add>(analyzer_, base,
443 IntervalSet(make_const(t, vstride * (op->lanes - 1)), make_zero(t)),
444 op->dtype);
445 }
446 }
447 DLOG(WARNING) << "cannot evaluate set on expression " << GetRef<PrimExpr>(op);
448 return IntervalSet::Everything();
449 }
450
451 IntervalSet VisitExpr_(const BroadcastNode* op) final {
452 ICHECK(eval_vec_);
453 return VisitExpr(op->value);
454 }
455
456 IntervalSet VisitExpr_(const SelectNode* op) final {
457 IntervalSet true_set = this->Eval(op->true_value);
458 IntervalSet false_set = this->Eval(op->false_value);
459 return Union(analyzer_, false_set, true_set);
460 }
461
462 IntervalSet VisitExpr_(const CastNode* op) final {
463 IntervalSet value_set = this->Eval(op->value);
464 PrimExpr min_value =
465 value_set->HasLowerBound() ? cast(op->dtype, value_set->min_value) : neg_inf();
466 PrimExpr max_value =
467 value_set->HasUpperBound() ? cast(op->dtype, value_set->max_value) : pos_inf();
468 return IntervalSet(min_value, max_value);
469 }
470
471 IntervalSet VisitExpr_(const BufferLoadNode* op) final {
472 if (!(op->dtype.is_int() || op->dtype.is_uint())) {
473 DLOG(WARNING) << "cannot evaluate set BufferLoad which loads from a " << op->dtype
474 << " buffer";
475 return IntervalSet::Everything();
476 }
477 // If the indices do not contain any variables to be relaxed, return the BufferLoad itself.
478 // Otherwise return `IntervalSet::everything()` since we have no knowledge on the buffer data.
479 for (const PrimExpr& index : op->indices) {
480 if (UsesVar(index, [dom_map = &this->dom_map_](const VarNode* var) {
481 return dom_map->find(GetRef<Var>(var)) != dom_map->end();
482 })) {
483 return IntervalSet::Everything();
484 }
485 }
486 return IntervalSet::SinglePoint(GetRef<PrimExpr>(op));
487 }
488
489 IntervalSet VisitExprDefault_(const Object* op) final {
490 DLOG(WARNING) << "cannot evaluate set type " << op->GetTypeKey();
491 return IntervalSet::Everything();
492 }
493
494 private:
495 // whether set is exactly single point that equals value.
496 bool MatchPoint(const IntervalSet& set, const PrimExpr& value) const {
497 return set->min_value.same_as(value) && set->max_value.same_as(value);
498 }
499
500 template <typename TOp, typename T>
501 inline IntervalSet VisitBinaryExpr_(const T* op) {
502 static_assert(std::is_same<typename TOp::ContainerType, T>::value, "constraint");
503 IntervalSet a = this->Eval(op->a);
504 IntervalSet b = this->Eval(op->b);
505 if (MatchPoint(a, op->a) && MatchPoint(b, op->b)) {
506 return IntervalSet::SinglePoint(GetRef<PrimExpr>(op));
507 }
508 return Combine<TOp>(analyzer_, a, b, op->dtype);
509 }
510
511 // recursive depth
512 int recur_depth_{0};
513 // analyzer
514 Analyzer* analyzer_;
515 const Map<Var, IntSet>& dom_map_;
516 bool eval_vec_{false};
517};

578void IntSetAnalyzer::Impl::Update(const Var& var, const IntSet& info, bool can_override) {
579 if (!can_override) {
580 auto it = dom_map_.find(var);
581 if (it != dom_map_.end()) {
582 const IntSet& old_info = (*it).second;
583
584 ICHECK(ExprDeepEqual()(old_info.min(), info.min()))
585 << "Trying to update var \'" << var << "\'"
586 << " with a different minimum value: "
587 << "original=" << old_info.min() << ", new=" << info.min();
588
589 ICHECK(ExprDeepEqual()(old_info.max(), info.max()))
590 << "Trying to update var \'" << var << "\'"
591 << " with a different maximum value: "
592 << "original=" << old_info.max() << ", new=" << info.max();
593 }
594 }
595 dom_map_.Set(var, info);
596}

602Map<Var, IntSet> IntSetAnalyzer::Impl::GetCurrentBounds() const {
603 // If either constraints_ or dom_map_ is empty, return the other to
604 // avoid constructing a new map.
605 if (constraints_.empty()) {
606 return dom_map_;
607 } else if (dom_map_.empty()) {
608 return constraints_;
609 }
610
611 // If neither is empty, construct a merged domain map with
612 // information from both sources.
613 Map<Var, IntSet> merged = dom_map_;
614 for (const auto& pair : constraints_) {
615 auto it = merged.find(pair.first);
616 if (it == merged.end()) {
617 merged.Set(pair.first, pair.second);
618 } else {
619 merged.Set(pair.first, Intersect({pair.second, (*it).second}));
620 }
621 }
622 return merged;
623}
624
625std::vector<std::pair<Var, IntSet>> IntSetAnalyzer::Impl::DetectBoundInfo(
626 const PrimExpr& constraint) {
627 PVar<Var> x;
628 PVar<PrimExpr> limit;
629
630 std::vector<std::pair<Var, IntSet>> bounds;
631 for (const PrimExpr& subconstraint : ExtractConstraints(constraint)) {
632 if ((x <= limit).Match(subconstraint)) {
633 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval())});
634 } else if ((x < limit).Match(subconstraint)) {
635 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval() - 1)});
636 } else if ((x >= limit).Match(subconstraint)) {
637 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval(), SymbolicLimits::pos_inf_)});
638 } else if ((x > limit).Match(subconstraint)) {
639 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval() + 1, SymbolicLimits::pos_inf_)});
640 } else if ((x == limit).Match(subconstraint)) {
641 bounds.push_back({x.Eval(), IntSet::SinglePoint(limit.Eval())});
642 }
643
644 if ((limit >= x).Match(subconstraint)) {
645 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval())});
646 } else if ((limit > x).Match(subconstraint)) {
647 bounds.push_back({x.Eval(), IntSet::Interval(SymbolicLimits::neg_inf_, limit.Eval() - 1)});
648 } else if ((limit <= x).Match(subconstraint)) {
649 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval(), SymbolicLimits::pos_inf_)});
650 } else if ((limit < x).Match(subconstraint)) {
651 bounds.push_back({x.Eval(), IntSet::Interval(limit.Eval() + 1, SymbolicLimits::pos_inf_)});
652 } else if ((limit == x).Match(subconstraint)) {
653 bounds.push_back({x.Eval(), IntSet::SinglePoint(limit.Eval())});
654 }
655 }
656 return bounds;
657}

660 return impl_->EnterConstraint(constraint);
661}
662
663std::function<void()> IntSetAnalyzer::Impl::EnterConstraint(const PrimExpr& constraint) {
664 Map<Var, IntSet> cached_values;
665
666 auto bounds = DetectBoundInfo(constraint);
667
668 if (bounds.size() == 0) return nullptr;
669
670 // Collect the current values of each var that is changes by this
671 // constraint.
672 for (const auto& pair : bounds) {
673 auto it = constraints_.find(pair.first);
674 if (it == constraints_.end()) {
675 cached_values.Set(pair.first, IntSet());
676 } else {
677 cached_values.Set(pair.first, (*it).second);
678 }
679 }
680
681 // Update all constraints
682 for (const auto& pair : bounds) {
683 auto it = constraints_.find(pair.first);
684 if (it == constraints_.end()) {
685 constraints_.Set(pair.first, pair.second);
686 } else {
687 constraints_.Set(pair.first, Intersect({pair.second, (*it).second}));
688 }
689 }
690
691 auto frecover = [cached_values, this]() {
692 for (const auto& it : cached_values) {
693 if (it.second.defined()) {
694 constraints_.Set(it.first, it.second);
695 } else {
696 constraints_.erase(it.first);
697 }
698 }
699 };
700 return frecover;
701}

878IntSet UnionLowerBound(const Array<IntSet>& sets) {
879 if (sets.size() == 0) return IntSet::Nothing();
880 if (sets.size() == 1) return sets[0];
881 Analyzer analyzer;
882 bool is_first_interval = true;
883 PrimExpr min_inclusive{nullptr};
884 PrimExpr max_inclusive(nullptr);
885 for (const IntSet& int_set : sets) {
886 if (const auto* interval_set = int_set.as<IntervalSetNode>()) {
887 PrimExpr new_min_inclusive = interval_set->min_value;
888 PrimExpr new_max_inclusive = interval_set->max_value;
889 if (is_first_interval) {
890 is_first_interval = false;
891 min_inclusive = std::move(new_min_inclusive);
892 max_inclusive = std::move(new_max_inclusive);
893 continue;
894 }
895 bool bound_1 = is_neg_inf(new_min_inclusive) || is_pos_inf(max_inclusive) ||
896 analyzer.CanProve(new_min_inclusive <= max_inclusive + 1);
897 bool bound_2 = is_neg_inf(min_inclusive) || is_pos_inf(new_max_inclusive) ||
898 analyzer.CanProve(min_inclusive <= new_max_inclusive + 1);
899 if (bound_1 && bound_2) {
900 min_inclusive = min(min_inclusive, new_min_inclusive);
901 max_inclusive = max(max_inclusive, new_max_inclusive);
902 }
903 }
904 }
905 if (is_first_interval) {
906 return IntSet::Nothing();
907 }
908 return IntSet::Interval(min_inclusive, max_inclusive);
909}

1051/*! \brief Helper function to convert IterSumExpr to the actual touched range. */
1052static Optional<IntSet> EvalIterSum(const IterSumExpr& iter_min, const PrimExpr& extent,
1053 Analyzer* analyzer) {
1054 if (iter_min->args.empty()) {
1055 return IntSet::FromMinExtent(iter_min->base, extent);
1056 }
1057 ICHECK_EQ(iter_min->args.size(), 1) << "The `EvalIterSum` expects fused iter sum expr";
1058 const IterSplitExpr& split = iter_min->args[0];
1059 if (!analyzer->CanProve(extent >= split->scale)) {
1060 return NullOpt;
1061 }
1062
1063 const PrimExpr& base = iter_min->base;
1064 // IterSplitExpr: (source // lower_factor) % extent * scale
1065 // where `(source // lower_factor) % extent` is within [0, extent - 1]
1066 if (analyzer->CanProve(split->scale < 0)) {
1067 // If scale is negative, the var dom is [(extent - 1) * scale, 0]
1068 // The total base is `base + (extent - 1) * scale`,
1069 // while total extent is `dom_extent + (extent - 1) * (-scale)`
1070 const PrimExpr& var_extent = (split->extent - 1) * split->scale;
1071 return IntSet::FromMinExtent(base + var_extent, extent - var_extent);
1072 } else {
1073 // If scale is positive, the var dom is [0, (extent - 1) * scale]
1074 // The total dom is [base, dom_extent + (extent - 1) * scale]
1075 return IntSet::FromMinExtent(base, extent + (split->extent - 1) * split->scale);
1076 }
1077}

1079Optional<Array<IntSet>> EstimateRegionStrictBound(const Array<Range>& region,
1080 const Map<Var, Range>& var_dom,
1081 const PrimExpr& predicate, Analyzer* analyzer) {
1082 int ndim = region.size();
1083 Array<IterSumExpr> iter_sum_exprs{nullptr};
1084 {
1085 Array<PrimExpr> affine_indices;
1086 affine_indices.reserve(ndim);
1087 for (const Range& range : region) {
1088 if (!is_const_number(range->extent)) {
1089 // dynamic extent is not supported yet.
1090 return NullOpt;
1091 }
1092 affine_indices.push_back(range->min);
1093 }
1094 auto res = DetectIterMap(
1095 /*indices=*/affine_indices, /*input_iters=*/var_dom,
1096 /*predicate=*/predicate, /*check_level=*/IterMapLevel::Surjective, analyzer);
1097 iter_sum_exprs = res->indices;
1098 }
1099 if (iter_sum_exprs.empty()) {
1100 return NullOpt;
1101 }
1102 ICHECK_EQ(iter_sum_exprs.size(), ndim);
1103 Array<IntSet> result;
1104 result.reserve(ndim);
1105 for (int i = 0; i < ndim; ++i) {
1106 const IterSumExpr& sum_expr = iter_sum_exprs[i];
1107 const Range& range = region[i];
1108 Optional<IntSet> int_set = EvalIterSum(sum_expr, range->extent, analyzer);
1109 if (int_set.defined()) {
1110 result.push_back(int_set.value());
1111 } else {
1112 return NullOpt;
1113 }
1114 }
1115 return result;
1116}

1118Optional<Array<IntSet>> EstimateRegionLowerBound(const Array<Range>& region,
1119 const Map<Var, Range>& var_dom,
1120 const PrimExpr& predicate,
1121 arith::Analyzer* analyzer) {
1122 return EstimateRegionStrictBound(region, var_dom, predicate, analyzer);
1123}
1124
1125Array<IntSet> EstimateRegionUpperBound(const Array<Range>& region, const Map<Var, Range>& var_dom,
1126 const PrimExpr& predicate, Analyzer* analyzer) {
1127 if (Optional<Array<arith::IntSet>> result = EstimateRegionStrictBound(
1128 /*region=*/region,
1129 /*var_dom=*/var_dom,
1130 /*predicate=*/predicate, /*analyzer=*/analyzer)) {
1131 return result.value();
1132 }
1133 Array<IntSet> result;
1134 result.reserve(region.size());
1135 // try estimate each dimension independently
1136 for (const Range& range : region) {
1137 auto res = DetectIterMap(
1138 /*indices=*/{range->min}, /*input_iters=*/var_dom,
1139 /*predicate=*/predicate, /*check_level=*/IterMapLevel::Surjective, analyzer);
1140 if (!res->indices.empty()) {
1141 ICHECK_EQ(res->indices.size(), 1U);
1142 IterSumExpr sum_expr = res->indices[0];
1143
1144 // dynamic extent is not supported yet.
1145 PrimExpr extent = range->extent;
1146 if (!is_const_number(extent)) {
1147 IntSet relaxed = EvalSet(extent, AsIntSet(var_dom));
1148 ICHECK(relaxed.HasUpperBound());
1149 extent = relaxed.max();
1150 }
1151
1152 if (Optional<IntSet> int_set = EvalIterSum(sum_expr, range->extent, analyzer)) {
1153 result.push_back(int_set.value());
1154 continue;
1155 }
1156 }
1157 // fallback to coarse grained evalset
1158 result.push_back(EvalSet(range, AsIntSet(var_dom)));
1159 }
1160 return result;
1161}
