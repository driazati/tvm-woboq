62inline PrimExpr ModImpl(PrimExpr a, PrimExpr b, DivMode mode) {
63 if (mode == kTruncDiv) {
64 return truncmod(a, b);
65 } else {
66 ICHECK_EQ(mode, kFloorDiv);
67 return floormod(a, b);
68 }
69}
70
71inline PrimExpr DivImpl(PrimExpr a, PrimExpr b, DivMode mode) {
72 if (mode == kTruncDiv) {
73 return truncdiv(a, b);
74 } else {
75 ICHECK_EQ(mode, kFloorDiv);
76 return floordiv(a, b);
77 }
78}

80/*!
81 * \brief check if value fits in dtype
82 * \param value The value to be analyzed
83 * \param dtype The target dtype
84 * \param analyzer The analyzer
85 * \return whether value fits in dtype
86 */
87bool CastIsSafe(DataType dtype, PrimExpr value, Analyzer* analyzer) {
88 if (!IsIndexType(dtype)) {
89 return false;
90 }
91 ConstIntBound bound = analyzer->const_int_bound(value);
92 int64_t ubound = Downcast<IntImm>(max_value(dtype))->value;
93 int64_t lbound = Downcast<IntImm>(min_value(dtype))->value;
94 if (value.dtype().bits() <= dtype.bits() || // upcast is safe
95 (bound->max_value <= ubound && bound->min_value >= lbound)) {
96 return true;
97 }
98 return false;
99}

101/*!
102 * \brief Internal "Split normal form" of expression.
103 *
104 * This is a special expression that represents
105 * a scaled value derived from a split of an index.
106 *
107 * result = ((index % upper_factor) / lower_factor) * scale
108 */
109class SplitExprNode : public CanonicalExprNode {
110 public:
111 /*! \brief The base index expression. */
112 PrimExpr index;
113 /*! \brief The division factor ratio. */
114 int64_t lower_factor{1};
115 /*!
116 * \brief The upper factor.
117 * invariance: (upper_factor == kPosInf || upper_factor % lower_factor == 0)
118 */
119 int64_t upper_factor{kPosInf};
120 /*! \brief scale to the expression. */
121 int64_t scale{1};
122 /*! \brief Division mode. */
123 DivMode div_mode{kTruncDiv};
124
125 /*! \brief verify that this is a valid entry. */
126 void Verify() const { ICHECK(upper_factor == kPosInf || upper_factor % lower_factor == 0); }
127
128 PrimExpr NormalizeWithScale(int64_t sscale) const {
129 PrimExpr res = this->index;
130 DataType dtype = this->dtype;
131 if (this->scale == 0) {
132 return make_const(dtype, 0);
133 }
134 if (this->upper_factor != SplitExprNode::kPosInf) {
135 res = ModImpl(res, make_const(dtype, this->upper_factor), div_mode);
136 }
137 if (this->lower_factor != 1) {
138 res = DivImpl(res, make_const(dtype, this->lower_factor), div_mode);
139 }
140 sscale *= this->scale;
141 if (sscale != 1) {
142 ICHECK(!dtype.is_uint() || sscale > 0);
143 res = res * make_const(dtype, sscale);
144 }
145 return res;
146 }
147
148 PrimExpr Normalize() const final { return NormalizeWithScale(1); }
149
150 void MulToSelf(int64_t scale) { this->scale *= scale; }
151
152 /*!
153 * \brief check if cast can be pushed to sub-expressions
154 * \param dtype The target datatype
155 * \param analyzer The analyzer
156 * \return whether the cast can be safely pushed to children
157 */
158 bool CanPushCastToChildren(DataType dtype, Analyzer* analyzer) const {
159 // cast(dtype, index % upper_factor / lower_factor * scale) ==
160 // cast(dtype, index) % upper_factor / lower_factor * scale
161 // iff it is an upcast (dtype.bits >= self.dtype.bits) or all of
162 // its intermediate results fit in the range of dtype
163 if (dtype.bits() >= this->dtype.bits()) {
164 return true; // upcast is safe
165 }
166 PrimExpr res = this->index;
167 if (this->scale == 0) {
168 return true;
169 }
170 if (!CastIsSafe(dtype, res, analyzer)) {
171 return false;
172 }
173 if (this->upper_factor != SplitExprNode::kPosInf) {
174 res = ModImpl(res, make_const(this->dtype, this->upper_factor), div_mode);
175 if (!CastIsSafe(dtype, res, analyzer)) {
176 return false;
177 }
178 }
179 if (this->lower_factor != 1) {
180 res = DivImpl(res, make_const(this->dtype, this->lower_factor), div_mode);
181 if (!CastIsSafe(dtype, res, analyzer)) {
182 return false;
183 }
184 }
185 if (this->scale != 1) {
186 ICHECK(!this->dtype.is_uint() || this->scale > 0);
187 res = res * make_const(this->dtype, this->scale);
188 if (!CastIsSafe(dtype, res, analyzer)) {
189 return false;
190 }
191 }
192 return true;
193 }
194
195 /*!
196 * \brief self = cast(dtype, self)
197 * \param dtype The target datatype
198 */
199 void PushCastToChildren(DataType dtype) {
200 this->index = cast(dtype, this->index);
201 this->dtype = dtype;
202 }
203
204 inline bool IndexEqual(const SplitExpr& other) const;
205 inline bool DivModeCompatibleTo(DivMode mode) const;
206
207 /*! \brief positive infty */
208 static const constexpr int64_t kPosInf = ConstIntBoundNode::kPosInf;
209 static constexpr const char* _type_key = "arith.SplitExpr";
210 TVM_DECLARE_FINAL_OBJECT_INFO(SplitExprNode, CanonicalExprNode);
211};

230/*!
231 * \brief Normal form that represents sum of expressions.
232 *
233 * result = sum(args) + base.
234 */
235class SumExprNode : public CanonicalExprNode {
236 public:
237 /*!
238 * \brief arguments to be summed up.
239 *
240 * args are divided into segments with the same index.
241 * within each segment, the SplitExpr is ordered in descending order of lower_factor.
242 */
243 std::vector<SplitExpr> args;
244 /*! \brief Base value in the summation. */
245 int64_t base{0};
246 /*! \brief The expression equals zero. */
247 bool IsZero() const { return base == 0 && args.size() == 0; }
248 /*!
249 * \brief Return the normal Expr that is equivalent to self.
250 * \return The normal expression.
251 */
252 PrimExpr Normalize() const final {
253 // quick path 1.
254 if (this->args.size() == 0) {
255 return make_const(this->dtype, this->base);
256 }
257 return Normalize_(this->dtype, SimplifySplitExprs(args), base);
258 }
259 /*!
260 * \brief Whether self is divisible by scale.
261 * \param scale The scale to be applied.
262 */
263 bool DivisibleBy(int64_t scale) {
264 if (base % scale != 0) return false;
265 for (size_t i = 0; i < this->args.size(); ++i) {
266 if (args[i]->scale % scale != 0) return false;
267 }
268 return true;
269 }
270 /*!
271 * \brief mul scale to self.
272 * \param scale The scale to be applied.
273 */
274 void MulToSelf(int64_t scale) {
275 this->base *= scale;
276 for (size_t i = 0; i < this->args.size(); ++i) {
277 args[i].CopyOnWrite()->scale *= scale;
278 }
279 }
280 /*!
281 * \brief divide by scale.
282 * \param scale The scale to be applied.
283 */
284 void DivideBy(int64_t scale) {
285 ICHECK_EQ(this->base % scale, 0);
286 this->base /= scale;
287 for (size_t i = 0; i < this->args.size(); ++i) {
288 ICHECK_EQ(args[i]->scale % scale, 0);
289 args[i].CopyOnWrite()->scale /= scale;
290 }
291 }
292 /*!
293 * \brief add constant value to self.
294 * \param value to be added.
295 */
296 void AddToSelf(int64_t value) { this->base += value; }
297 /*!
298 * \brief self += other * scale;
299 * \param other The expression to be added.
300 * \param scale The additional scale on value.
301 */
302 void AddToSelf(SplitExpr other, int64_t scale) {
303 if (other->scale == 0) return;
304 // We need to maintain the segment invariance:
305 // Same index are stored close to each other.
306 // sorted from big lower_factor to small one.
307 size_t start = 0;
308 for (; start < args.size(); ++start) {
309 if (args[start]->IndexEqual(other)) break;
310 }
311 for (size_t j = start; j < args.size(); ++j) {
312 if (!args[j]->IndexEqual(other) || other->lower_factor > args[j]->lower_factor) {
313 other.CopyOnWrite()->scale *= scale;
314 this->args.insert(this->args.begin() + j, other);
315 return;
316 }
317 if (other->lower_factor == args[j]->lower_factor &&
318 other->upper_factor == args[j]->upper_factor &&
319 other->DivModeCompatibleTo(args[j]->div_mode)) {
320 args[j].CopyOnWrite()->scale += other->scale * scale;
321 return;
322 }
323 }
324 // Insert other in the end.
325 other.CopyOnWrite()->scale *= scale;
326 this->args.emplace_back(std::move(other));
327 }

329 void AddToSelf(const SumExpr& other, int64_t scale);

331 /*!
332 * \brief check if cast can be pushed to sub-expressions
333 * \param dtype The target datatype
334 * \param analyzer The analyzer
335 * \return whether the cast can be safely pushed to children
336 */
337 bool CanPushCastToChildren(DataType dtype, Analyzer* analyzer) const {
338 // cast(dtype, arg_1 + arg_2 + ... arg_n) ==
339 // cast(dtype, arg_1) + ... + cast(dtype, arg_n)
340 // iff it is an upcast (dtype.bits >= self.dtype.bits) or all of
341 // its intermediate results fit in the range of dtype
342 if (dtype.bits() >= this->dtype.bits()) {
343 return true; // upcast is safe
344 }
345 PrimExpr res = make_const(dtype, 0);
346 for (size_t i = 0; i < args.size(); ++i) {
347 if (args[i]->scale > 0) {
348 res = res + args[i]->Normalize();
349 if (!CastIsSafe(dtype, res, analyzer)) {
350 return false;
351 }
352 }
353 }
354 if (base > 0) {
355 res = res + make_const(dtype, base);
356 if (!CastIsSafe(dtype, res, analyzer)) {
357 return false;
358 }
359 }
360 // negative scales follows using sub.
361 for (size_t i = 0; i < args.size(); ++i) {
362 if (args[i]->scale < 0) {
363 res = res - args[i]->NormalizeWithScale(-1);
364 if (!CastIsSafe(dtype, res, analyzer)) {
365 return false;
366 }
367 }
368 }
369 if (base < 0) {
370 res = res - make_const(dtype, -base);
371 if (!CastIsSafe(dtype, res, analyzer)) {
372 return false;
373 }
374 }
375 for (const auto& arg : args) {
376 if (!arg->CanPushCastToChildren(dtype, analyzer)) {
377 return false;
378 }
379 }
380 return true;
381 }
382
383 /*!
384 * \brief self = cast(dtype, self)
385 * \param dtype The target datatype
386 */
387 void PushCastToChildren(DataType dtype) {
388 for (auto& arg : args) {
389 arg.CopyOnWrite()->PushCastToChildren(dtype);
390 }
391 this->dtype = dtype;
392 }

394 static constexpr const char* _type_key = "arith.SumExpr";
395 TVM_DECLARE_FINAL_OBJECT_INFO(SumExprNode, CanonicalExprNode);

397 private:
398 /*!
399 * \brief Simplify the args by merging SplitExprs
400 * \param args The original list of arguments.
401 * \return simplified version.
402 */
403 static std::vector<SplitExpr> SimplifySplitExprs(std::vector<SplitExpr> args) {
404 // NOTE: This algorithm relies on the factor that args are divided into segments
405 // and each segment is sorted in descending order of lower_factor.
406 for (size_t i = 0; i < args.size(); ++i) {
407 if (args[i]->scale == 0) continue;
408 for (size_t j = i + 1; j < args.size(); ++j) {
409 SplitExpr& lhs = args[i];
410 SplitExpr& rhs = args[j];
411 if (!lhs->IndexEqual(rhs)) break;
412 if (lhs->upper_factor < rhs->lower_factor) break;
413 if (lhs->upper_factor == rhs->upper_factor && lhs->lower_factor == rhs->lower_factor &&
414 lhs->DivModeCompatibleTo(rhs->div_mode)) {
415 // folding same co-efficient.
416 rhs.CopyOnWrite()->scale += lhs->scale;
417 lhs.CopyOnWrite()->scale = 0;
418 } else if (lhs->lower_factor == rhs->upper_factor && rhs->scale != 0 &&
419 lhs->scale % rhs->scale == 0 &&
420 lhs->lower_factor == (lhs->scale / rhs->scale) * rhs->lower_factor &&
421 lhs->DivModeCompatibleTo(rhs->div_mode)) {
422 // Rules used in the proof:
423 //
424 // Rule 1: (x % (c * s)) / c = (x / c) % s
425 // Proof:
426 // x can always be decomposed into p * c * s + q * c + r
427 // where 0 <= q * c + r < c * s and 0 <= r < c.
428 // Then, lhs = ((p * c * s + q * c + r) % (c * s)) / c = (q * c + r) / c = q
429 // rhs = ((p * c * s + q * c + r) / c) % s = (p * s + q) % s = q
430 // Thus, lhs = rhs
431 //
432 // The above proof is for the floordiv.
433 // The same rule also holds for truncdiv(division rule in C).
434 // Because both sides only involve mul, div and mod,
435 // we can take abs of x, c and s, apply the floordiv proof,
436 // and finally add the sign back.
437 //
438 // Rule 2: (x / s) * s + x % s = x (true for both trunc and floor div)
439 //
440 // General merge condition and proof:
441 // - x = lhs->index % lhs->upper_factor
442 // - s = lhs->scale / rhs->scale
443 // - c = rhs->lower_factor
444 //
445 // (x / (c * s)) * s + (x % (c * s)) / c
446 // => ((x / c) / s) * s + ((x / c) % s)
447 // => (x / c)
448 //
449 // Examples:
450 //
451 // (z / 6) * 6 + ((z % 6) / 3) * 3
452 // => ((z / 6) * 2 + (z % 6) / 3) * 3
453 // => (z / 3) * 3
454 // note: x = z, c = 3, s = 2
455 //
456 // ((z % 12) / 6) * 6 + ((z % 6) / 3) * 3
457 // => (((z % 12) / 6) * 2 + ((z % 12) % 6) / 3) * 3
458 // => ((z % 12) / 3) * 3
459 // note: x = z % 12, c = 3, s = 2
460 // note also the invariance lhs->upper_factor % lhs->lower_factor == 0
461 //
462 SplitExprNode* merged = rhs.CopyOnWrite();
463 merged->upper_factor = lhs->upper_factor;
464 // reset args[i] to be zero.
465 lhs.CopyOnWrite()->scale = 0;
466 break;
467 }
468 }
469 }
470 // sort by the entry
471 // Here we simply sort by descending order of scales.
472 // For now, we do not compare by index because that comparison
473 // can be runtime dependent and create inderminism.
474 // we do not sort by index for now because it can be costly
475 // to deep compare Exprs, and address of Vars can be runtime dependent.
476 //
477 auto fcompare = [](const SplitExpr& lhs, const SplitExpr& rhs) {
478 // order by scale first
479 if (lhs->scale > rhs->scale) return true;
480 if (lhs->scale < rhs->scale) return false;
481 // then order by factor
482 if (lhs->lower_factor > rhs->lower_factor) return true;
483 if (lhs->lower_factor < rhs->lower_factor) return false;
484 // then order by upper factor
485 if (lhs->upper_factor > rhs->upper_factor) return true;
486 if (lhs->upper_factor < rhs->upper_factor) return false;
487 // then order by div mode
488 if (lhs->div_mode > rhs->div_mode) return true;
489 if (lhs->div_mode < rhs->div_mode) return false;
490 // tie.
491 // TODO(tvm-team) We might consider index as the last comparison point,
492 // after we make deep comparator more derministic.
493 // Specifically, we can consider comparing names of vars and break ties with address.
494 return false;
495 };
496 std::stable_sort(args.begin(), args.end(), fcompare);
497 return args;
498 }
499 static PrimExpr Normalize_(DataType dtype, const std::vector<SplitExpr>& args, int64_t base) {
500 // Positive scales first
501 PrimExpr res = make_const(dtype, 0);
502 for (size_t i = 0; i < args.size(); ++i) {
503 if (args[i]->scale > 0) {
504 res = res + args[i]->Normalize();
505 }
506 }
507 if (base > 0) {
508 res = res + make_const(dtype, base);
509 }
510 // negative scales follows using sub.
511 for (size_t i = 0; i < args.size(); ++i) {
512 if (args[i]->scale < 0) {
513 res = res - args[i]->NormalizeWithScale(-1);
514 }
515 }
516 if (base < 0) {
517 res = res - make_const(dtype, -base);
518 }
519 return res;
520 }
521};

529void SumExprNode::AddToSelf(const SumExpr& other, int64_t scale) {
530 // NOTE: it is rare to have a balanced long expression,
531 // linear scan is fine for our case.
532 for (size_t i = 0; i < other->args.size(); ++i) {
533 this->AddToSelf(other->args[i], scale);
534 }
535 this->AddToSelf(other->base * scale);
536}

606 private:
607 /*!
608 * \brief compute lhs / cval
609 * \param lhs The left operand.
610 * \param cval The constant value.
611 * \param div_mode The division mode.
612 * \return The result expression;
613 */
614 SplitExpr SplitDivConst(SplitExpr lhs, int64_t cval, DivMode div_mode);
615 /*!
616 * \brief compute lhs % cval
617 * \param lhs The left operand.
618 * \param cval The constant value.
619 * \param div_mode The division mode.
620 * \return The result expression;
621 */
622 SplitExpr SplitModConst(SplitExpr lhs, int64_t cval, DivMode div_mode);
623 /*!
624 * \brief Separate psum into divisible and non-divisible parts.
625 * \param psum The sum expression.
626 * \param coeff The co-efficient.
627 * \param out_divisible The result divisible component.
628 * \param out_non_divisible The non-divisible component.
629 */
630 void SeparateDivisibleParts(const SumExprNode* psum, int64_t coeff, SumExpr* out_divisible,
631 SumExpr* out_non_divisible);
632 /*!
633 * \brief Normalize expr to normal expr.
634 * \param expr The input expression.
635 * \return Normalized expr.
636 */
637 PrimExpr Normalize(PrimExpr expr) {
638 if (const auto* op = expr.as<CanonicalExprNode>()) {
639 return op->Normalize();
640 } else {
641 return expr;
642 }
643 }
644 /*!
645 * \brief Create a SplitExpr from expr.
646 * \param expr The input expr.
647 * \return The transformed SplitExpr.
648 */
649 SplitExpr ToSplitExpr(PrimExpr expr) {
650 if (const auto* op = expr.as<SplitExprNode>()) {
651 return GetRef<SplitExpr>(op);
652 }
653 if (const auto* op = expr.as<SumExprNode>()) {
654 if (op->base == 0 && op->args.size() == 1) return op->args[0];
655 }
656 if (const auto* op = expr.as<CanonicalExprNode>()) {
657 expr = op->Normalize();
658 }
659 ObjectPtr<SplitExprNode> n = make_object<SplitExprNode>();
660 n->dtype = expr.dtype();
661 n->index = std::move(expr);
662 n->div_mode = kTruncDiv;
663 return SplitExpr(n);
664 }
665 /*!
666 * \brief Convert expr to an equivalent SplitExpr
667 * that has the specified div_mode.
668 *
669 * This function will return the same expr if its
670 * div_mode already satisfies the need.
671 *
672 * \param expr The input expr.
673 * \param div_mode The new div_mode.
674 * \return The transformed SplitExpr.
675 */
676 SplitExpr ConvertDivMode(SplitExpr expr, DivMode div_mode) {
677 if (expr->div_mode == div_mode) return expr;
678 if (expr->DivModeCompatibleTo(div_mode)) {
679 expr.CopyOnWrite()->div_mode = div_mode;
680 return expr;
681 }
682 expr = ToSplitExpr(Normalize(expr));
683 ICHECK(expr->DivModeCompatibleTo(div_mode));
684 expr.CopyOnWrite()->div_mode = div_mode;
685 return expr;
686 }
687 /*!
688 * \brief Create a SumExpr from expr.
689 * \param expr The input expr.
690 * \return The transformed SumExpr.
691 */
692 SumExpr ToSumExpr(PrimExpr expr) {
693 if (const auto* op = expr.as<SumExprNode>()) {
694 return GetRef<SumExpr>(op);
695 }
696 ObjectPtr<SumExprNode> n = make_object<SumExprNode>();
697 n->dtype = expr.dtype();
698 if (const auto* op = expr.as<IntImmNode>()) {
699 n->base = op->value;
700 return SumExpr(n);
701 } else {
702 n->args.emplace_back(ToSplitExpr(expr));
703 return SumExpr(n);
704 }
705 }
706 // Simplify the combiner used in reduce.
707 PrimExpr SimplifyReduceCombiner(const ReduceNode* op);
708};

712 return Rewriter::VisitExpr_(op);
713 }

716 PrimExpr b = this->CanonicalMutate(op->b);
717
718 // const folding
719 if (auto const_res = TryConstFold<Add>(a, b)) return const_res.value();

726 } else if (const auto* op = b.as<SumExprNode>()) {
727 ret.CopyOnWrite()->AddToSelf(GetRef<SumExpr>(op), 1);
728 } else {
729 ret.CopyOnWrite()->AddToSelf(ToSplitExpr(b), 1);
730 }
731 return std::move(ret);

758PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const MulNode* op) {
759 if (!IsIndexType(op->dtype)) {
760 return Rewriter::VisitExpr_(op);
761 }
762 // normalize
763 PrimExpr a = this->CanonicalMutate(op->a);
764 PrimExpr b = this->CanonicalMutate(op->b);
765
766 // const folding
767 if (auto const_res = TryConstFold<Mul>(a, b)) return const_res.value();
768
769 // x * c
770 if (a.as<IntImmNode>()) {
771 std::swap(a, b);
772 }
773 if (const auto* bconst = b.as<IntImmNode>()) {
774 if (a.as<SumExprNode>()) {
775 SumExpr ret = Downcast<SumExpr>(std::move(a));
776 ret.CopyOnWrite()->MulToSelf(bconst->value);
777 return std::move(ret);
778 } else {
779 SplitExpr ret = ToSplitExpr(std::move(a));
780 ret.CopyOnWrite()->MulToSelf(bconst->value);
781 return std::move(ret);
782 }
783 }
784
785 // normal path.
786 a = Normalize(a);
787 b = Normalize(b);
788 if (op->a.same_as(a) && op->b.same_as(b)) {
789 return GetRef<PrimExpr>(op);
790 } else {
791 return Mul(a, b);
792 }

795void CanonicalSimplifier::Impl::SeparateDivisibleParts(const SumExprNode* psum, int64_t coeff,
796 SumExpr* out_divisible,
797 SumExpr* out_non_divisible) {
798 auto divisible = make_object<SumExprNode>();
799 auto non_divisible = make_object<SumExprNode>();
800 divisible->dtype = psum->dtype;
801 non_divisible->dtype = psum->dtype;
802
803 if (psum->base % coeff == 0) {
804 divisible->base = psum->base;
805 } else {
806 non_divisible->base = psum->base;
807 }
808 for (const auto& e : psum->args) {
809 if (e->scale % coeff == 0) {
810 divisible->args.push_back(e);
811 } else {
812 non_divisible->args.push_back(e);
813 }
814 }
815 *out_divisible = SumExpr(divisible);
816 *out_non_divisible = SumExpr(non_divisible);
817}

819SplitExpr CanonicalSimplifier::Impl::SplitDivConst(SplitExpr lhs, int64_t cval, DivMode div_mode) {
820 ICHECK_GT(cval, 0);
821 lhs = ConvertDivMode(lhs, div_mode);
822
823 // the following rule works for both floordiv and truncdiv
824 if (lhs->scale % cval == 0) {
825 lhs.CopyOnWrite()->scale /= cval;
826 return lhs;
827 }
828
829 if (cval % lhs->scale == 0) {
830 int64_t scaled_cval = cval / lhs->scale;
831 if (lhs->upper_factor == SplitExprNode::kPosInf ||
832 lhs->upper_factor % (lhs->lower_factor * scaled_cval) == 0) {
833 // directly fold division.
834 lhs.CopyOnWrite()->scale = 1;
835 lhs.CopyOnWrite()->lower_factor *= scaled_cval;
836 lhs->Verify();
837 return lhs;
838 } else if (lhs->upper_factor <= (lhs->lower_factor * scaled_cval)) {
839 // (x % c1) / c2 => 0 when c2 >= c1
840 return ToSplitExpr(make_zero(lhs.dtype()));
841 } else {
842 // move the upper_factor modular into index.
843 lhs.CopyOnWrite()->index =
844 ModImpl(lhs->index, make_const(lhs.dtype(), lhs->upper_factor), div_mode);
845 lhs.CopyOnWrite()->upper_factor = SplitExprNode::kPosInf;
846 lhs.CopyOnWrite()->scale = 1;
847 lhs.CopyOnWrite()->lower_factor *= scaled_cval;
848 lhs->Verify();
849 return lhs;
850 }
851 }
852 // directly return the split with cval == 1
853 lhs = ToSplitExpr(Normalize(lhs));
854 ICHECK(lhs->DivModeCompatibleTo(div_mode));
855 ICHECK_EQ(lhs->scale, 1);
856 lhs.CopyOnWrite()->lower_factor *= cval;
857 lhs.CopyOnWrite()->div_mode = div_mode;
858 return lhs;
859}

861PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const DivNode* op) {
862 if (!IsIndexType(op->dtype)) {
863 return Rewriter::VisitExpr_(op);
864 }
865
866 PrimExpr a = this->CanonicalMutate(op->a);
867 PrimExpr b = this->CanonicalMutate(op->b);
868
869 // const folding
870 if (auto const_res = TryConstFold<Div>(a, b)) return const_res.value();
871 PVar<IntImm> c1;
872 // x / c1
873 if (c1.Match(b) && c1.Eval()->value > 0) {
874 int64_t cval = c1.Eval()->value;
875 if (cval == 1) return a;
876
877 if (const auto* psum = a.as<SumExprNode>()) {
878 SumExpr lhs, extra;
879 SeparateDivisibleParts(psum, cval, &lhs, &extra);
880 // can be divided by cval
881 if (extra->IsZero()) {
882 lhs.CopyOnWrite()->DivideBy(cval);
883 return std::move(lhs);
884 }
885 // both lhs and extra are non-negative
886 if (analyzer_->CanProveGreaterEqual(lhs->Normalize(), 0) &&
887 analyzer_->CanProveGreaterEqual(extra->Normalize(), 0)) {
888 lhs.CopyOnWrite()->DivideBy(cval);
889 PrimExpr temp = Normalize(extra);
890 if (const auto* pconst = temp.as<IntImmNode>()) {
891 lhs.CopyOnWrite()->AddToSelf(pconst->value / cval);
892 } else {
893 // if 0 <= extra < cval, it means the extra can be eliminated.
894 if (TryCompare(temp, cval) != CompareResult::kLT) {
895 lhs.CopyOnWrite()->AddToSelf(SplitDivConst(ToSplitExpr(temp), cval, kTruncDiv), 1);
896 }
897 }
898 return std::move(lhs);
899 }
900 } else {
901 // if a >= 0 && a < cval, then result == 0
902 auto cbound = analyzer_->const_int_bound(Normalize(a));
903 if (cbound->min_value >= 0 && cbound->max_value < cval) {
904 return make_zero(a.dtype());
905 }
906 }
907 return SplitDivConst(ToSplitExpr(std::move(a)), cval, kTruncDiv);
908 }
909 // normal path
910 a = Normalize(a);
911 b = Normalize(b);
912 if (op->a.same_as(a) && op->b.same_as(b)) {
913 return GetRef<PrimExpr>(op);
914 } else {
915 return Div(a, b);
916 }
917}

919PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const FloorDivNode* op) {
920 if (!IsIndexType(op->dtype)) {
921 return Rewriter::VisitExpr_(op);
922 }
923 PrimExpr a = this->CanonicalMutate(op->a);
924 PrimExpr b = this->CanonicalMutate(op->b);
925
926 // const folding
927 if (auto const_res = TryConstFold<FloorDiv>(a, b)) return const_res.value();
928 PVar<IntImm> c1;
929 // x / c1
930 if (c1.Match(b) && c1.Eval()->value > 0) {
931 int64_t cval = c1.Eval()->value;
932 if (cval == 1) return a;
933
934 if (const auto* psum = a.as<SumExprNode>()) {
935 SumExpr lhs, extra;
936 SeparateDivisibleParts(psum, cval, &lhs, &extra);
937 if (extra->IsZero()) {
938 lhs.CopyOnWrite()->DivideBy(cval);
939 return std::move(lhs);
940 }
941 // continue simplification.
942 lhs.CopyOnWrite()->DivideBy(cval);
943 PrimExpr temp = Normalize(extra);
944 if (const auto* pconst = temp.as<IntImmNode>()) {
945 lhs.CopyOnWrite()->AddToSelf(floordiv(pconst->value, cval));
946 } else {
947 // if 0 <= extra < cval, it means the extra can be eliminated.
948 if (!(TryCompare(temp, cval) == CompareResult::kLT &&
949 analyzer_->CanProveGreaterEqual(temp, 0))) {
950 lhs.CopyOnWrite()->AddToSelf(SplitDivConst(ToSplitExpr(temp), cval, kFloorDiv), 1);
951 }
952 }
953 return std::move(lhs);
954 } else {
955 // if a >= 0 && a < cval, then result == 0
956 auto cbound = analyzer_->const_int_bound(Normalize(a));
957 if (cbound->min_value >= 0 && cbound->max_value < cval) {
958 return make_zero(a.dtype());
959 }
960 }
961 return SplitDivConst(ToSplitExpr(std::move(a)), cval, kFloorDiv);
962 }
963 // normal path
964 a = Normalize(a);
965 b = Normalize(b);
966 if (op->a.same_as(a) && op->b.same_as(b)) {
967 return GetRef<PrimExpr>(op);
968 } else {
969 return FloorDiv(a, b);
970 }
971}

973SplitExpr CanonicalSimplifier::Impl::SplitModConst(SplitExpr lhs, int64_t cval, DivMode div_mode) {
974 ICHECK_GT(cval, 0);
975 lhs = ConvertDivMode(lhs, div_mode);
976
977 if (lhs->scale % cval == 0) {
978 lhs.CopyOnWrite()->scale = 0;
979 return lhs;
980 }
981 if (cval % lhs->scale == 0) {
982 // The rationale:
983 // (index % upper) / lower * scale % cval, given cval = scaled_cval * scale
984 // by the rule (x * c1) % (c2 * c1) => (x % c2) * c1,
985 // = (index % upper) / lower % scaled_cval * scale
986 // by the rule (x / c1) % c2 => (x % (c1 * c2)) / c1,
987 // = (index % upper) % (new_upper_factor) / lower * scale
988 int64_t scaled_cval = cval / lhs->scale;
989 int64_t new_upper_factor = lhs->lower_factor * scaled_cval;
990 // try to see if we can reduce the existing upper modular.
991 if (lhs->upper_factor == SplitExprNode::kPosInf || lhs->upper_factor % new_upper_factor == 0) {
992 // we gained a new upper factor that is smaller
993 // than the original one
994 // Perhaps there are more chances in simplifying the index
995 // Do a recursive call to simplify the mod with the new factor.
996 if (new_upper_factor < lhs->upper_factor && lhs->upper_factor != SplitExprNode::kPosInf) {
997 auto updated = ToSplitExpr(this->VisitExpr(
998 ModImpl(lhs->index, make_const(lhs.dtype(), new_upper_factor), div_mode)));
999 // re-apply the lower_factor
1000 if (lhs->lower_factor != 1) {
1001 auto ret = SplitDivConst(updated, lhs->lower_factor, div_mode);
1002 ret.CopyOnWrite()->MulToSelf(lhs->scale);
1003 return ret;
1004 } else {
1005 updated.CopyOnWrite()->MulToSelf(lhs->scale);
1006 return updated;
1007 }
1008 } else {
1009 lhs.CopyOnWrite()->upper_factor = new_upper_factor;
1010 return lhs;
1011 }
1012 } else if (new_upper_factor % lhs->upper_factor == 0) {
1013 // (x % 2) % 4 => x % 2
1014 return lhs;
1015 }
1016 }
1017 // Normalize the value.
1018 lhs = ToSplitExpr(Normalize(lhs));
1019 ICHECK(lhs->DivModeCompatibleTo(div_mode));
1020 ICHECK_EQ(lhs->scale, 1);
1021 ICHECK_EQ(lhs->lower_factor, 1);
1022 lhs.CopyOnWrite()->div_mode = div_mode;
1023 lhs.CopyOnWrite()->upper_factor = cval;
1024 return lhs;
1025}

1027PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const ModNode* op) {
1028 if (!IsIndexType(op->dtype)) {
1029 return Rewriter::VisitExpr_(op);
1030 }
1031 // normalize
1032 PrimExpr a = this->CanonicalMutate(op->a);
1033 PrimExpr b = this->CanonicalMutate(op->b);
1034
1035 // const folding
1036 if (auto const_res = TryConstFold<Mod>(a, b)) return const_res.value();
1037
1038 PVar<IntImm> c1;
1039 // x % c1
1040 if (c1.Match(b) && c1.Eval()->value > 0) {
1041 int64_t cval = c1.Eval()->value;
1042 if (const auto* psum = a.as<SumExprNode>()) {
1043 SumExpr lhs, extra;
1044 SeparateDivisibleParts(psum, cval, &lhs, &extra);
1045 if (extra->IsZero()) {
1046 return make_zero(a.dtype());
1047 }
1048 // both lhs and extra are non-negative
1049 if (analyzer_->CanProveGreaterEqual(lhs->Normalize(), 0) &&
1050 analyzer_->CanProveGreaterEqual(extra->Normalize(), 0)) {
1051 PrimExpr temp = Normalize(extra);
1052 if (temp.as<IntImmNode>()) {
1053 return truncmod(temp, c1.Eval());
1054 } else {
1055 // If temp < cval && temp >=0 then can remove the mod.
1056 if (TryCompare(temp, cval) == CompareResult::kLT) {
1057 return temp;
1058 } else {
1059 // contonue to use logic below.
1060 a = extra;
1061 psum = a.as<SumExprNode>();
1062 ICHECK(psum != nullptr);
1063 }
1064 }
1065 }
1066 // Simplify the offset constant if necessary.
1067 // (x - 5) % 3 => (x - 2) % 3 if x - 5 >= 0
1068 auto cbound = analyzer_->const_int_bound(Normalize(a));
1069 int64_t new_base = psum->base % cval;
1070 if (cbound->min_value >= 0 && cbound->min_value - psum->base + new_base >= 0) {
1071 SumExpr sum_expr = Downcast<SumExpr>(a);
1072 sum_expr.CopyOnWrite()->base = new_base;
1073 return SplitModConst(ToSplitExpr(std::move(sum_expr)), cval, kTruncDiv);
1074 }
1075 } else {
1076 // if a >= 0 && a < cval, then result == 0
1077 auto cbound = analyzer_->const_int_bound(Normalize(a));
1078 if (cbound->min_value >= 0 && cbound->max_value < cval) {
1079 return a;
1080 }
1081 }
1082 return SplitModConst(ToSplitExpr(std::move(a)), cval, kTruncDiv);
1083 }
1084 // normal path
1085 a = Normalize(a);
1086 b = Normalize(b);
1087 if (op->a.same_as(a) && op->b.same_as(b)) {
1088 return GetRef<PrimExpr>(op);
1089 } else {
1090 return Mod(a, b);
1091 }
1092}

1152// Simplify reduce expression.
1153PrimExpr CanonicalSimplifier::Impl::SimplifyReduceCombiner(const ReduceNode* op) {
1154 // First simplify the results
1155 Array<PrimExpr> simplified_result;
1156 for (const auto& res : op->combiner->result) {
1157 PrimExpr new_res = this->VisitExpr(res);
1158 simplified_result.push_back(new_res);
1159 }
1160
1161 // Which components to keep
1162 std::vector<int> used(op->combiner->result.size(), false);
1163
1164 // This function recursively marks the used components starting from
1165 // the index idx
1166 std::function<void(int)> mark_used;
1167 mark_used = [&used, &simplified_result, op, &mark_used](size_t idx) {
1168 // if the idx-th component was marked as used before, do nothing
1169 if (used[idx]) return;
1170 used[idx] = true;
1171
1172 // check if the idx-th result expr uses some lhs or rhs variables
1173 // and recursively mark the corresponding components
1174 for (size_t i = 0; i < simplified_result.size(); ++i)
1175 if (!used[i]) {
1176 if (UsesVar(simplified_result[idx],
1177 [v = op->combiner->lhs[i].get()](const VarNode* var) { return var == v; }) ||
1178 UsesVar(simplified_result[idx],
1179 [v = op->combiner->rhs[i].get()](const VarNode* var) { return var == v; }))
1180 mark_used(i);
1181 }
1182 };
1183
1184 // mark all used components starting from the value_index
1185 mark_used(op->value_index);
1186
1187 // components which have side effects should also be preserved
1188 for (size_t i = 0; i < used.size(); ++i) {
1189 if (SideEffect(op->source[i]) > CallEffectKind::kReadState ||
1190 SideEffect(op->combiner->identity_element[i]) > CallEffectKind::kReadState ||
1191 SideEffect(op->combiner->result[i]) > CallEffectKind::kReadState ||
1192 (!op->init.empty() && SideEffect(op->init[i]) > CallEffectKind::kReadState)) {
1193 mark_used(i);
1194 }
1195 }
1196
1197 int new_value_index = op->value_index;
1198 Array<PrimExpr> new_result;
1199 Array<PrimExpr> new_identity;
1200 Array<Var> new_lhs;
1201 Array<Var> new_rhs;
1202 Array<PrimExpr> new_source;
1203 Array<PrimExpr> new_init;
1204
1205 // new stuff is old stuff which is used
1206 for (size_t i = 0; i < used.size(); ++i) {
1207 if (used[i]) {
1208 // We simplify the result and identity, but not the source
1209 new_result.push_back(simplified_result[i]);
1210 new_identity.push_back(this->VisitExpr(op->combiner->identity_element[i]));
1211 new_lhs.push_back(op->combiner->lhs[i]);
1212 new_rhs.push_back(op->combiner->rhs[i]);
1213 new_source.push_back(op->source[i]);
1214 if (!op->init.empty()) new_init.push_back(op->init[i]);
1215 } else if (static_cast<int>(i) < op->value_index) {
1216 // value_index should also be adjusted
1217 new_value_index--;
1218 }
1219 }
1220
1221 CommReducer new_combiner = CommReducer(new_lhs, new_rhs, new_result, new_identity);
1222 return Reduce(new_combiner, new_source, op->axis, op->condition, new_value_index, new_init);
1223}
1224
1225PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const ReduceNode* op) {
1226 // Recursively call simplification when necessary.
1227 PrimExpr ret = RewriteSimplifier::Impl::VisitExpr_(op);
1228 op = ret.as<ReduceNode>();
1229 // already been simplified by const reduction axis removal
1230 if (op == nullptr) return ret;
1231 if (op->axis.empty()) {
1232 if (!op->init.empty()) {
1233 return this->VisitExpr(Select(op->condition,
1234 (*op->combiner.get())(op->init, op->source)[op->value_index],
1235 op->init[op->value_index]));
1236 }
1237 // Note that here we assume that the identity element is indeed identity. Without this
1238 // assumption we would have to perform a single iteration of the loop, i.e. use
1239 // `(*op->combiner.get())(op->combineop->identity_element, op->source)[op->value_index]`
1240 // instead of `op->source[op->value_index]`. The former may be more difficult to simplify.
1241 return this->VisitExpr(Select(op->condition, op->source[op->value_index],
1242 op->combiner->identity_element[op->value_index]));
1243 }
1244 // combiner simplification.
1245 ret = SimplifyReduceCombiner(op);
1246 return ret;
1247}

1249PrimExpr CanonicalSimplifier::Impl::VisitExpr_(const CastNode* op) {
1250 if (!IsIndexType(op->dtype)) {
1251 return Rewriter::VisitExpr_(op);
1252 }
1253 // normalize
1254 PrimExpr value = this->CanonicalMutate(op->value);
1255 // PushCastToChildren
1256 if (value.as<SumExprNode>()) {
1257 SumExpr se = Downcast<SumExpr>(value);
1258 if (se->CanPushCastToChildren(op->dtype, analyzer_)) {
1259 se.CopyOnWrite()->PushCastToChildren(op->dtype);
1260 return std::move(se);
1261 }
1262 }
1263 if (value.as<SplitExprNode>()) {
1264 SplitExpr se = Downcast<SplitExpr>(value);
1265 if (se->CanPushCastToChildren(op->dtype, analyzer_)) {
1266 se.CopyOnWrite()->PushCastToChildren(op->dtype);
1267 return std::move(se);
1268 }
1269 }
1270 return Rewriter::VisitExpr_(op);
1271}
