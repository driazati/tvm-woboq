42/*!
43 * \brief Rewrite-based simplifier.
44 *
45 * This class can be inheritated for other simplifiers.
46 */
47class RewriteSimplifier::Impl : public IRMutatorWithAnalyzer {
48 public:
49 using IRMutatorWithAnalyzer::VisitExpr_;
50
51 explicit Impl(Analyzer* parent) : IRMutatorWithAnalyzer(parent) {}
52
53 void Update(const Var& var, const PrimExpr& info, bool override_info);
54 PrimExpr VisitExpr_(const AddNode* op) override;
55 PrimExpr VisitExpr_(const SubNode* op) override;
56 PrimExpr VisitExpr_(const MulNode* op) override;
57 PrimExpr VisitExpr_(const DivNode* op) override;
58 PrimExpr VisitExpr_(const ModNode* op) override;
59 PrimExpr VisitExpr_(const FloorDivNode* op) override;
60 PrimExpr VisitExpr_(const FloorModNode* op) override;
61 PrimExpr VisitExpr_(const MinNode* op) override;
62 PrimExpr VisitExpr_(const MaxNode* op) override;
63 PrimExpr VisitExpr_(const EQNode* op) override;
64 PrimExpr VisitExpr_(const NENode* op) override;
65 PrimExpr VisitExpr_(const LTNode* op) override;
66 PrimExpr VisitExpr_(const LENode* op) override;
67 PrimExpr VisitExpr_(const GTNode* op) override;
68 PrimExpr VisitExpr_(const GENode* op) override;
69 PrimExpr VisitExpr_(const AndNode* op) override;
70 PrimExpr VisitExpr_(const OrNode* op) override;
71 PrimExpr VisitExpr_(const NotNode* op) override;
72 PrimExpr VisitExpr_(const SelectNode* op) override;
73 PrimExpr VisitExpr_(const CallNode* op) override;
74 PrimExpr VisitExpr_(const VarNode* op) override;
75 PrimExpr VisitExpr_(const CastNode* op) override;
76 PrimExpr VisitExpr_(const LetNode* op) override;
77
78 std::function<void()> EnterConstraint(const PrimExpr& constraint);
79
80 protected:
81 /*! \brief internal structure for comparison. */
82 enum CompareResult { kUnknown, kEQ, kGT, kGE, kLT, kLE, kNE };
83 // counter to record recursive rewrite depth.
84 int recur_depth_{0};
85 // internal variable map
86 std::unordered_map<Var, PrimExpr, ObjectPtrHash, ObjectPtrEqual> var_map_;
87
88 std::vector<PrimExpr> literal_constraints_;
89
90 // maximum number of recursion allowed during a single pass.
91 static const constexpr int kMaxRecurDepth = 5;
92
93 /*!
94 * \brief try to compare x against val.
95 * \param x The expression to be evaluated.
96 * \param val The constant value.
97 * \return comparison result.
98 */
99 CompareResult TryCompare(const PrimExpr& x, int64_t val);
100
101 /*!
102 * \brief Internal function to check whether or not to inline let.
103 * \param op The let expr.
104 * \return The inline decision.
105 */
106 bool CanInlineLet(const LetNode* op);
107
108 /*! \brief Internal function to apply constraints
109 *
110 * Tests whether the expression is known to be true or false based
111 * on existing constraints. If the expression or its negation
112 * matches a constraint, return the boolean it should be replaced
113 * with. Otherwise, return false.
114 */
115 Optional<PrimExpr> TryMatchLiteralConstraint(const PrimExpr& expr) const;
116
117 private:
118 // Whether x >= val
119 bool CanProveGreaterEqual(const PrimExpr& x, int64_t val) {
120 return analyzer_->CanProveGreaterEqual(x, val);
121 }
122 // Whether x < val
123 bool CanProveLess(const PrimExpr& x, int64_t val) { return analyzer_->CanProveLess(x, val); }
124 // Whether x == val
125 bool CanProveEqual(const PrimExpr& x, int64_t val) {
126 // TODO(tqchen) refer back to super-analyzer.
127 return TryCompare(x, val) == kEQ;
128 }
129
130 // Recursive rewrite x
131 // we limit maximum depth of recursive rewrite allowed to
132 // avoid infinite loop
133 PrimExpr RecursiveRewrite(const PrimExpr& x) {
134 if (recur_depth_ >= kMaxRecurDepth) return x;
135 ++recur_depth_;
136 PrimExpr res = this->VisitExpr(x);
137 --recur_depth_;
138 return res;
139 }
140
141 template <typename TA>
142 PConstWithTypeLike<TA> ZeroWithTypeLike(const Pattern<TA>& pattern) {
143 return PConstWithTypeLike<TA>(pattern.derived(), 0);
144 }
145
146 template <typename TA>
147 PConstWithTypeLike<TA> OneWithTypeLike(const Pattern<TA>& pattern) {
148 return PConstWithTypeLike<TA>(pattern.derived(), 1);
149 }
150};
