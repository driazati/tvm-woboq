21 * \file narrow_predicate_expression.h
22 * \brief Utility for extracting and interacting with buffer touch points

24

26#include <tvm/tir/var.h>

28#ifndef TVM_ARITH_NARROW_PREDICATE_EXPRESSION_H_
29#define TVM_ARITH_NARROW_PREDICATE_EXPRESSION_H_
30
31namespace tvm {
32namespace arith {
33
34/* \brief Narrow a true expression to remove free parameters
35 *
36 * This function provides two guarantees:
37 *
38 * 1. If the resulting expression evaluates to True, then the original
39 * expression also evaluates to True.
40 *
41 * 2. The resulting expression does not contain any of the free
42 * parameters.
43 *
44 * 3. The resulting expression does not contain any BufferLoad
45 *
46 * \param expr The expression to be examined.
47 *
48 * \param ranges The variables to be removed from the expression
49 *
50 * \returns An expression that, if true, implies that the original
51 * expression is also true.
52 */
53PrimExpr NarrowPredicateExpression(PrimExpr expr, Map<tir::Var, Range> free_parameters);
54
55} // namespace arith
56} // namespace tvm
57#endif // TVM_ARITH_NARROW_PREDICATE_EXPRESSION_H_
