29#include "pattern_match.h"

31namespace tvm {
32namespace arith {
33
34void CollectConstraints(const PrimExpr& expr, Analyzer* analyzer, std::vector<PrimExpr>* collect) {
35 collect->push_back(expr);
36
37 PVar<PrimExpr> x, y;
38 if ((x && y).Match(expr)) {
39 CollectConstraints(x.Eval(), analyzer, collect);
40 CollectConstraints(y.Eval(), analyzer, collect);
41 } else if ((!(x || y)).Match(expr)) {
42 CollectConstraints(analyzer->rewrite_simplify(tir::Not(x.Eval())), analyzer, collect);
43 CollectConstraints(analyzer->rewrite_simplify(tir::Not(y.Eval())), analyzer, collect);
44 }
45}
46
47std::vector<PrimExpr> ExtractConstraints(const PrimExpr& expr) {
48 std::vector<PrimExpr> out;
49 Analyzer analyzer;
50 CollectConstraints(expr, &analyzer, &out);
51 return out;
52}
