38class TransitiveComparisonAnalyzer::Impl {
39 public:
40 /* \brief Using previously specified knowns, compare the expressions provided
41 *
42 * \param lhs The left-hand side of the comparison
43 *
44 * \param rhs The right-hand side of the comparison
45 *
46 * \param propagate_inequalities If true, attempt to find a sequence
47 * of transitive inequalities that allow the lhs and rhs to be
48 * compared. If false, only use the known comparison that have been
49 * directly provided. Using `propagate_inequalities = false` is
50 * roughly equivalent to comparing against all known values with
51 * `ExprDeepEqual`, but also allowing for constant offsets on either
52 * side of the inequality.
53 *
54 * \return The most specific result that can be proven about the
55 * comparison. If nothing can be proven, returns kUnknown.
56 */
57 CompareResult TryCompare(const PrimExpr& lhs, const PrimExpr& rhs,
58 bool propagate_inequalities = true) const;
59
60 /*! \brief Bind a variable as being equal to a known expression
61 *
62 * \param var The variable of interest.
63 * \param expr The bound expression
64 * \param allow_override Whether to allow override of existing information.
65 */
66 void Bind(const tir::Var& var, const PrimExpr& expr, bool allow_override = false);
67
68 /*! \brief Bind a variable as being within a specified range
69 *
70 * \param var The variable of interest.
71 * \param range The known range
72 * \param allow_override Whether to allow override of existing information.
73 */
74 void Bind(const tir::Var& var, const Range& expr, bool allow_override = false);
75
76 /*!
77 * \brief Update the internal state to enter constraint.
78 * \param constraint A constraint expression.
79 *
80 * \return An exit function that must be called to cleanup. May be
81 * `nullptr`, if no cleanup is required.
82 */
83 std::function<void()> EnterConstraint(const PrimExpr& expr);
84
85 private:
86 /* \brief Internal representation of a PrimExpr
87 *
88 * The Key enum serves two purposes.
89 *
90 * 1. Providing efficiency, as compared to a PrimExpr. Two keys are
91 * equal if and only if the corresponding PrimExprs would satisfy
92 * ExprDeepEqual. This allows two expressions to be checked for
93 * equivalency, without requiring a call to ExprDeepEqual for
94 * each comparison.
95 *
96 * 2. Providing type-safety, as compared to using `size_t` directly.
97 * Requiring an explicit conversion from an integer to a Key
98 * prevents accidental comparisons, especially if both loop
99 * iterators and Keys are used in the same scope.
100 *
101 * A Key should only be obtained using the methods `ExprToKey` and
102 * `ExprToPreviousKey`.
103 */
104 enum class Key : size_t {};
105
106 /*! \brief Convert an expression to internal representation
107 *
108 * If the expression has previously been converted to the internal
109 * representation, returns the same Key as has been used previously.
110 * Otherwise, generate and return a new Key.
111 *
112 * \param expr The PrimExpr to be converted
113 *
114 * \returns The Key representing the expression
115 *
116 * \see ExprToPreviousKey
117 */
118 Key ExprToKey(const PrimExpr& expr);
119
120 /*! \brief Convert an expression to internal representation
121 *
122 * If the expression has previously been converted to the internal
123 * representation, returns the same Key as has been used previously.
124 * Otherwise, return `std::nullopt`.
125 *
126 * \param expr The PrimExpr to be converted
127 *
128 * \returns The Key representing the expression, if one exists.
129 *
130 * \see ExprToKey
131 */
132 std::optional<Key> ExprToPreviousKey(const PrimExpr& expr) const;
133
134 /*! \brief The mapping from expression to Key
135 *
136 * Should not be used directly. Instead, use the helper functions
137 * `ExprToKey` and `ExprToPreviousKey`.
138 *
139 * \see ExprToKey
140 * \see ExprToPreviousKey
141 */
142 std::unordered_map<PrimExpr, Key, StructuralHash, StructuralEqual> expr_to_key;
143
144 /*! \brief Internal representation of a comparison operator */
145 struct Comparison {
146 /*! \brief Construct a comparison that represents `lhs OP rhs +
147 * offset`, where the operation is specified by the CompareResult.
148 */
149 Comparison(Key lhs, Key rhs, int64_t offset, CompareResult result);
150
151 /*! \brief Utility function to validate that all GT and LT results
152 * have been normalized out
153 */
154 bool IsNormalized() const;
155
156 /*! \brief Move the specified expression to the LHS.
157 *
158 * \param new_lhs The argument that should be moved to the LHS of the
159 * comparison.
160 *
161 * \return If possible, returns a comparison that is equivalent to
162 * the current comparison, but with the specified LHS. If not
163 * possible, returns nullopt.
164 */
165 std::optional<Comparison> WithLHS(Key new_lhs) const;
166
167 /*! \brief Create the negation of the current comparison */
168 Comparison Negated() const;
169
170 /*! \brief Check the this comparison implies
171 *
172 * Returns true if this comparison being true implies that the
173 * other comparison must also be true. Returns false if the other
174 * comparison cannot be shown to be true.
175 */
176 bool Implies(const Comparison& other) const;
177
178 // The LHS of the comparison
179 Key lhs_;
180
181 // The RHS of the comparison, not including any constant offset.
182 Key rhs_;
183
184 // Additive offset on rhs
185 int64_t offset_{0};
186
187 // The comparison operator.
188 CompareResult result_{CompareResult::kInconsistent};
189 };
190
191 /*! \brief Generate a Comparison representing the given expression */
192 std::optional<Comparison> FromExpr(const PrimExpr& expr);
193
194 /*! \brief Utility function used by Bind and EnterConstraint
195 *
196 * \param expr The comparison expression, to be converted into
197 * internal Comparison objects.
198 *
199 * \param vec The vector to which the Comparison objects should be
200 * appended.
201 */
202 void AddKnown(const PrimExpr& expr, std::vector<Comparison>* vec);
203
204 /*! Collect known comparisons between LHS and RHS, without propagation
205 *
206 * Allows the internal representation to handle any constant
207 * offsets, without searching for a sequence of inequalities.
208 *
209 * \param lhs_key The left-hand side of the comparison
210 *
211 * \param rhs_key The right-hand side of the comparison
212 *
213 * \returns A subset of `knowns_` and `scoped_knowns_`, filtered to
214 * only include comparisons between `lhs_key` and `rhs_key`,
215 * normalized such that `lhs_key` is on the left-hand side.
216 */
217 std::vector<Comparison> CollectDirectComparisons(Key lhs_key, Key rhs_key) const;
218
219 /*! Collect known comparisons between LHS and RHS, with propagation
220 *
221 * \param lhs_key The left-hand side of the comparison
222 *
223 * \param rhs_key The right-hand side of the comparison
224 *
225 * \returns All comparisons between `lhs_key` and `rhs_key`,
226 * including the explicitly-provided comparisons in `knowns_` and
227 * `scoped_knowns_`, and comparisons provable through a series of
228 * comparisons through other values. All comparisons returned are
229 * between `lhs_key` and `rhs_key`, and are normalized such that
230 * `lhs_key` is on the left-hand side.
231 */
232 std::vector<Comparison> CollectIndirectComparisons(Key lhs_key, Key rhs_key) const;
233
234 /*! \brief Internal function used by CollectIndirectComparisons
235 *
236 * Perform a depth-first search through the space of known
237 * expressions, starting at the LHS of a comparison. In this
238 * search, each expression is a node of a graph, and each known
239 * comparison is an edge of the graph.
240 *
241 * For example, suppose we have previous knowns of (A<=B), (B<=C+1)
242 * and (C<=D-5). The expressions [A,B,C,D] are the nodes of the
243 * search space. Each comparison is an edge connecting two
244 * expressions, such as (B<=C+1) connecting the expressions B and D.
245 * If we are attempting to compare expressions A and D, a search
246 * starting at expression A could follow each edge until reaching
247 * expression D, then combine the comparisons that compose the path
248 * into the expression A<=D-4.
249 *
250 * \param lhs_key The left-hand side of the comparison
251 *
252 * \param rhs_key The right-hand side of the comparison
253 *
254 * \returns A vector of comparisons between the two expressions.
255 */
256 std::vector<Comparison> DFSFromLHS(Key lhs_key, Key rhs_key) const;
257
258 /*! \brief Combine a set of comparisons that share a LHS and RHS
259 *
260 * \param lhs_to_rhs The comparisons to merge. These should all
261 * have the same LHS and RHS. This parameter will typically be the
262 * result from `CollectDirectComparisons` or
263 * `CollectIndirectComparisons`.
264 *
265 * \param offset The constant offset in the comparison being proven.
266 * This is extracted from any additive/subtractive constants in the
267 * `PrimExpr` arguments to `TryCompare`.
268 *
269 * \returns The possible comparisons between LHS and RHS provided
270 * inequalities.
271 */
272 CompareResult MergeComparisons(const std::vector<Comparison>& lhs_to_rhs, int64_t offset) const;
273
274 /*! \brief Previous Range bindings
275 *
276 * Tracked separatedly to handle the `allow_override` option used by
277 * all sub-analyzers when binding variables.
278 */
279 Map<Var, Range> prev_bindings_;
280
281 /*! \brief Known comparisons based on definitionally-true statements
282 *
283 * For example, a Let binding, or the range of an iterator. These
284 * known statements are always true, based on the definition site of
285 * the variable. e.g. A loop iterator may never exceed the bounds
286 * of its loop.
287 */
288 std::vector<Comparison> knowns_;
289
290 /*! \brief Known comparisons based on scoped conditions
291 *
292 * For example, the condition of an IfThenElse. These known
293 * statements may only be used within the scope of the statement
294 * from which they were derived. e.g. After exiting an IfThenElse,
295 * the condition may no longer be true.
296 */
297 std::vector<Comparison> scoped_knowns_;
298};

302// Internal utility, return the CompareResult resulting from swapping
303// the left-hand side with the right-hand side.
304CompareResult Reverse(CompareResult res) {
305 switch (res) {
306 case CompareResult::kInconsistent:
307 return CompareResult::kInconsistent;
308 case CompareResult::kEQ:
309 return CompareResult::kEQ;
310 case CompareResult::kLT:
311 return CompareResult::kGT;
312 case CompareResult::kLE:
313 return CompareResult::kGE;
314 case CompareResult::kGT:
315 return CompareResult::kLT;
316 case CompareResult::kGE:
317 return CompareResult::kLE;
318 case CompareResult::kNE:
319 return CompareResult::kNE;
320 case CompareResult::kUnknown:
321 return CompareResult::kUnknown;
322 default:
323 LOG(FATAL) << "Invalid CompareResult: " << static_cast<int>(res);
324 return CompareResult::kInconsistent;
325 }
326}
327
328// Internal utility, return the CompareResult resulting from negating
329// the comparison.
330CompareResult Negate(CompareResult res) {
331 switch (res) {
332 case CompareResult::kInconsistent:
333 return CompareResult::kInconsistent;
334 case CompareResult::kUnknown:
335 return CompareResult::kUnknown;
336 default:
337 return CompareResult(~static_cast<int>(res) & static_cast<int>(CompareResult::kUnknown));
338 }
339}

341// Internal utility, extract constant offsets out of the two sides of
342// a comparison. Given lhs and rhs, return a tuple of three elements
343// (lhs_inner, rhs_inner, offset), such that (lhs OP rhs) and
344// (lhs_inner OP rhs_inner + offset) are equivalent.
345std::tuple<PrimExpr, PrimExpr, int64_t> ExtractOffsets(const PrimExpr& lhs, const PrimExpr& rhs) {
346 auto extract_offset = [](const PrimExpr& expr) -> std::pair<PrimExpr, int64_t> {
347 PVar<PrimExpr> x;
348 PVar<IntImm> c;
349 if ((x + c).Match(expr)) {
350 return {x.Eval(), c.Eval()->value};
351 } else if ((x - c).Match(expr)) {
352 return {x.Eval(), -c.Eval()->value};
353 } else if (c.Match(expr)) {
354 return {0, c.Eval()->value};
355 } else {
356 return {expr, 0};
357 }
358 };
359
360 auto lhs_split = extract_offset(lhs);
361 auto rhs_split = extract_offset(rhs);
362 return {lhs_split.first, rhs_split.first, rhs_split.second - lhs_split.second};
363}

367std::optional<TransitiveComparisonAnalyzer::Impl::Comparison>
368TransitiveComparisonAnalyzer::Impl::FromExpr(const PrimExpr& expr) {
369 CompareResult res;
370 PVar<PrimExpr> x, y;
371 if ((x <= y).Match(expr)) {
372 res = CompareResult::kLE;
373 } else if ((x >= y).Match(expr)) {
374 res = CompareResult::kGE;
375 } else if ((x < y).Match(expr)) {
376 res = CompareResult::kLT;
377 } else if ((x > y).Match(expr)) {
378 res = CompareResult::kGT;
379 } else if ((x == y).Match(expr)) {
380 res = CompareResult::kEQ;
381 } else if ((x != y).Match(expr)) {
382 res = CompareResult::kNE;
383 } else {
384 return std::nullopt;
385 }
386
387 PrimExpr lhs_expr = x.Eval();
388 PrimExpr rhs_expr = y.Eval();
389
390 if (lhs_expr.as<IntImmNode>() && rhs_expr.as<IntImmNode>()) {
391 return std::nullopt;
392 }
393
394 auto [lhs, rhs, offset] = ExtractOffsets(lhs_expr, rhs_expr);
395 Key lhs_key = ExprToKey(lhs);
396 Key rhs_key = ExprToKey(rhs);
397
398 return Comparison(lhs_key, rhs_key, offset, res);
399}

401TransitiveComparisonAnalyzer::Impl::Comparison::Comparison(Key lhs, Key rhs, int64_t offset,
402 CompareResult result)
403 : lhs_(lhs), rhs_(rhs), offset_(offset), result_(result) {
404 // Normalize the comparison to remove LT and GT expressions,
405 // reducing the number of operators that must be handled later. By
406 // eliminating LT and GT, instead of eliminating LE or GE, a
407 // potential off-by-one error is avoided.
408 //
409 // For floating-point numbers, (x < y + c1) and (y < z + c2) implies
410 // that (x < z + (c1 + c2)). For integer types, which the
411 // TransitiveComparisonAnalyzer is intended for use with integers,
412 // LT or GT can give a tighter constraint, though with a less
413 // convenient symmetry.
414 //
415 // i < j + c1, j < k + c2
416 // i <= j + c1 - 1, j <= k + c2 - 1
417 // i + 1 - c1 <= j, j <= k + c2 - 1
418 // i + 1 - c1 <= k + c2 - 1
419 // i <= k + c1 + c2 - 2
420 // i < k + (c1 + c2 - 1)
421 //
422 // By always working with LE and GE comparisons, we avoid needing to
423 // handle the offset of one that would be introduced by LT and GT at
424 // all points of use. The only point of use for LT and GT is when
425 // normalizing comparisons (i.e. this constructor).
426
427 if (result_ == CompareResult::kLT) {
428 result_ = CompareResult::kLE;
429 offset_ -= 1;
430 }
431 if (result_ == CompareResult::kGT) {
432 result_ = CompareResult::kGE;
433 offset_ += 1;
434 }
435}

437std::optional<TransitiveComparisonAnalyzer::Impl::Key>
438TransitiveComparisonAnalyzer::Impl::ExprToPreviousKey(const PrimExpr& expr) const {
439 auto it = expr_to_key.find(expr);
440 if (it != expr_to_key.end()) {
441 return it->second;
442 } else {
443 return std::nullopt;
444 }
445}
446
447TransitiveComparisonAnalyzer::Impl::Key TransitiveComparisonAnalyzer::Impl::ExprToKey(
448 const PrimExpr& expr) {
449 if (auto prev = ExprToPreviousKey(expr)) {
450 return prev.value();
451 } else {
452 Key new_key = Key(expr_to_key.size());
453 expr_to_key[expr] = new_key;
454 return new_key;
455 }
456}

458bool TransitiveComparisonAnalyzer::Impl::Comparison::IsNormalized() const {
459 // These < and > should be removed during normalization. See the
460 // `Comparison::Comparison` constructor for further details.
461 return result_ != CompareResult::kLT && result_ != CompareResult::kGT;
462}

464std::optional<TransitiveComparisonAnalyzer::Impl::Comparison>
465TransitiveComparisonAnalyzer::Impl::Comparison::WithLHS(Key new_lhs) const {
466 if (new_lhs == lhs_) {
467 return *this;
468 } else if (new_lhs == rhs_) {
469 return Comparison(rhs_, lhs_, -offset_, Reverse(result_));
470 } else {
471 return std::nullopt;
472 }
473}

480bool TransitiveComparisonAnalyzer::Impl::Comparison::Implies(
481 const TransitiveComparisonAnalyzer::Impl::Comparison& other) const {
482 ICHECK(lhs_ == other.lhs_);
483 ICHECK(rhs_ == other.rhs_);
484 ICHECK(IsNormalized());
485 ICHECK(other.IsNormalized());
486
487 if (result_ == other.result_ && offset_ == other.offset_) {
488 // if c1 == c2, x != y + c1 => x != y + c2
489 // if c1 == c2, x == y + c1 => x == y + c2
490 return true;
491 }
492
493 if (other.result_ == CompareResult::kLE && offset_ <= other.offset_) {
494 if (result_ == CompareResult::kEQ || result_ == CompareResult::kLE) {
495 // if c1 <= c2, x <= y + c1 => x <= y + c2
496 // if c1 <= c2, x == y + c1 => x <= y + c2
497 return true;
498 }
499 }
500
501 if (other.result_ == CompareResult::kGE && offset_ >= other.offset_) {
502 if (result_ == CompareResult::kEQ || result_ == CompareResult::kGE) {
503 // if c1 >= c2, x == y + c1 => x >= y + c2
504 // if c1 >= c2, x >= y + c1 => x >= y + c2
505 return true;
506 }
507 }
508
509 if (other.result_ == CompareResult::kNE) {
510 if (result_ == CompareResult::kEQ && offset_ != other.offset_) {
511 // if c1 != c2, x == y + c1 => x != y + c2
512 return true;
513 }
514
515 if (result_ == CompareResult::kLE && offset_ < other.offset_) {
516 // if c1 < c2, x <= y + c1 => x < y + c2 => x != y + c2
517 return true;
518 }
519
520 if (result_ == CompareResult::kGE && offset_ > other.offset_) {
521 // if c1 != c2, x >= y + c1 => x > y + c2 => x != y + c2
522 return true;
523 }
524 }
525
526 return false;
527}

548void TransitiveComparisonAnalyzer::Impl::AddKnown(const PrimExpr& expr,
549 std::vector<Comparison>* vec) {
550 for (const auto& subexpr : ExtractConstraints(expr)) {
551 if (tir::SideEffect(expr) <= tir::CallEffectKind::kPure) {
552 if (auto cmp = FromExpr(subexpr)) {
553 vec->push_back(cmp.value());
554 }
555 }
556 }

558
559void TransitiveComparisonAnalyzer::Impl::Bind(const tir::Var& var, const Range& range,
560 bool allow_override) {
561 auto it = prev_bindings_.find(var);
562 if (it != prev_bindings_.end()) {
563 ExprDeepEqual expr_equal;
564 bool differs_from_previous = !expr_equal(range->min, (*it).second->min) ||
565 !expr_equal(range->extent, (*it).second->extent);
566 if (differs_from_previous) {
567 ICHECK(allow_override) << "Binding of variable " << var << " as " << range
568 << " conflicts with previous binding as " << (*it).second;
569 if (auto key = ExprToPreviousKey(var)) {
570 knowns_.erase(std::remove_if(knowns_.begin(), knowns_.end(),
571 [&](const auto& known) { return known.lhs_ == key.value(); }),
572 knowns_.end());
573 }
574 }
575 }
576
577 prev_bindings_.Set(var, range);
578
579 if (is_const_int(range->extent, 1)) {
580 AddKnown(var == range->min, &knowns_);
581 } else {
582 AddKnown(var >= range->min, &knowns_);
583 AddKnown(var < range->min + range->extent, &knowns_);
584 }
585}

587void TransitiveComparisonAnalyzer::Impl::Bind(const tir::Var& var, const PrimExpr& expr,
588 bool allow_override) {
589 Bind(var, Range::FromMinExtent(expr, 1), allow_override);
590}

592std::function<void()> TransitiveComparisonAnalyzer::Impl::EnterConstraint(const PrimExpr& expr) {
593 size_t old_literal_size = scoped_knowns_.size();
594 AddKnown(expr, &scoped_knowns_);
595 size_t new_literal_size = scoped_knowns_.size();
596
597 auto frecover = [old_literal_size, new_literal_size, this]() {
598 ICHECK_EQ(scoped_knowns_.size(), new_literal_size);
599 scoped_knowns_.erase(scoped_knowns_.begin() + old_literal_size, scoped_knowns_.end());
600 };
601 return frecover;
602}
603
604CompareResult TransitiveComparisonAnalyzer::Impl::TryCompare(const PrimExpr& lhs_expr,
605 const PrimExpr& rhs_expr,
606 bool propagate_inequalities) const {
607 // Currently only supports integer checks
608 if (!lhs_expr.dtype().is_int() || !rhs_expr.dtype().is_int()) {
609 return CompareResult::kUnknown;
610 }
611
612 // Bail out early if possible. This int check should have been
613 // constant-folded earlier, so this check shouldn't occur.
614 auto* x_int = lhs_expr.as<IntImmNode>();
615 auto* y_int = rhs_expr.as<IntImmNode>();
616 if (x_int && y_int) {
617 if (x_int->value < y_int->value) {
618 return CompareResult::kLT;
619 } else if (x_int->value > y_int->value) {
620 return CompareResult::kGT;
621 } else {
622 return CompareResult::kEQ;
623 }
624 }
625
626 auto [lhs, rhs, offset] = ExtractOffsets(lhs_expr, rhs_expr);
627 auto lhs_key = ExprToPreviousKey(lhs);
628 auto rhs_key = ExprToPreviousKey(rhs);
629
630 if (!lhs_key.has_value() || !rhs_key.has_value()) {
631 return CompareResult::kUnknown;
632 }
633
634 auto lhs_to_rhs = [&]() {
635 if (propagate_inequalities) {
636 return CollectIndirectComparisons(lhs_key.value(), rhs_key.value());
637 } else {
638 return CollectDirectComparisons(lhs_key.value(), rhs_key.value());
639 }
640 }();
641 return MergeComparisons(lhs_to_rhs, offset);
642}

644std::vector<TransitiveComparisonAnalyzer::Impl::Comparison>
645TransitiveComparisonAnalyzer::Impl::CollectDirectComparisons(Key lhs_key, Key rhs_key) const {
646 std::vector<Comparison> output;
647
648 auto append_known = [&](Comparison cmp) {
649 if (auto normalized = cmp.WithLHS(lhs_key)) {
650 if (normalized.value().rhs_ == rhs_key) {
651 output.push_back(normalized.value());
652 }
653 }
654 };
655
656 for (const auto& known : knowns_) {
657 append_known(known);
658 }
659 for (const auto& known : scoped_knowns_) {
660 append_known(known);
661 }
662
663 return output;
664}
665
666std::vector<TransitiveComparisonAnalyzer::Impl::Comparison>
667TransitiveComparisonAnalyzer::Impl::CollectIndirectComparisons(Key lhs_key, Key rhs_key) const {
668 auto output = DFSFromLHS(lhs_key, rhs_key);
669 for (Comparison cmp : DFSFromLHS(rhs_key, lhs_key)) {
670 auto opt_normalized = cmp.WithLHS(lhs_key);
671 ICHECK(opt_normalized.has_value());
672 output.push_back(opt_normalized.value());
673 }
674 return output;
675}

677std::vector<TransitiveComparisonAnalyzer::Impl::Comparison>
678TransitiveComparisonAnalyzer::Impl::DFSFromLHS(Key lhs_key, Key rhs_key) const {
679 // Everything in `to_visit` has lhs as its lhs.
680 std::unordered_set<Key> seen;
681 std::unordered_set<Key> to_visit;
682 std::unordered_map<Key, std::vector<Comparison>> compared_to_lhs;
683
684 // Utility function to add a new known statement
685 auto declare_known = [&](Comparison cmp) {
686 std::vector<Comparison>& knowns = compared_to_lhs[cmp.rhs_];
687
688 // The comparison adds no new information, no modification
689 // required.
690 for (auto& prev_known : knowns) {
691 if (prev_known.Implies(cmp)) {
692 return;
693 }
694 }
695
696 // New information may require visiting a new expression.
697 if (cmp.rhs_ != rhs_key && !seen.count(cmp.rhs_)) {
698 to_visit.insert(cmp.rhs_);
699 seen.insert(cmp.rhs_);
700 }
701
702 // This comparison is a stronger version of a previous constraint.
703 // Therefore, replace the old version entirely.
704 for (auto& prev_known : knowns) {
705 if (cmp.Implies(prev_known)) {
706 prev_known = cmp;
707 return;
708 }
709 }
710
711 // Neither a superset nor a subset of previously known
712 // constraints, must be tracked separately.
713 knowns.push_back(cmp);
714 };
715
716 // Initialize the search based on any known (in)equalities that use
717 // the LHS of the comparison.
718 for (const auto& known : knowns_) {
719 if (auto normalized = known.WithLHS(lhs_key)) {
720 declare_known(normalized.value());
721 }
722 }
723 for (const auto& known : scoped_knowns_) {
724 if (auto normalized = known.WithLHS(lhs_key)) {
725 declare_known(normalized.value());
726 }
727 }
728
729 // Walk through the space of all comparisons that can be made with
730 // LHS.
731 while (to_visit.size()) {
732 Key middle_key = *to_visit.begin();
733 to_visit.erase(to_visit.begin());
734
735 std::vector<Comparison>& prev_knowns_using_middle = compared_to_lhs.at(middle_key);
736 ICHECK(compared_to_lhs.count(middle_key));
737
738 std::vector<Comparison> new_knowns_using_lhs;
739
740 auto attempt_transitive = [&](Comparison cmp) {
741 ICHECK(cmp.IsNormalized());
742
743 Key right_key = cmp.rhs_;
744
745 if (right_key == lhs_key) {
746 return;
747 }
748
749 for (const auto& prev : prev_knowns_using_middle) {
750 CompareResult new_result = CompareResult::kUnknown;
751 int64_t new_offset = prev.offset_ + cmp.offset_;
752
753 if (prev.result_ == CompareResult::kEQ) {
754 // x == y + c1 && y OP z + c2, x OP z + (c1 + c2)
755 new_result = cmp.result_;
756 } else if (cmp.result_ == CompareResult::kEQ) {
757 // x OP y + c1 && y == z + c2, x OP z + (c1 + c2)
758 new_result = prev.result_;
759 } else if (prev.result_ == cmp.result_ &&
760 (prev.result_ == CompareResult::kLE || prev.result_ == CompareResult::kGE)) {
761 // x <= y + c1 && y <= z + c2, x <= z + (c1 + c2)
762 // x >= y + c1 && y >= z + c2, x >= z + (c1 + c2)
763 //
764 // This condition is much simpler to write than the
765 // equivalent handling of < or of >, which is why the
766 // inequalities are normalized to <= and to >=. See
767 // `TransitiveComparisonAnalyzer::Impl::Comparison::Comparison`
768 // for further details.
769 new_result = prev.result_;
770 }
771
772 if (new_result != CompareResult::kUnknown) {
773 Comparison new_known(lhs_key, right_key, new_offset, new_result);
774 new_knowns_using_lhs.push_back(new_known);
775 }
776 }
777 };
778
779 // Attempt to prove a new comparison using one of the original
780 // known comparisons. We want to find a known such that
781 // `(LHS OP1 middle) && (middle OP2 right)` can be simplified
782 // into `(LHS OP3 right)`.
783 //
784 // Note: The right side is this step is not necessarily the RHS of
785 // the comparison we're trying to prove, as we may need to find
786 // intermediate comparisons first. For example, if we know that
787 // `a<=b`, `b<=c`, and `c<=d`, and we wish to prove that `a<=d`,
788 // we must first combine `a<=b` and `b<=c` into `a<=c`. During
789 // this first step, `b` is the "middle" and `c` is the "right".
790 // The next step can then combind `a<=c` and `c<=d` into `a<=d`.
791 for (const auto& known : knowns_) {
792 if (auto cmp = known.WithLHS(middle_key)) {
793 attempt_transitive(cmp.value());
794 }
795 }
796
797 for (const auto& known : scoped_knowns_) {
798 if (auto cmp = known.WithLHS(middle_key)) {
799 attempt_transitive(cmp.value());
800 }
801 }
802
803 // Collect together all new knowns, marking new nodes for visiting
804 // as needed.
805 for (const auto& new_known : new_knowns_using_lhs) {
806 declare_known(new_known);
807 }
808 }
809
810 if (auto it = compared_to_lhs.find(rhs_key); it != compared_to_lhs.end()) {
811 return it->second;
812 } else {
813 // There are known comparisons involving the LHS and the RHS, but
814 // no path that connects the two expressions.
815 return {};
816 }
817}
818
819CompareResult TransitiveComparisonAnalyzer::Impl::MergeComparisons(
820 const std::vector<Comparison>& lhs_to_rhs, int64_t offset) const {
821 // Just because we found a comparison involving LHS and RHS doesn't
822 // mean that it's useful. e.g. Knowing that `x < y` doesn't let us
823 // prove whether `x + 5 < y`.
824 CompareResult result = CompareResult::kUnknown;
825 for (const auto& cmp : lhs_to_rhs) {
826 switch (cmp.result_) {
827 case CompareResult::kInconsistent:
828 result = CompareResult::kInconsistent;
829 break;
830
831 case CompareResult::kEQ:
832 if (offset == cmp.offset_) {
833 result = result & CompareResult::kEQ;
834 } else {
835 result = result & CompareResult::kNE;
836 }
837 break;
838
839 case CompareResult::kLE:
840 if (cmp.offset_ < offset) {
841 result = result & CompareResult::kLT;
842 } else if (cmp.offset_ <= offset) {
843 result = result & CompareResult::kLE;
844 }
845 break;
846
847 case CompareResult::kGE:
848 if (cmp.offset_ > offset) {
849 result = result & CompareResult::kGT;
850 } else if (cmp.offset_ >= offset) {
851 result = result & CompareResult::kGE;
852 }
853 break;
854
855 case CompareResult::kNE:
856 if (offset == cmp.offset_) {
857 result = result & CompareResult::kNE;
858 }
859 break;
860
861 case CompareResult::kUnknown:
862 break;
863
864 case CompareResult::kGT:
865 case CompareResult::kLT:
866 LOG(FATAL) << "Internal error, normalized comparisons should only include <= and >=";
867 return CompareResult::kInconsistent;
868
869 default:
870 LOG(FATAL) << "Invalid CompareResult: " << static_cast<int>(cmp.result_);
871 return CompareResult::kInconsistent;
872 }
873 }
874
875 return result;
876}
