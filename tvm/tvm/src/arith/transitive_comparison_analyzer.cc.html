38class TransitiveComparisonAnalyzer::Impl {
39 public:
40 /* \brief Using previously specified knowns, compare the expressions provided
41 *
42 * \param lhs The left-hand side of the comparison
43 *
44 * \param rhs The right-hand side of the comparison
45 *
46 * \return The most specific result that can be proven about the
47 * comparison. If nothing can be proven, returns kUnknown.
48 */
49 CompareResult TryCompare(const PrimExpr& lhs, const PrimExpr& rhs) const;
50
51 /*! \brief Bind a variable as being equal to a known expression
52 *
53 * \param var The variable of interest.
54 * \param expr The bound expression
55 * \param allow_override Whether to allow override of existing information.
56 */
57 void Bind(const tir::Var& var, const PrimExpr& expr, bool allow_override = false);
58
59 /*! \brief Bind a variable as being within a specified range
60 *
61 * \param var The variable of interest.
62 * \param range The known range
63 * \param allow_override Whether to allow override of existing information.
64 */
65 void Bind(const tir::Var& var, const Range& expr, bool allow_override = false);
66
67 /*!
68 * \brief Update the internal state to enter constraint.
69 * \param constraint A constraint expression.
70 *
71 * \return An exit function that must be called to cleanup. May be
72 * `nullptr`, if no cleanup is required.
73 */
74 std::function<void()> EnterConstraint(const PrimExpr& expr);
75
76 private:
77 /* \brief Internal representation of a PrimExpr
78 *
79 * The Key enum serves two purposes.
80 *
81 * 1. Providing efficiency, as compared to a PrimExpr. Two keys are
82 * equal if and only if the corresponding PrimExprs would satisfy
83 * ExprDeepEqual. This allows two expressions to be checked for
84 * equivalency, without requiring a call to ExprDeepEqual for
85 * each comparison.
86 *
87 * 2. Providing type-safety, as compared to using `size_t` directly.
88 * Requiring an explicit conversion from an integer to a Key
89 * prevents accidental comparisons, especially if both loop
90 * iterators and Keys are used in the same scope.
91 *
92 * A Key should only be obtained using the methods `ExprToKey` and
93 * `ExprToPreviousKey`.
94 */
95 enum class Key : size_t {};
96
97 /*! \brief Convert an expression to internal representation
98 *
99 * If the expression has previously been converted to the internal
100 * representation, returns the same Key as has been used previously.
101 * Otherwise, generate and return a new Key.
102 *
103 * \param expr The PrimExpr to be converted
104 *
105 * \returns The Key representing the expression
106 *
107 * \see ExprToPreviousKey
108 */
109 Key ExprToKey(const PrimExpr& expr);
110
111 /*! \brief Convert an expression to internal representation
112 *
113 * If the expression has previously been converted to the internal
114 * representation, returns the same Key as has been used previously.
115 * Otherwise, return `std::nullopt`.
116 *
117 * \param expr The PrimExpr to be converted
118 *
119 * \returns The Key representing the expression, if one exists.
120 *
121 * \see ExprToKey
122 */
123 std::optional<Key> ExprToPreviousKey(const PrimExpr& expr) const;
124
125 /*! \brief The mapping from expression to Key
126 *
127 * Should not be used directly. Instead, use the helper functions
128 * `ExprToKey` and `ExprToPreviousKey`.
129 *
130 * \see ExprToKey
131 * \see ExprToPreviousKey
132 */
133 std::unordered_map<PrimExpr, Key, StructuralHash, StructuralEqual> expr_to_key;
134
135 /*! \brief Internal representation of a comparison operator */
136 struct Comparison {
137 /*! \brief Construct a comparison that represents `lhs OP rhs +
138 * offset`, where the operation is specified by the CompareResult.
139 */
140 Comparison(Key lhs, Key rhs, int64_t offset, CompareResult result);
141
142 /*! \brief Utility function to validate that all GT and LT results
143 * have been normalized out
144 */
145 bool IsNormalized() const;
146
147 /*! \brief Move the specified expression to the LHS.
148 *
149 * \param new_lhs The argument that should be moved to the LHS of the
150 * comparison.
151 *
152 * \return If possible, returns a comparison that is equivalent to
153 * the current comparison, but with the specified LHS. If not
154 * possible, returns nullopt.
155 */
156 std::optional<Comparison> WithLHS(Key new_lhs) const;
157
158 /*! \brief Create the negation of the current comparison */
159 Comparison Negated() const;
160
161 /*! \brief Check the this comparison implies
162 *
163 * Returns true if this comparison being true implies that the
164 * other comparison must also be true. Returns false if the other
165 * comparison cannot be shown to be true.
166 */
167 bool Implies(const Comparison& other) const;
168
169 // The LHS of the comparison
170 Key lhs_;
171
172 // The RHS of the comparison, not including any constant offset.
173 Key rhs_;
174
175 // Additive offset on rhs
176 int64_t offset_{0};
177
178 // The comparison operator.
179 CompareResult result_{CompareResult::kInconsistent};
180 };
181
182 /*! \brief Generate a Comparison representing the given expression */
183 std::optional<Comparison> FromExpr(const PrimExpr& expr);
184
185 /*! \brief Utility function used by Bind and EnterConstraint
186 *
187 * \param expr The comparison expression, to be converted into
188 * internal Comparison objects.
189 *
190 * \param vec The vector to which the Comparison objects should be
191 * appended.
192 */
193 void AddKnown(const PrimExpr& expr, std::vector<Comparison>* vec);
194
195 /*! \brief Attempt to compare the expressions, starting at the lhs.
196 *
197 * Perform a depth-first search through the space of known
198 * expressions, starting at the LHS of a comparison. In this
199 * search, each expression is a node of a graph, and each known
200 * comparison is an edge of the graph.
201 *
202 * For example, suppose we have previous knowns of (A<=B), (B<=C+1)
203 * and (C<=D-5). The expressions [A,B,C,D] are the nodes of the
204 * search space. Each comparison is an edge connecting two
205 * expressions, such as (B<=C+1) connecting the expressions B and D.
206 * If we are attempting to compare expressions A and D, a search
207 * starting at expression A could follow each edge until reaching
208 * expression D, then combine the comparisons that compose the path
209 * into the expression A<=D-4.
210 *
211 * \param lhs The left-hand side of the comparison
212 *
213 * \param rhs The right-hand side of the comparison
214 *
215 * \return The result of the comparison
216 */
217 CompareResult DFSFromLHS(Key lhs_key, Key rhs_key, int64_t offset, const PrimExpr& lhs,
218 const PrimExpr& rhs) const;
219
220 /*! \brief Previous Range bindings
221 *
222 * Tracked separatedly to handle the `allow_override` option used by
223 * all sub-analyzers when binding variables.
224 */
225 Map<Var, Range> prev_bindings_;
226
227 /*! \brief Known comparisons based on definitionally-true statements
228 *
229 * For example, a Let binding, or the range of an iterator. These
230 * known statements are always true, based on the definition site of
231 * the variable. e.g. A loop iterator may never exceed the bounds
232 * of its loop.
233 */
234 std::vector<Comparison> knowns_;
235
236 /*! \brief Known comparisons based on scoped conditions
237 *
238 * For example, the condition of an IfThenElse. These known
239 * statements may only be used within the scope of the statement
240 * from which they were derived. e.g. After exiting an IfThenElse,
241 * the condition may no longer be true.
242 */
243 std::vector<Comparison> scoped_knowns_;
244};

248// Internal utility, return the CompareResult resulting from swapping
249// the left-hand side with the right-hand side.
250CompareResult Reverse(CompareResult res) {
251 switch (res) {
252 case CompareResult::kInconsistent:
253 return CompareResult::kInconsistent;
254 case CompareResult::kEQ:
255 return CompareResult::kEQ;
256 case CompareResult::kLT:
257 return CompareResult::kGT;
258 case CompareResult::kLE:
259 return CompareResult::kGE;
260 case CompareResult::kGT:
261 return CompareResult::kLT;
262 case CompareResult::kGE:
263 return CompareResult::kLE;
264 case CompareResult::kNE:
265 return CompareResult::kNE;
266 case CompareResult::kUnknown:
267 return CompareResult::kUnknown;
268 default:
269 LOG(FATAL) << "Invalid CompareResult: " << static_cast<int>(res);
270 return CompareResult::kInconsistent;
271 }
272}
273
274// Internal utility, return the CompareResult resulting from negating
275// the comparison.
276CompareResult Negate(CompareResult res) {
277 switch (res) {
278 case CompareResult::kInconsistent:
279 return CompareResult::kInconsistent;
280 case CompareResult::kUnknown:
281 return CompareResult::kUnknown;
282 default:
283 return CompareResult(~static_cast<int>(res) & static_cast<int>(CompareResult::kUnknown));
284 }
285}

287// Internal utility, extract constant offsets out of the two sides of
288// a comparison. Given lhs and rhs, return a tuple of three elements
289// (lhs_inner, rhs_inner, offset), such that (lhs OP rhs) and
290// (lhs_inner OP rhs_inner + offset) are equivalent.
291std::tuple<PrimExpr, PrimExpr, int64_t> ExtractOffsets(const PrimExpr& lhs, const PrimExpr& rhs) {
292 auto extract_offset = [](const PrimExpr& expr) -> std::pair<PrimExpr, int64_t> {
293 PVar<PrimExpr> x;
294 PVar<IntImm> c;
295 if ((x + c).Match(expr)) {
296 return {x.Eval(), c.Eval()->value};
297 } else if ((x - c).Match(expr)) {
298 return {x.Eval(), -c.Eval()->value};
299 } else if (c.Match(expr)) {
300 return {0, c.Eval()->value};
301 } else {
302 return {expr, 0};
303 }
304 };
305
306 auto lhs_split = extract_offset(lhs);
307 auto rhs_split = extract_offset(rhs);
308 return {lhs_split.first, rhs_split.first, rhs_split.second - lhs_split.second};
309}

313std::optional<TransitiveComparisonAnalyzer::Impl::Comparison>
314TransitiveComparisonAnalyzer::Impl::FromExpr(const PrimExpr& expr) {
315 CompareResult res;
316 PVar<PrimExpr> x, y;
317 if ((x <= y).Match(expr)) {
318 res = CompareResult::kLE;
319 } else if ((x >= y).Match(expr)) {
320 res = CompareResult::kGE;
321 } else if ((x < y).Match(expr)) {
322 res = CompareResult::kLT;
323 } else if ((x > y).Match(expr)) {
324 res = CompareResult::kGT;
325 } else if ((x == y).Match(expr)) {
326 res = CompareResult::kEQ;
327 } else if ((x != y).Match(expr)) {
328 res = CompareResult::kNE;
329 } else {
330 return std::nullopt;
331 }
332
333 PrimExpr lhs_expr = x.Eval();
334 PrimExpr rhs_expr = y.Eval();
335
336 if (lhs_expr.as<IntImmNode>() && rhs_expr.as<IntImmNode>()) {
337 return std::nullopt;
338 }
339
340 auto [lhs, rhs, offset] = ExtractOffsets(lhs_expr, rhs_expr);
341 Key lhs_key = ExprToKey(lhs);
342 Key rhs_key = ExprToKey(rhs);
343
344 return Comparison(lhs_key, rhs_key, offset, res);
345}

347TransitiveComparisonAnalyzer::Impl::Comparison::Comparison(Key lhs, Key rhs, int64_t offset,
348 CompareResult result)
349 : lhs_(lhs), rhs_(rhs), offset_(offset), result_(result) {
350 // Normalize the comparison to remove LT and GT expressions,
351 // reducing the number of operators that must be handled later. By
352 // eliminating LT and GT, instead of eliminating LE or GE, a
353 // potential off-by-one error is avoided.
354 //
355 // For floating-point numbers, (x < y + c1) and (y < z + c2) implies
356 // that (x < z + (c1 + c2)). For integer types, which the
357 // TransitiveComparisonAnalyzer is intended for use with integers,
358 // LT or GT can give a tighter constraint, though with a less
359 // convenient symmetry.
360 //
361 // i < j + c1, j < k + c2
362 // i <= j + c1 - 1, j <= k + c2 - 1
363 // i + 1 - c1 <= j, j <= k + c2 - 1
364 // i + 1 - c1 <= k + c2 - 1
365 // i <= k + c1 + c2 - 2
366 // i < k + (c1 + c2 - 1)
367 //
368 // By always working with LE and GE comparisons, we avoid needing to
369 // handle the offset of one that would be introduced by LT and GT at
370 // all points of use. The only point of use for LT and GT is when
371 // normalizing comparisons (i.e. this constructor).
372
373 if (result_ == CompareResult::kLT) {
374 result_ = CompareResult::kLE;
375 offset_ -= 1;
376 }
377 if (result_ == CompareResult::kGT) {
378 result_ = CompareResult::kGE;
379 offset_ += 1;
380 }
381}

384TransitiveComparisonAnalyzer::Impl::ExprToPreviousKey(const PrimExpr& expr) const {
385 auto it = expr_to_key.find(expr);
386 if (it != expr_to_key.end()) {
387 return it->second;
388 } else {
389 return std::nullopt;
390 }

393TransitiveComparisonAnalyzer::Impl::Key TransitiveComparisonAnalyzer::Impl::ExprToKey(
394 const PrimExpr& expr) {
395 if (auto prev = ExprToPreviousKey(expr)) {
396 return prev.value();
397 } else {
398 Key new_key = Key(expr_to_key.size());
399 expr_to_key[expr] = new_key;
400 return new_key;
401 }

404bool TransitiveComparisonAnalyzer::Impl::Comparison::IsNormalized() const {
405 // These < and > should be removed during normalization. See the
406 // `Comparison::Comparison` constructor for further details.
407 return result_ != CompareResult::kLT && result_ != CompareResult::kGT;
408}

410std::optional<TransitiveComparisonAnalyzer::Impl::Comparison>
411TransitiveComparisonAnalyzer::Impl::Comparison::WithLHS(Key new_lhs) const {
412 if (new_lhs == lhs_) {
413 return *this;
414 } else if (new_lhs == rhs_) {
415 return Comparison(rhs_, lhs_, -offset_, Reverse(result_));
416 } else {
417 return std::nullopt;
418 }
419}

426bool TransitiveComparisonAnalyzer::Impl::Comparison::Implies(
427 const TransitiveComparisonAnalyzer::Impl::Comparison& other) const {
428 ICHECK(lhs_ == other.lhs_);
429 ICHECK(rhs_ == other.rhs_);
430 ICHECK(IsNormalized());
431 ICHECK(other.IsNormalized());
432
433 if (result_ == other.result_ && offset_ == other.offset_) {
434 // if c1 == c2, x != y + c1 => x != y + c2
435 // if c1 == c2, x == y + c1 => x == y + c2
436 return true;
437 }
438
439 if (other.result_ == CompareResult::kLE && offset_ <= other.offset_) {
440 if (result_ == CompareResult::kEQ || result_ == CompareResult::kLE) {
441 // if c1 <= c2, x <= y + c1 => x <= y + c2
442 // if c1 <= c2, x == y + c1 => x <= y + c2
443 return true;
444 }
445 }
446
447 if (other.result_ == CompareResult::kGE && offset_ >= other.offset_) {
448 if (result_ == CompareResult::kEQ || result_ == CompareResult::kGE) {
449 // if c1 >= c2, x == y + c1 => x >= y + c2
450 // if c1 >= c2, x >= y + c1 => x >= y + c2
451 return true;
452 }
453 }
454
455 if (other.result_ == CompareResult::kNE) {
456 if (result_ == CompareResult::kEQ && offset_ != other.offset_) {
457 // if c1 != c2, x == y + c1 => x != y + c2
458 return true;
459 }
460
461 if (result_ == CompareResult::kLE && offset_ < other.offset_) {
462 // if c1 < c2, x <= y + c1 => x < y + c2 => x != y + c2
463 return true;
464 }
465
466 if (result_ == CompareResult::kGE && offset_ > other.offset_) {
467 // if c1 != c2, x >= y + c1 => x > y + c2 => x != y + c2
468 return true;
469 }
470 }
471
472 return false;
473}

493void TransitiveComparisonAnalyzer::Impl::AddKnown(const PrimExpr& expr,
494 std::vector<Comparison>* vec) {
495 for (const auto& subexpr : ExtractConstraints(expr)) {
496 if (tir::SideEffect(expr) <= tir::CallEffectKind::kPure) {
497 if (auto cmp = FromExpr(subexpr)) {
498 vec->push_back(cmp.value());
499 }
500 }
501 }
502}
503
504void TransitiveComparisonAnalyzer::Impl::Bind(const tir::Var& var, const Range& range,
505 bool allow_override) {
506 auto it = prev_bindings_.find(var);
507 if (it != prev_bindings_.end()) {
508 ExprDeepEqual expr_equal;
509 bool differs_from_previous = !expr_equal(range->min, (*it).second->min) ||
510 !expr_equal(range->extent, (*it).second->extent);
511 if (differs_from_previous) {
512 ICHECK(allow_override) << "Binding of variable " << var << " as " << range
513 << " conflicts with previous binding as " << (*it).second;
514 if (auto key = ExprToPreviousKey(var)) {
515 knowns_.erase(std::remove_if(knowns_.begin(), knowns_.end(),
516 [&](const auto& known) { return known.lhs_ == key.value(); }),
517 knowns_.end());
518 }
519 }
520 }
521
522 prev_bindings_.Set(var, range);
523
524 if (is_const_int(range->extent, 1)) {
525 AddKnown(var == range->min, &knowns_);
526 } else {
527 AddKnown(var >= range->min, &knowns_);
528 AddKnown(var < range->min + range->extent, &knowns_);
529 }
530}

532void TransitiveComparisonAnalyzer::Impl::Bind(const tir::Var& var, const PrimExpr& expr,
533 bool allow_override) {
534 Bind(var, Range::FromMinExtent(expr, 1), allow_override);
535}

537std::function<void()> TransitiveComparisonAnalyzer::Impl::EnterConstraint(const PrimExpr& expr) {
538 size_t old_literal_size = scoped_knowns_.size();
539 AddKnown(expr, &scoped_knowns_);
540 size_t new_literal_size = scoped_knowns_.size();
541
542 auto frecover = [old_literal_size, new_literal_size, this]() {
543 ICHECK_EQ(scoped_knowns_.size(), new_literal_size);
544 scoped_knowns_.erase(scoped_knowns_.begin() + old_literal_size, scoped_knowns_.end());
545 };
546 return frecover;
547}

549CompareResult TransitiveComparisonAnalyzer::Impl::TryCompare(const PrimExpr& lhs_expr,
550 const PrimExpr& rhs_expr) const {
551 // Currently only supports integer checks
552 if (!lhs_expr.dtype().is_int() || !rhs_expr.dtype().is_int()) {
553 return CompareResult::kUnknown;
554 }
555
556 // Bail out early if possible. This int check should have been
557 // constant-folded earlier, so this check shouldn't occur.
558 auto* x_int = lhs_expr.as<IntImmNode>();
559 auto* y_int = rhs_expr.as<IntImmNode>();
560 if (x_int && y_int) {
561 if (x_int->value < y_int->value) {
562 return CompareResult::kLT;
563 } else if (x_int->value > y_int->value) {
564 return CompareResult::kGT;
565 } else {
566 return CompareResult::kEQ;
567 }
568 }
569
570 auto [lhs, rhs, offset] = ExtractOffsets(lhs_expr, rhs_expr);
571 auto lhs_key = ExprToPreviousKey(lhs);
572 auto rhs_key = ExprToPreviousKey(rhs);
573
574 if (!lhs_key.has_value() || !rhs_key.has_value()) {
575 return CompareResult::kUnknown;
576 }
577
578 auto from_lhs = DFSFromLHS(lhs_key.value(), rhs_key.value(), offset, lhs, rhs);
579 auto from_rhs = Reverse(DFSFromLHS(rhs_key.value(), lhs_key.value(), -offset, rhs, lhs));
580 auto output = from_lhs & from_rhs;
581
582 return output;
583}
584
585CompareResult TransitiveComparisonAnalyzer::Impl::DFSFromLHS(Key lhs_key_input, Key rhs_key_input,
586 int64_t offset_input,
587 const PrimExpr& lhs_input,
588 const PrimExpr& rhs_input) const {
589 Key lhs_key = lhs_key_input;
590 Key rhs_key = rhs_key_input;
591 int64_t offset = offset_input;
592
593 // Everything in `to_visit` has lhs as its lhs.
594 std::unordered_set<Key> seen;
595 std::unordered_set<Key> to_visit;
596 std::unordered_map<Key, std::vector<Comparison>> compared_to_x;
597
598 // Utility function to add a new known statement
599 auto declare_known = [&](Comparison cmp) {
600 std::vector<Comparison>& knowns = compared_to_x[cmp.rhs_];
601
602 // The comparison adds no new information, no modification
603 // required.
604 for (auto& prev_known : knowns) {
605 if (prev_known.Implies(cmp)) {
606 return;
607 }
608 }
609
610 // New information may require visiting a new expression.
611 if (cmp.rhs_ != rhs_key && !seen.count(cmp.rhs_)) {
612 to_visit.insert(cmp.rhs_);
613 seen.insert(cmp.rhs_);
614 }
615
616 // This comparison is a stronger version of a previous constraint.
617 // Therefore, replace the old version entirely.
618 for (auto& prev_known : knowns) {
619 if (cmp.Implies(prev_known)) {
620 prev_known = cmp;
621 return;
622 }
623 }
624
625 // Neither a superset nor a subset of previously known
626 // constraints, must be tracked separately.
627 knowns.push_back(cmp);
628 };
629
630 // Initialize the search based on any known (in)equalities that use
631 // the LHS of the comparison.
632 for (const auto& known : knowns_) {
633 if (auto normalized = known.WithLHS(lhs_key)) {
634 declare_known(normalized.value());
635 }
636 }
637 for (const auto& known : scoped_knowns_) {
638 if (auto normalized = known.WithLHS(lhs_key)) {
639 declare_known(normalized.value());
640 }
641 }
642
643 // Walk through the space of all comparisons that can be made with
644 // LHS.
645 while (to_visit.size()) {
646 Key middle_key = *to_visit.begin();
647 to_visit.erase(to_visit.begin());
648
649 std::vector<Comparison>& prev_knowns_using_middle = compared_to_x.at(middle_key);
650 ICHECK(compared_to_x.count(middle_key));
651
652 std::vector<Comparison> new_knowns_using_lhs;
653
654 auto attempt_transitive = [&](Comparison cmp) {
655 ICHECK(cmp.IsNormalized());
656
657 Key right_key = cmp.rhs_;
658
659 if (right_key == lhs_key) {
660 return;
661 }
662
663 for (const auto& prev : prev_knowns_using_middle) {
664 CompareResult new_result = CompareResult::kUnknown;
665 int64_t new_offset = prev.offset_ + cmp.offset_;
666
667 if (prev.result_ == CompareResult::kEQ) {
668 // x == y + c1 && y OP z + c2, x OP z + (c1 + c2)
669 new_result = cmp.result_;
670 } else if (cmp.result_ == CompareResult::kEQ) {
671 // x OP y + c1 && y == z + c2, x OP z + (c1 + c2)
672 new_result = prev.result_;
673 } else if (prev.result_ == cmp.result_ &&
674 (prev.result_ == CompareResult::kLE || prev.result_ == CompareResult::kGE)) {
675 // x <= y + c1 && y <= z + c2, x <= z + (c1 + c2)
676 // x >= y + c1 && y >= z + c2, x >= z + (c1 + c2)
677 //
678 // This condition is much simpler to write than the
679 // equivalent handling of < or of >, which is why the
680 // inequalities are normalized to <= and to >=. See
681 // `TransitiveComparisonAnalyzer::Impl::Comparison::Comparison`
682 // for further details.
683 new_result = prev.result_;
684 }
685
686 if (new_result != CompareResult::kUnknown) {
687 Comparison new_known(lhs_key, right_key, new_offset, new_result);
688 new_knowns_using_lhs.push_back(new_known);
689 }
690 }
691 };
692
693 // Attempt to prove a new comparison using one of the original
694 // known comparisons. We want to find a known such that
695 // `(LHS OP1 middle) && (middle OP2 right)` can be simplified
696 // into `(LHS OP3 right)`.
697 //
698 // Note: The right side is this step is not necessarily the RHS of
699 // the comparison we're trying to prove, as we may need to find
700 // intermediate comparisons first. For example, if we know that
701 // `a<=b`, `b<=c`, and `c<=d`, and we wish to prove that `a<=d`,
702 // we must first combine `a<=b` and `b<=c` into `a<=c`. During
703 // this first step, `b` is the "middle" and `c` is the "right".
704 // The next step can then combind `a<=c` and `c<=d` into `a<=d`.
705 for (const auto& known : knowns_) {
706 if (auto cmp = known.WithLHS(middle_key)) {
707 attempt_transitive(cmp.value());
708 }
709 }
710
711 for (const auto& known : scoped_knowns_) {
712 if (auto cmp = known.WithLHS(middle_key)) {
713 attempt_transitive(cmp.value());
714 }
715 }
716
717 // Collect together all new knowns, marking new nodes for visiting
718 // as needed.
719 for (const auto& new_known : new_knowns_using_lhs) {
720 declare_known(new_known);
721 }
722 }
723
724 // It's possible that we don't have any transitive comparisons that
725 // can prove something about LHS and RHS.
726 auto it = compared_to_x.find(rhs_key);
727 if (it == compared_to_x.end()) {
728 return CompareResult::kUnknown;
729 }
730
731 const std::vector<Comparison>& known_between_lhs_and_rhs = it->second;
732
733 // Just because we found a comparison involving LHS and RHS doesn't
734 // mean that it's useful. e.g. Knowing that `x < y` doesn't let us
735 // prove whether `x + 5 < y`.
736 CompareResult result = CompareResult::kUnknown;
737 for (const auto& known : known_between_lhs_and_rhs) {
738 switch (known.result_) {
739 case CompareResult::kInconsistent:
740 result = CompareResult::kInconsistent;
741 break;
742
743 case CompareResult::kEQ:
744 if (offset == known.offset_) {
745 result = result & CompareResult::kEQ;
746 } else {
747 result = result & CompareResult::kNE;
748 }
749 break;
750
751 case CompareResult::kLE:
752 if (known.offset_ < offset) {
753 result = result & CompareResult::kLT;
754 } else if (known.offset_ <= offset) {
755 result = result & CompareResult::kLE;
756 }
757 break;
758
759 case CompareResult::kGE:
760 if (known.offset_ > offset) {
761 result = result & CompareResult::kGT;
762 } else if (known.offset_ >= offset) {
763 result = result & CompareResult::kGE;
764 }
765 break;
766
767 case CompareResult::kNE:
768 if (offset == known.offset_) {
769 result = result & CompareResult::kNE;
770 }
771 break;
772
773 case CompareResult::kUnknown:
774 break;
775
776 case CompareResult::kGT:
777 case CompareResult::kLT:
778 LOG(FATAL) << "Internal error, normalized comparisons should only include <= and >=";
779 return CompareResult::kInconsistent;
780
781 default:
782 LOG(FATAL) << "Invalid CompareResult: " << static_cast<int>(known.result_);
783 return CompareResult::kInconsistent;
784 }
785 }
786
787 return result;
788}
