40/*!
41 * \brief Try to run binary compute with constant folding.
42 *
43 * \param a The left operand.
44 * \param b The right operand.
45 * \tparam Op The operator type.
46 *
47 * \note a and b Must already matched data types with each other.
48 * \return NullOpt if constant fold fails, otherwise return folded result.
49 */
50template <typename Op>
51inline Optional<PrimExpr> TryConstFold(PrimExpr a, PrimExpr b);

53/*!
54 * \brief Try to run unary compute with constant folding.
55 *
56 * \param a The left operand.
57 * \tparam Op The operator type.

59 * \note a and b Must already matched data types with each other.
60 * \return NullOpt if constant fold fails, otherwise return folded result.

62template <typename Op>

65/*!
66 * \brief Check whether type is used to represent index.
67 *
68 * Index types are frequently used in shape computation
69 * and need to be aggressively constant-folded.
70 *
71 * \param type The type to represent index.
72 * \return the checked result.
73 */
74inline bool IsIndexType(const DataType& type) {
75 return type.is_int() && type.lanes() == 1 && (type.bits() == 32 || type.bits() == 64);
76}

78/*! \brief Helper to get const folding result repr in int64. */
79inline int64_t GetFoldResultInt64Repr(int64_t x, const DataType& dtype) {
80 if (dtype.bits() < 64) {
81 x &= (1LL << dtype.bits()) - 1;
82 }
83 if (dtype.is_int()) {
84 // get sign extended value of integer with specified bits
85 int64_t m = 1LL << (dtype.bits() - 1);
86 x = (x ^ m) - m;
87 }
88 return x;
89}
90
91/*! \brief Helper to get fp32 const folding result repr in double. */
92inline double GetFoldResultDoubleRepr(float x) {
93 double res = static_cast<double>(x);
94 if (std::isinf(res) || std::isnan(res)) {
95 return res;
96 }
97 // certain platform (eg, on gcc7-i386) do the folding arithmetic
98 // on float and write back to double is optimized to double
99 // precision arithmetic, this is legal and we check the output
100 // range thus to ensure consistency when the float result is inf.
101 if (res < std::numeric_limits<float>::lowest()) {
102 LOG(WARNING) << "underlying float value overflow";
103 return -std::numeric_limits<double>::infinity();
104 } else if (res > std::numeric_limits<float>::max()) {
105 LOG(WARNING) << "underlying float value overflow";
106 return std::numeric_limits<double>::infinity();
107 }
108 return res;
109}

111#define TVM_ARITH_CONST_PROPAGATION(BODY) \
112 using tir::FloatImmNode; \
113 const IntImmNode* pa = a.as<IntImmNode>(); \
114 const IntImmNode* pb = b.as<IntImmNode>(); \
115 const FloatImmNode* fa = a.as<FloatImmNode>(); \
116 const FloatImmNode* fb = b.as<FloatImmNode>(); \
117 BODY;
118
119#define TVM_INDEX_CONST_PROPAGATION(BODY) \
120 const IntImmNode* pa = a.as<IntImmNode>(); \
121 const IntImmNode* pb = b.as<IntImmNode>(); \
122 const DataType& ta = a.dtype(); \
123 const DataType& tb = b.dtype(); \
124 if (arith::IsIndexType(ta) && arith::IsIndexType(tb)) { \
125 BODY; \
126 }

128// specialization of constant folders.
129template <>
130inline Optional<PrimExpr> TryConstFold<tir::Add>(PrimExpr a, PrimExpr b) {
131 TVM_ARITH_CONST_PROPAGATION({
132 const DataType& rtype = a.dtype();
133 if (pa && pb) {
134 int64_t res = pa->value + pb->value;
135 return IntImm(rtype, GetFoldResultInt64Repr(res, rtype));
136 }
137 if (pa && pa->value == 0) return b;
138 if (pb && pb->value == 0) return a;
139 if (fa && fb) {
140 if (rtype.bits() == 32) {
141 return FloatImm(rtype, GetFoldResultDoubleRepr(static_cast<float>(fa->value) +
142 static_cast<float>(fb->value)));
143 } else if (rtype.bits() == 64) {
144 return FloatImm(rtype, fa->value + fb->value);
145 } else {
146 return NullOpt;
147 }
148 }
149 if (fa && fa->value == 0) return b;
150 if (fb && fb->value == 0) return a;
151 });
152 return NullOpt;
153}

155template <>
156inline Optional<PrimExpr> TryConstFold<tir::Sub>(PrimExpr a, PrimExpr b) {
157 TVM_ARITH_CONST_PROPAGATION({
158 ICHECK(!((pa && pa->dtype.is_uint() && pa->value == 0U) &&
159 (pb && pb->dtype.is_uint() && pb->value > 0U)))
160 << "Checked failed. Minuend 's value is 0U and it's dtype is uint "
161 << "while Subtrahend's dtype is uint; which will cause a negative uint";
162 const DataType& rtype = a.dtype();
163 if (pa && pb) {
164 int64_t res = pa->value - pb->value;
165 return IntImm(rtype, GetFoldResultInt64Repr(res, rtype));
166 }
167 if (pb && pb->value == 0) return a;
168 if (fa && fb) {
169 if (rtype.bits() == 32) {
170 return FloatImm(rtype, GetFoldResultDoubleRepr(static_cast<float>(fa->value) -
171 static_cast<float>(fb->value)));
172 } else if (rtype.bits() == 64) {
173 return FloatImm(rtype, fa->value - fb->value);
174 } else {
175 return NullOpt;
176 }
177 }
178 if (fb && fb->value == 0) return a;
179 });
180 return NullOpt;
181}

183template <>
184inline Optional<PrimExpr> TryConstFold<tir::Mul>(PrimExpr a, PrimExpr b) {
185 TVM_ARITH_CONST_PROPAGATION({
186 const DataType& rtype = a.dtype();
187 if (pa && pb) {
188 int64_t res = pa->value * pb->value;
189 return IntImm(rtype, GetFoldResultInt64Repr(res, rtype));
190 }
191 if (pa) {
192 if (pa->value == 1) return b;
193 if (pa->value == 0) return a;
194 }
195 if (pb) {
196 if (pb->value == 1) return a;
197 if (pb->value == 0) return b;
198 }
199 if (fa && fb) {
200 if (rtype.bits() == 32) {
201 return FloatImm(rtype, GetFoldResultDoubleRepr(static_cast<float>(fa->value) *
202 static_cast<float>(fb->value)));
203 } else if (rtype.bits() == 64) {
204 return FloatImm(rtype, fa->value * fb->value);
205 } else {
206 return NullOpt;
207 }
208 }
209 if (fa) {
210 if (fa->value == 1) return b;
211 if (fa->value == 0) return a;
212 }
213 if (fb) {
214 if (fb->value == 1) return a;
215 if (fb->value == 0) return b;
216 }
217 });
218 return NullOpt;
219}

221template <>
222inline Optional<PrimExpr> TryConstFold<tir::Div>(PrimExpr a, PrimExpr b) {
223 TVM_ARITH_CONST_PROPAGATION({
224 const DataType& rtype = a.dtype();
225 if (pa && pb) {
226 // due to division and mod can have different modes
227 // NOTE: this will assumes truc div.
228 ICHECK_NE(pb->value, 0) << "Divide by zero";
229 int64_t res = pa->value / pb->value;
230 return IntImm(rtype, GetFoldResultInt64Repr(res, rtype));
231 }
232 if (pa) {
233 if (pa->value == 0) return a;
234 }
235 if (pb) {
236 if (pb->value == 1) return a;
237 ICHECK_NE(pb->value, 0) << "Divide by zero";
238 }
239 if (fa && fb && fb->value != 0) {
240 if (rtype.bits() == 32) {
241 return FloatImm(rtype, GetFoldResultDoubleRepr(static_cast<float>(fa->value) /
242 static_cast<float>(fb->value)));
243 } else if (rtype.bits() == 64) {
244 return FloatImm(rtype, fa->value / fb->value);
245 } else {
246 return NullOpt;
247 }
248 }
249 if (fa && fa->value == 0) return a;
250 if (fb) {
251 if (fb->value == 1) return a;
252 ICHECK_NE(fb->value, 0) << "Divide by zero";
253 }
254 });
255 return NullOpt;
256}

258template <>
259inline Optional<PrimExpr> TryConstFold<tir::Mod>(PrimExpr a, PrimExpr b) {
260 TVM_INDEX_CONST_PROPAGATION({
261 const DataType& rtype = a.dtype();
262 if (pa && pb) {
263 ICHECK_NE(pb->value, 0) << "Divide by zero";
264 int64_t res = pa->value % pb->value;
265 return IntImm(rtype, GetFoldResultInt64Repr(res, rtype));
266 }
267 if (pa) {
268 if (pa->value == 0) return a;
269 }
270 if (pb) {
271 if (pb->value == 1) return tir::make_zero(rtype);
272 ICHECK_NE(pb->value, 0) << "Divide by zero";
273 }
274 });
275 return NullOpt;
276}

278template <>
279inline Optional<PrimExpr> TryConstFold<tir::FloorDiv>(PrimExpr a, PrimExpr b) {
280 TVM_ARITH_CONST_PROPAGATION({
281 const DataType& rtype = a.dtype();
282 if (pa && pb) {
283 ICHECK_NE(pb->value, 0) << "Divide by zero";
284 int64_t res = arith::floordiv(pa->value, pb->value);
285 return IntImm(rtype, GetFoldResultInt64Repr(res, rtype));
286 }
287 if (pa) {
288 if (pa->value == 0) return a;
289 }
290 if (pb) {
291 if (pb->value == 1) return a;
292 ICHECK_NE(pb->value, 0) << "Divide by zero";
293 }
294 if (fa && fb && fb->value != 0) {
295 if (rtype.bits() == 32) {
296 return FloatImm(rtype, GetFoldResultDoubleRepr(std::floor(static_cast<float>(fa->value) /
297 static_cast<float>(fb->value))));
298 } else if (rtype.bits() == 64) {
299 return FloatImm(rtype, std::floor(fa->value / fb->value));
300 } else {
301 return NullOpt;
302 }
303 }
304 if (fa && fa->value == 0) return a;
305 if (fb) {
306 if (fb->value == 1) return a;
307 ICHECK_NE(fb->value, 0) << "Divide by zero";
308 }
309 });
310 return NullOpt;
311}

409template <>
410inline Optional<PrimExpr> TryConstFold<tir::And>(PrimExpr a, PrimExpr b) {
411 const IntImmNode* pa = a.as<IntImmNode>();
412 const IntImmNode* pb = b.as<IntImmNode>();
413 if (pa && pa->value) return b;
414 if (pa && !pa->value) return a;
415 if (pb && pb->value) return a;
416 if (pb && !pb->value) return b;
417 return NullOpt;
418}

422 const IntImmNode* pa = a.as<IntImmNode>();
423 const IntImmNode* pb = b.as<IntImmNode>();
424 if (pa && pa->value) return a;
425 if (pa && !pa->value) return b;
426 if (pb && pb->value) return b;
427 if (pb && !pb->value) return a;
428 return NullOpt;
429}

431template <>
432inline Optional<PrimExpr> TryConstFold<tir::Not>(PrimExpr a) {
433 const IntImmNode* pa = a.as<IntImmNode>();
434 if (pa) {
435 return IntImm(DataType::UInt(1), !(pa->value));
436 }
437 return NullOpt;

440/*! \brief Helper namespace for symbolic value limits */
441struct SymbolicLimits {
442 /*! \brief positive infinity */
443 static PrimExpr pos_inf_;
444 /*! \brief negative infinity */
445 static PrimExpr neg_inf_;
446};
447
448/*!
449 * \brief Opaque expression representing positive infinity.
450 *
451 * It can can only be used as parameter of by min/max
452 * for integer analysis and cannot be used in normal expressions.
453 *
454 * \return positive infinity.
455 */
456inline PrimExpr pos_inf() { return SymbolicLimits::pos_inf_; }
457
458/*!
459 * \brief Check if value is positive infinity.
460 * \param value The value to be checked.
461 *
462 * \return The check result.
463 */
464inline bool is_pos_inf(const PrimExpr& value) { return value.same_as(SymbolicLimits::pos_inf_); }
465
466/*!
467 * \brief Opaque expression representing negative infinity.
468 *
469 * It can can only be used as parameter of by min/max
470 * for integer analysis and cannot be used in normal expressions.
471 *
472 * \return negative infinity.
473 */
474inline PrimExpr neg_inf() { return SymbolicLimits::neg_inf_; }
475
476/*!
477 * \brief Check if value is negative infinity.
478 * \param value The value to be checked.
479 *
480 * \return The check result.
481 */
482inline bool is_neg_inf(const PrimExpr& value) { return value.same_as(SymbolicLimits::neg_inf_); }
