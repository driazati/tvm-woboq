39// a visitor to find the path to the target variable
40// from a expression.
41class VariablePathFinder : public ExprVisitor {
42 public:
43 explicit VariablePathFinder(PrimExpr target) : target_(target) {}
44
45 void VisitExpr(const PrimExpr& node) final {
46 if (visited_.count(node.get()) != 0) return;
47 visited_.insert(node.get());
48
49 if (!found_) path_.push_back(node.get());
50 if (node.same_as(target_)) found_ = true;
51 ExprVisitor::VisitExpr(node);
52 if (!found_) path_.pop_back();
53 }
54
55 std::vector<const Object*> path_;
56
57 private:
58 bool found_{false};
59 PrimExpr target_;
60 std::unordered_set<const Object*> visited_;
61};
62
63// get the path to the variable,
64// return empty vector to represent failure
65std::vector<const Object*> GetPath(PrimExpr target, PrimExpr expr) {
66 VariablePathFinder v(target);
67 v(expr);
68 return v.path_;
69}

71enum CompareOp { kGreater, kLess, kEqual };
72
73// a visitor to deduce the bound of a variable from a expression
74class BoundDeducer : public ExprFunctor<void(const PrimExpr&)> {
75 public:
76 friend class BoundDeduceInputChecker;
77 friend class Converter;
78 BoundDeducer(PrimExpr target, PrimExpr expr,
79 const std::unordered_map<const VarNode*, IntSet>& hint_map,
80 const std::unordered_map<const VarNode*, IntSet>& relax_map)
81 : target_(target), expr_(expr), hint_map_(hint_map), relax_map_(relax_map) {}
82
83 void Deduce();
84
85 void VisitExpr(const PrimExpr& e) final {
86 if (!success_) return;
87 if (iter_ < path_.size() && e.get() == path_[iter_++]) {
88 ExprFunctor::VisitExpr(e);
89 } else {
90 success_ = false;
91 return;
92 }
93 }
94
95 void VisitExprDefault_(const Object* op) final { success_ = false; }
96
97 void VisitExpr_(const VarNode* op) final {}
98
99 void VisitExpr_(const AddNode* op) final {
100 bool left = op->a.get() == path_[iter_];
101 result_ -= left ? op->b : op->a;
102 this->VisitExpr(left ? op->a : op->b);
103 }
104
105 void VisitExpr_(const SubNode* op) final {
106 bool left = op->a.get() == path_[iter_];
107 if (left) {
108 result_ += op->b;
109 } else {
110 result_ -= op->a;
111 result_ = -result_;
112 comp_op = ReverseOp(comp_op);
113 }
114 this->VisitExpr(left ? op->a : op->b);
115 }
116
117 void VisitExpr_(const MulNode* op) final {
118 bool left = op->a.get() == path_[iter_];
119 PrimExpr operand = left ? op->b : op->a;
120 PrimExpr target_var = left ? op->a : op->b;
121
122 SignType sign_operand;
123 if (operand.dtype().is_uint()) {
124 sign_operand = kPositive;
125 } else {
126 sign_operand = expr_map_[operand].GetSignType();
127 }
128
129 if (sign_operand == SignType::kNegative) {
130 comp_op = ReverseOp(comp_op);
131 } else if (sign_operand == SignType::kUnknown) {
132 // unable to get the sign of operand
133 success_ = false;
134 return;
135 }
136
137 // always use relax bound
138 bool divided = analyzer_.CanProve(floormod(result_, operand) == 0);
139
140 result_ = floordiv(result_, operand); // rounding down here
141
142 if (!divided) {
143 if (comp_op == kGreater) {
144 // System will round down in all the cases, so add one for result_ for kGreater
145 // (x >= 3/2 --> x >= 2)
146 // (x >= -3/2 --> x >= -1)
147 // (x >= 3/-2 --> x >= -1)
148 // (x >= -3/-2 --> x >= 2)
149 result_ += 1;
150 } else if (comp_op == kEqual) {
151 // condition unsatisfiable as with floor div, it will change the expression
152 success_ = false;
153 return;
154 } else {
155 // System rounds down in all cases, do nothing for kLess.
156 // ( x <= 3/2 --> x <= 1)
157 // ( x <= -3/2 --> x <= -2)
158 // ( x <= 3/-2 --> x <= -2)
159 // ( x <= -3/-2 --> x <= 1)
160 }
161 }
162 this->VisitExpr(left ? op->a : op->b);
163 }
164
165 PrimExpr result_;
166 CompareOp comp_op{kGreater};
167 bool success_{true};
168
169 private:
170 void Init();
171 void Transform();
172 void Relax();
173 CompareOp ReverseOp(CompareOp comp_op);
174 PrimExpr target_;
175 PrimExpr expr_;
176 const std::unordered_map<const VarNode*, IntSet>& hint_map_;
177 const std::unordered_map<const VarNode*, IntSet>& relax_map_;
178 ExprIntSetMap expr_map_;
179 std::vector<const Object*> path_;
180 size_t iter_{0};
181 // internal analzyer
182 Analyzer analyzer_;
183};
184
185class BoundDeduceInputChecker : public ExprVisitor {
186 public:
187 bool Check(BoundDeducer* deducer) {
188 deducer_ = deducer;
189 this->VisitExpr(deducer_->expr_);
190 return target_count == 1;
191 }
192
193 void VisitExpr(const PrimExpr& e) final {
194 if (e.same_as(deducer_->target_)) ++target_count;
195 ExprVisitor::VisitExpr(e);
196 }
197
198 private:
199 BoundDeducer* deducer_;
200 size_t target_count{0};
201};
202
203void BoundDeducer::Init() {
204 BoundDeduceInputChecker checker;
205 if (!checker.Check(this)) success_ = false;
206 Transform();
207}
208
209CompareOp BoundDeducer::ReverseOp(CompareOp comp_op) {
210 switch (comp_op) {
211 case kEqual:
212 return kEqual; // IntSet can not represent range for `NE
213 case kGreater:
214 return kLess;
215 case kLess:
216 return kGreater;
217 default:
218 LOG(FATAL) << "Not a valid compare op";
219 return kGreater; // return some default value
220 }
221}
222
223void BoundDeducer::Transform() {
224 // We will ensure to set expr_ such that it contains target_
225 if (const LTNode* op = expr_.as<LTNode>()) {
226 if (GetPath(target_, op->a).empty()) {
227 // a < b -> b >= a + 1
228 comp_op = kGreater;
229 expr_ = op->b;
230 result_ = op->a + 1;
231 } else {
232 // a < b -> a <= b - 1
233 comp_op = kLess;
234 expr_ = op->a;
235 result_ = op->b - 1;
236 }
237 } else if (const LENode* op = expr_.as<LENode>()) {
238 if (GetPath(target_, op->a).empty()) {
239 // a <= b -> b >= a
240 comp_op = kGreater;
241 expr_ = op->b;
242 result_ = op->a;
243 } else {
244 comp_op = kLess;
245 expr_ = op->a;
246 result_ = op->b;
247 }
248 } else if (const GTNode* op = expr_.as<GTNode>()) {
249 if (GetPath(target_, op->a).empty()) {
250 // a > b -> b <= a - 1
251 comp_op = kLess;
252 expr_ = op->b;
253 result_ = op->a - 1;
254 } else {
255 // a > b -> a >= b + 1
256 comp_op = kGreater;
257 expr_ = op->a;
258 result_ = op->b + 1;
259 }
260 } else if (const GENode* op = expr_.as<GENode>()) {
261 if (GetPath(target_, op->a).empty()) {
262 // a >= b -> b <= a
263 comp_op = kLess;
264 expr_ = op->b;
265 result_ = op->a;
266 } else {
267 comp_op = kGreater;
268 expr_ = op->a;
269 result_ = op->b;
270 }
271 } else if (const EQNode* op = expr_.as<EQNode>()) {
272 comp_op = kEqual;
273 if (GetPath(target_, op->a).empty()) {
274 // if the b == a -> a == b
275 expr_ = op->b;
276 result_ = op->a;
277 } else {
278 expr_ = op->a;
279 result_ = op->b;
280 }
281 } else {
282 success_ = false;
283 }
284}

286void BoundDeducer::Deduce() {
287 Init();
288 if (!success_) return;
289
290 Relax();
291 if (!success_) return;
292 // get the path
293 path_ = GetPath(target_, expr_);
294 if (!path_.size()) {
295 success_ = false;
296 return;
297 }
298 expr_map_ = EvalSetForEachSubExpr(expr_, hint_map_);
299
300 this->VisitExpr(expr_);
301}
302
303void BoundDeducer::Relax() {
304 IntSet a = EvalSet(expr_, relax_map_);
305 IntSet b = EvalSet(result_, relax_map_);
306 if (a.IsEverything() || b.IsEverything()) {
307 success_ = false;
308 return;
309 }
310 // Both LHS and RHS of the EQ should behave as constants e.g. i == j,
311 // can not be resolved when either `i` or `j` or both are variables with
312 // some Range OR `i` and `j` both should be a single point in IntSet
313 if (comp_op == kEqual &&
314 (!analyzer_.CanProve(b.min() == b.max()) || !analyzer_.CanProve(a.min() == a.max()))) {
315 success_ = false;
316 return;
317 }
318 expr_ = (comp_op == kGreater) ? a.min() : a.max();
319 result_ = (comp_op == kGreater) ? b.max() : b.min();
320}
321
322IntSet DeduceBound(PrimExpr v, PrimExpr e,
323 const std::unordered_map<const VarNode*, IntSet>& hint_map,
324 const std::unordered_map<const VarNode*, IntSet>& relax_map) {
325 BoundDeducer d(v, e, hint_map, relax_map);
326 d.Deduce();
327 if (!d.success_) return IntSet::Nothing();
328 PrimExpr min = neg_inf(), max = pos_inf();
329 if (d.comp_op == kEqual) {
330 min = d.result_;
331 max = d.result_;
332 } else if (d.comp_op == kGreater) {
333 min = d.result_;
334 } else {
335 max = d.result_;
336 }
337 return IntSet::Interval(min, max);
338}
339
340// assuming e >= 0, deduce the bound of variable from it.
341// return empty set to represent deduce failure.
342IntSet DeduceBound(PrimExpr v, PrimExpr e, const Map<Var, IntSet>& hint_map,
343 const Map<Var, IntSet>& relax_map) {
344 std::unordered_map<const VarNode*, IntSet> hmap;
345 for (auto kv : hint_map) {
346 hmap[kv.first.get()] = kv.second;
347 }
348 std::unordered_map<const VarNode*, IntSet> rmap;
349 for (auto kv : relax_map) {
350 rmap[kv.first.get()] = kv.second;
351 }
352 return DeduceBound(v, e, hmap, rmap);
353}
354
355TVM_REGISTER_GLOBAL("arith.DeduceBound")
356 .set_body_typed([](PrimExpr v, PrimExpr cond, const Map<Var, IntSet> hint_map,
357 const Map<Var, IntSet> relax_map) {
358 return DeduceBound(v, cond, hint_map, relax_map);
359 });
