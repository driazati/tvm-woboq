41/*! \brief The helper mutator that transforms ProducerLoad to BufferLoad */
42class ProducerToBufferTransformer : public StmtExprMutator {
43 public:
44 explicit ProducerToBufferTransformer(const std::unordered_map<te::Tensor, Buffer>& tensor2buffers)
45 : tensor2buffers_(tensor2buffers) {}
46
47 PrimExpr VisitExpr_(const ProducerLoadNode* op) final {
48 auto visited_op = Downcast<ProducerLoad>(StmtExprMutator::VisitExpr_(op));
49 te::Tensor tensor = Downcast<te::Tensor>(visited_op->producer);
50 auto it = tensor2buffers_.find(tensor);
51 ICHECK(it != tensor2buffers_.end()) << "IndexError: Cannot find the tensor " << tensor;
52 const Buffer& buffer = it->second;
53 return BufferLoad(buffer, visited_op->indices);
54 }
55
56 private:
57 /*! \brief The Map from Operations to buffers */
58 const std::unordered_map<te::Tensor, Buffer>& tensor2buffers_;
59};

61/*! \brief Helper data structure to store information. */
62struct CreateFuncInfo {
63 /*! \brief The Tensor arg_list. */
64 Array<te::Tensor> arg_list;
65 /*! \brief The map from each Tensor to its corresponding buffer. */
66 std::unordered_map<te::Tensor, Buffer> tensor2buffers;
67 /*! \brief The transformer from ProducerLoad to BufferLoad. */
68 ProducerToBufferTransformer transformer;
69 /*! \brief The buffers should be allocated at function root. */
70 Array<Buffer> root_alloc;
71 /*! \brief The NameSupply to make block name unique. */
72 NameSupply name_supply = NameSupply("");
73
74 String FreshName(String base_name) { return name_supply->FreshName(base_name); }
75
76 explicit CreateFuncInfo(Array<te::Tensor> arg_list)
77 : arg_list(std::move(arg_list)), transformer(tensor2buffers) {}
78
79 bool IsArg(const te::Tensor& tensor) const {
80 return std::any_of(arg_list.begin(), arg_list.end(),
81 [&tensor](const te::Tensor& arg) { return tensor == arg; });
82 }
83};

85class LayoutFreePlaceholdersNormalizer : public StmtMutator {
86 public:
87 PrimFunc Process(PrimFunc func) {
88 for (int i = 0, n = func->params.size(); i < n; ++i) {
89 if (const auto* v = func->params[i].as<VarNode>()) {
90 if (Optional<Buffer> buffer = func->buffer_map.Get(GetRef<Var>(v))) {
91 buffer2index_[buffer.value()] = i;
92 }
93 }
94 }
95 PrimFuncNode* f = func.CopyOnWrite();
96 f->body = VisitStmt(std::move(f->body));
97 if (this->layout_free_buffer_indices_.empty()) {
98 return func;
99 }
100 Array<Integer> indices;
101 indices.reserve(this->layout_free_buffer_indices_.size());
102 for (int i : this->layout_free_buffer_indices_) {
103 indices.push_back(Integer(i));
104 }
105 return WithAttr(std::move(func), tir::attr::layout_free_buffers, indices);
106 }
107
108 Stmt VisitStmt_(const BlockNode* _block) final {
109 Block block = Downcast<Block>(StmtMutator::VisitStmt_(_block));
110 if (Optional<ObjectRef> ann = block->annotations.Get(topi_attr)) {
111 Array<Buffer> buffers = Downcast<Array<Buffer>>(ann);
112 for (Buffer buffer : buffers) {
113 auto it = buffer2index_.find(buffer);
114 if (it != buffer2index_.end()) {
115 layout_free_buffer_indices_.insert(it->second);
116 }
117 }
118 block.CopyOnWrite()->annotations.erase(topi_attr);
119 }
120 return std::move(block);
121 }
122
123 std::unordered_map<tir::Buffer, int, ObjectPtrHash, ObjectPtrEqual> buffer2index_;
124 std::set<int> layout_free_buffer_indices_;
125 String topi_attr = "layout_free_placeholders";
126};

128BlockRealize GenerateBlockFromTensors(const te::ComputeOp& compute_op,
129 const Array<te::Tensor>& tensors, Array<PrimExpr> bindings,
130 PrimExpr expr_body, CreateFuncInfo* info,
131 arith::Analyzer* analyzer) {
132 // Step 1. Push_back data_par axis and reduce_axis into block_vars.
133 Array<IterVar> iter_vars;
134 std::unordered_map<const VarNode*, PrimExpr> var_map;
135 iter_vars.reserve(compute_op->axis.size() + compute_op->reduce_axis.size());
136 auto f_push_block_vars = [&iter_vars, &var_map, &analyzer](const Array<IterVar>& iters) {
137 for (IterVar iter_var : iters) {
138 // Create new var
139 Var new_var(iter_var->var->name_hint, iter_var->var->dtype);
140 var_map[iter_var->var.get()] = new_var;
141
142 const PrimExpr& dom_min = analyzer->Simplify(iter_var->dom->min);
143 const PrimExpr& dom_extent = analyzer->Simplify(iter_var->dom->extent);
144 iter_vars.push_back(IterVar(Range::FromMinExtent(dom_min, dom_extent), new_var,
145 iter_var->iter_type, iter_var->thread_tag, iter_var->span));
146 }
147 };
148 f_push_block_vars(compute_op->axis);
149 f_push_block_vars(compute_op->reduce_axis);
150
151 // Step 2.
152 // - Declare buffers
153 // - Update `op2buffers`
154 // - Add the non-argument tensors to `alloc_buffer` of the root block
155 Array<Buffer> buffers;
156 for (const te::Tensor& tensor : tensors) {
157 Buffer buffer = decl_buffer(tensor->shape, tensor->dtype, tensor->GetNameHint(), "global");
158 info->tensor2buffers[tensor] = buffer;
159 buffers.push_back(buffer);
160
161 if (!info->IsArg(tensor)) {
162 info->root_alloc.push_back(info->tensor2buffers[tensor]);
163 }
164 }
165
166 // Step 3. Calculate indices for BufferStore
167 Array<PrimExpr> indices;
168 indices.reserve(compute_op->axis.size());
169 for (const IterVar& iter_var : compute_op->axis) {
170 auto it = var_map.find(iter_var->var.get());
171 ICHECK(it != var_map.end());
172 indices.push_back(it->second);
173 }
174
175 // Step 4. Create block body.
176 String block_name{nullptr};
177 Optional<Stmt> init = NullOpt;
178 Stmt body;
179 if (const auto* reduce = expr_body.as<ReduceNode>()) {
180 // Case 1. Reduce compute
181 block_name = info->FreshName(compute_op->name);
182 int n_buffers = buffers.size();
183
184 Array<PrimExpr> lhs;
185 Array<PrimExpr> rhs;
186 lhs.reserve(n_buffers);
187 rhs.reserve(n_buffers);
188
189 // Make the LHS operands and RHS operands:
190 // - A LHS operand is the buffer storing the reduction result, with corresponding indices.
191 // - A RHS operand is the value to be reduced.
192 for (int i = 0; i < n_buffers; ++i) {
193 const PrimExpr& left = BufferLoad(buffers[i], indices);
194 const PrimExpr& right =
195 analyzer->Simplify(Substitute(info->transformer(reduce->source[i]), var_map));
196 lhs.push_back(left);
197 rhs.push_back(right);
198 ICHECK_EQ(left->dtype, right->dtype);
199 }
200
201 Array<Var> temp_vars;
202 Array<Stmt> body_stmts;
203 Array<Stmt> init_stmts;
204 temp_vars.reserve(n_buffers);
205 body_stmts.reserve(n_buffers);
206 init_stmts.reserve(n_buffers);
207
208 // - When there is only one buffer, we directly create a BufferStore which stores "combiner(lhs,
209 // rhs)" into the target buffer position.
210 // - In case there are multiple buffers, to avoid incorrect results, we create some intermediate
211 // variables and use LetStmts to bind the variables with "combiner(lhs, rhs)". After that, we
212 // then store the value of the variables into the target buffer positions.
213 for (int i = 0; i < n_buffers; ++i) {
214 const Buffer& buffer = buffers[i];
215 init_stmts.push_back(BufferStore(buffer, reduce->combiner->identity_element[i], indices));
216 PrimExpr value{nullptr};
217 if (n_buffers > 1) {
218 temp_vars.push_back(Var("v_" + buffer->name, PrimType(lhs[i].dtype())));
219 value = temp_vars.back();
220 } else {
221 value = reduce->combiner.get()->operator()(lhs, rhs)[i];
222 }
223 body_stmts.push_back(BufferStore(buffer, value, indices));
224 }
225
226 init = SeqStmt::Flatten(init_stmts);
227 body = SeqStmt::Flatten(body_stmts);
228 if (n_buffers > 1) {
229 // When there are multiple buffers, we wrap the body with LetStmts.
230 for (int i = n_buffers - 1; i >= 0; --i) {
231 PrimExpr value = reduce->combiner.get()->operator()(lhs, rhs)[i];
232 body = LetStmt(temp_vars[i], std::move(value), std::move(body));
233 }
234 }
235 } else {
236 // Case 2. Data parallel compute
237 ICHECK_EQ(tensors.size(), 1);
238 block_name = info->FreshName(tensors[0]->GetNameHint());
239 const PrimExpr& compute_body = Substitute(info->transformer(expr_body), var_map);
240 body = BufferStore(info->tensor2buffers[tensors[0]], analyzer->Simplify(compute_body), indices);
241 }
242
243 // Step 5. Add script_parsing_detect_access attr for auto complete the whole IR.
244 Map<String, ObjectRef> annotations;
245 auto mutate_attr = [&info](const ObjectRef& value) -> ObjectRef {
246 if (const auto* tensor_value = value.as<te::TensorNode>()) {
247 return info->tensor2buffers.at(GetRef<te::Tensor>(tensor_value));
248 } else {
249 return value;
250 }
251 };
252
253 for (const auto& pair : compute_op->attrs) {
254 const String& key = pair.first;
255 const ObjectRef& value = pair.second;
256 // TensorIR will not allow Tensor data structure
257 if (value->IsInstance<ArrayNode>()) {
258 const auto array_value = Downcast<Array<ObjectRef>>(value);
259 annotations.Set(key, MutateArray(array_value, mutate_attr));
260 } else {
261 annotations.Set(key, mutate_attr(value));
262 }
263 }
264 // Set script_parsing_detect_access
265 annotations.Set(tir::attr::script_parsing_detect_access, IntImm(DataType::Int(32), 3));
266 if (iter_vars.empty()) {
267 IterVar iter(Range::FromMinExtent(0, 1), Var("vi", DataType::Int(32)), IterVarType::kDataPar);
268 PrimExpr binding(0);
269 iter_vars.push_back(iter);
270 bindings.push_back(binding);
271 }
272
273 // Step 6. Create Block and BlockRealize.
274 return BlockRealize(/*iter_values=*/std::move(bindings),
275 /*predicate=*/Bool(true),
276 /*block=*/
277 Block(/*iter_vars=*/std::move(iter_vars),
278 /*reads=*/{},
279 /*writes=*/{},
280 /*name_hint=*/block_name,
281 /*body=*/std::move(body),
282 /*init=*/std::move(init),
283 /*alloc_buffers=*/{},
284 /*match_buffers=*/{},
285 /*annotations=*/std::move(annotations)));
286}

288Stmt GenerateStmtFromCompute(const te::ComputeOp& compute_op, CreateFuncInfo* info,
289 arith::Analyzer* analyzer) {
290 // Step 1. Creating loop vars for block bindings.
291 Array<IterVar> axes = compute_op->axis;
292 axes.insert(axes.end(), compute_op->reduce_axis.begin(), compute_op->reduce_axis.end());
293 Array<PrimExpr> bindings;
294 for (size_t i = 0; i < axes.size(); ++i) {
295 const IterVar& axis = axes[i];
296 int bits = std::max(axis->dom->min.dtype().bits(), axis->dom->extent.dtype().bits());
297 bindings.push_back(Var("i" + std::to_string(i), runtime::DataType::Int(bits)));
298 }
299 // Step 2. Generate block bodies.
300 Array<Stmt> seq_stmt;
301 if (compute_op->body[0]->IsInstance<ReduceNode>()) {
302 auto f_reducer_equal = [](const ReduceNode* a, const ReduceNode* b) -> bool {
303 return a->combiner.same_as(b->combiner) && //
304 a->source.same_as(b->source) && //
305 a->axis.same_as(b->axis) && //
306 a->condition.same_as(b->condition) && //
307 ((a->init.empty() && b->init.empty()) || a->init.same_as(b->init));
308 };
309
310 PrimExpr expr_body = compute_op->body[0];
311 Array<te::Tensor> tensors = {compute_op.output(0)};
312 const tir::ReduceNode* reduce = expr_body.as<tir::ReduceNode>();
313 // specially handle reduction inline for multiplre reductions.
314 for (size_t k = 1; k < compute_op->body.size(); ++k) {
315 const tir::ReduceNode* reduce_ = compute_op->body[k].as<tir::ReduceNode>();
316 ICHECK(reduce_);
317 ICHECK(f_reducer_equal(reduce_, reduce))
318 << "The Reduce inputs of ComputeOp should have the same attribute except value_index";
319 tensors.push_back(compute_op.output(k));
320 }
321
322 seq_stmt.push_back(GenerateBlockFromTensors(compute_op, tensors, bindings, std::move(expr_body),
323 info, analyzer));
324 } else {
325 for (int i = 0; i < compute_op->num_outputs(); ++i) {
326 const te::Tensor& tensor = compute_op.output(i);
327 PrimExpr expr_body = compute_op->body[i];
328 seq_stmt.push_back(GenerateBlockFromTensors(compute_op, {tensor}, bindings,
329 std::move(expr_body), info, analyzer));
330 }
331 }
332
333 Stmt body = SeqStmt::Flatten(seq_stmt);
334
335 // Step 3. Generate loop nesting.
336 for (size_t i = axes.size(); i > 0; --i) {
337 const IterVar& axis = axes[i - 1];
338 PrimExpr dom_min = analyzer->Simplify(axis->dom->min);
339 PrimExpr dom_extent = analyzer->Simplify(axis->dom->extent);
340 const Var& loop_var = Downcast<Var>(bindings[i - 1]);
341 body = For(loop_var, dom_min, dom_extent, ForKind::kSerial, body);
342 }
343
344 return body;
345}

347Stmt GenerateStmtFromExternOp(const te::ExternOp& extern_op, CreateFuncInfo* info) {
348 // Step 1. Check all inputs are visited before and update var_map.
349 std::unordered_map<const VarNode*, PrimExpr> var_map;
350 ICHECK_EQ(extern_op->inputs.size(), extern_op->input_placeholders.size());
351 for (size_t i = 0; i < extern_op->inputs.size(); ++i) {
352 const Buffer& placeholder = extern_op->input_placeholders[i];
353 const te::Tensor& input_tensor = extern_op->inputs[i];
354 auto it = info->tensor2buffers.find(input_tensor);
355 ICHECK(it != info->tensor2buffers.end());
356 var_map[placeholder->data.get()] = it->second->data;
357 }
358
359 // Step 2. Update info with its output tensor and placeholder buffer.
360 ICHECK_EQ(extern_op->num_outputs(), extern_op->output_placeholders.size());
361 for (int i = 0; i < extern_op->num_outputs(); ++i) {
362 const Buffer& placeholder = extern_op->output_placeholders[i];
363 const te::Tensor& output_tensor = extern_op.output(i);
364 info->tensor2buffers[output_tensor] = placeholder;
365 if (!info->IsArg(output_tensor)) {
366 info->root_alloc.push_back(placeholder);
367 }
368 }
369
370 // Step 3. Collect Access Region
371 Array<BufferRegion> reads, writes;
372 for (const te::Tensor& tensor : extern_op->inputs) {
373 // We have ICHECK before so it is not needed here.
374 reads.push_back(BufferRegion::FullRegion(info->tensor2buffers[tensor]));
375 }
376 for (const Buffer& buffer : extern_op->output_placeholders) {
377 writes.push_back(BufferRegion::FullRegion(buffer));
378 }
379
380 Stmt body = Substitute(extern_op->body, var_map);
381
382 // Step 4. Generate opaque block as body.
383 return BlockRealize(/*iter_values=*/{},
384 /*predicate=*/Bool(true),
385 /*block=*/
386 Block(/*iter_vars=*/{},
387 /*reads=*/std::move(reads),
388 /*writes=*/std::move(writes),
389 /*name_hint=*/info->FreshName(extern_op->name),
390 /*body=*/std::move(body),
391 /*init=*/NullOpt,
392 /*alloc_buffers=*/{},
393 /*match_buffers=*/{},
394 /*annotations=*/extern_op->attrs));
395}

397Array<te::Operation> CollectOrderedOps(const Array<te::Tensor>& arg_list) {
398 Array<te::Operation> arg_ops;
399 for (const te::Tensor& arg : arg_list) {
400 arg_ops.push_back(arg->op);
401 }
402 te::ReadGraph g = te::CreateReadGraph(arg_ops);
403 Array<te::Operation> order = te::PostDFSOrder(arg_ops, g);
404
405 for (const te::Operation& op : order) {
406 if (!(op->IsInstance<te::PlaceholderOpNode>() || op->IsInstance<te::ComputeOpNode>() ||
407 op->IsInstance<te::ExternOpNode>()))
408 LOG(FATAL) << "TypeError: Unsupported Operation: " << op->GetTypeKey() << ". "
409 << "Only te.placeholder and te.compute are allowed for now.";
410 }
411 return order;
412}

414void InitializeBufferBinds(const Array<te::Operation>& ordered_ops, CreateFuncInfo* info) {
415 // Process any TE operations which contain user defined buffers
416 for (const auto& op : ordered_ops) {
417 // Initialize the tensor2buffer binds map with buffers defined by the te.extern
418 if (const auto* extern_op = op.as<te::ExternOpNode>()) {
419 ICHECK_EQ(extern_op->inputs.size(), extern_op->input_placeholders.size());
420 for (size_t i = 0; i < extern_op->inputs.size(); ++i) {
421 const te::Tensor& input = extern_op->inputs[i];
422 const Buffer& buffer = extern_op->input_placeholders[i];
423 info->tensor2buffers[input] = buffer;
424 }
425 }
426 }
427}

429void RewriteStageToBlock(const te::Operation& op, CreateFuncInfo* info, Array<Stmt>* root_stmts,
430 arith::Analyzer* analyzer) {
431 if (const auto* placeholder = op.as<te::PlaceholderOpNode>()) {
432 // Case 1. PlaceholderOp (te.placeholder)
433 ICHECK_EQ(op->num_outputs(), 1);
434 const te::Tensor& tensor = op.output(0);
435 // Check op is in op list
436 ICHECK(info->IsArg(tensor));
437 // Declare a buffer for any argument tensors without a pre-existing
438 // buffer declaration recorded in the tensor2buffer binds map
439 if (info->tensor2buffers.count(tensor) == 0) {
440 const Buffer& buffer =
441 decl_buffer(placeholder->shape, placeholder->dtype, placeholder->name, "global");
442 info->tensor2buffers[tensor] = buffer;
443 }
444 } else if (const auto* compute_op = op.as<te::ComputeOpNode>()) {
445 // Case 2. ComputeOp (te.compute)
446 root_stmts->push_back(
447 GenerateStmtFromCompute(GetRef<te::ComputeOp>(compute_op), info, analyzer));
448 } else if (const auto extern_op = op.as<te::ExternOpNode>()) {
449 // Case 3. ExternOp (te.extern)
450 root_stmts->push_back(GenerateStmtFromExternOp(GetRef<te::ExternOp>(extern_op), info));
451 } else {
452 ICHECK(false) << "TypeError: Unsupported Operation: " << op->GetTypeKey() << ". "
453 << "Only te.placeholder and te.compute are allowed for now.";
454 }
455}

457PrimFunc GenerateAndCompletePrimFunc(const Array<te::Tensor>& arg_list,
458 const Array<Stmt>& root_stmts, CreateFuncInfo* info) {
459 Array<Var> parameters;
460 Map<Var, Buffer> buffer_map;
461 for (const te::Tensor& tensor : arg_list) {
462 Var arg("var_" + tensor->GetNameHint(), PrimType(DataType::Handle()));
463 parameters.push_back(arg);
464 auto it = info->tensor2buffers.find(tensor);
465 ICHECK(it != info->tensor2buffers.end());
466 buffer_map.Set(arg, it->second);
467 }
468 PrimFunc func = WithAttrs(PrimFunc(/*params=*/std::move(parameters),
469 /*body=*/SeqStmt::Flatten(root_stmts),
470 /*ret_type=*/VoidType(),
471 /*buffer_map=*/std::move(buffer_map)),
472 {{"global_symbol", String("main")}, {"tir.noalias", Bool(true)}});
473 const auto* complete = runtime::Registry::Get("script.Complete");
474 ICHECK(complete);
475 func = (*complete)(std::move(func), info->root_alloc);
476 return LayoutFreePlaceholdersNormalizer().Process(std::move(func));
477}

479PrimFunc CreatePrimFunc(const Array<te::Tensor>& arg_list) {
480 // Infomations used in CreatePrimFunc and its sub-functions.
481 CreateFuncInfo info(arg_list);
482 // Root body stmts.
483 Array<Stmt> root_stmts;
484 // Analyzer
485 arith::Analyzer analyzer;
486
487 // Step 1. Create ordered array of operations and validate they are supported.
488 Array<te::Operation> order = CollectOrderedOps(arg_list);
489
490 // Step 2. Initialize buffer binds map
491 InitializeBufferBinds(order, &info);
492
493 // Step 3. Rewrite compute stages into blocks.
494 for (const te::Operation& op : order) {
495 RewriteStageToBlock(op, &info, &root_stmts, &analyzer);
496 }
497 // Step 4. Create func and complete prim func.
498 return GenerateAndCompletePrimFunc(arg_list, root_stmts, &info);
499}

501PrimFunc CreatePrimFuncWithConstants(const Array<te::Tensor>& arg_list,
502 const Array<runtime::NDArray>& constants) {
503 PrimFunc func = CreatePrimFunc(arg_list);
504 return tir::BindParams(func, constants);
