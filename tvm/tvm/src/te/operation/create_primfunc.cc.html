43/*! \brief The helper mutator that transforms ProducerLoad to BufferLoad */
44class ProducerToBufferTransformer : public StmtExprMutator {
45 public:
46 explicit ProducerToBufferTransformer(const std::unordered_map<te::Tensor, Buffer>& tensor2buffers)
47 : tensor2buffers_(tensor2buffers) {}
48
49 PrimExpr VisitExpr_(const ProducerLoadNode* op) final {
50 auto visited_op = Downcast<ProducerLoad>(StmtExprMutator::VisitExpr_(op));
51 te::Tensor tensor = Downcast<te::Tensor>(visited_op->producer);
52 auto it = tensor2buffers_.find(tensor);
53 ICHECK(it != tensor2buffers_.end()) << "IndexError: Cannot find the tensor " << tensor;
54 const Buffer& buffer = it->second;
55 return BufferLoad(buffer, visited_op->indices);
56 }
57
58 private:
59 /*! \brief The Map from Operations to buffers */
60 const std::unordered_map<te::Tensor, Buffer>& tensor2buffers_;
61};

63/*! \brief Helper data structure to store information. */
64struct CreateFuncInfo {
65 /*! \brief The Tensor arg_list. */
66 Array<te::Tensor> arg_list;
67 /*! \brief The map from each Tensor to its corresponding buffer. */
68 std::unordered_map<te::Tensor, Buffer> tensor2buffers;
69 /*! \brief The transformer from ProducerLoad to BufferLoad. */
70 ProducerToBufferTransformer transformer;
71 /*! \brief The buffers should be allocated at function root. */
72 Array<Buffer> root_alloc;
73 /*! \brief The NameSupply to make block name unique. */
74 NameSupply name_supply = NameSupply("");
75
76 String FreshName(String base_name) { return name_supply->FreshName(base_name); }
77
78 explicit CreateFuncInfo(Array<te::Tensor> arg_list)
79 : arg_list(std::move(arg_list)), transformer(tensor2buffers) {}
80
81 bool IsArg(const te::Tensor& tensor) const {
82 return std::any_of(arg_list.begin(), arg_list.end(),
83 [&tensor](const te::Tensor& arg) { return tensor == arg; });
84 }
85};

87class LayoutFreePlaceholdersNormalizer : public StmtMutator {
88 public:
89 PrimFunc Process(PrimFunc func) {
90 for (int i = 0, n = func->params.size(); i < n; ++i) {
91 if (const auto* v = func->params[i].as<VarNode>()) {
92 if (Optional<Buffer> buffer = func->buffer_map.Get(GetRef<Var>(v))) {
93 buffer2index_[buffer.value()] = i;
94 }
95 }
96 }
97 PrimFuncNode* f = func.CopyOnWrite();
98 f->body = VisitStmt(std::move(f->body));
99 if (this->layout_free_buffer_indices_.empty()) {
100 return func;
101 }
102 Array<Integer> indices;
103 indices.reserve(this->layout_free_buffer_indices_.size());
104 for (int i : this->layout_free_buffer_indices_) {
105 indices.push_back(Integer(i));
106 }
107 return WithAttr(std::move(func), tir::attr::layout_free_buffers, indices);
108 }
109
110 Stmt VisitStmt_(const BlockNode* _block) final {
111 Block block = Downcast<Block>(StmtMutator::VisitStmt_(_block));
112 BlockNode* n = block.CopyOnWrite();
113 if (Optional<ObjectRef> ann = n->annotations.Get(topi_attr)) {
114 Array<Buffer> new_buffers;
115 for (Buffer buffer : Downcast<Array<Buffer>>(ann)) {
116 auto it = buffer2index_.find(buffer);
117 if (it != buffer2index_.end()) {
118 layout_free_buffer_indices_.insert(it->second);
119 } else {
120 new_buffers.push_back(buffer);
121 }
122 }
123 if (new_buffers.empty()) {
124 n->annotations.erase(topi_attr);
125 } else {
126 n->annotations.Set(topi_attr, new_buffers);
127 }
128 }
129 for (const String& attr : this->blocklist) {
130 auto it = n->annotations.find(attr);
131 if (it != n->annotations.end()) {
132 n->annotations.erase(attr);
133 }
134 }
135 return std::move(block);
136 }
137
138 std::unordered_map<tir::Buffer, int, ObjectPtrHash, ObjectPtrEqual> buffer2index_;
139 std::set<int> layout_free_buffer_indices_;
140 String topi_attr = "layout_free_placeholders";
141 std::vector<String> blocklist = {"const_matrix", "auto_scheduler_simplify_const_tensor_indices",
142 "workload"};
143};

145BlockRealize GenerateBlockFromTensors(const te::ComputeOp& compute_op,
146 const Array<te::Tensor>& tensors, Array<PrimExpr> bindings,
147 PrimExpr expr_body, CreateFuncInfo* info,
148 arith::Analyzer* analyzer) {
149 // Step 1. Push_back data_par axis and reduce_axis into block_vars.
150 Array<IterVar> iter_vars;
151 std::unordered_map<const VarNode*, PrimExpr> var_map;
152 iter_vars.reserve(compute_op->axis.size() + compute_op->reduce_axis.size());
153 auto f_push_block_vars = [&iter_vars, &var_map, &analyzer](const Array<IterVar>& iters) {
154 for (IterVar iter_var : iters) {
155 // Create new var
156 Var new_var("v_" + iter_var->var->name_hint, iter_var->var->dtype);
157 var_map[iter_var->var.get()] = new_var;
158
159 PrimExpr dom_min = analyzer->Simplify(iter_var->dom->min);
160 PrimExpr dom_extent = analyzer->Simplify(iter_var->dom->extent);
161 iter_vars.push_back(IterVar(Range::FromMinExtent(dom_min, dom_extent), new_var,
162 iter_var->iter_type, iter_var->thread_tag, iter_var->span));
163 }
164 };
165 f_push_block_vars(compute_op->axis);
166 f_push_block_vars(compute_op->reduce_axis);
167
168 // Step 2.
169 // - Declare buffers
170 // - Update `op2buffers`
171 // - Add the non-argument tensors to `alloc_buffer` of the root block
172 Array<Buffer> buffers;
173 for (const te::Tensor& tensor : tensors) {
174 Buffer buffer = decl_buffer(tensor->shape, tensor->dtype, tensor->GetNameHint(), "global");
175 info->tensor2buffers[tensor] = buffer;
176 buffers.push_back(buffer);
177
178 if (!info->IsArg(tensor)) {
179 info->root_alloc.push_back(info->tensor2buffers[tensor]);
180 }
181 }
182
183 // Step 3. Calculate indices for BufferStore
184 Array<PrimExpr> indices;
185 indices.reserve(compute_op->axis.size());
186 for (const IterVar& iter_var : compute_op->axis) {
187 auto it = var_map.find(iter_var->var.get());
188 ICHECK(it != var_map.end());
189 indices.push_back(it->second);
190 }
191
192 // Step 4. Create block body.
193 String block_name{nullptr};
194 Optional<Stmt> init = NullOpt;
195 Stmt body;
196 if (const auto* reduce = expr_body.as<ReduceNode>()) {
197 // Case 1. Reduce compute
198 block_name = info->FreshName(compute_op->name);
199 int n_buffers = buffers.size();
200
201 Array<PrimExpr> lhs;
202 Array<PrimExpr> rhs;
203 lhs.reserve(n_buffers);
204 rhs.reserve(n_buffers);
205
206 // Make the LHS operands and RHS operands:
207 // - A LHS operand is the buffer storing the reduction result, with corresponding indices.
208 // - A RHS operand is the value to be reduced.
209 for (int i = 0; i < n_buffers; ++i) {
210 const PrimExpr& left = BufferLoad(buffers[i], indices);
211 const PrimExpr& right =
212 analyzer->Simplify(Substitute(info->transformer(reduce->source[i]), var_map));
213 lhs.push_back(left);
214 rhs.push_back(right);
215 ICHECK_EQ(left->dtype, right->dtype);
216 }
217
218 Array<Var> temp_vars;
219 Array<Stmt> body_stmts;
220 Array<Stmt> init_stmts;
221 temp_vars.reserve(n_buffers);
222 body_stmts.reserve(n_buffers);
223 init_stmts.reserve(n_buffers);
224
225 // - When there is only one buffer, we directly create a BufferStore which stores "combiner(lhs,
226 // rhs)" into the target buffer position.
227 // - In case there are multiple buffers, to avoid incorrect results, we create some intermediate
228 // variables and use LetStmts to bind the variables with "combiner(lhs, rhs)". After that, we
229 // then store the value of the variables into the target buffer positions.
230 for (int i = 0; i < n_buffers; ++i) {
231 const Buffer& buffer = buffers[i];
232 init_stmts.push_back(BufferStore(buffer, reduce->combiner->identity_element[i], indices));
233 PrimExpr value{nullptr};
234 if (n_buffers > 1) {
235 temp_vars.push_back(Var("v_" + buffer->name, PrimType(lhs[i].dtype())));
236 value = temp_vars.back();
237 } else {
238 value = reduce->combiner.get()->operator()(lhs, rhs)[i];
239 }
240 body_stmts.push_back(BufferStore(buffer, value, indices));
241 }
242
243 init = SeqStmt::Flatten(init_stmts);
244 body = SeqStmt::Flatten(body_stmts);
245 if (n_buffers > 1) {
246 // When there are multiple buffers, we wrap the body with LetStmts.
247 for (int i = n_buffers - 1; i >= 0; --i) {
248 PrimExpr value = reduce->combiner.get()->operator()(lhs, rhs)[i];
249 body = LetStmt(temp_vars[i], std::move(value), std::move(body));
250 }
251 }
252 } else {
253 // Case 2. Data parallel compute
254 ICHECK_EQ(tensors.size(), 1);
255 block_name = info->FreshName(tensors[0]->GetNameHint());
256 const PrimExpr& compute_body = Substitute(info->transformer(expr_body), var_map);
257 body = BufferStore(info->tensor2buffers[tensors[0]], analyzer->Simplify(compute_body), indices);
258 }
259
260 // Step 5. Add script_parsing_detect_access attr for auto complete the whole IR.
261 Map<String, ObjectRef> annotations;
262 auto mutate_attr = [&info](const ObjectRef& value) -> ObjectRef {
263 if (const auto* tensor_value = value.as<te::TensorNode>()) {
264 return info->tensor2buffers.at(GetRef<te::Tensor>(tensor_value));
265 } else {
266 return value;
267 }
268 };
269
270 for (const auto& pair : compute_op->attrs) {
271 const String& key = pair.first;
272 const ObjectRef& value = pair.second;
273 // TensorIR will not allow Tensor data structure
274 if (value->IsInstance<ArrayNode>()) {
275 const auto array_value = Downcast<Array<ObjectRef>>(value);
276 annotations.Set(key, array_value.Map(mutate_attr));
277 } else {
278 annotations.Set(key, mutate_attr(value));
279 }
280 }
281 // Set script_parsing_detect_access
282 annotations.Set(tir::attr::script_parsing_detect_access, IntImm(DataType::Int(32), 3));
283 if (iter_vars.empty()) {
284 IterVar iter(Range::FromMinExtent(0, 1), Var("vi", DataType::Int(32)), IterVarType::kDataPar);
285 PrimExpr binding(0);
286 iter_vars.push_back(iter);
287 bindings.push_back(binding);
288 }
289
290 // Step 6. Create Block and BlockRealize.
291 return BlockRealize(/*iter_values=*/std::move(bindings),
292 /*predicate=*/Bool(true),
293 /*block=*/
294 Block(/*iter_vars=*/std::move(iter_vars),
295 /*reads=*/{},
296 /*writes=*/{},
297 /*name_hint=*/block_name,
298 /*body=*/std::move(body),
299 /*init=*/std::move(init),
300 /*alloc_buffers=*/{},
301 /*match_buffers=*/{},
302 /*annotations=*/std::move(annotations)));
303}

305Stmt GenerateStmtFromCompute(const te::ComputeOp& compute_op, CreateFuncInfo* info,
306 arith::Analyzer* analyzer) {
307 // Step 1. Creating loop vars for block bindings.
308 Array<IterVar> axes = compute_op->axis;
309 axes.insert(axes.end(), compute_op->reduce_axis.begin(), compute_op->reduce_axis.end());
310
311 Array<PrimExpr> bindings = axes.Map([&](IterVar iter_var) -> PrimExpr {
312 int bits = std::max(iter_var->dom->min.dtype().bits(), iter_var->dom->extent.dtype().bits());
313 return Var(iter_var->var->name_hint, runtime::DataType::Int(bits));
314 });
315
316 // Step 2. Generate block bodies.
317 Array<Stmt> seq_stmt;
318 if (compute_op->body[0]->IsInstance<ReduceNode>()) {
319 auto f_reducer_equal = [](const ReduceNode* a, const ReduceNode* b) -> bool {
320 return a->combiner.same_as(b->combiner) && //
321 a->source.same_as(b->source) && //
322 a->axis.same_as(b->axis) && //
323 a->condition.same_as(b->condition) && //
324 ((a->init.empty() && b->init.empty()) || a->init.same_as(b->init));
325 };
326
327 PrimExpr expr_body = compute_op->body[0];
328 Array<te::Tensor> tensors = {compute_op.output(0)};
329 const tir::ReduceNode* reduce = expr_body.as<tir::ReduceNode>();
330 // specially handle reduction inline for multiplre reductions.
331 for (size_t k = 1; k < compute_op->body.size(); ++k) {
332 const tir::ReduceNode* reduce_ = compute_op->body[k].as<tir::ReduceNode>();
333 ICHECK(reduce_);
334 ICHECK(f_reducer_equal(reduce_, reduce))
335 << "The Reduce inputs of ComputeOp should have the same attribute except value_index";
336 tensors.push_back(compute_op.output(k));
337 }
338
339 seq_stmt.push_back(GenerateBlockFromTensors(compute_op, tensors, bindings, std::move(expr_body),
340 info, analyzer));
341 } else {
342 for (int i = 0; i < compute_op->num_outputs(); ++i) {
343 const te::Tensor& tensor = compute_op.output(i);
344 PrimExpr expr_body = compute_op->body[i];
345 seq_stmt.push_back(GenerateBlockFromTensors(compute_op, {tensor}, bindings,
346 std::move(expr_body), info, analyzer));
347 }
348 }
349
350 Stmt body = SeqStmt::Flatten(seq_stmt);
351
352 // Step 3. Generate loop nesting.
353 for (size_t i = axes.size(); i > 0; --i) {
354 const IterVar& axis = axes[i - 1];
355 PrimExpr dom_min = analyzer->Simplify(axis->dom->min);
356 PrimExpr dom_extent = analyzer->Simplify(axis->dom->extent);
357 const Var& loop_var = Downcast<Var>(bindings[i - 1]);
358 body = For(loop_var, dom_min, dom_extent, ForKind::kSerial, body);
359 }
360
361 return body;
362}

364Stmt GenerateStmtFromExternOp(const te::ExternOp& extern_op, CreateFuncInfo* info) {
365 // Step 1. Check all inputs are visited before and update var_map.
366 std::unordered_map<const VarNode*, PrimExpr> var_map;
367 ICHECK_EQ(extern_op->inputs.size(), extern_op->input_placeholders.size());
368 for (size_t i = 0; i < extern_op->inputs.size(); ++i) {
369 const Buffer& placeholder = extern_op->input_placeholders[i];
370 const te::Tensor& input_tensor = extern_op->inputs[i];
371 auto it = info->tensor2buffers.find(input_tensor);
372 ICHECK(it != info->tensor2buffers.end());
373 var_map[placeholder->data.get()] = it->second->data;
374 }
375
376 // Step 2. Update info with its output tensor and placeholder buffer.
377 ICHECK_EQ(extern_op->num_outputs(), extern_op->output_placeholders.size());
378 for (int i = 0; i < extern_op->num_outputs(); ++i) {
379 const Buffer& placeholder = extern_op->output_placeholders[i];
380 const te::Tensor& output_tensor = extern_op.output(i);
381 info->tensor2buffers[output_tensor] = placeholder;
382 if (!info->IsArg(output_tensor)) {
383 info->root_alloc.push_back(placeholder);
384 }
385 }
386
387 // Step 3. Collect Access Region
388 Array<BufferRegion> reads, writes;
389 for (const te::Tensor& tensor : extern_op->inputs) {
390 // We have ICHECK before so it is not needed here.
391 reads.push_back(BufferRegion::FullRegion(info->tensor2buffers[tensor]));
392 }
393 for (const Buffer& buffer : extern_op->output_placeholders) {
394 writes.push_back(BufferRegion::FullRegion(buffer));
395 }
396
397 Stmt body = Substitute(extern_op->body, var_map);
398
399 // Step 4. Generate opaque block as body.
400 return BlockRealize(/*iter_values=*/{},
401 /*predicate=*/Bool(true),
402 /*block=*/
403 Block(/*iter_vars=*/{},
404 /*reads=*/std::move(reads),
405 /*writes=*/std::move(writes),
406 /*name_hint=*/info->FreshName(extern_op->name),
407 /*body=*/std::move(body),
408 /*init=*/NullOpt,
409 /*alloc_buffers=*/{},
410 /*match_buffers=*/{},
411 /*annotations=*/extern_op->attrs));
412}

414Array<te::Operation> CollectOrderedOps(const Array<te::Tensor>& arg_list) {
415 Array<te::Operation> arg_ops;
416 for (const te::Tensor& arg : arg_list) {
417 arg_ops.push_back(arg->op);
418 }
419 te::ReadGraph g = te::CreateReadGraph(arg_ops);
420 Array<te::Operation> order = te::PostDFSOrder(arg_ops, g);
421
422 for (const te::Operation& op : order) {
423 if (!(op->IsInstance<te::PlaceholderOpNode>() || op->IsInstance<te::ComputeOpNode>() ||
424 op->IsInstance<te::ExternOpNode>()))
425 LOG(FATAL) << "TypeError: Unsupported Operation: " << op->GetTypeKey() << ". "
426 << "Only te.placeholder and te.compute are allowed for now.";
427 }
428 return order;
429}

431void InitializeBufferBinds(const Array<te::Operation>& ordered_ops, CreateFuncInfo* info) {
432 // Process any TE operations which contain user defined buffers
433 for (const auto& op : ordered_ops) {
434 // Initialize the tensor2buffer binds map with buffers defined by the te.extern
435 if (const auto* extern_op = op.as<te::ExternOpNode>()) {
436 ICHECK_EQ(extern_op->inputs.size(), extern_op->input_placeholders.size());
437 for (size_t i = 0; i < extern_op->inputs.size(); ++i) {
438 const te::Tensor& input = extern_op->inputs[i];
439 const Buffer& buffer = extern_op->input_placeholders[i];
440 info->tensor2buffers[input] = buffer;
441 }
442 }
443 }
444}

446void RewriteStageToBlock(const te::Operation& op, CreateFuncInfo* info, Array<Stmt>* root_stmts,
447 arith::Analyzer* analyzer) {
448 if (const auto* placeholder = op.as<te::PlaceholderOpNode>()) {
449 // Case 1. PlaceholderOp (te.placeholder)
450 ICHECK_EQ(op->num_outputs(), 1);
451 const te::Tensor& tensor = op.output(0);
452 // Check op is in op list
453 ICHECK(info->IsArg(tensor));
454 // Declare a buffer for any argument tensors without a pre-existing
455 // buffer declaration recorded in the tensor2buffer binds map
456 if (info->tensor2buffers.count(tensor) == 0) {
457 const Buffer& buffer =
458 decl_buffer(placeholder->shape, placeholder->dtype, placeholder->name, "global");
459 info->tensor2buffers[tensor] = buffer;
460 }
461 } else if (const auto* compute_op = op.as<te::ComputeOpNode>()) {
462 // Case 2. ComputeOp (te.compute)
463 root_stmts->push_back(
464 GenerateStmtFromCompute(GetRef<te::ComputeOp>(compute_op), info, analyzer));
465 } else if (const auto extern_op = op.as<te::ExternOpNode>()) {
466 // Case 3. ExternOp (te.extern)
467 root_stmts->push_back(GenerateStmtFromExternOp(GetRef<te::ExternOp>(extern_op), info));
468 } else {
469 ICHECK(false) << "TypeError: Unsupported Operation: " << op->GetTypeKey() << ". "
470 << "Only te.placeholder and te.compute are allowed for now.";
471 }
472}

474PrimFunc GenerateAndCompletePrimFunc(const Array<te::Tensor>& arg_list,
475 const Array<Stmt>& root_stmts, CreateFuncInfo* info) {
476 Array<Var> parameters;
477 Map<Var, Buffer> buffer_map;
478 for (const te::Tensor& tensor : arg_list) {
479 Var arg("var_" + tensor->GetNameHint(), PrimType(DataType::Handle()));
480 parameters.push_back(arg);
481 auto it = info->tensor2buffers.find(tensor);
482 ICHECK(it != info->tensor2buffers.end());
483 buffer_map.Set(arg, it->second);
484 }
485 PrimFunc func = WithAttrs(PrimFunc(/*params=*/std::move(parameters),
486 /*body=*/SeqStmt::Flatten(root_stmts),
487 /*ret_type=*/VoidType(),
488 /*buffer_map=*/std::move(buffer_map)),
489 {{"global_symbol", String("main")}, {"tir.noalias", Bool(true)}});
490 const auto* complete = runtime::Registry::Get("script.Complete");
491 ICHECK(complete);
492 func = (*complete)(std::move(func), info->root_alloc);
493 return func;
494}

496PrimFunc CreatePrimFuncWithConstants(const Array<te::Tensor>& arg_list,
497 const Array<runtime::NDArray>& constants,
498 std::optional<DataType> index_dtype_override) {
499 // Information used in CreatePrimFunc and its sub-functions.
500 CreateFuncInfo info(arg_list);
501 // Root body stmts.
502 Array<Stmt> root_stmts;
503 // Analyzer
504 arith::Analyzer analyzer;
505
506 // Step 1. Create ordered array of operations and validate they are supported.
507 Array<te::Operation> order = CollectOrderedOps(arg_list);
508
509 // Step 2. Initialize buffer binds map
510 InitializeBufferBinds(order, &info);
511
512 // Step 3. Rewrite compute stages into blocks.
513 for (const te::Operation& op : order) {
514 RewriteStageToBlock(op, &info, &root_stmts, &analyzer);
515 }
516
517 // Step 4. Create func and complete prim func.
518 auto func = GenerateAndCompletePrimFunc(arg_list, root_stmts, &info);
519 func = tir::BindParams(func, constants);
520 if (index_dtype_override.has_value()) {
521 func = IndexDataTypeNormalizer(index_dtype_override.value()).Rewrite(std::move(func));
522 }
523 auto result = LayoutFreePlaceholdersNormalizer().Process(std::move(func));
524 return result;
525}

527PrimFunc CreatePrimFunc(const Array<te::Tensor>& arg_list,
528 std::optional<DataType> index_dtype_override) {
529 return CreatePrimFuncWithConstants(arg_list, {}, index_dtype_override);
530}
