42/*! \brief The helper mutator that transforms ProducerLoad to BufferLoad */
43class ProducerToBufferTransformer : public StmtExprMutator {
44 public:
45 explicit ProducerToBufferTransformer(const std::unordered_map<te::Tensor, Buffer>& tensor2buffers)
46 : tensor2buffers_(tensor2buffers) {}
47
48 PrimExpr VisitExpr_(const ProducerLoadNode* op) final {
49 auto visited_op = Downcast<ProducerLoad>(StmtExprMutator::VisitExpr_(op));
50 te::Tensor tensor = Downcast<te::Tensor>(visited_op->producer);
51 auto it = tensor2buffers_.find(tensor);
52 ICHECK(it != tensor2buffers_.end()) << "IndexError: Cannot find the tensor " << tensor;
53 const Buffer& buffer = it->second;
54 return BufferLoad(buffer, visited_op->indices);
55 }
56
57 private:
58 /*! \brief The Map from Operations to buffers */
59 const std::unordered_map<te::Tensor, Buffer>& tensor2buffers_;
60};

62/*! \brief Helper data structure to store information. */
63struct CreateFuncInfo {
64 /*! \brief The Tensor arg_list. */
65 Array<te::Tensor> arg_list;
66 /*! \brief The map from each Tensor to its corresponding buffer. */
67 std::unordered_map<te::Tensor, Buffer> tensor2buffers;
68 /*! \brief The transformer from ProducerLoad to BufferLoad. */
69 ProducerToBufferTransformer transformer;
70 /*! \brief The buffers should be allocated at function root. */
71 Array<Buffer> root_alloc;
72 /*! \brief The NameSupply to make block name unique. */
73 NameSupply name_supply = NameSupply("");
74
75 String FreshName(String base_name) { return name_supply->FreshName(base_name); }
76
77 explicit CreateFuncInfo(Array<te::Tensor> arg_list)
78 : arg_list(std::move(arg_list)), transformer(tensor2buffers) {}
79
80 bool IsArg(const te::Tensor& tensor) const {
81 return std::any_of(arg_list.begin(), arg_list.end(),
82 [&tensor](const te::Tensor& arg) { return tensor == arg; });
83 }
84};

86class LayoutFreePlaceholdersNormalizer : public StmtMutator {
87 public:
88 PrimFunc Process(PrimFunc func) {
89 for (int i = 0, n = func->params.size(); i < n; ++i) {
90 if (const auto* v = func->params[i].as<VarNode>()) {
91 if (Optional<Buffer> buffer = func->buffer_map.Get(GetRef<Var>(v))) {
92 buffer2index_[buffer.value()] = i;
93 }
94 }
95 }
96 PrimFuncNode* f = func.CopyOnWrite();
97 f->body = VisitStmt(std::move(f->body));
98 if (this->layout_free_buffer_indices_.empty()) {
99 return func;
100 }
101 Array<Integer> indices;
102 indices.reserve(this->layout_free_buffer_indices_.size());
103 for (int i : this->layout_free_buffer_indices_) {
104 indices.push_back(Integer(i));
105 }
106 return WithAttr(std::move(func), tir::attr::layout_free_buffers, indices);
107 }
108
109 Stmt VisitStmt_(const BlockNode* _block) final {
110 Block block = Downcast<Block>(StmtMutator::VisitStmt_(_block));
111 BlockNode* n = block.CopyOnWrite();
112 if (Optional<ObjectRef> ann = n->annotations.Get(topi_attr)) {
113 Array<Buffer> new_buffers;
114 for (Buffer buffer : Downcast<Array<Buffer>>(ann)) {
115 auto it = buffer2index_.find(buffer);
116 if (it != buffer2index_.end()) {
117 layout_free_buffer_indices_.insert(it->second);
118 } else {
119 new_buffers.push_back(buffer);
120 }
121 }
122 if (new_buffers.empty()) {
123 n->annotations.erase(topi_attr);
124 } else {
125 n->annotations.Set(topi_attr, new_buffers);
126 }
127 }
128 for (const String& attr : this->blocklist) {
129 auto it = n->annotations.find(attr);
130 if (it != n->annotations.end()) {
131 n->annotations.erase(attr);
132 }
133 }
134 return std::move(block);
135 }
136
137 std::unordered_map<tir::Buffer, int, ObjectPtrHash, ObjectPtrEqual> buffer2index_;
138 std::set<int> layout_free_buffer_indices_;
139 String topi_attr = "layout_free_placeholders";
140 std::vector<String> blocklist = {"const_matrix", "auto_scheduler_simplify_const_tensor_indices",
141 "workload"};
142};

144BlockRealize GenerateBlockFromTensors(const te::ComputeOp& compute_op,
145 const Array<te::Tensor>& tensors, Array<PrimExpr> bindings,
146 PrimExpr expr_body, CreateFuncInfo* info,
147 arith::Analyzer* analyzer) {
148 // Step 1. Push_back data_par axis and reduce_axis into block_vars.
149 Array<IterVar> iter_vars;
150 std::unordered_map<const VarNode*, PrimExpr> var_map;
151 iter_vars.reserve(compute_op->axis.size() + compute_op->reduce_axis.size());
152 auto f_push_block_vars = [&iter_vars, &var_map, &analyzer](const Array<IterVar>& iters) {
153 for (IterVar iter_var : iters) {
154 // Create new var
155 Var new_var(iter_var->var->name_hint, iter_var->var->dtype);
156 var_map[iter_var->var.get()] = new_var;
157
158 const PrimExpr& dom_min = analyzer->Simplify(iter_var->dom->min);
159 const PrimExpr& dom_extent = analyzer->Simplify(iter_var->dom->extent);
160 iter_vars.push_back(IterVar(Range::FromMinExtent(dom_min, dom_extent), new_var,
161 iter_var->iter_type, iter_var->thread_tag, iter_var->span));
162 }
163 };
164 f_push_block_vars(compute_op->axis);
165 f_push_block_vars(compute_op->reduce_axis);
166
167 // Step 2.
168 // - Declare buffers
169 // - Update `op2buffers`
170 // - Add the non-argument tensors to `alloc_buffer` of the root block
171 Array<Buffer> buffers;
172 for (const te::Tensor& tensor : tensors) {
173 Buffer buffer = decl_buffer(tensor->shape, tensor->dtype, tensor->GetNameHint(), "global");
174 info->tensor2buffers[tensor] = buffer;
175 buffers.push_back(buffer);
176
177 if (!info->IsArg(tensor)) {
178 info->root_alloc.push_back(info->tensor2buffers[tensor]);
179 }
180 }
181
182 // Step 3. Calculate indices for BufferStore
183 Array<PrimExpr> indices;
184 indices.reserve(compute_op->axis.size());
185 for (const IterVar& iter_var : compute_op->axis) {
186 auto it = var_map.find(iter_var->var.get());
187 ICHECK(it != var_map.end());
188 indices.push_back(it->second);
189 }
190
191 // Step 4. Create block body.
192 String block_name{nullptr};
193 Optional<Stmt> init = NullOpt;
194 Stmt body;
195 if (const auto* reduce = expr_body.as<ReduceNode>()) {
196 // Case 1. Reduce compute
197 block_name = info->FreshName(compute_op->name);
198 int n_buffers = buffers.size();
199
200 Array<PrimExpr> lhs;
201 Array<PrimExpr> rhs;
202 lhs.reserve(n_buffers);
203 rhs.reserve(n_buffers);
204
205 // Make the LHS operands and RHS operands:
206 // - A LHS operand is the buffer storing the reduction result, with corresponding indices.
207 // - A RHS operand is the value to be reduced.
208 for (int i = 0; i < n_buffers; ++i) {
209 const PrimExpr& left = BufferLoad(buffers[i], indices);
210 const PrimExpr& right =
211 analyzer->Simplify(Substitute(info->transformer(reduce->source[i]), var_map));
212 lhs.push_back(left);
213 rhs.push_back(right);
214 ICHECK_EQ(left->dtype, right->dtype);
215 }
216
217 Array<Var> temp_vars;
218 Array<Stmt> body_stmts;
219 Array<Stmt> init_stmts;
220 temp_vars.reserve(n_buffers);
221 body_stmts.reserve(n_buffers);
222 init_stmts.reserve(n_buffers);
223
224 // - When there is only one buffer, we directly create a BufferStore which stores "combiner(lhs,
225 // rhs)" into the target buffer position.
226 // - In case there are multiple buffers, to avoid incorrect results, we create some intermediate
227 // variables and use LetStmts to bind the variables with "combiner(lhs, rhs)". After that, we
228 // then store the value of the variables into the target buffer positions.
229 for (int i = 0; i < n_buffers; ++i) {
230 const Buffer& buffer = buffers[i];
231 init_stmts.push_back(BufferStore(buffer, reduce->combiner->identity_element[i], indices));
232 PrimExpr value{nullptr};
233 if (n_buffers > 1) {
234 temp_vars.push_back(Var("v_" + buffer->name, PrimType(lhs[i].dtype())));
235 value = temp_vars.back();
236 } else {
237 value = reduce->combiner.get()->operator()(lhs, rhs)[i];
238 }
239 body_stmts.push_back(BufferStore(buffer, value, indices));
240 }
241
242 init = SeqStmt::Flatten(init_stmts);
243 body = SeqStmt::Flatten(body_stmts);
244 if (n_buffers > 1) {
245 // When there are multiple buffers, we wrap the body with LetStmts.
246 for (int i = n_buffers - 1; i >= 0; --i) {
247 PrimExpr value = reduce->combiner.get()->operator()(lhs, rhs)[i];
248 body = LetStmt(temp_vars[i], std::move(value), std::move(body));
249 }
250 }
251 } else {
252 // Case 2. Data parallel compute
253 ICHECK_EQ(tensors.size(), 1);
254 block_name = info->FreshName(tensors[0]->GetNameHint());
255 const PrimExpr& compute_body = Substitute(info->transformer(expr_body), var_map);
256 body = BufferStore(info->tensor2buffers[tensors[0]], analyzer->Simplify(compute_body), indices);
257 }
258
259 // Step 5. Add script_parsing_detect_access attr for auto complete the whole IR.
260 Map<String, ObjectRef> annotations;
261 auto mutate_attr = [&info](const ObjectRef& value) -> ObjectRef {
262 if (const auto* tensor_value = value.as<te::TensorNode>()) {
263 return info->tensor2buffers.at(GetRef<te::Tensor>(tensor_value));
264 } else {
265 return value;
266 }
267 };
268
269 for (const auto& pair : compute_op->attrs) {
270 const String& key = pair.first;
271 const ObjectRef& value = pair.second;
272 // TensorIR will not allow Tensor data structure
273 if (value->IsInstance<ArrayNode>()) {
274 const auto array_value = Downcast<Array<ObjectRef>>(value);
275 annotations.Set(key, array_value.Map(mutate_attr));
276 } else {
277 annotations.Set(key, mutate_attr(value));
278 }
279 }
280 // Set script_parsing_detect_access
281 annotations.Set(tir::attr::script_parsing_detect_access, IntImm(DataType::Int(32), 3));
282 if (iter_vars.empty()) {
283 IterVar iter(Range::FromMinExtent(0, 1), Var("vi", DataType::Int(32)), IterVarType::kDataPar);
284 PrimExpr binding(0);
285 iter_vars.push_back(iter);
286 bindings.push_back(binding);
287 }
288
289 // Step 6. Create Block and BlockRealize.
290 return BlockRealize(/*iter_values=*/std::move(bindings),
291 /*predicate=*/Bool(true),
292 /*block=*/
293 Block(/*iter_vars=*/std::move(iter_vars),
294 /*reads=*/{},
295 /*writes=*/{},
296 /*name_hint=*/block_name,
297 /*body=*/std::move(body),
298 /*init=*/std::move(init),
299 /*alloc_buffers=*/{},
300 /*match_buffers=*/{},
301 /*annotations=*/std::move(annotations)));
302}

304Stmt GenerateStmtFromCompute(const te::ComputeOp& compute_op, CreateFuncInfo* info,
305 arith::Analyzer* analyzer) {
306 // Step 1. Creating loop vars for block bindings.
307 Array<IterVar> axes = compute_op->axis;
308 axes.insert(axes.end(), compute_op->reduce_axis.begin(), compute_op->reduce_axis.end());
309 Array<PrimExpr> bindings;
310 for (size_t i = 0; i < axes.size(); ++i) {
311 const IterVar& axis = axes[i];
312 int bits = std::max(axis->dom->min.dtype().bits(), axis->dom->extent.dtype().bits());
313 bindings.push_back(Var("i" + std::to_string(i), runtime::DataType::Int(bits)));
314 }
315 // Step 2. Generate block bodies.
316 Array<Stmt> seq_stmt;
317 if (compute_op->body[0]->IsInstance<ReduceNode>()) {
318 auto f_reducer_equal = [](const ReduceNode* a, const ReduceNode* b) -> bool {
319 return a->combiner.same_as(b->combiner) && //
320 a->source.same_as(b->source) && //
321 a->axis.same_as(b->axis) && //
322 a->condition.same_as(b->condition) && //
323 ((a->init.empty() && b->init.empty()) || a->init.same_as(b->init));
324 };
325
326 PrimExpr expr_body = compute_op->body[0];
327 Array<te::Tensor> tensors = {compute_op.output(0)};
328 const tir::ReduceNode* reduce = expr_body.as<tir::ReduceNode>();
329 // specially handle reduction inline for multiplre reductions.
330 for (size_t k = 1; k < compute_op->body.size(); ++k) {
331 const tir::ReduceNode* reduce_ = compute_op->body[k].as<tir::ReduceNode>();
332 ICHECK(reduce_);
333 ICHECK(f_reducer_equal(reduce_, reduce))
334 << "The Reduce inputs of ComputeOp should have the same attribute except value_index";
335 tensors.push_back(compute_op.output(k));
336 }
337
338 seq_stmt.push_back(GenerateBlockFromTensors(compute_op, tensors, bindings, std::move(expr_body),
339 info, analyzer));
340 } else {
341 for (int i = 0; i < compute_op->num_outputs(); ++i) {
342 const te::Tensor& tensor = compute_op.output(i);
343 PrimExpr expr_body = compute_op->body[i];
344 seq_stmt.push_back(GenerateBlockFromTensors(compute_op, {tensor}, bindings,
345 std::move(expr_body), info, analyzer));
346 }
347 }
348
349 Stmt body = SeqStmt::Flatten(seq_stmt);
350
351 // Step 3. Generate loop nesting.
352 for (size_t i = axes.size(); i > 0; --i) {
353 const IterVar& axis = axes[i - 1];
354 PrimExpr dom_min = analyzer->Simplify(axis->dom->min);
355 PrimExpr dom_extent = analyzer->Simplify(axis->dom->extent);
356 const Var& loop_var = Downcast<Var>(bindings[i - 1]);
357 body = For(loop_var, dom_min, dom_extent, ForKind::kSerial, body);
358 }
359
360 return body;
361}

363Stmt GenerateStmtFromExternOp(const te::ExternOp& extern_op, CreateFuncInfo* info) {
364 // Step 1. Check all inputs are visited before and update var_map.
365 std::unordered_map<const VarNode*, PrimExpr> var_map;
366 ICHECK_EQ(extern_op->inputs.size(), extern_op->input_placeholders.size());
367 for (size_t i = 0; i < extern_op->inputs.size(); ++i) {
368 const Buffer& placeholder = extern_op->input_placeholders[i];
369 const te::Tensor& input_tensor = extern_op->inputs[i];
370 auto it = info->tensor2buffers.find(input_tensor);
371 ICHECK(it != info->tensor2buffers.end());
372 var_map[placeholder->data.get()] = it->second->data;
373 }
374
375 // Step 2. Update info with its output tensor and placeholder buffer.
376 ICHECK_EQ(extern_op->num_outputs(), extern_op->output_placeholders.size());
377 for (int i = 0; i < extern_op->num_outputs(); ++i) {
378 const Buffer& placeholder = extern_op->output_placeholders[i];
379 const te::Tensor& output_tensor = extern_op.output(i);
380 info->tensor2buffers[output_tensor] = placeholder;
381 if (!info->IsArg(output_tensor)) {
382 info->root_alloc.push_back(placeholder);
383 }
384 }
385
386 // Step 3. Collect Access Region
387 Array<BufferRegion> reads, writes;
388 for (const te::Tensor& tensor : extern_op->inputs) {
389 // We have ICHECK before so it is not needed here.
390 reads.push_back(BufferRegion::FullRegion(info->tensor2buffers[tensor]));
391 }
392 for (const Buffer& buffer : extern_op->output_placeholders) {
393 writes.push_back(BufferRegion::FullRegion(buffer));
394 }
395
396 Stmt body = Substitute(extern_op->body, var_map);
397
398 // Step 4. Generate opaque block as body.
399 return BlockRealize(/*iter_values=*/{},
400 /*predicate=*/Bool(true),
401 /*block=*/
402 Block(/*iter_vars=*/{},
403 /*reads=*/std::move(reads),
404 /*writes=*/std::move(writes),
405 /*name_hint=*/info->FreshName(extern_op->name),
406 /*body=*/std::move(body),
407 /*init=*/NullOpt,
408 /*alloc_buffers=*/{},
409 /*match_buffers=*/{},
410 /*annotations=*/extern_op->attrs));
411}

413Array<te::Operation> CollectOrderedOps(const Array<te::Tensor>& arg_list) {
414 Array<te::Operation> arg_ops;
415 for (const te::Tensor& arg : arg_list) {
416 arg_ops.push_back(arg->op);
417 }
418 te::ReadGraph g = te::CreateReadGraph(arg_ops);
419 Array<te::Operation> order = te::PostDFSOrder(arg_ops, g);
420
421 for (const te::Operation& op : order) {
422 if (!(op->IsInstance<te::PlaceholderOpNode>() || op->IsInstance<te::ComputeOpNode>() ||
423 op->IsInstance<te::ExternOpNode>()))
424 LOG(FATAL) << "TypeError: Unsupported Operation: " << op->GetTypeKey() << ". "
425 << "Only te.placeholder and te.compute are allowed for now.";
426 }
427 return order;
428}

430void InitializeBufferBinds(const Array<te::Operation>& ordered_ops, CreateFuncInfo* info) {
431 // Process any TE operations which contain user defined buffers
432 for (const auto& op : ordered_ops) {
433 // Initialize the tensor2buffer binds map with buffers defined by the te.extern
434 if (const auto* extern_op = op.as<te::ExternOpNode>()) {
435 ICHECK_EQ(extern_op->inputs.size(), extern_op->input_placeholders.size());
436 for (size_t i = 0; i < extern_op->inputs.size(); ++i) {
437 const te::Tensor& input = extern_op->inputs[i];
438 const Buffer& buffer = extern_op->input_placeholders[i];
439 info->tensor2buffers[input] = buffer;
440 }
441 }
442 }
443}

445void RewriteStageToBlock(const te::Operation& op, CreateFuncInfo* info, Array<Stmt>* root_stmts,
446 arith::Analyzer* analyzer) {
447 if (const auto* placeholder = op.as<te::PlaceholderOpNode>()) {
448 // Case 1. PlaceholderOp (te.placeholder)
449 ICHECK_EQ(op->num_outputs(), 1);
450 const te::Tensor& tensor = op.output(0);
451 // Check op is in op list
452 ICHECK(info->IsArg(tensor));
453 // Declare a buffer for any argument tensors without a pre-existing
454 // buffer declaration recorded in the tensor2buffer binds map
455 if (info->tensor2buffers.count(tensor) == 0) {
456 const Buffer& buffer =
457 decl_buffer(placeholder->shape, placeholder->dtype, placeholder->name, "global");
458 info->tensor2buffers[tensor] = buffer;
459 }
460 } else if (const auto* compute_op = op.as<te::ComputeOpNode>()) {
461 // Case 2. ComputeOp (te.compute)
462 root_stmts->push_back(
463 GenerateStmtFromCompute(GetRef<te::ComputeOp>(compute_op), info, analyzer));
464 } else if (const auto extern_op = op.as<te::ExternOpNode>()) {
465 // Case 3. ExternOp (te.extern)
466 root_stmts->push_back(GenerateStmtFromExternOp(GetRef<te::ExternOp>(extern_op), info));
467 } else {
468 ICHECK(false) << "TypeError: Unsupported Operation: " << op->GetTypeKey() << ". "
469 << "Only te.placeholder and te.compute are allowed for now.";
470 }
471}

473PrimFunc GenerateAndCompletePrimFunc(const Array<te::Tensor>& arg_list,
474 const Array<Stmt>& root_stmts, CreateFuncInfo* info) {
475 Array<Var> parameters;
476 Map<Var, Buffer> buffer_map;
477 for (const te::Tensor& tensor : arg_list) {
478 Var arg("var_" + tensor->GetNameHint(), PrimType(DataType::Handle()));
479 parameters.push_back(arg);
480 auto it = info->tensor2buffers.find(tensor);
481 ICHECK(it != info->tensor2buffers.end());
482 buffer_map.Set(arg, it->second);
483 }
484 PrimFunc func = WithAttrs(PrimFunc(/*params=*/std::move(parameters),
485 /*body=*/SeqStmt::Flatten(root_stmts),
486 /*ret_type=*/VoidType(),
487 /*buffer_map=*/std::move(buffer_map)),
488 {{"global_symbol", String("main")}, {"tir.noalias", Bool(true)}});
489 const auto* complete = runtime::Registry::Get("script.Complete");
490 ICHECK(complete);
491 func = (*complete)(std::move(func), info->root_alloc);
492 return func;
493}

495PrimFunc CreatePrimFuncWithConstants(const Array<te::Tensor>& arg_list,
496 const Array<runtime::NDArray>& constants) {
497 // Infomations used in CreatePrimFunc and its sub-functions.
498 CreateFuncInfo info(arg_list);
499 // Root body stmts.
500 Array<Stmt> root_stmts;
501 // Analyzer
502 arith::Analyzer analyzer;
503
504 // Step 1. Create ordered array of operations and validate they are supported.
505 Array<te::Operation> order = CollectOrderedOps(arg_list);
506
507 // Step 2. Initialize buffer binds map
508 InitializeBufferBinds(order, &info);
509
510 // Step 3. Rewrite compute stages into blocks.
511 for (const te::Operation& op : order) {
512 RewriteStageToBlock(op, &info, &root_stmts, &analyzer);
513 }
514
515 // Step 4. Create func and complete prim func.
516 auto func = GenerateAndCompletePrimFunc(arg_list, root_stmts, &info);
517 func = tir::BindParams(func, constants);
518 return LayoutFreePlaceholdersNormalizer().Process(std::move(func));
519}
520
521PrimFunc CreatePrimFunc(const Array<te::Tensor>& arg_list) {
522 return CreatePrimFuncWithConstants(arg_list, {});
523}
