20#ifndef TVM_TE_OPERATION_CREATE_PRIMFUNC_H_
21#define TVM_TE_OPERATION_CREATE_PRIMFUNC_H_
22
23#include <tvm/runtime/container/array.h>
24#include <tvm/te/tensor.h>
25#include <tvm/tir/function.h>
26
27#include <optional>
28
29namespace tvm {
30namespace tir {
31
32/*! \brief Use Tensor Expression to create a schedulable TensorIR func. */
33PrimFunc CreatePrimFunc(const Array<te::Tensor>& arg_list,
34 std::optional<DataType> index_dtype_override = std::nullopt);
35
36/*! \brief The same as above but create a PrimFunc with AllocateConstNode. If the size of the
37 * constants array is N, the last N tensors in arg_list will be treated as constant tensors.
38 * Constant tensors will not be part of the parameters of the created PrimFunc, instead constants
39 * will be embedded in the body as AllocateConstNode.
40 */
41PrimFunc CreatePrimFuncWithConstants(const Array<te::Tensor>& arg_list,
42 const Array<runtime::NDArray>& constants,
43 std::optional<DataType> index_dtype_override = std::nullopt);
44
45} // namespace tir
46} // namespace tvm
47
48#endif // TVM_TE_OPERATION_CREATE_PRIMFUNC_H_
