20#ifndef TVM_TE_OPERATION_CREATE_PRIMFUNC_H_
21#define TVM_TE_OPERATION_CREATE_PRIMFUNC_H_
22
23#include <tvm/runtime/container/array.h>
24#include <tvm/te/tensor.h>
25#include <tvm/tir/function.h>
26
27namespace tvm {
28namespace tir {
29
30/*! \brief Use Tensor Expression to create a schedulable TensorIR func. */
31PrimFunc CreatePrimFunc(const Array<te::Tensor>& arg_list);
32
33/*! \brief The same as above but create a PrimFunc with AllocateConstNode. If the size of the
34 * constants array is N, the last N tensors in arg_list will be treated as constant tensors.
35 * Constant tensors will not be part of the parameters of the created PrimFunc, instead constants
36 * will be embedded in the body as AllocateConstNode.
37 */
38PrimFunc CreatePrimFuncWithConstants(const Array<te::Tensor>& arg_list,
39 const Array<runtime::NDArray>& constants);
40
41} // namespace tir
42} // namespace tvm

44#endif // TVM_TE_OPERATION_CREATE_PRIMFUNC_H_
