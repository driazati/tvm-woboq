39/*!
40 * \brief Clone iter vars and return both the new vars and the substitution from old to new.

42 * \param vars The original iter vars.

53/*!
54 * \brief Create a tensor from an expression. The expression may be a reduction, in which
55 * case its body will be correctly duplicated if it is a multi-valued reduction.
56 *
57 * \param expr The expr which will be the tensor's body.
58 * \param axis The input variables with ranges.
59 * \param name The tensor's name.
60 * \param tag The tensor's tag.
61 * \param attrs The tensor's attrs.
62 * \param clone_axis Whether to clone the given axis and perform substitution.
63 * \return A tensor.
64 */
65Tensor TensorFromExpr(const PrimExpr& expr, const Array<IterVar>& axis,
66 const std::string& name = "tensor", const std::string& tag = "",
67 const Map<String, ObjectRef>& attrs = {}, bool clone_axis = true);

76/*!
77 * \brief Inline tensors access recursively.
78 *
79 * This function will inline tensors recursively until it reaches a tensor which is impossible to
80 * inline (a reduction if \p inline_reductions is false, a non-compute tensor, a tensor which is
81 * not from \p inlineable). It won't descend into non-inlinable tensors' bodies.
82 *
83 * \param tensor The tensor whose body to transform.
84 * \param inlineable A list of tensors which are allowed to be inlined. If empty, try
85 * to inline all tensors.
86 * \param inline_reductions Whether to inline reductions (this may result in top-level reduction
87 * nodes).
88 *
89 * \return An inlined tensor
90 */
91TVM_DLL Tensor InlineTensorAccess(const Tensor& tensor,
92 const Array<Tensor>& inlineable = Array<Tensor>(),
93 bool inline_reductions = false);
94
95/*!
96 * \brief Inline tensors access at the tail.
97 * \param tensor The tensor whose body to transform.
98 * \return An inlined tensor
99 */
100TVM_DLL Tensor InlineTailTensorAccess(const Tensor& tensor);
101
102/*!
103 * \brief Simplify an iteration domain.
104 *
105 * An iteration domain is basically an array of variables and a condition. The function will do the
106 * following:
107 * - Replace div and mod operations with new variables (optional).
108 * - Extract (in)equalities from the condition.
109 * - Perform Fourier-Motzkin elimination.
110 * - Shear the domain of iteration (e.g. if `y <= x <= y + 2` then x will be replaced with `y + d`
111 * where `d` is a new variable such that `0 <= d <= 2`).
112 * - Remove redundant variables.
113 * - Infer new variable ranges (hopefully more precise).
114 *
115 * \param iter_domains The original domain.
116 * \param eliminate_div_mod Whether to eliminate div and mod by introducing new variables.
117 */
118TVM_DLL arith::IntConstraintsTransform SimplifyDomain(const arith::IntConstraints& iter_domains,
119 bool eliminate_div_mod = true);
120
121/*!
122 * \brief Perform lifting of conditions of being possible to be non-zero together with
123 * applying some transformations like simplifying the reduction domain. Works only with
124 * this particular tensor's body, i.e. doesn't perform inlining.
125 *
126 * \param tensor The original tensor;
127 * \param vranges Optional map from free variables to their value ranges.
128 * \return An optimized tensor.
129 */
130TVM_DLL Tensor RemoveJacobianAndLiftNonzeroCond(const Tensor& tensor,
131 const Map<Var, Range>& vranges = Map<Var, Range>());
