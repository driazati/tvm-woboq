20/*!
21 * \file ad_simplify.cc
22 * \brief Simplify tensor compute generated by tensor-level autodiff.
23 *
24 * The major simplification we do in this file is to eliminate
25 * the Jacobian tensor created by autodiff.
26 *
27 * Jacobian tensor is sparse because one output element usually relates
28 * to a small portion of the inputs. For example, element-wise function has a one-to-one mapping
29 * between input tensor and output tensor, thus the Jacobian is diagonal.
30 *
31 * Generally, we have Out_{\beta} = f( In_{A \alpha} ) in which A is a matrix,
32 * \alpha and \beta are vectors represent the indices of In and Out respectively.
33 * i.e., the non-zero Jacobian indices is a linear combination of the input indices.
34 * Thereby we solve linear equations of \beta = A \alpha,
35 * as well as linear inequalities of their domain ranges.
36 *
37 * Refer to Urban S, van der Smagt P. Automatic differentiation for tensor algebras[J].
38 * arXiv preprint arXiv:1711.01348, 2017. for more details.
39 *
40 * Implement-wise, we extract the equations in the compute definition via NonzeronessCondition,
41 * replace the compute expression with solved new axes, and create a selection node
42 * (non-zero-condition ? new_compute_expression : 0).
43 *
44 * Due to TVM's restriction, we also lift the reduction to the top of the compute stage.
45 *
46 */

69// Combine all expressions from the container using &&.
70template <class container>
71PrimExpr All(const container& c) {
72 PrimExpr res;
73 for (const auto& e : c) {
74 if (res.get()) {
75 res = res && e;
76 } else {
77 res = e;
78 }
79 }
80 if (res.get()) {
81 return res;
82 } else {
83 return const_true();
84 }
85}

95// Given a map from vars to ranges create an array of itervars
96Array<IterVar> IterVarsFromMap(const Array<Var>& vars, const Map<Var, Range>& vranges,
97 IterVarType iter_type = kDataPar, std::string thread_tag = "") {
98 Array<IterVar> res;
99 for (const Var& v : vars) {

102 res.push_back(IterVar(vranges[v], v, iter_type, thread_tag));
103 }
104 return res;

115template <typename ValueType>
116bool is_const_value(const PrimExpr& e, ValueType value) {
117 static_assert(std::is_integral<ValueType>::value,
118 "Comparison to non-integer values is forbidden.");
119 if (const tir::IntImmNode* i = e.as<tir::IntImmNode>()) {
120 return i->value == value;
121 } else if (const tir::FloatImmNode* i = e.as<tir::FloatImmNode>()) {
122 return i->value == value;
123 } else if (const tir::CastNode* c = e.as<tir::CastNode>()) {
124 return is_const_value(c->value, value);
125 } else if (const tir::BroadcastNode* b = e.as<tir::BroadcastNode>()) {
126 return is_const_value(b->value, value);
127 } else {
128 return false;
129 }
130}

132// Return true if this combiner is just a sum.
133bool IsSumCombiner(const CommReducer& combiner, const Map<Var, Range>& vranges) {
134 arith::Analyzer analyzer;
135 analyzer.Bind(vranges);
136 if (combiner->result.size() != 1) {
137 return false;
138 }
139
140 if (!is_const_value(
141 analyzer.Simplify(combiner->identity_element[0], kSimplifyRewriteCanonicalRewrite), 0)) {
142 return false;
143 }
144
145 PrimExpr combiner_result =
146 analyzer.Simplify(combiner->result[0], kSimplifyRewriteCanonicalRewrite);
147
148 return tir::ExprDeepEqual()(combiner_result, combiner->lhs[0] + combiner->rhs[0]) ||
149 tir::ExprDeepEqual()(combiner_result, combiner->rhs[0] + combiner->lhs[0]);
150}
151
152bool CanFactorZeroFromCombiner(const CommReducer& combiner, int value_index,
153 const Map<Var, Range>& vranges) {
154 arith::Analyzer analyzer;
155 analyzer.Bind(vranges);
156 if (!is_const_value(analyzer.Simplify(combiner->identity_element[value_index],
157 kSimplifyRewriteCanonicalRewrite),
158 0)) {
159 return false;
160 }
161
162 PrimExpr zero = make_zero(combiner->result[value_index].dtype());
163 PrimExpr in = Substitute(combiner->result[value_index], {{combiner->lhs[value_index], zero},
164 {combiner->rhs[value_index], zero}});
165 in = analyzer.Simplify(in, kSimplifyRewriteCanonicalRewrite);
166
167 return is_const_value(in, 0);
168}

170struct NonzeroConditionResult {
171 PrimExpr cond;
172 PrimExpr value;
173
174 PrimExpr to_expr() const { return Select(cond, value, make_zero(value.dtype())); }
175
176 friend std::ostream& operator<<(std::ostream& os, const NonzeroConditionResult& r) {
177 return os << r.to_expr();
178 }
179};
180
181// The implementation of NonzeroCondition
182// transform expression to cond ? value : 0
183class NonzeroConditionFunctor : public ExprFunctor<NonzeroConditionResult(const PrimExpr&)> {
184 public:
185 NonzeroConditionResult NonzeroCondition(const PrimExpr& e) {
186 if (e.dtype().is_bool()) {
187 // Boolean expressions are non-zero whenever they are true themselves
188 return {e, const_true()};
189 } else {
190 return VisitExpr(e);
191 }
192 }
193
194 // Most of the cases are implemented using helpers below
195 result_type VisitExpr_(const VarNode* op) final { return Default_(GetRef<PrimExpr>(op)); }
196 result_type VisitExpr_(const IntImmNode* op) final { return Const_(GetRef<IntImm>(op)); }
197 result_type VisitExpr_(const FloatImmNode* op) final { return Const_(GetRef<FloatImm>(op)); }
198 result_type VisitExpr_(const StringImmNode* op) final { return Default_(GetRef<PrimExpr>(op)); }
199 result_type VisitExpr_(const AddNode* op) final { return BinOpAddLike_(GetRef<Add>(op)); }
200 result_type VisitExpr_(const SubNode* op) final { return BinOpAddLike_(GetRef<Sub>(op)); }
201 result_type VisitExpr_(const MulNode* op) final { return BinOpMulLike_(GetRef<Mul>(op)); }
202 result_type VisitExpr_(const DivNode* op) final { return BinOpDivLike_(GetRef<Div>(op)); }
203 result_type VisitExpr_(const ModNode* op) final { return BinOpDivLike_(GetRef<Mod>(op)); }
204 result_type VisitExpr_(const FloorDivNode* op) final {
205 return BinOpDivLike_(GetRef<FloorDiv>(op));
206 }
207 result_type VisitExpr_(const FloorModNode* op) final {
208 return BinOpDivLike_(GetRef<FloorMod>(op));
209 }
210 result_type VisitExpr_(const MinNode* op) final { return BinOpAddLike_(GetRef<Min>(op)); }
211 result_type VisitExpr_(const MaxNode* op) final { return BinOpAddLike_(GetRef<Max>(op)); }
212
213 result_type VisitExpr_(const CastNode* op) final {
214 auto nz_a = NonzeroCondition(op->value);
215 return {nz_a.cond, Cast(op->dtype, nz_a.value)};
216 }
217
218 result_type VisitExpr_(const SelectNode* op) final {
219 PrimExpr cond = op->condition, true_val = op->true_value, false_val = op->false_value;
220 auto nz_a = NonzeroCondition(true_val);
221 auto nz_b = NonzeroCondition(false_val);
222
223 // If the false part is zero, we can get rid of the select
224 if (is_const_value(nz_b.value, 0)) {
225 PrimExpr new_cond = analyzer_.Simplify(nz_a.cond && cond, kSimplifyRewriteCanonicalRewrite);
226 return {new_cond, nz_a.value};
227 }
228
229 // If the true part is zero, we can also get rid of the select
230 if (is_const_value(nz_a.value, 0)) {
231 PrimExpr new_cond = analyzer_.Simplify(nz_b.cond && !cond, kSimplifyRewriteCanonicalRewrite);
232 return {new_cond, nz_b.value};
233 }
234
235 // Otherwise we retain the select and combine the conditions into this
236 PrimExpr new_cond = analyzer_.Simplify((cond && nz_a.cond) || (!cond && nz_b.cond),
237 kSimplifyRewriteCanonicalRewrite);
238 if (nz_a.value.same_as(true_val) && nz_b.value.same_as(false_val)) {
239 return {new_cond, GetRef<PrimExpr>(op)};
240 } else {
241 return {new_cond, Select(cond, nz_a.value, nz_b.value)};
242 }
243 }
244
245 result_type VisitExpr_(const CallNode* op) final {
246 if (op->op.same_as(op_if_then_else_)) {
247 PrimExpr cond = op->args[0], true_val = op->args[1], false_val = op->args[2];
248 auto nz_a = NonzeroCondition(true_val);
249 auto nz_b = NonzeroCondition(false_val);
250
251 // We don't have as much freedom here as in the select case
252 // since the `if` must be preserved in any case
253 PrimExpr new_cond = analyzer_.Simplify((cond && nz_a.cond) || (!cond && nz_b.cond),
254 kSimplifyRewriteCanonicalRewrite);
255 if (nz_a.value.same_as(true_val) && nz_b.value.same_as(false_val)) {
256 return {new_cond, GetRef<PrimExpr>(op)};
257 } else {
258 return {new_cond, if_then_else(cond, nz_a.value, nz_b.value)};
259 }
260 } else {
261 return Default_(GetRef<PrimExpr>(op));
262 }
263 }
264
265 result_type VisitExpr_(const ProducerLoadNode* op) final {
266 return Default_(GetRef<PrimExpr>(op));
267 }

269 NonzeroConditionResult Default_(const PrimExpr& e) {
270 // This is always correct, so it's the default
271 return {const_true(), e};
272 }

274 template <class T>
275 NonzeroConditionResult Const_(const T& op) {
276 if (op->value == 0) {
277 return {const_false(), op};
278 } else {
279 return {const_true(), op};
280 }
281 }

283 template <class T>
284 NonzeroConditionResult BinOpAddLike_(const T& op) {
285 auto nz_a = NonzeroCondition(op->a);
286 auto nz_b = NonzeroCondition(op->b);
287
288 // For addition and similar ops the result may be nonzero if either of the arguments is
289 // nonzero, so we combine the conditions with Or.
290 if (tir::ExprDeepEqual()(nz_a.cond, nz_b.cond)) {
291 // If the conditions are the same, we don't need Or
292 if (nz_a.value.same_as(op->a) && nz_b.value.same_as(op->b)) {
293 return {nz_a.cond, op};
294 } else {
295 return {nz_a.cond, T(nz_a.value, nz_b.value)};
296 }
297 } else {
298 // Otherwise use Or
299 PrimExpr new_cond =
300 analyzer_.Simplify(nz_a.cond || nz_b.cond, kSimplifyRewriteCanonicalRewrite);
301 // A little optimization: if the combined condition is the same as one of the inner
302 // conditions, we don't need to guard the inner value with a select, otherwise
303 // we create a select in the `to_expr` call.
304 PrimExpr new_a = tir::ExprDeepEqual()(nz_a.cond, new_cond) ? nz_a.value : nz_a.to_expr();
305 PrimExpr new_b = tir::ExprDeepEqual()(nz_b.cond, new_cond) ? nz_b.value : nz_b.to_expr();
306 PrimExpr new_expr = T(new_a, new_b);
307 return {new_cond, new_expr};
308 }
309 }
310
311 template <class T>
312 NonzeroConditionResult BinOpMulLike_(const T& op) {
313 auto nz_a = NonzeroCondition(op->a);
314 auto nz_b = NonzeroCondition(op->b);
315
316 // For multiplication and similar ops the result may be nonzero if
317 // both the arguments are nonzero, so we combine with And.
318 PrimExpr new_cond =
319 analyzer_.Simplify(nz_a.cond && nz_b.cond, kSimplifyRewriteCanonicalRewrite);
320
321 if (nz_a.value.same_as(op->a) && nz_b.value.same_as(op->b)) {
322 return {new_cond, op};
323 } else {
324 return {new_cond, T(nz_a.value, nz_b.value)};
325 }
326 }
327
328 template <class T>
329 NonzeroConditionResult BinOpDivLike_(const T& op) {
330 auto nz_a = NonzeroCondition(op->a);
331
332 // For Div we simply use the condition of the numerator.
333
334 if (nz_a.value.same_as(op->a)) {
335 return {nz_a.cond, op};
336 } else {
337 return {nz_a.cond, T(nz_a.value, op->b)};
338 }
339 }
340
341 private:
342 arith::Analyzer analyzer_;
343 const Op& op_if_then_else_ = Op::Get("tir.if_then_else");
344};

350struct FactorOutAtomicFormulasResult {
351 std::vector<PrimExpr> atomic_formulas;
352 PrimExpr rest;
353
354 PrimExpr to_expr() const {
355 PrimExpr res = rest;
356 for (const PrimExpr& e : atomic_formulas) {
357 res = And(e, res);
358 }
359 return res;
360 }
361
362 Array<PrimExpr> to_array() const {
363 Array<PrimExpr> res = atomic_formulas;
364 res.push_back(rest);
365 return res;
366 }
367};

369// The implementation of FactorOutAtomicFormulas
370class FactorOutAtomicFormulasFunctor
371 : public ExprFunctor<FactorOutAtomicFormulasResult(const PrimExpr&)> {
372 public:
373 result_type Atomic_(const PrimExpr& e) {
374 // For atomic expressions the result is the expr itself with True as the residual
375 return {{e}, make_const(e.dtype(), 1)};
376 }
377
378 // This is basically the list of expression kinds that are considered atomic
379 result_type VisitExpr_(const VarNode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
380 result_type VisitExpr_(const CallNode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
381 result_type VisitExpr_(const IntImmNode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
382 result_type VisitExpr_(const EQNode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
383 result_type VisitExpr_(const NENode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
384 result_type VisitExpr_(const LENode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
385 result_type VisitExpr_(const LTNode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
386 result_type VisitExpr_(const GENode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
387 result_type VisitExpr_(const GTNode* op) final { return Atomic_(GetRef<PrimExpr>(op)); }
388
389 result_type VisitExpr_(const SelectNode* op) final {
390 // Select can be rewritten through other logical ops
391 PrimExpr expr = (op->condition && op->true_value) || (!op->condition && op->false_value);
392 return VisitExpr(expr);
393 }
394
395 result_type VisitExpr_(const NotNode* op) final {
396 // Not should be moved down
397 if (const OrNode* or_expr = op->a.as<OrNode>()) {
398 PrimExpr expr = !or_expr->a && !or_expr->b;
399 return VisitExpr(expr);
400 } else if (const AndNode* and_expr = op->a.as<AndNode>()) {
401 PrimExpr expr = !and_expr->a || !and_expr->b;
402 return VisitExpr(expr);
403 } else if (const SelectNode* sel_expr = op->a.as<SelectNode>()) {
404 PrimExpr expr = ((!sel_expr->condition || !sel_expr->true_value) &&
405 (sel_expr->condition || !sel_expr->false_value));
406 return VisitExpr(expr);
407 }
408 return Atomic_(GetRef<PrimExpr>(op));
409 }
410
411 result_type VisitExpr_(const AndNode* op) final {
412 auto res_a = VisitExpr(op->a);
413 auto res_b = VisitExpr(op->b);
414
415 // For the And case we return the union of the sets of atomic formulas
416 std::unordered_set<PrimExpr, StructuralHash, StructuralEqual> res_a_set;
417 res_a_set.reserve(res_a.atomic_formulas.size());
418 std::copy(res_a.atomic_formulas.begin(), res_a.atomic_formulas.end(),
419 std::inserter(res_a_set, res_a_set.end()));
420
421 std::vector<PrimExpr> res = res_a.atomic_formulas;
422 for (const auto& e : res_b.atomic_formulas) {
423 if (res_a_set.find(e) == res_a_set.end()) {
424 res.emplace_back(e);
425 }
426 }
427 // And the residuals are combined with &&
428 return {res, res_a.rest && res_b.rest};
429 }
430
431 result_type VisitExpr_(const MulNode* op) final {
432 // Since we work with bools, for multiplication we do the same thing as for And
433 PrimExpr e_and = op->a && op->b;
434 return VisitExpr(e_and);
435 }
436
437 result_type VisitExpr_(const OrNode* op) final {
438 auto res_a = VisitExpr(op->a);
439 auto res_b = VisitExpr(op->b);
440
441 std::unordered_set<PrimExpr, StructuralHash, StructuralEqual> res_a_set{
442 res_a.atomic_formulas.begin(), res_a.atomic_formulas.end()};
443 std::unordered_set<PrimExpr, StructuralHash, StructuralEqual> res_b_set{
444 res_b.atomic_formulas.begin(), res_b.atomic_formulas.end()};
445
446 // For the Or case we intersect the sets of atomic formulas
447 std::unordered_set<PrimExpr, StructuralHash, StructuralEqual> res_set;
448 std::vector<PrimExpr> res;
449 res_set.reserve(std::min(res_a.atomic_formulas.size(), res_b.atomic_formulas.size()));
450 res.reserve(std::min(res_a.atomic_formulas.size(), res_b.atomic_formulas.size()));
451 for (const auto& res_b_formula : res_b.atomic_formulas) {
452 if (res_a_set.count(res_b_formula)) {
453 res_set.insert(res_b_formula);
454 res.push_back(res_b_formula);
455 }
456 }
457
458 // Computing the residual is more complex: we have to compute the sets of atomic formulas
459 // which are left behind, and then combine them with the residuals into the new residual.
460 std::vector<PrimExpr> new_cond_a;
461 new_cond_a.reserve(res_a.atomic_formulas.size() - res_set.size());
462 for (const auto& formula : res_a.atomic_formulas) {
463 if (!res_set.count(formula)) new_cond_a.emplace_back(formula);
464 }
465
466 std::vector<PrimExpr> new_cond_b;
467 new_cond_b.reserve(res_b.atomic_formulas.size() - res_set.size());
468 for (const auto& formula : res_b.atomic_formulas) {
469 if (!res_set.count(formula)) new_cond_b.emplace_back(formula);
470 }
471
472 res_a.atomic_formulas = std::move(new_cond_a);
473 res_b.atomic_formulas = std::move(new_cond_b);
474
475 PrimExpr new_rest = res_a.to_expr() || res_b.to_expr();
476
477 return {res, new_rest};
478 }
479};

481// Transform the given formula into a conjunction of atomic formulas (represented as an array)
482// and a non-atomic residual. Atomic formulas are consts, calls, variables and comparisons (a <= b,
483// etc), i.e. formulas which are not logical operators (||, &&, !) on the top level.
484FactorOutAtomicFormulasResult FactorOutAtomicFormulas(const PrimExpr& e) {
485 ICHECK(e.dtype().is_bool());
486 return FactorOutAtomicFormulasFunctor().VisitExpr(e);
487}

497inline PrimExpr ModImpl(PrimExpr a, PrimExpr b, DivMode mode) {
498 if (mode == kTruncDiv) {
499 return truncmod(a, b);
500 } else {
501 ICHECK_EQ(mode, kFloorDiv);
502 return floormod(a, b);
503 }
504}

506inline PrimExpr DivImpl(PrimExpr a, PrimExpr b, DivMode mode) {
507 if (mode == kTruncDiv) {
508 return truncdiv(a, b);
509 } else {
510 ICHECK_EQ(mode, kFloorDiv);
511 return floordiv(a, b);
512 }

515class EliminateDivModMutator : public ExprMutator {
516 public:
517 Map<Var, PrimExpr> substitution;
518 Array<Var> new_variables;
519 Array<PrimExpr> conditions;
520 Map<Var, Range> ranges;
521
522 explicit EliminateDivModMutator(Map<Var, Range> ranges) : ranges(std::move(ranges)) {}
523
524 virtual PrimExpr VisitExpr_(const DivNode* op) {
525 const IntImmNode* imm = op->b.as<IntImmNode>();
526 if (imm && imm->value != 0) {
527 if (imm->value < 0) {
528 // x / -c == -(x/c) for truncated division
529 return make_zero(op->dtype) -
530 VisitExpr(truncdiv(op->a, make_const(op->dtype, -imm->value)));
531 }
532
533 // Try to find the already existing variables for this expression
534 auto it = expr_to_vars_.find(std::make_tuple(kTruncDiv, op->a, imm->value));
535 if (it != expr_to_vars_.end()) {
536 return it->second.first;
537 }
538
539 // Otherwise recursively mutate the left hand side, and create new variables
540 PrimExpr mutated_a = VisitExpr(op->a);
541 if (auto var_pair_opt = AddNewVarPair(op->a, mutated_a, imm->value, kTruncDiv)) {
542 return var_pair_opt.value().first;
543 } else {
544 return truncdiv(mutated_a, op->b);
545 }
546 }
547
548 return div(VisitExpr(op->a), VisitExpr(op->b));
549 }
550
551 virtual PrimExpr VisitExpr_(const ModNode* op) {
552 const IntImmNode* imm = op->b.as<IntImmNode>();
553 if (imm && imm->value != 0) {
554 if (imm->value < 0) {
555 // x % -c == x % c for truncated division
556 return VisitExpr(truncmod(op->a, make_const(op->dtype, -imm->value)));
557 }
558
559 // Try to find the already existing variables for this expression
560 auto it = expr_to_vars_.find(std::make_tuple(kTruncDiv, op->a, imm->value));
561 if (it != expr_to_vars_.end()) {
562 return it->second.second;
563 }
564
565 // Otherwise recursively mutate the left hand side, and create new variables
566 PrimExpr mutated_a = VisitExpr(op->a);
567 if (auto var_pair_opt = AddNewVarPair(op->a, mutated_a, imm->value, kTruncDiv)) {
568 return var_pair_opt.value().second;
569 } else {
570 return truncmod(mutated_a, op->b);
571 }
572 }
573
574 return truncmod(VisitExpr(op->a), VisitExpr(op->b));
575 }
576
577 virtual PrimExpr VisitExpr_(const FloorDivNode* op) {
578 const IntImmNode* imm = op->b.as<IntImmNode>();
579 if (imm && imm->value != 0) {
580 if (imm->value < 0) {
581 // x / -c == (-x) / c for flooring division
582 return VisitExpr(
583 floordiv(make_zero(op->dtype) - op->a, make_const(op->dtype, -imm->value)));
584 }
585
586 // Try to find the already existing variables for this expression
587 auto it = expr_to_vars_.find(std::make_tuple(kFloorDiv, op->a, imm->value));
588 if (it != expr_to_vars_.end()) {
589 return it->second.first;
590 }
591
592 // Otherwise recursively mutate the left hand side, and create new variables
593 PrimExpr mutated_a = VisitExpr(op->a);
594 if (auto var_pair_opt = AddNewVarPair(op->a, mutated_a, imm->value, kFloorDiv)) {
595 return var_pair_opt.value().first;
596 } else {
597 return floordiv(mutated_a, op->b);
598 }
599 }
600
601 return floordiv(VisitExpr(op->a), VisitExpr(op->b));
602 }
603
604 virtual PrimExpr VisitExpr_(const FloorModNode* op) {
605 const IntImmNode* imm = op->b.as<IntImmNode>();
606 if (imm && imm->value != 0) {
607 if (imm->value < 0) {
608 // x % -c == -(-x % c) for flooring division
609 return VisitExpr(make_zero(op->dtype) - floormod(make_zero(op->dtype) - op->a,
610 make_const(op->dtype, -imm->value)));
611 }
612
613 // Try to find the already existing variables for this expression
614 auto it = expr_to_vars_.find(std::make_tuple(kFloorDiv, op->a, imm->value));
615 if (it != expr_to_vars_.end()) {
616 return it->second.second;
617 }
618
619 // Otherwise recursively mutate the left hand side, and create new variables
620 PrimExpr mutated_a = VisitExpr(op->a);
621 if (auto var_pair_opt = AddNewVarPair(op->a, mutated_a, imm->value, kFloorDiv)) {
622 return var_pair_opt.value().second;
623 } else {
624 return floormod(mutated_a, op->b);
625 }
626 }
627
628 return floormod(VisitExpr(op->a), VisitExpr(op->b));
629 }
630
631 private:
632 std::optional<std::pair<Var, Var>> AddNewVarPair(const PrimExpr& e, const PrimExpr& mut,
633 int64_t val, DivMode mode) {
634 using tresult = std::optional<std::pair<Var, Var>>;
635
636 // Try to find the variables using the mutated expressions
637 if (!e.same_as(mut)) {
638 auto it = expr_to_vars_.find(std::make_tuple(mode, mut, val));
639 if (it != expr_to_vars_.end()) {
640 return tresult(it->second);
641 }
642 }
643
644 PrimExpr val_e = make_const(e.dtype(), val);
645 idx_ += 1;
646
647 // Convert `ranges` to IntSets
648 std::unordered_map<const VarNode*, IntSet> var_intsets;
649 for (const auto& p : ranges) {
650 var_intsets[p.first.get()] = IntSet::FromRange(p.second);
651 }
652
653 // Infer ranges for the expressions we want to replace with variables
654 Range div_range = EvalSet(DivImpl(mut, val_e, mode), var_intsets).CoverRange(Range());
655 Range mod_range = EvalSet(ModImpl(mut, val_e, mode), var_intsets).CoverRange(Range());
656
657 // We don't want to add unbounded variables
658 if (!div_range.get() || !mod_range.get()) {
659 LOG(WARNING) << "EliminateDivMod: won't eliminate " << DivImpl(e, val_e, mode)
660 << " because its bounds cannot be inferred";
661 return tresult();
662 }
663 if (!mod_range.get()) {
664 LOG(WARNING) << "EliminateDivMod: won't eliminate " << ModImpl(e, val_e, mode)
665 << " because its bounds cannot be inferred";
666 return tresult();
667 }
668
669 // Create new variables for the expressions
670 auto div = Var((mode == kTruncDiv ? "tdiv" : "fdiv") + std::to_string(idx_), e.dtype());
671 auto mod = Var((mode == kTruncDiv ? "tmod" : "fmod") + std::to_string(idx_), e.dtype());
672
673 new_variables.push_back(div);
674 new_variables.push_back(mod);
675
676 // Note that we have to perform substitution to mut because mut may contain new variables
677 substitution.Set(div, DivImpl(Substitute(mut, substitution), val_e, mode));
678 substitution.Set(mod, ModImpl(Substitute(mut, substitution), val_e, mode));
679
680 ranges.Set(div, div_range);
681 ranges.Set(mod, mod_range);
682
683 // This additional condition works as a definition for the new variables
684 conditions.push_back(mut == div * val_e + mod);
685
686 if (!analyzer_.CanProve(mod_range->extent <= val_e)) {
687 // If we use the C/C++ definition of mod, there may be multiple values of `mod`
688 // satisfying the added condition if the expr `e` may change its sign, so we
689 // have to add another condition.
690 LOG(WARNING) << "EliminateDivMod: cannot fully eliminate div or mod because "
691 << ModImpl(e, val_e, mode) << " probably may change its sign";
692 conditions.push_back(Select(e >= 0, mod >= 0, mod <= 0));
693 }
694
695 auto p = std::make_pair(div, mod);
696 expr_to_vars_[std::make_tuple(mode, e, val)] = p;
697 if (!e.same_as(mut)) {
698 expr_to_vars_[std::make_tuple(mode, mut, val)] = p;
699 }
700 return tresult(p);
701 }

704 public:
705 bool operator()(const std::tuple<DivMode, PrimExpr, int64_t>& lhs,
706 const std::tuple<DivMode, PrimExpr, int64_t>& rhs) const {
707 return std::get<0>(lhs) == std::get<0>(rhs) &&
708 tir::ExprDeepEqual()(std::get<1>(lhs), std::get<1>(rhs)) &&
709 std::get<2>(lhs) == std::get<2>(rhs);

713 class TupleHasher_ {
714 public:
715 size_t operator()(const std::tuple<DivMode, PrimExpr, int64_t>& key) const {
716 return ((std::hash<int>()(std::get<0>(key)) ^ (StructuralHash()(std::get<1>(key)) << 1)) >>
717 1) ^
718 (std::hash<int64_t>()(std::get<2>(key)) << 1);

727 TupleEqual_>
728 expr_to_vars_;
729 arith::Analyzer analyzer_;

732// Replace every subexpr of the form e/const and e % const with a new variable.
733// Syntactically equal expressions will be mapped to the same variable.
734EliminateDivModResult EliminateDivMod(const PrimExpr& expr, Map<Var, Range> ranges) {
735 EliminateDivModResult res;
736 EliminateDivModMutator mutator(ranges);
737 res.expr = mutator(expr);
738 res.conditions = std::move(mutator.conditions);
739 res.new_variables = std::move(mutator.new_variables);
740 res.substitution = std::move(mutator.substitution);
741 res.ranges = std::move(mutator.ranges);
742 return res;

745arith::IntConstraintsTransform EliminateDivModFromDomainConditions(
746 const arith::IntConstraints& domain) {
747 auto elim_res = EliminateDivMod(All(domain->relations), domain->ranges);
748
749 Map<Var, Range> new_vranges = elim_res.ranges;
750 Array<Var> new_axis = Concat(domain->variables, elim_res.new_variables);
751 PrimExpr new_cond = elim_res.expr && All(elim_res.conditions);
752
753 arith::IntConstraints new_domain(new_axis, new_vranges,
754 FactorOutAtomicFormulas(new_cond).to_array());
755
756 Map<Var, PrimExpr> src_to_dst;
757 Map<Var, PrimExpr> dst_to_src = elim_res.substitution;
758 for (const Var& v : domain->variables) {
759 src_to_dst.Set(v, v);
760 dst_to_src.Set(v, v);
761 }
762
763 return arith::IntConstraintsTransform(domain, new_domain, src_to_dst, dst_to_src);
764}

766inline arith::IntConstraintsTransform IdentityTransformation(const arith::IntConstraints& domain) {
767 Map<Var, PrimExpr> identity_map;
768 for (const Var& v : domain->variables) {
769 identity_map.Set(v, v);
770 }
771 return arith::IntConstraintsTransform(domain, domain, identity_map, identity_map);
772}

774// Simplify an iteration domain.
775arith::IntConstraintsTransform SimplifyDomain(const arith::IntConstraints& iter_domains,
776 bool eliminate_div_mod) {
777 arith::IntConstraintsTransform transf = IdentityTransformation(iter_domains);
778
779 if (eliminate_div_mod) {
780 transf = transf + EliminateDivModFromDomainConditions(transf->dst);
781 }
782
783 // TODO(sgrechanik-h): Repeating the following steps has a positive effect, however we probably
784 // should find a better terminating criterion (like stop when the domain volume stops decreasing)
785 // Also 2 steps seems to be slightly better than 3
786 for (size_t i = 0; i < 2; ++i) {
787 transf = transf + arith::SolveLinearEquations(transf->dst);
788 transf = transf + arith::SolveInequalitiesDeskewRange(transf->dst);
789 }
790
791 return transf;
792}

794// Use the condition of a reduction op to simplify its domain (axis)
795PrimExpr SimplifyReductionDomain(const PrimExpr& expr, const Map<Var, Range>& outer_vranges) {
796 if (const ReduceNode* red = expr.as<ReduceNode>()) {
797 Array<Var> vars = IterVarsToVars(red->axis);
798 Map<Var, Range> vranges = Merge(outer_vranges, IterVarsToMap(red->axis));
799 Array<PrimExpr> relations = FactorOutAtomicFormulas(red->condition).to_array();
800
801 arith::IntConstraints domain(vars, vranges, relations);
802 auto res = SimplifyDomain(domain);
803

806 new_source.push_back(Substitute(src, res->src_to_dst));
807 }
808
809 Array<IterVar> new_axis = IterVarsFromMap(res->dst->variables, res->dst->ranges, kCommReduce);
810
811 // Perform simplification mainly to remove a possibly empty reduction.
812 arith::Analyzer analyzer;
813 return analyzer.Simplify(Reduce(red->combiner, new_source, new_axis, All(res->dst->relations),
814 red->value_index, red->init),
815 kSimplifyRewriteCanonicalRewrite);
816 } else {
817 return expr;
818 }

821// Extract from cond an implication of cond not containing vars
822std::pair<PrimExpr, PrimExpr> ImplicationNotContainingVars(
823 const PrimExpr& cond, const std::unordered_set<const VarNode*>& vars) {
824 ICHECK(cond.dtype().is_bool()) << "The type of cond must be bool";
825 // TODO(sgrechanik-h): NOTs could be pushed down using De Morgan laws
826 // before running this function but this case didn't seem to be important enough.
827 if (const AndNode* op = cond.as<AndNode>()) {
828 auto pair_a = ImplicationNotContainingVars(op->a, vars);
829 auto pair_b = ImplicationNotContainingVars(op->b, vars);
830 return {pair_a.first && pair_b.first, pair_a.second && pair_b.second};
831 } else if (const OrNode* op = cond.as<OrNode>()) {
832 auto pair_a = ImplicationNotContainingVars(op->a, vars);
833 auto pair_b = ImplicationNotContainingVars(op->b, vars);
834 return {pair_a.first || pair_b.first, (pair_a.first || pair_b.second) &&
835 (pair_b.first || pair_a.second) &&
836 (pair_a.second || pair_b.second)};
837 } else if (!tir::UsesVar(cond, [&vars](const VarNode* var) { return vars.count(var); })) {
838 return {cond, const_true()};
839 } else {
840 return {const_true(), cond};
841 }
842}

844// Factor conditions out of a reduction by applying Fourier-Motzkin elimination and moving out
845// (in)equalities which do not depend on the reduction variables.
846std::pair<PrimExpr, PrimExpr> LiftConditionsThroughReduction(const PrimExpr& cond,
847 const Array<IterVar>& red_axis,
848 const Array<IterVar>& outer_axis) {
849 // Factor out atomics so that we can consider this as a system of inequalities
850 auto factor_atomic_res = FactorOutAtomicFormulas(cond);
851 Array<PrimExpr> atomics = factor_atomic_res.atomic_formulas;
852 const PrimExpr& rest = factor_atomic_res.rest;
853
854 Array<Var> allvars;
855 for (const IterVar& v : red_axis) {
856 allvars.push_back(v->var);
857 }
858 for (const IterVar& v : outer_axis) {
859 allvars.push_back(v->var);
860 }
861
862 auto vranges = Merge(IterVarsToMap(red_axis), IterVarsToMap(outer_axis));
863 // start from reduction vars, so that input vars don't depend on them
864 arith::IntConstraints ineq_to_solve(allvars, vranges, atomics);
865 auto res_ineq = arith::SolveLinearInequalities(ineq_to_solve);
866 atomics = arith::AsConditions(allvars, res_ineq.first, res_ineq.second);
867
868 // Append the rest part
869 PrimExpr rewritten_cond = All(atomics) && rest;
870
871 std::unordered_set<const VarNode*> vset;
872 for (const IterVar& v : red_axis) {
873 vset.insert(v->var.get());
874 }
875
876 // The outer (first) condition does not contain reduction vars,
877 // the inner (second) condition is everything else
878 auto res = ImplicationNotContainingVars(rewritten_cond, vset);
879 return res;
880}

882// Convert an array of itervars to an array of inequalities
883Array<PrimExpr> IterVarsToInequalities(const Array<IterVar>& itervars) {
884 Array<PrimExpr> res;
885 for (const IterVar& v : itervars) {
886 res.push_back(GE(v->var, v->dom->min));
887 res.push_back(LT(v->var, v->dom->min + v->dom->extent));
888 }
889 return res;
890}

892class RemoveRedundantInequalitiesMutator : public ExprMutator {
893 public:
894 explicit RemoveRedundantInequalitiesMutator(Array<PrimExpr> known) {
895 for (const PrimExpr& cond : known) {
896 known_.push_back(analyzer_.Simplify(cond, kSimplifyRewriteCanonicalRewrite));
897 }
898 }
899
900 virtual PrimExpr VisitExpr_(const SelectNode* op) {
901 bool has_side_effect = (SideEffect(GetRef<PrimExpr>(op)) > CallEffectKind::kReadState);
902 PrimExpr new_cond =
903 analyzer_.Simplify(VisitExpr(op->condition), kSimplifyRewriteCanonicalRewrite);
904 if (is_one(new_cond) && !has_side_effect) {
905 return VisitExpr(op->true_value);
906 } else if (is_zero(new_cond) && !has_side_effect) {
907 return VisitExpr(op->false_value);
908 } else {
909 Array<PrimExpr> new_known = known_;
910 for (const PrimExpr& atomic : FactorOutAtomicFormulas(new_cond).atomic_formulas) {
911 new_known.push_back(atomic);
912 }
913 RemoveRedundantInequalitiesMutator new_mutator(new_known);
914 // Note that we mutate only the true value with the new mutator
915 // TODO(sgrechanik-h): Update known conditions for the false value as well
916 return Select(new_cond, new_mutator(op->true_value), VisitExpr(op->false_value));
917 }
918 }
919
920 virtual PrimExpr VisitExpr_(const CallNode* op) {
921 if (op->op.same_as(op_if_then_else_)) {
922 PrimExpr new_cond =
923 analyzer_.Simplify(VisitExpr(op->args[0]), kSimplifyRewriteCanonicalRewrite);
924 if (is_one(new_cond)) {
925 return VisitExpr(op->args[1]);
926 } else if (is_zero(new_cond)) {
927 return VisitExpr(op->args[2]);
928 } else {
929 Array<PrimExpr> new_known = known_;
930 for (const PrimExpr& atomic : FactorOutAtomicFormulas(new_cond).atomic_formulas) {
931 new_known.push_back(atomic);
932 }
933 RemoveRedundantInequalitiesMutator new_mutator(new_known);
934 // Note that we mutate only the true value with the new mutator
935 // TODO(sgrechanik-h): Update known conditions for the false value as well
936 return if_then_else(new_cond, new_mutator(op->args[1]), VisitExpr(op->args[2]));
937 }
938 } else {
939 return ExprMutator::VisitExpr_(op);
940 }
941 }
942
943 virtual PrimExpr VisitExpr_(const ReduceNode* op) {
944 Array<PrimExpr> known_with_axes = known_;
945 ICHECK(op->init.empty()) << "Derivative of Reduction with initialization is not implemented";
946 for (const PrimExpr& axis_cond : IterVarsToInequalities(op->axis)) {
947 known_with_axes.push_back(axis_cond);
948 }
949 RemoveRedundantInequalitiesMutator mutator_with_axes(known_with_axes);
950
951 PrimExpr new_cond = mutator_with_axes(op->condition);
952
953 Array<PrimExpr> new_known = known_with_axes;
954 for (const PrimExpr& atomic : FactorOutAtomicFormulas(new_cond).atomic_formulas) {
955 new_known.push_back(atomic);
956 }
957 RemoveRedundantInequalitiesMutator new_mutator(new_known);
958
959 Array<PrimExpr> new_source;
960 for (const PrimExpr& src : op->source) {
961 new_source.push_back(new_mutator(src));
962 }
963
964 return Reduce(op->combiner, new_source, op->axis, new_cond, op->value_index, op->init);
965 }
966
967 virtual PrimExpr VisitExpr_(const EQNode* op) { return MutateAtomic_(GetRef<PrimExpr>(op)); }
968 virtual PrimExpr VisitExpr_(const NENode* op) { return MutateAtomic_(GetRef<PrimExpr>(op)); }
969 virtual PrimExpr VisitExpr_(const LTNode* op) { return MutateAtomic_(GetRef<PrimExpr>(op)); }
970 virtual PrimExpr VisitExpr_(const LENode* op) { return MutateAtomic_(GetRef<PrimExpr>(op)); }
971 virtual PrimExpr VisitExpr_(const GTNode* op) { return MutateAtomic_(GetRef<PrimExpr>(op)); }
972 virtual PrimExpr VisitExpr_(const GENode* op) { return MutateAtomic_(GetRef<PrimExpr>(op)); }
973
974 virtual PrimExpr VisitExpr_(const AndNode* op) { return VisitExpr(op->a) && VisitExpr(op->b); }
975
976 private:
977 PrimExpr MutateAtomic_(const PrimExpr& e) {
978 PrimExpr simplified = analyzer_.Simplify(e, kSimplifyRewriteCanonicalRewrite);
979 for (const PrimExpr& other : known_) {
980 if (ExprDeepEqual()(simplified, other)) {
981 return const_true();
982 }
983 }
984 return simplified;
985 }
986
987 Array<PrimExpr> known_;
988 arith::Analyzer analyzer_;
989 const Op& op_if_then_else_ = Op::Get("tir.if_then_else");
990};

997// Extract the given expr under the given condition as a separate tensor if the volume of the
998// extracted tensor will be less than the volume of the outer_axis
999PrimExpr TrySimplifyCompute(const PrimExpr& expr, const PrimExpr& cond,
1000 const Array<Var>& outer_axis, const Map<Var, Range>& vranges) {
1001 // solve cond, e.g., (jac_i0 == i) && (jac_i1 == j)
1002 arith::IntConstraints domain_to_solve(outer_axis, vranges,
1003 FactorOutAtomicFormulas(cond).to_array());
1004 auto res = SimplifyDomain(domain_to_solve);
1005
1006 arith::Analyzer analyzer;
1007 analyzer.Bind(res->dst->ranges);
1008 PrimExpr new_expr =
1009 analyzer.Simplify(Substitute(expr, res->src_to_dst), kSimplifyRewriteCanonicalRewrite);
1010 // TODO(yzhliu): This is mostly done to simplify if_then_else
1011 // which is not realized by the canonical simplifier
1012 new_expr = RemoveRedundantInequalities(new_expr, res->dst->relations);
1013
1014 // Keep only those variables of the new vars which are used in the new_expr
1015 Array<Var> used_res_variables;
1016 for (const Var& var : res->dst->variables) {
1017 if (tir::UsesVar(new_expr, [&var](const VarNode* var_) { return var_ == var.get(); })) {
1018 ICHECK(res->dst->ranges.count(var)) << "Range of " << var << " cannot be inferred.";
1019 used_res_variables.push_back(var);
1020 }
1021 }
1022
1023 // If the expression does not use vars then it is probably better to keep it inlined
1024 if (used_res_variables.empty()) {
1025 // We can return the new_expr here instead of the old expr because it doesn't use variables
1026 // otherwise we would need to replace the new vars or create a let-expression
1027 return new_expr;
1028 }
1029
1030 // If it's already tensor[...] then it will probably be useless to further simplify it.
1031 if (new_expr.as<ProducerLoadNode>()) {
1032 return expr;
1033 }
1034
1035 // Compute volumes before and after
1036 PrimExpr old_volume = make_const(DataType::Int(64), 1);
1037 for (const Var& var : outer_axis) {
1038 ICHECK(vranges.count(var)) << "Range of " << var << " was not provided.";
1039 old_volume = old_volume * vranges[var]->extent;
1040 }
1041
1042 PrimExpr new_volume = make_const(DataType::Int(64), 1);
1043 for (const Var& var : used_res_variables) {
1044 new_volume = new_volume * res->dst->ranges[var]->extent;
1045 }
1046
1047 // if we can prove that the old volume is not greater than the new volume then
1048 // prefer the old expression.
1049 arith::Analyzer ana_vranges;
1050 ana_vranges.Bind(vranges);
1051 if (ana_vranges.CanProve(old_volume <= new_volume)) {
1052 return expr;
1053 }
1054
1055 Tensor tensor = TensorFromExpr(new_expr, IterVarsFromMap(used_res_variables, res->dst->ranges),
1056 "extracted_tensor");
1057
1058 Array<PrimExpr> args;
1059 for (const Var& var : used_res_variables) {
1060 args.push_back(res->dst_to_src[var]);
1061 }
1062
1063 return ProducerLoad(tensor, args);
1064}

1148PrimExpr RemoveJacobianAndLiftNonzeroCondImpl(const PrimExpr& expr_orig, const Array<IterVar>& axis,

1158 if (const ReduceNode* red = expr.as<ReduceNode>()) {
1159 ICHECK(red->init.empty()) << "Derivative of Reduction with initialization is not implemented";
1160 // TODO(sgrechanik-h): There are some other operations which behave like sum
1161 bool is_sum = IsSumCombiner(red->combiner, vranges);
1162 if (is_sum || CanFactorZeroFromCombiner(red->combiner, red->value_index, vranges)) {
1163 PrimExpr new_red = expr;
1164
1165 // Here we simplify the reduction
1166 PrimExpr cond = red->condition;
1167 Array<PrimExpr> source = red->source;
1168
1169 // If it is a summation then we can lift nonzeroness conditions from the source
1170 // and add them to the reduction conditions
1171 if (is_sum) {
1172 auto nz = NonzeronessCondition(red->source[red->value_index]);
1173 cond = nz.cond && cond;
1174 source.Set(0, nz.value);
1175 }
1176
1177 new_red = Reduce(red->combiner, source, red->axis, cond, red->value_index, red->init);
1178 new_red = SimplifyReductionDomain(new_red, combined_vranges);
1179 // Update original red pointer for later use.
1180 red = new_red.as<ReduceNode>();
1181 // If the reduction disappears completely then transform the result as a non-reduction
1182 if (!red) {
1183 return RemoveJacobianAndLiftNonzeroCondImpl(new_red, axis, vranges);
1184 }
1185
1186 Array<PrimExpr> new_source = red->source;
1187
1188 // Partially lift conditions from the reduce condition
1189 auto [new_outer_cond, new_reduce_cond] =
1190 LiftConditionsThroughReduction(red->condition, red->axis, axis);
1191
1192 // If it's not sum then we haven't yet lifted nonzeroness cond from the source
1193 if (!is_sum) {
1194 auto nz = NonzeronessCondition(red->source[red->value_index]);
1195 // Append conditions from the reduction
1196 PrimExpr nz_source = nz.value;
1197 auto [outer_nz_cond, nz_cond] =
1198 LiftConditionsThroughReduction(new_reduce_cond && nz.cond, red->axis, axis);
1199 new_outer_cond = new_outer_cond && outer_nz_cond;
1200 new_source.Set(red->value_index, Select(nz_cond, nz_source, make_zero(nz_source.dtype())));
1201 }
1202
1203 PrimExpr new_reduce = Reduce(red->combiner, new_source, red->axis, new_reduce_cond,
1204 red->value_index, red->init);
1205 new_reduce =
1206 TrySimplifyCompute(new_reduce, new_outer_cond, IterVarsToVars(axis), combined_vranges);
1207 result = Select(new_outer_cond, new_reduce, make_zero(new_reduce.dtype()));
1208 } else {
1209 return SimplifyReductionDomain(expr, combined_vranges);
1210 }
1211 } else {
1212 auto nz = NonzeronessCondition(expr);
1213 PrimExpr new_expr =
1214 TrySimplifyCompute(nz.value, nz.cond, IterVarsToVars(axis), combined_vranges);
1215 result = Select(nz.cond, new_expr, make_zero(new_expr.dtype()));
1216 }
1217
1218 // Note that RemoveRedundantInequalities can sometimes propagate equalities which
1219 // other simplifiers cannot, like (i % 3) == 0.
1220 Array<PrimExpr> axis_conds = IterVarsToInequalities(axis);
1221 result = RemoveRedundantInequalities(result, axis_conds);
1222
1223 // Currently in TVM reductions are only allowed at the top level of compute,
1224 // we need to extract intermediate inlined reduction as a separate stage (tensor).
1225 // Sometimes TrySimplifyCompute doesn't perform lift / extraction,
1226 // so there may be some non-top reductions left, take care of them.
1227 result = LiftReductions(result, IterVarsToVars(axis), combined_vranges);
1228 return analyzer.Simplify(result, kSimplifyRewriteCanonicalRewrite);
1229}

1231Tensor RemoveJacobianAndLiftNonzeroCond(const Tensor& tensor, const Map<Var, Range>& vranges) {
1232 auto transform_func = [&vranges](const PrimExpr& expr, const Array<IterVar>& axis) {
1233 return RemoveJacobianAndLiftNonzeroCondImpl(expr, axis, vranges);
1234 };
1235 return TransformTensorBody(tensor, transform_func);
1236}
