20/*!
21 * \file adjoint.cc
22 * \brief Perform reverse-mode autodiff.
23 * Suppose we have f(x) = g(h1(x), h2(x), ..., hn(x)),
24 * df/dx = \sum_i df/dhi * dhi/dx
25 * We call df/dx as adjoint(x), df/dhi as adjoint(hi), dhi/dx is the Jacobian
26 * The idea is to first construct the reverse-dependency {input->outputs} between tensors,
27 * start from one input,
28 * (1) collect adjoints from all its dependencies (outputs),
29 * (2) multiply the Jacobian (PartialAdjoint),
30 * (3) and sum them together to get the adjoint of the input itself.
31 * The three steps are computed recursively.
32 */

47Tensor Identity(const Tensor& output) {
48 Array<PrimExpr> shape = output->shape;
49 for (auto e : output->shape) {
50 // add extra dimension for Jacobian
51 shape.push_back(e);
52 }
53 auto func = [&output](const Array<Var>& input_indices) {
54 PrimExpr res = const_true();
55 for (size_t i = 0; i < output->shape.size(); ++i) {
56 res =
57 res && (PrimExpr(input_indices[i]) == PrimExpr(input_indices[output->shape.size() + i]));
58 }
59 return Cast(output->dtype, res);
60 };
61 return te::compute(shape, func, "identity");
62}
63
64Tensor VectorJacobianProduct(const Tensor& output, const Tensor& input, const Tensor& head) {
65 Tensor jac = Jacobian(output, input);
66 Tensor result = topi::tensordot(head, jac, /*axes=*/output->shape.size(),
67 output->op->name + "." + input->op->name + ".grad");
68 result = InlineTensorAccess(result, {jac}, false);
69 result = RemoveJacobianAndLiftNonzeroCond(result);
70 // inline tail call
71 result = InlineTailTensorAccess(result);
72 return result;
73}

75Array<Tensor> Gradient(const Tensor& output, const Array<Tensor>& inputs,
76 const Tensor& head_or_null) {
77 // Diagonal identity tensor
78 Tensor head = head_or_null.get() ? head_or_null : Identity(output);
79
80 // This Map{input -> outputs} maps a tensor to the list of tensors
81 // immediately depending on it (using it in their bodies)
82 std::unordered_map<Tensor, std::vector<Tensor>> reverse_dependencies;
83 std::vector<Tensor> stack({output});
84 while (!stack.empty()) {
85 Tensor tensor = stack.back();
86 stack.pop_back();
87 for (const Tensor& input : tensor->op->InputTensors()) {
88 if (!reverse_dependencies.count(input)) {
89 stack.push_back(input);
90 }
91 reverse_dependencies[input].push_back(tensor);
92 }
93 }
94
95 // This map maps tensors to the corresponding adjoints (dLoss/dTensor)
96 std::unordered_map<Tensor, Tensor> adjoints;
97 // head is the adjoint of output by definition
98 adjoints[output] = head;
99
100 // This is a recursive function that does all the work. It computes the adjoint for a given
101 // tensor, adds it to the map, and returns it
102 std::function<Tensor(const Tensor&)> compute_adjoint;
103 compute_adjoint = [&compute_adjoint, &adjoints, &reverse_dependencies, &head,
104 &output](const Tensor& tensor) {
105 if (!adjoints.count(tensor)) {
106 // Here the adjoint hasn't been computed yet
107 Tensor res_adjoint;
108 std::vector<Tensor> direct_consumers = reverse_dependencies[tensor];
109 if (direct_consumers.empty()) {
110 // No reverse dependencies means that the output does not depend on this tensor,
111 // return a zero tensor of the appropriate shape
112 // (i.e., output shape + tensor shape, aka shape of Jacobian)
113 Array<PrimExpr> result_shape(head->shape.begin(), head->shape.end() - output->shape.size());
114 for (auto e : tensor->shape) {
115 result_shape.push_back(e);
116 }
117 res_adjoint = topi::full(result_shape, output->dtype, make_zero(output->dtype));
118 } else {
119 // The new adjoint is computed as a sum of the reverse dependencies' adjoints multiplied
120 // by the corresponding "local" jacobians (dDep/dTensor). The computation of the jacobian
121 // and the multiplication is done in the function VectorJacobianProduct
122 for (const Tensor& direct_consumer : direct_consumers) {
123 // part = (adjoint of direct_consumer) * Jacobian(direct_consumer, tensor)
124 Tensor part =
125 VectorJacobianProduct(direct_consumer, tensor, compute_adjoint(direct_consumer));
126 res_adjoint = res_adjoint.get() ? topi::add(res_adjoint, part) : part;
127 }
128 }
129
130 adjoints[tensor] = res_adjoint;
131 return res_adjoint;
132 } else {
133 return adjoints[tensor];
134 }
135 };
136
137 // Adjoints corresponding to inputs
138 Array<Tensor> result;
139 // Compute an adjoint for each input
140 for (const Tensor& input : inputs) {
141 result.push_back(compute_adjoint(input));
142 }
143
144 return result;
145}

147TVM_REGISTER_GLOBAL("te.Gradient").set_body([](TVMArgs args, TVMRetValue* ret) {
148 LOG(WARNING) << "te.Gradient is an experimental feature.";
149 if (args.size() == 2) {
150 *ret = Gradient(args[0], args[1]);
151 } else if (args.size() == 3) {
152 *ret = Gradient(args[0], args[1], args[2]);
153 }
154});
