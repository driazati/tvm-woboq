38#define NOT_IMPLEMENTED \
39 { \
40 LOG(FATAL) << "Derivative of this expr is not implemented: " << GetRef<PrimExpr>(op); \
41 throw; \
42 }

44/*! \brief Differentiate an expression wrt a variable or a tensor element */
45class JacobianMutator : public ExprMutator {
46 public:
47 /*!
48 * \brief Differentiate wrt `input(indices)`.
49 * \param input The input tensor.
50 * \param indices The indices of the element with respect to which to differentiate.
51 */
52 explicit JacobianMutator(Tensor input, Array<PrimExpr> indices)
53 : input_(input), indices_(indices) {}
54 /*!
55 * \brief Differentiate wrt the input variable.
56 * \param input The input variable.
57 */
58 explicit JacobianMutator(Var input) : input_var_(input) {}
59
60 PrimExpr Mutate(PrimExpr e) {
61 if (e.dtype().is_int() || e.dtype().is_uint()) {
62 LOG(WARNING) << "For now we assume that the derivative of any integer expression is always 0."
63 << " e = " << e;
64 return make_zero(e.dtype());
65 } else {
66 return ExprMutator::VisitExpr(e);
67 }
68 }
69
70 PrimExpr VisitExpr_(const VarNode* op) {
71 if (input_var_.get() && input_var_.get() == op && op->dtype.is_float()) {
72 return FloatImm(op->dtype, 1.0);
73 } else {
74 return make_zero(op->dtype);
75 }
76 }
77
78 PrimExpr VisitExpr_(const LoadNode* op) NOT_IMPLEMENTED;
79 PrimExpr VisitExpr_(const LetNode* op) NOT_IMPLEMENTED;
80
81 PrimExpr VisitExpr_(const ProducerLoadNode* op) final {
82 auto tensor = Downcast<te::Tensor>(op->producer);
83 if (input_.get() && tensor == input_) {
84 // Tensor(indices)
85 ICHECK_EQ(indices_.size(), op->indices.size());
86 PrimExpr condition = const_true();
87 for (size_t i = 0; i < input_.ndim(); ++i) {
88 condition = And(condition, EQ(indices_[i], op->indices[i]));
89 }
90 return Cast(op->dtype, condition);
91 } else {
92 return make_zero(op->dtype);
93 }
94 }
95
96 PrimExpr VisitExpr_(const CallNode* op) {
97 PrimExpr expr = GetRef<PrimExpr>(op);
98 if (op->op.same_as(op_exp_)) {
99 return Mul(Mutate(op->args[0]), expr);
100 } else if (op->op.same_as(op_log_)) {
101 return Div(Mutate(op->args[0]), op->args[0]);
102 } else if (op->op.same_as(op_sigmoid_)) {
103 return Mul(Mutate(op->args[0]), Mul(expr, Sub(FloatImm(expr.dtype(), 1.0), expr)));
104 } else if (op->op.same_as(op_sqrt_)) {
105 return Div(Mutate(op->args[0]), Mul(expr, FloatImm(expr.dtype(), 2.0)));
106 } else if (op->op.same_as(op_tanh_)) {
107 return Mul(Mutate(op->args[0]), Sub(FloatImm(expr.dtype(), 1.0), Mul(expr, expr)));
108 } else if (op->op.same_as(op_pow_)) {
109 auto x = op->args[0], y = op->args[1];
110 return expr * (Mutate(y) * log(x) + Mutate(x) * y / x);
111 } else if (op->op.same_as(op_fabs_)) {
112 auto type = op->args[0].dtype();
113 return Mul(Mutate(op->args[0]), Select(GE(op->args[0], make_zero(type)), FloatImm(type, 1.0),
114 FloatImm(type, -1.0)));
115 } else if (op->op.same_as(op_if_then_else_)) {
116 Array<PrimExpr> new_args = {op->args[0], Mutate(op->args[1]), Mutate(op->args[2])};
117 return Call(op->dtype, op->op, new_args);
118 } else if (piecewise_const.count(op->op)) {
119 return FloatImm(expr.dtype(), 0.0);
120 } else {
121 LOG(FATAL) << "Derivative of this intrinsic is not implemented: " << op->op;
122 return PrimExpr();
123 }
124 }

126 PrimExpr VisitExpr_(const AddNode* op) { return Add(Mutate(op->a), Mutate(op->b)); }
127
128 PrimExpr VisitExpr_(const SubNode* op) { return Sub(Mutate(op->a), Mutate(op->b)); }
129
130 PrimExpr VisitExpr_(const MulNode* op) {
131 return Add(Mul(Mutate(op->a), op->b), Mul(op->a, Mutate(op->b)));
132 }
133
134 PrimExpr VisitExpr_(const DivNode* op) {
135 return Div(Sub(Mul(Mutate(op->a), op->b), Mul(op->a, Mutate(op->b))), Mul(op->b, op->b));
136 }

138 PrimExpr VisitExpr_(const ModNode* op) NOT_IMPLEMENTED;
139
140 PrimExpr VisitExpr_(const FloorDivNode* op) {
141 return FloorDiv(Sub(Mul(Mutate(op->a), op->b), Mul(op->a, Mutate(op->b))), Mul(op->b, op->b));
142 }

144 PrimExpr VisitExpr_(const FloorModNode* op) NOT_IMPLEMENTED;

146 PrimExpr VisitExpr_(const MinNode* op) {
147 return Select(LE(op->a, op->b), Mutate(op->a), Mutate(op->b));
148 }

150 PrimExpr VisitExpr_(const MaxNode* op) {
151 return Select(GE(op->a, op->b), Mutate(op->a), Mutate(op->b));

163 PrimExpr VisitExpr_(const ReduceNode* op) {
164 // This case is relatively difficult because a reduction expression
165 // may use an arbitrary combiner.
166 // The resulting reduction expression will return a tuple containing
167 // both derivatives and the original results (in exactly this order).
168 // The order matters when original init value is different from its derivative init value,
169 // and they depend on each other during gradient calculation,
170 // we must calculate derivatives first (using origin's init value),
171 // switching the order (original results first, then derivatives)
172 // makes the origin value be replaced before using,
173 // produces incorrect results.
174
175 // Example of a ReduceNode,
176 // reduce(combiner=comm_reducer(result=[(x + y)], lhs=[x], rhs=[y], identity_element=[0f]),
177 // source=[A(k)], axis=[iter_var(k, range(min=0, ext=5))], where=(bool)1, value_index=0)
178
179 // We have to clone the reduction axes because otherwise the original expression
180 // cannot be used together with the derivative (it will lead to errors during lowering)
181 PrimExpr expr_with_new_axes = te::CloneReduction(GetRef<PrimExpr>(op));
182 const ReduceNode* new_op = expr_with_new_axes.as<ReduceNode>();
183
184 ICHECK(new_op->init.empty())
185 << "Derivative of Reduction with initialization is not implemented";
186
187 // New lhs and rhs variables of the new combiner consist of
188 // variables representing derivatives (which are later derived from new_op->source)
189 // followed by the original variables.
190 Array<Var> new_lhs;
191 for (const auto& var : new_op->combiner->lhs) {
192 new_lhs.push_back(var.copy_with_suffix(".jac"));
193 }
194 for (const auto& var : new_op->combiner->lhs) {
195 new_lhs.push_back(var);
196 }
197
198 Array<Var> new_rhs;
199 for (const auto& var : new_op->combiner->rhs) {
200 new_rhs.push_back(var.copy_with_suffix(".jac"));
201 }
202 for (const auto& var : new_op->combiner->rhs) {
203 new_rhs.push_back(var);
204 }
205
206 // The new combiner result also consists of the resulting derivatives
207 // followed by the original results.
208 Array<PrimExpr> new_result;
209 for (const auto& res : new_op->combiner->result) {
210 // Each resulting derivative is computed as a sum of derivatives
211 // wrt lhs and rhs multiplied by the derivatives of lhs and rhs
212 PrimExpr new_res = make_zero(res.dtype());
213 for (size_t i = 0; i < new_op->combiner->lhs.size(); ++i) {
214 PrimExpr res_di = Derivative(res, new_op->combiner->lhs[i]);
215 // new_lhs[i] is the derivative of lhs[i] (wrt our input tensor)
216 new_res = Add(new_res, Mul(new_lhs[i], res_di));
217 }
218 for (size_t i = 0; i < new_op->combiner->rhs.size(); ++i) {
219 PrimExpr res_di = Derivative(res, new_op->combiner->rhs[i]);
220 // new_rhs[i] is the derivative of rhs[i] (wrt our input tensor)
221 new_res = Add(new_res, Mul(new_rhs[i], res_di));
222 }
223 new_result.push_back(new_res);
224 }
225 // add original results
226 for (const auto& res : new_op->combiner->result) {
227 new_result.push_back(res);
228 }
229
230 // The identity is transformed in a similar way
231 Array<PrimExpr> new_identity;
232 for (const auto& id : new_op->combiner->identity_element) {
233 new_identity.push_back(Mutate(id));
234 }
235 for (const auto& id : new_op->combiner->identity_element) {
236 new_identity.push_back(id);
237 }
238
239 // Same as source
240 Array<PrimExpr> new_source;
241 for (const auto& src : new_op->source) {
242 new_source.push_back(Mutate(src));
243 }
244 for (const auto& src : new_op->source) {
245 new_source.push_back(src);
246 }
247
248 CommReducer new_combiner = CommReducer(new_lhs, new_rhs, new_result, new_identity);
249 // Also simplify the resulting combiner
250 // (mostly to get rid of unused components, e.g., the original expressions)
251 return analyzer_.Simplify(Reduce(new_combiner, new_source, new_op->axis, new_op->condition,
252 new_op->value_index, new_op->init));
253 }

255 PrimExpr VisitExpr_(const CastNode* op) {
256 if (op->dtype.is_float()) {
257 return Cast(op->dtype, Mutate(op->value));
258 } else {
259 return make_zero(op->dtype);
260 }
261 }

285 const Op& op_exp_ = Op::Get("tir.exp");
286 const Op& op_log_ = Op::Get("tir.log");
287 const Op& op_sigmoid_ = Op::Get("tir.sigmoid");
288 const Op& op_sqrt_ = Op::Get("tir.sqrt");
289 const Op& op_tanh_ = Op::Get("tir.tanh");
290 const Op& op_pow_ = Op::Get("tir.pow");
291 const Op& op_fabs_ = Op::Get("tir.fabs");
292 const Op& op_if_then_else_ = Op::Get("tir.if_then_else");
293 std::unordered_set<RelayExpr, ObjectPtrHash, ObjectPtrEqual> piecewise_const = {
294 Op::Get("tir.floor"), Op::Get("tir.ceil"), Op::Get("tir.trunc"), Op::Get("tir.round")};

297PrimExpr Derivative(const PrimExpr& expr, const Var& var) {
298 return JacobianMutator(var).Mutate(expr);
299}

301PrimExpr Jacobian(const PrimExpr& expr, const Tensor& input, const Array<PrimExpr>& indices) {
302 return JacobianMutator(input, indices).Mutate(expr);
303}
304
305Tensor Jacobian(const Tensor& output, const Tensor& input) {
306 const ComputeOpNode* op = output->op.as<ComputeOpNode>();
307 ICHECK(op) << "Derivative of this operation is not implemented: " << output->op;
308 bool is_input_tensor = false;
309 for (const Tensor& child : op->InputTensors()) {
310 if (input == child) {
311 is_input_tensor = true;
312 break;
313 }
314 }
315 ICHECK(is_input_tensor) << "Jacobian is called on a pair of tensors such that the output "
316 << "does not directly depend on the input.";
317
318 // We have to clone the iteration axes because otherwise the original expression
319 // cannot be used together with the derivative (it will lead to errors during lowering)
320 auto [new_axis, vmap] = te::CloneIterVars(op->axis);
321
322 Array<PrimExpr> input_indices;
323 size_t i = 0;
324 for (PrimExpr ext : input->shape) {
325 IterVar new_v =
326 IterVar(Range(0, ext), Var("jac_i" + std::to_string(i++)), IterVarType::kDataPar);
327 // Append jacobian iter to new_axis
328 new_axis.push_back(new_v);
329 // Differentiate wrt input[input_indices]
330 input_indices.push_back(new_v);
331 }
332 arith::Analyzer analzyer;
333 // Compute Jacobian
334 PrimExpr new_body =
335 Jacobian(Substitute(op->body[output->value_index], vmap), input, input_indices);
336 new_body = analzyer.Simplify(new_body);
337
338 int value_index = 0;
339 Array<PrimExpr> new_bodies;
340
341 // If this is a reduction then it may return a tuple and we have
342 // to repeat the body several times
343 if (const ReduceNode* red = new_body.as<ReduceNode>()) {
344 value_index = red->value_index;
345 for (size_t idx = 0; idx < red->source.size(); ++idx) {
346 new_bodies.push_back(
347 Reduce(red->combiner, red->source, red->axis, red->condition, idx, red->init));
348 }
349 } else {
350 new_bodies.push_back(new_body);
351 }
352
353 auto new_op = ComputeOp(op->name + ".jacobian", op->tag, op->attrs, new_axis, new_bodies);
354
355 // Jacobian shape = output.shape + input.shape
356 Array<PrimExpr> new_shape = output->shape;
357 for (const auto& e : input->shape) {
358 new_shape.push_back(e);
359 }
360
361 Tensor ret = Tensor(new_shape, output->dtype, new_op, value_index);
362 ret = RemoveJacobianAndLiftNonzeroCond(ret);
363 return ret;
364}
