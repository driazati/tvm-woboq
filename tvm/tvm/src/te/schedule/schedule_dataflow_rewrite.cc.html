36// find first occurance location in leaf
37template <typename T>
38size_t FindNodeRef(ArrayNode* array_node, const T& v) {
39 const Object* n = v.get();
40 for (size_t i = 0; i < array_node->size(); ++i) {
41 if (array_node->at(i).get() == n) return i;
42 }
43 return array_node->size();
44}

46// The replacer of cache.
47class VarReplacer : public tir::StmtExprMutator {
48 public:
49 explicit VarReplacer(const std::unordered_map<const VarNode*, PrimExpr>& vsub) : vsub_(vsub) {}
50 PrimExpr VisitExpr_(const VarNode* op) final {
51 auto it = vsub_.find(op);
52 if (it != vsub_.end()) return it->second;
53 return GetRef<PrimExpr>(op);
54 }
55
56 tir::CommReducer MutateCommReducer(tir::CommReducer combiner) {
57 // Replace free variables in combiner
58 auto new_identity = tir::UpdateArray(combiner->identity_element,
59 [this](const PrimExpr& e) { return this->VisitExpr(e); });
60 auto new_result = tir::UpdateArray(combiner->result,
61 [this](const PrimExpr& e) { return this->VisitExpr(e); });
62
63 if (combiner->identity_element.same_as(new_identity) &&
64 combiner->identity_element.same_as(new_result)) {
65 return combiner;
66 } else {
67 return tir::CommReducer(combiner->lhs, combiner->rhs, new_result, new_identity);
68 }
69 }
70
71 PrimExpr VisitExpr_(const tir::ReduceNode* op) final {
72 PrimExpr new_e = StmtExprMutator::VisitExpr_(op);
73 const tir::ReduceNode* new_reduce = new_e.as<tir::ReduceNode>();
74 tir::CommReducer new_combiner = MutateCommReducer(op->combiner);
75 if (op->combiner.same_as(new_combiner)) {
76 return new_e;
77 } else {
78 return tir::Reduce(new_combiner, new_reduce->source, new_reduce->axis, new_reduce->condition,
79 new_reduce->value_index, new_reduce->init);
80 }
81 }
82
83 private:
84 const std::unordered_map<const VarNode*, PrimExpr>& vsub_;
85};
86
87PrimExpr InjectPredicate(const Array<PrimExpr>& predicates, PrimExpr body) {
88 using tir::ReduceNode;
89 using tir::SelectNode;
90 if (predicates.size() == 0) return body;
91 const ReduceNode* reduce = body.as<ReduceNode>();
92
93 if (reduce) {
94 auto n = make_object<ReduceNode>(*reduce);
95 n->condition = foldl([](PrimExpr a, PrimExpr b, Span span) { return logical_and(a, b, span); },
96 n->condition, predicates);
97 return PrimExpr(n);
98 }
99 return Select(foldl([](PrimExpr a, PrimExpr b, Span span) { return logical_and(a, b, span); },
100 const_true(1), predicates),
101 body, make_zero(body.dtype()));
102}

104// Replace data flow appears in all stages given the tensor change.
105// Also update vmap if subsequent dataflow need to be replaced.
106// Need to keep an update to the date transitive closure property on the vmap by a reverse map.
107void ReplaceDataFlow(const Array<Stage>& stages, std::unordered_map<Tensor, Tensor>* vmap,
108 std::unordered_map<Tensor, Tensor>* rvmap) {
109 for (Stage s : stages) {
110 Operation op = s->op->ReplaceInputs(s->op, *vmap);
111 if (!op.same_as(s->op)) {
112 for (int i = 0; i < op->num_outputs(); ++i) {
113 auto it = rvmap->find(s->op.output(i));
114 if (it != rvmap->end()) {
115 (*vmap)[it->second] = op.output(i);
116 } else {
117 (*vmap)[s->op.output(i)] = op.output(i);
118 (*rvmap)[op.output(i)] = s->op.output(i);
119 }
120 }
121 s->op = op;
122 }
123 }
124}

126inline bool ReduceEqual(const tir::ReduceNode* a, const tir::ReduceNode* b) {
127 return (a->combiner.same_as(b->combiner)) && (a->source.same_as(b->source)) &&
128 (a->axis.same_as(b->axis)) && (a->condition.same_as(b->condition)) &&
129 ((a->init.empty() && b->init.empty()) || (a->init.same_as(b->init)));
130}

132Tensor Schedule::cache_read(const Tensor& tensor, const std::string& scope,
133 const Array<Operation>& readers) {
134 (*this)->InvalidateCache();
135 // create identity mapping.
136 std::ostringstream os;
137 os << tensor->op->name;
138 if (tensor->op->num_outputs() != 1) {
139 os << ".v" << tensor->value_index;
140 }
141
142 // when a schedule has multiple cache_read on the same tensor,
143 // we make sure their op names are unique. e.g., w.shared, w_d.shared, w_d_d.shared
144 for (auto pair : (*this)->stage_map) {
145 auto stage = pair.second;
146 if (stage->op->name == os.str() + "." + scope) {
147 os << ".d";
148 }
149 }
150 os << "." << scope;
151
152 std::unordered_map<Tensor, Tensor> vsub;
153 Stage s = operator[](tensor->op);
154 Tensor sugar_tensor = s->op.output(tensor->value_index);
155 Tensor cache = compute(
156 sugar_tensor->shape,
157 [&sugar_tensor](const Array<Var>& i) {
158 return sugar_tensor(Array<PrimExpr>(i.begin(), i.end()));
159 },
160 os.str());
161 vsub[sugar_tensor] = cache;
162
163 std::unordered_map<Tensor, Tensor> vmap;
164 std::unordered_map<Tensor, Tensor> rvmap;
165 for (Operation op : readers) {
166 Stage s = operator[](op);
167 Operation repl_op = s->op->ReplaceInputs(s->op, vsub);
168 ICHECK(!repl_op.same_as(s->op)) << "Cannot find " << tensor << " in the inputs of " << s->op;
169 vmap[s->op.output(0)] = repl_op.output(0);
170 rvmap[repl_op.output(0)] = s->op.output(0);
171 s->op = repl_op;
172 }
173 ReplaceDataFlow((*this)->stages, &vmap, &rvmap);
174 Array<Stage>& stages = (*this)->stages;
175 Stage op_stage = operator[](tensor->op);
176 size_t pos = FindNodeRef(stages.GetArrayNode(), op_stage);
177 Stage cache_stage = Stage(cache->op);
178 cache_stage.set_scope(scope);
179 ICHECK_LT(pos, stages.size());
180 stages.insert(stages.begin() + pos + 1, cache_stage);
181 (*this)->stage_map.Set(cache->op, cache_stage);
182 // Update group
183 cache_stage->group = op_stage->group;
184 if (cache_stage->group.defined()) {
185 ++cache_stage->group->num_child_stages;
186 }
187 return cache;
188}

190template <typename OpType>
191void PrepareAxisMapping(Stage orig_stage, OpType* op, std::unordered_set<IterVar>* p_red_axis,
192 Array<IterVar>* p_new_axis, std::unordered_map<IterVar, Range>* p_dom_map,
193 std::unordered_map<const VarNode*, PrimExpr>* p_vsub,
194 std::unordered_map<const VarNode*, PrimExpr>* p_vsub2newvar,
195 std::vector<PrimExpr>* p_predicates) {
196 auto& red_axis = *p_red_axis;
197 auto& new_axis = *p_new_axis;
198 auto& dom_map = *p_dom_map;
199 auto& vsub = *p_vsub;
200 auto& vsub2newvar = *p_vsub2newvar;
201 auto& predicates = *p_predicates;
202 arith::Analyzer analyzer;
203
204 for (IterVar iv : op->reduce_axis) {
205 red_axis.insert(iv);
206 }
207 for (IterVar iv : op->axis) {
208 dom_map[iv] = iv->dom;
209 analyzer.Bind(iv->var, iv->dom);
210 }
211 te::PassDownDomain(orig_stage, &dom_map, &analyzer, true);
212 {
213 // The source->cache
214 std::unordered_map<IterVar, PrimExpr> value_map;
215 for (IterVar iv : orig_stage->leaf_iter_vars) {
216 if (red_axis.count(iv)) continue;
217 ICHECK_EQ(iv->iter_type, kDataPar) << "Can only relayout with in data parallel dimensions";
218 Range dom = dom_map.at(iv);
219 IterVar new_iv = IterVar(dom, iv->var.copy_with_suffix(".c"), iv->iter_type);
220 new_axis.push_back(new_iv);
221 if (is_one(dom->min)) {
222 value_map[iv] = dom->min;
223 } else {
224 value_map[iv] = iv->var;
225 vsub2newvar[iv->var.get()] = new_iv->var;
226 }
227 }
228 // skip reduction iteration.
229 std::unordered_set<IterVar> skip_bound_check;
230 for (IterVar iv : op->reduce_axis) {
231 skip_bound_check.insert(iv);
232 }
233 PassUpIndex(orig_stage, dom_map, &value_map, true);
234 predicates = MakeBoundCheck(orig_stage, dom_map, value_map, true, skip_bound_check);
235 // The root axis
236 for (IterVar iv : op->axis) {
237 if (value_map.count(iv)) {
238 vsub[iv->var.get()] = value_map.at(iv);
239 } // to handle tensor axis
240 }
241 }
242}

244Array<Tensor> ReplaceOriginalOp(Schedule sch, Stage orig_stage, const std::string& scope,
245 Operation cache_op, Operation orig_new_op, size_t tensor_size) {
246 Array<Tensor> cache_tensor_list;
247 for (size_t i = 0; i < tensor_size; i++) {
248 Tensor cache_tensor = cache_op.output(i);
249 cache_tensor_list.push_back(cache_tensor);
250 }
251 // The replace of the dataflow
252 std::unordered_map<Tensor, Tensor> vmap;
253 std::unordered_map<Tensor, Tensor> rvmap;
254 vmap[orig_stage->op.output(0)] = orig_new_op.output(0);
255 rvmap[orig_new_op.output(0)] = orig_stage->op.output(0);
256 for (size_t i = 0; i < tensor_size; i++) {
257 vmap[orig_stage->op.output(0)] = orig_new_op.output(0);
258 rvmap[orig_new_op.output(0)] = orig_stage->op.output(0);
259 }
260 ReplaceDataFlow(sch->stages, &vmap, &rvmap);
261 // mutate orig stage
262 orig_stage->op = orig_new_op;
263 orig_stage->all_iter_vars = orig_stage->op->root_iter_vars();
264 orig_stage->leaf_iter_vars = orig_stage->all_iter_vars;
265 orig_stage->relations = Array<IterVarRelation>();
266 // create schedule for new cached stage.
267 Array<Stage>& stages = sch->stages;
268 size_t pos = FindNodeRef(stages.GetArrayNode(), orig_stage);
269 Stage cache_stage = Stage(cache_op);
270 cache_stage.set_scope(scope);
271 ICHECK_LT(pos, stages.size());
272 stages.insert(stages.begin() + pos, cache_stage);
273 sch->stage_map.Set(cache_op, cache_stage);
274 // Update group
275 cache_stage->group = orig_stage->group;
276 if (cache_stage->group.defined()) {
277 ++cache_stage->group->num_child_stages;
278 }
279 return cache_tensor_list;
280}

282// Cache write and relayout the data according to loop pattern
283Array<Tensor> CacheWriteWithReLayout(Schedule sch, const Array<Tensor>& tensor_array,
284 const std::string& scope) {
285 size_t tensor_size = tensor_array.size();
286 sch->InvalidateCache();
287 Tensor tensor = tensor_array[0];
288 Stage orig_stage = sch[tensor->op];
289 const ComputeOpNode* compute = orig_stage->op.as<ComputeOpNode>();
290
291 std::unordered_set<IterVar> red_axis;
292 Array<IterVar> new_axis;
293 std::unordered_map<IterVar, Range> dom_map;
294
295 std::unordered_map<const VarNode*, PrimExpr> vsub;
296 std::unordered_map<const VarNode*, PrimExpr> vsub2newvar;
297 std::vector<PrimExpr> predicates;
298
299 PrepareAxisMapping(orig_stage, compute, &red_axis, &new_axis, &dom_map, &vsub, &vsub2newvar,
300 &predicates);
301
302 PrimExpr body;
303 Array<PrimExpr> body_list;
304 const tir::ReduceNode* first_reduce = nullptr;
305 for (auto cbody : compute->body) {
306 body = VarReplacer(vsub)(cbody);
307 body = InjectPredicate(predicates, body);
308 body = VarReplacer(vsub2newvar)(body);
309 // Reduce nodes in ONE computeOp must be the same except value_index
310 // This is right only if the original body ensures Reduce nodes are the same
311 if (body->IsInstance<tir::ReduceNode>()) {
312 const tir::ReduceNode* reduce_body = body.as<tir::ReduceNode>();
313 if (first_reduce != nullptr) {
314 ICHECK(ReduceEqual(reduce_body, first_reduce));
315 body = tir::Reduce(first_reduce->combiner, first_reduce->source, first_reduce->axis,
316 first_reduce->condition, reduce_body->value_index, reduce_body->init);
317 } else {
318 first_reduce = reduce_body;
319 }
320 } else {
321 ICHECK(first_reduce == nullptr) << "cannot mix reduce and other node in ONE compute bodys";
322 }
323 body_list.push_back(body);
324 }
325 // The reader args
326 Array<PrimExpr> args;
327 {
328 // cache->compute
329 std::unordered_map<IterVar, PrimExpr> value_map;
330 for (IterVar iv : compute->axis) {
331 value_map[iv] = iv->var;
332 }
333 te::PassDownIndex(orig_stage, dom_map, &value_map, true);
334 for (IterVar iv : orig_stage->leaf_iter_vars) {
335 if (red_axis.count(iv)) continue;
336 args.push_back(value_map.at(iv));
337 }
338 }
339 Operation cache_op =
340 ComputeOp(compute->name + "." + scope, compute->tag, compute->attrs, new_axis, body_list);
341
342 Array<PrimExpr> cache_expr_list;
343 for (size_t i = 0; i < tensor_size; i++) {
344 Tensor cache_tensor = cache_op.output(i);
345 cache_expr_list.push_back(cache_tensor(args));
346 }
347 Operation orig_new_op =
348 ComputeOp(compute->name, compute->tag, compute->attrs, compute->axis, cache_expr_list);
349 return ReplaceOriginalOp(sch, orig_stage, scope, cache_op, orig_new_op, tensor_size);
350}

352// for tensor compute op
353Array<Tensor> CacheWriteWithReLayoutTensor(Schedule sch, const Array<Tensor>& tensor_array,
354 const std::string& scope) {
355 size_t tensor_size = tensor_array.size();
356 sch->InvalidateCache();
357 Tensor tensor = tensor_array[0];
358 Stage orig_stage = sch[tensor->op];
359 const TensorComputeOpNode* tensor_op = orig_stage->op.as<TensorComputeOpNode>();
360 ICHECK_EQ(tensor_op->num_outputs(), 1)
361 << "cache write only support single output tensor_compute_op";
362
363 std::unordered_set<IterVar> red_axis;
364 Array<IterVar> new_axis;
365 std::unordered_map<IterVar, Range> dom_map;
366
367 std::unordered_map<const VarNode*, PrimExpr> vsub;
368 std::unordered_map<const VarNode*, PrimExpr> vsub2newvar;
369 std::vector<PrimExpr> predicates;
370
371 PrepareAxisMapping(orig_stage, tensor_op, &red_axis, &new_axis, &dom_map, &vsub, &vsub2newvar,
372 &predicates);
373
374 for (int i = tensor_op->schedulable_ndim; i < static_cast<int>(tensor_op->axis.size()); ++i) {
375 IterVar iv = tensor_op->axis[i];
376 IterVar new_iv = IterVar(iv->dom, iv->var.copy_with_suffix(".c"), iv->iter_type);
377 new_axis.push_back(new_iv);
378 }
379 Array<Region> new_regions;
380 for (Region old_region : tensor_op->input_regions) {
381 Region region;
382 for (Range r : old_region) {
383 PrimExpr min = VarReplacer(vsub2newvar)(r->min);
384 PrimExpr extent = VarReplacer(vsub2newvar)(r->extent);
385 region.push_back(Range::FromMinExtent(min, extent));
386 }
387 new_regions.push_back(region);
388 }
389
390 Array<PrimExpr> new_scalar_inputs;
391 for (PrimExpr old_input : tensor_op->scalar_inputs) {
392 new_scalar_inputs.push_back(VarReplacer(vsub2newvar)(old_input));
393 }
394
395 Operation cache_op =
396 TensorComputeOp(tensor_op->name + "." + scope, tensor_op->tag, new_axis,
397 tensor_op->reduce_axis, tensor_op->schedulable_ndim, tensor_op->intrin,
398 tensor_op->inputs, new_regions, new_scalar_inputs);
399
400 // axis will be used in generating compute op
401 Array<IterVar> compute_axis = tensor_op->axis;
402 for (size_t i = tensor_op->schedulable_ndim; i < tensor_op->axis.size(); ++i) {
403 IterVar iv = tensor_op->axis[i];
404 IterVar aiv = IterVar(iv->dom, iv->var, kDataPar);
405 compute_axis.Set(i, aiv);
406 }
407
408 // The reader args
409 Array<PrimExpr> args;
410 {
411 // cache->compute
412 std::unordered_map<IterVar, PrimExpr> value_map;
413 for (IterVar iv : compute_axis) {
414 value_map[iv] = iv->var;
415 }
416 PassDownIndex(orig_stage, dom_map, &value_map, true);
417 for (IterVar iv : orig_stage->leaf_iter_vars) {
418 if (red_axis.count(iv)) continue;
419 args.push_back(value_map.at(iv));
420 }
421 // tensorized region axis
422 for (size_t i = tensor_op->schedulable_ndim; i < tensor_op->axis.size(); ++i) {
423 IterVar iv = compute_axis[i];
424 args.push_back(value_map.at(iv));
425 }
426 }
427
428 Array<PrimExpr> cache_expr_list;
429 for (size_t i = 0; i < tensor_size; i++) {
430 Tensor cache_tensor = cache_op.output(i);
431 cache_expr_list.push_back(cache_tensor(args));
432 }
433 Operation orig_new_op =
434 ComputeOp(tensor_op->name, tensor_op->tag, {}, compute_axis, cache_expr_list);
435 return ReplaceOriginalOp(sch, orig_stage, scope, cache_op, orig_new_op, tensor_size);
436}

438Array<Tensor> Schedule::cache_write(const Array<Tensor>& tensor_array, const std::string& scope) {
439 (*this)->InvalidateCache();
440 ICHECK(tensor_array.size() > 0) << "size of tensor_array must be greater than 0";
441 Tensor tensor = tensor_array[0];
442 Stage orig_stage = operator[](tensor->op);
443 const ComputeOpNode* compute = tensor->op.as<ComputeOpNode>();
444 ICHECK(static_cast<size_t>(compute->num_outputs()) == tensor_array.size())
445 << "size of input tensor list must be same as number of stage outputs";
446 for (size_t i = 1; i < tensor_array.size(); i++) {
447 Stage tmp_stage = operator[](tensor_array[i]->op);
448 ICHECK(orig_stage.same_as(tmp_stage)) << "Input tensor list must be generated by ONE computeOp";
449 }
450 return CacheWriteWithReLayout(*this, tensor_array, scope);
451}
452
453Tensor Schedule::cache_write(const Tensor& tensor, const std::string& scope) {
454 // support original compute and tensor compute both
455 (*this)->InvalidateCache();
456 if (tensor->op.as<ComputeOpNode>()) {
457 return (CacheWriteWithReLayout(*this, {tensor}, scope))[0];
458 } else if (tensor->op.as<TensorComputeOpNode>()) {
459 return (CacheWriteWithReLayoutTensor(*this, {tensor}, scope))[0];
460 } else {
461 LOG(FATAL) << "cache write only take ComputeOp or TensorComputeOp as writers";
462 return Tensor();
463 }
464}

466void RebaseNonZeroMinLoop(ScheduleNode* sch) {
467 std::unordered_map<IterVar, IterVar> rebase_map;
468 for (Stage s : sch->stages) {
469 if (s->attach_type == kInlinedAlready) continue;
470
471 auto root_iter_vars = s->op->root_iter_vars();
472 ArrayNode* leaf_vars = s->leaf_iter_vars.CopyOnWrite();
473 for (IterVar iv : root_iter_vars) {
474 size_t idx = FindNodeRef(leaf_vars, iv);
475 auto it = s->iter_var_attrs.find(iv);
476 // don;t need to rebase path that are binded.
477 if (it != s->iter_var_attrs.end() && (*it).second->bind_thread.defined()) {
478 continue;
479 }
480 if (idx < leaf_vars->size()) {
481 // insert rebase
482 IterVar rebased = IterVar(Range(), iv->var.copy_with_suffix(""), iv->iter_type);
483 s->relations.push_back(te::Rebase(iv, rebased));
484 if (s->iter_var_attrs.count(iv)) {
485 s->iter_var_attrs.Set(rebased, s->iter_var_attrs.at(iv));
486 }
487 leaf_vars->SetItem(idx, rebased);
488 rebase_map[iv] = rebased;
489 }
490 }
491 }
492 // remap the parent relation
493 for (Stage s : sch->stages) {
494 if (s->attach_type != kScope) continue;
495 if (rebase_map.count(s->attach_ivar)) {
496 s->attach_ivar = rebase_map.at(s->attach_ivar);
497 }
498 }
499 for (Stage s : sch->groups) {
500 if (s->attach_type != kScope) continue;
501 if (rebase_map.count(s->attach_ivar)) {
502 s->attach_ivar = rebase_map.at(s->attach_ivar);
503 }
504 }
505}

507void InjectInline(ScheduleNode* sch, bool feature_extraction_mode) {
508 sch->InvalidateCache();
509
510 std::vector<Array<PrimExpr>> new_body(sch->stages.size());
511 std::vector<bool> changed(sch->stages.size(), false);
512 std::vector<Stmt> new_hybrid_body(sch->stages.size());
513 std::vector<bool> hybrid_changed(sch->stages.size(), false);
514 // (sshtin): this workaround allows to inline extern ops into their consumer.
515 // All inputs for extern op should not be inlined because inlining may happen
516 // before TE generation for particular extern op. That may lead to
517 // crash during lowering or building stages.
518 // The problem description:
519 // In case of operations fusing, arguments inlining
520 // prevents creation of ProducerNode for extern operation.
521 // Instead of the creation it is supposed to use operation argument as inlined buffer
522 // but extern_op TIR generation can be peformed after inlining procedure so
523 // newly generated TIR does not have reference to input data at all.
524 std::unordered_map<Operation, Operation> ext_ops;
525 for (size_t i = 0; i < sch->stages.size(); i++) {
526 Stage stage = sch->stages[i];
527 auto ext_op = stage->op.as<ExternOpNode>();
528 if (ext_op) {
529 auto inps = ext_op->InputTensors();
530 for (size_t ii = 0; ii < inps.size(); ++ii) {
531 if (ext_ops.find(inps[ii]->op) == ext_ops.end()) {
532 ext_ops[inps[ii]->op] = stage->op;
533 }
534 }
535 }
536 }
537 // inline all the ops
538 for (size_t i = sch->stages.size(); i != 0; --i) {
539 Stage stage = sch->stages[i - 1];
540 if (stage->attach_type == kInline) {
541 stage->attach_type = kInlinedAlready;
542 Array<Var> args;
543 PrimExpr body;
544 {
545 // setup args
546 const ComputeOpNode* compute = stage->op.as<ComputeOpNode>();
547 ICHECK(compute) << "can only inline compute op";
548 for (auto iv : compute->axis) {
549 args.push_back(iv->var);
550 }
551 if (ext_ops.find(stage->op) != ext_ops.end()) {
552 // sshtin: The extern op can try to get access to the input tensors as a raw data,
553 // that can lead to error in IR builder.
554 stage->attach_type = kGroupRoot;
555 continue;
556 }
557 ICHECK_EQ(compute->body.size(), 1U) << "can only inline compute op with 1 output";
558 if (feature_extraction_mode && compute->attrs.count("const_matrix")) {
559 // Use constant value to replace access of const matrices.
560 // This produces wrong IR but is good enough for feature extraction purposes.
561 // This simplification can accelerate the feature extration and evolutionary search.
562 body = make_const(compute->output_dtype(0), 1.0f);
563 } else {
564 body = compute->body[0];
565 }
566 }
567 for (size_t j = i; j < sch->stages.size(); ++j) {
568 Stage s = sch->stages[j];
569 const ComputeOpNode* compute = s->op.as<ComputeOpNode>();
570 const HybridOpNode* hybrid = s->op.as<HybridOpNode>();
571 if (compute) {
572 if (!new_body[j].size()) {
573 new_body[j] = compute->body;
574 }
575 if (new_body[j][0]->IsInstance<tir::ReduceNode>()) {
576 // specially handle reduction inline for multiplre reductions.
577 const tir::ReduceNode* reduce = new_body[j][0].as<tir::ReduceNode>();
578 for (size_t k = 1; k < new_body[j].size(); ++k) {
579 const tir::ReduceNode* reduce_ = new_body[j][k].as<tir::ReduceNode>();
580 ICHECK(reduce_);
581 ICHECK(ReduceEqual(reduce_, reduce)) << "The Reduce inputs of ComputeOp should "
582 << "have the same attribute except value_index";
583 }
584 PrimExpr new_value = Inline(tir::Evaluate(new_body[j][0]), stage->op, args, body)
585 .as<tir::EvaluateNode>()
586 ->value;
587 if (!new_value.same_as(new_body[j][0])) {
588 changed[j] = true;
589 const tir::ReduceNode* r = new_value.as<tir::ReduceNode>();
590 ICHECK(r != nullptr);
591 ICHECK_EQ(new_body[j].size(), r->source.size());
592 for (size_t k = 0; k < new_body[j].size(); ++k) {
593 auto n = make_object<tir::ReduceNode>(*r);
594 n->value_index = static_cast<int>(k);
595 n->dtype = r->source[k].dtype();
596 new_body[j].Set(k, PrimExpr(n));
597 }
598 }
599 } else {
600 for (size_t k = 0; k < new_body[j].size(); ++k) {
601 PrimExpr new_value = Inline(tir::Evaluate(new_body[j][k]), stage->op, args, body)
602 .as<tir::EvaluateNode>()
603 ->value;
604 if (!new_value.same_as(new_body[j][k])) {
605 new_body[j].Set(k, new_value);
606 changed[j] = true;
607 }
608 }
609 }
610 } else if (hybrid) {
611 if (!new_hybrid_body[j].defined()) {
612 new_hybrid_body[j] = hybrid->body;
613 }
614 Stmt new_stmt = Inline(new_hybrid_body[j], stage->op, args, body);
615 if (!new_stmt.same_as(new_hybrid_body[j])) {
616 new_hybrid_body[j] = new_stmt;
617 hybrid_changed[j] = true;
618 }
619 }
620 }
621 }
622 }
623 std::unordered_map<Tensor, Tensor> repl;
624 // rewrite dataflow
625 for (size_t i = 0; i < sch->stages.size(); ++i) {
626 Stage s = sch->stages[i];
627 if (s->attach_type == kInlinedAlready) continue;
628 if (new_body[i].size()) {
629 // Logics from ReplaceDataFlow
630 const ComputeOpNode* compute = sch->stages[i]->op.as<ComputeOpNode>();
631 ICHECK(compute);
632 Operation op = s->op;
633 if (changed[i]) {
634 op = ComputeOp(compute->name, compute->tag, compute->attrs, compute->axis, new_body[i]);
635 }
636 op = op->ReplaceInputs(op, repl);
637 if (!op.same_as(s->op)) {
638 for (int idx = 0; idx < s->op->num_outputs(); ++idx) {
639 repl[s->op.output(idx)] = op.output(idx);
640 }
641 s->op = op;
642 }
643 } else if (hybrid_changed[i]) {
644 const HybridOpNode* hybrid = sch->stages[i]->op.as<HybridOpNode>();
645 ICHECK(hybrid);
646 Operation op = HybridOp(hybrid->name, hybrid->tag, hybrid->attrs, hybrid->inputs,
647 hybrid->outputs, new_hybrid_body[i]);
648 op = op->ReplaceInputs(op, repl);
649 for (int idx = 0; idx < s->op->num_outputs(); ++idx) {
650 repl[s->op.output(idx)] = op.output(idx);
651 }
652 s->op = op;
653 } else {
654 Operation op = s->op->ReplaceInputs(s->op, repl);
655 if (!op.same_as(s->op)) {
656 for (int j = 0; j < op->num_outputs(); ++j) {
657 repl[s->op.output(j)] = op.output(j);
658 }
659 s->op = op;
660 }
661 }
662 }
663}

665void LegalizeInvalidAttach(ScheduleNode* sch) {
666 // Legalize the compute_at location if the target iterator of compute_at is split or fused.
667 // Case 1: If the target of compute_at is split,
668 // we will move the compute_at location to the inner iterator.
669 // Case 2: If the target of compute_at is fused,
670 // we will move the compute_at location to the newly fused iterator.
671 // Note that case 2 can only happen if the target of compute_at
672 // is the innermost operand of fuse operation.
673
674 // Map an old invalid attach point to its new valid attach point
675 std::unordered_map<IterVar, IterVar> replace_map;
676
677 for (Stage stage : sch->stages) {
678 std::unordered_set<const Object*> visited;
679 for (Stage s = stage; s.defined();) {
680 // The following logic is simiar to the `CreateAttachPath` in `src/te/schedule/graph.h`,
681 // because we follow the validation check in that function to legalize the attach.
682 ICHECK(!visited.count(s.get())) << "Find loop in compute_at attach group";
683 visited.insert(s.get());
684 Stage spec = s.GetAttachSpec();
685 if (spec->attach_type != kScope) {
686 break;
687 }
688 bool start_attach = false;
689 IterVar attach_ivar = spec->attach_ivar;
690 s = spec->attach_stage;
691 ICHECK(attach_ivar.defined());
692 ICHECK(s.defined());
693
694 for (size_t i = s->leaf_iter_vars.size(); i != 0; --i) {
695 IterVar iv = s->leaf_iter_vars[i - 1];
696 if (!start_attach && iv.same_as(attach_ivar)) {
697 start_attach = true;
698 break;
699 }
700 }
701
702 if (!start_attach) {
703 IterVar new_attach_ivar = attach_ivar;
704 bool updated = true;
705 // recursively update the relations
706 while (updated) {
707 updated = false;
708 for (const auto& rel : s->relations) {
709 if (const FuseNode* r = rel.as<FuseNode>()) {
710 if (new_attach_ivar.same_as(r->inner)) {
711 new_attach_ivar = r->fused;
712 updated = true;
713 }
714 } else if (const SplitNode* r = rel.as<SplitNode>()) {
715 if (new_attach_ivar.same_as(r->parent)) {
716 new_attach_ivar = r->inner;
717 updated = true;
718 }
719 }
720 }
721 replace_map[attach_ivar] = new_attach_ivar;
722 }
723 }
724 }
725 }
726
727 // remap the parent relation
728 for (Stage s : sch->stages) {
729 if (s->attach_type != kScope) continue;
730 if (replace_map.count(s->attach_ivar)) {
731 s->attach_ivar = replace_map.at(s->attach_ivar);
732 }
733 }
734 for (Stage s : sch->groups) {
735 if (s->attach_type != kScope) continue;
736 if (replace_map.count(s->attach_ivar)) {
737 s->attach_ivar = replace_map.at(s->attach_ivar);
738 }
739 }
740}

742Schedule Schedule::normalize() {
743 Schedule sn = copy();
744 InjectInline(sn.operator->(), false);
745 RebaseNonZeroMinLoop(sn.operator->());
746 LegalizeInvalidAttach(sn.operator->());
747 return sn;
748}

750Schedule Schedule::normalize_for_feature_extraction() {
751 Schedule sn = copy();
752 InjectInline(sn.operator->(), true);
753 RebaseNonZeroMinLoop(sn.operator->());
754 LegalizeInvalidAttach(sn.operator->());
755 return sn;
756}

758// Handle reduction factor.
759Array<Tensor> Schedule::rfactor(const Tensor& tensor, const IterVar& axis, int factor_axis) {
760 (*this)->InvalidateCache();
761 using tir::ReduceNode;
762 ICHECK_EQ(axis->iter_type, kCommReduce) << "Can only factor reduction axis";
763 Stage reduce_stage = operator[](tensor->op);
764 const ComputeOpNode* compute_op = reduce_stage->op.as<ComputeOpNode>();
765 ICHECK(compute_op) << "Can only factor ComputeOp";
766 ArrayNode* leaf_vars = reduce_stage->leaf_iter_vars.CopyOnWrite();
767 {
768 size_t axis_pos = FindNodeRef(leaf_vars, axis);
769 ICHECK_NE(axis_pos, leaf_vars->size())
770 << "Cannot find IterVar " << axis << " in leaf iter vars";
771 }
772 // Find touched reduction axis.
773 std::unordered_map<IterVar, int> touch_map;
774 touch_map[axis] = 1;
775 te::PassUpBitMaskOr(reduce_stage, &touch_map, true);
776 te::PassDownBitMaskOr(reduce_stage, &touch_map, true);
777 // skip reduction iteration.
778 std::unordered_set<IterVar> skip_bound_check;
779 // Verify normal axis are not touched.
780 for (IterVar iv : compute_op->axis) {
781 ICHECK(!touch_map.count(iv)) << "Factor axis touches normal axis.";
782 skip_bound_check.insert(iv);
783 }
784 // get analyzer.
785 arith::Analyzer analyzer;
786 // Get the replace index
787 std::unordered_map<IterVar, Range> dom_map;
788 std::unordered_map<IterVar, PrimExpr> value_map;
789 for (IterVar iv : compute_op->reduce_axis) {
790 if (touch_map.count(iv)) {
791 dom_map[iv] = iv->dom;
792 } else {
793 skip_bound_check.insert(iv);
794 }
795 analyzer.Bind(iv->var, iv->dom);
796 }
797 te::PassDownDomain(reduce_stage, &dom_map, &analyzer, true);
798 for (IterVar iv : reduce_stage->leaf_iter_vars) {
799 if (touch_map.count(iv)) {
800 Range dom = dom_map.at(iv);
801 if (is_one(dom->extent)) {
802 value_map[iv] = dom->min;
803 } else {
804 value_map[iv] = iv->var;
805 }
806 }
807 }
808 te::PassUpIndex(reduce_stage, dom_map, &value_map, true);
809 std::vector<PrimExpr> predicates =
810 MakeBoundCheck(reduce_stage, dom_map, value_map, true, skip_bound_check);
811
812 // Get the factored op node.
813 const int factor_axis_pos =
814 factor_axis >= 0 ? factor_axis : static_cast<int>(compute_op->axis.size() + 1) + factor_axis;
815 ICHECK_LE(factor_axis_pos, compute_op->axis.size());
816 auto n = make_object<ComputeOpNode>();
817 n->name = compute_op->name + ".rf";
818 {
819 // axis relacement.
820 IterVar iv(dom_map.at(axis), axis->var, kDataPar);
821 ICHECK(is_zero(iv->dom->min)) << "Can only factor reduction domain starting from 0";
822
823 const int size = compute_op->axis.size();
824 for (int idx = 0; idx < size; ++idx) {
825 if (factor_axis_pos == idx) {
826 n->axis.push_back(iv);
827 }
828 n->axis.push_back(compute_op->axis[idx]);
829 }
830 if (factor_axis_pos == size) {
831 n->axis.push_back(iv);
832 }
833 }
834 // predicate generation, copy not touched axis.
835 int idx = tensor->value_index;
836 const ReduceNode* reduce = compute_op->body[idx].as<ReduceNode>();
837 ICHECK(reduce) << "Can only rfactor non-inline reductions";
838 predicates.push_back(reduce->condition);
839
840 PrimExpr predicate =
841 likely(foldl([](PrimExpr a, PrimExpr b, Span span) { return logical_and(a, b, span); },
842 const_true(1), predicates));
843
844 std::unordered_map<const VarNode*, PrimExpr> vsub;
845
846 for (IterVar iv : compute_op->reduce_axis) {
847 if (!touch_map.count(iv)) {
848 n->reduce_axis.push_back(iv);
849 } else {
850 ICHECK(value_map.count(iv));
851 PrimExpr index = value_map.at(iv);
852 vsub[iv->var.get()] = index;
853 }
854 }
855
856 // Copy touched axis.
857 for (IterVar iv : reduce_stage->leaf_iter_vars) {
858 if (touch_map.count(iv) && !iv.same_as(axis)) {
859 ICHECK_EQ(iv->iter_type, kCommReduce);
860 IterVar ncpy(dom_map.at(iv), iv->var, iv->iter_type, iv->thread_tag, iv->span);
861 n->reduce_axis.push_back(ncpy);
862 }
863 }
864 VarReplacer replacer(vsub);
865 Array<PrimExpr> new_source =
866 tir::UpdateArray(reduce->source, [&replacer](const PrimExpr& e) { return replacer(e); });
867
868 PrimExpr new_pred = replacer(predicate);
869
870 std::vector<PrimExpr> body;
871 for (size_t idx = 0; idx < reduce->source.size(); ++idx) {
872 body.emplace_back(Reduce(reduce->combiner, new_source, n->reduce_axis, new_pred, idx, {}));
873 }
874 n->body = Array<PrimExpr>(body);
875 // refresh relations, keep the un-touched relations.
876 Array<IterVarRelation> rels;
877 for (IterVarRelation rel : reduce_stage->relations) {
878 bool touched = false;
879 if (const SplitNode* r = rel.as<SplitNode>()) {
880 if (touch_map.count(r->parent)) touched = true;
881 } else if (const FuseNode* r = rel.as<FuseNode>()) {
882 if (touch_map.count(r->fused)) touched = true;
883 } else if (const RebaseNode* r = rel.as<RebaseNode>()) {
884 if (touch_map.count(r->parent)) touched = true;
885 } else {
886 LOG(FATAL) << "unknown relation type";
887 }
888 if (!touched) {
889 rels.push_back(rel);
890 }
891 }
892 // initialize the factored stage.
893 Operation factor_op(n);
894 Array<Stage>& stages = (*this)->stages;
895 size_t stage_pos = FindNodeRef(stages.GetArrayNode(), reduce_stage);
896 Stage factor_stage = Stage(factor_op);
897 factor_stage->relations = rels;
898 ICHECK_LT(stage_pos, stages.size());
899 stages.insert(stages.begin() + stage_pos, factor_stage);
900 (*this)->stage_map.Set(factor_op, factor_stage);
901 factor_stage->group = reduce_stage->group;
902 if (factor_stage->group.defined()) {
903 ++factor_stage->group->num_child_stages;
904 }
905 // Replace the old reduction.
906 IterVar repl_red_axis = reduce_axis(dom_map.at(axis), axis->var->name_hint + ".v");
907 Array<Tensor> factor_tensors;
908 Array<Tensor> old_tensors;
909 int size = factor_op->num_outputs();
910 for (int idx = 0; idx < size; ++idx) {
911 factor_tensors.push_back(factor_op.output(idx));
912 old_tensors.push_back(reduce_stage->op.output(idx));
913 }
914 Array<Tensor> repl_tensors = compute(
915 old_tensors[0]->shape,
916 [&](const Array<Var>& i) {
917 Array<PrimExpr> indices;
918 const int idx_size = static_cast<int>(i.size());
919 for (int idx = 0; idx < idx_size; ++idx) {
920 if (factor_axis_pos == idx) {
921 indices.push_back(repl_red_axis->var);
922 }
923 indices.push_back(i[idx]);
924 }
925 Array<PrimExpr> new_init = reduce->init;
926 if (!reduce->init.empty()) {
927 std::unordered_map<const VarNode*, PrimExpr> init_vsub;
928 for (const auto& init : reduce->init) {
929 if (init->IsInstance<ProducerLoadNode>()) {
930 ICHECK_EQ(compute_op->axis.size(), idx_size)
931 << "'init' should have the number of dimensions as output when using with "
932 "rfactor";
933 for (int idx = 0; idx < idx_size; idx++) {
934 init_vsub[compute_op->axis[idx]->var.get()] = i[idx];
935 }
936 }
937 }
938 VarReplacer init_replacer(init_vsub);
939 new_init = tir::UpdateArray(
940 reduce->init, [&init_replacer](const PrimExpr& e) { return init_replacer(e); });
941 }
942 if (factor_axis_pos == idx_size) {
943 indices.push_back(repl_red_axis->var);
944 }
945 Array<PrimExpr> factor_exprs;
946 for (int idx = 0; idx < size; ++idx) {
947 factor_exprs.push_back(factor_tensors[idx](indices));
948 }
949 Array<PrimExpr> reductions;
950 Array<IterVar> axis = {repl_red_axis};
951 PrimExpr cond = const_true();
952 for (int idx = 0; idx < size; ++idx) {
953 reductions.push_back(Reduce(reduce->combiner, factor_exprs, axis, cond, idx, new_init));
954 }
955 return reductions;
956 },
957 reduce_stage->op->name + ".repl");
958
959 std::unordered_map<Tensor, Tensor> vmap;
960 std::unordered_map<Tensor, Tensor> rvmap;
961 for (int idx = 0; idx < size; ++idx) {
962 vmap[old_tensors[idx]] = repl_tensors[idx];
963 rvmap[repl_tensors[idx]] = old_tensors[idx];
964 }
965 ReplaceDataFlow((*this)->stages, &vmap, &rvmap);
966 // revamp the reduction stage.
967 reduce_stage->op = repl_tensors[0]->op;
968 reduce_stage->all_iter_vars = repl_tensors[0]->op->root_iter_vars();
969 reduce_stage->leaf_iter_vars = reduce_stage->all_iter_vars;
970 reduce_stage->relations = Array<IterVarRelation>();
971 return factor_tensors;
972}
