20/*!
21 * \file verify_compact_buffer.cc
22 * \brief Verify if there was any compact buffer bound to a statement.
23 */
24#include <tvm/runtime/registry.h>
25#include <tvm/te/schedule_pass.h>
26#include <tvm/te/tensor.h>
27#include <tvm/tir/buffer.h>
28#include <tvm/tir/expr.h>
29#include <tvm/tir/stmt.h>
30#include <tvm/tir/stmt_functor.h>
31
32#include <unordered_map>
33
34namespace tvm {
35namespace te {
36
37class VerifyBuffer : public StmtVisitor {
38 public:
39 bool Verify(const Stmt& stmt) {
40 this->VisitStmt(stmt);
41 return is_compact_;
42 }
43
44 void VisitStmt_(const AttrStmtNode* op) final {
45 StmtVisitor::VisitStmt_(op);
46 if (op->attr_key == tir::attr::buffer_bind_scope) {
47 is_compact_ = true;
48 }
49 }
50
51 private:
52 bool is_compact_{false};
53};
54
55bool VerifyCompactBuffer(const Stmt& stmt) {
56 VerifyBuffer verifier;
57 return verifier.Verify(stmt);
58}
59
60TVM_REGISTER_GLOBAL("schedule.VerifyCompactBuffer").set_body_typed(VerifyCompactBuffer);
