20/*!
21 *
22 * \file support/hexdump.h
23 * \brief Print hex representation of BLOBs.
24 */
25#ifndef TVM_SUPPORT_HEXDUMP_H_
26#define TVM_SUPPORT_HEXDUMP_H_
27
28#include <iomanip>
29#include <sstream>
30#include <string>
31
32namespace tvm {
33namespace support {
34
35/*! \brief generate a hexdump of some binary data.
36 * \param s Binary data to print.
37 * \param os stream that receives the hexdump.
38 */
39void HexDump(const std::string& s, std::ostream& os);
40
41/*! \brief return a string containing a hexdump of the data in s */
42inline std::string HexDump(const std::string& s) {
43 std::stringstream ss;
44 HexDump(s, ss);
45 return ss.str();
46}
47
48} // namespace support
49} // namespace tvm
50#endif // TVM_SUPPORT_HEXDUMP_H_
