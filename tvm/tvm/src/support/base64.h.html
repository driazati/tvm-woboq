57/*!
58 * \brief Buffer reader from stream to avoid
59 * virtual call overhead on each read.
60 */
61class StreamBufferReader {
62 public:
63 explicit StreamBufferReader(size_t buffer_size) { buffer_.resize(buffer_size); }
64 /*!
65 * \brief set input stream
66 * \param stream The stream to be set
67 */
68 void set_stream(dmlc::Stream* stream) {
69 stream_ = stream;
70 read_len_ = read_ptr_ = 1;
71 }
72 /*!
73 * \return allows quick read using get char
74 */
75 int GetChar() {
76 while (true) {
77 if (read_ptr_ < read_len_) {
78 return static_cast<int>(buffer_[read_ptr_++]);
79 } else {
80 read_len_ = stream_->Read(&buffer_[0], buffer_.length());
81 if (read_len_ == 0) return EOF;
82 read_ptr_ = 0;
83 }
84 }
85 }
86 /*! \return whether we are reaching the end of file */
87 bool AtEnd() const { return read_len_ == 0; }
88
89 private:
90 /*! \brief the underlying stream */
91 dmlc::Stream* stream_{nullptr};
92 /*! \brief buffer to hold data */
93 std::string buffer_;
94 /*! \brief length of valid data in buffer */
95 size_t read_len_{1};
96 /*! \brief pointer in the buffer */
97 size_t read_ptr_{1};
98};

100/*!
101 * \brief Input stream from base64 encoding
102 */
103class Base64InStream : public dmlc::Stream {
104 public:
105 explicit Base64InStream(dmlc::Stream* fs) : reader_(256) { reader_.set_stream(fs); }
106 /*!
107 * \brief initialize the stream position to beginning of next base64 stream
108 * \note call this function before actually start read
109 */
110 void InitPosition(void) {
111 // get a character
112 do {
113 temp_ch_ = reader_.GetChar();
114 } while (isspace(temp_ch_));
115 }
116 /*! \brief whether current position is end of a base64 stream */
117 bool IsEOF(void) const { return num_prev_ == 0 && (temp_ch_ == EOF || isspace(temp_ch_)); }
118 // override read function.
119 virtual size_t Read(void* ptr, size_t size) {
120 using base64::DecodeTable;
121 if (size == 0) return 0;
122 // use tlen to record left size
123 size_t tlen = size;
124 unsigned char* cptr = static_cast<unsigned char*>(ptr);
125 // if anything left, load from previous buffered result
126 if (num_prev_ != 0) {
127 if (num_prev_ == 2) {
128 if (tlen >= 2) {
129 *cptr++ = buf_prev[0];
130 *cptr++ = buf_prev[1];
131 tlen -= 2;
132 num_prev_ = 0;
133 } else {
134 // assert tlen == 1
135 *cptr++ = buf_prev[0];
136 --tlen;
137 buf_prev[0] = buf_prev[1];
138 num_prev_ = 1;
139 }
140 } else {
141 // assert num_prev_ == 1
142 *cptr++ = buf_prev[0];
143 --tlen;
144 num_prev_ = 0;
145 }
146 }
147 if (tlen == 0) return size;
148 int nvalue;
149 // note: everything goes with 4 bytes in Base64
150 // so we process 4 bytes a unit
151 while (tlen && temp_ch_ != EOF && !isspace(temp_ch_)) {
152 // first byte
153 nvalue = DecodeTable[temp_ch_] << 18;
154 {
155 // second byte
156 temp_ch_ = reader_.GetChar();
157 ICHECK(temp_ch_ != EOF && !isspace(temp_ch_)) << "invalid base64 format";
158 nvalue |= DecodeTable[temp_ch_] << 12;
159 *cptr++ = (nvalue >> 16) & 0xFF;
160 --tlen;
161 }
162 {
163 // third byte
164 temp_ch_ = reader_.GetChar();
165 ICHECK(temp_ch_ != EOF && !isspace(temp_ch_)) << "invalid base64 format";
166 // handle termination
167 if (temp_ch_ == '=') {
168 temp_ch_ = reader_.GetChar();
169 ICHECK(temp_ch_ == '=') << "invalid base64 format";
170 temp_ch_ = reader_.GetChar();
171 ICHECK(temp_ch_ == EOF || isspace(temp_ch_)) << "invalid base64 format";
172 break;
173 }
174 nvalue |= DecodeTable[temp_ch_] << 6;
175 if (tlen) {
176 *cptr++ = (nvalue >> 8) & 0xFF;
177 --tlen;
178 } else {
179 buf_prev[num_prev_++] = (nvalue >> 8) & 0xFF;
180 }
181 }
182 {
183 // fourth byte
184 temp_ch_ = reader_.GetChar();
185 ICHECK(temp_ch_ != EOF && !isspace(temp_ch_)) << "invalid base64 format";
186 if (temp_ch_ == '=') {
187 temp_ch_ = reader_.GetChar();
188 ICHECK(temp_ch_ == EOF || isspace(temp_ch_)) << "invalid base64 format";
189 break;
190 }
191 nvalue |= DecodeTable[temp_ch_];
192 if (tlen) {
193 *cptr++ = nvalue & 0xFF;
194 --tlen;
195 } else {
196 buf_prev[num_prev_++] = nvalue & 0xFF;
197 }
198 }
199 // get next char
200 temp_ch_ = reader_.GetChar();
201 }
202 if (kStrictCheck) {
203 ICHECK_EQ(tlen, 0) << "Base64InStream: read incomplete";
204 }
205 return size - tlen;
206 }
207 virtual void Write(const void* ptr, size_t size) {
208 LOG(FATAL) << "Base64InStream do not support write";
209 }
210
211 private:
212 // internal reader
213 StreamBufferReader reader_;
214 int temp_ch_{0};
215 int num_prev_{0};
216 unsigned char buf_prev[2];
217 // whether we need to do strict check
218 static const bool kStrictCheck = false;
219};

221/*!
222 * \brief Stream to write to base64 format.
223 */
224class Base64OutStream : public dmlc::Stream {
225 public:
226 explicit Base64OutStream(dmlc::Stream* fp) : fp_(fp) {}
227 virtual void Write(const void* ptr, size_t size) {
228 using base64::EncodeTable;
229 size_t tlen = size;
230 const unsigned char* cptr = static_cast<const unsigned char*>(ptr);
231 while (tlen) {
232 while (buf__top_ < 3 && tlen != 0) {
233 buf_[++buf__top_] = *cptr++;
234 --tlen;
235 }
236 if (buf__top_ == 3) {
237 // flush 4 bytes out
238 PutChar(EncodeTable[buf_[1] >> 2]);
239 PutChar(EncodeTable[((buf_[1] << 4) | (buf_[2] >> 4)) & 0x3F]);
240 PutChar(EncodeTable[((buf_[2] << 2) | (buf_[3] >> 6)) & 0x3F]);
241 PutChar(EncodeTable[buf_[3] & 0x3F]);
242 buf__top_ = 0;
243 }
244 }
245 }
246 virtual size_t Read(void* ptr, size_t size) {
247 LOG(FATAL) << "Base64OutStream do not support read";
248 return 0;
249 }
250 /*!
251 * \brief finish writing of all current base64 stream, do some post processing
252 * \param endch character to put to end of stream, if it is EOF, then nothing will be appended.
253 */
254 void Finish(int endch = EOF) {
255 using base64::EncodeTable;
256 if (buf__top_ == 1) {
257 PutChar(EncodeTable[buf_[1] >> 2]);
258 PutChar(EncodeTable[(buf_[1] << 4) & 0x3F]);
259 PutChar('=');
260 PutChar('=');
261 }
262 if (buf__top_ == 2) {
263 PutChar(EncodeTable[buf_[1] >> 2]);
264 PutChar(EncodeTable[((buf_[1] << 4) | (buf_[2] >> 4)) & 0x3F]);
265 PutChar(EncodeTable[(buf_[2] << 2) & 0x3F]);
266 PutChar('=');
267 }
268 buf__top_ = 0;
269 if (endch != EOF) PutChar(endch);
270 this->Flush();
271 }
272
273 private:
274 static constexpr size_t kBufferSize = 256;
275
276 dmlc::Stream* fp_{nullptr};
277 int buf__top_{0};
278 unsigned char buf_[4];
279 std::string out_buf_;
280
281 void PutChar(char ch) {
282 out_buf_ += ch;
283 if (out_buf_.length() >= kBufferSize) Flush();
284 }
285 void Flush(void) {
286 if (out_buf_.length() != 0) {
287 fp_->Write(&out_buf_[0], out_buf_.length());
288 out_buf_.clear();
289 }
290 }
291};
