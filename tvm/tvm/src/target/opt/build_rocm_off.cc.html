20/*!
21 * Optional module when build rocm is switched to off
22 */
23#include "../../runtime/rocm/rocm_module.h"
24#include "../source/codegen_source_base.h"
25
26namespace tvm {
27namespace runtime {
28
29Module ROCMModuleCreate(std::string data, std::string fmt,
30 std::unordered_map<std::string, FunctionInfo> fmap, std::string rocm_source,
31 std::string assembly) {
32 LOG(WARNING) << "ROCM runtime is not enabled, return a source module...";
33 auto fget_source = [rocm_source, assembly](const std::string& format) {
34 if (format.length() == 0) return assembly;
35 if (format == "ll" || format == "llvm") return rocm_source;
36 if (format == "asm") return assembly;
37 return std::string("");
38 };
39 return codegen::DeviceSourceModuleCreate(data, fmt, fmap, "hsaco", fget_source);
40}
