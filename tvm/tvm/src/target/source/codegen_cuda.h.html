20/*!
21 * \file codegen_cuda.h
22 * \brief Utility to generate cuda code
23 */
24#ifndef TVM_TARGET_SOURCE_CODEGEN_CUDA_H_
25#define TVM_TARGET_SOURCE_CODEGEN_CUDA_H_
26
27#include <tvm/target/codegen.h>
28#include <tvm/tir/expr.h>
29#include <tvm/tir/op.h>
30
31#include <string>
32#include <unordered_map>
33
34#include "codegen_c.h"
35
36namespace tvm {
37namespace codegen {
38
39class CodeGenCUDA final : public CodeGenC {
40 public:
41 CodeGenCUDA();
42 void Init(bool output_ssa);
43 std::string Finish();
44 bool need_include_path() {
45 return (enable_fp16_ || enable_bf16_ || enable_int8_ || need_math_constants_h_ || need_mma_h_);
46 }
47 // override behavior
48 void PrintFuncPrefix() final;
49 void PrintExtraAttrs(const PrimFunc& f) final;
50 void VisitStmt_(const ForNode* op) final;
51 void PrintStorageSync(const CallNode* op) final;
52 void PrintStorageScope(const std::string& scope, std::ostream& os) final; // NOLINT(*)
53 void PrintVecBinaryOp(const std::string& op, DataType t, PrimExpr lhs, PrimExpr rhs,
54 std::ostream& os) final; // NOLINT(*)
55 void PrintType(DataType t, std::ostream& os) final; // NOLINT(*)
56 void PrintVecElemLoad(const std::string& vec, DataType t, int i,
57 std::ostream& os) final; // NOLINT(*)
58 void PrintVecElemStore(const std::string& vec, DataType t, int i, const std::string& value) final;
59 void BindThreadIndex(const IterVar& iv) final; // NOLINT(*)
60 void PrintVecElemLoadExpr(DataType t, int i, const std::string& value, std::ostream& os) final;
61 // overload visitor
62 void VisitExpr_(const RampNode* op, std::ostream& os) final; // NOLINT(*)
63 void VisitExpr_(const ShuffleNode* op, std::ostream& os) final; // NOLINT(*)
64 void VisitExpr_(const SelectNode* op, std::ostream& os) final; // NOLINT(*)
65 void VisitExpr_(const BroadcastNode* op, std::ostream& os) final; // NOLINT(*)
66 void VisitExpr_(const FloatImmNode* op, std::ostream& os) final;
67 void VisitExpr_(const CallNode* op, std::ostream& os) final;
68 void VisitExpr_(const CastNode* op, std::ostream& os) final;
69 void VisitStmt_(const EvaluateNode* op) final;
70 void VisitStmt_(const AllocateNode* op) final;
71 void VisitStmt_(const AttrStmtNode* op) final;
72
73 protected:
74 void PrintCallExtern(Type ret_type, String global_symbol, const Array<PrimExpr>& args,
75 bool skip_first_arg, std::ostream& os) final; // NOLINT(*)
76
77 private:
78 // Handle volatile loads
79 void HandleVolatileLoads(const std::string& value, const BufferLoadNode* op,
80 std::ostream& os) final;
81
82 // Whether scope such as "__shared__" or "__constant__" is part of type.
83 bool IsScopePartOfType() const final { return false; }
84
85 // Whether global barrier is needed.
86 bool need_global_barrier_{false};
87 // Global barrier state
88 std::string vid_global_barrier_state_;
89 // Global barrier expected node.
90 std::string vid_global_barrier_expect_;
91 // whether enable fp16
92 bool enable_fp16_{false};
93 // whether enable bf16
94 bool enable_bf16_{false};
95 // whether enable int8
96 bool enable_int8_{false};
97 // whether enable warp shuffle intrinsics
98 bool enable_warp_shuffle_{false};
99 // whether need math_constants.h
100 bool need_math_constants_h_{false};
101 // whether need mma.h
102 bool need_mma_h_{false};
103 // Op attribute map
104 OpAttrMap<bool> op_need_warp_shuffle_ = Op::GetAttrMap<bool>("cuda.need_warp_shuffle");
105
106 std::unordered_map<const VarNode*, std::string> fragment_shapes;
107 std::unordered_map<const VarNode*, std::string> fragment_layouts;
108 friend void PrintConst(const FloatImmNode* op, std::ostream& os, CodeGenCUDA* p);
109 void PrintWmmaScope(const std::string& scope, DataType t, const VarNode* variable,
110 std::ostream& os);
111 int32_t GetWmmaFragmentSize(const std::string& scope, const VarNode* variable, int32_t size);
112};
113
114} // namespace codegen
115} // namespace tvm
116
117#endif // TVM_TARGET_SOURCE_CODEGEN_CUDA_H_
