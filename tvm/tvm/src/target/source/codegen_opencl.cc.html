37class InferTextureAccess : public StmtExprVisitor {
38 public:
39 static constexpr const uint8_t kReadAccess = 1;
40 static constexpr const uint8_t kWriteAccess = 2;
41
42 InferTextureAccess() {}
43 std::unordered_map<const VarNode*, std::string> Infer(const Stmt& n) {
44 StmtExprVisitor::VisitStmt(n);
45 std::unordered_map<const VarNode*, std::string> storage_scope_qualifiers;
46 for (auto& texture : var_access_map_) {
47 if (texture.second == kReadAccess) {
48 storage_scope_qualifiers.insert({texture.first, "texture_read"});
49 } else if (texture.second == kWriteAccess) {
50 storage_scope_qualifiers.insert({texture.first, "texture_write"});
51 } else if (texture.second == (kReadAccess | kWriteAccess)) {
52 storage_scope_qualifiers.insert({texture.first, ""});
53 }
54 }
55 return storage_scope_qualifiers;
56 }
57 void VisitExpr_(const CallNode* op) {
58 if (op->op.same_as(builtin::texture2d_load())) {
59 var_access_map_[op->args[0].as<VarNode>()] |= kReadAccess;
60 } else if (op->op.same_as(builtin::texture2d_store())) {
61 var_access_map_[op->args[0].as<VarNode>()] |= kWriteAccess;
62 }
63 StmtExprVisitor::VisitExpr_(op);
64 }
65
66 private:
67 std::unordered_map<const VarNode*, uint8_t> var_access_map_;
68};

75void CodeGenOpenCL::InitFuncState(const PrimFunc& f) {
76 CodeGenC::InitFuncState(f);
77 this->SetTextureScope(InferTextureAccess().Infer(f->body));
78 for (Var arg : f->params) {
79 auto ptr_type = arg->type_annotation.as<PointerTypeNode>();
80 if (ptr_type && runtime::IsTextureStorage(std::string(ptr_type->storage_scope))) {
81 // Storage scope qualifiers for textures are inferred
82 // and set prior to function codegen.
83 continue;
84 } else if (arg.dtype().is_handle()) {
85 alloc_storage_scope_[arg.get()] = "global";
86 }
87 }

92std::string CodeGenOpenCL::Finish() {
93 // inject extension enable pragma for fp16 and fp64
94 if (enable_fp16_) {
95 decl_stream << "#ifdef cl_khr_fp16\n"
96 "#pragma OPENCL EXTENSION cl_khr_fp16 : enable\n"
97 "#elif defined(cl_amd_fp16)\n"
98 "#pragma OPENCL EXTENSION cl_amd_fp16 : enable\n"
99 "#else\n"
100 "#error \"Half precision floating point not supported"
101 " by OpenCL implementation on your device.\" \n"
102 "#endif\n\n";
103 }
104
105 if (enable_fp64_) {
106 decl_stream << "#ifdef cl_khr_fp64\n"
107 "#pragma OPENCL EXTENSION cl_khr_fp64 : enable\n"
108 "#elif defined(cl_amd_fp64)\n"
109 "#pragma OPENCL EXTENSION cl_amd_fp64 : enable\n"
110 "#else\n"
111 "#error \"Double precision floating point not supported"
112 " by OpenCL implementation on your device.\" \n"
113 "#endif\n\n";
114 }
115
116 // Enable atomic_add used by get_valid_counts. Only needed for OpenCL < 1.1.
117 if (enable_atomics_) {
118 decl_stream << "#pragma OPENCL EXTENSION cl_khr_global_int32_base_atomics : enable\n"
119 "#pragma OPENCL EXTENSION cl_khr_global_int32_extended_atomics : enable\n\n";
120 }
121
122 // Enable OpenCL 1.2 sampler-less texture reads, but utilize
123 // provided sampler in OpenCL 2.0.
124 if (enable_compliant_texture_reads_) {
125 // TODO(csullivan, lunderberg): Extend device attribute querying to support remote devices
126 // generically through the device API such that a target can be created from a specific device's
127 // attributes and utilized during codegen. Potential generlization of #8127 (c02cafb) for remote
128 // devices.
129 //
130 // E.g. Only provide an image sampler when the local or remote device supports OpenCL 2.0,
131 // see below for context.
132 //
133 // For backwards compatibility with OpenCL 1.2, sampler-less read_image calls are used.
134 // By default in sampler-less read_image calls OpenCL defaults to
135 // sampler_ = "CLK_NORMALIZED_COORDS_FALSE | CLK_ADDRESS_NONE | CLK_FILTER_NEAREST";
136 // See section 6.12.14.3 Built-in Image Sampler-less Read Functions in the OpenCL 1.2
137 // specification. For OpenCL 2.0 it can be preferable to use,
138 // sampler_ = "CLK_NORMALIZED_COORDS_FALSE | CLK_ADDRESS_CLAMP | CLK_FILTER_NEAREST";
139 // For now we rely on OpenCL preprocessor directives to utilize the correct behavior
140 // depending on the OpenCL version detected at OpenCL compile time.
141 decl_stream << "#ifdef __OPENCL_VERSION__\n"
142 << "#if __OPENCL_VERSION__ == CL_VERSION_2_0\n"
143 << "#define READ_IMAGEH(image, sampler, coord) "
144 << "read_imageh(image, sampler, coord)\n"
145 << "#define READ_IMAGEF(image, sampler, coord) "
146 << "read_imagef(image, sampler, coord)\n"
147 << "#else\n"
148 << "#define READ_IMAGEH(image, sampler, coord) "
149 << "read_imageh(image, coord)\n"
150 << "#define READ_IMAGEF(image, sampler, coord) "
151 << "read_imagef(image, coord)\n"
152 << "#endif\n"
153 << "#endif\n\n";
154 }
155 return CodeGenC::Finish();
156}

158void CodeGenOpenCL::BindThreadIndex(const IterVar& iv) {
159 ICHECK(!var_idmap_.count(iv->var.get()));
160 runtime::ThreadScope ts = runtime::ThreadScope::Create(iv->thread_tag);
161 std::ostringstream os;
162 if (ts.rank == 1) {
163 os << "get_local_id(" << ts.dim_index << ")";
164 } else {
165 os << "get_group_id(" << ts.dim_index << ")";
166 }
167 var_idmap_[iv->var.get()] = CastFromTo(os.str(), DataType::UInt(64), iv->var.dtype());
168}

170void CodeGenOpenCL::PrintType(DataType t, std::ostream& os) { // NOLINT(*)
171 int lanes = t.lanes();
172 if (t.is_handle()) {
173 ICHECK_EQ(lanes, 1) << "do not yet support vector types";
174 os << "void*";
175 return;
176 }
177 if (t.is_void()) {
178 os << "void";
179 return;
180 }
181 if (t == DataType::Bool()) {
182 os << "bool";
183 return;
184 }
185 bool fail = false;
186 if (t.is_float()) {
187 switch (t.bits()) {
188 case 16:
189 os << "half";
190 enable_fp16_ = true;
191 break;
192 case 32:
193 os << "float";
194 break;
195 case 64:
196 os << "double";
197 enable_fp64_ = true;
198 break;
199 default:
200 fail = true;
201 break;
202 }
203 if (!fail && lanes == 1) return;
204 if (!fail && ((lanes >= 2 && lanes <= 4) || lanes == 8 || lanes == 16)) {
205 os << lanes;
206 return;
207 }
208 } else if (t.is_uint() || t.is_int()) {
209 if (t.is_uint()) {
210 os << 'u';
211 }
212 if (t.bits() == 8 && t.lanes() == 4) {
213 // directly 4 8 bit int in integer.
214 os << "int";
215 return;
216 }
217 switch (t.bits()) {
218 case 8:
219 os << "char";
220 break;
221 case 16:
222 os << "short";
223 break;
224 case 32:
225 os << "int";
226 break;
227 case 64:
228 os << "long";
229 break;
230 case 1:
231 os << "int";
232 break;
233 default:
234 fail = true;
235 break;
236 }
237 if (!fail && lanes == 1) return;
238 if (!fail && ((lanes >= 2 && lanes <= 4) || lanes == 8 || lanes == 16)) {
239 os << lanes;
240 return;
241 }
242 }
243 LOG(FATAL) << "Cannot convert type " << t << " to OpenCL type";
244}

246void CodeGenOpenCL::PrintType(const Type& type, std::ostream& os) { // NOLINT(*)
247 if (auto* ptr = type.as<PrimTypeNode>()) {
248 return PrintType(ptr->dtype, os);
249 } else if (auto* ptr = type.as<PointerTypeNode>()) {
250 if (runtime::IsTextureStorage(std::string(ptr->storage_scope))) {
251 os << "image2d_t";
252 } else {
253 PrintType(ptr->element_type, os);
254 os << '*';
255 }
256 } else if (IsVoidType(type)) {
257 os << "void";
258 } else {
259 LOG(FATAL) << "Type " << type << " does not have a corresponding C Type";
260 }

263void CodeGenOpenCL::PrintVecAddr(const BufferNode* buffer, DataType t, PrimExpr base,
264 std::ostream& os) { // NOLINT(*)
265 const VarNode* buffer_var = buffer->data.get();
266 if (!HandleTypeMatch(buffer_var, t.element_of())) {
267 os << '(';
268 auto it = alloc_storage_scope_.find(buffer_var);
269 if (it != alloc_storage_scope_.end()) {
270 PrintStorageScope(it->second, os);
271 }
272 PrintType(t.element_of(), os);
273 os << "*)";
274 }
275 os << GetVarID(buffer_var) << " + ";
276 PrintExpr(base, os);

333std::string CodeGenOpenCL::CastTo(std::string value, DataType target) {
334 std::ostringstream os;
335 if (target.lanes() == 1) {
336 os << "((";
337 this->PrintType(target, os);

339 } else { // convert vector type
340 os << "(";
341 os << "convert_";
342 this->PrintType(target, os);
343 os << "(" << value << "))";
344 }

352void CodeGenOpenCL::VisitStmt_(const BufferStoreNode* op) {
353 if (auto call = op->value.as<CallNode>()) {
354 if (call->op.same_as(builtin::texture2d_load())) {
355 need_texture_ssa_ = false;
356 // If storing a texture load into a buffer, don't use an
357 // intermediate local unless the buffer allocation is a
358 // single element selected from the texture read.
359 auto it = allocation_size_.find(op->buffer->data.get());
360 if (it != allocation_size_.end() && it->second == 1) {
361 need_texture_ssa_ = true;
362 }
363 }
364 }
365 CodeGenC::VisitStmt_(op);
366 need_texture_ssa_ = true;
367}

369void CodeGenOpenCL::VisitExpr_(const CastNode* op, std::ostream& os) {
370 if (auto call = op->value.as<CallNode>()) {
371 if (call->op.same_as(builtin::texture2d_load())) {
372 need_texture_ssa_ = false;
373 }
374 }

376 need_texture_ssa_ = true;

384void CodeGenOpenCL::VisitExpr_(const CallNode* op, std::ostream& os) {
385 if (op->op.same_as(builtin::address_of())) {
386 // Overload tvm_address_of to add storage scope (e.g. __global).
387 const BufferLoadNode* load = op->args[0].as<BufferLoadNode>();
388 ICHECK(op->args.size() == 1 && load);
389 ICHECK_EQ(load->indices.size(), 1) << "CodeGenOpenCL only supports flat memory allocations.";
390 os << "((";
391 auto it = alloc_storage_scope_.find(load->buffer->data.get());
392 if (it != alloc_storage_scope_.end()) {
393 PrintStorageScope(it->second, os);
394 }
395 this->PrintType(load->dtype.element_of(), os);
396 os << " *)" << this->GetVarID(load->buffer->data.get()) << " + ";
397 this->PrintExpr(load->indices[0], os);
398 os << ')';
399 } else if (op->op.same_as(builtin::texture2d_store())) {
400 auto* ptr_type = op->args[0].as<VarNode>()->type_annotation.as<PointerTypeNode>();
401 ICHECK(ptr_type != nullptr) << "Texture Var's must be of PointerType";
402 ICHECK(runtime::IsTextureStorage(std::string(ptr_type->storage_scope)))
403 << "builtin::texture2d_store() only supports storing to texture buffers";
404 DataType buffer_type = ptr_type->element_type.as<PrimTypeNode>()->dtype;
405 if (buffer_type.is_float16()) {
406 os << "write_imageh(";
407 } else if (buffer_type.is_float()) {
408 os << "write_imagef(";
409 } else {
410 LOG(FATAL) << "Unsupported type: " << buffer_type
411 << ", currently only float and half are supported for image2d OpenCL codegen.";
412 }
413 this->PrintExpr(op->args[0], os);
414 os << ", ";
415 os << "(int2)(";
416 this->PrintExpr(op->args[1], os);
417 os << ", ";
418 this->PrintExpr(op->args[2], os);
419 os << "), ";
420 this->PrintExpr(op->args[3], os);
421 os << ")";
422 } else if (op->op.same_as(builtin::texture2d_load())) {
423 enable_compliant_texture_reads_ = true;
424 std::stringstream ss;
425 if (op->dtype.is_float16()) {
426 ss << "READ_IMAGEH(";
427 } else if (op->dtype.is_float()) {
428 ss << "READ_IMAGEF(";
429 } else {
430 LOG(FATAL) << "Unsupported type: " << op->dtype
431 << ", currently only float and half are supported for image2d OpenCL codegen.";
432 }
433 this->PrintExpr(op->args[0], ss);
434 ss << ", ";
435 ss << "CLK_NORMALIZED_COORDS_FALSE | CLK_ADDRESS_CLAMP | CLK_FILTER_NEAREST, ";
436 ss << "((int2)(";
437 this->PrintExpr(op->args[1], ss);
438 ss << ", ";
439 this->PrintExpr(op->args[2], ss);
440 ss << ")))";
441
442 // Only use local SSA if texture is not already being stored
443 if (need_texture_ssa_) {
444 std::string rhs = SSAGetID(ss.str(), op->dtype.with_lanes(4));
445 if (op->args.back().as<RampNode>()) {
446 os << rhs;
447 } else {
448 os << "((";
449 this->PrintType(op->dtype.with_lanes(1), os);
450 os << "*)&" << rhs << ")[";
451 this->PrintExpr(op->args.back(), os);
452 os << "]";
453 }
454 } else {
455 os << ss.str();
456 }
457 } else if (op->op.same_as(builtin_call_extern_)) {
458 auto func = Downcast<StringImm>(op->args[0]);
459 // Enable atomics extension if used.
460 if (func->value == "atomic_add") {
461 enable_atomics_ = true;
462 }
463 CodeGenC::VisitExpr_(op, os);
464 } else {
465 CodeGenC::VisitExpr_(op, os);
466 }
467}

494template <typename T>
495inline void PrintBinaryExpr(const T* op, const char* opstr, std::ostream& os, CodeGenOpenCL* p) {
496 if (op->dtype.lanes() == 1) {
497 os << opstr << "((";
498 p->PrintType(op->a->dtype, os);
499 os << ")";
500 p->PrintExpr(op->a, os);
501 os << ", (";
502 p->PrintType(op->b->dtype, os);
503 os << ")";
504 p->PrintExpr(op->b, os);
505 os << ')';
506 } else {
507 p->PrintVecBinaryOp(opstr, op->dtype, op->a, op->b, os);

543void CodeGenOpenCL::VisitExpr_(const SelectNode* op, std::ostream& os) {
544 std::ostringstream oss;
545 os << "select(";
546 PrintExpr(op->false_value, oss);
547 os << CastFromTo(oss.str(), op->false_value.dtype(), op->dtype);
548 oss.str("");
549 os << ", ";
550 PrintExpr(op->true_value, oss);
551 os << CastFromTo(oss.str(), op->true_value.dtype(), op->dtype);
552 oss.str("");
553 os << ", ";
554 PrintExpr(op->condition, oss);
555 if (op->dtype.is_float()) {
556 if (op->condition.dtype().is_uint() || op->condition.dtype().is_int()) {
557 os << oss.str();
558 } else {
559 os << CastTo(oss.str(), DataType::Int(op->dtype.bits(), op->dtype.lanes()));
560 }
561 } else {
562 os << CastFromTo(oss.str(), op->condition.dtype(), op->dtype);
563 }
564 os << ")";
565}

574runtime::Module BuildOpenCL(IRModule mod, Target target) {
575 using tvm::runtime::Registry;
576 bool output_ssa = false;
577
578 std::stringstream code;
579 const auto* fpostproc = Registry::Get("tvm_callback_opencl_postproc");
580 for (auto kv : mod->functions) {
581 ICHECK(kv.second->IsInstance<PrimFuncNode>()) << "CodeGenOpenCL: Can only take PrimFunc";
582 code << "// Function: " << kv.first->name_hint << std::endl;
583 CodeGenOpenCL cg;
584 cg.Init(output_ssa);
585 auto f = Downcast<PrimFunc>(kv.second);
586 auto calling_conv = f->GetAttr<Integer>(tvm::attr::kCallingConv);
587 ICHECK(calling_conv == CallingConv::kDeviceKernelLaunch)
588 << "CodeGenOpenCL: expect calling_conv equals CallingConv::kDeviceKernelLaunch";
589 cg.AddFunction(f);
590 std::string fsource = cg.Finish();
591 if (fpostproc) {
592 fsource = (*fpostproc)(fsource).operator std::string();
593 }
594 code << fsource;
595 }
596
597 return OpenCLModuleCreate(code.str(), "cl", ExtractFuncInfo(mod), code.str());
598}
