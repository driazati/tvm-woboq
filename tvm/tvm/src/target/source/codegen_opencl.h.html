20/*!
21 * \file codegen_opencl.h
22 * \brief Generate OpenCL device code.
23 */
24#ifndef TVM_TARGET_SOURCE_CODEGEN_OPENCL_H_
25#define TVM_TARGET_SOURCE_CODEGEN_OPENCL_H_
26
27#include <tvm/target/codegen.h>
28
29#include <string>
30#include <unordered_map>
31
32#include "codegen_c.h"
33
34namespace tvm {
35namespace codegen {
36
37class CodeGenOpenCL final : public CodeGenC {
38 public:
39 CodeGenOpenCL();
40 std::string Finish();
41
42 // override print thread tag.
43 void InitFuncState(const PrimFunc& f) final;
44 void PrintFuncPrefix() final; // NOLINT(*)
45 void BindThreadIndex(const IterVar& iv) final; // NOLINT(*)
46 void PrintStorageScope(const std::string& scope, std::ostream& os) final; // NOLINT(*)
47 void PrintStorageSync(const CallNode* op) final; // NOLINT(*)
48 void PrintType(DataType t, std::ostream& os) final; // NOLINT(*)
49 void PrintType(const Type& type, std::ostream& os) final; // NOLINT(*)
50 std::string GetVecLoad(DataType t, const BufferNode* buffer, PrimExpr base) final;
51 void PrintVecStore(const BufferNode* buffer, DataType t, PrimExpr base,
52 const std::string& value) final; // NOLINT(*)
53 // the address of load/store
54 void PrintVecAddr(const BufferNode* buffer, DataType t, PrimExpr base,
55 std::ostream& os); // NOLINT(*)
56 void PrintRestrict(const Var& v, std::ostream& os) final; // NOLINT(*)
57 std::string CastFromTo(std::string value, DataType from, DataType target); // NOLINT(*)
58 std::string CastTo(std::string value, DataType target); // NOLINT(*)
59 void SetTextureScope(const std::unordered_map<const VarNode*, std::string>&); // NOLINT(*)
60
61 // overload visitor
62 void VisitStmt_(const AllocateNode* op) final; // NOLINT(*)
63 void VisitExpr_(const BroadcastNode* op, std::ostream& os) final; // NOLINT(*)
64 void VisitExpr_(const CallNode* op, std::ostream& os) final; // NOLINT(*)
65 void VisitExpr_(const CastNode* op, std::ostream& os) final; // NOLINT(*)
66 void VisitExpr_(const FloatImmNode* op, std::ostream& os) final; // NOLINT(*)
67 void VisitStmt_(const StoreNode* op) final; // NOLINT(*)
68 void VisitStmt_(const BufferStoreNode* op) final; // NOLINT(*)
69
70 // overload min and max to avoid ambiguous call errors
71 void VisitExpr_(const MinNode* op, std::ostream& os) final;
72 void VisitExpr_(const MaxNode* op, std::ostream& os) final;
73 void VisitExpr_(const AndNode* op, std::ostream& os) final;
74 void VisitExpr_(const OrNode* op, std::ostream& os) final;
75 void VisitExpr_(const SelectNode* op, std::ostream& os) final;
76
77 private:
78 // whether enable fp16 and fp64 extension
79 bool enable_fp16_{false};
80 bool enable_fp64_{false};
81 // Whether to enable atomics extension.
82 bool enable_atomics_{false};
83 // Whether to enable sampler or sampler-less texture reads,
84 // where the choice depends on the OpenCL version used.
85 bool enable_compliant_texture_reads_{false};
86 // Key to disable use of texture SSA in certain scenarios. For example,
87 // when loaded value is stored directly to a user declared l-value buffer
88 bool need_texture_ssa_{true};
89 // Mapping from buffer to allocation size.
90 // Useful to track when a scalar store of a vectorized texture load is required.
91 std::unordered_map<const Object*, size_t> allocation_size_;
92};
93
94} // namespace codegen
95} // namespace tvm
96
97#endif // TVM_TARGET_SOURCE_CODEGEN_OPENCL_H_
