47void CodeGenCHost::Init(bool output_ssa, bool emit_asserts, bool emit_fwd_func_decl,
48 std::string target_str, const std::unordered_set<std::string>& devices) {
49 emit_asserts_ = emit_asserts;
50 emit_fwd_func_decl_ = emit_fwd_func_decl;
51 declared_globals_.clear();
52 decl_stream << "// tvm target: " << target_str << "\n";
53 decl_stream << "#define TVM_EXPORTS\n";
54 decl_stream << "#include \"tvm/runtime/c_runtime_api.h\"\n";
55 decl_stream << "#include \"tvm/runtime/c_backend_api.h\"\n";
56 decl_stream << "#include <math.h>\n";
57 if (devices.find("ethos-u") != devices.end()) {
58 decl_stream << "#include <tvm_ethosu_runtime.h>\n";
59 }
60 if (devices.find("cmsis-nn") != devices.end()) {
61 decl_stream << "#include <stdio.h>\n";
62 decl_stream << "#include <stdlib.h>\n";
63 decl_stream << "#include <dlpack/dlpack.h>\n";
64 decl_stream << "#include <arm_nnfunctions.h>\n";
65 decl_stream << "#include <arm_nn_types.h>\n";
66 decl_stream << "#include <arm_nn_math_types.h>\n";
67 }
68 CodeGenC::Init(output_ssa);
69}

77void CodeGenCHost::AddFunction(const PrimFunc& f, bool emit_fwd_func_decl) {
78 auto global_symbol = f->GetAttr<String>(tvm::attr::kGlobalSymbol);
79 ICHECK(global_symbol.defined())
80 << "CodeGenCHost: Expect PrimFunc to have the global_symbol attribute";
81 function_names_.push_back(global_symbol.value());
82
83 emit_fwd_func_decl_ = emit_fwd_func_decl;
84 CodeGenC::AddFunction(f);
85 if (f->HasNonzeroAttr(tir::attr::kIsEntryFunc)) {
86 function_names_.push_back(runtime::symbol::tvm_module_main);
87 stream << "// CodegenC: NOTE: Auto-generated entry function\n";
88 PrintFuncPrefix(stream);
89 stream << " " << tvm::runtime::symbol::tvm_module_main
90 << "(void* args, int* arg_type_ids, int num_args, void* out_ret_value, "
91 << "int* out_ret_tcode, void* resource_handle) {\n";
92 stream << " return " << global_symbol.value()
93 << "(args, arg_type_ids, num_args, out_ret_value, out_ret_tcode, resource_handle);\n";
94 stream << "}\n";
95 }
96}

98void CodeGenCHost::GenerateForwardFunctionDeclarations(String global_symbol,
99 const Array<PrimExpr>& args) {
100 if (!emit_fwd_func_decl_) {
101 return;
102 }
103 for (auto& func_already_defined : GetFunctionNames()) {
104 if (global_symbol == func_already_defined) {
105 return;
106 }
107 }
108 this->PrintFuncPrefix(fwd_decl_stream);
109 fwd_decl_stream << " " << global_symbol << "(";
110 for (size_t i = 1; i < args.size(); ++i) {
111 CodeGenSourceBase::PrintType(GetType(args[i]), fwd_decl_stream);
112 fwd_decl_stream << " ", this->PrintExpr(args[i], fwd_decl_stream);
113 if (i < args.size() - 1) {
114 fwd_decl_stream << ", ";
115 }
116 }
117 fwd_decl_stream << ");\n";
118}

141void CodeGenCHost::PrintType(DataType t, std::ostream& os) { // NOLINT(*)
142 int lanes = t.lanes();
143 if (t.is_handle()) {
144 ICHECK_EQ(lanes, 1) << "does not support vector types";
145 os << "void*";
146 return;
147 }
148 if (t.is_void()) {
149 os << "void";
150 return;
151 }
152 if (t == DataType::Bool()) {
153 os << "bool";
154 return;
155 }
156 bool fail = false;
157 if (t.is_float()) {
158 switch (t.bits()) {
159 case 16:
160 os << "half";
161 break;
162 case 32:
163 os << "float";
164 break;
165 case 64:
166 os << "double";
167 break;
168 default:
169 fail = true;
170 break;
171 }
172 if (!fail && lanes == 1) return;
173 if (!fail && (lanes >= 2 && lanes <= 16)) {
174 os << lanes;
175 return;
176 }
177 } else if (t.is_uint() || t.is_int()) {
178 if (t.is_uint()) {
179 os << 'u';
180 }
181 switch (t.bits()) {
182 case 8:
183 os << "int8_t";
184 break;
185 case 16:
186 os << "int16_t";
187 break;
188 case 32:
189 os << "int32_t";
190 break;
191 case 64:
192 os << "int64_t";
193 break;
194 case 1:
195 os << "int32_t";
196 break;
197 default:
198 fail = true;
199 break;
200 }
201 if (!fail && lanes == 1) return;
202 if (!fail && (lanes >= 2 && lanes <= 16)) {
203 os << lanes;
204 return;
205 }
206 }
207 LOG(FATAL) << "Cannot convert type " << t << " to C type";
208}

210void CodeGenCHost::VisitExpr_(const BroadcastNode* op, std::ostream& os) { // NOLINT(*)
211 std::string v = PrintExpr(op->value);
212 os << "((";
213 PrintType(op->dtype, os);
214 os << ")(";
215 for (int i = 0; i < op->lanes; ++i) {
216 if (i != 0) os << ", ";
217 os << v;
218 }
219 os << "))";
220}

222void CodeGenCHost::PrintGetFuncFromBackend(const std::string& func_name,
223 const std::string& packed_func_name) {
224 this->PrintIndent();
225 this->stream << "if (" << packed_func_name << " == NULL) {\n";
226 int packed_func_if_scope = this->BeginScope();
227 this->PrintIndent();
228 this->stream << "if (TVMBackendGetFuncFromEnv(" << module_name_ << ", \"" << func_name << "\""
229 << ", &" << packed_func_name << ") != 0) {\n";
230 int get_func_env_scope = this->BeginScope();
231 this->PrintIndent();
232 this->stream << "return -1;\n";
233 this->EndScope(get_func_env_scope);
234 this->PrintIndent();
235 this->stream << "}\n";
236 this->EndScope(packed_func_if_scope);
237 this->PrintIndent();
238 this->stream << "}\n";
239}

241void CodeGenCHost::PrintFuncCall(const std::string& packed_func_name, int num_args) {
242 this->PrintIndent();
243 std::string ret_val = name_supply_->FreshName("ret_val");
244 std::string ret_type_code = name_supply_->FreshName("ret_type_code");
245 this->stream << "TVMValue " << ret_val << ";\n";
246 this->PrintIndent();
247 this->stream << "int " << ret_type_code << ";\n";
248 this->PrintIndent();
249 this->stream << "if (TVMFuncCall(" << packed_func_name << ", "
250 << "(TVMValue*) stack_value"
251 << ", "
252 << "(int*) stack_tcode"
253 << ", " << num_args << ", "
254 << "&" << ret_val << ", "
255 << "&" << ret_type_code << ") != 0) {\n";
256 int func_call_scope = this->BeginScope();
257 this->PrintIndent();
258 this->stream << "return -1;\n";
259 this->EndScope(func_call_scope);
260 this->PrintIndent();
261 this->stream << "}\n";
262}

264void CodeGenCHost::PrintFuncCallC(const std::string& packed_func_name, int num_args,
265 const std::string& resource_handle_name) {
266 this->PrintIndent();
267 std::string ret_val = name_supply_->FreshName("ret_val");
268 std::string ret_type_code = name_supply_->FreshName("ret_type_code");
269 this->stream << "TVMValue " << ret_val << ";\n";
270 this->PrintIndent();
271 this->stream << "int " << ret_type_code << ";\n";
272 this->PrintIndent();
273
274 this->stream << "if (" << packed_func_name << "( "
275 << "(TVMValue*) stack_value "
276 << ", "
277 << "(int*) stack_tcode"
278 << ", " << num_args << ", "
279 << "&" << ret_val << ", "
280 << "&" << ret_type_code << ", " << resource_handle_name << ") != 0){\n";
281
282 int func_call_scope = this->BeginScope();
283 this->PrintIndent();
284 this->stream << "return -1;\n";
285 this->EndScope(func_call_scope);
286 this->PrintIndent();
287 this->stream << "}\n";
288}

290std::string CodeGenCHost::GetPackedName(const CallNode* op) {
291 const StringImmNode* s = op->args[0].as<StringImmNode>();
292 ICHECK(s != nullptr) << "tvm_call_packed_lowered expects first argument as function name";
293 std::string func_name = s->value;
294 std::string packed_func_name = func_name + "_packed";
295 std::string unique_name;
296 auto it = declared_globals_.find(packed_func_name);
297 if (it != declared_globals_.end()) {
298 unique_name = it->second;
299 } else {
300 unique_name = name_supply_->FreshName(packed_func_name);
301 declared_globals_[packed_func_name] = unique_name;
302 decl_stream << "static void* " << unique_name << " = NULL;\n";
303 }
304 return unique_name;
305}

307CodeGenCHost::FunctionInfo CodeGenCHost::GetFunctionInfo(const CallNode* op,
308 bool has_resource_handle) {
309 const StringImmNode* s = op->args[0].as<StringImmNode>();
310 ICHECK(s != nullptr) << "tvm_call_[c]packed_lowered expects first argument as function name";
311 int64_t begin = op->args[3].as<IntImmNode>()->value;
312 int64_t end = op->args[4].as<IntImmNode>()->value;
313 int64_t num_args = end - begin;
314 ICHECK_GE(num_args, 0);
315 std::string func_name = s->value;
316
317 if (has_resource_handle) {
318 const StringImmNode* resource_handle_var = op->args[5].as<StringImmNode>();
319 if (resource_handle_var != nullptr) {
320 std::string resource_handle_name = resource_handle_var->value;
321 return {func_name, num_args - 1, resource_handle_name};
322 } else {
323 // The final arg should be "(void*) NULL" to indicate the empty resource_handle.
324 num_args--;
325
326 const CallNode* reinterpret_call = op->args[5].as<CallNode>();
327 ICHECK_NE(reinterpret_call, (void*)nullptr)
328 << "At CallNode to " << s
329 << "arg 5: Expect either StringImm naming the resource_handle var from interface API or "
330 << "reinterpret(0); got: " << op->args[5];
331 ICHECK_EQ(reinterpret_call->op, builtin::reinterpret())
332 << "At CallNode to " << s
333 << "arg 5: Expect either StringImm naming the resource_handle var from interface API or "
334 << "reinterpret(0); got: " << op->args[5];
335 ICHECK(is_zero(reinterpret_call->args[0])) << "At CallNode to " << s
336 << " arg 5: Expect either StringImm naming the "
337 "resource_handle var from interface API, or "
338 << "zero; got " << op->args[5];
339 }
340 }
341 return {func_name, num_args, "NULL"};
342}

344void CodeGenCHost::VisitExpr_(const CallNode* op, std::ostream& os) { // NOLINT(*)
345 if (op->op.same_as(builtin::tvm_stack_alloca())) {
346 std::string stack_name = name_supply_->FreshName("stack");
347 const std::string& type = op->args[0].as<StringImmNode>()->value;
348 const IntImmNode* num = op->args[1].as<IntImmNode>();
349 ICHECK(num != nullptr);
350 static_assert(alignof(TVMValue) % alignof(DLTensor) == 0, "invariant");
351 size_t unit = sizeof(TVMValue);
352 size_t size = 0;
353 if (type == "shape") {
354 size = (num->value * sizeof(tvm_index_t) + unit - 1) / unit;
355 } else if (type == "arg_value") {
356 size = (num->value * sizeof(TVMValue) + unit - 1) / unit;
357 } else if (type == "arg_tcode") {
358 size = (num->value * sizeof(int) + unit - 1) / unit;
359 } else if (type == "array") {
360 size = (num->value * sizeof(DLTensor) + unit - 1) / unit;
361 } else {
362 LOG(FATAL) << "Unknown stack alloca type " << type;
363 }
364 this->PrintIndent();
365 this->stream << "TVMValue " << stack_name << "[" << size << "];\n";
366 os << stack_name;
367 } else if (op->op.same_as(builtin::tvm_call_packed_lowered())) {
368 auto function_info = GetFunctionInfo(op, false /* has_resource_handle */);
369 std::string func_name_packed = GetPackedName(op);
370 this->PrintGetFuncFromBackend(function_info.func_name, func_name_packed);
371 this->PrintFuncCall(func_name_packed, function_info.num_args);
372 } else if (op->op.same_as(builtin::tvm_call_cpacked_lowered())) {
373 auto function_info = GetFunctionInfo(op, true /* has_resource_handle */);
374 this->PrintFuncCallC(function_info.func_name, function_info.num_args,
375 function_info.resource_handle_name);
376 } else if (op->op.same_as(builtin::tvm_throw_last_error())) {
377 this->PrintIndent();
378 this->stream << "return -1;\n";
379 } else {
380 CodeGenC::VisitExpr_(op, os);
381 }
382}

384void CodeGenCHost::VisitStmt_(const AssertStmtNode* op) { // NOLINT(*)
385 if (emit_asserts_) {
386 std::string cond = PrintExpr(op->condition);
387 PrintIndent();
388 stream << "if (!(" << cond << ")) {\n";
389 int assert_if_scope = this->BeginScope();
390 PrintIndent();
391 stream << "TVMAPISetLastError(\"" << op->message.as<StringImmNode>()->value << "\");\n";
392 PrintIndent();
393 stream << "return -1;\n";
394 this->EndScope(assert_if_scope);
395 PrintIndent();
396 stream << "}\n";
397 }
398 this->PrintStmt(op->body);
399}

409template <typename T>
410inline void CodeGenCHost::PrintTernaryCondExpr(const T* op, const char* compare,
411 std::ostream& os) { // NOLINT(*)
412 std::ostringstream temp_a;
413 VisitExpr(op->a, temp_a);
414 std::string a_id = SSAGetID(temp_a.str(), op->a.dtype());
415 std::ostringstream temp_b;
416 VisitExpr(op->b, temp_b);
417 std::string b_id = SSAGetID(temp_b.str(), op->b.dtype());
418
419 os << "((" << a_id << ") " << compare << " (" << b_id << ") "
420 << "? (" << a_id << ") : (" << b_id << "))";
421}

423runtime::Module BuildCHost(IRModule mod, Target target) {
424 using tvm::runtime::Registry;
425 bool output_ssa = false;
426 bool emit_asserts = false;
427 bool emit_fwd_func_decl = true;
428
429 std::unordered_set<std::string> devices;
430 if (mod->GetAttr<Map<GlobalVar, String>>("device_contexts") != nullptr) {
431 Map<GlobalVar, String> device_contexts =
432 mod->GetAttr<Map<GlobalVar, String>>("device_contexts").value();
433 for (auto const& context : device_contexts) {
434 devices.insert(context.second.data());
435 }
436 }
437
438 CodeGenCHost cg;
439 cg.Init(output_ssa, emit_asserts, emit_fwd_func_decl, target->str(), devices);
440 cg.SetConstantsByteAlignment(target->GetAttr<Integer>("constants-byte-alignment").value_or(16));
441 PrimFunc aot_executor_fn;
442
443 std::vector<std::pair<tvm::GlobalVar, tvm::BaseFunc>> funcs;
444 for (auto kv : mod->functions) {
445 // Make sure that the executor function is the last one to be code generated so that all the
446 // symbols are available to __tvm_main__
447 auto fun_name = std::string(kv.first->name_hint);
448 bool is_aot_executor_fn = kv.second->GetAttr<Bool>("runner_function", Bool(false)).value();
449
450 if (is_aot_executor_fn) {
451 aot_executor_fn = Downcast<PrimFunc>(kv.second);
452 continue;
453 }
454 funcs.push_back(kv);
455 }
456
457 // Sort functions
458 std::sort(funcs.begin(), funcs.end(),
459 [](std::pair<tvm::GlobalVar, tvm::BaseFunc> kv_a,
460 std::pair<tvm::GlobalVar, tvm::BaseFunc> kv_b) {
461 std::string name_hint_a = kv_a.first->name_hint;
462 std::string name_hint_b = kv_b.first->name_hint;
463 return name_hint_a < name_hint_b;
464 });
465
466 // Add all functions except __tvm_main__
467 for (auto& kv : funcs) {
468 ICHECK(kv.second->IsInstance<PrimFuncNode>()) << "CodegenCHost: Can only take PrimFunc";
469 auto f = Downcast<PrimFunc>(kv.second);
470 cg.AddFunction(f);
471 }
472
473 // Add __tvm_main__
474 if (aot_executor_fn.defined()) {
475 emit_fwd_func_decl = true;
476 cg.AddFunction(aot_executor_fn, emit_fwd_func_decl);
477 }
478
479 // NOTE: it's possible that kRuntime attr is not attached when the mod was built with tvm.build().
480 // See issue #10373.
481 auto opt_runtime = mod->GetAttr<relay::Runtime>(tvm::attr::kRuntime);
482 relay::Runtime runtime;
483 if (opt_runtime.get() != nullptr) {
484 runtime = opt_runtime.value();
485 } else {
486 runtime = relay::Runtime::Create("cpp", {});
487 }
488 if (aot_executor_fn.defined() && runtime->name == relay::kTvmRuntimeCpp) {
489 cg.InitGlobalContext();
490 }
491
492 if (target->GetAttr<Bool>("system-lib").value_or(Bool(false))) {
493 ICHECK_EQ(target->GetAttr<String>("runtime").value_or(""), "c")
494 << "c target only supports generating C runtime SystemLibs";
495 }
496
497 std::string code = cg.Finish();
498 return CSourceModuleCreate(code, "c", cg.GetFunctionNames());
499}
