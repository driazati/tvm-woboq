47void CodeGenCHost::Init(bool output_ssa, bool emit_asserts, std::string target_str,
48 const std::unordered_set<std::string>& devices) {
49 emit_asserts_ = emit_asserts;
50 declared_globals_.clear();
51 decl_stream << "// tvm target: " << target_str << "\n";
52 decl_stream << "#define TVM_EXPORTS\n";
53 decl_stream << "#include \"tvm/runtime/c_runtime_api.h\"\n";
54 decl_stream << "#include \"tvm/runtime/c_backend_api.h\"\n";
55 decl_stream << "#include <math.h>\n";
56 if (devices.find("ethos-u") != devices.end()) {
57 decl_stream << "#include <tvm_ethosu_runtime.h>\n";
58 }
59 if (devices.find("cmsis-nn") != devices.end()) {
60 decl_stream << "#include <stdio.h>\n";
61 decl_stream << "#include <stdlib.h>\n";
62 decl_stream << "#include <dlpack/dlpack.h>\n";
63 decl_stream << "#include <arm_nnfunctions.h>\n";
64 decl_stream << "#include <arm_nn_types.h>\n";
65 decl_stream << "#include <arm_nn_math_types.h>\n";
66 }
67 CodeGenC::Init(output_ssa);
68}

76void CodeGenCHost::AddFunction(const PrimFunc& f) {
77 auto global_symbol = f->GetAttr<String>(tvm::attr::kGlobalSymbol);
78 ICHECK(global_symbol.defined())
79 << "CodeGenCHost: Expect PrimFunc to have the global_symbol attribute";
80 function_names_.push_back(global_symbol.value());
81
82 CodeGenC::AddFunction(f);
83 if (f->HasNonzeroAttr(tir::attr::kIsEntryFunc)) {
84 function_names_.push_back(runtime::symbol::tvm_module_main);
85 stream << "// CodegenC: NOTE: Auto-generated entry function\n";
86 PrintFuncPrefix();
87 stream << " " << tvm::runtime::symbol::tvm_module_main
88 << "(void* args, int* arg_type_ids, int num_args, void* out_ret_value, "
89 << "int* out_ret_tcode, void* resource_handle) {\n";
90 stream << " return " << global_symbol.value()
91 << "(args, arg_type_ids, num_args, out_ret_value, out_ret_tcode, resource_handle);\n";
92 stream << "}\n";
93 }
94}

108void CodeGenCHost::PrintType(DataType t, std::ostream& os) { // NOLINT(*)
109 int lanes = t.lanes();
110 if (t.is_handle()) {
111 ICHECK_EQ(lanes, 1) << "does not support vector types";
112 os << "void*";
113 return;
114 }
115 if (t.is_void()) {
116 os << "void";
117 return;
118 }
119 if (t == DataType::Bool()) {
120 os << "bool";
121 return;
122 }
123 bool fail = false;
124 if (t.is_float()) {
125 switch (t.bits()) {
126 case 16:
127 os << "half";
128 break;
129 case 32:
130 os << "float";
131 break;
132 case 64:
133 os << "double";
134 break;
135 default:
136 fail = true;
137 break;
138 }
139 if (!fail && lanes == 1) return;
140 if (!fail && (lanes >= 2 && lanes <= 16)) {
141 os << lanes;
142 return;
143 }
144 } else if (t.is_uint() || t.is_int()) {
145 if (t.is_uint()) {
146 os << 'u';
147 }
148 switch (t.bits()) {
149 case 8:
150 os << "int8_t";
151 break;
152 case 16:
153 os << "int16_t";
154 break;
155 case 32:
156 os << "int32_t";
157 break;
158 case 64:
159 os << "int64_t";
160 break;
161 case 1:
162 os << "int32_t";
163 break;
164 default:
165 fail = true;
166 break;
167 }
168 if (!fail && lanes == 1) return;
169 if (!fail && (lanes >= 2 && lanes <= 16)) {
170 os << lanes;
171 return;
172 }
173 }
174 LOG(FATAL) << "Cannot convert type " << t << " to C type";
175}

177void CodeGenCHost::VisitExpr_(const BroadcastNode* op, std::ostream& os) { // NOLINT(*)
178 std::string v = PrintExpr(op->value);
179 os << "((";
180 PrintType(op->dtype, os);
181 os << ")(";
182 for (int i = 0; i < op->lanes; ++i) {
183 if (i != 0) os << ", ";
184 os << v;
185 }
186 os << "))";
187}

189void CodeGenCHost::PrintGetFuncFromBackend(const std::string& func_name,
190 const std::string& packed_func_name) {
191 this->PrintIndent();
192 this->stream << "if (" << packed_func_name << " == NULL) {\n";
193 int packed_func_if_scope = this->BeginScope();
194 this->PrintIndent();
195 this->stream << "if (TVMBackendGetFuncFromEnv(" << module_name_ << ", \"" << func_name << "\""
196 << ", &" << packed_func_name << ") != 0) {\n";
197 int get_func_env_scope = this->BeginScope();
198 this->PrintIndent();
199 this->stream << "return -1;\n";
200 this->EndScope(get_func_env_scope);
201 this->PrintIndent();
202 this->stream << "}\n";
203 this->EndScope(packed_func_if_scope);
204 this->PrintIndent();
205 this->stream << "}\n";
206}

208void CodeGenCHost::PrintFuncCall(const std::string& packed_func_name, int num_args) {
209 this->PrintIndent();
210 std::string ret_val = name_supply_->FreshName("ret_val");
211 std::string ret_type_code = name_supply_->FreshName("ret_type_code");
212 this->stream << "TVMValue " << ret_val << ";\n";
213 this->PrintIndent();
214 this->stream << "int " << ret_type_code << ";\n";
215 this->PrintIndent();
216 this->stream << "if (TVMFuncCall(" << packed_func_name << ", "
217 << "(TVMValue*) stack_value"
218 << ", "
219 << "(int*) stack_tcode"
220 << ", " << num_args << ", "
221 << "&" << ret_val << ", "
222 << "&" << ret_type_code << ") != 0) {\n";
223 int func_call_scope = this->BeginScope();
224 this->PrintIndent();
225 this->stream << "return -1;\n";
226 this->EndScope(func_call_scope);
227 this->PrintIndent();
228 this->stream << "}\n";
229}

231void CodeGenCHost::PrintFuncCallC(const std::string& packed_func_name, int num_args,

233 this->PrintIndent();

235 std::string ret_type_code = name_supply_->FreshName("ret_type_code");
236 this->stream << "TVMValue " << ret_val << ";\n";

239 this->PrintIndent();

241 this->stream << "if (" << packed_func_name << "( "
242 << "(TVMValue*) stack_value "
243 << ", "
244 << "(int*) stack_tcode"
245 << ", " << num_args << ", "
246 << "&" << ret_val << ", "
247 << "&" << ret_type_code << ", " << resource_handle_name << ") != 0){\n";
248
249 int func_call_scope = this->BeginScope();
250 this->PrintIndent();
251 this->stream << "return -1;\n";
252 this->EndScope(func_call_scope);
253 this->PrintIndent();
254 this->stream << "}\n";
255}

257std::string CodeGenCHost::GetPackedName(const CallNode* op) {
258 const StringImmNode* s = op->args[0].as<StringImmNode>();
259 ICHECK(s != nullptr) << "tvm_call_packed_lowered expects first argument as function name";
260 std::string func_name = s->value;
261 std::string packed_func_name = func_name + "_packed";
262 std::string unique_name;
263 auto it = declared_globals_.find(packed_func_name);
264 if (it != declared_globals_.end()) {
265 unique_name = it->second;
266 } else {
267 unique_name = name_supply_->FreshName(packed_func_name);
268 declared_globals_[packed_func_name] = unique_name;
269 decl_stream << "static void* " << unique_name << " = NULL;\n";
270 }
271 return unique_name;
272}

274CodeGenCHost::FunctionInfo CodeGenCHost::GetFunctionInfo(const CallNode* op,
275 bool has_resource_handle) {
276 const StringImmNode* s = op->args[0].as<StringImmNode>();
277 ICHECK(s != nullptr) << "tvm_call_[c]packed_lowered expects first argument as function name";
278 int64_t begin = op->args[3].as<IntImmNode>()->value;
279 int64_t end = op->args[4].as<IntImmNode>()->value;
280 int64_t num_args = end - begin;
281 ICHECK_GE(num_args, 0);
282 std::string func_name = s->value;
283
284 if (has_resource_handle) {
285 const StringImmNode* resource_handle_var = op->args[5].as<StringImmNode>();
286 if (resource_handle_var != nullptr) {
287 std::string resource_handle_name = resource_handle_var->value;
288 return {func_name, num_args - 1, resource_handle_name};
289 } else {
290 // The final arg should be "(void*) NULL" to indicate the empty resource_handle.
291 num_args--;
292
293 const CallNode* reinterpret_call = op->args[5].as<CallNode>();
294 ICHECK_NE(reinterpret_call, (void*)nullptr)
295 << "At CallNode to " << s
296 << "arg 5: Expect either StringImm naming the resource_handle var from interface API or "
297 << "reinterpret(0); got: " << op->args[5];
298 ICHECK_EQ(reinterpret_call->op, builtin::reinterpret())
299 << "At CallNode to " << s
300 << "arg 5: Expect either StringImm naming the resource_handle var from interface API or "
301 << "reinterpret(0); got: " << op->args[5];
302 ICHECK(is_zero(reinterpret_call->args[0])) << "At CallNode to " << s
303 << " arg 5: Expect either StringImm naming the "
304 "resource_handle var from interface API, or "
305 << "zero; got " << op->args[5];
306 }
307 }
308 return {func_name, num_args, "NULL"};
309}

311void CodeGenCHost::VisitExpr_(const CallNode* op, std::ostream& os) { // NOLINT(*)
312 if (op->op.same_as(builtin::tvm_stack_alloca())) {
313 std::string stack_name = name_supply_->FreshName("stack");
314 const std::string& type = op->args[0].as<StringImmNode>()->value;
315 const IntImmNode* num = op->args[1].as<IntImmNode>();
316 ICHECK(num != nullptr);
317 static_assert(alignof(TVMValue) % alignof(DLTensor) == 0, "invariant");
318 size_t unit = sizeof(TVMValue);
319 size_t size = 0;
320 if (type == "shape") {
321 size = (num->value * sizeof(tvm_index_t) + unit - 1) / unit;
322 } else if (type == "arg_value") {
323 size = (num->value * sizeof(TVMValue) + unit - 1) / unit;
324 } else if (type == "arg_tcode") {
325 size = (num->value * sizeof(int) + unit - 1) / unit;
326 } else if (type == "array") {
327 size = (num->value * sizeof(DLTensor) + unit - 1) / unit;
328 } else {
329 LOG(FATAL) << "Unknown stack alloca type " << type;
330 }
331 this->PrintIndent();
332 this->stream << "TVMValue " << stack_name << "[" << size << "];\n";
333 os << stack_name;
334 } else if (op->op.same_as(builtin::tvm_call_packed_lowered())) {
335 auto function_info = GetFunctionInfo(op, false /* has_resource_handle */);
336 std::string func_name_packed = GetPackedName(op);
337 this->PrintGetFuncFromBackend(function_info.func_name, func_name_packed);
338 this->PrintFuncCall(func_name_packed, function_info.num_args);
339 } else if (op->op.same_as(builtin::tvm_call_cpacked_lowered())) {
340 auto function_info = GetFunctionInfo(op, true /* has_resource_handle */);
341 this->PrintFuncCallC(function_info.func_name, function_info.num_args,
342 function_info.resource_handle_name);
343 } else if (op->op.same_as(builtin::tvm_throw_last_error())) {
344 this->PrintIndent();
345 this->stream << "return -1;\n";
346 } else {
347 CodeGenC::VisitExpr_(op, os);
348 }
349}

351void CodeGenCHost::VisitStmt_(const AssertStmtNode* op) { // NOLINT(*)
352 if (emit_asserts_) {
353 std::string cond = PrintExpr(op->condition);
354 PrintIndent();
355 stream << "if (!(" << cond << ")) {\n";
356 int assert_if_scope = this->BeginScope();
357 PrintIndent();
358 stream << "TVMAPISetLastError(\"" << op->message.as<StringImmNode>()->value << "\");\n";
359 PrintIndent();
360 stream << "return -1;\n";
361 this->EndScope(assert_if_scope);
362 PrintIndent();
363 stream << "}\n";
364 }
365 this->PrintStmt(op->body);
366}

376template <typename T>
377inline void CodeGenCHost::PrintTernaryCondExpr(const T* op, const char* compare,
378 std::ostream& os) { // NOLINT(*)
379 std::ostringstream temp_a;
380 VisitExpr(op->a, temp_a);
381 std::string a_id = SSAGetID(temp_a.str(), op->a.dtype());
382 std::ostringstream temp_b;
383 VisitExpr(op->b, temp_b);
384 std::string b_id = SSAGetID(temp_b.str(), op->b.dtype());
385
386 os << "((" << a_id << ") " << compare << " (" << b_id << ") "
387 << "? (" << a_id << ") : (" << b_id << "))";
388}

390runtime::Module BuildCHost(IRModule mod, Target target) {
391 using tvm::runtime::Registry;
392 bool output_ssa = false;
393 bool emit_asserts = false;
394
395 std::unordered_set<std::string> devices;
396 if (mod->GetAttr<Map<GlobalVar, String>>("device_contexts") != nullptr) {
397 Map<GlobalVar, String> device_contexts =
398 mod->GetAttr<Map<GlobalVar, String>>("device_contexts").value();
399 for (auto const& context : device_contexts) {
400 devices.insert(context.second.data());
401 }
402 }
403
404 CodeGenCHost cg;
405 cg.Init(output_ssa, emit_asserts, target->str(), devices);
406 cg.SetConstantsByteAlignment(target->GetAttr<Integer>("constants-byte-alignment").value_or(16));
407 PrimFunc aot_executor_fn;
408
409 std::vector<std::pair<tvm::GlobalVar, tvm::BaseFunc>> funcs;
410 for (auto kv : mod->functions) {
411 // Make sure that the executor function is the last one to be code generated so that all the
412 // symbols are available to __tvm_main__
413 auto fun_name = std::string(kv.first->name_hint);
414 bool is_aot_executor_fn = kv.second->GetAttr<Bool>("runner_function", Bool(false)).value();
415
416 if (is_aot_executor_fn) {
417 aot_executor_fn = Downcast<PrimFunc>(kv.second);
418 continue;
419 }
420 funcs.push_back(kv);
421 }
422
423 // Sort functions
424 std::sort(funcs.begin(), funcs.end(),
425 [](std::pair<tvm::GlobalVar, tvm::BaseFunc> kv_a,
426 std::pair<tvm::GlobalVar, tvm::BaseFunc> kv_b) {
427 std::string name_hint_a = kv_a.first->name_hint;
428 std::string name_hint_b = kv_b.first->name_hint;
429 return name_hint_a < name_hint_b;
430 });
431
432 // Add all functions except __tvm_main__
433 for (auto& kv : funcs) {
434 ICHECK(kv.second->IsInstance<PrimFuncNode>()) << "CodegenCHost: Can only take PrimFunc";
435 auto f = Downcast<PrimFunc>(kv.second);
436 cg.AddFunction(f);
437 }
438
439 // Add __tvm_main__
440 if (aot_executor_fn.defined()) {
441 cg.AddFunction(aot_executor_fn);
442 }
443
444 // NOTE: it's possible that kRuntime attr is not attached when the mod was built with tvm.build().
445 // See issue #10373.
446 auto opt_runtime = mod->GetAttr<relay::Runtime>(tvm::attr::kRuntime);
447 relay::Runtime runtime;
448 if (opt_runtime.get() != nullptr) {
449 runtime = opt_runtime.value();
450 } else {
451 runtime = relay::Runtime::Create("cpp", {});
452 }
453 if (aot_executor_fn.defined() && runtime->name == relay::kTvmRuntimeCpp) {
454 cg.InitGlobalContext();
455 }
456
457 if (target->GetAttr<Bool>("system-lib").value_or(Bool(false))) {
458 ICHECK_EQ(target->GetAttr<String>("runtime").value_or(""), "c")
459 << "c target only supports generating C runtime SystemLibs";
460 }
461
462 std::string code = cg.Finish();
463 return CSourceModuleCreate(code, "c", cg.GetFunctionNames());
464}
