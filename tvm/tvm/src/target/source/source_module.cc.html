92class CSourceModuleNode : public runtime::ModuleNode {
93 public:
94 CSourceModuleNode(const std::string& code, const std::string& fmt,
95 const Array<String>& func_names, const Array<String>& const_vars)
96 : code_(code), fmt_(fmt), const_vars_(const_vars), func_names_(func_names) {}
97 const char* type_key() const final { return "c"; }
98
99 PackedFunc GetFunction(const std::string& name, const ObjectPtr<Object>& sptr_to_self) final {
100 // Currently c-source module is used as demonstration purposes with binary metadata module
101 // that expects get_symbol interface. When c-source module is used as external module, it
102 // will only contain one function. However, when its used as an internal module (e.g., target
103 // "c") it can have many functions.
104 if (name == "get_symbol") {
105 return PackedFunc(
106 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->func_names_[0]; });
107 } else if (name == "get_const_vars") {
108 return PackedFunc(
109 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->const_vars_; });
110 } else if (name == "get_func_names") {
111 return PackedFunc(
112 [sptr_to_self, this](TVMArgs args, TVMRetValue* rv) { *rv = this->func_names_; });
113 } else {
114 return PackedFunc(nullptr);
115 }
116 }
117
118 std::string GetSource(const std::string& format) final { return code_; }
119
120 std::string GetFormat() { return fmt_; }
121
122 void SaveToFile(const std::string& file_name, const std::string& format) final {
123 std::string fmt = GetFileFormat(file_name, format);
124 std::string meta_file = GetMetaFilePath(file_name);
125 if (fmt == "c" || fmt == "cc" || fmt == "cpp" || fmt == "cu") {
126 ICHECK_NE(code_.length(), 0);
127 SaveBinaryToFile(file_name, code_);
128 } else {
129 ICHECK_EQ(fmt, fmt_) << "Can only save to format=" << fmt_;
130 }
131 }
132
133 bool IsDSOExportable() const final { return true; }
134
135 bool ImplementsFunction(const String& name, bool query_imports) final {
136 return std::find(func_names_.begin(), func_names_.end(), name) != func_names_.end();
137 }
138
139 protected:
140 std::string code_;
141 std::string fmt_;
142 Array<String> const_vars_;
143 Array<String> func_names_;
144};

260 if (metadata_->pool_inputs.defined()) {
261 for (const auto& kv : metadata_->pool_inputs.value()) {
262 tir::usmp::AllocatedPoolInfo allocated_pool_info = kv.second;
263 if (allocated_pool_info->pool_info->is_internal) {
264 if (const auto* pool_info = allocated_pool_info->pool_info.as<ConstantPoolInfoNode>()) {

275 void GenerateIOWorkspaceMapFunction(const std::string& struct_type,
276 const std::string& function_name,
277 const Array<String>& tensor_names) {
278 std::string map_function = runtime::get_name_mangled(metadata_->mod_name, function_name);
279 code_ << "struct " << struct_type << " " << map_function << "(\n";
280 std::string pools_struct = runtime::get_name_mangled(metadata_->mod_name, "workspace_pools");
281 code_ << " struct " << pools_struct << "* workspace_pools\n";
282 code_ << "\n){\n";
283 code_ << "struct " << struct_type << " ret = {\n";
284 for (const String& name : tensor_names) {
285 tir::usmp::PoolAllocation pool_allocation = metadata_->io_pool_allocations[name];
286 code_ << "\t." << name << " = "
287 << "&((uint8_t*)workspace_pools->" << pool_allocation->pool_info->pool_name << ")["
288 << pool_allocation->byte_offset << "],\n";
289 }
290 code_ << "};\n";
291 code_ << "return ret;\n";
292 code_ << "}\n\n";
293 }

295 void GenerateConstantBuffer(const ConstantPoolInfoNode* pool_info, size_t allocated_size) {
296 size_t ord = 0;
297 if (pool_info->constant_info_array.size() > 0) {
298 // Pool is RO, form an initialized struct
299 code_ << "__attribute__((section(\".rodata.tvm\"), ";
300 code_ << "))\n";
301 code_ << "static struct " << pool_info->pool_name << " {\n";
302 // emit struct field names
303 std::vector<ConstantInfo> const_info_vec(pool_info->constant_info_array.begin(),
304 pool_info->constant_info_array.end());
305 std::sort(const_info_vec.begin(), const_info_vec.end(),
306 [](const ConstantInfo& a, const ConstantInfo& b) {
307 return a->byte_offset->value < b->byte_offset->value;
308 });
309 for (const auto& const_info : const_info_vec) {
310 const auto& data = const_info->data;
311 const auto& offs = const_info->byte_offset;
312 int64_t num_elements = std::accumulate(data.Shape().begin(), data.Shape().end(), 1,
313 std::multiplies<int64_t>());
314 code_ << " ";
315 codegen_c_base_.PrintType(data.DataType(), code_);
316 code_ << " " << const_info->name_hint << "[" << num_elements << "] __attribute__(("
317 << (ord++ ? "packed, " : "") << "aligned(" << metadata_->constant_alignment << ")));";
318 code_ << " // " << num_elements * data.DataType().bytes()
319 << " bytes, aligned offset: " << offs << "\n";
320 }
321 code_ << "} " << pool_info->pool_name << " = {\n";
322
323 // emit struct field initialization data
324 for (const auto& const_info : const_info_vec) {
325 code_ << " ." << const_info->name_hint << " = {\n";
326 codegen::NDArrayDataToC(const_info->data, 4, code_);
327 code_ << " },\n";
328 }
329 code_ << "};";
330 code_ << "// of total size " << allocated_size << " bytes\n";
331 } else {
332 LOG(FATAL) << "No constant data in constant pool found "
333 << PrettyPrint(GetRef<ObjectRef>(pool_info));
334 }
335 }

337 void GenerateWorkspaceBuffer(const WorkspacePoolInfoNode* pool_info, size_t allocated_size) {
338 code_ << "__attribute__((section(\".bss.noinit.tvm\"), ";
339 code_ << "aligned(" << metadata_->workspace_alignment << ")))\n";
340 code_ << "static uint8_t " << pool_info->pool_name << "[";
341 code_ << allocated_size << "];\n";
342 }

344 bool IsInternalWorkspaceBuffer(const tir::Var& pool_var) {
345 if (metadata_->pool_inputs.defined()) {
346 Map<tir::Var, tir::usmp::AllocatedPoolInfo> allocated_pool_infos =
347 metadata_->pool_inputs.value();
348 if (allocated_pool_infos.find(pool_var) != allocated_pool_infos.end()) {
349 tir::usmp::AllocatedPoolInfo allocate_pool_info = allocated_pool_infos[pool_var];
350 if (allocate_pool_info->pool_info->is_internal) {
351 return true;
352 }
353 }
354 }
355 return false;
356 }

358 void GenerateEntrypointForUnpackedAPI(const std::string& entrypoint_name,
359 const std::string& run_func) {
360 code_ << "TVM_DLL int32_t " << run_func << "(";
361
362 {
363 std::stringstream call_args_ss;
364 if (metadata_->io_pool_allocations.empty()) {
365 for (const tir::Var& input_var : metadata_->inputs) {
366 if (input_var->type_annotation.defined()) {
367 codegen_c_base_.PrintType(input_var->type_annotation, call_args_ss);
368 } else {
369 codegen_c_base_.PrintType(input_var.dtype(), call_args_ss);
370 }
371 call_args_ss << " " << input_var->name_hint << ",";
372 }
373 for (unsigned int i = 0; i < metadata_->outputs.size(); ++i) {
374 call_args_ss << "void* output" << i << ",";
375 }
376 }
377 for (const tir::Var& pool_var : metadata_->pools) {
378 if (pool_var->type_annotation.defined()) {
379 codegen_c_base_.PrintType(pool_var->type_annotation, call_args_ss);
380 } else {
381 codegen_c_base_.PrintType(pool_var.dtype(), call_args_ss);
382 }
383 call_args_ss << " " << pool_var->name_hint << ",";
384 }
385 std::string call_args_str = call_args_ss.str();
386 call_args_str.pop_back();
387 code_ << call_args_str;
388 }
389
390 code_ << ");\n";
391 code_ << "int32_t " << entrypoint_name;
392 code_ << "(void* args, void* type_code, int num_args, void* out_value, void* "
393 "out_type_code, void* resource_handle) {\n";
394 code_ << "return " << run_func << "(";
395
396 {
397 std::stringstream call_args_ss;
398 if (metadata_->io_pool_allocations.empty()) {
399 for (unsigned int i = 0; i < metadata_->inputs.size(); ++i) {
400 call_args_ss << "((DLTensor*)(((TVMValue*)args)[" << i << "].v_handle))[0].data,";
401 }
402 for (unsigned int i = 0; i < metadata_->outputs.size(); ++i) {
403 int j = metadata_->inputs.size() + i;
404 call_args_ss << "((DLTensor*)(((TVMValue*)args)[" << j << "].v_handle))[0].data,";
405 }
406 }
407 for (const tir::Var& pool_var : metadata_->pools) {
408 if (IsInternalWorkspaceBuffer(pool_var)) {
409 call_args_ss << "&" << metadata_->pool_inputs.value()[pool_var]->pool_info->pool_name
410 << ",";
411 }
412 }
413 std::string call_args_str = call_args_ss.str();
414 call_args_str.pop_back();
415 code_ << call_args_str;
416 code_ << ");\n";
417 code_ << "}\n";
418 }

421 std::unordered_map<int, ObjectRef> GenerateRunFuncToEntryPointArgMap() {
422 std::unordered_map<int, ObjectRef> run_func_to_entry_point_args;
423 int entrypoint_arg_count = 0;
424 int run_func_arg_count = 0;
425
426 if (metadata_->io_pool_allocations.empty()) {
427 for (unsigned int i = 0; i < metadata_->inputs.size(); i++) {
428 run_func_to_entry_point_args[run_func_arg_count] = Integer(entrypoint_arg_count);
429 entrypoint_arg_count++;
430 run_func_arg_count++;
431 }
432 for (unsigned int i = 0; i < metadata_->outputs.size(); i++) {
433 run_func_to_entry_point_args[run_func_arg_count] = Integer(entrypoint_arg_count);
434 entrypoint_arg_count++;
435 run_func_arg_count++;
436 }
437 }
438 for (const tir::Var& pool_var : metadata_->pools) {
439 if (IsInternalWorkspaceBuffer(pool_var)) {
440 tir::usmp::AllocatedPoolInfo allocated_pool_info = metadata_->pool_inputs.value()[pool_var];
441 run_func_to_entry_point_args[run_func_arg_count] =
442 allocated_pool_info->pool_info->pool_name;
443 run_func_arg_count++;
444 }
445 }
446 return run_func_to_entry_point_args;
447 }

449 void GenerateEntrypointForPackedAPI(const std::string& entrypoint_name,
450 const std::string& run_func) {
451 code_ << "TVM_DLL int32_t " << run_func;
452 code_ << "(TVMValue* args, int* type_code, int num_args, TVMValue* out_value, int* "
453 "out_type_code, void* resource_handle);\n\n";
454
455 code_ << "int32_t " << entrypoint_name;
456 code_ << "(TVMValue* args, int* type_code, int num_args, TVMValue* out_value, int* "
457 "out_type_code, void* resource_handle) {\n";
458
459 // We are creating a copy of the set of pointers
460 size_t number_of_io_tensors = metadata_->inputs.size() + metadata_->outputs.size() +
461 metadata_->pools.size() - metadata_->io_pool_allocations.size();
462 code_ << "TVMValue tensors[" << number_of_io_tensors << "];\n";
463
464 std::unordered_map<int, ObjectRef> run_func_to_entry_point_args =
465 GenerateRunFuncToEntryPointArgMap();
466 for (unsigned int i = 0; i < number_of_io_tensors; i++) {
467 if (run_func_to_entry_point_args.find(i) != run_func_to_entry_point_args.end()) {
468 if (run_func_to_entry_point_args[i]->IsInstance<StringObj>()) {
469 String pool_name = Downcast<String>(run_func_to_entry_point_args[i]);
470 String pool_name_tvmv = GenerateDLTensorStructWrapper(pool_name);
471 code_ << "tensors[" << i << "] = " << pool_name_tvmv << ";\n";
472 } else {
473 code_ << "tensors[" << i << "] = ((TVMValue*)args)[" << run_func_to_entry_point_args[i]
474 << "];\n";
475 }
476 }
477 }
478
479 code_ << "return " << run_func;
480 code_ << "((void*)tensors, type_code, num_args, out_value, out_type_code, resource_handle);\n";
481 code_ << "}\n";
482 }

486 void GenerateCInterfaceEntrypoint(const std::string& entrypoint_name, const std::string& run_func,
487 const std::string& mod_name) {
488 code_ << "#include <" << mod_name << ".h>\n";
489 if (!metadata_->io_pool_allocations.empty()) {
490 const std::string input_struct_type =
491 runtime::get_name_mangled(metadata_->mod_name, "inputs");
492 Array<String> input_tensor_names;
493 for (const tir::Var& input_var : metadata_->inputs) {
494 input_tensor_names.push_back(input_var->name_hint);
495 }
496 GenerateIOWorkspaceMapFunction(input_struct_type, "map_inputs", input_tensor_names);
497 const std::string output_struct_type =
498 runtime::get_name_mangled(metadata_->mod_name, "outputs");
499 GenerateIOWorkspaceMapFunction(output_struct_type, "map_outputs", metadata_->outputs);
500 }
501 code_ << "TVM_DLL int32_t " << run_func << "(";
502 {
503 std::stringstream call_args_ss;
504 if (metadata_->io_pool_allocations.empty()) {
505 for (const tir::Var& input_var : metadata_->inputs) {
506 if (input_var->type_annotation.defined()) {
507 codegen_c_base_.PrintType(input_var->type_annotation, call_args_ss);
508 } else {
509 codegen_c_base_.PrintType(input_var.dtype(), call_args_ss);
510 }
511 call_args_ss << " " << tvm::runtime::SanitizeName(input_var->name_hint) << ",";
512 }
513 for (unsigned int i = 0; i < metadata_->outputs.size(); ++i) {
514 call_args_ss << "void* output" << i << ",";
515 }
516 }
517 for (const tir::Var& pool_var : metadata_->pools) {
518 if (pool_var->type_annotation.defined()) {
519 codegen_c_base_.PrintType(pool_var->type_annotation, call_args_ss);
520 } else {
521 codegen_c_base_.PrintType(pool_var.dtype(), call_args_ss);
522 }
523 call_args_ss << " " << pool_var->name_hint << ",";
524 }
525 for (const String& device : metadata_->devices) {
526 call_args_ss << "void* " << device << ",";
527 }
528 std::string call_args_str = call_args_ss.str();
529 call_args_str.pop_back();
530 code_ << call_args_str;
531 }
532
533 code_ << ");\n";
534 code_ << "int32_t " << entrypoint_name << "(";
535 {
536 std::stringstream call_args_ss;
537 if (metadata_->io_pool_allocations.empty()) {
538 call_args_ss << "struct " << runtime::get_name_mangled(mod_name, "inputs") << "* inputs,";
539 call_args_ss << "struct " << runtime::get_name_mangled(mod_name, "outputs") << "* outputs,";
540 }
541 if (!metadata_->pools.empty()) {
542 bool is_external_pools_present = false;
543 for (tir::Var pool_var : metadata_->pools) {
544 if (!IsInternalWorkspaceBuffer(pool_var)) {
545 is_external_pools_present = true;
546 break;
547 }
548 }
549 if (is_external_pools_present) {
550 call_args_ss << "struct " << runtime::get_name_mangled(mod_name, "workspace_pools")
551 << "* workspace_pools,";
552 }
553 }
554 if (!metadata_->devices.empty()) {
555 call_args_ss << "struct " << runtime::get_name_mangled(mod_name, "devices") << "* devices,";
556 }
557 std::string call_args_str = call_args_ss.str();
558 call_args_str.pop_back();
559 code_ << call_args_str;
560 }
561
562 code_ << ") {"
563 << "return " << run_func << "(";
564
565 {
566 std::stringstream call_args_ss;
567 if (metadata_->io_pool_allocations.empty()) {
568 for (const auto& input : metadata_->inputs) {
569 call_args_ss << "inputs->" << tvm::runtime::SanitizeName(input->name_hint) << ",";
570 }
571 for (const auto& output : metadata_->outputs) {
572 call_args_ss << "outputs->" << tvm::runtime::SanitizeName(output);
573 call_args_ss << ",";
574 }
575 }
576
577 for (const tir::Var& pool_var : metadata_->pools) {
578 String pool_name = metadata_->pool_inputs.value()[pool_var]->pool_info->pool_name;
579 if (IsInternalWorkspaceBuffer(pool_var)) {
580 call_args_ss << "&" << pool_name << ",";
581 } else {
582 call_args_ss << "workspace_pools->" << tvm::runtime::SanitizeName(pool_name) << ",";
583 }
584 }
585 for (const String& device : metadata_->devices) {
586 call_args_ss << "devices->" << device << ",";
587 }
588 std::string call_args_str = call_args_ss.str();
589 call_args_str.pop_back();
590 code_ << call_args_str;
591 }
592 code_ << ");\n";
593 code_ << "}\n";
594 }

596 void GenerateAOTDescriptor() {
597 const std::string run_func_suffix = ::tvm::runtime::symbol::tvm_module_main;
598 const std::string tvm_entrypoint_suffix = ::tvm::runtime::symbol::tvm_entrypoint_suffix;
599 const std::string run_func_mangled =
600 runtime::get_name_mangled(metadata_->mod_name, run_func_suffix);
601 const std::string entrypoint_mangled =
602 runtime::get_name_mangled(metadata_->mod_name, tvm_entrypoint_suffix);
603 const std::string network_mangled = runtime::get_name_mangled(metadata_->mod_name, "network");
604
605 code_ << "#include \"tvm/runtime/c_runtime_api.h\"\n";
606 code_ << "#ifdef __cplusplus\n";
607 code_ << "extern \"C\" {\n";
608 code_ << "#endif\n";
609
610 GenerateInternalBuffers();
611
612 if (metadata_->unpacked_api) {
613 if (metadata_->interface_api == "c") {
614 GenerateCInterfaceEntrypoint(entrypoint_mangled, run_func_mangled, metadata_->mod_name);
615 } else {
616 GenerateEntrypointForUnpackedAPI(entrypoint_mangled, run_func_mangled);
617 }
618 } else {
619 ICHECK_EQ(metadata_->interface_api, "packed")
620 << "Packed interface required for packed operators";
621 GenerateEntrypointForPackedAPI(entrypoint_mangled, run_func_mangled);
622 }
623
624 code_ << "#ifdef __cplusplus\n";
625 code_ << "}\n";
626 code_ << "#endif\n";
627 }

631 CreateFuncRegistry();
632 GenerateCrtSystemLib();
633 }
634 if (metadata_.defined() && metadata_->executor == runtime::kTvmExecutorAot) {
635 GenerateAOTDescriptor();
636 }
637 code_ << ";";
638 }

641class MetadataSerializer : public AttrVisitor {
642 public:
643 static constexpr const char* kGlobalSymbol = "kTvmgenMetadata";
644 using MetadataKind = ::tvm::runtime::metadata::MetadataKind;
645
646 MetadataSerializer() : is_first_item_{true} {}
647
648 void WriteComma() {
649 if (is_first_item_) {
650 is_first_item_ = false;
651 } else {
652 code_ << ", " << std::endl;
653 }
654 }
655
656 void WriteKey(const char* key) {
657 if (key != nullptr) {
658 code_ << " /* " << key << "*/";
659 }
660 }
661
662 void Visit(const char* key, double* value) final {
663 WriteComma();
664 code_.setf(std::ios::hex | std::ios::showbase | std::ios::fixed | std::ios::scientific,
665 std::ios::basefield | std::ios::showbase | std::ios::floatfield);
666 code_ << *value;
667 WriteKey(key);
668 }
669
670 void Visit(const char* key, int64_t* value) final {
671 WriteComma();
672 code_ << *value << "L";
673 WriteKey(key);
674 }
675
676 void Visit(const char* key, uint64_t* value) final {
677 WriteComma();
678 code_ << *value << "UL";
679 WriteKey(key);
680 }
681 void Visit(const char* key, int* value) final {
682 WriteComma();
683 code_ << *value;
684 WriteKey(key);
685 }
686 void Visit(const char* key, bool* value) final {
687 WriteComma();
688 code_ << *value;
689 WriteKey(key);
690 }
691 void Visit(const char* key, std::string* value) final {
692 WriteComma();
693 code_ << "\"" << *value << "\"";
694 WriteKey(key);
695 }
696 void Visit(const char* key, void** value) final {
697 WriteComma();
698 code_ << *value;
699 WriteKey(key);
700 }
701 void Visit(const char* key, DataType* value) final {
702 WriteComma();
703 code_ << "{" << value->code() << ", " << value->bits() << ", " << value->lanes() << "}";
704 WriteKey(key);
705 }
706
707 // Serialiding NDArray as tuple of len, data
708 void Visit(const char* key, runtime::NDArray* value) final {
709 WriteComma();
710 std::string bytes;
711 dmlc::MemoryStringStream stream(&bytes);
712 value->Save(&stream);
713 // Serializing length of the data of NDArray
714 code_ << stream.Tell();
715 WriteComma();
716 // Serializing NDArray as bytestream
717 code_ << "\"";
718 std::stringstream ss;
719 char buf[6] = {0};
720 for (uint8_t c : bytes) {
721 snprintf(buf, sizeof(buf), "\\x%02x", c);
722 ss << buf;
723 }
724 std::string as_bytes(ss.str());
725 code_ << as_bytes;
726 code_ << "\"\n";
727 }
728
729 void VisitArray(runtime::metadata::MetadataArray array) {
730 auto old_is_first_item = is_first_item_;
731 is_first_item_ = true;
732 for (unsigned int i = 0; i < array->array.size(); ++i) {
733 ObjectRef o = array->array[i];
734
735 switch (array->kind) {
736 case MetadataKind::kUint64: {
737 int64_t i = Downcast<Integer>(o).IntValue();
738 CHECK_GT(i, 0)
739 << "Metadata is of type uint64_t, but array type contains a negative number";
740 uint64_t ui = static_cast<uint64_t>(i);
741 Visit(nullptr, &ui);
742 continue;
743 }
744 case MetadataKind::kInt64: {
745 int64_t i = Downcast<Integer>(o).IntValue();
746 Visit(nullptr, &i);
747 continue;
748 }
749 case MetadataKind::kBool: {
750 bool b = Downcast<Bool>(o);
751 Visit(nullptr, &b);
752 break;
753 }
754 case MetadataKind::kString: {
755 std::string s = Downcast<String>(o);
756 Visit(nullptr, &s);
757 break;
758 }
759 case MetadataKind::kHandle:
760 CHECK(false) << "Don't know how to serialize handle";
761 break;
762
763 case MetadataKind::kMetadata: {
764 runtime::metadata::MetadataBase metadata = Downcast<runtime::metadata::MetadataBase>(o);
765 std::stringstream i_str;
766 i_str << i;
767 address_.push_back(i_str.str());
768 Visit(nullptr, &metadata);
769 address_.pop_back();
770 break;
771 }
772 default:
773 CHECK(false) << "Unknown MetadataKind for array: " << array->kind;
774 break;
775 }
776 is_first_item_ = false;
777 }
778 is_first_item_ = old_is_first_item;
779 }
780
781 void Visit(const char* key, ObjectRef* value) final {
782 const runtime::metadata::MetadataArrayNode* arr =
783 value->as<runtime::metadata::MetadataArrayNode>();
784 if (arr != nullptr) {
785 WriteComma();
786 if (key != nullptr) {
787 address_.push_back(key);
788 }
789 code_ << metadata::AddressFromParts(address_);
790 if (key != nullptr) {
791 address_.pop_back();
792 }
793 return;
794 }
795
796 runtime::metadata::MetadataBase metadata = Downcast<runtime::metadata::MetadataBase>(*value);
797 if (key != nullptr) { // NOTE: outermost call passes nullptr key
798 address_.push_back(key);
799 }
800 WriteComma();
801 code_ << "{\n";
802 is_first_item_ = true;
803 ReflectionVTable::Global()->VisitAttrs(metadata.operator->(), this);
804 code_ << "}\n";
805 if (key != nullptr) { // NOTE: outermost call passes nullptr key
806 address_.pop_back();
807 }
808 }
809
810 private:
811 void EmitCType(const runtime::metadata::MetadataArrayNode* arr, std::ostream& os) {
812 switch (arr->kind) {
813 case MetadataKind::kUint64:
814 os << "uint64_t";
815 break;
816 case MetadataKind::kInt64:
817 os << "int64_t";
818 break;
819 case MetadataKind::kBool:
820 os << "bool";
821 break;
822 case MetadataKind::kString:
823 os << "const char*";
824 break;
825 case MetadataKind::kHandle:
826 os << "void*";
827 break;
828 case MetadataKind::kMetadata:
829 os << "struct " << arr->get_element_c_struct_name();
830 break;
831 default:
832 CHECK(false) << "Unknown kind in MetadataArray: " << arr->kind
833 << " (struct_name=" << arr->get_c_struct_name() << ")";
834 break;
835 }
836 }
837
838 public:
839 void CodegenMetadata(::tvm::runtime::metadata::Metadata metadata) {
840 decl_ << "#include <inttypes.h>" << std::endl
841 << "#include <tvm/runtime/metadata_types.h>" << std::endl
842 << "#include <tvm/runtime/c_runtime_api.h>" << std::endl;
843 std::vector<metadata::DiscoverArraysVisitor::DiscoveredArray> queue;
844 metadata::DiscoverArraysVisitor array_discover{&queue};
845 array_discover.Visit(metadata::kMetadataGlobalSymbol, &metadata);
846
847 for (auto item : queue) {
848 auto struct_address = std::get<0>(item);
849 address_.push_back(struct_address);
850
851 auto arr = std::get<1>(item);
852
853 // Prepend const with everything except C-string, which needs appending.
854 code_ << "static ";
855 if (arr->kind != MetadataKind::kString) {
856 code_ << "const ";
857 }
858 EmitCType(arr.operator->(), code_);
859 if (arr->kind == MetadataKind::kString) {
860 code_ << " const";
861 }
862 code_ << " " << struct_address << "[" << arr->array.size() << "] = {" << std::endl;
863 is_first_item_ = true;
864
865 VisitArray(arr);
866 address_.pop_back();
867 code_ << "};" << std::endl;
868 }
869
870 // Finally, emit overall struct.
871 address_.push_back(metadata::kMetadataGlobalSymbol);
872 code_ << "static const struct TVMMetadata " << metadata::AddressFromParts(address_) << "[1] = {"
873 << std::endl;
874 Visit(nullptr, &metadata);
875 code_ << "};" << std::endl;
876 address_.pop_back();
877 }
878
879 std::string GetOutput() { return decl_.str() + code_.str(); }
880
881 private:
882 std::vector<std::string> address_;
883 std::stringstream decl_;
884 std::stringstream code_;
885 bool is_first_item_;
886 std::unordered_set<std::string> generated_struct_decls_;
887 std::vector<bool> is_defining_struct_;
888};

928runtime::Module CreateCSourceCrtMetadataModule(const Array<runtime::Module>& modules, Target target,
929 relay::Runtime runtime,
930 relay::backend::ExecutorCodegenMetadata metadata,

943 func_names.push_back(fname);

946 }
947
948 auto n = make_object<CSourceCrtMetadataModuleNode>(func_names, "c", target, runtime, metadata);
949 auto csrc_metadata_module = runtime::Module(n);
950 for (const auto& mod : final_modules) {
951 csrc_metadata_module.Import(mod);
952 }
953
954 return std::move(csrc_metadata_module);
