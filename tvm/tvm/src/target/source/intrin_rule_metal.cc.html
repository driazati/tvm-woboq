94// There is no erf function in Metal. When erf is used, we use fast_erf instead
95static PrimExpr DispatchFastErf(const PrimExpr& e) {
96 LOG(WARNING) << " Metal doesn't have built-in erf function. fast_erf will be used instead.";
97 const CallNode* call = e.as<CallNode>();
98 ICHECK(call != nullptr);
99 ICHECK_EQ(call->args.size(), 1);
100 PrimExpr arg = call->args[0];
101 int bits = arg.dtype().bits();
102 bool isFloat = arg.dtype().is_float();
103 PrimExpr res;
104 if (isFloat && (bits == 16 || bits == 32))
105 res = topi::fast_erf_float_expr(arg, bits);
106 else
107 LOG(FATAL) << "Unsupported type in Metal fast_erf";
108 return res;
109}
110TVM_REGISTER_OP("tir.erf").set_attr<FLowerIntrinsic>("metal.FLowerIntrinsic", DispatchFastErf);
