44class InterfaceCNode : public runtime::ModuleNode {
45 public:
46 InterfaceCNode(std::string module_name, Array<String> inputs, Array<String> outputs,
47 Array<tir::usmp::AllocatedPoolInfo> pools,
48 Map<String, tir::usmp::PoolAllocation> io_pool_allocations, Array<String> devices,
49 int workspace_size)
50 : module_name_(module_name),
51 inputs_(inputs),
52 outputs_(outputs),
53 devices_(devices),
54 pools_(FilterExternalPools(pools)),
55 io_pool_allocations_(io_pool_allocations),
56 workspace_size_(workspace_size) {}
57 const char* type_key() const final { return "h"; }
58
59 std::string GetSource(const std::string& format) final {
60 std::stringstream code;
61
62 EmitUpperHeaderGuard(code);
63 EmitBrief(code, "Input tensor pointers");
64 EmitStruct(code, "inputs", inputs_);
65 EmitBrief(code, "Output tensor pointers");
66 EmitStruct(code, "outputs", outputs_);
67
68 if (!devices_.empty()) {
69 EmitBrief(code, "Device context pointers");
70 EmitStruct(code, "devices", devices_);
71 }
72 if (!pools_.empty()) {
73 EmitBrief(code, "Workspace pool pointers");
74 Array<String> pool_names;
75 for (const tir::usmp::AllocatedPoolInfo pool : pools_) {
76 pool_names.push_back(pool->pool_info->pool_name);
77 }
78 EmitStruct(code, "workspace_pools", pool_names);
79 }
80
81 if (!io_pool_allocations_.empty()) {
82 std::string inputs_struct = ToCVariableStyle(PrefixGeneratedName({module_name_, "inputs"}));
83 EmitMapIOToPoolsFunction(code, inputs_struct, "map_inputs", inputs_);
84 std::string outputs_struct = ToCVariableStyle(PrefixGeneratedName({module_name_, "outputs"}));
85 EmitMapIOToPoolsFunction(code, outputs_struct, "map_outputs", outputs_);
86 }
87
88 EmitRunFunction(code);
89 // Emit workspace
90 EmitIntegerValueMacro(code, "Workspace size", "WORKSPACE_SIZE", workspace_size_);
91 // Emit memory pool sizes
92 for (const tir::usmp::AllocatedPoolInfo pool : pools_) {
93 String pool_name = pool->pool_info->pool_name;
94 Integer pool_size = pool->allocated_size;
95 if (const auto* pool_info = pool->pool_info.as<ConstantPoolInfoNode>()) {
96 EmitConstantPool(code, SanitizeName(pool_name) + " initialization data", pool_info);
97 } else {
98 EmitIntegerValueMacro(code, SanitizeName(pool_name) + " size",
99 SanitizeName(pool_name) + _macro_workspace_pool_size_postfix,
100 pool_size->value);
101 }
102 }
103 EmitLowerHeaderGuard(code);
104
105 return code.str();
106 }
107
108 PackedFunc GetFunction(const std::string& name, const ObjectPtr<Object>& sptr_to_self) final {
109 return PackedFunc();
110 }
111
112 private:
113 constexpr static const char* _macro_workspace_pool_size_postfix = "_WORKSPACE_POOL_SIZE";
114 constexpr static const char* _macro_constant_pool_size_postfix = "_CONSTANT_POOL_SIZE";
115 constexpr static const char* _macro_constant_pool_data_postfix = "_CONSTANT_POOL_DATA";
116
117 void EmitUpperHeaderGuard(std::stringstream& code_stream) {
118 std::string header_guard_name = ToCConstantStyle(PrefixGeneratedName({module_name_, "H"}));
119 code_stream << "#ifndef " << header_guard_name << "_\n"
120 << "#define " << header_guard_name << "_\n"
121 << "#include <stdint.h>\n\n"
122 << "#ifdef __cplusplus\n"
123 << "extern \"C\" {\n"
124 << "#endif\n\n";
125 }
126
127 void EmitLowerHeaderGuard(std::stringstream& code_stream) {
128 std::string header_guard_name = ToCConstantStyle(PrefixGeneratedName({module_name_, "H"}));
129 code_stream << "\n#ifdef __cplusplus\n"
130 << "}\n"
131 << "#endif\n\n"
132 << "#endif // " << header_guard_name << "_\n";
133 }
134
135 void EmitBrief(std::stringstream& code_stream, const std::string& description) {
136 code_stream << "/*!\n"
137 << " * \\brief " << description << " for TVM module \"" << module_name_ << "\" \n"
138 << " */\n";
139 }
140
141 void EmitStruct(std::stringstream& code_stream, const std::string& suffix,
142 Array<String> properties) {
143 std::string struct_name = ToCVariableStyle(PrefixGeneratedName({module_name_, suffix}));
144 code_stream << "struct " << struct_name << " {\n";
145
146 std::vector<std::string> sanitized_properties;
147 for (const String& property : properties) {
148 std::string sanitized_property = SanitizeName(property);
149 ICHECK(std::find(sanitized_properties.begin(), sanitized_properties.end(),
150 sanitized_property) == sanitized_properties.end())
151 << "Sanitized input tensor name clash" << sanitized_property;
152 code_stream << " void* " << sanitized_property << ";\n";
153 sanitized_properties.push_back(sanitized_property);
154 }
155 code_stream << "};\n\n";
156 }
157
158 void EmitIntegerValueMacro(std::stringstream& code_stream, const std::string& brief_description,
159 const std::string& macro_name, int macro_value) {
160 EmitBrief(code_stream, brief_description);
161 std::string macro_name_prefixed =
162 ToCConstantStyle(PrefixGeneratedName({module_name_, macro_name}));
163 code_stream << "#define " << macro_name_prefixed << " " << macro_value << "\n";
164 }
165
166 void EmitConstantPool(std::stringstream& code_, const std::string& brief_description,
167 const ConstantPoolInfoNode* pool_info) {
168 EmitBrief(code_, brief_description);
169 std::string name_prefixed =
170 ToCConstantStyle(PrefixGeneratedName({module_name_, SanitizeName(pool_info->pool_name)}));
171
172 if (pool_info->constant_info_array.size() > 0) {
173 std::vector<ConstantInfo> const_info_vec(pool_info->constant_info_array.begin(),
174 pool_info->constant_info_array.end());
175 std::sort(const_info_vec.begin(), const_info_vec.end(),
176 [](const ConstantInfo& a, const ConstantInfo& b) {
177 return a->byte_offset->value < b->byte_offset->value;
178 });
179 int64_t accumulated_pool_len =
180 const_info_vec.back()->byte_offset.IntValue() +
181 runtime::GetDataSize(*const_info_vec.back()->data.operator->());
182 const auto& accumulated_pool = runtime::NDArray::Empty(
183 {accumulated_pool_len}, DataType::UInt(8), const_info_vec.back()->data->device);
184 for (const auto& const_info : const_info_vec) {
185 const auto& data = const_info->data;
186 const auto& offs = const_info->byte_offset;
187 data.CopyToBytes(static_cast<uint8_t*>(accumulated_pool->data) + offs.IntValue(),
188 runtime::GetDataSize(*data.operator->()));
189 }
190
191 code_ << "#define " << name_prefixed << _macro_constant_pool_size_postfix << " "
192 << accumulated_pool_len << "\n";
193 code_ << "#define " << name_prefixed << _macro_constant_pool_data_postfix << " \\\n";
194 codegen::NDArrayDataToC(accumulated_pool, 4, code_, "\\\n");
195 code_ << '\n';
196
197 } else {
198 LOG(FATAL) << "No constant data in constant pool found "
199 << PrettyPrint(GetRef<ObjectRef>(pool_info));
200 }
201 }
202
203 void EmitRunFunction(std::stringstream& code_stream) {
204 std::string run_function = ToCVariableStyle(PrefixGeneratedName({module_name_, "run"}));
205 std::string inputs_struct = ToCVariableStyle(PrefixGeneratedName({module_name_, "inputs"}));
206 std::string outputs_struct = ToCVariableStyle(PrefixGeneratedName({module_name_, "outputs"}));
207 std::string devices_struct = ToCVariableStyle(PrefixGeneratedName({module_name_, "devices"}));
208 std::string pools_struct =
209 ToCVariableStyle(PrefixGeneratedName({module_name_, "workspace_pools"}));
210
211 code_stream << "/*!\n"
212 << " * \\brief entrypoint function for TVM module \"" << module_name_ << "\"\n";
213 if (io_pool_allocations_.empty()) {
214 code_stream << " * \\param inputs Input tensors for the module \n";
215 code_stream << " * \\param outputs Output tensors for the module \n";
216 }
217
218 if (!pools_.empty()) {
219 code_stream << " * \\param workspace_pools Workspace memory pool pointers for the module \n";
220 }
221 if (!devices_.empty()) {
222 code_stream << " * \\param devices Device context pointers for the module \n";
223 }
224
225 code_stream << " */\n"
226 << "int32_t " << run_function << "(\n";
227
228 std::stringstream call_args_ss;
229 if (io_pool_allocations_.empty()) {
230 call_args_ss << " struct " << inputs_struct << "* inputs,\n";
231 call_args_ss << " struct " << outputs_struct << "* outputs,\n";
232 }
233 if (!pools_.empty()) {
234 call_args_ss << " struct " << pools_struct << "* workspace_pools,\n";
235 }
236 if (!devices_.empty()) {
237 call_args_ss << " struct " << devices_struct << "* devices,\n";
238 }
239 std::string call_args_str = call_args_ss.str();
240 call_args_str.pop_back();
241 call_args_str.pop_back();
242 code_stream << call_args_str << "\n);\n";
243 }
244
245 void EmitMapIOToPoolsFunction(std::stringstream& code_stream, const std::string& struct_type,
246 const std::string& function_name,
247 const Array<String>& tensor_names) {
248 code_stream << "/*!\n"
249 << " * \\brief Maps I/O inside the workspace pools for TVM module \""
250 << module_name_ << "\"\n"
251 << " * \\param workspace_pools Workspace memory pool struct for the module \n"
252 << " * \\return I/O tensor struct for the module \n";
253 std::string map_function = ToCVariableStyle(PrefixGeneratedName({module_name_, function_name}));
254 code_stream << " */\n"
255 << "struct " << struct_type << " " << map_function << "(\n";
256 std::string pools_struct =
257 ToCVariableStyle(PrefixGeneratedName({module_name_, "workspace_pools"}));
258 code_stream << " struct " << pools_struct << "* workspace_pools\n";
259 code_stream << ");\n\n";
260 }
261
262 Array<tir::usmp::AllocatedPoolInfo> FilterExternalPools(
263 const Array<tir::usmp::AllocatedPoolInfo>& pools) {
264 Array<tir::usmp::AllocatedPoolInfo> external_pools;
265 for (tir::usmp::AllocatedPoolInfo pool : pools) {
266 if (!pool->pool_info->is_internal) {
267 external_pools.push_back(pool);
268 }
269 }
270 return external_pools;
271 }
272
273 std::string module_name_;
274 Array<String> inputs_;
275 Array<String> outputs_;
276 Array<String> devices_;
277 Array<tir::usmp::AllocatedPoolInfo> pools_;
278 Map<String, tir::usmp::PoolAllocation> io_pool_allocations_;
279 int workspace_size_;
280};
281
282runtime::Module InterfaceCCreate(std::string module_name, Array<String> inputs,
283 Array<String> outputs, Array<tir::usmp::AllocatedPoolInfo> pools,
284 Map<String, tir::usmp::PoolAllocation> io_pool_allocations,
285 Array<String> devices, int workspace_size) {
286 auto n = make_object<InterfaceCNode>(module_name, inputs, outputs, pools, io_pool_allocations,
287 devices, workspace_size);
288 return runtime::Module(n);
289}
