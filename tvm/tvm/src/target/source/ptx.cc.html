38/*!
39 * \brief PTX data type.
40 * \note
41 * PTX fundamental data types:
42 * https://docs.nvidia.com/cuda/parallel-thread-execution/index.html#fundamental-types
43 * PTX matrix data types:
44 * https://docs.nvidia.com/cuda/parallel-thread-execution/index.html#warp-level-matrix-data-types
45 */
46enum class DataType : int {
47 kInt4 = 0,
48 kUInt4 = 1,
49 kInt8 = 2,
50 kUInt8 = 3,
51 kInt16 = 4,
52 kUInt16 = 5,
53 kInt32 = 6,
54 kUInt32 = 7,
55 kInt64 = 8,
56 kUInt64 = 9,
57 kFloat16 = 10,
58 kBFloat16 = 11,
59 kFloat16x2 = 12,
60 kFloat32 = 13,
61 kTensorFloat32 = 14,
62 kFloat64 = 15,
63 kBit1 = 16,
64 kBit8 = 17,
65 kBit16 = 18,
66 kBit32 = 19,
67 kBit64 = 20,
68};
69
70static const char* dtype_str[] = {".s4", ".u4", ".s8", ".u8", ".s16", ".u16", ".s32",
71 ".u32", ".s64", ".u64", ".f16", ".bf16", ".f16x2", ".f32",
72 ".tf32", ".f64", ".b1", ".b8", ".b16", ".b32", ".b64"};
73static const uint32_t num_bits[] = {4, 4, 8, 8, 16, 16, 32, 32, 64, 64, 16,
74 16, 32, 32, 32, 64, 1, 8, 16, 32, 64};

76/*!
77 * \brief Create PTX data type from string.
78 */
79inline DataType DTypeFromString(const std::string str) {
80 if (str == "int4" || str == ".s4") {
81 return DataType::kInt4;
82 } else if (str == "uint4" || str == ".u4") {
83 return DataType::kUInt4;
84 } else if (str == "int8" || str == ".s8") {
85 return DataType::kInt8;
86 } else if (str == "uint8" || str == ".u8") {
87 return DataType::kUInt8;
88 } else if (str == "int16" || str == ".s16") {
89 return DataType::kInt16;
90 } else if (str == "uint16" || str == ".u16") {
91 return DataType::kUInt16;
92 } else if (str == "int32" || str == ".s32") {
93 return DataType::kInt32;
94 } else if (str == "uint32" || str == ".u32") {
95 return DataType::kUInt32;
96 } else if (str == "int64" || str == ".s64") {
97 return DataType::kInt64;
98 } else if (str == "uint64" || str == ".u64") {
99 return DataType::kUInt64;
100 } else if (str == "float16" || str == "fp16" || str == ".f16") {
101 return DataType::kFloat16;
102 } else if (str == "bfloat16" || str == "bf16") {
103 return DataType::kBFloat16;
104 } else if (str == ".f16x2") {
105 return DataType::kFloat16x2;
106 } else if (str == "float32" || str == "fp32" || str == ".f32") {
107 return DataType::kFloat32;
108 } else if (str == "tf32") {
109 return DataType::kTensorFloat32;
110 } else if (str == "float64" || str == "fp64" || str == ".f64") {
111 return DataType::kFloat64;
112 } else if (str == "int1" || str == ".b1") {
113 return DataType::kBit1;
114 } else if (str == ".b8") {
115 return DataType::kBit8;
116 } else if (str == ".b16") {
117 return DataType::kBit16;
118 } else if (str == ".b32") {
119 return DataType::kBit32;
120 } else if (str == ".b64") {
121 return DataType::kBit64;
122 } else {
123 LOG(FATAL) << "Unrecognized PTX data type " << str;
124 return DataType(0);
125 }
126}

128/*!
129 * \brief Get the string representation of given PTX data type.

131inline std::string DTypeToString(DataType dtype) { return dtype_str[static_cast<int>(dtype)]; }

138/*!
139 * \brief Extract the value m, n, k from string m*n*k*
140 */
141inline std::tuple<int, int, int> ParseMMAShape(const std::string& str) {
142 size_t pos_m = str.find("m"), pos_n = str.find("n"), pos_k = str.find("k");
143 CHECK(pos_m != str.npos && pos_n != str.npos && pos_k != str.npos)
144 << "Cannot parse MMA shape " << str;
145 int m = std::stoi(str.substr(pos_m + 1, pos_n - pos_m - 1)),
146 n = std::stoi(str.substr(pos_n + 1, pos_k - pos_n - 1)), k = std::stoi(str.substr(pos_k + 1));
147 return std::make_tuple(m, n, k);
148}

189 return m == other.m && n == other.n && k == other.k && dtype_mul == other.dtype_mul &&

191 }

239/*!
240 * \brief Check whether the multiplicand data type and accumulator data type is valid for MMA
241 * computation.
242 * \param dtype_a The data type of multiplicand a.
243 * \param dtype_b The data type of multiplicand b.
244 * \param dtype_c The data type of accumulator c.
245 * \note Reference:
246 * https://docs.nvidia.com/cuda/parallel-thread-execution/index.html#warp-level-matrix-data-types
247 */
248void CheckMMADTypeCompatible(DataType dtype_a, DataType dtype_b, DataType dtype_c) {
249 std::string ab_not_match_err_str = "The multiplicands' data type " + DTypeToString(dtype_a) +
250 DTypeToString(dtype_b) + " do not match.";
251 // check a and b
252 switch (dtype_a) {
253 case DataType::kBit1:
254 case DataType::kFloat16:
255 case DataType::kBFloat16:
256 case DataType::kTensorFloat32:
257 case DataType::kFloat64:
258 CHECK(dtype_a == dtype_b) << ab_not_match_err_str;
259 break;
260 case DataType::kInt4:
261 case DataType::kUInt4:
262 CHECK(dtype_b == DataType::kInt4 || dtype_b == DataType::kUInt4) << ab_not_match_err_str;
263 break;
264 case DataType::kInt8:
265 case DataType::kUInt8:
266 CHECK(dtype_b == DataType::kInt8 || dtype_b == DataType::kUInt8) << ab_not_match_err_str;
267 break;
268 default:
269 CHECK(false) << "Invalid multiplicand data types: " << DTypeToString(dtype_a)
270 << DTypeToString(dtype_b);
271 }
272 // check a,b and c
273 switch (dtype_a) {
274 case DataType::kBit1:
275 case DataType::kInt4:
276 case DataType::kUInt4:
277 case DataType::kInt8:
278 case DataType::kUInt8:
279 CHECK(dtype_c == DataType::kInt32)
280 << "For multiplicand data type " << DTypeToString(dtype_a) << DTypeToString(dtype_b)
281 << ", accumulator data type should be s32.";
282 break;
283 case DataType::kFloat16:
284 CHECK(dtype_c == DataType::kFloat16 || dtype_c == DataType::kFloat32)
285 << "For multiplicand data type f16, accumulator data type should be f16/f32.";
286 break;
287 case DataType::kBFloat16:
288 case DataType::kTensorFloat32:
289 CHECK(dtype_c == DataType::kFloat32)
290 << "For multiplicand data type bf16/tf32, accumulator data type can only be f32.";
291 break;
292 case DataType::kFloat64:
293 CHECK(dtype_c == DataType::kFloat64)
294 << "For multiplicand data type f64, accumulator data type can only be f64.";
295 break;
296 default:
297 CHECK(false) << "Invalid multiplicand/accumulator data types: " << DTypeToString(dtype_a)
298 << DTypeToString(dtype_b) << DTypeToString(dtype_c) << ".";
299 }
300}

302/*!
303 * \brief Check whether the given configuration is valid for MMA computation.
304 * \param m The M in mMnNkK of MMA instructions.
305 * \param n The N in mMnNkK of MMA instructions.
306 * \param k The K in mMnNkK of MMA instructions.
307 * \param layout_a The layout of multiplicand A (row/col).
308 * \param layout_b The layout of multiplicand B (row/col).
309 * \param dtype_a The data type of multiplicand A.
310 * \param dtype_b The data type of multiplicand B.
311 * \param dtype_c The data type of accumulator C.
312 * \param bit_op The bit operator for 1-bit MMA computation, can be "xor"/"and" or ""(if it's not
313 * 1-bit MMA).
314 * \param sparse Whether it's Sparse MMA or not.
315 * \param saturate Whether saturate output or not.
316 */
317void CheckMMAConfigValidity(int m, int n, int k, LayoutType layout_a, LayoutType layout_b,
318 DataType dtype_a, DataType dtype_b, DataType dtype_c,
319 const std::string& bit_op, bool sparse, bool saturate) {
320 CHECK(bit_op == "xor" || bit_op == "and" || bit_op == "")
321 << "Unrecognized 1-bit operation " << bit_op << " , can only be xor/and.";
322 bool use_bit_op = !bit_op.empty();
323 if (use_bit_op) {
324 CHECK(dtype_a == DataType::kBit1) << "Bit operator is only compatible with 1-bit multiplicand.";
325 }
326 CheckMMADTypeCompatible(dtype_a, dtype_b, dtype_c);
327 if (saturate) {
328 CHECK(dtype_a == DataType::kInt4 || dtype_a == DataType::kUInt4 || dtype_a == DataType::kInt8 ||
329 dtype_a == DataType::kUInt8)
330 << "Output saturation only applicable to multiplicand type s4/u4/s8/u8.";
331 }
332
333 if (!(m == 8 && n == 8 && k == 4 && dtype_a == ptx::DataType::kFloat16)) {
334 // Only MMA on m8n8k4 for fp16 supports customized layouts.
335 CHECK(layout_a == LayoutType::kRowMajor && layout_b == LayoutType::kColumnMajor)
336 << "Invalid layout combination " << LayoutTypeToString(layout_a) << ","
337 << LayoutTypeToString(layout_b) << ".";
338 }
339
340 MMAConfig config(m, n, k, dtype_a, use_bit_op, sparse);
341 bool match = false;
342 for (const MMAConfig& valid_config : valid_mma_configs) {
343 if (config == valid_config) {
344 match = true;
345 break;
346 }
347 }
348 CHECK(match) << "Cannot find matched MMA configurations.";
349}

351/*!
352 * \brief Fragment attributes
353 */
354class FragAttrs {
355 public:
356 explicit FragAttrs(char reg_type, uint32_t size, std::string ptr_type)
357 : reg_type(reg_type), size(size), ptr_type(ptr_type) {}
358 /*! \brief PTX register type */
359 char reg_type;
360 /*! \brief Fragment size */
361 uint32_t size;
362 /*! \brief Fragment pointer type */
363 std::string ptr_type;
364};
365
366/*!
367 * \brief Fragment attributes of given data type.
368 */
369inline FragAttrs GetFragAttrs(DataType dtype) {
370 switch (dtype) {
371 case DataType::kBit1:
372 case DataType::kInt4:
373 case DataType::kUInt4:
374 case DataType::kInt8:
375 case DataType::kUInt8:
376 case DataType::kBit16:
377 case DataType::kFloat16: // .f16x2 register
378 case DataType::kBFloat16:
379 case DataType::kTensorFloat32:
380 return FragAttrs('r', 32, "(unsigned *)");
381 case DataType::kInt32:
382 return FragAttrs('r', 32, "(int *)");
383 case DataType::kFloat32:
384 return FragAttrs('f', 32, "(float *)");
385 case DataType::kFloat64:
386 return FragAttrs('d', 64, "(double *)");
387 default:
388 ICHECK(false) << DTypeToString(dtype) << " is not matrix data type in MMA.";
389 return FragAttrs('\0', 0, "");
390 }
391}

395/*!
396 * \brief Replace patterns with replacement strings.
397 * \note should use std::format instead when codebase is ported to C++20.
398 */
399class Replacer {
400 public:
401 void register_rule(const std::string& pattern, const std::string& replacement) {
402 _rules.emplace_back(pattern, replacement);
403 }
404 std::string rewrite(std::string str) {
405 for (auto&& rule : _rules) {
406 auto [pattern, replacement] = rule;
407 size_t len = pattern.size();
408 size_t new_len = replacement.size();
409 size_t pos = str.find(pattern);
410 while (pos != std::string::npos) {
411 str = str.replace(pos, len, replacement);
412 pos = str.find(pattern, pos + new_len);
413 }
414 }
415 return str;
416 }
417 void empty_rules() { _rules.clear(); }
418
419 private:
420 std::vector<std::pair<std::string, std::string>> _rules;
421};

423/*!
424 * \brief Get the number of MMA computations for given shape and datatype.
425 */
426inline uint32_t GetNumMMAComputations(int m, int n, int k, ptx::DataType dtype) {
427 if (m == 8 && n == 8 && k == 4 && dtype == ptx::DataType::kFloat16) {
428 // MMA for m8n8k4 on fp16 would launch 4 MMA computations instead of one.
429 return 4;
430 } else {
431 return 1;
432 }
433}

435/*!
436 * \brief Return template string, input operands string and output operands string.
437 * \param m The M in mMnNkK of MMA instructions.
438 * \param n The N in mMnNkK of MMA instructions.
439 * \param k The K in mMnNkK of MMA instructions.
440 * \param dtype_a The data type of multiplicand a.
441 * \param dtype_b The data type of multiplicand b.
442 * \param dtype_c The data type of accumulator c.
443 * \param sparse Whether it's Sparse MMA or not.
444 */
445inline std::tuple<std::string, std::string, std::string> GetMMAOperands(int m, int n, int k,
446 ptx::DataType dtype_a,
447 ptx::DataType dtype_b,
448 ptx::DataType dtype_c,
449 bool sparse) {
450 std::stringstream templates, inputs, outputs;
451 const ptx::FragAttrs frag_attr_a = ptx::GetFragAttrs(dtype_a),
452 frag_attr_b = ptx::GetFragAttrs(dtype_b),
453 frag_attr_c = ptx::GetFragAttrs(dtype_c);
454 constexpr uint32_t warp_size = 32;
455 const uint32_t threads = warp_size / GetNumMMAComputations(m, n, k, dtype_a);
456 const int num_operands_a =
457 (m * k) * ptx::DTypeBits(dtype_a) / frag_attr_a.size / threads / (sparse ? 2 : 1),
458 num_operands_b = (k * n) * ptx::DTypeBits(dtype_b) / frag_attr_b.size / threads,
459 num_operands_c = (m * n) * ptx::DTypeBits(dtype_c) / frag_attr_c.size / threads;
460
461 // generate templates;
462 int arg_counter = 0;
463 templates << "{"
464 << "%" << arg_counter++;
465 for (int i = 1; i < num_operands_c; ++i) {
466 templates << ", %" << arg_counter++;
467 }
468 templates << "}, {"
469 << "%" << arg_counter++;
470 for (int i = 1; i < num_operands_a; ++i) {
471 templates << ", %" << arg_counter++;
472 }
473 templates << "}, {"
474 << "%" << arg_counter++;
475 for (int i = 1; i < num_operands_b; ++i) {
476 templates << ", %" << arg_counter++;
477 }
478 templates << "}, {"
479 << "%" << arg_counter++;
480 for (int i = 1; i < num_operands_c; ++i) {
481 templates << ", %" << arg_counter++;
482 }
483 templates << "}";
484 // templates of metadata and sparse selector for sparse mma.
485 if (sparse) {
486 templates << ", %" << (arg_counter++) << ", F";
487 }
488
489 // generate inputs
490 for (int i = 0; i < num_operands_a; ++i) {
491 if (i != 0) {
492 inputs << ", ";
493 }
494 inputs << "\"" << frag_attr_a.reg_type << "\"((" << frag_attr_a.ptr_type << "(A))[" << i
495 << "])";
496 }
497 for (int i = 0; i < num_operands_b; ++i) {
498 inputs << ", \"" << frag_attr_b.reg_type << "\"((" << frag_attr_b.ptr_type << "(B))[" << i
499 << "])";
500 }
501 for (int i = 0; i < num_operands_c; ++i) {
502 inputs << ", \"" << frag_attr_c.reg_type << "\"((" << frag_attr_c.ptr_type << "(C))[" << i
503 << "])";
504 }
505 // input of metadata for sparse mma.
506 if (sparse) {
507 inputs << ", \"r\"(((unsigned *)(E))[0])";
508 }
509
510 // generate outputs
511 for (int i = 0; i < num_operands_c; ++i) {
512 if (i != 0) {
513 outputs << ",";
514 }
515 outputs << " \"=" << frag_attr_c.reg_type << "\"((" << frag_attr_c.ptr_type << "(D))[" << i
516 << "])";
517 }
518 return std::make_tuple(templates.str(), inputs.str(), outputs.str());
519}

521std::string PrintMMAAssembly(const std::string& shape, const std::string& A_layout,
522 const std::string& B_layout, const std::string& A_dtype,
523 const std::string& B_dtype, const std::string& C_dtype,
524 const std::string& a_ptr, const std::string& a_elem_offset,
525 const std::string& b_ptr, const std::string& b_elem_offset,
526 const std::string& c_ptr, const std::string& c_elem_offset,
527 const std::string& metadata, const std::string& metadata_offset,
528 const std::string& sparsity_selector, const std::string& bit_op,
529 bool sparse, bool saturate) {
530 ptx::DataType dtype_a = ptx::DTypeFromString(A_dtype), dtype_b = ptx::DTypeFromString(B_dtype),
531 dtype_c = ptx::DTypeFromString(C_dtype);
532 ptx::LayoutType layout_a = ptx::LayoutTypeFromString(A_layout),
533 layout_b = ptx::LayoutTypeFromString(B_layout);
534 auto [m, n, k] = ptx::ParseMMAShape(shape);
535 CheckMMAConfigValidity(m, n, k, layout_a, layout_b, dtype_a, dtype_b, dtype_c, bit_op, sparse,
536 saturate);
537 std::string asm_code = R"(
538 {
539 __asm__ __volatile__(
540 "mma{.sparse}.sync.aligned{.shape}{.alayout}{.blayout}{.saturate}{.dtype}{.atype}{.btype}{.ctype}{.bitop}"
541 "{templates};\n"
542 : {outputs}
543 : {inputs});
544 }
545)";
546 auto [templates_str, inputs_str, outputs_str] =
547 GetMMAOperands(m, n, k, dtype_a, dtype_b, dtype_c, sparse);
548
549 // replace patterns
550 Replacer replacer;
551 replacer.register_rule("{.sparse}", sparse ? ".sp" : "");
552 replacer.register_rule("{.shape}", "." + shape);
553 replacer.register_rule("{.saturate}", saturate ? ".satfinite" : "");
554 replacer.register_rule("{.alayout}", "." + A_layout);
555 replacer.register_rule("{.blayout}", "." + B_layout);
556 replacer.register_rule("{.atype}", ptx::DTypeToString(dtype_a));
557 replacer.register_rule("{.btype}", ptx::DTypeToString(dtype_b));
558 replacer.register_rule("{.ctype}", ptx::DTypeToString(dtype_c));
559 replacer.register_rule("{.dtype}", ptx::DTypeToString(dtype_c));
560 replacer.register_rule("{.bitop}", bit_op.empty() ? "" : "." + bit_op + ".popc");
561 replacer.register_rule("{templates}", templates_str);
562 replacer.register_rule("{outputs}", outputs_str);
563 replacer.register_rule("{inputs}", inputs_str);
564 asm_code = replacer.rewrite(asm_code);
565 replacer.empty_rules();
566 replacer.register_rule("A", a_ptr + " + " + a_elem_offset);
567 replacer.register_rule("B", b_ptr + " + " + b_elem_offset);
568 replacer.register_rule("C", c_ptr + " + " + c_elem_offset);
569 replacer.register_rule("D", c_ptr + " + " + c_elem_offset);
570 replacer.register_rule("E", metadata + " + " + metadata_offset);
571 replacer.register_rule("F", sparsity_selector);
572 asm_code = replacer.rewrite(asm_code);
573 return asm_code;
574}
575
576inline std::tuple<std::string, std::string> GetLoadMatrixOperands(
577 int num, const std::string& local_ptr, const std::string& local_elem_offset) {
578 std::stringstream templates, outputs;
579 int arg_counter = 0;
580 // generate templates
581 templates << "{%" << arg_counter++;
582 for (int i = 1; i < num; ++i) {
583 templates << ", %" << arg_counter++;
584 }
585 templates << "}, [%" << arg_counter++ << "]";
586 // generate outputs
587 std::string ptr_type = "(unsigned *)";
588 for (int i = 0; i < num; ++i) {
589 if (i != 0) {
590 outputs << ", ";
591 }
592 outputs << "\"=r\"((" << ptr_type << "(" << local_ptr << " + " << local_elem_offset << "))["
593 << i << "])";
594 }
595 return std::make_tuple(templates.str(), outputs.str());
596}
597
598std::string PrintLoadMatrixAssembly(bool trans, int num, const std::string& type,
599 const std::string& local_ptr,
600 const std::string& local_elem_offset,
601 const std::string& smem_ptr,
602 const std::string& smem_elem_offset) {
603 CHECK(num == 1 || num == 2 || num == 4) << "ldmatrix only accept loading 1/2/4 matrices.";
604 ptx::DataType data_type = ptx::DTypeFromString(type);
605 CHECK(data_type == ptx::DataType::kBit16) << "ldmatrix only accept matrix with type .b16.";
606 std::string asm_code = R"(
607 {
608 unsigned int addr;
609 __asm__ __volatile__(
610 "{ .reg .u64 addr; cvta.to.shared.u64 addr, %1; cvt.u32.u64 %0, addr; }\n"
611 : "=r"(addr)
612 : "l"((void *)({smem_addr}))
613 );
614 __asm__ __volatile__(
615 "ldmatrix.sync.aligned{.shape}{.num}{.trans}{.ss}{.type}"
616 "{templates};\n"
617 : {outputs}
618 : "r"(addr)
619 );
620 }
621)";
622 auto [templates_str, outputs_str] = GetLoadMatrixOperands(num, local_ptr, local_elem_offset);
623
624 Replacer replacer;
625 replacer.register_rule("{.shape}", ".m8n8");
626 replacer.register_rule("{.num}", ".x" + std::to_string(num));
627 replacer.register_rule("{.trans}", trans ? ".trans" : "");
628 replacer.register_rule("{.ss}", ".shared");
629 replacer.register_rule("{.type}", ptx::DTypeToString(data_type));
630 replacer.register_rule("{smem_addr}", smem_ptr + " + " + smem_elem_offset);
631 replacer.register_rule("{templates}", templates_str);
632 replacer.register_rule("{outputs}", outputs_str);
633 asm_code = replacer.rewrite(asm_code);
634 return asm_code;
635}

637std::string PrintCpAsyncAssembly(const std::string& shared_ptr,
638 const std::string& shared_elem_offset,
639 const std::string& global_ptr,
640 const std::string& global_elem_offset, const std::string& bytes) {
641 std::string asm_code = R"(
642 {
643 unsigned int addr;
644 __asm__ __volatile__(
645 "{ .reg .u64 addr; cvta.to.shared.u64 addr, %1; cvt.u32.u64 %0, addr; }\n"
646 : "=r"(addr)
647 : "l"((void *)({smem_addr}))
648 );
649 __asm__ __volatile__(
650 "cp.async.{cg_or_ca}.shared.global [%0], [%1], %2;"
651 :: "r"(addr), "l"((void*)({global_ptr})), "n"({bytes})
652 );
653 }
654)";
655 Replacer replacer;
656 replacer.register_rule("{smem_addr}", shared_ptr + " + " + shared_elem_offset);
657 replacer.register_rule("{global_ptr}", global_ptr + " + " + global_elem_offset);
658 replacer.register_rule("{bytes}", bytes);
659 replacer.register_rule("{cg_or_ca}", bytes == "16" ? "cg" : "ca");
660 asm_code = replacer.rewrite(asm_code);
661 return asm_code;
662}
