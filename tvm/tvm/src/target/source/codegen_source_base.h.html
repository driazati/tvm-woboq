44/*!
45 * \brief A base class to generate source code.
46 * Contains helper utilities to generate nest and ssa form.
47 */
48class CodeGenSourceBase {
49 public:
50 virtual ~CodeGenSourceBase() = default;
51 /*!
52 * \brief Register constant value appeared in expresion tree
53 * This avoid generated a ssa id for each appearance of the value
54 * \param value The constant value.
55 */
56 void MarkConst(std::string value);
57 /*!
58 * Print Type representation of type type.
59 * \param t The type representation.
60 * \param os The stream to print the ctype into
61 */
62 virtual void PrintType(DataType type, std::ostream& os); // NOLINT(*)
63 /*!
64 * Print Type representation of type type.
65 * \param type The type representation.
66 * \param os The stream to print the ctype into
67 */
68 virtual void PrintType(const Type& type, std::ostream& os); // NOLINT(*)
69
70 protected:
71 /*! \brief entry in ssa assign map */
72 struct SSAEntry {
73 /*! \brief The value id */
74 std::string vid;
75 /*! \brief The scope id, used to check if this entry is invalid. */
76 int scope_id;
77 };
78 /*! \brief Clear the states that might relates to function generation */
79 void ClearFuncState();
80 /*! \brief print the current indented value */
81 void PrintIndent();
82 /*!
83 * \brief Allocate a variable name for a newly defined var.
84 * \param v The variable.
85 * \return the variable name.
86 */
87 std::string AllocVarID(const tir::VarNode* v);
88 /*!
89 * \brief Get a variable name.
90 * \param v The variable.
91 * \return the variable name.
92 */
93 std::string GetVarID(const tir::VarNode* v) const;
94 /*!
95 * \brief Get the SSA ID corresponds to src
96 * If necessary, generate new assignment
97 * \param src The source expression
98 * \param t The type of the expression.
99 */
100 std::string SSAGetID(std::string src, DataType t);
101 /*!
102 * \brief mark the beginning of a new scope
103 * \return The scope id.
104 */
105 int BeginScope();
106 /*!
107 * \brief mark the end of an old scope.
108 * \param scope_id The scope id to be ended.
109 */
110 void EndScope(int scope_id);
111 /*!
112 * \brief Print assignment of src to the id in ssa entry.
113 * \param target id of target variable.
114 * \param src The source expression.
115 * \param t The type of target.
116 */
117 virtual void PrintSSAAssign(const std::string& target, const std::string& src, DataType t) = 0;
118
119 /*! \brief the declaration stream */
120 std::ostringstream decl_stream;
121 /*! \brief the stream to be printed */
122 std::ostringstream stream;
123 /*! \brief name of each variable */
124 std::unordered_map<const tir::VarNode*, std::string> var_idmap_;
125 /*! \brief NameSupply for allocation */
126 NameSupply name_supply_ = NameSupply("");
127
128 private:
129 /*! \brief assignment map of ssa */
130 std::unordered_map<std::string, SSAEntry> ssa_assign_map_;
131 /*! \brief array to check whether we are inside certain scope */
132 std::vector<bool> scope_mark_;
133 /*! \brief The current indentation value */
134 int indent_{0};
135};

137/*!
138 * \brief Create a source module for viewing.
139 * \param code The code to be viewed.
140 * \param fmt The code. format.
141 */
142runtime::Module SourceModuleCreate(std::string code, std::string fmt);
143
144/*!
145 * \brief Create a C source module for viewing and compiling GCC code.
146 * \param code The code to be viewed.
147 * \param fmt The code format.
148 * \param func_names The name of functions inside the runtime module.
149 * \param const_vars. The constant variables that the c source module needs.
150 * \return The created module.
151 */
152runtime::Module CSourceModuleCreate(const String& code, const String& fmt,
153 const Array<String>& func_names,
154 const Array<String>& const_vars = {});
155
156/*!
157 * \brief Wrap the submodules in a metadata module.
158 * \param params The variable to constant mapping that is collected by the host
159 * module.
160 * \param target_module The main TIR-lowered internal runtime module
161 * \param modules All the external modules that needs to be imported inside the metadata module(s).
162 * \param target The target that all the modules are compiled for
163 * \param metadata Metadata which should be exported to the runtime.
164 * \return The wrapped module.
165 */
166runtime::Module CreateMetadataModule(
167 const std::unordered_map<std::string, runtime::NDArray>& params, runtime::Module target_module,
168 const Array<runtime::Module>& ext_modules, Target target, runtime::metadata::Metadata metadata);
169
170/*!
171 * \brief Create a source module for viewing and limited saving for device.
172 * \param data The code data to be viewed.
173 * \param fmt The code. format.
174 * \param fmap The map function information map of each function.
175 * \param type_key The type_key of the runtime module of this source code
176 * \param fget_source a closure to replace default get source behavior.
177 */
178runtime::Module DeviceSourceModuleCreate(
179 std::string data, std::string fmt, std::unordered_map<std::string, runtime::FunctionInfo> fmap,
180 std::string type_key, std::function<std::string(const std::string&)> fget_source = nullptr);
181
182/*!
183 * \brief Wrap the submodules that are to be wrapped in a c-source metadata module for C runtime.
184 * \param modules The modules to be wrapped.
185 * \param target the target the modules are compiled for.
186 * \param metadata the metadata needed for code generation.
187 * \return The wrapped module.
188 */
189runtime::Module CreateCSourceCrtMetadataModule(const Array<runtime::Module>& modules, Target target,
190 runtime::metadata::Metadata metadata);
