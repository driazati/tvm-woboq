53class ThreadIdxExtractor : public tir::StmtVisitor {
54 private:
55 void VisitStmt_(const AttrStmtNode* op) final {
56 if (op->attr_key == tir::attr::thread_extent) {
57 IterVar iv = Downcast<IterVar>(op->node);
58 if (iv->var->name_hint == "threadIdx.x" || iv->thread_tag == "threadIdx.x") {
59 threadIdx_x_ext = op->value;
60 }
61 if (iv->var->name_hint == "threadIdx.y" || iv->thread_tag == "threadIdx.y") {
62 threadIdx_y_ext = op->value;
63 }
64 if (iv->var->name_hint == "threadIdx.z" || iv->thread_tag == "threadIdx.z") {
65 threadIdx_z_ext = op->value;
66 }
67 }
68 StmtVisitor::VisitStmt_(op);
69 }
70
71 public:
72 PrimExpr threadIdx_x_ext = Integer(1);
73 PrimExpr threadIdx_y_ext = Integer(1);
74 PrimExpr threadIdx_z_ext = Integer(1);
75};
76
77void CodeGenCUDA::PrintExtraAttrs(const PrimFunc& f) {
78 ThreadIdxExtractor extractor;
79 extractor(f->body);
80 arith::Analyzer analyzer;
81 PrimExpr threadIdx_ext = analyzer.Simplify(extractor.threadIdx_x_ext * extractor.threadIdx_y_ext *
82 extractor.threadIdx_z_ext);
83 if (const IntImmNode* const threadIdx_ext_int = threadIdx_ext.as<IntImmNode>()) {
84 if (threadIdx_ext_int->value == 1) {
85 // unable to extract the number of threads per block, hence directly return
86 return;
87 }
88 stream << " __launch_bounds__(" << threadIdx_ext_int->value << ")";
89 }
90}

92std::string CodeGenCUDA::Finish() {
93 if (enable_fp16_) {
94 decl_stream << "#if defined(__CUDA_ARCH__) && (__CUDA_ARCH__ >= 530)\n";
95 decl_stream << "#include <cuda_fp16.h>\n";
96 decl_stream << "__device__ half max"
97 << "(half a, half b)\n"
98 << "{\n return __hgt(__half(a), __half(b)) ? a : b;\n}\n";
99 decl_stream << "__device__ half min(half a, half b)\n"
100 << "{\n return __hlt(__half(a), __half(b)) ? a : b;\n}\n";
101 decl_stream << "#else\n";
102 decl_stream << _cuda_half_t_def;
103 decl_stream << "#endif\n\n";
104 decl_stream << _cuda_half_util;
105 }
106
107 if (enable_bf16_) {
108 decl_stream << "#if defined(__CUDA_ARCH__) && (__CUDA_ARCH__ >= 800)\n";
109 decl_stream << "#include <cuda_bf16.h>\n";
110 decl_stream << "__device__ nv_bfloat16 max"
111 << "(nv_bfloat16 a, nv_bfloat16 b)\n"
112 << "{\n return __hgt(a, b) ? a : b;\n}\n";
113 decl_stream << "__device__ nv_bfloat16 min(nv_bfloat16 a, nv_bfloat16 b)\n"
114 << "{\n return __hlt(a, b) ? a : b;\n}\n";
115 decl_stream << "#endif\n\n";
116 decl_stream << _cuda_bfloat16_util;
117 }
118
119 if (enable_warp_shuffle_) {
120 decl_stream << _cuda_warp_intrinsic_util;
121 }
122
123 if (enable_int8_) {
124 decl_stream << "#if defined(__CUDA_ARCH__) && (__CUDA_ARCH__ >= 610)\n";
125 decl_stream << "#include <sm_61_intrinsics.h>\n";
126 decl_stream << "#endif\n";
127 }
128
129 if (need_math_constants_h_) {
130 decl_stream << "#include <math_constants.h>\n";
131 }
132
133 if (need_mma_h_) {
134 decl_stream << "#include <mma.h>\n";
135 }
136
137 decl_stream << "\n#ifdef _WIN32\n";
138 decl_stream << " using uint = unsigned int;\n";
139 decl_stream << " using uchar = unsigned char;\n";
140 decl_stream << " using ushort = unsigned short;\n";
141 decl_stream << " using int64_t = long long;\n";
142 decl_stream << " using uint64_t = unsigned long long;\n";
143 decl_stream << "#else\n";
144 decl_stream << " #define uint unsigned int\n";
145 decl_stream << " #define uchar unsigned char\n";
146 decl_stream << " #define ushort unsigned short\n";
147 decl_stream << " #define int64_t long long\n";
148 decl_stream << " #define uint64_t unsigned long long\n";
149 decl_stream << "#endif\n";
150
151 return CodeGenC::Finish();
152}

168void CodeGenCUDA::PrintType(DataType t, std::ostream& os) { // NOLINT(*)
169 int lanes = t.lanes();
170 if (t.is_handle()) {
171 ICHECK(t.is_scalar()) << "do not yet support vector types";
172 os << "void*";
173 return;
174 }
175
176 if (t.is_void()) {
177 os << "void";
178 return;
179 }
180
181 bool fail = false;
182 if (t.is_float()) {
183 switch (t.bits()) {
184 case 16:
185 enable_fp16_ = true;
186 if (t.is_scalar()) {
187 os << "half";
188 } else if (lanes <= 8) {
189 // Emit CUDA code to access fp16 vector elements.
190 //
191 // half4 is stored as uint2
192 //
193 // h4.x is emitted as *(half2*)(&(u2.x)).x
194 // h4.y is emitted as *(half2*)(&(u2.x)).y
195 // h4.z is emitted as *(half2*)(&(u2.y)).x
196 // h4.w is emitted as *(half2*)(&(u2.y)).y
197 //
198 ICHECK_EQ(lanes % 2, 0) << "only support even lane for half type";
199 os << "uint" << lanes / 2;
200 } else {
201 fail = true;
202 }
203 break;
204 case 32:
205 if (lanes <= 4) {
206 os << "float";
207 } else if (lanes <= 8) {
208 // Emit CUDA code to access fp32 vector elements for 4 < lanes <= 8.
209 //
210 // float8 is stored as ulonglong4
211 //
212 // f8.v1 is emitted as *(float2*)(&(ul4.x)).x
213 // f8.v2 is emitted as *(float2*)(&(ul4.x)).y
214 //
215 ICHECK_EQ(lanes % 2, 0) << "only support even lane for float type with lanes > 4";
216 os << "ulonglong" << lanes / 2;
217 } else {
218 fail = true;
219 }
220 break;
221 case 64:
222 os << "double";
223 break;
224 default:
225 fail = true;
226 break;
227 }
228 if (!fail && (t.is_scalar() || t.bits() == 16)) return;
229 if (!fail && (lanes > 4 && lanes <= 8 && t.bits() == 32)) return;
230 if (!fail && (lanes >= 2 && lanes <= 4)) {
231 os << lanes;
232 return;
233 }
234 } else if (t.is_bfloat16()) {
235 enable_bf16_ = true;
236 if (t.is_scalar()) {
237 os << "nv_bfloat16";
238 } else if (lanes <= 8) {
239 ICHECK_EQ(lanes % 2, 0) << "only support even lane for half type";
240 os << "uint" << lanes / 2;
241 } else {
242 fail = true;
243 }
244 if (!fail) return;
245 } else if (t == DataType::Bool()) {
246 os << "bool";
247 return;
248 } else if (t.is_vector_bool()) {
249 // CUDA does not support bool vectors.
250 // Use ushort vectors to represent instead.
251 int n = t.lanes();
252 if (n <= 4) {
253 os << "ushort" << n;
254 return;
255 }
256 } else if (t.is_uint() || t.is_int()) {
257 if (t.is_uint()) {
258 os << "u";
259 }
260 switch (t.bits()) {
261 case 1: {
262 if (t.is_scalar()) {
263 os << "int";
264 return;
265 } else if (t.lanes() == 8) {
266 os << "int8_t";
267 return;
268 } else if (t.lanes() == 16) {
269 os << "int16_t";
270 return;
271 } else if (t.lanes() == 32) {
272 os << "int";
273 return;
274 } else {
275 LOG(FATAL) << "Cannot convert type " << t << " to CUDA type!";
276 }
277 }
278 case 4: {
279 if (t.is_scalar()) {
280 os << "int";
281 return;
282 } else if (t.lanes() == 4) {
283 os << "int16_t";
284 return;
285 } else if (t.lanes() == 8) {
286 // directly 8 4-bit int in integer.
287 os << "int";
288 return;
289 } else if (t.lanes() == 16) {
290 os << "int2";
291 return;
292 } else if (t.lanes() == 32) {
293 os << "int4";
294 return;
295 } else if (t.lanes() == 64) {
296 os << "int8";
297 return;
298 } else {
299 LOG(FATAL) << "Cannot convert type " << t << " to CUDA type!";
300 }
301 }
302 case 8: {
303 if (t.lanes() == 4) {
304 // directly 4 8 bit int in integer.
305 enable_int8_ = true;
306
307 // We use int for int8x4 instead of char4 because using char4 is
308 // likely to produce extra instructions to pack four int8 elements
309 // into 32-bit data.
310 os << "int";
311 return;
312 } else if (t.lanes() == 8) {
313 enable_int8_ = true;
314 os << "int2";
315 return;
316 } else if (t.lanes() == 16) {
317 enable_int8_ = true;
318 os << "int4";
319 return;
320 } else if (!t.is_uint() && t.is_scalar()) {
321 os << "signed char";
322 break;
323 } else {
324 os << "char";
325 break;
326 }
327 }
328 case 16: {
329 if (t.is_scalar()) {
330 os << "short";
331 } else if (t.lanes() <= 4) {
332 os << "short" << lanes;
333 } else if (t.lanes() <= 8) {
334 // Emit CUDA code to access int16 vector elements.
335 //
336 // short4 is stored as int2
337 //
338 // s4.x is emitted as *(short2*)(&(i2.x)).x
339 // s4.y is emitted as *(short2*)(&(i2.x)).y
340 // s4.z is emitted as *(short2*)(&(i2.y)).x
341 // s4.w is emitted as *(short2*)(&(i2.y)).y
342 //
343 ICHECK_EQ(t.lanes() % 2, 0) << "only support even lane for shorT type with lanes > 4";
344 os << "int" << t.lanes() / 2;
345 } else {
346 fail = true;
347 }
348 if (!fail) {
349 return;
350 }
351 break;
352 }
353 case 32: {
354 if (t.is_scalar()) {
355 os << "int";
356 } else if (t.lanes() <= 4) {
357 os << "int" << t.lanes();
358 } else if (t.lanes() <= 8) {
359 // Emit CUDA code to access int32 vector elements for 4 < lanes <= 8.
360 //
361 // int8 is stored as longlong4
362 //
363 // i8.v1 is emitted as *(int2*)(&(l4.x)).x
364 // i8.v2 is emitted as *(int2*)(&(l4.x)).y
365 //
366 ICHECK_EQ(lanes % 2, 0) << "only support even lane for int32 type with lanes > 4";
367 os << "longlong" << lanes / 2;
368 } else {
369 fail = true;
370 }
371 if (!fail) {
372 return;
373 }
374 break;
375 }
376 case 64: {
377 if (t.is_scalar()) {
378 os << "int64_t";
379 } else if (t.lanes() == 2) {
380 os << "longlong2";
381 } else if (t.lanes() == 3) {
382 os << "longlong3";
383 } else if (t.lanes() == 4) {
384 os << "longlong4";
385 }
386 return;
387 }
388 default:
389 fail = true;
390 break;
391 }
392 if (!fail && lanes == 1) {
393 return;
394 }
395 if (!fail && (lanes >= 2 && lanes <= 4)) {
396 os << lanes;
397 return;
398 }
399 }
400 LOG(FATAL) << "Cannot convert type " << t << " to CUDA type";
401}

403void CodeGenCUDA::PrintVecBinaryOp(const std::string& op, DataType t, PrimExpr lhs, PrimExpr rhs,
404 std::ostream& os) { // NOLINT(*)
405 // Delcare the result.
406 std::string sret = name_supply_->FreshName("_");
407 this->PrintIndent();
408 this->PrintType(t, stream);
409 stream << ' ' << sret << ";\n";
410 int ssa_scope = BeginScope();
411 {
412 // Unpack into individual ops.
413 std::string vlhs = SSAGetID(PrintExpr(lhs), lhs.dtype());
414 std::string vrhs = SSAGetID(PrintExpr(rhs), rhs.dtype());
415
416 for (int i = 0, lanes = t.lanes(); i < lanes; ++i) {
417 std::ostringstream value_temp;
418 if (isalpha(op[0])) {
419 value_temp << op << "(";
420 PrintVecElemLoad(vlhs, lhs.dtype(), i, value_temp);
421 value_temp << ", ";
422 PrintVecElemLoad(vrhs, rhs.dtype(), i, value_temp);
423 value_temp << ")";
424 } else {
425 value_temp << "(";
426 PrintVecElemLoad(vlhs, lhs.dtype(), i, value_temp);
427 value_temp << op;
428 PrintVecElemLoad(vrhs, rhs.dtype(), i, value_temp);
429 value_temp << ")";
430 }
431 PrintVecElemStore(sret, t, i, value_temp.str());
432 }
433 }
434 EndScope(ssa_scope);
435 os << sret;
436}

438void CodeGenCUDA::PrintVecElemLoad(const std::string& vec, DataType t, int i,
439 std::ostream& os) { // NOLINT(*)
440 if (t.is_scalar()) {
441 os << vec;
442 return;
443 }
444
445 static const char access[] = {'x', 'y', 'z', 'w'};
446 ICHECK(i >= 0 && i < (t.bits() == 8 ? 16 : (t.bits() == 16 || t.bits() == 32) ? 8 : 4));
447 if (t.bits() == 8 && (t.is_int() || t.is_uint())) {
448 std::string type_name = t.is_int() ? "char" : "unsigned char";
449 if (t.lanes() == 2 || t.lanes() == 3) {
450 os << vec << "." << access[i % t.lanes()];
451 } else {
452 std::string ac = t.lanes() == 4 ? vec : (vec + "." + access[i / 4]);
453 os << "((" << type_name << ")(" << ac << " >> " << i % 4 * 8 << "))";
454 }
455 } else if (t.is_float16()) {
456 os << "((half2*)(&(" << vec << "." << access[i / 2] << ")))->" << access[i % 2];
457 } else if (t.is_bfloat16()) {
458 os << "((nv_bfloat162*)(&(" << vec << "." << access[i / 2] << ")))->" << access[i % 2];
459 } else if (t.lanes() > 4 && t.lanes() <= 8) {
460 std::string type_name;
461 if (t.bits() == 16) {
462 if (t.is_int()) {
463 type_name = "short";
464 } else if (t.is_uint()) {
465 type_name = "ushort";
466 }
467 } else if (t.bits() == 32) {
468 if (t.is_int()) {
469 type_name = "int";
470 } else if (t.is_uint()) {
471 type_name = "uint";
472 } else if (t.is_float()) {
473 type_name = "float";
474 }
475 }
476 ICHECK(!type_name.empty());
477 os << "((" << type_name << "2*)(&(" << vec << "." << access[i / 2] << ")))->" << access[i % 2];
478 } else {
479 os << vec << "." << access[i];
480 }
481}

483void CodeGenCUDA::PrintVecElemStore(const std::string& vec, DataType t, int i,
484 const std::string& value) {
485 this->PrintIndent();
486 static const char access[] = {'x', 'y', 'z', 'w'};
487 ICHECK(i >= 0 && i < (t.bits() == 8 ? 16 : (t.bits() == 16 || t.bits() == 32) ? 8 : 4));
488 if (t.bits() == 8 && (t.is_int() || t.is_uint())) {
489 if (t.lanes() == 2 || t.lanes() == 3) {
490 stream << vec << '.' << access[i % t.lanes()] << "="
491 << "(" << value << ");\n";
492 } else {
493 std::string ac = t.lanes() == 4 ? vec : (vec + "." + access[i / 4]);
494 stream << ac << "=";
495 // Do not read the first undef lane.
496 if (i != 0) {
497 stream << ac << " & ~(0x000000ff << " << i % 4 * 8 << ") |";
498 }
499 stream << "(" << value << " << " << i % 4 * 8 << ");\n";
500 }
501 } else if (t.is_float16()) {
502 stream << "((half2*)(&(" << vec << "." << access[i / 2] << ")))->" << access[i % 2] << " = "
503 << value << ";\n";
504 } else if (t.is_bfloat16()) {
505 stream << "((nv_bfloat162*)(&(" << vec << "." << access[i / 2] << ")))->" << access[i % 2]
506 << " = " << value << ";\n";
507 } else if (t.lanes() > 4 && t.lanes() <= 8) {
508 std::string type_name;
509 if (t.bits() == 16) {
510 if (t.is_int()) {
511 type_name = "short";
512 } else if (t.is_uint()) {
513 type_name = "ushort";
514 }
515 } else if (t.bits() == 32) {
516 if (t.is_int()) {
517 type_name = "int";
518 } else if (t.is_uint()) {
519 type_name = "uint";
520 } else if (t.is_float()) {
521 type_name = "float";
522 }
523 }
524 ICHECK(!type_name.empty());
525 stream << "((" << type_name << "2*)(&(" << vec << "." << access[i / 2] << ")))->"
526 << access[i % 2] << " = " << value << ";\n";
527 } else {
528 stream << vec << "." << access[i] << " = " << value << ";\n";
529 }
530}

532void CodeGenCUDA::PrintStorageSync(const CallNode* op) {
533 const std::string& sync = op->args[0].as<StringImmNode>()->value;
534 if (sync == "warp") {
535 // DO nothing.
536 } else if (sync == "shared" || sync == "shared.dyn") {
537 this->PrintIndent();
538 this->stream << "__syncthreads();\n";
539 } else if (sync == "global") {
540 if (!need_global_barrier_) {
541 need_global_barrier_ = true;
542 this->decl_stream << "extern \"C\" __device__ unsigned " << vid_global_barrier_state_
543 << ";\n";
544 }
545 // global synchronizer
546 std::string is_load = PrintExpr(op->args[1]);
547 std::string num_blocks = PrintExpr(op->args[2]);
548 this->PrintIndent();
549 // In theory only threadfence is needed
550 // but we observed problems with only threadfence
551 this->stream << "__threadfence_system();\n";
552 this->PrintIndent();
553 this->stream << "if (" << is_load << ") {\n";
554 int wb = this->BeginScope();
555 this->PrintIndent();
556 this->stream << "atomicAdd(&" << vid_global_barrier_state_ << ", 1);\n";
557 this->PrintIndent();
558 std::string ptr = name_supply_->FreshName("pf");
559 this->stream << "volatile unsigned* " << ptr << " = &" << vid_global_barrier_state_ << ";\n";
560 this->PrintIndent();
561 this->stream << vid_global_barrier_expect_ << " += " << num_blocks << ";\n";
562 this->PrintIndent();
563 this->stream << "while (" << ptr << "[0] < " << vid_global_barrier_expect_ << ");\n";
564 this->EndScope(wb);
565 this->PrintIndent();
566 this->stream << "}\n";
567 this->PrintIndent();
568 this->stream << "__syncthreads();\n";
569 }
570}

581
582void CodeGenCUDA::VisitExpr_(const CastNode* op, std::ostream& os) {
583 DataType from_ty = op->value.dtype();
584 DataType target_ty = op->dtype;
585 ICHECK_EQ(target_ty.lanes(), from_ty.lanes());
586
587 // Emit simple C-style type conversion.
588 if (from_ty.is_scalar()) return CodeGenC::VisitExpr_(op, os);
589
590 // We could emit make_float4 like calls, but the emitted code looks
591 // too compact to read. Emit this as vectorized unary ops.
592 std::string sret = name_supply_->FreshName("_");
593 this->PrintIndent();
594 this->PrintType(target_ty, stream);
595 stream << ' ' << sret << ";\n";
596 {
597 std::string src = SSAGetID(PrintExpr(op->value), from_ty);
598 for (int i = 0, lanes = from_ty.lanes(); i < lanes; ++i) {
599 std::ostringstream val;
600 val << "(";
601 PrintType(target_ty.element_of(), val);
602 val << ")(";
603 PrintVecElemLoad(src, from_ty, i, val);
604 val << ")";
605 PrintVecElemStore(sret, target_ty, i, val.str());
606 }
607 }
608 os << sret;
609}

611void CodeGenCUDA::PrintCallExtern(Type ret_type, String global_symbol, const Array<PrimExpr>& args,
612 bool skip_first_arg, std::ostream& os) { // NOLINT(*)
613 DataType ret_dtype = GetRuntimeDataType(ret_type);
614 if (ret_dtype.is_vector()) {
615 //
616 // Emit an unsupported vector call
617 //
618 // v = intrin_f((float4*)A[0], (float4*)B[0])
619 //
620 // as
621 //
622 // float4 __ret;
623 // {
624 // float4 __arg0 = ((float4*)A)[0];
625 // float4 __arg1 = ((float4*)B)[0];
626 // __ret.x = intrin_f(__arg0.x, __arg1.x);
627 // __ret.y = intrin_f(__arg0.y, __arg1.y);
628 // __ret.z = intrin_f(__arg0.z, __arg1.z);
629 // __ret.w = intrin_f(__arg0.w, __arg1.w);
630 // }
631 // v = __ret;
632 //
633 // Declare the result vector.
634 std::string sret = name_supply_->FreshName("_");
635 this->PrintIndent();
636 this->PrintType(ret_dtype, stream);
637 stream << ' ' << sret << ";\n";
638 {
639 // Load arguments.
640 std::vector<std::string> sargs;
641 size_t arg_begin = static_cast<size_t>(skip_first_arg);
642 for (size_t i = arg_begin; i < args.size(); ++i) {
643 std::string val = SSAGetID(PrintExpr(args[i]), args[i].dtype());
644 sargs.push_back(std::move(val));
645 }
646
647 // Emit a scalar call for each lane.
648 for (int i = 0; i < ret_dtype.lanes(); ++i) {
649 std::ostringstream scall;
650 scall << global_symbol << "(";
651 for (size_t j = 0; j < sargs.size(); ++j) {
652 if (j > 0) scall << ", ";
653 PrintVecElemLoad(sargs[j], args[arg_begin + j].dtype(), i, scall);
654 }
655 scall << ")";
656 PrintVecElemStore(sret, ret_dtype, i, scall.str());
657 }
658 }
659 os << sret;
660 } else {
661 CodeGenC::PrintCallExtern(ret_type, global_symbol, args, skip_first_arg, os);
662 }
663}

665void CodeGenCUDA::VisitExpr_(const CallNode* op, std::ostream& os) {
666 if (auto* ptr_op = op->op.as<OpNode>()) {
667 Op call_op = GetRef<Op>(ptr_op);
668 // This is only for backward compatibility with __shfl_{up/down}.
669 // A macro will be used to replace *_sync calls to legacy ones.
670 if (op_need_warp_shuffle_.get(call_op, false)) {
671 enable_warp_shuffle_ = true;
672 }
673 }
674
675 if (op->op.same_as(builtin::tvm_fill_fragment())) {
676 need_mma_h_ = true;
677 ICHECK_EQ(op->args.size(), 6U);
678 os << "nvcuda::wmma::fill_fragment(";
679 this->PrintExpr(op->args[0], os);
680 os << "[";
681 this->PrintExpr(op->args[4], os);
682 os << "], ";
683 this->PrintExpr(op->args[5], os);
684 os << ")";
685 } else if (op->op.same_as(builtin::tvm_load_matrix_sync())) {
686 need_mma_h_ = true;
687 ICHECK_EQ(op->args.size(), 8U);
688 os << "nvcuda::wmma::load_matrix_sync(";
689 this->PrintExpr(op->args[0], os);
690 os << "[";
691 this->PrintExpr(op->args[4], os);
692 os << "], ";
693 this->PrintExpr(op->args[5], os);
694 os << ", ";
695 this->PrintExpr(op->args[6], os);
696 os << ")";
697 } else if (op->op.same_as(builtin::tvm_store_matrix_sync())) {
698 need_mma_h_ = true;
699 ICHECK_EQ(op->args.size(), 8U);
700 os << "nvcuda::wmma::store_matrix_sync(";
701 this->PrintExpr(op->args[5], os);
702 os << ", ";
703 this->PrintExpr(op->args[0], os);
704 os << "[";
705 this->PrintExpr(op->args[4], os);
706 os << "], ";
707 this->PrintExpr(op->args[6], os);
708 if (const StringImmNode* str = op->args[7].as<StringImmNode>()) {
709 os << ", nvcuda::wmma::mem_" << str->value;
710 } else {
711 LOG(FATAL) << "Invalid parameters";
712 }
713 os << ")";
714 } else if (op->op.same_as(builtin::tvm_mma_sync())) {
715 need_mma_h_ = true;
716 ICHECK_EQ(op->args.size(), 8U);
717 os << "nvcuda::wmma::mma_sync(";
718 for (int i = 0; i < 4; ++i) {
719 this->PrintExpr(op->args[i * 2], os);
720 os << "[";
721 this->PrintExpr(op->args[i * 2 + 1], os);
722 os << "]" << ((i < 3) ? ", " : ")");
723 }
724 } else if (op->op.same_as(builtin::tvm_bmma_sync())) {
725 need_mma_h_ = true;
726 ICHECK_EQ(op->args.size(), 8U);
727 os << "nvcuda::wmma::bmma_sync(";
728 for (int i = 0; i < 4; ++i) {
729 this->PrintExpr(op->args[i * 2], os);
730 os << "[";
731 this->PrintExpr(op->args[i * 2 + 1], os);
732 os << "]" << ((i < 3) ? ", " : ")");
733 }
734 } else if (op->op.same_as(builtin::ptx_mma())) {
735 // arg 0: shape: mXnXkX
736 // arg 1: A layout: row/col
737 // arg 2: B layout: row/col
738 // arg 3: A precision: fp16, fp64, ...
739 // arg 4: B precision: fp16, fp64, ...
740 // arg 5: C precision: fp32, fp64, ...
741 // arg 6: A multiplicand
742 // arg 7: A multiplicand index
743 // arg 8: B multiplicand
744 // arg 9: B multiplicand index
745 // arg 10: C accumulator
746 // arg 11: C accumulator index
747 // arg 12: saturate
748 // arg 13: (optional) 1-bit operator (xor or and)
749 ICHECK(op->args.size() == 13U || op->args.size() == 14U);
750 std::string shape = Downcast<StringImm>(op->args[0])->value;
751 std::string A_layout = Downcast<StringImm>(op->args[1])->value;
752 std::string B_layout = Downcast<StringImm>(op->args[2])->value;
753 std::string A_dtype = Downcast<StringImm>(op->args[3])->value;
754 std::string B_dtype = Downcast<StringImm>(op->args[4])->value;
755 std::string C_dtype = Downcast<StringImm>(op->args[5])->value;
756 std::string a_ref = this->PrintExpr(op->args[6]);
757 std::string a_bias = this->PrintExpr(op->args[7]);
758 std::string b_ref = this->PrintExpr(op->args[8]);
759 std::string b_bias = this->PrintExpr(op->args[9]);
760 std::string c_ref = this->PrintExpr(op->args[10]);
761 std::string c_bias = this->PrintExpr(op->args[11]);
762 bool saturate = Downcast<Bool>(op->args[12])->value;
763 std::string bit_op = op->args.size() > 13 ? Downcast<StringImm>(op->args[13])->value : "";
764 std::string asm_code =
765 PrintMMAAssembly(shape, A_layout, B_layout, A_dtype, B_dtype, C_dtype, a_ref, a_bias, b_ref,
766 b_bias, c_ref, c_bias, "", "", "", bit_op, false, saturate);
767
768 this->stream << asm_code;
769 } else if (op->op.same_as(builtin::ptx_mma_sp())) {
770 // arg 0: shape: mXnXkX
771 // arg 1: A layout: row/col
772 // arg 2: B layout: row/col
773 // arg 3: A precision: fp16, fp32, ...
774 // arg 4: B precision: fp16, fp32, ...
775 // arg 5: C precision: fp16, fp32, ...
776 // arg 6: A multiplicand pointer
777 // arg 7: A multiplicand index
778 // arg 8: B multiplicand pointer
779 // arg 9: B multiplicand index
780 // arg 10: C accumulator pointer
781 // arg 11: C accumulator index
782 // arg 12: metadata
783 // arg 13: metadata index
784 // arg 14: sparse_selector
785 // arg 15: saturate
786 ICHECK_EQ(op->args.size(), 16U);
787 std::string shape = Downcast<StringImm>(op->args[0])->value;
788 std::string A_layout = Downcast<StringImm>(op->args[1])->value;
789 std::string B_layout = Downcast<StringImm>(op->args[2])->value;
790 std::string A_dtype = Downcast<StringImm>(op->args[3])->value;
791 std::string B_dtype = Downcast<StringImm>(op->args[4])->value;
792 std::string C_dtype = Downcast<StringImm>(op->args[5])->value;
793 std::string a_ref = this->PrintExpr(op->args[6]);
794 std::string a_offset = this->PrintExpr(op->args[7]);
795 std::string b_ref = this->PrintExpr(op->args[8]);
796 std::string b_offset = this->PrintExpr(op->args[9]);
797 std::string c_ref = this->PrintExpr(op->args[10]);
798 std::string c_offset = this->PrintExpr(op->args[11]);
799 std::string metadata = this->PrintExpr(op->args[12]);
800 std::string metadata_offset = this->PrintExpr(op->args[13]);
801 std::string sparse_selector = this->PrintExpr(op->args[14]);
802 bool saturate = Downcast<Bool>(op->args[15])->value;
803 std::string asm_code = PrintMMAAssembly(
804 shape, A_layout, B_layout, A_dtype, B_dtype, C_dtype, a_ref, a_offset, b_ref, b_offset,
805 c_ref, c_offset, metadata, metadata_offset, sparse_selector, "", true, saturate);
806 this->stream << asm_code;
807 } else if (op->op.same_as(builtin::ptx_ldmatrix())) {
808 // arg 0: whether the matrix is loaded in column major format or not.
809 // arg 1: number of matrices to load.
810 // arg 2: The data type in the matrix, .b16 is the only accepted data type.
811 // arg 3: pointer to local buffer.
812 // arg 4: The offset of the element to store in the local buffer.
813 // arg 5: pointer to the shared memory buffer to load.
814 // arg 6: The offset of the start element of the row to load in shared memory.
815 ICHECK_EQ(op->args.size(), 7U);
816 bool trans = Downcast<Bool>(op->args[0])->value;
817 int num = Downcast<Integer>(op->args[1])->value;
818 std::string type = Downcast<StringImm>(op->args[2])->value;
819 std::string local_ptr = this->PrintExpr(op->args[3]);
820 std::string local_elem_offset = this->PrintExpr(op->args[4]);
821 std::string smem_ptr = this->PrintExpr(op->args[5]);
822 if (trans && op->dtype.bits() == 8) {
823 // Since ldmatrix assumes that a matrix element is 16 bit, it cannot properly transpose an
824 // int8 matrix.
825 std::string smem_stride = this->PrintExpr(op->args[6]);
826 ICHECK(num == 4);
827 os << "for (int i = 0; i < 16; ++i) {\n";
828 os << local_ptr << "[" + local_elem_offset + " + i] = " << smem_ptr
829 << "[(i % 8) / 4 * " + smem_stride + " * 16 + (threadIdx.x % 4) * 4 * " + smem_stride +
830 "+ (i % 4) * " + smem_stride + " + threadIdx.x / 4 + (i / 8) * 8];\n";
831 os << "}\n";
832 } else {
833 std::string smem_elem_offset = this->PrintExpr(op->args[6]);
834 this->stream << PrintLoadMatrixAssembly(trans, num, type, local_ptr, local_elem_offset,
835 smem_ptr, smem_elem_offset);
836 }
837 } else if (op->op.same_as(builtin::mma_store())) {
838 int m = Downcast<Integer>(op->args[0])->value;
839 int n = Downcast<Integer>(op->args[1])->value;
840 std::string dst = this->PrintExpr(op->args[2]);
841 std::string src = this->PrintExpr(op->args[3]);
842 std::string src_offset = this->PrintExpr(op->args[4]);
843 PrimExpr stride = op->args[5];
844
845 ICHECK(m == 16 && n == 16) << "Only m == 16 && n == 16 case supported for now";
846
847 // Each thread in a warp holds a certain number of elements of an MMA output.
848 // For example, if we compute a 16x16 tile using MMA, each thread holds 8 elements
849 // in its registers. So conceptually, a warp memory is organized as a 32x8 block.
850 // A map from a 16x16 tile to a 32x8 block of memory is specified by the index map below.
851
852 // To store the 32x8 output back to a 16x16 tile in shared or global memory, we invert this map
853 // to determine the output location for each 8 element.
854
855 const auto* index_map_func =
856 runtime::Registry::Get("tir.index_map.shared_16x16_to_ldmatrix_32x8_layout");
857 ICHECK(index_map_func);
858
859 auto inverse_index_map =
860 IndexMap::FromFunc(2, *index_map_func).Inverse({Range(0, m), Range(0, n)});
861 auto indices_16x16 = inverse_index_map->final_indices;
862
863 // "//" and "%" in the index map are translated to FloorDiv/Mod, but the plain Div/Mod are fine.
864 // FloorDiv/Mod are supposed to be lowered before they reach codegen, so manually replace them
865 // to the plain ones here.
866 class LowerFloorDivMod : public ExprMutator {
867 public:
868 PrimExpr VisitExpr_(const FloorDivNode* op) {
869 return tir::Div(this->VisitExpr(op->a), this->VisitExpr(op->b));
870 }
871 PrimExpr VisitExpr_(const FloorModNode* op) {
872 return tir::Mod(this->VisitExpr(op->a), this->VisitExpr(op->b));
873 }
874 };
875
876 auto dst_ind = LowerFloorDivMod()(indices_16x16[0] * stride + indices_16x16[1]);
877
878 var_idmap_[inverse_index_map->initial_indices[0].get()] = "threadIdx.x";
879 var_idmap_[inverse_index_map->initial_indices[1].get()] = "local_id";
880
881 os << "for (int local_id = 0; local_id < 8; ++local_id) {\n";
882 os << dst << "[" + this->PrintExpr(dst_ind) + "]"
883 << " = " << src << "[" << src_offset << " + local_id];\n";
884 os << "}\n";
885
886 } else if (op->op.same_as(builtin::mma_fill())) {
887 std::string num_elem = this->PrintExpr(op->args[0]);
888 std::string dst = this->PrintExpr(op->args[1]);
889 std::string dst_offset = this->PrintExpr(op->args[2]);
890
891 os << "for (int i = 0; i < " << num_elem << "; ++i) {\n";
892 os << dst << "[" << dst_offset << " + i] = 0.0;";
893 os << "}\n";
894 } else if (op->op.same_as(builtin::ptx_cp_async())) {
895 std::string dst = this->PrintExpr(op->args[0]);
896 std::string dst_offset = this->PrintExpr(op->args[1]);
897 std::string src = this->PrintExpr(op->args[2]);
898 std::string src_offset = this->PrintExpr(op->args[3]);
899 std::string size = this->PrintExpr(op->args[4]);
900 this->stream << PrintCpAsyncAssembly(dst, dst_offset, src, src_offset, size);
901 } else if (op->op.same_as(builtin::ptx_commit_group())) {
902 this->stream << "__asm__ __volatile__(\"cp.async.commit_group;\");\n\n";
903 } else if (op->op.same_as(builtin::ptx_wait_group())) {
904 std::string N = this->PrintExpr(op->args[0]);
905 this->stream << "__asm__ __volatile__(\"cp.async.wait_group " + N + ";\");\n\n";
906 } else {
907 CodeGenC::VisitExpr_(op, os);
908 }
909}

936 this->VisitStmt(inner->body);
937 return;
938 }
939 CodeGenC::VisitStmt_(op);

1017void CodeGenCUDA::VisitExpr_(const BroadcastNode* op, std::ostream& os) { // NOLINT(*)
1018 if ((op->dtype.is_int() || op->dtype.is_uint()) && op->dtype.bits() == 8 && op->lanes == 4) {
1019 // make_int8x4
1020 const int64_t* p = as_const_int(op->value);
1021 ICHECK(p);
1022 int64_t v = *p & 0xFF;
1023 v = (v << 24) | (v << 16) | (v << 8) | v;
1024 if (op->dtype.is_uint()) {
1025 os << "(uint)" << v;
1026 } else {
1027 os << "(int)" << v;
1028 }
1029 return;
1030 }
1031
1032 if (op->dtype.is_float16()) {
1033 std::string v = PrintExpr(op->value);
1034 os << "make_";
1035 PrintType(op->dtype, os);
1036 os << '(';
1037 for (int i = 0; i < op->lanes / 2; ++i) {
1038 if (i != 0) os << ", ";
1039 os << "__pack_half2(" << v << ", " << v << ")";
1040 }
1041 os << ')';
1042 return;
1043 }
1044
1045 if (op->dtype.is_bfloat16()) {
1046 std::string v = PrintExpr(op->value);
1047 os << "make_";
1048 PrintType(op->dtype, os);
1049 os << '(';
1050 for (int i = 0; i < op->lanes / 2; ++i) {
1051 if (i != 0) os << ", ";
1052 os << "__pack_nv_bfloat162(" << v << ", " << v << ")";
1053 }
1054 os << ')';
1055 return;
1056 }
1057
1058 if ((op->dtype.is_int() || op->dtype.is_uint()) && op->dtype.bits() == 4) {
1059 bool fail = false;
1060 const int64_t* p = as_const_int(op->value);
1061 ICHECK(p);
1062 int64_t v = *p & 0xF;
1063
1064 if (op->lanes == 4) {
1065 v = (v << 12) | (v << 8) | (v << 4) | v;
1066 if (op->dtype.is_uint()) {
1067 os << "(uint16_t)" << v;
1068 } else {
1069 os << "(int16_t)" << v;
1070 }
1071 } else {
1072 v = (v << 28) | (v << 24) | (v << 20) | (v << 16) | (v << 12) | (v << 8) | (v << 4) | v;
1073 if (op->lanes == 8) {
1074 if (op->dtype.is_uint()) {
1075 os << "(uint)" << v;
1076 } else {
1077 os << "(int)" << v;
1078 }
1079 } else if (op->lanes == 16 || op->lanes == 32) {
1080 os << "make_";
1081 PrintType(op->dtype, os);
1082 os << '(';
1083 for (int i = 0; i < op->lanes / 8; ++i) {
1084 if (i != 0) os << ", ";
1085 if (op->dtype.is_uint()) {
1086 os << "(uint)" << v;
1087 } else {
1088 os << "(int)" << v;
1089 }
1090 }
1091 os << ')';
1092 } else {
1093 fail = true;
1094 }
1095 }
1096
1097 if (!fail) {
1098 return;
1099 }
1100 }
1101
1102 std::string v = PrintExpr(op->value);
1103 os << "make_";
1104 PrintType(op->dtype, os);
1105 os << '(';
1106 for (int i = 0; i < op->lanes; ++i) {
1107 if (i != 0) os << ", ";
1108 os << v;
1109 }
1110 os << ')';
1111}

1131void CodeGenCUDA::VisitExpr_(const SelectNode* op, std::ostream& os) {
1132 // Non-vector cases.
1133 if (!op->dtype.is_vector()) {
1134 CodeGenC::VisitExpr_(op, os);
1135 return;
1136 }
1137
1138 // Codegen vector condition case by serializing the select op.
1139 ICHECK(op->false_value->dtype == op->dtype && op->true_value->dtype == op->dtype &&
1140 op->dtype.lanes() == op->condition.dtype().lanes());
1141
1142 std::string r_var = name_supply_->FreshName("_");
1143 this->PrintIndent();
1144 this->PrintType(op->dtype, stream);
1145 stream << ' ' << r_var << ";\n";
1146 {
1147 std::string c_var = SSAGetID(PrintExpr(op->condition), op->dtype);
1148 std::string t_var = SSAGetID(PrintExpr(op->true_value), op->dtype);
1149 std::string f_var = SSAGetID(PrintExpr(op->false_value), op->dtype);
1150
1151 // The condition is stored as an ushort vector.
1152 int lanes = op->dtype.lanes();
1153 DataType memory_ty(DataType::TypeCode::kUInt, 16, lanes);
1154
1155 for (int i = 0; i < lanes; ++i) {
1156 std::ostringstream item;
1157 item << "(bool(";
1158 PrintVecElemLoad(c_var, memory_ty, i, item);
1159 item << ")?";
1160 PrintVecElemLoad(t_var, op->dtype, i, item);
1161 item << ':';
1162 PrintVecElemLoad(f_var, op->dtype, i, item);
1163 item << ')';
1164 PrintVecElemStore(r_var, op->dtype, i, item.str());
1165 }
1166 }
1167 os << r_var;
1168}

1170inline void PrintConst(const FloatImmNode* op, std::ostream& os, CodeGenCUDA* p) { // NOLINT(*)
1171 // Type code is kBFloat
1172 if (op->dtype.is_bfloat16()) {
1173 os << "__float2bfloat16_rn";
1174 os << '(' << std::scientific << op->value << 'f' << ')';
1175 return;
1176 }
1177 // Type code is kFloat
1178 switch (op->dtype.bits()) {
1179 case 64:
1180 case 32: {
1181 std::ostringstream temp;
1182 if (std::isinf(op->value)) {
1183 if (op->value < 0) {
1184 temp << "-";
1185 }
1186 temp << ((op->dtype.bits() == 32) ? "CUDART_INF_F" : "CUDART_INF");
1187 p->need_math_constants_h_ = true;
1188 } else if (std::isnan(op->value)) {
1189 temp << ((op->dtype.bits() == 32) ? "CUDART_NAN_F" : "CUDART_NAN");
1190 p->need_math_constants_h_ = true;
1191 } else {
1192 temp << std::scientific << op->value;
1193 if (op->dtype.bits() == 32) temp << 'f';
1194 }
1195 p->MarkConst(temp.str());
1196 os << temp.str();
1197 break;
1198 }
1199 case 16: {
1200 os << "__float2half_rn" << '(';
1201 FloatImm const_f32 = FloatImm(DataType::Float(32), op->value);
1202 PrintConst(const_f32.get(), os, p);
1203 os << ')';
1204 break;
1205 }
1206 default:
1207 LOG(FATAL) << "Bad bit-width for float: " << op->dtype << "\n";
1208 }
1209}

1266int32_t CodeGenCUDA::GetWmmaFragmentSize(const std::string& scope, const VarNode* variable,
1267 int32_t size) {
1268 std::string shape_str = fragment_shapes.at(variable);
1269 size_t m, n, k;
1270 size_t last_pos = 0, pos = 0;
1271 pos = shape_str.find(", ", last_pos);
1272 m = tvm::codegen::stoi(shape_str.substr(last_pos, pos - last_pos));
1273 last_pos = pos + 2;
1274 pos = shape_str.find(", ", last_pos);
1275 n = tvm::codegen::stoi(shape_str.substr(last_pos, pos - last_pos));
1276 last_pos = pos + 2;
1277 k = tvm::codegen::stoi(shape_str.substr(last_pos, shape_str.length() - last_pos));
1278 if (scope == "wmma.matrix_a") {
1279 return size / m / k;
1280 } else if (scope == "wmma.matrix_b") {
1281 return size / n / k;
1282 } else if (scope == "wmma.accumulator") {
1283 return size / m / n;
1284 }
1285 return 0;
1286}

1302void CodeGenCUDA::PrintVecElemLoadExpr(DataType t, int i, const std::string& value,
1303 std::ostream& os) {
1304 ICHECK_GT(t.lanes(), 1);
1305 if (t.bits() == 8 && (t.is_int() || t.is_uint())) {
1306 if (!(t.lanes() == 2 || t.lanes() == 3)) {
1307 if (i != 0) {
1308 os << "|";
1309 }
1310 os << "((0x000000ff << " << i * 8 << ") & (" << value << " << " << i * 8 << "))";
1311 return;
1312 }
1313 }
1314
1315 if (t.is_float16()) {
1316 if (i == 0) {
1317 os << "make_";
1318 PrintType(t, os);
1319 os << '(';
1320 }
1321 if (i % 2 == 0) {
1322 os << "__pack_half2(" << value;
1323 } else {
1324 os << "," << value << ")";
1325 if (i != t.lanes() - 1) {
1326 os << ",";
1327 } else {
1328 os << ")";
1329 }
1330 }
1331 return;
1332 }
1333
1334 if (t.is_bfloat16()) {
1335 if (i == 0) {
1336 os << "make_";
1337 PrintType(t, os);
1338 os << '(';
1339 }
1340 if (i % 2 == 0) {
1341 os << "__pack_bfloat162(" << value;
1342 } else {
1343 os << "," << value << ")";
1344 if (i != t.lanes() - 1) {
1345 os << ",";
1346 } else {
1347 os << ")";
1348 }
1349 }
1350 return;
1351 }
1352
1353 if (i == 0) {
1354 os << "make_";
1355 PrintType(t, os);
1356 os << "(";
1357 }
1358 os << value;
1359 if (i != t.lanes() - 1) {
1360 os << ",";
1361 } else {
1362 os << ")";
