49/*!
50 * \brief A base class to generate C code.
51 *
52 * CodeGenC have two modes: generate SSA formed C code or normal form.
53 *
54 * **NOTE** CodeGenC does not aim at generating C codes consumed by MSVC or GCC,
55 * Rather, it's providing infrastructural abstraction for C variants like CUDA
56 * and OpenCL-C. You might find some odd variant features, e.g., type `int3` for
57 * a vector of 3 `int`s. For native C code generator, see `CodeGenLLVM`.
58 */
59class CodeGenC : public ExprFunctor<void(const PrimExpr&, std::ostream&)>,
60 public StmtFunctor<void(const Stmt&)>,
61 public CodeGenSourceBase {
62 public:
63 /*!
64 * \brief Initialize the code generator.
65 * \param output_ssa Whether output SSA.
66 */
67 void Init(bool output_ssa);
68 /*!
69 * \brief Add the function to the generated module.
70 * \param f The function to be compiled.
71 * \param whether to append return 0 in the end.
72 */
73 void AddFunction(const PrimFunc& f);
74 /*!
75 * \brief Finalize the compilation and return the code.
76 * \return The code.
77 */
78 std::string Finish();
79 /*!
80 * \brief Print the Stmt n to CodeGenC->stream
81 * \param n The statement to be printed.
82 */
83 void PrintStmt(const Stmt& n) { VisitStmt(n); }
84 /*!
85 * \brief Print the expression n(or its ssa id if in ssa mode) into os
86 * \param n The expression to be printed.
87 * \param os The output stream
88 */
89 void PrintExpr(const PrimExpr& n, std::ostream& os);
90 /*!
91 * \brief Same as PrintExpr, but simply returns result string
92 * \param n The expression to be printed.
93 */
94 std::string PrintExpr(const PrimExpr& n) {
95 std::ostringstream os;
96 PrintExpr(n, os);
97 return os.str();
98 }
99 // The following parts are overloadable print operations.
100 /*!
101 * \brief Print the function header before the argument list
102 *
103 * Example: stream << "void";
104 */
105 virtual void PrintFuncPrefix(); // NOLINT(*)
106 /*!
107 * \brief Print extra function attributes
108 *
109 * Example: __launch_bounds__(256) for CUDA functions
110 */
111 virtual void PrintExtraAttrs(const PrimFunc& f);
112 /*!
113 * \brief Print the final return at the end the function.
114 */
115 virtual void PrintFinalReturn(); // NOLINT(*)
116 /*!
117 * \brief Insert statement before function body.
118 * \param f The function to be compiled.
119 */
120 virtual void PreFunctionBody(const PrimFunc& f) {}
121 /*!
122 * \brief Initialize codegen state for generating f.
123 * \param f The function to be compiled.
124 */
125 virtual void InitFuncState(const PrimFunc& f);
126 // expression
127 void VisitExpr_(const VarNode* op, std::ostream& os) override; // NOLINT(*)
128 void VisitExpr_(const LoadNode* op, std::ostream& os) override; // NOLINT(*)
129 void VisitExpr_(const BufferLoadNode* op, std::ostream& os) override; // NOLINT(*)
130 void VisitExpr_(const LetNode* op, std::ostream& os) override; // NOLINT(*)
131 void VisitExpr_(const CallNode* op, std::ostream& os) override; // NOLINT(*)
132 void VisitExpr_(const AddNode* op, std::ostream& os) override; // NOLINT(*)
133 void VisitExpr_(const SubNode* op, std::ostream& os) override; // NOLINT(*)
134 void VisitExpr_(const MulNode* op, std::ostream& os) override; // NOLINT(*)
135 void VisitExpr_(const DivNode* op, std::ostream& os) override; // NOLINT(*)
136 void VisitExpr_(const ModNode* op, std::ostream& os) override; // NOLINT(*)
137 void VisitExpr_(const MinNode* op, std::ostream& os) override; // NOLINT(*)
138 void VisitExpr_(const MaxNode* op, std::ostream& os) override; // NOLINT(*)
139 void VisitExpr_(const EQNode* op, std::ostream& os) override; // NOLINT(*)
140 void VisitExpr_(const NENode* op, std::ostream& os) override; // NOLINT(*)
141 void VisitExpr_(const LTNode* op, std::ostream& os) override; // NOLINT(*)
142 void VisitExpr_(const LENode* op, std::ostream& os) override; // NOLINT(*)
143 void VisitExpr_(const GTNode* op, std::ostream& os) override; // NOLINT(*)
144 void VisitExpr_(const GENode* op, std::ostream& os) override; // NOLINT(*)
145 void VisitExpr_(const AndNode* op, std::ostream& os) override; // NOLINT(*)
146 void VisitExpr_(const OrNode* op, std::ostream& os) override; // NOLINT(*)
147 void VisitExpr_(const CastNode* op, std::ostream& os) override; // NOLINT(*)
148 void VisitExpr_(const NotNode* op, std::ostream& os) override; // NOLINT(*)
149 void VisitExpr_(const SelectNode* op, std::ostream& os) override; // NOLINT(*)
150 void VisitExpr_(const RampNode* op, std::ostream& os) override; // NOLINT(*)
151 void VisitExpr_(const ShuffleNode* op, std::ostream& os) override; // NOLINT(*)
152 void VisitExpr_(const BroadcastNode* op, std::ostream& os) override; // NOLINT(*)
153 void VisitExpr_(const IntImmNode* op, std::ostream& os) override; // NOLINT(*)
154 void VisitExpr_(const FloatImmNode* op, std::ostream& os) override; // NOLINT(*)
155 void VisitExpr_(const StringImmNode* op, std::ostream& os) override; // NOLINT(*)
156 // statment
157 void VisitStmt_(const LetStmtNode* op) override;
158 void VisitStmt_(const StoreNode* op) override;
159 void VisitStmt_(const BufferStoreNode* op) override;
160 void VisitStmt_(const ForNode* op) override;
161 void VisitStmt_(const WhileNode* op) override;
162 void VisitStmt_(const IfThenElseNode* op) override;
163 void VisitStmt_(const AllocateNode* op) override;
164 void VisitStmt_(const AttrStmtNode* op) override;
165 void VisitStmt_(const AssertStmtNode* op) override;
166 void VisitStmt_(const EvaluateNode* op) override;
167 void VisitStmt_(const SeqStmtNode* op) override;
168 void VisitStmt_(const AllocateConstNode* op) override;
169 void VisitStmt_(const DeclBufferNode* op) override;
170
171 /*!
172 * \brief Print expr representing the thread tag
173 * \param IterVar iv The thread index to be binded;
174 */
175 virtual void BindThreadIndex(const IterVar& iv); // NOLINT(*)
176 virtual void PrintStorageScope(const std::string& scope, std::ostream& os); // NOLINT(*)
177 virtual void PrintStorageSync(const CallNode* op); // NOLINT(*)
178 // Binary vector op.
179 virtual void PrintVecBinaryOp(const std::string& op, DataType op_type, PrimExpr lhs, PrimExpr rhs,
180 std::ostream& os); // NOLINT(*)
181 // print vector load
182 virtual std::string GetVecLoad(DataType t, const BufferNode* buffer, PrimExpr base);
183 // print vector store
184 virtual void PrintVecStore(const BufferNode* buffer, DataType t, PrimExpr base,
185 const std::string& value); // NOLINT(*)
186 // print load of single element
187 virtual void PrintVecElemLoad(const std::string& vec, DataType t, int i,
188 std::ostream& os); // NOLINT(*)
189 // print store of single element.
190 virtual void PrintVecElemStore(const std::string& vec, DataType t, int i,
191 const std::string& value);
192 // Get a cast type from to
193 virtual std::string CastFromTo(std::string value, DataType from, DataType target);
194 // Get load of single element with expression
195 virtual void PrintVecElemLoadExpr(DataType t, int i, const std::string& value, std::ostream& os);
196 // Print restrict keyword for a given Var if applicable
197 virtual void PrintRestrict(const Var& v, std::ostream& os);
198
199 virtual void SetConstantsByteAlignment(Integer constants_byte_alignment) {
200 constants_byte_alignment_ = constants_byte_alignment;
201 }
202
203 protected:
204 // Print reference to struct location
205 std::string GetStructRef(DataType t, const PrimExpr& buffer, const PrimExpr& index, int kind);
206 // Print reference to a buffer as type t in index.
207 virtual std::string GetBufferRef(DataType t, const BufferNode* buffer, PrimExpr index);
208
209 /*!
210 * \brief Handle volatile loads.
211 *
212 * This is to workaround a bug in CUDA cuda_fp16.h. Volatile accesses
213 * to shared memory are required for reductions. However, __half class
214 * does not implement volatile member functions. CUDA codegen will cast
215 * away volatile qualifier from CUDA __half types.
216 */
217 virtual void HandleVolatileLoads(const std::string& value, const BufferLoadNode* op,
218 std::ostream& os) {
219 // By default, do nothing but print the loaded value.
220 os << value;
221 }
222
223 /*!
224 * \brief Check if scope is part of type in the target language.
225 *
226 * **NOTE** In OpenCL, __local is part of type, so "__local int *"
227 * is legal. This is not the case for CUDA, where "__shared__"
228 * or "__constant__" is not part of type but a storage class (like
229 * C/C++ static).
230 */
231 virtual bool IsScopePartOfType() const { return true; }
232
233 /*!
234 * \brief Print external function call.
235 * \param ret_type The return type.
236 * \param global_symbol The symbolc of the target function.
237 * \param args The arguments to the function.
238 * \param skip_first_arg Whether to skip the first arguments.
239 * \param os The output stream.
240 */
241 virtual void PrintCallExtern(Type ret_type, String global_symbol, const Array<PrimExpr>& args,
242 bool skip_first_arg, std::ostream& os); // NOLINT(*)
243 /*!
244 * \brief If buffer is allocated as type t.
245 * \param buf_var The buffer variable.
246 * \param t The type to be checked.
247 */
248 bool HandleTypeMatch(const VarNode* buf_var, DataType t) const;
249 /*!
250 * \brief Register the data type of buf_var
251 * \param buf_var The buffer variable.
252 * \param t The type to be checked.
253 */
254 void RegisterHandleType(const VarNode* buf_var, DataType t);
255 // override
256 void PrintSSAAssign(const std::string& target, const std::string& src, DataType t) final;
257 /*! \brief reserves common C keywords */
258 void ReserveKeywordsAsUnique();
259
260 /*! \brief Check if buf_var is volatile or not. */
261 bool IsVolatile(const VarNode* buf_var) const { return volatile_buf_.count(buf_var) != 0; }
262
263 /*! \brief restrict keyword */
264 std::string restrict_keyword_{""};
265 /*! \brief the storage scope of allocation */
266 std::unordered_map<const VarNode*, std::string> alloc_storage_scope_;
267 /*! \brief the data type of allocated buffers */
268 std::unordered_map<const VarNode*, DataType> handle_data_type_;
269 /*! \brief Record of ops that have pre-defined global symbol. */
270 OpAttrMap<TGlobalSymbol> op_attr_global_symbol_ = Op::GetAttrMap<TGlobalSymbol>("TGlobalSymbol");
271 // cache commonly used ops
272 const Op& builtin_call_extern_ = builtin::call_extern();
273 const Op& builtin_call_pure_extern_ = builtin::call_pure_extern();
274 Integer constants_byte_alignment_ = 16;
275
276 private:
277 /*! \brief whether to print in SSA form */
278 bool print_ssa_form_{false};
279 /*! \brief set of volatile buf access */
280 std::unordered_set<const VarNode*> volatile_buf_;
281 // deep comparison of PrimExpr
282 ExprDeepEqual deep_equal_;
283 // binding of let variables. Enables duplicate var defs that map to same value
284 std::unordered_map<Var, const LetNode*, ObjectPtrHash, ObjectPtrEqual> let_binding_;
285};
