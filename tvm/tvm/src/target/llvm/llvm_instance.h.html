58/*!
59 * \class LLVMInstance
60 * \brief LLVMInstance is a class that (conceptually) starts and stops LLVM. All
61 * uses of LLVM should take place within a lifetime of an object of this class.
62 *
63 * E.g.
64 * ```{.cpp}
65 * {
66 * LLVMInstance llvm_instance;
67 * ...
68 * someFunctionFromLLVM(...);
69 * ...
70 * }
71 * // no more calls to LLVM here
72 * ```
73 * In addition to that, LLVMInstance provides an LLVM context (llvm::LLVMContext).
74 * The context is a structure in LLVM where common IR constructs are maintained,
75 * (such as types, constants, etc.) so that they can be identified by their
76 * address (i.e. pointer comparison). Because of that, it's important to use
77 * the same context throughout compilation.
78 *
79 * At the moment the "starting" of LLVM performs initialization of LLVM, but
80 * "stopping" doesn't do anything. In the future, if such a need arises, this
81 * functionality may be extended to perform dlopen/dlclose of the LLVM-based
82 * code in TVM.
83 *
84 * This class provides means to deserialize an LLVM module, either from text
85 * (in a string), or from a file. In either case, the serialized module can
86 * be LLVM IR assembly, or binary bitcode enconding.
87 */
88class LLVMInstance {
89 public:
90 /*!
91 * \brief Constructs LLVMInstance
92 */
93 LLVMInstance();
94 /*!
95 * \brief Destroys LLVMInstance object
96 */
97 ~LLVMInstance(); // Must not be "= default" here in the header file.
98
99 /*!
100 * \brief Get the LLVM context for this scope.
101 */
102 std::shared_ptr<llvm::LLVMContext> GetContext() const { return ctx_; }
103
104 /*!
105 * \brief Create `llvm::Module` from a string.
106 *
107 * Parse the string in \param llvm_ir, and return the `llvm::Module`.
108 * At the moment this function will abort if the parsing fails.
109 * \param llvm_ir string with the LLVM IR assembly or bitcode
110 * \return created `llvm::Module`
111 */
112 std::unique_ptr<llvm::Module> ParseIR(const std::string& llvm_ir) const;
113 /*!
114 * \brief Load `llvm::Module` from a given file
115 *
116 * Read the file \param file_name, and return the `llvm::Module`.
117 * At the moment this function will abort if reading of the file or creation
118 * of the module fails.
119 * \param file_name file with the LLVM IR assembly or bitcode
120 * \return created `llvm::Module`
121 */
122 std::unique_ptr<llvm::Module> LoadIR(const std::string& file_name) const;
123
124 private:
125 std::unique_ptr<llvm::Module> ParseBuffer(const llvm::MemoryBuffer& buffer) const;
126
127 std::shared_ptr<llvm::LLVMContext> ctx_;
128};

130/*!
131 * \class LLVMTarget
132 * \brief Information used by LLVM for code generation for particular target
133 *
134 * This class contains all information that LLVM needs for code generation for
135 * a particular target. Since Target in TVM will soon contain command line
136 * flags for LLVM, objects of this class will handle saving and restoring
137 * global LLVM state that may be affected by these flags. This way, code
138 * generation for each LLVM-based target in TVM will start with the same LLVM
139 * global state.
140 *
141 * Note that objects of this class must be created within the lifetime of an
142 * LLVMInstance object.
143 */
144class LLVMTarget {
145 public:
146 /*!
147 * \brief Constructs LLVMTarget from `Target`
148 * \param scope LLVMInstance object
149 * \param target TVM Target object for target "llvm"
150 */
151 LLVMTarget(LLVMInstance& scope, const Target& target); // NOLINT(runtime/references)
152 /*!
153 * \brief Constructs LLVMTarget from target string
154 * \param scope LLVMInstance object
155 * \param target TVM target string for target "llvm"
156 */
157 LLVMTarget(LLVMInstance& scope, const std::string& target_str); // NOLINT(runtime/references)
158 /*!
159 * \brief Destroys LLVMTarget object
160 */
161 ~LLVMTarget();
162
163 /*!
164 * \brief Returns string representation (as TVM target) of the LLVMTarget
165 * \return Target string
166 *
167 * Note: If the LLVMTarget object was created from a string `s`, the string
168 * returned here may not be exactly equal to `s`. For example, if the CPU
169 * was "default", the returned string will have CPU set to the detected host
170 * CPU.
171 */
172 std::string str() const;
173
174 /*!
175 * \brief Get the LLVMInstance object from which the LLVMTarget object was
176 * created
177 * \return The enclosing LLVMInstance object
178 */
179 const LLVMInstance& GetInstance() const { return instance_; }
180 /*!
181 * \brief Get the current LLVM context
182 * \return the current LLVM context
183 */
184 llvm::LLVMContext* GetContext() const;
185 /*!
186 * \brief Return LLVM's `TargetMachine`, or nullptr
187 * \param allow_missing do not abort if the target machine cannot be created,
188 * return nullptr instead
189 * \return Pointer to the `TargetMachine` object (or nullptr if it cannot be
190 * created, \see allow_missing)
191 */
192 llvm::TargetMachine* GetOrCreateTargetMachine(bool allow_missing = false);
193
194 /*!
195 * \brief Get the target triple
196 * \return the target triple
197 */
198 const std::string& GetTargetTriple() const { return triple_; }
199 /*!
200 * \brief Get the CPU name
201 * \return the CPU name: the detected host CPU if the original TVM target
202 * specified it as "default"
203 */
204 const std::string& GetCPU() const { return cpu_; }
205 /*!
206 * \brief Get the list of LLVM target features
207 * \return array of individual feature strings
208 */
209 llvm::ArrayRef<std::string> GetTargetFeatures() const { return attrs_; }
210 /*!
211 * \brief Get the LLVM target feature string
212 * \return comma-separated list of LLVM target features
213 */
214 std::string GetTargetFeatureString() const;
215 /*!
216 * \brief Get the LLVM target options
217 * \return `llvm::TargetOptions` object for this target
218 */
219 const llvm::TargetOptions& GetTargetOptions() const { return target_options_; }
220 /*!
221 * \brief Get fast math flags
222 * \return `llvm::FastMathFlags` for this target
223 */
224 llvm::FastMathFlags GetFastMathFlags() const { return fast_math_flags_; }
225 /*!
226 * \brief Get the LLVM optimization level
227 * \return optimization level for this target
228 */
229 llvm::CodeGenOpt::Level GetOptLevel() const { return opt_level_; }
230
231 /*!
232 * \brief Extract the target string from given `llvm::Module`
233 * \param module LLVM module with the TVM target string embedded as metadata
234 * \return the target string from module's metadata
235 */
236 static std::string GetTargetMetadata(const llvm::Module& module);
237 /*!
238 * \brief Embed target string as metadata in given `llvm::Module`
239 * \param module the module to insert the target string into
240 */
241 void SetTargetMetadata(llvm::Module* module) const;
242
243 // Stubs to enable use with `With`.
244 void EnterWithScope() {}
245 void ExitWithScope() {}
246
247 private:
248 const LLVMInstance& instance_;
249 std::weak_ptr<llvm::LLVMContext> ctx_;
250
251 std::string triple_;
252 std::string cpu_;
253 std::vector<std::string> attrs_;
254 llvm::TargetOptions target_options_;
255 llvm::FastMathFlags fast_math_flags_;
256 llvm::CodeGenOpt::Level opt_level_;
257 llvm::Reloc::Model reloc_model_ = llvm::Reloc::PIC_;
258 llvm::CodeModel::Model code_model_ = llvm::CodeModel::Small;
259 std::shared_ptr<llvm::TargetMachine> target_machine_;
260};
