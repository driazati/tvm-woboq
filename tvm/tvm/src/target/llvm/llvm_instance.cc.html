80bool InitializeLLVM() {
81 static std::atomic_flag initialized = ATOMIC_FLAG_INIT;
82 if (!initialized.test_and_set()) {
83 llvm::InitializeAllTargetInfos();
84 llvm::InitializeAllTargets();
85 llvm::InitializeAllTargetMCs();
86 llvm::InitializeAllAsmParsers();
87 llvm::InitializeAllAsmPrinters();
88 }
89 return true;
90}

92std::string Join(std::string sep, llvm::ArrayRef<std::string> strings) {
93 std::string result;
94 bool is_first = true;
95 for (const std::string& s : strings) {
96 if (!is_first) {
97 result += sep;
98 }
99 result += s;
100 is_first = false;
101 }
102 return result;
103}

118 auto buffer = llvm::MemoryBuffer::getMemBuffer(llvm_ir, /*BufferName=*/"",

123std::unique_ptr<llvm::Module> LLVMInstance::LoadIR(const std::string& file_name) const {
124 llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> maybe_buffer =
125 llvm::MemoryBuffer::getFileAsStream(file_name);
126 if (std::error_code ec = maybe_buffer.getError()) {
127 LOG(FATAL) << ec.message();
128 }
129 return ParseBuffer(**maybe_buffer);
130}

132std::unique_ptr<llvm::Module> LLVMInstance::ParseBuffer(const llvm::MemoryBuffer& buffer) const {
133 llvm::SMDiagnostic error;
134 std::unique_ptr<llvm::Module> module = llvm::parseIR(buffer.getMemBufferRef(), error, *ctx_);
135 if (module == nullptr) {
136 std::string message;
137 llvm::raw_string_ostream ostream(message);
138 error.print(/*ProgName=*/nullptr, ostream, /*ShowColors=*/false, /*ShowKindLabel=*/true);
139 LOG(FATAL) << ostream.str();
140 }
141
142 return module;
143}

147std::ostream& operator<<(std::ostream& os, const LLVMTargetInfo::Option& opt) {
148 os << '-' << opt.name;
149 switch (opt.type) {
150 case LLVMTargetInfo::Option::OptType::Bool:
151 return os << ":bool=" << (opt.value.b ? "true" : "false");
152 case LLVMTargetInfo::Option::OptType::Int:
153 return os << ":int=" << opt.value.i;
154 case LLVMTargetInfo::Option::OptType::UInt:
155 return os << ":uint=" << opt.value.u;
156 case LLVMTargetInfo::Option::OptType::String:
157 return os << ":string=" << opt.value.s;
158 default:
159 os << ":?(" << static_cast<int>(opt.type) << ")";
160 break;
161 }
162 return os;
163}

165LLVMTargetInfo::LLVMTargetInfo(LLVMInstance& instance, const Target& target) {
166 triple_ = target->GetAttr<String>("mtriple").value_or("default");
167
168 if (triple_.empty() || triple_ == "default") {
169 triple_ = llvm::sys::getDefaultTargetTriple();
170 }
171 cpu_ = target->GetAttr<String>("mcpu").value_or(defaults::cpu);
172
173 if (const Optional<Array<String>>& v = target->GetAttr<Array<String>>("mattr")) {
174 for (const String& s : v.value()) {
175 attrs_.push_back(s);
176 }
177 }
178
179 if (const Optional<Array<String>>& v = target->GetAttr<Array<String>>("cl-opt")) {
180 llvm::StringMap<llvm::cl::Option*>& options = llvm::cl::getRegisteredOptions();
181 bool parse_error = false;
182 for (const String& s : v.value()) {
183 Option opt = ParseOptionString(s);
184 if (opt.type == Option::OptType::Invalid) {
185 parse_error = true;
186 continue;
187 }
188 if (options.count(opt.name)) {
189 llvm_options_.push_back(opt);
190 } else {
191 // Flag an error, but don't abort. LLVM flags may change, and this would
192 // give the code a chance to run even if the option no longer applies.
193 LOG(ERROR) << "\"" << opt.name << "\" is not an LLVM option, option ignored";
194 }
195 }
196 ICHECK(!parse_error) << "there were errors parsing command-line options";
197 }
198
199 llvm::FloatABI::ABIType float_abi = llvm::FloatABI::Default;
200 if (const Optional<String>& v = target->GetAttr<String>("mfloat-abi")) {
201 String value = v.value();
202 if (value == "hard") {
203 float_abi = llvm::FloatABI::Hard;
204 } else if (value == "soft") {
205 float_abi = llvm::FloatABI::Soft;
206 } else {
207 LOG(FATAL) << "invalid -mfloat-abi option " << value;
208 }
209 }
210
211 // Target options
212
213#if TVM_LLVM_VERSION < 50
214 target_options_.LessPreciseFPMADOption = true;
215#endif
216 // In clang, these are fed from LangOpts which describe language specific features
217 // TODO(AndrewZhaoLuo): figure out how these relate to fast math flags
218 target_options_.AllowFPOpFusion = llvm::FPOpFusion::Fast;
219 target_options_.UnsafeFPMath = false;
220 target_options_.NoInfsFPMath = false;
221 target_options_.NoNaNsFPMath = true;
222 target_options_.FloatABIType = float_abi;
223 if (const Optional<String>& v = target->GetAttr<String>("mabi")) {
224 target_options_.MCOptions.ABIName = v.value();
225 }
226
227 auto maybe_level = target->GetAttr<Integer>("opt-level");
228
229 if (maybe_level.defined()) {
230 int level = maybe_level.value()->value;
231 if (level <= 0) {
232 opt_level_ = llvm::CodeGenOpt::None;
233 } else if (level == 1) {
234 opt_level_ = llvm::CodeGenOpt::Less;
235 } else if (level == 2) {
236 opt_level_ = llvm::CodeGenOpt::Default;
237 } else {
238 // level >= 3
239 opt_level_ = llvm::CodeGenOpt::Aggressive;
240 }
241 } else {
242 opt_level_ = defaults::opt_level;
243 }
244
245 // Fast math options
246
247 auto GetBoolFlag = [&target](llvm::StringRef flag) -> bool {
248 return target->GetAttr<Bool>(flag.str()).value_or(Bool(false));
249 };
250 if (GetBoolFlag("fast-math")) {
251#if TVM_LLVM_VERSION >= 60
252 fast_math_flags_.setFast();
253#else
254 fast_math_flags_.setUnsafeAlgebra();
255#endif
256 } else {
257#if TVM_LLVM_VERSION >= 50
258 // This option was added in 5.x, and has a boolean argument,
259 // unlike the rest of options at the time.
260 fast_math_flags_.setAllowContract(GetBoolFlag("fast-math-contract"));
261#endif
262#if TVM_LLVM_VERSION >= 70
263 fast_math_flags_.setNoNaNs(GetBoolFlag("fast-math-nnan"));
264 fast_math_flags_.setNoInfs(GetBoolFlag("fast-math-ninf"));
265 fast_math_flags_.setNoSignedZeros(GetBoolFlag("fast-math-nsz"));
266 fast_math_flags_.setAllowReciprocal(GetBoolFlag("fast-math-arcp"));
267 fast_math_flags_.setAllowContract(GetBoolFlag("fast-math-contract"));
268 fast_math_flags_.setAllowReassoc(GetBoolFlag("fast-math-reassoc"));
269 fast_math_flags_.setApproxFunc(GetBoolFlag("fast-math-afn"));
270#else
271 // LLVM 4.x, 5.x, and 6.x
272 if (GetBoolFlag("fast-math-nnan")) fast_math_flags_.setNoNaNs();
273 if (GetBoolFlag("fast-math-ninf")) fast_math_flags_.setNoInfs();
274 if (GetBoolFlag("fast-math-nsz")) fast_math_flags_.setNoSignedZeros();
275 if (GetBoolFlag("fast-math-arcp")) fast_math_flags_.setAllowReciprocal();
276#if TVM_LLVM_VERSION >= 60
277 if (GetBoolFlag("fast-math-reassoc")) fast_math_flags_.setAllowReassoc();
278 if (GetBoolFlag("fast-math-afn")) fast_math_flags_.setApproxFunc();
279#endif
280#endif
281 }
282}

309std::string LLVMTargetInfo::str() const {
310 std::ostringstream os;
311 os << "llvm";
312 if (!triple_.empty()) {
313 os << " -mtriple=" << triple_;
314 }
315 if (!cpu_.empty() && cpu_ != defaults::cpu) {
316 os << " -mcpu=" << cpu_;
317 }
318 if (!attrs_.empty()) {
319 os << " -mattr=" << GetTargetFeatureString();
320 }
321
322 switch (target_options_.FloatABIType) {
323 case llvm::FloatABI::Soft:
324 os << " -mfloat-abi=soft";
325 break;
326 case llvm::FloatABI::Hard:
327 os << " -mfloat-abi=hard";
328 break;
329 case llvm::FloatABI::Default:
330 break;
331 }
332 if (!target_options_.MCOptions.ABIName.empty()) {
333 os << " -mabi=" << target_options_.MCOptions.ABIName;
334 }
335
336 bool do_individual = true;
337#if TVM_LLVM_VERSION >= 60
338 if (fast_math_flags_.isFast()) {
339 os << " -fast-math";
340 do_individual = false;
341 }
342#else
343 if (fast_math_flags_.unsafeAlgebra()) {
344 os << " -fast-math";
345 do_individual = false;
346 }
347#endif
348
349 if (do_individual) {
350 if (fast_math_flags_.noNaNs()) os << " -fast-math-nnan";
351 if (fast_math_flags_.noInfs()) os << " -fast-math-ninf";
352 if (fast_math_flags_.noSignedZeros()) os << " -fast-math-nsz";
353 if (fast_math_flags_.allowReciprocal()) os << " -fast-math-arcp";
354#if TVM_LLVM_VERSION >= 50
355 if (fast_math_flags_.allowContract()) os << " -fast-math-contract";
356#endif
357#if TVM_LLVM_VERSION >= 60
358 if (fast_math_flags_.allowReassoc()) os << " -fast-math-reassoc";
359 if (fast_math_flags_.approxFunc()) os << " -fast-math-afn";
360#endif
361 }
362
363 if (opt_level_ != defaults::opt_level) {
364 os << " -opt-level=";
365 switch (opt_level_) {
366 case llvm::CodeGenOpt::None:
367 os << "0";
368 break;
369 case llvm::CodeGenOpt::Less:
370 os << "1";
371 break;
372 case llvm::CodeGenOpt::Default:
373 os << "2";
374 break;
375 case llvm::CodeGenOpt::Aggressive:
376 os << "3";
377 break;
378 }
379 }
380
381 if (size_t num = llvm_options_.size(); num > 0) {
382 os << " -cl-opt=";
383 std::vector<std::string> opts;
384 for (const Option& opt : llvm_options_) {
385 std::stringstream os;
386 os << opt;
387 opts.emplace_back(os.str());
388 }
389 auto* quote = num > 1 ? "'" : "";
390 os << quote << Join(",", opts) << quote;
391 }
392
393 return os.str();
394}

396LLVMTargetInfo::Option LLVMTargetInfo::ParseOptionString(const std::string& str) {
397 Option opt;
398 opt.type = Option::OptType::Invalid;
399
400 // Option string: "-"+ <option_name> ":" <type> "=" <value>
401 //
402 // Note: "-"+ means 1 or more dashes, but only "-" are "--" valid.
403
404 // The first step is to do "lexing" of the option string, i.e. to break
405 // it up into parts (like "tokens") according to the syntax above. These
406 // parts will be non-overlapping substrings of the option string, and
407 // concatenated together, they will be equal to the option string.
408 // The literal elements are parts on their own.
409 //
410 // Note that the option string may be malformed, so any of the literal
411 // elements in the syntax may be missing.
412
413 std::vector<std::string> parts;
414
415 auto find_first_of = [](const std::string& str, const std::string& chars, auto start = 0) {
416 auto pos = str.find_first_of(chars, start);
417 return pos != std::string::npos ? pos : str.size();
418 };
419 auto find_first_not_of = [](const std::string& str, const std::string& chars, auto start = 0) {
420 auto pos = str.find_first_not_of(chars, start);
421 return pos != std::string::npos ? pos : str.size();
422 };
423
424 // "-"+
425 std::string::size_type pos_start = 0, pos_end = str.size();
426 std::string::size_type pos_at = find_first_not_of(str, "-", pos_start);
427 if (pos_at > 0) {
428 parts.push_back(str.substr(pos_start, pos_at));
429 }
430 // <option_name>, always present, may be empty string
431 pos_start = pos_at;
432 pos_at = find_first_of(str, ":=", pos_start);
433 parts.push_back(str.substr(pos_start, pos_at - pos_start));
434
435 // ":" or "=", if any
436 pos_start = pos_at;
437 char c = pos_start < pos_end ? str[pos_start] : 0;
438 if (c != 0) {
439 parts.emplace_back(1, c);
440 pos_start++;
441 }
442 // If the character found in the previous step wasn't '=', look for '='.
443 if (c == ':') {
444 // <type>
445 pos_at = find_first_of(str, "=", pos_start);
446 if (pos_at > pos_start) { // if non-empty
447 parts.push_back(str.substr(pos_start, pos_at - pos_start));
448 }
449
450 // "="
451 if (pos_at < pos_end) {
452 parts.emplace_back(1, str[pos_at]);
453 pos_start = pos_at + 1;
454 }
455 }
456 if (pos_start < pos_end) {
457 // <value>
458 parts.push_back(str.substr(pos_start));
459 }
460
461 // After breaking up the option string, examine and validate the individual
462 // parts.
463
464 int part_this = 0, part_end = parts.size();
465
466 const std::string error_header = "while parsing option \"" + str + "\": ";
467
468 // Check for "-" or "--".
469 if (part_this < part_end) {
470 auto& p = parts[part_this++];
471 if ((p.size() != 1 && p.size() != 2) || p.find_first_not_of('-') != std::string::npos) {
472 LOG(ERROR) << error_header << "option must start with \"-\" or \"--\"";
473 return opt;
474 }
475 }
476
477 // Validate option name.
478 if (part_this < part_end) {
479 auto& p = parts[part_this++];
480 if (p.empty()) {
481 LOG(ERROR) << error_header << "option name must not be empty";
482 return opt;
483 }
484 opt.name = std::move(p);
485 }
486
487 // Check type, if present.
488 Option::OptType type = Option::OptType::Invalid;
489 if (part_this < part_end) {
490 auto& p0 = parts[part_this];
491 if (p0 == ":") {
492 part_this++; // Only advance if we saw ":".
493 if (part_this < part_end) {
494 auto& p1 = parts[part_this];
495 ICHECK(!p1.empty()) << "tokenizing error"; // This shouldn't happen.
496 if (p1 != "=") {
497 part_this++;
498 if (p1 == "bool") {
499 type = Option::OptType::Bool;
500 } else if (p1 == "int") {
501 type = Option::OptType::Int;
502 } else if (p1 == "uint") {
503 type = Option::OptType::UInt;
504 } else if (p1 == "string") {
505 type = Option::OptType::String;
506 }
507 }
508 }
509 // If there was ":", there must be a type.
510 if (type == Option::OptType::Invalid) {
511 LOG(ERROR) << error_header << "invalid type";
512 return opt;
513 }
514 }
515 }
516
517 // Check value, if present.
518 std::optional<std::string> value;
519 if (part_this < part_end) {
520 auto& p0 = parts[part_this];
521 if (p0 == "=") {
522 part_this++;
523 if (part_this < part_end) {
524 value = std::move(parts[part_this]);
525 } else {
526 value = "";
527 }
528 } else {
529 // If there are still any parts left to be processed, there must be "=".
530 LOG(ERROR) << error_header << "expecting \"=\"";
531 return opt;
532 }
533 }
534
535 // NOLINTNEXTLINE(runtime/int)
536 auto to_integer = [](const std::string& s) -> std::optional<long long> {
537 // std::stoll takes "long long"
538 long long number; // NOLINT(runtime/int)
539 size_t pos;
540 try {
541 number = std::stoll(s, &pos);
542 } catch (...) {
543 return std::nullopt;
544 }
545 if (pos == s.size()) {
546 return number;
547 } else {
548 return std::nullopt;
549 }
550 };
551
552 auto to_boolean = [&to_integer](const std::string& s) -> std::optional<bool> {
553 // Return 0 or 1, if string corresponds to a valid boolean value,
554 // otherwise return 2.
555 auto ti = to_integer(s);
556 if (ti.has_value() && (ti.value() == 0 || ti.value() == 1)) {
557 return static_cast<bool>(ti.value());
558 }
559
560 std::string lower;
561 std::transform(s.begin(), s.end(), std::back_inserter(lower),
562 [](unsigned char c) { return std::tolower(c); });
563 if (lower == "true") {
564 return true;
565 } else if (lower == "false") {
566 return false;
567 }
568 return std::nullopt;
569 };
570
571 if (value.has_value()) {
572 if (type == Option::OptType::Int || type == Option::OptType::UInt) {
573 auto v = to_integer(value.value());
574 if (!v.has_value()) {
575 LOG(ERROR) << error_header << "invalid integer value \"" << value.value() << "\"";
576 return opt;
577 }
578 if (type == Option::OptType::Int) {
579 opt.value.i = static_cast<int>(v.value());
580 if (opt.value.i != v.value()) {
581 LOG(WARNING) << error_header << "value exceeds int range, assuming " << opt.value.i;
582 }
583 } else {
584 // NOLINTNEXTLINE(runtime/int)
585 opt.value.u = static_cast<unsigned>(static_cast<unsigned long long>(v.value()));
586 if (opt.value.u != static_cast<unsigned long long>(v.value())) { // NOLINT(runtime/int)
587 LOG(WARNING) << error_header << "value exceeds int range, assuming " << opt.value.u;
588 }
589 }
590 } else if (type == Option::OptType::String) {
591 opt.value.s = std::move(value.value());
592 } else {
593 // "type" is either Bool (given explicitly) or Invalid (type not present in string)
594 auto v = to_boolean(value.value());
595 if (!v.has_value()) {
596 LOG(ERROR) << error_header << "invalid boolean value \"" << value.value() << "\"";
597 return opt;
598 }
599 opt.value.b = v.value();
600 type = Option::OptType::Bool;
601 }
602 } else {
603 // Value was not present in string. Assume "true" if "type" is Bool or Invalid
604 if (type == Option::OptType::Bool || type == Option::OptType::Invalid) {
605 opt.value.b = true;
606 type = Option::OptType::Bool;
607 } else {
608 LOG(ERROR) << error_header << "must have a value";
609 return opt;
610 }
611 }
612
613 ICHECK(type != Option::OptType::Invalid);
614 opt.type = type;
615 return opt;
616}

618bool LLVMTargetInfo::MatchesGlobalState() const {
619 for (const Option& opt : GetCommandLineOptions()) {
620 Option current_opt = opt;
621 GetOptionValue(&current_opt);
622 ICHECK(current_opt.type != Option::OptType::Invalid);
623 switch (current_opt.type) {
624 case Option::OptType::Bool:
625 if (current_opt.value.b != opt.value.b) return false;
626 continue;
627 case Option::OptType::Int:
628 if (current_opt.value.i != opt.value.i) return false;
629 continue;
630 case Option::OptType::UInt:
631 if (current_opt.value.u != opt.value.u) return false;
632 continue;
633 case Option::OptType::String:
634 if (current_opt.value.s != opt.value.s) return false;
635 continue;
636 default:; // NOLINT(whitespace/semicolon)
637 }
638 }
639 return true;
640}

642void LLVMTargetInfo::GetOptionValue(LLVMTargetInfo::Option* opt) const {
643 llvm::StringMap<llvm::cl::Option*>& options = llvm::cl::getRegisteredOptions();
644 llvm::cl::Option* base_op = options[opt->name];
645
646 if (opt->type == Option::OptType::Bool) {
647 auto* bool_op = static_cast<llvm::cl::opt<bool>*>(base_op);
648 opt->value.b = bool_op->getValue();
649 } else if (opt->type == Option::OptType::Int) {
650 auto* int_op = static_cast<llvm::cl::opt<int>*>(base_op);
651 opt->value.i = int_op->getValue();
652 } else if (opt->type == Option::OptType::UInt) {
653 auto* uint_op = static_cast<llvm::cl::opt<unsigned>*>(base_op);
654 opt->value.u = uint_op->getValue();
655 } else if (opt->type == Option::OptType::String) {
656 auto* str_op = static_cast<llvm::cl::opt<std::string>*>(base_op);
657 opt->value.s = str_op->getValue();
658 } else {
659 opt->type = Option::OptType::Invalid;
660 }
661}

667LLVMTarget::LLVMTarget(LLVMInstance& instance, const LLVMTargetInfo& target_info)
668 : LLVMTargetInfo(target_info), instance_(instance), ctx_(instance.GetContext()) {
669 // Populate the list of saved options with the current values.
670 for (const Option& opt : GetCommandLineOptions()) {
671 GetOptionValue(&saved_llvm_options_.emplace_back(opt));
672 }
673
674 if (modified_llvm_state_) {
675 ICHECK(!ApplyLLVMOptions(true));
676 } else {
677 modified_llvm_state_ = ApplyLLVMOptions(true);
678 }
679}

685 : LLVMTarget(scope, Target(target_str)) {}

687LLVMTarget::~LLVMTarget() {
688 // Revert all applied LLVM options.
689 if (ApplyLLVMOptions(false)) {
690 modified_llvm_state_ = false;
691 }
692}

715bool LLVMTarget::ApplyLLVMOptions(bool apply_otherwise_revert, bool dry_run) {
716 llvm::StringMap<llvm::cl::Option*>& options = llvm::cl::getRegisteredOptions();
717 bool changed = false;
718
719#define HANDLE_OPTION_VALUE(option, new_val, saved_val) \
720 do { \
721 auto current = (option)->getValue(); \
722 auto replacement = apply_otherwise_revert ? (new_val) : (saved_val); \
723 if (current != replacement) { \
724 changed = true; \
725 if (!dry_run) { \
726 (option)->setValue(replacement); \
727 } \
728 } \
729 } while (false);
730
731 const auto& new_options = GetCommandLineOptions();
732 for (size_t i = 0, e = saved_llvm_options_.size(); i != e; ++i) {
733 const Option& new_opt = new_options[i];
734 const Option& saved_opt = saved_llvm_options_[i];
735
736 llvm::cl::Option* base_op = options[new_opt.name];
737
738 if (new_opt.type == Option::OptType::Bool) {
739 auto* bool_op = static_cast<llvm::cl::opt<bool>*>(base_op);
740 HANDLE_OPTION_VALUE(bool_op, new_opt.value.b, saved_opt.value.b);
741 } else if (new_opt.type == Option::OptType::Int) {
742 auto* int_op = static_cast<llvm::cl::opt<int>*>(base_op);
743 HANDLE_OPTION_VALUE(int_op, new_opt.value.i, saved_opt.value.i);
744 } else if (new_opt.type == Option::OptType::UInt) {
745 auto* uint_op = static_cast<llvm::cl::opt<unsigned>*>(base_op);
746 HANDLE_OPTION_VALUE(uint_op, new_opt.value.u, saved_opt.value.u);
747 } else if (new_opt.type == Option::OptType::String) {
748 auto* str_op = static_cast<llvm::cl::opt<std::string>*>(base_op);
749 HANDLE_OPTION_VALUE(str_op, new_opt.value.s, saved_opt.value.s);
750 } else {
751 LOG(FATAL) << "unexpected type in option " << new_opt;
752 }
753
754 if (dry_run && changed) {
755 return true;
756 }
757 }
758
759#undef HANDLE_OPTION_VALUE
760
761 return changed;
762}
