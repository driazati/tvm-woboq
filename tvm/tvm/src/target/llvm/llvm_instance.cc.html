74bool InitializeLLVM() {
75 static std::atomic_flag initialized = ATOMIC_FLAG_INIT;
76 if (!initialized.test_and_set()) {
77 llvm::InitializeAllTargetInfos();
78 llvm::InitializeAllTargets();
79 llvm::InitializeAllTargetMCs();
80 llvm::InitializeAllAsmParsers();
81 llvm::InitializeAllAsmPrinters();
82 }
83 return true;
84}
85
86std::string Join(std::string sep, llvm::ArrayRef<std::string> strings) {
87 std::string result;
88 bool is_first = true;
89 for (const std::string& s : strings) {
90 if (!is_first) {
91 result += sep;
92 }
93 result += s;
94 is_first = false;
95 }
96 return result;
97}

111std::unique_ptr<llvm::Module> LLVMInstance::ParseIR(const std::string& llvm_ir) const {
112 auto buffer = llvm::MemoryBuffer::getMemBuffer(llvm_ir, /*BufferName=*/"",
113 /*RequiresNullTerminator=*/false);
114 return ParseBuffer(*buffer);
115}
116
117std::unique_ptr<llvm::Module> LLVMInstance::LoadIR(const std::string& file_name) const {
118 llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> maybe_buffer =
119 llvm::MemoryBuffer::getFileAsStream(file_name);
120 if (std::error_code ec = maybe_buffer.getError()) {
121 LOG(FATAL) << ec.message();
122 }
123 return ParseBuffer(**maybe_buffer);
124}
125
126std::unique_ptr<llvm::Module> LLVMInstance::ParseBuffer(const llvm::MemoryBuffer& buffer) const {
127 llvm::SMDiagnostic error;
128 std::unique_ptr<llvm::Module> module = llvm::parseIR(buffer.getMemBufferRef(), error, *ctx_);
129 if (module == nullptr) {
130 std::string message;
131 llvm::raw_string_ostream ostream(message);
132 error.print(/*ProgName=*/nullptr, ostream, /*ShowColors=*/false, /*ShowKindLabel=*/true);
133 LOG(FATAL) << ostream.str();
134 }
135
136 return module;
137}

141LLVMTarget::LLVMTarget(LLVMInstance& instance, const Target& target)
142 : instance_(instance), ctx_(instance.GetContext()) {
143 triple_ = target->GetAttr<String>("mtriple").value_or("default");
144
145 if (triple_.empty() || triple_ == "default") {
146 triple_ = llvm::sys::getDefaultTargetTriple();
147 }
148 cpu_ = target->GetAttr<String>("mcpu").value_or(defaults::cpu);
149
150 if (const Optional<Array<String>>& v = target->GetAttr<Array<String>>("mattr")) {
151 for (const String& s : v.value()) {
152 attrs_.push_back(s);
153 }
154 }
155
156 llvm::FloatABI::ABIType float_abi = llvm::FloatABI::Default;
157 if (const Optional<String>& v = target->GetAttr<String>("mfloat-abi")) {
158 String value = v.value();
159 if (value == "hard") {
160 float_abi = llvm::FloatABI::Hard;
161 } else if (value == "soft") {
162 float_abi = llvm::FloatABI::Soft;
163 } else {
164 LOG(FATAL) << "invalid -mfloat-abi option " << value;
165 }
166 }
167
168 // Target options
169
170#if TVM_LLVM_VERSION < 50
171 target_options_.LessPreciseFPMADOption = true;
172#endif
173 // In clang, these are fed from LangOpts which describe language specific features
174 // TODO(AndrewZhaoLuo): figure out how these relate to fast math flags
175 target_options_.AllowFPOpFusion = llvm::FPOpFusion::Fast;
176 target_options_.UnsafeFPMath = false;
177 target_options_.NoInfsFPMath = false;
178 target_options_.NoNaNsFPMath = true;
179 target_options_.FloatABIType = float_abi;
180 if (const Optional<String>& v = target->GetAttr<String>("mabi")) {
181 target_options_.MCOptions.ABIName = v.value();
182 }
183
184 auto maybe_level = target->GetAttr<Integer>("opt-level");
185
186 if (maybe_level.defined()) {
187 int level = maybe_level.value()->value;
188 if (level <= 0) {
189 opt_level_ = llvm::CodeGenOpt::None;
190 } else if (level == 1) {
191 opt_level_ = llvm::CodeGenOpt::Less;
192 } else if (level == 2) {
193 opt_level_ = llvm::CodeGenOpt::Default;
194 } else {
195 // level >= 3
196 opt_level_ = llvm::CodeGenOpt::Aggressive;
197 }
198 } else {
199 opt_level_ = defaults::opt_level;
200 }
201
202 // Fast math options
203
204 auto GetBoolFlag = [&target](llvm::StringRef flag) -> bool {
205 return target->GetAttr<Bool>(flag.str()).value_or(Bool(false));
206 };
207 if (GetBoolFlag("fast-math")) {
208#if TVM_LLVM_VERSION >= 60
209 fast_math_flags_.setFast();
210#else
211 fast_math_flags_.setUnsafeAlgebra();
212#endif
213 } else {
214#if TVM_LLVM_VERSION >= 50
215 // This option was added in 5.x, and has a boolean argument,
216 // unlike the rest of options at the time.
217 fast_math_flags_.setAllowContract(GetBoolFlag("fast-math-contract"));
218#endif
219#if TVM_LLVM_VERSION >= 70
220 fast_math_flags_.setNoNaNs(GetBoolFlag("fast-math-nnan"));
221 fast_math_flags_.setNoInfs(GetBoolFlag("fast-math-ninf"));
222 fast_math_flags_.setNoSignedZeros(GetBoolFlag("fast-math-nsz"));
223 fast_math_flags_.setAllowReciprocal(GetBoolFlag("fast-math-arcp"));
224 fast_math_flags_.setAllowContract(GetBoolFlag("fast-math-contract"));
225 fast_math_flags_.setAllowReassoc(GetBoolFlag("fast-math-reassoc"));
226 fast_math_flags_.setApproxFunc(GetBoolFlag("fast-math-afn"));
227#else
228 // LLVM 4.x, 5.x, and 6.x
229 if (GetBoolFlag("fast-math-nnan")) fast_math_flags_.setNoNaNs();
230 if (GetBoolFlag("fast-math-ninf")) fast_math_flags_.setNoInfs();
231 if (GetBoolFlag("fast-math-nsz")) fast_math_flags_.setNoSignedZeros();
232 if (GetBoolFlag("fast-math-arcp")) fast_math_flags_.setAllowReciprocal();
233#if TVM_LLVM_VERSION >= 60
234 if (GetBoolFlag("fast-math-reassoc")) fast_math_flags_.setAllowReassoc();
235 if (GetBoolFlag("fast-math-afn")) fast_math_flags_.setApproxFunc();
236#endif
237#endif
238 }

251llvm::TargetMachine* LLVMTarget::GetOrCreateTargetMachine(bool allow_missing) {
252 if (target_machine_) return target_machine_.get();
253
254 std::string error;
255 if (const llvm::Target* llvm_instance = llvm::TargetRegistry::lookupTarget(triple_, error)) {
256 llvm::TargetMachine* tm =
257 llvm_instance->createTargetMachine(triple_, cpu_, GetTargetFeatureString(), target_options_,
258 reloc_model_, code_model_, opt_level_);
259 target_machine_ = std::unique_ptr<llvm::TargetMachine>(tm);
260 if (!allow_missing) {
261 ICHECK(target_machine_ != nullptr) << error;
262 }
263 }
264 return target_machine_.get();
265}

271std::string LLVMTarget::str() const {
272 std::ostringstream os;
273 os << "llvm";
274 if (!triple_.empty()) {
275 os << " -mtriple=" << triple_;
276 }
277 if (!cpu_.empty() && cpu_ != defaults::cpu) {
278 os << " -mcpu=" << cpu_;
279 }
280 if (!attrs_.empty()) {
281 os << " -mattr=" << GetTargetFeatureString();
282 }
283
284 switch (target_options_.FloatABIType) {
285 case llvm::FloatABI::Soft:
286 os << " -mfloat-abi=soft";
287 break;
288 case llvm::FloatABI::Hard:
289 os << " -mfloat-abi=hard";
290 break;
291 case llvm::FloatABI::Default:
292 break;
293 }
294 if (!target_options_.MCOptions.ABIName.empty()) {
295 os << " -mabi=" << target_options_.MCOptions.ABIName;
296 }
297
298 bool do_individual = true;
299#if TVM_LLVM_VERSION >= 60
300 if (fast_math_flags_.isFast()) {
301 os << " -fast-math";
302 do_individual = false;
303 }
304#else
305 if (fast_math_flags_.unsafeAlgebra()) {
306 os << " -fast-math";
307 do_individual = false;
308 }
309#endif
310
311 if (do_individual) {
312 if (fast_math_flags_.noNaNs()) os << " -fast-math-nnan";
313 if (fast_math_flags_.noInfs()) os << " -fast-math-ninf";
314 if (fast_math_flags_.noSignedZeros()) os << " -fast-math-nsz";
315 if (fast_math_flags_.allowReciprocal()) os << " -fast-math-arcp";
316#if TVM_LLVM_VERSION >= 50
317 if (fast_math_flags_.allowContract()) os << " -fast-math-contract";
318#endif
319#if TVM_LLVM_VERSION >= 60
320 if (fast_math_flags_.allowReassoc()) os << " -fast-math-reassoc";
321 if (fast_math_flags_.approxFunc()) os << " -fast-math-afn";
322#endif
323 }
324
325 if (opt_level_ != defaults::opt_level) {
326 os << " -opt-level=";
327 switch (opt_level_) {
328 case llvm::CodeGenOpt::None:
329 os << "0";
330 break;
331 case llvm::CodeGenOpt::Less:
332 os << "1";
333 break;
334 case llvm::CodeGenOpt::Default:
335 os << "2";
336 break;
337 case llvm::CodeGenOpt::Aggressive:
338 os << "3";
339 break;
340 }
341 }
342
343 return os.str();
344}

346std::string LLVMTarget::GetTargetMetadata(const llvm::Module& module) {
347 if (llvm::Metadata* tvm_target = module.getModuleFlag("tvm_target")) {
348 auto* mdstr = llvm::cast<llvm::MDString>(tvm_target);
349 llvm::StringRef meta = mdstr->getString();
350 if (meta.startswith("llvm")) {
351 return meta.str();
352 }
353 }
354 return "llvm -mtriple " + module.getTargetTriple();
355}
356
357void LLVMTarget::SetTargetMetadata(llvm::Module* module) const {
358 module->addModuleFlag(llvm::Module::Warning, "tvm_target",
359 llvm::MDString::get(*GetContext(), str()));
360}
