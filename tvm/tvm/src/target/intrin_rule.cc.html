124TVM_REGISTER_OP("tir.rsqrt")
125 .set_attr<FLegalize>("default.FLegalize", [](const PrimExpr& e) -> PrimExpr {
126 const CallNode* call = e.as<CallNode>();
127 ICHECK(call != nullptr);
128 auto one = make_const(call->args[0].dtype(), 1);
129 return one / sqrt(call->args[0]);
130 });

154TVM_REGISTER_OP("tir.q_multiply_shift")
155 .set_attr<FLegalize>("default.FLegalize", [](const PrimExpr& e) -> PrimExpr {
156 using tir::make_const;
157
158 const tir::CallNode* call = e.as<tir::CallNode>();
159 ICHECK(call != nullptr);
160
161 PrimExpr x = call->args[0];
162 PrimExpr y = call->args[1];
163 PrimExpr q = call->args[2];
164 PrimExpr s = call->args[3];
165
166 // Lambda function to extract the int value from PrimExpr
167 auto get_int_value = [](const PrimExpr node) {
168 if (auto int_node = node.as<IntImmNode>()) {
169 return int_node->value;
170 }
171 auto broadcast_node = node.as<BroadcastNode>();
172 CHECK(broadcast_node != nullptr);
173 auto int_node = broadcast_node->value.as<IntImmNode>();
174 CHECK(int_node != nullptr);
175 return int_node->value;
176 };
177 // Power of 2 is determined by the fixed_point_multiplier == 1 << 30. In case of power of
178 // 2, fixed point multiplier will represent a float value of 0.5. In fixed point, this is
179 // represented by 1 << 30.
180 if (get_int_value(y) == (1 << 30)) {
181 PrimExpr exp = s - 1;
182 int exp_val = get_int_value(s) - 1;
183 if (exp_val > 0) {
184 // power of 2 is greater than 0, apply left shift.
185 return x << exp;
186 } else {
187 // power of 2 is less than 0, round and then apply right shift.
188 DataType lp_dtype = DataType::Int(32, x.dtype().lanes());
189 PrimExpr one = make_const(lp_dtype, 1);
190 exp = -exp;
191 PrimExpr rounding_factor = one << (exp - 1);
192 PrimExpr rounded_t = x + rounding_factor;
193 return rounded_t >> exp;
194 }
195 } else {
196 // Only int32 types are supported (any number of lanes is allowed)
197 ICHECK(y.dtype().code() == DLDataTypeCode::kDLInt && y.dtype().bits() == 32);
198 ICHECK(s.dtype().code() == DLDataTypeCode::kDLInt && s.dtype().bits() == 32);
199
200 DataType hp_dtype = DataType::Int(64, x.dtype().lanes());
201 DataType lp_dtype = DataType::Int(32, x.dtype().lanes());
202
203 // 1) Calculating the integer multiplier and integer shift
204 PrimExpr zero = make_const(s.dtype(), 0);
205 PrimExpr left_shift = tir::Select(s > zero, s, zero);
206 PrimExpr right_shift = tir::Select(s > zero, zero, -s);
207
208 // 2) Cast and Multiply the integer multiplier
209 PrimExpr one = make_const(hp_dtype, 1);
210 x = cast(hp_dtype, x);
211 y = cast(hp_dtype, y);
212 x = tir::Select(left_shift != zero, x << left_shift, x);
213
214 // 3) Perform the multiplication in higher precision.
215 x = x * y;
216
217 // 4) Find the rounding scalar
218 PrimExpr total_right_shift = right_shift + q;
219 PrimExpr pos_rounding_value = (one << (total_right_shift - 1));
220 x = x + pos_rounding_value;
221
222 // 5) Simply right shift the result to get the final output.
223 x = x >> total_right_shift;
224
225 // 6) The fixed point multiplication keeps the value in int32 range. Casting back to
226 // int32.
227 return cast(lp_dtype, x);
228 }
229 });
