20/*!
21 * Defines functions that generate FuncRegistry structs for C runtime.
22 * \file func_registry_generator.cc
23 */
24
25#include "func_registry_generator.h"
26
27#include <sstream>
28
29namespace tvm {
30namespace target {
31
32std::string GenerateFuncRegistryNames(const Array<String>& function_names) {
33 std::stringstream ss;
34
35 unsigned char function_nums[sizeof(uint16_t)];
36 *reinterpret_cast<uint16_t*>(function_nums) = function_names.size();
37 for (auto f : function_nums) {
38 ss << f;
39 }
40
41 for (auto f : function_names) {
42 ss << f << '\0';
43 }
44
45 return ss.str();
46}

48} // namespace target
