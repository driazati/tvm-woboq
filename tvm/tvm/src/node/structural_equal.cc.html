55// Define the dispatch function here since primary user is in this file.
56bool ReflectionVTable::SEqualReduce(const Object* self, const Object* other,
57 SEqualReducer equal) const {
58 uint32_t tindex = self->type_index();
59 if (tindex >= fsequal_reduce_.size() || fsequal_reduce_[tindex] == nullptr) {
60 LOG(FATAL) << "TypeError: SEqualReduce of " << self->GetTypeKey()
61 << " is not registered via TVM_REGISTER_NODE_TYPE."
62 << " Did you forget to set _type_has_method_sequal_reduce=true?";
63 }
64 return fsequal_reduce_[tindex](self, other, equal);
65}

73 ObjectPathPair GetPathsForAttrs(const ObjectRef& lhs, const ObjectRef& rhs) const {
74 Optional<String> lhs_attr_key = GetAttrKeyByAddress(lhs_object.get(), &lhs);
75 Optional<String> rhs_attr_key = GetAttrKeyByAddress(rhs_object.get(), &rhs);
76 return ObjectPathPair(current_paths->lhs_path->Attr(lhs_attr_key),
77 current_paths->rhs_path->Attr(rhs_attr_key));
78 }

81bool SEqualReducer::operator()(const ObjectRef& lhs, const ObjectRef& rhs) const {
82 if (tracing_data_ == nullptr) {
83 // Fast path: no tracing
84 return handler_->SEqualReduce(lhs, rhs, map_free_vars_, NullOpt);
85 }
86 return ObjectAttrsEqual(lhs, rhs, map_free_vars_, nullptr);

89bool SEqualReducer::DefEqual(const ObjectRef& lhs, const ObjectRef& rhs) {

97/* static */ void SEqualReducer::GetPathsFromAttrAddressesAndStoreMismatch(
98 const void* lhs_address, const void* rhs_address, const PathTracingData* tracing_data) {
99 if (tracing_data != nullptr && !tracing_data->first_mismatch->defined()) {
100 Optional<String> lhs_attr_key =
101 GetAttrKeyByAddress(tracing_data->lhs_object.get(), lhs_address);
102 Optional<String> rhs_attr_key =
103 GetAttrKeyByAddress(tracing_data->rhs_object.get(), rhs_address);
104 *tracing_data->first_mismatch =
105 ObjectPathPair(tracing_data->current_paths->lhs_path->Attr(lhs_attr_key),
106 tracing_data->current_paths->rhs_path->Attr(rhs_attr_key));
107 }
108}

110template <typename T>
111/* static */ bool SEqualReducer::CompareAttributeValues(const T& lhs, const T& rhs,
112 const PathTracingData* tracing_data) {
113 if (BaseValueEqual()(lhs, rhs)) {
114 return true;
115 } else {
116 GetPathsFromAttrAddressesAndStoreMismatch(&lhs, &rhs, tracing_data);
117 return false;
118 }
119}

173bool SEqualReducer::ObjectAttrsEqual(const ObjectRef& lhs, const ObjectRef& rhs, bool map_free_vars,
174 const ObjectPathPair* paths) const {
175 if (tracing_data_ == nullptr) {
176 // Fast path: no tracing
177 return handler_->SEqualReduce(lhs, rhs, map_free_vars, NullOpt);
178 }
179
180 // Slow path: tracing object paths for better error reporting
181
182 ObjectPathPair new_paths = paths == nullptr ? tracing_data_->GetPathsForAttrs(lhs, rhs) : *paths;
183
184 if (handler_->SEqualReduce(lhs, rhs, map_free_vars, new_paths)) {
185 return true;
186 } else {
187 if (!tracing_data_->first_mismatch->defined()) {
188 *tracing_data_->first_mismatch = new_paths;
189 }
190 return false;
191 }

194/*!
195 * \brief A non recursive stack based SEqual handler that can remaps vars.
196 *
197 * This handler pushs the Object equality cases into a stack, and
198 * traverses the stack to expand the necessary children that need to be checked.
199 *
200 * The order of SEqual being called is the same as the order as if we
201 * eagerly do recursive calls in SEqualReduce.
202 */
203class SEqualHandlerDefault::Impl {
204 public:
205 Impl(SEqualHandlerDefault* parent, bool assert_mode, Optional<ObjectPathPair>* first_mismatch)
206 : parent_(parent), assert_mode_(assert_mode), first_mismatch_(first_mismatch) {}
207
208 bool SEqualReduce(const ObjectRef& lhs, const ObjectRef& rhs, bool map_free_vars,
209 const Optional<ObjectPathPair>& current_paths) {
210 // We cannot use check lhs.same_as(rhs) to check equality.
211 // if we choose to enable var remapping.
212 //
213 // Counter example below (%x, %y) are shared vars
214 // between the two functions(possibly before/after rewriting).
215 //
216 // - function0: fn (%x, %y) { %x + %y }
217 // - function1. fn (%y, %x) { %x + %y }
218 //
219 // Because we choose to enable var remapping,
220 // %x is mapped to %y, and %y is mapped to %x,
221 // the body of the function no longer means the same thing.
222 //
223 // Take away: We can either choose only compare Var by address,
224 // in which case we can use same_as for quick checking,
225 // or we have to run deep comparison and avoid to use same_as checks.
226 auto run = [=]() {
227 if (!lhs.defined() && !rhs.defined()) return true;
228 if (!lhs.defined() && rhs.defined()) return false;
229 if (!rhs.defined() && lhs.defined()) return false;
230 if (lhs->type_index() != rhs->type_index()) return false;
231 auto it = equal_map_lhs_.find(lhs);
232 if (it != equal_map_lhs_.end()) {
233 return it->second.same_as(rhs);
234 }
235 if (equal_map_rhs_.count(rhs)) return false;
236
237 // need to push to pending tasks in this case
238 pending_tasks_.emplace_back(lhs, rhs, map_free_vars, current_paths);
239 return true;
240 };
241 return CheckResult(run(), lhs, rhs, current_paths);
242 }
243
244 void DeferFail(const ObjectPathPair& mismatch_paths) {
245 pending_tasks_.emplace_back(Task::ForceFailTag{}, mismatch_paths);
246 }
247
248 void MarkGraphNode() {
249 // need to push to pending tasks in this case
250 ICHECK(!allow_push_to_stack_ && !task_stack_.empty());
251 task_stack_.back().graph_equal = true;
252 }
253
254 ObjectRef MapLhsToRhs(const ObjectRef& lhs) {
255 auto it = equal_map_lhs_.find(lhs);
256 if (it != equal_map_lhs_.end()) return it->second;
257 return ObjectRef(nullptr);
258 }
259
260 // Function that implements actual equality check.
261 bool Equal(const ObjectRef& lhs, const ObjectRef& rhs, bool map_free_vars) {
262 if (!lhs.defined() && !rhs.defined()) return true;
263 task_stack_.clear();
264 pending_tasks_.clear();
265 equal_map_lhs_.clear();
266 equal_map_rhs_.clear();
267
268 Optional<ObjectPathPair> current_paths;
269 if (IsPathTracingEnabled()) {
270 auto root_path = ObjectPath::Root();
271 current_paths = ObjectPathPair(root_path, root_path);
272 }
273 if (!SEqualReduce(lhs, rhs, map_free_vars, current_paths)) {
274 return false;
275 }
276
277 ICHECK_EQ(pending_tasks_.size(), 1U);
278 ICHECK(allow_push_to_stack_);
279 task_stack_.emplace_back(std::move(pending_tasks_.back()));
280 pending_tasks_.clear();
281 return RunTasks();
282 }

284 // The default equal as registered in the structural equal vtable.
285 bool DispatchSEqualReduce(const ObjectRef& lhs, const ObjectRef& rhs, bool map_free_vars,
286 const Optional<ObjectPathPair>& current_paths) {
287 auto compute = [=]() {
288 ICHECK(lhs.defined() && rhs.defined() && lhs->type_index() == rhs->type_index());
289 // skip entries that already have equality maps.
290 auto it = equal_map_lhs_.find(lhs);
291 if (it != equal_map_lhs_.end()) {
292 return it->second.same_as(rhs);
293 }
294 if (equal_map_rhs_.count(rhs)) return false;
295
296 if (!IsPathTracingEnabled()) {
297 return vtable_->SEqualReduce(lhs.get(), rhs.get(),
298 SEqualReducer(parent_, nullptr, map_free_vars));
299 } else {
300 PathTracingData tracing_data = {current_paths.value(), lhs, rhs, first_mismatch_};
301 return vtable_->SEqualReduce(lhs.get(), rhs.get(),
302 SEqualReducer(parent_, &tracing_data, map_free_vars));
303 }
304 };
305 return CheckResult(compute(), lhs, rhs, current_paths);
306 }
307
308 protected:
309 // Check the result.
310 bool CheckResult(bool result, const ObjectRef& lhs, const ObjectRef& rhs,
311 const Optional<ObjectPathPair>& current_paths) {
312 if (IsPathTracingEnabled() && !result && !first_mismatch_->defined()) {
313 *first_mismatch_ = current_paths;
314 }
315 if (assert_mode_ && !result) {
316 LOG(FATAL) << "ValueError: StructuralEqual check failed, caused by lhs:" << std::endl
317 << PrettyPrint(lhs) << std::endl
318 << "and rhs:" << std::endl
319 << PrettyPrint(rhs);
320 }
321 return result;
322 }
323 /*!
324 * \brief Run tasks until the stack reaches the stack begin
325 * \param stack_begin The expected beginning of the stack.
326 * \return The checks we encountered throughout the process.
327 */
328 bool RunTasks() {
329 while (task_stack_.size() != 0) {
330 // Caution: entry becomes invalid when the stack changes
331 auto& entry = task_stack_.back();
332
333 if (entry.force_fail) {
334 if (IsPathTracingEnabled() && !first_mismatch_->defined()) {
335 *first_mismatch_ = entry.current_paths;
336 }
337 return false;
338 }
339
340 if (entry.children_expanded) {
341 // When all the children has expanded and visited.
342 // This means all the condition checks for
343 // the current entry has been passed
344 // We can safely mark lhs and rhs as equal to each other.
345 auto it = equal_map_lhs_.find(entry.lhs);
346 if (it != equal_map_lhs_.end()) {
347 ICHECK(it->second.same_as(entry.rhs));
348 }
349 // create the map if the quality is graph equal.
350 if (entry.graph_equal) {
351 equal_map_lhs_[entry.lhs] = entry.rhs;
352 equal_map_rhs_[entry.rhs] = entry.lhs;
353 }
354 task_stack_.pop_back();
355 } else {
356 // mark before expand
357 // Important: because entry becomes invalid when stack changes.
358 entry.children_expanded = true;
359 // Expand the objects
360 // The SEqual of the object can call into this->SEqualReduce
361 // which populates the pending tasks.
362 ICHECK_EQ(pending_tasks_.size(), 0U);
363 allow_push_to_stack_ = false;
364 if (!parent_->DispatchSEqualReduce(entry.lhs, entry.rhs, entry.map_free_vars,
365 entry.current_paths))
366 return false;
367 allow_push_to_stack_ = true;
368 // Push pending tasks in reverse order, so earlier tasks get to
369 // expand first in the stack
370 while (pending_tasks_.size() != 0) {
371 task_stack_.emplace_back(std::move(pending_tasks_.back()));
372 pending_tasks_.pop_back();
373 }
374 }
375 }
376 return true;
377 }
378
379 private:
380 /*! \brief Pending reduce tasks. */
381 struct Task {
382 /*! \brief The lhs operand to be compared. */
383 ObjectRef lhs;
384 /*! \brief The rhs operand to be compared. */
385 ObjectRef rhs;
386 /*! \brief If path tracing is enabled, paths taken so far from the root to `lhs` and `rhs`
387 * objects. */
388 Optional<ObjectPathPair> current_paths;
389 /*! \brief The map free var argument. */
390 bool map_free_vars;
391 /*! \brief Whether the children has been expanded via SEqualReduce */
392 bool children_expanded{false};
393 /*! \brief whether the task is about graph equality(need remap). */
394 bool graph_equal{false};
395 /*! \brief whether the task should return "false" without actually comparing anything */
396 bool force_fail{false};
397
398 Task() = default;
399 Task(ObjectRef lhs, ObjectRef rhs, bool map_free_vars, Optional<ObjectPathPair> current_paths)
400 : lhs(lhs),
401 rhs(rhs),
402 current_paths(std::move(current_paths)),
403 map_free_vars(map_free_vars) {}
404
405 struct ForceFailTag {}; // dispatch tag for the constructor below
406 Task(ForceFailTag, const ObjectPathPair& current_paths)
407 : current_paths(current_paths), force_fail(true) {}
408 };
409
410 bool IsPathTracingEnabled() const { return first_mismatch_ != nullptr; }
411
412 // The owner of this impl
413 SEqualHandlerDefault* parent_;
414 // list of pending tasks to be pushed to the stack.
415 std::vector<Task> pending_tasks_;
416 // Internal task stack to executed the task.
417 std::vector<Task> task_stack_;
418 // Whether we allow push to stack.
419 bool allow_push_to_stack_{true};
420 // If in assert mode, must return true, and will throw error otherwise.
421 bool assert_mode_{false};
422 // Location to store the paths to the first detected mismatch, or nullptr to disable path tracing.
423 Optional<ObjectPathPair>* first_mismatch_;
424 // reflection vtable
425 ReflectionVTable* vtable_ = ReflectionVTable::Global();
426 // map from lhs to rhs
427 std::unordered_map<ObjectRef, ObjectRef, ObjectPtrHash, ObjectPtrEqual> equal_map_lhs_;
428 // map from rhs to lhs
429 std::unordered_map<ObjectRef, ObjectRef, ObjectPtrHash, ObjectPtrEqual> equal_map_rhs_;
430};

481bool NDArrayEqual(const runtime::NDArray::Container* lhs, const runtime::NDArray::Container* rhs,
482 SEqualReducer equal, bool compare_data) {
483 if (lhs == rhs) return true;
484
485 auto ldt = lhs->dl_tensor.dtype;
486 auto rdt = rhs->dl_tensor.dtype;
487 ICHECK_EQ(lhs->dl_tensor.device.device_type, kDLCPU) << "can only compare CPU tensor";
488 ICHECK_EQ(rhs->dl_tensor.device.device_type, kDLCPU) << "can only compare CPU tensor";
489 ICHECK(runtime::IsContiguous(lhs->dl_tensor)) << "Can only compare contiguous tensor";
490 ICHECK(runtime::IsContiguous(rhs->dl_tensor)) << "Can only compare contiguous tensor";
491
492 if (lhs->dl_tensor.ndim != rhs->dl_tensor.ndim) return false;
493 for (int i = 0; i < lhs->dl_tensor.ndim; ++i) {
494 if (!equal(lhs->dl_tensor.shape[i], rhs->dl_tensor.shape[i])) return false;
495 }
496 if (ldt.code == rdt.code && ldt.lanes == rdt.lanes && ldt.bits == rdt.bits) {
497 size_t data_size = runtime::GetDataSize(lhs->dl_tensor);
498 if (compare_data) {
499 return std::memcmp(lhs->dl_tensor.data, rhs->dl_tensor.data, data_size) == 0;
500 } else {
501 return true;
502 }
503 } else {
504 return false;
505 }
506}
