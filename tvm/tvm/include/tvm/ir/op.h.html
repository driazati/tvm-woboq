47/*!
48 * \brief Primitive Op(builtin intrinsics)
49 *
50 * This data structure stores the meta-data
51 * about primitive operators that can be invoked via Call.
52 *
53 * Low-level IR intrinsics(such as libc.expf) are also
54 * implemented via Op.
55 *
56 * \sa Op
57 */
58class OpNode : public RelayExprNode {
59 public:
60 /*! \brief name of the operator */
61 String name;
62 /*! \brief the type of the operator */
63 mutable FuncType op_type;
64 /*!
65 * \brief detailed description of the operator
66 * This can be used to generate docstring automatically for the operator.
67 */
68 String description;
69 /* \brief Information of input arguments to the operator */
70 Array<AttrFieldInfo> arguments;
71 /*!
72 * \brief The type key of the attribute field
73 * This can be empty, in which case it defaults to anything.
74 */
75 String attrs_type_key;
76 /*!
77 * \brief attribute type index,
78 * this field varies in each run and is not exposed to frontend.
79 */
80 uint32_t attrs_type_index{0};
81 /*!
82 * \brief number of input arguments to the operator,
83 * -1 means it is variable length
84 */
85 int32_t num_inputs = -1;
86 /*!
87 * \brief support level of the operator,
88 * The lower the more priority it contains.
89 * This is in analogies to BLAS levels.
90 */
91 int32_t support_level = 10;
92
93 void VisitAttrs(AttrVisitor* v) {
94 v->Visit("name", &name);
95 v->Visit("op_type", &op_type);
96 v->Visit("description", &description);
97 v->Visit("arguments", &arguments);
98 v->Visit("attrs_type_key", &attrs_type_key);
99 v->Visit("num_inputs", &num_inputs);
100 v->Visit("support_level", &support_level);
101 }
102
103 bool SEqualReduce(const OpNode* other, SEqualReducer equal) const {
104 // pointer equality is fine as there is only one op with the same name.
105 return this == other;
106 }
107
108 void SHashReduce(SHashReducer hash_reduce) const {
109 // Name uniquely identifies an Op.
110 hash_reduce(name);
111 }
112
113 /*!
114 * \brief Check that if current op is a "primtive operator".
115 * That is the arguments are all type variables, and there is a single
116 * type relation applied to the input and output types.
117 */
118 bool IsPrimitiveOp() const {
119 if (is_primitive_ != -1) return is_primitive_ != 0;
120 is_primitive_ = this->IsPrimitiveOp_() ? 1 : 0;
121 return is_primitive_ != 0;
122 }
123
124 static constexpr const char* _type_key = "Op";
125 TVM_DECLARE_FINAL_OBJECT_INFO(OpNode, RelayExprNode);
126
127 private:
128 /*! \return the internal attr registry index. */
129 uint32_t AttrRegistryIndex() const { return index_; }
130 /*! \brief repr to be printed in registry*/
131 std::string AttrRegistryName() const { return name; }
132
133 // friend class
134 template <typename>
135 friend class AttrRegistryMapContainerMap;
136 template <typename, typename>
137 friend class AttrRegistry;
138 friend class OpRegEntry;
139
140 friend bool IsPrimitiveOp(const RelayExpr&);
141 // Program internal unique index of operator.
142 // Used to help index the program.
143 uint32_t index_{0};
144 // whether this is a primitive op. -1 means unknown.
145 mutable int is_primitive_{-1};
146 // Internal function to compute if it is primitive op
147 bool IsPrimitiveOp_() const {
148 const auto& fn_ty = this->op_type;
149 ICHECK(fn_ty.get() != nullptr) << "op_type of " << this->name << " is not registered";
150 if (fn_ty->type_constraints.size() != 1) return false;
151 const TypeRelationNode* rel = fn_ty->type_constraints[0].as<TypeRelationNode>();
152 if (rel == nullptr) return false;
153 // validate if the type parameter matches up
154 for (size_t i = 0; i < fn_ty->type_params.size(); ++i) {
155 if (!fn_ty->type_params[i].same_as(rel->args[i])) return false;
156 }
157 return true;
158 }
159};

161/*!
162 * \brief Managed reference class to OpNode.
163 * \sa OpNode
164 */
165class Op : public RelayExpr {
166 public:
167 /*! \brief default constructor */
168 Op() {}
169 /*! \brief constructor from node pointer */
170 explicit Op(ObjectPtr<Object> n) : RelayExpr(n) {}
171 /*!
172 * \brief access the internal node container
173 * \return the pointer to the internal node container
174 */
175 inline const OpNode* operator->() const;
176 /*!
177 * \brief Get additional registered attribute about operators.
178 * If nothing has been registered, an empty OpAttrMap will be returned.
179 * \param attr_name The name of the attribute.
180 * \return An OpAttrMap of specified attr_name.
181 * \tparam ValueType The type of the attribute.
182 */
183 template <typename ValueType>
184 inline static OpAttrMap<ValueType> GetAttrMap(const String& attr_name);
185 /*!
186 * \brief Checks if an attr map is present in the registry.
187 * \param attr_name The name of the attribute.
188 * \return bool True if the attr is present.
189 */
190 TVM_DLL static bool HasAttrMap(const String& attr_name);
191 /*!
192 * \brief Get an Op for a given operator name.
193 * Will raise an error if the op has not been registered.
194 * \param op_name Name of the operator.
195 * \return Pointer to a Op, valid throughout program lifetime.
196 */
197 TVM_DLL static const Op& Get(const String& op_name);
198
199 /*! \brief specify container node */
200 using ContainerType = OpNode;
201
202 private:
203 /*!
204 * \brief Get generic attrmap given attr name
205 * \param key The attribute key
206 * \return The attr map.
207 */
208 TVM_DLL static const AttrRegistryMapContainerMap<Op>& GetAttrMapContainer(const String& key);
209};

211/*!
212 * \brief Helper structure to register operators
213 * \sa TVM_REGISTER_OP
214 */
215class OpRegEntry {
216 public:
217 /*! \return the operator */
218 const Op& op() const { return op_; }
219 /*!
220 * \brief setter function during registration
221 * Set the description of operator
222 * \param descr the description string.
223 * \return reference to self.
224 */
225 inline OpRegEntry& describe(const std::string& descr); // NOLINT(*)
226 /*!
227 * \brief Add argument information to the function.
228 * \param name Name of the argument.
229 * \param type Type of the argument.
230 * \param description Description of the argument.
231 * \return reference to self.
232 */
233 inline OpRegEntry& add_argument(const std::string& name, const std::string& type,
234 const std::string& description);
235 /*!
236 * \brief Attach the type function corresponding to the return type.
237 * \param rel_name The type relation name to register.
238 * \param type_rel_func The backing relation function which can solve an arbitrary
239 * relation on variables.
240 * \return reference to self.
241 */
242 inline OpRegEntry& add_type_rel(
243 const std::string& rel_name,
244 runtime::TypedPackedFunc<bool(const Array<Type>&, int, const Attrs&, const TypeReporter&)>
245 type_rel_func);
246 /*!
247 * \brief Set the attrs type key and index to be AttrsType.
248 * \tparam AttrsType the attribute type to b set.
249 * \return reference to self.
250 */
251 template <typename AttrsType>
252 inline OpRegEntry& set_attrs_type();
253 /*!
254 * \brief Set the attrs type key and index to be AttrsType.
255 * \param key The attribute type key to be set.
256 * \return reference to self.
257 */
258 inline OpRegEntry& set_attrs_type_key(const String& key);
259 /*!
260 * \brief Set the num_inputs
261 * \param n The number of inputs to be set.
262 * \return reference to self.
263 */
264 inline OpRegEntry& set_num_inputs(int32_t n); // NOLINT(*)
265 /*!
266 * \brief Set the support level of op.
267 * \param level The support level.
268 * \return reference to self.
269 */
270 inline OpRegEntry& set_support_level(int32_t level); // NOLINT(*)
271 /*!
272 * \brief Register additional attributes to operator.
273 * \param attr_name The name of the attribute.
274 * \param value The value to be set.
275 * \param plevel The priority level of this set,
276 * an higher priority level attribute
277 * will replace lower priority level attribute.
278 * Must be bigger than 0.
279 *
280 * Cannot set with same plevel twice in the code.
281 *
282 * \tparam ValueType The type of the value to be set.
283 */
284 template <typename ValueType>
285 inline OpRegEntry& set_attr(const std::string& attr_name, // NOLINT(*)
286 const ValueType& value, int plevel = 10);
287
288 /*!
289 * \brief Resets an attr of the registry.
290 * \param attr_name The name of the attribute.
291 */
292 inline void reset_attr(const std::string& attr_name);
293
294 // set the name of the op to be the same as registry
295 inline OpRegEntry& set_name() { // NOLINT(*)
296 if (get()->name.length() == 0) {
297 get()->name = name;
298 }
299 return *this;
300 }
301 /*!
302 * \brief Register or get a new entry.
303 * \param name The name of the operator.
304 * \return the corresponding entry.
305 */
306 TVM_DLL static OpRegEntry& RegisterOrGet(const String& name);
307
308 private:
309 template <typename, typename>
310 friend class AttrRegistry;
311 // the name
312 std::string name;
313 /*! \brief The operator */
314 Op op_;
315 // private constructor
316 TVM_DLL OpRegEntry(uint32_t reg_index);
317 // return internal pointer to op.
318 inline OpNode* get();
319 // update the attribute OpAttrMap
320 TVM_DLL void UpdateAttr(const String& key, runtime::TVMRetValue value, int plevel);
321};

323/*!
324 * \brief Map<Op,ValueType> used to store meta-information about Op.
325 * \tparam ValueType The type of the value stored in map.

327template <typename ValueType>
328class OpAttrMap : public AttrRegistryMap<Op, ValueType> {
329 public:
330 /*!
331 * \brief get the corresponding value element at op with default value.
332 * \param expr The key to the map
333 * \param def_value The default value when the key does not exist
334 * or if expr is not an Op.
335 * \return the const reference to the content value.
336 */
337 inline ValueType get(const RelayExpr& expr, ValueType def_value) const;
338
339 using TParent = AttrRegistryMap<Op, ValueType>;
340 using TParent::count;
341 using TParent::get;
342 using TParent::operator[];
343
344 private:
345 friend class Op;
346 // constructor
347 explicit OpAttrMap(const AttrRegistryMapContainerMap<Op>& map) : TParent(map) {}
348};

397inline OpRegEntry& OpRegEntry::add_type_rel(
398 const std::string& rel_name,
399 runtime::TypedPackedFunc<bool(const Array<Type>&, int, const Attrs&, const TypeReporter&)>
400 type_rel_func) {
401 auto func_name = std::string("tvm.relay.type_relation.") + rel_name;
402 TypeRelationFn env_type_rel_func;
403
404 if (runtime::Registry::Get(func_name)) {
405 auto env_func = EnvFunc::Get(func_name);
406 env_type_rel_func = env_func;
407 } else {
408 runtime::Registry::Register(func_name).set_body(type_rel_func.packed());
409 auto env_func = EnvFunc::Get(func_name);
410 env_type_rel_func = env_func;
411 }
412
413 Array<TypeVar> type_params;
414 Array<Type> arg_types;
415
416 // Add inputs.
417 std::string input_name_prefix = "in";
418 for (int i = 0; i < get()->num_inputs; i++) {
419 auto name = input_name_prefix + std::to_string(i);
420 auto param = TypeVar(name, TypeKind::kType);
421 type_params.push_back(param);
422 arg_types.push_back(param);
423 }
424
425 Array<Type> ty_call_args = arg_types;
426
427 // Add output type.
428 auto out_param = TypeVar("out", TypeKind::kType);
429 type_params.push_back(out_param);
430 // this will trigger copy on write.
431 ty_call_args.push_back(out_param);
432
433 // The attributes of primitive op is nullptr
434 //
435 // The attributes of primitive operator can vary at the call site.
436 // The type of sum is also dependent on Attrs being passed.
437 // So puting nullptr in the Attrs means that the operator is polymorphic on Attrs.
438 //
439 // A common example is sum(x, axis), where the choice of axis
440 // can affect the type of the function.
441 TypeConstraint type_rel =
442 TypeRelation(env_type_rel_func, ty_call_args, arg_types.size(), Attrs());
443
444 auto func_type = FuncType(arg_types, out_param, type_params, {type_rel});
445
446 get()->op_type = func_type;
447
448 return *this;
449}

474template <typename ValueType>
475inline OpRegEntry& OpRegEntry::set_attr( // NOLINT(*)
476 const std::string& attr_name, const ValueType& value, int plevel) {
477 ICHECK_GT(plevel, 0) << "plevel in set_attr must be greater than 0";
478 runtime::TVMRetValue rv;

480 UpdateAttr(attr_name, rv, plevel);
481 return *this;
482}

486template <typename ValueType>
487inline ValueType OpAttrMap<ValueType>::get(const RelayExpr& expr, ValueType def_value) const {
488 ICHECK(expr.defined());
489 if (const OpNode* op = expr.as<OpNode>()) {
490 return this->map_.get(GetRef<Op>(op), def_value);
491 } else {
492 return def_value;
493 }

496/*!
497 * \brief Check that an expression is a "primitive operator".
498 *
499 * Will return true if the expression is an operator which
500 * matches the form of primitive operators registered directly
501 * by the Relay codebase.
502 *
503 * That is the arguments are all type variables, and there is a single
504 * type relation applied to the input and output types.
505 *
506 * \param expr An expression.
507 * \return Whether the expression is primitive op.
508 */
509inline bool IsPrimitiveOp(const RelayExpr& expr) {
510 const auto* op = expr.as<OpNode>();
511 return op != nullptr && op->IsPrimitiveOp();
512}
