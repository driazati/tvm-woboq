46/*!
47 * \brief IRModule that holds functions and type definitions.
48 *
49 * IRModule is the basic unit for all IR transformations across the stack.
50 *
51 * Many operations require access to the global IRModule.
52 * We pass the IRModule by value in a functional style as an explicit argument,
53 * but we mutate the Module while optimizing programs.
54 * \sa IRModule
55 */
56class IRModuleNode : public Object {
57 public:
58 /*! \brief A map from ids to all global functions. */
59 Map<GlobalVar, BaseFunc> functions;
60 /*! \brief A map from global type vars to ADT type data. */
61 Map<GlobalTypeVar, TypeData> type_definitions;
62 /*! \brief The source map for the module. */
63 parser::SourceMap source_map;
64 /* \brief Additional attributes storing meta-data about the module. */
65 DictAttrs attrs;
66
67 /*!
68 * \brief Get a module attribute.
69 *
70 * \param attr_key The attribute key.
71 * \param default_value The default value if the key does not exist, defaults to nullptr.
72 *
73 * \return The result
74 *
75 * \tparam TOBjectRef the expected object type.
76 * \throw Error if the key exists but the value does not match TObjectRef
77 *
78 * \code
79 *
80 * void GetAttrExample(const IRModule& mod) {
81 * auto value = f->GetAttr<Integer>("AttrKey", 0);
82 * }
83 *
84 * \endcode
85 */
86 template <typename TObjectRef>
87 Optional<TObjectRef> GetAttr(
88 const std::string& attr_key,
89 Optional<TObjectRef> default_value = Optional<TObjectRef>(nullptr)) const {
90 return attrs.GetAttr(attr_key, default_value);
91 }
92 // variant that uses TObjectRef to enable implicit conversion to default value.
93 template <typename TObjectRef>
94 Optional<TObjectRef> GetAttr(const std::string& attr_key, TObjectRef default_value) const {
95 return GetAttr<TObjectRef>(attr_key, Optional<TObjectRef>(default_value));
96 }
97
98 /*!
99 * \brief Check whether the module has an non-zero integer attr.
100 *
101 * This function can be used to check whether an optional
102 * attribute mark(e.g. inline) exists.
103 *
104 * \param attr_key The key to the attribute.
105 * \return The check result.
106 *
107 * \code
108 *
109 * void HasNonzeroAttrExample(const IRModule& mod) {
110 * if (mod->HasNonzeroAttr(attr::kInline)) {
111 * // inline the function.
112 * }
113 * }
114 *
115 * \endcode
116 */
117 bool HasNonzeroAttr(const std::string& attr_key) const { return attrs.HasNonzeroAttr(attr_key); }
118
119 IRModuleNode() : source_map() {}
120
121 void VisitAttrs(AttrVisitor* v) {
122 v->Visit("functions", &functions);
123 v->Visit("type_definitions", &type_definitions);
124 v->Visit("global_var_map_", &global_var_map_);
125 v->Visit("global_type_var_map_", &global_type_var_map_);
126 v->Visit("source_map", &source_map);
127 v->Visit("attrs", &attrs);
128 }

348/*!
349 * \brief Managed reference class to IRModuleNode.
350 * \sa IRModuleNode
351 */
352class IRModule : public ObjectRef {
353 public:
354 /*!
355 * \brief constructor
356 * \param functions Functions in the module.
357 * \param type_definitions Type definitions in the module.
358 * \param import_set Set of imported files in the module.
359 * \param map The module source map.
360 * \param attrs The module attributes.
361 */
362 TVM_DLL explicit IRModule(Map<GlobalVar, BaseFunc> functions,
363 Map<GlobalTypeVar, TypeData> type_definitions = {},
364 std::unordered_set<String> import_set = {}, parser::SourceMap map = {},
365 DictAttrs attrs = {});
366
367 /*! \brief default constructor */
368 IRModule() : IRModule(Map<GlobalVar, BaseFunc>({})) {}
369 /*!
370 * \brief constructor
371 * \param n The object pointer.
372 */
373 explicit IRModule(ObjectPtr<Object> n) : ObjectRef(n) {}
374 /*! \return mutable pointers to the node. */
375 IRModuleNode* operator->() const {
376 auto* ptr = get_mutable();
377 ICHECK(ptr != nullptr);
378 return static_cast<IRModuleNode*>(ptr);
379 }
380
381 /*!
382 * \brief Constructs a module from a standalone expression \p expr.
383 *
384 * If \p expr is a function it will be bound directly. Otherwise a function over the free
385 * variables of \p expr (possibly none) with \p expr as body is created and bound.
386 *
387 * The function is bound to, in preference order:
388 * - The "global_symbol" attribute of \p expr, if it is a function with that attribute.
389 * - 'main'
390 * - A unique name derived from 'main' if 'main' is already bound in \p global_funcs.
391 *
392 * Additional global functions and type definitions may be included in the result module.
393 *
394 * See also \p FromExpr.
395 *
396 * \param expr The expression to set as the main function to the module.
397 * \param global_funcs The global function map. Default empty.
398 * \param type_definitions The global type definition map. Default empty.
399 * \param import_set Set of external modules already imported. Default empty.
400 *
401 * \returns A module with \p expr set as the main function, and the global var to which
402 * \p expr was bound (typcially 'main').
403 *
404 * TODO(mbs): Does import_set and the bound global var need to be exposed via ffi?
405 */
406 static std::pair<IRModule, GlobalVar> FromExprInContext(
407 const RelayExpr& expr, const Map<GlobalVar, BaseFunc>& global_funcs = {},
408 const Map<GlobalTypeVar, TypeData>& type_definitions = {},
409 std::unordered_set<String> import_set = {});
410
411 /*!
412 * \brief As for \p FromExprInContext, but assuming \p expr is bound to 'main' and no
413 * imports.
414 */
415 TVM_DLL static IRModule FromExpr(const RelayExpr& expr,
416 const Map<GlobalVar, BaseFunc>& global_funcs = {},
417 const Map<GlobalTypeVar, TypeData>& type_definitions = {});
418
419 /*!
420 * \brief Parse text format source file into an IRModule.
421 * \param text A string of Relay source code.
422 * \param source_path The path to the source file.
423 * \return A Relay module.
424 */
425 TVM_DLL static IRModule FromText(const String& text, const String& source_path);
426
427 /*!
428 * \brief Create a shallow copy of an IRModule.
429 * \param mod The module to copy.
430 * \return The copied module.
431 */
432 IRModule ShallowCopyIRModule(IRModule mod);
433
434 /*! \brief Declare the container type. */
435 using ContainerType = IRModuleNode;
436
437 /*! \brief Declare whether Ref is nullable. */
438 static constexpr bool _type_is_nullable = false;
439
440 // allow copy on write.
441 TVM_DEFINE_OBJECT_REF_COW_METHOD(IRModuleNode);
442};
