20/*!
21 * \file tvm/ir/transform.h
22 *
23 * This file implements a pass manager. The pass manager manages a sequence
24 * of IRModule -> IRModule transformation passes over a particlar unit of AST. The
25 * design is largely inspired from LLVM's pass manager and modern deep learning
26 * frameworks that perform tensor->tensor transformations.
27 *
28 * The responsibilities of a traditional compiler pass manager usually involves:
29 * - Organizing the execution order of optimization passes though not
30 * necessarily in the optimal sequence.
31 * - Collecting required analysis information and keep them up-to-date.
32 * - Reducing the effort required to implement new passes for compiler
33 * developers, etc.
34 *
35 * Similar to LLVM's pass manager, we designed the Relay pass manager to work
36 * different granularity, i.e. module level, function level, and even sequential
37 * passe that contains a host of passes.
38 *
39 * However, we also extend the functionality of the traditional pass manager
40 * with the consideration of requirements/convention from deep learning
41 * frameworks, such as Pytorch and Gluon, etc. Each pass in the Relay pass
42 * manager performs the IRModule -> IRModule transformation. All
43 * different types of passes, including the sequential-level pass object, are
44 * essentially pass objects. This design, therefore, effectively provides users
45 * a consistent and convenient interface, i.e. Pass, to play with. It offers a
46 * means to ease the development and testing of Relay passes. For example, with
47 * the pass manager, external users will be able to have custom passes correctly
48 * scheduled without having to modify a single handcrafted pass order.
49 *
50 * In the future we need to describe constraints between passes. For example,
51 * we may want to preserve dependencies between different passes and validate
52 * them on the completion of a certain pass.
53 *
54 * We also need to store side information and import the error reporting system.
55 */

73/*!
74 * \brief PassContextNode contains the information that a pass can rely on,
75 * such as analysis results.
76 * \sa PassContext
77 */
78class PassContextNode : public Object {
79 public:
80 /*! \brief The default optimization level. */
81 int opt_level{2};
82
83 /*! \brief The list of required passes. */
84 Array<String> required_pass;
85 /*! \brief The list of disabled passes. */
86 Array<String> disabled_pass;
87 /*! \brief The diagnostic context. */
88 mutable Optional<DiagnosticContext> diag_ctx;
89 /*! \brief Pass specific configurations. */
90 Map<String, ObjectRef> config;
91
92 /*! \brief A list of pass instrument implementations. */
93 Array<instrument::PassInstrument> instruments;
94
95 PassContextNode() = default;
96
97 /*!
98 * \brief Get a config value from the pass context.
99 *
100 * \param key The config key.
101 * \param default_value The default value if the key does not exist, defaults to nullptr.
102 *
103 * \return The result
104 *
105 * \tparam TOBjectRef the expected object type.
106 * \throw Error if the key exists but the value does not match TObjectRef.
107 */
108 template <typename TObjectRef>
109 Optional<TObjectRef> GetConfig(const std::string& key, Optional<TObjectRef> default_value =
110 Optional<TObjectRef>(nullptr)) const {
111 static_assert(std::is_base_of<ObjectRef, TObjectRef>::value,
112 "Can only call GetAttr with ObjectRef types.");
113 if (!config.defined()) return default_value;
114 auto it = config.find(key);
115 if (it != config.end()) {
116 return Downcast<Optional<TObjectRef>>((*it).second);
117 } else {
118 return default_value;
119 }
120 }
121 // variant that uses TObjectRef to enable implicit conversion to default value.
122 template <typename TObjectRef>
123 Optional<TObjectRef> GetConfig(const std::string& key, TObjectRef default_value) const {
124 return GetConfig<TObjectRef>(key, Optional<TObjectRef>(default_value));
125 }
126
127 void VisitAttrs(AttrVisitor* v) {
128 v->Visit("opt_level", &opt_level);
129 v->Visit("required_pass", &required_pass);
130 v->Visit("disabled_pass", &disabled_pass);
131 v->Visit("instruments", &instruments);
132 v->Visit("config", &config);
133 v->Visit("diag_ctx", &diag_ctx);
134 }
135
136 static constexpr const char* _type_key = "transform.PassContext";
137 static constexpr bool _type_has_method_sequal_reduce = false;
138 TVM_DECLARE_FINAL_OBJECT_INFO(PassContextNode, Object);
139};

141/*!
142 * \brief PassContext that is used to configure the pass behavior.
143 *
144 * \code
145 *
146 * auto new_ctx = PassContext::Create();
147 * ctx->opt_level = 2;
148 * With<PassContext> scope(ctx);
149 * // pass context in effect.
150 *
151 * \endcode
152 * \sa PassContextNode
153 */
154class PassContext : public ObjectRef {
155 public:
156 PassContext() {}
157 explicit PassContext(ObjectPtr<Object> n) : ObjectRef(n) {}
158 /*!
159 * \brief const accessor.
160 * \return const access pointer.
161 */
162 const PassContextNode* operator->() const {
163 ICHECK(get() != nullptr);
164 return static_cast<const PassContextNode*>(get());
165 }
166 /*!
167 * \brief mutable accessor.
168 * \return mutable access pointer.
169 */
170 PassContextNode* operator->() {
171 ICHECK(get() != nullptr);
172 return static_cast<PassContextNode*>(get_mutable());
173 }
174
175 /*!
176 * \brief Construct a PassContext containing the default configurations.
177 * \return The new PassContext.
178 */
179 TVM_DLL static PassContext Create();
180 /*!
181 * \brief Get the default pass context in the current scope.
182 * \return The pass context.
183 */
184 TVM_DLL static PassContext Current();
185
186 /*!
187 * \brief Get all supported configuration names and metadata, registered within the PassContext.
188 * \return Map indexed by the config name, pointing to the metadata map as key-value
189 */
190 TVM_DLL static Map<String, Map<String, String>> ListConfigs();
191
192 /*!
193 * \brief Call instrument implementations' callbacks when entering PassContext.
194 * The callbacks are called in order, and if one raises an exception, the rest will not be
195 * called.
196 */
197 TVM_DLL void InstrumentEnterPassContext();
198
199 /*!
200 * \brief Call instrument implementations' callbacks when exiting PassContext.
201 * The callbacks are called in order, and if one raises an exception, the rest will not be
202 * called.
203 */
204 TVM_DLL void InstrumentExitPassContext();
205
206 /*!
207 * \brief Call instrument implementations' callbacks before a pass run.
208 * The callbacks are called in order, and if one raises an exception, the rest will not be
209 * called.
210 *
211 * \param mod The module that an optimization pass runs on.
212 * \param info The pass information.
213 *
214 * \return false: the pass is skipped; true: the pass runs.
215 */
216 TVM_DLL bool InstrumentBeforePass(const IRModule& mod, const PassInfo& info) const;
217
218 /*!
219 * \brief Call instrument implementations callbacks after a pass run.
220 * The callbacks are called in order, and if one raises an exception, the rest will not be
221 * called.
222 *
223 * \param mod The module that an optimization pass runs on.
224 * \param info The pass information.
225 */
226 TVM_DLL void InstrumentAfterPass(const IRModule& mod, const PassInfo& info) const;
227
228 /*!
229 * \brief Check whether a pass is enabled.
230 * \param info The pass information.
231 * \return true if the pass is enabled. Otherwise, false.

233 TVM_DLL bool PassEnabled(const PassInfo& info) const;

235 /*!
236 * \brief Register a valid configuration option and its ValueType for validation.
237 *
238 * \param key The configuration key.
239 * \tparam ValueType The value type to be registered
240 */
241 template <typename ValueType>
242 static uint32_t RegisterConfigOption(const char* key) {
243 using ValueNodeType = typename ValueType::ContainerType;
244 // NOTE: we could further update the function later.
245 uint32_t tindex = ValueNodeType::_GetOrAllocRuntimeTypeIndex();
246 RegisterConfigOption(key, tindex);
247 return tindex;
248 }

250 // accessor.
251 using ContainerType = PassContextNode;
252 class Internal;
253
254 private:
255 // The entry of a pass context scope.
256 TVM_DLL void EnterWithScope();
257 // The exit of a pass context scope.
258 TVM_DLL void ExitWithScope();
259 // Register configuration key value type.
260 TVM_DLL static void RegisterConfigOption(const char* key, uint32_t value_type_index);
261
262 // Classes to get the Python `with` like syntax.
263 friend class Internal;
264 friend class With<PassContext>;
265};

324/*!
325 * \brief PassNode is the base type of differnt types of optimization passes.
326 * It is designed as a pure class and implemented by different pass subclasses
327 * at different granularity of Relay nodes.
328 */
329class PassNode : public Object {
330 public:
331 virtual ~PassNode() {}
332 /*!
333 * \brief Get the pass information/meta data. */
334 virtual PassInfo Info() const = 0;
335
336 /*!
337 * \brief Transform mod using the default PassContext in the current scope.
338 *
339 * \param mod The module that an optimization pass runs on.
340 *
341 * \return The transformed module.
342 */
343 IRModule operator()(IRModule mod) const {
344 return this->operator()(std::move(mod), PassContext::Current());
345 }
346
347 /*!
348 * \brief Transform mod using a functor under a given pass context.
349 *
350 * \param mod The module that an optimization pass runs on.
351 * \param pass_ctx The pass context that can provide information for the optimization.
352 *
353 * \return The transformed module.
354 */
355 virtual IRModule operator()(IRModule mod, const PassContext& pass_ctx) const = 0;
356
357 void VisitAttrs(AttrVisitor* v) {}
358
359 static constexpr const char* _type_key = "transform.Pass";
360 TVM_DECLARE_BASE_OBJECT_INFO(PassNode, Object);
361};

363class Pass : public ObjectRef {
364 public:
365 /*!
366 * \brief Transform mod using the default PassContext in the current scope.
367 *
368 * \code
369 *
370 * // If you do no longer need the input module
371 * // it is recommended to use std::move to move your input module.
372 * mod = pass(std::move(mod));
373 *
374 * \endcode
375 *
376 * \param mod The module that an optimization pass runs on.
377 *
378 * \return The transformed module.
379 */
380 IRModule operator()(IRModule mod) const;
381
382 /*!
383 * \brief Transform mod using a functor under a given pass context.
384 *
385 * \param mod The module that an optimization pass runs on.
386 * \param pass_ctx The pass context that can provide information for the optimization.
387 *
388 * \return The transformed module.
389 */
390 IRModule operator()(IRModule mod, const PassContext& pass_ctx) const;
391
392 TVM_DEFINE_OBJECT_REF_METHODS(Pass, ObjectRef, PassNode);
393
394 private:
395 IRModule static AssertImmutableModule(const IRModule& mod, const PassNode* node,
396 const PassContext& pass_ctx);
397};

399/*!
400 * \brief The SequentialNode contains a set of passes that transform Relay
401 * programs from one AST to another semantically equivalent one.
402 *
403 * One example of this level of pass is that the pass manager needs to correctly
404 * perform a host of optimizations with a given optimization level and disabled
405 * passes.
406 */
407class SequentialNode : public PassNode {
408 public:
409 /* \brief The pass meta data.*/
410 PassInfo pass_info;
411
412 /*! \brief A list of passes that used to compose a sequential pass. */
413 tvm::Array<Pass> passes;
414
415 void VisitAttrs(tvm::AttrVisitor* v) {
416 v->Visit("pass_info", &pass_info);
417 v->Visit("passes", &passes);
418 }
419
420 /*!
421 * \brief Get the pass information/meta data.
422 */
423 PassInfo Info() const override { return pass_info; }
424
425 /*!
426 * \brief Resolve the pass dependency. It globs all required passes by
427 * a given pass and executes them.
428 *
429 * \param mod The module that an optimization pass runs on.
430 *
431 * \return The updated module after resolving pass dependencies.
432 *
433 * TODO(zhiics) Build a dependency graph among the passes using provided
434 * metadata, i.e. required_passes. Likely, we can have a data structure, i.e.
435 * PassInfo, to store the relevant information including the parent passes.
436 */
437 void ResolveDependency(const IRModule& mod);
438
439 /*!
440 * \brief Perform optimizations on a series of passes. The aforementioned
441 * typical pass manager jobs could be done by it. This function could
442 * be overloaded to focus on different metrics, i.e. performance,
443 * memory footprint, etc.
444 *
445 * \param mod The module that these passes are applied on.
446 * \param pass_ctx The context that these passes execute on.
447 *
448 * \return Return the updated module.
449 */
450 IRModule operator()(IRModule mod, const PassContext& pass_ctx) const final;
451
452 static constexpr const char* _type_key = "transform.Sequential";
453 TVM_DECLARE_FINAL_OBJECT_INFO(SequentialNode, PassNode);
454};
