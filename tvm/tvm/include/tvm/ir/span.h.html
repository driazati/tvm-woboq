20/*!
21 * \file tvm/ir/span.h
22 * \brief Span information for debugging purposes.
23 */
24#ifndef TVM_IR_SPAN_H_
25#define TVM_IR_SPAN_H_
26
27#include <tvm/node/node.h>
28#include <tvm/runtime/object.h>
29
30#include <string>
31
32namespace tvm {
33/*!
34 * \brief The source name in the Span
35 * \sa SourceNameNode, Span
36 */
37class SourceName;
38/*!
39 * \brief The name of a source fragment.
40 */
41class SourceNameNode : public Object {
42 public:
43 /*! \brief The source name. */
44 String name;
45 // override attr visitor
46 void VisitAttrs(AttrVisitor* v) { v->Visit("name", &name); }
47
48 static constexpr bool _type_has_method_sequal_reduce = true;
49
50 bool SEqualReduce(const SourceNameNode* other, SEqualReducer equal) const {
51 return equal(name, other->name);
52 }
53
54 static constexpr const char* _type_key = "SourceName";
55 TVM_DECLARE_FINAL_OBJECT_INFO(SourceNameNode, Object);
56};
57
58/*!
59 * \brief The source name of a file span.
60 * \sa SourceNameNode, Span
61 */
62class SourceName : public ObjectRef {
63 public:
64 /*!
65 * \brief Get an SourceName for a given operator name.
66 * Will raise an error if the source name has not been registered.
67 * \param name Name of the operator.
68 * \return SourceName valid throughout program lifetime.
69 */
70 TVM_DLL static SourceName Get(const String& name);
71
72 TVM_DEFINE_OBJECT_REF_METHODS(SourceName, ObjectRef, SourceNameNode);
73};
74
75/*!
76 * \brief Span information for debugging purposes
77 */
78class Span;
79/*!
80 * \brief Stores locations in frontend source that generated a node.
81 */
82class SpanNode : public Object {
83 public:
84 /*! \brief The source name. */
85 SourceName source_name;
86 /*! \brief The line number. */
87 int line;
88 /*! \brief The column offset. */
89 int column;
90 /*! \brief The end line number. */
91 int end_line;
92 /*! \brief The end column number. */
93 int end_column;
94
95 // override attr visitor
96 void VisitAttrs(AttrVisitor* v) {
97 v->Visit("source_name", &source_name);
98 v->Visit("line", &line);
99 v->Visit("column", &column);
100 v->Visit("end_line", &end_line);
101 v->Visit("end_column", &end_column);
102 }
103 static constexpr bool _type_has_method_sequal_reduce = true;
104
105 bool SEqualReduce(const SpanNode* other, SEqualReducer equal) const {
106 return equal(source_name, other->source_name) && equal(line, other->line) &&
107 equal(column, other->column) && equal(end_line, other->end_line) &&
108 equal(end_column, other->end_column);
109 }
110
111 static constexpr const char* _type_key = "Span";
112 TVM_DECLARE_FINAL_OBJECT_INFO(SpanNode, Object);
113};
114
115class Span : public ObjectRef {
116 public:
117 TVM_DLL Span(SourceName source_name, int line, int end_line, int column, int end_column);
118
119 /*! \brief Merge two spans into one which captures the combined regions. */
120 TVM_DLL Span Merge(const Span& other) const;
121
122 TVM_DEFINE_OBJECT_REF_METHODS(Span, ObjectRef, SpanNode);
123};
124
125} // namespace tvm
126#endif // TVM_IR_SPAN_H_
