40/*! \brief The diagnostic level, controls the printing of the message. */
41enum class DiagnosticLevel : int {
42 kBug = 10,
43 kError = 20,
44 kWarning = 30,
45 kNote = 40,
46 kHelp = 50,
47};
48
49class DiagnosticBuilder;
50
51/*! \brief A compiler diagnostic. */
52class Diagnostic;
53
54/*! \brief A compiler diagnostic message. */
55class DiagnosticNode : public Object {
56 public:
57 /*! \brief The level. */
58 DiagnosticLevel level;
59 /*! \brief The span at which to report an error. */
60 Span span;
61 /*! \brief The diagnostic message. */
62 String message;
63
64 // override attr visitor
65 void VisitAttrs(AttrVisitor* v) {
66 v->Visit("level", &level);
67 v->Visit("span", &span);
68 v->Visit("message", &message);
69 }
70
71 bool SEqualReduce(const DiagnosticNode* other, SEqualReducer equal) const {
72 return equal(this->level, other->level) && equal(this->span, other->span) &&
73 equal(this->message, other->message);
74 }
75
76 static constexpr const char* _type_key = "Diagnostic";
77 TVM_DECLARE_FINAL_OBJECT_INFO(DiagnosticNode, Object);
78};
79
80class Diagnostic : public ObjectRef {
81 public:
82 TVM_DLL Diagnostic(DiagnosticLevel level, Span span, const std::string& message);
83
84 static DiagnosticBuilder Bug(Span span);
85 static DiagnosticBuilder Error(Span span);
86 static DiagnosticBuilder Warning(Span span);
87 static DiagnosticBuilder Note(Span span);
88 static DiagnosticBuilder Help(Span span);
89
90 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(Diagnostic, ObjectRef, DiagnosticNode);
91};
92
93/*!
94 * \brief A wrapper around std::stringstream to build a diagnostic.
95 */
96class DiagnosticBuilder {
97 public:
98 /*! \brief The level. */
99 DiagnosticLevel level;
100
101 /*! \brief The source name. */
102 SourceName source_name;
103
104 /*! \brief The span of the diagnostic. */
105 Span span;
106
107 template <typename T>
108 DiagnosticBuilder& operator<<(const T& val) { // NOLINT(*)
109 stream_ << val;
110 return *this;
111 }
112
113 DiagnosticBuilder() : level(DiagnosticLevel::kError), source_name(), span(Span()) {}
114
115 DiagnosticBuilder(const DiagnosticBuilder& builder)
116 : level(builder.level), source_name(builder.source_name), span(builder.span) {}
117
118 DiagnosticBuilder(DiagnosticLevel level, Span span) : level(level), span(span) {}
119
120 operator Diagnostic() { return Diagnostic(this->level, this->span, this->stream_.str()); }
121
122 private:
123 std::stringstream stream_;
124 friend class Diagnostic;
125};
126
127/*!
128 * \brief A diagnostic context for recording errors against a source file.
129 */
130class DiagnosticContext;
131
132/*! \brief Display diagnostics in a given display format.
133 *
134 * A diagnostic renderer is responsible for converting the
135 * raw diagnostics into consumable output.
136 *
137 * For example the terminal renderer will render a sequence
138 * of compiler diagnostics to std::out and std::err in
139 * a human readable form.
140 */
141class DiagnosticRendererNode : public Object {
142 public:
143 TypedPackedFunc<void(DiagnosticContext ctx)> renderer;
144
145 // override attr visitor
146 void VisitAttrs(AttrVisitor* v) {}
147
148 static constexpr const char* _type_key = "DiagnosticRenderer";
149 TVM_DECLARE_FINAL_OBJECT_INFO(DiagnosticRendererNode, Object);
150};
151
152class DiagnosticRenderer : public ObjectRef {
153 public:
154 TVM_DLL DiagnosticRenderer(TypedPackedFunc<void(DiagnosticContext ctx)> render);
155 TVM_DLL DiagnosticRenderer()
156 : DiagnosticRenderer(TypedPackedFunc<void(DiagnosticContext ctx)>()) {}
157
158 void Render(const DiagnosticContext& ctx);
159
160 DiagnosticRendererNode* operator->() {
161 ICHECK(get() != nullptr);
162 return static_cast<DiagnosticRendererNode*>(get_mutable());
163 }
164
165 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(DiagnosticRenderer, ObjectRef, DiagnosticRendererNode);
166};
167
168class DiagnosticContextNode : public Object {
169 public:
170 /*! \brief The Module to report against. */
171 IRModule module;
172
173 /*! \brief The set of diagnostics to report. */
174 Array<Diagnostic> diagnostics;
175
176 /*! \brief The renderer set for the context. */
177 DiagnosticRenderer renderer;
178
179 void VisitAttrs(AttrVisitor* v) {
180 v->Visit("module", &module);
181 v->Visit("diagnostics", &diagnostics);
182 }
183
184 bool SEqualReduce(const DiagnosticContextNode* other, SEqualReducer equal) const {
185 return equal(module, other->module) && equal(diagnostics, other->diagnostics);
186 }
187
188 static constexpr const char* _type_key = "DiagnosticContext";
189 TVM_DECLARE_FINAL_OBJECT_INFO(DiagnosticContextNode, Object);
190};
191
192class DiagnosticContext : public ObjectRef {
193 public:
194 TVM_DLL DiagnosticContext(const IRModule& module, const DiagnosticRenderer& renderer);
195 TVM_DLL static DiagnosticContext Default(const IRModule& source_map);
196
197 /*! \brief Emit a diagnostic.
198 * \param diagnostic The diagnostic to emit.
199 */
200 void Emit(const Diagnostic& diagnostic);
201
202 /*! \brief Emit a diagnostic and then immediately attempt to render all errors.
203 *
204 * \param diagnostic The diagnostic to emit.
205 *
206 * Note: this will raise an exception if you would like to instead continue execution
207 * use the Emit method instead.
208 */
209 void EmitFatal(const Diagnostic& diagnostic);
210
211 /*! \brief Render the errors and raise a DiagnosticError exception. */
212 void Render();
213
214 DiagnosticContextNode* operator->() {
215 ICHECK(get() != nullptr);
216 return static_cast<DiagnosticContextNode*>(get_mutable());
217 }
218
219 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(DiagnosticContext, ObjectRef, DiagnosticContextNode);
220};
