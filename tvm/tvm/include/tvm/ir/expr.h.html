73/*!
74 * \brief Base node of all primitive expressions.
75 *
76 * A primitive expression deals with low-level
77 * POD data types and handles without
78 * doing life-cycle management for objects.
79 *
80 * PrimExpr is used in the low-level code
81 * optimizations and integer analysis.
82 *
83 * \sa PrimExpr
84 */
85class PrimExprNode : public BaseExprNode {
86 public:
87 /*!
88 * \brief The runtime data type of the primitive expression.
89 *
90 * runtime::DataType(dtype) provides coarse grained type information
91 * during compile time and runtime. It is eagerly built in
92 * PrimExpr expression construction and can be used for
93 * quick type checking.
94 *
95 * dtype is sufficient to decide the Type of the PrimExpr
96 * when it corresponds to POD value types such as i32.
97 *
98 * When dtype is DataType::Handle(), the expression could corresponds to
99 * a more fine-grained Type, and we can get the type by running lazy type inference.
100 */
101 DataType dtype;
102
103 static constexpr const char* _type_key = "PrimExpr";
104 static constexpr const uint32_t _type_child_slots = 38;
105 TVM_DECLARE_BASE_OBJECT_INFO(PrimExprNode, BaseExprNode);
106};

350/*!
351 * \brief Base node of all non-primitive expressions.
352 *
353 * RelayExpr supports tensor types, functions and ADT as
354 * first class citizens. The life-cycle of the corresponding
355 * objects are implicitly managed by the language.
356 *
357 * \sa RelayExpr
358 */
359class RelayExprNode : public BaseExprNode {
360 public:
361 /*!
362 * \brief Stores the result of type inference(type checking).
363 *
364 * \note This can be undefined before type inference.
365 * This value is discarded during serialization.
366 */
367 mutable Type checked_type_ = Type(nullptr);
368 /*!
369 * \return The checked_type
370 */
371 inline const Type& checked_type() const;
372 /*!
373 * \brief Check if the inferred(checked) type of the Expr
374 * is backed by a TTypeNode and return it.
375 *
376 * \note This function will thrown an error if the node type
377 * of this Expr is not TTypeNode.
378 *
379 * \return The corresponding TTypeNode pointer.
380 * \tparam The specific TypeNode we look for.
381 */
382 template <typename TTypeNode>
383 inline const TTypeNode* type_as() const;
384
385 /*!
386 * \brief The virtual device (VirtualDevice) for this node (the result of device planning).
387 * For first-order expressions (non functions), this describes where the result of evaluating the
388 * expression should be stored. Note that currently, all composite first-order values (tuples,
389 * references, ADTs) must be stored on the same virtual device. This means that it is not possible
390 * to store two tuple fields on different devices, so we only need one virtual device for these
391 * types.
392 *
393 * For expressions that have the function type, the virtual device describes where the result of
394 * the call to the function or closure is stored (instead of where the function itself is stored).
395 * For example, the virtual device of f = fn(x) { body } is the virtual device of f(y), not where
396 * the function itself is stored. Note that f(y)'s virtual device will be the same as the virtual
397 * device of body. For more details, see the documentation in
398 * src/relay/transforms/device_planner.cc.
399 *
400 * The VirtualDevice's Target field describes how the body of the function should be compiled.
401 *
402 * Set to VirtualDevice::FullyUnconstrained by default.
403 *
404 * \note Unfortunately, the type of virtual_device_ needs to be ObjectRef to avoid a circular
405 * import.
406 */
407 mutable ObjectRef virtual_device_;
408
409 /*!
410 * \return The virtual device (VirtualDevice).
411 * If the virtual device is not defined, returns VirtualDevice::FullyUnconstrained().
412 * Note that for function types, the virtual device is the device where the result of a
413 * call to the function is stored, not where the function itself lives.
414 * For example, the virtual device of f = fn(x) { body } is the virtual device of f(y), not where
415 * the function itself is stored. Note that f(y)'s virtual device will be the same as the virtual
416 * device of body.
417 *
418 * See the documentation of the virtual_device_ field (above) for more details.
419 */
420 VirtualDevice virtual_device() const;
421
422 static constexpr const char* _type_key = "RelayExpr";
423 static constexpr const uint32_t _type_child_slots = 22;
424 TVM_DECLARE_BASE_OBJECT_INFO(RelayExprNode, BaseExprNode);
425};

436class GlobalVar;
437/*!
438 * \brief Global variable that lives in the top-level module.
439 *
440 * A GlobalVar only refers to function definitions.
441 * This is used to enable recursive calls between function.
442 *
443 * \sa GlobalVarNode
444 */
445class GlobalVarNode : public RelayExprNode {
446 public:
447 /*! \brief The name of the variable, this only acts as a hint. */
448 String name_hint;
449
450 void VisitAttrs(AttrVisitor* v) {
451 v->Visit("name_hint", &name_hint);
452 v->Visit("virtual_device_", &virtual_device_);
453 v->Visit("span", &span);
454 v->Visit("_checked_type_", &checked_type_);
455 }
456
457 bool SEqualReduce(const GlobalVarNode* other, SEqualReducer equal) const {
458 // name matters for global var.
459 return equal(name_hint, other->name_hint) && equal.FreeVarEqualImpl(this, other);
460 }
461
462 void SHashReduce(SHashReducer hash_reduce) const {
463 hash_reduce(name_hint);
464 hash_reduce.FreeVarHashImpl(this);
465 }
466
467 static constexpr const char* _type_key = "GlobalVar";
468 TVM_DECLARE_FINAL_OBJECT_INFO(GlobalVarNode, RelayExprNode);
469};

577/*!
578 * \brief Boolean constant.
579 *
580 * This reference type is useful to add additional compile-time

583class Bool : public IntImm {
584 public:
585 explicit Bool(bool value, Span span = Span()) : IntImm(DataType::Bool(), value, span) {}

592// Overload operators to make sure we have the most fine grained types.
593inline Bool operator||(const Bool& a, bool b) { return Bool(a.operator bool() || b); }
594inline Bool operator||(bool a, const Bool& b) { return Bool(a || b.operator bool()); }
595inline Bool operator||(const Bool& a, const Bool& b) {
596 return Bool(a.operator bool() || b.operator bool());
597}
598inline Bool operator&&(const Bool& a, bool b) { return Bool(a.operator bool() && b); }
599inline Bool operator&&(bool a, const Bool& b) { return Bool(a && b.operator bool()); }
600inline Bool operator&&(const Bool& a, const Bool& b) {
601 return Bool(a.operator bool() && b.operator bool());
602}
603
604inline bool operator==(const Bool& a, bool b) { return a.operator bool() == b; }
605inline bool operator==(bool a, const Bool& b) { return a == b.operator bool(); }
606inline bool operator==(const Bool& a, const Bool& b) {
607 return a.operator bool() == b.operator bool();
608}

610/*!
611 * \brief Container of constant int that adds more constructors.
612 *
613 * This is used to store and automate type check
614 * attributes that must be constant integer.
615 *
616 * \sa IntImm
617 */
618class Integer : public IntImm {
619 public:
620 Integer() {}
621 /*!
622 * \brief constructor from node.
623 */
624 explicit Integer(ObjectPtr<Object> node) : IntImm(node) {}
625 /*!
626 * \brief Construct integer from int value.
627 */
628 Integer(int value, Span span = Span()) : IntImm(DataType::Int(32), value, span) {} // NOLINT(*)
629 /*!
630 * \brief Construct integer from int imm.
631 * \param other The other value.
632 */
633 Integer(IntImm other) : IntImm(std::move(other)) {} // NOLINT(*)
634 /*!
635 * \brief Constructor from enum
636 * \tparam Enum The enum type.
637 * \param value The enum value.
638 */
639 template <typename Enum, typename = typename std::enable_if<std::is_enum<Enum>::value>::type>
640 explicit Integer(Enum value) : Integer(static_cast<int>(value)) {
641 static_assert(std::is_same<int, typename std::underlying_type<Enum>::type>::value,
642 "declare enum to be enum int to use visitor");
643 }
644 /*!
645 * \brief Assign an expression to integer.
646 * \param other another expression.
647 */
648 Integer& operator=(const IntImm& other) {
649 data_ = ObjectRef::GetDataPtr<Object>(other);
650 return *this;
651 }
652 /*!
653 * \brief convert to int64_t
654 */
655 int64_t IntValue() const {
656 ICHECK(data_ != nullptr) << " Trying to reference a null Integer";
657 return (*this)->value;
658 }
659 // comparators
660 Bool operator==(int other) const {
661 if (data_ == nullptr) return Bool(false);
662 return Bool((*this)->value == other);
663 }
664 Bool operator!=(int other) const { return !(*this == other); }
665 template <typename Enum, typename = typename std::enable_if<std::is_enum<Enum>::value>::type>
666 Bool operator==(Enum other) const {
667 return *this == static_cast<int>(other);
668 }
669 template <typename Enum, typename = typename std::enable_if<std::is_enum<Enum>::value>::type>
670 Bool operator!=(Enum other) const {
671 return *this != static_cast<int>(other);
672 }
673};

675/*! \brief range over one dimension */
676class RangeNode : public Object {
677 public:
678 /*! \brief beginning of the node */
679 PrimExpr min;
680 /*! \brief the extend of range */
681 PrimExpr extent;
682 /*! \brief the location of this range in the source */
683 mutable Span span;
684 /*! \brief constructor */
685 RangeNode() {}
686 RangeNode(PrimExpr min, PrimExpr extent, Span span = Span())
687 : min(min), extent(extent), span(span) {}
688
689 void VisitAttrs(AttrVisitor* v) {
690 v->Visit("min", &min);
691 v->Visit("extent", &extent);
692 v->Visit("span", &span);
693 }
694
695 bool SEqualReduce(const RangeNode* other, SEqualReducer equal) const {
696 return equal(min, other->min) && equal(extent, other->extent);
697 }
698
699 void SHashReduce(SHashReducer hash_reduce) const {
700 hash_reduce(min);
701 hash_reduce(extent);
702 }
703
704 static constexpr const char* _type_key = "Range";
705 static constexpr const bool _type_has_method_sequal_reduce = true;
706 static constexpr const bool _type_has_method_shash_reduce = true;
707 TVM_DECLARE_FINAL_OBJECT_INFO(RangeNode, Object);
708};

736inline const Type& RelayExprNode::checked_type() const {
737 ICHECK(checked_type_.defined()) << "internal error: the type checker has "
738 << "not populated the checked_type "
739 << "field for " << GetRef<RelayExpr>(this);
740 return this->checked_type_;

743template <typename TTypeNode>
744inline const TTypeNode* RelayExprNode::type_as() const {
745 static_assert(std::is_base_of<TypeNode, TTypeNode>::value,
746 "TType must be a special case of type");
747 ICHECK(checked_type_.defined())
748 << "Type inference for this Expr has not completed. Try to call infer_type pass.";
749 const TTypeNode* node = checked_type_.as<TTypeNode>();
750 ICHECK(node != nullptr) << "Expected type to be " << TTypeNode::_type_key << ", but get "
751 << checked_type_->GetTypeKey();
752 return node;
753}

759// common rule for RetValue and ArgValue
760template <>
761struct PackedFuncValueConverter<PrimExpr> {
762 static PrimExpr From(const TVMPODValue_& val) {
763 if (val.type_code() == kTVMNullptr) {
764 return PrimExpr(ObjectPtr<Object>(nullptr));
765 }
766 if (val.type_code() == kDLInt) {
767 return IntImm(runtime::DataType::Int(32), val.operator int());
768 }
769 if (val.type_code() == kDLFloat) {
770 return FloatImm(runtime::DataType::Float(32), val.operator double());
771 }
772
773 return PrimExpr::FromObject_(val.AsObjectRef<ObjectRef>());
774 }

778struct PackedFuncValueConverter<tvm::Integer> {
779 static tvm::Integer From(const TVMPODValue_& val) {
780 if (val.type_code() == kTVMNullptr) {
781 return Integer(ObjectPtr<Object>(nullptr));
782 }
783 if (val.type_code() == kTVMArgInt) {
784 return Integer(val.operator int());
785 }
786 return val.AsObjectRef<tvm::Integer>();
787 }
788};

790template <>
791struct PackedFuncValueConverter<tvm::Bool> {
792 static tvm::Bool From(const TVMPODValue_& val) {
793 if (val.type_code() == kTVMNullptr) {
794 return Bool(ObjectPtr<Object>(nullptr));
795 }
796 if (val.type_code() == kTVMArgInt) {
797 int v = val.operator int();
798 ICHECK(v == 0 || v == 1) << "ValueError: boolean value can only be 0 or 1, but get " << v;
799 return Bool(static_cast<bool>(v));
800 }
801 return val.AsObjectRef<tvm::Bool>();
802 }
