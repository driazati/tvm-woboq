19/*!
20 * \file tvm/node/repr_printer.h
21 * \brief Printer class to print repr string of each AST/IR nodes.
22 */
23#ifndef TVM_NODE_REPR_PRINTER_H_
24#define TVM_NODE_REPR_PRINTER_H_
25
26#include <tvm/node/functor.h>
27
28#include <iostream>
29
30namespace tvm {
31/*! \brief A printer class to print the AST/IR nodes. */
32class ReprPrinter {
33 public:
34 /*! \brief The output stream */
35 std::ostream& stream;
36 /*! \brief The indentation level. */
37 int indent{0};
38
39 explicit ReprPrinter(std::ostream& stream) // NOLINT(*)
40 : stream(stream) {}
41
42 /*! \brief The node to be printed. */
43 TVM_DLL void Print(const ObjectRef& node);
44 /*! \brief Print indent to the stream */
45 TVM_DLL void PrintIndent();
46 // Allow registration to be printer.
47 using FType = NodeFunctor<void(const ObjectRef&, ReprPrinter*)>;
48 TVM_DLL static FType& vtable();
49};
50
51/*!
52 * \brief Dump the node to stderr, used for debug purposes.
53 * \param node The input node
54 */
55TVM_DLL void Dump(const runtime::ObjectRef& node);
56
57/*!
58 * \brief Dump the node to stderr, used for debug purposes.
59 * \param node The input node
60 */
61TVM_DLL void Dump(const runtime::Object* node);
62
63} // namespace tvm
64
65namespace tvm {
66namespace runtime {
67// default print function for all objects
68// provide in the runtime namespace as this is where objectref originally comes from.
69inline std::ostream& operator<<(std::ostream& os, const ObjectRef& n) { // NOLINT(*)
70 ReprPrinter(os).Print(n);
71 return os;
72}
73} // namespace runtime
74} // namespace tvm
75#endif // TVM_NODE_REPR_PRINTER_H_
