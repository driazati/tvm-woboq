20/*!
21 * Utility functions for serialization.
22 * \file tvm/node/serialization.h
23 */
24#ifndef TVM_NODE_SERIALIZATION_H_
25#define TVM_NODE_SERIALIZATION_H_
26
27#include <tvm/runtime/c_runtime_api.h>
28#include <tvm/runtime/object.h>
29
30#include <string>
31
32namespace tvm {
33/*!
34 * \brief save the node as well as all the node it depends on as json.
35 * This can be used to serialize any TVM object
36 *
37 * \return the string representation of the node.
38 */
39TVM_DLL std::string SaveJSON(const runtime::ObjectRef& node);
40
41/*!
42 * \brief Internal implementation of LoadJSON
43 * Load tvm Node object from json and return a shared_ptr of Node.
44 * \param json_str The json string to load from.
45 *
46 * \return The shared_ptr of the Node.
47 */
48TVM_DLL runtime::ObjectRef LoadJSON(std::string json_str);
49
50} // namespace tvm
51#endif // TVM_NODE_SERIALIZATION_H_
