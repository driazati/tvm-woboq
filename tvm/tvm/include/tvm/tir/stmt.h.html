37/*! \brief Base node of all statements. */

41 * \brief Span that points to the original source code.

48
49 static constexpr const char* _type_key = "tir.Stmt";
50 static constexpr const bool _type_has_method_sequal_reduce = true;
51 static constexpr const bool _type_has_method_shash_reduce = true;
52 static constexpr const uint32_t _type_child_slots = 15;
53 TVM_DECLARE_BASE_OBJECT_INFO(StmtNode, Object);
54};

62/*!
63 * \brief Let binding, bind var to value, then run body.
64 */
65class LetStmtNode : public StmtNode {
66 public:
67 /*! \brief The variable. */
68 Var var;
69 /*! \brief The value to be binded. */
70 PrimExpr value;
71 /*! \brief The body block. */
72 Stmt body;
73
74 void VisitAttrs(AttrVisitor* v) {
75 v->Visit("var", &var);
76 v->Visit("value", &value);
77 v->Visit("body", &body);
78 v->Visit("span", &span);
79 }
80
81 bool SEqualReduce(const LetStmtNode* other, SEqualReducer equal) const {
82 return equal.DefEqual(var, other->var) && equal(value, other->value) &&
83 equal(body, other->body);
84 }
85
86 void SHashReduce(SHashReducer hash_reduce) const {
87 hash_reduce.DefHash(var);
88 hash_reduce(value);
89 hash_reduce(body);
90 }
91
92 static constexpr const char* _type_key = "tir.LetStmt";
93 TVM_DECLARE_FINAL_OBJECT_INFO(LetStmtNode, StmtNode);
94};

107/*!
108 * \brief Define certain auxiliary attribute for the body to be a symbolic value.
109 * This provide auxiliary information for IR passes that transforms body.
110 *
111 * In terms of effect, this is equivalent to Block(Evaluate(value), body).
112 *
113 * Examples of possible usage:
114 * - Bound of function, variables.
115 * - Hint which block corresponds to a parallel region.
116 */
117class AttrStmtNode : public StmtNode {
118 public:
119 /*! \brief this is attribute about certain node */
120 ObjectRef node;
121 /*! \brief the type key of the attribute */
122 String attr_key;
123 /*! \brief The attribute value, value is well defined at current scope. */
124 PrimExpr value;
125 /*! \brief The body statement to be executed */
126 Stmt body;
127
128 void VisitAttrs(AttrVisitor* v) {
129 v->Visit("node", &node);
130 v->Visit("attr_key", &attr_key);
131 v->Visit("value", &value);
132 v->Visit("body", &body);
133 v->Visit("span", &span);
134 }
135
136 bool SEqualReduce(const AttrStmtNode* other, SEqualReducer equal) const {
137 return equal(node, other->node) && equal(attr_key, other->attr_key) &&
138 equal(value, other->value) && equal(body, other->body);
139 }
140
141 void SHashReduce(SHashReducer hash_reduce) const {
142 hash_reduce(node);
143 hash_reduce(attr_key);
144 hash_reduce(value);
145 hash_reduce(body);
146 }
147
148 static constexpr const char* _type_key = "tir.AttrStmt";
149 TVM_DECLARE_FINAL_OBJECT_INFO(AttrStmtNode, StmtNode);
150};

211/*!
212 * \brief Store value to the buffer.
213 *
214 * Equivalent to ((DType*)buffer_var)[index] = value.
215 * where DType is the type specified by type().element_of().
216 *
217 * For example, if type = float32x3, then the store will corresponds to
218 *
219 * \code
220 *
221 * auto buffer = static_cast<float*>(buffer_var);
222 * buffer[index.v0] = value.v0;
223 * buffer[index.v1] = value.v1;
224 * buffer[index.v2] = value.v2;
225 *
226 * \endcode
227 * \sa LoadNode
228 */
229class StoreNode : public StmtNode {
230 public:
231 /*! \brief The buffer variable. */
232 Var buffer_var;
233 /*! \brief The value to be stored. */
234 PrimExpr value;
235 /*! \brief The index locations to be stored. */
236 PrimExpr index;
237 /*! \brief The predicate to mask which lanes would be stored. */
238 PrimExpr predicate;
239
240 void VisitAttrs(AttrVisitor* v) {
241 v->Visit("buffer_var", &buffer_var);
242 v->Visit("value", &value);
243 v->Visit("index", &index);
244 v->Visit("predicate", &predicate);
245 v->Visit("span", &span);
246 }
247
248 bool SEqualReduce(const StoreNode* other, SEqualReducer equal) const {
249 return equal(buffer_var, other->buffer_var) && equal(value, other->value) &&
250 equal(index, other->index) && equal(predicate, other->predicate);
251 }
252
253 void SHashReduce(SHashReducer hash_reduce) const {
254 hash_reduce(buffer_var);
255 hash_reduce(value);
256 hash_reduce(index);
257 hash_reduce(predicate);
258 }
259
260 static constexpr const char* _type_key = "tir.Store";
261 TVM_DECLARE_FINAL_OBJECT_INFO(StoreNode, StmtNode);
262};

330/*!
331 * \brief Annotate the region where the buffer need to
332 * be read and write in the body.
333 * We only need to allocate the space for the corresponding region.
334 *
335 * \note There should be at most one BufferRealize for each buffer.
336 * BufferRealize is not necessary for external buffers,
337 * since they are assumed to be fully allocated.
338 *
339 * \sa BufferLoad, BufferStore
340 */
341class BufferRealizeNode : public StmtNode {
342 public:
343 /*! \brief The buffer variable. */
344 Buffer buffer;
345 /*! \brief Bounds to be realized */
346 Array<Range> bounds;
347 /*! \brief Only realize if condition holds. */
348 PrimExpr condition;
349 /*! \brief The body of realization. */
350 Stmt body;
351
352 void VisitAttrs(AttrVisitor* v) {
353 v->Visit("buffer", &buffer);
354 v->Visit("bounds", &bounds);
355 v->Visit("condition", &condition);
356 v->Visit("body", &body);
357 v->Visit("span", &span);
358 }
359
360 bool SEqualReduce(const BufferRealizeNode* other, SEqualReducer equal) const {
361 return equal(buffer, other->buffer) && equal(bounds, other->bounds) &&
362 equal(condition, other->condition) && equal(body, other->body);
363 }
364
365 void SHashReduce(SHashReducer hash_reduce) const {
366 hash_reduce(buffer);
367 hash_reduce(bounds);
368 hash_reduce(condition);
369 hash_reduce(body);
370 }
371
372 BufferRealizeNode() = default;
373 BufferRealizeNode(Buffer buffer, Array<Range> bounds, PrimExpr condition, Stmt body,
374 Span span = Span())
375 : StmtNode(span), buffer(buffer), bounds(bounds), condition(condition), body(body) {}
376
377 static constexpr const char* _type_key = "tir.BufferRealize";
378 TVM_DECLARE_FINAL_OBJECT_INFO(BufferRealizeNode, StmtNode);
379};

751 public:
752 /*!
753 * \brief Construct SeqStmt.
754 * \param seq The sequence.
755 * \param span The location of this object in the source code.
756 */
757 TVM_DLL explicit SeqStmt(Array<Stmt> seq, Span span = Span());
758
759 /*! \return get the size of the sequence */
760 size_t size() const { return operator->()->size(); }
761 /*!
762 * \brief Get the index-th element in the sequence.
763 */
764 Stmt operator[](size_t index) const { return (*(operator->()))[index]; }
765 /*!
766 * \brief Construct a sequence statement by flattening
767 * all the arrays and sequences in the arguments
768 * recursively.
769 *
770 * - When an argument is nullptr, it will be ignored.
771 * - When an argument is an array or a SeqStmt, it will be flattened recursively.
772 * - A normal Stmt will be appended to the end of the sequence.
773 *
774 * \note This function can directly return an element
775 * if it is the only element in the sequence.
776 *
777 * \param seq_args The list of arguments to be flattened.
778 * \tparam Args arguments
779 * \return The constructed statement
780 */
781 template <typename... Args>
782 static Stmt Flatten(Args&&... seq_args) {
783 Array<Stmt> seq;
784 runtime::detail::for_each(Flattener(&seq), std::forward<Args>(seq_args)...);
785 if (seq.size() == 1) return seq[0];
786 return SeqStmt(seq);
787 }
788 /*! \brief Helper class to flatten sequence of arguments into Array. */
789 class Flattener {
790 public:
791 explicit Flattener(Array<Stmt>* seq) : seq_(seq) {}
792
793 void operator()(size_t i, const Stmt& stmt) const {
794 if (!stmt.defined()) return;
795 if (auto* op = stmt.as<SeqStmtNode>()) {
796 operator()(0, op->seq);
797 } else {
798 seq_->push_back(stmt);
799 }
800 }
801
802 template <typename T>
803 void operator()(size_t i, const T& seq) const {
804 for (auto v : seq) {
805 this->operator()(0, v);
806 }
807 }
808
809 private:
810 Array<Stmt>* seq_;
811 };
812
813 TVM_DEFINE_OBJECT_REF_METHODS(SeqStmt, Stmt, SeqStmtNode);

1206/*!
1207 * \brief A block is a basic schedule unit in TIR.
1208 * \note Block's body is parameterized by iter vars.
1209 * \code
1210 *
1211 * with T.block(name):
1212 * v0 = T.axis.S(domain, value0)
1213 * v1 = T.axis.R(domain, value1)
1214 * ...
1215 * T.reads([buffer0[start:end, ...], ...])
1216 * T.writes([buffer1[start:end, ...], ...])
1217 * T.where(predicate)
1218 * buffer2 = T.alloc_buffer(shape, dtype)
1219 * buffer3 = T.match_buffer(source_buffer[start:end, ...])
1220 * T.attr({attr_key: attr_value, ...})
1221 * with T.init():
1222 * // init body
1223 * // body
1224 *
1225 * \endcode

1241 * reduction block. The optional init field allows us to represent initialization and

1244 * Init field is `NullOpt` if there is no reduction iter_vars
