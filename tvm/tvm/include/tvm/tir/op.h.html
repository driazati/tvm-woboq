655/*!
656 * \brief Execute a multiplication between two Q-numbers x and y
657 * followed by a right shift s. The mathematical expression is:
658 *
659 * out = round(x*y*2^-s)
660 *
661 * Please note that the two Q-numbers x and y are supposed to have
662 * the same number of fractional bits q.
663 *
664 * More about Q-numbers here: https://en.wikipedia.org/wiki/Q_(number_format)
665 *
666 * The rounding rule is to the nearest value, rounding half up
667 * (i.e., round(x.1) = x and round (x.5) = x+1)
668 * \param x first Q-number
669 * \param y second Q-number
670 * \param q number of fractional bits in x and y. Needs to be > 0
671 * \param s integer right shift
672 * \param span The location of this operation in the source.
673 * \return The constructed expression.
674 */
675TVM_DLL PrimExpr q_multiply_shift(PrimExpr x, PrimExpr y, PrimExpr q, PrimExpr s,
676 Span span = Span());

678// Intrinsic operators
679#define TVM_DECLARE_INTRIN_UNARY(OpName) \
680 inline PrimExpr OpName(PrimExpr x, Span span = Span()) { \
681 static const Op& op = Op::Get("tir." #OpName); \
682 if (x.dtype().is_bfloat16()) { \
683 DataType bf16_dtype = x.dtype(); \
684 DataType fp32_dtype(kDLFloat, 32, bf16_dtype.lanes()); \
685 PrimExpr x_fp32 = tir::Cast(fp32_dtype, {x}, span); \
686 PrimExpr result_fp32 = tir::Call(fp32_dtype, op, {x_fp32}, span); \
687 return tir::Cast(bf16_dtype, {result_fp32}, span); \
688 } else { \
689 return tir::Call(x.dtype(), op, {x}, span); \
690 } \
691 }

907template <typename ValueType>
908inline PrimExpr MakeConstScalar(DataType t, ValueType value, Span span = Span()) {
909 if (t.is_int()) return IntImm(t, static_cast<int64_t>(value), span);
910 if (t.is_uint()) {
911 // Use IntImm if it is a small integer
912 uint64_t uval = static_cast<uint64_t>(value);
913 if (uval <= static_cast<uint64_t>(std::numeric_limits<int64_t>::max())) {
914 return IntImm(t, static_cast<int64_t>(value), span);
915 } else {
916 uint64_t mask = (static_cast<uint64_t>(1) << 32U) - 1U;
917 uint64_t low = uval & mask;
918 uint64_t high = uval >> 32U;
919 return LargeUIntImm(t, static_cast<int64_t>(low), static_cast<int64_t>(high), span);
920 }
921 }
922 if (t.is_float() || t.is_bfloat16()) return FloatImm(t, static_cast<double>(value), span);
923 // For now, we store const scalar values of custom datatypes within doubles; later, during the
924 // datatypes lowering pass, we will lower the value to its true representation in the format
925 // specified by the datatype.
926 // TODO(gus) when do we need to start worrying about doubles not being precise enough?
927 if (static_cast<uint8_t>(t.code()) >= static_cast<uint8_t>(DataType::kCustomBegin)) {
928 return FloatImm(t, static_cast<double>(value), span);
929 }
930 LOG(FATAL) << "cannot make const for type " << t;
931 return PrimExpr();
932}

934template <typename ValueType, typename>
935inline PrimExpr make_const(DataType t, ValueType value, Span span) {
936 if (t.lanes() == 1) {
937 return MakeConstScalar(t, value, span);
938 } else {
939 return tir::Broadcast(MakeConstScalar(t.element_of(), value, span), t.lanes(), span);
940 }
