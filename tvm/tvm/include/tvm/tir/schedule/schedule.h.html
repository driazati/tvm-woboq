112 /*! \brief Get the IRModule associated with this schedule. */

119 * \brief Instruct the schedule to work on a function in the IRModule.
120 *
121 * By default, the schedule works on the function with the name "main", or the only function in

294 /******** Schedule: Transform loops ********/
295 /*!
296 * \brief Fuse a list of consecutive loops into one. It requires:
297 * 1) The loops can't have annotations or thread bindings.
298 * 2) The (i+1)-th loop must be the only child of the i-th loop.
299 * 3) All loops must start with 0.
300 * 4) The domain of a loop to be fused cannot depend on another loop to be fused.
301 * \param loop_rvs The loops to be fused
302 * \param preserve_unit_iters Whether or not to preserve unit iterators in block bindings
303 * \return The new loop after fusion
304 */
305 virtual LoopRV Fuse(const Array<LoopRV>& loop_rvs, bool preserve_unit_iters = true) = 0;
306 /*!
307 * \brief Split a loop into a list of consecutive loops. It requires:
308 * 1) The loop can't have annotation or thread binding.
309 * 2) The loop must start with 0.
310 * \param loop_rv The loop to be split
311 * \param factors The positive tiling factors, and at most one of which is `NullOpt`, which means
312 * that factor is inferred.
313 * \param preserve_unit_iters Whether or not to preserve unit iterators in block bindings
314 * \return The new loops after split
315 */
316 virtual Array<LoopRV> Split(const LoopRV& loop_rv, const Array<Optional<ExprRV>>& factors,
317 bool preserve_unit_iters = true) = 0;
318 /*!
319 * \brief Reorder a list of loops. It doesn't require the loops to be consecutive.
320 * It requires:
321 * 1) The loops are in the same chain. That means: the loops can be ordered to [l_1, l_2, ... ,
322 * l_n] where l_i is an ancestor of l_{i+1} and there are only single-branch loops between
323 * l_1 and l_n (which also indicates they are under the same scope).
324 * 2) After reordering, the domain of an outer loop cannot depend on any of the inner loops.
325 * 3) For every block under the loop nests, its block binding must be affine, and the block
326 * variables must be either data parallel or reduction.
327 * 4) No duplicated loops are allowed in the arguments.
328 * \param ordered_loop_rvs The loops in the new order
329 */
330 virtual void Reorder(const Array<LoopRV>& ordered_loop_rvs) = 0;
331 /*!
332 * \brief Create a new unit loop on top of the specific block.
333 * \param block_rv The block above which the new loop is created
334 * \return The new loop created
335 */
336 virtual LoopRV AddUnitLoop(const BlockRV& block_rv) = 0;
337 /*!
338 * \brief Create a new unit loop on top of the specific loop.
339 * \param loop_rv The loop above which the new loop is created
340 * \return The new loop created
341 */
342 virtual LoopRV AddUnitLoop(const LoopRV& loop_rv) = 0;

349 * 3) For each block under the loop, the loop can only be contained in data-parallel block iters'
350 * bindings
351 * \param loop_rv The loop to be parallelized
352 */
353 virtual void Parallel(const LoopRV& loop_rv) = 0;
354 /*!
355 * \brief Vectorize the input loop. It requires:
356 * 1) The scope block that the loop is in should have stage-pipeline property
357 * 2) All the blocks under the loop are complete blocks or reduction blocks, and have affine
358 * bindings
359 * 3) For each block under the loop, the loop can only be contained in data-parallel block iters'
360 * bindings
361 * \param loop_rv The loop to be vectorized
362 */
363 virtual void Vectorize(const LoopRV& loop_rv) = 0;
364 /*!
365 * \brief Bind the input loop to the given thread axis. It requires:
366 * 1) The scope block that the loop is in should have stage-pipeline property
367 * 2) All the blocks under the loop are complete blocks or reduction blocks, and have affine
368 * bindings
369 * 3) For each block under the loop, if the thread axis starts with "threadIdx`, the loop can only
370 * be contained in data-parallel block iter and reduction block iters' bindings. Otherwise the
371 * loop can only be contained in data-parallel block iters' bindings
372 * \param loop_rv The loop to be bound to the thread axis
373 * \param thread_axis The thread axis to be bound to the loop
374 */
375 virtual void Bind(const LoopRV& loop_rv, const String& thread_axis) = 0;
376 /*!
377 * \brief Unroll the input loop. It requires nothing
378 * \param loop_rv The loop to be unrolled
379 */
380 virtual void Unroll(const LoopRV& loop_rv) = 0;
381 /******** Schedule: Insert cache stages ********/
382 /*!
383 * \brief Create a block that reads a buffer region into a read cache. It requires:
384 * 1) There is at most one block who writes the buffer in the scope.
385 * 2) The scope block have stage-pipeline property.
386 * \param block_rv The consumer block of the target buffer.
387 * \param read_buffer_index The index of the buffer in block's read region.
388 * \param storage_scope The target storage scope.
389 * \param consumer_blocks An optional list of consumers of the cache to rewrite.
390 * \return The cache stage block.
391 */
392 virtual BlockRV CacheRead(const BlockRV& block_rv, int read_buffer_index,
393 const String& storage_scope,
394 const Array<BlockRV> consumer_blocks = {}) = 0;
395 /*!
396 * \brief Create a block that writes a buffer region into a write cache. It requires:
397 * 1) There is only one block who writes the target buffer.
398 * 2) The scope block have stage-pipeline property.
399 * \param block_rv The producer of the buffer
400 * \param write_buffer_index The index of the buffer in block's write region
401 * \param storage_scope The target storage scope
402 * \return The cache stage block.
403 */
404 virtual BlockRV CacheWrite(const BlockRV& block_rv, int write_buffer_index,
405 const String& storage_scope) = 0;
406 /*!
407 * \brief Create a block that read/write a buffer region into a read/write cache with reindexing.
408 * The layout of the cache will be the same as by the iterators of the block that reads/writes the
409 * buffer. It requires:
410 * 1) There is only one block who reads/writes the target buffer
411 * 2) There is only one buffer load/store of this buffer in the block
412 * \param block_rv The block operates on the target buffer.
413 * \param buffer_index The index of the buffer in block's read or write region.
414 * \param buffer_index_type The type of the buffer index, kRead or kWrite.
415 * \return The reindex stage block.
416 */
417 virtual BlockRV ReIndex(const BlockRV& block_rv, int buffer_index,
418 BufferIndexType buffer_index_type) = 0;
419 /******** Schedule: Compute location ********/
420 /*!
421 * \brief Move a producer block under the specific loop, and regenerate the
422 * loops induced by the block so that the buffer region produced by the producer block could
423 * cover those regions consumed by its consumer blocks under the given loop. It requires:
424 * 1) `block` and `loop` are under the same scope, `loop` is not the ancestor of `block`
425 * 2) The scope block has stage-pipeline property
426 * 3) The subtree of the scope block, where the given block is in, satisfies the compact dataflow
427 * condition. i.e. all the blocks in the scope block's subtree must be either complete block or
428 * reduction block
429 * 4) The block is not an output block with regard to the scope block, i.e. the buffers written by
430 * the block are allocated under the scope block
431 * 5) All the consumers of the block are under the given loop
432 * \param block_rv The block to be moved
433 * \param loop_rv The loop where the block to be moved under
434 * \param preserve_unit_loops Whether to keep the trivial loops whose extents are 1
435 * \param index The block index of the loop body subtree blocks:
436 * - `index = -1` means inserted into the last possible insertion point;
437 * - `index = -2` means inserted into the first possible insertion point;
438 * - Otherwise, `index` is a nonnegative number that indicates the insertion point
439 */
440 virtual void ComputeAt(const BlockRV& block_rv, const LoopRV& loop_rv, bool preserve_unit_loops,
441 int index = -1) = 0;
442 /*!
443 * \brief Move a consumer block under the specific loop, and regenerate the
444 * loops induced by the block so that the buffer region consumed by the consumer block could
445 * cover those regions produced by its producer blocks under the given loop. It requires:
446 * 1) `block` and `loop` are under the same scope, `loop` is not the ancestor of `block`
447 * 2) The scope block has stage-pipeline property
448 * 3) The subtree of the scope block, where the given block is in, satisfies the compact dataflow
449 * condition. i.e. all the blocks in the scope block's subtree must be either complete block or
450 * reduction block
451 * 4) All the producers of the block are under the given loop
452 *
453 * \param block_rv The block to be moved
454 * \param loop_rv The loop where the block to be moved under
455 * \param preserve_unit_loops Whether to keep the trivial loops whose extents are 1
456 * \param index The block index of the loop body subtree blocks:
457 * - `index = -1` means inserted into the last possible insertion point;
458 * - `index = -2` means inserted into the first possible insertion point;
459 * - Otherwise, `index` is a nonnegative number that indicates the insertion point
460 */
461 virtual void ReverseComputeAt(const BlockRV& block_rv, const LoopRV& loop_rv,
462 bool preserve_unit_loops, int index = -1) = 0;
463 /*!
464 * \brief Inline a block into its consumer(s). It requires:
465 * 1) The block is a complete non-root block, which only produces one buffer
466 * 2) The block must not be the only leaf in the scope.
467 * 3) The body of the block must be a BufferStore statement in the form of,
468 * A[i, j, k, ...] = ...
469 * where the indices of the LHS are all distinct atomic variables,
470 * and no variables other than those indexing variables are allowed in the statement.
471 * \param block The block to be inlined to its consumer(s)
472 */
473 virtual void ComputeInline(const BlockRV& block) = 0;
474 /*!
475 * \brief Inline a block into its only producer. It requires:
476 * 1) The block is a complete non-root block, which only produces and consumers one buffer
477 * 2) The block must not be the only leaf in the scope.
478 * 3) The only producer of the block is a read-after-write producer and a complete non-root block
479 * 4) The body of the block must be a BufferStore statement in the form of,
480 * B[f(i, j, k, ...)] = g(i, j, k, A[i, j, k, ...] ...)
481 * where the indices of each `BufferLoad` on the RHS are all distinct atomic variables,
482 * and no variables other than those indexing variables are allowed in the statement.
483 * \param block The block to be inlined to its producer
484 */
485 virtual void ReverseComputeInline(const BlockRV& block) = 0;
486 /******** Schedule: Reduction ********/
487 /*!
488 * \brief Decompose a reduction block into two separate blocks.
489 * a) The init block, which is translated from the init statement of the reduction block;
490 * b) The update block, which is the original block without init statement.
491 *
492 * The init block is inserted right before the given loop.
493 *
494 * The schedule primitive requires:
495 * 1) The input block is a reduction block.
496 * 2) The input loop is the ancestor of the block.
497 * 3) The input loop is not lower than all the loops related to reduce block var.
498 * \param block_rv The reduction block to be decomposed
499 * \param loop_rv The loop above which the init block is inserted before.
500 * \return The init block
501 */
502 virtual BlockRV DecomposeReduction(const BlockRV& block_rv, const LoopRV& loop_rv) = 0;
503 /*!
504 * \brief Factorize an associative reduction block by the specified loop.
505 * \details An associative reduction cannot be parallelized directly,
506 * because it leads to potential race condition during accumulation.
507 * Alternatively, the reduction could be factorized on a loop with the following steps:
508 * - Step 1: evenly slice the reduction into `n` separate chunks, where `n` is the loop extent
509 * - Step 2: compute the chunks separately and write the result into `n` intermediate buffers;
510 * - Step 3: accumulate the `n` separate buffer into the result buffer.
511 * Note that the Step 2 above introduces opportunities for parallelization.
512 * RFactor is a schedule primitive that implements the transformation described above.
513 * \param loop_rv The loop outside block we want to do rfactor
514 * \param factor_axis The position where the new dimension is placed in the new introduced rfactor
515 * buffer. Suppose the original reduction block writes to buffer `B` with
516 * ndim(B) dimensions, then `factor_axis` should be in range `[-ndim(B) - 1,
517 * ndim(B)]`, and the negative index will be normalized to a non-negative one
518 * \return The rfactor block
519 */
520 virtual BlockRV RFactor(const LoopRV& loop_rv, int factor_axis) = 0;
521 /******** Schedule: Block annotation ********/
522 /*!
523 * \brief Set alignment requirement for specific dimension such that
524 * stride[axis] == k * factor + offset for some k. This is useful to set memory layout for
525 * more friendly memory access pattern. For example, we can set alignment to be factor=2,
526 * offset=1 to avoid bank conflict for thread access on higher dimension in GPU shared
527 * memory.
528 * \param block_rv The producer block of the buffer
529 * \param buffer_index The index of the buffer in block's write region
530 * \param axis The dimension to be specified for alignment
531 * \param factor The factor multiple of alignment
532 * \param offset The required offset factor
533 */
534 virtual void StorageAlign(const BlockRV& block_rv, int buffer_index, int axis, int factor,
535 int offset) = 0;
536 /*!
537 * \brief Set the storage scope of a buffer, where the buffer is specified by the a block and a
538 * write-index
539 * \param block_rv The producer block of the buffer
540 * \param buffer_index The index of the buffer in block's write region
541 * \param storage_scope The storage scope to be set
542 */
543 virtual void SetScope(const BlockRV& block_rv, int buffer_index, const String& storage_scope) = 0;

593 /******** Schedule: Layout transformation ********/
594 /*!
595 * \brief Apply a transformation represented by IndexMap to buffer
596 * \details The indices and the access region to the target buffer is transformed by the given
597 * index_map. The index_map is used to infer the new shape of the buffer. Buffer must be either
598 * a function parameter, or allocated in a block (it cannot be a buffer subregion created via
599 * 'match_buffer').
600 * \param block_rv The block that accesses the target buffer.
601 * \param buffer_index The index of the buffer in block's read or write region.
602 * \param buffer_index_type The type of the buffer index, kRead or kWrite.
603 * \param index_map The transformation to apply.
604 *
605 * \param pad_value The value to write into padding introduced by
606 * the transformation. If the schedule contains a producer block
607 * for the specified buffer, the pad value will be written as
608 * part of the producer block if possible, or after the producer
609 * block otherwise. Otherwise, if the buffer is an input, will
610 * insert an annotation block to state that the padding contains
611 * the known value.
612 *
613 * Note: If applied to an input buffer, the calling scope is
614 * responsible for ensuring that the pad_value is present.
615 * Algebraic symplifications, branch elimination, and other
616 * optimizations may assume that this precondition is met, and
617 * may result in incorrect results being returned.
618 */
619 virtual void TransformLayout(const BlockRV& block_rv, int buffer_index,
620 BufferIndexType buffer_index_type, const IndexMap& index_map,
621 const Optional<IndexMap>& pad_value = NullOpt) = 0;
622
623 /*!
624 * \brief Apply a transformation represented by IndexMap to block
625 * \details The block iters and the block body are transformed by the given index_map.
626 * Outer loops corresponding to each new block iter are regenerated.
627 * The index_map is required to be bijective affine since we need its inverse mapping.
628 * \param block_rv The block to be transformed
629 * \param index_map The transformation to apply.
630 */
631 virtual void TransformBlockLayout(const BlockRV& block_rv, const IndexMap& index_map) = 0;
632
633 /*!
634 * \brief Set the axis separator of a buffer, where the buffer is specified by a block and a read
635 * or write index
636 * \param block_rv The block that accesses the target buffer.
637 * \param buffer_index The index of the buffer in block's read or write region.
638 * \param buffer_index_type The type of the buffer index, kRead or kWrite.
639 * \param axis_separators The axis separator of the buffer
640 */
641 virtual void SetAxisSeparator(const BlockRV& block_rv, int buffer_index,
642 BufferIndexType buffer_index_type,
643 const Array<IntImm>& axis_separators) = 0;
644
645 /******** Schedule: Padding ********/
646 /*!
647 * \brief Decompose a padding block into a block filling const pad values and a block
648 * writing in-bound values.
649 * \param block_rv The block that match the padding pattern.
650 * \param loop_rv The loop above which the const filling block is inserted before.
651 * \return The const pad value filling block.
652 */
653 virtual BlockRV DecomposePadding(const BlockRV& block_rv, const LoopRV& loop_rv) = 0;
654
655 /*!
656 * \brief Pad the computation of Einsum.
657 * \param block_rv The block that matches the Einsum pattern.
658 * \param padding The padding for each block iter.
659 * \details This schedule primitives identifies the Einsum pattern in the block body, and find its
660 * producer blocks. It then pads the computation of the Einsum pattern and its producer blocks.
661 * The output buffer and the producer buffer is resized according to the padding size. It requires
662 * the output buffer and the producer buffer to be allocated inside the PrimFunc.
663 *
664 * The padding is a list of non-negative integers, each element corresponds to the padding for
665 * each block iter in the order of block iters. The block and its producer blocks should have
666 * trivial bindings, i.e. each block iter is bound to a single loop variable. After padding, the
667 * block iter extent and the corresponding outer loop is extended by the padding size.
668 *
669 * The size of the producer buffers are infered from the padding size of the Einsum computation.
670 * The producer buffers are padded by the initial value of the corresponding reduction.
671 */
672 virtual void PadEinsum(const BlockRV& block_rv, const Array<Integer>& padding) = 0;
673
674 /******** Schedule: Misc ********/
675 /*! \brief A no-op that marks the start of postprocessing phase of scheduling */
676 virtual void EnterPostproc() = 0;
677};
