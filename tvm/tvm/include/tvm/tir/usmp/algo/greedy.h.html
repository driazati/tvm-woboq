36namespace tvm {
37namespace tir {
38namespace usmp {
39namespace algo {
40
41/*!
42 * \brief This is the base class for Greedy Algorithms where the sorting
43 * is specialized in the extended classes based on the greedy criteria.
44 */
45class GreedyBase {
46 public:
47 GreedyBase() {}
48 /*!
49 * \brief This function should be implemented by the extended classes to sort the BufferInfo
50 * objects based on a criteria and then calling PostSortAllocation.
51 */
52 virtual Map<BufferInfo, PoolAllocation> PlanMemory(const Array<BufferInfo>& buffer_info_arr) = 0;
53
54 protected:
55 /*!
56 * \brief Rounds up the offset to satisfy the alignement requirement
57 */
58 size_t round_up_to_byte_alignment(const size_t& non_aligned_byte_offset,
59 const int& byte_alignment);
60
61 /*!
62 * \brief A helper function check whether a offset is valid given the constraints
63 */
64 bool IsValidPlacement(const PoolInfo& candidate_pool, const size_t& next_offset,
65 const size_t& size_bytes);
66
67 /*!
68 * \brief Selects a pool for placement in the given set of ordered pool candidates
69 */
70 PoolInfo SelectPlacementPool(
71 const BufferInfo& buf_info,
72 const std::unordered_map<PoolInfo, size_t, ObjectPtrHash, ObjectPtrEqual>& pool_offsets);
73
74 /*!
75 * \brief This is the base allocation function that works on sorted BufferInfo objects based
76 * on the greedy heuristic. The sorting algorithm has to be called before calling this.
77 */
78 Map<BufferInfo, PoolAllocation> PostSortAllocation(
79 const std::vector<BufferInfo>& buffer_info_vec);
80};
