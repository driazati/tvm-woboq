42class LayoutAxis {
43 public:
44 static const LayoutAxis& Get(const char name);
45
46 // Get the singleton LayoutAxis using itvar->var->name_hint
47 static const LayoutAxis& Get(const tir::IterVar& itvar);
48
49 // Get the singleton LayoutAxis using name[0] (size of name must be 1).
50 static const LayoutAxis& Get(const std::string& name);
51
52 inline bool IsPrimal() const { return name_ >= 'A' && name_ <= 'Z'; }
53 inline std::string name() const { return std::string(1, name_); }
54
55 // if current axis is primal, switch the axis to its subordinate one,
56 // else switch to the primal.
57 inline const LayoutAxis& ToDual() const {
58 if (name_ >= 'A' && name_ <= 'Z') {
59 return LayoutAxis::Get(name_ - 'A' + 'a');
60 } else {
61 return LayoutAxis::Get(name_ - 'a' + 'A');
62 }
63 }
64
65 // return the primal axis. If it is already primal, return itself.
66 const LayoutAxis& ToPrimal() const { return IsPrimal() ? *this : ToDual(); }
67
68 // return the subordinate axis. If it is already subordinate, return itself.
69 const LayoutAxis& ToSubordinate() const { return IsPrimal() ? ToDual() : *this; }
70
71 inline bool operator==(const LayoutAxis& rhs) const { return name_ == rhs.name_; }
72
73 friend std::ostream& operator<<(std::ostream& os, const LayoutAxis& l) {
74 os << l.name();
75 return os;
76 }
77
78 private:
79 static const LayoutAxis UPPER_CASE[];
80 static const LayoutAxis LOWER_CASE[];
81 LayoutAxis(const LayoutAxis&);
82 LayoutAxis& operator=(const LayoutAxis&);
83 explicit LayoutAxis(const char name) : name_(name) {}
84
85 const char name_;
86};
87
88/*!
89 * \brief Layout is to describe how data is organized within an N-dimention tensor.
90 * It is composed of upper cases, lower cases and numbers,
91 * where upper case indicates a primal axis and
92 * the corresponding lower case with factor size indicates the subordinate axis.
93 * For example, NCHW16c can describe a 5-D tensor of
94 * [batch_size, channel, height, width, channel_block].
95 * Here subordinate axis channel_block=16 is the factor size of the primal axis C (channel).
96 * Layout for scalar is defined, while both its name and axes have size 0.
97 */
98class LayoutNode : public Object {
99 public:
100 /*! \brief string representation of layout, "" for scalar. */
101 String name;
102 /*! \brief specify each axis of the layout,
103 * in which the variable name is the name of the axis.
104 * The IterVar's extent indicates the size of the axis,
105 * it is a variable for a primal axis, but a constant for a subordinate axis.
106 * Empty for scalar's layout.
107 */
108 Array<tir::IterVar> axes;
109
110 void VisitAttrs(AttrVisitor* v) {
111 v->Visit("name", &name);
112 v->Visit("axes", &axes);
113 }
114
115 static constexpr const char* _type_key = "tir.Layout";
116 TVM_DECLARE_FINAL_OBJECT_INFO(LayoutNode, Object);
117};
118
119/*!
120 * \brief Managed reference to LayoutNode
121 * \sa LayoutNode
122 */
123class Layout : public ObjectRef {
124 public:
125 explicit Layout(const Array<tir::IterVar>& axes);
126
127 /*! \brief construct from a string */
128 Layout(const tvm::String& name) : Layout(name.operator std::string()) {} // NOLINT(*)
129
130 /*! \brief construct from a string */
131 Layout(const char* name) : Layout(std::string(name)) {} // NOLINT(*)
132
133 /*!
134 * \brief construct from a string.
135 * \param name input in layout convention:
136 * upper case indicates a dimension and
137 * the corresponding lower case with factor size
138 * indicates the split dimension.
139 * return undefined layout if "__undef__" is passed.
140 */
141 TVM_DLL Layout(const std::string& name); // NOLINT(*)
142
143 /*!
144 * \brief access the internal node container
145 * \return the pointer to the internal node container
146 */
147 LayoutNode* operator->() { return static_cast<LayoutNode*>(get_mutable()); }
148
149 /*!
150 * \brief Return an undefined layout.
151 * \return a (global) undefined layout.
152 */
153 static const Layout& Undef() {
154 static Layout undef;
155 return undef;
156 }
157
158 /*!
159 * \brief Returns a sub-layout which is the portion of the object
160 * that starts at dimension \p pos and spans \p len dimensions
161 * (or until the end of the layout, whichever comes first).
162 * \param pos The start position.
163 * \param len The length of the sub-layout. if 0, return layout of scalar
164 * \return A newly constructed Layout object.
165 */
166 Layout SubLayout(size_t pos, size_t len) const;
167
168 /*!
169 * \brief Split \p axis by \p size and put the sub-axis to position \p target_pos.
170 * \param axis The source axis to be split. It must be a primal-axis;
171 * \param target_pos The target position of the newly split subordinate-axis.
172 * \param factor size of the sub-dimension.
173 * \return A newly constructed Layout object.
174 */
175 Layout Split(const LayoutAxis& axis, size_t target_pos, int32_t factor) const;
176
177 /*! \return number of dimensions */
178 inline size_t ndim() const {
179 if (!defined()) return 0;
180 return operator->()->axes.size();
181 }
182
183 /*! \return number of super dimensions */
184 inline size_t ndim_primal() const {
185 if (!defined()) return 0;
186 size_t ct = 0;
187 for (auto x : operator->()->axes) {
188 if (LayoutAxis::Get(x).IsPrimal()) {
189 ct++;
190 }
191 }
192 return ct;
193 }
194
195 /*!
196 * \brief Returns a new layout where the dims have been expanded to match the primal dimensions.
197 * \param dst_layout The dst layout to which current layout has to be expanded.
198 * \return The expanded Layout.
199 */
200 inline Layout ExpandPrimal(const Layout& dst_layout) {
201 Layout new_src_layout;
202 // 1) Find the axis which are missing in the current layout. Make them the prefix.
203 std::string new_src_layout_str = "";
204 for (auto dst_axis : dst_layout->axes) {
205 if (LayoutAxis::Get(dst_axis).IsPrimal()) {
206 if (!this->Contains(LayoutAxis::Get(dst_axis))) {
207 new_src_layout_str += dst_axis->var->name_hint;
208 }
209 }
210 }
211 // 2) Now, add the primal axis of the current layout.
212 new_src_layout_str += this->name();
213 new_src_layout = Layout(new_src_layout_str);
214 return new_src_layout;
215 }
216
217 /*!
218 * \brief return the index of the input axis.
219 * If it is not found in the layout or the layout is undefined,
220 * return -1.
221 * \param axis the input axis.
222 * \return the index or -1 if not found.
223 */
224 inline int32_t IndexOf(const LayoutAxis& axis) const {
225 if (!this->defined()) return -1;
226 const auto axes = operator->()->axes;
227 for (size_t i = 0; i < axes.size(); ++i) {
228 if (axes[i]->var->name_hint == axis.name()) return static_cast<int32_t>(i);
229 }
230 return -1;
231 }
232
233 /*!
234 * \brief Get the factor size of the subordinate axis.
235 * \param axis the input primal-axis or subordinate-axis.
236 * \return the size of the subordinate-axis of \p axis (if \p axis is a primal-axis),
237 * or the size of \p axis itself (if \p axis is a subordinate-axis).
238 * Return -1 if \p axis is not in the layout the layout is undefined.
239 */
240 int32_t FactorOf(const LayoutAxis& axis) const;
241
242 /*!
243 * \brief Whether the layout contains an axis.
244 * \param axis axis to be checked.
245 * \return Whether the layout contains the axis.
246 */
247 bool Contains(const LayoutAxis& axis) const {
248 if (!defined()) return false;
249 for (const tir::IterVar var : operator->()->axes) {
250 if (var->var->name_hint == axis.name()) {
251 return true;
252 }
253 }
254 return false;
255 }
256
257 const LayoutAxis& operator[](int32_t i) const {
258 ICHECK(defined()) << "Try to access axis from an undefined layout.";
259 int32_t index = i < 0 ? static_cast<int32_t>(ndim() + i) : i;
260 ICHECK(index >= 0 && static_cast<size_t>(index) < ndim()) << "Invalid index " << i;
261 const tir::IterVar axis = operator->()->axes[index];
262 return LayoutAxis::Get(axis);
263 }
264
265 /*! \return the string description of the layout */
266 inline std::string name() const {
267 if (!defined()) return "__undef__";
268 return operator->()->name;
269 }
270
271 /*!
272 * \brief Whether the two layouts are equal.
273 * \param rhs Another layout.
274 * \return whether the two layouts are equal.
275 */
276 inline bool Equals(const Layout& rhs) const { return name() == rhs.name(); }
277
278 /*!
279 * \brief allow output string of layout to ostream
280 * \param os the output stream
281 * \param l the layout
282 * \return the ostream
283 */
284 friend std::ostream& operator<<(std::ostream& os, const Layout& l) {
285 os << l.name();
286 return os;
287 }
288
289 TVM_DEFINE_OBJECT_REF_METHODS(Layout, ObjectRef, LayoutNode);
290};

292// Internal node container BijectiveLayout
293class BijectiveLayoutNode : public Object {
294 public:
295 /*! \brief Describes how source axes can be mapped to the destination axes,
296 * e.g., [i0 / 16, i1, i0 % 16] can describe NC -> NC16n
297 */
298 Array<PrimExpr> index_forward_rule;
299 /*! \brief Describes how destination axes can be mapped to the source axes */
300 Array<PrimExpr> index_backward_rule;
301 /*! \brief Describes how source shapes can be mapped to the destination shapes */
302 Array<PrimExpr> shape_forward_rule;
303 /*! \brief Describes how destination shapes can be mapped to the source shapes */
304 Array<PrimExpr> shape_backward_rule;
305
306 /*! \brief The source layout */
307 Layout src_layout;
308 /*! \brief The destination layout */
309 Layout dst_layout;
310
311 void VisitAttrs(AttrVisitor* v) {
312 v->Visit("src_layout", &src_layout);
313 v->Visit("dst_layout", &dst_layout);
314 v->Visit("index_forward_rule", &index_forward_rule);
315 v->Visit("index_backward_rule", &index_backward_rule);
316 v->Visit("shape_forward_rule", &shape_forward_rule);
317 v->Visit("shape_backward_rule", &shape_backward_rule);
318 }
319
320 static constexpr const char* _type_key = "tir.BijectiveLayout";
321 TVM_DECLARE_FINAL_OBJECT_INFO(BijectiveLayoutNode, Object);
322};

324/*!
325 * \brief Bijective function mapping for data layout transformation.
326 * Given two Layout, BijectiveLayout build and store the mapping rules,
327 * provides API to transform N-dimention tensor from the source indices (i0, i1, .., im)
328 * to the destination indices (j0, j1, .., jm).
329 */
330class BijectiveLayout : public ObjectRef {
331 public:
332 /*!
333 * \brief The constructor
334 * \param src_layout The source layout
335 * \param dst_layout The destination layout
336 */
337 TVM_DLL BijectiveLayout(Layout src_layout, Layout dst_layout);
338
339 // Given the source shape, infer the destination shape.
340 TVM_DLL Array<PrimExpr> ForwardShape(const Array<PrimExpr>& shape) const;
341 // Given the destination shape, recover the source shape.
342 TVM_DLL Array<PrimExpr> BackwardShape(const Array<PrimExpr>& dst_shape) const;
343 // Given the destination indices, infer the destination indices.
344 TVM_DLL Array<PrimExpr> ForwardIndex(const Array<PrimExpr>& index) const;
345 // Given the destination indices, recover the source indices.
346 TVM_DLL Array<PrimExpr> BackwardIndex(const Array<PrimExpr>& dst_index) const;
347
348 TVM_DEFINE_OBJECT_REF_METHODS(BijectiveLayout, ObjectRef, BijectiveLayoutNode);
349};
