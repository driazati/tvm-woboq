20/*!
21 * \file tvm/tir/op_attr_types.h
22 * \brief Attribute types in the Op registry for TIR ops.
23 *
24 * These attributes can be set via OpRegEntry::set_attr
25 *
26 * \sa tvm/ir/op.h
27 */
28#ifndef TVM_TIR_OP_ATTR_TYPES_H_
29#define TVM_TIR_OP_ATTR_TYPES_H_
30
31#include <tvm/ir/expr.h>
32#include <tvm/runtime/container/string.h>
33#include <tvm/runtime/packed_func.h>
34
35#include <ostream>
36
37namespace tvm {
38namespace tir {
39/*!
40 * \brief Global symbol of the op after lowering.
41 */
42using TGlobalSymbol = String;
43
44/*!
45 * \brief Whether the op is overloaded for vector form.
46 */
47using TVectorizable = bool;
48
49/*!
50 * \brief The intrinsic lowering function for given op.
51 */
52using FLowerIntrinsic = runtime::TypedPackedFunc<PrimExpr(PrimExpr)>;
53
54/*!
55 * \brief The legalization function for given tir op.
56 */
57using FLegalize = runtime::TypedPackedFunc<PrimExpr(PrimExpr)>;
58
59/*!
60 * \brief The effect type of the call.
61 */
62enum class CallEffectKind : int {
63 /*! \brief Function corresponds to an annotation(e.g. likely) and can translate to identity. */
64 kExprAnnotation = 0,
65 /*!
66 * \brief Pure function that do not interacts
67 * with any external state.
68 */
69 kPure = 1,
70 /*!
71 * \brief Function's that may read from states(e.g. RAM)
72 */
73 kReadState = 2,
74 /*!
75 * \brief Function that may read/write from states(e.g. RAM).
76 */
77 kUpdateState = 3,
78 /*!
79 * \brief Opaque function, cannot make any assumption
80 */
81 kOpaque = kUpdateState,
82 /*!
83 * \brief Special intrinsic to annotate call arguments info
84 * only valid as a direct argument to a call.
85 */
86 kSpecialCallArg = 4,
87 /*!
88 * \brief Embed opaque information in the Expr, cannot be codegen.
89 */
90 kEmbedInfo = 5,
91 /*!
92 * \brief Function that changes control flow
93 */
94 kControlJump = 6,
95};
96
97inline std::ostream& operator<<(std::ostream& os, CallEffectKind side_effect) {
98 switch (side_effect) {
99 case CallEffectKind::kExprAnnotation:
100 return os << "kExprAnnotation";
101
102 case CallEffectKind::kPure:
103 return os << "kPure";
104
105 case CallEffectKind::kReadState:
106 return os << "kReadState";
107
108 case CallEffectKind::kUpdateState:
109 return os << "kUpdateState";
110
111 case CallEffectKind::kSpecialCallArg:
112 return os << "kSpecialCallArg";
113
114 case CallEffectKind::kEmbedInfo:
115 return os << "kEmbedInfo";
116
117 case CallEffectKind::kControlJump:
118 return os << "kControlJump";
119
120 default:
121 LOG(FATAL) << "Unknown CallEffectKind: " << static_cast<int>(side_effect);
122 }
123}
124
125/*! \brief Use integer to record the kind. */
126using TCallEffectKind = Integer;
127
128} // namespace tir
129} // namespace tvm
130#endif // TVM_TIR_OP_ATTR_TYPES_H_
