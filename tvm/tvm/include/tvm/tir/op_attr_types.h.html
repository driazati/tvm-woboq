20/*!
21 * \file tvm/tir/op_attr_types.h
22 * \brief Attribute types in the Op registry for TIR ops.
23 *
24 * These attributes can be set via OpRegEntry::set_attr
25 *
26 * \sa tvm/ir/op.h
27 */
28#ifndef TVM_TIR_OP_ATTR_TYPES_H_
29#define TVM_TIR_OP_ATTR_TYPES_H_
30
31#include <tvm/ir/expr.h>
32#include <tvm/runtime/container/string.h>
33#include <tvm/runtime/packed_func.h>
34
35namespace tvm {
36namespace tir {
37/*!
38 * \brief Global symbol of the op after lowering.
39 */
40using TGlobalSymbol = String;
41
42/*!
43 * \brief Whether the op is overloaded for vector form.
44 */
45using TVectorizable = bool;
46
47/*!
48 * \brief The intrinsic lowering function for given op.
49 */
50using FLowerIntrinsic = runtime::TypedPackedFunc<PrimExpr(PrimExpr)>;
51
52/*!
53 * \brief The legalization function for given tir op.
54 */
55using FLegalize = runtime::TypedPackedFunc<PrimExpr(PrimExpr)>;
56
57/*!
58 * \brief The effect type of the call.
59 */
60enum class CallEffectKind : int {
61 /*! \brief Function corresponds to an annotation(e.g. likely) and can translate to identity. */
62 kExprAnnotation = 0,
63 /*!
64 * \brief Pure function that do not interacts
65 * with any external state.
66 */
67 kPure = 1,
68 /*!
69 * \brief Function's that may read from states(e.g. RAM)
70 */
71 kReadState = 2,
72 /*!
73 * \brief Function that may read/write from states(e.g. RAM).
74 */
75 kUpdateState = 3,
76 /*!
77 * \brief Opaque function, cannot make any assumption
78 */
79 kOpaque = kUpdateState,
80 /*!
81 * \brief Special intrinsic to annotate call arguments info
82 * only valid as a direct argument to a call.
83 */
84 kSpecialCallArg = 4,
85 /*!
86 * \brief Embed opaque information in the Expr, cannot be codegen.
87 */
88 kEmbedInfo = 5,
89 /*!
90 * \brief Function that changes control flow
91 */
92 kControlJump = 6,
93};
94
95/*! \brief Use integer to record the kind. */
96using TCallEffectKind = Integer;
97
98} // namespace tir
99} // namespace tvm
100#endif // TVM_TIR_OP_ATTR_TYPES_H_
