20#ifndef TVM_META_SCHEDULE_POSTPROC_H_
21#define TVM_META_SCHEDULE_POSTPROC_H_
22
23#include <tvm/node/reflection.h>
24#include <tvm/runtime/object.h>
25#include <tvm/runtime/packed_func.h>
26#include <tvm/tir/schedule/schedule.h>
27
28namespace tvm {
29namespace meta_schedule {
30
31class TuneContext;
32
33/*!
34 * \brief Rules to apply a postprocessor to a schedule.
35 */
36class PostprocNode : public runtime::Object {
37 public:
38 /*! \brief Virtual destructor. */
39 virtual ~PostprocNode() = default;
40
41 void VisitAttrs(tvm::AttrVisitor* v) {}
42
43 /*!
44 * \brief Initialize the design space generator with tuning context.
45 * \param context The tuning context for initialization.
46 * \note This method is supposed to be called only once before every other method.
47 */
48 virtual void InitializeWithTuneContext(const TuneContext& context) = 0;
49
50 /*!
51 * \brief Apply a postprocessor to the given schedule.
52 * \param sch The schedule to be post processed.
53 * \return Whether the postprocessor was successfully applied.
54 */
55 virtual bool Apply(const tir::Schedule& sch) = 0;
56
57 static constexpr const char* _type_key = "meta_schedule.Postproc";
58 TVM_DECLARE_BASE_OBJECT_INFO(PostprocNode, Object);
59};
60
61/*! \brief The postprocessor with customized methods on the python-side. */
62class PyPostprocNode : public PostprocNode {
63 public:
64 /*!
65 * \brief The function type of `InitializeWithTuneContext` method.
66 * \param context The tuning context for initialization.
67 */
68 using FInitializeWithTuneContext = runtime::TypedPackedFunc<void(const TuneContext&)>;
69 /*!
70 * \brief Apply a postprocessor to the given schedule.
71 * \param sch The schedule to be post processed.
72 * \return Whether the postprocessor was successfully applied.
73 */
74 using FApply = runtime::TypedPackedFunc<bool(const tir::Schedule&)>;
75 /*!
76 * \brief Get the postprocessor function as string with name.
77 * \return The string of the postprocessor function.
78 */
79 using FAsString = runtime::TypedPackedFunc<String()>;
80
81 /*! \brief The packed function to the `InitializeWithTuneContext` function. */
82 FInitializeWithTuneContext f_initialize_with_tune_context;
83 /*! \brief The packed function to the `Apply` function. */
84 FApply f_apply;
85 /*! \brief The packed function to the `AsString` function. */
86 FAsString f_as_string;
87
88 void VisitAttrs(tvm::AttrVisitor* v) {
89 // `f_initialize_with_tune_context` is not visited
90 // `f_apply` is not visited
91 // `f_as_string` is not visited
92 }
93
94 void InitializeWithTuneContext(const TuneContext& context) final;
95 bool Apply(const tir::Schedule& sch) final;
96
97 static constexpr const char* _type_key = "meta_schedule.PyPostproc";
98 TVM_DECLARE_FINAL_OBJECT_INFO(PyPostprocNode, PostprocNode);
99};
100
101/*!
102 * \brief Managed reference to PostprocNode
103 * \sa PostprocNode
104 */
105class Postproc : public runtime::ObjectRef {
106 public:
107 /*!
108 * \brief Create a postprocessor with customized methods on the python-side.
109 * \param f_initialize_with_tune_context The packed function of `InitializeWithTuneContext`.
110 * \param f_apply The packed function of `Apply`.
111 * \param f_as_string The packed function of `AsString`.
112 * \return The postprocessor created.
113 */
114 TVM_DLL static Postproc PyPostproc(
115 PyPostprocNode::FInitializeWithTuneContext f_initialize_with_tune_context, //
116 PyPostprocNode::FApply f_apply, //
117 PyPostprocNode::FAsString f_as_string);
118 /*!
119 * \brief Create a postprocessor that checks if all loops are static
120 * \return The postprocessor created
121 */
122 TVM_DLL static Postproc DisallowDynamicLoop();
123 /*!
124 * \brief Create a postprocessor that rewrites the cooperative fetch annotation to
125 * actual vectorized cooperative fetching in loop bindings.
126 * \return The postprocessor created.
127 */
128 TVM_DLL static Postproc RewriteCooperativeFetch();
129 /*!
130 * \brief Creates a postprocessor that applies parallelization, vectorization and auto unrolling
131 * according to the annotation of each block
132 * \return The postprocessor created
133 */
134 TVM_DLL static Postproc RewriteParallelVectorizeUnroll();
135 /*!
136 * \brief Create a postprocessor that rewrites reduction block by moving the init block out.
137 * \return The postprocessor created.
138 */
139 TVM_DLL static Postproc RewriteReductionBlock();
140 /*!
141 * \brief Create a postprocessor that adds thread binding to unbound blocks
142 * \param max_threadblocks The max number of threadblocks in the cuda device.
143 * \return The postprocessor created.
144 */
145 TVM_DLL static Postproc RewriteUnboundBlock(int max_threadblocks);
146 /*!
147 * \brief Create a postprocessor that applies tensorization to annotated blocks
148 * \param vectorize_init_loop Whether or not vectorize the initialization loop produced by
149 * DecomposeReduction
150 * \return The postprocessor created.
151 */
152 TVM_DLL static Postproc RewriteTensorize(bool vectorize_init_loop = false);
153 /*!
154 * \brief Creates a postprocessor that verifies if the GPU code is correct
155 * \return The postprocessor created
156 */
157 TVM_DLL static Postproc VerifyGPUCode();
158 /*!
159 * \brief Creates a postprocessor that rewrites the layout of input tensor
160 * \note Weight layout rewrite is supported so far, activation layout rewrite will be added.
161 * \return The postprocessor created
162 */
163 TVM_DLL static Postproc RewriteLayout();
164 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(Postproc, ObjectRef, PostprocNode);
165};
