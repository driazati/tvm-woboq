23#include <tvm/node/reflection.h>

25#include <tvm/runtime/object.h>
26#include <tvm/runtime/packed_func.h>
27#include <tvm/support/random_engine.h>
28#include <tvm/tir/schedule/schedule.h>
29#include <tvm/tir/schedule/trace.h>

31namespace tvm {
32namespace meta_schedule {
33
34class TuneContext;
35
36/*! \brief Mutator is designed to mutate the trace to explore the design space. */
37class MutatorNode : public runtime::Object {
38 public:
39 /*! \brief Virtual destructor. */
40 virtual ~MutatorNode() = default;
41
42 void VisitAttrs(tvm::AttrVisitor* v) {}
43
44 /*!
45 * \brief Initialize the design space generator with tuning context.
46 * \param context The tuning context for initialization.
47 * \note This method is supposed to be called only once before every other method.
48 */
49 virtual void InitializeWithTuneContext(const TuneContext& context) = 0;
50
51 /*!
52 * \brief Apply the mutator function to the given trace.
53 * \param trace The given trace for mutation.
54 * \param rand_state The random state for mutation.
55 * \return None if mutator failed, otherwise return the mutated trace.
56 */
57 virtual Optional<tir::Trace> Apply(const tir::Trace& trace,
58 support::LinearCongruentialEngine::TRandState* rand_state) = 0;
59
60 static constexpr const char* _type_key = "meta_schedule.Mutator";
61 TVM_DECLARE_BASE_OBJECT_INFO(MutatorNode, Object);
62};
63
64/*! \brief The mutator with customized methods on the python-side. */
65class PyMutatorNode : public MutatorNode {
66 public:
67 /*!
68 * \brief The function type of `InitializeWithTuneContext` method.
69 * \param context The tuning context for initialization.
70 */
71 using FInitializeWithTuneContext = runtime::TypedPackedFunc<void(const TuneContext&)>;
72 /*!
73 * \brief Apply the mutator function to the given trace.
74 * \param trace The given trace for mutation.
75 * \return None if mutator failed, otherwise return the mutated trace.
76 */
77 using FApply = runtime::TypedPackedFunc<Optional<tir::Trace>(
78 const tir::Trace&, support::LinearCongruentialEngine::TRandState rand_state)>;
79 /*!
80 * \brief Get the mutator as string with name.
81 * \return The string of the mutator.
82 */
83 using FAsString = runtime::TypedPackedFunc<String()>;
84
85 /*! \brief The packed function to the `InitializeWithTuneContext` function. */
86 FInitializeWithTuneContext f_initialize_with_tune_context;
87 /*! \brief The packed function to the `Apply` function. */
88 FApply f_apply;
89 /*! \brief The packed function to the `AsString` function. */
90 FAsString f_as_string;
91
92 void VisitAttrs(tvm::AttrVisitor* v) {
93 // `f_initialize_with_tune_context` is not visited
94 // `f_apply` is not visited
95 // `f_as_string` is not visited
96 }
97
98 void InitializeWithTuneContext(const TuneContext& context) final;
99 Optional<tir::Trace> Apply(const tir::Trace& trace,
100 support::LinearCongruentialEngine::TRandState* rand_state) final;
101
102 static constexpr const char* _type_key = "meta_schedule.PyMutator";
103 TVM_DECLARE_FINAL_OBJECT_INFO(PyMutatorNode, MutatorNode);
104};
105
106/*!
107 * \brief Managed reference to MutatorNode
108 * \sa MutatorNode
109 */
110class Mutator : public runtime::ObjectRef {
111 public:
112 /*! \brief Create a Mutator that mutates the decision of instruction Sample-Perfect-Tile */
113 TVM_DLL static Mutator MutateTileSize();
114 /*!
115 * \brief Create a Mutator that mutates the parallel extent
116 * \param max_jobs_per_core The maximum number of parallel jobs per core.
117 * \return The created mutator.
118 */
119 TVM_DLL static Mutator MutateParallel(int64_t max_jobs_per_core);
120 /*!
121 * \brief Create a Mutator that mutates auto unroll step
122 * \return The mutator created
123 */
124 TVM_DLL static Mutator MutateUnroll();
125 /*!
126 * \brief Create a Mutator that mutates the outcome of SampleComputeLocation
127 * \return The mutator created
128 */
129 TVM_DLL static Mutator MutateComputeLocation();
130 /*!
131 * \brief Create a Mutator that mutates auto thread binding.
132 * \return The mutator created
133 */
134 TVM_DLL static Mutator MutateThreadBinding();
135 /*!
136 * \brief Create a mutator with customized methods on the python-side.
137 * \param f_initialize_with_tune_context The packed function of `InitializeWithTuneContext`.
138 * \param f_apply The packed function of `Apply`.
139 * \param f_as_string The packed function of `AsString`.
140 * \return The mutator created.
141 */
142 TVM_DLL static Mutator PyMutator(
143 PyMutatorNode::FInitializeWithTuneContext f_initialize_with_tune_context, //
144 PyMutatorNode::FApply f_apply, //
145 PyMutatorNode::FAsString f_as_string);
146 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(Mutator, ObjectRef, MutatorNode);
147};
