32// Forward declaration
33class TuneContext;

35/*!
36 * \brief The abstract class for design space generation.
37 * \note The relationship between SpaceGenerator and other classes are as follows:
38 ┌──────────────────────────────────────────────────────────────┐
39 ┌──┴───────────────────────────────────────────────────────────┐ │
40┌──┴────────────────── Tune Context ───────────────────────────┐ │ │
41│ ┌─────────────────────┐ │ │ │
42│ │ │ Generate │ │ │
43│ │ Space Generator ├──────────────┐ │ │ │
44│ │ │ │ │ │ │
45│ └─────────────────────┘ ▼ │ │ │
46│ Design Space │ │ │
47│ ┌─────────────────────┐ │ │ │ │
48│ Generate │ │ Pretuning │ │ │ │
49│ ┌───────────┤ Search Strategy │◄─────────────┘ │ │ │
50│ │ │ │ │ ├──┘
51│ │ └─────────────────────┘ ├──┘
52└────┼─────────────────────────────────────────────────────────┘
53 │
54 │
55┌────┼──────────────── Managed By Task Scheduler ─────────────────────┐
56│ │ ┌───────────┐ │
57│ │ Send to │ │ Send to │
58│ ▼ ┌─────────────►│ Builder ├──────────┐ │
59│ Measure Candidate │ Builder │ │ Runner │ │
60│ │ │ └───────────┘ │ │
61│ │ ┌────────────┴────────┐ │ │
62│ │ │ │ ┌───────────┐ │ │
63│ └────►│ Task Scheduler │ │ │ │ │
64│ │ │ │ Runner │◄─────────┘ │
65│ └─────────────────────┘ │ │ │
66│ ▲ └─────┬─────┘ │
67│ │ │ │
68│ └─── Runner Future ◄────┘ │
69└─────────────────────────────────────────────────────────────────────┘
70*/
71class SpaceGeneratorNode : public runtime::Object {
72 public:
73 /*! \brief Default destructor */
74 virtual ~SpaceGeneratorNode() = default;
75
76 /*!
77 * \brief Initialize the design space generator with tuning context.
78 * \param context The tuning context for initialization.
79 * \note This method is supposed to be called only once before every other method.
80 */
81 virtual void InitializeWithTuneContext(const TuneContext& context) = 0;
82
83 /*!
84 * \brief Generate design spaces given a module.
85 * \param mod The module used for design space generation.
86 * \return The generated design spaces, i.e., schedules.
87 */
88 virtual Array<tir::Schedule> GenerateDesignSpace(const IRModule& mod) = 0;
89
90 static constexpr const char* _type_key = "meta_schedule.SpaceGenerator";
91 TVM_DECLARE_BASE_OBJECT_INFO(SpaceGeneratorNode, Object);
92};
93
94/*! \brief The design space generator with customized methods on the python-side. */
95class PySpaceGeneratorNode : public SpaceGeneratorNode {
96 public:
97 /*!
98 * \brief The function type of `InitializeWithTuneContext` method.
99 * \param context The tuning context for initialization.
100 */
101 using FInitializeWithTuneContext = runtime::TypedPackedFunc<void(const TuneContext&)>;
102 /*!
103 * \brief The function type of `GenerateDesignSpace` method.
104 * \param mod The module used for design space generation.
105 * \return The generated design spaces, i.e., schedules.
106 */
107 using FGenerateDesignSpace = runtime::TypedPackedFunc<Array<tir::Schedule>(const IRModule&)>;
108
109 /*! \brief The packed function to the `InitializeWithTuneContext` function. */
110 FInitializeWithTuneContext f_initialize_with_tune_context;
111 /*! \brief The packed function to the `GenerateDesignSpace` function. */
112 FGenerateDesignSpace f_generate_design_space;
113
114 void VisitAttrs(tvm::AttrVisitor* v) {
115 // `f_initialize_with_tune_context` is not visited
116 // `f_generate_design_space` is not visited
117 }
118
119 void InitializeWithTuneContext(const TuneContext& context) final;
120 Array<tir::Schedule> GenerateDesignSpace(const IRModule& mod) final;
121
122 static constexpr const char* _type_key = "meta_schedule.PySpaceGenerator";
123 TVM_DECLARE_FINAL_OBJECT_INFO(PySpaceGeneratorNode, SpaceGeneratorNode);
124};
125
126/*!
127 * \brief Managed reference to SpaceGeneratorNode.
128 * \sa SpaceGeneratorNode
129 */
130class SpaceGenerator : public runtime::ObjectRef {
131 protected:
132 SpaceGenerator() = default;
133
134 public:
135 /*!
136 * \brief Create a design space generator with customized methods on the python-side.
137 * \param f_initialize_with_tune_context The packed function of `InitializeWithTuneContext`.
138 * \param f_generate_design_space The packed function of `GenerateDesignSpace`.
139 * \return The design space generator created.
140 */
141 TVM_DLL static SpaceGenerator PySpaceGenerator(
142 PySpaceGeneratorNode::FInitializeWithTuneContext f_initialize_with_tune_context,
143 PySpaceGeneratorNode::FGenerateDesignSpace f_generate_design_space);
144
145 /*!
146 * \brief Create a design space generator that is union of multiple design space generators.
147 * \param space_generators An array of design space generators to be unioned.
148 * \return The design space generator created.
149 */
150 TVM_DLL static SpaceGenerator SpaceGeneratorUnion(Array<SpaceGenerator, void> space_generators);
151 /*!
152 * \brief Create a design space generator that generates design spaces by applying schedule rules
153 * to blocks in post-DFS order.
154 * \return The design space generator created.
155 */
156 TVM_DLL static SpaceGenerator PostOrderApply(runtime::PackedFunc f_block_filter = nullptr);
157 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(SpaceGenerator, ObjectRef, SpaceGeneratorNode);
158};
