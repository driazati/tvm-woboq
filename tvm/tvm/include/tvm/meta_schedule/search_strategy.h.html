41/*!
42 * \brief The search strategy for measure candidates generation.
43 * \note The relationship between SearchStrategy and other classes are as follows:
44 ┌──────────────────────────────────────────────────────────────┐
45 ┌──┴───────────────────────────────────────────────────────────┐ │
46┌──┴────────────────── Tune Context ───────────────────────────┐ │ │
47│ ┌─────────────────────┐ │ │ │
48│ │ │ Generate │ │ │
49│ │ Space Generator ├──────────────┐ │ │ │
50│ │ │ │ │ │ │
51│ └─────────────────────┘ ▼ │ │ │
52│ Design Space │ │ │
53│ ┌─────────────────────┐ │ │ │ │
54│ Generate │ │ Pretuning │ │ │ │
55│ ┌───────────┤ Search Strategy │◄─────────────┘ │ │ │
56│ │ │ │ │ ├──┘
57│ │ └─────────────────────┘ ├──┘
58└────┼─────────────────────────────────────────────────────────┘
59 │
60 │
61┌────┼──────────────── Managed By Task Scheduler ─────────────────────┐
62│ │ ┌───────────┐ │
63│ │ Send to │ │ Send to │
64│ ▼ ┌─────────────►│ Builder ├──────────┐ │
65│ Measure Candidate │ Builder │ │ Runner │ │
66│ │ │ └───────────┘ │ │
67│ │ ┌────────────┴────────┐ │ │
68│ │ │ │ ┌───────────┐ │ │
69│ └────►│ Task Scheduler │ │ │ │ │
70│ │ │ │ Runner │◄─────────┘ │
71│ └─────────────────────┘ │ │ │
72│ ▲ └─────┬─────┘ │
73│ │ │ │
74│ └─── Runner Future ◄────┘ │
75└─────────────────────────────────────────────────────────────────────┘
76*/
77class SearchStrategyNode : public runtime::Object {
78 public:
79 /*! \brief Virtual destructor */
80 virtual ~SearchStrategyNode() = default;
81
82 /*!
83 * \brief Initialize the search strategy with tuning context.
84 * \param context The tuning context for initialization.
85 * \note This method is supposed to be called only once before every other method.
86 */
87 virtual void InitializeWithTuneContext(const TuneContext& context) = 0;
88
89 /*!
90 * \brief Pre-tuning for the search strategy.
91 * \param design_spaces The design spaces used during tuning process.
92 * \param database The database used during tuning process.
93 * \param cost_model The cost model used during tuning process.
94 * \note Pre-tuning is supposed to be called before the tuning process and after the
95 * initialization. Because the search strategy is stateful, we can always call pretuning
96 * and reset the search strategy.
97 */
98 virtual void PreTuning(const Array<tir::Schedule>& design_spaces,
99 const Optional<Database>& database,
100 const Optional<CostModel>& cost_model) = 0;
101
102 /*!
103 * \brief Post-tuning for the search strategy.
104 * \note Post-tuning is supposed to be called after the tuning process and before we reset the
105 * search strategy with another pre-tuning. Post-tuning can be empty.
106 */
107 virtual void PostTuning() = 0;
108
109 /*!
110 * \brief Generate measure candidates from design spaces for measurement.
111 * \return The measure candidates generated, nullptr if finished.
112 */
113 virtual Optional<Array<MeasureCandidate>> GenerateMeasureCandidates() = 0;
114
115 /*!
116 * \brief Update the search strategy with measurement results.
117 * \param measure_candidates The candidates to be measured.
118 * \param results The measurement results from the runner.
119 */
120 virtual void NotifyRunnerResults(const Array<MeasureCandidate>& measure_candidates,
121 const Array<RunnerResult>& results) = 0;
122
123 /*!
124 * \brief Clone the search strategy.
125 * \return The cloned search strategy.
126 */
127 virtual SearchStrategy Clone() const = 0;
128
129 static constexpr const char* _type_key = "meta_schedule.SearchStrategy";
130 TVM_DECLARE_BASE_OBJECT_INFO(SearchStrategyNode, Object);
131};
132
133/*!
134 * \brief Managed reference to SearchStrategyNode.
135 * \sa SearchStrategyNode
136 */
137class SearchStrategy : public runtime::ObjectRef {
138 public:
139 /*!
140 * \brief The function type of `InitializeWithTuneContext` method.
141 * \param context The tuning context for initialization.
142 */
143 using FInitializeWithTuneContext = runtime::TypedPackedFunc<void(const TuneContext&)>;
144 /*!
145 * \brief The function type of `PreTuning` method.
146 * \param design_spaces The design spaces for pre-tuning.
147 */
148 using FPreTuning = runtime::TypedPackedFunc<void(
149 const Array<tir::Schedule>&, const Optional<Database>&, const Optional<CostModel>&)>;
150 /*! \brief The function type of `PostTuning` method. */
151 using FPostTuning = runtime::TypedPackedFunc<void()>;
152 /*!
153 * \brief The function type of `GenerateMeasureCandidates` method.
154 * \return The measure candidates generated, nullptr if finished.
155 */
156 using FGenerateMeasureCandidates = runtime::TypedPackedFunc<Optional<Array<MeasureCandidate>>()>;
157 /*!
158 * \brief The function type of `NotifyRunnerResults` method.
159 * \param results The measurement results from the runner.
160 */
161 using FNotifyRunnerResults =
162 runtime::TypedPackedFunc<void(const Array<MeasureCandidate>&, const Array<RunnerResult>&)>;
163 /*!
164 * \brief The function type of `Clone` method.
165 * \return The cloned search strategy.
166 */
167 using FClone = runtime::TypedPackedFunc<SearchStrategy()>;
168 /*!
169 * \brief Create a search strategy with customized methods on the python-side.
170 * \param f_initialize_with_tune_context The packed function of `InitializeWithTuneContext`.
171 * \param f_pre_tuning The packed function of `PreTuning`.
172 * \param f_post_tuning The packed function of `PostTuning`.
173 * \param f_generate_measure_candidates The packed function of `GenerateMeasureCandidates`.
174 * \param f_notify_runner_results The packed function of `NotifyRunnerResults`.
175 * \param f_clone The packed function of `Clone`.
176 * \return The search strategy created.
177 */
178 TVM_DLL static SearchStrategy PySearchStrategy(
179 FInitializeWithTuneContext f_initialize_with_tune_context, //
180 FPreTuning f_pre_tuning, //
181 FPostTuning f_post_tuning, //
182 FGenerateMeasureCandidates f_generate_measure_candidates, //
183 FNotifyRunnerResults f_notify_runner_results, //
184 FClone f_clone);
185
186 /*!
187 * \brief Constructor of replay trace search strategy.
188 * \param num_trials_per_iter The number of trials per iteration, i.e., the batch size.
189 * \param max_trials_per_task The total number of trials for trace replaying.
190 * \param max_fail_count The max number of failures during trace replaying.
191 */
192 TVM_DLL static SearchStrategy ReplayTrace(int num_trials_per_iter, int max_trials_per_task,
193 int max_fail_count);
194
195 /*!
196 * \brief Constructor of replay func search strategy.
197 * \param num_trials_per_iter The number of trials per iteration, i.e., the batch size.
198 * \param max_trials_per_task The total number of trials for func replaying.
199 */
200 TVM_DLL static SearchStrategy ReplayFunc(int num_trials_per_iter, int max_trials_per_task);
201
202 /*!
203 * \brief Constructor of evolutionary search strategy.
204 * \param num_trials_per_iter The number of trials per iteration, i.e., the batch size.
205 * \param max_trials_per_task The total number of trials for evolutionary search.
206 * \param population_size The initial sample population.
207 * \param init_measured_ratio The ratio of measures samples in initial population.
208 * \param init_min_unmeasured The minimal size of unmeasured population in the initial sampling.
209 * \param genetic_num_iters The iterations to run the genetic algorithm.
210 * \param genetic_mutate_prob The probability of mutation.
211 * \param genetic_max_fail_count The maximum number to try evolving the given trace.
212 * \param eps_greedy The ratio to select samples in a greedy fashion via their predicted score.
213 */
214 TVM_DLL static SearchStrategy EvolutionarySearch(int num_trials_per_iter, //
215 int max_trials_per_task, //
216 int population_size, //
217 double init_measured_ratio, //
218 int init_min_unmeasured, //
219 int genetic_num_iters, //
220 double genetic_mutate_prob, //
221 int genetic_max_fail_count, //
222 double eps_greedy);
223
224 TVM_DEFINE_MUTABLE_OBJECT_REF_METHODS(SearchStrategy, ObjectRef, SearchStrategyNode);
225};

227/*! \brief The python side customizable class for measure candidate generation */
228class PySearchStrategyNode : public SearchStrategyNode {
229 public:
230 using FInitializeWithTuneContext = SearchStrategy::FInitializeWithTuneContext;
231 using FPreTuning = SearchStrategy::FPreTuning;
232 using FPostTuning = SearchStrategy::FPostTuning;
233 using FGenerateMeasureCandidates = SearchStrategy::FGenerateMeasureCandidates;
234 using FNotifyRunnerResults = SearchStrategy::FNotifyRunnerResults;
235 using FClone = SearchStrategy::FClone;
236
237 /*! \brief The packed function to the `InitializeWithTuneContext` method. */
238 FInitializeWithTuneContext f_initialize_with_tune_context;
239 /*! \brief The packed function to the `PreTuning` method. */
240 FPreTuning f_pre_tuning;
241 /*! \brief The packed function to the `PostTuning` method. */
242 FPostTuning f_post_tuning;
243 /*! \brief The packed function to the `GenerateMeasureCandidates` method. */
244 FGenerateMeasureCandidates f_generate_measure_candidates;
245 /*! \brief The packed function to the `NotifyRunnerResults` method. */
246 FNotifyRunnerResults f_notify_runner_results;
247 /*! \brief The packed function to the `Clone` method. */
248 FClone f_clone;
249
250 void VisitAttrs(tvm::AttrVisitor* v) {
251 // `f_initialize_with_tune_context` is not visited
252 // `f_pre_tuning` is not visited
253 // `f_post_tuning` is not visited
254 // `f_generate_measure_candidates` is not visited
255 // `f_notify_runner_results` is not visited
256 // `f_clone` is not visited
257 }
258
259 void InitializeWithTuneContext(const TuneContext& context) final;
260 void PreTuning(const Array<tir::Schedule>& design_spaces, const Optional<Database>& database,
261 const Optional<CostModel>& cost_model) final;
262 void PostTuning() final;
263 Optional<Array<MeasureCandidate>> GenerateMeasureCandidates() final;
264 void NotifyRunnerResults(const Array<MeasureCandidate>& measure_candidates,
265 const Array<RunnerResult>& results);
266 SearchStrategy Clone() const final;
267
268 static constexpr const char* _type_key = "meta_schedule.PySearchStrategy";
269 TVM_DECLARE_FINAL_OBJECT_INFO(PySearchStrategyNode, SearchStrategyNode);
270};
