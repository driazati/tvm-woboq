36/*! \brief A workload, i.e. an IRModule and its structural hash. */
37class WorkloadNode : public runtime::Object {
38 public:
39 /*! \brief The type of structural hash */
40 using THashCode = size_t;
41 /*! \brief The workload's IRModule. */
42 IRModule mod;
43 /*! \brief The workload's structural hash. */
44 THashCode shash;
45
46 void VisitAttrs(tvm::AttrVisitor* v) {
47 v->Visit("mod", &mod);
48 // `shash` is not visited because TVM FFI doesn't support uint64_t
49 }
50
51 static constexpr const char* _type_key = "meta_schedule.Workload";
52 TVM_DECLARE_FINAL_OBJECT_INFO(WorkloadNode, runtime::Object);
53
54 /*!
55 * \brief Export the workload to a JSON string.
56 * \return An array containing the structural hash and the base64 json string.
57 */
58 ObjectRef AsJSON() const;
59};
60
61/*!
62 * \brief Managed reference to WorkloadNode.
63 * \sa WorkloadNode
64 */
65class Workload : public runtime::ObjectRef {
66 public:
67 using THashCode = WorkloadNode::THashCode;
68 /*!
69 * \brief Constructor of Workload.
70 * \param mod The workload's IRModule.
71 */
72 TVM_DLL explicit Workload(IRModule mod);
73 /*!
74 * \brief Constructor of Workload.
75 * \param mod The workload's IRModule.
76 * \param shash The workload's structural hash.
77 */
78 TVM_DLL explicit Workload(IRModule mod, THashCode shash);
79 /*!
80 * \brief Create a workload from a json object.
81 * \param json_obj The json object.
82 * \return The created workload.
83 */
84 TVM_DLL static Workload FromJSON(const ObjectRef& json_obj);
85
86 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(Workload, runtime::ObjectRef, WorkloadNode);
87};
88
89/*! \brief The hash method for Workload */
90struct WorkloadHash {
91 size_t operator()(const Workload& a) const { return a->shash; }
92};
93
94/*! \brief The equality check for Workload */
95struct WorkloadEqual {
96 bool operator()(const Workload& a, const Workload& b) const {
97 return a->shash == b->shash && tvm::StructuralEqual()(a->mod, b->mod);
98 }
99};

104/*! \brief The class of tuning records. */
105class TuningRecordNode : public runtime::Object {
106 public:
107 /*! \brief The trace tuned. */
108 tir::Trace trace;
109 /*! \brief The workload. */
110 Workload workload{nullptr};
111 /*! \brief The profiling result in seconds. */
112 Optional<Array<FloatImm>> run_secs;
113 /*! \brief The target for tuning. */
114 Optional<Target> target;
115 /*! \brief The argument information. */
116 Optional<Array<ArgInfo>> args_info;
117
118 void VisitAttrs(tvm::AttrVisitor* v) {
119 v->Visit("trace", &trace);
120 v->Visit("workload", &workload);
121 v->Visit("run_secs", &run_secs);
122 v->Visit("target", &target);
123 v->Visit("args_info", &args_info);
124 }
125
126 static constexpr const char* _type_key = "meta_schedule.TuningRecord";
127 TVM_DECLARE_FINAL_OBJECT_INFO(TuningRecordNode, runtime::Object);
128
129 /*! \brief Construct the measure candidate given the initial IR module and trace
130 * stored in the tuning record. */
131 MeasureCandidate AsMeasureCandidate() const;
132 /*!
133 * \brief Export the tuning record to a JSON string.
134 * \return An array containing the trace, running secs, serialized target, and
135 * argument information.
136 */
137 ObjectRef AsJSON() const;
138};

167/* \brief The abstract interface of database. */
168class DatabaseNode : public runtime::Object {
169 public:
170 /*! \brief Default destructor */
171 virtual ~DatabaseNode() = default;
172 /*!
173 * \brief Check if the database has the given workload.
174 * \param mod The IRModule to be searched for.
175 * \return Whether the database has the given workload.
176 */
177 virtual bool HasWorkload(const IRModule& mod) = 0;
178 /*!
179 * \brief Look up or add workload to the database if missing.
180 * \param mod The IRModule to be searched for or added.
181 * \return The workload corresponding to the given IRModule.
182 */
183 virtual Workload CommitWorkload(const IRModule& mod) = 0;
184 /*!
185 * \brief Add a tuning record to the database.
186 * \param record The tuning record to be added.
187 */
188 virtual void CommitTuningRecord(const TuningRecord& record) = 0;
189 /*!
190 * \brief Get the top K tuning records of given workload from the database.
191 * \param workload The workload to be searched for.
192 * \param top_k The number of top records to be returned.
193 * \return An array of top K tuning records for the given workload.
194 */
195 virtual Array<TuningRecord> GetTopK(const Workload& workload, int top_k) = 0;
196 /*!
197 * \brief Get all tuning records from the database.
198 * \return An Array of all the tuning records in the database.
199 */
200 virtual Array<TuningRecord> GetAllTuningRecords() = 0;
201 /*!
202 * \brief Get the size of the database.
203 * \return The size of the database.
204 */
205 virtual int64_t Size() = 0;
206
207 static constexpr const char* _type_key = "meta_schedule.Database";
208 TVM_DECLARE_BASE_OBJECT_INFO(DatabaseNode, runtime::Object);
209};

211/*! \brief The database with customized methods on the python-side. */
212class PyDatabaseNode : public DatabaseNode {
213 public:
214 /*!
215 * \brief The function type of `HasWorkload` method.
216 * \param mod The IRModule to be searched for.
217 * \return Whether the database has the given workload.
218 */
219 using FHasWorkload = runtime::TypedPackedFunc<bool(const IRModule&)>;
220 /*!
221 * \brief The function type of `CommitWorkload` method.
222 * \param mod The IRModule to be searched for or added.
223 * \return The workload corresponding to the given IRModule.
224 */
225 using FCommitWorkload = runtime::TypedPackedFunc<Workload(const IRModule&)>;
226 /*!
227 * \brief The function type of `CommitTuningRecord` method.
228 * \param record The tuning record to be added.
229 */
230 using FCommitTuningRecord = runtime::TypedPackedFunc<void(const TuningRecord&)>;
231 /*!
232 * \brief The function type of `GetTopK` method.
233 * \param workload The workload to be searched for.
234 * \param top_k The number of top records to be returned.
235 * \return An array of top K tuning records for the given workload.
236 */
237 using FGetTopK = runtime::TypedPackedFunc<Array<TuningRecord>(const Workload&, int)>;
238 /*!
239 * \brief The function type of `GetAllTuningRecords` method.
240 * \return An Array of all the tuning records in the database.
241 */
242 using FGetAllTuningRecords = runtime::TypedPackedFunc<Array<TuningRecord>()>;
243 /*!
244 * \brief The function type of `Size` method.
245 * \return The size of the database.
246 */
247 using FSize = runtime::TypedPackedFunc<int64_t()>;
248
249 /*! \brief The packed function to the `HasWorkload` function. */
250 FHasWorkload f_has_workload;
251 /*! \brief The packed function to the `CommitWorkload` function. */
252 FCommitWorkload f_commit_workload;
253 /*! \brief The packed function to the `CommitTuningRecord` function. */
254 FCommitTuningRecord f_commit_tuning_record;
255 /*! \brief The packed function to the `GetTopK` function. */
256 FGetTopK f_get_top_k;
257 /*! \brief The packed function to the `GetAllTuningRecords` function. */
258 FGetAllTuningRecords f_get_all_tuning_records;
259 /*! \brief The packed function to the `Size` function. */
260 FSize f_size;
261
262 void VisitAttrs(tvm::AttrVisitor* v) {
263 // PackedFuncs are all not visited, because the reflection system doesn't take care of them,
264 // so it cannot be accessible on the python side. If there is such need from the future,
265 // we can then add corresponding accessor methods to help access on python.
266 // `f_has_workload` is not visited
267 // `f_commit_workload` is not visited
268 // `f_commit_tuning_record` is not visited
269 // `f_get_top_k` is not visited
270 // `f_get_all_tuning_records` is not visited
271 // `f_size` is not visited
272 }
273
274 bool HasWorkload(const IRModule& mod) final {
275 ICHECK(f_has_workload != nullptr) << "PyDatabase's HasWorkload method not implemented!";
276 return f_has_workload(mod);
277 }
278
279 Workload CommitWorkload(const IRModule& mod) final {
280 ICHECK(f_commit_workload != nullptr) << "PyDatabase's CommitWorkload method not implemented!";
281 return f_commit_workload(mod);
282 }
283
284 void CommitTuningRecord(const TuningRecord& record) final {
285 ICHECK(f_commit_tuning_record != nullptr)
286 << "PyDatabase's CommitTuningRecord method not implemented!";
287 f_commit_tuning_record(record);
288 }
289
290 Array<TuningRecord> GetTopK(const Workload& workload, int top_k) final {
291 ICHECK(f_get_top_k != nullptr) << "PyDatabase's GetTopK method not implemented!";
292 return f_get_top_k(workload, top_k);
293 }
294
295 Array<TuningRecord> GetAllTuningRecords() final {
296 ICHECK(f_get_all_tuning_records != nullptr)
297 << "PyDatabase's GetAllTuningRecords method not implemented!";
298 return f_get_all_tuning_records();
299 }
300
301 int64_t Size() final {
302 ICHECK(f_size != nullptr) << "PyDatabase's Size method not implemented!";
303 return f_size();
304 }
305
306 static constexpr const char* _type_key = "meta_schedule.PyDatabase";
307 TVM_DECLARE_FINAL_OBJECT_INFO(PyDatabaseNode, DatabaseNode);
308};

310/*!
311 * \brief Managed reference to DatabaseNode.
312 * \sa DatabaseNode
313 */
314class Database : public runtime::ObjectRef {
315 public:
316 /*!
317 * \brief Create a default database that uses JSON file for tuning records.
318 * \param path_workload The path to the workload table.
319 * \param path_tuning_record The path to the database table.
320 * \param allow_missing Whether to create new file when the given path is not found.
321 */
322 TVM_DLL static Database JSONDatabase(String path_workload, String path_tuning_record,
323 bool allow_missing);
324 /*!
325 * \brief Create a database with customized methods on the python-side.
326 * \param f_has_workload The packed function of `HasWorkload`.
327 * \param f_commit_workload The packed function of `CommitWorkload`.
328 * \param f_commit_tuning_record The packed function of `CommitTuningRecord`.
329 * \param f_get_top_k The packed function of `GetTopK`.
330 * \param f_get_all_tuning_records The packed function of `GetAllTuningRecords`.
331 * \param f_size The packed function of `Size`.
332 * \return The created database.
333 */
334 TVM_DLL static Database PyDatabase(PyDatabaseNode::FHasWorkload f_has_workload,
335 PyDatabaseNode::FCommitWorkload f_commit_workload,
336 PyDatabaseNode::FCommitTuningRecord f_commit_tuning_record,
337 PyDatabaseNode::FGetTopK f_get_top_k,
338 PyDatabaseNode::FGetAllTuningRecords f_get_all_tuning_records,
339 PyDatabaseNode::FSize f_size);
340 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(Database, runtime::ObjectRef, DatabaseNode);
341};
