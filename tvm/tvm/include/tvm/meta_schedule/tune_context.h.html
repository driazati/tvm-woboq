48/*! \brief The auto tuning context. */
49class TuneContextNode : public runtime::Object {
50 public:
51 /*! \brief The workload to be tuned. */
52 Optional<IRModule> mod;
53 /*! \brief The target to be tuned for. */
54 Optional<Target> target;
55 /*! \brief The design space generator. */
56 Optional<SpaceGenerator> space_generator;
57 /*! \brief The search strategy. */
58 Optional<SearchStrategy> search_strategy;
59 /*! \brief The schedule rules. */
60 Array<ScheduleRule> sch_rules;
61 /*! \brief The postprocessors. */
62 Array<Postproc> postprocs;
63 /*! \brief The probability of using certain mutator. */
64 Map<Mutator, FloatImm> mutator_probs;
65 /*! \brief The name of the tuning task. */
66 Optional<String> task_name;
67 /*! \brief The tuning task's logging function. t*/
68 PackedFunc logging_func;
69 /*! \brief The random state. */
70 support::LinearCongruentialEngine::TRandState rand_state;
71 /*! \brief The number of threads to be used. */
72 int num_threads;
73
74 /*! \brief Whether the tuning task has been stopped or finished. */
75 bool is_terminated; // TODO(@junrushao1994): move to TaskScheduler
76 /*! \brief The measure candidates. */
77 Optional<Array<MeasureCandidate>> measure_candidates;
78 /*! \brief The building results. */
79 Optional<Array<BuilderResult>> builder_results;
80 /*! \brief Packed functions to fetch the runner results asynchronously. */
81 Optional<Array<RunnerFuture>> runner_futures;
82
83 void VisitAttrs(tvm::AttrVisitor* v) {
84 v->Visit("mod", &mod);
85 v->Visit("target", &target);
86 v->Visit("space_generator", &space_generator);
87 v->Visit("search_strategy", &search_strategy);
88 v->Visit("sch_rules", &sch_rules);
89 v->Visit("postprocs", &postprocs);
90 v->Visit("mutator_probs", &mutator_probs);
91 v->Visit("task_name", &task_name);
92 // `logging_func` is not visited
93 v->Visit("rand_state", &rand_state);
94 v->Visit("num_threads", &num_threads);
95 v->Visit("is_terminated", &is_terminated);
96 v->Visit("measure_candidates", &measure_candidates);
97 v->Visit("builder_results", &builder_results);
98 v->Visit("runner_futures", &runner_futures);
99 }
100
101 /*! \brief Initialize members that needs initialization with tune context. */
102 void Initialize();
103 /*!
104 * \brief Clone the tune context.
105 * \return The cloned tune context.
106 */
107 TuneContext Clone() const;
108 /*! \brief Set the measure candidates from the SearchStrategy */
109 void _SetMeasureCandidates(const Array<MeasureCandidate>& candidates);
110 /*!
111 * \brief Send the measure candidates to builder.
112 * \param builder The builder to send the candidates to.
113 */
114 void _SendToBuilder(const Builder& builder);
115 /*!
116 * \brief Send the built measure candidates to runner.
117 * \param runner The runner to send the candidates to.
118 */
119 void _SendToRunner(const Runner& runner);
120 /*!
121 * \brief Join the running tasks.
122 * \returns The results from the runner
123 */
124 Array<RunnerResult> _Join();
125 /*! \brief Set `measure_candidates`, `builder_results` and `runner_futures` to null. */
126 void _ClearMeasureState();
127 static constexpr const char* _type_key = "meta_schedule.TuneContext";
128 TVM_DECLARE_FINAL_OBJECT_INFO(TuneContextNode, Object);
129};
130
131/*!
132 * \brief Managed reference to TuneContextNode.
133 * \sa TuneContextNode
134 */
135class TuneContext : public runtime::ObjectRef {
136 public:
137 /*!
138 * \brief Constructor.
139 * \param mod The workload to be tuned.
140 * \param target The target to be tuned for.
141 * \param space_generator The design space generator.
142 * \param search_strategy The search strategy.
143 * \param sch_rules The schedule rules.
144 * \param postprocs The postprocessors.
145 * \param mutator_probs The probability of using certain mutator.
146 * \param task_name The name of the tuning task.
147 * \param logging_func The tuning task's logging function.
148 * \param rand_state The random state.
149 * \param num_threads The number of threads to be used.
150 */
151 TVM_DLL explicit TuneContext(Optional<IRModule> mod, //
152 Optional<Target> target, //
153 Optional<SpaceGenerator> space_generator, //
154 Optional<SearchStrategy> search_strategy, //
155 Optional<Array<ScheduleRule>> sch_rules, //
156 Optional<Array<Postproc>> postprocs, //
157 Optional<Map<Mutator, FloatImm>> mutator_probs, //
158 Optional<String> task_name, //
159 PackedFunc logging_func, //
160 support::LinearCongruentialEngine::TRandState rand_state, //
161 int num_threads);
162 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(TuneContext, ObjectRef, TuneContextNode);
163};
