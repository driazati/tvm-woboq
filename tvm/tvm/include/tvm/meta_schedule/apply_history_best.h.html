33namespace tvm {
34namespace meta_schedule {
35
36/*!
37 * \brief An integration context that allows application of historically best records from a
38 * database
39 */
40class ApplyHistoryBestNode : public runtime::Object {
41 public:
42 /*! \brief A callback function that filters TE compute */
43 using FTEFilterFunc = runtime::TypedPackedFunc<Optional<tir::PrimFunc>(
44 const Array<te::Tensor, void>&, const Array<runtime::NDArray>&)>;
45 /*! \brief A callback function that takes a tuning record and does something with it */
46 using FTakeTuningRecord = runtime::TypedPackedFunc<void(const TuningRecord&)>;
47 using FDirectDispatch = runtime::TypedPackedFunc<Optional<IRModule>(const IRModule&)>;
48
49 /*! \brief The database to be queried from */
50 Database database{nullptr};
51 /*! \brief The filtering function for TE computation */
52 FTEFilterFunc te_filter_func{nullptr};
53 /*! \brief The logging function to be used */
54 PackedFunc logging_func;
55
56 void VisitAttrs(tvm::AttrVisitor* v) {
57 v->Visit("database", &database);
58 // `te_filter_func` is not visited
59 // `logging_func` is not visited
60 }
61 /*!
62 * \brief Query the best entry from the database
63 * \param task_name The name of the task to be queried
64 * \param mod The module to be queried
65 * \param target The target to be queried
66 * \param dispatched The IRs after dispatch
67 * \param f_take_tuning_record A callback function that takes a tuning record and does something
68 * with it.
69 * \param f_direct_dispatch A function that directly dispatches an IRModule to the given workload
70 * as result if available, skipping the database query.
71 */
72 Optional<IRModule> Query(runtime::String task_name, IRModule mod, Target target,
73 Optional<Array<IRModule>> dispatched,
74 FTakeTuningRecord f_take_tuning_record,
75 FDirectDispatch f_direct_dispatch = nullptr);
76
77 static constexpr const char* _type_key = "meta_schedule.ApplyHistoryBest";
78 TVM_DECLARE_FINAL_OBJECT_INFO(ApplyHistoryBestNode, runtime::Object);
79};
80
81/*!
82 * \brief Managed reference to ApplyHistoryBestNode
83 * \sa ApplyHistoryBestNode
84 */
85class ApplyHistoryBest : public runtime::ObjectRef {
86 public:
87 /*!
88 * \brief Constructor
89 * \param database The database to be queried from
90 * \param te_filter_func The filtering function for TE computation
91 * \param logging_func The logging function to use
92 */
93 explicit ApplyHistoryBest(Database database, ApplyHistoryBestNode::FTEFilterFunc te_filter_func,
94 PackedFunc logging_func);
95 /*!
96 * \brief The current ApplyHistoryBest in the context
97 * \return The ApplyHistoryBest in the current scope.
98 */
99 static Optional<ApplyHistoryBest> Current();
100
101 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(ApplyHistoryBest, runtime::ObjectRef,
102 ApplyHistoryBestNode);
103
104 protected:
105 friend class ApplyHistoryBestInternal;
106 /*! \brief Entering the scope of the context manager */
107 void EnterWithScope();
108 /*! \brief Exiting the scope of the context manager */
109 void ExitWithScope();
110};
