38/*!
39 * \brief The abstract interface of task schedulers.
40 * \note The relationship between SpaceGenerator and other classes are as follows:
41 ┌──────────────────────────────────────────────────────────────┐
42 ┌──┴───────────────────────────────────────────────────────────┐ │
43┌──┴────────────────── Tune Context ───────────────────────────┐ │ │
44│ ┌─────────────────────┐ │ │ │
45│ │ │ Generate │ │ │
46│ │ Space Generator ├──────────────┐ │ │ │
47│ │ │ │ │ │ │
48│ └─────────────────────┘ ▼ │ │ │
49│ Design Space │ │ │
50│ ┌─────────────────────┐ │ │ │ │
51│ Generate │ │ Pretuning │ │ │ │
52│ ┌───────────┤ Search Strategy │◄─────────────┘ │ │ │
53│ │ │ │ │ ├──┘
54│ │ └─────────────────────┘ ├──┘
55└────┼─────────────────────────────────────────────────────────┘
56 │
57 │
58┌────┼──────────────── Managed By Task Scheduler ─────────────────────┐
59│ │ ┌───────────┐ │
60│ │ Send to │ │ Send to │
61│ ▼ ┌─────────────►│ Builder ├──────────┐ │
62│ Measure Candidate │ Builder │ │ Runner │ │
63│ │ │ └───────────┘ │ │
64│ │ ┌────────────┴────────┐ │ │
65│ │ │ │ ┌───────────┐ │ │
66│ └────►│ Task Scheduler │ │ │ │ │
67│ │ │ │ Runner │◄─────────┘ │
68│ └─────────────────────┘ │ │ │
69│ ▲ └─────┬─────┘ │
70│ │ │ │
71│ └─── Runner Future ◄────┘ │
72└─────────────────────────────────────────────────────────────────────┘
73*/
74class TaskSchedulerNode : public runtime::Object {
75 public:
76 /*! \brief The tasks to be tuned */
77 Array<TuneContext> tasks;
78 /*! \brief The builder of the scheduler. */
79 Builder builder{nullptr};
80 /*! \brief The runner of the scheduler. */
81 Runner runner{nullptr};
82 /*! \brief The database of the scheduler. */
83 Optional<Database> database;
84 /*! \brief The cost model of the scheduler. */
85 Optional<CostModel> cost_model;
86 /*! \brief The list of measure callbacks of the scheduler. */
87 Array<MeasureCallback> measure_callbacks;
88 /*! \brief The maximum number of trials allowed. */
89 int max_trials;
90 /*! \brief The number of trials already conducted. */
91 int num_trials_already;
92 /*! \brief The tuning task's logging function. t*/
93 PackedFunc logging_func;
94
95 /*! \brief The default destructor. */
96 virtual ~TaskSchedulerNode() = default;
97
98 void VisitAttrs(tvm::AttrVisitor* v) {
99 v->Visit("tasks", &tasks);
100 v->Visit("builder", &builder);
101 v->Visit("runner", &runner);
102 v->Visit("database", &database);
103 v->Visit("cost_model", &cost_model);
104 v->Visit("measure_callbacks", &measure_callbacks);
105 v->Visit("max_trials", &max_trials);
106 v->Visit("num_trials_already", &num_trials_already);
107 // `logging_func` is not visited
108 }
109
110 /*! \brief Auto-tuning. */
111 virtual void Tune();
112
113 /*!
114 * \brief Initialize modules of the given task.
115 * \param task_id The task id to be initialized.
116 */
117 virtual void InitializeTask(int task_id);
118
119 /*!
120 * \brief Touch the task and update its status
121 * \param task_id The task id to be checked.
122 */
123 virtual void TouchTask(int task_id);
124
125 /*!
126 * \brief Wait until the task is finished.
127 * \param task_id The task id to be joined.
128 */
129 virtual Array<RunnerResult> JoinRunningTask(int task_id);
130
131 /*!
132 * \brief Fetch the next task id.
133 * \return The next task id.
134 */
135 virtual int NextTaskId() = 0;
136
137 static constexpr const char* _type_key = "meta_schedule.TaskScheduler";
138 TVM_DECLARE_BASE_OBJECT_INFO(TaskSchedulerNode, Object);
139};

143/*! \brief The task scheduler with customized methods on the python-side. */
144class PyTaskSchedulerNode : public TaskSchedulerNode {
145 public:
146 /*! \brief The function type of `Tune` method. */
147 using FTune = runtime::TypedPackedFunc<void()>;
148
149 /*! \brief The function type of `InitializeTask` method. */
150 using FInitializeTask = runtime::TypedPackedFunc<void(int)>;
151
152 /*!
153 * \brief The function type of `TouchTask` method.
154 * \param task_id The task id to be checked.
155 * \return Whether the task is running.
156 */
157 using FTouchTask = runtime::TypedPackedFunc<void(int)>;
158
159 /*!
160 * \brief The function type of `JoinRunningTask` method.
161 * \param task_id The task id to be joined.
162 */
163 using FJoinRunningTask = runtime::TypedPackedFunc<Array<RunnerResult>(int)>;
164
165 /*!
166 * \brief The function type of `NextTaskId` method.
167 * \return The next task id.
168 */
169 using FNextTaskId = runtime::TypedPackedFunc<int()>;
170
171 /*! \brief The packed function to the `Tune` function. */
172 FTune f_tune;
173 /*! \brief The packed function to the `InitializeTask` function. */
174 FInitializeTask f_initialize_task;
175 /*! \brief The packed function to the `TouchTask` function. */
176 FTouchTask f_touch_task;
177 /*! \brief The packed function to the `JoinRunningTask` function. */
178 FJoinRunningTask f_join_running_task;
179 /*! \brief The packed function to the `NextTaskId` function. */
180 FNextTaskId f_next_task_id;
181
182 void VisitAttrs(tvm::AttrVisitor* v) {
183 // `f_tune` is not visited
184 // `f_initialize_task` is not visited
185 // `f_touch_task` is not visited
186 // `f_join_running_task` is not visited
187 // `f_next_task_id` is not visited
188 }
189
190 void Tune() final;
191 void InitializeTask(int task_id) final;
192 void TouchTask(int task_id) final;
193 Array<RunnerResult> JoinRunningTask(int task_id) final;
194 int NextTaskId() final;
195
196 static constexpr const char* _type_key = "meta_schedule.PyTaskScheduler";
197 TVM_DECLARE_FINAL_OBJECT_INFO(PyTaskSchedulerNode, TaskSchedulerNode);
198};

200/*!
201 * \brief Managed reference to TaskSchedulerNode.
202 * \sa TaskSchedulerNode
203 */
204class TaskScheduler : public runtime::ObjectRef {
205 public:
206 /*!
207 * \brief Create a task scheduler that fetches tasks in a round-robin fashion.
208 * \param tasks The tasks to be tuned.
209 * \param builder The builder of the scheduler.
210 * \param runner The runner of the scheduler.
211 * \param database The database of the scheduler.
212 * \param max_trials The maximum number of trials.
213 * \param cost_model The cost model of the scheduler.
214 * \param measure_callbacks The measure callbacks of the scheduler.
215 * \param logging_func The tuning task's logging function.
216 * \return The task scheduler created.
217 */
218 TVM_DLL static TaskScheduler RoundRobin(Array<TuneContext> tasks, //
219 Builder builder, //
220 Runner runner, //
221 Optional<Database> database, //
222 Optional<CostModel> cost_model, //
223 Optional<Array<MeasureCallback>> measure_callbacks, //
224 int max_trials, //
225 PackedFunc logging_func);
226 /*!
227 * \brief Create a task scheduler that fetches tasks in a gradient based fashion.
228 * \param tasks The tasks to be tuned.
229 * \param task_weights The weights of each task.
230 * \param builder The builder of the scheduler.
231 * \param runner The runner of the scheduler.
232 * \param database The database of the scheduler.
233 * \param max_trials The maximum number of trials.
234 * \param cost_model The cost model of the scheduler.
235 * \param measure_callbacks The measure callbacks of the scheduler.
236 * \param logging_func The tuning task's logging function.
237 * \param alpha The parameter alpha to control gradient computation.
238 * \param window_size The parameter to control backward window size.
239 * \param seed The random seed.
240 * \return The task scheduler created.
241 */
242 TVM_DLL static TaskScheduler GradientBased(Array<TuneContext> tasks,
243 Array<FloatImm> task_weights, //
244 Builder builder, //
245 Runner runner, //
246 Optional<Database> database, //
247 Optional<CostModel> cost_model, //
248 Optional<Array<MeasureCallback>> measure_callbacks, //
249 int max_trials, //
250 PackedFunc logging_func, //
251 double alpha, //
252 int window_size, //
253 support::LinearCongruentialEngine::TRandState seed);
254 /*!
255 * \brief Create a task scheduler with customized methods on the python-side.
256 * \param tasks The tasks to be tuned.
257 * \param builder The builder of the scheduler.
258 * \param runner The runner of the scheduler.
259 * \param database The database of the scheduler.
260 * \param max_trials The maximum number of trials.
261 * \param cost_model The cost model of the scheduler.
262 * \param measure_callbacks The measure callbacks of the scheduler.
263 * \param logging_func The tuning task's logging function.
264 * \param f_tune The packed function of `Tune`.
265 * \param f_initialize_task The packed function of `InitializeTask`.
266 * \param f_touch_task The packed function of `TouchTask`.
267 * \param f_join_running_task The packed function of `JoinRunningTask`.
268 * \param f_next_task_id The packed function of `NextTaskId`.
269 * \return The task scheduler created.
270 */
271 TVM_DLL static TaskScheduler PyTaskScheduler(
272 Array<TuneContext> tasks, //
273 Builder builder, //
274 Runner runner, //
275 Optional<Database> database, //
276 Optional<CostModel> cost_model, //
277 Optional<Array<MeasureCallback>> measure_callbacks, //
278 int max_trials, //
279 PackedFunc logging_func, //
280 PyTaskSchedulerNode::FTune f_tune, //
281 PyTaskSchedulerNode::FInitializeTask f_initialize_task, //
282 PyTaskSchedulerNode::FTouchTask f_touch_task, //
283 PyTaskSchedulerNode::FJoinRunningTask f_join_running_task, //
284 PyTaskSchedulerNode::FNextTaskId f_next_task_id);
285 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(TaskScheduler, ObjectRef, TaskSchedulerNode);
286};
