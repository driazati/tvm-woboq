20/*!
21 * \file tvm/relay/attrs/annotation.h
22 * \brief Attribute for annotation operators.
23 */
24#ifndef TVM_RELAY_ATTRS_ANNOTATION_H_
25#define TVM_RELAY_ATTRS_ANNOTATION_H_
26
27#include <tvm/ir/attrs.h>
28
29#include <string>
30
31namespace tvm {
32namespace relay {
33
34/*!
35 * \brief Annotate an expression to be cast into specific data type.
36 */
37struct CastHintAttrs : public tvm::AttrsNode<CastHintAttrs> {
38 DataType dtype;
39
40 TVM_DECLARE_ATTRS(CastHintAttrs, "relay.attrs.CastHintAttrs") {
41 TVM_ATTR_FIELD(dtype).describe("The data type denoted to be cast.");
42 }
43};
44
45/*!
46 * \brief Options for the operators used to annotate a compiler.
47 */
48struct CompilerAttrs : public tvm::AttrsNode<CompilerAttrs> {
49 /*! \brief A 3rd party compiler for code generation. */
50 std::string compiler;
51
52 TVM_DECLARE_ATTRS(CompilerAttrs, "relay.attrs.CompilerAttrs") {
53 TVM_ATTR_FIELD(compiler).describe("A 3rd party compiler used for code generation.");
54 }
55};
56
57} // namespace relay
58} // namespace tvm
59#endif // TVM_RELAY_ATTRS_ANNOTATION_H_
