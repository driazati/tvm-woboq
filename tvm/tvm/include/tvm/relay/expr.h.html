62/*!
63 * \brief Constant tensor, backed by an NDArray on the cpu(0) device.
64 *
65 * \note Scalar constants are represented by rank-0 const tensor.
66 * Constant folding are handled uniformly via Tensor types.
67 */
68class Constant;
69/*!
70 * \brief Constant tensor type.
71 */
72class ConstantNode : public ExprNode {
73 public:
74 /*! \brief The data of the tensor */
75 runtime::NDArray data;
76
77 /*! \return The corresponding tensor type of the data */
78 TensorType tensor_type() const;
79
80 /*! \return Whether it is scalar(rank-0 tensor) */
81 bool is_scalar() const { return data->ndim == 0; }
82
83 void VisitAttrs(tvm::AttrVisitor* v) {
84 v->Visit("data", &data);
85 v->Visit("virtual_device_", &virtual_device_);
86 v->Visit("span", &span);
87 v->Visit("_checked_type_", &checked_type_);
88 }
89
90 bool SEqualReduce(const ConstantNode* other, SEqualReducer equal) const {
91 return equal(data, other->data);
92 }
93
94 void SHashReduce(SHashReducer hash_reduce) const { hash_reduce(data); }
95
96 static constexpr const char* _type_key = "relay.Constant";

121/*! \brief Tuple of multiple Exprs */
122class Tuple;
123/*! \brief Tuple container */
124class TupleNode : public ExprNode {
125 public:
126 /*! \brief the fields of the tuple */
127 tvm::Array<relay::Expr> fields;
128
129 void VisitAttrs(tvm::AttrVisitor* v) {
130 v->Visit("fields", &fields);
131 v->Visit("virtual_device_", &virtual_device_);
132 v->Visit("span", &span);
133 v->Visit("_checked_type_", &checked_type_);
134 }
135
136 bool SEqualReduce(const TupleNode* other, SEqualReducer equal) const {
137 // specially handle empty tuple as a constant is not a graph node.
138 if (fields.size() == other->fields.size() && fields.size() == 0) {
139 return true;
140 } else {
141 equal->MarkGraphNode();
142 return equal(fields, other->fields);
143 }
144 }
145
146 void SHashReduce(SHashReducer hash_reduce) const {
147 if (fields.size() != 0) {
148 hash_reduce->MarkGraphNode();
149 hash_reduce(fields);
150 }
151 }
152
153 static constexpr const char* _type_key = "relay.Tuple";
154 TVM_DECLARE_FINAL_OBJECT_INFO(TupleNode, ExprNode);

179/*!
180 * \brief Local variables used in the let expression.
181 *
182 * Its semantics are similar to tvm.Var node used in TVM's low level
183 * tensor expression language.
184 *
185 * \note Each Var is bind only once and is immutable.
186 */
187class Var;
188/*! \brief Container for Var */
189class VarNode : public ExprNode {
190 public:
191 /*!
192 * \brief The unique identifier of the Var.
193 *
194 * vid will be preserved for the same Var during type inference
195 * and other rewritings, while the VarNode might be recreated
196 * to attach additional information.
197 * This property can be used to keep track of parameter Var
198 * information across passes.
199 */
200 Id vid;
201 /*!
202 * \brief type annotaion of the variable.
203 * This field records user provided type annotation of the Var.
204 * This field is optional and can be None.
205 */
206 Type type_annotation;
207
208 /*! \return The name hint of the variable */
209 const String& name_hint() const { return vid->name_hint; }
210
211 void VisitAttrs(tvm::AttrVisitor* v) {
212 v->Visit("vid", &vid);
213 v->Visit("type_annotation", &type_annotation);
214 v->Visit("virtual_device_", &virtual_device_);
215 v->Visit("span", &span);
216 v->Visit("_checked_type_", &checked_type_);
217 }
218
219 bool SEqualReduce(const VarNode* other, SEqualReducer equal) const {
220 equal->MarkGraphNode();
221 return equal(type_annotation, other->type_annotation) && equal(vid, other->vid) &&
222 equal(virtual_device_, other->virtual_device_);
223 }
224
225 void SHashReduce(SHashReducer hash_reduce) const {
226 hash_reduce->MarkGraphNode();
227 hash_reduce(type_annotation);
228 hash_reduce(vid);
229 }
230
231 static constexpr const char* _type_key = "relay.Var";
232 TVM_DECLARE_FINAL_OBJECT_INFO(VarNode, ExprNode);

277/*!
278 * \brief Call corresponds to operator invocation.
279 * Corresponds to the operator in computational graph terminology.
280 */
281class Call;
282/*! \brief Call container. */
283class CallNode : public ExprNode {
284 protected:
285 // CallNode uses own deleter to indirectly call non-recursive destructor
286 Object::FDeleter saved_deleter_;
287 static void Deleter_(Object* ptr);
288
289 public:
290 /*!
291 * \brief The operator(function) being invoked
292 *
293 * - It can be tvm::Op which corresponds to the primitive operators.
294 * - It can also be user defined functions (Function, GlobalVar, Var).
295 */
296 Expr op;
297
298 /*! \brief The arguments(inputs) of the call */
299 tvm::Array<relay::Expr> args;
300
301 /*! \brief The additional attributes */
302 Attrs attrs;
303
304 /*!
305 * \brief The type arguments passed to polymorphic(template) function.
306 *
307 * This is the advance feature that is only used when the function is
308 * polymorphic. It is safe to be ignored in most cases. For example, in the
309 * following code, the type_args of addone call is [int].
310 *
311 * \code
312 *
313 * template<typename T>
314 * T addone(T a) { return a + 1; }
315 *
316 * void main() {
317 * int x = addone<int>(10);
318 * }
319 *
320 * \endcode
321 */
322 tvm::Array<Type> type_args;
323
324 void VisitAttrs(tvm::AttrVisitor* v) {
325 v->Visit("op", &op);
326 v->Visit("args", &args);
327 v->Visit("attrs", &attrs);
328 v->Visit("type_args", &type_args);
329 v->Visit("virtual_device_", &virtual_device_);
330 v->Visit("span", &span);
331 v->Visit("_checked_type_", &checked_type_);
332 }
333
334 bool SEqualReduce(const CallNode* other, SEqualReducer equal) const {
335 // skip type_args check for primitive ops.
336 equal->MarkGraphNode();
337 return equal(op, other->op) && equal(args, other->args) && equal(attrs, other->attrs) &&
338 (IsPrimitiveOp(op) || equal(type_args, other->type_args));
339 }
340
341 void SHashReduce(SHashReducer hash_reduce) const {
342 hash_reduce->MarkGraphNode();
343 hash_reduce(op);
344 hash_reduce(args);
345 hash_reduce(attrs);
346 if (!IsPrimitiveOp(op)) {
347 hash_reduce(type_args);
348 }
349 }
350
351 static constexpr const char* _type_key = "relay.Call";
352 TVM_DECLARE_FINAL_OBJECT_INFO(CallNode, ExprNode);
353 template <typename>
354 friend class runtime::ObjAllocatorBase;
355 friend class Call;
356};

392/*!
393 * \brief Let binding that binds a local var and optionally a type annotation.
394 *
395 * \note Let is useful to transform the program to be A-normal form.
396 * where each of the expression corresponds to a let binding.
397 *
398 * For developers who are familar with the computational graph.
399 * Each of the let can be viewed as a operator node in the computational graph.
400 * Traversing the list of let bindings is similar to running
401 * PostDFS-order(topo-order) traversal on the computational graph.
402 */
403class Let;
404/*! \brief A binding of a sub-network. */
405class LetNode : public ExprNode {
406 protected:
407 // LetNode uses own deleter to indirectly call non-recursive destructor
408 Object::FDeleter saved_deleter_;
409 static void Deleter_(Object* ptr);
410
411 public:
412 /*! \brief The variable we bind to */
413 Var var;
414 /*! \brief The value we bind var to */
415 Expr value;
416 /*! \brief The body of the let binding */
417 Expr body;
418
419 void VisitAttrs(tvm::AttrVisitor* v) {
420 v->Visit("var", &var);
421 v->Visit("value", &value);
422 v->Visit("body", &body);
423 v->Visit("virtual_device_", &virtual_device_);
424 v->Visit("span", &span);
425 v->Visit("_checked_type_", &checked_type_);
426 }
427
428 bool SEqualReduce(const LetNode* other, SEqualReducer equal) const {
429 equal->MarkGraphNode();
430 return equal.DefEqual(var, other->var) && equal(value, other->value) &&
431 equal(body, other->body);
432 }
433
434 void SHashReduce(SHashReducer hash_reduce) const {
435 hash_reduce->MarkGraphNode();
436 hash_reduce.DefHash(var);
437 hash_reduce(value);
438 hash_reduce(body);
439 }
440
441 static constexpr const char* _type_key = "relay.Let";
442 TVM_DECLARE_FINAL_OBJECT_INFO(LetNode, ExprNode);
443 template <typename>
444 friend class runtime::ObjAllocatorBase;
445 friend class Let;
446};

479/*!
480 * \brief Condition expression
481 *
482 * Unlike traditional statement `if`s, the if evalutes
483 * to the result of the branch taken.
484 *
485 * let x = if (true) { 1 } else { 0 }; // x is 1
486 * let y = if (false) { 1 } else { 0 }; // y is 0
487 *
488 * \note This is similar to C's ternary operator.
489 */
490class If;
491/*! \brief container of If */
492class IfNode : public ExprNode {
493 public:
494 /*! \brief The condition */
495 Expr cond;
496 /*! \brief The expression evaluated when condition is true. */
497 Expr true_branch;
498 /*! \brief The expression evaluated when condition is false */
499 Expr false_branch;
500
501 void VisitAttrs(tvm::AttrVisitor* v) {
502 v->Visit("cond", &cond);
503 v->Visit("true_branch", &true_branch);
504 v->Visit("false_branch", &false_branch);
505 v->Visit("virtual_device_", &virtual_device_);
506 v->Visit("span", &span);
507 v->Visit("_checked_type_", &checked_type_);
508 }
509
510 bool SEqualReduce(const IfNode* other, SEqualReducer equal) const {
511 equal->MarkGraphNode();
512 return equal(cond, other->cond) && equal(true_branch, other->true_branch) &&
513 equal(false_branch, other->false_branch);
514 }
515
516 void SHashReduce(SHashReducer hash_reduce) const {
517 hash_reduce->MarkGraphNode();
518 hash_reduce(cond);
519 hash_reduce(true_branch);
520 hash_reduce(false_branch);
521 }
522
523 static constexpr const char* _type_key = "relay.If";
524 TVM_DECLARE_FINAL_OBJECT_INFO(IfNode, ExprNode);
525};

716 void VisitAttrs(tvm::AttrVisitor* v) {
717 v->Visit("ref", &ref);
718 v->Visit("value", &value);
