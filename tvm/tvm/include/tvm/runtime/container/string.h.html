78/*! \brief An object representing string. It's POD type. */
79class StringObj : public Object {
80 public:
81 /*! \brief The pointer to string data. */
82 const char* data;
83
84 /*! \brief The length of the string object. */
85 uint64_t size;
86
87 static constexpr const uint32_t _type_index = TypeIndex::kRuntimeString;
88 static constexpr const char* _type_key = "runtime.String";
89 TVM_DECLARE_FINAL_OBJECT_INFO(StringObj, Object);
90
91 private:
92 /*! \brief String object which is moved from std::string container. */
93 class FromStd;
94
95 friend class String;
96};

98/*!
99 * \brief Reference to string objects.
100 *
101 * \code
102 *
103 * // Example to create runtime String reference object from std::string
104 * std::string s = "hello world";
105 *
106 * // You can create the reference from existing std::string
107 * String ref{std::move(s)};
108 *
109 * // You can rebind the reference to another string.
110 * ref = std::string{"hello world2"};
111 *
112 * // You can use the reference as hash map key
113 * std::unordered_map<String, int32_t> m;
114 * m[ref] = 1;
115 *
116 * // You can compare the reference object with other string objects
117 * assert(ref == "hello world", true);
118 *
119 * // You can convert the reference to std::string again
120 * string s2 = (string)ref;
121 *
122 * \endcode
123 */
124class String : public ObjectRef {
125 public:
126 /*!
127 * \brief Construct an empty string.
128 */
129 String() : String(std::string()) {}
130 /*!
131 * \brief Construct a new String object
132 *
133 * \param other The moved/copied std::string object
134 *
135 * \note If user passes const reference, it will trigger copy. If it's rvalue,
136 * it will be moved into other.
137 */
138 String(std::string other); // NOLINT(*)
139
140 /*!
141 * \brief Construct a new String object
142 *
143 * \param other a char array.
144 */
145 String(const char* other) // NOLINT(*)
146 : String(std::string(other)) {}
147
148 /*!
149 * \brief Construct a new null object
150 */
151 String(std::nullptr_t) // NOLINT(*)
152 : ObjectRef(nullptr) {}
153
154 /*!
155 * \brief Change the value the reference object points to.
156 *
157 * \param other The value for the new String
158 *
159 */
160 inline String& operator=(std::string other);
161
162 /*!
163 * \brief Change the value the reference object points to.
164 *
165 * \param other The value for the new String
166 */
167 inline String& operator=(const char* other);
168
169 /*!
170 * \brief Compares this String object to other
171 *
172 * \param other The String to compare with.
173 *
174 * \return zero if both char sequences compare equal. negative if this appear
175 * before other, positive otherwise.
176 */
177 int compare(const String& other) const {
178 return memncmp(data(), other.data(), size(), other.size());
179 }
180
181 /*!
182 * \brief Compares this String object to other
183 *
184 * \param other The string to compare with.
185 *
186 * \return zero if both char sequences compare equal. negative if this appear
187 * before other, positive otherwise.
188 */
189 int compare(const std::string& other) const {
190 return memncmp(data(), other.data(), size(), other.size());
191 }
192
193 /*!
194 * \brief Compares this to other
195 *
196 * \param other The character array to compare with.
197 *
198 * \return zero if both char sequences compare equal. negative if this appear
199 * before other, positive otherwise.
200 */
201 int compare(const char* other) const {
202 return memncmp(data(), other, size(), std::strlen(other));
203 }
204
205 /*!
206 * \brief Returns a pointer to the char array in the string.
207 *
208 * \return const char*
209 */
210 const char* c_str() const { return get()->data; }
211
212 /*!
213 * \brief Return the length of the string
214 *
215 * \return size_t string length
216 */
217 size_t size() const {
218 const auto* ptr = get();
219 return ptr->size;
220 }
221
222 /*!
223 * \brief Return the length of the string
224 *
225 * \return size_t string length
226 */
227 size_t length() const { return size(); }
228
229 /*!
230 * \brief Retun if the string is empty
231 *
232 * \return true if empty, false otherwise.
233 */
234 bool empty() const { return size() == 0; }
235
236 /*!
237 * \brief Read an element.
238 * \param pos The position at which to read the character.
239 *
240 * \return The char at position
241 */
242 char at(size_t pos) const {
243 if (pos < size()) {
244 return data()[pos];
245 } else {
246 throw std::out_of_range("tvm::String index out of bounds");
247 }
248 }
249
250 /*!
251 * \brief Return the data pointer
252 *
253 * \return const char* data pointer
254 */
255 const char* data() const { return get()->data; }
256
257 /*!
258 * \brief Convert String to an std::string object
259 *
260 * \return std::string
261 */
262 operator std::string() const { return std::string{get()->data, size()}; }
263
264 /*!
265 * \brief Check if a TVMArgValue can be converted to String, i.e. it can be std::string or String
266 * \param val The value to be checked
267 * \return A boolean indicating if val can be converted to String
268 */
269 inline static bool CanConvertFrom(const TVMArgValue& val);
270
271 /*!
272 * \brief Hash the binary bytes
273 * \param data The data pointer
274 * \param size The size of the bytes.
275 * \return the hash value.
276 */
277 static size_t HashBytes(const char* data, size_t size) {
278 // This function falls back to string copy with c++11 compiler and is
279 // recommended to be compiled with c++14
280#if TVM_USE_CXX17_STRING_VIEW_HASH
281 return std::hash<std::string_view>()(std::string_view(data, size));
282#elif TVM_USE_CXX14_STRING_VIEW_HASH
283 return std::hash<std::experimental::string_view>()(std::experimental::string_view(data, size));
284#else
285 return std::hash<std::string>()(std::string(data, size));
286#endif
287 }
288
289 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(String, ObjectRef, StringObj);
290
291 private:
292 /*!
293 * \brief Compare two char sequence
294 *
295 * \param lhs Pointers to the char array to compare
296 * \param rhs Pointers to the char array to compare
297 * \param lhs_count Length of the char array to compare
298 * \param rhs_count Length of the char array to compare
299 * \return int zero if both char sequences compare equal. negative if this
300 * appear before other, positive otherwise.
301 */
302 static int memncmp(const char* lhs, const char* rhs, size_t lhs_count, size_t rhs_count);
303
304 /*!
305 * \brief Concatenate two char sequences
306 *
307 * \param lhs Pointers to the lhs char array
308 * \param lhs_size The size of the lhs char array
309 * \param rhs Pointers to the rhs char array
310 * \param rhs_size The size of the rhs char array
311 *
312 * \return The concatenated char sequence
313 */
314 static String Concat(const char* lhs, size_t lhs_size, const char* rhs, size_t rhs_size) {
315 std::string ret(lhs, lhs_size);
316 ret.append(rhs, rhs_size);
317 return String(ret);
318 }
319
320 // Overload + operator
321 friend String operator+(const String& lhs, const String& rhs);
322 friend String operator+(const String& lhs, const std::string& rhs);
323 friend String operator+(const std::string& lhs, const String& rhs);
324 friend String operator+(const String& lhs, const char* rhs);
325 friend String operator+(const char* lhs, const String& rhs);
326
327 friend struct tvm::runtime::ObjectEqual;
328};

350inline String::String(std::string other) {
351 auto ptr = make_object<StringObj::FromStd>(std::move(other));
352 ptr->size = ptr->data_container.size();
353 ptr->data = ptr->data_container.data();
354 data_ = std::move(ptr);
355}

357inline String& String::operator=(std::string other) {
358 String replace{std::move(other)};
359 data_.swap(replace.data_);
360 return *this;
361}

466inline int String::memncmp(const char* lhs, const char* rhs, size_t lhs_count, size_t rhs_count) {
467 if (lhs == rhs && lhs_count == rhs_count) return 0;
468
469 for (size_t i = 0; i < lhs_count && i < rhs_count; ++i) {
470 if (lhs[i] < rhs[i]) return -1;
471 if (lhs[i] > rhs[i]) return 1;
472 }
473 if (lhs_count < rhs_count) {
474 return -1;
475 } else if (lhs_count > rhs_count) {
476 return 1;
477 } else {
478 return 0;
479 }
480}

482inline size_t ObjectHash::operator()(const ObjectRef& a) const {
483 if (const auto* str = a.as<StringObj>()) {
484 return String::HashBytes(str->data, str->size);
485 }
486 return ObjectPtrHash()(a);
487}
488
489inline bool ObjectEqual::operator()(const ObjectRef& a, const ObjectRef& b) const {
490 if (a.same_as(b)) {
491 return true;
492 }
493 if (const auto* str_a = a.as<StringObj>()) {
494 if (const auto* str_b = b.as<StringObj>()) {
495 return String::memncmp(str_a->data, str_b->data, str_a->size, str_b->size) == 0;
496 }
497 }
498 return false;
499}
