20/*
21 * \file tvm/runtime/c_runtime_api.h
22 * \brief TVM runtime library.
23 *
24 * The philosophy of TVM project is to customize the compilation
25 * stage to generate code that can used by other projects transparently.
26 * So this is a minimum runtime code gluing, and some limited
27 * memory management code to enable quick testing.
28 *
29 * The runtime API is independent from TVM compilation stack and can
30 * be linked via libtvm_runtime.
31 *
32 * The common flow is:
33 * - Use TVMFuncListGlobalNames to get global function name
34 * - Use TVMFuncCall to call these functions.
35 *
36 * Possible return values of the API functions:
37 * * 0: success
38 * * -1: the error can be retrieved through TVMGetLastError.
39 * * -2: a frontend error occurred and recorded in the frontend.
40 */

94/*!
95 * \brief The type code in used and only used in TVM FFI for argument passing.
96 *
97 * DLPack consistency:
98 * 1) kTVMArgInt is compatible with kDLInt
99 * 2) kTVMArgFloat is compatible with kDLFloat
100 * 3) kDLUInt is not in ArgTypeCode, but has a spared slot
101 *
102 * Downstream consistency:
103 * The kDLInt, kDLUInt, kDLFloat are kept consistent with the original ArgType code
104 *
105 * It is only used in argument passing, and should not be confused with
106 * DataType::TypeCode, which is DLPack-compatible.
107 *
108 * \sa tvm::runtime::DataType::TypeCode
109 */
110typedef enum {
111 kTVMArgInt = kDLInt,
112 kTVMArgFloat = kDLFloat,
113 kTVMOpaqueHandle = 3U,
114 kTVMNullptr = 4U,
115 kTVMDataType = 5U,
116 kDLDevice = 6U,
117 kTVMDLTensorHandle = 7U,
118 kTVMObjectHandle = 8U,
119 kTVMModuleHandle = 9U,
120 kTVMPackedFuncHandle = 10U,
121 kTVMStr = 11U,
122 kTVMBytes = 12U,
123 kTVMNDArrayHandle = 13U,
124 kTVMObjectRValueRefArg = 14U,
125 // Extension codes for other frameworks to integrate TVM PackedFunc.
126 // To make sure each framework's id do not conflict, use first and
127 // last sections to mark ranges.
128 // Open an issue at the repo if you need a section of code.
129 kTVMExtBegin = 15U,
130 kTVMNNVMFirst = 16U,
131 kTVMNNVMLast = 20U,
132 // The following section of code is used for non-reserved types.
133 kTVMExtReserveEnd = 64U,
134 kTVMExtEnd = 128U,
135} TVMArgTypeCode;

246/*!
247 * \brief Call a Packed TVM Function.
248 *
249 * \param func node handle of the function.
250 * \param arg_values The arguments
251 * \param type_codes The type codes of the arguments
252 * \param num_args Number of arguments.
253 *
254 * \param ret_val The return value.
255 * \param ret_type_code the type code of return value.
256 *
257 * \return 0 when success, nonzero when failure happens
258 * \note TVM calls always exchanges with type bits=64, lanes=1
259 *
260 * \note API calls always exchanges with type bits=64, lanes=1
261 * If API call returns container handles (e.g. FunctionHandle)
262 * these handles should be managed by the front-end.
263 * The front-end need to call free function (e.g. TVMFuncFree)
264 * to free these handles.
265 */
266TVM_DLL int TVMFuncCall(TVMFunctionHandle func, TVMValue* arg_values, int* type_codes, int num_args,
267 TVMValue* ret_val, int* ret_type_code);
