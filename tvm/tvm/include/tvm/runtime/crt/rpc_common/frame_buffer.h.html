20/*!
21 * \file tvm/runtime/crt/rpc_common/frame_buffer.h
22 * \brief Defines a buffer for use by the RPC framing layer.
23 */
24
25#ifndef TVM_RUNTIME_CRT_RPC_COMMON_FRAME_BUFFER_H_
26#define TVM_RUNTIME_CRT_RPC_COMMON_FRAME_BUFFER_H_
27
28#include <inttypes.h>
29#include <stdlib.h>
30
31namespace tvm {
32namespace runtime {
33namespace micro_rpc {
34
35class FrameBuffer {
36 public:
37 FrameBuffer(uint8_t* data, size_t data_size_bytes)
38 : data_{data}, capacity_{data_size_bytes}, num_valid_bytes_{0}, read_cursor_{0} {}
39
40 size_t Write(const uint8_t* data, size_t data_size_bytes);
41
42 size_t Read(uint8_t* data, size_t data_size_bytes);
43
44 size_t Peek(uint8_t* data, size_t data_size_bytes);
45
46 void Clear();
47
48 size_t ReadAvailable() const { return num_valid_bytes_ - read_cursor_; }
49
50 size_t Size() const { return num_valid_bytes_; }
51
52 private:
53 /*! \brief pointer to data buffer. */
54 uint8_t* data_;
55
56 /*! \brief The total number of bytes available in data_. Always a power of 2. */
57 size_t capacity_;
58
59 /*! \brief index into data_ of the next potentially-available byte in the buffer.
60 * The byte is available when tail_ != data_ + capacity_.
61 */
62 size_t num_valid_bytes_;
63
64 /*! \brief Read cursor position. */
65 size_t read_cursor_;
66};
67
68} // namespace micro_rpc
69} // namespace runtime
70} // namespace tvm
71
72#endif // TVM_RUNTIME_CRT_RPC_COMMON_FRAME_BUFFER_H_
