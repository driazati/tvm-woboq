47class Unframer {
48 public:
49 explicit Unframer(WriteStream* stream)
50 : stream_{stream},
51 state_{State::kFindPacketStart},
52 saw_escape_start_{false},
53 num_buffer_bytes_valid_{0} {}
54
55 /*!
56 * \brief Push data into unframer and try to decode one packet.
57 *
58 * This function will return when exactly one packet has been decoded. It may not consume all of
59 * `data` in this case, and valid bytes may remain at the end of data.
60 *
61 * \param data The new data to unframe and send downstream.
62 * \param data_size_bytes The number of valid bytes in data.
63 * \param bytes_consumed Pointer written with the number of bytes consumed from data.
64 * \return
65 * - kTvmErrorNoError when successful -- continue writing data.
66 * - kTvmErrorFramingInvalidState when the Unframer was in or enters an invalid state
67 * (probably indicates memory corruption).
68 * - kTvmErrorFramingShortPacket when a new packet started before the current one ended.
69 * - kTvmErrorFramingInvalidEscape when an invalid escape sequence was seen
70 */
71 tvm_crt_error_t Write(const uint8_t* data, size_t data_size_bytes, size_t* bytes_consumed);
72
73 /*! \brief Reset unframer to initial state. */
74 void Reset();
75
76 /*! \brief Return an underestimate of the number of bytes needed from the wire. */
77 size_t BytesNeeded();
78
79 private:
80 tvm_crt_error_t FindPacketStart();
81 tvm_crt_error_t FindPacketLength();
82 tvm_crt_error_t FindPacketCrc();
83 tvm_crt_error_t FindCrcEnd();
84
85 bool IsBufferFull(size_t buffer_full_bytes) {
86 return num_buffer_bytes_valid_ >= buffer_full_bytes;
87 }
88
89 /*! \brief Consume input into buffer_ until buffer_ has buffer_full_bytes. */
90 tvm_crt_error_t AddToBuffer(size_t buffer_full_bytes, bool update_crc);
91
92 void ClearBuffer();
93
94 /*! \brief Unescape and consume input bytes, storing into buffer.
95 *
96 * \param buffer A buffer to fill with consumed, unescaped bytes.
97 * \param buffer_size_bytes Size of buffer, in bytes.
98 * \param bytes_filled A pointer to an accumulator to which is added the number of bytes written
99 * to `buffer`.
100 * \param update_crc true when the CRC should be updated with the escaped bytes.
101 * \return
102 * - kTvmErrorNoError if successful
103 * - kTvmErrorFramingShortPacket if a start-of-packet escape code was encountered. If so,
104 * *bytes_filled indicates the number of bytes before the Escape::kEscapeStart byte.
105 * - kTvmErrorFramingInvalidEscape if an invalid escape sequence was seen.
106 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
107 * function returns 0.
108 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
109 * function returns an invalid positive number.
110 * - Any negative value (i.e. with bits in kTvmErrorSystemErrorMask set) returned by the
111 * WriteStream's Write() function.
112 */
113 tvm_crt_error_t ConsumeInput(uint8_t* buffer, size_t buffer_size_bytes, size_t* bytes_filled,
114 bool update_crc);
115
116 WriteStream* stream_;
117
118 enum class State : uint8_t {
119 kFindPacketStart = 0,
120 kFindPacketLength = 1,
121 kFindPacketCrc = 2,
122 kFindCrcEnd = 3,
123 };
124 State state_;
125
126 const uint8_t* input_;
127 size_t input_size_bytes_;
128
129 bool saw_escape_start_;
130
131 /*! \brief unframe buffer, sized to the longest framing field. */
132 uint8_t buffer_[128];
133
134 /*! \brief number of bytes in buffer that are currently valid. */
135 size_t num_buffer_bytes_valid_;
136
137 /*! \brief number of payload bytes left to receive before the CRC begins. */
138 size_t num_payload_bytes_remaining_;
139
140 /*! \brief Running CRC value. */
141 uint16_t crc_;
142};
143
144class Framer {
145 public:
146 typedef ssize_t (*WriteFunc)(const uint8_t* data, size_t data_size_bytes);
147
148 explicit Framer(WriteStream* stream)
149 : stream_{stream}, state_{State::kReset}, num_payload_bytes_remaining_{0} {}
150
151 /*! \brief Frame and write a full packet.
152 * \param payload The entire packet payload.
153 * \param payload_size_bytes Number of bytes in the packet.
154 * \return
155 * - kTvmErrorNoError when no error occurs
156 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
157 * function returns 0.
158 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
159 * function returns an invalid positive number.
160 * - Any negative value (i.e. with bits in kTvmErrorSystemErrorMask set) returned by the
161 * WriteStream's Write() function.
162 */
163 tvm_crt_error_t Write(const uint8_t* payload, size_t payload_size_bytes);
164
165 /*! \brief Start framing and writing a new packet to the wire.
166 *
167 * When transmitting payloads that are too large to be buffered, call this function first to send
168 * the packet header and length fields.
169 *
170 * \param payload_size_bytes Number of payload bytes included as part of this packet.
171 * \return
172 * - kTvmErrorNoError when no error occurs
173 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
174 * function returns 0.
175 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
176 * function returns an invalid positive number.
177 * - Any negative value (i.e. with bits in kTvmErrorSystemErrorMask set) returned by the
178 * WriteStream's Write() function.
179 */
180 tvm_crt_error_t StartPacket(size_t payload_size_bytes);
181
182 /*! \brief Write payload data to the wire.
183 *
184 * When transmitting payloads that are too large to be buffered, call this function after calling
185 * StartPacket to escape and transmit framed payloads. This function can be called multiple times
186 * for a single packet.
187 *
188 * \param payload_chunk A piece of the packet payload.
189 * \param payload_chunk_size_bytes Number of valid bytes in payload_chunk.
190 * \return
191 * - kTvmErrorNoError when no error occurs
192 * - kTvmErrorFramingInvalidState when StartPacket() has not been called.
193 * - kTvmErrorFramingPayloadOverflow when more bytes were requested to be written than were
194 * declared in the payload_size_bytes parameter given to StartPacket().
195 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
196 * function returns 0.
197 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
198 * function returns an invalid positive number.
199 * - Any negative value (i.e. with bits in kTvmErrorSystemErrorMask set) returned by the
200 * WriteStream's Write() function.
201 */
202 tvm_crt_error_t WritePayloadChunk(const uint8_t* payload_chunk, size_t payload_chunk_size_bytes);
203
204 /* \brief Finish writing one packet by sending the CRC.
205 *
206 * When transmitting paylaods that are too large to be buffered, call this function after sending
207 * the entire payload using WritePayloadChunk.
208 *
209 * \return
210 * - kTvmErrorNoError when no error occurs
211 * - kTvmErrorFramingInvalidState when StartPacket() has not been called.
212 * - kTvmErrorFramingPayloadIncomplete when less bytes were written using WritePayloadChunk()
213 * than were declared in the payload_size_bytes parameter given to StartPacket().
214 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
215 * function returns 0.
216 * - kTvmErrorWriteStreamShortWrite if the WriteStream passed to constructor's Write()
217 * function returns an invalid positive number.
218 * - Any negative value (i.e. with bits in kTvmErrorSystemErrorMask set) returned by the
219 * WriteStream's Write() function.
220 */
221 tvm_crt_error_t FinishPacket();
222
223 /* \brief Reset state of the Framer. */
224 void Reset();
225
226 private:
227 /*! \brief Maximum size of stack-based buffer. */
228 static constexpr const size_t kMaxStackBufferSizeBytes = 128;
229
230 enum class State : uint8_t {
231 /*! \brief State entered at construction time or after write error, before first packet sent. */
232 kReset = 0,
233
234 /*! \brief State entered after a packet has successfully finished transmitting. */
235 kIdle = 1,
236
237 /*! \brief State entered when a packet payload or CRC needs to be transmitted. */
238 kTransmitPacketPayload = 2,
239 };
240
241 /*!
242 * \brief Escape data and write the result to wire, and update crc_.
243 *
244 * \param data Unescaped data to write.
245 * \param data_size_bytes Number of valid bytes in data.
246 * \param escape true if escaping should be applied.
247 * \param update_crc true if escaping should be applied.
248 * \return kTvmErrorNoError on success, negative value on error.
249 */
250 tvm_crt_error_t WriteAndCrc(const uint8_t* data, size_t data_size_bytes, bool escape,
251 bool update_crc);
252
253 /*! \brief Called to write framed data to the transport. */
254 WriteStream* stream_;
255
256 /*! \brief State fo the Framer. */
257 State state_;
258
259 /*! \brief When state_ == kTransmitPacketPayload, number of payload bytes left to transmit. */
260 size_t num_payload_bytes_remaining_;
261
262 /*! \brief Running CRC value. */
263 uint16_t crc_;
264};
