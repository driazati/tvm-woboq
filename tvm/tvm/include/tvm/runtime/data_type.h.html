35/*!
36 * \brief Runtime primitive data type.
37 *
38 * This class is a thin wrapper of DLDataType.
39 * We also make use of DataType in compiler to store quick hint
40 */
41class DataType {
42 public:
43 /*!
44 * \brief Type code for the DataType.
45 *
46 * DLPack consistency:
47 * 1) kInt is consistent with kDLInt
48 * 2) kUInt is consistent with kDLUInt
49 * 3) kFloat is consistent with kDLFloat
50 */
51 enum TypeCode {
52 kInt = kDLInt,
53 kUInt = kDLUInt,
54 kFloat = kDLFloat,
55 kHandle = TVMArgTypeCode::kTVMOpaqueHandle,
56 kBFloat = kDLBfloat,
57 kCustomBegin = 129
58 };
59 /*! \brief default constructor */
60 DataType() { data_ = DataType::Void(); }
61 /*!
62 * \brief Constructor
63 * \param dtype The DLDataType
64 */
65 explicit DataType(DLDataType dtype) : data_(dtype) {}
66 /*!
67 * \brief Constructor
68 * \param code The type code.
69 * \param bits The number of bits in the type.
70 * \param lanes The number of lanes.
71 */
72 DataType(int code, int bits, int lanes) {
73 data_.code = static_cast<uint8_t>(code);
74 data_.bits = static_cast<uint8_t>(bits);
75 data_.lanes = static_cast<uint16_t>(lanes);
76 if (code == kBFloat) {
77 ICHECK_EQ(bits, 16);
78 }
79 }
80 /*! \return The type code. */
81 int code() const { return static_cast<int>(data_.code); }
82 /*! \return number of bits in the data. */
83 int bits() const { return static_cast<int>(data_.bits); }
84 /*! \return number of bytes to store each scalar. */
85 int bytes() const { return (bits() + 7) / 8; }
86 /*! \return number of lanes in the data. */
87 int lanes() const { return static_cast<int>(data_.lanes); }
88 /*! \return whether type is a scalar type. */
89 bool is_scalar() const { return lanes() == 1; }
90 /*! \return whether type is a scalar type. */
91 bool is_bool() const { return code() == DataType::kUInt && bits() == 1; }
92 /*! \return whether type is a float type. */
93 bool is_float() const { return code() == DataType::kFloat; }
94 /*! \return whether type is a float16 type. */
95 bool is_float16() const { return is_float() && bits() == 16; }
96 /*! \return whether type is a bfloat16 type. */
97 bool is_bfloat16() const { return code() == DataType::kBFloat && bits() == 16; }
98 /*! \return whether type is an int type. */
99 bool is_int() const { return code() == DataType::kInt; }
100 /*! \return whether type is an uint type. */
101 bool is_uint() const { return code() == DataType::kUInt; }
102 /*! \return whether type is a handle type. */
103 bool is_handle() const { return code() == DataType::kHandle && !is_void(); }
104 /*! \return whether type is a vector type. */
105 bool is_vector() const { return lanes() > 1; }
106 /*! \return whether type is a bool vector type. */
107 bool is_vector_bool() const { return is_vector() && bits() == 1; }
108 /*! \return whether type is a Void type. */
109 bool is_void() const { return code() == DataType::kHandle && bits() == 0 && lanes() == 0; }
110 /*!
111 * \brief Create a new data type by change lanes to a specified value.
112 * \param lanes The target number of lanes.
113 * \return the result type.
114 */
115 DataType with_lanes(int lanes) const { return DataType(data_.code, data_.bits, lanes); }
116 /*!
117 * \brief Create a new data type by change bits to a specified value.
118 * \param bits The target number of bits.
119 * \return the result type.
120 */
121 DataType with_bits(int bits) const { return DataType(data_.code, bits, data_.lanes); }
122 /*!
123 * \brief Get the scalar version of the type.
124 * \return the result type.
125 */
126 DataType element_of() const { return with_lanes(1); }
127 /*!
128 * \brief Equal comparator.
129 * \param other The data type to compare against.
130 * \return The comparison result.
131 */
132 bool operator==(const DataType& other) const {
133 return data_.code == other.data_.code && data_.bits == other.data_.bits &&
134 data_.lanes == other.data_.lanes;
135 }
136 /*!
137 * \brief NotEqual comparator.
138 * \param other The data type to compare against.
139 * \return The comparison result.
140 */
141 bool operator!=(const DataType& other) const { return !operator==(other); }
142 /*!
143 * \brief Converter to DLDataType
144 * \return the result.
145 */
146 operator DLDataType() const { return data_; }
147
148 /*!
149 * \brief Construct an int type.
150 * \param bits The number of bits in the type.
151 * \param lanes The number of lanes.
152 * \return The constructed data type.
153 */
154 static DataType Int(int bits, int lanes = 1) { return DataType(kDLInt, bits, lanes); }
155 /*!
156 * \brief Construct an uint type.
157 * \param bits The number of bits in the type.
158 * \param lanes The number of lanes
159 * \return The constructed data type.
160 */
161 static DataType UInt(int bits, int lanes = 1) { return DataType(kDLUInt, bits, lanes); }
162 /*!
163 * \brief Construct an float type.
164 * \param bits The number of bits in the type.
165 * \param lanes The number of lanes
166 * \return The constructed data type.
167 */
168 static DataType Float(int bits, int lanes = 1) { return DataType(kDLFloat, bits, lanes); }
169 /*!
170 * \brief Construct an bfloat type.
171 * \param bits The number of bits in the type.
172 * \param lanes The number of lanes
173 * \return The constructed data type.
174 */
175 static DataType BFloat(int bits, int lanes = 1) { return DataType(kDLBfloat, bits, lanes); }
176 /*!
177 * \brief Construct a bool type.
178 * \param lanes The number of lanes
179 * \return The constructed data type.
180 */
181 static DataType Bool(int lanes = 1) { return DataType::UInt(1, lanes); }
182 /*!
183 * \brief Construct a handle type.
184 * \param bits The number of bits in the type.
185 * \param lanes The number of lanes
186 * \return The constructed data type.
187 */
188 static DataType Handle(int bits = 64, int lanes = 1) { return DataType(kHandle, bits, lanes); }
189 /*!
190 * \brief Construct a Void type.
191 * \return The constructed data type.
192 */
193 static DataType Void() { return DataType(kHandle, 0, 0); }
194 /*!
195 * \brief Get the corresponding type of TVMShapeIndex.
196 * \return The type of TVM shape index.
197 */
198 static DataType ShapeIndex() {
199 if (std::is_signed<tvm_index_t>::value) {
200 return DataType::Int(sizeof(tvm_index_t) * 8);
201 } else {
202 return DataType::UInt(sizeof(tvm_index_t) * 8);
203 }
204 }
205
206 private:
207 DLDataType data_;
208};

210/*!
211 * \brief Get the number of bytes needed in a vector.
212 * \param dtype The data type.
213 * \return Number of bytes needed.
214 */
215inline int GetVectorBytes(DataType dtype) {
216 int data_bits = dtype.bits() * dtype.lanes();
217 // allow bool to exist
218 if (dtype == DataType::Bool() || dtype == DataType::Int(4) || dtype == DataType::UInt(4) ||
219 dtype == DataType::Int(1)) {
220 return 1;
221 }
222 ICHECK_EQ(data_bits % 8, 0U) << "Need to load/store by multiple of bytes";
223 return data_bits / 8;
224}
225
226/*!
227 * \brief Check whether type matches the given spec.
228 * \param t The type
229 * \param code The type code.
230 * \param bits The number of bits to be matched.
231 * \param lanes The number of lanes in the type.
232 */
233inline bool TypeMatch(DLDataType t, int code, int bits, int lanes = 1) {
234 return t.code == code && t.bits == bits && t.lanes == lanes;
235}
236/*!
237 * \brief Check whether two types are equal .
238 * \param lhs The left operand.
239 * \param rhs The right operand.
240 */
241inline bool TypeEqual(DLDataType lhs, DLDataType rhs) {
242 return lhs.code == rhs.code && lhs.bits == rhs.bits && lhs.lanes == rhs.lanes;
243}

288// implementation details
289inline const char* DLDataTypeCode2Str(DLDataTypeCode type_code) {
290 switch (static_cast<int>(type_code)) {
291 case kDLInt:
292 return "int";
293 case kDLUInt:
294 return "uint";
295 case kDLFloat:
296 return "float";
297 case DataType::kHandle:
298 return "handle";
299 case kDLBfloat:
300 return "bfloat";
301 default:
302 LOG(FATAL) << "unknown type_code=" << static_cast<int>(type_code);
303 return "";
304 }
305}
306
307inline std::ostream& operator<<(std::ostream& os, DLDataType t) { // NOLINT(*)
308 if (t.bits == 1 && t.lanes == 1 && t.code == kDLUInt) {
309 os << "bool";
310 return os;
311 }
312 if (DataType(t).is_void()) {
313 return os << "void";
314 }
315 if (t.code < DataType::kCustomBegin) {
316 os << DLDataTypeCode2Str(static_cast<DLDataTypeCode>(t.code));
317 } else {
318 os << "custom[" << GetCustomTypeName(t.code) << "]";
319 }
320 if (t.code == kTVMOpaqueHandle) return os;
321 os << static_cast<int>(t.bits);
322 if (t.lanes != 1) {
323 os << 'x' << static_cast<int>(t.lanes);
324 }
325 return os;
326}
327
328inline std::ostream& operator<<(std::ostream& os, const DataType& dtype) { // NOLINT(*)
329 return os << dtype.operator DLDataType();
330}
331
332inline std::string DLDataType2String(DLDataType t) {
333 if (t.bits == 0) return "";
334 std::ostringstream os;
335 os << t;
336 return os.str();
337}

339inline DLDataType String2DLDataType(std::string s) {
340 DLDataType t;
341 // handle void type
342 if (s.length() == 0) {
343 t = DataType::Void();
344 return t;
345 }
346 t.bits = 32;
347 t.lanes = 1;
348 const char* scan;
349 if (s.substr(0, 3) == "int") {
350 t.code = kDLInt;
351 scan = s.c_str() + 3;
352 } else if (s.substr(0, 4) == "uint") {
353 t.code = kDLUInt;
354 scan = s.c_str() + 4;
355 } else if (s.substr(0, 5) == "float") {
356 t.code = kDLFloat;
357 scan = s.c_str() + 5;
358 } else if (s.substr(0, 6) == "handle") {
359 t.code = kTVMOpaqueHandle;
360 t.bits = 64; // handle uses 64 bit by default.
361 scan = s.c_str() + 6;
362 } else if (s == "bool") {
363 t.code = kDLUInt;
364 t.bits = 1;
365 t.lanes = 1;
366 return t;
367 } else if (s.substr(0, 6) == "bfloat") {
368 t.code = DataType::kBFloat;
369 scan = s.c_str() + 6;
370 } else if (s.substr(0, 6) == "custom") {
371 t.code = ParseCustomDatatype(s, &scan);
372 } else {
373 scan = s.c_str();
374 LOG(FATAL) << "unknown type " << s;
375 }
376 char* xdelim; // emulate sscanf("%ux%u", bits, lanes)
377 uint8_t bits = static_cast<uint8_t>(strtoul(scan, &xdelim, 10));
378 if (bits != 0) t.bits = bits;
379 char* endpt = xdelim;
380 if (*xdelim == 'x') {
381 t.lanes = static_cast<uint16_t>(strtoul(xdelim + 1, &endpt, 10));
382 }
383 ICHECK(endpt == s.c_str() + s.length()) << "unknown type " << s;
384 return t;
385}

393namespace std {
394template <>
395struct hash<tvm::DataType> {
396 inline int cantor_pairing_function(int a, int b) const { return (a + b) * (a + b + 1) / 2 + b; }
397 std::size_t operator()(tvm::DataType const& dtype) const {
398 int a = dtype.code();
399 int b = dtype.bits();
400 int c = dtype.lanes();
401 int d = cantor_pairing_function(a, b);
402 return cantor_pairing_function(c, d);
403 }
404};
