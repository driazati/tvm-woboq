34/*!
35 * \brief A frame that represents the IRModule frame with functions and global variables.
36 *
37 * \sa IRModuleFrame
38 */
39class IRModuleFrameNode : public IRBuilderFrameNode {
40 public:
41 Array<GlobalVar> global_vars;
42 Array<BaseFunc> functions;
43
44 void VisitAttrs(tvm::AttrVisitor* v) {
45 IRBuilderFrameNode::VisitAttrs(v);
46 v->Visit("global_vars", &global_vars);
47 v->Visit("functions", &functions);
48 }
49
50 static constexpr const char* _type_key = "script.ir_builder.IRModuleFrame";
51 TVM_DECLARE_FINAL_OBJECT_INFO(IRModuleFrameNode, IRBuilderFrameNode);
52
53 public:
54 void ExitWithScope() final;
55};
56
57/*!
58 * \brief Managed reference to IRModuleFrameNode.
59 *
60 * \sa IRModuleFrameNode
61 */
62class IRModuleFrame : public IRBuilderFrame {
63 public:
64 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(IRModuleFrame, IRBuilderFrame,
65 IRModuleFrameNode);
66};
