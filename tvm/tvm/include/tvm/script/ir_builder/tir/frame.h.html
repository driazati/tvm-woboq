63/*!
64 * \brief A frame that represents the PrimFunc containing TIR statements.
65 *
66 * \sa PrimFuncFrame
67 */
68class PrimFuncFrameNode : public TIRFrameNode {
69 public:
70 /*! \brief The name of the block. */
71 Optional<String> name;
72 /*! \brief Function parameters. */
73 Array<tvm::tir::Var> args;
74 /*! \brief The return type of the function. */
75 Optional<Type> ret_type;
76 /*! \brief Maps some parameters to specific Buffer data structures. */
77 Map<tvm::tir::Var, tvm::tir::Buffer> buffer_map;
78 /*! \brief The buffer map prior to flattening. */
79 Map<tvm::tir::Var, tvm::tir::Buffer> preflattened_buffer_map;
80 /*! \brief Additional attributes storing the meta-data */
81 Optional<Map<String, ObjectRef>> attrs;
82 /*! \brief The variable map bound to thread env. */
83 Map<tvm::tir::Var, tvm::tir::IterVar> env_threads;
84 /*! \brief The buffer allocated in root block. */
85 Array<tvm::tir::Buffer> root_alloc_buffers;
86
87 void VisitAttrs(tvm::AttrVisitor* v) {
88 TIRFrameNode::VisitAttrs(v);
89 v->Visit("name", &name);
90 v->Visit("args", &args);
91 v->Visit("ret_type", &ret_type);
92 v->Visit("buffer_map", &buffer_map);
93 v->Visit("preflattened_buffer_map", &preflattened_buffer_map);
94 v->Visit("attrs", &attrs);
95 v->Visit("env_threads", &env_threads);
96 v->Visit("root_alloc_buffers", &root_alloc_buffers);
97 }
98
99 static constexpr const char* _type_key = "script.ir_builder.tir.PrimFuncFrame";
100 TVM_DECLARE_FINAL_OBJECT_INFO(PrimFuncFrameNode, TIRFrameNode);
101
102 public:
103 /*!
104 * \brief The method called when exiting RAII scope.
105 * \sa tvm::support::With
106 */
107 void ExitWithScope() final;
108};

120/*!
121 * \brief A frame that represents the block.
122 *
123 * \sa BlockFrame
124 */
125class BlockFrameNode : public TIRFrameNode {
126 public:
127 /*! \brief The name of the block. */
128 String name;
129 /*! \brief The variables of the block. */
130 Array<tvm::tir::IterVar> iter_vars;
131 /*! \brief The read buffer regions of the block. */
132 Optional<Array<tvm::tir::BufferRegion>> reads;
133 /*! \brief The write buffer regions of the block. */
134 Optional<Array<tvm::tir::BufferRegion>> writes;
135 /*! \brief The init statement of the bolck. */
136 Optional<tvm::tir::Stmt> init;
137 /*! \brief The buffer allocated in the block. */
138 Array<tvm::tir::Buffer> alloc_buffers;
139 /*! \brief The match buffer regions. */
140 Array<tvm::tir::MatchBufferRegion> match_buffers;
141 /*! \brief The annotation of the block. */
142 Optional<Map<String, ObjectRef>> annotations;
143 /*! \brief The corresponding values of the iter vars. */
144 Array<PrimExpr> iter_values;
145 /*!
146 * \brief The predicate of the block realization, the block will only be executed when the
147 * predicate is true.
148 */
149 Optional<PrimExpr> predicate;
150 /*! \brief The flag whether to construct BlockRealize or Block. */
151 bool no_realize;
152
153 void VisitAttrs(tvm::AttrVisitor* v) {
154 TIRFrameNode::VisitAttrs(v);
155 v->Visit("name", &name);
156 v->Visit("iter_vars", &iter_vars);
157 v->Visit("reads", &reads);
158 v->Visit("writes", &writes);
159 v->Visit("init", &init);
160 v->Visit("alloc_buffers", &alloc_buffers);
161 v->Visit("match_buffers", &match_buffers);
162 v->Visit("annotations", &annotations);
163 v->Visit("iter_values", &iter_values);
164 v->Visit("predicate", &predicate);
165 v->Visit("no_realize", &no_realize);
166 }
167
168 static constexpr const char* _type_key = "script.ir_builder.tir.BlockFrame";
169 TVM_DECLARE_FINAL_OBJECT_INFO(BlockFrameNode, TIRFrameNode);
170
171 public:
172 /*!
173 * \brief The method called when exiting RAII scope.
174 * \sa tvm::support::With
175 */
176 void ExitWithScope() final;
177};

225/*!
226 * \brief A frame that represents the for loop.
227 *
228 * \sa ForFrame
229 */
230class ForFrameNode : public TIRFrameNode {
231 public:
232 /*!
233 * \brief Functions that generate loop nests.
234 * \param loop_vars The loop variables, from outer to inner
235 * \param loop_extents The loop extents that correspond to loop variables
236 * \param loop_body The loop body
237 * \return A stmt, the loop nest
238 */
239 using FMakeForLoop = runtime::TypedPackedFunc<tvm::tir::Stmt(
240 Array<tvm::tir::Var> loop_vars, Array<Range> loop_extents, tvm::tir::Stmt loop_body)>;
241 /*! \brief The loop variable. */
242 Array<tvm::tir::Var> vars;
243 /*! \brief The domains of iteration. */
244 Array<Range> doms;
245 /*! \brief The for loop generating function. */
246 FMakeForLoop f_make_for_loop;
247
248 void VisitAttrs(tvm::AttrVisitor* v) {
249 TIRFrameNode::VisitAttrs(v);
250 v->Visit("vars", &vars);
251 v->Visit("doms", &doms);
252 // `f_make_for_loop` is not visited.
253 }

255 static constexpr const char* _type_key = "script.ir_builder.tir.ForFrame";

257
258 public:
259 /*!
260 * \brief The method called when exiting RAII scope.
261 * \sa tvm::support::With
262 */
263 void ExitWithScope() final;
264};
