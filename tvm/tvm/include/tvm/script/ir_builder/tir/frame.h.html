31/*!
32 * \brief A base frame that represents the TIR fame with body of statements.
33 *
34 * \sa TIRFrame
35 */
36class TIRFrameNode : public IRBuilderFrameNode {
37 public:
38 /*! \brief The Stmt within in this frame. */
39 Array<tvm::tir::Stmt> stmts;
40
41 void VisitAttrs(tvm::AttrVisitor* v) {
42 IRBuilderFrameNode::VisitAttrs(v);
43 v->Visit("stmts", &stmts);
44 }
45
46 static constexpr const char* _type_key = "script.ir_builder.tir.TIRFrame";
47 TVM_DECLARE_BASE_OBJECT_INFO(TIRFrameNode, IRBuilderFrameNode);
48};

63/*!
64 * \brief A frame that represents the PrimFunc containing TIR statements.
65 *
66 * \sa PrimFuncFrame
67 */
68class PrimFuncFrameNode : public TIRFrameNode {
69 public:
70 /*! \brief The name of the block. */
71 Optional<String> name;
72 /*! \brief Function parameters. */
73 Array<tvm::tir::Var> args;
74 /*! \brief The return type of the function. */
75 Optional<Type> ret_type;
76 /*! \brief Maps some parameters to specific Buffer data structures. */
77 Map<tvm::tir::Var, tvm::tir::Buffer> buffer_map;
78 /*! \brief The buffer map prior to flattening. */
79 Map<tvm::tir::Var, tvm::tir::Buffer> preflattened_buffer_map;
80 /*! \brief Additional attributes storing the meta-data */
81 Optional<Map<String, ObjectRef>> attrs;
82 /*! \brief The variable map bound to thread env. */
83 Map<tvm::tir::Var, tvm::tir::IterVar> env_threads;
84 /*! \brief The buffer allocated in root block. */
85 Array<tvm::tir::Buffer> root_alloc_buffers;
86
87 void VisitAttrs(tvm::AttrVisitor* v) {
88 TIRFrameNode::VisitAttrs(v);
89 v->Visit("name", &name);
90 v->Visit("args", &args);
91 v->Visit("ret_type", &ret_type);
92 v->Visit("buffer_map", &buffer_map);
93 v->Visit("preflattened_buffer_map", &preflattened_buffer_map);
94 v->Visit("attrs", &attrs);
95 v->Visit("env_threads", &env_threads);
96 v->Visit("root_alloc_buffers", &root_alloc_buffers);
97 }
98
99 static constexpr const char* _type_key = "script.ir_builder.tir.PrimFuncFrame";
100 TVM_DECLARE_FINAL_OBJECT_INFO(PrimFuncFrameNode, TIRFrameNode);
101
102 public:
103 /*!
104 * \brief The method called when exiting RAII scope.
105 * \sa tvm::support::With
106 */
107 void ExitWithScope() final;
108};

120/*!
121 * \brief A frame that represents the assert statement. Proceeds if the condition is true,
122 * otherwise aborts with the message.
123 *
124 * \sa AssertFrame
125 */
126class AssertFrameNode : public TIRFrameNode {
127 public:
128 /*! \brief The PrimExpr to test. */
129 PrimExpr condition;
130 /*! \brief The output error message when the assertion failed. */
131 PrimExpr message;
132
133 void VisitAttrs(tvm::AttrVisitor* v) {
134 TIRFrameNode::VisitAttrs(v);
135 v->Visit("condition", &condition);
136 v->Visit("message", &message);
137 }
138
139 static constexpr const char* _type_key = "script.ir_builder.tir.AssertFrame";
140 TVM_DECLARE_FINAL_OBJECT_INFO(AssertFrameNode, TIRFrameNode);
141
142 public:
143 /*!
144 * \brief The method called when exiting RAII scope.
145 * \sa tvm::support::With
146 */
147 void ExitWithScope() final;
148};
