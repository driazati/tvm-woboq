425#define TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(FuncName, DType) \
426 inline PrimExpr FuncName(Optional<PrimExpr> expr = NullOpt) { \
427 DataType dtype = DType; \
428 return expr.defined() ? tvm::cast(dtype, expr.value()) : tvm::tir::Var("", dtype); \
429 }

431#define TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST_SIZES(DType, FDType) \
432 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##8, FDType(8)); \
433 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##16, FDType(16)); \
434 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##32, FDType(32)); \
435 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##64, FDType(64));
