445#define TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(FuncName, DType) \
446 inline PrimExpr FuncName(Optional<PrimExpr> expr = NullOpt) { \
447 DataType dtype = DType; \
448 return expr.defined() ? tvm::cast(dtype, expr.value()) : tvm::tir::Var("", dtype); \
449 }

451#define TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST_SIZES(DType, FDType) \
452 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##8, FDType(8)); \
453 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##16, FDType(16)); \
454 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##32, FDType(32)); \
455 TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(DType##64, FDType(64));
