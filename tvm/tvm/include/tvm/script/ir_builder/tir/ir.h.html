34/*!
35 * \brief The buffer declaration function.
36 * \param shape The type of the buffer prior to flattening.
37 * \param dtype The data type in the content of the buffer.
38 * \param buffer_name The name of the buffer.
39 * \param data The pointer to the head of the data.
40 * \param strides The strides of each dimension.
41 * \param elem_offset The offset in terms of number of dtype elements (including lanes).
42 * \param storage_scope The optional storage scope of buffer data pointer.
43 * \param align The alignment requirement of data pointer in bytes.
44 * \param offset_factor The factor of elem_offset field.
45 * \param buffer_type The buffer type.
46 * \param axis_separators The separators between input axes when generating flattened output axes.
47 * \return The declared buffer.
48 */
49Buffer BufferDecl(Array<PrimExpr> shape, DataType dtype, String buffer_name, Optional<Var> data,
50 Optional<Array<PrimExpr>> strides, Optional<PrimExpr> elem_offset,
51 String storage_scope, int align, int offset_factor, String buffer_type,
52 Optional<Array<IntImm>> axis_separators);
53

74Buffer Arg(String name, Buffer buffer);

93Type FuncRet(Type ret_type);

95/*!
96 * \brief The buffer match statement.
97 * \param param The parameter of the PrimFunc to match.
98 * \param shape The type of the buffer prior to flattening.
99 * \param dtype The data type in the content of the buffer.
100 * \param data The pointer to the head of the data.
101 * \param strides The strides of each dimension.
102 * \param elem_offset The offset in terms of number of dtype elements (including lanes).
103 * \param storage_scope The optional storage scope of buffer data pointer.
104 * \param align The alignment requirement of data pointer in bytes.
105 * \param offset_factor The factor of elem_offset field.
106 * \param buffer_type The buffer type.
107 * \param axis_separators The separators between input axes when generating flattened output axes.
108 * \return The matched buffer.
109 */
110Buffer MatchBuffer(ObjectRef param, Array<PrimExpr> shape, DataType dtype = DataType::Float(32),
111 Optional<Var> data = NullOpt, Array<PrimExpr> strides = {},
112 PrimExpr elem_offset = PrimExpr(), String storage_scope = "global",
113 int align = -1, int offset_factor = 0, String buffer_type = "default",
114 Array<IntImm> axis_separators = {});
115
116/*!
117 * \brief The pre-flattened buffer statement.
118 * \param postflattened_buffer The original buffer to be flattened.
119 * \param shape The type of the buffer prior to flattening.
120 * \param dtype The data type in the content of the buffer.
121 * \param data The pointer to the head of the data.
122 * \param strides The strides of each dimension.
123 * \param elem_offset The offset in terms of number of dtype elements (including lanes).
124 * \param storage_scope The optional storage scope of buffer data pointer.
125 * \param align The alignment requirement of data pointer in bytes.
126 * \param offset_factor The factor of elem_offset field.
127 * \param buffer_type The buffer type.
128 * \param axis_separators The separators between input axes when generating flattened output axes.
129 */
130void PreflattenedBuffer(Buffer postflattened_buffer, Array<PrimExpr> shape,
131 DataType dtype = DataType::Float(32), Optional<Var> data = NullOpt,
132 Array<PrimExpr> strides = {}, PrimExpr elem_offset = PrimExpr(),
133 String storage_scope = "global", int align = -1, int offset_factor = 0,
134 String buffer_type = "default", Array<IntImm> axis_separators = {});

150#define TVM_TIR_IR_BUILDER_DEF_DTYPE_CAST(FuncName, DType) \
151 inline PrimExpr FuncName(Optional<PrimExpr> expr = NullOpt) { \
152 DataType dtype = DType; \
153 return expr.defined() ? tvm::cast(dtype, expr.value()) : tvm::tir::Var("", dtype); \
154 }
