30/*!
31 * \brief The base class of all Doc.
32 *
33 * Doc is an intermediate representation between IR from TVM
34 * and the TVMScript code.
35 * During printing, IR graph is first translated into Doc tree,
36 * then the Doc tree is translated to the target language in
37 * text format.
38 *
39 * \sa Doc
40 */
41class DocNode : public Object {
42 public:
43 /*!
44 * \brief The list of object paths of the source IR node.
45 *
46 * This is used to trace back to the IR node position where
47 * this Doc is generated, in order to position the diagnostic
48 * message.
49 */
50 mutable Array<ObjectPath> source_paths;
51
52 void VisitAttrs(AttrVisitor* v) { v->Visit("source_paths", &source_paths); }
53
54 static constexpr const char* _type_key = "script.printer.Doc";
55 TVM_DECLARE_BASE_OBJECT_INFO(DocNode, Object);
56
57 public:
58 virtual ~DocNode() = default;
59};
60
61/*!
62 * \brief Reference type of DocNode.
63 *
64 * \sa DocNode
65 */
66class Doc : public ObjectRef {
67 protected:
68 Doc() = default;
69
70 public:
71 virtual ~Doc() = default;
72 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(Doc, ObjectRef, DocNode);
73};

77/*!
78 * \brief The base class of expression doc.
79 *
80 * \sa ExprDoc
81 */
82class ExprDocNode : public DocNode {
83 public:
84 /*!
85 * \brief Create a doc representing attribute access on the current ExprDoc
86 * \param attr The attribute to access.
87 */
88 ExprDoc Attr(String attr) const;
89
90 /*!
91 * \brief Create a doc representing index access on the current ExprDoc
92 * \param indices The indices to access.
93 */
94 ExprDoc operator[](Array<Doc> indices) const;
95
96 /*!
97 * \brief Create a doc representing calling the current ExprDoc
98 * \param args The positional arguments of the function call.
99 */
100 ExprDoc Call(Array<ExprDoc, void> args) const;
101
102 /*!
103 * \brief Create a doc representing attribute access on the current ExprDoc
104 * \param args The positional arguments of the function call.
105 * \param kwargs_keys Keys of keywords arguments of the function call.
106 * \param kwargs_values Values of keywords arguments of the function call.
107 */
108 ExprDoc Call(Array<ExprDoc, void> args, //
109 Array<String> kwargs_keys, //
110 Array<ExprDoc, void> kwargs_values) const;
111
112 void VisitAttrs(AttrVisitor* v) { DocNode::VisitAttrs(v); }
113
114 static constexpr const char* _type_key = "script.printer.ExprDoc";
115 TVM_DECLARE_BASE_OBJECT_INFO(ExprDocNode, DocNode);
116};

442/*!
443 * \brief Doc that represents operation.
444 *
445 * It can be unary, binary and other special operators (for example,
446 * the if-then-else expression).
447 *
448 * \sa OperationDoc
449 */
450class OperationDocNode : public ExprDocNode {
451 public:
452 enum class Kind : int32_t {
453 // Unary operators
454 kUnaryStart = 0,
455 kUSub = 1, // -x
456 kInvert = 2, // ~x
457 kUnaryEnd = 3,
458
459 // Binary operators
460 kBinaryStart = 4,
461 kAdd = 5, // +
462 kSub = 6, // -
463 kMult = 7, // *
464 kDiv = 8, // /
465 kFloorDiv = 9, // // in Python
466 kMod = 10, // % in Python
467 kPow = 11, // ** in Python
468 kLShift = 12, // <<
469 kRShift = 13, // >>
470 kBitAnd = 14, // &
471 kBitOr = 15, // |
472 kBitXor = 16, // ^
473 kLt = 17, // <
474 kLtE = 18, // <=
475 kEq = 19, // ==
476 kNotEq = 20, // !=
477 kGt = 21, // >
478 kGtE = 22, // >=
479 kBinaryEnd = 23,
480
481 // Special
482 kSpecialStart = 24,
483 kIfThenElse = 25, // <operands[1]> if <operands[0]> else <operands[2]>
484 kSpecialEnd = 26
485 };
486
487 /*! \brief The kind of operation (operator) */
488 Kind kind;
489 /*! \brief Operands of this expression */
490 Array<ExprDoc> operands;
491
492 void VisitAttrs(AttrVisitor* v) {
493 ExprDocNode::VisitAttrs(v);
494 v->Visit("kind", &kind);
495 v->Visit("operands", &operands);
496 }
497
498 static constexpr const char* _type_key = "script.printer.OperationDoc";
499 TVM_DECLARE_FINAL_OBJECT_INFO(OperationDocNode, ExprDocNode);
500};
