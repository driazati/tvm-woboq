32/*! \brief Default values in the TVMScript printer */
33struct Default {
34 /*! \brief Default data type of TIR buffer */
35 DataType buffer_dtype = DataType::Float(32);
36 /*! \brief Default data type of integer literals */
37 DataType int_dtype = DataType::Int(32);
38 /*!
39 * \brief Default data type of float literals. Right now we always print out the explicit type
40 * of floating point values, so setting it to Void means we do not print without the
41 * T.float32/T.float64 wrapper.
42 */
43 DataType float_dtype = DataType::Void();
44 /*! \brief Returns a singleton of the configuration */
45 static Default* Instance();
46 static DataType& BufferDType() { return Instance()->buffer_dtype; }
47 static DataType& IntDType() { return Instance()->int_dtype; }
48 static DataType& FloatDType() { return Instance()->float_dtype; }
49};
50
51/*!
52 * \brief The entry method for TVMScript printing
53 * \param obj The object to be printed
54 * \param ir_prefix The prefix of IR nodes
55 * \param indent_spaces Number of spaces used for indentation
56 * \param print_line_numbers Whether to print line numbers
57 * \param num_context_lines Number of context lines to print around the underlined text
58 * \param path_to_underline Object path to be underlined
59 * \return The TVMScript text format
60 */
61String Script(ObjectRef obj, //
62 Map<String, String> ir_prefix = {{"ir", "I"}, {"tir", "T"}}, //
63 int indent_spaces = 4, //
64 bool print_line_numbers = false, //
65 int num_context_lines = -1, //
66 Optional<ObjectPath> path_to_underline = NullOpt);
67
68/*!
69 * \brief Convert Doc into Python script.
70 * \param doc Doc to be converted
71 * \param indent_spaces Number of spaces used for indentation
72 * \param print_line_numbers Whether to print line numbers
73 * \param num_context_lines Number of context lines to print around the underlined text
74 * \param path_to_underline Object path to be underlined
75 */
76String DocToPythonScript(Doc doc, //
77 int indent_spaces = 4, //
78 bool print_line_numbers = false, //
79 int num_context_lines = -1, //
80 Optional<ObjectPath> path_to_underline = NullOpt);
