32/*!
33 * Frame is the core data structure for semantic information
34 * when printing IR graph into TVMScript code.
35 */
36class FrameNode : public Object {
37 public:
38 void VisitAttrs(tvm::AttrVisitor* v) {}
39
40 virtual ~FrameNode() = default;
41
42 /*!
43 * \brief Add a callback function to be called when this frame exits.
44 * \param cb The callback function. It should have signature void().
45 */
46 template <typename TCallback>
47 void AddExitCallback(TCallback&& cb) {
48 callbacks_.emplace_back(std::forward<TCallback>(cb));
49 }
50
51 /*!
52 * \brief Method that's called when Frame enters the scope.
53 */
54 virtual void EnterWithScope() {}
55
56 /*!
57 * \brief Method that's called when Frame exits the scope.
58 */
59 virtual void ExitWithScope() {
60 for (const std::function<void()>& callback : callbacks_) {
61 callback();
62 }
63 callbacks_.clear();
64 }
65
66 static constexpr const char* _type_key = "script.printer.Frame";
67 TVM_DECLARE_BASE_OBJECT_INFO(FrameNode, Object);
68
69 private:
70 std::vector<std::function<void()>> callbacks_;
71};
72
73/*!
74 * \brief Reference type of FrameNode
75 */
76class Frame : public ObjectRef {
77 protected:
78 Frame() = default;
79
80 public:
81 virtual ~Frame() = default;
82 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(Frame, ObjectRef, FrameNode);
83};
84
85/*!
86 * \brief MetadataFrame contains information like contant parameter array.
87 */
88class MetadataFrameNode : public FrameNode {
89 public:
90 Array<ObjectRef> metadata;
91
92 void VisitAttrs(tvm::AttrVisitor* v) {
93 FrameNode::VisitAttrs(v);
94 v->Visit("metadata", &metadata);
95 }
96
97 static constexpr const char* _type_key = "script.printer.MetadataFrame";
98 TVM_DECLARE_FINAL_OBJECT_INFO(MetadataFrameNode, FrameNode);
99};
100
101/*!
102 * \brief Reference type of MetadataFrameNode
103 */
104class MetadataFrame : public Frame {
105 public:
106 MetadataFrame();
107 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(MetadataFrame, Frame, MetadataFrameNode);
108};
109
110/*!
111 * \brief VarDefFrame contains information about the free variables that needs to be defined
112 * at the beginning of the printed snippet.
113 */
114class VarDefFrameNode : public FrameNode {
115 public:
116 Array<StmtDoc> stmts;
117
118 void VisitAttrs(tvm::AttrVisitor* v) {
119 FrameNode::VisitAttrs(v);
120 v->Visit("stmts", &stmts);
121 }
122
123 static constexpr const char* _type_key = "script.printer.VarDefFrame";
124 TVM_DECLARE_FINAL_OBJECT_INFO(VarDefFrameNode, FrameNode);
125};
126
127/*!
128 * \brief Reference type of VarDefFrameNode
129 */
130class VarDefFrame : public Frame {
131 public:
132 VarDefFrame();
133 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(VarDefFrame, Frame, VarDefFrameNode);
134};
