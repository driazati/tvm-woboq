41/*!
42 * Frame is the core data structure for semantic information
43 * when printing IR graph into TVMScript code.
44 */
45class FrameNode : public Object {
46 public:
47 /*! The docs generated in the frame */
48 Array<StmtDoc> stmts;
49 /*! The corresponding IRDocsifier */
50 IRDocsifierNode* d;
51 /*! The callbacks that are going to be invoked when the frame exits */
52 std::vector<std::function<void()>> callbacks;
53
54 void VisitAttrs(tvm::AttrVisitor* v) {
55 v->Visit("stmts", &stmts);
56 // `d` is not visited
57 // `callbacks` is not visited
58 }
59
60 static constexpr const char* _type_key = "script.printer.Frame";
61 TVM_DECLARE_BASE_OBJECT_INFO(FrameNode, Object);
62
63 public:
64 virtual ~FrameNode() = default;
65
66 /*!
67 * \brief Add a callback function to be called when this frame exits.
68 * \param cb The callback function. It should have signature void().
69 */
70 template <typename TCallback>
71 void AddExitCallback(TCallback&& cb) {
72 callbacks.emplace_back(std::forward<TCallback>(cb));
73 }
74 /*!
75 * \brief Add a dispatch token to the docsifier, and a callback that pops the token when this
76 * frame exits.
77 * \param d The docsifier.
78 * \param token The token to be added.
79 */
80 void AddDispatchToken(const IRDocsifier& d, const String& token);
81 /*!
82 * \brief Method that's called when Frame enters the scope.
83 */
84 virtual void EnterWithScope();
85 /*!
86 * \brief Method that's called when Frame exits the scope.
87 */
88 virtual void ExitWithScope();
89};
90
91/*!
92 * \brief Reference type of FrameNode
93 */
94class Frame : public ObjectRef {
95 protected:
96 Frame() = default;
97
98 public:
99 virtual ~Frame() = default;
100
101 /*! \brief Method that's called when Frame enters the scope. */
102 void EnterWithScope() { get()->EnterWithScope(); }
103
104 /*! \brief Method that's called when Frame exits the scope. */
105 void ExitWithScope() { get()->ExitWithScope(); }
106
107 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(Frame, ObjectRef, FrameNode);
108};

110//////////////////////// IRDocsifier ////////////////////////
111
112/*!
113 * \brief IRDocsifier is the top-level interface in the IR->Doc process.
114 *
115 * It provides methods to convert IR node object to Doc, operate on Frame
116 * objects and change dispatch tokens.
117 */
118class IRDocsifierNode : public Object {
119 public:
120 /*! \brief A function that creates the doc for a variable */
121 using DocCreator = std::function<ExprDoc()>;
122 /*! \brief Information about a variable, including its optional name and its doc creator */
123 struct VariableInfo {
124 /*! \brief The creator */
125 DocCreator creator;
126 /*! \brief The name of the variable */
127 Optional<String> name;
128 };
129 /*!
130 * \brief This map connects IR dispatch token to the name of identifier.
131 */
132 Map<String, String> ir_prefix;
133 /*!
134 * \brief The stack of frames.
135 * \sa FrameNode
136 */
137 Array<Frame> frames;
138 /*!
139 * \brief The stack of dispatch tokens.
140 *
141 * The dispatch token on the top decides which dispatch function to use
142 * when converting IR node object to Doc.
143 */
144 Array<String> dispatch_tokens;
145 /*! \brief The IRModule to be docsifier is handling */
146 Optional<IRModule> mod;
147 /*! \brief Mapping from a var to its info */
148 std::unordered_map<ObjectRef, VariableInfo, ObjectPtrHash, ObjectPtrEqual> obj2info;
149 /*! \brief The variable names used already */
150 std::unordered_set<String> defined_names;
151 /*! \brief Common prefixes of variable usages */
152 std::unordered_map<const Object*, std::vector<const Object*>> common_prefix;
153
154 void VisitAttrs(tvm::AttrVisitor* v) {
155 v->Visit("ir_prefix", &ir_prefix);
156 v->Visit("frames", &frames);
157 v->Visit("dispatch_tokens", &dispatch_tokens);
158 v->Visit("mod", &mod);
159 // `obj2info` is not visited
160 // `defined_names` is not visited
161 // `common_prefix` is not visited
162 }
163
164 static constexpr const char* _type_key = "script.printer.IRDocsifier";
165 TVM_DECLARE_FINAL_OBJECT_INFO(IRDocsifierNode, Object);
166
167 public:
168 /*!
169 * \brief Define variable by name.
170 * \param obj The variable object.
171 * \param frame The frame that this variable is defined in.
172 * \param name_hint The hint for variable name.
173 *
174 * \return The id doc for this variable.
175 *
176 * This function will rename the variable to avoid name conflict with other variables
177 * in the table.
178 */
179 IdDoc Define(const ObjectRef& obj, const Frame& frame, const String& name_hint);
180
181 /*!
182 * \brief Define variable by doc factory.
183 * \param obj The variable object.
184 * \param frame The frame that this variable is defined in.
185 * \param doc_factory The function to return an ExprDoc object for this variable.
186 *
187 * This function is a special form of `Define`. Variable is mapped to ExprDoc rather
188 * than IdDoc. It's useful when a variable is implicitly defined without a name, like
189 * the buf->data in TIR, which should be mapped to `AttrDoc(IdDoc("<buffer_name>"), "data")`.
190 *
191 * This function takes a DocFactory instead of Doc. It's because GetVarDoc needs to
192 * return a new Doc object every time it's called, as the returned doc will have
193 * different `source_path`. Currently there isn't a good way to deep copy a TVMObject
194 * so VarTable needs to call a factory function to get a freshly-constructed Doc object
195 * every time GetVarDoc is called.
196 */
197 void Define(const ObjectRef& obj, const Frame& frame, DocCreator doc_factory);
198
199 /*!
200 * \brief Get the doc for variable.
201 * \param obj The variable object.
202 *
203 * \return The doc for variable, if it exists in the table. Otherwise it returns NullOpt.
204 */
205 Optional<ExprDoc> GetVarDoc(const ObjectRef& obj) const;
206
207 /*!
208 * \brief Check if a variable exists in the table.
209 * \param obj The variable object.
210 *
211 * \return a boolean for whether variable exists.
212 */
213 bool IsVarDefined(const ObjectRef& obj) const;
214 /*! \brief Remove the variable defined */
215 void RemoveVar(const ObjectRef& obj);
216 /*!
217 * \brief Set the common prefix information of variable usage.
218 * \param root The root of the AST.
219 * \param is_var A function that returns true if the given object is considered a variable.
220 */
221 void SetCommonPrefix(const ObjectRef& root, runtime::TypedPackedFunc<bool(ObjectRef)> is_var);
222 /*!
223 * \brief Transform the input object into TDoc.
224 * \param obj The object to be transformed.
225 * \param path The path to this object.
226 *
227 * \return The Doc object.
228 */
229 template <class TDoc = Doc>
230 inline TDoc AsDoc(const ObjectRef& obj, const ObjectPath& path) const;
231};
232
233/*!
234 * \brief Reference type of IRDocsifierNode.
235 */
236class IRDocsifier : public ObjectRef {
237 public:
238 using FType = IRDocsifierFunctor<printer::Doc, ObjectPath, IRDocsifier>;
239 /*!
240 * \brief Create a IRDocsifier.
241 * \param ir_prefix The ir_prefix to use for this IRDocsifier.
242 */
243 explicit IRDocsifier(Map<String, String> ir_prefix);
244 /*! \brief The registration table for IRDocsifier. */
245 TVM_DLL static FType& vtable();
246
247 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(IRDocsifier, ObjectRef, IRDocsifierNode);
248};

250//////////////////////// Implementation ////////////////////////
251
252inline void FrameNode::EnterWithScope() {
253 if (d != nullptr) {
254 d->frames.push_back(GetRef<Frame>(this));
255 }
256}
257
258inline void FrameNode::ExitWithScope() {
259 for (const std::function<void()>& callback : callbacks) {
260 callback();
261 }
262 callbacks.clear();
263 if (d != nullptr) {
264 d->frames.pop_back();
265 }
266}
267
268template <class TDoc>
269inline TDoc IRDocsifierNode::AsDoc(const ObjectRef& obj, const ObjectPath& path) const {
270 if (!obj.defined()) {
271 return Downcast<TDoc>(LiteralDoc::None());
272 }
273 return Downcast<TDoc>(
274 IRDocsifier::vtable()(dispatch_tokens.back(), obj, path, GetRef<IRDocsifier>(this)));
275}
276
277inline void FrameNode::AddDispatchToken(const IRDocsifier& d, const String& token) {
278 d->dispatch_tokens.push_back(token);
279 this->AddExitCallback([doc = d.get()]() { doc->dispatch_tokens.pop_back(); });
280}
