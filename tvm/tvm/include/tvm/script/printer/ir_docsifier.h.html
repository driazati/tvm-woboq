35using WithCtx = With<ContextManager>;

37/*!
38 * \brief IRDocsifier is the top-level interface in the IR->Doc process.
39 *
40 * It provides methods to convert IR node object to Doc, operate on Frame
41 * objects and change dispatch tokens.
42 *
43 * Example usage:
44 * \code
45 * TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
46 * .set_dispatch([](TracedObject<tir::Var> obj, IRDocsifier p) { return IdDoc("x"); });
47 *
48 * TracedObject<tir::Var> var = ...;
49 * IRDocsifier p;
50 * p->AsDoc(var); // returns an IdDoc("x")
51 * \endcode
52 *
53 */
54class IRDocsifierNode : public Object {
55 public:
56 /*!
57 * \brief The var table to use during the printing process.
58 * \sa VarTableNode
59 */
60 VarTable vars;
61 /*!
62 * \brief The stack of frames.
63 * \sa FrameNode
64 */
65 Array<Frame> frames;
66 /*!
67 * \brief The stack of dispatch tokens.
68 *
69 * The dispatch token on the top decides which dispatch function to use
70 * when converting IR node object to Doc.
71 */
72 Array<String> dispatch_tokens;
73 /*!
74 * \brief This map connects IR dipatch token to the name of identifier.
75 */
76 Map<String, String> ir_prefix;
77
78 void VisitAttrs(tvm::AttrVisitor* v) {
79 v->Visit("vars", &vars);
80 v->Visit("frames", &frames);
81 v->Visit("dispatch_tokens", &dispatch_tokens);
82 v->Visit("ir_prefix", &ir_prefix);
83 }
84
85 static constexpr const char* _type_key = "script.printer.IRDocsifier";
86 TVM_DECLARE_FINAL_OBJECT_INFO(IRDocsifierNode, Object);
87
88 public:
89 /*!
90 * \brief Transform the input object into TDoc.
91 * \param obj The object to be transformed.
92 *
93 * \return The Doc object.
94 */
95 template <class TDoc>
96 TDoc AsDoc(const TracedObject<ObjectRef>& obj) const {
97 auto result = Downcast<TDoc>(AsDocImpl(obj));
98 result->source_paths.push_back(obj.GetPath());
99 return result;
100 }
101
102 /*!
103 * \brief Helper method to transform object into ExprDoc.
104 * \param obj The object to be transformed.
105 *
106 * \return The ExprDoc object.
107 */
108 ExprDoc AsExprDoc(const TracedObject<ObjectRef>& obj) { return AsDoc<ExprDoc>(obj); }
109
110 /*!
111 * \brief Push a new dispatch token into the stack
112 * \details The top dispatch token decides which dispatch table to use
113 * when printing Object. This method returns a RAII guard which
114 * pops the token when going out of the scope.
115 *
116 * \param token The dispatch token to push.
117 *
118 * \return A RAII guard to pop dispatch token when going out of scope.
119 */
120 WithCtx WithDispatchToken(const String& token) {
121 this->dispatch_tokens.push_back(token);
122 return WithCtx(nullptr, [this]() { this->dispatch_tokens.pop_back(); });
123 }
124
125 /*!
126 * \brief Push a new frame the stack
127 * \details Frame contains the contextual information that's needed during printing,
128 * for example, variables in the scope. This method returns a RAII guard which
129 * pops the frame and call the cleanup method of frame when going out of the scope.
130 *
131 * \param frame The frame to push.
132 *
133 * \return A RAII guard to pop frame and call the exit method of frame
134 * when going out of scope
135 */
136 WithCtx WithFrame(const Frame& frame) {
137 frame->EnterWithScope();
138 this->frames.push_back(frame);
139 return WithCtx(nullptr, [this, pushed_frame = frame]() {
140 Frame last_frame = this->frames.back();
141 ICHECK_EQ(last_frame, pushed_frame);
142 this->frames.pop_back();
143 last_frame->ExitWithScope();
144 });
145 }
146
147 /*!
148 * \brief Get the top frame with type FrameType
149 * \tparam FrameType The type of frame to get.
150 */
151 template <typename FrameType>
152 Optional<FrameType> GetFrame() const {
153 for (auto it = frames.rbegin(); it != frames.rend(); ++it) {
154 if (const auto* f = (*it).as<typename FrameType::ContainerType>()) {
155 return GetRef<FrameType>(f);
156 }
157 }
158 return NullOpt;
159 }
160
161 private:
162 Doc AsDocImpl(const TracedObject<ObjectRef>& obj) const;
163};

165/*!
166 * \brief Reference type of IRDocsifierNode.
167 */
168class IRDocsifier : public ObjectRef {
169 public:
170 /*!
171 * \brief Create a IRDocsifier.
172 * \param ir_prefix The ir_prefix to use for this IRDocsifier.
173 */
174 explicit IRDocsifier(Map<String, String> ir_prefix);
175
176 using FType = TracedObjectFunctor<printer::Doc, IRDocsifier>;
177 /*!
178 * \brief The registration table for IRDocsifier.
179 */
180 TVM_DLL static FType& vtable();
181
182 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(IRDocsifier, ObjectRef, IRDocsifierNode);
183};

185/*!
186 * \brief A wrapper object to provide injection point for printer of each IR.
187 *
188 * For any IR node to be transformed by IRDocsifier, it will be wrapped by RootNodeContainer
189 * and be dispatched to the corresponding function first. This provides an injection point for
190 * each IR's printer implemention to add specialized logic, for example, pushing a special
191 * Frame to the IRDocsifier before doing any IR->Doc transformation.
192 *
193 * \code
194 * TVM_STATIC_IR_FUNCTOR(IRDocsifier, vtable)
195 * .set_dispatch("relax", [](TracedObject<RootNodeContainer> obj, IRDocsifier p) {
196 * const ObjectRef& root_node = obj.Get()->root_node;
197 * // For example, relax printer can create a Frame specialized to Relax here
198 * RelaxGeneralFrame frame;
199 * auto ctx = p->WithFrame(frame);
200 * // More specialized logic for your IR.
201 * return p->AsDoc<Doc>(MakeTraced(root_node));
202 * });
203 * \endcode
204 */
205class RootNodeContainerNode : public Object {
206 public:
207 /*! \brief The root node to print. */
208 ObjectRef root_node;
209
210 void VisitAttrs(tvm::AttrVisitor* v) { v->Visit("root_node", &root_node); }
211
212 static constexpr const char* _type_key = "script.printer.RootNodeContainer";
213 TVM_DECLARE_FINAL_OBJECT_INFO(RootNodeContainerNode, Object);
214};
215
216class RootNodeContainer : public ObjectRef {
217 public:
218 /*!
219 * \brief Constructor of RootNodeContainer.
220 * \param root_node The root node to print.
221 * */
222 explicit RootNodeContainer(ObjectRef root_node);
223 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(RootNodeContainer, ObjectRef, RootNodeContainerNode);
224};
