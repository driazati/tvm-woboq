51template <typename T, bool IsObject = std::is_base_of<ObjectRef, T>::value>
52struct TracedObjectWrapperSelector;
53
54template <typename T>
55struct TracedObjectWrapperSelector<T, false> {
56 using Type = TracedBasicValue<T>;
57};
58
59template <typename T>
60struct TracedObjectWrapperSelector<T, true> {
61 using Type = TracedObject<T>;
62};
63
64template <typename K, typename V>
65struct TracedObjectWrapperSelector<Map<K, V>, true> {
66 using Type = TracedMap<K, V>;
67};
68
69template <typename T>
70struct TracedObjectWrapperSelector<Array<T>, true> {
71 using Type = TracedArray<T>;
72};
73
74template <typename T>
75struct TracedObjectWrapperSelector<Optional<T>, true> {
76 using Type = TracedOptional<T>;
77};

81/*!
82 * \brief Traced wrapper for regular (non-container) TVM objects.
83 */
84template <typename RefT>
85class TracedObject {
86 using ObjectType = typename RefT::ContainerType;
87
88 public:
89 using ObjectRefType = RefT;
90
91 // Don't use this direcly. For convenience, call MakeTraced() instead.
92 explicit TracedObject(const RefT& object_ref, ObjectPath path)
93 : ref_(object_ref), path_(std::move(path)) {}
94
95 // Implicit conversion from a derived reference class
96 template <typename DerivedRef>
97 TracedObject(const TracedObject<DerivedRef>& derived)
98 : ref_(derived.Get()), path_(derived.GetPath()) {}
99
100 /*!
101 * \brief Get a traced wrapper for an attribute of the wrapped object.
102 */
103 template <typename T, typename BaseType>
104 typename detail::TracedObjectWrapperSelector<T>::Type GetAttr(T BaseType::*member_ptr) const {
105 using WrapperType = typename detail::TracedObjectWrapperSelector<T>::Type;
106 const ObjectType* node = static_cast<const ObjectType*>(ref_.get());
107 const T& attr = node->*member_ptr;
108 Optional<String> attr_key = ICHECK_NOTNULL(GetAttrKeyByAddress(node, &attr));
109 return WrapperType(attr, path_->Attr(attr_key));
110 }
111
112 /*!
113 * \brief Access the wrapped object.
114 */
115 const RefT& Get() const { return ref_; }
116
117 /*!
118 * \brief Check if the reference to the wrapped object can be converted to `RefU`.
119 */
120 template <typename RefU>
121 bool IsInstance() const {
122 return ref_->template IsInstance<typename RefU::ContainerType>();
123 }
124
125 /*!
126 * \brief Same as Get().defined().
127 */
128 bool defined() const { return ref_.defined(); }
129
130 /*!
131 * \brief Convert the wrapped reference type to a subtype.
132 *
133 * Throws an exception if IsInstance<RefU>() is false.
134 */
135 template <typename RefU>
136 TracedObject<RefU> Downcast() const {
137 return TracedObject<RefU>(tvm::runtime::Downcast<RefU>(ref_), path_);
138 }
139
140 /*!
141 * \brief Convert the wrapped reference type to a subtype.
142 *
143 * Returns an empty optional if IsInstance<RefU>() is false.
144 */
145 template <typename RefU>
146 TracedOptional<RefU> TryDowncast() const {
147 if (ref_->template IsInstance<typename RefU::ContainerType>()) {
148 return Downcast<RefU>();
149 } else {
150 return TracedOptional<RefU>(NullOpt, path_);
151 }
152 }
153
154 /*!
155 * \brief Get the path of the wrapped object.
156 */
157 const ObjectPath& GetPath() const { return path_; }
158
159 private:
160 RefT ref_;
161 ObjectPath path_;
162};

164/*!
165 * \brief Iterator class for TracedMap<K, V>
166 */
167template <typename K, typename V>
168class TracedMapIterator {
169 public:
170 using WrappedV = typename detail::TracedObjectWrapperSelector<V>::Type;
171 using MapIter = typename Map<K, V>::iterator;
172
173 using iterator_category = std::bidirectional_iterator_tag;
174 using difference_type = ptrdiff_t;
175 using value_type = const std::pair<K, WrappedV>;
176 using pointer = value_type*;
177 using reference = value_type;
178
179 explicit TracedMapIterator(MapIter iter, ObjectPath map_path)
180 : iter_(iter), map_path_(std::move(map_path)) {}
181
182 bool operator==(const TracedMapIterator& other) const { return iter_ == other.iter_; }
183
184 bool operator!=(const TracedMapIterator& other) const { return iter_ != other.iter_; }
185
186 pointer operator->() const = delete;
187
188 reference operator*() const {
189 auto kv = *iter_;
190 return std::make_pair(kv.first, WrappedV(kv.second, map_path_->MapValue(kv.first)));
191 }
192
193 TracedMapIterator& operator++() {
194 ++iter_;
195 return *this;
196 }
197
198 TracedMapIterator operator++(int) {
199 TracedMapIterator copy = *this;
200 ++(*this);
201 return copy;
202 }
203
204 private:
205 MapIter iter_;
206 ObjectPath map_path_;
207};

209/*!
210 * \brief Traced wrapper for Map objects.
211 */
212template <typename K, typename V>
213class TracedMap {
214 public:
215 using WrappedV = typename detail::TracedObjectWrapperSelector<V>::Type;
216
217 using iterator = TracedMapIterator<K, V>;
218
219 // Don't use this direcly. For convenience, call MakeTraced() instead.
220 explicit TracedMap(Map<K, V> map, ObjectPath path)
221 : map_(std::move(map)), path_(std::move(path)) {}
222
223 /*!
224 * \brief Get a value by its key, wrapped in a traced wrapper.
225 */
226 WrappedV at(const K& key) const {
227 auto it = map_.find(key);
228 ICHECK(it != map_.end()) << "No such key in Map";
229 auto kv = *it;
230 return WrappedV(kv.second, path_->MapValue(kv.first));
231 }
232
233 /*!
234 * \brief Access the wrapped map object.
235 */
236 const Map<K, V>& Get() const { return map_; }
237
238 /*!
239 * \brief Get the path of the wrapped object.
240 */
241 const ObjectPath& GetPath() const { return path_; }
242
243 /*!
244 * \brief Get an iterator to the first item of the map.
245 */
246 iterator begin() const { return iterator(map_.begin(), path_); }
247
248 /*!
249 * \brief Get an iterator to the end of the map.
250 */
251 iterator end() const { return iterator(map_.end(), path_); }
252
253 /*!
254 * \brief Returns true iff the wrapped map is empty.
255 */
256 bool empty() const { return map_.empty(); }
257
258 private:
259 Map<K, V> map_;
260 ObjectPath path_;
261};

263/*!
264 * \brief Iterator class for TracedArray<T>
265 */
266template <typename T>
267class TracedArrayIterator {
268 public:
269 using WrappedT = typename detail::TracedObjectWrapperSelector<T>::Type;
270
271 using difference_type = ptrdiff_t;
272 using value_type = WrappedT;
273 using pointer = WrappedT*;
274 using reference = WrappedT&;
275 using iterator_category = std::random_access_iterator_tag;
276
277 explicit TracedArrayIterator(Array<T> array, size_t index, ObjectPath array_path)
278 : array_(array), index_(index), array_path_(array_path) {}
279
280 TracedArrayIterator& operator++() {
281 ++index_;
282 return *this;
283 }
284 TracedArrayIterator& operator--() {
285 --index_;
286 return *this;
287 }
288 TracedArrayIterator operator++(int) {
289 TracedArrayIterator copy = *this;
290 ++index_;
291 return copy;
292 }
293 TracedArrayIterator operator--(int) {
294 TracedArrayIterator copy = *this;
295 --index_;
296 return copy;
297 }
298
299 TracedArrayIterator operator+(difference_type offset) const {
300 return TracedArrayIterator(array_, index_ + offset, array_path_);
301 }
302
303 TracedArrayIterator operator-(difference_type offset) const {
304 return TracedArrayIterator(array_, index_ - offset, array_path_);
305 }
306
307 difference_type operator-(const TracedArrayIterator& rhs) const { return index_ - rhs.index_; }
308
309 bool operator==(TracedArrayIterator other) const {
310 return array_.get() == other.array_.get() && index_ == other.index_;
311 }
312 bool operator!=(TracedArrayIterator other) const { return !(*this == other); }
313 value_type operator*() const { return WrappedT(array_[index_], array_path_->ArrayIndex(index_)); }
314
315 private:
316 Array<T> array_;
317 size_t index_;
318 ObjectPath array_path_;
319};

321/*!
322 * \brief Traced wrapper for Array objects.
323 */
324template <typename T>
325class TracedArray {
326 public:
327 using WrappedT = typename detail::TracedObjectWrapperSelector<T>::Type;
328
329 using iterator = TracedArrayIterator<T>;
330
331 // Don't use this direcly. For convenience, call MakeTraced() instead.
332 explicit TracedArray(Array<T> array, ObjectPath path)
333 : array_(std::move(array)), path_(std::move(path)) {}
334
335 /*!
336 * \brief Access the wrapped array object.
337 */
338 const Array<T>& Get() const { return array_; }
339
340 /*!
341 * \brief Get the path of the wrapped array object.
342 */
343 const ObjectPath& GetPath() const { return path_; }
344
345 /*!
346 * \brief Get an element by index, wrapped in a traced wrapper.
347 */
348 WrappedT operator[](size_t index) const {
349 return WrappedT(array_[index], path_->ArrayIndex(index));
350 }
351
352 /*!
353 * \brief Get an iterator to the first array element.
354 *
355 * The iterator's dereference operator will automatically wrap each element in a traced wrapper.
356 */
357 iterator begin() const { return iterator(array_, 0, path_); }
358
359 /*!
360 * \brief Get an iterator to the end of the array.
361 *
362 * The iterator's dereference operator will automatically wrap each element in a traced wrapper.
363 */
364 iterator end() const { return iterator(array_, array_.size(), path_); }
365
366 /*!
367 * \brief Returns true iff the wrapped array is empty.
368 */
369 bool empty() const { return array_.empty(); }
370
371 /*!
372 * \brief Get the size of the wrapped array.
373 */
374 size_t size() const { return array_.size(); }
375
376 private:
377 Array<T> array_;
378 ObjectPath path_;
379};

381/*!
382 * \brief Traced wrapper for Optional objects.
383 */
384template <typename T>
385class TracedOptional {
386 public:
387 using WrappedT = typename detail::TracedObjectWrapperSelector<T>::Type;
388
389 /*!
390 * \brief Implicit conversion from the corresponding non-optional traced wrapper.
391 */
392 TracedOptional(const WrappedT& value) // NOLINT(runtime/explicit)
393 : optional_(value.Get().defined() ? value.Get() : Optional<T>(NullOpt)),
394 path_(value.GetPath()) {}
395
396 // Don't use this direcly. For convenience, call MakeTraced() instead.
397 explicit TracedOptional(Optional<T> optional, ObjectPath path)
398 : optional_(std::move(optional)), path_(std::move(path)) {}
399
400 /*!
401 * \brief Access the wrapped optional object.
402 */
403 const Optional<T>& Get() const { return optional_; }
404
405 /*!
406 * \brief Get the path of the wrapped optional object.
407 */
408 const ObjectPath& GetPath() const { return path_; }
409
410 /*!
411 * \brief Returns true iff the object is present.
412 */
413 bool defined() const { return optional_.defined(); }
414
415 /*!
416 * \brief Returns a non-optional traced wrapper, throws if defined() is false.
417 */
418 WrappedT value() const { return WrappedT(optional_.value(), path_); }
419
420 /*!
421 * \brief Same as defined().
422 */
423 explicit operator bool() const { return optional_.defined(); }
424
425 private:
426 Optional<T> optional_;
427 ObjectPath path_;
428};

430/*!
431 * \brief Traced wrapper for basic values (i.e. non-TVM objects)
432 */
433template <typename T>
434class TracedBasicValue {
435 public:
436 explicit TracedBasicValue(const T& value, ObjectPath path)
437 : value_(value), path_(std::move(path)) {}
438
439 /*!
440 * \brief Access the wrapped value.
441 */
442 const T& Get() const { return value_; }
443
444 /*!
445 * \brief Get the path of the wrapped value.
446 */
447 const ObjectPath& GetPath() const { return path_; }
448
449 /*!
450 * \brief Transform the wrapped value without changing its path.
451 */
452 template <typename F>
453 typename detail::TracedObjectWrapperSelector<typename std::invoke_result<F, const T&>::type>::Type
454 ApplyFunc(F&& f) const {
455 return MakeTraced(f(value_), path_);
456 }
457
458 private:
459 T value_;
460 ObjectPath path_;
461};
462
463/*!
464 * \brief Wrap the given root object in an appropriate traced wrapper class.
465 */
466template <typename RefT>
467typename detail::TracedObjectWrapperSelector<RefT>::Type MakeTraced(const RefT& object) {
468 using WrappedT = typename detail::TracedObjectWrapperSelector<RefT>::Type;
469 return WrappedT(object, ObjectPath::Root());
470}
471
472/*!
473 * \brief Wrap the given object with the given path in an appropriate traced wrapper class.
474 */
475template <typename RefT>
476typename detail::TracedObjectWrapperSelector<RefT>::Type MakeTraced(const RefT& object,
477 ObjectPath path) {
478 using WrappedT = typename detail::TracedObjectWrapperSelector<RefT>::Type;
479 return WrappedT(object, std::move(path));
480}
