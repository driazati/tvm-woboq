37/*
38 * This type alias and the following free functions are created to reduce the binary bloat
39 * from template and also hide implementation details from this header
40 */
41using DispatchTable = std::unordered_map<std::string, std::vector<runtime::PackedFunc>>;
42
43/*!
44 * \brief Get function from dispatch table.
45 * \param dispatch_table The dispatch table.
46 * \param token The dispatch token.
47 * \param type_index The type index of the Object type to be dispatched.
48 *
49 * \return The dispatch function.
50 */
51const runtime::PackedFunc& GetDispatchFunction(const DispatchTable& dispatch_table,
52 const String& token, uint32_t type_index);
53
54/*!
55 * \brief Set function in dispatch table.
56 * \param dispatch_table The dispatch table.
57 * \param token The dispatch token.
58 * \param type_index The type index of the Object type to be dispatched.
59 * \param f The dispatch function.
60 */
61void SetDispatchFunction(DispatchTable* dispatch_table, const String& token, uint32_t type_index,
62 runtime::PackedFunc f);
63
64/*!
65 * \brief Remove function from dispatch table.
66 * \param dispatch_table The dispatch table.
67 * \param token The dispatch token.
68 * \param type_index The TVM object type index for the dispatch function to be removed.
69 */
70void RemoveDispatchFunction(DispatchTable* dispatch_table, const String& token,
71 uint32_t type_index);
72
73constexpr const char* kDefaultDispatchToken = "";
74
75/*!
76 * \brief Dynamic dispatch functor based on TracedObject.
77 *
78 * This functor dispatches based on the type of object ref inside the input TracedObject,
79 * and the input dispatch token.
80 */
81template <typename R, typename... Args>
82class TracedObjectFunctor {
83 private:
84 using TSelf = TracedObjectFunctor<R, Args...>;
85
86 template <class TObjectRef, class TCallable>
87 using IsDispatchFunction =
88 typename std::is_convertible<TCallable, std::function<R(TracedObject<TObjectRef>, Args...)>>;
89
90 public:
91 /*!
92 * \brief Call the dispatch function.
93 * \param token The dispatch token.
94 * \param traced_object The traced object.
95 * \param args Other args.
96 *
97 * \return The return value of the dispatch function
98 *
99 * If the TObjectRef isn't registered with the token, it will try to find
100 * dispatch function for TObjectRef with kDefaultDispatchToken.
101 */
102 template <class TObjectRef>
103 R operator()(const String& token, TracedObject<TObjectRef> traced_object, Args... args) const {
104 const runtime::PackedFunc& dispatch_function =
105 GetDispatchFunction(dispatch_table_, token, traced_object.Get()->type_index());
106 return dispatch_function(traced_object.Get(), traced_object.GetPath(), args...);
107 }
108
109 /*!
110 * \brief Set the dispatch function
111 * \param token The dispatch token.
112 * \param type_index The TVM object type index for this dispatch function.
113 * \param f The dispatch function.
114 *
115 * This takes a type-erased packed function as input. It should be used
116 * through FFI boundary, for example, registering dispatch function from Python.
117 */
118 TSelf& set_dispatch(String token, uint32_t type_index, runtime::PackedFunc f) {
119 SetDispatchFunction(&dispatch_table_, token, type_index, std::move(f));
120 return *this;
121 }
122
123 /*!
124 * \brief Set the dispatch function
125 * \param token The dispatch token.
126 * \param f The dispatch function.
127 *
128 * The diaptch function should have signature `R(TracedObject<TObjectRef>, Args...)`.
129 */
130 template <typename TObjectRef, typename TCallable,
131 typename = std::enable_if_t<IsDispatchFunction<TObjectRef, TCallable>::value>>
132 TSelf& set_dispatch(String token, TCallable f) {
133 return set_dispatch(
134 token, //
135 TObjectRef::ContainerType::RuntimeTypeIndex(), //
136 runtime::TypedPackedFunc<R(TObjectRef, ObjectPath, Args...)>(
137 [f = std::move(f)](TObjectRef object, ObjectPath path, Args... args) -> R {
138 return f(MakeTraced(object, path), args...);
139 }));
140 }
141 /*!
142 * \brief Set the default dispatch function
143 * \param f The dispatch function.
144 *
145 * Default dispatch function will be used if there is no function registered
146 * with the requested dispatch token.
147 *
148 * Default dispatch function has an empty string as dispatch token.
149 */
150 template <typename TObjectRef, typename TCallable,
151 typename = std::enable_if_t<IsDispatchFunction<TObjectRef, TCallable>::value>>
152 TSelf& set_dispatch(TCallable&& f) {
153 return set_dispatch<TObjectRef>(kDefaultDispatchToken, std::forward<TCallable>(f));
154 }
155
156 /*!
157 * \brief Remove dispatch function
158 * \param token The dispatch token.
159 * \param type_index The TVM object type index for the dispatch function to be removed.
160 *
161 * This is useful when dispatch function comes from other language's runtime, and
162 * those function should be removed before that language runtime shuts down.
163 */
164 void remove_dispatch(String token, uint32_t type_index) {
165 RemoveDispatchFunction(&dispatch_table_, token, type_index);
166 }
167
168 private:
169 DispatchTable dispatch_table_;
170};
