40/*!
41 * \brief Helper template class to extract the type of first argument of a function
42 * \tparam FType The function type.
43 */
44template <typename FType>
45struct FirstArgTypeGetter;
46
47template <typename R, typename ArgOne, typename... OtherArgs>
48struct FirstArgTypeGetter<R(ArgOne, OtherArgs...)> {
49 using T = ArgOne;
50};
51
52/*!
53 * \brief Template alias for the type of first argument of a function
54 * \tparam FType The function type.
55 *
56 * The name of public functions are in snake case to be consistent with
57 * tvm/node/functor.h
58 */
59template <typename FType>
60using FirstArgType = typename detail::FirstArgTypeGetter<
61 typename tvm::runtime::detail::function_signature<FType>::FType>::T;
62} // namespace detail
63
64} // namespace
65
66/*
67 * This type alias and the following free functions are created to reduce the binary bloat
68 * from template and also hide implementation details from this header
69 */
70using DispatchTable = std::unordered_map<std::string, std::vector<runtime::PackedFunc>>;
71
72/*!
73 * \brief Get function from dispatch table.
74 * \param dispatch_table The dispatch table.
75 * \param token The dispatch token.
76 * \param type_index The type index of the Object type to be dispatched.
77 *
78 * \return The dispatch function.
79 */
80const runtime::PackedFunc& GetDispatchFunction(const DispatchTable& dispatch_table,
81 const String& token, uint32_t type_index);
82
83/*!
84 * \brief Set function in dispatch table.
85 * \param dispatch_table The dispatch table.
86 * \param token The dispatch token.
87 * \param type_index The type index of the Object type to be dispatched.
88 * \param f The dispatch function.
89 */
90void SetDispatchFunction(DispatchTable* dispatch_table, const String& token, uint32_t type_index,
91 runtime::PackedFunc f);
92
93/*!
94 * \brief Remove function from dispatch table.
95 * \param dispatch_table The dispatch table.
96 * \param token The dispatch token.
97 * \param type_index The TVM object type index for the dispatch function to be removed.
98 */
99void RemoveDispatchFunction(DispatchTable* dispatch_table, const String& token,
100 uint32_t type_index);
101
102constexpr const char* kDefaultDispatchToken = "";
103
104/*!
105 * \brief Dynamic dispatch functor based on TracedObject.
106 *
107 * This functor dispatches based on the type of object ref inside the input TracedObject,
108 * and the input dispatch token.
109 */
110template <typename R, typename... Args>
111class TracedObjectFunctor {
112 private:
113 using TSelf = TracedObjectFunctor<R, Args...>;
114
115 template <class TObjectRef, class TCallable>
116 using IsDispatchFunction =
117 typename std::is_convertible<TCallable, std::function<R(TracedObject<TObjectRef>, Args...)>>;
118
119 public:
120 /*!
121 * \brief Call the dispatch function.
122 * \param token The dispatch token.
123 * \param traced_object The traced object.
124 * \param args Other args.
125 *
126 * \return The return value of the dispatch function
127 *
128 * If the TObjectRef isn't registered with the token, it will try to find
129 * dispatch function for TObjectRef with kDefaultDispatchToken.
130 */
131 template <class TObjectRef>
132 R operator()(const String& token, TracedObject<TObjectRef> traced_object, Args... args) const {
133 const runtime::PackedFunc& dispatch_function =
134 GetDispatchFunction(dispatch_table_, token, traced_object.Get()->type_index());
135 return dispatch_function(traced_object.Get(), traced_object.GetPath(), args...);
136 }
137
138 /*!
139 * \brief Set the dispatch function
140 * \param token The dispatch token.
141 * \param type_index The TVM object type index for this dispatch function.
142 * \param f The dispatch function.
143 *
144 * This takes a type-erased packed function as input. It should be used
145 * through FFI boundary, for example, registering dispatch function from Python.
146 */
147 TSelf& set_dispatch(String token, uint32_t type_index, runtime::PackedFunc f) {
148 SetDispatchFunction(&dispatch_table_, token, type_index, std::move(f));
149 return *this;
150 }
151
152 /*!
153 * \brief Set the dispatch function
154 * \param token The dispatch token.
155 * \param f The dispatch function.
156 *
157 * The diaptch function should have signature `R(TracedObject<TObjectRef>, Args...)`.
158 */
159 template <typename TCallable,
160 typename TObjectRef = typename detail::FirstArgType<TCallable>::ObjectRefType,
161 typename = std::enable_if_t<IsDispatchFunction<TObjectRef, TCallable>::value>>
162 TSelf& set_dispatch(String token, TCallable f) {
163 return set_dispatch(
164 token, //
165 TObjectRef::ContainerType::RuntimeTypeIndex(), //
166 runtime::TypedPackedFunc<R(TObjectRef, ObjectPath, Args...)>(
167 [f = std::move(f)](TObjectRef object, ObjectPath path, Args... args) -> R {
168 return f(MakeTraced(object, path), args...);
169 }));
170 }
171 /*!
172 * \brief Set the default dispatch function
173 * \param f The dispatch function.
174 *
175 * Default dispatch function will be used if there is no function registered
176 * with the requested dispatch token.
177 *
178 * Default dispatch function has an empty string as dispatch token.
179 */
180 template <typename TCallable>
181 TSelf& set_dispatch(TCallable&& f) {
182 return set_dispatch(kDefaultDispatchToken, std::forward<TCallable>(f));
183 }
184
185 /*!
186 * \brief Remove dispatch function
187 * \param token The dispatch token.
188 * \param type_index The TVM object type index for the dispatch function to be removed.
189 *
190 * This is useful when dispatch function comes from other language's runtime, and
191 * those function should be removed before that language runtime shuts down.
192 */
193 void remove_dispatch(String token, uint32_t type_index) {
194 RemoveDispatchFunction(&dispatch_table_, token, type_index);
195 }
196
197 private:
198 DispatchTable dispatch_table_;
199};
