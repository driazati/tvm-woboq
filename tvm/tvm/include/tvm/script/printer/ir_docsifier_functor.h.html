28#include <unordered_map>
29#include <utility>

32namespace tvm {
33namespace script {
34namespace printer {

36/*!
37 * \brief Dynamic dispatch functor based on ObjectPath.
38 *
39 * This functor dispatches based on the type of object and the input dispatch token.
40 */
41template <typename R, typename... Args>
42class IRDocsifierFunctor {
43 private:
44 using TSelf = IRDocsifierFunctor<R, Args...>;
45
46 template <class TObjectRef, class TCallable>
47 using IsDispatchFunction =
48 typename std::is_convertible<TCallable, std::function<R(TObjectRef, Args...)>>;
49
50 public:
51 /*!
52 * \brief Call the dispatch function.
53 * \param token The dispatch token.
54 * \param obj The object.
55 * \param args Other args.
56 *
57 * \return The return value of the dispatch function
58 *
59 * If the TObjectRef isn't registered with the token, it will try to find
60 * dispatch function for TObjectRef with the default dispatch token (empty string).
61 */
62 template <class TObjectRef>
63 R operator()(const String& token, TObjectRef obj, Args... args) const {
64 uint32_t type_index = obj.defined() ? obj->type_index() : 0;
65 const runtime::PackedFunc* pf = nullptr;
66 if ((pf = LookupDispatchTable(token, type_index)) != nullptr) {
67 return (*pf)(obj, args...);
68 }
69 if ((pf = LookupDispatchTable("", type_index)) != nullptr) {
70 return (*pf)(obj, args...);
71 }
72 ICHECK(false) << "ObjectFunctor calls un-registered function on type: "
73 << runtime::Object::TypeIndex2Key(type_index) << " (token: " << token << ")"
74 << ". ObjectType: " << obj->GetTypeKey() << ". Object: " << obj;
75 }
76
77 /*!
78 * \brief Set the dispatch function
79 * \param token The dispatch token.
80 * \param type_index The TVM object type index for this dispatch function.
81 * \param f The dispatch function.
82 *
83 * This takes a type-erased packed function as input. It should be used
84 * through FFI boundary, for example, registering dispatch function from Python.
85 */
86 TSelf& set_dispatch(String token, uint32_t type_index, runtime::PackedFunc f) {
87 std::vector<runtime::PackedFunc>* table = &dispatch_table_[token];
88 if (table->size() <= type_index) {
89 table->resize(type_index + 1, nullptr);
90 }
91 runtime::PackedFunc& slot = (*table)[type_index];
92 if (slot != nullptr) {
93 ICHECK(false) << "Dispatch for type is already registered: "
94 << runtime::Object::TypeIndex2Key(type_index);
95 }
96 slot = f;
97 return *this;
98 }
99
100 /*!
101 * \brief Set the dispatch function
102 * \param token The dispatch token.
103 * \param f The dispatch function.
104 */
105 template <typename TObjectRef, typename TCallable,
106 typename = std::enable_if_t<IsDispatchFunction<TObjectRef, TCallable>::value>>
107 TSelf& set_dispatch(String token, TCallable f) {
108 return set_dispatch(token, TObjectRef::ContainerType::RuntimeTypeIndex(),
109 runtime::TypedPackedFunc<R(TObjectRef, Args...)>(f));
110 }
111
112 /*!
113 * \brief Remove dispatch function
114 * \param token The dispatch token.
115 * \param type_index The TVM object type index for the dispatch function to be removed.
116 *
117 * This is useful when dispatch function comes from other language's runtime, and
118 * those function should be removed before that language runtime shuts down.
119 */
120 void remove_dispatch(String token, uint32_t type_index) {
121 std::vector<runtime::PackedFunc>* table = &dispatch_table_[token];
122 if (table->size() <= type_index) {
123 return;
124 }
125 (*table)[type_index] = nullptr;
126 }
127
128 private:
129 /*!
130 * \brief Look up the dispatch table for the given token and type_index.
131 * \param token The dispatch token.
132 * \param type_index The TVM object type index.
133 * \return Returns the functor if the lookup succeeds, nullptr otherwise.
134 */
135 const runtime::PackedFunc* LookupDispatchTable(const String& token, uint32_t type_index) const {
136 auto it = dispatch_table_.find(token);
137 if (it == dispatch_table_.end()) {
138 return nullptr;
139 }
140 const std::vector<runtime::PackedFunc>& tab = it->second;
141 if (type_index >= tab.size()) {
142 return nullptr;
143 }
144 const PackedFunc* f = &tab[type_index];
145 if (f->defined()) {
146 return f;
147 } else {
148 return nullptr;
149 }
150 }
151 /*
152 * This type alias and the following free functions are created to reduce the binary bloat
153 * from template and also hide implementation details from this header
154 */
155 using DispatchTable = std::unordered_map<std::string, std::vector<runtime::PackedFunc>>;
156 /*! \brief The dispatch table. */
157 DispatchTable dispatch_table_;
158};
