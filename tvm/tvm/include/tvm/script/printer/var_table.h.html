35/*!
36 * \brief Variable Table manages mapping from variable object to ExprDoc during
37 * the process of printing TVMScript.
38 *
39 * The value type of this map is ExprDoc rather than IdDoc or String. It's
40 * because variables can be implicitly defined. For example in TIR buffer (tir::Buffer),
41 * `buf->data` is a variable, while its representation in TVMScript should be an
42 * expression `x.data`, where `x` is the variable for the buffer itself.
43 */
44class VarTableNode : public Object {
45 public:
46 void VisitAttrs(AttrVisitor*) {}
47
48 /*!
49 * \brief Define variable by name.
50 * \param obj The variable object.
51 * \param name_hint The hint for variable name.
52 * \param object_path The object_path for the returned ExprDoc.
53 * \param frame The frame that this variable is defined in.
54 *
55 * \return The id doc for this variable.
56 *
57 * This function will rename the variable to avoid name conflict with other variables
58 * in the table.
59 */
60 IdDoc Define(const ObjectRef& obj, const String& name_hint, const ObjectPath& object_path,
61 const Frame& frame);
62
63 /*!
64 * \brief Define variable by name.
65 * \param obj The variable object.
66 * \param name_hint The hint for variable name.
67 * \param frame The frame that this variable is defined in.
68 *
69 * \return The id doc for this variable.
70 *
71 * This is a shortcut version of `Define` which accepts a traced string.
72 */
73 IdDoc Define(const ObjectRef& obj, const TracedObject<String>& name_hint, const Frame& frame) {
74 return Define(obj, name_hint.Get(), name_hint.GetPath(), frame);
75 }
76
77 using DocFactory = std::function<ExprDoc()>;
78
79 /*!
80 * \brief Define variable by doc factory.
81 * \param obj The variable object.
82 * \param doc_factory The function to return an ExprDoc object for this variable.
83 * \param frame The frame that this variable is defined in.
84 *
85 * This function is a special form of `Define`. Variable is mapped to ExprDoc rather
86 * than IdDoc. It's useful when a variable is implicitly defined without a name, like
87 * the buf->data in TIR, which should be mapped to `AttrDoc(IdDoc("<buffer_name>"), "data")`.
88 *
89 * This function takes a DocFactory instead of Doc. It's because GetVarDoc needs to
90 * return a new Doc object every time it's called, as the returned doc will have
91 * different `soruce_path`. Currently there isn't a good way to deep copy a TVMObject
92 * so VarTable needs to call a factory function to get a freshly-constructed Doc object
93 * every time GetVarDoc is called.
94 */
95 void DefineByDoc(const ObjectRef& obj, DocFactory doc_factory, const Frame& frame);
96
97 /*!
98 * \brief Get the doc for variable.
99 * \param obj The variable object.
100 * \param object_path The object path for the variable.
101 *
102 * \return The doc for variable, if it exists in the table. Otherwise it returns NullOpt.
103 */
104 Optional<ExprDoc> GetVarDoc(const ObjectRef& obj, const ObjectPath& object_path) const;
105
106 /*!
107 * \brief Check if a variable exists in the table.
108 * \param obj The variable object.
109 *
110 * \return a boolean for whether variable exists.
111 */
112 bool IsVarDefined(const ObjectRef& obj) const;
113
114 static constexpr const char* _type_key = "script.printer.VarTable";
115 TVM_DECLARE_FINAL_OBJECT_INFO(VarTableNode, Object);
116
117 private:
118 void RemoveVar(const ObjectRef& obj);
119
120 struct VariableInfo {
121 DocFactory doc_factory;
122 Optional<String> name;
123 };
124 std::unordered_map<ObjectRef, VariableInfo, ObjectPtrHash, ObjectPtrEqual> obj2info;
125 std::unordered_set<String> defined_names;
126};
127
128/*!
129 * \brief Reference type of VarTableNode.
130 */
131class VarTable : public ObjectRef {
132 public:
133 /*!
134 * \brief Create an empty VarTable.
135 */
136 VarTable();
137 TVM_DEFINE_MUTABLE_NOTNULLABLE_OBJECT_REF_METHODS(VarTable, ObjectRef, VarTableNode);
138};
