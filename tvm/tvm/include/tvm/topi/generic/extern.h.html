25#define TVM_TOPI_GENERIC_EXTERN_H_

30#include <tvm/topi/detail/fuse.h>
31#include <tvm/topi/generic/injective.h>
32#include <tvm/topi/tags.h>
33
34namespace tvm {
35namespace topi {
36
37using namespace tvm::te;
38
39namespace generic {
40/*!
41 * \brief Schedule an extern op followed by injective operations
42 *
43 * \param target The target to generate a schedule for.
44 * \param outs The output tensors.
45 *
46 * \return A schedule for the op.
47 */
48inline Schedule schedule_extern(const Target& target, const Array<Tensor>& outs) {
49 Array<Operation> out_ops;
50 for (auto t : outs) {
51 out_ops.push_back(t->op);
52 }
53 auto s = create_schedule(out_ops);
54
55 tvm::te::AutoInlineInjective(s);
56 for (auto out : outs) {
57 if (out->op->IsInstance<ExternOpNode>()) {
58 continue;
59 }
60 tvm::GenericFunc::Get("schedule_injective_from_existing")(s, out);
61 }
62
63 return s;
64}
65
66} // namespace generic
67} // namespace topi
68} // namespace tvm
69#endif // TVM_TOPI_GENERIC_EXTERN_H_
