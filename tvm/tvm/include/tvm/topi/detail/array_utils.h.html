28
29namespace tvm {
30namespace topi {
31namespace detail {
32
33using namespace tvm::te;
34
35/*!
36 * \brief Search an array for a specific item
37 *
38 * \param array The array to search
39 * \param item The item to search for
40 *
41 * \return True iff the given array contains the given item.
42 */
43template <typename T>
44inline bool contains(Array<T> array, T item) {
45 for (auto& i : array) {
46 if (i == item) {
47 return true;
48 }
49 }
50 return false;
51}
