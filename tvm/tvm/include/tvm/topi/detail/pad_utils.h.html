39/*!
40 * \brief Get padding size for each side given padding height and width
41 *
42 * \param pad_h The amount to pad each of the top and bottom sides
43 * \param pad_w The amount to pad each of the left and right sides
44 *
45 * \return An array of 4 elements, representing padding sizes for
46 * each individual side. The array is in the order { top, left, bottom, right }
47 */
48inline Array<PrimExpr> GetPadTuple(PrimExpr pad_h, PrimExpr pad_w) {
49 pad_h *= 2;
50 pad_w *= 2;
51
52 auto pad_top = indexdiv(pad_h + 1, 2);
53 auto pad_left = indexdiv(pad_w + 1, 2);
54
55 return {pad_top, pad_left, pad_h - pad_top, pad_w - pad_left};
56}
