20/*!
21 * \file fuse.h
22 * \brief Fuse operation
23 */
24#ifndef TVM_TOPI_DETAIL_FUSE_H_
25#define TVM_TOPI_DETAIL_FUSE_H_
26
27#include <tvm/te/operation.h>
28
29namespace tvm {
30namespace topi {
31namespace detail {
32
33using namespace tvm::te;
34
35/*!
36 * \brief Fuse all of the given args
37 *
38 * \param stage The stage in which to apply the fuse
39 * \param args The iteration variables to be fused
40 *
41 * \return The fused iteration variable
42 */
43inline IterVar Fuse(Stage stage, const Array<IterVar>& args) {
44 IterVar res;
45 stage.fuse(args, &res);
46 return res;
47}
48
49} // namespace detail
50} // namespace topi
51} // namespace tvm
52#endif // TVM_TOPI_DETAIL_FUSE_H_
