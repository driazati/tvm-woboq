35/*! \brief Canonicalize an argument that may be Array<Expr> or int to Array<Expr> */
36inline Array<Integer> ArrayOrInt(TVMArgValue arg) {
37 if (arg.type_code() == kDLInt || arg.type_code() == kDLUInt) {
38 Array<Integer> result;
39 result.push_back(arg.operator int());
40 return result;
41 } else {
42 return arg;
43 }
44}
