35namespace tvm {
36namespace parser {
37
38/*! \brief A program source in any language.
39 *
40 * Could represent the source from an ML framework or a source
41 * representing a tvm::IRModule.
42 */
43class Source;
44
45class SourceNode : public Object {
46 public:
47 /*! \brief The source name. */
48 SourceName source_name;
49
50 /*! \brief The raw source. */
51 String source;
52
53 /*! \brief A mapping of line breaks into the raw source. */
54 std::vector<std::pair<int, int>> line_map;
55
56 // override attr visitor
57 void VisitAttrs(AttrVisitor* v) {
58 v->Visit("source_name", &source_name);
59 v->Visit("source", &source);
60 }
61
62 static constexpr const char* _type_key = "Source";
63 TVM_DECLARE_FINAL_OBJECT_INFO(SourceNode, Object);
64};
65
66class Source : public ObjectRef {
67 public:
68 TVM_DLL Source(SourceName src_name, std::string source);
69 TVM_DLL tvm::String GetLine(int line);
70
71 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(Source, ObjectRef, SourceNode);
72};
73
74/*!
75 * \brief A mapping from a unique source name to source fragment.
76 */
77class SourceMap;
78/*!
79 * \brief Stores locations in frontend source that generated a node.
80 */
81class SourceMapNode : public Object {
82 public:
83 /*! \brief The source mapping. */
84 Map<SourceName, Source> source_map;
85
86 // override attr visitor
87 void VisitAttrs(AttrVisitor* v) { v->Visit("source_map", &source_map); }
88
89 bool SEqualReduce(const SourceMapNode* other, SEqualReducer equal) const {
90 return equal(source_map, other->source_map);
91 }
92
93 static constexpr const char* _type_key = "SourceMap";
94 TVM_DECLARE_FINAL_OBJECT_INFO(SourceMapNode, Object);
95};
96
97class SourceMap : public ObjectRef {
98 public:
99 TVM_DLL SourceMap(Map<SourceName, Source> source_map);
100
101 TVM_DLL SourceMap(std::initializer_list<std::pair<SourceName, Source>> source_map)
102 : SourceMap(Map<SourceName, Source>(source_map)) {}
103
104 TVM_DLL SourceMap() : SourceMap(Map<SourceName, Source>()) {}
105
106 void Add(const Source& source);
107
108 SourceMapNode* operator->() {
109 ICHECK(get() != nullptr);
110 return static_cast<SourceMapNode*>(get_mutable());
111 }
112
113 TVM_DEFINE_NOTNULLABLE_OBJECT_REF_METHODS(SourceMap, ObjectRef, SourceMapNode);
114};
