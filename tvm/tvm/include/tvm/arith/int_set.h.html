41//-----------------------------------------------
42// Integer set data structure.
43//
44// This is a API build on top of the base
45// integer analysis API to provide set analysis.
46//------------------------------------------------
47/*!
48 * \brief Sign type of an integer expression.
49 */
50enum SignType { kPositive, kNegative, kZero, kUnknown };
51
52/*!
53 * \brief Base class of all Integer set containers.
54 * represent a set of integers in one dimension.
55 * \sa IntSet
56 */
57class IntSetNode : public Object {
58 public:
59 static constexpr const char* _type_key = "IntSet";
60 static constexpr bool _type_has_method_sequal_reduce = false;
61 TVM_DECLARE_BASE_OBJECT_INFO(IntSetNode, Object);
62};
63
64/*!
65 * \brief Managed reference to IntSetNode.
66 * \sa IntSetNode
67 */
68class IntSet : public ObjectRef {
69 public:
70 /*!
71 * \brief Find a range that covers the region.
72 * \param max_range The range to be covered.
73 * \return The covering range.
74 */
75 Range CoverRange(Range max_range) const;
76 /*! \return Lower bound of the set */
77 PrimExpr min() const;
78 /*! \return upper bound of the set */
79 PrimExpr max() const;
80 /*! \return The sign of the elements in the integer set */
81 SignType GetSignType() const;
82 /*! \return Whether the set represent nothing */
83 bool IsNothing() const;
84 /*! \return Whether the set represent everything */
85 bool IsEverything() const;
86 /*! \return Whether the set is a single point */
87 bool IsSinglePoint() const;
88 /*! \return Whether the set is proved to be bigger than 0 */
89 bool CanProvePositive() const;
90 /*! \return Whether the set is proved to be smaller than 0 */
91 bool CanProveNegative() const;
92 /*! \return Whether the set is proved to be smaller than or equal to 0 */
93 bool CanProveNonPositive() const;
94 /*! \return Whether the set is proved to be larger than or equal to 0 */
95 bool CanProveNonNegative() const;
96 /*! \return Whether the set has upper bound. */
97 bool HasUpperBound() const;
98 /*! \return Whether the set has lower bound. */
99 bool HasLowerBound() const;
100
101 /*!
102 * \brief The single point value, call only if IsSinglePoint is true
103 * \return The point value.
104 */
105 PrimExpr PointValue() const;
106 /*!
107 * \brief Try to match IntSet with range r.
108 *
109 * \note It is guanrateed that IntSet::FromRange(r).MatchRange(r) == true
110 * \return true if we can prove they are the same.
111 */
112 bool MatchRange(const tvm::Range& r) const;
113 /*! \return The set contains nothing */
114 static IntSet Nothing();
115 /*! \return The set contains everything */
116 static IntSet Everything();
117 /*!
118 * \brief construct a point set.
119 * \param point The point in the set.
120 * \return construct a single point set
121 */
122 static IntSet SinglePoint(PrimExpr point);
123 /*!
124 * \brief construct a integer set from vector expression.
125 * \param vec The vector expression, can also be single point.
126 * \return The result set containing the indices in the vector.
127 */
128 static IntSet Vector(PrimExpr vec);
129 /*!
130 * \brief Construct a set representing a range [min, min + extent).
131 * \param min The minimum of the range range
132 * \param extent The extent of the range.
133 * \return The constructed set.
134 */
135 static IntSet FromMinExtent(PrimExpr min, PrimExpr extent);
136 /*!
137 * \brief Construct a set representing a range.
138 * \param r The range
139 * \return The constructed set.
140 */
141 static IntSet FromRange(tvm::Range r);
142 /*!
143 * \brief Construct a set representing a interval.
144 * \param min The minimum value of the interval.
145 * \param max The maximum value of the interval.
146 * \return The constructed set.
147 */
148 static IntSet Interval(PrimExpr min, PrimExpr max);
149
150 TVM_DEFINE_OBJECT_REF_METHODS(IntSet, ObjectRef, IntSetNode);
151};

153//-----------------------------------------------
154// Integer set legacy API.
155//------------------------------------------------
156/*!
157 * \brief Convert std::unordered_map<const VarNode*, IntSet> to Map<Var, IntSet>
158 *
159 * \param dom_map The domain map to convert.
160 * \return The converted map.
161 */
162Map<Var, IntSet> ConvertDomMap(const std::unordered_map<const VarNode*, IntSet>& dom_map);
163/*!
164 * \brief Find an symbolic integer set that contains all possible values of
165 * e given the domain of each iteration variables.
166 *
167 * \param e The expression to be evaluated.
168 * \param dom_map The domain of each variable.
169 * \return An integer set that can cover all the possible values of e.
170 */
171IntSet EvalSet(PrimExpr e, const Map<IterVar, IntSet>& dom_map);
172/*!
173 * \brief Same as EvalSet, but takes unordered_map
174 *
175 * \param e The expression to be evaluated.
176 * \param dom_map The domain of each variable.
177 * \return An integer set that can cover all the possible values of e.
178 */
179IntSet EvalSet(PrimExpr e, const std::unordered_map<const tir::VarNode*, IntSet>& dom_map);
180/*!
181 * \brief Find an symbolic integer set that contains is union over
182 * all the possible conditional values in dom_map.
183 *
184 * \param r The initial range.
185 * \param dom_map The domain of each variable.
186 * \return An integer set that can cover all the possible values.
187 */
188IntSet EvalSet(Range r, const Map<IterVar, IntSet>& dom_map);
189
190/*!
191 * \brief Find an symbolic integer set that contains is union over
192 * all the possible conditional values in dom_map.
193 *
194 * \param s The initial set.
195 * \param dom_map The domain of each variable.
196 * \return An integer set that can cover all the possible values.
197 */
198IntSet EvalSet(IntSet s, const std::unordered_map<const VarNode*, IntSet>& dom_map);
199/*!
200 * \brief Same as EvalSet, but takes unordered_map
201 *
202 * \param r The range to be evaluated.
203 * \param dom_map The domain of each variable.
204 * \return An integer set that can cover all the possible values of e.
205 */
206IntSet EvalSet(Range r, const std::unordered_map<const VarNode*, IntSet>& dom_map);
207/*!
208 * \brief Same as EvalSet, but takes Array<Range>
209 *
210 * \param region The range to be evaluated.
211 * \param dom_map The domain of each variable.
212 * \return An array of integer sets that can cover all the possible values.
213 */
214Array<IntSet> EvalSet(const Array<Range>& region, const Map<Var, IntSet>& dom_map);
215/*! \brief Map from Expr to IntSet */
216using ExprIntSetMap = std::unordered_map<PrimExpr, IntSet, ObjectPtrHash, ObjectPtrEqual>;
217/*!
218 * \brief Find the integer set of every sub-expression, given the
219 * domain of each iteration variables.
220 *
221 * \param e The expression to be evaluated.
222 * \param dom_map The domain of each variable.
223 * \return the map from the expression to its possible value.
224 */
225ExprIntSetMap EvalSetForEachSubExpr(PrimExpr e,
226 const std::unordered_map<const VarNode*, IntSet>& dom_map);
227
228/*!
229 * \brief Create a union set of all sets, possibly relaxed
230 * \param sets The sets to be combined
231 * \return the set after union
232 */
233IntSet Union(const Array<IntSet>& sets);
234
235/*!
236 * \brief The union of N-dimensional integer sets
237 * \param nd_int_sets A list of N-dimensional integer sets
238 * \return An N-dimensional integer set as the result of union
239 */
240Array<IntSet> UnionRegion(const Array<Array<IntSet>>& nd_int_sets);
241
242/*!
243 * \brief Create a lower-bound of union set, where some of the segments may be dropped
244 * \param sets The sets to be combined
245 * \return the set after union
246 */
247IntSet UnionLowerBound(const Array<IntSet>& sets);
248
249/*!
250 * \brief The union of N-dimensional integer sets
251 * \param nd_int_sets A list of N-dimensional integer sets
252 * \return An N-dimensional integer set as the result of union
253 */
254Array<IntSet> UnionRegionLowerBound(const Array<Array<IntSet>>& nd_int_sets);
255
256/*!
257 * \brief Create an intersected set of all sets
258 * \param sets The sets to be intersected
259 * \return the set after intersected
260 */
261IntSet Intersect(const Array<IntSet>& sets);
262
263/*!
264 * \brief Converts the Ranges to IntSets
265 * \param var_dom The ranges of variables
266 * \return The integer sets of the variables
267 */
268Map<Var, arith::IntSet> AsIntSet(const Map<Var, Range>& var_dom);
269
270/*!
271 * \brief Analyze the region with affine map, given the domain of variables and their predicate.
272 * The result should be strict, i.e. no region is discarded or relaxed.
273 * \param region The region to be analyzed
274 * \param var_dom The ranges of the variables
275 * \param predicate The predicate for the affine map
276 * \param analyzer The analyzer used
277 * \return NullOpt if the detection fails, or an array of arith::IntSet as the result of analysis
278 */
279TVM_DLL Optional<Array<IntSet>> EstimateRegionStrictBound(const Array<Range>& region,
280 const Map<Var, Range>& var_dom,
281 const PrimExpr& predicate,
282 arith::Analyzer* analyzer);
283
284/*!
285 * \brief Analyze the region with affine map, given the domain of variables and their predicate.
286 * Some subregion may be discarded during the lower-bound analysis.
287 * \param region The region to be analyzed
288 * \param var_dom The ranges of the variables
289 * \param predicate The predicate for the affine map
290 * \param analyzer The analyzer used
291 * \return NullOpt if the detection fails, or an array of arith::IntSet as the result of analysis
292 */
293TVM_DLL Optional<Array<IntSet>> EstimateRegionLowerBound(const Array<Range>& region,
294 const Map<Var, Range>& var_dom,
295 const PrimExpr& predicate,
296 arith::Analyzer* analyzer);
297
298/*!
299 * \brief Analyze the region with affine map, given the domain of variables and their predicate
300 * Relaxation of the region may be used in upper-bound analysis, i.e. some extra region may be added
301 * to the result.
302 * \param region The region to be analyzed
303 * \param var_dom The ranges of the variables
304 * \param predicate The predicate for the affine map
305 * \param analyzer The analyzer used
306 * \return an array of arith::IntSet as the result of analysis
307 */
308TVM_DLL Array<IntSet> EstimateRegionUpperBound(const Array<Range>& region,
309 const Map<Var, Range>& var_dom,
310 const PrimExpr& predicate,
311 arith::Analyzer* analyzer);
