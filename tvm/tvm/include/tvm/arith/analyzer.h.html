62/*!
63 * \brief Constant integer up and lower bound(inclusive).
64 * Useful for value bound analysis.
65 *
66 * set = [min_value, max_value]
67 */
68class ConstIntBoundNode : public Object {
69 public:
70 int64_t min_value;
71 int64_t max_value;
72
73 void VisitAttrs(tvm::AttrVisitor* v) {
74 v->Visit("min_value", &min_value);
75 v->Visit("max_value", &max_value);
76 }
77
78 bool SEqualReduce(const ConstIntBoundNode* other, SEqualReducer equal) const {
79 return equal(min_value, other->min_value) && equal(max_value, other->max_value);
80 }
81
82 /*! \brief Number to represent +inf */
83 static const constexpr int64_t kPosInf = std::numeric_limits<int64_t>::max();
84 /*!
85 * \brief Number to represent -inf
86 * \note We can make use the of fact that -kPosInf == kNegInf in the project.
87 */
88 static const constexpr int64_t kNegInf = -kPosInf;
89
90 static constexpr const char* _type_key = "arith.ConstIntBound";
91 TVM_DECLARE_FINAL_OBJECT_INFO(ConstIntBoundNode, Object);
92};

168/*!
169 * \brief Range of a linear integer function.
170 * Use to do specify the possible index values.
171 *
172 * set = { coeff * x + base | x in Z }
173 *
174 * When coeff != 0, it can also be written as
175 * set = { n | n % coeff == base }
176 *
177 * This is useful to decide if the index is dividable by certain value.
178 * For example, if index = 0 + 4 x, then we know it can be divided by 4.
179 */
180class ModularSetNode : public Object {
181 public:
182 /*! \brief linear co-efficient */
183 int64_t coeff;
184 /*! \brief The base */
185 int64_t base;
186
187 void VisitAttrs(tvm::AttrVisitor* v) {
188 v->Visit("coeff", &coeff);
189 v->Visit("base", &base);
190 }
191
192 bool SEqualReduce(const ModularSetNode* other, SEqualReducer equal) const {
193 return equal(coeff, other->coeff) && equal(base, other->base);
194 }
195
196 static constexpr const char* _type_key = "arith.ModularSet";
197 TVM_DECLARE_FINAL_OBJECT_INFO(ModularSetNode, Object);
198};

320/*!
321 * \brief Constraint context.
322 *
323 * \code
324 *
325 * Var("x");
326 * arith::Analyzer analyzer;
327 * {
328 * With<arith::ConstraintContext> scope(&analyzer, x % 3 == 0);
329 * ICHECK_EQ(analyzer.modular_set(x)->coeff, 3);
330 * }
331 * // constraint no longer in effect.
332 * ICHECK_NE(analyzer.modular_set(x)->coeff, 3);
333 *
334 * \endcode
335 */
336class ConstraintContext {
337 private:
338 // declare friend to enable with.
339 friend class With<ConstraintContext>;
340 /*!
341 * \brief Construct a constraint context.
342 * \param analyzer The analyzer.
343 * \param constraint The constraint to be applied.
344 */
345 ConstraintContext(Analyzer* analyzer, PrimExpr constraint)
346 : analyzer_(analyzer), constraint_(constraint) {}
347 // enter the scope.
348 void EnterWithScope();
349 // exit the scope.
350 void ExitWithScope();
351 /*! \brief The analyzer */
352 Analyzer* analyzer_;
353 /*! \brief The constraint */
354 PrimExpr constraint_;
355 /*! \brief function to be called in recovery */
356 std::vector<std::function<void()>> recovery_functions_;
357};

413/*!
414 * \brief Analyzer that contains bunch of sub-analyzers.
415 *
416 * Each sub-analyzer can make use of another sub-analyzer
417 * by weak reference of this.
418 *
419 * NOTE for sub-analyzer developers:
420 * If the analyzer uses memoization, we need to clear the internal
421 * cache when information about a Var has been overridden.
422 */
423class TVM_DLL Analyzer {
424 public:
425 /*
426 * Disable copy constructor.
427 */
428 Analyzer(const Analyzer&) = delete;
429 Analyzer& operator=(const Analyzer&) = delete;
430 /*! \brief sub-analyzer: const integer bound */
431 ConstIntBoundAnalyzer const_int_bound;
432 /*! \brief sub-analyzer: modular set */
433 ModularSetAnalyzer modular_set;
434 /*! \brief sub-analyzer rewrite simplify */
435 RewriteSimplifier rewrite_simplify;
436 /*! \brief sub-analyzer canonical simplify */
437 CanonicalSimplifier canonical_simplify;
438 /*! \brief sub-analyzer: int set */
439 IntSetAnalyzer int_set;
440 /*! \brief constructor */
441 Analyzer();
442 /*!
443 * \brief Notify all the sub-analyzers that var
444 * is created and binded to expr.
445 *
446 * Each var can only be bound once.
447 *
448 * \param var The variable.
449 * \param expr The expression we bind to.
450 * \param allow_override Whether we allow overriding an existing var's
451 * expression. This option should not be used if there is any dependency
452 * between variables.
453 */
454 void Bind(const Var& var, const PrimExpr& expr, bool allow_override = false);
455 /*!
456 * \brief Notify all the sub-analyzers that var
457 * is created and binded to a range.
458 *
459 * Each var can only be binded once.
460 *
461 * \param var The variable.
462 * \param range The range we bind to.
463 * \param allow_override Whether we allow overriding an existing var's
464 * expression. This option should not be used if there is any dependency
465 * between variables.
466 */
467 void Bind(const Var& var, const Range& range, bool allow_override = false);
468 /*!
469 * \brief Bind all the vars in the Map
470 *
471 * \param variables The {variable -> range} map.
472 * \param allow_override Whether we allow overriding an existing var's
473 * expression. This option should not be used if there is any dependency
474 * between variables.
475 */
476 void Bind(const Map<Var, Range>& variables, bool allow_override = false);
477 /*!
478 * \brief Whether can we prove expr >= val.
479
480 * Non-negative proof is very useful in integer analysis
481 * to lower divisions and mods given difference in trunc and ceil mode.
482 *
483 * \param expr The expression.
484 * \param lower_bound The lower bound.
485 * \return Whether we can prove it.
486 *
487 * \note Analyzer will call into sub-analyzers to get the result.
488 */
489 bool CanProveGreaterEqual(const PrimExpr& expr, int64_t lower_bound);
490 /*!
491 * \brief Whether can we prove expr < val.
492
493 * Non-negative proof is very useful in integer analysis
494 * to lower divisions and mods given difference in trunc and ceil mode.
495 *
496 * \param expr The expression.
497 * \param upper_bound The upper bound.
498 * \return Whether we can prove it.
499 *
500 * \note Analyzer will call into sub-analyzers to get the result.
501 */
502 bool CanProveLess(const PrimExpr& expr, int64_t upper_bound);
503 /*!
504 * \brief Whether can we prove lhs == rhs.
505 *
506 * \param lhs The input lhs.
507 * \param rhs The input rhs.
508 * \return Whether we can prove lhs == rhs.
509 *
510 * \note Analyzer will call into sub-analyzers to get the result.
511 */
512 bool CanProveEqual(const PrimExpr& lhs, const PrimExpr& rhs);
513 /*!
514 * \brief Whether can we prove condition.
515 *
516 * \param cond The expression to be proved.
517 * \return The result.
518 *
519 * \note Analyzer will call into sub-analyzers to get the result.
520 */
521 bool CanProve(const PrimExpr& cond);
522 /*!
523 * \brief Simplify expr.
524 *
525 * \param expr The expression to be simplified.
526 * \param steps The simplification runs in the order of
527 * rewrite_simplify (step 1) -> canonical_simplify (step 2) ->
528 * rewrite_simplify (step 3) -> canonical_simplify (step 4) -> ...
529 * param steps controls how many steps to run.
530 * Default is 2, i.e., rewrite_simplify + canonical_simplify.
531 * \return The result.
532 *
533 * \note Analyzer will call into sub-analyzers to get the result.
534 */
535 PrimExpr Simplify(const PrimExpr& expr, int steps = 2);
536};
