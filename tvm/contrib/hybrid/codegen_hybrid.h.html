39namespace tvm {
40namespace contrib {
41
42using namespace te;
43using namespace tir;
44/*!
45 * \brief A base class to generate Hybrid Script.
46 *
47 * **NOTE** CodeGenHybrid does not aim at generating Python scripts consumed by Python2/3.
48 * For runtime support, please refer the decorator in ``tvm/python/hybrid/api.py``.
49 */
50class CodeGenHybrid : public ExprFunctor<void(const PrimExpr&, std::ostream&)>,
51 public StmtFunctor<void(const Stmt&)> {
52 public:
53 /*!
54 * \brief Dump the given function body to hybrid script.
55 * \param stmt The function body to be dumped to hybrid script.
56 * \param inputs Input tensors of this schedule.
57 * \param outputs Output tensors of this schedule.
58 * \param name The name of the function.
59 */
60 void DumpStmt(const Stmt& stmt, const Array<ObjectRef>& inputs, const Array<Tensor>& outputs,
61 const std::string& name = "hybrid_func");
62 /*!
63 * \brief Finalize the compilation and return the code.
64 * \return The code.
65 */
66 std::string Finish();
67 /*! \brief Reserve keywords in avoid of name conflict. */
68 void ReserveKeywords();
69 /*!
70 * \brief Print the Stmt n to CodeGenHybrid->stream
71 * \param n The statement to be printed.
72 */
73 void PrintStmt(const Stmt& n) { this->VisitStmt(n); }
74 /*!
75 * \brief Print the expression n(or its ssa id if in ssa mode) into os
76 * \param n The expression to be printed.
77 * \param os The output stream
78 */
79 void PrintExpr(const PrimExpr& n, std::ostream& os) { this->VisitExpr(n, os); }
80 /*!
81 * \brief Same as PrintExpr, but simply returns result string
82 * \param n The expression to be printed.
83 */
84 std::string PrintExpr(const PrimExpr& n) {
85 std::ostringstream os;
86 PrintExpr(n, os);
87 return os.str();
88 }
89 // expression
90 void VisitExpr_(const VarNode* op, std::ostream& os) override; // NOLINT(*)
91 void VisitExpr_(const LoadNode* op, std::ostream& os) override; // NOLINT(*)
92 void VisitExpr_(const BufferLoadNode* op, std::ostream& os) override; // NOLINT(*)
93 void VisitExpr_(const LetNode* op, std::ostream& os) override; // NOLINT(*)
94 void VisitExpr_(const CallNode* op, std::ostream& os) override; // NOLINT(*)
95 void VisitExpr_(const ProducerLoadNode* op, std::ostream& os) override; // NOLINT(*)
96 void VisitExpr_(const AddNode* op, std::ostream& os) override; // NOLINT(*)
97 void VisitExpr_(const SubNode* op, std::ostream& os) override; // NOLINT(*)
98 void VisitExpr_(const MulNode* op, std::ostream& os) override; // NOLINT(*)
99 void VisitExpr_(const DivNode* op, std::ostream& os) override; // NOLINT(*)
100 void VisitExpr_(const ModNode* op, std::ostream& os) override; // NOLINT(*)
101 void VisitExpr_(const FloorDivNode* op, std::ostream& os) override; // NOLINT(*)
102 void VisitExpr_(const FloorModNode* op, std::ostream& os) override; // NOLINT(*)
103 void VisitExpr_(const MinNode* op, std::ostream& os) override; // NOLINT(*)
104 void VisitExpr_(const MaxNode* op, std::ostream& os) override; // NOLINT(*)
105 void VisitExpr_(const EQNode* op, std::ostream& os) override; // NOLINT(*)
106 void VisitExpr_(const NENode* op, std::ostream& os) override; // NOLINT(*)
107 void VisitExpr_(const LTNode* op, std::ostream& os) override; // NOLINT(*)
108 void VisitExpr_(const LENode* op, std::ostream& os) override; // NOLINT(*)
109 void VisitExpr_(const GTNode* op, std::ostream& os) override; // NOLINT(*)
110 void VisitExpr_(const GENode* op, std::ostream& os) override; // NOLINT(*)
111 void VisitExpr_(const AndNode* op, std::ostream& os) override; // NOLINT(*)
112 void VisitExpr_(const OrNode* op, std::ostream& os) override; // NOLINT(*)
113 void VisitExpr_(const CastNode* op, std::ostream& os) override; // NOLINT(*)
114 void VisitExpr_(const NotNode* op, std::ostream& os) override; // NOLINT(*)
115 void VisitExpr_(const SelectNode* op, std::ostream& os) override; // NOLINT(*)
116 void VisitExpr_(const RampNode* op, std::ostream& os) override; // NOLINT(*)
117 void VisitExpr_(const BroadcastNode* op, std::ostream& os) override; // NOLINT(*)
118 void VisitExpr_(const IntImmNode* op, std::ostream& os) override; // NOLINT(*)
119 void VisitExpr_(const FloatImmNode* op, std::ostream& os) override; // NOLINT(*)
120 void VisitExpr_(const StringImmNode* op, std::ostream& os) override; // NOLINT(*)
121 // statment
122 void VisitStmt_(const LetStmtNode* op) override;
123 void VisitStmt_(const StoreNode* op) override;
124 void VisitStmt_(const BufferStoreNode* op) override;
125 void VisitStmt_(const ProducerStoreNode* op) override;
126 void VisitStmt_(const ForNode* op) override;
127 void VisitStmt_(const IfThenElseNode* op) override;
128 void VisitStmt_(const AllocateNode* op) override;
129 void VisitStmt_(const ProducerRealizeNode* op) override;
130 void VisitStmt_(const AttrStmtNode* op) override;
131 void VisitStmt_(const AssertStmtNode* op) override;
132 void VisitStmt_(const EvaluateNode* op) override;
133 void VisitStmt_(const SeqStmtNode* op) override;
134 /*!
135 * \brief Print Type represetnation of type t.
136 * \param t The type representation.
137 * \param os The stream to print the ctype into
138 */
139 virtual void PrintType(DataType t, std::ostream& os); // NOLINT(*)
140
141 private:
142 /*! \brief The current indent of the code dump. */
143 int indent_{0};
144 /*! \brief The tab size of code indent. */
145 const int tab_{4};
146 /*! \brief Print the current indent spaces. */
147 inline void PrintIndent();
148 /*! \brief Keys are ids allocated, and values are the suffix to prevent double-name. */
149 std::map<std::string, int> ids_allocated_;
150 /*!
151 * \brief Keys are either (tensors, value_index) or (variables, 0).
152 * Values are the corresponding IDs.*/
153 std::map<std::pair<const Object*, int>, std::string> id_map_;
154 /*! \brief Variables (keys) binded to the threads (values). */
155 std::map<const VarNode*, std::string> binds_;
156 /*!
157 * \brief Find an unallocated name for the given prefix.
158 * \param prefix The given prefix.
159 */
160 std::string GetUniqueName(std::string prefix);
161 /*! \brief The output code string builder. */
162 std::stringstream stream;
163 /*!
164 * \brief Get or allocate the ID for the given variable.
165 * \param v The given variable.
166 */
167 std::string GetVarID(const VarNode* v);
168 /*!
169 * \brief Get or allocate the ID for the given tensor.
170 * \param tensor The tensor to allocate a name.
171 */
172 std::string GetTensorID(const Tensor& tensor);
173};
174
175} // namespace contrib
176} // namespace tvm
