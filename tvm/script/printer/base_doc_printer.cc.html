26DocPrinter::DocPrinter(int indent_spaces) : indent_spaces_(indent_spaces) {}
27
28void DocPrinter::Append(const Doc& doc) { PrintDoc(doc); }
29
30String DocPrinter::GetString() const {
31 std::string text = output_.str();
32 if (!text.empty() && text.back() != '\n') {
33 text.push_back('\n');
34 }
35 return text;
36}

38void DocPrinter::PrintDoc(const Doc& doc) {
39 if (const auto* doc_node = doc.as<LiteralDocNode>()) {
40 PrintTypedDoc(GetRef<LiteralDoc>(doc_node));
41 } else if (const auto* doc_node = doc.as<IdDocNode>()) {
42 PrintTypedDoc(GetRef<IdDoc>(doc_node));
43 } else if (const auto* doc_node = doc.as<AttrAccessDocNode>()) {
44 PrintTypedDoc(GetRef<AttrAccessDoc>(doc_node));
45 } else if (const auto* doc_node = doc.as<IndexDocNode>()) {
46 PrintTypedDoc(GetRef<IndexDoc>(doc_node));
47 } else if (const auto* doc_node = doc.as<OperationDocNode>()) {
48 PrintTypedDoc(GetRef<OperationDoc>(doc_node));
49 } else if (const auto* doc_node = doc.as<CallDocNode>()) {
50 PrintTypedDoc(GetRef<CallDoc>(doc_node));
51 } else if (const auto* doc_node = doc.as<LambdaDocNode>()) {
52 PrintTypedDoc(GetRef<LambdaDoc>(doc_node));
53 } else if (const auto* doc_node = doc.as<ListDocNode>()) {
54 PrintTypedDoc(GetRef<ListDoc>(doc_node));
55 } else if (const auto* doc_node = doc.as<TupleDocNode>()) {
56 PrintTypedDoc(GetRef<TupleDoc>(doc_node));
57 } else if (const auto* doc_node = doc.as<DictDocNode>()) {
58 PrintTypedDoc(GetRef<DictDoc>(doc_node));
59 } else if (const auto* doc_node = doc.as<SliceDocNode>()) {
60 PrintTypedDoc(GetRef<SliceDoc>(doc_node));
61 } else if (const auto* doc_node = doc.as<StmtBlockDocNode>()) {
62 PrintTypedDoc(GetRef<StmtBlockDoc>(doc_node));
63 } else if (const auto* doc_node = doc.as<AssignDocNode>()) {
64 PrintTypedDoc(GetRef<AssignDoc>(doc_node));
65 } else if (const auto* doc_node = doc.as<IfDocNode>()) {
66 PrintTypedDoc(GetRef<IfDoc>(doc_node));
67 } else if (const auto* doc_node = doc.as<WhileDocNode>()) {
68 PrintTypedDoc(GetRef<WhileDoc>(doc_node));
69 } else if (const auto* doc_node = doc.as<ForDocNode>()) {
70 PrintTypedDoc(GetRef<ForDoc>(doc_node));
71 } else if (const auto* doc_node = doc.as<ScopeDocNode>()) {
72 PrintTypedDoc(GetRef<ScopeDoc>(doc_node));
73 } else if (const auto* doc_node = doc.as<ExprStmtDocNode>()) {
74 PrintTypedDoc(GetRef<ExprStmtDoc>(doc_node));
75 } else if (const auto* doc_node = doc.as<AssertDocNode>()) {
76 PrintTypedDoc(GetRef<AssertDoc>(doc_node));
77 } else if (const auto* doc_node = doc.as<ReturnDocNode>()) {
78 PrintTypedDoc(GetRef<ReturnDoc>(doc_node));
79 } else if (const auto* doc_node = doc.as<FunctionDocNode>()) {
80 PrintTypedDoc(GetRef<FunctionDoc>(doc_node));
81 } else if (const auto* doc_node = doc.as<ClassDocNode>()) {
82 PrintTypedDoc(GetRef<ClassDoc>(doc_node));
83 } else {
84 LOG(FATAL) << "Do not know how to print " << doc->GetTypeKey();
85 throw;
86 }
87}
